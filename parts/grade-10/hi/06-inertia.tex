\chapter{बल और जड़ता का सिद्धांत}\label{ch:g10:inertia}

पैडल चलाना बंद करो और साइकिल रुक जाएगी; खींचना बंद करो और
स्लेज अभी भी \omterm{def:g10:universal-gravitation:weight}{खड़ा} है. अनुभव फुसफुसाता है कि गति को एक द्वारा पोषित किया जाना चाहिए
प्रयास---और दो हजार वर्षों तक भौतिकी सहमत रही। यह अध्याय
का निर्माण करता है बल, एक तीर मापा गया \omterm{def:g10:inertia:force}{न्यूटन}, और पहला बताता है
यांत्रिकी का महान नियम: वास्तव में अकेला छोड़ दिया गया, एक गतिशील शरीर अपने पास रखता है
गति, सीधी और स्थिर, हमेशा के लिए।

\section{किसी क्रिया को एक शक्ति के रूप में मॉडलिंग करना}

\begin{definition}[बल]\label{def:g10:inertia:force}
ए \emph{बल}\index{बल} एक यांत्रिक क्रिया को मॉडल करता है (एक धक्का, एक खींच,
एक आकर्षण) एक शरीर पर लगाया गया। इसकी विशेषता है:
\begin{itemize}
  \item इसका \emph{आवेदन का बिंदु}\index{आवेदन का बिंदु};
  \item इसकी दिशा--वह रेखा जिसके साथ यह कार्य करता है और तरीका
        उस रेखा के साथ;
  \item इसका परिमाण, में मापा जाता है \emph{न्यूटन}\index{न्यूटन}
        (प्रतीक \unit{N}).
\end{itemize}
एक बल को तीर के रूप में खींचा जाता है $\vect F $ (एक वेक्टर, जैसा कि गणित में होता है
आयतन) इसके अनुप्रयोग के बिंदु से, लंबाई के आनुपातिक के साथ
एक निर्धारित पैमाने पर परिमाण.
\end{definition}

\begin{example}[स्कूल बैग का वजन]\label{ex:g10:inertia:schoolbag}
जनसमूह का एक स्कूलबैग $ m = \qty{4.0}{kg}$ पृथ्वी द्वारा नीचे की ओर खींचा जाता है
इसके साथ \omterm{def:g10:universal-gravitation:weight}{वज़न} (\cref{ch:g10:universal-gravitation}):
$ P = mg = 4.0 \times 9.81 \approx \qty{39}{N}$ --- एक \omterm{def:g10:inertia:force}{न्यूटन} के बारे में है
एक सेब का \omterm{def:g10:universal-gravitation:weight}{वज़न}. पैमाने पर $\qty{1}{cm} \leftrightarrow
\qty{10}{N}$: एक लंबवत तीर \qty{3.9}{cm} लंबा, पर लागू किया गया
बैग का गुरुत्वाकर्षण केंद्र, नीचे की ओर इशारा करते हुए।
\end{example}

\section{बलों की एक सूची}

\begin{definition}[संपर्क और दूरी बल]\label{def:g10:inertia:contact}
ए \emph{संपर्क बल}\index{संपर्क बल} बिन्दुओं पर लगाया जाता है
जहां दो शरीर छूते हैं. ए \emph{दूरी पर बल}\index{दूरी पर बल} किसी संपर्क की आवश्यकता नहीं: \omterm{def:g10:universal-gravitation:weight}{वज़न} ---गुरुत्वाकर्षण खिंचाव
\cref{ch:g10:universal-gravitation} --- इसमें केवल एक ही प्रयोग किया जाता है
अध्याय.
\end{definition}

\begin{definition}[सामान्य संपर्क बल]\label{def:g10:inertia:inventory}
चार \omterm{def:g10:inertia:contact}{संपर्क बल}ों अधिकांश रोजमर्रा की स्थितियों को कवर करें:
\begin{itemize}
  \item  \emph{सामान्य प्रतिक्रिया}\index{सामान्य प्रतिक्रिया} $\vect R $ का
        एक समर्थन, इसकी सतह पर लंबवत, इससे दूर धकेलना;
  \item  \emph{तनाव}\index{तनाव} $\vect T $ तार, रस्सी या का
        श्रृंखला, इसके साथ निर्देशित, इसकी ओर खींच रही है;
  \item \emph{टकराव}\index{टकराव} $\vect f $ एक सतह से,
        हवा या पानी, फिसलन या गति का विरोध;
  \item  \emph{जोर}\index{जोर} किसी इंजन, जेट या हाथ का,
        प्रयास के साथ आगे बढ़ना।
\end{itemize}
\end{definition}

\begin{example}[खींचे गए स्लेज पर इन्वेंटरी]\label{ex:g10:inertia:sled}
बर्फ़ के पार खींची गई एक स्लेज: \omterm{def:g10:universal-gravitation:weight}{वज़न} $\vect P $ (दूरी), \omterm{def:g10:inertia:inventory}{सामान्य
प्रतिक्रिया} $\vect R $ बर्फ का, \omterm{def:g10:inertia:inventory}{तनाव} $\vect T $ रस्सी का, \omterm{def:g10:inertia:inventory}{टकराव}
$\vect f $ फिसलने का विरोध. स्लेज को और कुछ नहीं छूता:
सूची पूर्ण है --- \emph{\omterm{def:g10:universal-gravitation:weight}{वज़न} हमेशा, फिर एक \omterm{def:g10:inertia:contact}{संपर्क बल}
प्रति छूने वाली वस्तु}.
\end{example}

\section{जड़ता का सिद्धांत}

\begin{definition}[सीधी एकसमान गति]\label{def:g10:inertia:uniform}
एक शरीर अंदर है \emph{सीधी एकसमान गति}\index{सीधी एकसमान गति} एक फ्रेम में जब यह \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} एक सीधी रेखा है जिस पर यात्रा की जाती है
स्थिर गति: इसका वेग वेक्टर $\vect v $ हर जगह एक जैसा है
तुरंत. बाकी तो विशेष मामला है $\vect v = \vect 0$.
\end{definition}

\begin{definition}[क्षतिपूर्ति करने वाली ताकतें]\label{def:g10:inertia:compensate}
बल किसी शरीर पर अभिनय करना \emph{मुआवजा}\index{मुआवजा}
जब उनका संयुक्त प्रभाव शून्य होता है: सिरे से पूंछ तक, तो उनके तीर वापस लौट आते हैं
शुरुआत. दो बल ठीक उसी समय क्षतिपूर्ति करें जब वे एक ही पंक्ति साझा करें
समान परिमाण और विपरीत तरीकों से कार्य करना।
\end{definition}

\begin{theorem}[जड़ता का सिद्धांत]\label{thm:g10:inertia:principle}
\index{जड़ता का सिद्धांत}\index{जड़ता}
में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}, एक निकाय जो संख्या के अधीन है बल, या करने के लिए बल वह
\omterm{def:g10:inertia:compensate}{मुआवजा}, आराम की स्थिति में रहता है या रखता है \omterm{def:g10:inertia:uniform}{सीधी एकसमान गति};
इसके विपरीत, शरीर आराम की स्थिति में या अंदर \omterm{def:g10:inertia:uniform}{सीधी एकसमान गति} विषय है
को क्षतिपूर्ति करने वाली ताकतें (या किसी को नहीं)।
\end{theorem}
\admitted

\begin{remark}
कोई भी प्रयोग किसी पिंड को पूरी तरह से अलग नहीं कर सकता है, इसलिए सिद्धांत एक अभिधारणा है
--- गैलीलियो द्वारा झलक, \omterm{def:g10:inertia:force}{न्यूटन} द्वारा कहा गया। बाद के अध्याय में इसे अपग्रेड किया गया है
कानून संबंधी बल को \emph{परिवर्तन} वेग का; वर्ष 1 खंड
पूछता है कि यह वास्तव में किस फ्रेम में है।
\end{remark}

\begin{proposition}[बलों और पीछे से गति पढ़ना]\label{prop:g10:inertia:converse}
में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}:
\begin{enumerate}
  \item यदि किसी पिंड का वेग परिमाण में बदलता है \emph{या} में
        दिशा--- द बल इस पर करो \emph{नहीं} \omterm{def:g10:inertia:compensate}{मुआवजा};
  \item यदि कोई शरीर आराम की स्थिति में है या अंदर है \omterm{def:g10:inertia:uniform}{सीधी एकसमान गति}, इसका
        आविष्कार किया गया बल रद्द करना होगा --- अज्ञात का निर्धारण करना।
\end{enumerate}
\end{proposition}

\begin{proof}
पहला कथन प्रतिसकारात्मक है
\cref{thm:g10:inertia:principle}, \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} इसका विपरीत भाग लागू होता है
एक पूर्ण सूची के लिए.
\end{proof}

\begin{example}[हॉकी पक]\label{ex:g10:inertia:puck}
छड़ी से छोड़ा गया पक पार हो जाता है \qty{30}{m} अंदर चिकनी बर्फ़ का
\qty{2.5}{s}, सीधे स्थिरांक पर $ v = 30/2.5 = \qty{12}{m/s}$.
भंडार: \omterm{def:g10:universal-gravitation:weight}{वज़न} और \omterm{def:g10:inertia:inventory}{सामान्य प्रतिक्रिया} (\omterm{def:g10:inertia:inventory}{टकराव} नगण्य)। वे
\omterm{def:g10:inertia:compensate}{मुआवजा}, और सिद्धांत सटीक रूप से देखी गई गति की भविष्यवाणी करता है:
कोई भी चीज़ पक को आगे नहीं धकेलती --- और किसी चीज़ की ज़रूरत नहीं है।
\end{example}

\begin{remark}[अरस्तू के विरुद्ध गैलीलियो]
अरस्तू ने सिखाया कि प्रत्येक गति को एक प्रेरक की आवश्यकता होती है, और दैनिक जीवन ---
\omterm{def:g10:inertia:inventory}{टकराव} प्रत्येक संपर्क में --- सहमत प्रतीत होता है। गैलीलियो ने इसे आदर्श बनाया
\omterm{def:g10:inertia:inventory}{टकराव} दूर: हमेशा चिकनी सतहों को हमेशा कम धक्का की आवश्यकता होती है
\emph{बनाए रखना} गति। दृढ़ता, आराम नहीं, प्राकृतिक अवस्था है।
\end{remark}

\begin{remark}[लपकता हुआ यात्री]
जब कोई बस तेजी से ब्रेक लगाती है, तो खड़े यात्री आगे की ओर झुक जाते हैं। नहीं बल
उन्हें फेंकता है: जड़ता से उनके शरीर बस के दौरान अपना वेग बनाए रखते हैं
अपना खो देता है. सीट बेल्ट पिछड़े को आपूर्ति करती है बल सूची
कमी है.
\end{remark}

\section{संतुलित बल: एक रेखाचित्र से स्थैतिकता}

\begin{definition}[बल आरेख]\label{def:g10:inertia:diagram}
 \emph{बल आरेख}\index{बल आरेख} एक शरीर का शरीर को दर्शाता है
अकेले, सबके साथ बल अभिनय \emph{पर} इसे इसके पैमाने से खींचा गया
\omterm{def:g10:inertia:force}{आवेदन का बिंदु}; बल लगाए गए \emph{द्वारा} शरीर कभी प्रकट नहीं होता.
\end{definition}

\begin{method}[एक स्थिर स्थिति का विश्लेषण]\label{met:g10:inertia:method}
\begin{enumerate}
  \item अध्ययन के लिए शरीर चुनें; में काम करें \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}.
  \item भंडार: \omterm{def:g10:universal-gravitation:weight}{वज़न} हमेशा, एक \omterm{def:g10:inertia:contact}{संपर्क बल} प्रति संपर्क.
  \item खींचना \omterm{def:g10:inertia:diagram}{बल आरेख}.
  \item शरीर आराम पर है, इसलिए बल \omterm{def:g10:inertia:compensate}{मुआवजा}
        (\cref{thm:g10:inertia:principle}): अज्ञात परिमाण पढ़ें
        ड्राइंग (1डी में समान लंबाई, 2डी में घटक)।
\end{enumerate}
\end{method}

\begin{example}[मेज पर किताब है]\label{ex:g10:inertia:book}
जन की एक किताब \qty{0.50}{kg} एक मेज पर टिकी हुई है: \omterm{def:g10:universal-gravitation:weight}{वज़न}
$ P = 0.50 \times 9.81 \approx \qty{4.9}{N}$ और \omterm{def:g10:inertia:inventory}{सामान्य प्रतिक्रिया}
$\vect R $. \omterm{def:g10:relative-motion:frame}{आराम से}, $\vect R $ क्षतिपूर्ति $\vect P $: ऊर्ध्वाधर, ऊपर की ओर,
$ R = \qty{4.9}{N}$. दूसरी किताब जोड़ें और $ R $ समायोजित करता है --- एक समर्थन
बिल्कुल उतना ही ज़ोर से धक्का देता है जितनी आवश्यकता होती है।
\end{example}

  \node[gray] at (-0.42,1.38) {$45^\circ$};
  \draw[fill=omExo!25] (0,-0.35) circle (0.35);
  \draw[-Stealth, omDef, very thick] (0,0) -- (-1.06,1.06)
    node[left] {$\vect T_1$};
  \draw[-Stealth, omDef, very thick] (0,0) -- (1.06,1.06)
    node[right] {$\vect T_2$};
  \draw[-Stealth, omThm, very thick] (0,0) -- (0,-1.5)

\end{tikzpicture}

{\small लटकता हुआ दीपक: दो \omterm{def:g10:inertia:inventory}{तनाव}' क्षैतिज भाग रद्द कर देते हैं
समरूपता; उनके ऊर्ध्वाधर भाग एक साथ मिलकर चलते हैं \omterm{def:g10:universal-gravitation:weight}{वज़न}.}
\end{omfigure}

\begin{example}[दो तारों पर दीपक]\label{ex:g10:inertia:lamp}
द्रव्यमान का एक दीपक \qty{3.0}{kg} ($ P \approx \qty{29.4}{N}$) दो से लटकता है
तार, प्रत्येक पर $45^\circ $ ऊर्ध्वाधर से. समरूपता द्वारा \omterm{def:g10:inertia:inventory}{तनाव}
बराबर हैं ($ T_1 = T_2 = T $) और उनके क्षैतिज घटक रद्द हो जाते हैं; द
ऊर्ध्वाधर घटक, प्रत्येक $ T \cos 45^\circ $, ले जाओ \omterm{def:g10:universal-gravitation:weight}{वज़न}:
\[
2\,T \cos 45^\circ = P
\qquad\text{so}\qquad
T = \frac{29.4}{2 \times 0.707} \approx \qty{21}{N}
\]
--- आधे से ज्यादा \omterm{def:g10:universal-gravitation:weight}{वज़न} प्रति तार: एक कोण पर खींचने की लागत
\omterm{def:g10:inertia:inventory}{तनाव}.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:inertia:1}
सूची और वर्गीकृत करें (संपर्क या दूरी)। बल पर: (ए) एक किताब
एक मेज पर आराम करते समय; (बी) उसके तार से लटका हुआ एक दीपक; (सी) एक पक
चिकनी बर्फ पर फिसलना; (डी) एक सेब गिरना (वायु प्रतिरोध)।
नगण्य)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:2}
की गणना करें \omterm{def:g10:universal-gravitation:weight}{वज़न} एक का \qty{250}{g} किताब और एक की \qty{60}{kg}
व्यक्ति। पैमाने पर $\qty{1}{cm} \leftrightarrow \qty{1}{N}$, कैसे
किताब का तीर लंबा है, और पैमाना इसके लिए निराशाजनक क्यों है
व्यक्ति?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:3}
पैमाने पर एक आरेख पर $\qty{1}{cm} \leftrightarrow \qty{2}{N}$, ए
बल एक क्षैतिज तीर है \qty{3.5}{cm} बाईं ओर इंगित करने वाला लंबा, लागू किया गया
एक बिंदु पर $ A $. इसकी तीन विशेषताएँ बताइये।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:4}
द्रव्यमान का एक शब्दकोश \qty{1.2}{kg} एक शेल्फ पर पड़ा है. इसका चित्र बनाओ \omterm{def:g10:inertia:diagram}{बल
आरेख} और प्रत्येक का परिमाण और दिशा दीजिए बल.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:5}
सही या गलत --- गलत को सही करें: (ए) एक शरीर हमेशा समाप्त होता है
जब तक रुकना नहीं बल उसे धकेलता रहता है; (बी) \omterm{def:g10:inertia:uniform}{सीधी एकसमान गति}
किसी मुआवजे की आवश्यकता नहीं है बल; (सी) आराम पर एक शरीर संख्या के अधीन है
बल; (डी) यदि किसी पिंड का वेग बदलता है, तो बल इस पर मत करो
\omterm{def:g10:inertia:compensate}{मुआवजा}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:6}
रिहाई के बाद, एक पक ढक जाता है \qty{24}{m} में बर्फ का \qty{3.0}{s},
स्थिर गति से सीधे. इसकी गति की गणना करें; करो बल उस पर
\omterm{def:g10:inertia:compensate}{मुआवजा}? कंक्रीट पर वही पक अंदर ही रुक जाती है एम-eters: क्या बदला
सूची में, और क्या करता है \cref{prop:g10:inertia:converse}
निष्कर्ष?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:7}
ए \qty{70}{kg} यात्री स्थिर गति से ऊपर उठती हुई लिफ्ट में \omterm{def:g10:universal-gravitation:weight}{खड़ा} है
\qty{1.5}{m/s}. लिस्ट बल यात्री पर; क्या वे \omterm{def:g10:inertia:compensate}{मुआवजा}?
फर्श का आकार बताइये बल. क्या इससे कुछ भी बदल जाएगा
लिफ्ट आराम पर है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:8}
द्रव्यमान का एक दीपक \qty{1.8}{kg} एक ही ऊर्ध्वाधर तार से \omterm{def:g10:relative-motion:frame}{आराम से} लटका हुआ है।
की गणना करें \omterm{def:g10:inertia:inventory}{तनाव} तार का. तार को दो से बदल दिया गया है
\emph{\omterm{def:g10:universal-gravitation:weight}{खड़ा}} तार लोड को समान रूप से साझा करते हैं: क्या है \omterm{def:g10:inertia:inventory}{तनाव} का
प्रत्येक?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:9}
एक बस तेजी से ब्रेक लगाती है और एक \omterm{def:g10:universal-gravitation:weight}{खड़ा} यात्री आगे की ओर झुक जाता है; एक में
स्थिर गति से वक्र, वही यात्री बाहर की ओर झुक जाता है। समझाओ
दोनों ``फॉरवर्ड'' या ``आउटवर्ड'' का आविष्कार किए बिना बल.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:10}
कुल द्रव्यमान का एक पैराशूटिस्ट \qty{85}{kg} उतरता खड़ी पर
स्थिर \qty{5.0}{m/s}. खींचना \omterm{def:g10:inertia:diagram}{बल आरेख} और हवा की गणना करें
चंदवा पर प्रतिरोध.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:11}
एक वॉटर-स्कीयर को स्थिर गति से सीधा खींचा जाता है; क्षैतिज रस्सी
साथ खींचता है \qty{300}{N}. पानी का क्षैतिज खिंचाव क्या है?
स्की (औचित्य)? जो दो ऊर्ध्वाधर बल \omterm{def:g10:inertia:compensate}{मुआवजा} एक दूसरे?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:12}
द्रव्यमान का एक दीपक \qty{2.4}{kg} पर दो तारों से लटका हुआ है $40^\circ $ से
ऊर्ध्वाधर ($\cos 40^\circ \approx 0.766$). प्रत्येक की गणना करें \omterm{def:g10:inertia:inventory}{तनाव}.
क्यों करता है \omterm{def:g10:inertia:inventory}{तनाव} जैसे-जैसे तार क्षैतिज की ओर बढ़ते हैं, और
दो बिल्कुल क्षैतिज तार कभी भी लैंप को क्यों नहीं पकड़ पाते?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:13}
एक कर्लिंग पत्थर को धकेला जाता है, छोड़ा जाता है, बहुत धीमी गति से फिसलता है
धीरे-धीरे, फिर आराम करता है। प्रत्येक चरण (पुश, सरकना, आराम) के लिए, ड्रा करें
\omterm{def:g10:inertia:diagram}{बल आरेख} और बताएं कि क्या बल \omterm{def:g10:inertia:compensate}{मुआवजा}, उद्धरण
\cref{thm:g10:inertia:principle} या \cref{prop:g10:inertia:converse}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:14}
जांच \emph{मल्लाह 1} अंतरतारकीय अंतरिक्ष के माध्यम से तटों पर
\qty{17}{km/s}, इंजन बंद। इसे रखने के लिए ईंधन की आवश्यकता क्यों नहीं पड़ती?
गति, और अरस्तू क्या भविष्यवाणी करेगा? यह कितनी दूर तक यात्रा करता है?
वर्ष, में \unit{km} में फिर खगोलीय इकाइयाँ ($\qty{1}{au} \approx
\qty{1.496e8}{km}$)? इसे बदलने के लिए भी थ्रस्टर फायर क्यों करना चाहिए?
दिशा \emph{बिना} गति बदल रही है?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:15}
तीन रस्सियाँ आराम की स्थिति में एक रिंग को खींचती हैं। एक ग्रिड पर जहां एक वर्ग है
\qty{1}{N}, दो बल पढ़ना $\vect F_1\,(3, 2)$ और
$\vect F_2\,(-1, 2)$. तीसरा खोजें बलके घटक, परिमाण और
दिशा.
\end{exercise}

\section{समस्या: तीन कृत्यों में जड़ता}

\begin{problem}[{सप्ताहांत समस्या---हवाई अड्डे का रास्ता, आइस रिंक
और पैराशूट: मुआवजे के हस्ताक्षर के रूप में निरंतर वेग
बल, और क्यों एक पैराशूट खींचें को कम नहीं बल्कि गति को कम करता है}]
\label{pb:g10:inertia:1}
तीन दृश्य --- चलती पगडंडी पर एक सूटकेस, बर्फ की रिंक पर एक पक,
शाम की हवा में एक पैराशूटिस्ट --- और तीनों में एक कानून बात कर रहा है।

\medskip
\noindent\textbf{Part I --- The airport walkway.}
ए \qty{23}{kg} सूटकेस स्थिर गति से चलते हुए रास्ते पर \omterm{def:g10:universal-gravitation:weight}{खड़ा} है
गति \qty{0.75}{m/s}.
\begin{enumerate}
  \item सूटकेस की ग्राउंड-फ़्रेम गति? इन्वेंटरी बल उस पर.
  \item ये करो बल \omterm{def:g10:inertia:compensate}{मुआवजा}? औचित्य; दोनों परिमाणों की गणना करें।
  \item इसकी तुलना करें \omterm{def:g10:inertia:diagram}{बल आरेख} उसी सूटकेस के साथ
        मंज़िल. तुलना किस गहरे कथन को दर्शाती है?
  \item रास्ता झटके से रुक जाता है। बताएं कि सूटकेस क्या करता है,
        फॉरवर्ड का आविष्कार किए बिना बल.
  \item इसका मालिक चलता है \qty{1.2}{m/s} वॉकवे के सापेक्ष. उसका
        ग्राउंड-फ़्रेम गति? करो बल पर \emph{उसकी} \omterm{def:g10:inertia:compensate}{मुआवजा}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The ice rink.}
\begin{enumerate}[resume]
  \item ए \qty{160}{g} पक स्थिरांक पर सीधा ग्लाइड होता है
        \qty{8.0}{m/s}. इसका चित्र बनाओ \omterm{def:g10:inertia:diagram}{बल आरेख} परिमाण के साथ.
  \item अरस्तू का दावा है कि पक को एक मूवर की जरूरत है। रिंक क्या करता है
        उत्तर दें, और रोजमर्रा की सतहें उसके पक्ष में क्यों दिखती हैं?
  \item पक एक कठिन जगह को पार करती है और धीमी गति से चलती है। के बारे में निष्कर्ष
        बल वहाँ? कौन बल अपराधी है?
  \item एक स्केटर स्थिर गति से सुचारू रूप से घूमता है। करो बल पर
        उसे \omterm{def:g10:inertia:compensate}{मुआवजा}? सावधान: वेग किस प्रकार की मात्रा है?
  \item गति से आरेख का मिलान करें: (ए) \omterm{def:g10:inertia:compensate}{मुआवजा} बल; (बी) एक अतिरिक्त
        बल विलोम $\vect v $; (सी) एक अतिरिक्त बल के लिए सीधा
        $\vect v $ --- (i) सीधे धीमा करना; (ii) स्थिरांक पर घूमना
        गति; (iii) \omterm{def:g10:inertia:uniform}{सीधी एकसमान गति}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The parachute.}
एक जम्पर का कुल द्रव्यमान, गियर सहित, है \qty{80}{kg}.
\begin{enumerate}[resume]
  \item विमान को छोड़कर, ऊर्ध्वाधर गति और खिंचाव अभी भी शून्य है।
        की गणना करें बल; क्या वे \omterm{def:g10:inertia:compensate}{मुआवजा}? आगे क्या होना चाहिए?
  \item ड्रैग गति के साथ बढ़ता है। खींचते समय गिरने का वर्णन करें $<$ \omterm{def:g10:universal-gravitation:weight}{वज़न}.
  \item कैनोपी खुलने से पहले, जम्पर एक स्थिर स्थान पर पहुँच जाता है
        \qty{50}{m/s} (``टर्मिनल स्पीड'')। ड्रैग की गणना करें. क्या है?
        गति, और यह ''छिपी हुई जड़ता'' क्यों है?
  \item कैनोपी खुलती है: खिंचाव अब बहुत अधिक हो गया है \omterm{def:g10:universal-gravitation:weight}{वज़न}. क्या
        वेग से होता है? (जम्पर चलता है \emph{नीचे}
        लगातार।)
  \item एक नया स्थिर \qty{5.0}{m/s} पहुंच गया है. ड्रैग की गणना करें;
        प्रश्न 13 से तुलना करें।
  \item पंचलाइन: पैराशूट ने क्या बदला --- टर्मिनल
        खींचें, या वह गति जिस पर खींचें पहुंचती है \omterm{def:g10:universal-gravitation:weight}{वज़न}?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Verdicts.}
\begin{enumerate}[resume]
  \item पूरा करें और उचित ठहराएँ: ``आराम पर या अंदर \omterm{def:g10:inertia:uniform}{सीधी एकसमान गति}
        $\iff$ \dots''. बाकी के लिए अलग कानून क्यों नहीं चाहिए?
  \item अरस्तू की ``गति को एक प्रस्तावक की आवश्यकता है'' किस अधिनियम में होगी?
        सबसे स्पष्ट रूप से विफल रहा, और गैलीलियो किस ओर इशारा करेगा?
  \item एक परीक्षा स्क्रिप्ट का दावा है: ``पैराशूटिस्ट लगातार गिरता है
        गति, तो नहीं बल उस पर कार्रवाई करता है''. इसे एक लाइन में सुधारें.
  \item समापन: प्रत्येक कार्य के लिए, जोड़ी का नाम बताएं बल वह \omterm{def:g10:inertia:compensate}{मुआवजा},
        और इस समस्या में बीस बार प्रयुक्त एक कानून बताएं।
\end{enumerate}
\end{problem}
