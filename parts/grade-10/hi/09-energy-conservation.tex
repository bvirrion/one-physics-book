\chapter{ऊर्जाः स्वरूप एवं संरक्षण}\label{ch:g10:energy-conservation}

एक चलती कार, एक चार्ज किया हुआ फोन, एक गर्म ओवन और एक उठा हुआ हथौड़ा शेयर
एक संपत्ति: प्रत्येक कुछ घटित कर सकता है। भौतिकी उसे मापती है
एकल संख्या के साथ \omterm{def:g10:energy-conservation:efficiency}{क्षमता} --- ऊर्जा ---और इसका हिसाब किताब रखता है
एक लोहे के नियम के तहत: कुल कभी नहीं बदलता है, यह केवल चलता रहता है
वेशभूषा बदलता है. यह अध्याय खाता बही खोलता है: प्रपत्र ऊर्जा
लेता है, यह जिन जंजीरों से यात्रा करता है, और क्या \omterm{def:g10:energy-conservation:kwh}{किलोवाट्ट घन्ता} खरीदता है.

\section{ऊर्जा: सामान्य मुद्रा}

\begin{definition}[ऊर्जा और जूल]\label{def:g10:energy-conservation:energy}
\emph{ऊर्जा}\index{ऊर्जा} किसी सिस्टम की उत्पादन \omterm{def:g10:energy-conservation:efficiency}{क्षमता} को मापता है
परिवर्तन: पदार्थ को गति देना, उठाना, गर्म करना, जलाना। इसका \omterm{def:g10:orders-of-magnitude:unit}{इकाई}
है \emph{जौल}\index{जौल} (\unit{J}). संग्रहित, हस्तांतरित,
परिवर्तित --- भौतिकी की प्रत्येक शाखा इस एक मुद्रा में गिनी जाती है।
\end{definition}

\begin{definition}[गतिज ऊर्जा]\label{def:g10:energy-conservation:kinetic}
द्रव्यमान का एक पिंड $m $ (में \unit{kg}) गति से घूमना $ v $ (में
\unit{m/s}) ले जाता है \emph{गतिज ऊर्जा}\index{गतिज ऊर्जा}
$ E_k = \tfrac12\, m v^2$: द्रव्यमान को दोगुना करने से यह दोगुना हो जाता है, दोगुना हो जाता है
गति इसे चौगुनी कर देती है।
\end{definition}

\begin{example}[कार बनाम पैदल यात्री]\label{ex:g10:energy-conservation:kinetic}
ए \qty{1300}{kg} कार पर \qty{100}{km/h} $= \qty{27.8}{m/s}$ किया जाता है
$ E_k = \tfrac12 \times 1300 \times 27.8^2 \approx \qty{5.0e5}{J}$;
ए \qty{70}{kg} पैदल यात्री पर \qty{1.5}{m/s},
$\tfrac12 \times 70 \times 1.5^2 \approx \qty{79}{J}$ --- छह
हजार गुना कम. गति महँगी है: वर्ग इसका ध्यान रखता है।
\end{example}

\begin{definition}[गुरुत्वाकर्षण स्थितिज ऊर्जा]\label{def:g10:energy-conservation:potential}
द्रव्यमान का एक पिंड $ m $ ऊंचाई पर $ h $ (में \unit{m}) किसी चुने हुए से ऊपर
संदर्भ स्तर संग्रहीत करता है \emph{गुरुत्वाकर्षण क्षमता ऊर्जा}\index{गुरुत्वाकर्षण क्षमता ऊर्जा} $ E_p = m g h $, साथ
$ g = \qty{9.81}{N/kg}$ ---जमीन के पास मान्य, कहाँ $ g $ है
प्रभावी रूप से स्थिर. केवल \emph{मतभेद} ऊंचाई का मामला:
स्तर कहाँ $ E_p = 0$ स्वतंत्र रूप से चुना जाता है.
\end{definition}

\begin{example}[पानी का टावर]\label{ex:g10:energy-conservation:potential}
प्रत्येक घन \omterm{def:g10:orders-of-magnitude:unit}{मीटर} (\qty{1000}{kg}) संग्रहित जल का \qty{30}{m} ऊपर ए
जल मीनार धारण करता है $1000 \times 9.81 \times 30 \approx \qty{2.9e5}{J}$
--- कार का लगभग छह दसवां हिस्सा \omterm{def:g10:energy-conservation:kinetic}{गतिज ऊर्जा} ऊपर। शहर
भण्डार ऊर्जा उपरि.
\end{example}

\begin{definition}[अन्य रूप]\label{def:g10:energy-conservation:forms}
ऊर्जा कम दिखाई देने वाली वेशभूषा में भी छिपते हैं: \emph{थर्मल ऊर्जा}\index{थर्मल ऊर्जा}, शरीर की अव्यवस्थित हलचल
अणु (जितना गर्म, उतना अधिक); \emph{रसायन ऊर्जा}\index{रसायन ऊर्जा}, आणविक बंधों में संग्रहित --- भोजन,
लकड़ी, गैसोलीन, बैटरी; \emph{विद्युतीय ऊर्जा}\index{विद्युतीय ऊर्जा}, पौधे से धाराओं द्वारा ले जाया जाता है
सॉकेट; \emph{विकिरण ऊर्जा}\index{विकिरण ऊर्जा}, द्वारा ले जाया गया
ख़ाली जगह पर भी प्रकाश --- सूर्य का कैसा ऊर्जा हम तक पहुंचता है;
\emph{परमाणु ऊर्जा}\index{परमाणु ऊर्जा}, परमाणु के अंदर संग्रहीत
नाभिक, तारों में जारी (\cref{ch:g12:nuclear-energy}).
\end{definition}

\section{स्थानान्तरण और रूपांतरण श्रृंखलाएँ}

\begin{definition}[स्थानांतरण, रूपांतरण, श्रृंखला]\label{def:g10:energy-conservation:transfer}
एक \emph{ऊर्जा अंतरण}\index{ऊर्जा अंतरण} चाल ऊर्जा
सिस्टम के बीच --- \emph{जलाशयों}\index{जलाशयों};
एक \emph{ऊर्जा रूपांतरण}\index{ऊर्जा रूपांतरण} इसे बदलता है
प्रपत्र. एक उपकरण जो या तो (टरबाइन, मोटर, लैंप, मसल) कर रहा है वह है
\emph{कनवर्टर}\index{कनवर्टर}. ए \emph{रूपांतरण श्रृंखला}\index{रूपांतरण श्रृंखला} यात्रा के चित्र: बक्से के लिए
स्थानान्तरण के लिए जलाशयों और रूपों, लेबल वाले तीर।
\end{definition}

\begin{method}[एक रूपांतरण श्रृंखला खींचना]\label{met:g10:energy-conservation:chain}
\begin{enumerate}
  \item प्रारंभिक भंडार और स्वरूप की पहचान करें (क्या उपभोग किया जाता है?)।
  \item का पीछा करो ऊर्जा \omterm{def:g10:energy-conservation:transfer}{कनवर्टर} द्वारा \omterm{def:g10:energy-conservation:transfer}{कनवर्टर}: प्रति एक तीर
        स्थानांतरण, प्रत्येक बॉक्स में नामित प्रपत्र।
  \item प्रत्येक वास्तविक पर एक ताप तीर जोड़ें \omterm{def:g10:energy-conservation:transfer}{कनवर्टर} --- कुछ ऊर्जा
        हमेशा गर्मी के रूप में लीक होता है (\omterm{def:g10:inertia:inventory}{टकराव}, तार, निकास)।
\end{enumerate}
\end{method}

स्विंग के शीर्ष पर, सबसे निचले बिंदु के माध्यम से सभी गतिज (लेते हुए)।

\section{ऊर्जा संरक्षण}

\begin{theorem}[ऊर्जा संरक्षण]\label{thm:g10:energy-conservation:conservation}
\index{ऊर्जा संरक्षण}
संपूर्ण ऊर्जा एक पृथक प्रणाली का --- जो किसी भी चीज़ का आदान-प्रदान नहीं करता
बाहर के साथ --- स्थिर है। ऊर्जा कभी नहीं बनाया जाता और कभी नहीं
नष्ट हो गया: यह केवल रूप और स्थान बदलता है, और एक जलाशय क्या है
हारता है, दूसरों को लाभ होता है, \omterm{def:g10:energy-conservation:energy}{जौल} के लिए \omterm{def:g10:energy-conservation:energy}{जौल}.
\end{theorem}
\admitted

\begin{remark}\label{rem:g10:energy-conservation:admitted}
किसी भी प्रयोग ने कभी भी इस सिद्धांत को विफल होते नहीं देखा है; यहाँ हम लेते हैं
जैसा दिया गया है वैसा ही करें और इसे संख्यात्मक रूप से उपयोग करें। \cref{ch:g12:work-and-energy}
इसका यांत्रिक भाग प्राप्त होता है ($ E_k + E_p $) \omterm{def:g10:inertia:force}{न्यूटन} के नियमों से; द
पूरा लेखा-जोखा विश्वविद्यालय की मात्रा की प्रतीक्षा करता है।
\end{remark}

\begin{example}[निःशुल्क गिरावट, कीमत जूल में]\label{ex:g10:energy-conservation:fall}
ए \qty{2.0}{kg} से गेंद गिरती है \qty{10}{m} (वायु प्रतिरोध
नगण्य): इसकी शुरुआत होती है
$ E_p = 2.0 \times 9.81 \times 10 \approx \qty{196}{J}$, भूमि के साथ
$ E_k = \qty{196}{J}$, इसलिए $ v = \sqrt{2 \times 196/2.0} =
\qty{14}{m/s} \approx \qty{50}{km/h}$. नहीं बल, कोई स्टॉपवॉच नहीं:
केवल संरक्षण से ही प्रभाव पड़ता है।
\end{example}

\section{क्षमता}

\begin{definition}[क्षमता]\label{def:g10:energy-conservation:efficiency}
ए \omterm{def:g10:energy-conservation:transfer}{कनवर्टर} खपत $ E_{\text{consumed}}$ और वांछित का उद्धार करता है
प्रपत्र $ E_{\text{useful}}$; इसका \emph{क्षमता}\index{क्षमता} है
\[
\eta = \frac{E_{\text{useful}}}{E_{\text{consumed}}} < 1 .
\]
संरक्षण वर्जित है $\eta > 1$, और असली कन्वर्टर्स कभी नहीं पहुँचते $1$:
लापता ऊर्जा नष्ट नहीं होता --- यह गर्मी के रूप में निकल जाता है।
\end{definition}

\begin{example}[पेट्रोल कहां जाता है]\label{ex:g10:energy-conservation:engine}
ए \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} गैसोलीन दुकानों का \qty{45}{MJ} का \omterm{def:g10:energy-conservation:forms}{रसायन ऊर्जा}; ए
कार का इंजन इसे परिवर्तित करता है $\eta \approx 0.30$: \qty{13.5}{MJ} का
गति, \qty{31.5}{MJ} निकास, रेडिएटर और को गर्म करना
सड़क. प्रत्येक टैंक का दो तिहाई भाग वातावरण को गर्म करता है।
\end{example}

\begin{remark}[कार्यकुशलता कई गुना बढ़ जाती है]\label{rem:g10:energy-conservation:chain-eta}
श्रृंखलित कन्वर्टर्स उनकी कार्यकुशलता को गुणा करें: a \omterm{def:g10:energy-conservation:power}{शक्ति} पौधा
($0.38$) ग्रिड को खिलाना ($0.94$) चार्जर खिलाना ($0.85$)
पहुंचाता है $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.30$. लंबा
जंजीरें लीक पाइप हैं: प्रत्येक तीर प्रवाह पर कर लगाता है।
\end{remark}

\section{बिजली, वाट और किलोवाट-घंटा}

\begin{definition}[शक्ति]\label{def:g10:energy-conservation:power}
\emph{शक्ति}\index{शक्ति} वह दर है जिस पर ऊर्जा बहती है,
$ P = E/t $, में मापा गया \emph{वाट}\index{वाट} (\unit{W}), साथ
$\qty{1}{W} = \qty{1}{J/s}$. ऊर्जा एक राशि है, शक्ति एक प्रवाह ---
एक बाथटब बनाम उसका नल।
\end{definition}

\begin{example}[केतली बनाम मानव]\label{ex:g10:energy-conservation:power}
ए \qty{2200}{W} केतली चल रही है \qty{150}{s} उपयोग
$ E = 2200 \times 150 = \qty{3.3e5}{J}$. मनुष्य इधर-उधर दौड़ता रहता है
\qty{10}{MJ} प्रति दिन भोजन का औसत, औसत
$\num{1.0e7}/\num{86400} \approx \qty{116}{W}$: आप एक की तरह निष्क्रिय हैं
चमकीला पुराना प्रकाश बल्ब.
\end{example}

\begin{definition}[किलोवाट-घंटा]\label{def:g10:energy-conservation:kwh}
 \emph{किलोवाट्ट घन्ता}\index{किलोवाट्ट घन्ता} (\unit{kW.h}) है
ऊर्जा एक का \qty{1}{kW} एक घंटे तक चलता रहा प्रवाह:
$\qty{1}{kW.h} = 1000 \times 3600\,\unit{J} = \qty{3.6}{MJ}$ ---
द \omterm{def:g10:orders-of-magnitude:unit}{इकाई} बिजली \omterm{def:g10:orders-of-magnitude:unit}{मीटर}ों की संख्या: \omterm{def:g10:energy-conservation:power}{शक्ति} में \unit{kW} कई घंटे.
\end{definition}

\begin{example}[बिल पढ़ना]\label{ex:g10:energy-conservation:bill}
एक बिल एक घटाव और एक गुणा है: नई \omterm{def:g10:orders-of-magnitude:unit}{मीटर} रीडिंग
माइनस ओल्ड देता है किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे; कई बार \omterm{def:g10:orders-of-magnitude:unit}{इकाई} कीमत (लगभग)
$0.25$ यूरो प्रति \unit{kW.h}), प्लस एक निश्चित सदस्यता, देता है
कुल. ए \qty{2}{kW} हीटर चल रहा है \qty{2.5}{h} रात्रिकालीन उपयोग
\qty{5}{kW.h} एक दिन: $38$ यूरो प्रति माह अपने आप।
\end{example}

\begin{exercise}[$\star $]\label{exo:g10:energy-conservation:1}

इसकी गणना करें \omterm{def:g10:energy-conservation:kinetic}{गतिज ऊर्जा}. इसे फिर से करें \qty{130}{km/h}

\end{exercise}

किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे, और इसकी कीमत तय करें $0.25$ यूरो प्रति \unit{kW.h}.
\end{exercise}

इसे लगभग वितरित किया जाता है \qty{100}{\micro s}: गणना करें \omterm{def:g10:energy-conservation:power}{शक्ति},

\end{exercise}

\section{समस्या: एक दिन के लिए घर को बिजली देना}

\begin{problem}[{सप्ताहांत की समस्या---अनाज के कटोरे से लेकर छत तक
पैनल: घर पर एक सामान्य दिन का पूर्ण ऊर्जा ऑडिट, और क्या
जूल की लागत इस पर निर्भर करती है कि इसे कौन बेचता है}]
\label{pb:g10:energy-conservation:1}
एक सामान्य मंगलवार को, एक परिवार यह पता लगाने के लिए निकलता है कि वह कहाँ है
बिजली जाती है, इस अध्याय से लैस, हॉलवे \omterm{def:g10:orders-of-magnitude:unit}{मीटर} और
बिल के पीछे. आज रात का फैसला: उनका दिन एक के बराबर है
\omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} गैसोलीन का, सोलह वर्ग एम-eters धूप की --- या पांच
साइकिल चालक चौबीसों घंटे पैडल मारते रहते हैं।

\medskip
\noindent\textbf{Part I --- The audit.}
दिन का सामान: फ्रिज \qty{100}{W} के लिए \qty{24}{h}; वाटर हीटर \qty{2400}{W} के लिए \qty{2.5}{h}; ओवन \qty{2000}{W} के लिए
\qty{1.0}{h}; वॉशिंग मशीन \qty{500}{W} के लिए \qty{2.0}{h}; प्रकाश
\qty{60}{W} के लिए \qty{5.0}{h}; स्क्रीन \qty{150}{W} के लिए \qty{6.0}{h}.
\begin{enumerate}
  \item प्रत्येक उपकरण की गणना करें ऊर्जा दिन के लिए, में \unit{kW.h}.
  \item दिन का कुल योग किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे, फिर मेगाजूल में;
        एक से तुलना करें \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} गैसोलीन का (\qty{45}{MJ}).
  \item घर के औसत की गणना करें \omterm{def:g10:energy-conservation:power}{शक्ति} ऊपर \qty{24}{h};
        वह चलती केतली का कौन सा अंश है?
  \item कौन सा एकल उपकरण हावी है? दैनिक कुल की पुनः गणना करें
        यदि केवल हीटर चलता है \qty{1.25}{h}.
  \item स्टैंडबाय लाइटें स्थायी रूप से खींचती हैं \qty{15}{W}: दैनिक ऊर्जा,
        कुल, वार्षिक लागत का हिस्सा $0.25$ यूरो प्रति \unit{kW.h}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The bill.}
\begin{enumerate}[resume]
  \item 30 दिन में \omterm{def:g10:orders-of-magnitude:unit}{मीटर} चला गया \qty{56214}{kW.h} को
        \qty{56604}{kW.h}: खपत, और दैनिक औसत की जाँच की गई
        लेखापरीक्षा के विरुद्ध?
  \item बिल की गणना करें: $0.25$ यूरो प्रति \unit{kW.h} प्लस ए
        निश्चित $14$ यूरो सदस्यता.
  \item ग्रिड से एक मेगाजूल की कीमत? गैसोलीन से, पर
        $1.80$ यूरो प्रति लीटर भंडारण \qty{34}{MJ}?
  \item ए \qty{1.5}{MJ} अनाज के कटोरे की कीमत लगभग है $0.50$ यूरो: यह
        मेगाजूल कीमत? के तीन विक्रेताओं को रैंक करें जूल.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The rooftop.}
\begin{enumerate}[resume]
  \item पूर्ण सूर्य का प्रकाश प्रदान करता है \qty{1000}{W} प्रति वर्ग
        \omterm{def:g10:orders-of-magnitude:unit}{मीटर}; पैनल इसे परिवर्तित करते हैं $\eta = 0.20$: विद्युत
        \omterm{def:g10:energy-conservation:power}{शक्ति} का \qty{1}{m^2} पूर्ण सूर्य में?
  \item क्षेत्र का औसत इसके बराबर है \qty{4.0}{h} पूर्ण का
        प्रति दिन सूर्य: की दैनिक उपज \qty{1}{m^2}?
  \item कौन सा पैनल क्षेत्र परिवार को कवर करता है \qty{12.6}{kW.h} दिन?
        दक्षिण की छत प्रदान करती है \qty{20}{m^2}: क्या यह सही बैठता है?
  \item दिसंबर में समतुल्य गिर जाता है \qty{1.5}{h}: द
        पैनलों का उत्पादन, आवश्यकता का उसका हिस्सा, और क्या भरता है
        अंतर?
  \item पैनलों के निर्माण में लगभग लागत आती है \qty{700}{kW.h} प्रति वर्ग \omterm{def:g10:orders-of-magnitude:unit}{मीटर}: ऊर्जा प्रश्न 11 का उपयोग करते हुए पेबैक समय?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The cereal bowl.}
\begin{enumerate}[resume]
  \item एक जनरेटर साइकिल उपज देती है \qty{100}{W} बिजली का:
        पैडल चलाने में कितने घंटे लगते हैं \qty{12.6}{kW.h} ---कितने
        लोग पैडल मार रहे हैं \qty{24}{h} बिना रुके?
  \item मांसपेशियां दौड़ती हैं $\eta \approx 0.25$: खाना ऊर्जा पीछे
        वो \qty{12.6}{kW.h}, मेगाजूल में और अनाज के कटोरे में?
  \item पैडल वाले दिन की लागत (कटोरे पर $0.50$ यूरो), विरुद्ध
        उसी दिन के लिए ग्रिड कितना शुल्क लेता है?
  \item ए \qty{5}{GJ} बिजली का बोल्ट कितने गृह-दिवस के बराबर होता है?
        इसे देखते हुए \qty{100}{\micro s} \omterm{def:g10:signals-and-waves:period}{अवधि}, यह अभी भी क्यों हो सकता है
