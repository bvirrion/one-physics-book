\chapter{सापेक्ष गति}\label{ch:g10:relative-motion}

एक यात्री कॉफ़ी पीते हुए हाई-स्पीड ट्रेन के सामने की ओर चल रहा है
हाथ, इत्मीनान से चलने की गति से। ट्रैकसाइड ऑब्जर्वर उसी को देखता है
ऊपर यात्री \qty{300}{km/h}. दोनों माप सही हैं. न ही
यात्री की ``गति'' है, क्योंकि गति जैसी कोई चीज़ नहीं है
यह अकेले शरीर की संपत्ति नहीं है, बल्कि शरीर की संपत्ति है \emph{और} का मुद्दा
इसका वर्णन करने के लिए दृश्य को चुना गया। यह अध्याय उस दृष्टिकोण को सटीक बनाता है
--- और आपको एक अच्छा चुनना सिखाता है।

\section{गति सापेक्ष है}

\begin{definition}[संदर्भ फ्रेम]\label{def:g10:relative-motion:frame}
ए \emph{संदर्भ फ्रेम}\index{संदर्भ फ्रेम} एक कठोर शरीर है ---
ज़मीन, एक रेलगाड़ी, संपूर्ण पृथ्वी --- जिसके सापेक्ष
स्थितियों को दर्ज किया जाता है, साथ ही उनकी तारीख बताने वाली एक घड़ी भी। एक शरीर है
\emph{आराम से}\index{आराम से} एक फ्रेम में जब उसकी स्थिति उस फ्रेम में होती है
समय के साथ नहीं बदलता; अन्यथा यह उस फ्रेम में गतिमान है।
\end{definition}

\begin{definition}[प्रक्षेपवक्र]\label{def:g10:relative-motion:trajectory}
 \emph{प्रक्षेपवक्र}\index{प्रक्षेपवक्र} किसी दिए गए शरीर का \omterm{def:g10:relative-motion:frame}{संदर्भ फ्रेम} उस फ्रेम में उसकी क्रमिक स्थितियों से बना वक्र है: a
सीधी रेखा, एक वृत्त, एक चाप\dots{} प्रक्षेप पथ ही समझ में आता है
एक बार फ़्रेम का नाम रख दिया जाए।
\end{definition}

\begin{example}[ट्रेन पर चलने वाला]\label{ex:g10:relative-motion:walker}
एक ट्रेन यात्रा करती है \qty{300}{km/h} जमीन के सापेक्ष जबकि ए
यात्री सामने की ओर चलता है \qty{4}{km/h} ट्रेन के सापेक्ष.
\begin{itemize}
  \item में \emph{रेलगाड़ी} फ़्रेम: यात्री आगे बढ़ता है \qty{4}{km/h}
        गलियारे नीचे; सीटें हैं आराम; बाहर पेड़ दौड़ते हैं
        पीछे की ओर \qty{300}{km/h}.
  \item में \emph{मैदान} फ्रेम: सीटें चलती रहती हैं \qty{300}{km/h},
        यात्री पर \qty{304}{km/h}, और पेड़ पर हैं आराम.
\end{itemize}
एक ही दृश्य, दो फ़्रेम, दो पूरी तरह से अलग विवरण --- और
ट्रेन के अंदर किया गया कोई भी प्रयोग उनमें से किसी एक को ``द'' घोषित नहीं कर सकता
असली वाला''।
\end{example}

\begin{proposition}[गति की सापेक्षता]\label{prop:g10:relative-motion:relativity}
की अवस्था आराम या किसी पिंड की गति, इसकी \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}, और इसकी गति सब कुछ
पर निर्भर है \omterm{def:g10:relative-motion:frame}{संदर्भ फ्रेम} उनका वर्णन करने के लिए चुना गया।
\end{proposition}

\begin{proof}
एक उदाहरण प्रत्येक दावे का निपटारा करता है। का बैठा हुआ यात्री
\cref{ex:g10:relative-motion:walker} पर है आराम ट्रेन के फ्रेम में और
पर चलता है \qty{300}{km/h} में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} (आराम और गति पर निर्भर करता है
फ़्रेम)। के लिए \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}, साइकिल के पहिये पर वाल्व देखें: एक
साइकिल चलाने वाला प्रेक्षक देखता है कि यह हब के चारों ओर एक घेरा बना रहा है
सड़क पर \omterm{def:g10:universal-gravitation:weight}{खड़ा} एक पर्यवेक्षक इसे मेहराबों के अनुक्रम का पता लगाता हुआ देखता है (a
वक्र को ए कहा जाता है \emph{चक्रज}\index{चक्रज}) --- नीचे दिया गया चित्र देखें।
\end{proof}

\begin{remark}
ध्यान दें क्या करता है \emph{नहीं} दो चित्रों में परिवर्तन: वाल्व,
पहिया, भौतिकी. केवल विवरण बदलता है. पूछ रहे हैं ``वाल्व है
सच में एक वृत्त में जा रहे हैं?'' यह पूछने जैसा है कि क्या कोई पर्वत है
``वास्तव में'' बाईं ओर: यह इस बात पर निर्भर करता है कि आप कहां खड़े हैं।
\end{remark}

\section{औसत गति}

\begin{definition}[औसत गति]\label{def:g10:relative-motion:speed}
 \emph{औसत गति}\index{औसत गति} किसी शरीर का, किसी दिए गए में
\omterm{def:g10:relative-motion:frame}{संदर्भ फ्रेम}, दूरी है $d $ यह विभाजित उस फ्रेम में यात्रा करता है
यात्रा का समय $ t $:
\[
v = \frac{d}{t} .
\]
साथ $ d $ में एम-eters और $ t $ सेकंड में, $ v $ में है एम-eters प्रति सेकंड
(\unit{m/s}); रोजमर्रा की जिंदगी किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} प्रति घंटे को प्राथमिकता देती है (\unit{km/h}).
\end{definition}

\begin{method}[मी/से और किमी/घंटा के बीच कनवर्ट करना]\label{met:g10:relative-motion:convert}
एक किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} है \qty{1000}{m} और एक घंटा है \qty{3600}{s}, इसलिए
$\qty{1}{m/s} = \qty{3.6}{km/h}$:
\begin{itemize}
  \item से \unit{m/s} को \unit{km/h}: \emph{गुणा} द्वारा $3.6$;
  \item से \unit{km/h} को \unit{m/s}: \emph{विभाजित करना} द्वारा $3.6$.
\end{itemize}
स्वच्छता जांच: की संख्या \unit{km/h} हमेशा है \emph{बड़ा} का
दो.
\end{method}

\begin{example}[जानने लायक कुछ गति]\label{ex:g10:relative-motion:convert}
तेज़ चाल: $\qty{1.5}{m/s} \times 3.6 = \qty{5.4}{km/h}$. एक शहरी कार
पर \qty{54}{km/h}: $54 / 3.6 = \qty{15}{m/s}$ --- के बारे में \qty{15}{m}
प्रत्येक सेकंड को कवर किया गया, अंकुश से हटने से पहले याद रखने लायक।
हवा में ध्वनि: $\qty{340}{m/s} \times 3.6 = \qty{1224}{km/h}$.
\end{example}

\begin{example}[एक ब्रेक के साथ पदयात्रा]\label{ex:g10:relative-motion:hike}
एक यात्री कवर करता है \qty{12}{km} में \qty{2}{h} फिर चलने का टिकी हुई है के लिए
\qty{1}{h}, फिर कवर करता है \qty{6}{km} अंतिम में \qty{1.5}{h}. \omterm{def:g10:relative-motion:speed}{औसत गति} पूरी सैर के दौरान:
\[
v = \frac{d}{t} = \frac{12 + 6}{2 + 1 + 1.5}
= \frac{\qty{18}{km}}{\qty{4.5}{h}} = \qty{4.0}{km/h}.
\]
औसत मायने रखता है \emph{सभी} बीता हुआ समय, विराम शामिल --- यह है
आम तौर पर अलग-अलग पैरों की गति का औसत नहीं होता है।
\end{example}

\section{एक फ्रेम चुनना: जमीन, भूकेन्द्रित, सूर्यकेन्द्रित}

चूँकि प्रत्येक फ्रेम वैध है, हम वही चुनते हैं जिसमें हम गति करते हैं
देखभाल सबसे सरल दिखती है। तीन विकल्प अधिकांश भौतिकी को कवर करते हैं
यह किताब.

\begin{definition}[तीन मानक फ़्रेम]\label{def:g10:relative-motion:frames}
\begin{itemize}
  \item  \emph{ज़मीन का ढाँचा}\index{ज़मीन का ढाँचा} (या
        \emph{प्रयोगशाला फ्रेम}\index{प्रयोगशाला फ्रेम}) से जुड़ा हुआ है
        पृथ्वी की सतह जहाँ प्रयोग होता है। यह है
        रोजमर्रा की गति के लिए सही ढांचा: गिरती वस्तुएं, वाहन,
        खेल, प्रयोगशाला बेंच।
  \item  \emph{भूकेंद्रिक फ़्रेम}\index{भूकेंद्रिक फ़्रेम} केन्द्रित है
        पृथ्वी पर लेकिन करता है \emph{नहीं} इसके साथ घूमें (इसकी दिशाएँ)।
        दूर के तारों की ओर इशारा करें)। यह चंद्रमा के लिए सही फ्रेम है और
        कृत्रिम के लिए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों, कौन \omterm{def:g10:universal-gravitation:satellite}{कक्षा} समग्र रूप से पृथ्वी और
        इसके दैनिक चक्र को अनदेखा करें।
  \item  \emph{हेलिओसेंट्रिक फ़्रेम}\index{हेलिओसेंट्रिक फ़्रेम} है
        सूर्य पर केन्द्रित, दिशाएँ पुनः दूर पर निर्धारित
        सितारे. यह ग्रहों के लिए सही फ्रेम है: प्रत्येक तब अनुसरण करता है
        एक सरल, लगभग गोलाकार \omterm{def:g10:universal-gravitation:satellite}{कक्षा}.
\end{itemize}
\end{definition}

\begin{method}[फ़्रेम का चयन करना]\label{met:g10:relative-motion:choose}
उन पिंडों के नाम बताएं जिनकी गति का आप वर्णन करना चाहते हैं, फिर फ़्रेम चुनें
वह गति सबसे सरल है:
\begin{enumerate}
  \item पृथ्वी की सतह के निकट गति, कुछ मिनट तक चलने वाली --- ज़मीन
        ढाँचा;
  \item कक्षाओं \emph{आस-पास} पृथ्वी (चंद्रमा, \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों) --- भूकेन्द्रित
        ढाँचा;
  \item कक्षाओं सूर्य के चारों ओर (ग्रह, धूमकेतु) --- \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम}.
\end{enumerate}
किसी प्रस्ताव का वर्णन करने से पहले हमेशा अपनी पसंद बताएं: a \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} या ए
इसके फ्रेम के बिना उद्धृत गति अर्थहीन है
(\cref{prop:g10:relative-motion:relativity}).
\end{method}

\begin{remark}[कौन सही है, टॉलेमी या कॉपरनिकस?]\label{rem:g10:relative-motion:history}
``सूर्य पूर्व में उगता है'' और ``पृथ्वी पूर्व की ओर घूमती है''
एक ही अवलोकन को दो फ़्रेमों में रिपोर्ट किया गया --- पहले में ज़मीन
ढाँचा, जहां सूर्य आकाश में घूमता है, दूसरे में भूकेन्द्रित
ढाँचा, जहां पर्यवेक्षक को पृथ्वी की धुरी के चारों ओर ले जाया जाता है। के लिए
सदियों से तख्ते स्वयं युद्ध के मैदान थे: टॉलेमी के
पृथ्वी-केंद्रित विवरण ने आकाश को सटीक रूप से ट्रैक किया लेकिन लूप की आवश्यकता थी
ग्रहों का अनुसरण करने के लिए लूपों पर, जबकि सूर्य-केंद्रित वर्णन
1543 में कोपरनिकस द्वारा पुनर्जीवित करके प्रत्येक ग्रह को एक सरल निकट-वृत्त बनाया गया
मंगल ग्रह के अजीब पिछड़े चक्रों को महज एक फ्रेम प्रभाव के रूप में समझाया गया ---
पृथ्वी एक धीमे ग्रह से आगे निकल रही है। आधुनिक फैसला वैसा नहीं है
फ़्रेम सत्य है और \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} असत्य है, लेकिन वे समान रूप से नहीं हैं
\emph{सुविधाजनक}; कुछ फ़्रेमों को प्राथमिकता देने के लिए एक गहरा मानदंड
(जड़त्व वाले) जड़ता के सिद्धांत के साथ आते हैं
\cref{ch:g10:inertia}.
\end{remark}

\section{एक रेखा के अनुदिश गतियाँ लिखना}

क्या जोड़ता है \qty{4}{km/h} वॉकर (ट्रेन फ्रेम) के लिए
\qty{304}{km/h} (\omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा})? एक पंक्ति में, उत्तर स्पष्ट है
जोड़.

\begin{proposition}[एक आयाम में गति की संरचना]\label{prop:g10:relative-motion:compose}
मान लीजिए कि एक शरीर किसी गतिशील सहारे (वॉकवे, ए) पर एक रेखा के साथ चलता है
रेलगाड़ी, बहता पानी), सहारा स्वयं एक ही रेखा पर फिसल रहा है
जमीन के सापेक्ष. रेखा पर सकारात्मक दिशा का चयन करना तथा
एक संकेत के साथ गति की गिनती,
\[
v_{\text{body/ground}} \;=\; v_{\text{body/support}} \;+\;
v_{\text{support/ground}} .
\]
शब्दों में: एक ही दिशा में गति जोड़ें; विपरीत गति
दिशाएँ घटाना.
\end{proposition}

\begin{proof}
एक सेकंड के दौरान सहारा शरीर को ढोता है $ v_{\text{support/ground}}$
एम-eters रेखा के साथ, और शरीर आगे बढ़ता है $ v_{\text{body/support}}$
एम-eters समर्थन के साथ. दोनों विस्थापन एक ही रेखा पर होते हैं,
तो उस सेकंड में जमीन पर शरीर का कुल विस्थापन उनका है
बीजगणितीय योग --- जो जमीन के सापेक्ष इसकी गति है।
\end{proof}

\begin{example}[गतिशील पथ]\label{ex:g10:relative-motion:walkway}
एक हवाईअड्डे का पैदल मार्ग चलता है \qty{1.0}{m/s}. एक यात्री इस पर चलता है
\qty{1.5}{m/s} बेल्ट के सापेक्ष, यात्रा की दिशा में: गति
जमीन के सापेक्ष $1.5 + 1.0 = \qty{2.5}{m/s}$. ग़लत चलना
उसी गति से: $1.5 - 1.0 = \qty{0.5}{m/s}$, अभी भी (धीरे से)
बेल्ट के विपरीत प्रगति करना। बेल्ट पर स्थिर खड़े रहना:
\qty{1.0}{m/s} ज़मीन के सापेक्ष, \qty{0}{m/s} के सापेक्ष
बेल्ट --- पर आराम और एक ही समय में गति में, दो अलग-अलग में
तख्ते.
\end{example}

\begin{remark}[एक साथ दो दिशाएँ]\label{rem:g10:relative-motion:twodim}
जब दो गतियाँ होती हैं \emph{नहीं} उसी पंक्ति में --- एक नाव का लक्ष्य
सीधे नदी के उस पार, जबकि धारा उसे नीचे की ओर धकेलती है --- प्रत्येक
गति अपना मार्ग स्वतंत्र रूप से चलाती है, और वेग इस प्रकार बनते हैं
तीर सिरे से पूंछ तक रखे गए, बिल्कुल गणित के सदिशों की तरह
आयतन. नाव अभी भी अपने द्वारा निर्धारित समय पर सुदूर तट पर पहुँच जाती है
अकेले ही गति पार कर लेता है, लेकिन यह अपने लक्ष्य बिंदु से नीचे की ओर उतरता है, और
इसका ग्राउंड-फ़्रेम \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} एक तिरछी सीधी रेखा है. पूर्ण वेक्टर
उपचार बाद के अध्याय में आता है; यहाँ हम केवल चित्र पढ़ते हैं।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:relative-motion:1}
एक यात्री ट्रेन में सो रहा है \qty{100}{km/h}. राज्य
चाहे यात्री चालू हो आराम या गति में, और किस गति से,
के सापेक्ष: गाड़ी; आधार; ट्रैक के बगल में एक पेड़; एक
आने वाली ट्रेन यात्रा कर रही है \qty{100}{km/h} दूसरे तरीके से. फिर वर्णन करें
यात्री की ट्रेन के फ्रेम में पेड़ की गति।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:relative-motion:2}
बदलना: \qty{72}{km/h} और \qty{108}{km/h} में \unit{m/s};
\qty{25}{m/s} और \qty{340}{m/s} में \unit{km/h}. प्रत्येक उत्तर की जाँच करें
के विवेक नियम के साथ \cref{met:g10:relative-motion:convert}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:relative-motion:3}
एक साइकिल चालक कवर करता है \qty{36}{km} में \qty{1}{h}~\qty{30}{min}; एक धावक
कवर \qty{100}{m} में \qty{10.0}{s}. दोनों की गणना करें \omterm{def:g10:relative-motion:speed}{औसत गति} में
\unit{m/s} और में \unit{km/h}. कौन तेज़ है, और उत्तर देता है
आपको आश्चर्य हुआ?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:relative-motion:4}
एक चलता-फिरता रास्ता चलता है \qty{0.9}{m/s}. एक यात्री चल रहा है
\qty{1.4}{m/s} बेल्ट के सापेक्ष. यात्री की सापेक्ष गति बताएं
बेल्ट की दिशा में, उसके विपरीत, और कब चलते समय जमीन पर गिरना
बेल्ट पर अभी भी \omterm{def:g10:universal-gravitation:weight}{खड़ा} है.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:relative-motion:5}
प्रत्येक गति के लिए, जमीन के बीच सबसे सुविधाजनक फ्रेम चुनें,
भूकेन्द्रित और सूर्यकेन्द्रित, और एक वाक्य में उचित ठहराएँ: एक कलम गिर गई
चलती ट्रेन के अंदर (बोनस: इससे भी बेहतर फ्रेम उपलब्ध है);
\omterm{def:g10:universal-gravitation:satellite}{कक्षा} अंतर्राष्ट्रीय अंतरिक्ष स्टेशन का; एक मंगल ग्रह वर्ष; ए
\qty{100}{m} तेज़ी से दौड़ना
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:relative-motion:6}
एक रेलगाड़ी चलती है \qty{108}{km/h}. एक यात्री सामने की ओर चलता है
\qty{1.5}{m/s} ट्रेन के सापेक्ष.
\begin{enumerate}
  \item ट्रेन की गति को परिवर्तित करें \unit{m/s}, तो दे दो
        जमीन के सापेक्ष यात्री की गति, की ओर चलना
        आगे और पीछे की ओर.
  \item गाड़ी है \qty{60}{m} लंबा। तक पैदल चलना कितना लंबा है
        फ्रंट टेक, और यात्री कितनी दूरी के सापेक्ष यात्रा करता है
        इस दौरान जमीन?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:relative-motion:7}
शांत जल में नाव की गति होती है \qty{5.0}{m/s}; एक नदी बहती है
\qty{2.0}{m/s}. नाव यात्रा करती है \qty{420}{m} नीचे की ओर, चारों ओर मुड़ता है,
और वापस आ जाता है.
\begin{enumerate}
  \item प्रत्येक पैर पर किनारे के सापेक्ष नाव की गति की गणना करें, और
        प्रत्येक चरण की अवधि.
  \item की गणना करें \omterm{def:g10:relative-motion:speed}{औसत गति} राउंड ट्रिप पर. यह कम क्यों है?
        से \qty{5.0}{m/s}, भले ही करंट एक पैर पर मदद करता हो
        बिल्कुल उतना ही जितना यह दूसरे पर बाधा डालता है?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:relative-motion:8}
दो ट्रेनें समानांतर पटरियों पर एक दूसरे से गुजरती हैं: ट्रेन ए (\qty{180}{m}
लंबा) पर \qty{25}{m/s}, ट्रेन बी (\qty{90}{m} लंबा) पर \qty{20}{m/s}.
\begin{enumerate}
  \item वे विपरीत दिशाओं में यात्रा करते हैं। B की गति कितनी है?
        ए का फ्रेम, और पूरी क्रॉसिंग कितने समय तक चलती है
        (नाक मिलने से लेकर पूँछ अलग होने तक)?
  \item यदि वे एक ही दिशा में यात्रा करते हैं तो वही प्रश्न।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:relative-motion:9}
एक साइकिल सीधी सड़क पर चलती है। का वर्णन करें \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} का
टायर वाल्व में: साइकिल चालक का फ्रेम; \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}; और फ्रेम
वाल्व का ही. का भी वर्णन करें \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} पहिये के हब में
द \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}. कोई गणना नहीं --- प्रत्येक वक्र को नाम दें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:relative-motion:10}
पृथ्वी का भूमध्य रेखा लगभग है \qty{40000}{km} लंबा है, और पृथ्वी घूमती है
एक बार अंदर \qty{24}{h}.
\begin{enumerate}
  \item भूमध्य रेखा पर एक बिंदु की गति की गणना करें भूकेन्द्रित
        ढाँचा, में \unit{km/h} और में \unit{m/s}.
  \item प्रत्येक को एक वाक्य में स्पष्ट करें: हम इस गति को महसूस क्यों नहीं करते हैं, और
        क्यों ``सूर्य उगता है'' और ``पृथ्वी घूमती है'' एक ही रिपोर्ट करते हैं
        अवलोकन.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:relative-motion:11}
चलती हुई एस्केलेटर पर स्थिर खड़े होकर, ऊपर की ओर यात्रा करनी पड़ती है \qty{60}{s}.
ऊपर चलने पर वही एस्केलेटर बंद हो जाता है \qty{90}{s}. कब तक
क्या चलती हुई एस्केलेटर पर चढ़ने में समय लगता है? (गति के साथ काम करें, नहीं
समय.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:relative-motion:12}
एक ड्राइवर एक ही दूरी को दो बार तय करता है: आउटबाउंड पर \qty{60}{km/h},
वापस पर \qty{90}{km/h}.
\begin{enumerate}
  \item दिखाओ कि \omterm{def:g10:relative-motion:speed}{औसत गति} ओवर राउंड ट्रिप है
        \qty{72}{km/h} --- नहीं \qty{75}{km/h}.
  \item स्पष्ट करें कि सरल उत्तर कहां गलत हो जाता है, और बताएं कि कौन सा चरण गलत है
        औसत पर हावी है.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:relative-motion:13}
दो साइकिल चालक शुरू होते हैं \qty{35}{km} अलग हो जाओ और एक दूसरे की ओर बढ़ो, एक पर
\qty{15}{km/h}, दूसरे पर \qty{20}{km/h}.
\begin{enumerate}
  \item के फ्रेम में दूसरे साइकिल चालक की गति का वर्णन करें
        पहले, और जब वे मिलें तो निष्कर्ष निकालें।
  \item बैठक में प्रत्येक ने कितनी दूर तक यात्रा की है?
  \item यदि इसके बजाय वे एक ही दिशा में (तेज़ दिशा में) चलते हैं
        पीछे), पीछा कितनी देर तक चलता है?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:relative-motion:14}
एक तैराक एक नदी पार करता है \qty{60}{m} चौड़ा, सीधे पार की ओर जा रहा है
\qty{1.2}{m/s} पानी के सापेक्ष; धारा प्रवाहित होती है
\qty{0.9}{m/s}.
\begin{enumerate}
  \item बताएं कि तैराक द्वारा क्रॉसिंग का समय क्यों निर्धारित किया जाता है
        अकेले गति, और इसकी गणना करें।
  \item तैराक कितनी दूर तक धारा के अनुकूल उतरता है?
  \item समकोण त्रिभुज का उपयोग करके, तैराक की गति ज्ञात कीजिए ज़मीन
        ढाँचा और पथ की लंबाई वास्तव में जमीन के ऊपर तैरती है।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:relative-motion:15}
पृथ्वी कक्षाओं सूर्य त्रिज्या के निकट-वृत्त पर
\qty{1.50e11}{m}, एक साल में ($\num{3.16e7}$ सेकंड)।
\begin{enumerate}
  \item में पृथ्वी की गति की गणना करें \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम}, में
        \unit{km/s}.
  \item इसकी तुलना भूमध्य रेखा की घूमने की गति से करें
        \cref{exo:g10:relative-motion:10}.
  \item पृथ्वी से देखने पर, मंगल कभी-कभी रुकता और बहता हुआ प्रतीत होता है
        कुछ हफ़्तों तक सितारों के बीच पिछड़ गया। दो में स्पष्ट करें
        वाक्य, इस अध्याय की शब्दावली का उपयोग करते हुए, क्यों सूर्यकेन्द्रित
        ढाँचा रहस्य को सुलझाता है.
\end{enumerate}
\end{exercise}

\section{समस्या: रास्ता, नदी और उपग्रह}

\begin{problem}[{सप्ताहांत समस्या---चलता हुआ रास्ता, नदी और
उपग्रह: कौन चलता है, किसके सापेक्ष, और कुर्सी यात्री की
अंतिम गति रिपोर्ट}]
\label{pb:g10:relative-motion:1}
एक हवाई अड्डा, एक नौका पारगमन, और गोधूलि बेला में फिसलता हुआ एक चमकीला बिंदु
आकाश: तीन रोजमर्रा के दृश्य, एक प्रश्न तीन बार पूछा गया --- कौन
चलता है, किसके सापेक्ष? प्रत्येक भाग अपना फ़्रेम चुनता है, गणना करता है, और
दृष्टिकोण के साथ उत्तर बदलते हुए देखता है; समापन का योग है
कितनी तेजी से \emph{आप} अभी चल रहे हैं, बिल्कुल शांत बैठे हैं।

\medskip
\noindent\textbf{Part I --- The walkway race.}
एक एयरपोर्ट वॉकवे है \qty{90}{m} लंबा और लुढ़कता है \qty{1.0}{m/s}. ए
यात्री चलता है \qty{1.5}{m/s} (जो कुछ भी पैरों के नीचे है उसके सापेक्ष)।
\begin{enumerate}
  \item एक खड़े यात्री को ले जाने में पैदल मार्ग को कितना समय लगता है?
        अंत से अंत तक?
  \item चलते-फिरते यात्री को चलने में कितना समय लगता है?
        बेल्ट, इसकी दिशा में?
  \item वॉकवे के बगल में, निश्चित मंजिल पर चलने में कितनी देर लगेगी?
  \item एक मसखरा बेल्ट पर गलत रास्ते पर चलता है \qty{1.5}{m/s}
        इसके सापेक्ष. जमीन के सापेक्ष गति, और प्राप्त करने का समय
        सुदूर छोर से वापस प्रवेश द्वार तक?
  \item एक बच्चा बिल्कुल गलत रास्ते पर दौड़ता है \qty{1.0}{m/s} रिश्तेदार
        बेल्ट तक. में बच्चे की गति का वर्णन करें \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा},
        और उस जिम मशीन का नाम बताएं जिसे इसने पुनः आविष्कृत किया है।
  \item प्रश्न 1-5 को दो-स्तंभ तालिका में सारांशित करें: प्रत्येक प्रस्तावक का
        बेल्ट फ्रेम में और में गति \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}. कौन सा कॉलम
        क्या यात्रियों के पैर महसूस होते हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The river.}
एक नौका एक नदी को पार करती है \qty{120}{m} चौड़ा। के सापेक्ष इसकी गति
पानी है \qty{2.0}{m/s}; धारा प्रवाहित होती है \qty{1.5}{m/s}. पायलट
पहला लक्ष्य सीधे विपरीत किनारे पर होता है।
\begin{enumerate}[resume]
  \item क्रॉसिंग में कितना समय लगता है?
  \item नौका धारा के अनुकूल कितनी दूर उतरती है?
  \item समकोण त्रिभुज का उपयोग करके, नौका की गति की गणना करें ज़मीन
        ढाँचा और इसके ग्राउंड-फ़्रेम की लंबाई \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}. (ए
        संख्याओं का परिचित त्रिगुण प्रकट होता है।)
  \item नौका का वर्णन करें \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} पानी के फ्रेम में और में
        \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}. उनमें क्या समानता है और क्या अंतर है?
  \item एक तैराक जिसकी गति पानी के सापेक्ष होती है \qty{1.0}{m/s}
        एक ही बैंक से प्रस्थान करता है, सीधे लक्ष्य की तरह
        नौका. क्रॉसिंग में कितना समय लगता है, और कितनी दूरी नीचे की ओर है
        क्या तैराक उतरता है? तैराक के ग्राउंड-फ़्रेम की गणना करें
        गति (फिर से समकोण त्रिभुज)। धीमा तैराक क्यों करता है
        नौका से अधिक दूर बहें?
  \item एक लट्ठा नौका के पार तैरता है। में इसकी गति बताइये \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}
        और पानी के फ्रेम में. लॉग किस फ़्रेम में चल रहा है?
        टिप्पणी।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} pass.}
शाम के समय, अंतर्राष्ट्रीय अंतरिक्ष स्टेशन कुछ में आकाश को पार करता है
मिनट. यह कक्षाओं की ऊंचाई पर \qty{400}{km}; पृथ्वी की त्रिज्या
है \qty{6.37e6}{m}, और एक \omterm{def:g10:universal-gravitation:satellite}{कक्षा} लेता है \qty{5.56e3}{s} (के बारे में
\qty{93}{min}).
\begin{enumerate}[resume]
  \item किस मानक फ्रेम में स्टेशन की गति सबसे सरल है?
        उस फ़्रेम में इसकी गति की गणना करें \unit{km/s}.
  \item दो वाक्यों में स्पष्ट करें कि ऐसा क्यों है \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} का वर्णन करता है
        स्टेशन अजीब ढंग से --- जमीन के ऊपर इसके ट्रैक का क्या होता है
        एक के बीच \omterm{def:g10:universal-gravitation:satellite}{कक्षा} और अगला?
  \item स्टेशन की कक्षीय गति की तुलना भूमध्य रेखा के घूर्णन से करें
        गति ($\approx \qty{0.46}{km/s}$): किस कारक से है
        स्टेशन तेजी से?
  \item एक भूस्थैतिक \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} कक्षाओं त्रिज्या पर \qty{4.22e7}{m} में
        ठीक एक दिन ($\qty{8.64e4}{s}$). में इसकी गति की गणना करें
        \omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम}, फिर इसकी गति का वर्णन करें \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}.
        दोनों उत्तरों को एक वाक्य में मिलाएँ।
  \item में \omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम} चंद्रमा चारों ओर एक निकट-वृत्त का पता लगाता है
        पृथ्वी के बारे में $27$ दिन. इसे शब्दों में रेखांकित करें \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} में
        द \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम} कुछ महीनों में.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Who moves?}
\begin{enumerate}[resume]
  \item ``सूर्य पूर्व में उगता है।'' ``पृथ्वी की ओर घूमती है
        पूर्व।'' वह ढाँचा बताएं जिसमें प्रत्येक वाक्य इसका वर्णन करता है
        अवलोकन करें, और समझाएं कि इनमें से कोई भी गलत क्यों नहीं है।
  \item एक गंभीर पैराग्राफ में: टॉलेमी के फ्रेम ने क्या अच्छा किया, क्या
        क्या कोपरनिकस के फ्रेम को सरल बनाया गया (मंगल के पिछड़े होने का उल्लेख करें)।
        लूप्स), और आधुनिक फैसला क्यों के बारे में है \emph{सुविधा}
        एक फ्रेम के सच होने के बजाय?
  \item कुर्सी यात्री की गति रिपोर्ट। आप अभी भी एक में बैठते हैं
        भूमध्य रेखा के पास कुर्सी. में अपनी गति दीजिये \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा},
        में \omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम} ($\approx \qty{0.46}{km/s}$), और इसमें \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम} ($\approx \qty{30}{km/s}$). फिर गणना करें
        जितनी दूरी आप तय करते हैं \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम} एक के दौरान
        \qty{90}{min} फिल्म, और अध्याय के साथ समस्या का समापन करें
        एक वाक्य नैतिक.
\end{enumerate}
\end{problem}
