\chapter{परिमाण के आदेश: ब्रह्मांड को मापना}\label{ch:g10:orders-of-magnitude}

भौतिकी वहां शुरू होती है जहां राय समाप्त होती है: माप के साथ। लेकिन बातें
भौतिकी के माप एक बेतुकी सीमा तक फैले हुए हैं --- एक परमाणु नाभिक के बारे में है
\qty{e-15}{m} पार, अवलोकनीय ब्रह्मांड के बारे में \qty{e27}{m}: ए
का कारक $10^{42}$, किसी भी मानसिक तस्वीर से कहीं परे।
यह अध्याय टूलकिट स्थापित करता है जो सीमा को नियंत्रित करता है --- इकाइयां,
दस की शक्तियाँ, \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} --- और प्रत्येक पर दो आदतें प्रयोग की जाती हैं
इसके बाद वाला पेज: लिखें \omterm{def:g10:orders-of-magnitude:unit}{इकाई}, और जाँच करें \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार}.

\section{इकाइयाँ: माप का व्याकरण}

\begin{definition}[भौतिक मात्राएँ और SI इकाइयाँ]\label{def:g10:orders-of-magnitude:unit}
ए \emph{भौतिक मात्रा}\index{भौतिक मात्रा} की संपत्ति है
वह संसार जिसे मापा जा सकता है: एक लंबाई, एक अवधि, एक द्रव्यमान।
इसे मापने का अर्थ है किसी संदर्भ मात्रा को कितनी बार गिनना ---
द \emph{इकाई}\index{इकाई} --- इसमें फिट बैठता है.
\emph{इकाइयों की अंतर्राष्ट्रीय प्रणाली}\index{इकाइयों की अंतर्राष्ट्रीय प्रणाली} (एसआई) सात आधार इकाइयों को ठीक करता है, जिनमें से तीन इसे पूरा चलाती हैं
वॉल्यूम: द \emph{मीटर}\index{मीटर} (\unit{m}) लंबाई के लिए,
\emph{किलोग्राम}\index{किलोग्राम} (\unit{kg}) द्रव्यमान और के लिए
\emph{दूसरा}\index{दूसरा} (\unit{s}) समय के लिए. (अन्य चार ---
एम्पीयर, केल्विन, मोल, कैंडेला --- बिजली, गर्मी और जुड़ेंगे
रसायन विज्ञान कहानी में प्रवेश करें।)
\end{definition}

\begin{remark}[इकाई के बिना कोई संख्या परिणाम नहीं होती]\label{rem:g10:orders-of-magnitude:units}
``दूरी है $384$'' का कोई मतलब नहीं: \qty{384}{mm},
\qty{384}{km} और \qty{384}{Mm} तीन अलग-अलग दुनिया हैं (अंतिम)।
चंद्रमा से दूरी होती है)। 1999 में नासा ने खो दिया
\$327 मिलियन \emph{मंगल जलवायु ऑर्बिटर} क्योंकि एक प्रोग्राम की गणना की गई
थ्रस्टर बल पाउंड-बल में जबकि \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} उन्हें न्यूटन के रूप में पढ़ता है।
हाउस नियम, इस पंक्ति से पुस्तक के अंत तक: प्रत्येक मापा या
गणना की गई संख्या इसका वहन करती है \omterm{def:g10:orders-of-magnitude:unit}{इकाई}, गणना की प्रत्येक पंक्ति पर।
\end{remark}

\begin{definition}[एसआई उपसर्ग]\label{def:g10:orders-of-magnitude:prefixes}
ए \emph{उपसर्ग}\index{उपसर्ग} एक से चिपका हुआ \omterm{def:g10:orders-of-magnitude:unit}{इकाई} इसे a से गुणा करें
दस की निश्चित शक्ति:
\begin{center}
\begin{tabular}{llc@{\qquad\qquad}llc}
femto & f & $10^{-15}$ & centi & c & $10^{-2}$ \\
pico  & p & $10^{-12}$ & kilo  & k & $10^{3}$  \\
nano  & n & $10^{-9}$  & mega  & M & $10^{6}$  \\
micro & $\mu $ & $10^{-6}$ & giga & G & $10^{9}$ \\
milli & m & $10^{-3}$  & tera  & T & $10^{12}$ \\
\end{tabular}
\end{center}
इस प्रकार $\qty{1}{nm} = \qty{e-9}{m}$ और $\qty{1}{km} = \qty{e3}{m}$.
\end{definition}

\begin{example}[उपसर्ग पढ़ना]\label{ex:g10:orders-of-magnitude:prefixes}
हरे प्रकाश की तरंगदैर्घ्य लगभग होती है $\qty{500}{nm} = \qty{5.00e-7}{m}$; ए
मानव बाल के बारे में है $\qty{70}{\micro m} = \qty{7.0e-5}{m}$ मोटा; द
चंद्रमा परिक्रमा करता है $\qty{384}{Mm} = \qty{3.84e8}{m}$. एक बारिश की बूंद है
द्रव्यमान के बारे में $\qty{30}{mg} = \qty{3.0e-5}{kg}$ ---ध्यान दें कि एसआई आधार
\omterm{def:g10:orders-of-magnitude:unit}{इकाई} द्रव्यमान का, सात में से अकेला, एक वहन करता है \omterm{def:g10:orders-of-magnitude:prefixes}{उपसर्ग} पहले से।
\end{example}

\section{वैज्ञानिक संकेतन और महत्वपूर्ण आंकड़े}

\begin{definition}[वैज्ञानिक संकेतन]\label{def:g10:orders-of-magnitude:scinot}
एक नंबर अंदर है \emph{वैज्ञानिक संकेतन}\index{वैज्ञानिक संकेतन}
जब के रूप में लिखा जाता है
\[
a \times 10^{n}, \qquad 1 \leq a < 10, \quad n \text{ an integer.}
\]
कारक $ a $ है \emph{अपूर्णांश}\index{अपूर्णांश}, पूर्णांक
$ n $  \emph{प्रतिपादक}\index{प्रतिपादक}.
इस प्रकार $\num{149600000} = \num{1.496e8}$ और
$\num{0.00052} = \num{5.2e-4}$.
\end{definition}

\begin{proposition}[दस की शक्तियों के साथ कंप्यूटिंग]\label{prop:g10:orders-of-magnitude:powers}
सभी पूर्णांकों के लिए $ m $ और $ n $:
\[
10^{m} \times 10^{n} = 10^{m+n}, \qquad
\frac{1}{10^{n}} = 10^{-n}, \qquad
\left(10^{m}\right)^{n} = 10^{mn}.
\]
\end{proposition}

\begin{proof}[आंशिक प्रमाण]
सकारात्मक के लिए घातांक, कारकों की गणना करें: $10^{m} \times 10^{n}$ एक है
का उत्पाद $ m + n $ दसियों, और $(10^m)^n $ चेन $ n $ के समूह $ m $ दसियों.
लिखना $10^{-n}$ के लिए $1/10^{n}$ बिल्कुल यही परिपाटी है
सभी पूर्णांकों के लिए पहला नियम सत्य रखता है: यह बाध्य करता है
$10^{n} \times 10^{-n} = 10^{0} = 1$. साथी गणित
वॉल्यूम अध्ययन करता है \omterm{def:g10:orders-of-magnitude:scinot}{प्रतिपादक} पूर्ण नियम.
\end{proof}

\begin{example}[वैज्ञानिक संकेतन में गुणा करना]\label{ex:g10:orders-of-magnitude:scinot}
द्रव्यमान के आधार पर कितनी पृथ्वियाँ सूर्य में समाती हैं? साथ
$ m_{\text{Sun}} = \qty{1.99e30}{kg}$ और
$ m_{\text{Earth}} = \qty{5.97e24}{kg}$:
\[
\frac{\num{1.99e30}}{\num{5.97e24}}
= \frac{1.99}{5.97} \times 10^{30-24}
= 0.333 \times 10^{6}
= \num{3.33e5}.
\]
मंटिसास और घातांक अलग से संभाला जाता है; अंतिम उत्तर है
पुनः सामान्यीकृत किया गया ताकि \omterm{def:g10:orders-of-magnitude:scinot}{अपूर्णांश} वापस अंदर आ जाता है $[1, 10)$.
\end{example}

\begin{definition}[महत्वपूर्ण लोग]\label{def:g10:orders-of-magnitude:sigfig}
 \emph{महत्वपूर्ण लोग}\index{महत्वपूर्ण लोग} एक का
मापी गई संख्या वे अंक हैं जिनकी वास्तव में पुष्टि की गई है
माप: अग्रणी शून्य को छोड़कर सभी लिखित अंक। पीछे चल रहा है
शून्य गिनती --- $\qty{5.20e-3}{s}$ (तीन महत्वपूर्ण आंकड़े)
से अधिक का वादा करता है $\qty{5.2e-3}{s}$ (दो)। \omterm{def:g10:orders-of-magnitude:scinot}{वैज्ञानिक संकेतन} दिखाता है
गिनती एक नज़र में: \omterm{def:g10:orders-of-magnitude:scinot}{अपूर्णांश} बिल्कुल महत्वपूर्ण वहन करता है
आंकड़े.
\end{definition}

\begin{method}[कितने अंक रखना है]\label{met:g10:orders-of-magnitude:sigfig}
एक परिकलित परिणाम उसके अवयवों से अधिक सटीक नहीं हो सकता।
\begin{enumerate}
  \item किसी उत्पाद या भागफल में उतने ही रखें \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} जैसे
        द \emph{कम से कम} सटीक कारक.
  \item केवल अंतिम उत्तर को गोल करें, मध्यवर्ती चरणों को कभी नहीं।
  \item जब संदेह हो, तीन \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} काम कर रहा है
        इस पुस्तक की सटीकता.
\end{enumerate}
\end{method}

\begin{example}[पृथ्वी की परिधि]\label{ex:g10:orders-of-magnitude:sigfig}
पृथ्वी का व्यास है $ d = \qty{1.27e7}{m}$ (तीन \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग}),
तो इसकी परिधि है
$\pi d = \num{3.14159} \times \qty{1.27e7}{m} = \qty{3.99e7}{m}$ ---
तीन अंक, न कि आठ जो कैलकुलेटर प्रदर्शित करता है। लिखना
$\qty{39898227}{m}$ को ग्रह को जानने का दावा करेगा \omterm{def:g10:orders-of-magnitude:unit}{मीटर}.
\end{example}

\section{ब्रह्मांड की सीढ़ी}

\begin{definition}[आदेश का आकार]\label{def:g10:orders-of-magnitude:oom}
 \emph{आदेश का आकार}\index{आदेश का आकार} एक मात्रा का
इसके निकटतम दस की घात है: मात्रा को इस प्रकार लिखें
$ a \times 10^{n}$ साथ $1 \leq a < 10$ और ले लो $10^{n}$ अगर $ a < 5$,
$10^{n+1}$ अगर $ a \geq 5$. (सटीक सीमा परिपाटी कभी नहीं
मायने रखता है: परिमाण के क्रम का विवरण केवल भीतर के लिए ही होता है
दस का एक गुणनखंड.)
\end{definition}

\begin{example}[परिमाण के आदेश]\label{ex:g10:orders-of-magnitude:oom}
मनुष्य का जीवन लगभग चलता है $80$ वर्ष, यानी\
$80 \times \qty{3.16e7}{s} \approx \qty{2.5e9}{s}$: \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार}
$\qty{e9}{s}$ --- कुछ अरब सेकंड, उन्हें अच्छे से व्यतीत करें। एक इंसान है
$\qty{1.7}{m}$ कठिन काम $\qty{e0}{m}$. माउंट एवरेस्ट,
$\qty{8848}{m}$: आदेश $\qty{e4}{m}$, तब से $8.848 \geq 5$.
\end{example}

ऐसे अनुमानों को ढेर करके \omterm{def:g10:orders-of-magnitude:scinot}{प्रतिपादक} का मुख्य चित्र तैयार करता है
यह अध्याय: एक सीढ़ी जिसकी हर सीढ़ी दस गुना चौड़ी दुनिया है
नीचे वाले से.

\begin{definition}[खगोलीय इकाई एवं प्रकाश वर्ष]\label{def:g10:orders-of-magnitude:lightyear}
 \emph{खगोलीय इकाई}\index{खगोलीय इकाई}
($\qty{1}{au} = \qty{1.496e11}{m}$) औसत पृथ्वी है - सूर्य
दूरी. \emph{प्रकाश वर्ष}\index{प्रकाश वर्ष} (\unit{ly}) है
प्रकाश एक वर्ष में कितनी दूरी तय करता है? प्रकाश निर्वात में गति करता है
\[
c = \qty{3.00e8}{m/s},
\]
सबसे तेज़ गति प्रकृति अनुमति देती है --- इसलिए प्रकाश-वर्ष एक दूरी है,
एक समय नहीं.
\end{definition}

\begin{example}[प्रकाश वर्ष मीटर में]\label{ex:g10:orders-of-magnitude:ly}
एक वर्ष रहता है $365.25 \times 24 \times 3600 = \qty{3.156e7}{s}$
(सुखद रूप से करीब $\pi \times 10^{7}\,\unit{s}$). इस तरह
\[
\qty{1}{ly} = c \times \qty{3.156e7}{s}
= \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}.
\]
सूर्य के बाद सबसे निकटतम तारा प्रॉक्सिमा सेंटॉरी स्थित है
$\qty{4.2}{ly} \approx \qty{4.0e16}{m}$ दूर: प्रकाश, जो एक को पार करता है
पृथ्वी-सूर्य की दूरी आठ मिनट में, पहुंचने में चार साल लगेंगे
वहां से.
\end{example}

\begin{method}[परिमाण के क्रम से विवेक की जाँच]\label{met:g10:orders-of-magnitude:sanity}
किसी भी परिकलित परिणाम पर भरोसा करने से पहले:
\begin{enumerate}
  \item प्रत्येक इनपुट को उसके चारों ओर गोल करें \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार};
  \item का उपयोग करके अकेले दस की घातों के साथ गणना फिर से करें
        \cref{prop:g10:orders-of-magnitude:powers} --- यह लेता है
        सेकंड;
  \item तुलना करें: सटीक उत्तर को अनुमान से सहमत होना चाहिए
        दस या उससे अधिक के कारक के भीतर, और यह \omterm{def:g10:orders-of-magnitude:unit}{इकाई} सही होना चाहिए
        एक. यदि कोई भी चेक विफल हो जाता है, तो आगे बढ़ने से पहले त्रुटि का पता लगाएं
        पर.
\end{enumerate}
\end{method}

\begin{remark}[परेशान क्यों होना]\label{rem:g10:orders-of-magnitude:sanity}
एक कैलकुलेटर ख़ुशी से रिपोर्ट करेगा कि एक सेब का द्रव्यमान है
$\qty{4e8}{kg}$ --- यह बेहतर नहीं जान सकता. कोई भी फार्मूला इसकी पुष्टि नहीं करता
बेतुका; आदत होती है. भौतिक विज्ञानी पहले अनुमान लगाते हैं और बाद में गणना करते हैं:
अनुमान ग़लत दशमलव बिंदुओं को पकड़ता है, अदला-बदली की जाती है इकाइयां और
उल्टे भिन्न, अधिकांश गलत उत्तरों के पीछे की तीन त्रुटियाँ
हर स्तर.
\end{remark}

\section{माप और अनिश्चितता}

कोई भी उपकरण सही मूल्य नहीं पढ़ता है। कोई माप तभी ईमानदार होता है जब
यह यह भी बताता है कि यह कितनी दूर हो सकता है।

\begin{definition}[पूर्ण और सापेक्ष अनिश्चितता]\label{def:g10:orders-of-magnitude:uncertainty}
किसी मात्रा का माप $ x $ के रूप में बताया गया है
\[
x = x_{\text{m}} \pm \Delta x ,
\]
कहाँ $ x_{\text{m}}$ मापा मूल्य है और \emph{निरपेक्ष अनिश्चितता}\index{निरपेक्ष अनिश्चितता} $\Delta x > 0$, ठीक उसी प्रकार
\omterm{def:g10:orders-of-magnitude:unit}{इकाई}, प्रशंसनीय त्रुटि को सीमित करता है: सही मूल्य झूठ होने का दावा किया जाता है
के बीच $ x_{\text{m}} - \Delta x $ और $ x_{\text{m}} + \Delta x $.
\emph{सापेक्ष अनिश्चितता}\index{सापेक्ष अनिश्चितता} अनुपात है
$\Delta x / x_{\text{m}}$, आमतौर पर प्रतिशत में उद्धृत किया जाता है; यह वही है
शब्द ``सटीक'' उपाय.
\end{definition}

\begin{method}[एक स्नातक उपकरण पढ़ना]\label{met:g10:orders-of-magnitude:reading}
एक रूलर, एक मापने वाले सिलेंडर, एक डायल के लिए:
\begin{enumerate}
  \item निशान के निकटतम ग्रेजुएशन पढ़ें, आंख लंबवत
        पैमाना (झुकी हुई आँख पढ़ने को बदल देती है);
  \item सबसे छोटे ग्रेजुएशन का कम से कम आधा हिस्सा लें \omterm{def:g10:orders-of-magnitude:uncertainty}{निरपेक्ष
        अनिश्चितता} --- अधिक यदि वस्तु का किनारा या तरल का
        सतह धुंधली है;
  \item परिणाम को इस प्रकार लिखें $ x_{\text{m}} \pm \Delta x $ के साथ \omterm{def:g10:orders-of-magnitude:unit}{इकाई},
        और अनिश्चित अंक से आगे कोई अंक न रखें।
\end{enumerate}
\end{method}

\begin{example}[सापेक्ष परिशुद्धता]\label{ex:g10:orders-of-magnitude:reading}
ऊपर पढ़ने से पता चला है \omterm{def:g10:orders-of-magnitude:uncertainty}{सापेक्ष अनिश्चितता}
$\frac{0.5}{127} \approx 0.4\%$. वही शासक जो मापता है
$\qty{12}{mm}$ बोल्ट देता है $\frac{0.5}{12} \approx 4\%$: समान
उपकरण छोटी वस्तु पर दस गुना कम सटीक है। परिशुद्धता है
की एक संपत्ति \emph{माप}, औज़ार का नहीं.
\end{example}

\begin{remark}[छोटा निरपेक्ष, विशाल सापेक्ष --- और इसके विपरीत]\label{rem:g10:orders-of-magnitude:precision}
लेजर स्टेशन पृथ्वी-चंद्रमा की दूरी मापते हैं, $\qty{3.84e8}{m}$, को
के बारे में $\pm\qty{3}{cm}$: ए \omterm{def:g10:orders-of-magnitude:uncertainty}{सापेक्ष अनिश्चितता} का $10^{-10}$, बीच में
अब तक का सबसे तीव्र माप --- पूर्ण त्रुटि संख्या के साथ
शासक घमंड करेगा. हमेशा पूछिये \emph{किस के सापेक्ष}: जोड़ी
$(\Delta x,\ \Delta x / x_{\text{m}})$ पूरी कहानी बताता है, और
कोई भी संख्या अकेले गुमराह कर सकती है।
\end{remark}

था\dots''
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:orders-of-magnitude:8}
की एक किरण $500$ कागज की शीट, एक मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} रूलर से मापी जाती है

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:orders-of-magnitude:12}
मात्रा के तेल की एक बूंद $\qty{1.0}{mm^3}$ स्थिर जल पर जमा होता है
और व्यास की एक गोलाकार फिल्म में फैल जाता है $\qty{1.1}{m}$ --- एक फिल्म

\begin{enumerate}
  \item फिल्म का क्षेत्रफल, फिर उसकी मोटाई की गणना करें।
  \item एक निष्कर्ष निकालें \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार} एक अणु के आकार के लिए, और
        इसे सीढ़ी के परमाणु पायदान से जांचें।
\end{enumerate}
(यह बैक-ऑफ-द-लिफाफा प्रयोग, सबसे पहले बेंजामिन द्वारा आजमाया गया था
लंदन के एक तालाब पर फ्रैंकलिन, मानवता के पहले ईमानदार उपायों में से एक था
आणविक जगत का।)
\end{exercise}

एक \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार} केवल। इसे चार तक क्यों उद्धृत किया जाएगा? \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} निरर्थक हो?

\begin{exercise}[$\star\star\star $]\label{exo:g10:orders-of-magnitude:14}
कागज की एक शीट है $\qty{0.10}{mm}$ मोटा, और प्रत्येक मोड़ दोगुना हो जाता है
ढेर की मोटाई. का उपयोग करते हुए $2^{10} \approx 10^{3}$:
\begin{enumerate}
  \item कितनी तहों के बाद ढेर माउंट एवरेस्ट से ऊपर निकल जाएगा?
        ($\qty{8848}{m}$)?
  \item कितने गुना के बाद यह चंद्रमा तक पहुंचेगा?
        ($\qty{3.84e8}{m}$)?
  \item व्यवहार में एक शीट को लगभग सात से अधिक नहीं मोड़ा जा सकता
        बार. क्या इससे अंकगणित अमान्य हो जाता है, या केवल
        प्रयोग?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:orders-of-magnitude:15}
लेज़र रेंजिंग से पृथ्वी-चंद्रमा की दूरी ज्ञात होती है
$\qty{3.84e8}{m} \pm \qty{3}{cm}$; एक अच्छा शासक एक मेज देता है
$\qty{1.000}{m} \pm \qty{1}{mm}$.
\begin{enumerate}
  \item दोनों सापेक्ष अनिश्चितताओं की गणना करें। माप कौन सा है
        अधिक सटीक, और किस कारक से?
  \item साथ क्या पूर्ण अनिश्चितता क्या आपको मापने की आवश्यकता होगी
        चंद्रमा माप की सापेक्ष सटीकता से मेल खाने वाली तालिका?
        उस लंबाई की तुलना परमाणु के आकार से करें और निष्कर्ष निकालें
        उचित सम्मान के साथ.
\end{enumerate}
\end{exercise}

\section{समस्या: परमाणु से एंड्रोमेडा तक}

\begin{problem}[{सप्ताहांत समस्या--- रेत का एक कण जितना बड़ा होता है
परमाणु, सौर मंडल, आकाशगंगा और अवलोकनीय ब्रह्मांड, और
आपको उनके बीच की सीढ़ी पर स्थित करता है}]
\label{pb:g10:orders-of-magnitude:1}
रेत का एक कण भौतिकी में सबसे विनम्र वस्तु है, और इस सप्ताह के अंत में यह
ओवरटाइम काम करता है: पहले हम इसके परमाणुओं को गिनते हैं, फिर हम सूर्य को सिकोड़ते हैं
इसका आकार और गति ब्रह्मांड से बाहर है, फिर हमने प्रकाश को सर्वेक्षण करने दिया,
और अंत में हम परमाणुओं की गिनती करते हैं \emph{आप}. डेटा, भर में उपयोग किया गया:
अनाज का व्यास $\qty{0.5}{mm}$; परमाणु व्यास $\qty{2.5e-10}{m}$;
रेत का घनत्व $\qty{2500}{kg/m^3}$; व्यास: सूर्य
$\qty{1.39e9}{m}$, धरती $\qty{1.27e7}{m}$, आकाशगंगा
$\qty{9.5e20}{m}$, अवलोकनीय ब्रह्मांड $\qty{8.8e26}{m}$; दूरियाँ
हमारी ओर से: चंद्रमा $\qty{3.84e8}{m}$, सूरज $\qty{1.496e11}{m}$, नेपच्यून
$\qty{4.5e12}{m}$, प्रॉक्सिमा सेंटॉरी $\qty{4.0e16}{m}$, एंड्रोमेडा
आकाशगंगा $\qty{2.4e22}{m}$; $ c = \qty{3.00e8}{m/s}$; एक जल अणु
द्रव्यमान है $\qty{3.0e-26}{kg}$ और इसमें तीन परमाणु होते हैं।

\medskip
\noindent\textbf{Part I --- Anatomy of the grain.}
\begin{enumerate}
  \item अनाज का व्यास और परमाणु का व्यास लिखें एम-eters, में
        \omterm{def:g10:orders-of-magnitude:scinot}{वैज्ञानिक संकेतन}, और एसआई का नाम बताएं \omterm{def:g10:orders-of-magnitude:prefixes}{उपसर्ग} प्रत्येक निकटतम है
        को.
  \item एक कण के व्यास में कितने परमाणु एक साथ बैठते हैं?
  \item अनाज को किनारे के घन के रूप में मॉडल करें $\qty{0.5}{mm}$ के साथ पैक किया गया
        परमाणु. अनाज में परमाणुओं की संख्या का अनुमान लगाएं \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार}.
  \item उस घन के आयतन की गणना करें, फिर अनाज के द्रव्यमान की।
        द्रव्यमान को मिलीग्राम में व्यक्त करें।
  \item मॉडल की विवेक-जाँच करें: अनाज के द्रव्यमान को उसकी संख्या से विभाजित करें
        परमाणुओं की और ज्ञात से तुलना करें \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार} का
        परमाणु द्रव्यमान (कुछ $10^{-26}\,\unit{kg}$). निर्णय?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Sun as a grain of sand.}
हम ब्रह्मांड का एक स्केल मॉडल बनाते हैं जिसमें सूर्य कण है:
प्रत्येक मॉडल का आकार एक कारक से गुणा करके वास्तविक आकार के बराबर होता है $ s $.
\begin{enumerate}[resume]
  \item स्केल फैक्टर की गणना करें $ s $.
  \item मॉडल में, पृथ्वी का व्यास और उससे दूरी ज्ञात करें
        अनाज-सूर्य. भाग I की दुनिया की कौन सी वस्तु मोटे तौर पर है
        मॉडल पृथ्वी का आकार?
  \item मॉडल नेप्च्यून की कक्षा अनाज से कितनी दूर है? किसका टुकड़ा
        फर्नीचर पूरे ग्रह मंडल को धारण करेगा?
  \item प्रॉक्सिमा सेंटॉरी मॉडल कितनी दूर है?
  \item रेत के दो कण, $\qty{14}{km}$ अलग, और लगभग कुछ भी नहीं
        बीच में: एक या दो वाक्यों में बताएं कि मॉडल क्या है
        इससे पता चलता है कि आकाशगंगा कितनी खाली है।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Light does the surveying.}
\begin{enumerate}[resume]
  \item रेत के वास्तविक कण को पार करने में प्रकाश को कितना समय लगता है?
  \item चंद्रमा से और हम तक प्रकाश पहुंचने में कितना समय लगता है?
        सूर्य?
  \item पुनः गणना करें \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष} में एम-eters से $ c $ और की लंबाई
        वर्ष, और सत्यापित करें कि डेटा में प्रॉक्सिमा दूरी है
        के बारे में मेल खाता है $\qty{4.2}{ly}$.
  \item एंड्रोमेडा दूरी को इसमें बदलें \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष}. मोटे तौर पर जब
        क्या अब आपकी आंख में प्रवेश करने वाली रोशनी एंड्रोमेडा से बाहर निकल गई, और क्या
        उस समय पृथ्वी पर चल रहा था?
  \item प्रकाश एक नैनोसेकंड में कितनी दूरी तय करता है? एक प्रोसेसर
        पर देखा गया $\qty{3}{GHz}$ के हर तीसरे भाग में एक ऑपरेशन शुरू करता है
        नैनोसेकंड: एक वाक्य में बताएं कि तेज़ कंप्यूटर क्यों होना चाहिए
        एक छोटा कंप्यूटर हो.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- You, between atom and Andromeda.}
\begin{enumerate}[resume]
  \item एक इंसान मोटे तौर पर होता है $\qty{70}{kg}$ पानी डा। डेटा का उपयोग करना,
        पानी में एक परमाणु के औसत द्रव्यमान का अनुमान लगाएं, फिर
        मनुष्य में परमाणुओं की संख्या.
  \item उस गिनती की तुलना अनाज के परमाणुओं से करें (प्रश्न 3)
        और मोटे तौर पर के साथ $2\times10^{23}$ अवलोकनीय के सितारे
        ब्रह्माण्ड. दोनों कारक दीजिए।
  \item सीढ़ी पर, क्या आप परमाणु के करीब हैं या अवलोकनीय के
        ब्रह्माण्ड? अनुपातों की तुलना करें $\text{(you)}/\text{(atom)}$ और
        $\text{(universe)}/\text{(you)}$, ले रहा हूँ $\qty{1.7}{m}$ आपके लिए।
  \item आकार ज्ञात करें $ x $ सीढ़ी के बिल्कुल आधे ऊपर बैठा हूँ
        नाभिक के बीच ($\qty{e-15}{m}$) और देखने योग्य
        ब्रह्माण्ड: $ x $ संतुष्ट होना चाहिए $ x / 10^{-15} = 10^{26.9} / x $
        (का उपयोग करते हुए $\qty{8.8e26}{m} \approx 10^{26.9}\,\unit{m}$). क्या
        सौर-प्रणाली वस्तु का आकार बस इतना ही होता है?
  \item पंचलाइन. भाग II द्वारा अवलोकनीय ब्रह्माण्ड को सिकोड़ें
        कारक $ s$, ताकि सूर्य रेत का एक कण हो। मॉडल दीजिए
        ब्रह्माण्ड का व्यास में एम-eters, में फिर खगोलीय इकाइयाँ, और
        एक वाक्य में निष्कर्ष निकालें: यहां तक कि हर सितारा एक में बदल गया
        रेत का कण, ब्रह्मांड हर मानव पैमाने पर बहता है --- केवल
        दस की शक्तियाँ इसे धारण करती हैं।
\end{enumerate}
\end{problem}
