\chapter{प्रकाश का अपवर्तन}\label{ch:g10:refraction}

एक गिलास पानी में एक तिनका डुबोएं: यह पानी की रेखा पर टूटा हुआ दिखता है।
स्विमिंग पूल हमेशा दिखने से ज्यादा गहरे होते हैं, हीरे चमकते हैं
रंगहीन पत्थर से निकली आग, और जो वाक्य आप शायद पढ़ रहे हैं
एक बाल से भी पतले कांच के धागे के अंदर एक महासागर को पार किया। एक कानून
इन चारों की व्याख्या करता है --- स्नेल द्वारा पाई गई दो साइनों के बीच समानता
1621 भौतिकविदों द्वारा कोणों की तालिकाओं को घूरने की पंद्रह शताब्दियों के बाद
बिना पैटर्न देखे.

\section{प्रतिबिंब: एक अनुस्मारक}

एक सजातीय पारदर्शी माध्यम में, प्रकाश सीधी रेखाओं में यात्रा करता है ---
की किरणें \cref{ch:g10:light-spectra}. दो के बीच की सीमा पर
मीडिया, प्रकाश का हिस्सा वापस उछलता है (\emph{प्रतिबिंब}) और भाग
पार कर जाता है (\emph{\omterm{def:g10:refraction:refraction}{अपवर्तन}}). दोनों सटीक ज्यामितीय कानूनों का पालन करते हैं, ऐसा कहा गया है
एक परिपाटी के साथ: कोणों को से मापा जाता है \emph{सामान्य}, कभी नहीं
सतह से.

\begin{definition}[सामान्य और आपतन कोण]\label{def:g10:refraction:angles}
उस बिंदु पर जहां किरण किसी सतह से मिलती है
\emph{सामान्य}\index{सामान्य} सतह पर लंबवत रेखा है
वह बिंदु. \emph{आपतन कोण}\index{आपतन कोण} $i_1$
आने वाली किरण और अभिलंब के बीच का कोण है; के कोण
प्रतिबिंब और का \omterm{def:g10:refraction:refraction}{अपवर्तन} उसी में सामान्य से मापा जाता है
रास्ता.
\end{definition}

\begin{proposition}[परावर्तन का नियम]\label{prop:g10:refraction:reflection}
परावर्तित किरण आपतित किरण वाले तल में स्थित होती है
सामान्य, के दूसरी तरफ सामान्य, और प्रतिबिंब का कोण
के बराबर है \omterm{def:g10:refraction:angles}{आपतन कोण}:
\[
i_1' = i_1 .
\]
\end{proposition}
\admitted

\begin{example}[कोणों को सही ढंग से पढ़ना]\label{ex:g10:refraction:mirror}
एक लेज़र किरण एक दर्पण से टकराती है $35^\circ $'' --- से मापा गया
दर्पण की सतह, जैसा कि बीमों को अक्सर वर्णित किया जाता है। \omterm{def:g10:refraction:angles}{आपतन कोण} है
$90^\circ - 35^\circ = 55^\circ $, तो परावर्तित किरण भी निकल जाती है
$55^\circ $ से सामान्य, और किरण घूम जाती है
$180^\circ - 2 \times 55^\circ = 70^\circ $. को भूल जाना
सामान्य सम्मलेन इस अध्याय की सबसे आम त्रुटि है; इसकी लागत है
$90^\circ $ हर बार.
\end{example}

\section{अपवर्तन और अपवर्तनांक}

आधे भरे गिलास में एक पेंसिल रखें: यह सतह पर टूटा हुआ दिखता है।
पानी के नीचे से प्रकाश पानी से बाहर निकलते ही दिशा बदल देता है,
तो आंख --- जो सीधी किरणें ग्रहण करती है --- टिप का पुनर्निर्माण करती है
कहीं ऐसा नहीं है.

\begin{definition}[अपवर्तन]\label{def:g10:refraction:refraction}
\emph{अपवर्तन}\index{अपवर्तन} प्रकाश की दिशा में परिवर्तन है
क्योंकि यह दो पारदर्शी मीडिया के बीच की सीमा को पार करता है। में किरण
\omterm{def:g10:orders-of-magnitude:unit}{दूसरा} माध्यम है \emph{अपवर्तित किरण}\index{अपवर्तित किरण}, और यह
से कोण सामान्य अपवर्तन का कोण है $ i_2$.
\end{definition}

\omterm{def:g10:refraction:refraction}{अपवर्तन} ऐसा इसलिए होता है क्योंकि प्रकाश की अलग-अलग गति अलग-अलग होती है
मीडिया; प्रत्येक सामग्री की विशेषता यह होती है कि वह प्रकाश को कितना धीमा कर देती है।

\begin{definition}[अपवर्तनांक]\label{def:g10:refraction:index}
 \emph{अपवर्तनांक}\index{अपवर्तनांक} एक पारदर्शी का
माध्यम अनुपात है
\[
n = \frac{c}{v},
\]
कहाँ $ c = \qty{3.00e8}{m/s}$ निर्वात में प्रकाश की गति है और $ v $ इसका
माध्यम में गति. तब से $ v \leq c $, सूचकांक संतुष्ट करता है $ n \geq 1$; यह
नहीं है \omterm{def:g10:orders-of-magnitude:unit}{इकाई}. कुछ उपयोगी मूल्य:
\begin{center}
\begin{tabular}{lccccc}
medium & air & water & plexiglas & glass & diamond \\
$ n $ & $1.0003$ & $1.33$ & $1.49$ & $1.50$ & $2.42$ \\
\end{tabular}
\end{center}
\end{definition}

\begin{example}[पानी में प्रकाश की गति]\label{ex:g10:refraction:speed}
पानी में, $ v = \dfrac{c}{n} = \dfrac{\qty{3.00e8}{m/s}}{1.33}
= \qty{2.26e8}{m/s}$. हीरे में प्रकाश रेंगता है
$\qty{3.00e8}{m/s}/2.42 = \qty{1.24e8}{m/s}$ --- आधे से भी कम
निर्वात गति. हवा के लिए, $ n = 1.0003$ द्वारा गति बदलता है $0.03\%$: में
इस अध्याय में हम वायु को निर्वात मानते हैं और लेते हैं $ n_{\text{air}} = 1.00$.
\end{example}

\begin{theorem}[स्नेल का अपवर्तन नियम]\label{thm:g10:refraction:snell}
\index{स्नेल का नियम}
 \omterm{def:g10:refraction:refraction}{अपवर्तित किरण} आपतित किरण के तल में स्थित है और सामान्य,
के दूसरी तरफ सामान्य, और आपतन के कोण और
\omterm{def:g10:refraction:refraction}{अपवर्तन} संतुष्ट
\[
n_1 \sin i_1 = n_2 \sin i_2 ,
\]
कहाँ $ n_1$ और $ n_2$ दो मीडिया के अपवर्तक सूचकांक हैं। द
पथ प्रतिवर्ती है: प्रकाश पीछे की ओर यात्रा करता है \omterm{def:g10:refraction:refraction}{अपवर्तित किरण}
घटना किरण के साथ बाहर निकलता है।
\end{theorem}
\admitted

\begin{remark}\label{rem:g10:refraction:honest}
प्रतिबिंब के नियम और \omterm{def:g10:refraction:refraction}{अपवर्तन} यहां प्रायोगिक तौर पर बताया गया है
तथ्य. वे ईमानदारी से वर्ष 1 खंड में प्राप्त हुए हैं, जहां प्रकाश है
एक तरंग के रूप में व्यवहार किया गया: दोनों नियम --- और सूत्र $ n = c/v $ स्वयं---
फिर सघन माध्यम में धीमी होती तरंग का अनुसरण करें।
\end{remark}

\begin{example}[हवा से पानी]\label{ex:g10:refraction:airwater}
एक किरण शांत तालाब से टकराती है $ i_1 = 30^\circ $. साथ $ n_1 = 1.00$ और
$ n_2 = 1.33$:
\[
\sin i_2 = \frac{n_1 \sin i_1}{n_2} = \frac{\sin 30^\circ}{1.33}
= \frac{0.500}{1.33} = 0.376,
\qquad
i_2 = 22.1^\circ .
\]
किरण की ओर झुकती है सामान्य, जैसा कि यह हमेशा प्रवेश करते समय होता है
धीमा माध्यम ($ n_2 > n_1$). पानी छोड़ने पर वही गणना चलती है
पीछे की ओर और किरण झुक जाती है \emph{दूर} से सामान्य.
\end{example}

\begin{method}[अपवर्तन समस्या का समाधान]\label{met:g10:refraction:method}
\begin{enumerate}
  \item सतह को ड्रा करें, सामान्य, और किरण; पहचान करना $ n_1$
        (आपतित किरण का माध्यम) और $ n_2$.
  \item जांचें कि दिए गए कोणों को से मापा जाता है सामान्य;
        यदि उन्हें सतह से मापा जाए तो परिवर्तित करें।
  \item लिखना $ n_1 \sin i_1 = n_2 \sin i_2$ और अज्ञात को अलग करें
        साइन; कैलकुलेटर की व्युत्क्रम ज्या से कोण प्राप्त करें, जैसे कि
        गणित की मात्रा.
  \item विवेक जांच: किरण की ओर झुकती है सामान्य अगर $ n_2 > n_1$,
        इससे दूर अगर $ n_2 < n_1$, और $\sin i_2$ बाहर आना चाहिए
        के बीच $0$ और $1$.
\end{enumerate}
\end{method}

कानून के पीछे का प्रयोग एक डेस्क पर फिट बैठता है: प्लेक्सीग्लास की एक आधी डिस्क,
एक प्रोट्रैक्टर, एक लेज़र। कच्चे कोण जोड़े $(i_1, i_2)$ अनियमित देखो ---
टॉलेमी ने उन्हें वर्ष 150 के आसपास सारणीबद्ध किया और गलत निष्कर्ष निकाला
$ i_2$ के समानुपाती था $ i_1$. इसके बजाय साइन के विरुद्ध साइन प्लॉट करें, और
डेटा ढलान की उत्पत्ति के माध्यम से एक रेखा में सीधा हो जाता है $ n_1/n_2$: वह
लाइन \emph{है} \omterm{thm:g10:refraction:snell}{स्नेल का नियम}.

\begin{proposition}[स्पष्ट गहराई]\label{prop:g10:refraction:depth}
\index{स्पष्ट गहराई}
गहराई पर एक वस्तु $ d $ पानी के नीचे, ऊपर से छोटे कोणों से देखा गया,
पर प्रतीत होता है \emph{\omterm{prop:g10:refraction:depth}{स्पष्ट गहराई}}
\[
d' = \frac{d}{n} ,
\]
कहाँ $ n $ जल का सूचकांक है. गहराई का तालाब \qty{1.20}{m} दिखता है
केवल $1.20/1.33 = \qty{0.90}{m}$ गहरा --- अविवेक का एक क्लासिक कारण
गोता लगाता है।
\end{proposition}
\admitted

\begin{remark}\label{rem:g10:refraction:depthhonest}
सूत्र इस प्रकार है \omterm{thm:g10:refraction:snell}{स्नेल का नियम} लगभग-ऊर्ध्वाधर किरणों पर लागू किया गया
आँख तक पहुँचना; व्युत्पत्ति वर्ष 1 खंड में की गई है। द
प्रभाव की दिशा के लिए किसी सूत्र की आवश्यकता नहीं होती: पानी से निकलने वाली किरणें मुड़ जाती हैं
से दूर सामान्य, इसलिए आँख उन्हें वापस एक बिंदु ऊँचे पर ले जाती है
वस्तु की तुलना में.
\end{remark}

\section{फैलाव: प्रति रंग एक नियम}

एक प्रिज्म सफेद सूरज की रोशनी को इंद्रधनुष में बदल देता है, जैसे कि स्पेक्ट्रा
\cref{ch:g10:light-spectra} दिखाया. \omterm{def:g10:refraction:refraction}{अपवर्तन} बताते हैं \emph{क्यों}: द
\omterm{def:g10:refraction:index}{अपवर्तनांक} प्रत्येक रंग के लिए एक माध्यम बिल्कुल समान नहीं होता है।

\begin{definition}[फैलाव]\label{def:g10:refraction:dispersion}
एक माध्यम है \emph{फैलानेवाला}\index{फैलानेवाला} कब इसकी अपवर्तक
सूचकांक पर निर्भर होता है \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य} प्रकाश का. गिलास और पानी में, $ n $ है
नीली रोशनी के लिए थोड़ा बड़ा (छोटा)। \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य}) लाल बत्ती की तुलना में
(लंबा \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य}) --- इसलिए नीला अधिक झुकता है।
\end{definition}

\begin{example}[दो रंग, एक सतह]\label{ex:g10:refraction:twocolors}
की एक किरण \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} क्राउन ग्लास से टकराता है $ i_1 = 60^\circ $. इसके लिए
कांच $ n_{\text{red}} = 1.51$ और $ n_{\text{blue}} = 1.53$. \omterm{thm:g10:refraction:snell}{स्नेल का नियम},
प्रति रंग एक बार:
\[
\sin i_{\text{red}} = \frac{\sin 60^\circ}{1.51} = 0.574,
\quad i_{\text{red}} = 35.0^\circ ;
\qquad
\sin i_{\text{blue}} = \frac{\sin 60^\circ}{1.53} = 0.566,
\quad i_{\text{blue}} = 34.5^\circ .
\]
एक ही सतह फट जाती है \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} आधे डिग्री तक --- एक पर अदृश्य
खिड़की का फलक, लेकिन एक प्रिज्म एक ही घूर्णी अर्थ में दो बार अपवर्तित होता है,
पंखे को चौड़ा करता है, और कुछ \omterm{def:g10:orders-of-magnitude:unit}{मीटर} की दूरी पर रंग सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} अलग होते हैं।
\end{example}

\begin{remark}\label{rem:g10:refraction:rainbow}
इंद्रधनुष वही गणना है जो बारिश की बूंद के अंदर चलती है: सूरज की रोशनी
बूँद में प्रवेश करके अपवर्तित होता है, उसकी पिछली दीवार पर परावर्तित होता है और अपवर्तित होता है
फिर से बाहर जाने के रास्ते पर. जल का सूचकांक चलता है $1.331$ (लाल) को $1.343$
(नीला), इसलिए प्रत्येक रंग अपने स्वयं के कोण पर बाहर निकलता है --- और कोई दो पर्यवेक्षक नहीं
वही इंद्रधनुष देखें.
\end{remark}

\section{पूर्ण आंतरिक प्रतिबिंब}

\omterm{thm:g10:refraction:snell}{स्नेल का नियम} एक जाल शामिल है. पानी से हवा की ओर जाना, \omterm{def:g10:refraction:refraction}{अपवर्तित
किरण} से दूर झुक जाता है सामान्य: $\sin i_2 = 1.33 \sin i_1 > \sin i_1$.
बढ़ोतरी $ i_1$, और सूत्र जल्द ही मांगता है $\sin i_2 > 1$, जो नहीं
कोण वितरित कर सकता है। \omterm{def:g10:refraction:refraction}{अपवर्तन} तो फिर घटित होना ही बंद हो जाता है।

\begin{definition}[क्रांतिक कोण और पूर्ण आंतरिक परावर्तन]\label{def:g10:refraction:critical}
प्रकाश को सूचकांक के माध्यम में यात्रा करने दें $ n_1$ छोटे के एक माध्यम की ओर
सूचकांक $ n_2 < n_1$. \emph{महत्वपूर्ण कोण}\index{महत्वपूर्ण कोण}
$ i_c $ है \omterm{def:g10:refraction:angles}{आपतन कोण} जिसके लिए \omterm{def:g10:refraction:refraction}{अपवर्तित किरण} को चराता है
सतह ($ i_2 = 90^\circ $). के लिए $ i_1 > i_c $ नहीं \omterm{def:g10:refraction:refraction}{अपवर्तित किरण} मौजूद है:
सतह सारे प्रकाश को निम्नलिखित के बाद पहले माध्यम में वापस परावर्तित कर देती है
परावर्तन का नियम. यह है \emph{कुल आंतरिक प्रतिबिंब}\index{कुल आंतरिक प्रतिबिंब}.
\end{definition}

\begin{proposition}[क्रांतिक कोण सूत्र]\label{prop:g10:refraction:criticalformula}
के लिए $ n_1 > n_2$,
\[
\sin i_c = \frac{n_2}{n_1} .
\]
\end{proposition}

\begin{proof}
तय करना $ i_2 = 90^\circ $ में \omterm{thm:g10:refraction:snell}{स्नेल का नियम}: $ n_1 \sin i_c = n_2 \sin 90^\circ
= n_2$. के लिए $ i_1 > i_c $, \omterm{thm:g10:refraction:snell}{स्नेल का नियम} आवश्यकता होगी
$\sin i_2 = \frac{n_1}{n_2}\sin i_1 > 1$: कोई अपवर्तित दिशा मौजूद नहीं है।
(वह प्रकाश तब है \emph{पूरी तरह से} प्रतिबिंबित --- अवशोषित नहीं --- है
एक प्रायोगिक तथ्य, जिसकी पुष्टि वर्ष 1 के तरंग सिद्धांत द्वारा की गई है
आयतन.)
\end{proof}

\begin{example}[तीन महत्वपूर्ण कोण]\label{ex:g10:refraction:criticalvalues}
हवा की ओर: पानी के लिए, $\sin i_c = 1/1.33 = 0.752$, इसलिए
$ i_c = 48.8^\circ $; कांच के लिए, $\sin i_c = 1/1.50$, इसलिए
$ i_c = 41.8^\circ $; हीरे के लिए, $\sin i_c = 1/2.42 = 0.413$, इसलिए
$ i_c = 24.4^\circ $. सूचकांक जितना अधिक होगा, एस्केप कोन उतना ही संकीर्ण होगा:
हीरे में, किसी भी किरण से अधिक $24.4^\circ $ एक पहलू से हटकर सामान्य है
फँसा हुआ --- इसकी चमक का रहस्य, विच्छेदित
\cref{pb:g10:refraction:1}.
\end{example}

\section{ऑप्टिकल फाइबर}

पूर्ण आंतरिक प्रतिबिंब प्रकाश के लिए कांच के धागे को पाइप में बदल देता है:
एक किरण अक्ष के लगभग समानांतर प्रवेश करती हुई दीवार से काफी दूर तक टकराती है
द \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}, बिना किसी नुकसान के प्रतिबिंबित करता है, और लाखों बार ऐसा करता है
प्रति किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} बिना रिसाव के।

\begin{definition}[प्रकाशित तंतु]\label{def:g10:refraction:fiber}
एक \emph{प्रकाशित तंतु}\index{प्रकाशित तंतु} एक पतला कांच का धागा बना हुआ है
एक का \emph{मुख्य}\index{मुख्य} सूचकांक का $ n_1$ एक में लिपटा हुआ
\emph{आवरण}\index{आवरण} थोड़ा छोटे सूचकांक का $ n_2 < n_1$.
एक छोर पर डाला गया प्रकाश बार-बार कोर के साथ यात्रा करता है \omterm{def:g10:refraction:critical}{कुल
आंतरिक प्रतिबिंब} कोर-क्लैडिंग सीमा पर।
\end{definition}

\begin{example}[संख्याओं द्वारा एक फाइबर]\label{ex:g10:refraction:fibernumbers}
साथ $ n_1 = 1.48$ और $ n_2 = 1.46$,
\[
\sin i_c = \frac{1.46}{1.48} = 0.986,
\qquad i_c = 80.6^\circ :
\]
भीतर केवल किरणें $9^\circ $ अक्ष का मार्गदर्शन किया जाता है --- जो है
ठीक है, क्योंकि लेज़र प्रकाश को ठीक इसी प्रकार इंजेक्ट किया जाता है। के अंदर
\omterm{def:g10:refraction:fiber}{मुख्य} प्रकाश यात्रा करता है $ v = c/1.48 = \qty{2.03e8}{m/s}$: एक नाड़ी
एक को पार करता है \qty{100}{km} लगभग आधा मिलीसेकेंड में फाइबर।
\end{example}

\begin{remark}[डेटा लिंक के परिमाण का क्रम]\label{rem:g10:refraction:datarates}
पनडुब्बी केबल --- समुद्र तल पर कुछ दर्जन फाइबर के बंडल
--- लगभग सभी अंतरमहाद्वीपीय यातायात ले जाते हैं। एक एकल आधुनिक फाइबर
के आदेश पर परिवहन करता है $10^{13}$ कई भेजकर प्रति सेकंड बिट्स
\omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य} तुरंत; एक पूरी केबल पहुंचती है $10^{14}$ प्रति सेकंड बिट्स,
हर एक एम्पलीफायर के साथ \qty{80}{km} या ऐसा। तुलना के लिए उपग्रह,
ट्रैफ़िक के एक प्रतिशत का एक छोटा सा अंश संभालें।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:refraction:1}
एक लेज़र किरण एक सपाट दर्पण पर प्रहार करती है $25^\circ $ \emph{दर्पण से सतह}. दे दो \omterm{def:g10:refraction:angles}{आपतन कोण}, परावर्तन का कोण, और
घटना और परावर्तित किरणों के बीच का कोण।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:refraction:2}
\begin{enumerate}
  \item पानी में प्रकाश की गति की गणना करें ($ n = 1.33$) और हीरे में
        ($ n = 2.42$).
  \item एक निश्चित ठोस में, प्रकाश गति करता है \qty{1.97e8}{m/s}. गणना करें
        इसका \omterm{def:g10:refraction:index}{अपवर्तनांक} और सामग्री की पहचान करें.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:refraction:3}
हवा में एक किरण एक कांच के ब्लॉक से टकराती है ($ n = 1.50$) पर $ i_1 = 40^\circ $.
के कोण की गणना करें \omterm{def:g10:refraction:refraction}{अपवर्तन}. एक किरण किस कोण पर यात्रा करेगी?
कांच के अंदर हवा में बाहर निकलने के लिए सतह से टकराना पड़ता है
$40^\circ $?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:refraction:4}
सूर्य का प्रकाश झील की सतह तक पहुंचता है $55^\circ $ ऊर्ध्वाधर से.
के कोण की गणना करें \omterm{def:g10:refraction:refraction}{अपवर्तित किरण}, और बताएं कि क्या किरण मुड़ती है
की ओर या उससे दूर सामान्य --- एक शब्द के कारण के साथ।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:refraction:5}
किसी साफ़ तरल की पहचान करने के लिए उसकी सतह पर एक किरण भेजी जाती है
$ i_1 = 45^\circ $; \omterm{def:g10:refraction:refraction}{अपवर्तित किरण} पर मापा जाता है $ i_2 = 32^\circ $.
की गणना करें \omterm{def:g10:refraction:index}{अपवर्तनांक} और की तालिका से तरल की पहचान करें
\cref{def:g10:refraction:index}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:refraction:6}
एक पारदर्शी प्लास्टिक की आधी डिस्क निम्नलिखित माप देती है:
\begin{center}
\begin{tabular}{lccccc}
$ i_1$ & $20^\circ $ & $30^\circ $ & $40^\circ $ & $50^\circ $ & $60^\circ $ \\
$ i_2$ & $13.5^\circ $ & $19.5^\circ $ & $25.5^\circ $ & $31.0^\circ $
      & $35.5^\circ $ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item गणना करना $\dfrac{\sin i_1}{\sin i_2}$ प्रत्येक जोड़ी के लिए. आप क्या करते हैं?
        निरीक्षण करें?
  \item निष्कर्ष निकालें \omterm{def:g10:refraction:index}{अपवर्तनांक} प्लास्टिक की पहचान करें और इसकी पहचान करें।
  \item साजिश क्यों रची जा रही है $\sin i_2$ ख़िलाफ़ $\sin i_1$ का एक बेहतर परीक्षण
        \omterm{thm:g10:refraction:snell}{स्नेल का नियम} साजिश रचने की तुलना में $ i_2$ ख़िलाफ़ $ i_1$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:refraction:7}
की गणना करें \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} पानी, गिलास के लिए हवा की ओर ($ n = 1.50$) और
हीरा. फिर समझाओ, साथ \omterm{thm:g10:refraction:snell}{स्नेल का नियम}, क्यों नहीं है \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}
प्रकाश जाने के लिए \emph{से} वायु \emph{में} पानी।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:refraction:8}
\begin{enumerate}
  \item एक पूल है \qty{2.0}{m} गहरा। एक तैराक कितनी गहराई पर खड़ा होता है
        किनारे पर समझते हैं?
  \item एक बगुला तैरती हुई मछली को देखता है \qty{40}{cm} सतह के नीचे. पर
        मछली कितनी गहराई में दिखाई देती है, और क्या बगुले को हमला करना चाहिए
        यह जो छवि देखता है उसके ऊपर, पर या नीचे?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:refraction:9}
एक किरण खिड़की के शीशे से टकराती है ($ n = 1.50$, समानांतर चेहरे) पर $50^\circ $.
\begin{enumerate}
  \item के कोण की गणना करें \omterm{def:g10:refraction:refraction}{अपवर्तन} पहले चेहरे पर.
  \item  \omterm{def:g10:refraction:refraction}{अपवर्तित किरण} दूसरे चेहरे पर भी अंदर से वार करता है
        कोण. निकास कोण की गणना करें और इसकी तुलना करें $50^\circ $.
  \item तो फिर, एक मोटी खिड़की दिशा और दिशा पर क्या प्रभाव डालती है
        इसे पार करने वाली किरण की स्थिति?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:refraction:10}
\omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} चकमक कांच के एक टुकड़े पर प्रहार करता है $55^\circ $. इसके लिए
कांच, $ n_{\text{red}} = 1.61$ और $ n_{\text{blue}} = 1.66$. की गणना करें
\omterm{def:g10:refraction:refraction}{अपवर्तन} लाल और नीली किरणों के कोण और उनके बीच का कोण।
कौन सा रंग के करीब समाप्त होता है सामान्य?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:refraction:11}
एक \omterm{def:g10:refraction:fiber}{प्रकाशित तंतु} एक \omterm{def:g10:refraction:fiber}{मुख्य} सूचकांक का $1.48$ और ए \omterm{def:g10:refraction:fiber}{आवरण} सूचकांक का
$1.46$.
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} कोर-क्लैडिंग सीमा पर।
  \item एक निर्देशित किरण ठीक उसी दिशा में उछलती है \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}. दिखाओ
        वह खत्म \qty{1.0}{km} यह सीधे रेशे का होता है जिसके बारे में यह यात्रा करता है
        \qty{14}{m} अक्ष से सीधे नीचे जाने वाली किरण से अधिक।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:refraction:12}
गोताखोर की नजर होती है \qty{2.0}{m} बिल्कुल शांत सतह के नीचे। के कारण
\omterm{def:g10:refraction:refraction}{अपवर्तन}, गोताखोर देखता है \emph{पूरा} आकाश एक में संकुचित हो गया
चमकीली डिस्क सीधे ऊपर---स्नेल की खिड़की।
\begin{enumerate}
  \item बताएं कि क्षितिज से प्रकाश गोताखोर की आंख तक क्यों पहुंचता है
        द \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} पानी डा, $48.8^\circ $ ऊर्ध्वाधर से.
  \item का उपयोग करके सतह पर चमकदार डिस्क की त्रिज्या की गणना करें
        समकोण त्रिभुज में स्पर्शरेखा आँख--ऊर्ध्वाधर--डिस्क किनारा।
  \item गोताखोर सतह पर क्या देखता है? \emph{बाहर} डिस्क?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:refraction:13}
प्रिज्म में कोण होते हैं $30^\circ $, $60^\circ $, $90^\circ $. की एक पतली किरण
\omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} समकोण से सटे किसी एक पक्ष के लंबवत प्रवेश करता है,
बिना विचलित हुए पार करता है, और अंदर से लंबे चेहरे पर प्रहार करता है
$ i_1 = 30^\circ $. इस गिलास के लिए, $ n_{\text{red}} = 1.51$ और
$ n_{\text{blue}} = 1.53$.
\begin{enumerate}
  \item लाल और नीली किरणों और कोणीय के निकास कोण की गणना करें
        उनके बीच पंखे की चौड़ाई.
  \item प्रिज्म को a द्वारा प्रतिस्थापित किया जाता है $45^\circ $--$45^\circ $--$90^\circ $
        प्रिज्म, तो आंतरिक \omterm{def:g10:refraction:angles}{आपतन कोण} बन जाता है $45^\circ $.
        दिखाएँ कि लंबे चेहरे से कोई प्रकाश बाहर नहीं निकलता है।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:refraction:14}
पेरिस्कोप प्रकाश को मोड़ते हैं $45^\circ $--$45^\circ $--$90^\circ $ कांच
प्रिज्म ($ n = 1.50$): किरण एक छोटे चेहरे में लंबवत रूप से प्रवेश करती है और
अंदर से लंबे चेहरे पर वार करता है $45^\circ $.
\begin{enumerate}
  \item दिखाएँ कि लंबा चेहरा एक आदर्श दर्पण के रूप में कार्य करता है। ऐसा क्यों है?
        यह धातु के दर्पण से बेहतर है, जो कुछ प्रतिशत ही अवशोषित करता है
        प्रत्येक प्रतिबिंब?
  \item पेरिस्कोप बाढ़ और लंबे चेहरे के साथ अब संपर्क में है
        पानी. नये की गणना करें \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} गिलास का - पानी
        सीमा और दर्पण दिखाने में विफल रहता है.
  \item प्रश्न 2 का प्रकाश प्रिज्म से किस कोण पर निकलता है?
        पानी?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:refraction:15}
एक ट्रान्साटलांटिक केबल चलती है \qty{6200}{km} फाइबर का (\omterm{def:g10:refraction:fiber}{मुख्य} अनुक्रमणिका $1.47$)
लंदन और न्यूयॉर्क के बीच.
\begin{enumerate}
  \item में प्रकाश की गति की गणना करें \omterm{def:g10:refraction:fiber}{मुख्य} और एक तरफ़ा यात्रा
        एक नाड़ी का समय.
  \item एक रेडियो तरंग यात्रा करती है $ c $. इसके लिए कितने समय की आवश्यकता होगी
        समान दूरी, और फ़ाइबर का राउंड-ट्रिप समय कितना है
        रेडियो की तुलना में ग्लास की कीमत?
  \item कुछ वित्तीय कंपनियाँ इससे मिलीसेकंड कम करने के लिए भारी कीमत चुकाती हैं
        लिंक. प्रकाश बनाने के दो अलग-अलग भौतिक तरीके प्रस्तावित करें
        पहले पहुंचें.
\end{enumerate}
\end{exercise}

\section{समस्या: हीरा, पूल और फ़ाइबर}

\begin{problem}[{सप्ताहांत समस्या---स्नेल के नियम के तीन जीवन: एक पूल
जो इसकी गहराई के बारे में है, एक पत्थर काटा गया है ताकि प्रकाश बाहर न निकल सके, और
इंटरनेट के अंतर्गत साठ मिलीसेकंड महासागर}]
\label{pb:g10:refraction:1}
दो साइन की एक समानता, तीन करियर। स्विमिंग पूल में यह धोखा देता है
आपकी आंखें गहराई के बारे में सोचती हैं और पूरे आकाश को एक घेरे में समेट लेती हैं। ए में
जब हीरे को उसके टूटने के बिंदु पर धकेल दिया जाता है, तो वह प्रकाश को बाहर निकलने से इंकार कर देता है ---
और जौहरियों ने सदियों से उस इनकार के आसपास पत्थर तराशे हैं। के अंतर्गत
अटलांटिक यह कांच के एक स्ट्रैंड के अंदर लेजर पल्स का मार्गदर्शन करता है, और सेट करता है
वैश्विक इंटरनेट की गति सीमा. इस समस्या के माध्यम से काम करता है
हर बार समान कानून के साथ, तीन जिंदगियां क्रम में रहती हैं।

\medskip
\noindent\textbf{Part I --- Warming up.}
\begin{enumerate}
  \item पानी में प्रकाश की गति की गणना करें ($ n = 1.33$) और हीरे में
        ($ n = 2.42$). हीरा किस कारक से धीमी रोशनी देता है?
  \item हवा में एक किरण एक तालाब से टकराती है $ i_1 = 40^\circ $. की गणना करें
        का कोण \omterm{def:g10:refraction:refraction}{अपवर्तन}.
  \item पानी में एक किरण नीचे से सतह पर गिरती है $28.9^\circ $.
        यह किस कोण पर हवा में बाहर निकलता है? सिद्धांत बताइये
        यह दर्शाता है.
  \item की गणना करें \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} जल-वायु सीमा, और
        वर्णन करें कि पानी के अंदर की किरण सतह से टकराने पर क्या होता है
        पर $60^\circ $.
  \item वायु का सूचकांक है $1.0003$. हवा में प्रकाश की गति की गणना करें
        और सन्निकटन का औचित्य सिद्ध करें $ n_{\text{air}} = 1.00$ प्रयोग किया गया
        भर में.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The pool.}
\begin{enumerate}[resume]
  \item एक पूल है \qty{3.0}{m} गहरा। स्पष्ट-गहराई सूत्र का उपयोग करना
        (\cref{prop:g10:refraction:depth}), एक व्यक्ति की गहराई की गणना करें
        ऊपर खड़े होकर अनुभव करता है।
  \item एक दो-किरण रेखाचित्र (शब्दों में वर्णित) के साथ स्पष्ट करें कि सिक्का क्यों है
        तल पर ऊपर उठता हुआ प्रतीत होता है: पानी से निकलने वाली किरणें क्या करती हैं,
        और आंखें उनके प्रतिच्छेदन को कहां रखती हैं?
  \item गोताखोर की नजर है \qty{3.0}{m}. स्नेल की त्रिज्या की गणना करें
        खिड़की --- चमकदार डिस्क जिसके माध्यम से गोताखोर देखता है
        पूरा आकाश (\cref{exo:g10:refraction:12}).
  \item तल पर एक दीपक किरणों को ऊपर की ओर भेजता है $30^\circ $,
        $45^\circ $ और $50^\circ $ ऊर्ध्वाधर से. प्रत्येक किरण के लिए, कहें
        क्या यह बच निकलता है और किस कोण से, या इसके बजाय क्या होता है।
  \item किनारे पर एक भाला-मछुआरा एक मछली पर निशाना साधता है। क्या हड़ताल होनी चाहिए
        मछली की छवि के ऊपर, उस पर या नीचे लक्षित --- और करता है
        मछली, पीछे मुड़कर देखें तो क्या मछुआरे को भी विस्थापित देखा जा सकता है?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The diamond.}
\begin{enumerate}[resume]
  \item की गणना करें \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} हीरे का ($ n = 2.42$) हवा की ओर,
        और इसकी तुलना कांच से करें ($ n = 1.52$),
        जो है $41.1^\circ $.
  \item पत्थर के अंदर एक किरण एक पहलू से टकराती है $30^\circ $ उसमें से
        सामान्य. हीरे में क्या होता है? कांच की नकल में?
        एक वाक्य में स्पष्ट करें कि हीरे की कटाई के साथ संयोजन क्यों किया जाता है
        यह छोटा है \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}, चमक पैदा करता है।
  \item हीरा भी जोरदार होता है \omterm{def:g10:refraction:dispersion}{फैलानेवाला}: $ n_{\text{red}} = 2.41$,
        $ n_{\text{blue}} = 2.45$. \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} पर एक पहलू में प्रवेश करता है
        $45^\circ $; की गणना करें \omterm{def:g10:refraction:refraction}{अपवर्तन} प्रत्येक रंग के लिए कोण और
        कोणीय विभाजन.
  \item बाद में, लाल और नीली दोनों किरणें अंदर से एक निकास पहलू तक पहुंचती हैं
        पर $17.0^\circ $. दो निकास कोणों और विभाजन की गणना करें
        उभरता हुआ प्रशंसक. प्रवेश पर विभाजन से तुलना करें: क्या होता है
        को फैलाव के पास \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}?
  \item पानी की एक बूंद ($ n = 1.33$) एक पहलू को कोट करता है। नये की गणना करें
        \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} हीरे की - जल सीमा, भाग्य का फैसला करें
        का $30^\circ $ प्रश्न 12 की किरण, और स्पष्ट करें कि जौहरी क्यों
        हीरे को ईमानदारी से साफ रखें।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The fiber.}
\begin{enumerate}[resume]
  \item एक फाइबर में एक होता है \omterm{def:g10:refraction:fiber}{मुख्य} सूचकांक का $1.48$ और ए \omterm{def:g10:refraction:fiber}{आवरण} सूचकांक का
        $1.46$. की गणना करें \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} कोर पर--आवरण
        सीमा.
  \item हवा में एक नंगे कांच के धागे में एक होगा \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} का
        $42.5^\circ $ ---प्रतीत होता है कहीं बेहतर। दो कारण बताइये
        \omterm{def:g10:refraction:fiber}{आवरण} वैसे भी प्रयोग किया जाता है. (धूल, खरोंच और क्या के बारे में सोचो
        सतह को छूने से ऐसा होगा पूर्ण आंतरिक प्रतिबिंब.)
  \item में प्रकाश की गति की गणना करें \omterm{def:g10:refraction:fiber}{मुख्य}, फिर एक तरफ़ा यात्रा
        एक नाड़ी का समय साथ में \qty{6000}{km} फाइबर का.
  \item एक किरण बिल्कुल पर उछल रही है \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} ट्रेवल्स
        $1/\sin i_c $ अक्ष की लंबाई का गुना. अतिरिक्त की गणना करें
        की दूरी \qty{6000}{km} और इसमें अतिरिक्त समय खर्च होता है।
  \item पंचलाइन. एक आधुनिक फ़ाइबर जोड़ी घूमती रहती है $10^{13}$ बिट्स
        प्रति सेकंड, और यह पुस्तक इसके बारे में है $2.4 \times 10^8$ बिट्स
        समुद्र के पार राउंड-ट्रिप (``पिंग'') के समय की गणना करें
        इस पुस्तक की प्रति सेकंड फाइबर प्रतियों की संख्या ---
        तो बताओ, एक वाक्य में, क्या \omterm{thm:g10:refraction:snell}{स्नेल का नियम} के लिए करता है
        इंटरनेट.
\end{enumerate}
\end{problem}
