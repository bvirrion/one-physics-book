\chapter{संकेत और लहरें}\label{ch:g10:signals-and-waves}

आपको अब तक प्राप्त प्रत्येक संदेश एक के रूप में आया है \omterm{def:g10:signals-and-waves:signal}{संकेत}: एक आवाज दबायी गयी
आपके कानों पर, यह पृष्ठ रेडियो की झिलमिलाहट के रूप में आप तक पहुंचा। के पीछे
विविधता एक विचार बैठती है --- ए \omterm{def:g10:orders-of-magnitude:unit}{भौतिक मात्रा} समय में भिन्नता --- और एक
टूलकिट: \omterm{def:g10:signals-and-waves:period}{अवधि}, \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}, \omterm{def:g10:signals-and-waves:amplitude}{आयाम}, और देरी का समय निर्धारित करने की कला।
इसकी मदद से एक नाव समुद्र तल को मापती है और एक डॉक्टर दिल की धड़कनों को गिनता है।

\section{सिग्नल जानकारी ले जाते हैं}

\begin{definition}[संकेत]\label{def:g10:signals-and-waves:signal}
ए \emph{संकेत}\index{संकेत} एक है \omterm{def:g10:orders-of-magnitude:unit}{भौतिक मात्रा} जो समय के साथ बदलता रहता है
जानकारी ले जाने के लिए: हवा \omterm{def:g10:pressure:pressure}{दबाव} आपके कान के परदे पर, वोल्टेज a
माइक्रोफोन के टर्मिनल, एक में प्रकाश की चमक \omterm{def:g10:refraction:fiber}{प्रकाशित तंतु}.
किसी सिग्नल का अध्ययन करने के लिए, इसे समय के एक फ़ंक्शन के रूप में रिकॉर्ड करें और पढ़ें
वक्र से हटकर जानकारी.
\end{definition}

फ़ोन कॉल में एक ही जानकारी वाहक को बार-बार बदलती है:
\omterm{def:g10:pressure:pressure}{दबाव} हवा में, फिर माइक्रोफ़ोन में वोल्टेज, फिर रेडियो पर
एंटीना --- और दूसरे छोर पर पीछे की ओर। एक शब्दावली उन्हें संभालती है
सब; यह अध्याय इसका निर्माण करता है।

\section{आवधिक संकेत}

\begin{definition}[आवधिक संकेत, अवधि]\label{def:g10:signals-and-waves:period}
ए \omterm{def:g10:signals-and-waves:signal}{संकेत} है \emph{आवधिक}\index{आवधिक} जब यह दोहराता है
समान रूप से नियमित अंतराल पर। \emph{अवधि}\index{अवधि} $T $ है
एक पूर्ण चक्र की अवधि, सेकंड में।
\end{definition}

\begin{definition}[आवृत्ति]\label{def:g10:signals-and-waves:frequency}
 \emph{आवृत्ति}\index{आवृत्ति} एक का \omterm{def:g10:signals-and-waves:period}{आवधिक} \omterm{def:g10:signals-and-waves:signal}{संकेत} संख्या है
प्रति सेकंड चक्रों की संख्या:
\[
f = \frac{1}{T} .
\]
इसका \omterm{def:g10:orders-of-magnitude:unit}{इकाई}, प्रति सेकंड एक चक्र है \emph{हेटर्स}\index{हेटर्स}
(\unit{Hz}), सामान्य गुणकों के साथ \unit{kHz}, \unit{MHz}, \unit{GHz}.
\end{definition}

\begin{definition}[आयाम]\label{def:g10:signals-and-waves:amplitude}
 \emph{आयाम}\index{आयाम} एक का \omterm{def:g10:signals-and-waves:period}{आवधिक} \omterm{def:g10:signals-and-waves:signal}{संकेत} इसका अधिकतम है
विश्राम मूल्य से विचलन. सममित रूप से दोलन करने वाले वोल्टेज के लिए
शून्य के बारे में, आयाम चरम मान है; शिखा से गर्त तक के उपाय
\emph{दो बार} आयाम (शिखर-से-शिखर मान)।
\end{definition}

\begin{example}[मुख्य और संगीत कार्यक्रम की पिच]\label{ex:g10:signals-and-waves:mains}
\omterm{def:g10:refraction:fiber}{मुख्य} वोल्टेज पर दोलन होता है $ f = \qty{50}{Hz}$: इसका \omterm{def:g10:signals-and-waves:period}{अवधि} है
$ T = 1/f = \qty{0.020}{s} = \qty{20}{ms}$. एक ऑर्केस्ट्रा का संगीत कार्यक्रम A एक है
\omterm{def:g10:pressure:pressure}{दबाव} \omterm{def:g10:signals-and-waves:signal}{संकेत} पर \qty{440}{Hz}: एक चक्र चलता है
$ T = 1/440 = \qty{2.3}{ms}$. उच्च \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}, छोटा चक्र: $ f $ और $ T $
एक दूसरे के व्युत्क्रम हैं।
\end{example}

\section{ऑसिलोग्राम पढ़ना}

\begin{definition}[आस्टसीलस्कप]\label{def:g10:signals-and-waves:oscilloscope}
एक \emph{आस्टसीलस्कप}\index{आस्टसीलस्कप} समय के विरुद्ध वोल्टेज प्लॉट करता है;
इसकी ग्रिडयुक्त स्क्रीन पर वक्र एक है \emph{आस्टसीलग्रम}%
\index{आस्टसीलग्रम}. दो सेटिंग्स स्क्रीन डिवीजनों को भौतिक में परिवर्तित करती हैं
मान: \emph{समय आधार}\index{समय आधार} (प्रति क्षैतिज सेकंड
विभाजन) और ऊर्ध्वाधर \emph{पाना}\index{पाना} (वोल्ट प्रति ऊर्ध्वाधर
प्रभाग).
\end{definition}

\begin{method}[स्क्रीन से लेकर नंबर तक]\label{met:g10:signals-and-waves:oscillogram}
\begin{enumerate}
  \item एक पूर्ण चक्र द्वारा फैले क्षैतिज विभाजनों की गणना करें
        (शिखा से शिखा तक सबसे सुरक्षित है); से गुणा करें \omterm{def:g10:signals-and-waves:oscilloscope}{समय आधार} पाने के
        $ T $, तब $ f = 1/T $.
  \item केंद्र रेखा से शिखा तक ऊर्ध्वाधर विभाजनों की गणना करें
        से गुणा करें \omterm{def:g10:signals-and-waves:oscilloscope}{पाना} पाने के लिए \omterm{def:g10:signals-and-waves:amplitude}{आयाम} (शिखा से गर्त तक देता है
        शिखर-से-शिखर मान: इसे आधा कर दें)।
\end{enumerate}
\end{method}

\begin{example}[एक पूर्ण वाचन]\label{ex:g10:signals-and-waves:reading}
पर \omterm{def:g10:signals-and-waves:oscilloscope}{आस्टसीलग्रम} नीचे, \omterm{def:g10:signals-and-waves:oscilloscope}{समय आधार} है \qty{5}{ms} प्रति प्रभाग और
द \omterm{def:g10:signals-and-waves:oscilloscope}{पाना} \qty{2}{V} प्रति प्रभाग. एक चक्र फैलता है $4.0$ प्रभाग:
$ T = 4.0 \times \qty{5}{ms} = \qty{20}{ms}$ और
$ f = 1/\qty{0.020}{s} = \qty{50}{Hz}$ --- \omterm{def:g10:refraction:fiber}{मुख्य} बातें फिर से। एक शिखा बैठती है
$3.0$ धुरी के ऊपर विभाजन: \omterm{def:g10:signals-and-waves:amplitude}{आयाम} है
$3.0 \times \qty{2}{V} = \qty{6.0}{V}$.
\end{example}

\section{ध्वनि और विद्युत चुम्बकीय संकेत}

\begin{definition}[आवाज़]\label{def:g10:signals-and-waves:sound}
\emph{आवाज़}\index{आवाज़} एक है \omterm{def:g10:pressure:pressure}{दबाव} \omterm{def:g10:signals-and-waves:signal}{संकेत}: एक कम्पायमान वस्तु ---
तार, झिल्ली, स्वर रज्जु --- हवा पर लयबद्ध रूप से धक्का देते हैं, और
संपीडन लगभग बाहर की ओर यात्रा करते हैं \qty{340}{m/s}. एक माइक्रोफोन
आने वाले को परिवर्तित करता है \omterm{def:g10:pressure:pressure}{दबाव} एक वोल्टेज में; कान का पर्दा, तंत्रिका में
आवेग. ध्वनि को एक माध्यम की आवश्यकता होती है: वैक्यूम जार के नीचे बजाई गई घंटी गिरती है
जैसे ही हवा बाहर निकाली जाती है, चुप हो जाता है।
\end{definition}

\begin{definition}[लहर]\label{def:g10:signals-and-waves:wave}
एक यात्रा क्षोभ --- ए आवाज़का संपीड़न पैटर्न, तरंग
एक तालाब पर, एक रेडियो पल्स --- एक है \emph{लहर}\index{लहर}. माध्यम,
जब कोई होता है, तो वहीं रहता है: पानी का प्रत्येक टुकड़ा अपनी जगह पर ही रहता है
लहर तालाब को पार कर जाती है. तरंगों की ज्यामिति --- \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य},
व्यतिकरण, विवर्तन --- को बाद के अध्याय में लिया गया है; यह
अध्याय को केवल उनकी गति की आवश्यकता है।
\end{definition}

\begin{definition}[अल्ट्रासाउंड और इन्फ्रासाउंड]\label{def:g10:signals-and-waves:ultrasound}
मानव कान सुनता है \omterm{def:g10:signals-and-waves:sound}{आवाज़} के बारे में के बीच \qty{20}{Hz} और \qty{20}{kHz}
(उम्र के साथ छत गिरती है)। \omterm{def:g10:signals-and-waves:sound}{आवाज़} ऊपर \qty{20}{kHz} है
\emph{अल्ट्रासाउंड}\index{अल्ट्रासाउंड} --- चमगादड़ शिकार करते हैं \qty{50}{kHz},
मेडिकल स्कैनर कुछ मेगाहर्ट्ज़ पर शरीर की छवि बनाते हैं। \omterm{def:g10:signals-and-waves:sound}{आवाज़} नीचे
\qty{20}{Hz} है \emph{इन्फ्रासाउंड}\index{इन्फ्रासाउंड}, हाथियों द्वारा महसूस किया गया
और भूकंपमापी.
\end{definition}

\begin{proposition}[विद्युत चुम्बकीय संकेत]\label{prop:g10:signals-and-waves:em}
रेडियो, वाई-फाई और लाइट हैं \emph{विद्युत चुम्बकीय संकेत}\index{विद्युत चुम्बकीय संकेत}: एक परिवार, सभी यात्रारत
निर्वात के माध्यम से --- किसी माध्यम की आवश्यकता नहीं --- और सब एक गति से, गति
के प्रकाश का \cref{ch:g10:refraction},
\[
c = \qty{3.00e8}{m/s} \quad \text{(very nearly the same in air).}
\]
\end{proposition}
\admitted

\begin{remark}\label{rem:g10:signals-and-waves:honest}
यहाँ यह एक प्रायोगिक तथ्य है --- सूर्य का प्रकाश पार करता है $150$ मिलियन
खाली जगह के किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर}; यह ईमानदारी से वर्ष 2 खंड में लिया गया है
बिजली और चुंबकत्व के नियमों से. कंट्रास्ट रखें: \omterm{def:g10:signals-and-waves:sound}{आवाज़} है
एक कंपन \emph{का} माध्यम और इसके बिना मर जाता है; एक \omterm{prop:g10:signals-and-waves:em}{विद्युत चुम्बकीय
संकेत} निर्वात में अपने क्षेत्र को लाखों गुना तेजी से ले जाता है।
\end{remark}

\section{गूँज: देरी से दूरियाँ}

ए \omterm{def:g10:signals-and-waves:signal}{संकेत} गति से चल रहा है $ v $ कवर $ d = v\,t $ एक समय में $ t $: हर देरी
भेष में एक दूरी है. इससे भी बेहतर, एक छोटी पल्स भेजें
बाधा और समय यह \emph{गूंज}\index{गूंज} --- राउंड ट्रिप कवर करता है $2d $.

\begin{method}[प्रतिध्वनि लेकर]\label{met:g10:signals-and-waves:echo}
\begin{enumerate}
  \item एक छोटी पल्स भेजें; राउंड-ट्रिप का समय मापें $ t $ की गूंज.
  \item गति लो $ v $ की \omterm{def:g10:signals-and-waves:signal}{संकेत} \emph{माध्यम में पार कर गया}.
  \item नाड़ी ढकी हुई $2d = v\,t $, इसलिए $ d = \dfrac{v\,t}{2}$.
\end{enumerate}
\omterm{def:g10:signals-and-waves:sound}{आवाज़} पानी में दालें बनाती हैं \emph{सोनार}\index{सोनार}
($ v \approx \qty{1500}{m/s}$); हवा में रेडियो स्पंदन बनाते हैं
\emph{राडार}\index{राडार} ($ v = c $); \omterm{def:g10:signals-and-waves:ultrasound}{अल्ट्रासाउंड} शरीर में नाड़ियाँ बनती हैं
मेडिकल स्कैनर ($ v \approx \qty{1540}{m/s}$ कोमल ऊतकों में)।
\end{method}

\begin{example}[सोनार]\label{ex:g10:signals-and-waves:sonar}
एक जहाज़ का \omterm{met:g10:signals-and-waves:echo}{सोनार} पिंग्स और समुद्र तल गूंज के बाद लौटता है

\section{समस्या: सोनार, तूफ़ान और कार्डियोग्राम}

\begin{problem}[{सप्ताहांत समस्या--- दुनिया के संकेतों को पढ़ना: एक
कानून, $ d = v\,t $, तूफ़ान का पता लगाता है, समुद्र तल का नक्शा बनाता है, दिल की धड़कन गिनता है और
बताता है कि जीपीएस कैसे काम करता है}]
\label{pb:g10:signals-and-waves:1}
एक मछली पकड़ने वाली नाव तूफानी रात में काम करती है: क्षितिज पर बिजली चमकती है,
द \omterm{met:g10:signals-and-waves:echo}{सोनार} नीचे झाडू लगाते हुए, कप्तान का दिल डॉक्टर के कागज़ पर
पट्टी, एक जीपीएस रिसीवर सुन रहा है \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों. चार यंत्र, एक
कानून. लेना $ v_{\text{sound}} = \qty{340}{m/s}$ हवा में, \qty{1500}{m/s} में
समुद्री जल, और $ c = \qty{3.00e8}{m/s}$.

\medskip
\noindent\textbf{Part I --- The storm.}
\begin{enumerate}
  \item एक फ्लैश; गड़गड़ाहट इस प्रकार है \qty{9.0}{s} बाद में। यह कितनी दूर है?
  \item उस दूरी पर प्रकाश की यात्रा के समय की गणना करें, और औचित्य सिद्ध करें
        फ़्लैश को तात्कालिक मानकर।
  \item नाविक चमक और गड़गड़ाहट के बीच के सेकंड गिनते हैं और विभाजित करते हैं
        किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} पाने के लिए तीन से. नियम का औचित्य सिद्ध करें.
  \item पाँच मिनट बाद, एक फ्लैश आता है \qty{6.0}{s}. अब कितनी दूर?
        तूफ़ान की औसत दृष्टिकोण गति ज्ञात कीजिए \unit{m/s} और \unit{km/h}.
  \item कड़कने की बजाय गरजती है गड़गड़ाहट: बिजली का चैनल है
        किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} लंबा है, इसलिए इसके हिस्से अलग-अलग दूरी पर स्थित हैं। यदि
        चैनल से फैला है \qty{2.0}{km} को \qty{5.0}{km} से
        नाव, गड़गड़ाहट कितनी देर तक चलती है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{met:g10:signals-and-waves:echo}{सोनार}.}
\begin{enumerate}[resume]
  \item  \omterm{met:g10:signals-and-waves:echo}{सोनार} पिंग्स; समुद्र तल उत्तर देता है \qty{0.90}{s}. गहराई?
  \item शेल्फ के ऊपर गूंज को छोटा करता है \qty{0.20}{s}. गहराई?
  \item एक पिंग वापस आता है \emph{दो} इकोज, पर \qty{0.16}{s} और
        \qty{0.20}{s}. उनकी व्याख्या करें; नीचे से कितना ऊपर है
        मछली का किनारा तैरना?
  \item  \omterm{met:g10:signals-and-waves:echo}{सोनार} हर पिंग \qty{0.50}{s}. सबसे बड़ी गहराई क्या है
        यह बिना किसी भ्रम के माप सकता है, और इससे परे क्या गलत होता है?
  \item हवा में भी वैसा ही \qty{0.60}{s} देरी का मतलब कितनी दूरी होगी?
        नैतिक: विलंब को विलंब में बदलने से पहले आपको क्या जानना चाहिए?
        दूरी?

\medskip
\noindent\textbf{Part III --- The cardiogram.}
\begin{enumerate}[resume]
  \item स्किपर के ईसीजी स्पाइक्स हैं \qty{0.86}{s} अलग। \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}?
        प्रति मिनट धड़कता है?
  \item पट्टी आगे बढ़ती है \qty{25}{mm/s}. स्पाइक स्पेसिंग ने क्या किया
        डॉक्टर उपाय?
  \item जाल खींचने के बाद दूरी बनाई जाती है \qty{15}{mm}. नई हृदय गति?
