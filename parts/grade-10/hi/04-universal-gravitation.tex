\chapter{सार्वभौमिक गुरुत्वाकर्षण और वजन}\label{ch:g10:universal-gravitation}

एक सेब गिराओ: वह गिर जाता है। चंद्रमा, पृथ्वी की त्रिज्या से साठ ऊपर, कभी दिखाई नहीं देता।
न्यूटन की महान अंतर्दृष्टि चंद्रमा है \emph{है} गिरना --- के बारे में
1.4 मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} प्रति सेकंड, बिल्कुल आकर्षण की मात्रा
दूरी के वर्ग की भविष्यवाणी के अनुसार कमजोर होना। एक सूत्र, एक के साथ
सार्वभौमिक स्थिरांक, सेब, चंद्रमा, ज्वार और को नियंत्रित करता है
ग्रह. प्रथम खंड का वर्णन किया गया है \omterm{def:g10:universal-gravitation:force}{आकर्षण-शक्ति} शब्दों के साथ; यह
अध्याय इसे मापता है।

\section{सार्वभौमिक गुरुत्वाकर्षण का नियम}

\begin{definition}[गुरुत्वाकर्षण बल]\label{def:g10:universal-gravitation:force}
कोई भी दो शरीर एक-दूसरे को आकर्षित करते हैं, चाहे वे किसी भी चीज से बने हों और जैसे भी हों
वे दूर-दूर बैठते हैं। ये आकर्षण है \emph{गुरुत्वाकर्षण बल}\index{गुरुत्वाकर्षण बल}, और घटना है
\emph{आकर्षण-शक्ति}\index{आकर्षण-शक्ति}. इसे किसी संपर्क, किसी रस्सी आदि की आवश्यकता नहीं है
कोई माध्यम नहीं, और यह है \emph{आपसी}: प्रत्येक शरीर दूसरे को खींचता है।
\end{definition}

\begin{theorem}[सार्वभौमिक गुरुत्वाकर्षण का नियम]\label{thm:g10:universal-gravitation:law}
जनसमूह के दो शरीर $m_1$ और $ m_2$ (में \unit{kg}), जिनके केंद्र हैं a
दूरी $ d $ अलग (में) \unit{m}), दो शक्तियों से एक दूसरे को आकर्षित करते हैं
केंद्रों को जोड़ने वाली रेखा के साथ निर्देशित, प्रत्येक की ओर इशारा करता है
अन्य शरीर, समान परिमाण का
\[
F = G\,\frac{m_1 m_2}{d^2}
\qquad \text{(in \unit{N})},
\]
कहाँ $ G $ ब्रह्मांड में प्रत्येक जोड़ी पिंडों के लिए समान स्थिरांक है।
\end{theorem}
\admitted

\begin{remark}[कानून कहां से आता है]
न्यूटन ने ग्रहों की देखी गई गति से इस नियम का प्रतिपादन किया
(केप्लर के नियम); ईमानदार कटौती वर्ष 1 में की जाती है
आयतन. \Cref{ch:g12:satellites-kepler} कानून को पहले ही काम पर लगा देंगे
पर \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों और ग्रह.
\end{remark}

\begin{definition}[गुरुत्वाकर्षण स्थिरांक]\label{def:g10:universal-gravitation:constant}
स्थिरांक
\[
G = \qty{6.67e-11}{N.m^2/kg^2}
\]
है \emph{गुरुत्वाकर्षण स्थिरांक}\index{गुरुत्वाकर्षण स्थिरांक}. यह
रोजमर्रा की वस्तुओं के बीच गुरुत्वाकर्षण का कारण शून्य मान है
अगोचर: केवल तभी जब कम से कम एक द्रव्यमान खगोलीय हो
$ G m_1 m_2 / d^2$ कुछ भी करने के लिए राशि।
\end{definition}

\begin{remark}[सूत्र पढ़ना]\label{rem:g10:universal-gravitation:reading}
चार विशेषताएं ध्यान देने योग्य हैं।
\begin{itemize}
  \item \emph{परस्पर और समान.} धरती तुम्हें अपने साथ खींचती है \omterm{def:g10:universal-gravitation:weight}{वज़न}
        --- और आप पृथ्वी को बिल्कुल उसी बल से खींचते हैं। (द
        पृथ्वी बस कम प्रतिक्रिया देती है: यह इसके बारे में है $\num{e23}$ बार
        भारी.)
  \item \emph{उलटा वर्ग.} दोहरीकरण $ d $ बल को विभाजित करता है $4$;
        दस गुना आगे बढ़ने से यह विभाजित हो जाता है $100$.
  \item \emph{केन्द्रों के बीच की दूरी.} गोलाकार पिंडों के लिए --- तारे,
        ग्रह, तोप के गोले --- $ d $ केंद्र से केंद्र तक मापा जाता है, नहीं
        सतह से सतह तक.
  \item \emph{दोनों जनसमूह मायने रखते हैं।} किसी भी द्रव्यमान को दोगुना करने से द्रव्यमान दोगुना हो जाता है
        बल; सूत्र सममित है $ m_1$ और $ m_2$.
\end{itemize}
\end{remark}

\begin{example}[दो दोस्त]\label{ex:g10:universal-gravitation:friends}
दो \qty{70}{kg} मित्र अपने केंद्रों के साथ खड़े हैं \qty{1.0}{m} अलग:
\[
F = \num{6.67e-11} \times \frac{70 \times 70}{1.0^2}
\approx \qty{3.3e-7}{N}.
\]
प्रत्येक मित्र का \omterm{def:g10:universal-gravitation:weight}{वज़न} के बारे में है \qty{6.9e2}{N} --- दो \emph{अरब}
गुना बड़ा. उनका पारस्परिक आकर्षण वास्तविक है, लेकिन महसूस करना निराशाजनक है।
\end{example}

\begin{example}[पृथ्वी चंद्रमा को धारण करती है]\label{ex:g10:universal-gravitation:earthmoon}
साथ $ m_1 = \qty{5.97e24}{kg}$ (धरती), $ m_2 = \qty{7.35e22}{kg}$ (चंद्रमा)
और $ d = \qty{3.84e8}{m}$:
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{2.0e20}{N}.
\]
और चंद्रमा पृथ्वी को उसी के साथ पीछे खींचता है \qty{2.0e20}{N} --- ए
समुद्र उत्तर में दिन में दो बार ज्वार-भाटा की तरह खींचते हैं।
\end{example}

\begin{method}[गुरुत्वाकर्षण बल की गणना]\label{met:g10:universal-gravitation:compute}
\begin{enumerate}
  \item जनता को परिवर्तित करें \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} और केंद्र-से-केंद्र
        \omterm{def:g10:orders-of-magnitude:unit}{मीटर} की दूरी.
  \item आवेदन करना $ F = G m_1 m_2 / d^2$.
  \item दस की घात को अंकों से अलग-अलग जाँचें, जैसे कि
        \cref{ch:g10:orders-of-magnitude}: \omterm{def:g10:universal-gravitation:force}{गुरुत्वाकर्षण बल} से रेंज $\num{e-7}$ न्यूटन (दो लोग ए \omterm{def:g10:orders-of-magnitude:unit}{मीटर} अलग) से $\num{e22}$ न्यूटन
        (पृथ्वी पर सूर्य), तो एक गलत जगह \omterm{def:g10:orders-of-magnitude:scinot}{प्रतिपादक} पकड़ना आसान है.
\end{enumerate}
\end{method}

\section{वजन और गुरुत्वाकर्षण की शक्ति}

\begin{definition}[वज़न]\label{def:g10:universal-gravitation:weight}
 \emph{वज़न}\index{वज़न} $\vect P $ किसी वस्तु की सतह पर
एक तारा (एक ग्रह, एक चंद्रमा, \dots) है \omterm{def:g10:universal-gravitation:force}{गुरुत्वाकर्षण बल} सितारा
उस पर जोर देता है. यह तारे के केंद्र की ओर इशारा करता है --- जिस दिशा में हम हैं
बुलाओ \emph{खड़ा}\index{खड़ा} --- और न्यूटन में मापा जाता है।
\end{definition}

\begin{proposition}[किसी तारे की सतह पर भार]\label{prop:g10:universal-gravitation:surfaceg}
द्रव्यमान के एक गोलाकार तारे की सतह पर $ M $ और त्रिज्या $ R $, एक वस्तु
द्रव्यमान का $ m $ है \omterm{def:g10:universal-gravitation:weight}{वज़न}
\[
P = m\,g,
\qquad\text{where}\qquad
g = \frac{G M}{R^2}
\]
केवल तारे पर निर्भर करता है, वस्तु पर नहीं।
\end{proposition}

\begin{proof}
वस्तु का केंद्र और तारे का केंद्र एक दूरी हैं $ R $ अलग, तो
सार्वभौमिक का नियम \omterm{def:g10:universal-gravitation:force}{आकर्षण-शक्ति}
(\cref{thm:g10:universal-gravitation:law}) देता है
$ P = G M m / R^2 = m \times \left(G M / R^2\right)$. ब्रैकेट है
सतह पर प्रत्येक वस्तु के लिए समान: इसे कॉल करें $ g $.
\end{proof}

\begin{definition}[गुरुत्वाकर्षण क्षेत्र की ताकत]\label{def:g10:universal-gravitation:fieldstrength}
मात्रा $ g = G M / R^2$, प्रति न्यूटन में \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम}, है
\emph{गुरुत्वाकर्षण क्षेत्र की ताकत}\index{गुरुत्वाकर्षण क्षेत्र की ताकत}
तारे की सतह पर: तारा प्रत्येक को खींचता है \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} साथ $ g $ न्यूटन.
पृथ्वी पर, $ g = \qty{9.81}{N/kg}$.
\end{definition}

\begin{example}[कम्प्यूटिंग $ g $]\label{ex:g10:universal-gravitation:gvalues}
पृथ्वी के लिए ($ M = \qty{5.97e24}{kg}$, $ R = \qty{6.37e6}{m}$):
\[
g = \frac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.37e6})^2}
= \frac{\num{3.98e14}}{\num{4.06e13}}
\approx \qty{9.81}{N/kg}.
\]
चंद्रमा के लिए ($ M = \qty{7.35e22}{kg}$, $ R = \qty{1.74e6}{m}$) वही
गणना देती है $ g_{\text{Moon}} \approx \qty{1.62}{N/kg}$ --- छह
कई गुना कम: चंद्रमा बहुत हल्का है, और छोटा होने से कोई फर्क नहीं पड़ता
इसके लिए. मंगल ग्रह के लिए, $ g_{\text{Mars}} \approx \qty{3.73}{N/kg}$.
\end{example}

\begin{remark}[अंतरिक्ष यात्री क्यों उछलते हैं?]\label{rem:g10:universal-gravitation:bounce}
द्रव्यमान का एक अंतरिक्ष यात्री \qty{120}{kg} (बॉडी प्लस सूट) का वजन होता है
$120 \times 9.81 \approx \qty{1180}{N}$ पृथ्वी पर लेकिन केवल
$120 \times 1.62 \approx \qty{194}{N}$ चंद्रमा पर. मांसपेशियां प्रशिक्षित
के अंतर्गत \qty{1180}{N} अचानक भार का छठा हिस्सा ले जाना: इसलिए
प्रसिद्ध धीमी, बाउंडिंग चंद्र चाल। द्रव्यमान --- और इसके साथ प्रयास
की जरूरत है \emph{रुकना} या \emph{मोड़} --- अपरिवर्तित है, यही कारण है
चंद्र अंतरिक्ष यात्री बहुत बार गिरे।
\end{remark}

\section{द्रव्यमान या वज़न? संतुलन और स्प्रिंग स्केल}

रोज़मर्रा की भाषा कहती है कि आटे की एक बोरी का वजन तीन किलो होता है। भौतिकी
उस वाक्य को दो भागों में विभाजित करता है।

\begin{remark}[द्रव्यमान भार नहीं है]\label{rem:g10:universal-gravitation:massweight}
\begin{itemize}
  \item  \emph{द्रव्यमान} $ m $ पदार्थ की मात्रा को मापता है
        \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम}. यह अकेले वस्तु का गुण है: समान पर
        पृथ्वी, चंद्रमा पर, या गहरे अंतरिक्ष में भटकते हुए।
  \item  \emph{\omterm{def:g10:universal-gravitation:weight}{वज़न}} $ P = mg $ न्यूटन में, लगाया गया एक बल है
        \emph{एक सितारे द्वारा} वस्तु पर. यह एक तारे से दूसरे तारे में बदलता रहता है
        के साथ $ g $ --- और हर तारे से दूर हो जाता है।
\end{itemize}
उन्हें भ्रमित करना बाजार में हानिरहित है और भौतिकी में घातक है
समस्या: \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} नहीं है कोई \omterm{def:g10:orders-of-magnitude:unit}{इकाई} शक्ति के।
\end{remark}

\begin{method}[संतुलन और स्प्रिंग स्केल]\label{met:g10:universal-gravitation:instruments}
\begin{itemize}
  \item ए \emph{किरण संतुलन} वस्तु की तुलना मानक द्रव्यमान से करता है:
        दोनों पैन' तौल उसी के अनुसार स्केल करें $ g $, तो तुलना
        मापता है \emph{द्रव्यमान} ---किसी भी स्टार पर एक ही फैसला.
  \item ए \emph{वसंत पैमाना} इसके स्प्रिंग को खींचने वाले बल को मापता है:
        यह पढ़ता है \emph{\omterm{def:g10:universal-gravitation:weight}{वज़न}}, न्यूटन में, और इसकी रीडिंग बदल जाती है
        स्टार से स्टार तक. द्रव्यमान को पुनः प्राप्त करने के लिए, स्थानीय से विभाजित करें
        क्षेत्र की ताकत: $ m = P/g $.
\end{itemize}
\end{method}

\begin{example}[चंद्रमा पर आटा]\label{ex:g10:universal-gravitation:flour}
ए \qty{3.0}{kg} आटे की बोरी संतुलन बनाती है \qty{3.0}{kg} मानक का
पृथ्वी पर जनसमूह \emph{और} चंद्रमा पर. हालाँकि, एक स्प्रिंग स्केल पढ़ता है
$3.0 \times 9.81 \approx \qty{29}{N}$ पृथ्वी पर और
$3.0 \times 1.62 \approx \qty{4.9}{N}$ चंद्रमा पर. एक व्यापारी बेच रहा है
``न्यूटन द्वारा'' चंद्रमा पर छह गुना आटा वितरित किया जाएगा ---
व्यापार, विज्ञान की तरह, द्रव्यमान पर चलता है।
\end{example}

\section{पृथ्वी के चारों ओर गिरना: न्यूटन की तोप}

एक पत्थर को क्षैतिज रूप से फेंकें: गुरुत्वाकर्षण अपना मार्ग एक चाप में मोड़ देता है और वह
कुछ \omterm{def:g10:orders-of-magnitude:unit}{मीटर} की दूरी पर उतरता है। इसे तेजी से फेंको: यह दूर तक गिरता है। न्यूटन का
विचार प्रयोग विचार को उसकी सीमा तक धकेलता है। पृथ्वी गोल है, और
इसकी सतह लगभग गिर जाती है \qty{5}{m} प्रत्येक के लिए क्षैतिज से नीचे
\qty{8}{km} यात्रा की; एक स्वतंत्र रूप से गिरता हुआ शरीर गिरता है \qty{4.9}{m} इसके में
पहला \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} (\cref{ch:g12:projectile-motion} इसे स्थापित करेंगे)। तो ए
प्रक्षेप्य पृथ्वी पर चक्कर लगा रहा है \qty{8}{km/s} बिल्कुल उतनी ही तेजी से गिरता है
ज़मीन मुड़ जाती है: यह बिना उतरे ही हमेशा के लिए गिर जाती है।

\begin{definition}[उपग्रह और कक्षा]\label{def:g10:universal-gravitation:satellite}
ए \emph{उपग्रह}\index{उपग्रह} एक तारे का एक पिंड है जो गिरता है
तारे की सतह तक पहुंचे बिना उसके चारों ओर; जिस पथ को वह दोहराता है
इसका है \emph{कक्षा}\index{कक्षा}.
\end{definition}

\begin{remark}[चंद्रमा एक गिरता हुआ पत्थर है]\label{rem:g10:universal-gravitation:fallingmoon}
चंद्रमा पृथ्वी का प्राकृतिक रूप है \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}: यह लगभग हमारी ओर गिरता है
\qty{1.4}{mm} हर सेकंड, और इसकी बग़ल में गति इसे चारों ओर ले जाती है
इससे पहले कि यह और करीब आ सके
(\cref{exo:g10:universal-gravitation:15} आपको न्यूटन को फिर से बनाने की सुविधा देता है
इस नंबर की जांच करें)। एक परिक्रमा स्टेशन के अंतरिक्ष यात्री हैं
उनके जहाज के समान ही मुक्त पतन --- वह, और गुरुत्वाकर्षण की कोई अनुपस्थिति नहीं,
इसीलिए वे तैरते हैं। \omterm{def:g10:universal-gravitation:force}{आकर्षण-शक्ति} से प्रत्येक प्रक्षेप पथ को आकार देता है
ग्रह के दीर्घवृत्त के लिए पत्थर का चाप; की मात्रात्मक कहानी कक्षाओं
में बताया गया है \cref{ch:g12:satellites-kepler}.
\end{remark}

\begin{enumerate}
  \item उनके पारस्परिक गुरुत्वाकर्षण आकर्षण की गणना करें।
  \item कौन सा द्रव्यमान है a \omterm{def:g10:universal-gravitation:weight}{वज़न} पृथ्वी पर इस बल के बराबर?
  \item बल आश्चर्यजनक रूप से बड़ा है --- फिर भी टैंकर नहीं दिखाते हैं
        एक साथ बहने की प्रवृत्ति. त्वरण की गणना करें
        $ a = F/m $ यह एक टैंकर और टिप्पणी देता है।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:7}

  \item एक ग्राहक `` के लिए भुगतान करता है\qty{3.0}{kg}''. कौन सा यंत्र
        किसी भी दुनिया में उचित सौदे की गारंटी देता है, और क्यों?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:8}

  \item इसे सतही मान के प्रतिशत के रूप में व्यक्त करें।
  \item जहाज पर मौजूद अंतरिक्ष यात्री तैरते हैं। इसे अपने उत्तर से मिलाइये।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:10}

        पानी, फिर चंद्रमा द्वारा लगाया गया बल
        \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम}.
  \item सूर्य अधिक ज़ोर से खींचता है --- किस कारक से?
  \item फिर भी चंद्रमा ज्वार पर हावी है। ज्वार किसके द्वारा संचालित नहीं होता है?
        खुद को खींचो लेकिन द्वारा \emph{अंतर} खींचतान के बीच
        पृथ्वी के निकट और दूर के भाग। समझाओ, बिना
        कंप्यूटिंग, क्यों पास का चंद्रमा दूर के सूर्य को हरा सकता है
        वह कसौटी.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:11}
\begin{enumerate}
  \item एक ग्रह का द्रव्यमान पृथ्वी से दोगुना और उसकी त्रिज्या दोगुनी है।
        इसकी सतह क्षेत्र की ताकत को पृथ्वी के संदर्भ में व्यक्त करें
        $ g $, में फिर \unit{N/kg}.
  \item पृथ्वी के द्रव्यमान वाले किसी ग्रह को इसके लिए किस त्रिज्या की आवश्यकता होगी?
        सतही क्षेत्र की ताकत होनी चाहिए $2g $?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:12}
पृथ्वी पर कहीं - चंद्रमा रेखा, दोनों एक अंतरिक्ष जांच पर खींचते हैं
रद्द करें. होने देना $ x $ पृथ्वी के केंद्र से उस बिंदु तक की दूरी हो
और $ D $ पृथ्वी - चंद्रमा की दूरी।
\begin{enumerate}
  \item उस बिंदु पर दिखाओ
        $\left(\dfrac{x}{D - x}\right)^{2}
        = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.
  \item परिणाम निकालना $ x $. वह चंद्रमा की यात्रा का कितना अंश है?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:13}
न्यूटन की तोप, संख्याओं के साथ। एक प्रक्षेप्य पृथ्वी की ओर उड़ता है
क्षैतिज गति से $ v $.
\begin{enumerate}
  \item बाद $ x = \qty{8.0}{km}$ क्षैतिज यात्रा का, गोलाकार
        पृथ्वी ऊंचाई से ``गिर गई'' है $ h $ संतुष्टि देने वाला
        $(R + h)^2 = R^2 + x^2$. का उपयोग करते हुए
        $(R + h)^2 \approx R^2 + 2Rh $ छोटे के लिए $ h $, बताते हैं कि
        $ h \approx x^2 / (2R)$ और गणना करें $ h $.
  \item एक स्वतंत्र रूप से गिरता हुआ शरीर गिरता है \qty{4.9}{m} इसके पहले सेकंड में.
        साथ तुलना करें $ h $ और उस गति को कम करें जिस पर प्रक्षेप्य
        कभी नहीं उतरता.
  \item एक पूर्ण की अवधि की गणना करें \omterm{def:g10:universal-gravitation:satellite}{कक्षा} उस गति से, उड़ते हुए
        सतह.
\end{enumerate}
\end{exercise}

  \item क्या है \omterm{def:g10:universal-gravitation:weight}{वज़न}, वहाँ, का \qty{1.0}{mg} बालु के कन?

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:15}
न्यूटन का चंद्रमा परीक्षण (1666)। चंद्रमा का \omterm{def:g10:universal-gravitation:satellite}{कक्षा} त्रिज्या के बारे में है $60$ पृथ्वी
त्रिज्या.
\begin{enumerate}
  \item पृथ्वी का खिंचाव किस कारक से है? \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} पर कमजोर
        चंद्रमा पृथ्वी की सतह की तुलना में, यदि गुरुत्वाकर्षण का अनुसरण करता है
        उलटा वर्ग?
  \item सतह के निकट, एक गिरता हुआ पिंड गिरता है \qty{4.9}{m} इसके में
        पहला \omterm{def:g10:orders-of-magnitude:unit}{दूसरा}. चंद्रमा को पृथ्वी की ओर कितनी दूरी तक गिरना चाहिए?
        प्रत्येक सेकंड?
  \item के विरुद्ध जाँच करें \omterm{def:g10:universal-gravitation:satellite}{कक्षा}: चंद्रमा अपनी त्रिज्या के घेरे को कवर करता है
        $ d = \qty{3.84e8}{m}$ में $ T = 27.3$ दिन. इसकी गति की गणना करें
        $ v $, दूरी $ x $ यह एक सेकंड में ढक जाता है, फिर गिर जाता है
        $ h \approx x^2/(2d)$ (के रूप में
        \cref{exo:g10:universal-gravitation:13}).
  \item प्रश्न 2 और 3 के उत्तरों की तुलना करें और बताएं कि न्यूटन क्या है
        निष्कर्ष निकाला।
\end{enumerate}
\end{exercise}

\section{समस्या: पृथ्वी का वजन करना}

\begin{problem}[{सप्ताहांत समस्या --- एक शेड में दो सीसे के गोले से
पृथ्वी का द्रव्यमान और फिर सूर्य का द्रव्यमान: क्या एक छोटा सा स्थिरांक,
एक बार मापा गया, इसके लायक है}]
\label{pb:g10:universal-gravitation:1}
1798 में हेनरी कैवेंडिश ने दो छोटे सीसे वाली एक हल्की छड़ को लटका दिया
एक पतले तार से गोले, दो बड़े सीसे के गोले को पास लाया, और
तार के लगभग न के बराबर मोड़ को मापा। अखबारों ने कहा
उसके पास था \emph{पृथ्वी को तौला} --- और वे सही थे: एक बार $ G $ है
ज्ञात, पृथ्वी का अपना $ g $ और त्रिज्या उसके द्रव्यमान को सौंप देती है, और
पृथ्वी का वार्षिक \omterm{def:g10:universal-gravitation:satellite}{कक्षा} सूर्य को सौंप दो. यह समस्या फिर से सामने आती है
पूरी शृंखला. संपूर्ण डेटा कार्ड का उपयोग करें.

\medskip
\noindent\textbf{Part I --- The feeblest force.}
\begin{enumerate}
  \item एफिल टॉवर के लोहे का द्रव्यमान लगभग है \qty{7.3e6}{kg}. ए
        \qty{55}{kg} पर्यटक स्टैंड \qty{100}{m} इसके केंद्र से
        द्रव्यमान. पर्यटक पर टावर के खिंचाव की गणना करें।
  \item से इसकी तुलना करें \omterm{def:g10:universal-gravitation:weight}{वज़न} एक का \qty{1.0}{mg} बालु के कन।
  \item आनुपातिकता ड्रिल: क्या होता है? \omterm{def:g10:universal-gravitation:force}{गुरुत्वाकर्षण बल}
        यदि (ए) दूरी से गुणा किया जाता है $10$; (बी) दोनों द्रव्यमान हैं
        से गुणा किया गया $1000$; (सी) दोनों द्रव्यमान \emph{और} दूरी
        से गुणा किया जाता है $1000$?
  \item ए पर पृथ्वी के खिंचाव की गणना करें \qty{1.00}{kg} इस पर आपत्ति
        सतह, से $ M $, $ R $ और $ G $. क्या रोजमर्रा का नाम और प्रतीक
        क्या यह नंबर मौजूद है?
  \item गुरुत्वाकर्षण अब तक की सबसे कमज़ोर अंतःक्रिया है ---
        और फिर भी यह ब्रह्मांड को चलाता है। एक या दो वाक्यों में समझाइये
        आप पर पृथ्वी का खिंचाव आपके पड़ोसी पर हावी क्यों है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Cavendish's balance.}
मरोड़ संतुलन में, द्रव्यमान का एक बड़ा गोला $ m_1 = \qty{158}{kg}$
द्रव्यमान के एक छोटे से भाग को आकर्षित करता है $ m_2 = \qty{0.73}{kg}$; उनके केंद्र हैं
$ d = \qty{0.23}{m}$ अलग।
\begin{enumerate}[resume]
  \item के आधुनिक मूल्य के साथ $ G $, के बीच बल की गणना करें
        दो गोले.
  \item की गणना करें \omterm{def:g10:universal-gravitation:weight}{वज़न} छोटे गोले का और अनुपात का
        दो ताकतें. कैवेंडिश को मरोड़ वाले तार की आवश्यकता क्यों पड़ी?
        कोई सामान्य पैमाना?
  \item कैवेंडिश का मापा मोड़ (आधुनिक में) से मेल खाता है इकाइयां) को
        $ F = \qty{1.47e-7}{N}$. उसका मान निकालें $ G $ और इसकी तुलना करें
        आज के साथ.
  \item साथ $ G = \qty{6.67e-11}{N.m^2/kg^2}$, $ g = \qty{9.81}{N/kg}$
        और $ R = \qty{6.37e6}{m}$, पृथ्वी का द्रव्यमान निकालें। (यह
        यह वह कदम है जिसका अखबारों ने जश्न मनाया।)
  \item पृथ्वी का औसत घनत्व ज्ञात कीजिए $\rho = M / V $, साथ
        $ V = \frac43 \pi R^3$. सतही चट्टानों के चारों ओर घनत्व होता है
        \qty{2.7e3}{kg/m^3}: तुलना से क्या पता चलता है
        गहरा आंतरिक भाग?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Other worlds.}
\begin{enumerate}[resume]
  \item डेटा कार्ड से, चंद्रमा के सतह क्षेत्र की पुनः गणना करें
        ताकत.
  \item पृथ्वी पर, एक अच्छा पुश-ऑफ़ आपके द्रव्यमान के केंद्र को ऊपर उठा देता है
        \qty{0.50}{m}. समान पुश-ऑफ के लिए, पहुँची गई ऊँचाई है
        के व्युत्क्रमानुपाती $ g $ (ऊर्जा अध्याय के रूप में,
        \cref{ch:g10:energy-conservation}, उचित ठहराएगा)। कितना ऊँचा
        क्या वही छलांग आपको चंद्रमा पर ले जाती है?
  \item बुध: $ M = \qty{3.30e23}{kg}$, $ R = \qty{2.44e6}{m}$.
        इसकी सतह क्षेत्र की ताकत की गणना करें और मंगल ग्रह से तुलना करें।
        बुध मंगल से बहुत छोटा है --- दोनों मान कैसे हो सकते हैं
        इतने करीब हो?
  \item पृथ्वी की सतह से कितनी ऊँचाई पर क्षेत्र है?
        ताकत घटकर इसके सतह मूल्य से आधी रह गई?
  \item पृथ्वी के केंद्र से कितनी दूरी पर पृथ्वी का खिंचाव होता है?
        प्रति \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} चंद्रमा की सतह के मूल्य में गिरावट,
        \qty{1.62}{N/kg}? इसे पृथ्वी त्रिज्या में व्यक्त करें।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Weighing the Sun.}
पृथ्वी एक निकट-गोलाकार यात्रा करती है \omterm{def:g10:universal-gravitation:satellite}{कक्षा} त्रिज्या का
$ d = \qty{1.496e11}{m}$ एक साल में, $ T = \qty{3.156e7}{s}$. के रूप में लें
दिया गया (यह सिद्ध है) \cref{ch:g12:satellites-kepler}) वह एक शरीर पर
त्रिज्या का एक वृत्ताकार पथ $ d $ गति से $ v $ खींचने की जरूरत है $ v^2/d $
न्यूटन प्रति \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} केंद्र की ओर.
\begin{enumerate}[resume]
  \item पृथ्वी की कक्षीय गति की गणना करें $ v $.
  \item प्रति खिंचाव का अनुमान लगाएं \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} कि सूर्य पृथ्वी पर प्रभाव डालता है।
  \item वह खिंचाव भी है $ G M_{\text{Sun}} / d^2$. का द्रव्यमान ज्ञात कीजिए
        सूर्य.
  \item वह कितनी पृथ्वी है? तुलना करना $ M_{\text{Sun}}$ साथ
        $ M_{\text{Earth}}$.
  \item समापन. कैवेंडिश के गोले, पृथ्वी, सूर्य: एक स्थिरांक
        $ G$ तीनों के लिए सेवा की. एक वाक्य में बताएं कि कौन सा शब्द है
        \emph{सार्वभौमिक} ``सार्वभौमिक'' में \omterm{def:g10:universal-gravitation:force}{आकर्षण-शक्ति}''हमें खरीद लिया
        यह समस्या.
\end{enumerate}
\end{problem}
