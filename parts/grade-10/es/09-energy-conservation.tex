\chapter{Energía: formas y conservación}\label{ch:g10:energy-conservation}

Un coche en movimiento, un teléfono cargado, un horno caliente y un martillo levantado
una propiedad: cada uno puede hacer que algo suceda. La física mide que
capacidad con un solo número --- \omterm{def:g10:energy-conservation:energy}{energía} --- y lleva sus cuentas
bajo una regla de hierro: el total nunca cambia, sólo se mueve y
cambia de disfraz. Este capítulo abre el libro mayor: los formularios \omterm{def:g10:energy-conservation:energy}{energía}
lleva, las cadenas que recorre y lo que compra un \omterm{def:g10:energy-conservation:kwh}{kilovatio-hora}.

\section{Energía: la moneda común}

\begin{definition}[energía y julio]\label{def:g10:energy-conservation:energy}
\emph{Energía}\index{energía} mide la capacidad de un sistema para producir
cambios: poner la materia en movimiento, levantarla, calentarla, encenderla. Es \omterm{def:g10:orders-of-magnitude:unit}{unidad}
es el \emph{joule}\index{joule} (\unit{J}). Almacenado, transferido,
convertido --- cada rama de la física cuenta en esta única moneda.
\end{definition}

\begin{definition}[Energía cinetica]\label{def:g10:energy-conservation:kinetic}
Un cuerpo de masa $m$ (en \unit{kg}) que se mueve a la velocidad $v$ (en
\unit{m/s}) lleva el \emph{energía cinética}\index{energía cinética}
$E_k = \tfrac12\, m v^2$: duplicar la masa la duplica, duplicarla
la velocidad la cuadriplica.
\end{definition}

\begin{example}[Coche versus peatón]\label{ex:g10:energy-conservation:kinetic}
Un automóvil \qty{1300}{kg} en \qty{100}{km/h} $= \qty{27.8}{m/s}$ lleva
$E_k = \tfrac12 \times 1300 \times 27.8^2 \approx \qty{5.0e5}{J}$;
un peatón \qty{70}{kg} en \qty{1.5}{m/s},
$\tfrac12 \times 70 \times 1.5^2 \approx \qty{79}{J}$ --- seis
mil veces menos. La velocidad es cara: la plaza se encarga de ello.
\end{example}

\begin{definition}[Energía potencial gravitacional]\label{def:g10:energy-conservation:potential}
Un cuerpo de masa $m$ a la altura $h$ (en \unit{m}) por encima de un
El nivel de referencia almacena el \emph{potencial gravitacional
energía}\index{energía potencial!gravitacional} $E_p = m g h$, con
$g = \qty{9.81}{N/kg}$ --- válido cerca del suelo, donde $g$ es
efectivamente constante. Sólo \emph{diferencias} de altura importa: el
nivel donde $E_p = 0$ se elige libremente.
\end{definition}

\begin{example}[torre de agua]\label{ex:g10:energy-conservation:potential}
Cada \omterm{def:g10:orders-of-magnitude:unit}{metro} (\qty{1000}{kg}) cúbico de agua almacenada \qty{30}{m} hasta un
torre de agua sostiene $1000 \times 9.81 \times 30 \approx \qty{2.9e5}{J}$
--- aproximadamente seis décimas del \omterm{def:g10:energy-conservation:kinetic}{energía cinética} del automóvil anterior. Ciudades
Acumular \omterm{def:g10:energy-conservation:energy}{energía} gastos generales.
\end{example}

\begin{definition}[Las otras formas]\label{def:g10:energy-conservation:forms}
\omterm{def:g10:energy-conservation:energy}{Energía} también se esconde en trajes menos visibles: \emph{térmica
energía}\index{energía termal}, la agitación desordenada de un cuerpo.
moléculas (cuanto más calientes, más); \emph{químico
energía}\index{energía química}, almacenado en enlaces moleculares --- alimentos,
madera, gasolina, batterys; \emph{electrico
energía}\index{energía eléctrica}, transportado por corrientes de planta a
enchufe; \emph{energía radiativa}\index{energía radiativa}, llevado por
luz incluso a través del espacio vacío --- cómo llega hasta nosotros el \omterm{def:g10:energy-conservation:energy}{energía} del Sol;
\emph{energía nuclear}\index{energía nuclear}, almacenado dentro de atómico
núcleos, liberados en las estrellas (\cref{ch:g12:nuclear-energy}).
\end{definition}

\section{Transferencias y cadenas de conversión}

\begin{definition}[Transferencia, conversión, cadena]\label{def:g10:energy-conservation:transfer}
Un \emph{transferencia de energía}\index{transferencia de energía} mueve \omterm{def:g10:energy-conservation:energy}{energía}
entre sistemas --- \emph{embalses}\index{depósito (de energía)};
un \emph{conversión de energía}\index{conversión de energía} cambia su
forma. Un dispositivo que hace cualquiera de las dos cosas (turbina, motor, lamp, músculo) es un
\emph{convertidor}\index{convertidor}. A \emph{conversión
cadena}\index{cadena de conversión} ilustra el viaje: cajas para
embalses y formularios, flechas etiquetadas para transferencias.
\end{definition}

\begin{method}[Dibujando una cadena de conversión]\label{met:g10:energy-conservation:chain}
\begin{enumerate}
  \item Identificar el depósito inicial y su forma (¿qué se consume?).
  \item Siga \omterm{def:g10:energy-conservation:energy}{energía} \omterm{def:g10:energy-conservation:transfer}{convertidor} por \omterm{def:g10:energy-conservation:transfer}{convertidor}: una flecha por
        transferencia, la forma nombrada en cada casilla.
  \item Agregue una flecha de calor en cada \omterm{def:g10:energy-conservation:transfer}{convertidor} real --- algunos \omterm{def:g10:energy-conservation:energy}{energía}
        siempre pierde calor (\omterm{def:g10:inertia:inventory}{fricción}, cables, escape).
\end{enumerate}
\end{method}

parte superior del swing, todo cinético hasta el punto más bajo (tomando
$E_p = 0$ ahí):

\section{Conservación de energía}

\begin{theorem}[Conservación de energía]\label{thm:g10:energy-conservation:conservation}
\index{conservación de energía}
El \omterm{def:g10:energy-conservation:energy}{energía} total de un sistema aislado --- uno que no intercambia nada
con el exterior --- es constante. \omterm{def:g10:energy-conservation:energy}{Energía} nunca se crea y nunca
destruido: sólo cambia de forma y de lugar, y lo que un reservorio
pierde, los demás ganan, \omterm{def:g10:energy-conservation:energy}{joule} por \omterm{def:g10:energy-conservation:energy}{joule}.
\end{theorem}
\admitted

\begin{remark}\label{rem:g10:energy-conservation:admitted}
Ningún experimento ha descubierto jamás que este principio falla; aquí tomamos
explíquelo como se le da y utilícelo numéricamente. \cref{ch:g12:work-and-energy}
deriva su parte mecánica ($E_k + E_p$) de las leyes de Newton; el
la contabilidad completa espera los volúmenes universitarios.
\end{remark}

\begin{example}[Caída libre, precio en julios]\label{ex:g10:energy-conservation:fall}
Una bola \qty{2.0}{kg} cae desde \qty{10}{m} (resistencia del aire
insignificante): comienza con
$E_p = 2.0 \times 9.81 \times 10 \approx \qty{196}{J}$, aterriza con
$E_k = \qty{196}{J}$, entonces $v = \sqrt{2 \times 196/2.0} =
\qty{14}{m/s} \approx \qty{50}{km/h}$. No \omterm{def:g10:inertia:force}{efectivo}, sin cronómetro:
La conservación por sí sola valora el impacto.
\end{example}

\section{Eficiencia}

\begin{definition}[eficiencia]\label{def:g10:energy-conservation:efficiency}
Un \omterm{def:g10:energy-conservation:transfer}{convertidor} consume $E_{\text{consumed}}$ y entrega lo deseado
formulario $E_{\text{useful}}$; su \emph{eficiencia}\index{eficiencia} es
\[
\eta = \frac{E_{\text{useful}}}{E_{\text{consumed}}} < 1 .
\]
La conservación prohíbe $\eta > 1$, y el \omterm{def:g10:energy-conservation:transfer}{convertidores} real nunca llega a $1$:
el \omterm{def:g10:energy-conservation:energy}{energía} que falta no se destruye --- sale en forma de calor.
\end{definition}

\begin{example}[Donde va la gasolina]\label{ex:g10:energy-conservation:engine}
A \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de depósitos de gasolina \qty{45}{MJ} de \omterm{def:g10:energy-conservation:forms}{energía química}; un
El motor del coche lo convierte con $\eta \approx 0.30$: \qty{13.5}{MJ} de
movimiento, \qty{31.5}{MJ} calentando el escape, el radiador y el
calle. Dos tercios de cada tanque calientan la atmósfera.
\end{example}

\begin{remark}[Las eficiencias se multiplican]\label{rem:g10:energy-conservation:chain-eta}
\omterm{def:g10:energy-conservation:transfer}{convertidores} encadenados multiplican sus eficiencias: una planta \omterm{def:g10:energy-conservation:power}{fuerza}
($0.38$) alimentando la red ($0.94$) alimentando un cargador ($0.85$)
entrega $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.30$. largo
Las cadenas son tuberías con fugas: cada flecha grava el flujo.
\end{remark}

\section{La potencia, el vatio y el kilovatio-hora.}

\begin{definition}[poder]\label{def:g10:energy-conservation:power}
\emph{Fuerza}\index{fuerza} es la velocidad a la que fluye \omterm{def:g10:energy-conservation:energy}{energía},
$P = E/t$, medido en \emph{vatios}\index{vatio} (\unit{W}), con
$\qty{1}{W} = \qty{1}{J/s}$. \omterm{def:g10:energy-conservation:energy}{Energía} es una cantidad, potencia un flujo ---
una bañera versus su grifo.
\end{definition}

\begin{example}[Hervidor versus humano]\label{ex:g10:energy-conservation:power}
Una tetera \qty{2200}{W} funcionando con \qty{150}{s} utiliza
$E = 2200 \times 150 = \qty{3.3e5}{J}$. Un humano corre sobre
\qty{10}{MJ} de comida por día, un promedio
$\num{1.0e7}/\num{86400} \approx \qty{116}{W}$: estás inactivo como un
bombilla vieja y brillante.
\end{example}

\begin{definition}[El kilovatio-hora]\label{def:g10:energy-conservation:kwh}
El \emph{kilovatio-hora}\index{kilovatio-hora} (\unit{kW.h}) es el
\omterm{def:g10:energy-conservation:energy}{energía} de un flujo \qty{1}{kW} ejecutándose durante una hora:
$\qty{1}{kW.h} = 1000 \times 3600\,\unit{J} = \qty{3.6}{MJ}$ ---
los medidores de electricidad \omterm{def:g10:orders-of-magnitude:unit}{unidad} cuentan: \omterm{def:g10:energy-conservation:power}{fuerza} en \unit{kW} veces horas.
\end{definition}

\begin{example}[leyendo la factura]\label{ex:g10:energy-conservation:bill}
Un billete es una resta y una multiplicación: nueva lectura del contador
menos viejo da el \omterm{def:g10:energy-conservation:kwh}{kilovatios-hora}; veces el precio \omterm{def:g10:orders-of-magnitude:unit}{unidad} (aproximadamente
$0.25$ euros por \unit{kW.h}), más una suscripción fija, da la
Total. Un calentador \qty{2}{kW} funcionando \qty{2.5}{h} para usos nocturnos
\qty{5}{kW.h} al día: $38$ euros al mes por su cuenta.
\end{example}

\begin{example}[Órdenes de magnitud]\label{ex:g10:energy-conservation:orders}
Vale la pena saberlo de memoria: levantando una manzana una \omterm{def:g10:orders-of-magnitude:unit}{metro}, sobre \qty{1}{J};
una battery de teléfono, $\qty{10}{W.h} \approx \qty{4e4}{J}$; un cereal
cuenco, \qty{1.5}{MJ}; tu alimento para un día, \qty{10}{MJ}; a \omterm{def:g10:orders-of-magnitude:unit}{kilogramo}
de gasolina, \qty{45}{MJ} --- casi exactamente el consumo diario de un hogar
electricidad ($\approx \qty{12}{kW.h}$); un rayo, \qty{5}{GJ}:
cien días hogar, entregados en un instante.
\end{example}

\section{Problema: dar energía a una casa por un día}

\begin{problem}[{Problema del fin de semana --- del tazón de cereal a la azotea
paneles: una auditoría energética completa de un día cualquiera en casa, y qué
costos en julios dependiendo de quién lo vende}]
\label{pb:g10:energy-conservation:1}
Un martes cualquiera, una familia se propone encontrar dónde está
La electricidad va, armado con este capítulo, el medidor del pasillo y
el reverso del billete. Veredicto de esta noche: su día es igual a uno
\omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de gasolina, dieciséis cuadrados \omterm{def:g10:orders-of-magnitude:unit}{metros} de sol --- o cinco
ciclistas pedaleando las 24 horas del día.

\medskip
\noindent\textbf{Parte I --- The audit.}
El inventario del día: frigorífico \qty{100}{W} para \qty{24}{h}; agua
calentador \qty{2400}{W} para \qty{2.5}{h}; horno \qty{2000}{W} para
\qty{1.0}{h}; lavadora \qty{500}{W} para \qty{2.0}{h}; iluminación
\qty{60}{W} para \qty{5.0}{h}; pantallas \qty{150}{W} para \qty{6.0}{h}.
\begin{enumerate}
  \item Calcule el \omterm{def:g10:energy-conservation:energy}{energía} de cada aparato para el día, en \unit{kW.h}.
  \item Totaliza el día en \omterm{def:g10:energy-conservation:kwh}{kilovatios-hora}, luego en megajulios;
        comparar con un \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de gasolina (\qty{45}{MJ}).
  \item Calcule el \omterm{def:g10:energy-conservation:power}{fuerza} promedio de la casa sobre el \qty{24}{h};
        ¿Qué fracción de una tetera en funcionamiento es esa?
  \item ¿Qué aparato domina? Recalcular el total diario
        si el calentador funcionó solo \qty{1.25}{h}.
  \item Las luces de espera dibujan un \qty{15}{W} permanente: diario \omterm{def:g10:energy-conservation:energy}{energía},
        participación del costo total anual en $0.25$ euros por \unit{kW.h}?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The bill.}
\begin{enumerate}[resume]
  \item En 30 días el medidor pasó de \qty{56214}{kW.h} a
        \qty{56604}{kW.h}: consumo y promedio diario comprobados
        ¿En contra de la auditoría?
  \item Calcular la factura: $0.25$ euros por \unit{kW.h} más un
        suscripción fija $14$ euros.
  \item ¿Precio de un megajulio de la red? De gasolina, en
        $1.80$ euros por litro almacenando \qty{34}{MJ}?
  \item Un bol de cereal \qty{1.5}{MJ} cuesta unos $0.50$ euros:
        precio megajulio? Clasifique a los tres vendedores de \omterm{def:g10:energy-conservation:energy}{julios}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The rooftop.}
\begin{enumerate}[resume]
  \item La luz solar plena proporciona aproximadamente \qty{1000}{W} por cuadrado
        \omterm{def:g10:orders-of-magnitude:unit}{metro}; paneles convertirlo con $\eta = 0.20$: eléctrico
        ¿\omterm{def:g10:energy-conservation:power}{fuerza} de \qty{1}{m^2} a pleno sol?
  \item La región promedia el equivalente a \qty{4.0}{h} de total
        sol por día: rendimiento diario de \qty{1}{m^2}?
  \item ¿Qué área del panel cubre el día \qty{12.6}{kW.h} de la familia?
        El techo sur ofrece \qty{20}{m^2}: ¿encaja?
  \item En diciembre el equivalente baja a \qty{1.5}{h}: el
        producción de paneles, su participación en la necesidad y lo que cubre
        la brecha?
  \item Fabricar los paneles cuesta aproximadamente \qty{700}{kW.h} por
        cuadrado \omterm{def:g10:orders-of-magnitude:unit}{metro}: \omterm{def:g10:energy-conservation:energy}{energía} tiempo de recuperación, ¿usando la pregunta 11?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The cereal bowl.}
\begin{enumerate}[resume]
  \item Una bicicleta con generador produce \qty{100}{W} de electricidad:
        cuantas horas de pedaleo hace \qty{12.6}{kW.h} --- cuantas
        ¿Gente pedaleando \qty{24}{h} sin parar?
  \item Los músculos corren en $\eta \approx 0.25$: la comida \omterm{def:g10:energy-conservation:energy}{energía} detrás
        esos \qty{12.6}{kW.h}, en megajulios y en tazones de cereal?
  \item Coste del día pedaleado (bochas a $0.50$ euros), contra
        ¿Cuánto cobra la red por el mismo día?
  \item ¿A cuántos días-casa equivale un rayo \qty{5}{GJ}?
        Dada su duración \qty{100}{\micro s}, ¿por qué todavía puede
        no \omterm{def:g10:energy-conservation:power}{fuerza} la casa?
  \item Los julios nunca se pierden, pero red, gasolina y alimentos \omterm{def:g10:energy-conservation:energy}{julios}
        vender a precios muy diferentes: ¿por qué se paga realmente?
  \item Final: indique el veredicto del día en una frase --- una
        \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de gasolina, dieciséis cuadrados \omterm{def:g10:orders-of-magnitude:unit}{metros} de techo solar o
        cinco ciclistas las 24 horas del día --- y nombra las dos cantidades
        (un stock, un flujo) para no volver a confundir nunca más.
\end{enumerate}
\end{problem}
