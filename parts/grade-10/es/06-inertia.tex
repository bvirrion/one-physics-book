\chapter{Fuerzas y el principio de inercia.}\label{ch:g10:inertia}

Deja de pedalear y la bicicleta se detiene; deja de tirar y el
el trineo se detiene. La experiencia susurra que el movimiento debe ser alimentado por un
esfuerzo --- y durante dos mil años la física estuvo de acuerdo. este capitulo
construye el \omterm{def:g10:inertia:force}{fuerza}, una flecha medida en \omterm{def:g10:inertia:force}{newton}, y establece la primera
Gran ley de la mecánica: si se lo deja verdaderamente solo, un cuerpo en movimiento mantiene su
movimiento, recto y constante, para siempre.

\section{Modelar una acción como una fuerza.}

\begin{definition}[fuerza]\label{def:g10:inertia:force}
Un \emph{fuerza}\index{fuerza} modela una acción mecánica (un empujón, un tirón,
una atracción) ejercida sobre un cuerpo. Se caracteriza por:
\begin{itemize}
  \item es \emph{punto de aplicación}\index{punto de aplicación};
  \item su dirección --- la línea por la que actúa y el camino
        en esa línea;
  \item su magnitud, medida en \emph{newton}\index{Newton}
        (símbolo \unit{N}).
\end{itemize}
Una fuerza se dibuja como una flecha $\vect F$ (un vector, como en las matemáticas
volumen) desde su punto de aplicación, con longitud proporcional al
magnitud a una escala determinada.
\end{definition}

\begin{example}[El peso de una mochila]\label{ex:g10:inertia:schoolbag}
Una mochila de masa $m = \qty{4.0}{kg}$ es arrastrada hacia abajo por la Tierra
con su \omterm{def:g10:universal-gravitation:weight}{peso} (\cref{ch:g10:universal-gravitation}):
$P = mg = 4.0 \times 9.81 \approx \qty{39}{N}$ --- un \omterm{def:g10:inertia:force}{Newton} es aproximadamente
\omterm{def:g10:universal-gravitation:weight}{peso} de una manzana. En la escala $\qty{1}{cm} \leftrightarrow
\qty{10}{N}$: una flecha vertical \qty{3.9}{cm} larga, aplicada en la
centro de gravedad de la bolsa, apuntando hacia abajo.
\end{example}

\section{Un inventario de fuerzas}

\begin{definition}[Fuerzas de contacto y distancia]\label{def:g10:inertia:contact}
Se ejerce una \emph{fuerza de contacto}\index{fuerza de contacto} en los puntos
donde dos cuerpos se tocan. Un \emph{fuerza a distancia}\index{fuerza en un
distancia} no necesita contacto: \omterm{def:g10:universal-gravitation:weight}{peso} --- la atracción gravitacional de
\cref{ch:g10:universal-gravitation} --- es el único utilizado en este
capítulo.
\end{definition}

\begin{definition}[Las fuerzas de contacto comunes]\label{def:g10:inertia:inventory}
Cuatro \omterm{def:g10:inertia:contact}{fuerzas de contacto} cubren la mayoría de situaciones cotidianas:
\begin{itemize}
  \item el \emph{reacción normal}\index{reacción normal} $\vect R$ de
        un soporte, perpendicular a su superficie, que se aleja de ella;
  \item el \emph{tensión}\index{tensión} $\vect T$ de un alambre, cuerda o
        cadena, dirigida a lo largo de ella, tirando hacia ella;
  \item \emph{fricción}\index{fricción} $\vect f$ desde una superficie, el
        aire o agua, oponiéndose al deslizamiento o al movimiento;
  \item el \emph{empuje}\index{empuje} de un motor, jet o mano,
        impulsando el esfuerzo.
\end{itemize}
\end{definition}

\begin{example}[Inventario en trineo remolcado]\label{ex:g10:inertia:sled}
Un trineo remolcado por la nieve: \omterm{def:g10:universal-gravitation:weight}{peso} $\vect P$ (distancia), \omterm{def:g10:inertia:inventory}{normales
reacción} $\vect R$ de la nieve, \omterm{def:g10:inertia:inventory}{tensión} $\vect T$ de la cuerda, \omterm{def:g10:inertia:inventory}{fricción}
$\vect f$ oponiéndose al deslizamiento. Nada más toca el trineo: el
el inventario está completo --- \emph{\omterm{def:g10:universal-gravitation:weight}{peso} always, then one \omterm{def:g10:inertia:contact}{fuerza de contacto}
per touching object}.
\end{example}

\section{El principio de inercia.}

\begin{definition}[Movimiento uniforme y recto]\label{def:g10:inertia:uniform}
Un cuerpo está en \emph{movimiento uniforme y rectilíneo}\index{uniforme recto
movimiento} en un marco cuando su \omterm{def:g10:relative-motion:trajectory}{trayectoria} es una línea recta recorrida en
velocidad constante: su vector velocidad $\vect v$ es el mismo en cada
instantáneo. El resto es el caso especial $\vect v = \vect 0$.
\end{definition}

\begin{definition}[Fuerzas compensadoras]\label{def:g10:inertia:compensate}
\omterm{def:g10:inertia:force}{Efectivo} actuando sobre un cuerpo \emph{compensar}\index{fuerzas compensadoras}
cuando su efecto combinado es nulo: punta a cola, sus flechas regresan a
el comienzo. Dos \omterm{def:g10:inertia:force}{efectivo} compensan exactamente cuando comparten la misma línea
de acción con iguales magnitudes y caminos opuestos.
\end{definition}

\begin{theorem}[Principio de inercia]\label{thm:g10:inertia:principle}
\index{principio de inercia}\index{inercia}
En el \omterm{def:g10:relative-motion:frames}{marco de tierra}, un cuerpo sujeto a ningún \omterm{def:g10:inertia:force}{fuerza}, o a \omterm{def:g10:inertia:force}{efectivo} que
\omterm{def:g10:inertia:compensate}{compensar}, permanece en reposo o mantiene un \omterm{def:g10:inertia:uniform}{movimiento uniforme y rectilíneo};
por el contrario, un cuerpo en reposo o en \omterm{def:g10:inertia:uniform}{movimiento uniforme y rectilíneo} está sujeto
a \omterm{def:g10:inertia:compensate}{fuerzas compensadoras} (o a ninguno).
\end{theorem}
\admitted

\begin{remark}
Ningún experimento puede aislar completamente un cuerpo, por lo que el principio es un postulado.
--- vislumbrado por Galileo, afirmado por Newton. Un capítulo posterior lo actualiza a un
ley que relaciona \omterm{def:g10:inertia:force}{efectivo} con \emph{cambios} de velocidad; el volumen del año 1
pregunta en qué fotogramas se encuentra realmente.
\end{remark}

\begin{proposition}[Lectura del movimiento a partir de fuerzas y viceversa.]\label{prop:g10:inertia:converse}
En el \omterm{def:g10:relative-motion:frames}{marco de tierra}:
\begin{enumerate}
  \item si la velocidad de un cuerpo cambia --- en magnitud \emph{or} en
        dirección --- el \omterm{def:g10:inertia:force}{efectivo} en él hace \emph{no} \omterm{def:g10:inertia:compensate}{compensar};
  \item si un cuerpo está en reposo o en \omterm{def:g10:inertia:uniform}{movimiento uniforme y rectilíneo}, su
        inventariado \omterm{def:g10:inertia:force}{efectivo} debe cancelar --- determinando incógnitas.
\end{enumerate}
\end{proposition}

\begin{proof}
La primera afirmación es la contrapositiva de
\cref{thm:g10:inertia:principle}, la segunda se aplicó su parte inversa
a un inventario completo.
\end{proof}

\begin{example}[La discoteca de hockey.]\label{ex:g10:inertia:puck}
Liberado por el palo, un disco atraviesa \qty{30}{m} de hielo liso en
\qty{2.5}{s}, directo a la constante $v = 30/2.5 = \qty{12}{m/s}$.
Inventario: \omterm{def:g10:universal-gravitation:weight}{peso} y \omterm{def:g10:inertia:inventory}{reacción normal} (\omterm{def:g10:inertia:inventory}{fricción} insignificante). ellos
\omterm{def:g10:inertia:compensate}{compensar}, y el principio predice exactamente el movimiento observado:
nada empuja el disco hacia adelante\dots y nada necesita hacerlo.
\end{example}

\begin{remark}[Galileo contra Aristóteles]
Aristóteles enseñó que todo movimiento necesita un motor, y la vida diaria ---
\omterm{def:g10:inertia:inventory}{fricción} en cada contacto --- parece estar de acuerdo. Galileo idealizó el
\omterm{def:g10:inertia:inventory}{fricción} lejos: las superficies cada vez más lisas necesitan cada vez menos empuje para
\emph{mantener} movimiento. La persistencia, no el descanso, es el estado natural.
\end{remark}

\begin{remark}[El pasajero tambaleante]
Cuando un autobús frena bruscamente, los pasajeros que están parados se inclinan hacia adelante. No \omterm{def:g10:inertia:force}{fuerza}
los lanza: por inercia sus cuerpos mantienen su velocidad mientras el autobús
pierde lo suyo. El cinturón de seguridad suministra el \omterm{def:g10:inertia:force}{fuerza} hacia atrás el inventario
carece.
\end{remark}

\section{Fuerzas equilibradas: estática a partir de un dibujo.}

\begin{definition}[Diagrama de fuerza]\label{def:g10:inertia:diagram}
El \emph{diagrama de fuerza}\index{diagrama de fuerza} de un cuerpo muestra el cuerpo
solo, con cada \omterm{def:g10:inertia:force}{fuerza} actuando \emph{on} dibujado a escala desde su
\omterm{def:g10:inertia:force}{punto de aplicación}; \omterm{def:g10:inertia:force}{efectivo} ejerció \emph{by} el cuerpo nunca apareció.
\end{definition}

\begin{method}[Analizando una situación estática]\label{met:g10:inertia:method}
\begin{enumerate}
  \item Elige el cuerpo a estudiar; Trabajar en el \omterm{def:g10:relative-motion:frames}{marco de tierra}.
  \item Inventario: \omterm{def:g10:universal-gravitation:weight}{peso} siempre, un \omterm{def:g10:inertia:contact}{fuerza de contacto} por contacto.
  \item Dibuja el \omterm{def:g10:inertia:diagram}{diagrama de fuerza}.
  \item El cuerpo está en reposo, por lo que el \omterm{def:g10:inertia:force}{efectivo} \omterm{def:g10:inertia:compensate}{compensar}
        (\cref{thm:g10:inertia:principle}): lectura de magnitudes desconocidas
        el dibujo (longitudes iguales en 1D, componentes en 2D).
\end{enumerate}
\end{method}

\begin{example}[El libro sobre la mesa]\label{ex:g10:inertia:book}
Un libro de masa \qty{0.50}{kg} descansa sobre una mesa: \omterm{def:g10:universal-gravitation:weight}{peso}
$P = 0.50 \times 9.81 \approx \qty{4.9}{N}$ y \omterm{def:g10:inertia:inventory}{reacción normal}
$\vect R$. En reposo, $\vect R$ \omterm{def:g10:inertia:compensate}{compensa} $\vect P$: vertical, hacia arriba,
$R = \qty{4.9}{N}$. Agregue un segundo libro y $R$ se ajusta --- un soporte
empuja exactamente tan fuerte como sea necesario.
\end{example}

\begin{example}[La lamp de dos cables]\label{ex:g10:inertia:lamp}
Una lamp de masa \qty{3.0}{kg} ($P \approx \qty{29.4}{N}$) cuelga de dos
cables, cada uno en $45^\circ$ desde la vertical. Por simetría el \omterm{def:g10:inertia:inventory}{tensiones}
son iguales ($T_1 = T_2 = T$) y sus componentes horizontales se cancelan; el
Los componentes verticales, cada $T \cos 45^\circ$, llevan el \omterm{def:g10:universal-gravitation:weight}{peso}:
\[
2\,T \cos 45^\circ = P
\qquad\text{so}\qquad
T = \frac{29.4}{2 \times 0.707} \approx \qty{21}{N}
\]
--- más de la mitad del \omterm{def:g10:universal-gravitation:weight}{peso} por cable: tirar en ángulo cuesta
\omterm{def:g10:inertia:inventory}{tensión}.
\end{example}

\section{Problema: Inercia en tres actos}

\begin{problem}[{Problema del fin de semana --- la pasarela del aeropuerto, la pista de hielo
y el paracaídas: velocidad constante como firma de compensación
fuerzas, y por qué una paracaídas no reduce la resistencia sino la velocidad}]
\label{pb:g10:inertia:1}
Tres escenas: una maleta sobre una pasarela móvil, un disco sobre una pista de hielo,
un paracaidista en el aire de la tarde\dots y una ley hablando en los tres.

\medskip
\noindent\textbf{Parte I --- The airport walkway.}
Una maleta \qty{23}{kg} se encuentra sobre una pasarela que se mueve a velocidad constante.
velocidad \qty{0.75}{m/s}.
\begin{enumerate}
  \item ¿Movimiento de la maleta en el marco del suelo? Haga un inventario del \omterm{def:g10:inertia:force}{efectivo} en él.
  \item ¿Estos \omterm{def:g10:inertia:force}{efectivo} \omterm{def:g10:inertia:compensate}{compensar}? Justificar; calcular ambas magnitudes.
  \item Compare su \omterm{def:g10:inertia:diagram}{diagrama de fuerza} con la misma maleta en el
        piso. ¿Qué afirmación profunda ilustra la comparación?
  \item La pasarela se detiene bruscamente. Describe qué hace la maleta,
        sin inventar un delantero \omterm{def:g10:inertia:force}{fuerza}.
  \item Su propietario camina en \qty{1.2}{m/s} con respecto a la pasarela. ella
        velocidad del marco del suelo? ¿El \omterm{def:g10:inertia:force}{efectivo} en \emph{su} \omterm{def:g10:inertia:compensate}{compensar}?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The ice rink.}
\begin{enumerate}[resume]
  \item Un disco \qty{160}{g} se desliza directamente hacia la constante
        \qty{8.0}{m/s}. Dibuja su \omterm{def:g10:inertia:diagram}{diagrama de fuerza} con magnitudes.
  \item Aristóteles afirma que el disco necesita alguien que lo mueva. ¿Qué hace la pista?
        responder, y ¿por qué las superficies cotidianas parecen estar de su lado?
  \item El disco atraviesa una mala racha y frena. Conclusión sobre el
        \omterm{def:g10:inertia:force}{efectivo} ahí? ¿Cuál \omterm{def:g10:inertia:force}{fuerza} es el culpable?
  \item Un patinador toma una curva suavemente a velocidad constante. Haz el \omterm{def:g10:inertia:force}{efectivo} en
        ¿Su \omterm{def:g10:inertia:compensate}{compensar}? Cuidado: ¿qué tipo de cantidad es la velocidad?
  \item Haga coincidir el diagrama con el movimiento: (a) compensado \omterm{def:g10:inertia:force}{efectivo}; (b) un extra
        \omterm{def:g10:inertia:force}{fuerza} frente a $\vect v$; (c) un \omterm{def:g10:inertia:force}{fuerza} adicional perpendicular a
        $\vect v$ --- (i) desacelerar en línea recta; (ii) girando a velocidad constante
        velocidad; (iii) \omterm{def:g10:inertia:uniform}{movimiento uniforme y rectilíneo}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The parachute.}
La masa total de un saltador, incluido el equipo, es \qty{80}{kg}.
\begin{enumerate}[resume]
  \item Al abandonar el avión, la velocidad vertical y la resistencia siguen siendo cero.
        Calcule el \omterm{def:g10:inertia:force}{efectivo}; ¿Son \omterm{def:g10:inertia:compensate}{compensar}? ¿Qué debe pasar a continuación?
  \item La resistencia crece con la velocidad. Describe la caída mientras arrastras $<$ \omterm{def:g10:universal-gravitation:weight}{peso}.
  \item Antes de que se abra la capota, el saltador alcanza una posición estable.
        \qty{50}{m/s} ("velocidad terminal"). Calcula el arrastre. que es
        el movimiento, y ¿por qué es "inercia disfrazada"?
  \item Se abre la capota: la resistencia ahora supera con creces a la \omterm{def:g10:universal-gravitation:weight}{peso}. que
        ¿Qué pasa con la velocidad? (El saltador se mueve \emph{abajo}
        en todas partes.)
  \item Se alcanza un nuevo \qty{5.0}{m/s} estable. Calcule la resistencia;
        comparar con la pregunta 13.
  \item El remate: ¿qué cambió el paracaídas? --- la terminal
        arrastre, o la velocidad a la que el arrastre llega al \omterm{def:g10:universal-gravitation:weight}{peso}?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Verdicts.}
\begin{enumerate}[resume]
  \item Completa y justifica: ``en reposo o en \omterm{def:g10:inertia:uniform}{movimiento uniforme y rectilíneo}
        $\iff$ \dots''. ¿Por qué el descanso no merece una ley separada?
  \item ¿En qué acto se tendría el “movimiento necesita un motor” de Aristóteles?
        fracasó de manera más visible, y ¿qué señalaría Galileo?
  \item Un guión de examen dice: ``el paracaidista cae a velocidad constante
        velocidad, por lo que ningún \omterm{def:g10:inertia:force}{fuerza} actúa sobre él''. Corrígelo en una línea.
  \item Final: para cada acto, nombra la pareja de \omterm{def:g10:inertia:force}{efectivo} que \omterm{def:g10:inertia:compensate}{compensar},
        y enuncie la ley utilizada veinte veces en este problema.
\end{enumerate}
\end{problem}
