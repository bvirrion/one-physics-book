\section{Las señales transportan información.}

\begin{definition}[Señal]\label{def:g10:signals-and-waves:signal}
Un \emph{señal}\index{señal} es un \omterm{def:g10:orders-of-magnitude:unit}{cantidad fisica} que varía en el tiempo
para transportar información: el aire \omterm{def:g10:pressure:pressure}{presión} en su tímpano, el voltaje en un
terminales del micrófono, el brillo de la luz en un \omterm{def:g10:refraction:fiber}{fibra óptica}.
Para estudiar una señal, regístrela en función del tiempo y lea la
información fuera de la curva.
\end{definition}

En una llamada telefónica la misma información cambia de operador repetidamente:
\omterm{def:g10:pressure:pressure}{presión} en el aire, luego voltaje en el micrófono, luego radio en el
antena --- y hacia atrás en el otro extremo. Un vocabulario los maneja.
todos; este capítulo lo construye.

\section{Señales periódicas}

\begin{definition}[Señal periódica, punto]\label{def:g10:signals-and-waves:period}
Un \omterm{def:g10:signals-and-waves:signal}{señal} es \emph{periódico}\index{señal periódica} cuando se repite
idénticamente a intervalos regulares. El \emph{período}\index{período} $T$ es
la duración de un ciclo completo, en segundos.
\end{definition}

\begin{definition}[Frecuencia]\label{def:g10:signals-and-waves:frequency}
El \emph{frecuencia}\index{frecuencia} de un \omterm{def:g10:signals-and-waves:period}{señal periódica} es el número
de ciclos por segundo:
\[
f = \frac{1}{T} .
\]
Su \omterm{def:g10:orders-of-magnitude:unit}{unidad}, un ciclo por segundo, es el \emph{hercios}\index{hercios}
(\unit{Hz}), con los múltiplos habituales \unit{kHz}, \unit{MHz}, \unit{GHz}.
\end{definition}

\begin{definition}[amplitud]\label{def:g10:signals-and-waves:amplitude}
El \emph{amplitud}\index{amplitud} de un \omterm{def:g10:signals-and-waves:period}{señal periódica} es su máximo
desviación del valor en reposo. Para un voltaje que oscila simétricamente
aproximadamente cero, la amplitud es el valor máximo; medidas de cresta a valle
\emph{dos veces} la amplitud (el valor pico a pico).
\end{definition}

\begin{example}[Pista principal y de concierto]\label{ex:g10:signals-and-waves:mains}
La tensión de red oscila en $f = \qty{50}{Hz}$: su \omterm{def:g10:signals-and-waves:period}{período} es
$T = 1/f = \qty{0.020}{s} = \qty{20}{ms}$. El concierto A de una orquesta es un
\omterm{def:g10:pressure:pressure}{presión} \omterm{def:g10:signals-and-waves:signal}{señal} en \qty{440}{Hz}: dura un ciclo
$T = 1/440 = \qty{2.3}{ms}$. \omterm{def:g10:signals-and-waves:frequency}{frecuencia} superior, ciclo más corto: $f$ y $T$
son inversos entre sí.
\end{example}

\section{Leyendo un oscilograma}

\begin{definition}[osciloscopio]\label{def:g10:signals-and-waves:oscilloscope}
Un \emph{osciloscopio}\index{osciloscopio} traza un voltaje contra el tiempo;
la curva en su pantalla cuadriculada es \emph{oscilograma}%
\index{oscilograma}. Dos configuraciones convierten las divisiones de pantalla en físicas
valores: \emph{base de tiempo}\index{base de tiempo} (segundos por horizontal)
división) y la vertical \emph{ganar}\index{ganar} (voltios por vertical)
división).
\end{definition}

\begin{method}[De la pantalla a los números]\label{met:g10:signals-and-waves:oscillogram}
\begin{enumerate}
  \item Cuente las divisiones horizontales abarcadas por un ciclo completo
        (de cresta a cresta es lo más seguro); multiplica por \omterm{def:g10:signals-and-waves:oscilloscope}{base de tiempo} para obtener
        $T$, luego $f = 1/T$.
  \item Cuente las divisiones verticales desde la línea central hasta una cresta y
        multiplique por \omterm{def:g10:signals-and-waves:oscilloscope}{ganar} para obtener \omterm{def:g10:signals-and-waves:amplitude}{amplitud} (de la cresta al valle se obtiene
        el valor pico a pico: reducirlo a la mitad).
\end{enumerate}
\end{method}

\begin{example}[Una lectura completa]\label{ex:g10:signals-and-waves:reading}
En el \omterm{def:g10:signals-and-waves:oscilloscope}{oscilograma} a continuación, el \omterm{def:g10:signals-and-waves:oscilloscope}{base de tiempo} es \qty{5}{ms} por división y
el \omterm{def:g10:signals-and-waves:oscilloscope}{ganar} \qty{2}{V} por división. Un ciclo abarca las divisiones $4.0$:
$T = 4.0 \times \qty{5}{ms} = \qty{20}{ms}$ y
$f = 1/\qty{0.020}{s} = \qty{50}{Hz}$ --- red eléctrica nuevamente. Una cresta se sienta
$3.0$ divisiones sobre el eje: el \omterm{def:g10:signals-and-waves:amplitude}{amplitud} es
$3.0 \times \qty{2}{V} = \qty{6.0}{V}$.
\end{example}

\section{Señales sonoras y electromagnéticas.}

\begin{definition}[sonido]\label{def:g10:signals-and-waves:sound}
\emph{Sonido}\index{sonido} es un \omterm{def:g10:pressure:pressure}{presión} \omterm{def:g10:signals-and-waves:signal}{señal}: un objeto que vibra ---
cuerda, membrana, cuerdas vocales --- empuja rítmicamente en el aire y
las compresiones viajan hacia afuera alrededor de \qty{340}{m/s}. un micrófono
convierte el \omterm{def:g10:pressure:pressure}{presión} que llega en voltaje; el tímpano, en el nervio
impulsos. El sonido necesita un medio: cae una campana debajo de un frasco de vacío
silencioso mientras se bombea el aire.
\end{definition}

\begin{definition}[onda]\label{def:g10:signals-and-waves:wave}
Una perturbación viajera --- el patrón de compresión de \omterm{def:g10:signals-and-waves:sound}{sonido}, la ondulación
en un estanque, un pulso de radio --- es un \emph{ola}\index{ola}. el medio,
cuando hay uno, se queda quieto: cada parche de agua se balancea en su lugar mientras
la onda cruza el estanque. La geometría de las ondas --- \omterm{def:g10:light-spectra:wavelength}{longitud de onda},
interferencia, difracción --- se trata en un capítulo posterior; esto
El capítulo solo necesita sus velocidades.
\end{definition}

\begin{definition}[Ultrasonido e infrasonido]\label{def:g10:signals-and-waves:ultrasound}
El oído humano escucha \omterm{def:g10:signals-and-waves:sound}{sonido} entre aproximadamente \qty{20}{Hz} y \qty{20}{kHz}.
(el techo cae con la edad). \omterm{def:g10:signals-and-waves:sound}{Sonido} arriba de \qty{20}{kHz} es
\emph{ultrasonido}\index{ultrasonido} --- los murciélagos cazan en \qty{50}{kHz},
Los escáneres médicos toman imágenes del cuerpo a unos pocos megahercios. \omterm{def:g10:signals-and-waves:sound}{Sonido} a continuación
\qty{20}{Hz} es \emph{infrasonido}\index{infrasonido}, sentido por los elefantes.
y sismómetros.
\end{definition}

\begin{proposition}[Señales electromagnéticas]\label{prop:g10:signals-and-waves:em}
Radio, Wi-Fi y luz son \emph{electromagnético
señales}\index{señal electromagnética}: una familia, todos viajando
a través del vacío --- no se necesita medio --- y todo a una velocidad, la velocidad
de luz de \cref{ch:g10:refraction},
\[
c = \qty{3.00e8}{m/s} \quad \text{(very nearly the same in air).}
\]
\end{proposition}
\admitted

\begin{remark}\label{rem:g10:signals-and-waves:honest}
Este es un hecho experimental: la luz del sol cruza $150$ millones
kilómetros de espacio vacío; Honestamente, se deriva del volumen del año 2.
de las leyes de la electricidad y el magnetismo. Mantenga el contraste: \omterm{def:g10:signals-and-waves:sound}{sonido} es
una vibración \emph{of} el medio y muere sin ella; un \omterm{prop:g10:signals-and-waves:em}{electromagnético
señal} transporta su propio campo a través del vacío, un millón de veces más rápido.
\end{remark}

\section{Ecos: distancias de los retrasos}

Un \omterm{def:g10:signals-and-waves:signal}{señal} moviéndose a velocidad $v$ cubre $d = v\,t$ en un tiempo $t$: cada retraso
Es una distancia disfrazada. Mejor aún, envía un pulso corto a un
obstáculo y el tiempo es \emph{eco}\index{eco} --- el viaje de ida y vuelta cubre $2d$.

\begin{method}[Rango de eco]\label{met:g10:signals-and-waves:echo}
\begin{enumerate}
  \item Enviar un pulso corto; Mida el tiempo de ida y vuelta $t$ del \omterm{rem:g10:signals-and-waves:honest}{eco}.
  \item Tome la velocidad $v$ del \omterm{def:g10:signals-and-waves:signal}{señal} \emph{en el medio cruzado}.
  \item El pulso cubrió $2d = v\,t$, por lo que $d = \dfrac{v\,t}{2}$.
\end{enumerate}
\omterm{def:g10:signals-and-waves:sound}{Sonido} pulsos en agua hacen \emph{sonar}\index{sonar}
($v \approx \qty{1500}{m/s}$); pulsos de radio en el aire hacen
\emph{Radar}\index{Radar} ($v = c$); \omterm{def:g10:signals-and-waves:ultrasound}{ultrasonido} pulsos en el cuerpo hacen
el escáner médico ($v \approx \qty{1540}{m/s}$ en tejido blando).
\end{method}

\begin{example}[Sonda]\label{ex:g10:signals-and-waves:sonar}
El \omterm{met:g10:signals-and-waves:echo}{sonar} de un barco hace ping y el fondo marino \omterm{rem:g10:signals-and-waves:honest}{eco} regresa después
$t = \qty{0.80}{s}$: la profundidad es
$d = \frac{1500 \times 0.80}{2} = \qty{600}{m}$. olvidando el factor
$2$ duplica el océano.
\end{example}

\begin{example}[Radar]\label{ex:g10:signals-and-waves:radar}
Un aeropuerto \omterm{met:g10:signals-and-waves:echo}{Radar} recibe el \omterm{rem:g10:signals-and-waves:honest}{eco} $\qty{0.30}{ms}$ de una aeronave después del
pulso:
$d = \frac{\qty{3.00e8}{m/s} \times \qty{3.0e-4}{s}}{2} = \qty{45}{km}$.
Debido a que $c$ es tan grande, \omterm{met:g10:signals-and-waves:echo}{Radar} vive en relojes de microsegundos: sincronización de luz
es un trabajo de precisión.
\end{example}

\begin{example}[El corazón como señal]\label{ex:g10:signals-and-waves:ecg}
Cada latido envía un pequeño pulso de voltaje a través del pecho; electrodos
regístrelo como \emph{electrocardiograma}\index{electrocardiograma} (ECG), casi \omterm{def:g10:signals-and-waves:period}{periódico
señal}. Si las púas altas están separadas $T = \qty{0.75}{s}$, el corazón
late en $f = 1/0.75 = \qty{1.33}{Hz}$, es decir, \ $1.33 \times 60 = 80$
latidos por minuto: leer $T$ de un rastro es como toma un cardiólogo
tu pulso.
\end{example}
