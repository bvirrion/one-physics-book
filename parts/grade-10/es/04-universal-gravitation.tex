\chapter{Gravitación y peso universales}\label{ch:g10:universal-gravitation}

Dejar caer una manzana: cae. La Luna, a sesenta radios terrestres de altura, parece no hacerlo nunca.
La gran idea de Newton es que la Luna \emph{is} cayendo --- aproximadamente
1,4 milímetros por segundo, exactamente la cantidad de una atracción
debilitándose como lo predice el cuadrado de la distancia. Una fórmula, con una
constante universal, rige la manzana, la Luna, las mareas y el
planetas. El primer volumen describía \omterm{def:g10:universal-gravitation:force}{gravitación} con palabras; esto
El capítulo lo mide.

\section{La ley de la gravitación universal.}

\begin{definition}[fuerza gravitacional]\label{def:g10:universal-gravitation:force}
Dos cuerpos cualesquiera se atraen entre sí, cualquiera que sea el material del que estén hechos y por más que
muy separados se sientan. Esta atracción es la \emph{gravitacional
fuerza}\index{fuerza gravitacional}, y el fenómeno es
\emph{gravitación}\index{gravitación}. No necesita contacto, ni cuerda y
no hay medio, y es \emph{mutual}: cada cuerpo tira del otro.
\end{definition}

\begin{theorem}[ley de gravitación universal]\label{thm:g10:universal-gravitation:law}
Dos cuerpos de masas $m_1$ y $m_2$ (en \unit{kg}), cuyos centros son un
distancia $d$ (en \unit{m}), se atraen entre sí con dos fuerzas
dirigido a lo largo de la línea que une los centros, cada uno apuntando hacia el
otro cuerpo, de igual magnitud
\[
F = G\,\frac{m_1 m_2}{d^2}
\qquad \text{(in \unit{N})},
\]
donde $G$ es la misma constante para cada par de cuerpos del universo.
\end{theorem}
\admitted

\begin{remark}[De donde viene la ley]
Newton destiló esta ley del movimiento observado de los planetas.
(leyes de Kepler); la deducción honesta se realiza en el Año 1
volumen. \Cref{ch:g12:satellites-kepler} ya pondrá a funcionar la ley
en \omterm{def:g10:universal-gravitation:satellite}{satélites} y planetas.
\end{remark}

\begin{definition}[La constante gravitacional]\label{def:g10:universal-gravitation:constant}
la constante
\[
G = \qty{6.67e-11}{N.m^2/kg^2}
\]
es el \emph{constante gravitacional}\index{constante gravitacional}. su
El valor minúsculo es la razón por la que la gravedad entre los objetos cotidianos es
imperceptible: sólo cuando al menos una masa es astronómica
$G m_1 m_2 / d^2$ equivalen a cualquier cosa.
\end{definition}

\begin{remark}[Leyendo la fórmula]\label{rem:g10:universal-gravitation:reading}
Cuatro características merecen atención.
\begin{itemize}
  \item \emph{Mutuos e iguales.} La Tierra te atrae con tu \omterm{def:g10:universal-gravitation:weight}{peso}
        --- y tiras de la Tierra exactamente con la misma fuerza. (El
        La Tierra simplemente responde menos: es aproximadamente $\num{e23}$ veces
        más pesado.)
  \item \emph{Cuadrado inverso.} Duplicar $d$ divide la fuerza por $4$;
        moverse diez veces más lo divide por $100$.
  \item \emph{Distancia entre centros.} Para cuerpos esféricos --- estrellas,
        planetas, balas de cañón --- $d$ se mide de centro a centro, no
        superficie a superficie.
  \item \emph{Ambas masas importan.} Duplicar cualquiera de las masas duplica la
        fuerza; la fórmula es simétrica en $m_1$ y $m_2$.
\end{itemize}
\end{remark}

\begin{example}[dos amigos]\label{ex:g10:universal-gravitation:friends}
Dos amigos \qty{70}{kg} están con sus centros \qty{1.0}{m} separados:
\[
F = \num{6.67e-11} \times \frac{70 \times 70}{1.0^2}
\approx \qty{3.3e-7}{N}.
\]
El \omterm{def:g10:universal-gravitation:weight}{peso} de cada amigo es aproximadamente \qty{6.9e2}{N} --- dos \emph{mil millones}
veces mayor. Su atracción mutua es real, pero es imposible sentirla.
\end{example}

\begin{example}[La Tierra sostiene a la Luna]\label{ex:g10:universal-gravitation:earthmoon}
Con $m_1 = \qty{5.97e24}{kg}$ (Tierra), $m_2 = \qty{7.35e22}{kg}$ (Luna)
y $d = \qty{3.84e8}{m}$:
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{2.0e20}{N}.
\]
Y la Luna atrae a la Tierra hacia atrás con el mismo \qty{2.0e20}{N} --- un
Los océanos responden dos veces al día, como lo hacen las mareas.
\end{example}

\begin{method}[Calcular una fuerza gravitacional]\label{met:g10:universal-gravitation:compute}
\begin{enumerate}
  \item Convierte las masas a \omterm{def:g10:orders-of-magnitude:unit}{kilogramos} y el centro a centro.
        distancia a metros.
  \item Aplicar $F = G m_1 m_2 / d^2$.
  \item Compruebe la potencia de diez por separado de los dígitos, como en
        \cref{ch:g10:orders-of-magnitude}: rango \omterm{def:g10:universal-gravitation:force}{fuerzas gravitacionales}
        de $\num{e-7}$ newtons (dos personas separadas por \omterm{def:g10:orders-of-magnitude:unit}{metro}) a $\num{e22}$ newtons
        (Sol en la Tierra), por lo que un \omterm{def:g10:orders-of-magnitude:scinot}{exponente} fuera de lugar es fácil de detectar.
\end{enumerate}
\end{method}

\section{El peso y la fuerza de la gravedad.}

\begin{definition}[Peso]\label{def:g10:universal-gravitation:weight}
El \emph{peso}\index{peso} $\vect P$ de un objeto en la superficie de
una estrella (un planeta, una luna, \dots) es la \omterm{def:g10:universal-gravitation:force}{fuerza gravitacional} la estrella
ejerce sobre él. Apunta hacia el centro de la estrella, la dirección en la que
llame a \emph{vertical}\index{vertical} --- y se mide en newtons.
\end{definition}

\begin{proposition}[Peso en la superficie de una estrella]\label{prop:g10:universal-gravitation:surfaceg}
En la superficie de una estrella esférica de masa $M$ y radio $R$, un objeto
de masa $m$ tiene \omterm{def:g10:universal-gravitation:weight}{peso}
\[
P = m\,g,
\qquad\text{where}\qquad
g = \frac{G M}{R^2}
\]
Depende sólo de la estrella, no del objeto.
\end{proposition}

\begin{proof}
El centro del objeto y el centro de la estrella están separados por una distancia $R$, por lo que
la ley del universal \omterm{def:g10:universal-gravitation:force}{gravitación}
(\cref{thm:g10:universal-gravitation:law}) da
$P = G M m / R^2 = m \times \left(G M / R^2\right)$. El soporte es el
Lo mismo para cada objeto en la superficie: llámelo $g$.
\end{proof}

\begin{definition}[Intensidad del campo gravitacional]\label{def:g10:universal-gravitation:fieldstrength}
La cantidad $g = G M / R^2$, en newtons por \omterm{def:g10:orders-of-magnitude:unit}{kilogramo}, es la
\emph{intensidad del campo gravitacional}\index{intensidad del campo gravitacional}
en la superficie de la estrella: la estrella atrae cada \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} con $g$ newtons.
En la Tierra, $g = \qty{9.81}{N/kg}$.
\end{definition}

\begin{example}[Computación $g$]\label{ex:g10:universal-gravitation:gvalues}
Para la Tierra ($M = \qty{5.97e24}{kg}$, $R = \qty{6.37e6}{m}$):
\[
g = \frac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.37e6})^2}
= \frac{\num{3.98e14}}{\num{4.06e13}}
\approx \qty{9.81}{N/kg}.
\]
Para la Luna ($M = \qty{7.35e22}{kg}$, $R = \qty{1.74e6}{m}$) lo mismo
el cálculo da $g_{\text{Moon}} \approx \qty{1.62}{N/kg}$ --- seis
veces menos: la Luna es mucho más clara, y ser más pequeña no compensa
por ello. Para Marte, $g_{\text{Mars}} \approx \qty{3.73}{N/kg}$.
\end{example}

\begin{remark}[¿Por qué los astronautas rebotan?]\label{rem:g10:universal-gravitation:bounce}
Un astronauta de masa \qty{120}{kg} (cuerpo más traje) pesa
$120 \times 9.81 \approx \qty{1180}{N}$ en la Tierra pero sólo
$120 \times 1.62 \approx \qty{194}{N}$ en la Luna. Músculos entrenados
bajo \qty{1180}{N} de repente lleva una sexta parte de la carga: de ahí el
famoso andar lunar lento y saltador. La masa --- y con ella el esfuerzo
necesario para \emph{detener} o \emph{doblar} --- no ha cambiado, por lo que
Los astronautas lunares se caían con tanta frecuencia.
\end{remark}

\section{¿Masa o peso? La balanza y la balanza de resorte.}

El lenguaje corriente dice que un saco de harina "pesa tres kilos". Física
divide esa frase en dos.

\begin{remark}[La masa no es peso]\label{rem:g10:universal-gravitation:massweight}
\begin{itemize}
  \item El \emph{masa} $m$ mide la cantidad de materia, en
        \omterm{def:g10:orders-of-magnitude:unit}{kilogramos}. Es una propiedad exclusiva del objeto: lo mismo en
        La Tierra, en la Luna o a la deriva en el espacio profundo.
  \item La \emph{\omterm{def:g10:universal-gravitation:weight}{peso}} $P = mg$ es una fuerza, en newtons, ejercida
        \emph{por una estrella} en el objeto. Cambia de estrella en estrella
        con $g$ --- y se desvanece lejos de cada estrella.
\end{itemize}
Confundirlos es inofensivo en el mercado y fatal en física.
Problema: el \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} no es un \omterm{def:g10:orders-of-magnitude:unit}{unidad} de fuerza.
\end{remark}

\begin{method}[La balanza y la balanza de resorte.]\label{met:g10:universal-gravitation:instruments}
\begin{itemize}
  \item A \emph{balanza de viga} compara el objeto con masas estándar:
        La escala \omterm{def:g10:universal-gravitation:weight}{pesas} de ambos platos por el mismo $g$, por lo que la comparación
        mide el \emph{masa} --- mismo veredicto sobre cualquier estrella.
  \item A \emph{escala de primavera} mide la fuerza que estira su resorte:
        lee el \emph{\omterm{def:g10:universal-gravitation:weight}{peso}}, en newtons, y cambia su lectura
        de estrella en estrella. Para recuperar la masa, divida por el local
        intensidad de campo: $m = P/g$.
\end{itemize}
\end{method}

\begin{example}[Harina en la Luna]\label{ex:g10:universal-gravitation:flour}
Un saco de harina \qty{3.0}{kg} se equilibra con \qty{3.0}{kg} de estándar
masas en la Tierra \emph{y} en la Luna. Una escala de resorte, sin embargo, lee
$3.0 \times 9.81 \approx \qty{29}{N}$ en la Tierra y
$3.0 \times 1.62 \approx \qty{4.9}{N}$ en la Luna. un comerciante vendiendo
``por el newton'' en la Luna se repartiría seis veces la harina ---
el comercio, como la ciencia, funciona con la masa.
\end{example}

\section{Cayendo alrededor de la Tierra: el cañón de Newton}

Lanzar una piedra horizontalmente: la gravedad curva su trayectoria formando un arco y
aterriza a unos metros de distancia. Lánzalo más rápido: aterriza más lejos. Newton
El experimento mental lleva la idea a su límite. La Tierra es redonda y
su superficie cae aproximadamente \qty{5}{m} por debajo de la horizontal por cada
\qty{8}{km} viajó; un cuerpo en caída libre deja caer \qty{4.9}{m} en su
primer segundo (\cref{ch:g12:projectile-motion} establecerá esto). entonces un
El proyectil que roza la Tierra en \qty{8}{km/s} cae exactamente tan rápido como
el suelo se curva: cae para siempre sin aterrizar jamás.

\begin{definition}[satélite y órbita]\label{def:g10:universal-gravitation:satellite}
Un \emph{satélite}\index{satélite} de una estrella es un cuerpo que cae
alrededor de la estrella sin llegar nunca a su superficie; el camino se repite
es su \emph{órbita}\index{órbita}.
\end{definition}

\begin{remark}[La Luna es una piedra que cae.]\label{rem:g10:universal-gravitation:fallingmoon}
La Luna es el \omterm{def:g10:universal-gravitation:satellite}{satélite} natural de la Tierra: cae hacia nosotros aproximadamente
\qty{1.4}{mm} cada segundo, y su movimiento lateral lo lleva
antes de que pueda acercarse más
(\cref{exo:g10:universal-gravitation:15} te permite rehacer el propio Newton

misma caída libre que su nave, eso, y no cualquier ausencia de gravedad,
Por eso flotan. \omterm{def:g10:universal-gravitation:force}{Gravitación} da forma a cada trayectoria, desde el
el arco de la piedra a la elipse del planeta; la historia cuantitativa de \omterm{def:g10:universal-gravitation:satellite}{órbitas}
se cuenta en \cref{ch:g12:satellites-kepler}.
\end{remark}
