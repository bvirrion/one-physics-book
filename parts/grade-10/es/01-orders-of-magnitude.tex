\section{Unidades: la gramática de la medida.}

\begin{definition}[Cantidades físicas y unidades SI]\label{def:g10:orders-of-magnitude:unit}
A \emph{cantidad fisica}\index{cantidad fisica} es una propiedad de
el mundo que se puede medir: una longitud, una duración, una masa.
Medirlo significa contar cuantas veces una cantidad de referencia ---
el \emph{unidad}\index{unidad} --- encaja en él. el
\emph{Sistema Internacional de Unidades}\index{Sistema Internacional de
Unidades (SI)} (SI) repara siete unidades base, de las cuales tres ejecutan todo esto
volumen: el \emph{metro}\index{metro} (\unit{m}) para longitud, el
\emph{kilogramo}\index{kilogramo} (\unit{kg}) para masa y
\emph{segundo}\index{segundo} (\unit{s}) por tiempo. (Los otros cuatro ---
amperio, kelvin, topo, candela --- se unirán cuando la electricidad, el calor y
la química entra en la historia.)
\end{definition}

\begin{remark}[Un número sin unidad no es un resultado]\label{rem:g10:orders-of-magnitude:units}
``La distancia es $384$'' no significa nada: \qty{384}{mm},
\qty{384}{km} y \qty{384}{Mm} son tres mundos diferentes (el último
resulta ser la distancia a la Luna). En 1999, la NASA perdió el
\$327~millones \emph{Orbitador climático de Marte} porque un programa calculó
fuerzas del propulsor en libras-fuerza, mientras que otro las lee como newtons.
Regla de la casa, desde esta línea hasta el final del libro: cada medida o
El número calculado lleva su \omterm{def:g10:orders-of-magnitude:unit}{unidad}, en cada línea del cálculo.
\end{remark}

\begin{definition}[prefijos SI]\label{def:g10:orders-of-magnitude:prefixes}
Un \emph{prefijo}\index{prefijo (SI)} pegado a un \omterm{def:g10:orders-of-magnitude:unit}{unidad} lo multiplica por un
potencia fija de diez:
\begin{center}
\begin{tabular}{llc@{\qquad\qquad}llc}
femto & f & $10^{-15}$ & centi & c & $10^{-2}$ \\
pico  & p & $10^{-12}$ & kilo  & k & $10^{3}$  \\
nano  & n & $10^{-9}$  & mega  & M & $10^{6}$  \\
micro & $\mu$ & $10^{-6}$ & giga & G & $10^{9}$ \\
milli & m & $10^{-3}$  & tera  & T & $10^{12}$ \\
\end{tabular}
\end{center}
Por tanto, $\qty{1}{nm} = \qty{e-9}{m}$ y $\qty{1}{km} = \qty{e3}{m}$.
\end{definition}

\begin{example}[Prefijos de lectura]\label{ex:g10:orders-of-magnitude:prefixes}
La luz verde tiene una longitud de onda de aproximadamente $\qty{500}{nm} = \qty{5.00e-7}{m}$; un
el cabello humano tiene aproximadamente $\qty{70}{\micro m} = \qty{7.0e-5}{m}$ de espesor; el
La luna orbita en $\qty{384}{Mm} = \qty{3.84e8}{m}$. Una gota de lluvia tiene
masa aproximadamente $\qty{30}{mg} = \qty{3.0e-5}{kg}$ --- tenga en cuenta que la base SI
\omterm{def:g10:orders-of-magnitude:unit}{unidad} de masa, solo entre los siete, ya lleva un \omterm{def:g10:orders-of-magnitude:prefixes}{prefijo}.
\end{example}

\section{Notación científica y cifras significativas.}

\begin{definition}[Notación científica]\label{def:g10:orders-of-magnitude:scinot}
Un número está en \emph{notación científica}\index{notación científica}
cuando se escribe como
\[
a \times 10^{n}, \qquad 1 \leq a < 10, \quad n \text{ an integer.}
\]
El factor $a$ es el \emph{mantisa}\index{mantisa}, el número entero
$n$ o \emph{exponente}\index{exponente}.
Así $\num{149600000} = \num{1.496e8}$ y
$\num{0.00052} = \num{5.2e-4}$.
\end{definition}

\begin{proposition}[Cálculo con potencias de diez]\label{prop:g10:orders-of-magnitude:powers}
Para todos los números enteros $m$ y $n$:
\[
10^{m} \times 10^{n} = 10^{m+n}, \qquad
\frac{1}{10^{n}} = 10^{-n}, \qquad
\left(10^{m}\right)^{n} = 10^{mn}.
\]
\end{proposition}

\begin{proof}[prueba parcial]
Para \omterm{def:g10:orders-of-magnitude:scinot}{exponentes} positivo, cuente los factores: $10^{m} \times 10^{n}$ es un
producto de $m + n$ decenas y $(10^m)^n$ cadenas $n$ grupos de $m$ decenas.
Escribir $10^{-n}$ para $1/10^{n}$ es precisamente la convención que
mantiene la primera regla verdadera para todos los números enteros: obliga
$10^{n} \times 10^{-n} = 10^{0} = 1$. Las matemáticas compañeras
El volumen estudia las reglas \omterm{def:g10:orders-of-magnitude:scinot}{exponente} en su totalidad.
\end{proof}

\begin{example}[Multiplicar en notación científica]\label{ex:g10:orders-of-magnitude:scinot}
¿Cuántas Tierras caben en el Sol, en masa? con
$m_{\text{Sun}} = \qty{1.99e30}{kg}$ y
$m_{\text{Earth}} = \qty{5.97e24}{kg}$:
\[
\frac{\num{1.99e30}}{\num{5.97e24}}
= \frac{1.99}{5.97} \times 10^{30-24}
= 0.333 \times 10^{6}
= \num{3.33e5}.
\]
\omterm{def:g10:orders-of-magnitude:scinot}{mantisas} y \omterm{def:g10:orders-of-magnitude:scinot}{exponentes} se manejan por separado; la respuesta final es
renormalizado para que \omterm{def:g10:orders-of-magnitude:scinot}{mantisa} vuelva a aterrizar en $[1, 10)$.
\end{example}

\begin{definition}[Cifras significativas]\label{def:g10:orders-of-magnitude:sigfig}
El \emph{cifras significativas}\index{cifras significativas} de un
El número medido son los dígitos realmente avalados por el
Medición: todos los dígitos escritos excepto los ceros iniciales. Seguimiento
recuento de ceros --- $\qty{5.20e-3}{s}$ (tres cifras significativas)
promete más de $\qty{5.2e-3}{s}$ (dos). \omterm{def:g10:orders-of-magnitude:scinot}{Notación científica} espectáculos
el recuento de un vistazo: el \omterm{def:g10:orders-of-magnitude:scinot}{mantisa} lleva exactamente los datos significativos
cifras.
\end{definition}

\begin{method}[Cuántos dígitos conservar]\label{met:g10:orders-of-magnitude:sigfig}
Un resultado calculado no puede ser más preciso que sus ingredientes.
\begin{enumerate}
  \item En un producto o cociente, mantener tantos \omterm{def:g10:orders-of-magnitude:sigfig}{cifras significativas} como
        el factor preciso \emph{el menos}.
  \item Redondea sólo la respuesta final, nunca los pasos intermedios.
  \item En caso de duda, tres \omterm{def:g10:orders-of-magnitude:sigfig}{cifras significativas} son los que funcionan.
        precisión de este libro.
\end{enumerate}
\end{method}

\begin{example}[circunferencia de la Tierra]\label{ex:g10:orders-of-magnitude:sigfig}
El diámetro de la Tierra es $d = \qty{1.27e7}{m}$ (tres \omterm{def:g10:orders-of-magnitude:sigfig}{cifras significativas}),
entonces su circunferencia es
$\pi d = \num{3.14159} \times \qty{1.27e7}{m} = \qty{3.99e7}{m}$ ---
tres cifras, no las ocho que muestra la calculadora. escribiendo
$\qty{39898227}{m}$ afirmaría conocer el planeta hasta el \omterm{def:g10:orders-of-magnitude:unit}{metro}.
\end{example}

\section{La escalera del universo.}

\begin{definition}[Orden de magnitud]\label{def:g10:orders-of-magnitude:oom}
El \emph{orden de magnitud}\index{orden de magnitud} de una cantidad
es la potencia de diez más cercana a ella: escribe la cantidad como
$a \times 10^{n}$ con $1 \leq a < 10$ y toma $10^{n}$ si $a < 5$,
$10^{n+1}$ si $a \geq 5$. (La convención de límites exacta nunca
importa: una declaración de orden de magnitud sólo está destinada a dentro
un factor de diez.)
\end{definition}

\begin{example}[Órdenes de magnitud]\label{ex:g10:orders-of-magnitude:oom}
Una vida humana dura aproximadamente $80$ años, es decir\
$80 \times \qty{3.16e7}{s} \approx \qty{2.5e9}{s}$: \omterm{def:g10:orders-of-magnitude:oom}{orden de magnitud}
$\qty{e9}{s}$ --- unos miles de millones de segundos, gástalos bien. un humano es
$\qty{1.7}{m}$ alto: pedido $\qty{e0}{m}$. Monte Everest,
$\qty{8848}{m}$: pedido $\qty{e4}{m}$, desde $8.848 \geq 5$.
\end{example}

La acumulación de tales estimaciones por \omterm{def:g10:orders-of-magnitude:scinot}{exponente} produce la imagen maestra de
este capítulo: una escalera en la que cada peldaño es un mundo diez veces más ancho
que el de abajo.

\begin{definition}[Unidad astronómica y año luz]\label{def:g10:orders-of-magnitude:lightyear}
El \emph{unidad astronómica}\index{unidad astronómica}
($\qty{1}{au} = \qty{1.496e11}{m}$) es el promedio Tierra--Sol
distancia. El \emph{año luz}\index{año luz} (\unit{ly}) es el
distancia que recorre la luz en un año. La luz se mueve en el vacío a
\[
c = \qty{3.00e8}{m/s},
\]
la velocidad más rápida que permite la naturaleza --- entonces el año luz es una distancia,
no es un tiempo.
\end{definition}

\begin{method}[Comprobación de cordura por orden de magnitud]\label{met:g10:orders-of-magnitude:sanity}
Antes de confiar en cualquier resultado calculado:
\begin{enumerate}
  \item redondea cada entrada a su \omterm{def:g10:orders-of-magnitude:oom}{orden de magnitud};
  \item rehacer el cálculo sólo con potencias de diez, utilizando
        \cref{prop:g10:orders-of-magnitude:powers} --- esto toma
        segundos;
  \item comparar: la respuesta precisa debe coincidir con la estimación para
        dentro de un factor de diez aproximadamente, y su \omterm{def:g10:orders-of-magnitude:unit}{unidad} debe ser el correcto
        uno. Si cualquiera de las comprobaciones falla, busque el error antes de continuar.
        encendido.
\end{enumerate}
\end{method}

\begin{remark}[¿Por qué molestarse?]\label{rem:g10:orders-of-magnitude:sanity}
Una calculadora informará felizmente que una manzana tiene masa.
$\qty{4e8}{kg}$ --- no puede saberlo mejor. Ninguna fórmula marca la
absurdo; el hábito sí. Los físicos estiman primero y calculan segundo:
la estimación detecta puntos decimales mal colocados, intercambia \omterm{def:g10:orders-of-magnitude:unit}{unidades} y
fracciones invertidas, los tres errores detrás de la mayoría de las respuestas incorrectas en
cada nivel.
\end{remark}

\section{Medición e incertidumbre}

Ningún instrumento lee el valor real. Una medición es honesta sólo cuando
también informa qué tan lejos podría estar.

\begin{definition}[Incertidumbre absoluta y relativa]\label{def:g10:orders-of-magnitude:uncertainty}
Una medida de una cantidad $x$ se expresa como
\[
x = x_{\text{m}} \pm \Delta x ,
\]
donde $x_{\text{m}}$ es el valor medido y \emph{absoluto
incertidumbre}\index{incertidumbre!absoluto} $\Delta x > 0$, en el mismo
\omterm{def:g10:orders-of-magnitude:unit}{unidad}, limita el error plausible: se afirma que el valor verdadero miente
entre $x_{\text{m}} - \Delta x$ y $x_{\text{m}} + \Delta x$. el
\emph{incertidumbre relativa}\index{incertidumbre!relativo} es la relación
$\Delta x / x_{\text{m}}$, normalmente expresado en porcentaje; es lo que el
palabra medidas "precisas".
\end{definition}

\begin{method}[Lectura de un instrumento graduado]\label{met:g10:orders-of-magnitude:reading}
Para una regla, una probeta o un dial:
\begin{enumerate}
  \item lea la graduación más cercana a la marca, ojo perpendicular a la
        escala (un ojo rasgado cambia la lectura);
  \item tome al menos la mitad de la graduación más pequeña que el \omterm{def:g10:orders-of-magnitude:uncertainty}{absoluto
        incertidumbre} --- más si el borde del objeto o el líquido
        la superficie es borrosa;
  \item escribe el resultado como $x_{\text{m}} \pm \Delta x$ con su \omterm{def:g10:orders-of-magnitude:unit}{unidad},
        y no guardes ningún dígito más allá del incierto.
\end{enumerate}
\end{method}

\begin{example}[Precisión relativa]\label{ex:g10:orders-of-magnitude:reading}
La lectura anterior tiene \omterm{def:g10:orders-of-magnitude:uncertainty}{incertidumbre relativa}
$\frac{0.5}{127} \approx 0.4\%$. La misma regla que mide un

El instrumento es diez veces menos preciso en el objeto pequeño. La precisión es

\end{example}

\begin{remark}[Pequeño absoluto, enorme relativo --- y a la inversa]\label{rem:g10:orders-of-magnitude:precision}
Las estaciones láser miden la distancia Tierra-Luna, $\qty{3.84e8}{m}$, para
sobre $\pm\qty{3}{cm}$: un \omterm{def:g10:orders-of-magnitude:uncertainty}{incertidumbre relativa} de $10^{-10}$, entre
las mediciones más nítidas jamás realizadas, con un error absoluto sin
gobernante se jactaría. Pregunta siempre \emph{en relación con lo que}: la pareja
$(\Delta x,\ \Delta x / x_{\text{m}})$ cuenta toda la historia, y
cualquiera de los dos números por sí solo puede inducir a error.
\end{remark}

\section{Problema: del átomo a Andrómeda}

\begin{problem}[{Problema de fin de semana --- un solo grano de arena mide el
átomo, el Sistema Solar, la galaxia y el universo observable, y
te ubicas en la escalera entre ellos}]
\label{pb:g10:orders-of-magnitude:1}
Un grano de arena es el objeto más humilde de la física y este fin de semana
trabaja horas extras: primero contamos sus átomos, luego reducimos el Sol a
su tamaño y su ritmo en el cosmos, luego dejamos que la luz haga la inspección,
y finalmente contamos los átomos en \emph{tú}. Datos utilizados en:
diámetro de grano $\qty{0.5}{mm}$; diámetro del átomo $\qty{2.5e-10}{m}$;
densidad de arena $\qty{2500}{kg/m^3}$; diámetros: Sol
$\qty{1.39e9}{m}$, Tierra $\qty{1.27e7}{m}$, Vía Láctea
$\qty{9.5e20}{m}$, universo observable $\qty{8.8e26}{m}$; distancias
de nosotros: Luna $\qty{3.84e8}{m}$, Sol $\qty{1.496e11}{m}$, Neptuno
$\qty{4.5e12}{m}$, Próxima Centauri $\qty{4.0e16}{m}$, Andrómeda
galaxia $\qty{2.4e22}{m}$; $c = \qty{3.00e8}{m/s}$; una molécula de agua
tiene masa $\qty{3.0e-26}{kg}$ y contiene tres átomos.

\medskip
\noindent\textbf{Parte I --- Anatomy of the grain.}
\begin{enumerate}
  \item Escriba el diámetro del grano y el diámetro del átomo en \omterm{def:g10:orders-of-magnitude:unit}{metros}, en
        \omterm{def:g10:orders-of-magnitude:scinot}{notación científica} y nombre el SI \omterm{def:g10:orders-of-magnitude:prefixes}{prefijo} más cercano a cada uno de ellos.
        a.
  \item ¿Cuántos átomos se encuentran uno al lado del otro a lo largo de un diámetro de grano?
  \item Modele el grano como un cubo de lado $\qty{0.5}{mm}$ lleno de
        átomos. Estima el número de átomos en el grano, como \omterm{def:g10:orders-of-magnitude:oom}{orden
        de magnitud}.
  \item Calcula el volumen de ese cubo, luego la masa del grano.
        Expresa la masa en miligramos.
  \item Control de cordura del modelo: dividir la masa del grano por su número
        de átomos y comparar con el conocido \omterm{def:g10:orders-of-magnitude:oom}{orden de magnitud} de
        masas atómicas (algunas $10^{-26}\,\unit{kg}$). ¿Veredicto?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The Sun as a grain of sand.}
Construimos una maqueta del cosmos en la que el Sol es el grano:
cada tamaño de modelo es igual al tamaño real multiplicado por un factor $s$.
\begin{enumerate}[resume]
  \item Calcular el factor de escala $s$.
  \item En el modelo, encuentre el diámetro de la Tierra y su distancia desde
        el grano-Sol. ¿Qué objeto del mundo de la Parte I tiene aproximadamente la misma
        ¿Modelo del tamaño de la Tierra?
  \item ¿A qué distancia del grano orbita el modelo Neptuno? ¿Qué pedazo de
        ¿Los muebles albergarían todo el sistema planetario?
  \item ¿A qué distancia está el modelo Proxima Centauri?
  \item Dos granos de arena, $\qty{14}{km}$ de diferencia, y casi nada
        en el medio: indique en una o dos oraciones cuál es el modelo
        revela cuán vacía está la galaxia.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Light does the surveying.}
\begin{enumerate}[resume]
  \item ¿Cuánto tarda la luz en atravesar el grano de arena real?
  \item ¿Cuánto tarda en llegarnos la luz desde la Luna, y desde
        el sol?
  \item Vuelva a calcular el \omterm{def:g10:orders-of-magnitude:lightyear}{año luz} en \omterm{def:g10:orders-of-magnitude:unit}{metros} a partir de $c$ y la longitud de
        el año, y verificar que la distancia Proxima en los datos
        corresponde aproximadamente a $\qty{4.2}{ly}$.
  \item Convierte la distancia de Andrómeda a \omterm{def:g10:orders-of-magnitude:lightyear}{años luz}. Aproximadamente cuando
        ¿La luz que ahora entra en tus ojos abandonó Andrómeda y qué
        caminaba por la Tierra en ese momento?
  \item ¿Qué distancia recorre la luz en un nanosegundo? un procesador
        registrado en $\qty{3}{GHz}$ inicia una operación cada tercio de un
        nanosegundo: explique en una frase por qué una computadora rápida debe
        ser una computadora pequeña.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- You, between atom and Andromeda.}
\begin{enumerate}[resume]
  \item Un ser humano es aproximadamente $\qty{70}{kg}$ de agua. Usando los datos,
        estimar la masa promedio de un átomo en agua, entonces la
        Número de átomos en un ser humano.
  \item Compara esa cuenta con los átomos del grano (pregunta 3)
        y con las aproximadamente $2\times10^{23}$ estrellas del observable
        universo. Da ambos factores.
  \item En la escalera, ¿estás más cerca del átomo o de lo observable?
        universo? Compare las relaciones $\text{(you)}/\text{(atom)}$ y
        $\text{(universe)}/\text{(you)}$, tomando $\qty{1.7}{m}$ para
        usted.
  \item Encuentra la talla $x$ exactamente a mitad de la escalera.
        entre el núcleo ($\qty{e-15}{m}$) y el observable
        universo: $x$ debe satisfacer $x / 10^{-15} = 10^{26.9} / x$
        (usando $\qty{8.8e26}{m} \approx 10^{26.9}\,\unit{m}$). que
        ¿El objeto del Sistema Solar tiene exactamente ese tamaño?
  \item El remate. Reducir el universo observable en la Parte II
        factor $s$, de modo que el Sol es un grano de arena. dar el modelo
        diámetro del universo en \omterm{def:g10:orders-of-magnitude:unit}{metros}, luego en \omterm{def:g10:orders-of-magnitude:lightyear}{unidades astronómicas}, y
        concluir en una frase: incluso con cada estrella convertida en
        grano de arena, el universo desborda toda escala humana --- sólo
        potencias de diez lo mantienen.
\end{enumerate}
\end{problem}
