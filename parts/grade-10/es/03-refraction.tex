\section{Reflexión: un recordatorio}

En un medio transparente homogéneo, la luz viaja en línea recta ---
los rayos de \cref{ch:g10:light-spectra}. En el límite entre dos
medios, parte de la luz rebota (\emph{reflexión}) y parte
cruza (\emph{\omterm{def:g10:refraction:refraction}{refracción}}). Ambos obedecen leyes geométricas exactas, establecidas
con una convención: los ángulos se miden desde el \emph{\omterm{def:g10:refraction:angles}{normal}}, nunca
desde la superficie.

\begin{definition}[Normal y ángulo de incidencia]\label{def:g10:refraction:angles}
En el punto donde un rayo toca una superficie, el
\emph{normal}\index{normal} es la línea perpendicular a la superficie en
ese punto. El \emph{ángulo de incidencia}\index{ángulo de incidencia} $i_1$
es el ángulo entre el rayo entrante y la normal; los ángulos de
reflexión y de \omterm{def:g10:refraction:refraction}{refracción} se miden desde la normal en el mismo
manera.
\end{definition}

\begin{proposition}[Ley de reflexión]\label{prop:g10:refraction:reflection}
El rayo reflejado se encuentra en el plano que contiene al rayo incidente y al
\omterm{def:g10:refraction:angles}{normal}, al otro lado del \omterm{def:g10:refraction:angles}{normal}, y el ángulo de reflexión
es igual al \omterm{def:g10:refraction:angles}{ángulo de incidencia}:
\[
i_1' = i_1 .
\]
\end{proposition}
\admitted

\begin{example}[Ángulos de lectura correctamente]\label{ex:g10:refraction:mirror}
Un rayo láser incide en un espejo ``en $35^\circ$'' --- medido desde el
superficie del espejo, como a menudo se describen las vigas. El \omterm{def:g10:refraction:angles}{ángulo de incidencia} es
$90^\circ - 35^\circ = 55^\circ$, por lo que el rayo reflejado también sale por
$55^\circ$ del \omterm{def:g10:refraction:angles}{normal}, y el haz gira
$180^\circ - 2 \times 55^\circ = 70^\circ$. olvidando el
La convención \omterm{def:g10:refraction:angles}{normal} es el error más común de este capítulo; cuesta
$90^\circ$ cada vez.
\end{example}

\section{Refracción y el índice de refracción.}

Sostenga un lápiz en un vaso medio lleno: parece roto en la superficie.
La luz de la parte submarina cambia de dirección a medida que sale del agua,
entonces el ojo --- que asume rayos rectos --- reconstruye la punta
en algún lugar no lo es.

\begin{definition}[Refracción]\label{def:g10:refraction:refraction}
\emph{Refracción}\index{refracción} es el cambio de dirección de la luz
cuando cruza el límite entre dos medios transparentes. El rayo en el
El segundo medio es el \emph{rayo refractado}\index{rayo refractado}, y su
El ángulo del \omterm{def:g10:refraction:angles}{normal} es el ángulo de refracción $i_2$.
\end{definition}

\omterm{def:g10:refraction:refraction}{Refracción} sucede porque la luz tiene diferentes velocidades en diferentes
medios de comunicación; Cada material se caracteriza por cuánto ralentiza la luz.

\begin{definition}[Índice de refracción]\label{def:g10:refraction:index}
El \emph{índice de refracción}\index{índice de refracción} de un transparente
medio es la proporción
\[
n = \frac{c}{v},
\]
donde $c = \qty{3.00e8}{m/s}$ es la velocidad de la luz en el vacío y $v$ su
velocidad en el medio. Desde $v \leq c$, el índice satisface $n \geq 1$; eso
no tiene \omterm{def:g10:orders-of-magnitude:unit}{unidad}. Algunos valores útiles:
\begin{center}
\begin{tabular}{lccccc}
medium & air & water & plexiglas & glass & diamond \\
$n$ & $1.0003$ & $1.33$ & $1.49$ & $1.50$ & $2.42$ \\
\end{tabular}
\end{center}
\end{definition}

\begin{example}[Velocidad de la luz en el agua]\label{ex:g10:refraction:speed}
En agua, $v = \dfrac{c}{n} = \dfrac{\qty{3.00e8}{m/s}}{1.33}
= \qty{2.26e8}{m/s}$. En el diamante, la luz se arrastra hacia
$\qty{3.00e8}{m/s}/2.42 = \qty{1.24e8}{m/s}$ --- menos de la mitad de su
velocidad de vacío. Para aire, $n = 1.0003$ cambia la velocidad en $0.03\%$: en
En este capítulo tratamos el aire como vacío y tomamos $n_{\text{air}} = 1.00$.
\end{example}

\begin{theorem}[Ley de refracción de Snell]\label{thm:g10:refraction:snell}
\index{ley de snell}
El \omterm{def:g10:refraction:refraction}{rayo refractado} se encuentra en el plano del rayo incidente y el \omterm{def:g10:refraction:angles}{normal},
en el otro lado del \omterm{def:g10:refraction:angles}{normal}, y los ángulos de incidencia y
\omterm{def:g10:refraction:refraction}{refracción} satisfacer
\[
n_1 \sin i_1 = n_2 \sin i_2 ,
\]
donde $n_1$ y $n_2$ son los índices de refracción de los dos medios. el
El camino es reversible: la luz viaja hacia atrás a lo largo del \omterm{def:g10:refraction:refraction}{rayo refractado}.
sale a lo largo del rayo incidente.
\end{theorem}
\admitted

hechos. Honestamente, se derivan del volumen del Año 1, donde la luz es

\end{remark}

\begin{example}[aire y agua]\label{ex:g10:refraction:airwater}
Un rayo impacta en un estanque en calma en $i_1 = 30^\circ$. Con $n_1 = 1.00$ y
$n_2 = 1.33$:
\[
\sin i_2 = \frac{n_1 \sin i_1}{n_2} = \frac{\sin 30^\circ}{1.33}
= \frac{0.500}{1.33} = 0.376,
\qquad
i_2 = 22.1^\circ .
\]
El rayo se inclina hacia el \omterm{def:g10:refraction:angles}{normal}, como siempre lo hace al entrar en una
medio más lento ($n_2 > n_1$). Al salir del agua, se ejecuta el mismo cálculo.
hacia atrás y el rayo se curva \emph{lejos} desde el \omterm{def:g10:refraction:angles}{normal}.
\end{example}

\begin{method}[Resolviendo un problema de refracción]\label{met:g10:refraction:method}
\begin{enumerate}
  \item Dibuja la superficie, el \omterm{def:g10:refraction:angles}{normal} y el rayo; identificar $n_1$
        (medio del rayo incidente) y $n_2$.
  \item Compruebe que los ángulos indicados se midan desde el \omterm{def:g10:refraction:angles}{normal};
        convertir si se miden desde la superficie.
  \item Escribe $n_1 \sin i_1 = n_2 \sin i_2$ y aísla lo desconocido.
        seno; obtenga el ángulo con el seno inverso de la calculadora, como en
        El volumen de matemáticas.
  \item Control de cordura: el rayo se inclina hacia \omterm{def:g10:refraction:angles}{normal} si $n_2 > n_1$,
        lejos de él si $n_2 < n_1$ y $\sin i_2$ deben salir
        entre $0$ y $1$.
\end{enumerate}
\end{method}

\begin{proposition}[profundidad aparente]\label{prop:g10:refraction:depth}
\index{profundidad aparente}
Un objeto a una profundidad $d$ bajo el agua, visto desde arriba en ángulos pequeños,
parece estar en el \emph{\omterm{prop:g10:refraction:depth}{profundidad aparente}}
\[
d' = \frac{d}{n} ,
\]
donde $n$ es el índice de agua. Un charco de profundidad \qty{1.20}{m} mira
sólo $1.20/1.33 = \qty{0.90}{m}$ profundo --- una causa clásica de imprudencia
inmersiones.
\end{proposition}
\admitted

llegando al ojo; la derivación se realiza en el volumen del Año 1. el

lejos del \omterm{def:g10:refraction:angles}{normal}, por lo que el ojo los rastrea hasta un punto más alto
que el objeto.
\end{remark}

\section{Dispersión: una ley por color}

Un prisma convierte la luz blanca del sol en un arco iris, mientras los espectros de
\cref{ch:g10:light-spectra} mostró. \omterm{def:g10:refraction:refraction}{Refracción} explica \emph{por qué}: el
\omterm{def:g10:refraction:index}{índice de refracción} de un medio no es exactamente igual para todos los colores.

\begin{definition}[Dispersión]\label{def:g10:refraction:dispersion}
Un medio es \emph{dispersivo}\index{dispersión} cuando su \omterm{def:g10:refraction:index}{refractivo
índice} depende del \omterm{def:g10:light-spectra:wavelength}{longitud de onda} de la luz. En vidrio y agua, $n$ es
ligeramente más grande para luz azul (corto \omterm{def:g10:light-spectra:wavelength}{longitud de onda}) que para luz roja
(largo \omterm{def:g10:light-spectra:wavelength}{longitud de onda}) --- entonces el azul se dobla más.
\end{definition}

\begin{example}[Dos colores, una superficie.]\label{ex:g10:refraction:twocolors}
Un rayo de \omterm{def:g10:light-spectra:white}{luz blanca} golpea el cristal de corona en $i_1 = 60^\circ$. Para esto
vidrio $n_{\text{red}} = 1.51$ y $n_{\text{blue}} = 1.53$. \omterm{thm:g10:refraction:snell}{ley de snell},
una vez por color:
\[
\sin i_{\text{red}} = \frac{\sin 60^\circ}{1.51} = 0.574,
\quad i_{\text{red}} = 35.0^\circ ;
\qquad
\sin i_{\text{blue}} = \frac{\sin 60^\circ}{1.53} = 0.566,
\quad i_{\text{blue}} = 34.5^\circ .
\]
Una sola superficie divide \omterm{def:g10:light-spectra:white}{luz blanca} en medio grado --- invisible en un
cristal de una ventana, pero un prisma se refracta dos veces en el mismo sentido de rotación,
Se ensancha el abanico, y a unos metros los colores se encuentran a centímetros de distancia.
\end{example}

\begin{remark}\label{rem:g10:refraction:rainbow}
El arco iris es el mismo cálculo realizado dentro de una gota de lluvia: la luz del sol
se refracta al entrar en la gota, se refleja en su pared posterior y se refracta
de nuevo a la salida. El índice del agua va desde $1.331$ (rojo) hasta $1.343$
(azul), por lo que cada color sale en su propio ángulo --- y no hay dos observadores
ver el mismo arcoiris.
\end{remark}

\section{Reflexión interna total}

\omterm{thm:g10:refraction:snell}{ley de snell} contiene una trampa. Al pasar del agua al aire, el \omterm{def:g10:refraction:refraction}{refractado
rayo} se aleja del \omterm{def:g10:refraction:angles}{normal}: $\sin i_2 = 1.33 \sin i_1 > \sin i_1$.
Aumente $i_1$ y la fórmula pronto exigirá $\sin i_2 > 1$, lo cual no
El ángulo puede ofrecer. \omterm{def:g10:refraction:refraction}{Refracción} simplemente deja de ocurrir.

\begin{definition}[Ángulo crítico y reflexión interna total]\label{def:g10:refraction:critical}
Dejemos que la luz viaje en un medio de índice $n_1$ hacia un medio de menor
índice $n_2 < n_1$. El \emph{ángulo crítico}\index{ángulo crítico}
$i_c$ es el \omterm{def:g10:refraction:angles}{ángulo de incidencia} para el cual el \omterm{def:g10:refraction:refraction}{rayo refractado} roza el
superficie ($i_2 = 90^\circ$). Para $i_1 > i_c$ no existe \omterm{def:g10:refraction:refraction}{rayo refractado}: el
La superficie refleja toda la luz de regreso al primer medio, siguiendo el
ley de la reflexión. Este es \emph{total interno
reflexión}\index{reflexión interna total}.
\end{definition}

\begin{proposition}[Fórmula del ángulo crítico]\label{prop:g10:refraction:criticalformula}
Para $n_1 > n_2$,
\[
\sin i_c = \frac{n_2}{n_1} .
\]
\end{proposition}

\begin{proof}
Configure $i_2 = 90^\circ$ en \omterm{thm:g10:refraction:snell}{ley de snell}: $n_1 \sin i_c = n_2 \sin 90^\circ
= n_2$. Para$i_1 > i_c$, \omterm{thm:g10:refraction:snell}{ley de snell} requeriría
$\sin i_2 = \frac{n_1}{n_2}\sin i_1 > 1$: no existe dirección refractada.
(Que la luz es entonces \emph{enteramente} reflejada --- no absorbida --- es
un hecho experimental, confirmado por la teoría ondulatoria del Año 1
volumen.)
\end{proof}

\begin{example}[Tres ángulos críticos]\label{ex:g10:refraction:criticalvalues}
Hacia el aire: para agua, $\sin i_c = 1/1.33 = 0.752$, entonces
$i_c = 48.8^\circ$; para vidrio, $\sin i_c = 1/1.50$, entonces
$i_c = 41.8^\circ$; para diamante, $\sin i_c = 1/2.42 = 0.413$, entonces
$i_c = 24.4^\circ$. Cuanto mayor sea el índice, más estrecho será el cono de escape:
en diamante, cualquier rayo a más de $24.4^\circ$ del \omterm{def:g10:refraction:angles}{normal} de una faceta es
atrapado --- el secreto de su brillo, disecado en
\cref{pb:g10:refraction:1}.
\end{example}

\section{Fibras ópticas}

\omterm{def:g10:refraction:critical}{Reflexión interna total} convierte un hilo de vidrio en un tubo para luz:
un rayo que entra casi paralelo al eje incide en la pared mucho más allá
el \omterm{def:g10:refraction:critical}{ángulo crítico}, refleja sin pérdida, y lo hace millones de veces
por kilómetro sin fugas.

\begin{definition}[fibra]\label{def:g10:refraction:fiber}
Un \emph{fibra óptica}\index{fibra óptica} es un fino hilo de vidrio hecho
de un \emph{centro}\index{centro} de índice $n_1$ envuelto en un
\emph{revestimiento}\index{revestimiento} de índice ligeramente más pequeño $n_2 < n_1$.
La luz inyectada en un extremo viaja a lo largo del núcleo repitiendo \omterm{def:g10:refraction:critical}{total
reflexión interna} en el límite núcleo-revestimiento.
\end{definition}

\begin{example}[Una fibra en números]\label{ex:g10:refraction:fibernumbers}
Con $n_1 = 1.48$ y $n_2 = 1.46$,
\[
\sin i_c = \frac{1.46}{1.48} = 0.986,
\qquad i_c = 80.6^\circ :
\]
sólo se guían los rayos dentro de aproximadamente $9^\circ$ del eje, lo cual es
Bien, ya que así es exactamente como se inyecta la luz láser. Dentro del
\omterm{def:g10:refraction:fiber}{centro} la luz viaja en $v = c/1.48 = \qty{2.03e8}{m/s}$: un pulso
cruza una fibra \qty{100}{km} en aproximadamente medio milisegundo.
\end{example}

\begin{remark}[Orden de magnitud de los enlaces de datos]\label{rem:g10:refraction:datarates}
Cables submarinos: haces de unas pocas docenas de fibras en el fondo del océano.
--- transporta casi todo el tráfico intercontinental. Una única fibra moderna
transporta del orden de $10^{13}$ bits por segundo enviando muchos
\omterm{def:g10:light-spectra:wavelength}{longitudes de onda} a la vez; un cable completo alcanza $10^{14}$ bits por segundo,
con un amplificador cada \qty{80}{km} más o menos. Los satélites, a modo de comparación,
manejar una pequeña fracción del porcentaje del tráfico.
\end{remark}

\section{Problema: El diamante, la piscina y la fibra}

\begin{problem}[{Problema del fin de semana --- tres vidas de la ley de Snell: una piscina
que yace en su profundidad, una piedra cortada para que la luz no pueda salir, y
el océano de sesenta milisegundos bajo Internet}]
\label{pb:g10:refraction:1}
Una igualdad de dos senos, tres carreras. En una piscina hace trampa
Tus ojos miran hacia la profundidad y aprietan todo el cielo en un círculo. en un
diamante, empujado hasta su punto de ruptura, se niega a dejar salir la luz ---
y los joyeros han tallado piedras en torno a esa negativa durante siglos. bajo
el Atlántico guía pulsos láser dentro de una hebra de vidrio y establece
el límite de velocidad de Internet global. Este problema funciona a través de la
tres vidas en orden, con la misma ley cada vez.

\medskip
\noindent\textbf{Parte I --- Warming up.}
\begin{enumerate}
  \item Calcula la velocidad de la luz en el agua ($n = 1.33$) y en el diamante.
        ($n = 2.42$). ¿Por qué factor el diamante ralentiza la luz?
  \item Un rayo en el aire impacta en un estanque en $i_1 = 40^\circ$. Calcular el
        ángulo de \omterm{def:g10:refraction:refraction}{refracción}.
  \item Un rayo en el agua golpea la superficie desde abajo en $28.9^\circ$.
        ¿En qué ángulo sale al aire? Establecer el principio
        esto ilustra.
  \item Calcular el \omterm{def:g10:refraction:critical}{ángulo crítico} del límite agua-aire, y
        describir lo que le sucede a un rayo submarino que golpea la superficie
        en $60^\circ$.
  \item El índice de aire es $1.0003$. Calcular la velocidad de la luz en el aire.
        y justificar la aproximación $n_{\text{air}} = 1.00$ utilizada
        en todas partes.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The pool.}
\begin{enumerate}[resume]
  \item Una piscina tiene una profundidad \qty{3.0}{m}. Usando la fórmula de profundidad aparente
        (\cref{prop:g10:refraction:depth}), calcula la profundidad de una persona
        estando arriba percibe.
  \item Explica con un dibujo de dos rayos (descrito con palabras) por qué una moneda
        en el fondo parece elevarse: ¿qué hacen los rayos que salen del agua?
        ¿Y dónde sitúa el ojo su intersección?
  \item El ojo de un buzo está en \qty{3.0}{m}. Calcular el radio de Snell.
        ventana --- el disco brillante a través del cual el buceador ve el
        cielo entero (\cref{exo:g10:refraction:12}).
  \item Una lamp en la parte inferior envía rayos hacia arriba en $30^\circ$,
        $45^\circ$ y $50^\circ$ desde la vertical. Para cada rayo, digamos
        si se escapa y en qué ángulo, o qué sucede en su lugar.
  \item Un pescador submarino en la orilla apunta a un pez. ¿Debería la huelga ser
        apunta arriba, hacia o debajo de la imagen del pez --- y ¿el
        pez, mirando hacia atrás, ¿ve al pescador desplazado también?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The diamond.}
\begin{enumerate}[resume]
  \item Calcule el \omterm{def:g10:refraction:critical}{ángulo crítico} del diamante ($n = 2.42$) hacia el aire,
        y compararlo con el de vidrio ($n = 1.52$),
        que es $41.1^\circ$.
  \item Un rayo dentro de la piedra golpea una faceta en $30^\circ$ desde su
        \omterm{def:g10:refraction:angles}{normal}. ¿Qué pasa en el diamante? ¿En la imitación de vidrio?
        Explique en una frase por qué la talla de un diamante, combinada con
        su diminuto \omterm{def:g10:refraction:critical}{ángulo crítico} produce el brillo.
  \item Diamond también es fuertemente \omterm{def:g10:refraction:dispersion}{dispersivo}: $n_{\text{red}} = 2.41$,
        $n_{\text{blue}} = 2.45$. \omterm{def:g10:light-spectra:white}{luz blanca} ingresa a una faceta en
        $45^\circ$; calcular el ángulo \omterm{def:g10:refraction:refraction}{refracción} para cada color y el
        división angular.
  \item Más tarde, los rayos rojo y azul alcanzan una faceta de salida desde el interior.
        en $17.0^\circ$. Calcule los dos ángulos de salida y la división de
        el aficionado emergente. Comparar con la división en la entrada: ¿qué pasa?
        a \omterm{def:g10:refraction:dispersion}{dispersión} cerca del \omterm{def:g10:refraction:critical}{ángulo crítico}?
  \item Una gota de agua ($n = 1.33$) recubre una faceta. Calcular el nuevo
        \omterm{def:g10:refraction:critical}{ángulo crítico} del límite diamante-agua, decide el destino
        del rayo $30^\circ$ de la pregunta 12, y explique por qué los joyeros
        Mantenga los diamantes escrupulosamente limpios.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The fiber.}
\begin{enumerate}[resume]
  \item Una fibra tiene un \omterm{def:g10:refraction:fiber}{centro} de índice $1.48$ y un \omterm{def:g10:refraction:fiber}{revestimiento} de índice
        $1.46$. Calcule el \omterm{def:g10:refraction:critical}{ángulo crítico} en el núcleo-revestimiento
        límite.
  \item Un hilo de vidrio desnudo en el aire tendría un \omterm{def:g10:refraction:critical}{ángulo crítico} de
        $42.5^\circ$ --- aparentemente mucho mejor. Da dos razones para
        \omterm{def:g10:refraction:fiber}{revestimiento} se utiliza de todos modos. (Piense en el polvo, los rayones y lo que
        tocar la superficie sería suficiente para \omterm{def:g10:refraction:critical}{reflexión interna total}.)
  \item Calcula la velocidad de la luz en el \omterm{def:g10:refraction:fiber}{centro}, luego el viaje en un sentido.
        tiempo de un pulso a lo largo de \qty{6000}{km} de fibra.
  \item Un rayo que rebota exactamente en el recorrido \omterm{def:g10:refraction:critical}{ángulo crítico}
        $1/\sin i_c$ veces la longitud del eje. Calcular el extra
        distancia sobre el \qty{6000}{km} y el tiempo extra que cuesta.
  \item El remate. Un par de fibras moderno transporta alrededor de bits $10^{13}$
        por segundo, y este libro trata sobre $2.4 \times 10^8$ bits.
        Calcule el tiempo de ida y vuelta ("ping") a través del océano y el
        número de copias de este libro la fibra se mueve por segundo ---
        luego indique, en una oración, qué hace \omterm{thm:g10:refraction:snell}{ley de snell} por el
        Internet.
\end{enumerate}
\end{problem}
