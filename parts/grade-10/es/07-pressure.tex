\chapter{Presión: del deporte al buceo}\label{ch:g10:pressure}

Un esquiador se desliza sobre la nieve en la que se hunde un caminante; un empujón \qty{5}{N}
clava una aguja a través del cuero que un puño no puede abollar. Mismo \omterm{def:g10:inertia:force}{efectivo},
efectos tremendamente diferentes: la cantidad que falta es \emph{\omterm{def:g10:pressure:pressure}{presión}},
\omterm{def:g10:inertia:force}{fuerza} dividido por el área que lo recibe. Este capítulo lo define,
lo sigue por una piscina, sube por un barómetro y entra en una jeringa exprimida\dots
y termina cuarenta \omterm{def:g10:orders-of-magnitude:unit}{metros} bajo el mar, donde se torna vital.

\section{Presión: fuerza repartida sobre un área.}

\begin{definition}[presión]\label{def:g10:pressure:pressure}
Cuando un \omterm{def:g10:inertia:force}{fuerza} de magnitud $F$ (en \unit{N}) presiona perpendicularmente sobre un
superficie del área $S$ (en \unit{m^2}), la \emph{presión}\index{presión}
ejercida sobre esa superficie es
\[
P = \frac{F}{S}.
\]
Su \omterm{def:g10:orders-of-magnitude:unit}{unidad}, el \unit{N/m^2}, es el \emph{pascal}\index{pascal}.
(\unit{Pa}) --- un \omterm{def:g10:orders-of-magnitude:unit}{unidad} diminuto, el \omterm{def:g10:universal-gravitation:weight}{peso} de una manzana repartido en un cuadrado
\omterm{def:g10:orders-of-magnitude:unit}{metro}, por lo que \unit{kPa} y \unit{MPa} son los \omterm{def:g10:orders-of-magnitude:unit}{unidades} en funcionamiento.
\end{definition}

\begin{example}[Tacón versus raqueta de nieve]\label{ex:g10:pressure:snowshoe}
Una persona \qty{60}{kg} pesa $F = 60 \times 9.81 \approx \qty{590}{N}$.
En un tacón de aguja de \qty{1.0}{cm^2} = \qty{1.0e-4}{m^2}:
$P = 590/\num{1.0e-4} \approx \qty{5.9e6}{Pa} = \qty{5.9}{MPa}$. en dos
raquetas de nieve de \qty{0.20}{m^2} cada una:
$P = 590/0.40 \approx \qty{1.5}{kPa}$. Las prensas \emph{mismo} \omterm{def:g10:universal-gravitation:weight}{peso}
$4000$ veces más duro debajo del talón: tacones marcados en suelos de madera,
Las raquetas de nieve flotan sobre la nieve en polvo.
\end{example}

\section{Presión en un líquido en reposo.}

¿Por qué el agua presiona más fuerte hacia abajo? Aislar una vertical imaginaria
columna de agua de sección $S$, desde la superficie hasta la profundidad $h$:
volumen $Sh$, masa $\rho S h$ ($\rho$ la densidad del líquido, en
\unit{kg/m^3}), \omterm{def:g10:universal-gravitation:weight}{peso} $\rho S h g$. La parte inferior de la columna soporta
que \omterm{def:g10:universal-gravitation:weight}{peso} \emph{más} la atmósfera empuja a $P_0 S$ hacia la cima.
Dividiendo el total \omterm{def:g10:inertia:force}{fuerza} por $S$: el \omterm{def:g10:pressure:pressure}{presión} en la profundidad $h$ debe ser
$P_0 + \rho g h$ --- y está, en todas partes del líquido.

\begin{theorem}[Presión en profundidad]\label{thm:g10:pressure:depth}
En un líquido de densidad $\rho$ en reposo, abierto al aire, el \omterm{def:g10:pressure:pressure}{presión} en
La profundidad $h$ debajo de la superficie es
\[
P = P_0 + \rho g h,
\]
donde $P_0$ es el \omterm{def:g10:pressure:atmospheric}{presión atmosférica} en la superficie. El \omterm{def:g10:pressure:pressure}{presión}
depende sólo de la profundidad, no de la forma del recipiente, y en un
En un punto dado, el líquido presiona con la misma fuerza en todas las direcciones.
\end{theorem}
\admitted

\begin{remark}
El argumento de la columna sólo hace que la ley sea plausible; la derivación honesta
(cualquier forma, todas las direcciones) está en el volumen del Año 1. el agua propia
término $\rho g h$ es el \emph{indicador}\index{presión manométrica} \omterm{def:g10:pressure:pressure}{presión} --- qué buzo
Se muestra \omterm{def:g10:pressure:pressure}{presión} \omterm{thm:g10:pressure:depth}{indicador}.
\end{remark}

\begin{example}[Diez metros de agua]\label{ex:g10:pressure:tenmetres}
En agua dulce ($\rho = \qty{1000}{kg/m^3}$), en $h = \qty{10}{m}$:
$\rho g h = 1000 \times 9.81 \times 10 \approx \qty{98}{kPa}$, entonces
$P = 101 + 98 \approx \qty{199}{kPa}$. Diez \omterm{def:g10:orders-of-magnitude:unit}{metros} de agua suman casi
exactamente una atmósfera: la regla general del buceador, \emph{one extra
atmosphere per ten \omterm{def:g10:orders-of-magnitude:unit}{metros}}.
\end{example}

\section{El océano de aire}

\begin{definition}[Presión atmosférica]\label{def:g10:pressure:atmospheric}
Vivimos en el fondo de un océano de aire, cuyo \omterm{def:g10:universal-gravitation:weight}{peso} crea el
\emph{presión atmosférica}\index{presión atmosférica}, al nivel del mar
\[
P_0 \approx \qty{101}{kPa} = \qty{1.01e5}{Pa}.
\]
En cada centímetro cuadrado esto es $F = \num{1.01e5} \times \num{1e-4}
\approx \qty{10}{N}$: el \omterm{def:g10:universal-gravitation:weight}{peso} de una masa \omterm{def:g10:orders-of-magnitude:unit}{un kilogramo} en cada
\unit{cm^2} de tu piel, de tu escritorio, de todo.
\end{definition}

\begin{example}[La carga invisible]\label{ex:g10:pressure:sheet}
Una hoja de papel A4 (\qty{21.0}{cm} $\times$ \qty{29.7}{cm}, por lo que
$S \approx \qty{0.0624}{m^2}$) recibe desde el aire sobre él
$F = \num{1.01e5} \times 0.0624 \approx \qty{6.3e3}{N}$ --- el \omterm{def:g10:universal-gravitation:weight}{peso}
de un coche pequeño. Nada se desgarra: el aire de abajo empuja
\emph{up} igual de duro. Notamos la atmósfera sólo cuando un lado
Lo pierde: una ventosa, una pajita, la cabina de un avión.
\end{example}

\begin{remark}[barómetro de Torricelli]\label{rem:g10:pressure:torricelli}
En 1643, Torricelli invirtió un tubo sellado de mercurio sobre un tubo de mercurio.
baño. La columna cayó hasta la altura a la que $\rho g h = P_0$:
$h = \num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$, saliendo
encima, el primer vacío jamás creado. La altura sigue el clima y
se reduce con la altitud: un barómetro pesa el aire sobre ti (agua
necesitaría \qty{10.3}{m}: \cref{exo:g10:pressure:9}).
\end{remark}

\section{Gases: bombardeo y ley de Boyle}

También un gas presiona su recipiente: es un enjambre de moléculas en
movimiento aleatorio incesante, cada colisión le da a la pared una pequeña
empuje, y miles de millones de impactos por segundo se desdibujan en el
constante \omterm{def:g10:inertia:force}{fuerza} medimos como gas \omterm{def:g10:pressure:pressure}{presión}. Exprime el gas hasta la mitad del
Volumen: las moléculas atacan el doble de veces, por lo que el \omterm{def:g10:pressure:pressure}{presión} debería
doble. Lo hace.

\begin{proposition}[ley de boyle]\label{prop:g10:pressure:boyle}
\index{ley de boyle}
Para una cantidad fija de gas mantenido a temperatura constante, \omterm{def:g10:pressure:pressure}{presión} y
el volumen es inversamente proporcional:
\[
P \, V = \text{constant},
\qquad\text{i.e.}\qquad
P_1 V_1 = P_2 V_2 .
\]
\end{proposition}
\admitted

\begin{remark}
Un hecho experimental a este nivel --- el volumen del Año 1 lo deriva
de la imagen de la colisión. El experimento (una jeringa sellada de aire,
un sensor \omterm{def:g10:pressure:pressure}{presión}, \qty{20}{\degreeCelsius}):
\begin{center}
\begin{tabular}{lccccc}
$V$ (\unit{mL}) & 60 & 50 & 40 & 30 & 20 \\
$P$ (\unit{kPa}) & 100 & 120 & 150 & 200 & 300 \\
$P \times V$ (\unit{kPa.mL}) & 6000 & 6000 & 6000 & 6000 & 6000
\end{tabular}
\end{center}
\end{remark}

\begin{example}[Una jeringa exprimida]\label{ex:g10:pressure:boyle}
El aire ocupa \qty{4.0}{L} en \qty{100}{kPa}; comprimido lentamente (para que el
temperatura permanece constante) a \qty{250}{kPa}, ocupa
$V_2 = P_1 V_1 / P_2 = 400/250 = \qty{1.6}{L}$.
\end{example}

\section{Buceo: presión puesta a trabajar}

\begin{method}[Presión absoluta en profundidad]\label{met:g10:pressure:dive}
Para un buzo en la profundidad $h$ en el mar ($\rho \approx \qty{1025}{kg/m^3}$,
entonces $\rho g h \approx \qty{101}{kPa}$ por \qty{10}{m}), calcule
$P = P_0 + \rho g h$ --- y en \omterm{prop:g10:pressure:boyle}{ley de boyle} siempre usa esto
\emph{absoluto}\index{presión absoluta} \omterm{def:g10:pressure:pressure}{presión}, nunca el término \omterm{thm:g10:pressure:depth}{indicador} solo. un buzo se sienta
bajo $2$ atmósferas en \qty{10}{m}, $3$ en \qty{20}{m}, $5$ en
\qty{40}{m}.
\end{method}

\begin{remark}[Nunca contiene la respiración]\label{rem:g10:pressure:breath}
Un regulador suministra aire a temperatura ambiente \omterm{def:g10:pressure:pressure}{presión}: los pulmones se llenan normalmente a
cualquier profundidad. Pero complete \qty{6.0}{L} en \qty{20}{m} ($\qty{302}{kPa}$),
contener la respiración y salir a la superficie: exige Boyle
$6.0 \times 302/101 \approx \qty{18}{L}$ --- tres veces que pecho
aguanta, y los pulmones se desgarran mucho antes de eso, incluso en unos pocos \omterm{def:g10:orders-of-magnitude:unit}{metros} de ascenso.
De ahí la primera regla del buceo: \emph{Nunca contengas la respiración mientras asciendes.}
--- sigue respirando y el aire en expansión simplemente sale. Un buzo libre
es seguro: su aire fue tomado en la superficie, por lo que en el camino hacia arriba solo
vuelve a expandirse a su volumen original.
\end{remark}

\begin{remark}[Descompresión]\label{rem:g10:pressure:deco}
Respirar aire \omterm{def:g10:pressure:pressure}{alta presión} también disuelve el exceso de nitrógeno en la sangre,
como gas en una botella de refresco tapada; emerge demasiado rápido y burbujea
burbujas dentro del cuerpo --- enfermedad de descompresión. De ahí la lentitud
ascenso (aproximadamente \qty{10}{m} por minuto) y paradas poco profundas, que permiten
El gas sale silenciosamente a través de los pulmones. La teoría espera al Año 1
volumen; la regla de conducta pertenece a cada buceador.
\end{remark}

  \item Para elevar el automóvil \qty{2.0}{cm}, ¿qué distancia debe alcanzar el pistón pequeño?
        ¿viajar? Compare el trabajo realizado en cada pistón.
\end{enumerate}
\end{exercise}

\section{Problema: una respiración hacia abajo, una respiración hacia arriba}

\begin{problem}[{Problema del fin de semana --- un suspiro hacia abajo, un respiro hacia arriba:
la física de una inmersión en apnea y un ascenso, desde apretar la máscara hasta
regla que los últimos diez metros son los más peligrosos}]
\label{pb:g10:pressure:1}
Inmersiones en apnea Lena: una respiración en la superficie, hasta \qty{30}{m} y viceversa.
Marco bucea junto a ella, respirando desde un tanque. Mismo mar
($\rho = \qty{1025}{kg/m^3}$, $g = \qty{9.81}{N/kg}$,
$P_0 = \qty{101}{kPa}$), mismos pulmones \qty{6.0}{L}, peligros opuestos ---
todo gobernado por $P = P_0 + \rho g h$ y $PV$ = constante (temperatura
constante en todo momento).

\medskip
\noindent\textbf{Parte I --- The \omterm{def:g10:universal-gravitation:weight}{peso} of water.}
\begin{enumerate}
  \item Calcule $\rho g$ para agua de mar en \unit{kPa} según \omterm{def:g10:orders-of-magnitude:unit}{metro}, y
        compruebe la regla general: una atmósfera por \qty{10}{m}.
  \item Calcule \omterm{met:g10:pressure:dive}{presión absoluta} en \qty{10}{m}, \qty{20}{m},
        \qty{30}{m} y \qty{40}{m}.
  \item Expresar cada uno como múltiplo de la superficie \omterm{def:g10:pressure:pressure}{presión}.
  \item Fundas de máscara de Lena \qty{150}{cm^2}. En \qty{30}{m}, si el aire
        dentro todavía estuviera en la superficie \omterm{def:g10:pressure:pressure}{presión}, ¿qué red \omterm{def:g10:inertia:force}{fuerza}
        el agua ejerce? (Los buzos exhalan dentro de la máscara por la nariz.
        para evitar esto \emph{apretón de máscara}.)
  \item Un tímpano tiene un área de aproximadamente \qty{0.60}{cm^2}. Calcular la red
        \omterm{def:g10:inertia:force}{fuerza} en \qty{3.0}{m} si el oído medio permanece en la superficie
        \omterm{def:g10:pressure:pressure}{presión}; ¿Por qué duelen los oídos en un simple estanque y qué se debe hacer?
        ¿Lograr la “igualación”?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- One breath down.}
Lena sale de la superficie con \qty{6.0}{L} de aire en sus pulmones.
\begin{enumerate}[resume]
  \item Usando \omterm{prop:g10:pressure:boyle}{ley de boyle} con \omterm{met:g10:pressure:dive}{presiones absolutas}, calcule su pulmón
        volumen en \qty{10}{m}, \qty{20}{m} y \qty{30}{m}.
  \item ¿A qué profundidad se reduce a la mitad su volumen pulmonar?
  \item ¿El volumen pulmonar es proporcional a la profundidad? Describe la curva de
        $V$ contra \omterm{met:g10:pressure:dive}{presión absoluta} $P$.
  \item Su volumen residual es \qty{1.5}{L}: los pulmones no pueden encogerse
        más lejos. Muestre que este límite se alcanza cerca de \qty{30}{m}.
  \item Los récords de buceo en apnea aún superan el \qty{100}{m}: ¿qué
        llena el volumen que falta? (¿Qué más puede fluir hacia el cofre?)
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- One breath up.}
En \qty{30}{m}, el regulador de Marco llena sus pulmones \qty{6.0}{L}.
\begin{enumerate}[resume]
  \item ¿A qué \omterm{def:g10:pressure:pressure}{presión} entrega ese aire el regulador? como
        ¿Cuántas veces es más denso que el aire de la superficie?
  \item Marco entra en pánico, contiene la respiración y sube a \qty{10}{m}:
        ¿Qué volumen demanda el aire atrapado allí?
  \item ¿Qué volumen hay en la superficie? ¿Cuántas veces su capacidad?
  \item Indique la primera regla del buzo y por qué respirar.
        normalmente elimina el peligro.
  \item Lena también contiene la respiración desde \qty{30}{m} hasta la superficie,
        sin embargo, sus pulmones están perfectamente a salvo. Explica la asimetría.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Bubbles, and the last ten \omterm{def:g10:orders-of-magnitude:unit}{metros}.}
\begin{enumerate}[resume]
  \item Marco lanza una burbuja \qty{1.0}{cm^3} en \qty{40}{m}.
        Calcule su volumen en \qty{30}{m}, \qty{20}{m}, \qty{10}{m}
        y justo debajo de la superficie.
  \item Para cada etapa \qty{10}{m} del ascenso, calcule el factor mediante
        que la burbuja crece. ¿Dónde es el crecimiento más rápido?
  \item Los buzos ascienden como máximo \qty{10}{m} por minuto: ¿cuánto tiempo desde
        \qty{40}{m}, y en qué minuto queda algún atrapado o disuelto
        ¿El gas se expande más?
  \item En profundidad, la sangre de Marco disuelve el exceso de nitrógeno, como gas en
        una botella de refresco tapada. ¿Qué hace un ascenso demasiado rápido y cómo?
        ¿Lo impiden el lento ascenso y una parada de seguridad \qty{5}{m}?
  \item Remate: en una frase cada uno, da la palabra del buceador libre.
        veredicto, el veredicto del buzo y la razón cuantificada
        Los últimos diez \omterm{def:g10:orders-of-magnitude:unit}{metros} de cualquier ascensión merecen el mayor respeto.
\end{enumerate}
\end{problem}
