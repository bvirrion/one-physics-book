\chapter{Forces and the Principle of Inertia}\label{ch:g10:inertia}

Stop pedaling and the bicycle coasts to a halt; stop pulling and the
sled stands still. Experience whispers that motion must be fed by an
effort --- and for two thousand years physics agreed. This chapter
builds the \omterm{def:g10:inertia:force}{force}, an arrow measured in \omterm{def:g10:inertia:force}{newtons}, and states the first
great law of mechanics: left truly alone, a moving body keeps its
motion, straight and steady, forever.

\section{Modeling an action as a force}

\begin{definition}[Force]\label{def:g10:inertia:force}
A \emph{force}\index{force} models a mechanical action (a push, a pull,
an attraction) exerted on a body. It is characterized by:
\begin{itemize}
  \item its \emph{point of application}\index{point of application};
  \item its direction --- the line along which it acts and the way
        along that line;
  \item its magnitude, measured in \emph{newtons}\index{newton}
        (symbol \unit{N}).
\end{itemize}
A force is drawn as an arrow $\vect F$ (a vector, as in the mathematics
volume) from its point of application, with length proportional to the
magnitude at a stated scale.
\end{definition}

\begin{example}[The weight of a schoolbag]\label{ex:g10:inertia:schoolbag}
A schoolbag of mass $m = \qty{4.0}{kg}$ is pulled downward by the Earth
with its \omterm{def:g10:universal-gravitation:weight}{weight} (\cref{ch:g10:universal-gravitation}):
$P = mg = 4.0 \times 9.81 \approx \qty{39}{N}$ --- one \omterm{def:g10:inertia:force}{newton} is about
an apple's \omterm{def:g10:universal-gravitation:weight}{weight}. At the scale $\qty{1}{cm} \leftrightarrow
\qty{10}{N}$: a vertical arrow \qty{3.9}{cm} long, applied at the
bag's center of gravity, pointing down.
\end{example}

\section{An inventory of forces}

\begin{definition}[Contact and distance forces]\label{def:g10:inertia:contact}
A \emph{contact force}\index{contact force} is exerted at the points
where two bodies touch. A \emph{force at a distance}\index{force at a
distance} needs no contact: \omterm{def:g10:universal-gravitation:weight}{weight} --- the gravitational pull of
\cref{ch:g10:universal-gravitation} --- is the only one used in this
chapter.
\end{definition}

\begin{definition}[The common contact forces]\label{def:g10:inertia:inventory}
Four \omterm{def:g10:inertia:contact}{contact forces} cover most everyday situations:
\begin{itemize}
  \item the \emph{normal reaction}\index{normal reaction} $\vect R$ of
        a support, perpendicular to its surface, pushing away from it;
  \item the \emph{tension}\index{tension} $\vect T$ of a wire, rope or
        chain, directed along it, pulling toward it;
  \item \emph{friction}\index{friction} $\vect f$ from a surface, the
        air or water, opposing the sliding or the motion;
  \item the \emph{thrust}\index{thrust} of an engine, jet or hand,
        pushing along the effort.
\end{itemize}
\end{definition}

\begin{example}[Inventory on a towed sled]\label{ex:g10:inertia:sled}
A sled towed across the snow: \omterm{def:g10:universal-gravitation:weight}{weight} $\vect P$ (distance), \omterm{def:g10:inertia:inventory}{normal
reaction} $\vect R$ of the snow, \omterm{def:g10:inertia:inventory}{tension} $\vect T$ of the rope, \omterm{def:g10:inertia:inventory}{friction}
$\vect f$ opposing the sliding. Nothing else touches the sled: the
inventory is complete --- \emph{\omterm{def:g10:universal-gravitation:weight}{weight} always, then one \omterm{def:g10:inertia:contact}{contact force}
per touching object}.
\end{example}

\section{The principle of inertia}

\begin{definition}[Straight uniform motion]\label{def:g10:inertia:uniform}
A body is in \emph{straight uniform motion}\index{straight uniform
motion} in a frame when its \omterm{def:g10:relative-motion:trajectory}{trajectory} is a straight line traveled at
constant speed: its velocity vector $\vect v$ is the same at every
instant. Rest is the special case $\vect v = \vect 0$.
\end{definition}

\begin{definition}[Compensating forces]\label{def:g10:inertia:compensate}
\omterm{def:g10:inertia:force}{Forces} acting on a body \emph{compensate}\index{compensating forces}
when their combined effect is nil: tip to tail, their arrows return to
the start. Two \omterm{def:g10:inertia:force}{forces} compensate exactly when they share the same line
of action with equal magnitudes and opposite ways.
\end{definition}

\begin{theorem}[Principle of inertia]\label{thm:g10:inertia:principle}
\index{principle of inertia}\index{inertia}
In the \omterm{def:g10:relative-motion:frames}{ground frame}, a body subject to no \omterm{def:g10:inertia:force}{force}, or to \omterm{def:g10:inertia:force}{forces} that
\omterm{def:g10:inertia:compensate}{compensate}, stays at rest or keeps a \omterm{def:g10:inertia:uniform}{straight uniform motion};
conversely, a body at rest or in \omterm{def:g10:inertia:uniform}{straight uniform motion} is subject
to \omterm{def:g10:inertia:compensate}{compensating forces} (or to none).
\end{theorem}
\admitted

\begin{remark}
No experiment can fully isolate a body, so the principle is a postulate
--- glimpsed by Galileo, stated by Newton. A later chapter upgrades it to a
law relating \omterm{def:g10:inertia:force}{forces} to \emph{changes} of velocity; the Year 1 volume
asks in which frames it actually holds.
\end{remark}

\begin{proposition}[Reading motion from forces, and back]\label{prop:g10:inertia:converse}
In the \omterm{def:g10:relative-motion:frames}{ground frame}:
\begin{enumerate}
  \item if the velocity of a body changes --- in magnitude \emph{or} in
        direction --- the \omterm{def:g10:inertia:force}{forces} on it do \emph{not} \omterm{def:g10:inertia:compensate}{compensate};
  \item if a body is at rest or in \omterm{def:g10:inertia:uniform}{straight uniform motion}, its
        inventoried \omterm{def:g10:inertia:force}{forces} must cancel --- determining unknowns.
\end{enumerate}
\end{proposition}

\begin{proof}
The first statement is the contrapositive of
\cref{thm:g10:inertia:principle}, the second its converse part applied
to a completed inventory.
\end{proof}

\begin{example}[The hockey puck]\label{ex:g10:inertia:puck}
Released by the stick, a puck crosses \qty{30}{m} of smooth ice in
\qty{2.5}{s}, straight at the constant $v = 30/2.5 = \qty{12}{m/s}$.
Inventory: \omterm{def:g10:universal-gravitation:weight}{weight} and \omterm{def:g10:inertia:inventory}{normal reaction} (\omterm{def:g10:inertia:inventory}{friction} negligible). They
\omterm{def:g10:inertia:compensate}{compensate}, and the principle predicts exactly the observed motion:
nothing pushes the puck forward --- and nothing needs to.
\end{example}

\begin{remark}[Galileo against Aristotle]
Aristotle taught that every motion needs a mover, and daily life ---
\omterm{def:g10:inertia:inventory}{friction} in every contact --- seems to agree. Galileo idealized the
\omterm{def:g10:inertia:inventory}{friction} away: ever smoother surfaces need ever less push to
\emph{maintain} motion. Persistence, not rest, is the natural state.
\end{remark}

\begin{remark}[The lurching passenger]
When a bus brakes sharply, standing passengers pitch forward. No \omterm{def:g10:inertia:force}{force}
throws them: by inertia their bodies keep their velocity while the bus
loses its own. The seat belt supplies the backward \omterm{def:g10:inertia:force}{force} the inventory
lacks.
\end{remark}

\section{Balanced forces: statics off a drawing}

\begin{definition}[Force diagram]\label{def:g10:inertia:diagram}
The \emph{force diagram}\index{force diagram} of a body shows the body
alone, with every \omterm{def:g10:inertia:force}{force} acting \emph{on} it drawn to scale from its
\omterm{def:g10:inertia:force}{point of application}; \omterm{def:g10:inertia:force}{forces} exerted \emph{by} the body never appear.
\end{definition}

\begin{method}[Analysing a static situation]\label{met:g10:inertia:method}
\begin{enumerate}
  \item Choose the body to study; work in the \omterm{def:g10:relative-motion:frames}{ground frame}.
  \item Inventory: \omterm{def:g10:universal-gravitation:weight}{weight} always, one \omterm{def:g10:inertia:contact}{contact force} per contact.
  \item Draw the \omterm{def:g10:inertia:diagram}{force diagram}.
  \item The body is at rest, so the \omterm{def:g10:inertia:force}{forces} \omterm{def:g10:inertia:compensate}{compensate}
        (\cref{thm:g10:inertia:principle}): read unknown magnitudes off
        the drawing (equal lengths in 1D, components in 2D).
\end{enumerate}
\end{method}

\begin{example}[The book on the table]\label{ex:g10:inertia:book}
A book of mass \qty{0.50}{kg} rests on a table: \omterm{def:g10:universal-gravitation:weight}{weight}
$P = 0.50 \times 9.81 \approx \qty{4.9}{N}$ and \omterm{def:g10:inertia:inventory}{normal reaction}
$\vect R$. At rest, $\vect R$ \omterm{def:g10:inertia:compensate}{compensates} $\vect P$: vertical, upward,
$R = \qty{4.9}{N}$. Add a second book and $R$ adjusts --- a support
pushes exactly as hard as needed.
\end{example}

\begin{example}[The lamp on two wires]\label{ex:g10:inertia:lamp}
A lamp of mass \qty{3.0}{kg} ($P \approx \qty{29.4}{N}$) hangs from two
wires, each at $45^\circ$ from the vertical. By symmetry the \omterm{def:g10:inertia:inventory}{tensions}
are equal ($T_1 = T_2 = T$) and their horizontal components cancel; the
vertical components, each $T \cos 45^\circ$, carry the \omterm{def:g10:universal-gravitation:weight}{weight}:
\[
2\,T \cos 45^\circ = P
\qquad\text{so}\qquad
T = \frac{29.4}{2 \times 0.707} \approx \qty{21}{N}
\]
--- more than half the \omterm{def:g10:universal-gravitation:weight}{weight} per wire: pulling at an angle costs
\omterm{def:g10:inertia:inventory}{tension}.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:inertia:1}
Inventory and classify (contact or distance) the \omterm{def:g10:inertia:force}{forces} on: (a) a book
at rest on a table; (b) a lamp hanging from its wire; (c) a puck
gliding on smooth ice; (d) an apple falling (air resistance
negligible).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:2}
Compute the \omterm{def:g10:universal-gravitation:weight}{weight} of a \qty{250}{g} book and of a \qty{60}{kg}
person. At the scale $\qty{1}{cm} \leftrightarrow \qty{1}{N}$, how
long is the book's arrow, and why is the scale hopeless for the
person?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:3}
On a diagram at the scale $\qty{1}{cm} \leftrightarrow \qty{2}{N}$, a
\omterm{def:g10:inertia:force}{force} is a horizontal arrow \qty{3.5}{cm} long pointing left, applied
at a point $A$. Give its three characteristics.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:4}
A dictionary of mass \qty{1.2}{kg} lies on a shelf. Draw its \omterm{def:g10:inertia:diagram}{force
diagram} and give the magnitude and direction of each \omterm{def:g10:inertia:force}{force}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:5}
True or false --- correct the false ones: (a) a body always ends up
stopping unless a \omterm{def:g10:inertia:force}{force} keeps pushing it; (b) \omterm{def:g10:inertia:uniform}{straight uniform motion}
requires no uncompensated \omterm{def:g10:inertia:force}{force}; (c) a body at rest is subject to no
\omterm{def:g10:inertia:force}{force}; (d) if a body's velocity changes, the \omterm{def:g10:inertia:force}{forces} on it do not
\omterm{def:g10:inertia:compensate}{compensate}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:6}
After release, a puck covers \qty{24}{m} of ice in \qty{3.0}{s},
straight at constant speed. Compute its speed; do the \omterm{def:g10:inertia:force}{forces} on it
\omterm{def:g10:inertia:compensate}{compensate}? On concrete the same puck stops within \omterm{def:g10:orders-of-magnitude:unit}{metres}: what changed
in the inventory, and what does \cref{prop:g10:inertia:converse}
conclude?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:7}
A \qty{70}{kg} passenger stands in an elevator rising at the constant
\qty{1.5}{m/s}. List the \omterm{def:g10:inertia:force}{forces} on the passenger; do they \omterm{def:g10:inertia:compensate}{compensate}?
Give the magnitude of the floor's \omterm{def:g10:inertia:force}{force}. Would anything change with
the elevator at rest?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:8}
A lamp of mass \qty{1.8}{kg} hangs at rest from a single vertical wire.
Compute the \omterm{def:g10:inertia:inventory}{tension} of the wire. The wire is replaced by two

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:9}
A bus brakes sharply and a standing passenger lurches forward; in a

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:10}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:11}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:12}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:13}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:15}

direction.
\end{exercise}

\section{Problem: Inertia in three acts}

\begin{problem}[{Weekend problem --- the airport walkway, the ice rink
and the parachute: constant velocity as the signature of compensated
forces, and why a parachute does not reduce the drag but the speed}]
\label{pb:g10:inertia:1}
Three scenes --- a suitcase on a moving walkway, a puck on an ice rink,
a parachutist in the evening air --- and one law talking in all three.

\medskip
\noindent\textbf{Part I --- The airport walkway.}
A \qty{23}{kg} suitcase stands on a walkway moving at the constant
speed \qty{0.75}{m/s}.
\begin{enumerate}
  \item Ground-frame motion of the suitcase? Inventory the \omterm{def:g10:inertia:force}{forces} on it.
  \item Do these \omterm{def:g10:inertia:force}{forces} \omterm{def:g10:inertia:compensate}{compensate}? Justify; compute both magnitudes.
  \item Compare its \omterm{def:g10:inertia:diagram}{force diagram} with the same suitcase's on the
        floor. What deep statement does the comparison illustrate?
  \item The walkway jerks to a halt. Describe what the suitcase does,
        without inventing a forward \omterm{def:g10:inertia:force}{force}.
  \item Its owner walks at \qty{1.2}{m/s} relative to the walkway. Her
        ground-frame speed? Do the \omterm{def:g10:inertia:force}{forces} on \emph{her} \omterm{def:g10:inertia:compensate}{compensate}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The ice rink.}
\begin{enumerate}[resume]
  \item A \qty{160}{g} puck glides straight at the constant
        \qty{8.0}{m/s}. Draw its \omterm{def:g10:inertia:diagram}{force diagram} with magnitudes.
  \item Aristotle claims the puck needs a mover. What does the rink
        reply, and why do everyday surfaces seem to side with him?
  \item The puck crosses a rough patch and slows. Conclusion about the
        \omterm{def:g10:inertia:force}{forces} there? Which \omterm{def:g10:inertia:force}{force} is the culprit?
  \item A skater curves smoothly at constant speed. Do the \omterm{def:g10:inertia:force}{forces} on
        her \omterm{def:g10:inertia:compensate}{compensate}? Careful: what kind of quantity is velocity?
  \item Match diagram to motion: (a) compensated \omterm{def:g10:inertia:force}{forces}; (b) an extra
        \omterm{def:g10:inertia:force}{force} opposite $\vect v$; (c) an extra \omterm{def:g10:inertia:force}{force} perpendicular to
        $\vect v$ --- (i) slowing straight; (ii) turning at constant
        speed; (iii) \omterm{def:g10:inertia:uniform}{straight uniform motion}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The parachute.}
A jumper's total mass, gear included, is \qty{80}{kg}.
\begin{enumerate}[resume]
  \item Leaving the plane, vertical speed and drag are still zero.
        Compute the \omterm{def:g10:inertia:force}{forces}; do they \omterm{def:g10:inertia:compensate}{compensate}? What must happen next?
  \item Drag grows with speed. Describe the fall while drag $<$ \omterm{def:g10:universal-gravitation:weight}{weight}.
  \item Before the canopy opens, the jumper reaches a steady
        \qty{50}{m/s} (``terminal speed''). Compute the drag. What is
        the motion, and why is it ``inertia in disguise''?
  \item The canopy opens: the drag now far exceeds the \omterm{def:g10:universal-gravitation:weight}{weight}. What
        happens to the velocity? (The jumper moves \emph{down}
