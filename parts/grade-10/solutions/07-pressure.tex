\section*{Chapter \ref{ch:g10:pressure} --- Pressure: From Sport to Diving}

\begin{solution}{exo:g10:pressure:1}
Large face: $S = \qty{0.15}{m^2}$, $P = 600/0.15 = \qty{4.0}{kPa}$.
Small face: $S = \qty{0.060}{m^2}$, $P = \qty{10}{kPa}$. On sand, the
large face: lower pressure, less sinking.
\end{solution}

\begin{solution}{exo:g10:pressure:2}
$\qty{0.25}{MPa} = \qty{250}{kPa} = \qty{2.5e5}{Pa}$.
$3.0/\num{2.0e-4} = \qty{15}{kPa}$ beats
$60/0.050 = \qty{1.2}{kPa}$: the small force presses harder.
\end{solution}

\begin{solution}{exo:g10:pressure:3}
$\rho g h = 1000 \times 9.81 \times 3.0 \approx \qty{29}{kPa}$;
absolute $P = 101 + 29 = \qty{130}{kPa}$.
\end{solution}

\begin{solution}{exo:g10:pressure:4}
$V_2 = 100 \times 4.0 / 250 = \qty{1.6}{L}$;
at \qty{50}{kPa}, $V_2 = 400/50 = \qty{8.0}{L}$.
\end{solution}

\begin{solution}{exo:g10:pressure:5}
$F = 65 \times 9.81 \approx \qty{638}{N}$. Blade:
$638/\num{1.2e-3} \approx \qty{5.3e5}{Pa} = \qty{530}{kPa}$. Soles:
$638/0.036 \approx \qty{18}{kPa}$. Ratio $\approx 30$: the blade bites
into the ice, the soles do not.
\end{solution}

\begin{solution}{exo:g10:pressure:6}
Thumb: $20/\num{1.2e-4} \approx \qty{1.7e5}{Pa} = \qty{170}{kPa}$.
Wall: $20/\num{1.0e-8} = \qty{2.0e9}{Pa}$. The same \qty{20}{N} acts on
an area $12\,000$ times smaller at the tip, so only there does the
pressure exceed what the material can withstand.
\end{solution}

\begin{solution}{exo:g10:pressure:7}
$\Delta P = \rho g h = 1025 \times 9.81 \times 30 \approx
\qty{3.0e5}{Pa}$, so
$F = \num{3.0e5} \times 0.50 \approx \qty{1.5e5}{N}$ --- the weight of
$F/g \approx \qty{1.5e4}{kg}$, fifteen tonnes. No crew opens that
outward against the sea.
\end{solution}

\begin{solution}{exo:g10:pressure:8}
$\rho = \dfrac{P}{g h} = \dfrac{\num{66.2e3}}{9.81 \times 5.00}
\approx \qty{1350}{kg/m^3}$. Not fresh water ($\qty{1000}{kg/m^3}$):
a markedly denser liquid, e.g.\ a concentrated brine.
\end{solution}

\begin{solution}{exo:g10:pressure:9}
$h = P_0/(\rho g)$: water
$\num{1.01e5}/9810 \approx \qty{10.3}{m}$; mercury
$\num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$. A
\qty{76}{cm} tube fits on a desk; a \qty{10}{m} water barometer needs
a stairwell --- mercury's density won.
\end{solution}

\begin{solution}{exo:g10:pressure:10}
$V = 6000/240 = \qty{25}{mL}$; $P = 6000/75 = \qty{80}{kPa}$.
$P$ against $V$ is a hyperbola; $P$ against $1/V$ is a straight line
through the origin, of slope \qty{6000}{kPa.mL}.
\end{solution}

\begin{solution}{exo:g10:pressure:11}
At \qty{20}{m}: $P = 101 + 2 \times 101 = \qty{303}{kPa}$, three
atmospheres. At the surface,
$V = 0.50 \times 303/101 = \qty{1.5}{cm^3}$: the bubble triples.
\end{solution}

\begin{solution}{exo:g10:pressure:12}
\emph{1.} $F = \num{1.01e5} \times \num{1.2e-3} \approx \qty{121}{N}$,
holding $m = 121/9.81 \approx \qty{12}{kg}$.

\emph{2.} $\Delta P = 101 - 20 = \qty{81}{kPa}$:
$F = \num{8.1e4} \times \num{1.2e-3} \approx \qty{97}{N}$, about
\qty{9.9}{kg}.
\end{solution}

\begin{solution}{exo:g10:pressure:13}
Boyle: $12 \times (200 - 50) = \qty{1800}{L}$ of surface-pressure air
is usable. At \qty{30}{m} each breath is drawn at \qty{4}{bar}, so
\qty{15}{L/min} at ambient pressure costs
$15 \times 4 = \qty{60}{L/min}$ of surface air: $1800/60 =
\qty{30}{min}$. At \qty{10}{m} ($\qty{2}{bar}$): \qty{30}{L/min},
so \qty{60}{min}. Depth is paid for in air.
\end{solution}

\begin{solution}{exo:g10:pressure:14}
$P = P_0 \dfrac{V_0}{V} = 101 \times \dfrac{6.0}{1.5} =
\qty{404}{kPa}$. With $\rho g \approx \qty{10.1}{kPa/m}$,
$h = (404 - 101)/10.1 \approx \qty{30}{m}$.
\end{solution}

\begin{solution}{exo:g10:pressure:15}
\emph{1.} Weight $F = 1200 \times 9.81 \approx \qty{1.18e4}{N}$;
pressure $P = F/S = \num{1.18e4}/0.040 \approx \qty{2.9e5}{Pa}$; small
piston: $f = P s = \num{2.9e5} \times \num{2.0e-4} \approx \qty{59}{N}$
--- the force is divided by $S/s = 200$.

\emph{2.} Liquid volume is conserved:
$400 \times 2.0 = \qty{800}{cm^3}$ must come from the small cylinder,
which travels $800/2.0 = \qty{400}{cm} = \qty{4.0}{m}$. Work:
$59 \times 4.0 \approx \qty{235}{J}$ on one side,
$\num{1.18e4} \times 0.020 \approx \qty{235}{J}$ on the other --- the
lift trades distance for force, never work.
\end{solution}

\begin{solution}{pb:g10:pressure:1}
\textbf{1.} $\rho g = 1025 \times 9.81 \approx \qty{10.1}{kPa}$ per
metre, so \qty{10}{m} of sea water add $\approx \qty{101}{kPa}$: one
atmosphere per ten metres, almost exactly.

\textbf{2.} $P = 101 + 10.1\,h$: \qty{202}{kPa} at \qty{10}{m},
\qty{302}{kPa} at \qty{20}{m}, \qty{403}{kPa} at \qty{30}{m},
\qty{503}{kPa} at \qty{40}{m}.

\textbf{3.} $2.0$, $3.0$, $4.0$ and $5.0$ times $P_0$.

\textbf{4.} $\Delta P = 403 - 101 = \qty{302}{kPa}$ over
$\qty{0.015}{m^2}$: $F = \num{3.02e5} \times 0.015 \approx
\qty{4.5e3}{N}$ --- the weight of nearly half a tonne on the face.

\textbf{5.} $\Delta P = 10.1 \times 3.0 \approx \qty{30}{kPa}$;
$F = \num{3.0e4} \times \num{6.0e-5} \approx \qty{1.8}{N}$ --- a
finger pressed on the eardrum, at pool depth already. Equalizing pushes
air into the middle ear until the inside pressure matches the water's.

\textbf{6.} $V = 6.0 \, P_0/P$: $6.0 \times 101/202 = \qty{3.0}{L}$;
$606/302 = \qty{2.0}{L}$; $606/403 = \qty{1.5}{L}$.

\textbf{7.} Halved when $P = 2P_0$, i.e.\ at \qty{10}{m} --- in the
very first stretch of the dive.

\textbf{8.} No: $V = 606/P$ is a hyperbola in the absolute pressure
(and $P$, not $h$, is what Boyle's law sees); each extra
\qty{10}{m} removes less volume than the previous one.

\textbf{9.} $V = \qty{1.5}{L}$ requires
$P = 101 \times 6.0/1.5 = \qty{404}{kPa}$, i.e.\
$h = 303/10.1 \approx \qty{30}{m}$ (\cref{exo:g10:pressure:14}).

\textbf{10.} Blood: plasma shifts into the vessels of the chest,
incompressible, and occupies the volume the air no longer fills.

\textbf{11.} At ambient pressure, \qty{403}{kPa}. By Boyle the same
air at \qty{101}{kPa} would fill $4$ times the volume: it is $4.0$
times denser than surface air.

\textbf{12.} $V = 6.0 \times 403/202 \approx \qty{12}{L}$ --- double,
after only \qty{20}{m} of rise.

\textbf{13.} $V = 6.0 \times 403/101 \approx \qty{24}{L}$: four times
his lung capacity.

\textbf{14.} \emph{Never hold your breath while ascending.} With the
airway open, the expanding air flows out through the regulator, and
lung volume never exceeds \qty{6.0}{L}.

\textbf{15.} Lena's \qty{1.5}{L} at \qty{30}{m} \emph{is} her surface
\qty{6.0}{L} compressed: on the way up it re-expands to exactly
\qty{6.0}{L}, never beyond. Marco's \qty{6.0}{L} were \emph{taken} at
\qty{403}{kPa}: they have \qty{24}{L} of surface air in them.

\textbf{16.} $V = 1.0 \times 503/P$: \qty{1.25}{cm^3} at \qty{30}{m},
\qty{1.67}{cm^3} at \qty{20}{m}, \qty{2.49}{cm^3} at \qty{10}{m},
\qty{4.98}{cm^3} at the surface.

\textbf{17.} Stage factors: $503/403 = 1.25$, then $1.33$, $1.49$,
and $202/101 = 2.0$ for the last ten metres --- growth accelerates as
the surface nears.

\textbf{18.} $40/10 = \qty{4.0}{min}$. The last minute: from
\qty{10}{m} to the surface the absolute pressure halves, the biggest
relative drop of the whole ascent.

\textbf{19.} A fast ascent lets the dissolved nitrogen fizz into
bubbles inside blood and joints (decompression sickness); ascending
slowly and pausing near \qty{5}{m} keeps the gas dissolved long enough
to leave quietly through the lungs.

\textbf{20.} Free diver: her air only returns to its original volume
--- Boyle protects her. Scuba diver: air taken at depth multiplies on
the way up --- breathe, never hold. And the last ten metres double
every trapped volume ($\times 2.0$, against $\times 1.25$ down at
\qty{40}{m}): the closer the surface, the slower you should approach
it.
\end{solution}
