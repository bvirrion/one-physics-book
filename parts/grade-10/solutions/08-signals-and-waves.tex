\begin{solution}{exo:g10:signals-and-waves:1}
$f = 1/T$: (a) $1/\qty{0.020}{s} = \qty{50}{Hz}$;
(b) $1/\qty{4.0e-3}{s} = \qty{250}{Hz}$;
(c) $1/\qty{5.0e-5}{s} = \qty{20}{kHz}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:2}
$T = 1/f$: (a) \qty{20}{ms}; (b) $1/440 = \qty{2.3}{ms}$;
(c) $1/\qty{2.4e9}{Hz} = \qty{0.42}{ns}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:3}

\end{solution}

\begin{solution}{exo:g10:signals-and-waves:5}
$d = 340 \times 6.0 = \qty{2040}{m} \approx \qty{2.0}{km}$. Light covers
this in $2040/(3.00\times10^{8}) \approx \qty{7}{\micro s}$, a million
times less than \qty{6.0}{s}: the flash marks the instant of the strike.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:6}
The sound goes down \emph{and back}, covering $2d$:
$d = \frac{1500 \times 0.60}{2} = \qty{450}{m}$. Over the trench:
$t = 2d/v = 2 \times 1200/1500 = \qty{1.6}{s}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:8}
$T = \qty{20}{mm}/\qty{25}{mm/s} = \qty{0.80}{s}$, so
$f = \qty{1.25}{Hz}$: $75$ beats per minute. For $100$ beats per minute,
$T = \qty{0.60}{s}$, i.e.\ $25 \times 0.60 = \qty{15}{mm}$: tachycardia

\end{solution}

\begin{solution}{exo:g10:signals-and-waves:9}
$T = 1/200 = \qty{5.0}{ms}$; two cycles last \qty{10}{ms}, spread over
$10$ divisions: time base \qty{1}{ms} per division. The crest must fit
in $4$ divisions: gain at least $3.0/4 = \qty{0.75}{V}$ per division ---
take \qty{1}{V} per division (crest $3.0$ divisions up).
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:11}

\begin{solution}{exo:g10:signals-and-waves:12}
While emitting for \qty{3.0}{ms} the bat is deaf to echoes, i.e.\ to
anything within $d = \frac{340 \times 3.0\times10^{-3}}{2} =
\qty{0.51}{m}$. Moth at \qty{1.7}{m}:
$t = 2 \times 1.7/340 = \qty{10}{ms}$. Closing in, the echo returns ever
sooner; the pulse must end before it arrives, so it must shorten.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:13}
$d = \frac{340 \times 1.5}{2} = \qty{255}{m}$. At \qty{155}{m}:
$t = 2 \times 155/340 = \qty{0.91}{s}$. Echo and clap merge when
$t < \qty{0.1}{s}$, i.e.\ within
$d < \frac{340 \times 0.1}{2} = \qty{17}{m}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:14}
\emph{1.} $T = 3.0 \times \qty{5}{ms} = \qty{15}{ms}$,
$f = 1/0.015 \approx \qty{67}{Hz}$.
\emph{2.} $\qty{15}{ms}/\qty{2}{ms} = 7.5$ divisions.
\emph{3.} The screen shows \qty{50}{ms}, then \qty{20}{ms}: $3$ complete
cycles ($50/15 = 3.3$), then $1$ ($20/15 = 1.3$).
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:15}
\emph{1.} $t = 2 \times 1.2\times10^{6}/(3.00\times10^{8}) =
\qty{8.0}{ms}$.
\emph{2.} $2 \times 1.2\times10^{6}/(2.0\times10^{8}) = \qty{12}{ms}$.

\end{solution}

\begin{solution}{pb:g10:signals-and-waves:1}
\textbf{1.} $d = 340 \times 9.0 = \qty{3060}{m} \approx \qty{3.1}{km}$.

\textbf{2.} $3060/(3.00\times10^{8}) \approx \qty{10}{\micro s}$, a
million times shorter than \qty{9.0}{s}: the flash marks the instant of
the strike.

\textbf{3.} In \qty{3}{s} sound covers $340 \times 3 = \qty{1020}{m}
\approx \qty{1}{km}$: seconds divided by three gives kilometres.

\textbf{4.} $d = 340 \times 6.0 = \qty{2040}{m}$. The storm closed
\qty{1020}{m} in \qty{300}{s}: $v = \qty{3.4}{m/s} \approx
\qty{12}{km/h}$, heading for the boat.

\textbf{5.} The near end is heard after $2000/340 = \qty{5.9}{s}$, the
far end after $5000/340 = \qty{14.7}{s}$: the rumble lasts about
\qty{8.8}{s}.

\textbf{6.} $d = 1500 \times 0.90/2 = \qty{675}{m}$.

\textbf{7.} $d = 1500 \times 0.20/2 = \qty{150}{m}$.

\textbf{8.} Two obstacles: a fish shoal at $1500 \times 0.16/2 =
\qty{120}{m}$ and the seabed at \qty{150}{m}. The shoal swims
\qty{30}{m} above the bottom.

\textbf{9.} The echo must return before the next ping:
$d_{\max} = 1500 \times 0.50/2 = \qty{375}{m}$. From deeper water the
echo arrives \emph{after} the next ping and is attributed to it: the
display shows a false, far too shallow bottom.

\textbf{10.} In air, $340 \times 0.60/2 = \qty{102}{m}$ instead of
\qty{450}{m}. A delay becomes a distance only once you know the carrier
and its speed in the medium crossed.

\textbf{11.} $f = 1/0.86 = \qty{1.16}{Hz}$, i.e.\ $1.16 \times 60
\approx 70$ beats per minute.

\textbf{12.} $25 \times 0.86 = \qty{21.5}{mm}$.

\textbf{13.} $T = 15/25 = \qty{0.60}{s}$: $100$ beats per minute.

\textbf{14.} $T = 60/50 = \qty{1.2}{s}$; spacing
$25 \times 1.2 = \qty{30}{mm}$.

\textbf{15.} No: the period drifts from beat to beat with effort,
breathing and stress. The doctor reads the period (the rate), its
regularity, and the shape of each cycle --- the diagnosis is in the
signal.

\textbf{16.} Radio: $5.0\times10^{4}/(3.00\times10^{8}) =
\qty{0.17}{ms}$ --- instantaneous to human senses. Sound:
$50\,000/340 \approx \qty{147}{s}$, two and a half minutes.

\textbf{17.} $t = 2.02\times10^{7}/(3.00\times10^{8}) = \qty{67}{ms}$.

\textbf{18.} $c \times \qty{1.0}{\micro s} = \qty{300}{m}$. For a
\qty{3.0}{m} fix: $3.0/(3.00\times10^{8}) = \qty{10}{ns}$ --- which is
why GPS satellites carry atomic clocks.

\textbf{19.} Satellites sit in vacuum, and sound needs a medium: no
signal would leave the satellite at all. (Even granting air the whole
way, \qty{20200}{km} at \qty{340}{m/s} is over \qty{16}{h} --- a
position fix hours out of date.)

\textbf{20.} \qty{340}{m/s} (sound in air), \qty{1500}{m/s} (sound in
seawater) and $c = \qty{3.00e8}{m/s}$ (radio and light); the law is
$d = v\,t$, halved for an echo. The habit: know your carrier, know its
speed in the medium, time the delay --- a delay is a distance in
disguise.
\end{solution}
