\section*{Chapter \ref{ch:g10:energy-conservation} --- Energy: Forms and Conservation}

\begin{solution}{exo:g10:energy-conservation:1}
$E_k = \tfrac12 \times 1200 \times 25^2 = \qty{3.75e5}{J}$. At
\qty{36.1}{m/s}: $\tfrac12 \times 1200 \times 36.1^2 \approx
\qty{7.8e5}{J}$ --- $44\%$ more speed, $(130/90)^2 \approx 2.1$
times the \omterm{def:g10:energy-conservation:kinetic}{kinetic energy}: the square at work.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:2}
$\Delta h = \qty{450}{m}$:
$E_p = 72 \times 9.81 \times 450 \approx \qty{3.2e5}{J}
= \num{3.2e5}/\num{3.6e6} \approx \qty{0.088}{kW.h}$, worth about
$0.022$ euros. A day's climbing is two cents of electricity.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:3}
$\qty{1}{kW.h} = \qty{3.6e6}{J}$;
$\qty{12}{W.h} = 12 \times \qty{3600}{J} \approx \qty{43}{kJ}$;
$\qty{45}{MJ} = 45/3.6 = \qty{12.5}{kW.h}$;
$\qty{10}{MJ} = 10/3.6 \approx \qty{2.8}{kW.h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:4}
$E = 2200 \times 150 = \qty{3.3e5}{J}
= \num{3.3e5}/\num{3.6e6} \approx \qty{0.092}{kW.h}$. Human:
$P = \num{1.0e7}/\num{86400} \approx \qty{116}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:5}
Dynamo: chemical (cyclist's food) $\to$ kinetic (wheel) $\to$
electrical (dynamo) $\to$ radiative $+$ thermal (lamp); heat leaks
at muscles, tire contact, dynamo, lamp. Gas heater: chemical (gas)
$\to$ thermal (water), leak in the flue gases. Panel: radiative
(Sun) $\to$ electrical (panel) $\to$ chemical (battery), heat leaks
at panel and charger.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:6}
Consumed $2.0 \times 45 = \qty{90}{MJ}$; $\eta = 27/90 = 0.30$.
Heat: $90 - 27 = \qty{63}{MJ}$, carried off by the exhaust gases
and the radiator into the surrounding air.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:7}
$E_k = E_p = 70 \times 9.81 \times 20 \approx \qty{1.37e4}{J}$;
$v = \sqrt{2 \times 13734/70} = \sqrt{392} \approx \qty{19.8}{m/s}
\approx \qty{71}{km/h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:8}
$41562 - 41236 = \qty{326}{kW.h}$; daily $326/30 \approx
\qty{10.9}{kW.h}$. Bill: $326 \times 0.26 + 12 = 96.76$ euros.
Average \omterm{def:g10:energy-conservation:power}{power}: $\qty{326}{kW.h}/\qty{720}{h} \approx \qty{453}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:9}
\emph{1.} Light: $0.05 \times 60 = \qty{3.0}{W}$; LED:
$\eta = 3/9 \approx 0.33$.
\emph{2.} Yearly ($\qty{1460}{h}$): bulb
$60 \times 1460 = \qty{87.6}{kW.h}$, LED $\qty{13.1}{kW.h}$;
saving $\qty{74.5}{kW.h} \approx 19$ euros per bulb per year.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:10}
\emph{1.} $P = \num{1.5e5} \times 9.81 \times 40 \approx
\qty{5.9e7}{W} = \qty{59}{MW}$.
\emph{2.} $0.90 \times 59 \approx \qty{53}{MW}$;
$\num{5.3e7}/525 \approx \num{1.0e5}$ households.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:11}
$v = \sqrt{2 \times 9.81 \times 0.050} \approx \qty{0.99}{m/s}$
(the mass cancels: $\tfrac12 m v^2 = mgh$). The initial
$E_p = 0.20 \times 9.81 \times 0.050 \approx \qty{0.098}{J}$ ends
as \omterm{def:g10:energy-conservation:forms}{thermal energy} in the air and the pivot: the pendulum is not
isolated, and the total --- swing plus heat --- is exactly
conserved.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:12}
\emph{1.} $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.304$;
consumed $12/0.304 \approx \qty{39.5}{W.h}
= 39.5 \times 3600 \approx \qty{142}{kJ}$.
\emph{2.} $\num{1.42e5}/\num{2.5e7} \approx \qty{5.7}{g}$ of coal
per charge; $\times 365 \approx \qty{2.1}{kg}$ per year.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:13}
\emph{1.} $\num{5e9}/\num{3.6e6} \approx \qty{1400}{kW.h}$;
$1400/12.6 \approx 110$ days.
\emph{2.} $P = \num{5e9}/\num{1e-4} = \qty{5e13}{W} = \qty{50}{TW}$
--- seventeen times humanity's average electric \omterm{def:g10:energy-conservation:power}{power}, for
\qty{100}{\micro s}, at an unpredictable place. Lightning is much
\omterm{def:g10:energy-conservation:power}{power} but little \omterm{def:g10:energy-conservation:energy}{energy}, delivered uselessly fast: no \omterm{def:g10:energy-conservation:transfer}{converter}
accepts such a spike.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:14}
\emph{1.} $E_k = \tfrac12 \times 1300 \times 30.6^2 \approx
\qty{6.1e5}{J}$, converted to \omterm{def:g10:energy-conservation:forms}{thermal energy} in the brake discs
and pads (then the air).
\emph{2.} $0.60 \times \qty{6.1e5}{J} \approx \qty{3.7e5}{J}$ per
stop; $\num{3.6e6}/\num{3.7e5} \approx 10$ stops per \omterm{def:g10:energy-conservation:kwh}{kilowatt-hour}.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:15}
$E_p = 85 \times 9.81 \times 1200 \approx \qty{1.0}{MJ}$; food:
$1.0/0.25 = \qty{4.0}{MJ}$, i.e.\ $4.0/1.5 \approx 2.7$ cereal
bowls --- or $4.0/45 \approx \qty{0.09}{kg}$ of gasoline. A whole
morning of honest effort fits in a coffee cup of fuel: \omterm{def:g10:energy-conservation:forms}{chemical
energy} is absurdly concentrated.
\end{solution}

\begin{solution}{pb:g10:energy-conservation:1}
\textbf{1.} Fridge $0.1 \times 24 = \qty{2.4}{kW.h}$; heater
$2.4 \times 2.5 = \qty{6.0}{kW.h}$; oven \qty{2.0}{kW.h}; washing
machine \qty{1.0}{kW.h}; lighting \qty{0.3}{kW.h}; screens
\qty{0.9}{kW.h}.
\textbf{2.} Total \qty{12.6}{kW.h} $= 12.6 \times 3.6 \approx
\qty{45}{MJ}$: one \omterm{def:g10:orders-of-magnitude:unit}{kilogram} of gasoline, to the \omterm{def:g10:energy-conservation:energy}{joule}.
\textbf{3.} $\qty{12600}{W.h}/\qty{24}{h} = \qty{525}{W}$ --- the
house idles at a quarter of a kettle.
\textbf{4.} The water heater (\qty{6.0}{kW.h}, nearly half). At
\qty{1.25}{h}: heater \qty{3.0}{kW.h}, total \qty{9.6}{kW.h}.
\textbf{5.} $15 \times 24 = \qty{0.36}{kW.h}$ per day, about
$2.8\%$ of the metered total; yearly $0.36 \times 365 \approx
\qty{131}{kW.h} \approx 33$ euros --- for lights nobody watches.
\textbf{6.} $56604 - 56214 = \qty{390}{kW.h}$;
$390/30 = \qty{13.0}{kW.h}$ per day: the audit's \qty{12.6}{kW.h}
plus the \qty{0.36}{kW.h} of standby --- confirmed.
\textbf{7.} $390 \times 0.25 + 14 = 111.50$ euros.
\textbf{8.} Grid: $0.25/3.6 \approx 0.07$ euros per megajoule.
Gasoline: $1.80/34 \approx 0.05$ euros per megajoule.
\textbf{9.} Bowl: $0.50/1.5 \approx 0.33$ euros per megajoule ---
five times the grid. Cheapest first: gasoline, grid, then food,
far behind.
\textbf{10.} $0.20 \times 1000 = \qty{200}{W}$ per square \omterm{def:g10:orders-of-magnitude:unit}{metre}.
\textbf{11.} $200 \times 4.0 = \qty{800}{W.h} = \qty{0.80}{kW.h}$
per square \omterm{def:g10:orders-of-magnitude:unit}{metre} per day.
\textbf{12.} $12.6/0.80 = 15.75 \approx \qty{16}{m^2}$: it fits
the \qty{20}{m^2} roof.
\textbf{13.} $16 \times 0.2 \times 1.5 = \qty{4.8}{kW.h}$: about
$38\%$ of the need; the grid (or a battery charged in better
months) must supply the rest.
\textbf{14.} One square \omterm{def:g10:orders-of-magnitude:unit}{metre} yields $0.80 \times 365 \approx
\qty{292}{kW.h}$ per year; payback $700/292 \approx 2.4$ years ---
short against a 25-year panel lifetime.
\textbf{15.} $\qty{12.6}{kW.h}/\qty{0.1}{kW} = \qty{126}{h}$:
more than five people ($126/24 = 5.25$) pedaling day and night.
\textbf{16.} Food $= 12.6/0.25 = \qty{50.4}{kW.h} \approx
\qty{181}{MJ}$, i.e.\ $181/1.5 \approx 121$ cereal bowls.
\textbf{17.} $121 \times 0.50 \approx 60$ euros, against
$12.6 \times 0.25 \approx 3.15$ euros from the grid: pedal \omterm{def:g10:energy-conservation:power}{power}
costs about $19$ times more --- before paying the cyclists.
\textbf{18.} $5000/45 \approx 110$ house-days. But
$\num{5e9}/\num{1e-4} = \qty{5e13}{W}$ for \qty{100}{\micro s}, at
a random place: no \omterm{def:g10:energy-conservation:transfer}{converter} or battery can drink from that hose.
A house needs a steady \qty{525}{W} flow, not a spike.
\textbf{19.} Not \omterm{def:g10:energy-conservation:energy}{joules} --- they are conserved and identical. You
pay for their \emph{form and availability}: \omterm{def:g10:energy-conservation:energy}{energy} that is
concentrated, storable and on demand costs the whole \omterm{def:g10:energy-conservation:transfer}{conversion
chain} that made it so.
\textbf{20.} One family day $= \qty{12.6}{kW.h} \approx
\qty{45}{MJ}$: one \omterm{def:g10:orders-of-magnitude:unit}{kilogram} of gasoline, sixteen square \omterm{def:g10:orders-of-magnitude:unit}{metres} of
sunny roof, or five round-the-clock cyclists. The stock is \omterm{def:g10:energy-conservation:energy}{energy}
(\omterm{def:g10:energy-conservation:energy}{joules}, \omterm{def:g10:energy-conservation:kwh}{kilowatt-hours}); the flow is \omterm{def:g10:energy-conservation:power}{power} (\omterm{def:g10:energy-conservation:power}{watts}) --- the bathtub
and the tap, never again confused.
\end{solution}
