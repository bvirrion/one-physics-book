\section*{Hoofdstuk \ref{ch:g10:signals-and-waves} --- Signalen en golven}

\begin{solution}{exo:g10:signals-and-waves:1}
$f = 1/T$: (a) $1/\qty{0.020}{s} = \qty{50}{Hz}$;
(b) $1/\qty{4.0e-3}{s} = \qty{250}{Hz}$;
(c) $1/\qty{5.0e-5}{s} = \qty{20}{kHz}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:2}
$T = 1/f$: (a) \qty{20}{ms}; (b) $1/440 = \qty{2.3}{ms}$;
(c) $1/\qty{2.4e9}{Hz} = \qty{0.42}{ns}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:3}
\qty{12}{Hz}: \omterm{def:g10:signals-and-waves:ultrasound}{infrageluid}. \qty{440}{Hz} en \qty{18}{kHz}: hoorbaar
(het laatste alleen voor jonge oren). \qty{40}{kHz} en \qty{5}{MHz}:
\omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid}.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:4}
$T = 5.0 \times \qty{2}{ms} = \qty{10}{ms}$, dus $f = \qty{100}{Hz}$;
\omterm{def:g10:signals-and-waves:amplitude}{amplitude} $4.0 \times \qty{0.5}{V} = \qty{2.0}{V}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:5}
$d = 340 \times 6.0 = \qty{2040}{m} \approx \qty{2.0}{km}$. Licht
overbrugt dit in $2040/(3.00\times10^{8}) \approx \qty{7}{\micro s}$,
een miljoen keer minder dan \qty{6.0}{s}: de flits markeert het
ogenblik van de inslag.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:6}
Het \omterm{def:g10:signals-and-waves:sound}{geluid} gaat heen \emph{en terug}, over $2d$:
$d = \frac{1500 \times 0.60}{2} = \qty{450}{m}$. Over de geul:
$t = 2d/v = 2 \times 1200/1500 = \qty{1.6}{s}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:7}
$d_1 = \frac{3.00\times10^{8} \times 2.4\times10^{-4}}{2} =
\qty{36.0}{km}$ en $d_2 = \qty{35.7}{km}$: het vliegtuig legde
\qty{300}{m} af in \qty{1.0}{s}, dus $v = \qty{300}{m/s} \approx
\qty{1100}{km/h}$, naderend.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:8}
$T = \qty{20}{mm}/\qty{25}{mm/s} = \qty{0.80}{s}$, dus
$f = \qty{1.25}{Hz}$: $75$ slagen per minuut. Voor $100$ slagen per
minuut, $T = \qty{0.60}{s}$, d.w.z.\ $25 \times 0.60 = \qty{15}{mm}$:
tachycardie toont onder \qty{15}{mm}.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:9}
$T = 1/200 = \qty{5.0}{ms}$; twee cycli duren \qty{10}{ms}, gespreid
over $10$ divisies: \omterm{def:g10:signals-and-waves:oscilloscope}{tijdbasis} \qty{1}{ms} per divisie. De top moet
in $4$ divisies passen: \omterm{def:g10:signals-and-waves:oscilloscope}{versterking} minstens $3.0/4 = \qty{0.75}{V}$
per divisie --- neem \qty{1}{V} per divisie (top $3.0$ divisies omhoog).
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:10}
$d = v\,t/2$: voorwand
$\frac{1540 \times 4.0\times10^{-5}}{2} = \qty{3.1}{cm}$; achterwand
$\frac{1540 \times 6.0\times10^{-5}}{2} = \qty{4.6}{cm}$; dikte
ongeveer \qty{1.5}{cm}.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:11}
\omterm{def:g10:signals-and-waves:sound}{Geluid}: $30/340 = \qty{0.088}{s}$. Radio:
$3.0\times10^{6}/(3.00\times10^{8}) = \qty{0.010}{s}$. De verre
radioluisteraar hoort de trom eerst, met ongeveer \qty{78}{ms}.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:12}
Tijdens \qty{3.0}{ms} zenden is de vleermuis doof voor
\omterm{rem:g10:signals-and-waves:honest}{echo's}, d.w.z.\ voor alles binnen
$d = \frac{340 \times 3.0\times10^{-3}}{2} = \qty{0.51}{m}$. Mot bij
\qty{1.7}{m}: $t = 2 \times 1.7/340 = \qty{10}{ms}$. Bij naderen
keert de \omterm{rem:g10:signals-and-waves:honest}{echo} steeds eerder terug; de puls moet eindigen
voordat die aankomt, dus moet ze korter worden.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:13}
$d = \frac{340 \times 1.5}{2} = \qty{255}{m}$. Bij \qty{155}{m}:
$t = 2 \times 155/340 = \qty{0.91}{s}$. \omterm{rem:g10:signals-and-waves:honest}{Echo} en klap
vervloeien wanneer $t < \qty{0.1}{s}$, d.w.z.\ binnen
$d < \frac{340 \times 0.1}{2} = \qty{17}{m}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:14}
\emph{1.} $T = 3.0 \times \qty{5}{ms} = \qty{15}{ms}$,
$f = 1/0.015 \approx \qty{67}{Hz}$.
\emph{2.} $\qty{15}{ms}/\qty{2}{ms} = 7.5$ divisies.
\emph{3.} Het scherm toont \qty{50}{ms}, dan \qty{20}{ms}: $3$ volledige
cycli ($50/15 = 3.3$), dan $1$ ($20/15 = 1.3$).
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:15}
\emph{1.} $t = 2 \times 1.2\times10^{6}/(3.00\times10^{8}) =
\qty{8.0}{ms}$.
\emph{2.} $2 \times 1.2\times10^{6}/(2.0\times10^{8}) = \qty{12}{ms}$.
\emph{3.} De vezel loopt niet recht tussen de twee machines, en elke
router en server onderweg voegt verwerkingsvertraging toe.
\end{solution}

\begin{solution}{pb:g10:signals-and-waves:1}
\textbf{1.} $d = 340 \times 9.0 = \qty{3060}{m} \approx \qty{3.1}{km}$.

\textbf{2.} $3060/(3.00\times10^{8}) \approx \qty{10}{\micro s}$, een
miljoen keer korter dan \qty{9.0}{s}: de flits markeert het ogenblik
van de inslag.

\textbf{3.} In \qty{3}{s} overbrugt \omterm{def:g10:signals-and-waves:sound}{geluid}
$340 \times 3 = \qty{1020}{m} \approx \qty{1}{km}$: seconden gedeeld
door drie geeft kilometers.

\textbf{4.} $d = 340 \times 6.0 = \qty{2040}{m}$. De storm naderde
\qty{1020}{m} in \qty{300}{s}: $v = \qty{3.4}{m/s} \approx
\qty{12}{km/h}$, op weg naar de boot.

\textbf{5.} Het nabije einde hoort men na $2000/340 = \qty{5.9}{s}$,
het verre na $5000/340 = \qty{14.7}{s}$: de rommel duurt ongeveer
\qty{8.8}{s}.

\textbf{6.} $d = 1500 \times 0.90/2 = \qty{675}{m}$.

\textbf{7.} $d = 1500 \times 0.20/2 = \qty{150}{m}$.

\textbf{8.} Twee obstakels: een visschool bij $1500 \times 0.16/2 =
\qty{120}{m}$ en de zeebodem bij \qty{150}{m}. De school zwemt
\qty{30}{m} boven de bodem.

\textbf{9.} De \omterm{rem:g10:signals-and-waves:honest}{echo} moet terugkeren vóór de volgende ping:
$d_{\max} = 1500 \times 0.50/2 = \qty{375}{m}$. Van dieper water
arriveert de \omterm{rem:g10:signals-and-waves:honest}{echo} \emph{na} de volgende ping en wordt
eraan toegeschreven: het scherm toont een valse, veel te ondiepe bodem.

\textbf{10.} In lucht, $340 \times 0.60/2 = \qty{102}{m}$ in plaats van
\qty{450}{m}. Een vertraging wordt een afstand alleen wanneer je de
drager en haar snelheid in het doorkruiste medium kent.

\textbf{11.} $f = 1/0.86 = \qty{1.16}{Hz}$, d.w.z.\ $1.16 \times 60
\approx 70$ slagen per minuut.

\textbf{12.} $25 \times 0.86 = \qty{21.5}{mm}$.

\textbf{13.} $T = 15/25 = \qty{0.60}{s}$: $100$ slagen per minuut.

\textbf{14.} $T = 60/50 = \qty{1.2}{s}$; afstand
$25 \times 1.2 = \qty{30}{mm}$.

\textbf{15.} Nee: de \omterm{def:g10:signals-and-waves:period}{periode} drijft van slag tot slag
met inspanning, ademhaling en stress. De arts leest de
\omterm{def:g10:signals-and-waves:period}{periode} (het tempo), haar regelmaat, en de vorm van
elke cyclus --- de diagnose zit in het \omterm{def:g10:signals-and-waves:signal}{signaal}.

\textbf{16.} Radio: $5.0\times10^{4}/(3.00\times10^{8}) =
\qty{0.17}{ms}$ --- onmiddellijk voor menselijke zintuigen.
\omterm{def:g10:signals-and-waves:sound}{Geluid}: $50\,000/340 \approx \qty{147}{s}$, twee en een
halve minuut.

\textbf{17.} $t = 2.02\times10^{7}/(3.00\times10^{8}) = \qty{67}{ms}$.

\textbf{18.} $c \times \qty{1.0}{\micro s} = \qty{300}{m}$. Voor een
\qty{3.0}{m}-fixatie: $3.0/(3.00\times10^{8}) = \qty{10}{ns}$ ---
daarom dragen GPS-\omterm{def:g10:universal-gravitation:satellite}{satellieten} atoomklokken.

\textbf{19.} \omterm{def:g10:universal-gravitation:satellite}{Satellieten} zitten in vacuüm, en
\omterm{def:g10:signals-and-waves:sound}{geluid} heeft een medium nodig: er zou helemaal geen
\omterm{def:g10:signals-and-waves:signal}{signaal} de \omterm{def:g10:universal-gravitation:satellite}{satelliet} verlaten. (Zelfs
lucht de hele weg toegestaan: \qty{20200}{km} bij \qty{340}{m/s} is
meer dan \qty{16}{h} --- een positiefix uren te laat.)

\textbf{20.} \qty{340}{m/s} (\omterm{def:g10:signals-and-waves:sound}{geluid} in lucht),
\qty{1500}{m/s} (\omterm{def:g10:signals-and-waves:sound}{geluid} in zeewater) en
$c = \qty{3.00e8}{m/s}$ (radio en licht); de wet is $d = v\,t$,
gehalveerd voor een \omterm{rem:g10:signals-and-waves:honest}{echo}. De gewoonte: ken je drager,
ken haar snelheid in het medium, tijd de vertraging --- een
vertraging is een afstand in vermomming.
\end{solution}
