\section*{Hoofdstuk \ref{ch:g10:orders-of-magnitude} --- Ordegrootten: het heelal meten}

\begin{solution}{exo:g10:orders-of-magnitude:1}
\emph{1.} $\qty{750}{nm} = 750 \times \qty{e-9}{m} = \qty{7.50e-7}{m}$.
\emph{2.} $\qty{2.5}{km} = \qty{2.5e3}{m}$.
\emph{3.} $\qty{3.2}{\micro m} = \qty{3.2e-6}{m}$.
\emph{4.} $\qty{0.15}{Tm} = 0.15 \times \qty{e12}{m} = \qty{1.5e11}{m}$
--- één \omterm{def:g10:orders-of-magnitude:lightyear}{astronomische eenheid}.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:2}
\emph{1.} $\qty{149600000}{km} = \qty{1.496e11}{m}$: $4$ \omterm{def:g10:orders-of-magnitude:sigfig}{significante
cijfers}.
\emph{2.} $\qty{1.06e-10}{m}$: $3$ \omterm{def:g10:orders-of-magnitude:sigfig}{significante cijfers} (voorloopnullen
tellen nooit mee).
\emph{3.} $\num{8.1e9}$: $2$ \omterm{def:g10:orders-of-magnitude:sigfig}{significante cijfers}.
\emph{4.} $\qty{5.20e-3}{s}$: $3$ \omterm{def:g10:orders-of-magnitude:sigfig}{significante cijfers} (de
achterloopnul is een belofte).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:3}
\omterm{def:g10:orders-of-magnitude:scinot}{Mantisses} en \omterm{def:g10:orders-of-magnitude:scinot}{exponenten} afzonderlijk
(\cref{prop:g10:orders-of-magnitude:powers}):
\[
3 \times 2 \times 10^{8-5} = \num{6e3}, \qquad
\tfrac{8}{4} \times 10^{4-(-3)} = \num{2e7},
\]
\[
5^{2} \times 10^{12} = \num{2.5e13}, \qquad
\tfrac{4 \times 3}{6} \times 10^{-3-9+7} = \num{2e-5} .
\]
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:4}
Mens: $1.7 < 5$, ordegrootte $\qty{e0}{m}$. Everest:
$\qty{8848}{m} = \qty{8.848e3}{m}$ en $8.848 \geq 5$: ordegrootte
$\qty{e4}{m}$. Bacterie: ordegrootte $\qty{e-6}{m}$. Aarde:
$5.97 \geq 5$, ordegrootte $\qty{e25}{kg}$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:5}
\emph{1.} Halve schaalverdeling: $L = (127.0 \pm 0.5)\,\unit{mm}$.
\emph{2.} $\Delta L / L = 0.5/127 \approx 0.4\%$.
\emph{3.} Gum: $0.5/12 \approx 4\%$. De potloodmeting is
ongeveer tien keer preciezer --- dezelfde liniaal, dezelfde
\omterm{def:g10:orders-of-magnitude:uncertainty}{absolute onzekerheid}, object tien keer langer.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:6}
\emph{1.} Proxima:
$4.24 \times \qty{9.47e15}{m} = \qty{4.01e16}{m}$. Sirius:
$8.6 \times \qty{9.47e15}{m} = \qty{8.1e16}{m}$ (twee \omterm{def:g10:orders-of-magnitude:sigfig}{significante
cijfers}, zoals de invoer).
\emph{2.} $t = d/c = \qty{2.4e13}{m} / \qty{3.00e8}{m/s}
= \qty{8.0e4}{s}$, d.w.z.\ $\num{8.0e4}/3600 \approx \qty{22}{h}$:
bijna een volle dag voor elk bericht van de sonde.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:7}
$t = d/c$ elke keer. Maan:
$\qty{3.84e8}{m}/\qty{3.00e8}{m/s} = \qty{1.28}{s}$. Zon:
$\qty{1.496e11}{m}/\qty{3.00e8}{m/s} = \qty{499}{s} \approx
\qty{8.3}{min}$. Neptunus:
$\qty{4.50e12}{m}/\qty{3.00e8}{m/s} = \qty{1.50e4}{s} \approx
\qty{4.2}{h}$. Als je naar Neptunus kijkt, zie je hem zoals hij ongeveer
vier uur geleden was --- elke telescoop is een milde tijdmachine.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:8}
\emph{1.} Eén vel: $\qty{52.0}{mm}/500 = \qty{0.104}{mm}
= \qty{1.04e-4}{m}$. De onzekerheid deelt mee:
$\qty{0.5}{mm}/500 = \qty{0.001}{mm}$, dus
$t = (104 \pm 1)\,\unit{\micro m}$.
\emph{2.} $1/104 \approx 1\%$.
\emph{3.} Eén vel rechtstreeks meten zou geven
$(0.1 \pm 0.5)\,\unit{mm}$: een \omterm{def:g10:orders-of-magnitude:uncertainty}{relatieve onzekerheid} van ongeveer $500\%$
--- de liniaal kan niet eens bewijzen dat het vel bestaat. $500$ vellen
stapelen vermenigvuldigt de gemeten lengte met $500$ terwijl de
\omterm{def:g10:orders-of-magnitude:uncertainty}{absolute onzekerheid} van de liniaal vast blijft: de
\omterm{def:g10:orders-of-magnitude:uncertainty}{relatieve onzekerheid} krimpt met diezelfde factor $500$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:9}
Deel elke lengte door $\qty{2e-10}{m}$. Bacterie:
$\qty{2.0e-6}{m}/\qty{2e-10}{m} = \num{1.0e4}$ atomen. Haar:
$\qty{7.0e-5}{m}/\qty{2e-10}{m} = \num{3.5e5}$. Mens:
$1.7/\num{2e-10} = \num{8.5e9}$, \omterm{def:g10:orders-of-magnitude:oom}{ordegrootte} $10^{10}$: tien
miljard atomen van top tot teen.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:10}
\emph{1.} $s = \qty{0.22}{m}/\qty{1.39e9}{m} = \num{1.58e-10}$.
\emph{2.} Aarde:
$\qty{1.27e7}{m} \times \num{1.58e-10} = \qty{2.0e-3}{m} =
\qty{2.0}{mm}$ --- een peperkorrel --- die wentelt op
$\qty{1.496e11}{m} \times \num{1.58e-10} = \qty{23.7}{m} \approx
\qty{24}{m}$ van de voetbal.
\emph{3.} Maan:
$\qty{3.84e8}{m} \times \num{1.58e-10} = \qty{6.1}{cm}$ van de
peperkorrel. Neptunus:
$\qty{4.50e12}{m} \times \num{1.58e-10} \approx \qty{710}{m}$: het
model past niet op de schoolspeelplaats.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:11}
\emph{1.} $21.0 \times 29.7 = \qty{623.7}{cm^2}$, afgerond op de drie
\omterm{def:g10:orders-of-magnitude:sigfig}{significante cijfers} van de invoer: $\qty{624}{cm^2}$.
\emph{2.} $v = \qty{100.0}{m}/\qty{12.3}{s} = \qty{8.13}{m/s}$ (drie
cijfers, bepaald door de tijd).
\emph{3.} Vijf \omterm{def:g10:orders-of-magnitude:sigfig}{significante cijfers} zou de oppervlakte tot
$\qty{0.01}{cm^2}$ claimen, een precisie die de millimeterniveau-invoer
nooit leverde --- een vervalste decimaal blijft een vervalsing.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:12}
\emph{1.} Straal $r = \qty{0.55}{m}$, oppervlakte
$A = \pi r^{2} = \pi \times 0.55^{2} \approx \qty{0.95}{m^2}$. Volume
$V = \qty{1.0}{mm^3} = \qty{1.0e-9}{m^3}$, en een film is (oppervlakte)
$\times$ (dikte), dus
\[
e = \frac{V}{A} = \frac{\qty{1.0e-9}{m^3}}{\qty{0.95}{m^2}}
= \qty{1.1e-9}{m} .
\]
\emph{2.} Een molecule is van ordegrootte $\qty{e-9}{m}$ --- één sport
boven de $\qty{e-10}{m}$ van het atoom, consistent met een molecule als
kleine cluster van atomen. Franklins theelepel olie kalmeerde een halve
acre vijver en mat, zonder dat hij het wist, de nanometer.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:13}
Aannames: ongeveer $70$ slagen per minuut in rust, een leven van $80$
jaar. Per dag: $70 \times 60 \times 24 \approx \num{1e5}$ slagen. Per
jaar: $\num{1e5} \times 365 \approx \num{4e7}$. Per leven:
$\num{4e7} \times 80 \approx \num{3e9}$ --- \omterm{def:g10:orders-of-magnitude:oom}{ordegrootte}
$10^{9}$, een paar miljard hartslagen. Vier \omterm{def:g10:orders-of-magnitude:sigfig}{significante cijfers} zou
doen alsof men een hartslag minuut voor minuut tachtig jaar kent: de
invoer is ordegrootte-giswerk, en geen uitvoer kan zijn invoer
overtreffen (\cref{met:g10:orders-of-magnitude:sigfig}).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:14}
Na $n$ vouwen is de dikte $2^{n} \times \qty{1.0e-4}{m}$.
\emph{1.} Nodig $2^{n} \geq 8848/\num{1.0e-4} = \num{8.8e7}$. Omdat
$2^{26} \approx \num{6.7e7}$ tekortschiet en
$2^{27} \approx \num{1.3e8}$ volstaat: $27$ vouwen (stapel
$\approx \qty{13}{km}$).
\emph{2.} Nodig $2^{n} \geq \num{3.84e12}$: $2^{41} \approx
\num{2.2e12}$ is te kort, $2^{42} \approx \num{4.4e12}$ haalt ---
$42$ vouwen, stapel $\qty{4.4e8}{m}$, comfortabel voorbij de Maan.
\emph{3.} Alleen het experiment: elke vouw halveert de oppervlakte, en
na zeven vouwen is de stapel dikker dan breed. De rekenkunde van het
verdubbelen blijft onaangetast --- exponentiële groei loopt simpelweg
vooruit op wat papier (en intuïtie) kan volgen.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:15}
\emph{1.} Maan: $\qty{0.03}{m}/\qty{3.84e8}{m} = \num{7.8e-11}$.
Tafel: $\qty{0.001}{m}/\qty{1.000}{m} = \num{1.0e-3}$. De Maan\-
meting is preciezer met een factor
$\num{1.0e-3}/\num{7.8e-11} \approx \num{1.3e7}$ --- meer dan tien
miljoen.
\emph{2.} $\num{7.8e-11}$ evenaren op een \omterm{def:g10:orders-of-magnitude:unit}{één-meter}-tafel betekent
$\Delta x = \qty{7.8e-11}{m}$, kleiner dan één atoom
($\approx \qty{e-10}{m}$): de tafelrand is niet eens zo scherp
\emph{gedefinieerd}. Relatief gezien is de Aarde--Maan-laserafstand
een van de fijnste metingen die onze soort bezit.
\end{solution}

\begin{solution}{pb:g10:orders-of-magnitude:1}
\textbf{1.} Korrel: $\qty{0.5}{mm} = \qty{5e-4}{m}$, op de milli\-
schaal; atoom: $\qty{2.5e-10}{m} = \qty{0.25}{nm}$, op de nanoschaal.

\textbf{2.} $\qty{5e-4}{m}/\qty{2.5e-10}{m} = \num{2.0e6}$: twee
miljoen atomen over.

\textbf{3.} Een kubus van $2 \times 10^{6}$ atomen per rand bevat
$(2\times10^{6})^{3} = 8 \times 10^{18}$ atomen: \omterm{def:g10:orders-of-magnitude:oom}{ordegrootte}
$10^{19}$.

\textbf{4.} $V = (\qty{5e-4}{m})^{3} = \qty{1.25e-10}{m^3}$, dus
$m = \rho V = \qty{2500}{kg/m^3} \times \qty{1.25e-10}{m^3}
= \qty{3.1e-7}{kg} = \qty{0.31}{mg}$.

\textbf{5.} Massa per atoom:
$\qty{3.1e-7}{kg}/(8\times10^{18}) = \qty{3.9e-26}{kg}$ --- precies
in het bereik «enkele $10^{-26}\,\unit{kg}$». De naïeve atoomkubus
reproduceert echte atomaire massa's: het model doorstaat de
sanity check (\cref{met:g10:orders-of-magnitude:sanity}) met vlag en wimpel.

\textbf{6.} $s = \qty{5e-4}{m}/\qty{1.39e9}{m} = \num{3.6e-13}$.

\textbf{7.} Model-Aarde: diameter
$\qty{1.27e7}{m} \times \num{3.6e-13} = \qty{4.6e-6}{m}$ --- de
grootte van een bacterie --- op afstand
$\qty{1.496e11}{m} \times \num{3.6e-13} = \qty{5.4}{cm}$ van de
korrel: de Aarde is een microbe vijf centimeter van een zandkorrel.

\textbf{8.} $\qty{4.5e12}{m} \times \num{3.6e-13} = \qty{1.6}{m}$: het
hele planetenstelsel wentelt binnen een eettafel.

\textbf{9.} $\qty{4.0e16}{m} \times \num{3.6e-13} = \qty{1.4e4}{m}
= \qty{14}{km}$.

\textbf{10.} De dichtstbijzijnde ster bij ons tafel-zonnestelsel is
een andere zandkorrel veertien kilometer verder, met vrijwel niets
ertussen: de melkweg is overweldigend lege ruimte, daarom zijn
botsingen tussen sterren vrijwel onbekend.

\textbf{11.} $t = \qty{5e-4}{m}/\qty{3.00e8}{m/s} = \qty{1.7e-12}{s}$
--- onder twee picoseconden.

\textbf{12.} Maan: $\qty{3.84e8}{m}/\qty{3.00e8}{m/s} =
\qty{1.28}{s}$. Zon: $\qty{1.496e11}{m}/\qty{3.00e8}{m/s} =
\qty{499}{s} \approx \qty{8.3}{min}$: zonlicht is altijd acht minuten
oud.

\textbf{13.} Eén jaar is $365.25 \times 86400 = \qty{3.156e7}{s}$, dus
$\qty{1}{ly} = \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}$; dan
$\qty{4.0e16}{m}/\qty{9.47e15}{m} = 4.2$: de gegevensafstand is
inderdaad ongeveer $\qty{4.2}{ly}$.

\textbf{14.} $\qty{2.4e22}{m}/\qty{9.47e15}{m} = \num{2.5e6}$: twee
en een half miljoen \omterm{def:g10:orders-of-magnitude:lightyear}{lichtjaren}. Dat licht verliet Andromeda
ongeveer $2.5$ miljoen jaar geleden, toen de eerste leden van het
geslacht \emph{Homo} leerden stenen te bewerken --- en nog geen
\emph{Homo sapiens} ergens bestond.

\textbf{15.} $\qty{3.00e8}{m/s} \times \qty{1e-9}{s} = \qty{0.30}{m}$.
In een derde nanoseconde overbrugt een signaal hoogstens ongeveer
$\qty{10}{cm}$, dus twee delen van de machine die binnen één
kloktik moeten praten, moeten binnen tien centimeter van elkaar
zitten: de snelheid begrensd de grootte.

\textbf{16.} Gemiddeld atoom in water:
$\qty{3.0e-26}{kg}/3 = \qty{1.0e-26}{kg}$. Aantal atomen:
$\qty{70}{kg}/\qty{1.0e-26}{kg} = \num{7e27}$.

\textbf{17.} Tegen de korrel:
$\num{7e27}/\num{8e18} \approx 10^{9}$ --- jij bevat een miljard
korrels aan atomen. Tegen de sterren:
$\num{7e27}/\num{2e23} = \num{3.5e4}$ --- jouw atomen overtreffen alle
sterren van het waarneembare heelal vijfendertigduizend tegen één.

\textbf{18.} $\text{(jij)}/\text{(atoom)} =
\qty{1.7}{m}/\qty{2.5e-10}{m} \approx \num{7e9} \approx 10^{10}$,
terwijl $\text{(heelal)}/\text{(jij)} =
\qty{8.8e26}{m}/\qty{1.7}{m} \approx \num{5e26}$. Op de ladder sta
je ongeveer tien sporten boven het atoom en bijna zevenentwintig onder
het heelal: multiplicatief leven mensen heel dicht bij hun atomen.

\textbf{19.} De voorwaarde geeft
$x^{2} = 10^{-15} \times 10^{26.9} = 10^{11.9}$, dus
$x = 10^{5.95} \approx \qty{9e5}{m} \approx \qty{900}{km}$ --- bijna
exact de diameter van Ceres, de grootste planetoïde
($\approx \qty{940}{km}$). Halverwege de ladder tussen een kern en
de kosmos zit een bescheiden dwergplaneet.

\textbf{20.} $\qty{8.8e26}{m} \times \num{3.6e-13} = \qty{3.2e14}{m}$,
d.w.z.\ $\qty{3.2e14}{m}/\qty{1.496e11}{m} \approx \num{2.1e3}$
\omterm{def:g10:orders-of-magnitude:lightyear}{astronomische eenheden}. Punchline: krimpt men de kosmos tot
de Zon een zandkorrel is, dan overspant het waarneembare heelal nog
steeds tweeduizend keer de echte Aarde--Zon-afstand --- geen
schaalmodel brengt het heelal tot menselijke grootte; alleen machten
van tien kunnen het dragen.
\end{solution}
