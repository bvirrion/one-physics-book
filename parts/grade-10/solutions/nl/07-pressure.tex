\section*{Hoofdstuk \ref{ch:g10:pressure} --- Druk: van sport tot duiken}

\begin{solution}{exo:g10:pressure:1}
Grote zijde: $S = \qty{0.15}{m^2}$, $P = 600/0.15 = \qty{4.0}{kPa}$.
Kleine zijde: $S = \qty{0.060}{m^2}$, $P = \qty{10}{kPa}$. Op zand de
grote zijde: lagere \omterm{def:g10:pressure:pressure}{druk}, minder inzinken.
\end{solution}

\begin{solution}{exo:g10:pressure:2}
$\qty{0.25}{MPa} = \qty{250}{kPa} = \qty{2.5e5}{Pa}$.
$3.0/\num{2.0e-4} = \qty{15}{kPa}$ verslaat
$60/0.050 = \qty{1.2}{kPa}$: de kleine \omterm{def:g10:inertia:force}{kracht} drukt harder.
\end{solution}

\begin{solution}{exo:g10:pressure:3}
$\rho g h = 1000 \times 9.81 \times 3.0 \approx \qty{29}{kPa}$;
absoluut $P = 101 + 29 = \qty{130}{kPa}$.
\end{solution}

\begin{solution}{exo:g10:pressure:4}
$V_2 = 100 \times 4.0 / 250 = \qty{1.6}{L}$;
bij \qty{50}{kPa}, $V_2 = 400/50 = \qty{8.0}{L}$.
\end{solution}

\begin{solution}{exo:g10:pressure:5}
$F = 65 \times 9.81 \approx \qty{638}{N}$. Schaats:
$638/\num{1.2e-3} \approx \qty{5.3e5}{Pa} = \qty{530}{kPa}$. Zolen:
$638/0.036 \approx \qty{18}{kPa}$. Verhouding $\approx 30$: de schaats
bijt in het ijs, de zolen niet.
\end{solution}

\begin{solution}{exo:g10:pressure:6}
Duim: $20/\num{1.2e-4} \approx \qty{1.7e5}{Pa} = \qty{170}{kPa}$.
Muur: $20/\num{1.0e-8} = \qty{2.0e9}{Pa}$. Dezelfde \qty{20}{N} werkt op
een $12\,000$ keer kleiner oppervlak aan de punt, dus alleen daar
overschrijdt de \omterm{def:g10:pressure:pressure}{druk} wat het materiaal kan weerstaan.
\end{solution}

\begin{solution}{exo:g10:pressure:7}
$\Delta P = \rho g h = 1025 \times 9.81 \times 30 \approx
\qty{3.0e5}{Pa}$, dus
$F = \num{3.0e5} \times 0.50 \approx \qty{1.5e5}{N}$ --- het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van $F/g \approx \qty{1.5e4}{kg}$, vijftien ton. Geen
bemanning opent die naar buiten tegen de zee.
\end{solution}

\begin{solution}{exo:g10:pressure:8}
$\rho = \dfrac{P}{g h} = \dfrac{\num{66.2e3}}{9.81 \times 5.00}
\approx \qty{1350}{kg/m^3}$. Geen zoet water ($\qty{1000}{kg/m^3}$):
een duidelijk dichtere vloeistof, bijv.\ geconcentreerde pekel.
\end{solution}

\begin{solution}{exo:g10:pressure:9}
$h = P_0/(\rho g)$: water
$\num{1.01e5}/9810 \approx \qty{10.3}{m}$; kwik
$\num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$. Een
\qty{76}{cm}-buis past op een bureau; een \qty{10}{m}-waterbarometer
heeft een trappenhuis nodig --- de dichtheid van kwik won.
\end{solution}

\begin{solution}{exo:g10:pressure:10}
$V = 6000/240 = \qty{25}{mL}$; $P = 6000/75 = \qty{80}{kPa}$.
$P$ tegen $V$ is een hyperbool; $P$ tegen $1/V$ is een rechte lijn
door de oorsprong, met helling \qty{6000}{kPa.mL}.
\end{solution}

\begin{solution}{exo:g10:pressure:11}
Bij \qty{20}{m}: $P = 101 + 2 \times 101 = \qty{303}{kPa}$, drie
atmosferen. Aan het oppervlak,
$V = 0.50 \times 303/101 = \qty{1.5}{cm^3}$: de bel verdrievoudigt.
\end{solution}

\begin{solution}{exo:g10:pressure:12}
\emph{1.} $F = \num{1.01e5} \times \num{1.2e-3} \approx \qty{121}{N}$,
houdt $m = 121/9.81 \approx \qty{12}{kg}$.

\emph{2.} $\Delta P = 101 - 20 = \qty{81}{kPa}$:
$F = \num{8.1e4} \times \num{1.2e-3} \approx \qty{97}{N}$, ongeveer
\qty{9.9}{kg}.
\end{solution}

\begin{solution}{exo:g10:pressure:13}
Boyle: $12 \times (200 - 50) = \qty{1800}{L}$
\omterm{def:g10:pressure:pressure}{oppervlaktedruk}-lucht is bruikbaar. Bij \qty{30}{m} wordt
elke ademhaling genomen bij \qty{4}{bar}, dus \qty{15}{L/min} bij
omgevings-\omterm{def:g10:pressure:pressure}{druk} kost $15 \times 4 = \qty{60}{L/min}$
oppervlaktelucht: $1800/60 = \qty{30}{min}$. Bij \qty{10}{m}
($\qty{2}{bar}$): \qty{30}{L/min}, dus \qty{60}{min}. Diepte wordt
betaald in lucht.
\end{solution}

\begin{solution}{exo:g10:pressure:14}
$P = P_0 \dfrac{V_0}{V} = 101 \times \dfrac{6.0}{1.5} =
\qty{404}{kPa}$. Met $\rho g \approx \qty{10.1}{kPa/m}$,
$h = (404 - 101)/10.1 \approx \qty{30}{m}$.
\end{solution}

\begin{solution}{exo:g10:pressure:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{Gewicht} $F = 1200 \times 9.81 \approx \qty{1.18e4}{N}$;
\omterm{def:g10:pressure:pressure}{druk} $P = F/S = \num{1.18e4}/0.040 \approx \qty{2.9e5}{Pa}$;
kleine zuiger: $f = P s = \num{2.9e5} \times \num{2.0e-4} \approx \qty{59}{N}$
--- de \omterm{def:g10:inertia:force}{kracht} wordt gedeeld door $S/s = 200$.

\emph{2.} Vloeistofvolume is behouden:
$400 \times 2.0 = \qty{800}{cm^3}$ moet uit de kleine cilinder komen,
die $800/2.0 = \qty{400}{cm} = \qty{4.0}{m}$ reist. Arbeid:
$59 \times 4.0 \approx \qty{235}{J}$ aan de ene kant,
$\num{1.18e4} \times 0.020 \approx \qty{235}{J}$ aan de andere --- de
lift ruilt afstand voor \omterm{def:g10:inertia:force}{kracht}, nooit arbeid.
\end{solution}

\begin{solution}{pb:g10:pressure:1}
\textbf{1.} $\rho g = 1025 \times 9.81 \approx \qty{10.1}{kPa}$ per
\omterm{def:g10:orders-of-magnitude:unit}{meter}, dus \qty{10}{m} zeewater voegen $\approx \qty{101}{kPa}$
toe: één atmosfeer per tien \omterm{def:g10:orders-of-magnitude:unit}{meters}, bijna exact.

\textbf{2.} $P = 101 + 10.1\,h$: \qty{202}{kPa} bij \qty{10}{m},
\qty{302}{kPa} bij \qty{20}{m}, \qty{403}{kPa} bij \qty{30}{m},
\qty{503}{kPa} bij \qty{40}{m}.

\textbf{3.} $2.0$, $3.0$, $4.0$ en $5.0$ keer $P_0$.

\textbf{4.} $\Delta P = 403 - 101 = \qty{302}{kPa}$ over
$\qty{0.015}{m^2}$: $F = \num{3.02e5} \times 0.015 \approx
\qty{4.5e3}{N}$ --- het \omterm{def:g10:universal-gravitation:weight}{gewicht} van bijna een halve ton op het
gezicht.

\textbf{5.} $\Delta P = 10.1 \times 3.0 \approx \qty{30}{kPa}$;
$F = \num{3.0e4} \times \num{6.0e-5} \approx \qty{1.8}{N}$ --- een
vinger op het trommelvlies, al op baddiepte. Equaliseren duwt lucht
in het middenoor tot de binnen-\omterm{def:g10:pressure:pressure}{druk} die van het water
evenaart.

\textbf{6.} $V = 6.0 \, P_0/P$: $6.0 \times 101/202 = \qty{3.0}{L}$;
$606/302 = \qty{2.0}{L}$; $606/403 = \qty{1.5}{L}$.

\textbf{7.} Gehalveerd wanneer $P = 2P_0$, d.w.z.\ bij \qty{10}{m} ---
in het allereerste stuk van de duik.

\textbf{8.} Nee: $V = 606/P$ is een hyperbool in de
\omterm{met:g10:pressure:dive}{absolute druk} (en $P$, niet $h$, is wat de
\omterm{prop:g10:pressure:boyle}{wet van Boyle} ziet); elke extra \qty{10}{m} verwijdert
minder volume dan de vorige.

\textbf{9.} $V = \qty{1.5}{L}$ vereist
$P = 101 \times 6.0/1.5 = \qty{404}{kPa}$, d.w.z.\
$h = 303/10.1 \approx \qty{30}{m}$ (\cref{exo:g10:pressure:14}).

\textbf{10.} Bloed: plasma verschuift naar de vaten van de borstkas,
oncompressibel, en bezet het volume dat de lucht niet meer vult.

\textbf{11.} Bij omgevings-\omterm{def:g10:pressure:pressure}{druk}, \qty{403}{kPa}. Volgens
Boyle zou dezelfde lucht bij \qty{101}{kPa} $4$ keer het volume vullen:
ze is $4.0$ keer dichter dan oppervlaktelucht.

\textbf{12.} $V = 6.0 \times 403/202 \approx \qty{12}{L}$ ---
dubbel, na slechts \qty{20}{m} stijging.

\textbf{13.} $V = 6.0 \times 403/101 \approx \qty{24}{L}$: vier keer
zijn longcapaciteit.

\textbf{14.} \emph{Houd nooit de adem in tijdens het opstijgen.} Met
de luchtweg open stroomt de expanderende lucht door de regulator naar
buiten, en het longvolume overschrijdt nooit \qty{6.0}{L}.

\textbf{15.} Lena's \qty{1.5}{L} bij \qty{30}{m} \emph{is} haar
oppervlakte-\qty{6.0}{L} samengeperst: op de weg omhoog expandeert die
precies tot \qty{6.0}{L}, nooit verder. Marco's \qty{6.0}{L} werden
\emph{genomen} bij \qty{403}{kPa}: ze bevatten \qty{24}{L}
oppervlaktelucht.

\textbf{16.} $V = 1.0 \times 503/P$: \qty{1.25}{cm^3} bij \qty{30}{m},
\qty{1.67}{cm^3} bij \qty{20}{m}, \qty{2.49}{cm^3} bij \qty{10}{m},
\qty{4.98}{cm^3} aan het oppervlak.

\textbf{17.} Etappefactoren: $503/403 = 1.25$, dan $1.33$, $1.49$,
en $202/101 = 2.0$ voor de laatste tien \omterm{def:g10:orders-of-magnitude:unit}{meters} --- groei
versnelt naarmate het oppervlak nadert.

\textbf{18.} $40/10 = \qty{4.0}{min}$. De laatste minuut: van
\qty{10}{m} naar het oppervlak halveert de
\omterm{met:g10:pressure:dive}{absolute druk}, de grootste relatieve daling van de hele
opstijging.

\textbf{19.} Een snelle opstijging laat opgeloste stikstof tot bellen
fizzelen in bloed en gewrichten (decompressieziekte); langzaam
opstijgen en pauzeren nabij \qty{5}{m} houdt het gas lang genoeg
opgelost om rustig via de longen te vertrekken.

\textbf{20.} Vrije duiker: haar lucht keert alleen terug naar haar
oorspronkelijke volume --- Boyle beschermt haar. Scuba-duiker: lucht
genomen op diepte vermenigvuldigt op de weg omhoog --- adem, houd
nooit in. En de laatste tien \omterm{def:g10:orders-of-magnitude:unit}{meters} verdubbelen elk
gevangen volume ($\times 2.0$, tegen $\times 1.25$ beneden bij
\qty{40}{m}): hoe dichter bij het oppervlak, hoe trager je moet
naderen.
\end{solution}
