\section*{Hoofdstuk \ref{ch:g10:light-spectra} --- Lichtspectra en de boodschap van het licht}

\begin{solution}{exo:g10:light-spectra:1}
$\qty{254}{nm} < \qty{400}{nm}$: \omterm{def:g10:light-spectra:wavelength}{ultraviolet}. $\qty{480}{nm}$: zichtbaar,
blauw. $\qty{589}{nm}$: zichtbaar, geeloranje. $\qty{656}{nm}$: zichtbaar,
rood. $\qty{1064}{nm} > \qty{800}{nm}$: \omterm{def:g10:light-spectra:wavelength}{infrarood} --- wat een
snijlaser dubbel gevaarlijk maakt: krachtig en onzichtbaar.
\end{solution}

\begin{solution}{exo:g10:light-spectra:2}
\emph{1.} Een continue kleurenband, van rood tot violet: het
\omterm{def:g10:light-spectra:white}{spectrum} van zonlicht.

\emph{2.} Het violette uiteinde --- het prisma buigt korte
\omterm{def:g10:light-spectra:wavelength}{golflengten} het meest af.

\emph{3.} Een enkele groene vlek: afgebogen, maar niet gespreid tot
een band. Laserlicht is \omterm{def:g10:light-spectra:white}{monochromatisch} --- één
\omterm{def:g10:light-spectra:wavelength}{golflengte} alleen, dus er is niets te sorteren.
\end{solution}

\begin{solution}{exo:g10:light-spectra:3}
(a) Continu (dichte hete gloeidraad). (b) Lijnemissie (aangeslagen gas
bij lage druk). (c) Continu (dicht gesmolten metaal). (d) Lijnemissie
(in wezen de enige lijn bij $\qty{589}{nm}$). (e) Absorptie: het
\omterm{def:g10:light-spectra:continuous}{continue spectrum} van het hete oppervlak, doorboord door de
donkere lijnen van de koelere atmosfeer van de Zon.
\end{solution}

\begin{solution}{exo:g10:light-spectra:4}
\emph{1.} De regendruppels: elk ontvangt wit zonlicht, ontbindt het
en stuurt het gesorteerd op kleur terug.

\emph{2.} De Zon levert het \omterm{def:g10:light-spectra:white}{witte licht}; de regen levert miljoenen
miniatuurprisma's; de meetkunde werkt alleen voor licht dat door de
druppels naar de waarnemer terugkomt, daarom moet de Zon in de rug van
de waarnemer staan en de regen ervoor.

\emph{3.} De druppels voegen geen eigen licht toe --- ze sorteren
alleen wat ze ontvangen. Als gesorteerd zonlicht elke kleur toont, dan
zat elke kleur al in het zonlicht: zonlicht is \omterm{def:g10:light-spectra:white}{wit licht}.
\end{solution}

\begin{solution}{exo:g10:light-spectra:5}
\emph{1.} Dof rood $<$ helder oranje $<$ bijna wit.

\emph{2.} Beide: de gloed wordt \emph{helderder} (effect 1), en haar
kleur schuift van rood naar de blauwe kant (effect 2, de piek
$\lambda_{\max}$ krimpt), zodat steeds meer van de zichtbare band
sterk verlicht is --- een volle band leest als wit.

\emph{3.} Een lichaam dat wit gloeit, zendt sterk uit over de hele
zichtbare band; een lichaam dat rood gloeit, haalt alleen het
\omterm{def:g10:light-spectra:wavelength}{langgolvige} uiteinde. Volgens de wet van Wien is de eerste de
hetere.
\end{solution}

\begin{solution}{exo:g10:light-spectra:6}
\emph{1.}
\[
T = \frac{\qty{2.90e-3}{m.K}}{\qty{5.00e-7}{m}} = \qty{5800}{K}.
\]

\emph{2.} $\theta = 5800 - 273 = \qty{5527}{\degreeCelsius} \approx
\qty{5500}{\degreeCelsius}$.

\emph{3.} $\qty{400}{nm} < \qty{500}{nm} < \qty{800}{nm}$: de piek is
zichtbaar, in het groen.
\end{solution}

\begin{solution}{exo:g10:light-spectra:7}
\emph{1.} Kaars:
$\lambda_{\max} = \qty{2.90e-3}{m.K} / \qty{1800}{K} = \qty{1.61e-6}{m}
= \qty{1.6}{\micro m}$, in het \omterm{def:g10:light-spectra:wavelength}{infrarood}. Kookplaat:
$\lambda_{\max} = \qty{2.90e-3}{m.K} / \qty{500}{K} = \qty{5.8e-6}{m}
= \qty{5.8}{\micro m}$, ver \omterm{def:g10:light-spectra:wavelength}{infrarood}.

\emph{2.} Bij $\qty{500}{K}$ is vrijwel alle straling
\omterm{def:g10:light-spectra:wavelength}{infrarood}: de huid absorbeert ze en voelt warmte, maar het
oog ontvangt niets --- heet, toch zwart.

\emph{3.} Een \omterm{def:g10:light-spectra:continuous}{continu spectrum} is breed: zelfs met piek bij
$\qty{1.6}{\micro m}$ houdt het \omterm{def:g10:light-spectra:white}{spectrum} van de kaars een
zichtbare staart aan het rood-gele uiteinde --- zwak, maar genoeg voor
het donker-geadapteerde oog.
\end{solution}

\begin{solution}{exo:g10:light-spectra:8}
Lamp A: de lijnen $434$, $486$, $\qty{656}{nm}$ matchen waterstof, en
de zwakke violette lijn voltooit de catalogus bij $\qty{410}{nm}$:
waterstof.

Lamp B: de lijn zit bij $\qty{589}{nm}$, de enige sterke zichtbare
lijn van natrium. Helium is uitgesloten niet door de lijn zelf (helium
heeft wel een lijn bij de nabije $\qty{588}{nm}$) maar door de
\emph{ontbrekende}: helium zou ook $447$, $502$ en $\qty{668}{nm}$
tonen, en geen daarvan verschijnt. Een element wordt geïdentificeerd
door zijn volle catalogus: lamp B bevat natrium.
\end{solution}

\begin{solution}{exo:g10:light-spectra:9}
\emph{1.} Een continue regenboog doorboord door één dunne donkere
lijn: het \omterm{def:g10:light-spectra:absorption}{absorptiespectrum} van natrium.

\emph{2.} Bij $\qty{589}{nm}$: een gas absorbeert precies de
\omterm{def:g10:light-spectra:wavelength}{golflengten} die het kan uitzenden
(\cref{prop:g10:light-spectra:kirchhoff}), en de sterke zichtbare lijn
van natrium zit bij $\qty{589}{nm}$.

\emph{3.} Het \omterm{def:g10:light-spectra:emission}{emissiespectrum} van natrium: één heldere
geeloranje lijn bij $\qty{589}{nm}$ op een zwarte achtergrond ---
dezelfde \omterm{def:g10:light-spectra:wavelength}{golflengte}, met de twee spectra die helder en donker
uitwisselen.
\end{solution}

\begin{solution}{exo:g10:light-spectra:10}
\emph{1.} $410$, $434$, $486$, $\qty{656}{nm}$: de volledige
waterstofcatalogus. $517$ en $\qty{518}{nm}$: het magnesiumpaar. Beide
elementen zijn aanwezig.

\emph{2.} In de atmosfeer van de Zon, het koelere en dunnere gas boven
het gloeiende oppervlak. Het oppervlak stuurt een
\omterm{def:g10:light-spectra:continuous}{continu spectrum} omhoog; de atmosfeer haalt haar eigen
\omterm{def:g10:light-spectra:wavelength}{golflengten} uit het passerende licht, zodat de lijnen als
ontbrekend licht verschijnen --- donker tegen de heldere achtergrond.
\end{solution}

\begin{solution}{exo:g10:light-spectra:11}
\emph{1.} $T = 37 + 273 = \qty{310}{K}$, dus
\[
\lambda_{\max} = \frac{\qty{2.90e-3}{m.K}}{\qty{310}{K}}
= \qty{9.4e-6}{m} = \qty{9.4}{\micro m}.
\]

\emph{2.} Ver \omterm{def:g10:light-spectra:wavelength}{infrarood} --- meer dan tien keer de
\omterm{def:g10:light-spectra:wavelength}{golflengte} van rood licht.

\emph{3.} Bij $\qty{310}{K}$ is het zichtbare deel van de straling
volkomen verwaarloosbaar: geen zichtbare gloed, hoe donker de kamer
ook. Een thermische camera heeft een sensor afgestemd op de straling
die werkelijk wordt uitgezonden, rond $\qty{9}{\micro m}$, en ziet
warme lichamen stralen tegen koelere omgevingen.
\end{solution}

\begin{solution}{exo:g10:light-spectra:12}
\emph{1.} $\lambda_{\max} = \qty{2.90e-3}{m.K} / \qty{2700}{K}
= \qty{1.07e-6}{m} \approx \qty{1.1}{\micro m}$: nabij
\omterm{def:g10:light-spectra:wavelength}{infrarood}.

\emph{2.} Het \omterm{def:g10:light-spectra:white}{spectrum} piekt voorbij het zichtbare: het merendeel
van het elektrische vermogen keert terug als onzichtbaar
\omterm{def:g10:light-spectra:wavelength}{infrarood} --- dat wil zeggen warmte --- en alleen een
kleine zichtbare staart verlicht de kamer. Als lamp verspilt de gloeilamp
het meeste van haar vermogen; als kacheltje is ze bijna perfect.

\emph{3.} $T = \qty{2.90e-3}{m.K} / \qty{5.50e-7}{m} \approx
\qty{5300}{K}$ --- ver boven het smeltpunt van wolfraam van
$\qty{3700}{K}$. Geen vaste gloeidraad kan tot daglichttemperaturen
worden geduwd, dus daglichtachtige lampen moesten de gloeitechniek
helemaal verlaten (fluorescentiebuizen, leds), en zichtbaar licht
produceren zonder een lichaam tot $\qty{5300}{K}$ te verhitten.
\end{solution}

\begin{solution}{exo:g10:light-spectra:13}
\emph{1.} $T = \qty{2.90e-3}{m.K} / \qty{4.20e-7}{m} \approx
\qty{6900}{K}$: heter dan de $\qty{5800}{K}$ van de Zon.

\emph{2.} Waterstof (de volledige reeks $410$, $434$, $486$,
$\qty{656}{nm}$ is aanwezig) en calcium (beide lijnen, $393$ en
$\qty{397}{nm}$).

\emph{3.} Nee. De natriumlijn bij $\qty{589}{nm}$ ontbreekt, terwijl
overvloedig natrium in de atmosfeer die noodzakelijk op de continue
achtergrond zou drukken. Als er natrium is, is het alleen in sporen te
zwak om te detecteren.
\end{solution}

\begin{solution}{exo:g10:light-spectra:14}
\emph{1.} Waterstof verklaart $410$, $434$, $486$ en $\qty{656}{nm}$;
helium verklaart $447$, $502$, $588$ en $\qty{668}{nm}$. Elke
waargenomen lijn is verantwoord: het mengsel is waterstof en helium.

\emph{2.} Helium is een \emph{voldoende} verklaring omdat zijn drie
andere lijnen ($447$, $502$, $\qty{668}{nm}$) allemaal aanwezig zijn
--- de lijn bij $\qty{588}{nm}$ is dan te verwachten van helium alleen,
en niets in het \omterm{def:g10:light-spectra:white}{spectrum} \emph{vereist} natrium. Om de zaak te
beslechten is resolutie nodig, geen argument: een fijnere
\omterm{def:g10:light-spectra:white}{spectroscoop} scheidt de enkele lijn van helium bij
$\qty{587.6}{nm}$ van die van natrium, die in feite een nauw paar is bij
$589.0$ en $\qty{589.6}{nm}$. Eén lijn daar vrijpleit natrium; drie
veroordeelt het.
\end{solution}

\begin{solution}{exo:g10:light-spectra:15}
\emph{1.} Betelgeuze:
$T = \qty{2.90e-3}{m.K} / \qty{8.5e-7}{m} \approx \qty{3400}{K}$. Rigel:
$T = \qty{2.90e-3}{m.K} / \qty{2.4e-7}{m} \approx \qty{12000}{K}$.

\emph{2.} Omdat $\lambda_{\max} \times T$ dezelfde constante is voor
beide,
$\dfrac{T_{\text{Rigel}}}{T_{\text{Betelgeuse}}}
= \dfrac{\qty{850}{nm}}{\qty{240}{nm}} \approx 3.5$.

\emph{3.} Betelgeuze piekt in het \omterm{def:g10:light-spectra:wavelength}{infrarood}: binnen de
zichtbare band is haar \omterm{def:g10:light-spectra:white}{spectrum} het sterkst aan het rode
uiteinde, dus ziet de ster er rood uit. Rigel piekt in het
\omterm{def:g10:light-spectra:wavelength}{ultraviolet}: haar zichtbare band is het sterkst aan het
blauwe uiteinde, vandaar blauw-wit.

\emph{4.} Een \omterm{def:g10:light-spectra:continuous}{continu spectrum} bestrijkt een enorm bereik van
\omterm{def:g10:light-spectra:wavelength}{golflengten}. Zelfs met haar piek buiten de zichtbare band
stort een ster voldoende licht \emph{in} die band --- de piek bepaalt
de tint, niet de zichtbaarheid.
\end{solution}

\begin{solution}{pb:g10:light-spectra:1}
\textbf{1.} Van ongeveer $\qty{400}{nm}$ (violet uiteinde) tot
ongeveer $\qty{800}{nm}$ (rood uiteinde); $\qty{550}{nm}$ zit ongeveer
in het midden, in het groen.

\textbf{2.} $\qty{410}{nm}$: violet; $\qty{434}{nm}$: violetblauw;
$\qty{486}{nm}$: blauwgroen; $\qty{656}{nm}$: rood.

\textbf{3.} Een \omterm{def:g10:light-spectra:white}{monochromatische} geeloranje gloed. De kleur van
een auto is het mengsel van \omterm{def:g10:light-spectra:wavelength}{golflengten} dat haar lak
reflecteert; onder een lamp die \emph{alleen} $\qty{589}{nm}$ biedt,
kan elke lak alleen meer of minder van die ene
\omterm{def:g10:light-spectra:wavelength}{golflengte} reflecteren --- alle auto's verschijnen in
tinten oranje en grijs.

\textbf{4.} Elk element zendt een vaste catalogus van lijnen uit, en
geen twee elementen delen dezelfde catalogus
(\cref{prop:g10:light-spectra:fingerprint}): een complete reeks lijnen
identificeert dus het element als een vingerafdruk. Maar een
\emph{gedeeltelijke} match bewijst weinig --- verschillende elementen
kunnen bijna samenvallende enkele lijnen bezitten (helium bij
$\qty{588}{nm}$, natrium bij $\qty{589}{nm}$) --- dus de
aanwezigheidsclaim vereist alle sterke lijnen van de catalogus.

\textbf{5.} (a) Een continue regenboog met een donkere lijn bij
$\qty{589}{nm}$: het \omterm{def:g10:light-spectra:absorption}{absorptiespectrum}. (b) Eén heldere
lijn bij $\qty{589}{nm}$ op zwart: het \omterm{def:g10:light-spectra:emission}{emissiespectrum} ---
dezelfde \omterm{def:g10:light-spectra:wavelength}{golflengte}, helder en donker uitgewisseld
(\cref{prop:g10:light-spectra:kirchhoff}).

\textbf{6.} Het \omterm{def:g10:light-spectra:white}{spectrum} wordt helderder bij elke
\omterm{def:g10:light-spectra:wavelength}{golflengte}, en haar piek $\lambda_{\max}$ beweegt naar
kortere \omterm{def:g10:light-spectra:wavelength}{golflengten}, volgens
$\lambda_{\max} \times T = \qty{2.90e-3}{m.K}$
(\cref{prop:g10:light-spectra:wien}).

\textbf{7.} $\lambda_{\max} = \qty{2.90e-3}{m.K} / \qty{3400}{K}
= \qty{8.5e-7}{m} = \qty{850}{nm}$: nabij \omterm{def:g10:light-spectra:wavelength}{infrarood}, net
voorbij de rode rand --- daarom is halogeenlicht warmer van toon dan
daglicht.

\textbf{8.} $T = \qty{2.90e-3}{m.K} / \qty{5.00e-7}{m} = \qty{5800}{K}$.

\textbf{9.} De piek is zacht, geen piek in de enge zin: de Zon zendt de
hele zichtbare band uit met vergelijkbare sterkte. Een volle
kleurenband gemengd leest als wit --- de Zon ziet er wit uit
(gelig door onze atmosfeer), nooit groen.

\textbf{10.} Koude pook: piek diep in het
\omterm{def:g10:light-spectra:wavelength}{infrarood}, geen zichtbare staart --- straling wordt
gevoeld, niets wordt gezien. Heter: het helder wordende
\omterm{def:g10:light-spectra:white}{spectrum} duwt een eerste zichtbare staart voorbij
$\qty{800}{nm}$ --- het rode uiteinde alleen licht op: dof rood.
Nog heter: de hele zichtbare band vult zich terwijl alles helderder
wordt: oranje, dan naar wit.

\textbf{11.} De continue achtergrond komt van het dichte, hete
gloeiende oppervlak; de donkere lijnen worden gedrukt door de koelere,
dunnere atmosfeer erboven, die haar eigen
\omterm{def:g10:light-spectra:wavelength}{golflengten} uit het passerende licht absorbeert.

\textbf{12.} Waterstof ($410$, $434$, $486$, $\qty{656}{nm}$:
compleet), natrium ($\qty{589}{nm}$) en calcium ($393$ en
$\qty{397}{nm}$: beide) --- alle drie zijn aanwezig in de atmosfeer
van de Zon.

\textbf{13.} Dat de atmosfeer door waterstof wordt gedomineerd. Moderne
metingen bevestigen en kwantificeren: de Zon is grofweg driekwart
waterstof naar massa, het merendeel van de rest is helium.

\textbf{14.} De lijn matchte met geen bekend element; omdat de
catalogus van elk element vast en uniek is, moet een lijn die tot
geen catalogus behoort tot een element behoren dat nog nooit in een
laboratorium gezien is. Een nieuw element werd aangekondigd en helium
genoemd, naar \emph{helios}, de Zon. Het werd in 1895 op Aarde
geïsoleerd --- de voorspelling bevestigd, zevenentwintig jaar later.

\textbf{15.} Tegen de schitterende continue achtergrond
\emph{verwijdert} de atmosfeer licht bij haar eigen
\omterm{def:g10:light-spectra:wavelength}{golflengten}: donkere lijnen. Tijdens de eclips
blokkeert de Maan de achtergrond; de dunne atmosfeer staat dan alleen
tegen een donkere hemel en haar heruitgezonden licht is alles wat er
te zien is: dezelfde \omterm{def:g10:light-spectra:wavelength}{golflengten}, nu helder. Eén gas, één
catalogus --- het teken hangt af van wat erachter staat.

\textbf{16.} $T = \qty{2.90e-3}{m.K} / \qty{2.90e-7}{m}
= \qty{10000}{K}$.

\textbf{17.} Blauw-wit: binnen de zichtbare band is het
\omterm{def:g10:light-spectra:white}{spectrum} het sterkst aan het blauwe uiteinde. En geen
onzichtbaarheid: het \omterm{def:g10:light-spectra:continuous}{continue spectrum} overstroomt de
hele zichtbare band op weg naar de \omterm{def:g10:light-spectra:wavelength}{ultraviolette} piek ---
de piek bepaalt de tint, niet de helderheid.

\textbf{18.} De donkere lijnen $410$, $434$, $486$, $\qty{656}{nm}$
zijn de volledige waterstofvingerafdruk: waterstof domineert de
atmosfeer van Vega.

\textbf{19.} $T_{\text{Vega}} / T_{\text{Sun}} = 10000 / 5800 \approx
1.7$: Vega is heter, dus blauwer (piek bij $\qty{290}{nm}$ tegen
$\qty{500}{nm}$) en, volgens het eerste deel van
\cref{prop:g10:light-spectra:wien}, helderder per
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} gloeiend oppervlak bij elke
\omterm{def:g10:light-spectra:wavelength}{golflengte}.

\textbf{20.} Identiteitsformulier van Vega ---
oppervlaktetemperatuur: ongeveer $\qty{10000}{K}$; dominant element
van de atmosfeer: waterstof; schijnbare kleur: blauw-wit. Alles
gelezen uit één bundel licht, met een prisma, een catalogus van
laboratoriumlijnen en één deling. Comte's «nooit» werd uitgevaardigd
in 1835 en verliep in 1859: het duurde vierentwintig jaar.
\end{solution}
