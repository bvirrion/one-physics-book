\section*{Hoofdstuk \ref{ch:g10:refraction} --- Breking van licht}

\begin{solution}{exo:g10:refraction:1}
Hoeken meet men vanaf de \omterm{def:g10:refraction:angles}{normaal}: de
\omterm{def:g10:refraction:angles}{invalshoek} is $90^\circ - 25^\circ = 65^\circ$, dus de
reflectiehoek is ook $65^\circ$. De invallende en gereflecteerde
bundels maken een hoek van $2 \times 65^\circ = 130^\circ$ met
elkaar.
\end{solution}

\begin{solution}{exo:g10:refraction:2}
\emph{1.} $v = c/n$: in water
$v = \qty{3.00e8}{m/s}/1.33 = \qty{2.26e8}{m/s}$; in diamant
$v = \qty{3.00e8}{m/s}/2.42 = \qty{1.24e8}{m/s}$.

\emph{2.} $n = \dfrac{c}{v} = \dfrac{\qty{3.00e8}{m/s}}
{\qty{1.97e8}{m/s}} = 1.52$: glas.
\end{solution}

\begin{solution}{exo:g10:refraction:3}
$\sin i_2 = \dfrac{\sin 40^\circ}{1.50} = \dfrac{0.643}{1.50} = 0.429$,
dus $i_2 = 25.4^\circ$. Door omkeerbaarheid van lichtpaden treedt een
straal binnenin het glas die het oppervlak onder $25.4^\circ$ treft
in de lucht uit onder precies $40^\circ$.
\end{solution}

\begin{solution}{exo:g10:refraction:4}
$\sin i_2 = \dfrac{\sin 55^\circ}{1.33} = \dfrac{0.819}{1.33} = 0.616$,
dus $i_2 = 38.0^\circ$. De straal buigt \emph{naar} de
\omterm{def:g10:refraction:angles}{normaal} toe, omdat water het tragere medium is
($n_2 > n_1$).
\end{solution}

\begin{solution}{exo:g10:refraction:5}
$n = \dfrac{\sin 45^\circ}{\sin 32^\circ} = \dfrac{0.707}{0.530}
= 1.33$: de vloeistof is water.
\end{solution}

\begin{solution}{exo:g10:refraction:6}
\emph{1.} De vijf verhoudingen zijn
\[
\frac{0.342}{0.233} = 1.47,\quad
\frac{0.500}{0.334} = 1.50,\quad
\frac{0.643}{0.431} = 1.49,\quad
\frac{0.766}{0.515} = 1.49,\quad
\frac{0.866}{0.581} = 1.49 :
\]
constant binnen meetfout --- de \omterm{thm:g10:refraction:snell}{wet van Snellius}.

\emph{2.} $n \approx 1.49$: plexiglas.

\emph{3.} De \omterm{thm:g10:refraction:snell}{wet van Snellius} voorspelt
$\sin i_2 = \sin i_1 / n$: sinus tegen sinus moeten de data op een
rechte lijn door de oorsprong liggen, wat een liniaal in één oogopslag
controleert. Hoek tegen hoek geplot liggen de data op een kromme; bij
kleine hoeken lijkt de kromme bedrieglijk op een rechte lijn, precies
wat Ptolemaeus misleidde.
\end{solution}

\begin{solution}{exo:g10:refraction:7}
$\sin i_c = n_2/n_1$ met $n_2 = 1.00$:
water $\sin i_c = 1/1.33 = 0.752$, $i_c = 48.8^\circ$;
glas $\sin i_c = 1/1.50 = 0.667$, $i_c = 41.8^\circ$;
diamant $\sin i_c = 1/2.42 = 0.413$, $i_c = 24.4^\circ$.

Van lucht naar water is $\sin i_2 = \sin i_1/1.33 \leq 1/1.33 < 1$
voor elke inval: er bestaat altijd een brekingshoek, dus gebeurt er
nooit iets bijzonders --- \omterm{def:g10:refraction:critical}{totale interne reflectie}
vereist $n_1 > n_2$.
\end{solution}

\begin{solution}{exo:g10:refraction:8}
\emph{1.} $d' = \dfrac{d}{n} = \dfrac{\qty{2.0}{m}}{1.33}
= \qty{1.5}{m}$.

\emph{2.} De vis lijkt op $\qty{40}{cm}/1.33 = \qty{30}{cm}$: het
beeld is \emph{boven} de vis, dus moet de reiger \emph{onder} het
beeld toeslaan dat hij ziet. Reigers, onbelast door de
\omterm{thm:g10:refraction:snell}{wet van Snellius}, leren dit door vallen en opstaan.
\end{solution}

\begin{solution}{exo:g10:refraction:9}
\emph{1.} $\sin r = \dfrac{\sin 50^\circ}{1.50} = \dfrac{0.766}{1.50}
= 0.511$, dus $r = 30.7^\circ$.

\emph{2.} Glas naar lucht onder $30.7^\circ$:
$\sin i_2 = 1.50 \times 0.511 = 0.766$, dus $i_2 = 50^\circ$ --- de
uittredehoek is gelijk aan de intredehoek.

\emph{3.} De twee \omterm{def:g10:refraction:refraction}{brekingen} heffen elkaar op: de straal
vertrekt \emph{evenwijdig} aan haar oorspronkelijke richting, slechts
zijdelings verschoven. Daarom vervormt een raam het tafereel niet,
alleen (onmerkbaar) verschuift het.
\end{solution}

\begin{solution}{exo:g10:refraction:10}
$\sin 55^\circ = 0.819$. Rood:
$\sin i_2 = 0.819/1.61 = 0.509$, $i_2 = 30.6^\circ$. Blauw:
$\sin i_2 = 0.819/1.66 = 0.493$, $i_2 = 29.6^\circ$. De twee kleuren
zijn $1.0^\circ$ gescheiden; blauw, met de grotere index, buigt meer
en eindigt dichter bij de \omterm{def:g10:refraction:angles}{normaal}.
\end{solution}

\begin{solution}{exo:g10:refraction:11}
\emph{1.} $\sin i_c = \dfrac{1.46}{1.48} = 0.986$, dus
$i_c = 80.6^\circ$.

\emph{2.} Een straal onder $i_c$ van de \omterm{def:g10:refraction:angles}{normaal} maakt
$90^\circ - i_c$ met de as, dus haar pad over aslengte $L$ is
$\dfrac{L}{\sin i_c}$:
\[
\qty{1000}{m} \times \frac{1.48}{1.46} = \qty{1013.7}{m},
\]
ongeveer \qty{14}{m} meer dan de assstraal.
\end{solution}

\begin{solution}{exo:g10:refraction:12}
\emph{1.} Licht dat van de horizon scheert, arriveert met $i_1$
dicht bij $90^\circ$; het breekt tot $\sin i_2 = \sin 90^\circ / 1.33
= 0.752$, d.w.z.\ $i_2 = 48.8^\circ$ --- de
\omterm{def:g10:refraction:critical}{grenshoek}, zoals omkeerbaarheid eist. Stralen van de
hele hemel bereiken het oog dus binnen een kegel van halfhoek
$48.8^\circ$.

\emph{2.} De rand van de schijf is waar die kegel het oppervlak
treft:
\[
r = d \tan i_c = \qty{2.0}{m} \times \tan 48.8^\circ
= \qty{2.0}{m} \times 1.14 = \qty{2.3}{m}.
\]

\emph{3.} Buiten de schijf treft licht van onderen het oppervlak
voorbij de \omterm{def:g10:refraction:critical}{grenshoek} en wordt totaal gereflecteerd: de
duiker ziet een zilveren spiegel die de bodem van het bad toont.
\end{solution}

\begin{solution}{exo:g10:refraction:13}
\emph{1.} Rood: $\sin i_2 = 1.51 \times \sin 30^\circ = 0.755$, dus
$i_2 = 49.0^\circ$. Blauw: $\sin i_2 = 1.53 \times 0.500 = 0.765$,
dus $i_2 = 49.9^\circ$. De witte bundel waaiert uit over
$49.9^\circ - 49.0^\circ = 0.9^\circ$ --- een prisma-\omterm{def:g10:light-spectra:white}{spectrum}
uit één \omterm{def:g10:refraction:refraction}{breking}.

\emph{2.} Bij $45^\circ$: $\sin i_2 = 1.51 \times 0.707 = 1.07 > 1$
(en $1.53 \times 0.707 = 1.08$ voor blauw). Er bestaat geen
brekingshoek voor geen van beide kleuren: de
\omterm{def:g10:refraction:critical}{grenshoek} is $\arcsin(1/1.51) = 41.5^\circ < 45^\circ$,
dus het lange vlak reflecteert de hele bundel totaal.
\end{solution}

\begin{solution}{exo:g10:refraction:14}
\emph{1.} \omterm{def:g10:refraction:critical}{Grenshoek} glas--lucht: $\sin i_c = 1/1.50$, dus
$i_c = 41.8^\circ < 45^\circ$: de bundel wordt totaal gereflecteerd
--- $100\%$ van het licht, zonder coating die kan aantasten. Een
metalen spiegel verliest enkele procenten bij elke stuit, en een
periscoop heeft er twee nodig.

\emph{2.} Glas--water: $\sin i_c = \dfrac{1.33}{1.50} = 0.887$, dus
$i_c = 62.5^\circ$. Nu is $45^\circ < i_c$: de inval ligt onder de
grens, het merendeel van het licht breekt naar het water, en de
spiegel --- dus het beeld --- faalt.

\emph{3.} $\sin i_2 = \dfrac{1.50}{1.33} \times \sin 45^\circ
= 1.128 \times 0.707 = 0.798$, dus $i_2 = 52.9^\circ$ naar het
water.
\end{solution}

\begin{solution}{exo:g10:refraction:15}
\emph{1.} $v = \dfrac{\qty{3.00e8}{m/s}}{1.47} = \qty{2.04e8}{m/s}$,
dus
\[
t = \frac{\qty{6.2e6}{m}}{\qty{2.04e8}{m/s}} = \qty{30.4}{ms} \approx
\qty{30}{ms}.
\]

\emph{2.} Radio: $t = \qty{6.2e6}{m}/\qty{3.00e8}{m/s} =
\qty{20.7}{ms}$. Het glas kost $30.4 - 20.7 = \qty{9.7}{ms}$ per
richting, ongeveer \qty{19}{ms} per retour.

\emph{3.} Verlaag de index --- \omterm{def:g10:refraction:fiber}{holle-kern}-vezels leiden
licht door lucht ($n \approx 1.00$), en microgolfverbindingen door de
atmosfeer doen hetzelfde --- of verkort het pad: een kabel dichter
bij de grootcirkelroute tussen de twee steden. Beide worden echt
verkocht, tegen prijzen per milliseconde.
\end{solution}

\begin{solution}{pb:g10:refraction:1}
\textbf{1.} $v_{\text{water}} = \qty{3.00e8}{m/s}/1.33 =
\qty{2.26e8}{m/s}$; $v_{\text{diamond}} = \qty{3.00e8}{m/s}/2.42 =
\qty{1.24e8}{m/s}$ --- diamant vertraagt licht met de factor $n = 2.42$.

\textbf{2.} $\sin i_2 = \sin 40^\circ / 1.33 = 0.643/1.33 = 0.483$,
dus $i_2 = 28.9^\circ$.

\textbf{3.} $\sin i_2 = 1.33 \times \sin 28.9^\circ = 1.33 \times
0.483 = 0.643$, dus $i_2 = 40^\circ$: licht hertekent zijn pad ---
omkeerbaarheid van lichtstralen.

\textbf{4.} $\sin i_c = 1/1.33 = 0.752$, dus $i_c = 48.8^\circ$. Bij
$60^\circ > i_c$ zou de \omterm{thm:g10:refraction:snell}{wet van Snellius}
$\sin i_2 = 1.33 \sin 60^\circ = 1.15 > 1$ nodig hebben: geen
\omterm{def:g10:refraction:refraction}{gebroken straal}; het oppervlak reflecteert het licht
totaal terug naar beneden.

\textbf{5.} $v = \qty{3.00e8}{m/s}/1.0003 = \qty{2.999e8}{m/s}$: licht
in lucht is $0.03\%$ trager dan in vacuüm, ver onder de precisie van
onze hoekmetingen, dus $n_{\text{air}} = 1.00$ is onschuldig.

\textbf{6.} $d' = \qty{3.0}{m}/1.33 = \qty{2.3}{m}$: het bad verbergt
\qty{70}{cm} van zichzelf.

\textbf{7.} Twee stralen die de munt verlaten breken aan het
oppervlak, buigend \emph{van} de \omterm{def:g10:refraction:angles}{normaal} \emph{af},
zodat ze steiler divergeren boven het water dan eronder. Het oog
verlengt de twee uittredende stralen achteruit in rechte lijnen; ze
snijden \emph{boven} de munt, en dat snijpunt is waar het oog het
beeld plaatst.

\textbf{8.} $r = d \tan i_c = \qty{3.0}{m} \times \tan 48.8^\circ =
\qty{3.0}{m} \times 1.14 = \qty{3.4}{m}$: een schijf van ongeveer
\qty{6.9}{m} doorsnede houdt de hele hemel.

\textbf{9.} Bij $30^\circ$: $\sin i_2 = 1.33 \times 0.500 = 0.665$,
de straal treedt uit onder $41.7^\circ$. Bij $45^\circ$:
$\sin i_2 = 1.33 \times 0.707 = 0.940$, ze treedt uit onder
$70.1^\circ$, bijna scherend. Bij $50^\circ$:
$1.33 \times 0.766 = 1.02 > 1$ --- \omterm{def:g10:refraction:critical}{totale interne
reflectie}, de straal duikt terug in het bad.

\textbf{10.} Het beeld is boven de echte vis
(\cref{prop:g10:refraction:depth}), dus mikt men \emph{onder} haar. En
ja: de vis ziet de visser ook verplaatst, samengeperst naar de
verticale binnen het venster van Snellius --- elk ziet de ander waar
de ander niet is.

\textbf{11.} Diamant: $\sin i_c = 1/2.42 = 0.413$, dus
$i_c = 24.4^\circ$, tegen $41.1^\circ$ voor glas: de
ontsnappingskegel van diamant is niet eens half zo breed.

\textbf{12.} In diamant is $30^\circ > 24.4^\circ$:
\omterm{def:g10:refraction:critical}{totale interne reflectie}, de straal blijft in de steen.
In glas is $30^\circ < 41.1^\circ$: de straal lekt uit de achterkant.
De facetten van een briljant-slijping zijn zo gehoekt dat invallend
licht ze bijna altijd voorbij $24.4^\circ$ treft en stuit tot het
omhoog naar het oog uittreedt --- een glazen imitatie, met haar
brede ontsnappingskegel, laat het licht er eenvoudig doorheen vallen.

\textbf{13.} $\sin 45^\circ = 0.707$. Rood: $0.707/2.41 = 0.293$, dus
$i_2 = 17.1^\circ$. Blauw: $0.707/2.45 = 0.289$, dus
$i_2 = 16.8^\circ$. Splitsing bij intrede: ongeveer $0.3^\circ$.

\textbf{14.} $\sin 17.0^\circ = 0.292$. Rood: $\sin i_2 = 2.41 \times
0.292 = 0.705$, dus $i_2 = 44.8^\circ$. Blauw: $\sin i_2 = 2.45 \times
0.292 = 0.716$, dus $i_2 = 45.8^\circ$. De waaier is nu $1.0^\circ$
breed: de uittrede-\omterm{def:g10:refraction:refraction}{breking}, werkend nabij de
\omterm{def:g10:refraction:critical}{grenshoek}, rekte de $0.3^\circ$-intredesplitsing
verdrievoudigd. Die gerekte kleurenwaaier is het \emph{vuur} van de
diamant.

\textbf{15.} Diamant--water: $\sin i_c = 1.33/2.42 = 0.550$, dus
$i_c = 33.3^\circ$. De $30^\circ$-straal, vast toen het facet lucht
aankeek, ontsnapt nu ($30^\circ < 33.3^\circ$) in de waterfilm:
elke druppel, vingerafdruk of vetfilm verhoogt de buitenindex,
verbreedt de ontsnappingskegel en lekt de fonkeling weg --- vandaar
de juweliersdoek.

\textbf{16.} $\sin i_c = \dfrac{1.46}{1.48} = 0.986$, dus
$i_c = 80.6^\circ$.

\textbf{17.} \omterm{def:g10:refraction:critical}{Totale interne reflectie} gebeurt \emph{aan het
oppervlak}, dus moet het oppervlak perfect blijven: op een kale
draad verhoogt elk stofje, kras, waterdruppel of aanrakende vinger
de buitenindex op dat punt en bloedt licht weg. De
\omterm{def:g10:refraction:fiber}{mantel} begraaft de spiegel in schoon glas waar niets
haar kan bereiken --- en beschermt bovendien de haardunne
\omterm{def:g10:refraction:fiber}{kern} mechanisch.

\textbf{18.} $v = \qty{3.00e8}{m/s}/1.48 = \qty{2.03e8}{m/s}$, dus
\[
t = \frac{\qty{6.0e6}{m}}{\qty{2.03e8}{m/s}} = \qty{29.6}{ms} \approx
\qty{30}{ms}.
\]

\textbf{19.} Padfactor $1/\sin i_c = 1.48/1.46 = 1.0137$: in het
slechtste geval $\qty{6000}{km} \times 0.0137 \approx \qty{82}{km}$
extra glas, kostend $\qty{8.2e4}{m} / \qty{2.03e8}{m/s} \approx
\qty{0.4}{ms}$ --- de zigzag is bijna gratis.

\textbf{20.} Ping: $2 \times (29.6 + 0.4) \approx \qty{60}{ms}$.
Doorvoer: $\dfrac{10^{13}}{2.4 \times 10^8} \approx 40\,000$
kopieën van dit boek per seconde. In één zin: door licht in glas te
vangen via \omterm{def:g10:refraction:critical}{totale interne reflectie} draagt de
\omterm{thm:g10:refraction:snell}{wet van Snellius} elke seconde een bibliotheek over de
oceaan, zestig milliseconden heen en terug.
\end{solution}
