\section*{Hoofdstuk \ref{ch:g10:inertia} --- Krachten en het traagheidsprincipe}

\begin{solution}{exo:g10:inertia:1}
(a) \omterm{def:g10:universal-gravitation:weight}{Gewicht} (op afstand), \omterm{def:g10:inertia:inventory}{normaalkracht} van de tafel (contact).
(b) \omterm{def:g10:universal-gravitation:weight}{Gewicht} (op afstand), \omterm{def:g10:inertia:inventory}{spankracht} van de draad (contact).
(c) \omterm{def:g10:universal-gravitation:weight}{Gewicht} (op afstand), \omterm{def:g10:inertia:inventory}{normaalkracht} van het ijs (contact); \omterm{def:g10:inertia:inventory}{wrijving}
verwaarloosbaar. (d) Alleen \omterm{def:g10:universal-gravitation:weight}{gewicht} --- niets raakt de appel.
\end{solution}

\begin{solution}{exo:g10:inertia:2}
Boek: $P = 0.250 \times 9.81 \approx \qty{2.45}{N}$, een pijl van ongeveer
\qty{2.5}{cm}. Persoon: $P = 60 \times 9.81 \approx \qty{589}{N}$
--- een pijl van \qty{5.89}{m}. Kies een schaal die past bij de grootste
\omterm{def:g10:inertia:force}{kracht} op de pagina.
\end{solution}

\end{solution}

\begin{solution}{exo:g10:inertia:4}
\omterm{def:g10:universal-gravitation:weight}{Gewicht} $\vect P$: verticaal, omlaag, $P = 1.2 \times 9.81 \approx
\qty{11.8}{N}$. \omterm{def:g10:inertia:inventory}{Normaalkracht} $\vect R$ van de plank: dezelfde lijn,
omhoog, $R = \qty{11.8}{N}$ (het woordenboek is in rust, dus de twee

\end{solution}

\begin{solution}{exo:g10:inertia:5}
(a) Onwaar: \omterm{def:g10:inertia:inventory}{wrijving} stopt dingen; zonder enige \omterm{def:g10:inertia:force}{kracht} houdt het
lichaam zijn \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging}. (b) Waar
(traagheidsprincipe). (c) Onwaar: het is onderhevig aan \omterm{def:g10:inertia:force}{krachten} die
\emph{\omterm{def:g10:inertia:compensate}{compenseren}} (bijv.\ \omterm{def:g10:universal-gravitation:weight}{gewicht} en reactie). (d) Waar --- de
contrapositieve lezing van het principe.
\end{solution}

\begin{solution}{exo:g10:inertia:6}
\emph{1.} $v = 24/3.0 = \qty{8.0}{m/s}$. \omterm{def:g10:inertia:force}{Krachten}: \omterm{def:g10:universal-gravitation:weight}{gewicht} en
\omterm{def:g10:inertia:inventory}{normaalkracht}; ze \omterm{def:g10:inertia:compensate}{compenseren} (\omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging},
\omterm{def:g10:inertia:inventory}{wrijving} verwaarloosbaar). \emph{2.} \omterm{def:g10:inertia:inventory}{Wrijving} van het beton is niet
langer verwaarloosbaar: de inventaris wint een niet-gecompenseerde
achterwaartse \omterm{def:g10:inertia:force}{kracht}, en inderdaad verandert de snelheid --- de puck
vertraagt en stopt.
\end{solution}

\begin{solution}{exo:g10:inertia:7}
\omterm{def:g10:universal-gravitation:weight}{Gewicht} $P = 70 \times 9.81 \approx \qty{687}{N}$ omlaag; \omterm{def:g10:inertia:force}{kracht}
van de vloer omhoog. \omterm{def:g10:inertia:uniform}{Rechtlijnig eenparige beweging}, dus ze
\omterm{def:g10:inertia:compensate}{compenseren}: de vloer duwt met \qty{687}{N}. Niets verandert in
rust --- rust en eenparige beweging geven hetzelfde
\omterm{def:g10:inertia:diagram}{krachtendiagram}.
\end{solution}

\begin{solution}{exo:g10:inertia:8}
\emph{1.} In rust \omterm{def:g10:inertia:compensate}{compenseert} \omterm{def:g10:inertia:inventory}{spankracht} het
\omterm{def:g10:universal-gravitation:weight}{gewicht}: $T = 1.8 \times 9.81 \approx \qty{17.7}{N}$.
\emph{2.} Elke verticale draad draagt de helft: $T = \qty{8.8}{N}$.
\end{solution}

\begin{solution}{exo:g10:inertia:9}
Remmen: het lichaam van de passagier houdt zijn snelheid (traagheid)
terwijl de bus de zijne verliest, dus beweegt de passagier naar voren
\emph{ten opzichte van de bus}. Bochten: het lichaam neigt rechtuit te
blijven terwijl de bus eronder draait, dus drijft het naar de buitenkant
van de bocht. In beide gevallen is de «\omterm{def:g10:inertia:force}{kracht}» fictief; wat ontbreekt
is een echte \omterm{def:g10:inertia:force}{kracht} (grip, handgreep) om de snelheid van de
passagier mee te veranderen met die van de bus.
\end{solution}

\begin{solution}{exo:g10:inertia:10}
\omterm{def:g10:inertia:uniform}{Rechtlijnig eenparige beweging}, dus \omterm{def:g10:universal-gravitation:weight}{gewicht} en
luchtweerstand \omterm{def:g10:inertia:compensate}{compenseren}: weerstand
$= P = 85 \times 9.81 \approx \qty{834}{N}$, verticaal en omhoog.
\end{solution}

\begin{solution}{exo:g10:inertia:11}
\emph{1.} \omterm{def:g10:inertia:uniform}{Rechtlijnig eenparige beweging}: horizontale
\omterm{def:g10:inertia:force}{krachten} \omterm{def:g10:inertia:compensate}{compenseren}, dus de weerstand is
\qty{300}{N}, tegengesteld aan het touw. \emph{2.} Het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van de skiër en de opwaartse duw van het water op de
ski's.
\end{solution}

\begin{solution}{exo:g10:inertia:12}
\emph{1.} $P = 2.4 \times 9.81 \approx \qty{23.5}{N}$; de verticale
componenten dragen het \omterm{def:g10:universal-gravitation:weight}{gewicht}: $2T\cos 40^\circ = P$, dus
$T = 23.5/(2 \times 0.766) \approx \qty{15.4}{N}$. \emph{2.}
$T = P/(2\cos\theta)$ groeit onbegrensd als $\theta \to 90^\circ$
omdat $\cos\theta \to 0$; exact horizontale draden hebben
\emph{geen} verticale component, dus zou niets het
\omterm{def:g10:universal-gravitation:weight}{gewicht} dragen.
\end{solution}

\begin{solution}{exo:g10:inertia:13}
Duw: \omterm{def:g10:universal-gravitation:weight}{gewicht}, reactie, \omterm{def:g10:inertia:inventory}{stuwkracht} van de hand, kleine
\omterm{def:g10:inertia:inventory}{wrijving}; de snelheid neemt toe, dus \omterm{def:g10:inertia:compensate}{compenseren} de
\omterm{def:g10:inertia:force}{krachten} niet (\cref{prop:g10:inertia:converse}). Glijden:
\omterm{def:g10:universal-gravitation:weight}{gewicht}, reactie, kleine \omterm{def:g10:inertia:inventory}{wrijving}; de steen vertraagt, dus
opnieuw geen compensatie --- \omterm{def:g10:inertia:inventory}{wrijving} is het niet-gecompenseerde
restant (op ideaal ijs zou ze eeuwig glijden). Rust:
\omterm{def:g10:universal-gravitation:weight}{gewicht} en reactie alleen, compenserend
(\cref{thm:g10:inertia:principle}).
\end{solution}

\begin{solution}{exo:g10:inertia:14}
\emph{1.} Volgens het traagheidsprincipe vereist het behouden van een
constante snelheid helemaal geen \omterm{def:g10:inertia:force}{kracht} --- en in de
interstellaire ruimte werkt er bijna geen. Aristoteles zou voorspellen
dat de sonde stopt wanneer de «beweger» ophoudt. \emph{2.} Eén jaar
$\approx \qty{3.156e7}{s}$, dus
$d = 17 \times \num{3.156e7} \approx \qty{5.4e8}{km} \approx
\qty{3.6}{au}$. \emph{3.} Snelheid is een vector: haar richting
veranderen verandert de snelheid, wat een niet-gecompenseerde
\omterm{def:g10:inertia:force}{kracht} vereist.
\end{solution}

\begin{solution}{exo:g10:inertia:15}
In rust \omterm{def:g10:inertia:compensate}{compenseren} de drie \omterm{def:g10:inertia:force}{krachten}:
$\vect F_3 = -(\vect F_1 + \vect F_2)$, dus $\vect F_3\,(-2, -4)$,
grootte $\sqrt{4 + 16} = \sqrt{20} \approx \qty{4.5}{N}$, wijzend
tegengesteld aan de resultante van de eerste twee (omlaag-links op het
rooster).
\end{solution}

\begin{solution}{pb:g10:inertia:1}
\textbf{1.} \omterm{def:g10:inertia:uniform}{Rechtlijnig eenparige beweging} met \qty{0.75}{m/s}.
\omterm{def:g10:inertia:force}{Krachten}: \omterm{def:g10:universal-gravitation:weight}{gewicht} en de \omterm{def:g10:inertia:inventory}{normaalkracht}
van de band (geen \omterm{def:g10:inertia:inventory}{wrijving} nodig: de koffer glijdt niet op de
band).
\textbf{2.} Ja --- \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging}
(\cref{thm:g10:inertia:principle}). $P = 23 \times 9.81 \approx
\qty{226}{N}$, dus $R = \qty{226}{N}$.
\textbf{3.} De diagrammen zijn identiek. Rust en
\omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging} zijn één en dezelfde
mechanische toestand: traagheid onderscheidt ze niet.
\textbf{4.} De koffer houdt haar \qty{0.75}{m/s} terwijl de band stopt:
ze glijdt of kantelt naar voren tot \omterm{def:g10:inertia:inventory}{wrijving} haar beweging
opneemt. Niets duwde haar --- ze hield slechts de snelheid die ze had.
\textbf{5.} Snelheden in dezelfde richting tellen op: $0.75 + 1.2 =
\qty{1.95}{m/s}$. Haar snelheid is constant, dus
\omterm{def:g10:inertia:compensate}{compenseren} de \omterm{def:g10:inertia:force}{krachten} op haar.
\textbf{6.} $P = 0.160 \times 9.81 \approx \qty{1.57}{N}$ omlaag,
$R = \qty{1.57}{N}$ omhoog, en \emph{geen} horizontale
\omterm{def:g10:inertia:force}{kracht}.
\textbf{7.} De puck houdt \qty{8.0}{m/s} zonder enige beweger. Alledaagse
oppervlakken verbergen \omterm{def:g10:inertia:inventory}{wrijving} --- een niet-gecompenseerde
achterwaartse \omterm{def:g10:inertia:force}{kracht} --- zodat beweging lijkt vanzelf te sterven.
\textbf{8.} Haar snelheid verandert, dus \omterm{def:g10:inertia:compensate}{compenseren} de
\omterm{def:g10:inertia:force}{krachten} niet meer (\cref{prop:g10:inertia:converse}); de
schuldige is de \omterm{def:g10:inertia:inventory}{wrijving} van het ruwe ijs.
\textbf{9.} Nee: snelheid is een \emph{vector}, en haar richting
verandert, dus \omterm{def:g10:inertia:compensate}{compenseren} de \omterm{def:g10:inertia:force}{krachten}
niet --- het ijs duwt zijwaarts op haar schaatsen.
\textbf{10.} (a)--(iii), (b)--(i), (c)--(ii).
\textbf{11.} \omterm{def:g10:universal-gravitation:weight}{Gewicht} $P = 80 \times 9.81 \approx \qty{785}{N}$,
weerstand nul: niets \omterm{def:g10:inertia:compensate}{compenseert} het
\omterm{def:g10:universal-gravitation:weight}{gewicht}, dus moet de snelheid veranderen --- de springer
versnelt omlaag.
\textbf{12.} Zolang weerstand $<$ \omterm{def:g10:universal-gravitation:weight}{gewicht}
\omterm{def:g10:inertia:compensate}{compenseren} de \omterm{def:g10:inertia:force}{krachten} nog niet: de
snelheid blijft groeien, en de weerstand groeit mee.
\textbf{13.} Constante snelheid: \omterm{def:g10:universal-gravitation:weight}{gewicht} en weerstand
\omterm{def:g10:inertia:compensate}{compenseren}, dus de weerstand is \qty{785}{N}. De
beweging is rechtlijnig en eenparig --- een vallend lichaam in precies
de mechanische toestand van de koffer op het rolpad: traagheid in
vermomming.
\textbf{14.} Weerstand overtreft nu het \omterm{def:g10:universal-gravitation:weight}{gewicht}: de
\omterm{def:g10:inertia:force}{krachten} \omterm{def:g10:inertia:compensate}{compenseren} niet, dus
verandert de snelheid --- de springer, nog steeds dalend, vertraagt; naarmate
de snelheid daalt, krimpt de weerstand terug naar het
\omterm{def:g10:universal-gravitation:weight}{gewicht}.
\textbf{15.} Opnieuw constant: weerstand $= P = \qty{785}{N}$ --- precies
hetzelfde als bij \qty{50}{m/s}.
\textbf{16.} De eindweerstand is altijd gelijk aan het
\omterm{def:g10:universal-gravitation:weight}{gewicht}, parachute of niet. De parachute verandert de
\emph{snelheid} waarbij de weerstand \qty{785}{N} bereikt:
\qty{5}{m/s} in plaats van \qty{50}{m/s}.
\textbf{17.} «\dots de \omterm{def:g10:inertia:force}{krachten} die op het lichaam werken
\omterm{def:g10:inertia:compensate}{compenseren} (of er werkt er geen)» --- de twee lezingen
van \cref{thm:g10:inertia:principle}. Rust is gewoon $\vect v = \vect 0$,
een \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging} als elke andere.
\textbf{18.} De ijsbaan: de puck glijdt zonder beweger in zicht. Galileo
wijst naar het steeds gladdere oppervlak --- \omterm{def:g10:inertia:inventory}{wrijving}, niet de
natuur, is wat dingen stopt.
\textbf{19.} \omterm{def:g10:inertia:force}{Krachten} werken wel --- \omterm{def:g10:universal-gravitation:weight}{gewicht}
\qty{785}{N} omlaag en weerstand \qty{785}{N} omhoog; ze
\emph{\omterm{def:g10:inertia:compensate}{compenseren}}, wat precies is waarom de snelheid
constant is.
\textbf{20.} Rolpad: \omterm{def:g10:universal-gravitation:weight}{gewicht} en \omterm{def:g10:inertia:inventory}{normaalkracht}.
IJsbaan: \omterm{def:g10:universal-gravitation:weight}{gewicht} en \omterm{def:g10:inertia:inventory}{normaalkracht}.
Parachute: \omterm{def:g10:universal-gravitation:weight}{gewicht} en weerstand. De ene wet: het
traagheidsprincipe, in zijn directe en omgekeerde lezing.
\end{solution}
