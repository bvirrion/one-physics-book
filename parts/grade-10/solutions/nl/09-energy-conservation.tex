\section*{Hoofdstuk \ref{ch:g10:energy-conservation} --- Energie: vormen en behoud}

\begin{solution}{exo:g10:energy-conservation:1}
$E_k = \tfrac12 \times 1200 \times 25^2 = \qty{3.75e5}{J}$. Bij
\qty{36.1}{m/s}: $\tfrac12 \times 1200 \times 36.1^2 \approx
\qty{7.8e5}{J}$ --- $44\%$ meer snelheid, $(130/90)^2 \approx 2.1$
keer de \omterm{def:g10:energy-conservation:kinetic}{kinetische energie}: het kwadraat aan het werk.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:2}
$\Delta h = \qty{450}{m}$:
$E_p = 72 \times 9.81 \times 450 \approx \qty{3.2e5}{J}
= \num{3.2e5}/\num{3.6e6} \approx \qty{0.088}{kW.h}$, waard ongeveer
$0.022$ euro. Een dag klimmen is twee cent elektriciteit.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:3}
$\qty{1}{kW.h} = \qty{3.6e6}{J}$;
$\qty{12}{W.h} = 12 \times \qty{3600}{J} \approx \qty{43}{kJ}$;
$\qty{45}{MJ} = 45/3.6 = \qty{12.5}{kW.h}$;
$\qty{10}{MJ} = 10/3.6 \approx \qty{2.8}{kW.h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:4}
$E = 2200 \times 150 = \qty{3.3e5}{J}
= \num{3.3e5}/\num{3.6e6} \approx \qty{0.092}{kW.h}$. Mens:
$P = \num{1.0e7}/\num{86400} \approx \qty{116}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:5}
Dynamo: chemisch (voedsel van de fietser) $\to$ kinetisch (wiel) $\to$
elektrisch (dynamo) $\to$ stralings- $+$ thermisch (lamp); warmtelekken
bij spieren, bandcontact, dynamo, lamp. Gasboiler: chemisch (gas)
$\to$ thermisch (water), lek in de rookgassen. Paneel: straling
(Zon) $\to$ elektrisch (paneel) $\to$ chemisch (batterij), warmtelekken
bij paneel en lader.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:6}
Verbruikt $2.0 \times 45 = \qty{90}{MJ}$; $\eta = 27/90 = 0.30$.
Warmte: $90 - 27 = \qty{63}{MJ}$, afgevoerd door de uitlaatgassen
en de radiateur naar de omgevingslucht.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:7}
$E_k = E_p = 70 \times 9.81 \times 20 \approx \qty{1.37e4}{J}$;
$v = \sqrt{2 \times 13734/70} = \sqrt{392} \approx \qty{19.8}{m/s}
\approx \qty{71}{km/h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:8}
$41562 - 41236 = \qty{326}{kW.h}$; per dag $326/30 \approx
\qty{10.9}{kW.h}$. Rekening: $326 \times 0.26 + 12 = 96.76$ euro.
Gemiddeld \omterm{def:g10:energy-conservation:power}{vermogen}: $\qty{326}{kW.h}/\qty{720}{h} \approx \qty{453}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:9}
\emph{1.} Licht: $0.05 \times 60 = \qty{3.0}{W}$; led:
$\eta = 3/9 \approx 0.33$.
\emph{2.} Jaarlijks ($\qty{1460}{h}$): gloeilamp
$60 \times 1460 = \qty{87.6}{kW.h}$, led $\qty{13.1}{kW.h}$;
besparing $\qty{74.5}{kW.h} \approx 19$ euro per lamp per jaar.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:10}
\emph{1.} $P = \num{1.5e5} \times 9.81 \times 40 \approx
\qty{5.9e7}{W} = \qty{59}{MW}$.
\emph{2.} $0.90 \times 59 \approx \qty{53}{MW}$;
$\num{5.3e7}/525 \approx \num{1.0e5}$ huishoudens.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:11}
$v = \sqrt{2 \times 9.81 \times 0.050} \approx \qty{0.99}{m/s}$
(de massa valt weg: $\tfrac12 m v^2 = mgh$). De begin-
$E_p = 0.20 \times 9.81 \times 0.050 \approx \qty{0.098}{J}$ eindigt
als \omterm{def:g10:energy-conservation:forms}{thermische energie} in de lucht en de as: de slinger is niet
geïsoleerd, en het totaal --- zwaai plus warmte --- is precies
behouden.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:12}
\emph{1.} $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.304$;
verbruikt $12/0.304 \approx \qty{39.5}{W.h}
= 39.5 \times 3600 \approx \qty{142}{kJ}$.
\emph{2.} $\num{1.42e5}/\num{2.5e7} \approx \qty{5.7}{g}$ steenkool
per lading; $\times 365 \approx \qty{2.1}{kg}$ per jaar.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:13}
\emph{1.} $\num{5e9}/\num{3.6e6} \approx \qty{1400}{kW.h}$;
$1400/12.6 \approx 110$ dagen.
\emph{2.} $P = \num{5e9}/\num{1e-4} = \qty{5e13}{W} = \qty{50}{TW}$
--- zeventien keer het gemiddelde elektrische
\omterm{def:g10:energy-conservation:power}{vermogen} van de mensheid, voor \qty{100}{\micro s}, op
een onvoorspelbare plek. Bliksem is veel \omterm{def:g10:energy-conservation:power}{vermogen}
maar weinig \omterm{def:g10:energy-conservation:energy}{energie}, nutteloos snel afgeleverd: geen
\omterm{def:g10:energy-conservation:transfer}{omzetter} accepteert zo'n piek.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:14}
\emph{1.} $E_k = \tfrac12 \times 1300 \times 30.6^2 \approx
\qty{6.1e5}{J}$, omgezet naar \omterm{def:g10:energy-conservation:forms}{thermische energie} in de
remschijven en -blokken (dan de lucht).
\emph{2.} $0.60 \times \qty{6.1e5}{J} \approx \qty{3.7e5}{J}$ per
stop; $\num{3.6e6}/\num{3.7e5} \approx 10$ stops per
\omterm{def:g10:energy-conservation:kwh}{kilowattuur}.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:15}
$E_p = 85 \times 9.81 \times 1200 \approx \qty{1.0}{MJ}$; voedsel:
$1.0/0.25 = \qty{4.0}{MJ}$, d.w.z.\ $4.0/1.5 \approx 2.7$ kommen
graan --- of $4.0/45 \approx \qty{0.09}{kg}$ benzine. Een hele ochtend
eerlijke inspanning past in een koffiekop brandstof:
\omterm{def:g10:energy-conservation:forms}{chemische energie} is absurd geconcentreerd.
\end{solution}

\begin{solution}{pb:g10:energy-conservation:1}
\textbf{1.} Koelkast $0.1 \times 24 = \qty{2.4}{kW.h}$; boiler
$2.4 \times 2.5 = \qty{6.0}{kW.h}$; oven \qty{2.0}{kW.h}; wasmachine
\qty{1.0}{kW.h}; verlichting \qty{0.3}{kW.h}; schermen
\qty{0.9}{kW.h}.
\textbf{2.} Totaal \qty{12.6}{kW.h} $= 12.6 \times 3.6 \approx
\qty{45}{MJ}$: één \omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine, tot op de
\omterm{def:g10:energy-conservation:energy}{joule}.
\textbf{3.} $\qty{12600}{W.h}/\qty{24}{h} = \qty{525}{W}$ --- het
huis idle't op een kwart van een waterkoker.
\textbf{4.} De boiler (\qty{6.0}{kW.h}, bijna de helft). Bij
\qty{1.25}{h}: boiler \qty{3.0}{kW.h}, totaal \qty{9.6}{kW.h}.
\textbf{5.} $15 \times 24 = \qty{0.36}{kW.h}$ per dag, ongeveer
$2.8\%$ van het getelde totaal; jaarlijks $0.36 \times 365 \approx
\qty{131}{kW.h} \approx 33$ euro --- voor lampen die niemand bekijkt.
\textbf{6.} $56604 - 56214 = \qty{390}{kW.h}$;
$390/30 = \qty{13.0}{kW.h}$ per dag: de audit van \qty{12.6}{kW.h}
plus de \qty{0.36}{kW.h} standby --- bevestigd.
\textbf{7.} $390 \times 0.25 + 14 = 111.50$ euro.
\textbf{8.} Net: $0.25/3.6 \approx 0.07$ euro per megajoule.
Benzine: $1.80/34 \approx 0.05$ euro per megajoule.
\textbf{9.} Kom: $0.50/1.5 \approx 0.33$ euro per megajoule ---
vijf keer het net. Goedkoopste eerst: benzine, net, dan voedsel, ver
erachter.
\textbf{10.} $0.20 \times 1000 = \qty{200}{W}$ per vierkante
\omterm{def:g10:orders-of-magnitude:unit}{meter}.
\textbf{11.} $200 \times 4.0 = \qty{800}{W.h} = \qty{0.80}{kW.h}$
per vierkante \omterm{def:g10:orders-of-magnitude:unit}{meter} per dag.
\textbf{12.} $12.6/0.80 = 15.75 \approx \qty{16}{m^2}$: past op het
\qty{20}{m^2}-dak.
\textbf{13.} $16 \times 0.2 \times 1.5 = \qty{4.8}{kW.h}$: ongeveer
$38\%$ van de behoefte; het net (of een batterij geladen in betere
maanden) moet de rest leveren.
\textbf{14.} Eén vierkante \omterm{def:g10:orders-of-magnitude:unit}{meter} levert
$0.80 \times 365 \approx \qty{292}{kW.h}$ per jaar; terugverdientijd
$700/292 \approx 2.4$ jaar --- kort tegen een paneelleven van 25 jaar.
\textbf{15.} $\qty{12.6}{kW.h}/\qty{0.1}{kW} = \qty{126}{h}$:
meer dan vijf mensen ($126/24 = 5.25$) die dag en nacht trappen.
\textbf{16.} Voedsel $= 12.6/0.25 = \qty{50.4}{kW.h} \approx
\qty{181}{MJ}$, d.w.z.\ $181/1.5 \approx 121$ kommen graan.
\textbf{17.} $121 \times 0.50 \approx 60$ euro, tegen
$12.6 \times 0.25 \approx 3.15$ euro van het net: trap-
\omterm{def:g10:energy-conservation:power}{vermogen} kost ongeveer $19$ keer meer --- vóór het
betalen van de fietsers.
\textbf{18.} $5000/45 \approx 110$ huisdagen. Maar
$\num{5e9}/\num{1e-4} = \qty{5e13}{W}$ voor \qty{100}{\micro s}, op
een willekeurige plek: geen \omterm{def:g10:energy-conservation:transfer}{omzetter} of batterij
kan uit die slang drinken. Een huis heeft een gestage
\qty{525}{W}-stroom nodig, geen piek.
\textbf{19.} Niet \omterm{def:g10:energy-conservation:energy}{joules} --- die zijn behouden en
identiek. Je betaalt voor hun \emph{vorm en beschikbaarheid}:
\omterm{def:g10:energy-conservation:energy}{energie} die geconcentreerd, opslagbaar en op aanvraag is,
kost de hele \omterm{def:g10:energy-conservation:transfer}{omzettingsketen} die haar zo maakte.
\textbf{20.} Eén gezinsdag $= \qty{12.6}{kW.h} \approx
\qty{45}{MJ}$: één \omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine, zestien vierkante
\omterm{def:g10:orders-of-magnitude:unit}{meters} zonnig dak, of vijf round-the-clock fietsers. De
voorraad is \omterm{def:g10:energy-conservation:energy}{energie}
(\omterm{def:g10:energy-conservation:energy}{joules}, \omterm{def:g10:energy-conservation:kwh}{kilowatturen}); de
stroom is \omterm{def:g10:energy-conservation:power}{vermogen}
(\omterm{def:g10:energy-conservation:power}{watts}) --- de badkuip en de kraan, nooit meer verward.
\end{solution}
