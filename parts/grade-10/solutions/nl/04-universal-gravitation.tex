\section*{Hoofdstuk \ref{ch:g10:universal-gravitation} --- Universele zwaartekracht en gewicht}

\begin{solution}{exo:g10:universal-gravitation:1}
\emph{1.} $F = \num{6.67e-11} \times \dfrac{60 \times 60}{0.80^2}
= \dfrac{\num{2.40e-7}}{0.64} \approx \qty{3.8e-7}{N}$.

\emph{2.} Het \omterm{def:g10:universal-gravitation:weight}{gewicht} van de mug is
$P = \num{2.5e-6} \times 9.81 \approx \qty{2.5e-5}{N}$ --- ongeveer
$65$ keer \emph{groter} dan de wederzijdse aantrekking van de
dansers. Zwaartekracht tussen mensen is zwakker dan het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van een mug.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:2}
\emph{1.} $P = mg = 55 \times 9.81 \approx \qty{5.4e2}{N}$.

\emph{2.} Haar massa is onveranderd: \qty{55}{kg} (dezelfde
hoeveelheid materie). Haar \omterm{def:g10:universal-gravitation:weight}{gewicht} wordt
$P = 55 \times 1.62 \approx \qty{89}{N}$ --- zes keer minder.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:3}
\emph{1.} $g_{\text{Mars}} = \dfrac{G M}{R^2}
= \dfrac{\num{6.67e-11} \times \num{6.42e23}}{(\num{3.39e6})^2}
= \dfrac{\num{4.28e13}}{\num{1.15e13}} \approx \qty{3.73}{N/kg}$.

\emph{2.} Op Mars: $P = 900 \times 3.73 \approx \qty{3.4e3}{N}$. Op
Aarde: $P = 900 \times 9.81 \approx \qty{8.8e3}{N}$ --- de rover
weegt ongeveer $2.6$ keer minder op Mars, met dezelfde massa.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:4}
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
= \frac{\num{2.93e37}}{\num{1.47e17}}
\approx \qty{2.0e20}{N}.
\]
De Maan oefent op de Aarde een kracht van de \emph{zelfde} grootte
uit, \qty{2.0e20}{N}, in de tegenovergestelde richting:
\omterm{def:g10:universal-gravitation:force}{zwaartekracht} is wederzijds.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:5}
\emph{1.} Afstand verdubbeld: $d^2$ wordt met $4$ vermenigvuldigd,
dus $F = F_0/4$.

\emph{2.} Afstand gehalveerd: $F = 4F_0$.

\emph{3.} Beide massa's verdubbeld: $m_1 m_2$ wordt met $4$
vermenigvuldigd, dus $F = 4F_0$.

\emph{4.} Teller $\times 3$, noemer $\times 9$:
$F = F_0/3$.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:6}
\emph{1.} $F = \num{6.67e-11} \times
\dfrac{(\num{5.0e8})^2}{100^2}
= \num{6.67e-11} \times \dfrac{\num{2.5e17}}{\num{1.0e4}}
\approx \qty{1.7e3}{N}$.

\emph{2.} $m = F/g = 1667/9.81 \approx \qty{1.7e2}{kg}$: de twee
schepen trekken elkaar aan met het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van een zware motorfiets.

\emph{3.} $a = F/m = \num{1.7e3}/\num{5.0e8} \approx
\qty{3.3e-6}{m/s^2}$. Zelfs ongehinderd zouden de tankers ongeveer
vijf dagen nodig hebben om wandeltempo te bereiken; in de praktijk
overspoelen waterweerstand en meertrossen de aantrekking volledig.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:7}
\emph{1.} De balkbalans geeft \qty{3.0}{kg} op beide werelden: de
zak en de standaardmassa's zien hun
\omterm{def:g10:universal-gravitation:weight}{gewichten} door dezelfde factor gedeeld, dus de
vergelijking is onveranderd. De veerweegschaal leest het
\omterm{def:g10:universal-gravitation:weight}{gewicht}: $3.0 \times 9.81 \approx \qty{29}{N}$
op Aarde, $3.0 \times 1.62 \approx \qty{4.9}{N}$ op de Maan.

\emph{2.} De balans: ze meet massa, wat de klant koopt. Een
veerweegschaal geijkt in «\omterm{def:g10:orders-of-magnitude:unit}{kilogram}» op Aarde zou
$4.9/9.81 \approx \qty{0.50}{kg}$ lezen voor dezelfde zak op de
Maan --- de winkelier zou zes zakken weggeven voor de prijs van één.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:8}
\emph{1.} Afstand tot het centrum:
$d = R + h = \num{6.37e6} + \num{4.0e5} = \qty{6.77e6}{m}$. Dan
$g = \dfrac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.77e6})^2}
= \dfrac{\num{3.98e14}}{\num{4.58e13}} \approx \qty{8.69}{N/kg}$.

\emph{2.} $8.69/9.81 \approx 89\%$ van de oppervlaktewaarde.

\emph{3.} De zwaartekracht is daar bijna op volle sterkte: de
astronauten zweven omdat het station en alles erin \emph{samen
vallen} rond de Aarde (vrije val), niet omdat de zwaartekracht
verdwenen is.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:9}
Uit $g = GM/R^2$,
\[
M = \frac{g R^2}{G}
= \frac{8.87 \times (\num{6.05e6})^2}{\num{6.67e-11}}
= \frac{\num{3.25e14}}{\num{6.67e-11}}
\approx \qty{4.87e24}{kg},
\]
ongeveer $0.8$ keer de massa van de Aarde --- Venus is bijna de
tweeling van de Aarde in grootte en massa.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:10}
\emph{1.} Zon:
$F = \dfrac{\num{6.67e-11} \times \num{1.99e30}}{(\num{1.496e11})^2}
\approx \qty{5.9e-3}{N}$. Maan:
$F = \dfrac{\num{6.67e-11} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{3.3e-5}{N}$.

\emph{2.} De Zon trekt ongeveer $\num{5.9e-3}/\num{3.3e-5} \approx 180$
keer harder.

\emph{3.} De diameter van de Aarde is een veel grotere
\emph{fractie} van de Aarde--Maan-afstand ($\approx 3\%$) dan van de
Aarde--Zon-afstand ($\approx 0.01\%$), dus de trek van de Maan
verandert veel meer van de nabije naar de verre kant van de globe.
Getijden voeden zich met dat verschil --- en daar wint de Maan.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:11}
\emph{1.} $g' = \dfrac{G (2M)}{(2R)^2} = \dfrac{2}{4}\,\dfrac{GM}{R^2}
= \dfrac{g}{2} \approx \qty{4.9}{N/kg}$: de verdubbelde straal wint
van de verdubbelde massa.

\emph{2.} We hebben $\dfrac{GM}{R'^2} = 2\,\dfrac{GM}{R^2}$ nodig, dus
$R'^2 = R^2/2$ en $R' = R/\sqrt{2} \approx \qty{4.5e6}{m}$ ---
ongeveer \qty{4500}{km}.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:12}
\emph{1.} Gelijke trekken per \omterm{def:g10:orders-of-magnitude:unit}{kilogram}:
$\dfrac{G M_{\text{Earth}}}{x^2} = \dfrac{G M_{\text{Moon}}}{(D-x)^2}$.
Kruislings vermenigvuldigen en delen door $G$:
$\left(\dfrac{x}{D-x}\right)^2 = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.

\emph{2.} $\dfrac{x}{D-x}
= \sqrt{\dfrac{\num{5.97e24}}{\num{7.35e22}}} = \sqrt{81.2} \approx
9.0$, dus $x = 9.0\,(D - x)$, d.w.z.\ $x = \dfrac{9.0}{10.0}\,D
\approx 0.90 \times \num{3.84e8} \approx \qty{3.5e8}{m}$. Het
evenwichtspunt zit op $90\%$ van de weg naar de Maan: de trek van de
Aarde domineert bijna de hele reis.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:13}
\emph{1.} Uitwerken: $R^2 + 2Rh + h^2 = R^2 + x^2$; omdat $h$ piepklein
is vergeleken met $R$, laat $h^2$ weg: $2Rh \approx x^2$, dus
$h \approx \dfrac{x^2}{2R}
= \dfrac{(\num{8.0e3})^2}{2 \times \num{6.37e6}}
= \dfrac{\num{6.4e7}}{\num{1.27e7}} \approx \qty{5.0}{m}$.

\emph{2.} In één seconde valt het projectiel \qty{4.9}{m} --- bijna
exact de \qty{5.0}{m} die de grond wegzakt over \qty{8.0}{km}. Dus bij
$v \approx \qty{8.0}{km/s}$ haalt de val de grond nooit in: het
projectiel \omterm{def:g10:universal-gravitation:satellite}{gaat in een baan}.

\emph{3.} $T = \dfrac{2\pi R}{v}
= \dfrac{2\pi \times \num{6.37e6}}{\num{8.0e3}}
\approx \qty{5.0e3}{s} \approx \qty{83}{min}$ --- dicht bij de
ongeveer $\qty{90}{min}$ van echte
\omterm{def:g10:universal-gravitation:satellite}{lage-omloop}-
\omterm{def:g10:universal-gravitation:satellite}{satellieten}, die iets hoger vliegen.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:14}
\emph{1.} $g = \dfrac{\num{6.67e-11} \times \num{2.8e30}}
{(\num{1.2e4})^2} = \dfrac{\num{1.87e20}}{\num{1.44e8}}
\approx \qty{1.3e12}{N/kg}.$

\emph{2.} $P = 70 \times \num{1.3e12} \approx \qty{9.1e13}{N}$, tegen
$\qty{6.9e2}{N}$ op Aarde: ongeveer $\num{1.3e11}$ keer meer. Geen

\emph{3.} $P = \num{1.0e-6} \times \num{1.3e12} \approx

\omterm{def:g10:universal-gravitation:weight}{gewicht} van
$m = \num{1.3e6}/9.81 \approx \qty{1.3e5}{kg}$ --- een zandkorrel die
evenveel weegt als een locomotief van honderd ton.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:15}
\emph{1.} De afstand is $60$ keer groter, dus de trek per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} is $60^2 = 3600$ keer zwakker.

\emph{2.} De val in één seconde schaalt op dezelfde manier:
$4.9/3600 \approx \qty{1.4e-3}{m}$ --- ongeveer \qty{1.4}{mm}.

\emph{3.} $T = 27.3 \times \qty{86400}{s} = \qty{2.36e6}{s}$, dus
$v = \dfrac{2\pi \times \num{3.84e8}}{\num{2.36e6}} \approx
\qty{1.02e3}{m/s}$, en $x = \qty{1.02e3}{m}$ in één seconde. De val
is $h \approx \dfrac{x^2}{2d}
= \dfrac{(\num{1.02e3})^2}{2 \times \num{3.84e8}}
\approx \qty{1.4e-3}{m}$.

\emph{4.} De twee getallen kloppen: de Maan valt elke seconde naar
de Aarde precies zoveel als de omgekeerd-kwadraatzwaartekracht,
geijkt op een vallende appel, voorspelt. Newton concludeerde dat de
kracht die de Maan vasthoudt \emph{de} kracht is die de appel laat
vallen --- zwaartekracht is universeel.
\end{solution}

\begin{solution}{pb:g10:universal-gravitation:1}
\textbf{1.} $F = \num{6.67e-11} \times
\dfrac{\num{7.3e6} \times 55}{100^2} \approx \qty{2.7e-6}{N}$.

\textbf{2.} De zandkorrel weegt
$\num{1.0e-6} \times 9.81 \approx \qty{9.8e-6}{N}$: de hele
Eiffeltoren trekt de toerist ongeveer $3.7$ keer \emph{minder} aan
dan het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een zandkorrel.

\textbf{3.} (a) Afstand $\times 10$: kracht gedeeld door $100$.
(b) Beide massa's $\times 1000$: kracht vermenigvuldigd met $10^6$.
(c) Massa's $\times 1000$ en afstand $\times 1000$:
$10^6 / 10^6$ --- de kracht is onveranderd.

\textbf{4.} $F = \dfrac{\num{6.67e-11} \times \num{5.97e24} \times
1.00}{(\num{6.37e6})^2} \approx \qty{9.81}{N}$: het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van één
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} --- dit is precies de
veldsterkte $g = \qty{9.81}{N/kg}$.

\textbf{5.} Zwaartekracht is zwak per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} maar \emph{trekt} alleen maar aan, dus de
trekken van alle $\num{5.97e24}$ \omterm{def:g10:orders-of-magnitude:unit}{kilogram}
van de Aarde tellen in dezelfde richting op --- en een astronomische
massa verslaat een buurlijke met tweeëntwintig machten van tien.

\textbf{6.} $F = \num{6.67e-11} \times \dfrac{158 \times 0.73}{0.23^2}
= \num{6.67e-11} \times \dfrac{115.3}{0.0529}
\approx \qty{1.5e-7}{N}$.

\textbf{7.} De kleine bol weegt
$0.73 \times 9.81 \approx \qty{7.2}{N}$ --- ongeveer $\num{5e7}$
keer de te detecteren aantrekking. Geen schaalweegschaal lost één
deel in vijftig miljoen op; een fijne torsiedraad, die zichtbaar
draait onder nanonewton-schaal-koppels, wel.

\textbf{8.} $G = \dfrac{F d^2}{m_1 m_2}
= \dfrac{\num{1.47e-7} \times 0.0529}{115.3}
\approx \qty{6.74e-11}{N.m^2/kg^2}$ --- binnen ongeveer $1\%$ van de
moderne \num{6.67e-11}, uit een houten schuur in 1798.

\textbf{9.} Uit $g = G M / R^2$:
\[
M = \frac{g R^2}{G}
= \frac{9.81 \times (\num{6.37e6})^2}{\num{6.67e-11}}
\approx \qty{5.97e24}{kg}.
\]
Zes miljoen miljard miljard \omterm{def:g10:orders-of-magnitude:unit}{kilogram}: de Aarde,
gewogen.

\textbf{10.} $V = \frac43 \pi R^3 = \frac43 \pi
(\num{6.37e6})^3 \approx \qty{1.08e21}{m^3}$, dus
$\rho = \num{5.97e24}/\num{1.08e21} \approx \qty{5.5e3}{kg/m^3}$ ---
twee keer de dichtheid van oppervlakterots. Het binnenste moet iets
veel dichters dan graniet verbergen: de Aarde heeft een zware
(ijzeren) kern.

\textbf{11.} $g_{\text{Moon}} = \dfrac{\num{6.67e-11} \times
\num{7.35e22}}{(\num{1.74e6})^2} = \dfrac{\num{4.90e12}}
{\num{3.03e12}} \approx \qty{1.62}{N/kg}$.

\textbf{12.} Hoogte $\propto 1/g$:
$h = 0.50 \times \dfrac{9.81}{1.62} \approx \qty{3.0}{m}$ --- een
staande sprong over een basketbalring, in slow motion.

\textbf{13.} $g_{\text{Mercury}} = \dfrac{\num{6.67e-11} \times
\num{3.30e23}}{(\num{2.44e6})^2} = \dfrac{\num{2.20e13}}
{\num{5.95e12}} \approx \qty{3.70}{N/kg}$ --- bijna exact de
\qty{3.73}{N/kg} van Mars. Mercurius heeft minder massa, maar haar
kleinere straal (een kleinere $R^2$ eronder) compenseert: ze is een
veel dichtere wereld.

\textbf{14.} De helft van de veldsterkte vereist
$(R + h)^2 = 2 R^2$, dus $R + h = R\sqrt{2}$ en
$h = (\sqrt{2} - 1) R \approx 0.414 \times \num{6.37e6}
\approx \qty{2.6e6}{m}$ --- ongeveer \qty{2600}{km} omhoog.

\textbf{15.} $\dfrac{G M_{\text{Earth}}}{d^2} = 1.62$ geeft
$d = \sqrt{\dfrac{\num{6.67e-11} \times \num{5.97e24}}{1.62}}
= \sqrt{\num{2.46e14}} \approx \qty{1.57e7}{m} \approx 2.5$
Aardstralen. Vanaf slechts anderhalve straal boven de grond trekt de
Aarde niet harder dan het oppervlak van de Maan.

\textbf{16.} $v = \dfrac{2\pi d}{T}
= \dfrac{2\pi \times \num{1.496e11}}{\num{3.156e7}}
\approx \qty{2.98e4}{m/s}$ --- dertig kilometer elke seconde.

\textbf{17.} $\dfrac{v^2}{d} = \dfrac{(\num{2.98e4})^2}{\num{1.496e11}}
\approx \qty{5.9e-3}{N/kg}$: de Zon trekt aan elk
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} van de Aarde met ongeveer zes millinewton.

\textbf{18.} $\dfrac{G M_{\text{Sun}}}{d^2} = \dfrac{v^2}{d}$ geeft
\[
M_{\text{Sun}} = \frac{v^2 d}{G}
= \frac{\num{8.87e8} \times \num{1.496e11}}{\num{6.67e-11}}
\approx \qty{1.99e30}{kg}.
\]

\textbf{19.} $\dfrac{\num{1.99e30}}{\num{5.97e24}} \approx
\num{3.3e5}$: de Zon is een derde miljoen Aardes.

\textbf{20.} Omdat dezelfde $G$ elk paar massa's regeert, liet haar
één keer meten tussen twee loden bollen in een schuur ons eerst de
planeet onder onze voeten, daarna de ster die we
\omterm{def:g10:universal-gravitation:satellite}{omlopen}, op de weegschaal zetten --- dat is
wat \emph{universeel} waard is.
\end{solution}
