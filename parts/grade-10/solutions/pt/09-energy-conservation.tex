\section*{Capítulo \ref{ch:g10:energy-conservation} --- Energia: Formas e Conservação}

\begin{solution}{exo:g10:energy-conservation:1}
$E_k = \tfrac12 \times 1200 \times 25^2 = \qty{3.75e5}{J}$. No
\qty{36.1}{m/s}: $\tfrac12 \times 1200 \times 36.1^2 \approx
\qty{7.8e5}{J}$ --- $44\%$ mais velocidade, $(130/90)^2 \approx 2.1$
vezes o \omterm{def:g10:energy-conservation:kinetic}{energia cinética}: a praça em ação.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:2}
$\Delta h = \qty{450}{m}$:
$ E_p = 72 \times 9.81 \times 450 \approx \qty{3.2e5}{J}
= \num{3.2e5}/\num{3.6e6} \approx \qty{0.088}{kW.h}$, vale cerca de
$0.022$ euros. Um dia de escalada custa dois centavos de eletricidade.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:3}
$\qty{1}{kW.h} = \qty{3.6e6}{J}$;
$\qty{12}{W.h} = 12 \times \qty{3600}{J} \approx \qty{43}{kJ}$;
$\qty{45}{MJ} = 45/3.6 = \qty{12.5}{kW.h}$;
$\qty{10}{MJ} = 10/3.6 \approx \qty{2.8}{kW.h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:4}
$ E = 2200 \times 150 = \qty{3.3e5}{J}
= \num{3.3e5}/\num{3.6e6} \approx \qty{0.092}{kW.h}$. Humano:
$ P = \num{1.0e7}/\num{86400} \approx \qty{116}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:5}
Dínamo: químico (comida do ciclista) $\to $ cinético (roda) $\to $
elétrica (dínamo) $\to $ radiativo $+$ térmico (lâmpada); vazamentos de calor
nos músculos, contato com pneus, dínamo, lâmpada. Aquecedor a gás: químico (gás)
$\to $ térmico (água), vazamento nos gases de combustão. Painel: radiativo
(Sol) $\to $ elétrica (painel) $\to $ produto químico (battery), vazamentos de calor
no painel e no carregador.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:6}
Consumido $2.0 \times 45 = \qty{90}{MJ}$; $\eta = 27/90 = 0.30$.
Aquecer: $90 - 27 = \qty{63}{MJ}$, levado pelos gases de escape
e o radiador para o ar circundante.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:7}

$ v = \sqrt{2 \times 13734/70} = \sqrt{392} \approx \qty{19.8}{m/s}

\end{solution}

\begin{solution}{exo:g10:energy-conservation:8}
$41562 - 41236 = \qty{326}{kW.h}$; diário $326/30 \approx

\end{solution}

\begin{solution}{exo:g10:energy-conservation:9}
\emph{1.} Luz: $0.05 \times 60 = \qty{3.0}{W}$; LIDERADO:
$\eta = 3/9 \approx 0.33$.
\emph{2.} Anualmente ($\qty{1460}{h}$): lâmpada
$60 \times 1460 = \qty{87.6}{kW.h}$, LIDERADO $\qty{13.1}{kW.h}$;

\end{solution}

\begin{solution}{exo:g10:energy-conservation:10}

\begin{solution}{exo:g10:energy-conservation:11}
$ v = \sqrt{2 \times 9.81 \times 0.050} \approx \qty{0.99}{m/s}$
(a missa cancela: $\tfrac12 m v^2 = mgh $). A inicial
$ E_p = 0.20 \times 9.81 \times 0.050 \approx \qty{0.098}{J}$ termina
como energia térmica no ar e o pivô: o pêndulo não está
isolado, e o total --- balanço mais calor --- é exatamente
conservado.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:12}
\emph{1.} $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.304$;
consumido $12/0.304 \approx \qty{39.5}{W.h}
= 39.5 \times 3600 \approx \qty{142}{kJ}$.
\emph{2.} $\num{1.42e5}/\num{2.5e7} \approx \qty{5.7}{g}$ de carvão

\end{solution}

\begin{solution}{exo:g10:energy-conservation:13}
\emph{1.} $\num{5e9}/\num{3.6e6} \approx \qty{1400}{kW.h}$;
$1400/12.6 \approx 110$ dias.
\emph{2.} $ P = \num{5e9}/\num{1e-4} = \qty{5e13}{W} = \qty{50}{TW}$
--- dezessete vezes a energia elétrica média da humanidade poder, para
\qty{100}{\micro s}, em um lugar imprevisível. Relâmpago é muito
poder mas pouco energia, entregue inutilmente rápido: não \omterm{def:g10:energy-conservation:transfer}{conversor}
aceita tal pico.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:14}
\emph{1.} $ E_k = \tfrac12 \times 1300 \times 30.6^2 \approx

\emph{2.} $0.60 \times \qty{6.1e5}{J} \approx \qty{3.7e5}{J}$ por

\begin{solution}{exo:g10:energy-conservation:15}
$ E_p = 85 \times 9.81 \times 1200 \approx \qty{1.0}{MJ}$; comida:
$1.0/0.25 = \qty{4.0}{MJ}$, ou seja\ $4.0/1.5 \approx 2.7$ cereal

energia} está absurdamente concentrado.
\end{solution}

\begin{solution}{pb:g10:energy-conservation:1}
\textbf{1.} Geladeira $0.1 \times 24 = \qty{2.4}{kW.h}$; aquecedor
$2.4 \times 2.5 = \qty{6.0}{kW.h}$; forno \qty{2.0}{kW.h}; lavar
máquina \qty{1.0}{kW.h}; iluminação \qty{0.3}{kW.h}; telas
\qty{0.9}{kW.h}.
\textbf{2.} Total \qty{12.6}{kW.h} $= 12.6 \times 3.6 \approx
\qty{45}{MJ}$: um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de gasolina, para \omterm{def:g10:energy-conservation:energy}{joule}.
\textbf{3.} $\qty{12600}{W.h}/\qty{24}{h} = \qty{525}{W}$ --- o
a casa fica parada com um quarto de chaleira.
\textbf{4.} O aquecedor de água (\qty{6.0}{kW.h}, quase metade). No
\qty{1.25}{h}: aquecedor \qty{3.0}{kW.h}, total \qty{9.6}{kW.h}.
\textbf{5.} $15 \times 24 = \qty{0.36}{kW.h}$ por dia, cerca
$2.8\%$ do total medido; anual $0.36 \times 365 \approx
\qty{131}{kW.h} \approx 33$ euros --- para luzes que ninguém olha.
\textbf{6.} $56604 - 56214 = \qty{390}{kW.h}$;
$390/30 = \qty{13.0}{kW.h}$ por dia: a auditoria \qty{12.6}{kW.h}
mais o \qty{0.36}{kW.h} de espera --- confirmado.
\textbf{7.} $390 \times 0.25 + 14 = 111.50$ euros.
\textbf{8.} Grade: $0.25/3.6 \approx 0.07$ euros por megajoule.
Gasolina: $1.80/34 \approx 0.05$ euros por megajoule.
\textbf{9.} Tigela: $0.50/1.5 \approx 0.33$ euros por megajoule ---
cinco vezes a grade. Mais barato primeiro: gasolina, rede, depois comida,
muito atrás.
\textbf{10.} $0.20 \times 1000 = \qty{200}{W}$ por quadrado \omterm{def:g10:orders-of-magnitude:unit}{medidor}.
\textbf{11.} $200 \times 4.0 = \qty{800}{W.h} = \qty{0.80}{kW.h}$
por quadrado \omterm{def:g10:orders-of-magnitude:unit}{medidor} por dia.
\textbf{12.} $12.6/0.80 = 15.75 \approx \qty{16}{m^2}$: cabe
o \qty{20}{m^2} teto.
\textbf{13.} $16 \times 0.2 \times 1.5 = \qty{4.8}{kW.h}$: sobre
$38\%$ da necessidade; a rede (ou uma battery carregada em melhor
meses) deve fornecer o restante.
\textbf{14.} Um quadrado \omterm{def:g10:orders-of-magnitude:unit}{medidor} rendimentos $0.80 \times 365 \approx
\qty{292}{kW.h}$ por ano; vingança $700/292 \approx 2.4$ anos ---
curto contra uma vida útil do painel de 25 anos.
\textbf{15.} $\qty{12.6}{kW.h}/\qty{0.1}{kW} = \qty{126}{h}$:
mais de cinco pessoas ($126/24 = 5.25$) pedalando dia e noite.
\textbf{16.} Comida $= 12.6/0.25 = \qty{50.4}{kW.h} \approx
\qty{181}{MJ}$, ou seja\ $181/1.5 \approx 121$ tigelas de cereais.
\textbf{17.} $121 \times 0.50 \approx 60$ euros, contra
$12.6 \times 0.25 \approx 3.15$ euros da rede: pedal poder
custa cerca de $19$ vezes mais --- antes de pagar aos ciclistas.
\textbf{18.} $5000/45 \approx 110$ dias de casa. Mas
$\num{5e9}/\num{1e-4} = \qty{5e13}{W}$ para \qty{100}{\micro s}, em
um lugar aleatório: não \omterm{def:g10:energy-conservation:transfer}{conversor} ou a battery pode beber dessa mangueira.
Uma casa precisa de uma estabilidade \qty{525}{W} fluxo, não um pico.
\textbf{19.} Não \omterm{def:g10:energy-conservation:energy}{joules} --- eles são conservados e idênticos. Você
pagar por eles \emph{forma e disponibilidade}: energia isso é
concentrado, armazenável e sob demanda custa todo o \omterm{def:g10:energy-conservation:transfer}{conversão
corrente} isso fez com que fosse assim.
\textbf{20.} Um dia em família $= \qty{12.6}{kW.h} \approx
\qty{45}{MJ}$: um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de gasolina, dezesseis quadrados metros de
telhado ensolarado ou cinco ciclistas 24 horas por dia. O estoque é energia
(\omterm{def:g10:energy-conservation:energy}{joules}, \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora}); o fluxo é poder (\omterm{def:g10:energy-conservation:power}{watts}) --- a banheira
e a torneira, nunca mais confusa.
\end{solution}
