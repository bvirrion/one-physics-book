\section*{Capítulo \ref{ch:g10:orders-of-magnitude} --- Ordens de Grandeza: Medindo o Universo}

\begin{solution}{exo:g10:orders-of-magnitude:1}
\emph{1.} $\qty{750}{nm} = 750 \times \qty{e-9}{m} = \qty{7.50e-7}{m}$.
\emph{2.} $\qty{2.5}{km} = \qty{2.5e3}{m}$.
\emph{3.} $\qty{3.2}{\micro m} = \qty{3.2e-6}{m}$.
\emph{4.} $\qty{0.15}{Tm} = 0.15 \times \qty{e12}{m} = \qty{1.5e11}{m}$
--- um \omterm{def:g10:orders-of-magnitude:lightyear}{unidade astronômica}.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:2}
\emph{1.} $\qty{149600000}{km} = \qty{1.496e11}{m}$: $4$ significativo
figuras.
\emph{2.} $\qty{1.06e-10}{m}$: $3$ \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} (zeros à esquerda
nunca conte).
\emph{3.} $\num{8.1e9}$: $2$ \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos}.
\emph{4.} $\qty{5.20e-3}{s}$: $3$ \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} (o rastro
zero é uma promessa).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:3}
\omterm{def:g10:orders-of-magnitude:scinot}{Mantissas} e expoentes separadamente
(\cref{prop:g10:orders-of-magnitude:powers}):
\[
3 \times 2 \times 10^{8-5} = \num{6e3}, \qquad
\tfrac{8}{4} \times 10^{4-(-3)} = \num{2e7},
\]
\[
5^{2} \times 10^{12} = \num{2.5e13}, \qquad
\tfrac{4 \times 3}{6} \times 10^{-3-9+7} = \num{2e-5} .
\]
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:4}
Humano: $1.7 < 5$, ordem $\qty{e0}{m}$. Everest:
$\qty{8848}{m} = \qty{8.848e3}{m}$ e $8.848 \geq 5$: ordem
$\qty{e4}{m}$. Bactéria: ordem $\qty{e-6}{m}$. Terra:
$5.97 \geq 5$, ordem $\qty{e25}{kg}$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:5}
\emph{1.} Meia formatura: $ L = (127.0 \pm 0.5)\,\unit{mm}$.
\emph{2.} $\Delta L / L = 0.5/127 \approx 0.4\%$.
\emph{3.} Apagador: $0.5/12 \approx 4\%$. A medida do lápis é
cerca de dez vezes mais preciso --- mesma régua, mesma \omterm{def:g10:orders-of-magnitude:uncertainty}{absoluto
incerteza}, objeto dez vezes mais longo.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:6}
\emph{1.} Próxima:
$4.24 \times \qty{9.47e15}{m} = \qty{4.01e16}{m}$. Sírius:
$8.6 \times \qty{9.47e15}{m} = \qty{8.1e16}{m}$ (dois significativo
figuras, como a entrada).
\emph{2.} $ t = d/c = \qty{2.4e13}{m} / \qty{3.00e8}{m/s}
= \qty{8.0e4}{s}$, ou seja\ $\num{8.0e4}/3600 \approx \qty{22}{h}$:
quase um dia inteiro para cada mensagem da sonda.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:7}
$ t = d/c $ cada vez. Lua:
$\qty{3.84e8}{m}/\qty{3.00e8}{m/s} = \qty{1.28}{s}$. Sol:
$\qty{1.496e11}{m}/\qty{3.00e8}{m/s} = \qty{499}{s} \approx
\qty{8.3}{min}$. Netuno:
$\qty{4.50e12}{m}/\qty{3.00e8}{m/s} = \qty{1.50e4}{s} \approx
\qty{4.2}{h}$. Quando você olha para Netuno, você o vê como se tratasse de
quatro horas atrás --- todo telescópio é uma máquina do tempo moderada.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:8}
\emph{1.} Uma folha: $\qty{52.0}{mm}/500 = \qty{0.104}{mm}
= \qty{1.04e-4}{m}$. A incerteza também divide:
$\qty{0.5}{mm}/500 = \qty{0.001}{mm}$, então
$ t = (104 \pm 1)\,\unit{\micro m}$.
\emph{2.} $1/104 \approx 1\%$.
\emph{3.} Medir uma folha diretamente daria
$(0.1 \pm 0.5)\,\unit{mm}$: um \omterm{def:g10:orders-of-magnitude:uncertainty}{incerteza relativa} de cerca de $500\%$
--- o governante não consegue nem provar que a folha existe. Empilhamento $500$
folhas multiplica o comprimento medido por $500$ enquanto o governante
incerteza absoluta permanece fixo: o \omterm{def:g10:orders-of-magnitude:uncertainty}{incerteza relativa} encolhe em
esse mesmo fator $500$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:9}
Divida cada comprimento por $\qty{2e-10}{m}$. Bactéria:
$\qty{2.0e-6}{m}/\qty{2e-10}{m} = \num{1.0e4}$ átomos. Cabelo:
$\qty{7.0e-5}{m}/\qty{2e-10}{m} = \num{3.5e5}$. Humano:
$1.7/\num{2e-10} = \num{8.5e9}$, \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza} $10^{10}$: dez
bilhões de átomos da cabeça aos pés.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:10}
\emph{1.} $ s = \qty{0.22}{m}/\qty{1.39e9}{m} = \num{1.58e-10}$.
\emph{2.} Terra:
$\qty{1.27e7}{m} \times \num{1.58e-10} = \qty{2.0e-3}{m} =
\qty{2.0}{mm}$ --- um grão de pimenta --- orbitando em
$\qty{1.496e11}{m} \times \num{1.58e-10} = \qty{23.7}{m} \approx
\qty{24}{m}$ do futebol.
\emph{3.} Lua:
$\qty{3.84e8}{m} \times \num{1.58e-10} = \qty{6.1}{cm}$ do
pimenta em grão. Netuno:
$\qty{4.50e12}{m} \times \num{1.58e-10} \approx \qty{710}{m}$: o
modelo não cabe no pátio da escola.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:11}
\emph{1.} $21.0 \times 29.7 = \qty{623.7}{cm^2}$, arredondado para três
\omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} das entradas: $\qty{624}{cm^2}$.
\emph{2.} $ v = \qty{100.0}{m}/\qty{12.3}{s} = \qty{8.13}{m/s}$ (três
números, definidos pela hora).
\emph{3.} Cinco \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} reivindicaria a área para
$\qty{0.01}{cm^2}$, uma precisão que as entradas em nível milimétrico nunca
entregue --- um decimal forjado ainda é uma falsificação.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:12}
\emph{1.} Raio $ r = \qty{0.55}{m}$, área
$ A = \pi r^{2} = \pi \times 0.55^{2} \approx \qty{0.95}{m^2}$. Volume
$ V = \qty{1.0}{mm^3} = \qty{1.0e-9}{m^3}$, e um filme é (área)
$\times $ (espessura), então
\[
e = \frac{V}{A} = \frac{\qty{1.0e-9}{m^3}}{\qty{0.95}{m^2}}
= \qty{1.1e-9}{m} .
\]
\emph{2.} Uma molécula é de ordem $\qty{e-9}{m}$ --- um degrau acima do
átomo $\qty{e-10}{m}$, consistente com uma molécula sendo pequena
aglomerado de átomos. A colher de chá de óleo de Franklin acalmou meio acre de
lagoa e, sem que ele soubesse, mediu o nanômetro.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:13}
Suposições: sobre $70$ batidas por minuto em repouso, uma vida de $80$
anos. Por dia: $70 \times 60 \times 24 \approx \num{1e5}$ batidas. Por
ano: $\num{1e5} \times 365 \approx \num{4e7}$. Por vida:
$\num{4e7} \times 80 \approx \num{3e9}$ --- \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza}
$10^{9}$, alguns bilhões de batimentos cardíacos. Quatro \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} faria
fingir conhecer a frequência cardíaca, minuto a minuto, durante oitenta anos: o
as entradas são suposições de \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza} e nenhuma saída pode superar sua
entradas (\cref{met:g10:orders-of-magnitude:sigfig}).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:14}
Depois $ n $ dobras a espessura é $2^{n} \times \qty{1.0e-4}{m}$.
\emph{1.} Precisar $2^{n} \geq 8848/\num{1.0e-4} = \num{8.8e7}$. Desde
$2^{26} \approx \num{6.7e7}$ fica aquém e
$2^{27} \approx \num{1.3e8}$ basta: $27$ dobras (pilha
$\approx \qty{13}{km}$).
\emph{2.} Precisar $2^{n} \geq \num{3.84e12}$: $2^{41} \approx
\num{2.2e12}$ é curto, $2^{42} \approx \num{4.4e12}$ atinge ---
$42$ dobras, pilha $\qty{4.4e8}{m}$, confortavelmente além da Lua.
\emph{3.} Apenas o experimento: cada dobra divide a área pela metade, e depois
sete dobras a pilha é mais espessa do que larga. A aritmética de
a duplicação permanece intocada --- o crescimento exponencial simplesmente ultrapassa o que o papel
(e intuição) podem seguir.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:15}
\emph{1.} Lua: $\qty{0.03}{m}/\qty{3.84e8}{m} = \num{7.8e-11}$.
Mesa: $\qty{0.001}{m}/\qty{1.000}{m} = \num{1.0e-3}$. A Lua
a medição é mais precisa por um fator
$\num{1.0e-3}/\num{7.8e-11} \approx \num{1.3e7}$ --- mais de dez milhões.
\emph{2.} Correspondência $\num{7.8e-11}$ em um um metro mesa significa
$\Delta x = \qty{7.8e-11}{m}$, menor que um átomo
($\approx \qty{e-10}{m}$): a borda da mesa não é uniforme
\emph{definido} isso bruscamente. Relativamente falando, a Terra-Lua
a distância do laser é uma das melhores medidas que nossa espécie possui.
\end{solution}

\begin{solution}{pb:g10:orders-of-magnitude:1}
\textbf{1.} Grão: $\qty{0.5}{mm} = \qty{5e-4}{m}$, no mili
escala; átomo: $\qty{2.5e-10}{m} = \qty{0.25}{nm}$, na escala nano.

\textbf{2.} $\qty{5e-4}{m}/\qty{2.5e-10}{m} = \num{2.0e6}$: dois
milhões de átomos de diâmetro.

\textbf{3.} Um cubo de $2 \times 10^{6}$ átomos por aresta contém
$(2\times10^{6})^{3} = 8 \times 10^{18}$ átomos: \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza}
$10^{19}$.

\textbf{4.} $ V = (\qty{5e-4}{m})^{3} = \qty{1.25e-10}{m^3}$, então
$ m = \rho V = \qty{2500}{kg/m^3} \times \qty{1.25e-10}{m^3}
= \qty{3.1e-7}{kg} = \qty{0.31}{mg}$.

\textbf{5.} Massa por átomo:
$\qty{3.1e-7}{kg}/(8\times10^{18}) = \qty{3.9e-26}{kg}$ --- diretamente
nos ``poucos $10^{-26}\,\unit{kg}$'' faixa. O ingênuo cubo de átomos
reproduz massas atômicas reais: o modelo passa na verificação de sanidade
(\cref{met:g10:orders-of-magnitude:sanity}) com honras.

\textbf{6.} $ s = \qty{5e-4}{m}/\qty{1.39e9}{m} = \num{3.6e-13}$.

\textbf{7.} Modelo Terra: diâmetro
$\qty{1.27e7}{m} \times \num{3.6e-13} = \qty{4.6e-6}{m}$ --- o tamanho
de uma bactéria --- à distância
$\qty{1.496e11}{m} \times \num{3.6e-13} = \qty{5.4}{cm}$ do
grão: a Terra é um micróbio a cinco centímetros de um grão de areia.

\textbf{8.} $\qty{4.5e12}{m} \times \num{3.6e-13} = \qty{1.6}{m}$: o
todo o sistema planetário orbita dentro de uma mesa de jantar.

\textbf{9.} $\qty{4.0e16}{m} \times \num{3.6e-13} = \qty{1.4e4}{m}
= \qty{14}{km}$.

\textbf{10.} A estrela mais próxima do nosso Sistema Solar de mesa é outra
grão de areia a quatorze quilômetros de distância, praticamente sem nada
entre: a galáxia é um espaço esmagadoramente vazio, e é por isso
colisões entre estrelas são quase inéditas.

\textbf{11.} $ t = \qty{5e-4}{m}/\qty{3.00e8}{m/s} = \qty{1.7e-12}{s}$
--- menos de dois picossegundos.

\textbf{12.} Lua: $\qty{3.84e8}{m}/\qty{3.00e8}{m/s} =
\qty{1.28}{s}$. Sol: $\qty{1.496e11}{m}/\qty{3.00e8}{m/s} =
\qty{499}{s} \approx \qty{8.3}{min}$: a luz do sol dura sempre oito minutos
velho.

\textbf{13.} Um ano é $365.25 \times 86400 = \qty{3.156e7}{s}$, então
$\qty{1}{ly} = \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}$; então
$\qty{4.0e16}{m}/\qty{9.47e15}{m} = 4.2$: a distância dos dados é de fato
sobre $\qty{4.2}{ly}$.

\textbf{14.} $\qty{2.4e22}{m}/\qty{9.47e15}{m} = \num{2.5e6}$: dois e
meio milhão anos-luz. Essa luz deixou Andrômeda por volta $2.5$
milhões de anos atrás, quando os primeiros membros do gênero \emph{Homo}
estávamos aprendendo a quebrar pedras --- e não \emph{Homo sapiens} ainda
existia em qualquer lugar.

\textbf{15.} $\qty{3.00e8}{m/s} \times \qty{1e-9}{s} = \qty{0.30}{m}$.
Em um terço de nanossegundo, um sinal cobre no máximo cerca de
$\qty{10}{cm}$, portanto, quaisquer duas partes da máquina que devam conversar dentro
um tique-taque do relógio deve ficar a dez centímetros um do outro:
tamanho dos limites de velocidade.

\textbf{16.} Átomo médio na água:
$\qty{3.0e-26}{kg}/3 = \qty{1.0e-26}{kg}$. Número de átomos:
$\qty{70}{kg}/\qty{1.0e-26}{kg} = \num{7e27}$.

\textbf{17.} Contra a corrente:
$\num{7e27}/\num{8e18} \approx 10^{9}$ --- você contém um bilhão
o valor dos átomos dos grãos. Contra as estrelas:
$\num{7e27}/\num{2e23} = \num{3.5e4}$ --- seus átomos superam todos os
estrelas do universo observável trinta e cinco mil para um.

\textbf{18.} $\text{(you)}/\text{(atom)} =
\qty{1.7}{m}/\qty{2.5e-10}{m} \approx \num{7e9} \approx 10^{10}$,
enquanto $\text{(universe)}/\text{(you)} =
\qty{8.8e26}{m}/\qty{1.7}{m} \approx \num{5e26}$. Na escada você
ficam cerca de dez degraus acima do átomo e quase vinte e sete abaixo do
universo: multiplicativamente, os humanos vivem muito perto de seus átomos.

\textbf{19.} A condição dá
$ x^{2} = 10^{-15} \times 10^{26.9} = 10^{11.9}$, então
$ x = 10^{5.95} \approx \qty{9e5}{m} \approx \qty{900}{km}$ --- quase
exatamente o diâmetro de Ceres, o maior asteróide
($\approx \qty{940}{km}$). A meio caminho da escada entre um núcleo e
o cosmos é um modesto planeta anão.

\textbf{20.} $\qty{8.8e26}{m} \times \num{3.6e-13} = \qty{3.2e14}{m}$,
ou seja\ $\qty{3.2e14}{m}/\qty{1.496e11}{m} \approx \num{2.1e3}$
\omterm{def:g10:orders-of-magnitude:unit}{unidades} astronômicas. Conclusão: encolher o cosmos até que o Sol seja um
grão de areia, e o universo observável ainda se estende por dois mil
vezes a distância real entre a Terra e o Sol --- nenhum modelo em escala traz a
universo até o tamanho humano; apenas potências de dez podem carregá-lo.
\end{solution}
