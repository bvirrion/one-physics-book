\begin{solution}{exo:g10:universal-gravitation:1}
\emph{1.} $F = \num{6.67e-11} \times \dfrac{60 \times 60}{0.80^2}
= \dfrac{\num{2.40e-7}}{0.64} \approx \qty{3.8e-7}{N}$.

\emph{2.} O mosquito \omterm{def:g10:universal-gravitation:weight}{peso} é
$ P = \num{2.5e-6} \times 9.81 \approx \qty{2.5e-5}{N}$ --- sobre $65$
vezes \emph{maior} do que a atração mútua dos dançarinos. Gravidade
entre as pessoas é mais fraco que o de um mosquito \omterm{def:g10:universal-gravitation:weight}{peso}.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:2}

\emph{2.} Sua massa permanece inalterada: \qty{55}{kg} (mesma quantidade de
importa). Dela \omterm{def:g10:universal-gravitation:weight}{peso} torna-se

\end{solution}

\begin{solution}{exo:g10:universal-gravitation:3}
\emph{1.} $ g_{\text{Mars}} = \dfrac{G M}{R^2}
= \dfrac{\num{6.67e-11} \times \num{6.42e23}}{(\num{3.39e6})^2}
= \dfrac{\num{4.28e13}}{\num{1.15e13}} \approx \qty{3.73}{N/kg}$.

\emph{2.} Em Marte: $ P = 900 \times 3.73 \approx \qty{3.4e3}{N}$. Ligado
Terra: $ P = 900 \times 9.81 \approx \qty{8.8e3}{N}$ --- o rover
pesa cerca de $2.6$ vezes menos em Marte, com a mesma massa.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:5}
\emph{1.} Distância duplicada: $ d^2$ é multiplicado por $4$, então
$ F = F_0/4$.

\emph{2.} Distância reduzida pela metade: $ F = 4F_0$.

\emph{3.} Ambas as massas dobraram: $ m_1 m_2$ é multiplicado por $4$, então
$ F = 4F_0$.

\emph{4.} Numerador $\times 3$, denominador $\times 9$:
$ F = F_0/3$.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:6}
\emph{1.} $ F = \num{6.67e-11} \times
\dfrac{(\num{5.0e8})^2}{100^2}
= \num{6.67e-11} \times \dfrac{\num{2.5e17}}{\num{1.0e4}}
\approx \qty{1.7e3}{N}$.

\emph{2.} $ m = F/g = 1667/9.81 \approx \qty{1.7e2}{kg}$: os dois navios
atrair uns aos outros com o \omterm{def:g10:universal-gravitation:weight}{peso} de uma motocicleta grande.

\emph{3.} $ a = F/m = \num{1.7e3}/\num{5.0e8} \approx
\qty{3.3e-6}{m/s^2}$. Mesmo sem oposição, os petroleiros levariam cerca de
cinco dias para acelerar o ritmo de caminhada; na prática, resistência à água e amarrações
inundar a atração completamente.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:7}
\emph{1.} O equilíbrio do feixe indica \qty{3.0}{kg} em ambos os mundos: o
bolsa e as massas padrão veem seus \omterm{def:g10:universal-gravitation:weight}{pesos} dividido pelo mesmo
fator, então a comparação permanece inalterada. A escala de mola lê o
\omterm{def:g10:universal-gravitation:weight}{peso}: $3.0 \times 9.81 \approx \qty{29}{N}$ na Terra,
$3.0 \times 1.62 \approx \qty{4.9}{N}$ na Lua.

\emph{2.} A balança: mede a massa, que é o que o cliente é
comprando. Uma escala de mola calibrada em ``\omterm{def:g10:orders-of-magnitude:unit}{quilogramas}'' na Terra leria
$4.9/9.81 \approx \qty{0.50}{kg}$ para a mesma bolsa na Lua --- o
O lojista daria seis sacolas pelo preço de uma.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:8}
\emph{1.} Distância ao centro:
$ d = R + h = \num{6.37e6} + \num{4.0e5} = \qty{6.77e6}{m}$. Então
$ g = \dfrac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.77e6})^2}
= \dfrac{\num{3.98e14}}{\num{4.58e13}} \approx \qty{8.69}{N/kg}$.

\emph{2.} $8.69/9.81 \approx 89\%$ do valor da superfície.

\emph{3.} A gravidade está quase com força total lá em cima: os astronautas
flutuar porque a estação e tudo nela são \emph{caindo
juntos} ao redor da Terra (queda livre), não porque a gravidade tenha
desapareceu.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:9}
De $ g = GM/R^2$,
\[
M = \frac{g R^2}{G}
= \frac{8.87 \times (\num{6.05e6})^2}{\num{6.67e-11}}
= \frac{\num{3.25e14}}{\num{6.67e-11}}
\approx \qty{4.87e24}{kg},
\]
sobre $0.8$ vezes a massa da Terra --- Vênus é quase a massa da Terra
gêmeo em tamanho e massa.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:10}
\emph{1.} Sol:
$ F = \dfrac{\num{6.67e-11} \times \num{1.99e30}}{(\num{1.496e11})^2}
\approx \qty{5.9e-3}{N}$. Lua:
$ F = \dfrac{\num{6.67e-11} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{3.3e-5}{N}$.

\emph{2.} O Sol puxa $\num{5.9e-3}/\num{3.3e-5} \approx 180$
vezes mais difícil.

\emph{3.} O diâmetro da Terra é muito maior \emph{fração} do
Distância Terra-Lua ($\approx 3\%$) do que a distância Terra-Sol
($\approx 0.01\%$), então a atração da Lua muda muito mais em relação ao próximo
lado para o outro lado do globo. As marés se alimentam dessa diferença ---
e aí a Lua vence.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:11}
\emph{1.} $ g' = \dfrac{G (2M)}{(2R)^2} = \dfrac{2}{4}\,\dfrac{GM}{R^2}
= \dfrac{g}{2} \approx \qty{4.9}{N/kg}$: o raio duplicado vence
a massa dobrada.

\emph{2.} Nós precisamos $\dfrac{GM}{R'^2} = 2\,\dfrac{GM}{R^2}$, então
$ R'^2 = R^2/2$ e $ R' = R/\sqrt{2} \approx \qty{4.5e6}{m}$ --- sobre
\qty{4500}{km}.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:12}
\emph{1.} Trações iguais por \omterm{def:g10:orders-of-magnitude:unit}{quilograma}:
$\dfrac{G M_{\text{Earth}}}{x^2} = \dfrac{G M_{\text{Moon}}}{(D-x)^2}$.
Multiplicação cruzada e divisão por $ G $:
$\left(\dfrac{x}{D-x}\right)^2 = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.

\emph{2.} $\dfrac{x}{D-x}
= \sqrt{\dfrac{\num{5.97e24}}{\num{7.35e22}}} = \sqrt{81.2} \approx
9.0$, então $ x = 9.0\,(D - x)$, ou seja\ $ x = \dfrac{9.0}{10.0}\,D
\approx 0.90 \times \num{3.84e8} \approx \qty{3.5e8}{m}$. O
ponto de equilíbrio fica $90\%$ do caminho para a Lua: a atração da Terra
domina quase toda a viagem.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:13}
\emph{1.} Expandindo: $ R^2 + 2Rh + h^2 = R^2 + x^2$; desde $ h $ é minúsculo
comparado com $ R $, derrubar $ h^2$: $2Rh \approx x^2$, então
$ h \approx \dfrac{x^2}{2R}
= \dfrac{(\num{8.0e3})^2}{2 \times \num{6.37e6}}
= \dfrac{\num{6.4e7}}{\num{1.27e7}} \approx \qty{5.0}{m}$.

\emph{2.} Em um \omterm{def:g10:orders-of-magnitude:unit}{segundo} o projétil cai \qty{4.9}{m} --- quase
exatamente o \qty{5.0}{m} o chão cai \qty{8.0}{km}. Então
em $ v \approx \qty{8.0}{km/s}$ a queda nunca atinge o chão: o
projétil \omterm{def:g10:universal-gravitation:satellite}{órbitas}.

\emph{3.} $ T = \dfrac{2\pi R}{v}
= \dfrac{2\pi \times \num{6.37e6}}{\num{8.0e3}}
\approx \qty{5.0e3}{s} \approx \qty{83}{min}$ --- perto do aproximadamente
$\qty{90}{min}$ de verdade \omterm{def:g10:universal-gravitation:satellite}{órbita} baixa \omterm{def:g10:universal-gravitation:satellite}{satélites}, que voam um pouco
mais alto.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:14}

\approx \qty{1.3e12}{N/kg}.$

\begin{solution}{exo:g10:universal-gravitation:15}
\emph{1.} A distância é $60$ vezes maior, então a atração por \omterm{def:g10:orders-of-magnitude:unit}{quilograma}
é $60^2 = 3600$ vezes mais fraco.

\emph{2.} A queda em um \omterm{def:g10:orders-of-magnitude:unit}{segundo} é dimensionada da mesma maneira:
$4.9/3600 \approx \qty{1.4e-3}{m}$ --- sobre \qty{1.4}{mm}.

\emph{3.} $ T = 27.3 \times \qty{86400}{s} = \qty{2.36e6}{s}$, então
$ v = \dfrac{2\pi \times \num{3.84e8}}{\num{2.36e6}} \approx
\qty{1.02e3}{m/s}$, e $ x = \qty{1.02e3}{m}$ em um \omterm{def:g10:orders-of-magnitude:unit}{segundo}. A queda
é $ h \approx \dfrac{x^2}{2d}
= \dfrac{(\num{1.02e3})^2}{2 \times \num{3.84e8}}
\approx \qty{1.4e-3}{m}$.

\emph{4.} Os dois números concordam: a Lua cai em direção à Terra cada
\omterm{def:g10:orders-of-magnitude:unit}{segundo} exatamente tanto quanto a gravidade do inverso do quadrado, calibrada em um
maçã caindo, prevê. Newton concluiu que a força que segurava o
Lua \emph{é} a força que derruba a maçã --- a gravidade é
universais.
\end{solution}

\begin{solution}{pb:g10:universal-gravitation:1}
\textbf{1.} $ F = \num{6.67e-11} \times
\dfrac{\num{7.3e6} \times 55}{100^2} \approx \qty{2.7e-6}{N}$.

\textbf{2.} O grão de areia pesa
$\num{1.0e-6} \times 9.81 \approx \qty{9.8e-6}{N}$: toda a Eiffel
Torre puxa o turista $3.7$ vezes \emph{menos} do que o \omterm{def:g10:universal-gravitation:weight}{peso}
de um grão de areia.

\textbf{3.} (a) Distância $\times 10$: força dividida por $100$.
(b) Ambas as massas $\times 1000$: força multiplicada por $10^6$.
(c) Missas $\times 1000$ e distância $\times 1000$:
$10^6 / 10^6$ --- a força permanece inalterada.

\textbf{4.} $ F = \dfrac{\num{6.67e-11} \times \num{5.97e24} \times
1.00}{(\num{6.37e6})^2} \approx \qty{9.81}{N}$: o \omterm{def:g10:universal-gravitation:weight}{peso} de um
\omterm{def:g10:orders-of-magnitude:unit}{quilograma} --- esta é exatamente a força do campo
$ g = \qty{9.81}{N/kg}$.

\textbf{5.} A gravidade é fraca por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} mas só sempre
\emph{atrai}, então a atração de todos $\num{5.97e24}$ \omterm{def:g10:orders-of-magnitude:unit}{quilogramas} de
A Terra se soma na mesma direção --- e uma massa astronômica bate
um vizinho por vinte e duas potências de dez.

\textbf{6.} $ F = \num{6.67e-11} \times \dfrac{158 \times 0.73}{0.23^2}
= \num{6.67e-11} \times \dfrac{115.3}{0.0529}
\approx \qty{1.5e-7}{N}$.

\textbf{7.} A pequena esfera pesa
$0.73 \times 9.81 \approx \qty{7.2}{N}$ --- sobre $\num{5e7}$ vezes
a atração a ser detectada. Nenhuma escala panorâmica resolve uma parte
cinquenta milhões; um fio de torção fino, que torce visivelmente sob
torques em escala de nanonewtons, podem.

\textbf{8.} $ G = \dfrac{F d^2}{m_1 m_2}
= \dfrac{\num{1.47e-7} \times 0.0529}{115.3}
\approx \qty{6.74e-11}{N.m^2/kg^2}$ --- dentro de cerca de $1\%$ do
moderno \num{6.67e-11}, de um galpão de madeira em 1798.

\textbf{9.} De $ g = G M / R^2$:
\[
M = \frac{g R^2}{G}
= \frac{9.81 \times (\num{6.37e6})^2}{\num{6.67e-11}}
\approx \qty{5.97e24}{kg}.
\]
Seis milhões de bilhões de bilhões \omterm{def:g10:orders-of-magnitude:unit}{quilogramas}: a Terra, pesada.

\textbf{10.} $ V = \frac43 \pi R^3 = \frac43 \pi
(\num{6.37e6})^3 \approx \qty{1.08e21}{m^3}$, então
$\rho = \num{5.97e24}/\num{1.08e21} \approx \qty{5.5e3}{kg/m^3}$ ---
duas vezes a densidade da rocha superficial. O interior deve esconder algo
muito mais denso que o granito: a Terra tem um núcleo pesado (de ferro).

\textbf{11.} $ g_{\text{Moon}} = \dfrac{\num{6.67e-11} \times
\num{7.35e22}}{(\num{1.74e6})^2} = \dfrac{\num{4.90e12}}
{\num{3.03e12}} \approx \qty{1.62}{N/kg}$.

\textbf{12.} Altura $\propto 1/g $:
$ h = 0.50 \times \dfrac{9.81}{1.62} \approx \qty{3.0}{m}$ --- um
saltar sobre uma cesta de basquete, em câmera lenta.

\textbf{13.} $ g_{\text{Mercury}} = \dfrac{\num{6.67e-11} \times
\num{3.30e23}}{(\num{2.44e6})^2} = \dfrac{\num{2.20e13}}
{\num{5.95e12}} \approx \qty{3.70}{N/kg}$ --- quase exatamente o de Marte
\qty{3.73}{N/kg}. Mercúrio tem menos massa, mas seu raio menor (um
menor $ R^2$ abaixo) compensa: é um mundo muito mais denso.

\textbf{14.} Metade da intensidade do campo requer
$(R + h)^2 = 2 R^2$, então $ R + h = R\sqrt{2}$ e
$ h = (\sqrt{2} - 1) R \approx 0.414 \times \num{6.37e6}
\approx \qty{2.6e6}{m}$ --- sobre \qty{2600}{km} acima.

\textbf{15.} $\dfrac{G M_{\text{Earth}}}{d^2} = 1.62$ dá
$ d = \sqrt{\dfrac{\num{6.67e-11} \times \num{5.97e24}}{1.62}}
= \sqrt{\num{2.46e14}} \approx \qty{1.57e7}{m} \approx 2.5$ Terra
raios. A apenas um raio e meio acima do solo, a Terra
não puxa com mais força do que a superfície da Lua.

\textbf{16.} $ v = \dfrac{2\pi d}{T}
= \dfrac{2\pi \times \num{1.496e11}}{\num{3.156e7}}
\approx \qty{2.98e4}{m/s}$ --- trinta quilômetros a cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}.

\textbf{17.} $\dfrac{v^2}{d} = \dfrac{(\num{2.98e4})^2}{\num{1.496e11}}
\approx \qty{5.9e-3}{N/kg}$: o Sol puxa cada um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} da Terra
com cerca de seis milinewtons.

\textbf{18.} $\dfrac{G M_{\text{Sun}}}{d^2} = \dfrac{v^2}{d}$ dá
\[
M_{\text{Sun}} = \frac{v^2 d}{G}
= \frac{\num{8.87e8} \times \num{1.496e11}}{\num{6.67e-11}}
\approx \qty{1.99e30}{kg}.
\]

\textbf{19.} $\dfrac{\num{1.99e30}}{\num{5.97e24}} \approx
\num{3.3e5}$: o Sol é um terço de um milhão de Terras.

\textbf{20.} Porque o mesmo $ G$ governa cada par de massas,
medindo-o uma vez entre duas esferas de chumbo em um galpão, vamos colocar o
escala primeiro o planeta sob nossos pés, depois a estrela que \omterm{def:g10:universal-gravitation:satellite}{órbita} ---
isso é o que \emph{universal} vale a pena.
\end{solution}
