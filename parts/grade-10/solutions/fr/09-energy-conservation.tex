\section*{Chapitre \ref{ch:g10:energy-conservation} --- Énergie~: formes et conservation}

\begin{solution}{exo:g10:energy-conservation:1}
$E_k = \tfrac12 \times 1200 \times 25^2 = \qty{3.75e5}{J}$. À
\qty{36.1}{m/s}~: $\tfrac12 \times 1200 \times 36.1^2 \approx
\qty{7.8e5}{J}$ --- $44\%$ de vitesse en plus, $(130/90)^2 \approx 2.1$
fois l'\omterm{def:g10:energy-conservation:kinetic}{énergie cinétique}~: le carré au travail.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:2}
$\Delta h = \qty{450}{m}$~:
$E_p = 72 \times 9.81 \times 450 \approx \qty{3.2e5}{J}
= \num{3.2e5}/\num{3.6e6} \approx \qty{0.088}{kW.h}$, valant environ
$0{,}022$~euro. Une journée de montée, c'est deux centimes
d'électricité.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:3}
$\qty{1}{kW.h} = \qty{3.6e6}{J}$~;
$\qty{12}{W.h} = 12 \times \qty{3600}{J} \approx \qty{43}{kJ}$~;
$\qty{45}{MJ} = 45/3.6 = \qty{12.5}{kW.h}$~;
$\qty{10}{MJ} = 10/3.6 \approx \qty{2.8}{kW.h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:4}
$E = 2200 \times 150 = \qty{3.3e5}{J}
= \num{3.3e5}/\num{3.6e6} \approx \qty{0.092}{kW.h}$. Humain~:
$P = \num{1.0e7}/\num{86400} \approx \qty{116}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:5}
Dynamo~: chimique (nourriture du cycliste) $\to$ cinétique (roue)
$\to$ électrique (dynamo) $\to$ radiative $+$ thermique (lampe)~; des
fuites de chaleur aux muscles, au contact du pneu, à la dynamo, à la
lampe. Chauffe-eau à gaz~: chimique (gaz) $\to$ thermique (eau), fuite
dans les gaz de combustion. Panneau~: radiative (Soleil) $\to$
électrique (panneau) $\to$ chimique (batterie), fuites de chaleur au
panneau et au chargeur.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:6}
Consommée $2.0 \times 45 = \qty{90}{MJ}$~; $\eta = 27/90 = 0.30$.
Chaleur~: $90 - 27 = \qty{63}{MJ}$, emportée par les gaz d'échappement
et le radiateur dans l'air environnant.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:7}
$E_k = E_p = 70 \times 9.81 \times 20 \approx \qty{1.37e4}{J}$~;
$v = \sqrt{2 \times 13734/70} = \sqrt{392} \approx \qty{19.8}{m/s}
\approx \qty{71}{km/h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:8}
$41562 - 41236 = \qty{326}{kW.h}$~; journalier $326/30 \approx
\qty{10.9}{kW.h}$. Facture~: $326 \times 0.26 + 12 = 96{,}76$~euros.
\omterm{def:g10:energy-conservation:power}{Puissance} moyenne~: $\qty{326}{kW.h}/\qty{720}{h} \approx
\qty{453}{W}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:9}
\emph{1.} Lumière~: $0.05 \times 60 = \qty{3.0}{W}$~; LED~:
$\eta = 3/9 \approx 0.33$.
\emph{2.} Annuel ($\qty{1460}{h}$)~: ampoule
$60 \times 1460 = \qty{87.6}{kW.h}$, LED $\qty{13.1}{kW.h}$~;
économie $\qty{74.5}{kW.h} \approx 19$~euros par ampoule par an.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:10}
\emph{1.} $P = \num{1.5e5} \times 9.81 \times 40 \approx
\qty{5.9e7}{W} = \qty{59}{MW}$.
\emph{2.} $0.90 \times 59 \approx \qty{53}{MW}$~;
$\num{5.3e7}/525 \approx \num{1.0e5}$ foyers.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:11}
$v = \sqrt{2 \times 9.81 \times 0.050} \approx \qty{0.99}{m/s}$
(la masse s'annule~: $\tfrac12 m v^2 = mgh$). L'$E_p$ initiale
$= 0.20 \times 9.81 \times 0.050 \approx \qty{0.098}{J}$ finit en
\omterm{def:g10:energy-conservation:forms}{énergie thermique} dans l'air et le pivot~: le pendule n'est
pas isolé, et le total --- balancement plus chaleur --- est
exactement conservé.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:12}
\emph{1.} $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.304$~;
consommée $12/0.304 \approx \qty{39.5}{W.h}
= 39.5 \times 3600 \approx \qty{142}{kJ}$.
\emph{2.} $\num{1.42e5}/\num{2.5e7} \approx \qty{5.7}{g}$ de charbon
par charge~; $\times 365 \approx \qty{2.1}{kg}$ par an.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:13}
\emph{1.} $\num{5e9}/\num{3.6e6} \approx \qty{1400}{kW.h}$~;
$1400/12.6 \approx 110$~jours.
\emph{2.} $P = \num{5e9}/\num{1e-4} = \qty{5e13}{W} = \qty{50}{TW}$
--- dix-sept fois la \omterm{def:g10:energy-conservation:power}{puissance} électrique moyenne de
l'humanité, pendant \qty{100}{\micro s}, en un lieu imprévisible. La
foudre, c'est beaucoup de \omterm{def:g10:energy-conservation:power}{puissance} mais peu
d'\omterm{def:g10:energy-conservation:energy}{énergie}, livrée inutilement vite~: aucun
\omterm{def:g10:energy-conservation:transfer}{convertisseur} n'accepte un tel pic.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:14}
\emph{1.} $E_k = \tfrac12 \times 1300 \times 30.6^2 \approx
\qty{6.1e5}{J}$, convertie en \omterm{def:g10:energy-conservation:forms}{énergie thermique} dans les
disques et plaquettes de frein (puis l'air).
\emph{2.} $0.60 \times \qty{6.1e5}{J} \approx \qty{3.7e5}{J}$ par
arrêt~; $\num{3.6e6}/\num{3.7e5} \approx 10$ arrêts par
\omterm{def:g10:energy-conservation:kwh}{kilowatt-heure}.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:15}
$E_p = 85 \times 9.81 \times 1200 \approx \qty{1.0}{MJ}$~; nourriture~:
$1.0/0.25 = \qty{4.0}{MJ}$, c'est-à-dire $4.0/1.5 \approx 2{,}7$ bols
de céréales --- ou $4.0/45 \approx \qty{0.09}{kg}$ d'essence. Toute une
matinée d'effort honnête tient dans une tasse de café de carburant~:
l'\omterm{def:g10:energy-conservation:forms}{énergie chimique} est absurde de concentration.
\end{solution}

\begin{solution}{pb:g10:energy-conservation:1}
\textbf{1.} Frigo $0.1 \times 24 = \qty{2.4}{kW.h}$~; chauffe-eau
$2.4 \times 2.5 = \qty{6.0}{kW.h}$~; four \qty{2.0}{kW.h}~; machine à
laver \qty{1.0}{kW.h}~; éclairage \qty{0.3}{kW.h}~; écrans
\qty{0.9}{kW.h}.
\textbf{2.} Total \qty{12.6}{kW.h} $= 12.6 \times 3.6 \approx
\qty{45}{MJ}$~: un \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} d'essence, au
\omterm{def:g10:energy-conservation:energy}{joule} près.
\textbf{3.} $\qty{12600}{W.h}/\qty{24}{h} = \qty{525}{W}$ --- la maison
tourne au quart d'une bouilloire.
\textbf{4.} Le chauffe-eau (\qty{6.0}{kW.h}, près de la moitié). À
\qty{1.25}{h}~: chauffe-eau \qty{3.0}{kW.h}, total \qty{9.6}{kW.h}.
\textbf{5.} $15 \times 24 = \qty{0.36}{kW.h}$ par jour, environ
$2{,}8\%$ du total compté~; annuel $0.36 \times 365 \approx
\qty{131}{kW.h} \approx 33$~euros --- pour des voyants que personne ne
regarde.
\textbf{6.} $56604 - 56214 = \qty{390}{kW.h}$~;
$390/30 = \qty{13.0}{kW.h}$ par jour~: les \qty{12.6}{kW.h} de
l'audit plus les \qty{0.36}{kW.h} de veille --- confirmé.
\textbf{7.} $390 \times 0.25 + 14 = 111{,}50$~euros.
\textbf{8.} Réseau~: $0.25/3.6 \approx 0{,}07$~euro par mégajoule.
Essence~: $1.80/34 \approx 0{,}05$~euro par mégajoule.
\textbf{9.} Bol~: $0.50/1.5 \approx 0{,}33$~euro par mégajoule ---
cinq fois le réseau. Moins cher d'abord~: essence, réseau, puis
nourriture, loin derrière.
\textbf{10.} $0.20 \times 1000 = \qty{200}{W}$ par
\omterm{def:g10:orders-of-magnitude:unit}{mètre} carré.
\textbf{11.} $200 \times 4.0 = \qty{800}{W.h} = \qty{0.80}{kW.h}$ par
\omterm{def:g10:orders-of-magnitude:unit}{mètre} carré par jour.
\textbf{12.} $12.6/0.80 = 15{,}75 \approx \qty{16}{m^2}$~: cela tient
sur le toit de \qty{20}{m^2}.
\textbf{13.} $16 \times 0.2 \times 1.5 = \qty{4.8}{kW.h}$~: environ
$38\%$ du besoin~; le réseau (ou une batterie chargée les meilleurs
mois) doit fournir le reste.
\textbf{14.} Un \omterm{def:g10:orders-of-magnitude:unit}{mètre} carré livre $0.80 \times 365 \approx
\qty{292}{kW.h}$ par an~; amortissement $700/292 \approx 2{,}4$~ans ---
court face à une durée de vie de panneau de $25$~ans.
\textbf{15.} $\qty{12.6}{kW.h}/\qty{0.1}{kW} = \qty{126}{h}$~: plus de
cinq personnes ($126/24 = 5{,}25$) pédalant jour et nuit.
\textbf{16.} Nourriture $= 12.6/0.25 = \qty{50.4}{kW.h} \approx
\qty{181}{MJ}$, c'est-à-dire $181/1.5 \approx 121$ bols de céréales.
\textbf{17.} $121 \times 0.50 \approx 60$~euros, contre
$12.6 \times 0.25 \approx 3{,}15$~euros du réseau~: la
\omterm{def:g10:energy-conservation:power}{puissance} pédalée coûte environ $19$~fois plus --- avant
de payer les cyclistes.
\textbf{18.} $5000/45 \approx 110$ jours-maison. Mais
$\num{5e9}/\num{1e-4} = \qty{5e13}{W}$ pendant \qty{100}{\micro s}, en
un lieu aléatoire~: aucun \omterm{def:g10:energy-conservation:transfer}{convertisseur} ni batterie ne
peut boire à ce tuyau. Une maison a besoin d'un flux stable de
\qty{525}{W}, non d'un pic.
\textbf{19.} Pas les \omterm{def:g10:energy-conservation:energy}{joules} --- ils sont conservés et
identiques. On paie leur \emph{forme et leur disponibilité}~:
l'\omterm{def:g10:energy-conservation:energy}{énergie} concentrée, stockable et à la demande coûte
toute la \omterm{def:g10:energy-conservation:transfer}{chaîne de conversion} qui l'a rendue telle.
\textbf{20.} Un jour de famille $= \qty{12.6}{kW.h} \approx
\qty{45}{MJ}$~: un \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} d'essence, seize
\omterm{def:g10:orders-of-magnitude:unit}{mètres} carrés de toit ensoleillé, ou cinq cyclistes
jour et nuit. Le stock est l'\omterm{def:g10:energy-conservation:energy}{énergie}
(\omterm{def:g10:energy-conservation:energy}{joules}, \omterm{def:g10:energy-conservation:kwh}{kilowatt-heures})~; le flux est
la \omterm{def:g10:energy-conservation:power}{puissance} (\omterm{def:g10:energy-conservation:power}{watts}) --- la baignoire et le
robinet, à ne plus jamais confondre.
\end{solution}
