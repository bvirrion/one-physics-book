\section*{Chapitre \ref{ch:g10:orders-of-magnitude} --- Ordres de grandeur~: mesurer l'Univers}

\begin{solution}{exo:g10:orders-of-magnitude:1}
\emph{1.} $\qty{750}{nm} = 750 \times \qty{e-9}{m} = \qty{7.50e-7}{m}$.
\emph{2.} $\qty{2.5}{km} = \qty{2.5e3}{m}$.
\emph{3.} $\qty{3.2}{\micro m} = \qty{3.2e-6}{m}$.
\emph{4.} $\qty{0.15}{Tm} = 0.15 \times \qty{e12}{m} = \qty{1.5e11}{m}$
--- une \omterm{def:g10:orders-of-magnitude:lightyear}{unité astronomique}.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:2}
\emph{1.} $\qty{149600000}{km} = \qty{1.496e11}{m}$~: $4$ \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres
significatifs}.
\emph{2.} $\qty{1.06e-10}{m}$~: $3$ \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs} (les zéros
de tête ne comptent jamais).
\emph{3.} $\num{8.1e9}$~: $2$ \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs}.
\emph{4.} $\qty{5.20e-3}{s}$~: $3$ \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs} (le zéro
de queue est une promesse).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:3}
\omterm{def:g10:orders-of-magnitude:scinot}{Mantisses} et \omterm{def:g10:orders-of-magnitude:scinot}{exposants} séparément
(\cref{prop:g10:orders-of-magnitude:powers})~:
\[
3 \times 2 \times 10^{8-5} = \num{6e3}, \qquad
\tfrac{8}{4} \times 10^{4-(-3)} = \num{2e7},
\]
\[
5^{2} \times 10^{12} = \num{2.5e13}, \qquad
\tfrac{4 \times 3}{6} \times 10^{-3-9+7} = \num{2e-5} .
\]
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:4}
Humain~: $1.7 < 5$, ordre $\qty{e0}{m}$. Everest~:
$\qty{8848}{m} = \qty{8.848e3}{m}$ et $8.848 \geq 5$~: ordre
$\qty{e4}{m}$. Bactérie~: ordre $\qty{e-6}{m}$. Terre~:
$5.97 \geq 5$, ordre $\qty{e25}{kg}$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:5}
\emph{1.} Une demi-graduation~: $L = (127.0 \pm 0.5)\,\unit{mm}$.
\emph{2.} $\Delta L / L = 0.5/127 \approx 0.4\%$.
\emph{3.} Gomme~: $0.5/12 \approx 4\%$. La mesure du crayon est
environ dix fois plus précise --- même règle, même \omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude
absolue}, objet dix fois plus long.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:6}
\emph{1.} Proxima~:
$4.24 \times \qty{9.47e15}{m} = \qty{4.01e16}{m}$. Sirius~:
$8.6 \times \qty{9.47e15}{m} = \qty{8.1e16}{m}$ (deux \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres
significatifs}, comme l'entrée).
\emph{2.} $t = d/c = \qty{2.4e13}{m} / \qty{3.00e8}{m/s}
= \qty{8.0e4}{s}$, c'est-à-dire $\num{8.0e4}/3600 \approx \qty{22}{h}$~:
presque une journée entière pour chaque message de la sonde.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:7}
$t = d/c$ à chaque fois. Lune~:
$\qty{3.84e8}{m}/\qty{3.00e8}{m/s} = \qty{1.28}{s}$. Soleil~:
$\qty{1.496e11}{m}/\qty{3.00e8}{m/s} = \qty{499}{s} \approx
\qty{8.3}{min}$. Neptune~:
$\qty{4.50e12}{m}/\qty{3.00e8}{m/s} = \qty{1.50e4}{s} \approx
\qty{4.2}{h}$. Lorsque vous regardez Neptune, vous la voyez telle
qu'elle était il y a environ quatre heures --- tout télescope est une
machine à remonter le temps en douceur.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:8}
\emph{1.} Une feuille~: $\qty{52.0}{mm}/500 = \qty{0.104}{mm}
= \qty{1.04e-4}{m}$. L'incertitude se divise aussi~:
$\qty{0.5}{mm}/500 = \qty{0.001}{mm}$, donc
$t = (104 \pm 1)\,\unit{\micro m}$.
\emph{2.} $1/104 \approx 1\%$.
\emph{3.} Mesurer une feuille directement donnerait
$(0.1 \pm 0.5)\,\unit{mm}$~: une \omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude relative} d'environ
$500\%$ --- la règle ne peut même pas prouver que la feuille existe.
Empiler $500$ feuilles multiplie la longueur mesurée par $500$ tandis
que l'\omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude absolue} de la règle reste fixe~: l'\omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude
relative} se réduit par ce même facteur $500$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:9}
Diviser chaque longueur par $\qty{2e-10}{m}$. Bactérie~:
$\qty{2.0e-6}{m}/\qty{2e-10}{m} = \num{1.0e4}$ atomes. Cheveu~:
$\qty{7.0e-5}{m}/\qty{2e-10}{m} = \num{3.5e5}$. Humain~:
$1.7/\num{2e-10} = \num{8.5e9}$, \omterm{def:g10:orders-of-magnitude:oom}{ordre de grandeur} $10^{10}$~: dix
milliards d'atomes de la tête aux pieds.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:10}
\emph{1.} $s = \qty{0.22}{m}/\qty{1.39e9}{m} = \num{1.58e-10}$.
\emph{2.} Terre~:
$\qty{1.27e7}{m} \times \num{1.58e-10} = \qty{2.0e-3}{m} =
\qty{2.0}{mm}$ --- un grain de poivre --- orbitant à
$\qty{1.496e11}{m} \times \num{1.58e-10} = \qty{23.7}{m} \approx
\qty{24}{m}$ du ballon.
\emph{3.} Lune~:
$\qty{3.84e8}{m} \times \num{1.58e-10} = \qty{6.1}{cm}$ du grain de
poivre. Neptune~:
$\qty{4.50e12}{m} \times \num{1.58e-10} \approx \qty{710}{m}$~: la
maquette ne tient pas dans la cour de l'école.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:11}
\emph{1.} $21.0 \times 29.7 = \qty{623.7}{cm^2}$, arrondi aux trois
\omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs} des entrées~: $\qty{624}{cm^2}$.
\emph{2.} $v = \qty{100.0}{m}/\qty{12.3}{s} = \qty{8.13}{m/s}$ (trois
chiffres, fixés par le temps).
\emph{3.} Cinq \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs} prétendraient l'aire à
$\qty{0.01}{cm^2}$, une précision que les entrées au millimètre n'ont
jamais fournie --- une décimale forgée reste un faux.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:12}
\emph{1.} Rayon $r = \qty{0.55}{m}$, aire
$A = \pi r^{2} = \pi \times 0.55^{2} \approx \qty{0.95}{m^2}$. Volume
$V = \qty{1.0}{mm^3} = \qty{1.0e-9}{m^3}$, et un film est (aire)
$\times$ (épaisseur), donc
\[
e = \frac{V}{A} = \frac{\qty{1.0e-9}{m^3}}{\qty{0.95}{m^2}}
= \qty{1.1e-9}{m} .
\]
\emph{2.} Une molécule est d'ordre $\qty{e-9}{m}$ --- un barreau
au-dessus de l'atome à $\qty{e-10}{m}$, cohérent avec une molécule
comme petit amas d'atomes. La cuillerée d'huile de Franklin a calmé
un demi-acre d'étang et, sans qu'il le sache, a mesuré le nanomètre.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:13}
Hypothèses~: environ $70$ battements par minute au repos, une vie de
$80$~ans. Par jour~: $70 \times 60 \times 24 \approx \num{1e5}$
battements. Par an~: $\num{1e5} \times 365 \approx \num{4e7}$. Par
vie~: $\num{4e7} \times 80 \approx \num{3e9}$ --- \omterm{def:g10:orders-of-magnitude:oom}{ordre de grandeur}
$10^{9}$, quelques milliards de battements. Quatre \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres
significatifs} prétendraient connaître une fréquence cardiaque, minute
par minute, pendant quatre-vingts ans~: les entrées sont des
estimations d'ordre de grandeur, et aucune sortie ne peut surclasser
ses entrées (\cref{met:g10:orders-of-magnitude:sigfig}).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:14}
Après $n$ plis l'épaisseur est $2^{n} \times \qty{1.0e-4}{m}$.
\emph{1.} Il faut $2^{n} \geq 8848/\num{1.0e-4} = \num{8.8e7}$. Comme
$2^{26} \approx \num{6.7e7}$ est insuffisant et
$2^{27} \approx \num{1.3e8}$ suffit~: $27$ plis (pile
$\approx \qty{13}{km}$).
\emph{2.} Il faut $2^{n} \geq \num{3.84e12}$~: $2^{41} \approx
\num{2.2e12}$ est court, $2^{42} \approx \num{4.4e12}$ atteint ---
$42$ plis, pile $\qty{4.4e8}{m}$, largement au-delà de la Lune.
\emph{3.} Seulement l'expérience~: chaque pli divise l'aire par deux,
et après sept plis la pile est plus épaisse que large. L'arithmétique
du doublement n'est pas touchée --- la croissance exponentielle
dépasse simplement ce que le papier (et l'intuition) peuvent suivre.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:15}
\emph{1.} Lune~: $\qty{0.03}{m}/\qty{3.84e8}{m} = \num{7.8e-11}$.
Table~: $\qty{0.001}{m}/\qty{1.000}{m} = \num{1.0e-3}$. La mesure
lunaire est plus précise d'un facteur
$\num{1.0e-3}/\num{7.8e-11} \approx \num{1.3e7}$ --- plus de dix
millions.
\emph{2.} Égaler $\num{7.8e-11}$ sur une table d'un \omterm{def:g10:orders-of-magnitude:unit}{mètre} signifie
$\Delta x = \qty{7.8e-11}{m}$, plus petit qu'un atome
($\approx \qty{e-10}{m}$)~: le bord de la table n'est même pas
\emph{défini} aussi finement. Relativement, la distance Terre--Lune
par laser est l'une des plus belles mesures que notre espèce possède.
\end{solution}

\begin{solution}{pb:g10:orders-of-magnitude:1}
\textbf{1.} Grain~: $\qty{0.5}{mm} = \qty{5e-4}{m}$, à l'échelle
milli~; atome~: $\qty{2.5e-10}{m} = \qty{0.25}{nm}$, à l'échelle nano.

\textbf{2.} $\qty{5e-4}{m}/\qty{2.5e-10}{m} = \num{2.0e6}$~: deux
millions d'atomes en travers.

\textbf{3.} Un cube de $2 \times 10^{6}$ atomes par arête contient
$(2\times10^{6})^{3} = 8 \times 10^{18}$ atomes~: \omterm{def:g10:orders-of-magnitude:oom}{ordre de grandeur}
$10^{19}$.

\textbf{4.} $V = (\qty{5e-4}{m})^{3} = \qty{1.25e-10}{m^3}$, donc
$m = \rho V = \qty{2500}{kg/m^3} \times \qty{1.25e-10}{m^3}
= \qty{3.1e-7}{kg} = \qty{0.31}{mg}$.

\textbf{5.} Masse par atome~:
$\qty{3.1e-7}{kg}/(8\times10^{18}) = \qty{3.9e-26}{kg}$ ---
parfaitement dans la fourchette «~quelques
$10^{-26}\,\unit{kg}$~». Le cube naïf d'atomes reproduit les masses
atomiques réelles~: le modèle passe le contrôle de vraisemblance
(\cref{met:g10:orders-of-magnitude:sanity}) avec les honneurs.

\textbf{6.} $s = \qty{5e-4}{m}/\qty{1.39e9}{m} = \num{3.6e-13}$.

\textbf{7.} Terre modèle~: diamètre
$\qty{1.27e7}{m} \times \num{3.6e-13} = \qty{4.6e-6}{m}$ --- la taille
d'une bactérie --- à une distance
$\qty{1.496e11}{m} \times \num{3.6e-13} = \qty{5.4}{cm}$ du grain~: la
Terre est un microbe à cinq centimètres d'un grain de sable.

\textbf{8.} $\qty{4.5e12}{m} \times \num{3.6e-13} = \qty{1.6}{m}$~: le
système planétaire tout entier orbite dans une table de salle à manger.

\textbf{9.} $\qty{4.0e16}{m} \times \num{3.6e-13} = \qty{1.4e4}{m}
= \qty{14}{km}$.

\textbf{10.} L'étoile la plus proche de notre Système solaire de table
est un autre grain de sable à quatorze kilomètres, avec essentiellement
rien entre les deux~: la galaxie est un vide écrasant, c'est pourquoi
les collisions entre étoiles sont presque inconnues.

\textbf{11.} $t = \qty{5e-4}{m}/\qty{3.00e8}{m/s} = \qty{1.7e-12}{s}$
--- moins de deux picosecondes.

\textbf{12.} Lune~: $\qty{3.84e8}{m}/\qty{3.00e8}{m/s} =
\qty{1.28}{s}$. Soleil~: $\qty{1.496e11}{m}/\qty{3.00e8}{m/s} =
\qty{499}{s} \approx \qty{8.3}{min}$~: la lumière du Soleil a toujours
huit minutes d'âge.

\textbf{13.} Une année compte $365.25 \times 86400 = \qty{3.156e7}{s}$,
donc $\qty{1}{ly} = \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}$~; alors
$\qty{4.0e16}{m}/\qty{9.47e15}{m} = 4.2$~: la distance des données est
bien environ $\qty{4.2}{ly}$.

\textbf{14.} $\qty{2.4e22}{m}/\qty{9.47e15}{m} = \num{2.5e6}$~: deux
millions et demi d'\omterm{def:g10:orders-of-magnitude:lightyear}{années-lumière}. Cette lumière a quitté
Andromède il y a environ $2.5$~millions d'années, lorsque les premiers
membres du genre \emph{Homo} apprenaient à tailler la pierre --- et
aucun \emph{Homo sapiens} n'existait encore nulle part.

\textbf{15.} $\qty{3.00e8}{m/s} \times \qty{1e-9}{s} = \qty{0.30}{m}$.
En un tiers de nanoseconde un signal couvre au plus environ
$\qty{10}{cm}$, donc deux parties de la machine qui doivent
communiquer en un tic d'horloge doivent se trouver à moins de dix
centimètres l'une de l'autre~: la vitesse borne la taille.

\textbf{16.} Atome moyen dans l'eau~:
$\qty{3.0e-26}{kg}/3 = \qty{1.0e-26}{kg}$. Nombre d'atomes~:
$\qty{70}{kg}/\qty{1.0e-26}{kg} = \num{7e27}$.

\textbf{17.} Face au grain~:
$\num{7e27}/\num{8e18} \approx 10^{9}$ --- vous contenez l'équivalent
d'un milliard de grains en atomes. Face aux étoiles~:
$\num{7e27}/\num{2e23} = \num{3.5e4}$ --- vos atomes surpassent toutes
les étoiles de l'univers observable de trente-cinq mille à un.

\textbf{18.} $\text{(vous)}/\text{(atome)} =
\qty{1.7}{m}/\qty{2.5e-10}{m} \approx \num{7e9} \approx 10^{10}$,
tandis que $\text{(univers)}/\text{(vous)} =
\qty{8.8e26}{m}/\qty{1.7}{m} \approx \num{5e26}$. Sur l'échelle vous
êtes à environ dix barreaux au-dessus de l'atome et près de
vingt-sept sous l'univers~: de façon multiplicative, les humains
vivent très près de leurs atomes.

\textbf{19.} La condition donne
$x^{2} = 10^{-15} \times 10^{26.9} = 10^{11.9}$, donc
$x = 10^{5.95} \approx \qty{9e5}{m} \approx \qty{900}{km}$ ---
presque exactement le diamètre de Cérès, le plus grand astéroïde
($\approx \qty{940}{km}$). À mi-chemin sur l'échelle entre un noyau
et le cosmos se trouve une modeste planète naine.

\textbf{20.} $\qty{8.8e26}{m} \times \num{3.6e-13} = \qty{3.2e14}{m}$,
c'est-à-dire $\qty{3.2e14}{m}/\qty{1.496e11}{m} \approx \num{2.1e3}$
\omterm{def:g10:orders-of-magnitude:lightyear}{unités astronomiques}. Chute~: réduire le cosmos jusqu'à ce que
le Soleil soit un grain de sable, et l'univers observable s'étend
encore sur deux mille fois la vraie distance Terre--Soleil --- aucune
maquette ne ramène l'univers à l'échelle humaine~; seules les
puissances de dix peuvent le porter.
\end{solution}
