\section*{Chapitre \ref{ch:g10:pressure} --- Pression~: du sport à la plongée}

\begin{solution}{exo:g10:pressure:1}
Grande face~: $S = \qty{0.15}{m^2}$, $P = 600/0.15 = \qty{4.0}{kPa}$.
Petite face~: $S = \qty{0.060}{m^2}$, $P = \qty{10}{kPa}$. Sur le
sable, la grande face~: \omterm{def:g10:pressure:pressure}{pression} plus basse, moins d'enfoncement.
\end{solution}

\begin{solution}{exo:g10:pressure:2}
$\qty{0.25}{MPa} = \qty{250}{kPa} = \qty{2.5e5}{Pa}$.
$3.0/\num{2.0e-4} = \qty{15}{kPa}$ bat
$60/0.050 = \qty{1.2}{kPa}$~: la petite \omterm{def:g10:inertia:force}{force} presse plus fort.
\end{solution}

\begin{solution}{exo:g10:pressure:3}
$\rho g h = 1000 \times 9.81 \times 3.0 \approx \qty{29}{kPa}$~;
absolue $P = 101 + 29 = \qty{130}{kPa}$.
\end{solution}

\begin{solution}{exo:g10:pressure:4}
$V_2 = 100 \times 4.0 / 250 = \qty{1.6}{L}$~;
à \qty{50}{kPa}, $V_2 = 400/50 = \qty{8.0}{L}$.
\end{solution}

\begin{solution}{exo:g10:pressure:5}
$F = 65 \times 9.81 \approx \qty{638}{N}$. Lame~:
$638/\num{1.2e-3} \approx \qty{5.3e5}{Pa} = \qty{530}{kPa}$. Semelles~:
$638/0.036 \approx \qty{18}{kPa}$. Rapport $\approx 30$~: la lame
mord dans la glace, les semelles non.
\end{solution}

\begin{solution}{exo:g10:pressure:6}
Pouce~: $20/\num{1.2e-4} \approx \qty{1.7e5}{Pa} = \qty{170}{kPa}$.
Mur~: $20/\num{1.0e-8} = \qty{2.0e9}{Pa}$. Les mêmes \qty{20}{N}
agissent sur une aire $12\,000$~fois plus petite à la pointe, donc
seule là la \omterm{def:g10:pressure:pressure}{pression} dépasse ce que le matériau peut
supporter.
\end{solution}

\begin{solution}{exo:g10:pressure:7}
$\Delta P = \rho g h = 1025 \times 9.81 \times 30 \approx
\qty{3.0e5}{Pa}$, donc
$F = \num{3.0e5} \times 0.50 \approx \qty{1.5e5}{N}$ --- le
\omterm{def:g10:universal-gravitation:weight}{poids} de $F/g \approx \qty{1.5e4}{kg}$, quinze tonnes.
Aucun équipage n'ouvre cela vers l'extérieur contre la mer.
\end{solution}

\begin{solution}{exo:g10:pressure:8}
$\rho = \dfrac{P}{g h} = \dfrac{\num{66.2e3}}{9.81 \times 5.00}
\approx \qty{1350}{kg/m^3}$. Pas de l'eau douce
($\qty{1000}{kg/m^3}$)~: un liquide nettement plus dense, p.ex.\ une
saumure concentrée.
\end{solution}

\begin{solution}{exo:g10:pressure:9}
$h = P_0/(\rho g)$~: eau
$\num{1.01e5}/9810 \approx \qty{10.3}{m}$~; mercure
$\num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$. Un tube
de \qty{76}{cm} tient sur un bureau~; un baromètre d'eau de
\qty{10}{m} a besoin d'une cage d'escalier --- la densité du mercure
l'a emporté.
\end{solution}

\begin{solution}{exo:g10:pressure:10}
$V = 6000/240 = \qty{25}{mL}$~; $P = 6000/75 = \qty{80}{kPa}$.
$P$ contre $V$ est une hyperbole~; $P$ contre $1/V$ est une droite
passant par l'origine, de pente \qty{6000}{kPa.mL}.
\end{solution}

\begin{solution}{exo:g10:pressure:11}
À \qty{20}{m}~: $P = 101 + 2 \times 101 = \qty{303}{kPa}$, trois
atmosphères. À la surface,
$V = 0.50 \times 303/101 = \qty{1.5}{cm^3}$~: la bulle triple.
\end{solution}

\begin{solution}{exo:g10:pressure:12}
\emph{1.} $F = \num{1.01e5} \times \num{1.2e-3} \approx \qty{121}{N}$,
tenant $m = 121/9.81 \approx \qty{12}{kg}$.

\emph{2.} $\Delta P = 101 - 20 = \qty{81}{kPa}$~:
$F = \num{8.1e4} \times \num{1.2e-3} \approx \qty{97}{N}$, environ
\qty{9.9}{kg}.
\end{solution}

\begin{solution}{exo:g10:pressure:13}
Boyle~: $12 \times (200 - 50) = \qty{1800}{L}$ d'air à
\omterm{def:g10:pressure:pressure}{pression de surface} sont utilisables. À \qty{30}{m} chaque
inspiration est prise à \qty{4}{bar}, donc \qty{15}{L/min} à la
\omterm{def:g10:pressure:pressure}{pression} ambiante coûtent $15 \times 4 = \qty{60}{L/min}$
d'air de surface~: $1800/60 = \qty{30}{min}$. À \qty{10}{m}
($\qty{2}{bar}$)~: \qty{30}{L/min}, donc \qty{60}{min}. La profondeur
se paie en air.
\end{solution}

\begin{solution}{exo:g10:pressure:14}
$P = P_0 \dfrac{V_0}{V} = 101 \times \dfrac{6.0}{1.5} =
\qty{404}{kPa}$. Avec $\rho g \approx \qty{10.1}{kPa/m}$,
$h = (404 - 101)/10.1 \approx \qty{30}{m}$.
\end{solution}

\begin{solution}{exo:g10:pressure:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{Poids} $F = 1200 \times 9.81 \approx \qty{1.18e4}{N}$~;
\omterm{def:g10:pressure:pressure}{pression} $P = F/S = \num{1.18e4}/0.040 \approx \qty{2.9e5}{Pa}$~;
petit piston~: $f = P s = \num{2.9e5} \times \num{2.0e-4} \approx
\qty{59}{N}$ --- la \omterm{def:g10:inertia:force}{force} est divisée par $S/s = 200$.

\emph{2.} Le volume de liquide se conserve~:
$400 \times 2.0 = \qty{800}{cm^3}$ doivent venir du petit cylindre,
qui parcourt $800/2.0 = \qty{400}{cm} = \qty{4.0}{m}$. Travail~:
$59 \times 4.0 \approx \qty{235}{J}$ d'un côté,
$\num{1.18e4} \times 0.020 \approx \qty{235}{J}$ de l'autre --- le
vérin échange de la distance contre de la \omterm{def:g10:inertia:force}{force}, jamais du
travail.
\end{solution}

\begin{solution}{pb:g10:pressure:1}
\textbf{1.} $\rho g = 1025 \times 9.81 \approx \qty{10.1}{kPa}$ par
\omterm{def:g10:orders-of-magnitude:unit}{mètre}, donc \qty{10}{m} d'eau de mer ajoutent
$\approx \qty{101}{kPa}$~: une atmosphère tous les dix
\omterm{def:g10:orders-of-magnitude:unit}{mètres}, presque exactement.

\textbf{2.} $P = 101 + 10.1\,h$~: \qty{202}{kPa} à \qty{10}{m},
\qty{302}{kPa} à \qty{20}{m}, \qty{403}{kPa} à \qty{30}{m},
\qty{503}{kPa} à \qty{40}{m}.

\textbf{3.} $2{,}0$, $3{,}0$, $4{,}0$ et $5{,}0$~fois $P_0$.

\textbf{4.} $\Delta P = 403 - 101 = \qty{302}{kPa}$ sur
$\qty{0.015}{m^2}$~: $F = \num{3.02e5} \times 0.015 \approx
\qty{4.5e3}{N}$ --- le \omterm{def:g10:universal-gravitation:weight}{poids} de près d'une demi-tonne sur
le visage.

\textbf{5.} $\Delta P = 10.1 \times 3.0 \approx \qty{30}{kPa}$~;
$F = \num{3.0e4} \times \num{6.0e-5} \approx \qty{1.8}{N}$ --- un
doigt pressé sur le tympan, déjà à profondeur de piscine. Équilibrer
pousse de l'air dans l'oreille moyenne jusqu'à ce que la
\omterm{def:g10:pressure:pressure}{pression} intérieure égale celle de l'eau.

\textbf{6.} $V = 6.0 \, P_0/P$~: $6.0 \times 101/202 = \qty{3.0}{L}$~;
$606/302 = \qty{2.0}{L}$~; $606/403 = \qty{1.5}{L}$.

\textbf{7.} Divisé par deux lorsque $P = 2P_0$, c'est-à-dire à
\qty{10}{m} --- dans le tout premier tronçon de la plongée.

\textbf{8.} Non~: $V = 606/P$ est une hyperbole en
\omterm{met:g10:pressure:dive}{pression absolue} (et $P$, non $h$, est ce que la
\omterm{prop:g10:pressure:boyle}{loi de Boyle--Mariotte} voit)~; chaque
\qty{10}{m} supplémentaires retirent moins de volume que les
précédents.

\textbf{9.} $V = \qty{1.5}{L}$ exige
$P = 101 \times 6.0/1.5 = \qty{404}{kPa}$, c'est-à-dire
$h = 303/10.1 \approx \qty{30}{m}$ (\cref{exo:g10:pressure:14}).

\textbf{10.} Le sang~: le plasma se déplace dans les vaisseaux de la
poitrine, incompressible, et occupe le volume que l'air ne remplit
plus.

\textbf{11.} À la \omterm{def:g10:pressure:pressure}{pression} ambiante, \qty{403}{kPa}. Par Boyle le
même air à \qty{101}{kPa} remplirait $4$~fois le volume~: il est
$4{,}0$~fois plus dense que l'air de surface.

\textbf{12.} $V = 6.0 \times 403/202 \approx \qty{12}{L}$ --- le
double, après seulement \qty{20}{m} de montée.

\textbf{13.} $V = 6.0 \times 403/101 \approx \qty{24}{L}$~: quatre
fois sa capacité pulmonaire.

\textbf{14.} \emph{Ne jamais bloquer sa respiration en remontant.} Avec
les voies aériennes ouvertes, l'air qui se dilate s'échappe par le
détendeur, et le volume pulmonaire ne dépasse jamais \qty{6.0}{L}.

\textbf{15.} Les \qty{1.5}{L} de Léna à \qty{30}{m} \emph{sont} ses
\qty{6.0}{L} de surface comprimés~: en remontant ils se redilatent
exactement à \qty{6.0}{L}, jamais au-delà. Les \qty{6.0}{L} de Marco
ont été \emph{pris} à \qty{403}{kPa}~: ils contiennent \qty{24}{L}
d'air de surface.

\textbf{16.} $V = 1.0 \times 503/P$~: \qty{1.25}{cm^3} à \qty{30}{m},
\qty{1.67}{cm^3} à \qty{20}{m}, \qty{2.49}{cm^3} à \qty{10}{m},
\qty{4.98}{cm^3} juste sous la surface.

\textbf{17.} Facteurs par palier~: $503/403 = 1.25$, puis $1.33$,
$1.49$, et $202/101 = 2.0$ pour les dix derniers
\omterm{def:g10:orders-of-magnitude:unit}{mètres} --- la croissance accélère à l'approche de la
surface.

\textbf{18.} $40/10 = \qty{4.0}{min}$. La dernière minute~: de
\qty{10}{m} à la surface la \omterm{met:g10:pressure:dive}{pression absolue} se divise par
deux, la plus grande chute relative de toute la remontée.

\textbf{19.} Une remontée trop rapide laisse l'azote dissous pétiller
en bulles dans le sang et les articulations (accident de
décompression)~; remonter lentement et s'arrêter près de \qty{5}{m}
garde le gaz dissous assez longtemps pour qu'il s'échappe calmement
par les poumons.

\textbf{20.} Apnéiste~: son air ne revient qu'à son volume d'origine
--- Boyle la protège. Plongeur bouteille~: l'air pris en profondeur se
multiplie en remontant --- respirer, ne jamais bloquer. Et les dix
derniers \omterm{def:g10:orders-of-magnitude:unit}{mètres} doublent tout volume piégé ($\times 2.0$,
contre $\times 1.25$ en bas à \qty{40}{m})~: plus on est près de la
surface, plus on devrait s'en approcher lentement.
\end{solution}
