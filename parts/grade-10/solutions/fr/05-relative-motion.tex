\section*{Chapitre \ref{ch:g10:relative-motion} --- Mouvement relatif}

\begin{solution}{exo:g10:relative-motion:1}
Par rapport à la voiture~: au \omterm{def:g10:relative-motion:frame}{repos}. Par rapport au sol et à
l'arbre~: en mouvement à \qty{100}{km/h}. Par rapport au train venant
en sens inverse~: les vitesses en sens opposés s'additionnent,
$100 + 100 = \qty{200}{km/h}$. Dans le référentiel du train du
passager, l'arbre se déplace en arrière à \qty{100}{km/h} le long
d'une droite.
\end{solution}

\begin{solution}{exo:g10:relative-motion:2}
$72 / 3.6 = \qty{20}{m/s}$~; $108 / 3.6 = \qty{30}{m/s}$~;
$25 \times 3.6 = \qty{90}{km/h}$~; $340 \times 3.6 = \qty{1224}{km/h}$.
Dans chaque paire le chiffre en \unit{km/h} est le plus grand, comme
l'exige la règle de vraisemblance.
\end{solution}

\begin{solution}{exo:g10:relative-motion:3}
Cycliste~: $v = \frac{\qty{36}{km}}{\qty{1.5}{h}} = \qty{24}{km/h}
= 24/3.6 \approx \qty{6.7}{m/s}$. Sprinter~:
$v = \frac{\qty{100}{m}}{\qty{10.0}{s}} = \qty{10.0}{m/s}
= \qty{36}{km/h}$. Le sprinter est une fois et demie plus rapide ---
mais seulement pendant dix secondes, tandis que le cycliste tient
l'allure une heure et demie. Les \omterm{def:g10:relative-motion:speed}{vitesses moyennes} ne se
comparent équitablement que sur des durées comparables.
\end{solution}

\begin{solution}{exo:g10:relative-motion:4}
Même sens~: $1.4 + 0.9 = \qty{2.3}{m/s}$. Contre la bande~:
$1.4 - 0.9 = \qty{0.5}{m/s}$. Immobile sur la bande~: les propres
\qty{0.9}{m/s} de la bande.
\end{solution}

\begin{solution}{exo:g10:relative-motion:5}
Stylo lâché~: le \omterm{def:g10:relative-motion:frames}{référentiel du sol} fonctionne, mais le
référentiel du \emph{train} est encore meilleur --- là le stylo tombe
simplement tout droit. Station spatiale~:
\omterm{def:g10:relative-motion:frames}{référentiel géocentrique} (elle \omterm{def:g10:universal-gravitation:satellite}{orbite} la Terre
dans son ensemble, ignorant sa rotation). Année martienne~:
\omterm{def:g10:relative-motion:frames}{référentiel héliocentrique} (Mars suit un quasi-cercle autour
du Soleil). Sprint~: \omterm{def:g10:relative-motion:frames}{référentiel du sol} (ligne de départ,
ligne d'arrivée et chronomètre sont tous fixés au sol).
\end{solution}

\begin{solution}{exo:g10:relative-motion:6}
\emph{1.} $108/3.6 = \qty{30}{m/s}$. Vers l'avant~:
$30 + 1.5 = \qty{31.5}{m/s}$~; vers l'arrière~:
$30 - 1.5 = \qty{28.5}{m/s}$.

\emph{2.} Par rapport au train le passager parcourt \qty{60}{m} à
\qty{1.5}{m/s}~: $t = 60/1.5 = \qty{40}{s}$. Par rapport au sol le
passager parcourt $31.5 \times 40 = \qty{1260}{m}$ --- les
\qty{1200}{m} emportés par le train plus les \qty{60}{m} marchés,
comme le prédit la règle de composition.
\end{solution}

\begin{solution}{exo:g10:relative-motion:7}
\emph{1.} Aval~: $5.0 + 2.0 = \qty{7.0}{m/s}$, donc
$t_1 = 420/7.0 = \qty{60}{s}$. Amont~: $5.0 - 2.0 = \qty{3.0}{m/s}$,
donc $t_2 = 420/3.0 = \qty{140}{s}$.

\emph{2.} \omterm{def:g10:relative-motion:speed}{Vitesse moyenne}~: $v = \frac{840}{60 + 140}
= \frac{840}{200} = \qty{4.2}{m/s} < \qty{5.0}{m/s}$. L'aide et la
gêne du courant sont égales en \emph{vitesse} mais non en
\emph{temps}~: le bateau passe \qty{140}{s} ralenti et seulement
\qty{60}{s} aidé, donc le trajet lent domine la moyenne.
\end{solution}

\begin{solution}{exo:g10:relative-motion:8}
\emph{1.} Sens opposés~: dans le référentiel de A, B s'approche à
$25 + 20 = \qty{45}{m/s}$. Le croisement est complet lorsque B a
glissé sur la longueur combinée $180 + 90 = \qty{270}{m}$~:
$t = 270/45 = \qty{6.0}{s}$.

\emph{2.} Même sens~: vitesse relative $25 - 20 = \qty{5.0}{m/s}$, et
$t = 270/5.0 = \qty{54}{s}$ --- neuf fois plus long, c'est pourquoi
dépasser un camion semble interminable.
\end{solution}

\begin{solution}{exo:g10:relative-motion:9}
Référentiel du cycliste~: un cercle (le cercle de la jante, centré sur
le moyeu). \omterm{def:g10:relative-motion:frames}{Référentiel du sol}~: une
\omterm{prop:g10:relative-motion:relativity}{cycloïde} --- une chaîne d'arches, une par tour de
roue. Référentiel de la valve elle-même~: la valve est au
\omterm{def:g10:relative-motion:frame}{repos}, un seul point. Le moyeu dans le
\omterm{def:g10:relative-motion:frames}{référentiel du sol}~: une droite horizontale à la hauteur d'un
rayon de roue.
\end{solution}

\begin{solution}{exo:g10:relative-motion:10}
\emph{1.} $v = \frac{\qty{40000}{km}}{\qty{24}{h}}
\approx \qty{1.7e3}{km/h} = 1667/3.6 \approx \qty{463}{m/s}$ ---
plus vite que le son.

\emph{2.} Nous ne le sentons pas parce que tout autour de nous ---
sol, air, bâtiments --- partage exactement le même mouvement, donc
rien ne se déplace par rapport à nous (et le principe d'inertie,
le~\cref{ch:g10:inertia}, expliquera pourquoi un mouvement uniforme
partagé est indétectable de l'intérieur). «~Le Soleil se lève~» est
le rapport dans le référentiel du sol et «~la Terre tourne~» le
rapport dans le référentiel géocentrique d'une seule et même
observation.
\end{solution}

\begin{solution}{exo:g10:relative-motion:11}
Soit $L$ la longueur de l'escalator. Vitesse de l'escalator~: $L/60$~;
vitesse de marche~: $L/90$. En marchant sur l'escalator en marche, les
vitesses s'additionnent~:
\[
v = \frac{L}{60} + \frac{L}{90} = L\left(\frac{3 + 2}{180}\right)
= \frac{L}{36},
\]
donc la montée dure \qty{36}{s}. (Pas la moyenne de $60$ et $90$~: les
vitesses s'additionnent, les temps non.)
\end{solution}

\begin{solution}{exo:g10:relative-motion:12}
\emph{1.} Sur une distance $d$ dans chaque sens~:
$t = \frac{d}{60} + \frac{d}{90} = d\,\frac{3 + 2}{180} = \frac{d}{36}$,
donc $v = \frac{2d}{d/36} = \qty{72}{km/h}$.

\emph{2.} Les $75$ naïfs moyennent les deux vitesses comme si le
conducteur passait des \emph{temps} égaux à chacune~; en fait les
distances sont égales, donc le trajet lent dure plus longtemps
($\frac{d}{60} > \frac{d}{90}$) et tire la moyenne vers
\qty{60}{km/h}. La \omterm{def:g10:relative-motion:speed}{vitesse moyenne} est la distance totale sur
le temps total --- rien d'autre.
\end{solution}

\begin{solution}{exo:g10:relative-motion:13}
\emph{1.} Dans le référentiel du premier cycliste, le second s'approche
en ligne droite à $15 + 20 = \qty{35}{km/h}$ depuis \qty{35}{km}~: ils
se rencontrent après exactement \qty{1.0}{h}.

\emph{2.} Le premier a parcouru $15 \times 1 = \qty{15}{km}$, le
second $20 \times 1 = \qty{20}{km}$ --- total \qty{35}{km}, comme il
se doit.

\emph{3.} Même sens~: vitesse de rapprochement
$20 - 15 = \qty{5}{km/h}$, donc la poursuite dure $35/5 = \qty{7}{h}$.
\end{solution}

\begin{solution}{exo:g10:relative-motion:14}
\emph{1.} Le courant pousse le nageur le long de la rivière, non en
travers~: les deux mouvements se composent indépendamment, et le
progrès vers la rive opposée se fait à \qty{1.2}{m/s} indépendamment
de la dérive. Temps de traversée~: $t = 60/1.2 = \qty{50}{s}$.

\emph{2.} Dérive~: $0.9 \times 50 = \qty{45}{m}$ en aval.

\emph{3.} La vitesse dans le référentiel du sol est l'hypoténuse d'un
triangle rectangle de côtés $1.2$ et $0.9$~:
$v = \sqrt{1.2^2 + 0.9^2} = \sqrt{2.25} = \qty{1.5}{m/s}$ (un triangle
$3$--$4$--$5$ mis à l'échelle de $0.3$). Longueur du trajet~:
$1.5 \times 50 = \qty{75}{m}$ --- ou directement
$\sqrt{60^2 + 45^2} = \qty{75}{m}$.
\end{solution}

\begin{solution}{exo:g10:relative-motion:15}
\emph{1.} $v = \frac{2\pi \times \num{1.50e11}}{\num{3.16e7}}
= \frac{\num{9.42e11}}{\num{3.16e7}} \approx \qty{2.98e4}{m/s}
\approx \qty{30}{km/s}$.

\emph{2.} $30 / 0.46 \approx 65$~: la Terre nous emporte autour du
Soleil environ soixante-cinq fois plus vite qu'elle ne nous fait
tourner autour de son axe.

\emph{3.} Dans le \omterm{def:g10:relative-motion:frames}{référentiel géocentrique}, la trajectoire
apparente de Mars combine son propre mouvement orbital avec celui de
la Terre, et lorsque la Terre plus rapide dépasse Mars la combinaison
recule pendant quelques semaines. Dans le
\omterm{def:g10:relative-motion:frames}{référentiel héliocentrique} les deux planètes tournent
simplement autour du Soleil à des allures régulières --- la «~boucle
en arrière~» n'est pas un mouvement de Mars mais un artefact du
référentiel.
\end{solution}

\begin{solution}{pb:g10:relative-motion:1}
\textbf{1.} $t = 90/1.0 = \qty{90}{s}$.

\textbf{2.} Vitesse au sol $1.5 + 1.0 = \qty{2.5}{m/s}$~:
$t = 90/2.5 = \qty{36}{s}$.

\textbf{3.} $t = 90/1.5 = \qty{60}{s}$.

\textbf{4.} $1.5 - 1.0 = \qty{0.5}{m/s}$ par rapport au sol, contre le
sens de la bande~: $t = 90/0.5 = \qty{180}{s}$ --- trois minutes
entières de marche rapide.

\textbf{5.} Vitesse au sol $1.0 - 1.0 = \qty{0}{m/s}$~: l'enfant court
fort et n'avance nulle part, au \omterm{def:g10:relative-motion:frame}{repos} dans le
\omterm{def:g10:relative-motion:frames}{référentiel du sol} tout en sprinter dans le référentiel de
la bande. C'est un tapis de course.

\textbf{6.} Référentiel de la bande / \omterm{def:g10:relative-motion:frames}{référentiel du sol}~:
voyageur immobile $0$ / \qty{1.0}{m/s}~; marcheur avec la bande
$1.5$ / \qty{2.5}{m/s}~; marcheur sur le sol $0.5$ (le sol glisse en
arrière dans le référentiel de la bande) / \qty{1.5}{m/s}~; farceur
$1.5$ / \qty{0.5}{m/s}~; enfant $1.0$ / \qty{0}{m/s}. Les jambes de
chaque acteur ressentent la colonne du référentiel de ce qui est sous
ses pieds --- la colonne bande pour ceux qui sont sur la bande, la
colonne sol pour le marcheur sur le sol~: l'effort est la vitesse de
marche par rapport à ce qui est sous les pieds.

\textbf{7.} Le progrès à travers la rivière se fait à \qty{2.0}{m/s}
quoi que fasse le courant~: $t = 120/2.0 = \qty{60}{s}$.

\textbf{8.} Dérive~: $1.5 \times 60 = \qty{90}{m}$ en aval.

\textbf{9.} Vitesse au sol~: hypoténuse d'un triangle rectangle de
côtés $2.0$ et $1.5$~: $v = \sqrt{2.0^2 + 1.5^2} = \sqrt{6.25}
= \qty{2.5}{m/s}$. Longueur de la \omterm{def:g10:relative-motion:trajectory}{trajectoire}~:
$2.5 \times 60 = \qty{150}{m}$ --- le triple $3$--$4$--$5$ encore,
mis à l'échelle de $90$--$120$--$150$.

\textbf{10.} Référentiel de l'eau~: une droite, \qty{120}{m},
perpendiculaire aux rives. \omterm{def:g10:relative-motion:frames}{Référentiel du sol}~: aussi une
droite, mais oblique, et longue de \qty{150}{m}. Même forme (droite),
direction et longueur différentes~: la
\omterm{def:g10:relative-motion:trajectory}{trajectoire} appartient au référentiel, non au ferry.

\textbf{11.} Le progrès en travers se fait à \qty{1.0}{m/s}, donc
$t = 120/1.0 = \qty{120}{s}$. Dérive~: $1.5 \times 120 = \qty{180}{m}$
--- le double des \qty{90}{m} du ferry. Vitesse au sol~: hypoténuse
avec côtés $1.0$ et $1.5$~: $v = \sqrt{1.0^2 + 1.5^2} = \sqrt{3.25}
\approx \qty{1.8}{m/s}$. Le nageur plus lent met deux fois plus de
temps à traverser, donc le même courant a deux fois plus de temps pour
l'emporter en aval.

\textbf{12.} \omterm{def:g10:relative-motion:frames}{Référentiel du sol}~: le tronc dérive à la vitesse du
courant \qty{1.5}{m/s}. Référentiel de l'eau~: il est au
\omterm{def:g10:relative-motion:frame}{repos}, flottant avec l'eau autour de lui. Savoir si le tronc
«~se déplace~» n'a pas de réponse indépendante du référentiel ---
c'est tout le point du chapitre.

\textbf{13.} Le \omterm{def:g10:relative-motion:frames}{référentiel géocentrique}. Rayon d'\omterm{def:g10:universal-gravitation:satellite}{orbite}
$r = \num{6.37e6} + \num{4.0e5} = \qty{6.77e6}{m}$, donc
\[
v = \frac{2\pi r}{T} = \frac{2\pi \times \num{6.77e6}}{\num{5.56e3}}
\approx \qty{7.65e3}{m/s} \approx \qty{7.7}{km/s}.
\]

\textbf{14.} Pendant que la station achève une
\omterm{def:g10:universal-gravitation:satellite}{orbite}, la Terre tourne en dessous d'environ
$23^\circ$, donc la trace sur le sol se décale vers l'ouest à chaque
passage. Dans le \omterm{def:g10:relative-motion:frames}{référentiel du sol} le trajet de la station
est une courbe ondulée compliquée qui ne se répète jamais exactement~;
dans le \omterm{def:g10:relative-motion:frames}{référentiel géocentrique} c'est un quasi-cercle fixe.

\textbf{15.} $7.65 / 0.46 \approx 17$~: la station devance l'équateur
en rotation d'environ un facteur dix-sept, c'est pourquoi elle traverse
tout le ciel en quelques minutes.

\textbf{16.} $v = \frac{2\pi \times \num{4.22e7}}{\num{8.64e4}}
\approx \qty{3.07e3}{m/s} \approx \qty{3.1}{km/s}$ dans le
\omterm{def:g10:relative-motion:frames}{référentiel géocentrique}. Dans le
\omterm{def:g10:relative-motion:frames}{référentiel du sol} il est \emph{au
\omterm{def:g10:relative-motion:frame}{repos}}, suspendu au-dessus d'un point fixe de l'équateur ---
il tourne autour de l'axe de la Terre en exactement le temps que le
sol lui-même le fait, donc les deux mouvements s'annulent dans le
\omterm{def:g10:relative-motion:frames}{référentiel du sol}. Au \omterm{def:g10:relative-motion:frame}{repos} et à
\qty{3.1}{km/s}~: les deux vrais, un par référentiel.

\textbf{17.} Dans le \omterm{def:g10:relative-motion:frames}{référentiel héliocentrique} la Lune voyage
essentiellement le long de l'\omterm{def:g10:universal-gravitation:satellite}{orbite} de la Terre --- un
énorme quasi-cercle autour du Soleil --- surmonté d'un léger
balancement du cycle de \qty{27}{jours}. L'influence du Soleil domine
si fortement que le trajet se courbe toujours vers le Soleil~; il ne
boucle jamais en arrière.

\textbf{18.} «~Le Soleil se lève~» est le rapport dans le
\omterm{def:g10:relative-motion:frames}{référentiel du sol}, où l'observateur est au
\omterm{def:g10:relative-motion:frame}{repos} et le Soleil balaie le ciel. «~La Terre tourne~» est
le rapport dans le \omterm{def:g10:relative-motion:frames}{référentiel géocentrique}, où la direction
du Soleil est stable et l'observateur est emporté vers l'est autour
de l'axe. Ils décrivent la même observation~; chacun est correct dans
son propre référentiel, et le~\cref{prop:g10:relative-motion:relativity}
dit qu'aucune observation du mouvement seul ne peut arbitrer.

\textbf{19.} Le référentiel centré-Terre de Ptolémée collait
naturellement au mouvement quotidien du ciel et prédisait
convenablement les positions planétaires, mais au prix de boucles sur
boucles (épicycles) pour reproduire les errances de chaque planète. Le
référentiel centré-Soleil de Copernic faisait de chaque planète un
quasi-cercle régulier et transformait les boucles en arrière de Mars
en un effet de référentiel --- la Terre dépassant un coureur plus lent
sur une voie intérieure. Le verdict moderne n'est pas que le Soleil
est «~vraiment~» immobile (il se déplace aussi, autour du centre de la
galaxie) mais que les référentiels sont des outils~: l'héliocentrique
rend le mouvement planétaire incomparablement plus simple, et la
simplicité, non la vérité, est ce à quoi sert un référentiel.

\textbf{20.} \omterm{def:g10:relative-motion:frames}{Référentiel du sol}~: \qty{0}{km/s} --- au
\omterm{def:g10:relative-motion:frame}{repos} dans le fauteuil.
\omterm{def:g10:relative-motion:frames}{Référentiel géocentrique}~: environ \qty{0.46}{km/s},
emporté autour de l'axe de la Terre.
\omterm{def:g10:relative-motion:frames}{Référentiel héliocentrique}~: environ \qty{30}{km/s},
emporté autour du Soleil. Pendant un film de \qty{90}{min}~:
$d = \num{3.0e4} \times 5400 \approx \qty{1.6e8}{m} =
\qty{160000}{km}$ --- presque à mi-chemin de la Lune sans quitter le
fauteuil. Morale~: le mouvement n'est pas une propriété d'un corps
mais d'un corps et d'un référentiel~; choisir le référentiel qui
rend l'histoire simple, et toujours dire lequel on a choisi.
\end{solution}
