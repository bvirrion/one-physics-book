\section*{अध्याय \ref{ch:g10:orders-of-magnitude} --- परिमाण के आदेश: ब्रह्मांड को मापना}

\begin{solution}{exo:g10:orders-of-magnitude:1}
\emph{1.} $\qty{750}{nm} = 750 \times \qty{e-9}{m} = \qty{7.50e-7}{m}$.
\emph{2.} $\qty{2.5}{km} = \qty{2.5e3}{m}$.
\emph{3.} $\qty{3.2}{\micro m} = \qty{3.2e-6}{m}$.
\emph{4.} $\qty{0.15}{Tm} = 0.15 \times \qty{e12}{m} = \qty{1.5e11}{m}$
--- एक \omterm{def:g10:orders-of-magnitude:lightyear}{खगोलीय इकाई}.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:4}
इंसान: $1.7 < 5$, आदेश $\qty{e0}{m}$. एवरेस्ट:
$\qty{8848}{m} = \qty{8.848e3}{m}$ और $8.848 \geq 5$: आदेश

\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:5}
\emph{1.} आधा स्नातक: $ L = (127.0 \pm 0.5)\,\unit{mm}$.
\emph{2.} $\Delta L / L = 0.5/127 \approx 0.4\%$.
\emph{3.} रबड़: $0.5/12 \approx 4\%$. पेंसिल माप है
लगभग दस गुना अधिक सटीक --- वही रूलर, वही \omterm{def:g10:orders-of-magnitude:uncertainty}{निरपेक्ष
अनिश्चितता}, दस गुना लंबी वस्तु।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:6}
\emph{1.} प्रॉक्सिमा:
$4.24 \times \qty{9.47e15}{m} = \qty{4.01e16}{m}$. सीरियस:
$8.6 \times \qty{9.47e15}{m} = \qty{8.1e16}{m}$ (दो \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग}, इनपुट की तरह)।
\emph{2.} $ t = d/c = \qty{2.4e13}{m} / \qty{3.00e8}{m/s}
= \qty{8.0e4}{s}$, यानी\ $\num{8.0e4}/3600 \approx \qty{22}{h}$:
जांच से प्रत्येक संदेश के लिए लगभग पूरा दिन।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:7}
$ t = d/c $ हर बार. चंद्रमा:
$\qty{3.84e8}{m}/\qty{3.00e8}{m/s} = \qty{1.28}{s}$. सूरज:
$\qty{1.496e11}{m}/\qty{3.00e8}{m/s} = \qty{499}{s} \approx
\qty{8.3}{min}$. नेपच्यून:
$\qty{4.50e12}{m}/\qty{3.00e8}{m/s} = \qty{1.50e4}{s} \approx
\qty{4.2}{h}$. जब आप नेप्च्यून को देखते हैं, तो आप उसे वैसा ही देखते हैं जैसा वह था
चार घंटे पहले --- प्रत्येक दूरबीन एक माइल्ड टाइम मशीन है।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:8}
\emph{1.} एक शीट: $\qty{52.0}{mm}/500 = \qty{0.104}{mm}
= \qty{1.04e-4}{m}$. अनिश्चितता भी विभाजित होती है:
$\qty{0.5}{mm}/500 = \qty{0.001}{mm}$, इसलिए
$ t = (104 \pm 1)\,\unit{\micro m}$.
\emph{2.} $1/104 \approx 1\%$.
\emph{3.} सीधे एक शीट मापने से मिलेगा
$(0.1 \pm 0.5)\,\unit{mm}$: ए \omterm{def:g10:orders-of-magnitude:uncertainty}{सापेक्ष अनिश्चितता} के विषय का $500\%$
--- शासक यह भी साबित नहीं कर सकता कि शीट मौजूद है। स्टैकिंग $500$
शीटें मापी गई लंबाई को इससे गुणा करती हैं $500$ जबकि शासक का
पूर्ण अनिश्चितता स्थिर रहता है: \omterm{def:g10:orders-of-magnitude:uncertainty}{सापेक्ष अनिश्चितता} से सिकुड़ जाता है
वही कारक $500$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:9}
प्रत्येक लंबाई को इससे विभाजित करें $\qty{2e-10}{m}$. जीवाणु:
$\qty{2.0e-6}{m}/\qty{2e-10}{m} = \num{1.0e4}$ परमाणु. बाल:
$\qty{7.0e-5}{m}/\qty{2e-10}{m} = \num{3.5e5}$. इंसान:
$1.7/\num{2e-10} = \num{8.5e9}$, \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार} $10^{10}$: दस
अरब परमाणु सिर से पाँव तक।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:10}
\emph{1.} $ s = \qty{0.22}{m}/\qty{1.39e9}{m} = \num{1.58e-10}$.
\emph{2.} धरती:
$\qty{1.27e7}{m} \times \num{1.58e-10} = \qty{2.0e-3}{m} =

$\qty{1.496e11}{m} \times \num{1.58e-10} = \qty{23.7}{m} \approx
\qty{24}{m}$ फुटबॉल से.
\emph{3.} चंद्रमा:
$\qty{3.84e8}{m} \times \num{1.58e-10} = \qty{6.1}{cm}$ से
कालीमिर्च. नेपच्यून:
$\qty{4.50e12}{m} \times \num{1.58e-10} \approx \qty{710}{m}$: द
मॉडल स्कूल प्रांगण में फिट नहीं बैठता.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:11}
\emph{1.} $21.0 \times 29.7 = \qty{623.7}{cm^2}$, तीन तक पूर्णांकित
\omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} इनपुट का: $\qty{624}{cm^2}$.
\emph{2.} $ v = \qty{100.0}{m}/\qty{12.3}{s} = \qty{8.13}{m/s}$ (तीन
आंकड़े, समय के अनुसार निर्धारित)।
\emph{3.} पाँच \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} क्षेत्र पर दावा करेगा
$\qty{0.01}{cm^2}$, मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} स्तर के इनपुट की सटीकता कभी नहीं होती
वितरित --- एक जाली दशमलव अभी भी एक जालसाजी है।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:12}
\emph{1.} RADIUS $ r = \qty{0.55}{m}$, क्षेत्र
$ A = \pi r^{2} = \pi \times 0.55^{2} \approx \qty{0.95}{m^2}$. आयतन
$ V = \qty{1.0}{mm^3} = \qty{1.0e-9}{m^3}$, और एक फिल्म (क्षेत्र) है
$\times $ (मोटाई), तो
\[
e = \frac{V}{A} = \frac{\qty{1.0e-9}{m^3}}{\qty{0.95}{m^2}}
= \qty{1.1e-9}{m} .
\]
\emph{2.} एक अणु क्रम का होता है $\qty{e-9}{m}$ --- से एक पायदान ऊपर
परमाणु का $\qty{e-10}{m}$, एक अणु के छोटे होने के अनुरूप
परमाणुओं का समूह. फ्रैंकलिन के एक चम्मच तेल से आधा एकड़ जमीन शांत हो गई
तालाब और, उसकी जानकारी के बिना, नैनो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} मापा।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:13}
धारणाएँ: के बारे में $70$ विश्राम के समय प्रति मिनट धड़कता है, एक जीवन $80$
साल। प्रति दिन: $70 \times 60 \times 24 \approx \num{1e5}$ धड़कता है. प्रति
वर्ष: $\num{1e5} \times 365 \approx \num{4e7}$. प्रति जीवन:
$\num{4e7} \times 80 \approx \num{3e9}$ --- \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार}
$10^{9}$, कुछ अरब दिल की धड़कनें। चार \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} होगा
अस्सी वर्षों तक, मिनट दर मिनट, हृदय गति जानने का नाटक करें: द
इनपुट परिमाण के क्रम का अनुमान है, और कोई भी आउटपुट इससे आगे नहीं बढ़ सकता है
इनपुट (\cref{met:g10:orders-of-magnitude:sigfig}).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:14}
बाद $ n $ तहों की मोटाई है $2^{n} \times \qty{1.0e-4}{m}$.
\emph{1.} ज़रूरत $2^{n} \geq 8848/\num{1.0e-4} = \num{8.8e7}$. तब से
$2^{26} \approx \num{6.7e7}$ छोटा पड़ जाता है और
$2^{27} \approx \num{1.3e8}$ पर्याप्त: $27$ तह (स्टैक
$\approx \qty{13}{km}$).
\emph{2.} ज़रूरत $2^{n} \geq \num{3.84e12}$: $2^{41} \approx
\num{2.2e12}$ छोटा है, $2^{42} \approx \num{4.4e12}$ पहुँचता है ---
$42$ तह, ढेर $\qty{4.4e8}{m}$, आराम से चंद्रमा के पार।
\emph{3.} केवल प्रयोग: प्रत्येक तह क्षेत्र को आधा कर देती है, और उसके बाद
सात गुना ढेर जितना चौड़ा है उससे कहीं अधिक मोटा है। का अंकगणित
दोहरीकरण अछूता है --- घातीय वृद्धि बस कागज से आगे निकल जाती है
(और अंतर्ज्ञान) अनुसरण कर सकता है।
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:15}
\emph{1.} चंद्रमा: $\qty{0.03}{m}/\qty{3.84e8}{m} = \num{7.8e-11}$.
मेज़: $\qty{0.001}{m}/\qty{1.000}{m} = \num{1.0e-3}$. चाँद
किसी कारक द्वारा मापन अधिक सटीक होता है
$\num{1.0e-3}/\num{7.8e-11} \approx \num{1.3e7}$ --- दस मिलियन से अधिक.
\emph{2.} मेल मिलाना $\num{7.8e-11}$ एक पर एक \omterm{def:g10:orders-of-magnitude:unit}{मीटर} तालिका का मतलब है
$\Delta x = \qty{7.8e-11}{m}$, एक परमाणु से भी छोटा
($\approx \qty{e-10}{m}$): मेज का किनारा सम नहीं है
\emph{परिभाषित} वह तेजी से. तुलनात्मक रूप से कहें तो, पृथ्वी--चंद्रमा
लेज़र दूरी हमारी प्रजाति के सर्वोत्तम मापों में से एक है।
\end{solution}

\begin{solution}{pb:g10:orders-of-magnitude:1}
\textbf{1.} अनाज: $\qty{0.5}{mm} = \qty{5e-4}{m}$, मिलि पर
पैमाना; परमाणु: $\qty{2.5e-10}{m} = \qty{0.25}{nm}$, नैनो स्केल पर।

\textbf{2.} $\qty{5e-4}{m}/\qty{2.5e-10}{m} = \num{2.0e6}$: दो
लाखों परमाणु पार।

\textbf{3.} का एक घन $2 \times 10^{6}$ प्रति धार परमाणु धारण करता है
$(2\times10^{6})^{3} = 8 \times 10^{18}$ परमाणु: \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार}
$10^{19}$.

\textbf{4.} $ V = (\qty{5e-4}{m})^{3} = \qty{1.25e-10}{m^3}$, इसलिए
$ m = \rho V = \qty{2500}{kg/m^3} \times \qty{1.25e-10}{m^3}
= \qty{3.1e-7}{kg} = \qty{0.31}{mg}$.

\textbf{5.} प्रति परमाणु द्रव्यमान:
$\qty{3.1e-7}{kg}/(8\times10^{18}) = \qty{3.9e-26}{kg}$ --- बिल्कुल
``कुछ में $10^{-26}\,\unit{kg}$'' श्रेणी। परमाणुओं का भोला घन
वास्तविक परमाणु द्रव्यमान को पुन: उत्पन्न करता है: मॉडल विवेक जांच से गुजरता है
(\cref{met:g10:orders-of-magnitude:sanity}) सम्मान के साथ।

\textbf{6.} $ s = \qty{5e-4}{m}/\qty{1.39e9}{m} = \num{3.6e-13}$.

\textbf{7.} मॉडल पृथ्वी: व्यास
$\qty{1.27e7}{m} \times \num{3.6e-13} = \qty{4.6e-6}{m}$ --- आकार
एक जीवाणु की --- दूरी पर
$\qty{1.496e11}{m} \times \num{3.6e-13} = \qty{5.4}{cm}$ से
कण: पृथ्वी रेत के कण से पाँच सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} की दूरी पर एक सूक्ष्म जीव है।

\textbf{8.} $\qty{4.5e12}{m} \times \num{3.6e-13} = \qty{1.6}{m}$: द
संपूर्ण ग्रह मंडल एक डाइनिंग टेबल के भीतर परिक्रमा करता है।

\textbf{9.} $\qty{4.0e16}{m} \times \num{3.6e-13} = \qty{1.4e4}{m}
= \qty{14}{km}$.

\textbf{10.} हमारे टेबल-टॉप सौर मंडल का निकटतम तारा एक और है
चौदह किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} दूर रेत का कण, जिसमें मूलतः कुछ भी नहीं है
के बीच: आकाशगंगा अत्यधिक खाली स्थान है, यही कारण है
तारों के बीच टकराव लगभग अनसुना है।

\textbf{11.} $ t = \qty{5e-4}{m}/\qty{3.00e8}{m/s} = \qty{1.7e-12}{s}$
--- दो पिकोसेकंड के अंतर्गत।

\textbf{12.} चंद्रमा: $\qty{3.84e8}{m}/\qty{3.00e8}{m/s} =
\qty{1.28}{s}$. सूरज: $\qty{1.496e11}{m}/\qty{3.00e8}{m/s} =
\qty{499}{s} \approx \qty{8.3}{min}$: सूरज की रोशनी हमेशा आठ मिनट की होती है
पुराना.

\textbf{13.} एक साल है $365.25 \times 86400 = \qty{3.156e7}{s}$, इसलिए
$\qty{1}{ly} = \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}$; तब
$\qty{4.0e16}{m}/\qty{9.47e15}{m} = 4.2$: डेटा दूरी वास्तव में है
के बारे में $\qty{4.2}{ly}$.

\textbf{14.} $\qty{2.4e22}{m}/\qty{9.47e15}{m} = \num{2.5e6}$: दो और
डेढ़ लाख \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष}. वह प्रकाश एंड्रोमेडा को छोड़कर चला गया $2.5$
लाखों साल पहले, जब जीनस के पहले सदस्य थे \emph{होमोसेक्सुअल}
पत्थर गिनना सीख रहे थे --- और नहीं \emph{होमो सेपियन्स} अभी तक
कहीं भी अस्तित्व में था.

\textbf{15.} $\qty{3.00e8}{m/s} \times \qty{1e-9}{s} = \qty{0.30}{m}$.
एक नैनोसेकंड के एक तिहाई भाग में एक सिग्नल अधिकतम लगभग दूरी तय करता है
$\qty{10}{cm}$, तो मशीन के कोई भी दो हिस्से जो भीतर से बात करने चाहिए
एक घड़ी की टिक को एक दूसरे से दस सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} के भीतर बैठना होगा:
स्पीड कैप का आकार।

\textbf{16.} पानी में औसत परमाणु:
$\qty{3.0e-26}{kg}/3 = \qty{1.0e-26}{kg}$. परमाणुओं की संख्या:
$\qty{70}{kg}/\qty{1.0e-26}{kg} = \num{7e27}$.

\textbf{17.} स्वभाव के विरुद्ध:
$\num{7e27}/\num{8e18} \approx 10^{9}$ --- आपके पास एक अरब हैं
अनाज के परमाणुओं के लायक। सितारों के ख़िलाफ़:
$\num{7e27}/\num{2e23} = \num{3.5e4}$ --- आपके परमाणुओं की संख्या सभी से अधिक है
अवलोकनीय ब्रह्माण्ड के तारे पैंतीस हजार से एक।

\textbf{18.} $\text{(you)}/\text{(atom)} =
\qty{1.7}{m}/\qty{2.5e-10}{m} \approx \num{7e9} \approx 10^{10}$,
जबकि $\text{(universe)}/\text{(you)} =
\qty{8.8e26}{m}/\qty{1.7}{m} \approx \num{5e26}$. सीढ़ी पर तुम
परमाणु से लगभग दस पायदान ऊपर और लगभग सत्ताईस पायदान नीचे खड़ा है
ब्रह्मांड: गुणात्मक रूप से, मनुष्य अपने परमाणुओं के बहुत करीब रहते हैं।

\textbf{19.} शर्त देती है
$ x^{2} = 10^{-15} \times 10^{26.9} = 10^{11.9}$, इसलिए
$ x = 10^{5.95} \approx \qty{9e5}{m} \approx \qty{900}{km}$ --- लगभग
बिल्कुल सबसे बड़े क्षुद्रग्रह सेरेस का व्यास
($\approx \qty{940}{km}$). एक नाभिक और के बीच की सीढ़ी से आधा ऊपर
ब्रह्मांड एक मामूली बौना ग्रह है।

\textbf{20.} $\qty{8.8e26}{m} \times \num{3.6e-13} = \qty{3.2e14}{m}$,
यानी\ $\qty{3.2e14}{m}/\qty{1.496e11}{m} \approx \num{2.1e3}$
खगोलीय इकाइयाँ. पंचलाइन: ब्रह्मांड को तब तक सिकोड़ें जब तक सूर्य न हो जाए
रेत का कण, और देखने योग्य ब्रह्मांड अभी भी दो हजार तक फैला हुआ है
वास्तविक पृथ्वी से गुणा--सूर्य की दूरी--- कोई स्केल मॉडल नहीं लाता है
मानव आकार तक का ब्रह्मांड; केवल दस की शक्तियाँ ही इसे ले जा सकती हैं।
\end{solution}
