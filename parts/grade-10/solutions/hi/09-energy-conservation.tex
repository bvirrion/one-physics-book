\section*{अध्याय \ref{ch:g10:energy-conservation} --- ऊर्जाः स्वरूप एवं संरक्षण}

\begin{solution}{exo:g10:energy-conservation:1}
$E_k = \tfrac12 \times 1200 \times 25^2 = \qty{3.75e5}{J}$. पर
\qty{36.1}{m/s}: $\tfrac12 \times 1200 \times 36.1^2 \approx
\qty{7.8e5}{J}$ --- $44\%$ अधिक गति, $(130/90)^2 \approx 2.1$
कई बार \omterm{def:g10:energy-conservation:kinetic}{गतिज ऊर्जा}: काम पर वर्ग.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:2}
$\Delta h = \qty{450}{m}$:
$ E_p = 72 \times 9.81 \times 450 \approx \qty{3.2e5}{J}
= \num{3.2e5}/\num{3.6e6} \approx \qty{0.088}{kW.h}$, के बारे में लायक
$0.022$ यूरो. एक दिन की चढ़ाई में दो सेंट बिजली खर्च होती है।
\end{solution}

\begin{solution}{exo:g10:energy-conservation:3}

$\qty{12}{W.h} = 12 \times \qty{3600}{J} \approx \qty{43}{kJ}$;
$\qty{45}{MJ} = 45/3.6 = \qty{12.5}{kW.h}$;
$\qty{10}{MJ} = 10/3.6 \approx \qty{2.8}{kW.h}$.
\end{solution}

\begin{solution}{exo:g10:energy-conservation:4}

\begin{solution}{exo:g10:energy-conservation:5}
डायनेमो: रसायन (साइकिल चालक का भोजन) $\to $ गतिज (पहिया) $\to $
विद्युत (डायनमो) $\to $ विकिरणवाला $+$ थर्मल (दीपक); गर्मी का रिसाव
मांसपेशियों, टायर संपर्क, डायनेमो, लैंप पर। गैस हीटर: रासायनिक (गैस)
$\to $ थर्मल (पानी), ग्रिप गैसों में रिसाव। पैनल: विकिरणकारी
(रविवार) $\to $ विद्युत (पैनल) $\to $ रसायन (बैटरी), गर्मी का रिसाव
पैनल और चार्जर पर.
\end{solution}

\end{solution}

\begin{solution}{exo:g10:energy-conservation:11}

\begin{solution}{exo:g10:energy-conservation:13}

$1400/12.6 \approx 110$ दिन.
\emph{2.} $ P = \num{5e9}/\num{1e-4} = \qty{5e13}{W} = \qty{50}{TW}$
--- मानवता की औसत विद्युत का सत्रह गुना \omterm{def:g10:energy-conservation:power}{शक्ति}, के लिए
\qty{100}{\micro s}, एक अप्रत्याशित जगह पर. बिजली बहुत है
\omterm{def:g10:energy-conservation:power}{शक्ति} लेकिन थोड़ा ऊर्जा, अनावश्यक रूप से तेजी से वितरित: नहीं \omterm{def:g10:energy-conservation:transfer}{कनवर्टर}
ऐसे स्पाइक को स्वीकार करता है.
\end{solution}

\begin{solution}{pb:g10:energy-conservation:1}
\textbf{1.} फ़्रिज $0.1 \times 24 = \qty{2.4}{kW.h}$; हीटर
$2.4 \times 2.5 = \qty{6.0}{kW.h}$; ओवन \qty{2.0}{kW.h}; धुलाई
मशीन \qty{1.0}{kW.h}; प्रकाश \qty{0.3}{kW.h}; स्क्रीन
\qty{0.9}{kW.h}.
\textbf{2.} कुल \qty{12.6}{kW.h} $= 12.6 \times 3.6 \approx
\qty{45}{MJ}$: एक \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} गैसोलीन का, को \omterm{def:g10:energy-conservation:energy}{जौल}.
\textbf{3.} $\qty{12600}{W.h}/\qty{24}{h} = \qty{525}{W}$ --- द
एक चौथाई केतली पर घर बेकार है।
\textbf{4.} वॉटर हीटर (\qty{6.0}{kW.h}, लगभग आधा)। पर
\qty{1.25}{h}: हीटर \qty{3.0}{kW.h}, कुल \qty{9.6}{kW.h}.
\textbf{5.} $15 \times 24 = \qty{0.36}{kW.h}$ प्रति दिन, लगभग
$2.8\%$ \omterm{def:g10:orders-of-magnitude:unit}{मीटर} किए गए कुल का; वार्षिक $0.36 \times 365 \approx
\qty{131}{kW.h} \approx 33$ यूरो --- रोशनी के लिए कोई नहीं देखता।
\textbf{6.} $56604 - 56214 = \qty{390}{kW.h}$;
$390/30 = \qty{13.0}{kW.h}$ प्रति दिन: लेखापरीक्षा \qty{12.6}{kW.h}
इसके अलावा \qty{0.36}{kW.h} स्टैंडबाय की --- पुष्टि की गई।
\textbf{7.} $390 \times 0.25 + 14 = 111.50$ यूरो.
\textbf{8.} ग्रिड: $0.25/3.6 \approx 0.07$ यूरो प्रति मेगाजूल।
गैसोलीन: $1.80/34 \approx 0.05$ यूरो प्रति मेगाजूल।
\textbf{9.} कटोरा: $0.50/1.5 \approx 0.33$ यूरो प्रति मेगाजूल ---
ग्रिड से पांच गुना. सबसे सस्ता पहले: गैसोलीन, ग्रिड, फिर भोजन,
बहुत पीछे.
\textbf{10.} $0.20 \times 1000 = \qty{200}{W}$ प्रति वर्ग \omterm{def:g10:orders-of-magnitude:unit}{मीटर}.
\textbf{11.} $200 \times 4.0 = \qty{800}{W.h} = \qty{0.80}{kW.h}$
प्रति वर्ग \omterm{def:g10:orders-of-magnitude:unit}{मीटर} प्रति दिन।
\textbf{12.} $12.6/0.80 = 15.75 \approx \qty{16}{m^2}$: यह फिट बैठता है
द \qty{20}{m^2} छत।
\textbf{13.} $16 \times 0.2 \times 1.5 = \qty{4.8}{kW.h}$: के बारे में
$38\%$ आवश्यकता का; ग्रिड (या बेहतर तरीके से चार्ज की गई बैटरी)।
महीने) को बाकी की आपूर्ति करनी होगी।
\textbf{14.} एक वर्ग \omterm{def:g10:orders-of-magnitude:unit}{मीटर} पैदावार $0.80 \times 365 \approx
\qty{292}{kW.h}$ प्रति वर्ष; कर्ज उतारने $700/292 \approx 2.4$ वर्ष ---
25 साल के पैनल जीवनकाल के मुकाबले कम।
\textbf{15.} $\qty{12.6}{kW.h}/\qty{0.1}{kW} = \qty{126}{h}$:
पांच से अधिक लोग ($126/24 = 5.25$) दिन-रात पैडल चलाना।
\textbf{16.} खाना $= 12.6/0.25 = \qty{50.4}{kW.h} \approx
\qty{181}{MJ}$, यानी\ $181/1.5 \approx 121$ अनाज के कटोरे.
\textbf{17.} $121 \times 0.50 \approx 60$ यूरो, विरुद्ध
$12.6 \times 0.25 \approx 3.15$ ग्रिड से यूरो: पेडल \omterm{def:g10:energy-conservation:power}{शक्ति}
के बारे में लागत $19$ साइकिल चालकों को भुगतान करने से पहले कई गुना अधिक।
\textbf{18.} $5000/45 \approx 110$ घर के दिन. लेकिन
$\num{5e9}/\num{1e-4} = \qty{5e13}{W}$ के लिए \qty{100}{\micro s}, पर
एक यादृच्छिक स्थान: नहीं \omterm{def:g10:energy-conservation:transfer}{कनवर्टर} या बैटरी उस नली से पानी पी सकती है।
एक घर को एक स्थिर स्थान की आवश्यकता होती है \qty{525}{W} प्रवाह, स्पाइक नहीं.
\textbf{19.} नहीं जूल --- वे संरक्षित और समान हैं। आप
उनके लिए भुगतान करें \emph{रूप और उपलब्धता}: ऊर्जा वह है
संकेंद्रित, भंडारण योग्य और मांग के अनुसार संपूर्ण लागत \omterm{def:g10:energy-conservation:transfer}{रूपांतरण
श्रृंखला} जिससे ऐसा हो गया.
\textbf{20.} एक परिवार दिवस $= \qty{12.6}{kW.h} \approx
\qty{45}{MJ}$: एक \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} गैसोलीन का, सोलह वर्ग एम-eters का
धूपदार छत, या पाँच चौबीसों घंटे साइकिल चलाने वाले। स्टॉक है ऊर्जा
(जूल, किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे); प्रवाह है \omterm{def:g10:energy-conservation:power}{शक्ति} (\omterm{def:g10:energy-conservation:power}{वाट}) --- बाथटब
और नल, फिर कभी भ्रमित नहीं हुआ।
\end{solution}
