\section*{अध्याय \ref{ch:g10:relative-motion} --- सापेक्ष गति}

\begin{solution}{exo:g10:relative-motion:1}
गाड़ी के सापेक्ष: पर आराम. जमीन से संबंधित और के लिए
पेड़: गति में \qty{100}{km/h}. आने वाली ट्रेन के सापेक्ष:
विपरीत दिशाओं में गति जोड़ें, $100 + 100 = \qty{200}{km/h}$. में
यात्री की ट्रेन का ढांचा, पेड़ पीछे की ओर बढ़ता है
\qty{100}{km/h} एक सीधी रेखा के साथ.
\end{solution}

\begin{solution}{exo:g10:relative-motion:2}
$72 / 3.6 = \qty{20}{m/s}$; $108 / 3.6 = \qty{30}{m/s}$;
$25 \times 3.6 = \qty{90}{km/h}$; $340 \times 3.6 = \qty{1224}{km/h}$.
प्रत्येक जोड़ी में \unit{km/h} यह आंकड़ा बड़ा है, विवेक के रूप में
नियम की मांग.
\end{solution}

\begin{solution}{exo:g10:relative-motion:3}
साइकिल चालक: $ v = \frac{\qty{36}{km}}{\qty{1.5}{h}} = \qty{24}{km/h}
= 24/3.6 \approx \qty{6.7}{m/s}$. धावक:
$ v = \frac{\qty{100}{m}}{\qty{10.0}{s}} = \qty{10.0}{m/s}
= \qty{36}{km/h}$. धावक फिर से आधा तेज़ है --- लेकिन केवल के लिए
दस सेकंड, जबकि साइकिल चालक डेढ़ घंटे तक गति बनाए रखता है।
\omterm{def:g10:relative-motion:speed}{औसत गति} केवल तुलनीय अवधियों में निष्पक्ष रूप से तुलना करें।
\end{solution}

\begin{solution}{exo:g10:relative-motion:4}
एक ही दिशा: $1.4 + 0.9 = \qty{2.3}{m/s}$. बेल्ट के विरुद्ध:
$1.4 - 0.9 = \qty{0.5}{m/s}$. बेल्ट पर स्थिर \omterm{def:g10:universal-gravitation:weight}{खड़ा} होना: बेल्ट का अपना
\qty{0.9}{m/s}.
\end{solution}

\begin{solution}{exo:g10:relative-motion:5}
गिरा हुआ पेन: \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} काम करता है, लेकिन \emph{रेलगाड़ी} फ्रेम बेहतर है
फिर भी --- वहां कलम सीधे नीचे गिरती है। अंतरिक्ष स्टेशन:
\omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम} (यह कक्षाओं पृथ्वी समग्र रूप से, इसके घूर्णन को अनदेखा करते हुए)।
मंगल ग्रह का वर्ष: \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम} (मंगल चारों ओर एक निकट-चक्र चलाता है
रवि). स्प्रिंट: \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} (स्टार्ट लाइन, फिनिश लाइन और स्टॉपवॉच हैं
सभी जमीन पर टिके हुए हैं)।
\end{solution}

\begin{solution}{exo:g10:relative-motion:6}
\emph{1.} $108/3.6 = \qty{30}{m/s}$. सामने की ओर:
$30 + 1.5 = \qty{31.5}{m/s}$; पीछे की ओर:
$30 - 1.5 = \qty{28.5}{m/s}$.

\emph{2.} यात्री जिस ट्रेन से यात्रा करता है उसके सापेक्ष \qty{60}{m} पर
\qty{1.5}{m/s}: $ t = 60/1.5 = \qty{40}{s}$. जमीन के सापेक्ष
यात्री कवर $31.5 \times 40 = \qty{1260}{m}$ --- द \qty{1200}{m}
ट्रेन द्वारा ले जाया गया प्लस \qty{60}{m} चला गया, रचना के रूप में
नियम भविष्यवाणी करता है.
\end{solution}

\begin{solution}{exo:g10:relative-motion:7}
\emph{1.} डाउनस्ट्रीम: $5.0 + 2.0 = \qty{7.0}{m/s}$, इसलिए
$ t_1 = 420/7.0 = \qty{60}{s}$. अपस्ट्रीम: $5.0 - 2.0 = \qty{3.0}{m/s}$,
इसलिए $ t_2 = 420/3.0 = \qty{140}{s}$.

\emph{2.} \omterm{def:g10:relative-motion:speed}{औसत गति}: $ v = \frac{840}{60 + 140}
= \frac{840}{200} = \qty{4.2}{m/s} < \qty{5.0}{m/s}$. वर्तमान का
सहायता और बाधा बराबर हैं \emph{रफ़्तार} लेकिन अंदर नहीं \emph{समय}:
नाव खर्च करती है \qty{140}{s} केवल और केवल धीमा किया जा रहा है \qty{60}{s} किया जा रहा है
मदद की, इसलिए धीमा पैर औसत पर हावी है।
\end{solution}

\begin{solution}{exo:g10:relative-motion:8}
\emph{1.} विपरीत दिशाएँ: A के फ्रेम में, B पर पहुँचता है
$25 + 20 = \qty{45}{m/s}$. जब B फिसल गया तो क्रॉसिंग पूरी हो गई
संयुक्त लंबाई से आगे $180 + 90 = \qty{270}{m}$:
$ t = 270/45 = \qty{6.0}{s}$.

\emph{2.} समान दिशा: सापेक्ष गति $25 - 20 = \qty{5.0}{m/s}$,
और $ t = 270/5.0 = \qty{54}{s}$ --- नौ गुना अधिक, यही कारण है
ट्रक को ओवरटेक करना अंतहीन लगता है।
\end{solution}

\begin{solution}{exo:g10:relative-motion:9}
साइकिल चालक का फ़्रेम: एक वृत्त (रिम का वृत्त, हब पर केन्द्रित)।
\omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}: ए \omterm{prop:g10:relative-motion:relativity}{चक्रज} --- मेहराबों की एक शृंखला, प्रत्येक मोड़ पर एक
पहिया. वाल्व का फ़्रेम स्वयं: वाल्व चालू है आराम, एक बिंदु.
हब में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}: ऊंचाई पर एक क्षैतिज सीधी रेखा
एक पहिये की त्रिज्या का.
\end{solution}

\begin{solution}{exo:g10:relative-motion:10}
\emph{1.} $ v = \frac{\qty{40000}{km}}{\qty{24}{h}}
\approx \qty{1.7e3}{km/h} = 1667/3.6 \approx \qty{463}{m/s}$ ---
ध्वनि से भी तेज़.

\emph{2.} हम इसे महसूस नहीं कर पाते क्योंकि हमारे चारों ओर सब कुछ --- ज़मीन,
हवा, इमारतें --- बिलकुल एक जैसी गति साझा करती हैं, इसलिए कुछ भी नहीं हिलता
हमारे सापेक्ष (और जड़ता का सिद्धांत,
\cref{ch:g10:inertia}, समझाएगा कि साझा समान गति क्यों है
अंदर से पता नहीं चल पाता) ``सूरज उगता है'' ज़मीन-फ़्रेम है
रिपोर्ट और ``पृथ्वी घूमती है'' एक और की भूकेन्द्रित-फ़्रेम रिपोर्ट
वही अवलोकन.
\end{solution}

\begin{solution}{exo:g10:relative-motion:11}
होने देना $ L $ एस्केलेटर की लंबाई हो. एस्केलेटर की गति: $ L/60$; चलना
गति: $ L/90$. चलती हुई एस्केलेटर पर चलने से गति बढ़ती है:
\[
v = \frac{L}{60} + \frac{L}{90} = L\left(\frac{3 + 2}{180}\right)
= \frac{L}{36},
\]
तो सवारी लगती है \qty{36}{s}. (औसत नहीं $60$ और $90$:
गतियाँ जुड़ती हैं, समय नहीं।)
\end{solution}

\begin{solution}{exo:g10:relative-motion:12}
\emph{1.} कुछ दूरी पर $ d $ हर तरह से:
$ t = \frac{d}{60} + \frac{d}{90} = d\,\frac{3 + 2}{180} = \frac{d}{36}$,
इसलिए $ v = \frac{2d}{d/36} = \qty{72}{km/h}$.

\emph{2.} भोला $75$ दो गतियों का औसत इस प्रकार निकालता है जैसे कि ड्राइवर ने खर्च किया हो
बराबर \emph{टाइम्स} हरेक पर; वास्तव में दूरियाँ समान हैं, इसलिए
धीमा पैर अधिक समय तक चलता है ($\frac{d}{60} > \frac{d}{90}$) और खींचता है
औसत की ओर \qty{60}{km/h}. \omterm{def:g10:relative-motion:speed}{औसत गति} कुल दूरी खत्म हो गई है
कुल समय---और कुछ नहीं।
\end{solution}

\begin{solution}{exo:g10:relative-motion:13}
\emph{1.} पहले साइकिल चालक के फ्रेम में, \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} आता है
सीधी रेखा पर $15 + 20 = \qty{35}{km/h}$ से \qty{35}{km} दूर:
वे बिल्कुल बाद में मिलते हैं \qty{1.0}{h}.

\emph{2.} पहले ने सवारी की है $15 \times 1 = \qty{15}{km}$, \omterm{def:g10:orders-of-magnitude:unit}{दूसरा}
$20 \times 1 = \qty{20}{km}$ --- कुल \qty{35}{km}, जैसा कि यह होना चाहिए।

\emph{3.} वही दिशा: समापन गति $20 - 15 = \qty{5}{km/h}$, तो
पीछा चलता रहता है $35/5 = \qty{7}{h}$.
\end{solution}

\begin{solution}{exo:g10:relative-motion:14}
\emph{1.} धारा तैराक को नदी के किनारे धकेलती है, पार नहीं
यह: दो गतियाँ स्वतंत्र रूप से बनती हैं, और दूर की ओर बढ़ती हैं
बैंक बनाया गया है \qty{1.2}{m/s} बहाव की परवाह किए बिना. पार करने का समय:
$ t = 60/1.2 = \qty{50}{s}$.

\emph{2.} बहाव: $0.9 \times 50 = \qty{45}{m}$ बहाव

\emph{3.} ग्राउंड-फ़्रेम वेग दाईं ओर का कर्ण है
पैरों के साथ त्रिकोण $1.2$ और $0.9$:
$ v = \sqrt{1.2^2 + 0.9^2} = \sqrt{2.25} = \qty{1.5}{m/s}$ (ए $3$--$4$--$5$
त्रिकोण द्वारा स्केल किया गया $0.3$). मार्ग की लंबाई: $1.5 \times 50 = \qty{75}{m}$
--- या सीधे $\sqrt{60^2 + 45^2} = \qty{75}{m}$.
\end{solution}

\begin{solution}{exo:g10:relative-motion:15}
\emph{1.} $ v = \frac{2\pi \times \num{1.50e11}}{\num{3.16e7}}
= \frac{\num{9.42e11}}{\num{3.16e7}} \approx \qty{2.98e4}{m/s}
\approx \qty{30}{km/s}$.

\emph{2.} $30 / 0.46 \approx 65$: पृथ्वी हमें सूर्य के चारों ओर ले जाती है
यह हमें अपनी धुरी पर घुमाने की तुलना में लगभग पैंसठ गुना अधिक तेजी से घुमाती है।

\emph{3.} में \omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम}, मंगल का स्पष्ट पथ इसे जोड़ता है
पृथ्वी के साथ स्वयं की कक्षीय गति, और जब पृथ्वी तेज़ होती है
मंगल ग्रह से आगे निकलने पर संयोजन कुछ हफ्तों के लिए पीछे चला जाता है। में
\omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम} दोनों ग्रह स्थिर गति से सूर्य की परिक्रमा करते हैं
--- ``बैकवर्ड लूप'' मंगल ग्रह की गति नहीं बल्कि एक कलाकृति है
फ़्रेम.
\end{solution}

\begin{solution}{pb:g10:relative-motion:1}
\textbf{1.} $ t = 90/1.0 = \qty{90}{s}$.

\textbf{2.} जमीन की गति $1.5 + 1.0 = \qty{2.5}{m/s}$:
$ t = 90/2.5 = \qty{36}{s}$.

\textbf{3.} $ t = 90/1.5 = \qty{60}{s}$.

\textbf{4.} $1.5 - 1.0 = \qty{0.5}{m/s}$ ज़मीन के सापेक्ष,
बेल्ट की दिशा के विपरीत: $ t = 90/0.5 = \qty{180}{s}$ --- तीन
तेज गति से चलने के पूरे मिनट।

\textbf{5.} जमीन की गति $1.0 - 1.0 = \qty{0}{m/s}$: बच्चा दौड़ता है
कठिन और कहीं नहीं जाता, पर आराम में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} अंदर दौड़ते समय
बेल्ट फ्रेम. वह एक ट्रेडमिल है.

\textbf{6.} बेल्ट फ़्रेम / \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}: खड़े यात्री $0$ /
\qty{1.0}{m/s}; बेल्ट के साथ वॉकर $1.5$ / \qty{2.5}{m/s}; वॉकर चालू
मंजिल $0.5$ (बेल्ट फ्रेम में फर्श पीछे की ओर खिसकता है) /
\qty{1.5}{m/s}; शरारती $1.5$ / \qty{0.5}{m/s}; बच्चा $1.0$ /
\qty{0}{m/s}. प्रत्येक मूवर के पैर किस फ्रेम के स्तंभ को महसूस करते हैं
पैरों के नीचे है --- बेल्ट की सवारी करने वालों के लिए बेल्ट-फ़्रेम कॉलम,
फर्श पर वॉकर के लिए ग्राउंड-फ़्रेम कॉलम: प्रयास चलना है
पैरों के नीचे जो है उसके सापेक्ष गति।

\textbf{7.} नदी के उस पार प्रगति हुई है \qty{2.0}{m/s}
धारा जो भी करे: $ t = 120/2.0 = \qty{60}{s}$.

\textbf{8.} बहाव: $1.5 \times 60 = \qty{90}{m}$ बहाव

\textbf{9.} ज़मीन की गति: पैरों के साथ समकोण त्रिभुज का कर्ण
$2.0$ और $1.5$: $ v = \sqrt{2.0^2 + 1.5^2} = \sqrt{6.25}
= \qty{2.5}{m/s}$. \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} लंबाई: $2.5 \times 60 = \qty{150}{m}$
--- द $3$--$4$--$5$ फिर से तिगुना, स्केल किया गया $90$--$120$--$150$.

\textbf{10.} जल फ़्रेम: एक सीधी रेखा, \qty{120}{m}, लंबवत
बैंकों को. \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}: एक सीधी रेखा भी, लेकिन तिरछी, और
\qty{150}{m} लंबा। एक ही आकार (सीधा), अलग दिशा और
लंबाई: \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} फ्रेम का है, फेरी का नहीं.

\textbf{11.} पार प्रगति हुई है \qty{1.0}{m/s}, इसलिए
$ t = 120/1.0 = \qty{120}{s}$. बहाव: $1.5 \times 120 = \qty{180}{m}$
--- नौका को दोगुना करें \qty{90}{m}. ज़मीन की गति: पैरों के साथ कर्ण
$1.0$ और $1.5$: $ v = \sqrt{1.0^2 + 1.5^2} = \sqrt{3.25}
\approx \qty{1.8}{m/s}$. धीमे तैराक को तैरने में दोगुना समय लगता है
पार करते हैं, इसलिए समान धारा के पास उन्हें ले जाने के लिए दोगुना समय होता है
बहाव

\textbf{12.} \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}: लॉग धारा के अनुसार बहता है
\qty{1.5}{m/s}. जल फ़्रेम: यह पर है आराम, पानी के साथ तैरना
इसके चारों ओर. क्या लॉग ``चाल'' का कोई फ़्रेम-मुक्त उत्तर नहीं है --- कौन सा
अध्याय का संपूर्ण बिंदु यही है।

\textbf{13.}  \omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम}. \omterm{def:g10:universal-gravitation:satellite}{कक्षा} RADIUS
$ r = \num{6.37e6} + \num{4.0e5} = \qty{6.77e6}{m}$, इसलिए
\[
v = \frac{2\pi r}{T} = \frac{2\pi \times \num{6.77e6}}{\num{5.56e3}}
\approx \qty{7.65e3}{m/s} \approx \qty{7.7}{km/s}.
\]

\textbf{14.} जबकि स्टेशन एक पूरा करता है \omterm{def:g10:universal-gravitation:satellite}{कक्षा}, पृथ्वी घूमती है
इसके नीचे लगभग $23^\circ $, इसलिए जमीन के ऊपर का ट्रैक बदल जाता है
प्रत्येक मार्ग पर पश्चिम की ओर। में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} स्टेशन का पथ है a
जटिल लहरदार वक्र जो कभी भी बिल्कुल दोहराता नहीं है; में भूकेन्द्रित
ढाँचा यह एक निश्चित निकट-वृत्त है।

\textbf{15.} $7.65 / 0.46 \approx 17$: स्टेशन कताई से आगे निकल जाता है
भूमध्य रेखा लगभग सत्रह गुना है, यही कारण है कि यह पूरे आकाश को पार करती है
मिनट.

\textbf{16.} $ v = \frac{2\pi \times \num{4.22e7}}{\num{8.64e4}}
\approx \qty{3.07e3}{m/s} \approx \qty{3.1}{km/s}$ में भूकेन्द्रित
ढाँचा. में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} यह है \emph{पर आराम}, एक स्थिर पर लटका हुआ
भूमध्य रेखा का बिंदु--- यह पृथ्वी की धुरी के ठीक परितः चक्कर लगाता है
समय जमीन ही करता है, तो दोनों गतियों में रद्द कर देते हैं ज़मीन
ढाँचा. पर आराम और कम से \qty{3.1}{km/s}: दोनों सत्य, प्रति फ्रेम एक।

\textbf{17.} में \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम} चंद्रमा अनिवार्य रूप से यात्रा करता है
पृथ्वी के साथ \omterm{def:g10:universal-gravitation:satellite}{कक्षा} --- सूर्य के चारों ओर एक विशाल निकट-वृत्त ---
की हल्की सी डगमगाहट से ढका हुआ \qty{27}{day} चक्र। सूरज का
प्रभाव इतना प्रबल होता है कि रास्ता हमेशा की ओर झुक जाता है
रवि; यह कभी भी पीछे की ओर नहीं मुड़ता।

\textbf{18.} ``सूरज उगता है'' रिपोर्ट है \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}, कहाँ
पर्यवेक्षक पर है आराम और सूर्य आकाश में फैल जाता है। ``पृथ्वी
स्पिन्स'' में रिपोर्ट है \omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम}, जहां सूर्य है
दिशा स्थिर है और प्रेक्षक को चारों ओर पूर्व की ओर ले जाया जाता है
अक्ष. वे उसी अवलोकन का वर्णन करते हैं; प्रत्येक अपने आप में सही है
फ़्रेम, और \cref{prop:g10:relative-motion:relativity} नहीं कहता
केवल गति का अवलोकन ही मध्यस्थता कर सकता है।

\textbf{19.} टॉलेमी का पृथ्वी-केंद्रित फ्रेम आकाश के दैनिक से मेल खाता था
गति स्वाभाविक रूप से और सेवापूर्वक ग्रहों की स्थिति की भविष्यवाणी की, लेकिन
प्रत्येक को पुन: पेश करने के लिए केवल लूप पर लूप (एपिसाइकिल) की कीमत पर
ग्रह का भ्रमण. कॉपरनिकस के सूर्य-केन्द्रित फ्रेम ने प्रत्येक ग्रह को एक बना दिया
स्थिर निकट-चक्र और मंगल के पिछड़े लूप को एक फ्रेम में बदल दिया
प्रभाव---पृथ्वी अंदर की लेन पर धीमे धावक से आगे निकल जाती है। द
आधुनिक निर्णय यह नहीं है कि सूर्य ``वास्तव में'' स्थिर \omterm{def:g10:universal-gravitation:weight}{खड़ा} है (यह भी)।
आकाशगंगा के केंद्र के चारों ओर घूमता है) लेकिन वह फ्रेम उपकरण हैं:
हेलियोसेंट्रिक ग्रहीय गति को अतुलनीय रूप से सरल बनाता है, और
एक ढाँचा सरलता के लिए होता है, सत्य के लिए नहीं।

\textbf{20.} \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा}: \qty{0}{km/s} --- पर आराम आरामकुर्सी में.
\omterm{def:g10:relative-motion:frames}{भूकेंद्रिक फ़्रेम}: के बारे में \qty{0.46}{km/s}, पृथ्वी के चारों ओर ले जाया गया
अक्ष. \omterm{def:g10:relative-motion:frames}{हेलिओसेंट्रिक फ़्रेम}: के बारे में \qty{30}{km/s}, सूर्य के चारों ओर ले जाया गया।
एक के दौरान \qty{90}{min} पतली परत: $ d = \num{3.0e4} \times 5400
\approx \qty{1.6e8}{m} = \qty{160000}{km}$ --- लगभग आधे रास्ते तक
चंद्रमा बिना कुर्सी छोड़े। नैतिक: गति किसी का गुण नहीं है
शरीर लेकिन एक शरीर और एक फ्रेम का; वह फ़्रेम चुनें जो कहानी बनाता है
सरल, और हमेशा बताएं कि आपने किसे चुना है।
\end{solution}
