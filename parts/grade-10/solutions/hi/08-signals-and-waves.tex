\end{solution}

\end{solution}

\begin{solution}{exo:g10:signals-and-waves:4}

\omterm{def:g10:signals-and-waves:amplitude}{आयाम} $4.0 \times \qty{0.5}{V} = \qty{2.0}{V}$.

\begin{solution}{exo:g10:signals-and-waves:5}
$ d = 340 \times 6.0 = \qty{2040}{m} \approx \qty{2.0}{km}$. प्रकाश \omterm{def:g10:refraction:fiber}{आवरण}
इस में $2040/(3.00\times10^{8}) \approx \qty{7}{\micro s}$, एक लाख
से कई गुना कम \qty{6.0}{s}: फ़्लैश हमले के तत्काल को चिह्नित करता है।
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:6}
 \omterm{def:g10:signals-and-waves:sound}{आवाज़} नीचे जाता है \emph{और वापस}, ढकना $2d $:
$ d = \frac{1500 \times 0.60}{2} = \qty{450}{m}$. खाई के ऊपर:
$ t = 2d/v = 2 \times 1200/1500 = \qty{1.6}{s}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:7}
$ d_1 = \frac{3.00\times10^{8} \times 2.4\times10^{-4}}{2} =
\qty{36.0}{km}$ और $ d_2 = \qty{35.7}{km}$: विमान को कवर किया गया
\qty{300}{m} में \qty{1.0}{s}, इसलिए $ v = \qty{300}{m/s} \approx
\qty{1100}{km/h}$, आ रहा है।
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:8}
$ T = \qty{20}{mm}/\qty{25}{mm/s} = \qty{0.80}{s}$, इसलिए
$ f = \qty{1.25}{Hz}$: $75$ प्रति मिनट धड़कता है। के लिए $100$ प्रति मिनट धड़कता है,
$ T = \qty{0.60}{s}$, यानी\ $25 \times 0.60 = \qty{15}{mm}$: तचीकार्डिया
नीचे दिखाता है \qty{15}{mm}.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:9}
$ T = 1/200 = \qty{5.0}{ms}$; दो चक्र चलते हैं \qty{10}{ms}, फैला हुआ
$10$ प्रभाग: \omterm{def:g10:signals-and-waves:oscilloscope}{समय आधार} \qty{1}{ms} प्रति प्रभाग. शिखा फिट होनी चाहिए
में $4$ प्रभाग: \omterm{def:g10:signals-and-waves:oscilloscope}{पाना} कम से कम $3.0/4 = \qty{0.75}{V}$ प्रति प्रभाग ---
ले लो \qty{1}{V} प्रति प्रभाग (शिखा $3.0$ विभाजन ऊपर)।
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:10}

$\frac{1540 \times 6.0\times10^{-5}}{2} = \qty{4.6}{cm}$; मोटाई

\end{solution}

\begin{solution}{exo:g10:signals-and-waves:11}
\omterm{def:g10:signals-and-waves:sound}{आवाज़}: $30/340 = \qty{0.088}{s}$. रेडियो:
$3.0\times10^{6}/(3.00\times10^{8}) = \qty{0.010}{s}$. दूर का रेडियो
श्रोता सबसे पहले ड्रम सुनता है, लगभग \qty{78}{ms}.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:12}
के लिए उत्सर्जन करते समय \qty{3.0}{ms} बल्ला बहरा है इकोज, यानी\ को
भीतर कुछ भी $ d = \frac{340 \times 3.0\times10^{-3}}{2} =
\qty{0.51}{m}$. कीट पर \qty{1.7}{m}:
$ t = 2 \times 1.7/340 = \qty{10}{ms}$. में बंद हो रहा है, गूंज कभी भी लौटता है
जल्दी; नाड़ी आने से पहले ख़त्म होनी चाहिए, इसलिए इसे छोटा करना होगा।
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:13}
$ d = \frac{340 \times 1.5}{2} = \qty{255}{m}$. पर \qty{155}{m}:
$ t = 2 \times 155/340 = \qty{0.91}{s}$. गूंज और ताली मर्ज जब
$ t < \qty{0.1}{s}$, यानी\ अंदर
$ d < \frac{340 \times 0.1}{2} = \qty{17}{m}$.
\end{solution}

\begin{solution}{exo:g10:signals-and-waves:14}
\emph{1.} $ T = 3.0 \times \qty{5}{ms} = \qty{15}{ms}$,
$ f = 1/0.015 \approx \qty{67}{Hz}$.
\emph{2.} $\qty{15}{ms}/\qty{2}{ms} = 7.5$ प्रभाग.
\emph{3.} स्क्रीन दिखाता है \qty{50}{ms}, तब \qty{20}{ms}: $3$ पूर्ण

\end{solution}

\begin{solution}{exo:g10:signals-and-waves:15}
\emph{1.} $ t = 2 \times 1.2\times10^{6}/(3.00\times10^{8}) =
\qty{8.0}{ms}$.
\emph{2.} $2 \times 1.2\times10^{6}/(2.0\times10^{8}) = \qty{12}{ms}$.
\emph{3.} फ़ाइबर सीधे दो मशीनों के बीच नहीं चलता है, और
रास्ते में आने वाला प्रत्येक राउटर और सर्वर प्रोसेसिंग में देरी जोड़ता है।
\end{solution}

\begin{solution}{pb:g10:signals-and-waves:1}
\textbf{1.} $ d = 340 \times 9.0 = \qty{3060}{m} \approx \qty{3.1}{km}$.

\textbf{2.} $3060/(3.00\times10^{8}) \approx \qty{10}{\micro s}$, ए
से लाख गुना छोटा \qty{9.0}{s}: फ़्लैश तत्काल को चिह्नित करता है
हड़ताल.

\textbf{3.} में \qty{3}{s} \omterm{def:g10:signals-and-waves:sound}{आवाज़} कवर $340 \times 3 = \qty{1020}{m}
\approx \qty{1}{km}$: सेकंड को तीन से विभाजित करने पर किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} मिलता है।

\textbf{4.} $ d = 340 \times 6.0 = \qty{2040}{m}$. तूफ़ान बंद हो गया
\qty{1020}{m} में \qty{300}{s}: $ v = \qty{3.4}{m/s} \approx
\qty{12}{km/h}$, नाव की ओर जा रहा हूँ।

\textbf{5.} निकट अंत बाद में सुना जाता है $2000/340 = \qty{5.9}{s}$, द
बहुत दूर के बाद $5000/340 = \qty{14.7}{s}$: गड़गड़ाहट लगभग बनी रहती है
\qty{8.8}{s}.

\textbf{6.} $ d = 1500 \times 0.90/2 = \qty{675}{m}$.

\textbf{7.} $ d = 1500 \times 0.20/2 = \qty{150}{m}$.

\textbf{8.} दो बाधाएँ: एक मछली का किनारा $1500 \times 0.16/2 =
\qty{120}{m}$ और समुद्र तल पर \qty{150}{m}. उथला तैरता है
\qty{30}{m} नीचे से ऊपर.

\textbf{9.}  गूंज अगले पिंग से पहले वापस लौटना होगा:
$ d_{\max} = 1500 \times 0.50/2 = \qty{375}{m}$. गहरे पानी से
गूंज आता है \emph{बाद} अगला पिंग और इसके लिए जिम्मेदार है: द
डिस्प्ले एक गलत, बहुत उथला तल दिखाता है।

\textbf{10.} हवा में, $340 \times 0.60/2 = \qty{102}{m}$ के बजाय
\qty{450}{m}. देरी तभी दूरी बन जाती है जब आप वाहक को जान लेते हैं
और माध्यम में इसकी गति पार हो गई।

\textbf{11.} $ f = 1/0.86 = \qty{1.16}{Hz}$, यानी\ $1.16 \times 60
\approx 70$ प्रति मिनट धड़कता है।

\textbf{12.} $25 \times 0.86 = \qty{21.5}{mm}$.

\textbf{13.} $ T = 15/25 = \qty{0.60}{s}$: $100$ प्रति मिनट धड़कता है।

\textbf{14.} $ T = 60/50 = \qty{1.2}{s}$; अंतर
$25 \times 1.2 = \qty{30}{mm}$.

\textbf{15.} नहीं: \omterm{def:g10:signals-and-waves:period}{अवधि} प्रयास से एक धड़कन से दूसरी धड़कन की ओर बढ़ता है,
श्वास और \omterm{def:g10:inertia:inventory}{तनाव}. डॉक्टर पढ़ता है \omterm{def:g10:signals-and-waves:period}{अवधि} (दर), इसका
नियमितता, और प्रत्येक चक्र का आकार --- निदान में है
\omterm{def:g10:signals-and-waves:signal}{संकेत}.

\textbf{16.} रेडियो: $5.0\times10^{4}/(3.00\times10^{8}) =
\qty{0.17}{ms}$ --- मानवीय संवेदनाओं के लिए तात्कालिक। \omterm{def:g10:signals-and-waves:sound}{आवाज़}:
$50\,000/340 \approx \qty{147}{s}$, ढाई मिनट.

\textbf{17.} $ t = 2.02\times10^{7}/(3.00\times10^{8}) = \qty{67}{ms}$.

\textbf{18.} $ c \times \qty{1.0}{\micro s} = \qty{300}{m}$. एक के लिए
\qty{3.0}{m} हल करना: $3.0/(3.00\times10^{8}) = \qty{10}{ns}$ --- जो है
जीपीएस क्यों? \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों परमाणु घड़ियाँ ले जाओ.

\textbf{19.} \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों शून्य में बैठो, और \omterm{def:g10:signals-and-waves:sound}{आवाज़} एक माध्यम की आवश्यकता है: नहीं
\omterm{def:g10:signals-and-waves:signal}{संकेत} छोड़ देंगे \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} बिल्कुल भी। (यहां तक कि पूरी हवा देना भी
रास्ता, \qty{20200}{km} पर \qty{340}{m/s} खत्म हो गया है \qty{16}{h} --- ए
स्थिति ठीक करने के घंटे पुराने हो गए हैं।)

\textbf{20.} \qty{340}{m/s} (\omterm{def:g10:signals-and-waves:sound}{आवाज़} हवा में), \qty{1500}{m/s} (\omterm{def:g10:signals-and-waves:sound}{आवाज़} में
समुद्री जल) और $ c = \qty{3.00e8}{m/s}$ (रेडियो और प्रकाश); कानून है
$ d = v\,t$, एक के लिए आधा कर दिया गया गूंज. आदत: अपने वाहक को जानें, उसे जानें
माध्यम में गति, समय विलंब --- विलंब एक दूरी है
भेष बदलना।
\end{solution}
