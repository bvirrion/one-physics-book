\section*{अध्याय \ref{ch:g10:pressure} --- दबाव: खेल से लेकर गोताखोरी तक}

\begin{solution}{exo:g10:pressure:1}
बड़ा चेहरा: $S = \qty{0.15}{m^2}$, $ P = 600/0.15 = \qty{4.0}{kPa}$.
छोटा चेहरा: $ S = \qty{0.060}{m^2}$, $ P = \qty{10}{kPa}$. रेत पर,
बड़ा चेहरा: निचला \omterm{def:g10:pressure:pressure}{दबाव}, कम डूबना।
\end{solution}

\begin{solution}{exo:g10:pressure:2}
$\qty{0.25}{MPa} = \qty{250}{kPa} = \qty{2.5e5}{Pa}$.
$3.0/\num{2.0e-4} = \qty{15}{kPa}$ धड़कता है
$60/0.050 = \qty{1.2}{kPa}$: छोटा बल \omterm{def:g10:inertia:inventory}{जोर} से दबाता है.
\end{solution}

\begin{solution}{exo:g10:pressure:3}
$\rho g h = 1000 \times 9.81 \times 3.0 \approx \qty{29}{kPa}$;
\omterm{met:g10:pressure:dive}{निरपेक्ष} $ P = 101 + 29 = \qty{130}{kPa}$.
\end{solution}

\begin{solution}{exo:g10:pressure:4}
$ V_2 = 100 \times 4.0 / 250 = \qty{1.6}{L}$;
पर \qty{50}{kPa}, $ V_2 = 400/50 = \qty{8.0}{L}$.
\end{solution}

\begin{solution}{exo:g10:pressure:5}
$ F = 65 \times 9.81 \approx \qty{638}{N}$. ब्लेड:
$638/\num{1.2e-3} \approx \qty{5.3e5}{Pa} = \qty{530}{kPa}$. तलवे:
$638/0.036 \approx \qty{18}{kPa}$. अनुपात $\approx 30$: ब्लेड काटता है
बर्फ में, तलवे नहीं डूबते।
\end{solution}

\begin{solution}{exo:g10:pressure:6}
अँगूठा: $20/\num{1.2e-4} \approx \qty{1.7e5}{Pa} = \qty{170}{kPa}$.
दीवार: $20/\num{1.0e-8} = \qty{2.0e9}{Pa}$. जो उसी \qty{20}{N} पर कार्य करता है
एक क्षेत्र $12\,000$ सिरे पर कई गुना छोटा, इसलिए केवल वहीं होता है
\omterm{def:g10:pressure:pressure}{दबाव} सामग्री जितना झेल सकती है उससे अधिक।
\end{solution}

\begin{solution}{exo:g10:pressure:7}
$\Delta P = \rho g h = 1025 \times 9.81 \times 30 \approx
\qty{3.0e5}{Pa}$, इसलिए
$ F = \num{3.0e5} \times 0.50 \approx \qty{1.5e5}{N}$ --- द \omterm{def:g10:universal-gravitation:weight}{वज़न} का
$ F/g \approx \qty{1.5e4}{kg}$, पन्द्रह टन. कोई दल उसे नहीं खोलता
समुद्र के विपरीत बाहर की ओर.
\end{solution}

\begin{solution}{exo:g10:pressure:8}
$\rho = \dfrac{P}{g h} = \dfrac{\num{66.2e3}}{9.81 \times 5.00}
\approx \qty{1350}{kg/m^3}$. ताजा पानी नहीं ($\qty{1000}{kg/m^3}$):
एक स्पष्ट रूप से सघन तरल, उदा.\ एक संकेंद्रित नमकीन पानी.
\end{solution}

\begin{solution}{exo:g10:pressure:9}
$ h = P_0/(\rho g)$: पानी
$\num{1.01e5}/9810 \approx \qty{10.3}{m}$; पारा
$\num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$. ए
\qty{76}{cm} ट्यूब डेस्क पर फिट होती है; ए \qty{10}{m} जल बैरो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} की आवश्यकता
एक सीढ़ी--- पारे का घनत्व जीत गया।
\end{solution}

\begin{solution}{exo:g10:pressure:10}
$ V = 6000/240 = \qty{25}{mL}$; $ P = 6000/75 = \qty{80}{kPa}$.
$ P $ ख़िलाफ़ $ V $ एक अतिपरवलय है; $ P $ ख़िलाफ़ $1/V $ एक सीधी रेखा है
ढलान की उत्पत्ति के माध्यम से \qty{6000}{kPa.mL}.
\end{solution}

\begin{solution}{exo:g10:pressure:11}
पर \qty{20}{m}: $ P = 101 + 2 \times 101 = \qty{303}{kPa}$, तीन
वातावरण. सतह पर,
$ V = 0.50 \times 303/101 = \qty{1.5}{cm^3}$: बुलबुला तीन गुना।
\end{solution}

\begin{solution}{exo:g10:pressure:12}
\emph{1.} $ F = \num{1.01e5} \times \num{1.2e-3} \approx \qty{121}{N}$,
धारण करना $ m = 121/9.81 \approx \qty{12}{kg}$.

\emph{2.} $\Delta P = 101 - 20 = \qty{81}{kPa}$:
$ F = \num{8.1e4} \times \num{1.2e-3} \approx \qty{97}{N}$, के बारे में
\qty{9.9}{kg}.
\end{solution}

\begin{solution}{exo:g10:pressure:13}
बॉयल: $12 \times (200 - 50) = \qty{1800}{L}$ का \omterm{def:g10:pressure:pressure}{सतह-दबाव} वायु
प्रयोग करने योग्य है. पर \qty{30}{m} प्रत्येक सांस खींची जाती है \qty{4}{bar}, इसलिए
\qty{15}{L/min} परिवेश पर \omterm{def:g10:pressure:pressure}{दबाव} लागत
$15 \times 4 = \qty{60}{L/min}$ सतही वायु का: $1800/60 =
\qty{30}{min}$. पर \qty{10}{m} ($\qty{2}{bar}$): \qty{30}{L/min},
इसलिए \qty{60}{min}. हवा में गहराई का भुगतान किया जाता है।
\end{solution}

\begin{solution}{exo:g10:pressure:14}
$ P = P_0 \dfrac{V_0}{V} = 101 \times \dfrac{6.0}{1.5} =
\qty{404}{kPa}$. साथ $\rho g \approx \qty{10.1}{kPa/m}$,
$ h = (404 - 101)/10.1 \approx \qty{30}{m}$.
\end{solution}

\begin{solution}{exo:g10:pressure:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{वज़न} $ F = 1200 \times 9.81 \approx \qty{1.18e4}{N}$;
\omterm{def:g10:pressure:pressure}{दबाव} $ P = F/S = \num{1.18e4}/0.040 \approx \qty{2.9e5}{Pa}$; छोटा
पिस्टन: $ f = P s = \num{2.9e5} \times \num{2.0e-4} \approx \qty{59}{N}$
--- द बल द्वारा विभाजित किया गया है $ S/s = 200$.

\emph{2.} द्रव की मात्रा संरक्षित रहती है:
$400 \times 2.0 = \qty{800}{cm^3}$ छोटे सिलेंडर से आना चाहिए,
जो यात्रा करता है $800/2.0 = \qty{400}{cm} = \qty{4.0}{m}$. काम:
$59 \times 4.0 \approx \qty{235}{J}$ एक तरफ पर,
$\num{1.18e4} \times 0.020 \approx \qty{235}{J}$ दूसरे पर ---
लिफ्ट व्यापार दूरी के लिए बल, कभी काम मत करो.
\end{solution}

\begin{solution}{pb:g10:pressure:1}
\textbf{1.} $\rho g = 1025 \times 9.81 \approx \qty{10.1}{kPa}$ प्रति
\omterm{def:g10:orders-of-magnitude:unit}{मीटर}, इसलिए \qty{10}{m} समुद्र का पानी जोड़ें $\approx \qty{101}{kPa}$: एक
प्रति दस वातावरण एम-eters, लगभग बिलकुल।

\textbf{2.} $ P = 101 + 10.1\,h $: \qty{202}{kPa} पर \qty{10}{m},
\qty{302}{kPa} पर \qty{20}{m}, \qty{403}{kPa} पर \qty{30}{m},
\qty{503}{kPa} पर \qty{40}{m}.

\textbf{3.} $2.0$, $3.0$, $4.0$ और $5.0$ टाइम्स $ P_0$.

\textbf{4.} $\Delta P = 403 - 101 = \qty{302}{kPa}$ ऊपर
$\qty{0.015}{m^2}$: $ F = \num{3.02e5} \times 0.015 \approx
\qty{4.5e3}{N}$ --- द \omterm{def:g10:universal-gravitation:weight}{वज़न} चेहरे पर लगभग आधा टन का.

\textbf{5.} $\Delta P = 10.1 \times 3.0 \approx \qty{30}{kPa}$;
$ F = \num{3.0e4} \times \num{6.0e-5} \approx \qty{1.8}{N}$ --- ए
पहले से ही पूल की गहराई पर, कान के परदे पर उंगली दबाई। बराबर धक्का
हवा मध्य कान में अंदर तक जाती है \omterm{def:g10:pressure:pressure}{दबाव} पानी से मेल खाता है.

\textbf{6.} $ V = 6.0 \, P_0/P $: $6.0 \times 101/202 = \qty{3.0}{L}$;
$606/302 = \qty{2.0}{L}$; $606/403 = \qty{1.5}{L}$.

\textbf{7.} कब आधा कर दिया $ P = 2P_0$, यानी\ पर \qty{10}{m} --- में
गोता लगाने का सबसे पहला चरण।

\textbf{8.} नहीं: $ V = 606/P $ में एक अतिपरवलय है पूर्ण \omterm{def:g10:pressure:pressure}{दबाव}
(और $ P $, नहीं $ h $, क्या है \omterm{prop:g10:pressure:boyle}{बॉयल के नियम} देखता है); प्रत्येक अतिरिक्त
\qty{10}{m} पिछले वाले की तुलना में कम वॉल्यूम हटाता है।

\textbf{9.} $ V = \qty{1.5}{L}$ आवश्यक है
$ P = 101 \times 6.0/1.5 = \qty{404}{kPa}$, अर्थात\
$ h = 303/10.1 \approx \qty{30}{m}$ (\cref{exo:g10:pressure:14}).

\textbf{10.} रक्त: प्लाज्मा छाती की वाहिकाओं में स्थानांतरित हो जाता है,
असम्पीडित, और इतना आयतन घेर लेता है कि हवा अब नहीं भरती।

\textbf{11.} परिवेश में \omterm{def:g10:pressure:pressure}{दबाव}, \qty{403}{kPa}. बॉयल द्वारा भी वैसा ही
पर हवा \qty{101}{kPa} भरेगा $4$ आयतन का गुना: यह है $4.0$
सतही हवा से कई गुना सघन।

\textbf{12.} $ V = 6.0 \times 403/202 \approx \qty{12}{L}$ --- दोहरा,
केवल के बाद \qty{20}{m} वृद्धि का.

\textbf{13.} $ V = 6.0 \times 403/101 \approx \qty{24}{L}$: चार बार
उसकी फेफड़ों की क्षमता.

\textbf{14.} \emph{चढ़ते समय कभी भी अपनी सांस न रोकें।} के साथ
वायुमार्ग खुला है, फैली हुई हवा नियामक के माध्यम से बाहर बहती है, और
फेफड़ों का आयतन कभी भी अधिक नहीं होता \qty{6.0}{L}.

\textbf{15.} लीना का \qty{1.5}{L} पर \qty{30}{m} \emph{है} उसकी सतह
\qty{6.0}{L} संपीड़ित: ऊपर जाते-जाते यह बिल्कुल फिर से विस्तारित हो जाता है
\qty{6.0}{L}, इससे आगे कभी नहीं। मार्को का \qty{6.0}{L} थे \emph{लिया} पर
\qty{403}{kPa}: उनके पास है \qty{24}{L} उनमें सतही हवा का.

\textbf{16.} $ V = 1.0 \times 503/P $: \qty{1.25}{cm^3} पर \qty{30}{m},
\qty{1.67}{cm^3} पर \qty{20}{m}, \qty{2.49}{cm^3} पर \qty{10}{m},
\qty{4.98}{cm^3} सतह पर.

\textbf{17.} स्टेज कारक: $503/403 = 1.25$, तब $1.33$, $1.49$,
और $202/101 = 2.0$ पिछले दस के लिए एम-eters --- जैसे-जैसे विकास तेज होता है
सतह निकट है.

\textbf{18.} $40/10 = \qty{4.0}{min}$. अंतिम मिनट: से
\qty{10}{m} सतह पर पूर्ण \omterm{def:g10:pressure:pressure}{दबाव} आधे, सबसे बड़े
संपूर्ण चढ़ाई की सापेक्ष गिरावट।

\textbf{19.} एक तेज़ चढ़ाई से घुली हुई नाइट्रोजन फ़िज़ हो जाती है
रक्त और जोड़ों के अंदर बुलबुले (डीकंप्रेसन बीमारी); आरोही
धीरे-धीरे और पास रुकते हुए \qty{5}{m} गैस को काफी देर तक घुलाए रखता है
फेफड़ों के माध्यम से चुपचाप निकल जाना।

\textbf{20.} मुक्त गोताखोर: उसकी हवा केवल अपनी मूल मात्रा में लौटती है
--- बॉयल उसकी रक्षा करता है। स्कूबा गोताखोर: गहराई पर ली गई हवा कई गुना बढ़ जाती है
ऊपर जाने का रास्ता --- साँस लें, कभी रोकें नहीं। और आखिरी दस एम-eters दोगुना
प्रत्येक फंसी हुई मात्रा ($\times 2.0$, ख़िलाफ़ $\times 1.25$ पर उतरना
\qty{40}{m}): सतह जितनी करीब होगी, आपको उतनी ही धीमी गति से आगे बढ़ना चाहिए
यह.
\end{solution}
