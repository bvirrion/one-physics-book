\begin{solution}{exo:g10:refraction:6}
\emph{1.} पाँच अनुपात हैं
\[
\frac{0.342}{0.233} = 1.47,\quad
\frac{0.500}{0.334} = 1.50,\quad
\frac{0.643}{0.431} = 1.49,\quad
\frac{0.766}{0.515} = 1.49,\quad
\frac{0.866}{0.581} = 1.49 :
\]
माप त्रुटि के भीतर स्थिर --- \omterm{thm:g10:refraction:snell}{स्नेल का नियम}.

\emph{2.} $ n \approx 1.49$: प्लेक्सीग्लास.

\emph{3.} \omterm{thm:g10:refraction:snell}{स्नेल का नियम} भविष्यवाणी $\sin i_2 = \sin i_1 / n $: साइन विरुद्ध
साइन, डेटा को मूल के माध्यम से एक सीधी रेखा पर संरेखित होना चाहिए, जो ए
शासक एक नज़र में जाँच करता है। कोण के विरुद्ध कोण प्लॉट किया गया, डेटा एक पर स्थित है
वक्र; छोटे कोणों पर वक्र भ्रामक रूप से एक सीधी रेखा के करीब होता है
रेखा, जिसने वास्तव में टॉलेमी को मूर्ख बनाया।
\end{solution}

\begin{solution}{exo:g10:refraction:7}
$\sin i_c = n_2/n_1$ साथ $ n_2 = 1.00$:
पानी $\sin i_c = 1/1.33 = 0.752$, $ i_c = 48.8^\circ $;
काँच $\sin i_c = 1/1.50 = 0.667$, $ i_c = 41.8^\circ $;
डायमंड $\sin i_c = 1/2.42 = 0.413$, $ i_c = 24.4^\circ $.

हवा से पानी में, $\sin i_2 = \sin i_1/1.33 \leq 1/1.33 < 1$ के लिए
प्रत्येक घटना: एक अपवर्तित कोण हमेशा मौजूद होता है, इसलिए कुछ खास नहीं
कभी भी होता है --- पूर्ण आंतरिक प्रतिबिंब आवश्यकताओं $ n_1 > n_2$.
\end{solution}

\begin{solution}{exo:g10:refraction:9}
\emph{1.} $\sin r = \dfrac{\sin 50^\circ}{1.50} = \dfrac{0.766}{1.50}
= 0.511$, इसलिए $ r = 30.7^\circ $.

\emph{2.} हवा देने के लिए ग्लास $30.7^\circ $:
$\sin i_2 = 1.50 \times 0.511 = 0.766$, इसलिए $ i_2 = 50^\circ $ --- द
निकास कोण प्रवेश कोण के बराबर होता है।

\emph{3.} दो \omterm{def:g10:refraction:refraction}{अपवर्तन} एक दूसरे को पूर्ववत करें: किरण निकल जाती है
\emph{समानांतर} अपनी मूल दिशा में, बस बग़ल में स्थानांतरित हो गया।
यही कारण है कि एक खिड़की दृश्य को विकृत नहीं करती, केवल (अगोचर रूप से)
इसका अनुवाद करता है.
\end{solution}

\begin{solution}{exo:g10:refraction:12}
\emph{1.} क्षितिज से प्रकाश चरता हुआ आता है $ i_1$ के करीब
$90^\circ $; यह अपवर्तित करता है $\sin i_2 = \sin 90^\circ / 1.33 = 0.752$,
यानी\ $ i_2 = 48.8^\circ $ --- द \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}, उत्क्रमणीयता के रूप में
मांग. इसलिए पूरे आकाश से किरणें एक शंकु के भीतर आंख तक पहुंचती हैं
आधे कोण का $48.8^\circ $.

\emph{2.} डिस्क का किनारा वह स्थान है जहां शंकु सतह से मिलता है:
\[
r = d \tan i_c = \qty{2.0}{m} \times \tan 48.8^\circ
= \qty{2.0}{m} \times 1.14 = \qty{2.3}{m}.
\]

\emph{3.} डिस्क के बाहर, नीचे से प्रकाश परे की सतह से टकराता है
द \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} और पूरी तरह से परिलक्षित होता है: गोताखोर को एक चांदी दिखाई देती है
पूल का तल दिखाने वाला दर्पण।
\end{solution}

\begin{solution}{exo:g10:refraction:13}
\emph{1.} लाल: $\sin i_2 = 1.51 \times \sin 30^\circ = 0.755$, इसलिए
$ i_2 = 49.0^\circ $. नीला: $\sin i_2 = 1.53 \times 0.500 = 0.765$, इसलिए
$ i_2 = 49.9^\circ $. सफेद किरण पंखे बाहर
$49.9^\circ - 49.0^\circ = 0.9^\circ $ --- एक प्रिज्म \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} ए से
एकल \omterm{def:g10:refraction:refraction}{अपवर्तन}.

\emph{2.} पर $45^\circ $: $\sin i_2 = 1.51 \times 0.707 = 1.07 > 1$
(और $1.53 \times 0.707 = 1.08$ नीले रंग के लिए)। कोई अपवर्तित कोण मौजूद नहीं है
किसी भी रंग के लिए: \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण} है
$\arcsin(1/1.51) = 41.5^\circ < 45^\circ $, तो पूरी तरह से लंबा चेहरा
संपूर्ण किरण को प्रतिबिंबित करता है.
\end{solution}

\begin{solution}{exo:g10:refraction:14}
\emph{1.} गंभीर कोण कांच--वायु: $\sin i_c = 1/1.50$, इसलिए
$ i_c = 41.8^\circ < 45^\circ $: किरण पूर्णतः परावर्तित होती है --- $100\%$
प्रकाश की, धूमिल करने के लिए कोई लेप नहीं। एक धातु का दर्पण कुछ खो देता है
प्रत्येक उछाल पर प्रतिशत, और एक पेरिस्कोप को दो की आवश्यकता होती है।

\emph{2.} पानी का ग्लास: $\sin i_c = \dfrac{1.33}{1.50} = 0.887$, इसलिए
$ i_c = 62.5^\circ $. अब $45^\circ < i_c $: घटना नीचे है
गंभीर, अधिकांश प्रकाश पानी में अपवर्तित हो जाता है, और
दर्पण --- इसलिए छवि --- विफल हो जाती है।

\emph{3.} $\sin i_2 = \dfrac{1.50}{1.33} \times \sin 45^\circ
= 1.128 \times 0.707 = 0.798$, इसलिए $ i_2 = 52.9^\circ $ पानी में.
\end{solution}

\begin{solution}{exo:g10:refraction:15}

\emph{3.} सूचकांक कम करें --- खोखला-कोर तंतु प्रकाश का मार्गदर्शन करते हैं

कीमतें प्रति मिलीसेकंड.
\end{solution}

\begin{solution}{pb:g10:refraction:1}
\textbf{1.} $ v_{\text{water}} = \qty{3.00e8}{m/s}/1.33 =
\qty{2.26e8}{m/s}$; $ v_{\text{diamond}} = \qty{3.00e8}{m/s}/2.42 =
\qty{1.24e8}{m/s}$ --- हीरा कारक द्वारा प्रकाश को धीमा कर देता है $ n = 2.42$.

\textbf{2.} $\sin i_2 = \sin 40^\circ / 1.33 = 0.643/1.33 = 0.483$, इसलिए
$ i_2 = 28.9^\circ $.

\textbf{3.} $\sin i_2 = 1.33 \times \sin 28.9^\circ = 1.33 \times
0.483 = 0.643$, इसलिए $ i_2 = 40^\circ $: प्रकाश अपने पथ पर वापस लौटता है ---
प्रकाश किरणों की उत्क्रमणीयता.

\textbf{4.} $\sin i_c = 1/1.33 = 0.752$, इसलिए $ i_c = 48.8^\circ $. पर
$60^\circ > i_c $, \omterm{thm:g10:refraction:snell}{स्नेल का नियम} जरूरत पड़ेगी $\sin i_2 = 1.33 \sin
60^\circ = 1.15 > 1$: नहीं \omterm{def:g10:refraction:refraction}{अपवर्तित किरण}; सतह पूरी तरह से प्रतिबिंबित करती है
प्रकाश वापस नीचे.

\textbf{5.} $ v = \qty{3.00e8}{m/s}/1.0003 = \qty{2.999e8}{m/s}$: प्रकाश
वायु में निर्वात की तुलना में गति धीमी होती है $0.03\%$ की परिशुद्धता से बहुत नीचे
हमारे कोण माप, तो $n_{\text{air}} = 1.00$ हानिरहित है.

\textbf{6.} $ d' = \qty{3.0}{m}/1.33 = \qty{2.3}{m}$: पूल छिप जाता है
\qty{70}{cm} खुद का.

\textbf{7.} सिक्के से निकलने वाली दो किरणें सतह पर झुककर अपवर्तित हो जाती हैं
\emph{दूर} से सामान्य, इसलिए वे अधिक तेजी से ऊपर की ओर मुड़ते हैं
नीचे से पानी. आँख बाहर निकलने वाली दो किरणों को पीछे की ओर खींचती है
सीधी रेखाएँ; वे पार करते हैं \emph{ऊपर} सिक्का, और वह क्रॉसिंग है
जहां आंख छवि रखती है.

\textbf{8.} $ r = d \tan i_c = \qty{3.0}{m} \times \tan 48.8^\circ =
\qty{3.0}{m} \times 1.14 = \qty{3.4}{m}$: के बारे में एक डिस्क \qty{6.9}{m}
पार संपूर्ण आकाश धारण करता है।

\textbf{9.} पर $30^\circ $: $\sin i_2 = 1.33 \times 0.500 = 0.665$, द
किरण बाहर निकलती है $41.7^\circ $. पर $45^\circ $: $\sin i_2 = 1.33 \times
0.707 = 0.940$, यह बाहर निकलता है $70.1^\circ $, लगभग चरने वाला। पर
$50^\circ $: $1.33 \times 0.766 = 1.02 > 1$ --- \omterm{def:g10:refraction:critical}{कुल आंतरिक
प्रतिबिंब}, किरण वापस पूल में गोता लगाती है।

\textbf{10.} छवि असली मछली के ऊपर है
(\cref{prop:g10:refraction:depth}), तो लक्ष्य \emph{नीचे} यह। और हाँ:
मछली देखती है कि मछुआरा भी विस्थापित हो गया है, ऊर्ध्वाधर की ओर दबा हुआ है
स्नेल की खिड़की के अंदर --- प्रत्येक दूसरे को देखता है जहाँ \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} नहीं है।

\textbf{11.} हीरा: $\sin i_c = 1/2.42 = 0.413$, इसलिए
$ i_c = 24.4^\circ $, ख़िलाफ़ $41.1^\circ $ कांच के लिए: हीरे का पलायन
शंकु आधा भी चौड़ा नहीं है।

\textbf{12.} हीरे में, $30^\circ > 24.4^\circ $: \omterm{def:g10:refraction:critical}{कुल आंतरिक
प्रतिबिंब}, किरण पत्थर में रहती है। कांच में,
$30^\circ < 41.1^\circ $: किरण पीछे की ओर रिसती है। ए के पहलू
ब्रिलियंट कट कोणीय होते हैं ताकि प्रवेश करने वाला प्रकाश लगभग हमेशा टकराए
उनसे परे $24.4^\circ $ और तब तक उछलता रहता है जब तक वह ऊपर की ओर बाहर नहीं निकल जाता
आंख --- एक कांच की नकल, इसके विस्तृत भागने वाले शंकु के साथ, बस जाने देती है
प्रकाश गिरना.

\textbf{13.} $\sin 45^\circ = 0.707$. लाल: $0.707/2.41 = 0.293$, इसलिए
$ i_2 = 17.1^\circ $. नीला: $0.707/2.45 = 0.289$, इसलिए $ i_2 = 16.8^\circ $.
प्रवेश पर विभाजन: के बारे में $0.3^\circ $.

\textbf{14.} $\sin 17.0^\circ = 0.292$. लाल: $\sin i_2 = 2.41 \times
0.292 = 0.705$, इसलिए $ i_2 = 44.8^\circ $. नीला: $\sin i_2 = 2.45 \times
0.292 = 0.716$, इसलिए $ i_2 = 45.8^\circ $. पंखा अब है $1.0^\circ $
चौड़ा: निकास \omterm{def:g10:refraction:refraction}{अपवर्तन}, के पास काम कर रहा है \omterm{def:g10:refraction:critical}{महत्वपूर्ण कोण}, फैला हुआ
द $0.3^\circ $ प्रवेश तीन गुना विभाजित। रंगों की वह फैली हुई फुहार
हीरे का है \emph{आग}.

\textbf{15.} हीरा--पानी: $\sin i_c = 1.33/2.42 = 0.550$, इसलिए
$ i_c = 33.3^\circ $. $30^\circ $ किरण, पहलू का सामना होने पर फंस गई
हवा, अब बच जाती है ($30^\circ < 33.3^\circ $) जल फिल्म में: प्रत्येक
बूंद, फिंगरप्रिंट या ग्रीस की फिल्म बाहरी सूचकांक को बढ़ाती है, चौड़ी करती है
भागने का शंकु और चमक को खत्म कर देता है --- इसलिए जौहरी का कपड़ा।

\textbf{16.} $\sin i_c = \dfrac{1.46}{1.48} = 0.986$, इसलिए
$ i_c = 80.6^\circ $.

\textbf{17.} पूर्ण आंतरिक प्रतिबिंब ह ाेती है \emph{सतह पर},
इसलिए सतह बिल्कुल सही रहनी चाहिए: एक नंगे धागे पर, हर कण पर
धूल, खरोंच, पानी की बूंद या उंगली छूने से बाहरी सूचकांक ऊपर उठ जाता है
उस बिंदु पर और खून बहता है। \omterm{def:g10:refraction:fiber}{आवरण} दर्पण को दफना देता है
साफ कांच के अंदर जहां कोई भी चीज उस तक नहीं पहुंच सकती --- और, एक बोनस के रूप में,
पतले बालों की सुरक्षा करता है \omterm{def:g10:refraction:fiber}{मुख्य} यंत्रवत्।

\textbf{18.} $ v = \qty{3.00e8}{m/s}/1.48 = \qty{2.03e8}{m/s}$, इसलिए
\[
t = \frac{\qty{6.0e6}{m}}{\qty{2.03e8}{m/s}} = \qty{29.6}{ms} \approx
\qty{30}{ms}.
\]

\textbf{19.} पथ कारक $1/\sin i_c = 1.48/1.46 = 1.0137$: सबसे खराब
$\qty{6000}{km} \times 0.0137 \approx \qty{82}{km}$ अतिरिक्त ग्लास का,
लागत $\qty{8.2e4}{m} / \qty{2.03e8}{m/s} \approx \qty{0.4}{ms}$ ---
ज़िगज़ैग लगभग मुफ़्त है।

\textbf{20.} गुनगुनाहट: $2 \times (29.6 + 0.4) \approx \qty{60}{ms}$.
थ्रूपुट: $\dfrac{10^{13}}{2.4 \times 10^8} \approx 40\,000$ प्रतिलिपियाँ
इस पुस्तक का प्रति सेकंड. एक वाक्य में: प्रकाश को अंदर फंसाकर
कांच के माध्यम से पूर्ण आंतरिक प्रतिबिंब, \omterm{thm:g10:refraction:snell}{स्नेल का नियम} एक पुस्तकालय रखता है
हर सेकंड समुद्र के पार, साठ मिलीसेकंड इधर-उधर।
