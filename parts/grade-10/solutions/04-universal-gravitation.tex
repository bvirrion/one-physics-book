\section*{Chapter \ref{ch:g10:universal-gravitation} --- Universal Gravitation and Weight}

\begin{solution}{exo:g10:universal-gravitation:1}
\emph{1.} $F = \num{6.67e-11} \times \dfrac{60 \times 60}{0.80^2}
= \dfrac{\num{2.40e-7}}{0.64} \approx \qty{3.8e-7}{N}$.

\emph{2.} The mosquito's weight is
$P = \num{2.5e-6} \times 9.81 \approx \qty{2.5e-5}{N}$ --- about $65$
times \emph{larger} than the dancers' mutual attraction. Gravity
between people is weaker than a mosquito's weight.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:2}
\emph{1.} $P = mg = 55 \times 9.81 \approx \qty{5.4e2}{N}$.

\emph{2.} Her mass is unchanged: \qty{55}{kg} (same quantity of
matter). Her weight becomes
$P = 55 \times 1.62 \approx \qty{89}{N}$ --- six times less.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:3}
\emph{1.} $g_{\text{Mars}} = \dfrac{G M}{R^2}
= \dfrac{\num{6.67e-11} \times \num{6.42e23}}{(\num{3.39e6})^2}
= \dfrac{\num{4.28e13}}{\num{1.15e13}} \approx \qty{3.73}{N/kg}$.

\emph{2.} On Mars: $P = 900 \times 3.73 \approx \qty{3.4e3}{N}$. On
Earth: $P = 900 \times 9.81 \approx \qty{8.8e3}{N}$ --- the rover
weighs about $2.6$ times less on Mars, with the same mass.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:4}
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
= \frac{\num{2.93e37}}{\num{1.47e17}}
\approx \qty{2.0e20}{N}.
\]
The Moon exerts on the Earth a force of the \emph{same} magnitude,
\qty{2.0e20}{N}, in the opposite direction: gravitation is mutual.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:5}
\emph{1.} Distance doubled: $d^2$ is multiplied by $4$, so
$F = F_0/4$.

\emph{2.} Distance halved: $F = 4F_0$.

\emph{3.} Both masses doubled: $m_1 m_2$ is multiplied by $4$, so
$F = 4F_0$.

\emph{4.} Numerator $\times 3$, denominator $\times 9$:
$F = F_0/3$.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:6}
\emph{1.} $F = \num{6.67e-11} \times
\dfrac{(\num{5.0e8})^2}{100^2}
= \num{6.67e-11} \times \dfrac{\num{2.5e17}}{\num{1.0e4}}
\approx \qty{1.7e3}{N}$.

\emph{2.} $m = F/g = 1667/9.81 \approx \qty{1.7e2}{kg}$: the two ships
attract each other with the weight of a large motorcycle.

\emph{3.} $a = F/m = \num{1.7e3}/\num{5.0e8} \approx
\qty{3.3e-6}{m/s^2}$. Even unopposed, the tankers would take about
five days to pick up walking pace; in practice, water resistance and moorings
swamp the attraction completely.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:7}
\emph{1.} The beam balance indicates \qty{3.0}{kg} on both worlds: the
bag and the standard masses see their weights divided by the same
factor, so the comparison is unchanged. The spring scale reads the
weight: $3.0 \times 9.81 \approx \qty{29}{N}$ on Earth,
$3.0 \times 1.62 \approx \qty{4.9}{N}$ on the Moon.

\emph{2.} The balance: it measures mass, which is what the customer is
buying. A spring scale calibrated in ``kilograms'' on Earth would read
$4.9/9.81 \approx \qty{0.50}{kg}$ for the same bag on the Moon --- the
shopkeeper would give away six bags for the price of one.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:8}
\emph{1.} Distance to the center:
$d = R + h = \num{6.37e6} + \num{4.0e5} = \qty{6.77e6}{m}$. Then
$g = \dfrac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.77e6})^2}
= \dfrac{\num{3.98e14}}{\num{4.58e13}} \approx \qty{8.69}{N/kg}$.

\emph{2.} $8.69/9.81 \approx 89\%$ of the surface value.

\emph{3.} Gravity is nearly full strength up there: the astronauts
float because the station and everything in it are \emph{falling
together} around the Earth (free fall), not because gravity has
vanished.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:9}
From $g = GM/R^2$,
\[
M = \frac{g R^2}{G}
= \frac{8.87 \times (\num{6.05e6})^2}{\num{6.67e-11}}
= \frac{\num{3.25e14}}{\num{6.67e-11}}
\approx \qty{4.87e24}{kg},
\]
about $0.8$ times the Earth's mass --- Venus is nearly the Earth's
twin in size and mass.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:10}
\emph{1.} Sun:
$F = \dfrac{\num{6.67e-11} \times \num{1.99e30}}{(\num{1.496e11})^2}
\approx \qty{5.9e-3}{N}$. Moon:
$F = \dfrac{\num{6.67e-11} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{3.3e-5}{N}$.

\emph{2.} The Sun pulls about $\num{5.9e-3}/\num{3.3e-5} \approx 180$
times harder.

\emph{3.} The Earth's diameter is a much larger \emph{fraction} of the
Earth--Moon distance ($\approx 3\%$) than of the Earth--Sun distance
($\approx 0.01\%$), so the Moon's pull changes far more from the near
side to the far side of the globe. Tides feed on that difference ---
and there the Moon wins.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:11}
\emph{1.} $g' = \dfrac{G (2M)}{(2R)^2} = \dfrac{2}{4}\,\dfrac{GM}{R^2}
= \dfrac{g}{2} \approx \qty{4.9}{N/kg}$: the doubled radius wins over
the doubled mass.

\emph{2.} We need $\dfrac{GM}{R'^2} = 2\,\dfrac{GM}{R^2}$, so
$R'^2 = R^2/2$ and $R' = R/\sqrt{2} \approx \qty{4.5e6}{m}$ --- about
\qty{4500}{km}.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:12}
\emph{1.} Equal pulls per kilogram:
$\dfrac{G M_{\text{Earth}}}{x^2} = \dfrac{G M_{\text{Moon}}}{(D-x)^2}$.
Cross-multiplying and dividing by $G$:
$\left(\dfrac{x}{D-x}\right)^2 = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.

\emph{2.} $\dfrac{x}{D-x}
= \sqrt{\dfrac{\num{5.97e24}}{\num{7.35e22}}} = \sqrt{81.2} \approx
9.0$, so $x = 9.0\,(D - x)$, i.e.\ $x = \dfrac{9.0}{10.0}\,D
\approx 0.90 \times \num{3.84e8} \approx \qty{3.5e8}{m}$. The
balance point sits $90\%$ of the way to the Moon: the Earth's pull
dominates almost the whole trip.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:13}
\emph{1.} Expanding: $R^2 + 2Rh + h^2 = R^2 + x^2$; since $h$ is tiny
compared with $R$, drop $h^2$: $2Rh \approx x^2$, so
$h \approx \dfrac{x^2}{2R}
= \dfrac{(\num{8.0e3})^2}{2 \times \num{6.37e6}}
= \dfrac{\num{6.4e7}}{\num{1.27e7}} \approx \qty{5.0}{m}$.

\emph{2.} In one second the projectile falls \qty{4.9}{m} --- almost
exactly the \qty{5.0}{m} the ground drops away over \qty{8.0}{km}. So
at $v \approx \qty{8.0}{km/s}$ the fall never catches the ground: the
projectile orbits.

\emph{3.} $T = \dfrac{2\pi R}{v}
= \dfrac{2\pi \times \num{6.37e6}}{\num{8.0e3}}
\approx \qty{5.0e3}{s} \approx \qty{83}{min}$ --- close to the roughly
$\qty{90}{min}$ of real low-orbit satellites, which fly a little
higher.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:14}
\emph{1.} $g = \dfrac{\num{6.67e-11} \times \num{2.8e30}}
{(\num{1.2e4})^2} = \dfrac{\num{1.87e20}}{\num{1.44e8}}
\approx \qty{1.3e12}{N/kg}.$

\emph{2.} $P = 70 \times \num{1.3e12} \approx \qty{9.1e13}{N}$, versus
$\qty{6.9e2}{N}$ on Earth: about $\num{1.3e11}$ times more. No
structure made of atoms survives standing there.

\emph{3.} $P = \num{1.0e-6} \times \num{1.3e12} \approx
\qty{1.3e6}{N}$. On Earth that is the weight of
$m = \num{1.3e6}/9.81 \approx \qty{1.3e5}{kg}$ --- a grain of sand
weighing as much as a hundred-tonne locomotive.
\end{solution}

\begin{solution}{exo:g10:universal-gravitation:15}
\emph{1.} The distance is $60$ times larger, so the pull per kilogram
is $60^2 = 3600$ times weaker.

\emph{2.} The fall in one second scales the same way:
$4.9/3600 \approx \qty{1.4e-3}{m}$ --- about \qty{1.4}{mm}.

\emph{3.} $T = 27.3 \times \qty{86400}{s} = \qty{2.36e6}{s}$, so
$v = \dfrac{2\pi \times \num{3.84e8}}{\num{2.36e6}} \approx
\qty{1.02e3}{m/s}$, and $x = \qty{1.02e3}{m}$ in one second. The fall
is $h \approx \dfrac{x^2}{2d}
= \dfrac{(\num{1.02e3})^2}{2 \times \num{3.84e8}}
\approx \qty{1.4e-3}{m}$.

\emph{4.} The two numbers agree: the Moon falls toward the Earth each
second exactly as much as inverse-square gravity, calibrated on a
falling apple, predicts. Newton concluded that the force holding the
Moon \emph{is} the force that drops the apple --- gravity is
universal.
\end{solution}

\begin{solution}{pb:g10:universal-gravitation:1}
\textbf{1.} $F = \num{6.67e-11} \times
\dfrac{\num{7.3e6} \times 55}{100^2} \approx \qty{2.7e-6}{N}$.

\textbf{2.} The sand grain weighs
$\num{1.0e-6} \times 9.81 \approx \qty{9.8e-6}{N}$: the whole Eiffel
Tower pulls the tourist about $3.7$ times \emph{less} than the weight
of a grain of sand.

\textbf{3.} (a) Distance $\times 10$: force divided by $100$.
(b) Both masses $\times 1000$: force multiplied by $10^6$.
(c) Masses $\times 1000$ and distance $\times 1000$:
$10^6 / 10^6$ --- the force is unchanged.

\textbf{4.} $F = \dfrac{\num{6.67e-11} \times \num{5.97e24} \times
1.00}{(\num{6.37e6})^2} \approx \qty{9.81}{N}$: the weight of one
kilogram --- this is exactly the field strength
$g = \qty{9.81}{N/kg}$.

\textbf{5.} Gravity is feeble per kilogram but only ever
\emph{attracts}, so the pulls of all $\num{5.97e24}$ kilograms of
Earth add up in the same direction --- and an astronomical mass beats
a neighborly one by twenty-two powers of ten.

\textbf{6.} $F = \num{6.67e-11} \times \dfrac{158 \times 0.73}{0.23^2}
= \num{6.67e-11} \times \dfrac{115.3}{0.0529}
\approx \qty{1.5e-7}{N}$.

\textbf{7.} The small sphere weighs
$0.73 \times 9.81 \approx \qty{7.2}{N}$ --- about $\num{5e7}$ times
the attraction to be detected. No pan scale resolves one part in
fifty million; a fine torsion wire, which twists visibly under
nanonewton-scale torques, can.

\textbf{8.} $G = \dfrac{F d^2}{m_1 m_2}
= \dfrac{\num{1.47e-7} \times 0.0529}{115.3}
\approx \qty{6.74e-11}{N.m^2/kg^2}$ --- within about $1\%$ of the
modern \num{6.67e-11}, from a wooden shed in 1798.

\textbf{9.} From $g = G M / R^2$:
\[
M = \frac{g R^2}{G}
= \frac{9.81 \times (\num{6.37e6})^2}{\num{6.67e-11}}
\approx \qty{5.97e24}{kg}.
\]
Six million billion billion kilograms: the Earth, weighed.

\textbf{10.} $V = \frac43 \pi R^3 = \frac43 \pi
(\num{6.37e6})^3 \approx \qty{1.08e21}{m^3}$, so
$\rho = \num{5.97e24}/\num{1.08e21} \approx \qty{5.5e3}{kg/m^3}$ ---
twice the density of surface rock. The interior must hide something
much denser than granite: the Earth has a heavy (iron) core.

\textbf{11.} $g_{\text{Moon}} = \dfrac{\num{6.67e-11} \times
\num{7.35e22}}{(\num{1.74e6})^2} = \dfrac{\num{4.90e12}}
{\num{3.03e12}} \approx \qty{1.62}{N/kg}$.

\textbf{12.} Height $\propto 1/g$:
$h = 0.50 \times \dfrac{9.81}{1.62} \approx \qty{3.0}{m}$ --- a
standing jump over a basketball hoop, in slow motion.

\textbf{13.} $g_{\text{Mercury}} = \dfrac{\num{6.67e-11} \times
\num{3.30e23}}{(\num{2.44e6})^2} = \dfrac{\num{2.20e13}}
{\num{5.95e12}} \approx \qty{3.70}{N/kg}$ --- almost exactly Mars's
\qty{3.73}{N/kg}. Mercury has less mass, but its smaller radius (a
smaller $R^2$ below) compensates: it is a much denser world.

\textbf{14.} Half the field strength requires
$(R + h)^2 = 2 R^2$, so $R + h = R\sqrt{2}$ and
$h = (\sqrt{2} - 1) R \approx 0.414 \times \num{6.37e6}
\approx \qty{2.6e6}{m}$ --- about \qty{2600}{km} up.

\textbf{15.} $\dfrac{G M_{\text{Earth}}}{d^2} = 1.62$ gives
$d = \sqrt{\dfrac{\num{6.67e-11} \times \num{5.97e24}}{1.62}}
= \sqrt{\num{2.46e14}} \approx \qty{1.57e7}{m} \approx 2.5$ Earth
radii. From only one and a half radii above the ground, the Earth
pulls no harder than the Moon's surface does.

\textbf{16.} $v = \dfrac{2\pi d}{T}
= \dfrac{2\pi \times \num{1.496e11}}{\num{3.156e7}}
\approx \qty{2.98e4}{m/s}$ --- thirty kilometers every second.

\textbf{17.} $\dfrac{v^2}{d} = \dfrac{(\num{2.98e4})^2}{\num{1.496e11}}
\approx \qty{5.9e-3}{N/kg}$: the Sun pulls each kilogram of the Earth
with about six millinewtons.

\textbf{18.} $\dfrac{G M_{\text{Sun}}}{d^2} = \dfrac{v^2}{d}$ gives
\[
M_{\text{Sun}} = \frac{v^2 d}{G}
= \frac{\num{8.87e8} \times \num{1.496e11}}{\num{6.67e-11}}
\approx \qty{1.99e30}{kg}.
\]

\textbf{19.} $\dfrac{\num{1.99e30}}{\num{5.97e24}} \approx
\num{3.3e5}$: the Sun is a third of a million Earths.

\textbf{20.} Because the same $G$ governs every pair of masses,
measuring it once between two lead spheres in a shed let us put on the
scales first the planet under our feet, then the star we orbit ---
that is what \emph{universal} is worth.
\end{solution}
