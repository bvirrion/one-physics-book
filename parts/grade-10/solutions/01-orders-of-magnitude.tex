\section*{Chapter \ref{ch:g10:orders-of-magnitude} --- Orders of Magnitude: Measuring the Universe}

\begin{solution}{exo:g10:orders-of-magnitude:1}
\emph{1.} $\qty{750}{nm} = 750 \times \qty{e-9}{m} = \qty{7.50e-7}{m}$.
\emph{2.} $\qty{2.5}{km} = \qty{2.5e3}{m}$.
\emph{3.} $\qty{3.2}{\micro m} = \qty{3.2e-6}{m}$.
\emph{4.} $\qty{0.15}{Tm} = 0.15 \times \qty{e12}{m} = \qty{1.5e11}{m}$
--- one astronomical unit.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:2}
\emph{1.} $\qty{149600000}{km} = \qty{1.496e11}{m}$: $4$ significant
figures.
\emph{2.} $\qty{1.06e-10}{m}$: $3$ significant figures (leading zeros
never count).
\emph{3.} $\num{8.1e9}$: $2$ significant figures.
\emph{4.} $\qty{5.20e-3}{s}$: $3$ significant figures (the trailing
zero is a promise).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:3}
Mantissas and exponents separately
(\cref{prop:g10:orders-of-magnitude:powers}):
\[
3 \times 2 \times 10^{8-5} = \num{6e3}, \qquad
\tfrac{8}{4} \times 10^{4-(-3)} = \num{2e7},
\]
\[
5^{2} \times 10^{12} = \num{2.5e13}, \qquad
\tfrac{4 \times 3}{6} \times 10^{-3-9+7} = \num{2e-5} .
\]
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:4}
Human: $1.7 < 5$, order $\qty{e0}{m}$. Everest:
$\qty{8848}{m} = \qty{8.848e3}{m}$ and $8.848 \geq 5$: order
$\qty{e4}{m}$. Bacterium: order $\qty{e-6}{m}$. Earth:
$5.97 \geq 5$, order $\qty{e25}{kg}$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:5}
\emph{1.} Half a graduation: $L = (127.0 \pm 0.5)\,\unit{mm}$.
\emph{2.} $\Delta L / L = 0.5/127 \approx 0.4\%$.
\emph{3.} Eraser: $0.5/12 \approx 4\%$. The pencil measurement is
about ten times more precise --- same ruler, same absolute
uncertainty, ten times longer object.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:6}
\emph{1.} Proxima:
$4.24 \times \qty{9.47e15}{m} = \qty{4.01e16}{m}$. Sirius:
$8.6 \times \qty{9.47e15}{m} = \qty{8.1e16}{m}$ (two significant
figures, like the input).
\emph{2.} $t = d/c = \qty{2.4e13}{m} / \qty{3.00e8}{m/s}
= \qty{8.0e4}{s}$, i.e.\ $\num{8.0e4}/3600 \approx \qty{22}{h}$:
almost a full day for every message from the probe.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:7}
$t = d/c$ each time. Moon:
$\qty{3.84e8}{m}/\qty{3.00e8}{m/s} = \qty{1.28}{s}$. Sun:
$\qty{1.496e11}{m}/\qty{3.00e8}{m/s} = \qty{499}{s} \approx
\qty{8.3}{min}$. Neptune:
$\qty{4.50e12}{m}/\qty{3.00e8}{m/s} = \qty{1.50e4}{s} \approx
\qty{4.2}{h}$. When you look at Neptune, you see it as it was about
four hours ago --- every telescope is a mild time machine.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:8}
\emph{1.} One sheet: $\qty{52.0}{mm}/500 = \qty{0.104}{mm}
= \qty{1.04e-4}{m}$. The uncertainty divides too:
$\qty{0.5}{mm}/500 = \qty{0.001}{mm}$, so
$t = (104 \pm 1)\,\unit{\micro m}$.
\emph{2.} $1/104 \approx 1\%$.
\emph{3.} Measuring one sheet directly would give
$(0.1 \pm 0.5)\,\unit{mm}$: a relative uncertainty of about $500\%$
--- the ruler cannot even prove the sheet exists. Stacking $500$
sheets multiplies the measured length by $500$ while the ruler's
absolute uncertainty stays fixed: the relative uncertainty shrinks by
that same factor $500$.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:9}
Divide each length by $\qty{2e-10}{m}$. Bacterium:
$\qty{2.0e-6}{m}/\qty{2e-10}{m} = \num{1.0e4}$ atoms. Hair:
$\qty{7.0e-5}{m}/\qty{2e-10}{m} = \num{3.5e5}$. Human:
$1.7/\num{2e-10} = \num{8.5e9}$, order of magnitude $10^{10}$: ten
billion atoms head to toe.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:10}
\emph{1.} $s = \qty{0.22}{m}/\qty{1.39e9}{m} = \num{1.58e-10}$.
\emph{2.} Earth:
$\qty{1.27e7}{m} \times \num{1.58e-10} = \qty{2.0e-3}{m} =
\qty{2.0}{mm}$ --- a peppercorn --- orbiting at
$\qty{1.496e11}{m} \times \num{1.58e-10} = \qty{23.7}{m} \approx
\qty{24}{m}$ from the football.
\emph{3.} Moon:
$\qty{3.84e8}{m} \times \num{1.58e-10} = \qty{6.1}{cm}$ from the
peppercorn. Neptune:
$\qty{4.50e12}{m} \times \num{1.58e-10} \approx \qty{710}{m}$: the
model does not fit in the schoolyard.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:11}
\emph{1.} $21.0 \times 29.7 = \qty{623.7}{cm^2}$, rounded to the three
significant figures of the inputs: $\qty{624}{cm^2}$.
\emph{2.} $v = \qty{100.0}{m}/\qty{12.3}{s} = \qty{8.13}{m/s}$ (three
figures, set by the time).
\emph{3.} Five significant figures would claim the area to
$\qty{0.01}{cm^2}$, a precision the millimetre-level inputs never
delivered --- a forged decimal is still a forgery.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:12}
\emph{1.} Radius $r = \qty{0.55}{m}$, area
$A = \pi r^{2} = \pi \times 0.55^{2} \approx \qty{0.95}{m^2}$. Volume
$V = \qty{1.0}{mm^3} = \qty{1.0e-9}{m^3}$, and a film is (area)
$\times$ (thickness), so
\[
e = \frac{V}{A} = \frac{\qty{1.0e-9}{m^3}}{\qty{0.95}{m^2}}
= \qty{1.1e-9}{m} .
\]
\emph{2.} A molecule is of order $\qty{e-9}{m}$ --- one rung above the
atom's $\qty{e-10}{m}$, consistent with a molecule being a small
cluster of atoms. Franklin's teaspoon of oil calmed half an acre of
pond and, without his knowing it, measured the nanometre.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:13}
Assumptions: about $70$ beats per minute at rest, a life of $80$
years. Per day: $70 \times 60 \times 24 \approx \num{1e5}$ beats. Per
year: $\num{1e5} \times 365 \approx \num{4e7}$. Per life:
$\num{4e7} \times 80 \approx \num{3e9}$ --- order of magnitude
$10^{9}$, a few billion heartbeats. Four significant figures would
pretend to know a heart rate, minute by minute, for eighty years: the
inputs are order-of-magnitude guesses, and no output can outrank its
inputs (\cref{met:g10:orders-of-magnitude:sigfig}).
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:14}
After $n$ folds the thickness is $2^{n} \times \qty{1.0e-4}{m}$.
\emph{1.} Need $2^{n} \geq 8848/\num{1.0e-4} = \num{8.8e7}$. Since
$2^{26} \approx \num{6.7e7}$ falls short and
$2^{27} \approx \num{1.3e8}$ suffices: $27$ folds (stack
$\approx \qty{13}{km}$).
\emph{2.} Need $2^{n} \geq \num{3.84e12}$: $2^{41} \approx
\num{2.2e12}$ is short, $2^{42} \approx \num{4.4e12}$ reaches ---
$42$ folds, stack $\qty{4.4e8}{m}$, comfortably past the Moon.
\emph{3.} Only the experiment: each fold halves the area, and after
seven folds the stack is thicker than it is wide. The arithmetic of
doubling is untouched --- exponential growth simply outruns what paper
(and intuition) can follow.
\end{solution}

\begin{solution}{exo:g10:orders-of-magnitude:15}
\emph{1.} Moon: $\qty{0.03}{m}/\qty{3.84e8}{m} = \num{7.8e-11}$.
Table: $\qty{0.001}{m}/\qty{1.000}{m} = \num{1.0e-3}$. The Moon
measurement is more precise by a factor
$\num{1.0e-3}/\num{7.8e-11} \approx \num{1.3e7}$ --- over ten million.
\emph{2.} Matching $\num{7.8e-11}$ on a one-metre table means
$\Delta x = \qty{7.8e-11}{m}$, smaller than one atom
($\approx \qty{e-10}{m}$): the table's edge is not even
\emph{defined} that sharply. Relatively speaking, the Earth--Moon
laser distance is one of the finest measurements our species owns.
\end{solution}

\begin{solution}{pb:g10:orders-of-magnitude:1}
\textbf{1.} Grain: $\qty{0.5}{mm} = \qty{5e-4}{m}$, on the milli
scale; atom: $\qty{2.5e-10}{m} = \qty{0.25}{nm}$, on the nano scale.

\textbf{2.} $\qty{5e-4}{m}/\qty{2.5e-10}{m} = \num{2.0e6}$: two
million atoms across.

\textbf{3.} A cube of $2 \times 10^{6}$ atoms per edge holds
$(2\times10^{6})^{3} = 8 \times 10^{18}$ atoms: order of magnitude
$10^{19}$.

\textbf{4.} $V = (\qty{5e-4}{m})^{3} = \qty{1.25e-10}{m^3}$, so
$m = \rho V = \qty{2500}{kg/m^3} \times \qty{1.25e-10}{m^3}
= \qty{3.1e-7}{kg} = \qty{0.31}{mg}$.

\textbf{5.} Mass per atom:
$\qty{3.1e-7}{kg}/(8\times10^{18}) = \qty{3.9e-26}{kg}$ --- squarely
in the ``few $10^{-26}\,\unit{kg}$'' range. The naive cube of atoms
reproduces real atomic masses: the model passes the sanity check
(\cref{met:g10:orders-of-magnitude:sanity}) with honors.

\textbf{6.} $s = \qty{5e-4}{m}/\qty{1.39e9}{m} = \num{3.6e-13}$.

\textbf{7.} Model Earth: diameter
$\qty{1.27e7}{m} \times \num{3.6e-13} = \qty{4.6e-6}{m}$ --- the size
of a bacterium --- at distance
$\qty{1.496e11}{m} \times \num{3.6e-13} = \qty{5.4}{cm}$ from the
grain: the Earth is a microbe five centimetres from a grain of sand.

\textbf{8.} $\qty{4.5e12}{m} \times \num{3.6e-13} = \qty{1.6}{m}$: the
entire planetary system orbits within a dining table.

\textbf{9.} $\qty{4.0e16}{m} \times \num{3.6e-13} = \qty{1.4e4}{m}
= \qty{14}{km}$.

\textbf{10.} The nearest star to our table-top Solar System is another
grain of sand fourteen kilometres away, with essentially nothing in
between: the galaxy is overwhelmingly empty space, which is why
collisions between stars are almost unheard of.

\textbf{11.} $t = \qty{5e-4}{m}/\qty{3.00e8}{m/s} = \qty{1.7e-12}{s}$
--- under two picoseconds.

\textbf{12.} Moon: $\qty{3.84e8}{m}/\qty{3.00e8}{m/s} =
\qty{1.28}{s}$. Sun: $\qty{1.496e11}{m}/\qty{3.00e8}{m/s} =
\qty{499}{s} \approx \qty{8.3}{min}$: sunlight is always eight minutes
old.

\textbf{13.} One year is $365.25 \times 86400 = \qty{3.156e7}{s}$, so
$\qty{1}{ly} = \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}$; then
$\qty{4.0e16}{m}/\qty{9.47e15}{m} = 4.2$: the data distance is indeed
about $\qty{4.2}{ly}$.

\textbf{14.} $\qty{2.4e22}{m}/\qty{9.47e15}{m} = \num{2.5e6}$: two and
a half million light-years. That light left Andromeda about $2.5$
million years ago, when the first members of the genus \emph{Homo}
were learning to knap stones --- and no \emph{Homo sapiens} yet
existed anywhere.

\textbf{15.} $\qty{3.00e8}{m/s} \times \qty{1e-9}{s} = \qty{0.30}{m}$.
In a third of a nanosecond a signal covers at most about
$\qty{10}{cm}$, so any two parts of the machine that must talk within
one clock tick have to sit within ten centimetres of each other:
speed caps size.

\textbf{16.} Average atom in water:
$\qty{3.0e-26}{kg}/3 = \qty{1.0e-26}{kg}$. Number of atoms:
$\qty{70}{kg}/\qty{1.0e-26}{kg} = \num{7e27}$.

\textbf{17.} Against the grain:
$\num{7e27}/\num{8e18} \approx 10^{9}$ --- you contain a billion
grains' worth of atoms. Against the stars:
$\num{7e27}/\num{2e23} = \num{3.5e4}$ --- your atoms outnumber all the
stars of the observable universe thirty-five thousand to one.

\textbf{18.} $\text{(you)}/\text{(atom)} =
\qty{1.7}{m}/\qty{2.5e-10}{m} \approx \num{7e9} \approx 10^{10}$,
while $\text{(universe)}/\text{(you)} =
\qty{8.8e26}{m}/\qty{1.7}{m} \approx \num{5e26}$. On the ladder you
stand about ten rungs above the atom and nearly twenty-seven below the
universe: multiplicatively, humans live very close to their atoms.

\textbf{19.} The condition gives
$x^{2} = 10^{-15} \times 10^{26.9} = 10^{11.9}$, so
$x = 10^{5.95} \approx \qty{9e5}{m} \approx \qty{900}{km}$ --- almost
exactly the diameter of Ceres, the largest asteroid
($\approx \qty{940}{km}$). Halfway up the ladder between a nucleus and
the cosmos sits a modest dwarf planet.

\textbf{20.} $\qty{8.8e26}{m} \times \num{3.6e-13} = \qty{3.2e14}{m}$,
i.e.\ $\qty{3.2e14}{m}/\qty{1.496e11}{m} \approx \num{2.1e3}$
astronomical units. Punchline: shrink the cosmos until the Sun is a
grain of sand, and the observable universe still spans two thousand
times the real Earth--Sun distance --- no scale model brings the
universe down to human size; only powers of ten can carry it.
\end{solution}
