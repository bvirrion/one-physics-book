\chapter{Pressão: Do Esporte ao Mergulho}\label{ch:g10:pressure}

Um esquiador desliza sobre a neve onde um caminhante afunda; um \qty{5}{N} empurrar
enfia uma agulha no couro que um punho não consegue amassar. Mesmo forças,
efeitos totalmente diferentes: a quantidade que falta é \emph{\omterm{def:g10:pressure:pressure}{pressão}},
\omterm{def:g10:inertia:force}{vigor} dividido pela área que o recebe. Este capítulo o define,
segue-o por uma piscina, por um barômetro, até uma seringa espremida ---
e termina quarenta metros no fundo do mar, onde se torna vital.

\section{Pressão: força espalhada por uma área}

\begin{definition}[Pressão]\label{def:g10:pressure:pressure}
Quando um \omterm{def:g10:inertia:force}{vigor} de magnitude $F $ (em \unit{N}) pressiona perpendicularmente um
superfície da área $ S $ (em \unit{m^2}), o \emph{pressão}\index{pressão}
exercido naquela superfície é
\[
P = \frac{F}{S}.
\]
Isso é \omterm{def:g10:orders-of-magnitude:unit}{unidade}, o \unit{N/m^2}, é o \emph{Pascal}\index{Pascal}
(\unit{Pa}) --- um minúsculo \omterm{def:g10:orders-of-magnitude:unit}{unidade}, uma maçã \omterm{def:g10:universal-gravitation:weight}{peso} espalhado por um quadrado
\omterm{def:g10:orders-of-magnitude:unit}{medidor}, então \unit{kPa} e \unit{MPa} são os trabalhadores \omterm{def:g10:orders-of-magnitude:unit}{unidades}.
\end{definition}

\begin{example}[Salto versus sapato de neve]\label{ex:g10:pressure:snowshoe}
UM \qty{60}{kg} pessoa pesa $ F = 60 \times 9.81 \approx \qty{590}{N}$.
Em um salto agulha de \qty{1.0}{cm^2} = \qty{1.0e-4}{m^2}:
$ P = 590/\num{1.0e-4} \approx \qty{5.9e6}{Pa} = \qty{5.9}{MPa}$. Em dois
raquetes de neve de \qty{0.20}{m^2} cada:
$ P = 590/0.40 \approx \qty{1.5}{kPa}$. O \emph{mesmo} \omterm{def:g10:universal-gravitation:weight}{peso} prensas
$4000$ vezes mais difícil sob o calcanhar: saltos marcam pisos de madeira,
raquetes de neve flutuam na neve em pó.
\end{example}

\begin{remark}[Concentrar ou espalhar]\label{rem:g10:pressure:tools}
Uma agulha de costura empurrada com \qty{5}{N} em uma ponta de \qty{0.01}{mm^2} =
\qty{1e-8}{m^2} exerce $\qty{5e8}{Pa} = \qty{500}{MPa}$ --- o suficiente para
parte fibras de couro. Concentrado de ferramentas de perfuração \omterm{def:g10:inertia:force}{vigor}; suporte de carga
projetos (esquis, lagartas, fundações) o espalham. O esporte é
aplicado \omterm{def:g10:pressure:pressure}{pressão} gestão: os patins mordem o gelo, os esquis deslizam na neve.
\end{remark}

\section{Pressão em um líquido em repouso}

Por que a água pressiona com mais força para baixo? Isole uma \omterm{def:g10:universal-gravitation:weight}{vertical} imaginária
coluna de água de seção transversal $ S $, da superfície à profundidade $ h $:
volume $ Sh $, massa $\rho S h $ ($\rho $ a densidade do líquido, em
\unit{kg/m^3}), \omterm{def:g10:universal-gravitation:weight}{peso} $\rho S h g $. A parte inferior da coluna suporta
isso \omterm{def:g10:universal-gravitation:weight}{peso} \emph{mais} o \omterm{def:g10:inertia:inventory}{impulso} da atmosfera $ P_0 S $ no topo.
Dividindo o total \omterm{def:g10:inertia:force}{vigor} por $ S $: o \omterm{def:g10:pressure:pressure}{pressão} em profundidade $ h $ deveria ser
$ P_0 + \rho g h $ --- e está, em todo lugar no líquido.

\begin{theorem}[Pressão em profundidade]\label{thm:g10:pressure:depth}
Em um líquido de densidade $\rho $ \omterm{def:g10:relative-motion:frame}{em repouso}, aberto ao ar, o \omterm{def:g10:pressure:pressure}{pressão} em
profundidade $ h $ abaixo da superfície está
\[
P = P_0 + \rho g h,
\]
onde $ P_0$ é o \omterm{def:g10:pressure:atmospheric}{pressão atmosférica} na superfície. O \omterm{def:g10:pressure:pressure}{pressão}
depende apenas da profundidade, não da forma do recipiente, e em um
dado ponto, o líquido pressiona igualmente com força em todas as direções.
\end{theorem}
\admitted

\begin{remark}
O argumento da coluna apenas torna a lei plausível; a derivação honesta
(qualquer formato, todas as direções) está no volume do Ano 1. A própria água
prazo $\rho g h $ é o \emph{medidor}\index{medidor} \omterm{def:g10:pressure:pressure}{pressão} --- que mergulhador
\omterm{def:g10:pressure:pressure}{pressão} \omterm{def:g10:orders-of-magnitude:unit}{medidor} exibe.
\end{remark}

\begin{example}[Dez metros de água]\label{ex:g10:pressure:tenmetres}
Em água doce ($\rho = \qty{1000}{kg/m^3}$), no $ h = \qty{10}{m}$:
$\rho g h = 1000 \times 9.81 \times 10 \approx \qty{98}{kPa}$, então
$ P = 101 + 98 \approx \qty{199}{kPa}$. Dez metros de água adicione quase
exatamente uma atmosfera: a regra prática do mergulhador, \emph{um extra
atmosfera por dez metros}.
\end{example}

\section{O oceano de ar}

\begin{definition}[Pressão atmosférica]\label{def:g10:pressure:atmospheric}
Vivemos no fundo de um oceano de ar, cujo \omterm{def:g10:universal-gravitation:weight}{peso} cria o
\emph{pressão atmosférica}\index{pressão atmosférica}, ao nível do mar
\[
P_0 \approx \qty{101}{kPa} = \qty{1.01e5}{Pa}.
\]
Em cada centímetro quadrado isso é $ F = \num{1.01e5} \times \num{1e-4}
\approx \qty{10}{N}$: o \omterm{def:g10:universal-gravitation:weight}{peso} de um um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} massa em cada
\unit{cm^2} da sua pele, da sua mesa, de tudo.
\end{definition}

\begin{example}[A carga invisível]\label{ex:g10:pressure:sheet}
Uma folha de papel A4 (\qty{21.0}{cm} $\times $ \qty{29.7}{cm}, então
$ S \approx \qty{0.0624}{m^2}$) recebe do ar acima dele
$ F = \num{1.01e5} \times 0.0624 \approx \qty{6.3e3}{N}$ --- o \omterm{def:g10:universal-gravitation:weight}{peso}
de um carro pequeno. Nada rasga: o ar abaixo empurra
\emph{acima} tão difícil. Notamos a atmosfera apenas quando um lado
perde --- uma ventosa, um canudo, uma cabine de avião.
\end{example}

\begin{remark}[Barômetro de Torricelli]\label{rem:g10:pressure:torricelli}
Em 1643, Torricelli inverteu um tubo selado de mercúrio sobre um recipiente de mercúrio.
banho. A coluna caiu à altura em que $\rho g h = P_0$:
$ h = \num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$, saindo
acima dele, o primeiro vácuo já feito. A altura acompanha o clima e
diminui com a altitude: um barômetro pesa o ar acima de você (água
precisaria \qty{10.3}{m}: \cref{exo:g10:pressure:9}).
\end{remark}

\section{Gases: bombardeio e lei de Boyle}

Um gás também pressiona o seu recipiente: é um enxame de moléculas em
movimento aleatório incessante, cada colisão dá à parede uma pequena
\omterm{def:g10:inertia:inventory}{impulso}, e bilhões de bilhões de impactos por \omterm{def:g10:orders-of-magnitude:unit}{segundo} se confundem no
constante \omterm{def:g10:inertia:force}{vigor} medimos como gás \omterm{def:g10:pressure:pressure}{pressão}. Esprema o gás até metade
volume: as moléculas atingem o dobro da frequência, então o \omterm{def:g10:pressure:pressure}{pressão} deveria
duplo. Isso acontece.

\begin{proposition}[Lei de Boyle]\label{prop:g10:pressure:boyle}
\index{Lei de Boyle}
Para uma quantidade fixa de gás mantida a temperatura constante, \omterm{def:g10:pressure:pressure}{pressão} e
volume são inversamente proporcionais:
\[
P \, V = \text{constant},
\qquad\text{i.e.}\qquad
P_1 V_1 = P_2 V_2 .
\]
\end{proposition}
\admitted

\begin{remark}
Um fato experimental neste nível --- o volume do Ano 1 deriva
da imagem da colisão. O experimento (uma seringa selada de ar,
um \omterm{def:g10:pressure:pressure}{pressão} sensor, \qty{20}{\degreeCelsius}):
\begin{center}
\begin{tabular}{lccccc}
$ V $ (\unit{mL}) & 60 & 50 & 40 & 30 & 20 \\
$ P $ (\unit{kPa}) & 100 & 120 & 150 & 200 & 300 \\
$ P \times V $ (\unit{kPa.mL}) & 6000 & 6000 & 6000 & 6000 & 6000
\end{tabular}
\end{center}
\end{remark}

\begin{example}[Uma seringa espremida]\label{ex:g10:pressure:boyle}
O ar ocupa \qty{4.0}{L} no \qty{100}{kPa}; comprimido lentamente (para que o
a temperatura permanece constante) para \qty{250}{kPa}, ocupa
$ V_2 = P_1 V_1 / P_2 = 400/250 = \qty{1.6}{L}$.
\end{example}

\section{Mergulho: pressão colocada em ação}

\begin{method}[Pressão absoluta em profundidade]\label{met:g10:pressure:dive}
Para um mergulhador em profundidade $ h $ no mar ($\rho \approx \qty{1025}{kg/m^3}$,
então $\rho g h \approx \qty{101}{kPa}$ por \qty{10}{m}), calcular
$ P = P_0 + \rho g h $ --- e em Lei de Boyle sempre use isso
\emph{absoluto}\index{absoluto} \omterm{def:g10:pressure:pressure}{pressão}, nunca o \omterm{def:g10:orders-of-magnitude:unit}{medidor} termo sozinho. Um mergulhador senta
sob $2$ atmosferas em \qty{10}{m}, $3$ no \qty{20}{m}, $5$ no
\qty{40}{m}.
\end{method}

\begin{remark}[Nunca prenda a respiração]\label{rem:g10:pressure:breath}
Um regulador fornece ar à temperatura ambiente \omterm{def:g10:pressure:pressure}{pressão}: os pulmões enchem normalmente em
qualquer profundidade. Mas preencha \qty{6.0}{L} no \qty{20}{m} ($\qty{302}{kPa}$),
prenda a respiração e venha à tona: Boyle exige
$6.0 \times 302/101 \approx \qty{18}{L}$ --- três vezes que baú
segura, e os pulmões rasgam bem antes disso, mesmo em alguns metros de subida.
Daí a primeira regra do mergulho autônomo: \emph{nunca prenda a respiração enquanto sobe}
--- continue respirando e o ar em expansão simplesmente fluirá. Um mergulhador livre
é seguro: o ar dela foi retirado da superfície, então na subida só
se expande novamente para seu volume original.
\end{remark}

\begin{remark}[Descompressão]\label{rem:g10:pressure:deco}
Respirando de alta \omterm{def:g10:pressure:pressure}{pressão} o ar também dissolve o nitrogênio extra no sangue,
como gás em uma garrafa de refrigerante tampada; superfície muito rápido e efervesce
bolhas dentro do corpo --- doença descompressiva. Daí a lentidão
subida (cerca de \qty{10}{m} por minuto) e paradas superficiais, que permitem
o gás sai silenciosamente pelos pulmões. A teoria espera pelo Ano 1
volume; a regra de conduta pertence a todo mergulhador.
\end{remark}

\begin{exercise}[$\star\star\star $]\label{exo:g10:pressure:15}

  \item Para levantar o carro \qty{2.0}{cm}, até que ponto o pequeno pistão deve

\end{enumerate}
\end{exercise}

\section{Problema: Uma respiração para baixo, uma respiração para cima}

\begin{problem}[{Problema de fim de semana --- uma respiração para baixo, uma respiração para cima:
a física de um mergulho livre e uma subida de mergulho autônomo, desde o aperto da máscara até o
regra que os últimos dez metros são os mais perigosos}]
\label{pb:g10:pressure:1}
Lena faz mergulho livre: uma respiração superficial, até \qty{30}{m} e de volta.
Marco mergulha ao lado dela, respirando em um tanque. Mesmo mar
($\rho = \qty{1025}{kg/m^3}$, $ g = \qty{9.81}{N/kg}$,
$ P_0 = \qty{101}{kPa}$), mesmo \qty{6.0}{L} pulmões, perigos opostos ---
todos governados por $ P = P_0 + \rho g h $ e $ PV $ = constante (temperatura
constante por toda parte).

\medskip
\noindent\textbf{Part I --- The \omterm{def:g10:universal-gravitation:weight}{peso} of water.}
\begin{enumerate}
  \item Calcular $\rho g $ para água do mar em \unit{kPa} por medidore
        verifique a regra prática: uma atmosfera por \qty{10}{m}.
  \item Calcule o \omterm{def:g10:pressure:pressure}{pressão} absoluta no \qty{10}{m}, \qty{20}{m},
        \qty{30}{m} e \qty{40}{m}.
  \item Expresse cada um como um múltiplo da superfície \omterm{def:g10:pressure:pressure}{pressão}.
  \item A máscara de Lena cobre \qty{150}{cm^2}. No \qty{30}{m}, se o ar
        dentro dele ainda estava na superfície \omterm{def:g10:pressure:pressure}{pressão}, que rede \omterm{def:g10:inertia:force}{vigor} faria
        a água exerce? (Os mergulhadores expiram na máscara pelo nariz
        para evitar isso \emph{aperto de máscara}.)
  \item Um tímpano tem área de cerca de \qty{0.60}{cm^2}. Calcule a rede
        \omterm{def:g10:inertia:force}{vigor} nisso em \qty{3.0}{m} se o ouvido médio permanecer na superfície
        \omterm{def:g10:pressure:pressure}{pressão}; por que os ouvidos doem em uma mera piscina, e o que deve
        ``equalização''alcançar?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- One breath down.}
Lena sai da superfície com \qty{6.0}{L} de ar em seus pulmões.
\begin{enumerate}[resume]
  \item Usando Lei de Boyle com pressões absolutas, calcule seu pulmão
        volume em \qty{10}{m}, \qty{20}{m} e \qty{30}{m}.
  \item A que profundidade o volume pulmonar dela é reduzido pela metade?
  \item O volume pulmonar é proporcional à profundidade? Descreva a curva de
        $ V $ contra \omterm{def:g10:pressure:pressure}{pressão} absoluta $ P$.
  \item Seu volume residual é \qty{1.5}{L}: os pulmões não podem encolher
        mais. Mostre que esse limite é atingido próximo \qty{30}{m}.
  \item Mesmo assim, os recordes de mergulho livre excedem \qty{100}{m}: o que
        preenche o volume que falta? (O que mais pode fluir para o peito?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- One breath up.}
No \qty{30}{m}, o regulador de Marco preenche seu \qty{6.0}{L} pulmões.
\begin{enumerate}[resume]
  \item Em que \omterm{def:g10:pressure:pressure}{pressão} o regulador fornece esse ar? Como
        muitas vezes mais denso que o ar da superfície?
  \item Marco entra em pânico, prende a respiração e sobe para \qty{10}{m}:
        que volume seu ar aprisionado exige ali?
  \item Qual volume na superfície --- quantas vezes sua capacidade?
  \item Enuncie a primeira regra do mergulhador e por que respirar
        normalmente elimina o perigo.
  \item Lena também prende a respiração \qty{30}{m} para a superfície,
        no entanto, seus pulmões estão perfeitamente seguros. Explique a assimetria.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Bubbles, and the last ten metros.}
\begin{enumerate}[resume]
  \item Marco libera um \qty{1.0}{cm^3} bolha em \qty{40}{m}.
        Calcule seu volume em \qty{30}{m}, \qty{20}{m}, \qty{10}{m}
        e logo abaixo da superfície.
  \item Para cada \qty{10}{m} estágio de ascensão, calcule o fator por
        qual a bolha cresce. Onde está o crescimento mais rápido?
  \item Os mergulhadores sobem no máximo \qty{10}{m} por minuto: quanto tempo
        \qty{40}{m}, e em que minuto qualquer líquido preso ou dissolvido
        o gás se expande mais?
  \item Nas profundezas, o sangue de Marco dissolve nitrogênio extra, como gás em
        uma garrafa de refrigerante tampada. O que faz uma subida muito rápida e como
        faça a subida lenta e um \qty{5}{m} parada de segurança evitá-lo?
  \item Punchline: em uma frase cada, dê ao mergulhador livre
        veredicto, o veredicto do mergulhador e a razão quantificada
        os últimos dez metros de qualquer ascensão merecem o maior respeito.
\end{enumerate}
\end{problem}
