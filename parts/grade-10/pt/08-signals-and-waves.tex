\chapter{Sinais e Ondas}\label{ch:g10:signals-and-waves}

Cada mensagem que você já recebeu chegou como um \omterm{def:g10:signals-and-waves:signal}{sinal}: uma voz pressionada
em seu tímpano, esta página chegou até você como um piscar de rádio. Atrás do
variedade fica em uma ideia --- uma \omterm{def:g10:orders-of-magnitude:unit}{quantidade física} variando no tempo --- e um
kit de ferramentas: \omterm{def:g10:signals-and-waves:period}{período}, freqüência, amplitudee a arte de cronometrar um atraso.
Com ele, um barco mede o fundo do mar e um médico conta os batimentos cardíacos.

\section{Sinais transportam informações}

\begin{definition}[Sinal]\label{def:g10:signals-and-waves:signal}
UM \emph{sinal}\index{sinal} é um \omterm{def:g10:orders-of-magnitude:unit}{quantidade física} isso varia no tempo
para transportar informações: o ar \omterm{def:g10:pressure:pressure}{pressão} no seu tímpano, a voltagem em um
terminais do microfone, o brilho da luz em um \omterm{def:g10:refraction:fiber}{fibra óptica}.
Para estudar um sinal, registre-o em função do tempo e leia o
informações fora da curva.
\end{definition}

Em uma ligação, a mesma informação muda de operadora repetidamente:
\omterm{def:g10:pressure:pressure}{pressão} no ar, depois voltagem no microfone, depois rádio no
antena --- e para trás na outra extremidade. Um vocabulário lida com eles
tudo; este capítulo o constrói.

\section{Sinais periódicos}

\begin{definition}[Sinal periódico, ponto final]\label{def:g10:signals-and-waves:period}
UM \omterm{def:g10:signals-and-waves:signal}{sinal} é \emph{periódico}\index{periódico} quando ele se repetir
de forma idêntica em intervalos regulares. O \emph{período}\index{período} $T $ é
a duração de um ciclo completo, em \omterm{def:g10:orders-of-magnitude:unit}{segundos}.
\end{definition}

\begin{definition}[Freqüência]\label{def:g10:signals-and-waves:frequency}
O \emph{frequência}\index{frequência} de um \omterm{def:g10:signals-and-waves:signal}{sinal} \omterm{def:g10:signals-and-waves:period}{periódico} é o número
de ciclos por \omterm{def:g10:orders-of-magnitude:unit}{segundo}:
\[
f = \frac{1}{T} .
\]
Isso é \omterm{def:g10:orders-of-magnitude:unit}{unidade}, um ciclo por \omterm{def:g10:orders-of-magnitude:unit}{segundo}, é o \emph{hertz}\index{hertz}
(\unit{Hz}), com os múltiplos usuais \unit{kHz}, \unit{MHz}, \unit{GHz}.
\end{definition}

\begin{definition}[Amplitude]\label{def:g10:signals-and-waves:amplitude}
O \emph{amplitude}\index{amplitude} de um \omterm{def:g10:signals-and-waves:signal}{sinal} \omterm{def:g10:signals-and-waves:period}{periódico} é o seu máximo
desvio do valor de repouso. Para uma \omterm{def:g10:inertia:inventory}{tensão} oscilando simetricamente
cerca de zero, a amplitude é o valor de pico; medidas da crista à calha
\emph{duas vezes} a amplitude (o valor pico a pico).
\end{definition}

\begin{example}[Rede elétrica e campo de concerto]\label{ex:g10:signals-and-waves:mains}
A \omterm{def:g10:inertia:inventory}{tensão} da rede oscila em $ f = \qty{50}{Hz}$: isso é \omterm{def:g10:signals-and-waves:period}{período} é
$ T = 1/f = \qty{0.020}{s} = \qty{20}{ms}$. O concerto A de uma orquestra é um
\omterm{def:g10:pressure:pressure}{pressão} \omterm{def:g10:signals-and-waves:signal}{sinal} no \qty{440}{Hz}: dura um ciclo
$ T = 1/440 = \qty{2.3}{ms}$. Mais alto freqüência, ciclo mais curto: $ f $ e $ T $
são inversos um do outro.
\end{example}

\section{Lendo um oscilograma}

\begin{definition}[Osciloscópio]\label{def:g10:signals-and-waves:oscilloscope}
Um \emph{osciloscópio}\index{osciloscópio} traça uma \omterm{def:g10:inertia:inventory}{tensão} em função do tempo;
a curva em sua tela gradeada é um \emph{oscilograma}%
\index{oscilograma}. Duas configurações convertem divisões de tela em físicas
valores: o \emph{base de tempo}\index{base de tempo} (\omterm{def:g10:orders-of-magnitude:unit}{segundos} por horizontal
divisão) e a \omterm{def:g10:universal-gravitation:weight}{vertical} \emph{ganho}\index{ganho} (volts por \omterm{def:g10:universal-gravitation:weight}{vertical}
divisão).
\end{definition}

\begin{method}[Da tela aos números]\label{met:g10:signals-and-waves:oscillogram}
\begin{enumerate}
  \item Conte as divisões horizontais abrangidas por um ciclo completo
        (de crista a crista é mais seguro); multiplique pelo \omterm{def:g10:signals-and-waves:oscilloscope}{base de tempo} obter
        $ T $, então $ f = 1/T $.
  \item Conte as divisões verticais da linha central até uma crista e
        multiplique pelo \omterm{def:g10:signals-and-waves:oscilloscope}{ganho} para obter o \omterm{def:g10:signals-and-waves:amplitude}{amplitude} (da crista ao vale dá
        o valor pico a pico: reduzi-lo à metade).
\end{enumerate}
\end{method}

\begin{example}[Uma leitura completa]\label{ex:g10:signals-and-waves:reading}
No \omterm{def:g10:signals-and-waves:oscilloscope}{oscilograma} abaixo, o \omterm{def:g10:signals-and-waves:oscilloscope}{base de tempo} é \qty{5}{ms} por divisão e
o \omterm{def:g10:signals-and-waves:oscilloscope}{ganho} \qty{2}{V} por divisão. Um ciclo abrange $4.0$ divisões:
$ T = 4.0 \times \qty{5}{ms} = \qty{20}{ms}$ e
$ f = 1/\qty{0.020}{s} = \qty{50}{Hz}$ --- a rede elétrica novamente. Uma crista fica
$3.0$ divisões acima do eixo: o \omterm{def:g10:signals-and-waves:amplitude}{amplitude} é
$3.0 \times \qty{2}{V} = \qty{6.0}{V}$.
\end{example}

\section{Sinais sonoros e eletromagnéticos}

\begin{definition}[Som]\label{def:g10:signals-and-waves:sound}
\emph{Som}\index{Som} é um \omterm{def:g10:pressure:pressure}{pressão} \omterm{def:g10:signals-and-waves:signal}{sinal}: um objeto vibrante ---
corda, membrana, cordas vocais --- empurra ritmicamente o ar e
as compressões viajam para fora em cerca de \qty{340}{m/s}. Um microfone
converte a chegada \omterm{def:g10:pressure:pressure}{pressão} em uma voltagem; o tímpano, em nervo
\omterm{def:g10:inertia:inventory}{impulsos}. O som precisa de um meio: um sino tocado sob uma jarra de vácuo cai
silencioso enquanto o ar é bombeado para fora.
\end{definition}

\begin{definition}[Aceno]\label{def:g10:signals-and-waves:wave}
Uma perturbação itinerante --- uma sompadrão de compressão, a ondulação
em uma lagoa, um pulso de rádio --- é um \emph{aceno}\index{aceno}. O meio,
quando há um, permanece onde está: cada pedaço de água flutua no lugar enquanto
a ondulação atravessa o lago. A geometria das ondas --- \omterm{def:g10:light-spectra:wavelength}{comprimento de onda},
interferência, difração --- será abordada em um capítulo posterior; isso
capítulo só precisa de suas velocidades.
\end{definition}

\begin{definition}[Ultrassom e infra-som]\label{def:g10:signals-and-waves:ultrasound}
O ouvido humano ouve som entre cerca de \qty{20}{Hz} e \qty{20}{kHz}
(o teto cai com a idade). \omterm{def:g10:signals-and-waves:sound}{Som} acima \qty{20}{kHz} é
\emph{ultrassonografia}\index{ultrassonografia} --- morcegos caçam em \qty{50}{kHz},
scanners médicos criam imagens do corpo em alguns megahertz. \omterm{def:g10:signals-and-waves:sound}{Som} abaixo
\qty{20}{Hz} é \emph{infra-som}\index{infra-som}, sentido por elefantes
e sismógrafos.
\end{definition}

\begin{proposition}[Sinais eletromagnéticos]\label{prop:g10:signals-and-waves:em}
Rádio, Wi-Fi e luz são \emph{eletromagnético
sinais}\index{eletromagnético
sinais}: uma família, todos viajando
através do vácuo --- nenhum meio necessário --- e tudo em uma velocidade, a velocidade
de luz de \cref{ch:g10:refraction},
\[
c = \qty{3.00e8}{m/s} \quad \text{(very nearly the same in air).}
\]
\end{proposition}
\admitted

\begin{remark}\label{rem:g10:signals-and-waves:honest}
Aqui este é um fato experimental --- a luz solar atravessa $150$ milhão
quilômetros de espaço vazio; é honestamente derivado no volume do ano 2
das leis da eletricidade e do magnetismo. Mantenha o contraste: som é
uma vibração \emph{de} o médium e morre sem ele; um eletromagnético
\omterm{def:g10:signals-and-waves:signal}{sinal} carrega seu próprio campo através do vácuo, um milhão de vezes mais rápido.
\end{remark}

\section{Ecos: distâncias dos atrasos}

UM \omterm{def:g10:signals-and-waves:signal}{sinal} movendo-se em velocidade $ v $ capas $ d = v\,t $ em um tempo $ t $: cada atraso
é uma distância disfarçada. Melhor ainda, envie um pulso curto com uma \omterm{def:g10:signals-and-waves:frequency}{frequência}
obstáculo e tempo é \emph{ecológico}\index{ecológico} --- as coberturas de ida e volta $2d $.

\begin{method}[Alcance do eco]\label{met:g10:signals-and-waves:echo}
\begin{enumerate}
  \item Envie um pulso curto; medir o tempo de ida e volta $ t $ do eco.
  \item Pegue a velocidade $ v $ do \omterm{def:g10:signals-and-waves:signal}{sinal} \emph{no meio cruzado}.
  \item O pulso coberto $2d = v\,t $, então $ d = \dfrac{v\,t}{2}$.
\end{enumerate}
\omterm{def:g10:signals-and-waves:sound}{Som} pulsos na água fazem \emph{sonar}\index{sonar}
($ v \approx \qty{1500}{m/s}$); pulsos de rádio no ar fazem
\emph{radar}\index{radar} ($ v = c $); ultrassom pulsos no corpo fazem
o scanner médico ($ v \approx \qty{1540}{m/s}$ em tecidos moles).
\end{method}

\begin{example}[Sonar]\label{ex:g10:signals-and-waves:sonar}
Um navio \omterm{met:g10:signals-and-waves:echo}{sonar} pings e o fundo do mar eco retorna depois
$ t = \qty{0.80}{s}$: a profundidade é
$ d = \frac{1500 \times 0.80}{2} = \qty{600}{m}$. Esquecendo o fator
$2$ duplica o oceano.
\end{example}

\begin{example}[Radar]\label{ex:g10:signals-and-waves:radar}
Um aeroporto \omterm{met:g10:signals-and-waves:echo}{radar} recebe uma aeronave eco $\qty{0.30}{ms}$ depois do
pulso:
$ d = \frac{\qty{3.00e8}{m/s} \times \qty{3.0e-4}{s}}{2} = \qty{45}{km}$.
Porque $ c $ é tão grande, \omterm{met:g10:signals-and-waves:echo}{radar} vive em relógios de microssegundos: luz de cronometragem
é um trabalho de precisão.
\end{example}

\begin{example}[O coração como sinal]\label{ex:g10:signals-and-waves:ecg}
Cada batimento cardíaco envia um pequeno pulso de voltagem pelo peito; eletrodos
registre-o como o \emph{eletrocardiograma}\index{eletrocardiograma} (ECG), quase \omterm{def:g10:signals-and-waves:period}{periódico}
\omterm{def:g10:signals-and-waves:signal}{sinal}. Se os espinhos altos forem $ T = \qty{0.75}{s}$ à parte, o coração
bate em $ f = 1/0.75 = \qty{1.33}{Hz}$, ou seja\ $1.33 \times 60 = 80$
batidas por minuto: leitura $ T $ fora de um rastro é como um cardiologista toma
seu pulso.
\end{example}

\section{Exercícios}

(c) $ T = \qty{50}{\micro s}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:signals-and-waves:2}

\begin{exercise}[$\star $]\label{exo:g10:signals-and-waves:3}

\begin{exercise}[$\star $]\label{exo:g10:signals-and-waves:4}

\section{Problema: O sonar, a tempestade e o eletrocardiograma}

\begin{problem}[{Problema de fim de semana --- lendo os sinais do mundo: um
lei, $ d = v\,t $, avalia uma tempestade, mapeia o fundo do mar, conta os batimentos cardíacos e
revela como o GPS funciona}]
\label{pb:g10:signals-and-waves:1}
Um barco de pesca trabalha numa noite de tempestade: relâmpagos no horizonte,
o \omterm{met:g10:signals-and-waves:echo}{sonar} varrendo o fundo, o coração do capitão no papel do médico
strip, um receptor GPS ouvindo \omterm{def:g10:universal-gravitation:satellite}{satélites}. Quatro instrumentos, um
lei. Pegar $ v_{\text{sound}} = \qty{340}{m/s}$ no ar, \qty{1500}{m/s} em
água do mar, e $ c = \qty{3.00e8}{m/s}$.

\medskip
\noindent\textbf{Part I --- The storm.}
\begin{enumerate}
  \item Um flash; trovão segue \qty{9.0}{s} mais tarde. A que distância fica?
  \item Calcule o tempo de viagem da luz nessa distância e justifique
        tratando o flash como instantâneo.
  \item Os marinheiros contam os \omterm{def:g10:orders-of-magnitude:unit}{segundos} entre o clarão e o trovão e dividem
        por três para obter quilômetros. Justifique a regra.
  \item Cinco minutos depois, um flash dá \qty{6.0}{s}. Quão longe agora?
        Encontre a \omterm{def:g10:relative-motion:speed}{velocidade média} de aproximação da tempestade em \unit{m/s} e \unit{km/h}.
  \item O trovão ressoa em vez de estalar: o canal do relâmpago está
        quilômetros de comprimento, então suas partes ficam em distâncias diferentes. Se o
        canal se estende de \qty{2.0}{km} para \qty{5.0}{km} do
        barco, quanto tempo dura o barulho?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{met:g10:signals-and-waves:echo}{sonar}.}
\begin{enumerate}[resume]
  \item O \omterm{met:g10:signals-and-waves:echo}{sonar} pings; o fundo do mar responde em \qty{0.90}{s}. Profundidade?
  \item Sobre a prateleira o eco encurta para \qty{0.20}{s}. Profundidade?
  \item Um ping retorna \emph{dois} ecoa, no \qty{0.16}{s} e
        \qty{0.20}{s}. Interprete-os; a que distância acima do fundo está o
        cardumes de peixes nadam?
  \item O \omterm{met:g10:signals-and-waves:echo}{sonar} pings a cada \qty{0.50}{s}. Qual é a maior profundidade
        pode medir sem confusão, e o que há de errado além disso?
  \item No ar, o mesmo \qty{0.60}{s} atraso significaria que distância?
        Moral: o que você deve saber antes de transformar um atraso em um
        distância?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The cardiogram.}
\begin{enumerate}[resume]
  \item Os picos de ECG do capitão são \qty{0.86}{s} separado. Freqüência?
        Batidas por minuto?
  \item A tira avança em \qty{25}{mm/s}. Qual foi o espaçamento dos picos
        medida do médico?
  \item Depois de içar as redes, o espaçamento é \qty{15}{mm}. Nova \omterm{def:g10:signals-and-waves:frequency}{frequência} cardíaca?
  \item O jovem marinheiro, remador treinado, descansa $50$ batidas por
        minuto. \omterm{def:g10:signals-and-waves:period}{Período}? Espaçamento na tira?
  \item Um ECG é estritamente \omterm{def:g10:signals-and-waves:period}{periódico}? O que, então, o médico lê
        do rastreamento?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Homeward by light.}
\begin{enumerate}[resume]
  \item O barco transmite rádio ao porto, \qty{50}{km} ausente. Quanto tempo dura o
        mensagem levar? Quanto tempo demoraria som pegar?
  \item Um GPS \omterm{def:g10:universal-gravitation:satellite}{satélite} \omterm{def:g10:universal-gravitation:satellite}{órbitas} sobre \qty{20200}{km} sobrecarga. Quanto tempo
        faz o seu \omterm{def:g10:signals-and-waves:signal}{sinal} leva para chegar ao barco?
  \item O GPS transforma o tempo em posição. Que erro de distância um relógio
        erro de \qty{1.0}{\micro s} causa? Que precisão de tempo um
        \qty{3.0}{m} correção requer?
  \item Por que não poderia existir GPS baseado em som, mesmo em princípio?
  \item Final: liste as três velocidades usadas esta noite e a lei única
        por trás de todos os quatro instrumentos e, em seguida, indique o hábito de alcance em um
        sentença.
\end{enumerate}
\end{problem}
