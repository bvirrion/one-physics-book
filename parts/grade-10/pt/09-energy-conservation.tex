\chapter{Energia: Formas e Conservação}\label{ch:g10:energy-conservation}

Um carro em movimento, um telefone carregado, um forno quente e um martelo levantado
uma propriedade: cada um pode fazer algo acontecer. A física mede que
capacidade com um único número --- energia --- e mantém suas contas
sob uma regra de ferro: o total nunca muda, apenas se move e
muda de roupa. Este capítulo abre o livro-razão: os formulários energia
leva, as correntes que viaja, e que \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} compra.

\section{Energia: a moeda comum}

\begin{definition}[Energia e o joule]\label{def:g10:energy-conservation:energy}
\emph{Energia}\index{Energia} mede a capacidade de um sistema de produzir
mudanças: colocar a matéria em movimento, levantá-la, aquecê-la, iluminá-la. Isso é \omterm{def:g10:orders-of-magnitude:unit}{unidade}
é o \emph{joule}\index{joule} (\unit{J}). Armazenado, transferido,
convertido --- cada ramo da física conta nesta moeda.
\end{definition}

\begin{definition}[Energia cinética]\label{def:g10:energy-conservation:kinetic}
Um corpo de massa $m $ (em \unit{kg}) movendo-se em velocidade $ v $ (em
\unit{m/s}) carrega o \emph{energia cinética}\index{energia cinética}
$ E_k = \tfrac12\, m v^2$: dobrando a massa dobra, dobrando
a velocidade quadruplica isso.
\end{definition}

\begin{example}[Carro versus pedestre]\label{ex:g10:energy-conservation:kinetic}
UM \qty{1300}{kg} carro em \qty{100}{km/h} $= \qty{27.8}{m/s}$ carrega
$ E_k = \tfrac12 \times 1300 \times 27.8^2 \approx \qty{5.0e5}{J}$;
um \qty{70}{kg} pedestre em \qty{1.5}{m/s},
$\tfrac12 \times 70 \times 1.5^2 \approx \qty{79}{J}$ --- seis
mil vezes menos. A velocidade custa caro: a praça cuida disso.
\end{example}

\begin{definition}[Energia potencial gravitacional]\label{def:g10:energy-conservation:potential}
Um corpo de massa $ m $ em altura $ h $ (em \unit{m}) acima de um escolhido
nível de referência armazena o \emph{potencial gravitacional
energia}\index{potencial gravitacional
energia} $ E_p = m g h $, com
$ g = \qty{9.81}{N/kg}$ --- válido próximo ao solo, onde $ g $ é
efetivamente constante. Apenas \emph{diferenças} da questão da altura: o
nível onde $ E_p = 0$ é escolhido livremente.
\end{definition}

\begin{example}[Torre de água]\label{ex:g10:energy-conservation:potential}
Cada cúbico \omterm{def:g10:orders-of-magnitude:unit}{medidor} (\qty{1000}{kg}) de água armazenada \qty{30}{m} até um
torre de água segura $1000 \times 9.81 \times 30 \approx \qty{2.9e5}{J}$
--- cerca de seis décimos do carro \omterm{def:g10:energy-conservation:kinetic}{energia cinética} acima. Cidades
estoque energia sobrecarga.
\end{example}

\begin{definition}[As outras formas]\label{def:g10:energy-conservation:forms}
\omterm{def:g10:energy-conservation:energy}{Energia} também se esconde em trajes menos visíveis: \emph{térmico
energia}\index{térmico
energia}, a agitação desordenada do corpo
moléculas (quanto mais quente, mais); \emph{químico
energia}\index{químico
energia}, armazenado em ligações moleculares --- alimentos,
madeira, gasolina, batterys; \emph{elétrica
energia}\index{elétrica
energia}, transportado pelas correntes da plant para
tomada; \emph{energia radiativa}\index{energia radiativa}, transportado por
luz mesmo através do espaço vazio --- como o Sol energia chega até nós;
\emph{energia nuclear}\index{energia nuclear}, armazenado dentro do atômico
núcleos, liberados em estrelas (\cref{ch:g12:nuclear-energy}).
\end{definition}

\section{Transferências e cadeias de conversão}

\begin{definition}[Transferência, conversão, cadeia]\label{def:g10:energy-conservation:transfer}
Um \emph{transferência de energia}\index{transferência de energia} movimentos energia
entre sistemas --- \emph{reservatórios}\index{reservatórios};
um \emph{conversão de energia}\index{conversão de energia} muda seu
formulário. Um dispositivo que faz qualquer uma dessas coisas (turbina, motor, lâmpada, músculo) é um
\emph{conversor}\index{conversor}. UM \emph{conversão
corrente}\index{conversão
corrente} fotos da viagem: caixas para
reservatórios e formas, rotulados com setas para transferências.
\end{definition}

\begin{method}[Desenhando uma cadeia de conversão]\label{met:g10:energy-conservation:chain}
\begin{enumerate}
  \item Identifique o reservatório inicial e a forma (o que é consumido?).
  \item Siga o energia \omterm{def:g10:energy-conservation:transfer}{conversor} por \omterm{def:g10:energy-conservation:transfer}{conversor}: uma flecha por
        transferência, o formulário nomeado em cada caixa.
  \item Adicione uma flecha de calor a cada real \omterm{def:g10:energy-conservation:transfer}{conversor} --- alguns energia
        sempre vaza como calor (\omterm{def:g10:inertia:inventory}{atrito}, fios, escapamento).
\end{enumerate}
\end{method}

\section{Conservação de energia}

\begin{theorem}[Conservação de energia]\label{thm:g10:energy-conservation:conservation}
\index{conservação de energia}
O total energia de um sistema isolado --- um que não troca nada
com o exterior --- é constante. \omterm{def:g10:energy-conservation:energy}{Energia} nunca é criado e nunca
destruído: apenas muda de forma e lugar, e qual reservatório
perde, os outros ganham, \omterm{def:g10:energy-conservation:energy}{joule} para \omterm{def:g10:energy-conservation:energy}{joule}.
\end{theorem}
\admitted

\begin{remark}\label{rem:g10:energy-conservation:admitted}
Nenhuma experiência jamais detectou a falha deste princípio; aqui nós pegamos
como dado e use-o numericamente. \cref{ch:g12:work-and-energy}
deriva sua parte mecânica ($ E_k + E_p $) das leis de Newton; o
a contabilidade completa aguarda os volumes da universidade.
\end{remark}

\begin{example}[Queda livre, preço em joules]\label{ex:g10:energy-conservation:fall}
UM \qty{2.0}{kg} bola cai de \qty{10}{m} (resistência do ar
insignificante): começa com
$ E_p = 2.0 \times 9.81 \times 10 \approx \qty{196}{J}$, terras com
$ E_k = \qty{196}{J}$, então $ v = \sqrt{2 \times 196/2.0} =
\qty{14}{m/s} \approx \qty{50}{km/h}$. Não forças, sem cronômetro:
a conservação por si só avalia o impacto.
\end{example}

\section{Eficiência}

\begin{definition}[Eficiência]\label{def:g10:energy-conservation:efficiency}
UM \omterm{def:g10:energy-conservation:transfer}{conversor} consome $ E_{\text{consumed}}$ e entrega o desejado
formulário $ E_{\text{useful}}$; isso é \emph{eficiência}\index{eficiência} é
\[
\eta = \frac{E_{\text{useful}}}{E_{\text{consumed}}} < 1 .
\]
A conservação proíbe $\eta > 1$ e verdadeiro \omterm{def:g10:energy-conservation:transfer}{conversores} nunca alcance $1$:
o desaparecido energia não é destruído --- sai como calor.
\end{definition}

\begin{example}[Para onde vai a gasolina]\label{ex:g10:energy-conservation:engine}
UM \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de lojas de gasolina \qty{45}{MJ} de energia química; um
motor de carro converte-o com $\eta \approx 0.30$: \qty{13.5}{MJ} de
movimento, \qty{31.5}{MJ} aquecimento do escapamento, do radiador e do
rua. Dois terços de cada tanque aquecem a atmosfera.
\end{example}

\begin{remark}[As eficiências se multiplicam]\label{rem:g10:energy-conservation:chain-eta}
Acorrentado \omterm{def:g10:energy-conservation:transfer}{conversores} multiplicar suas \omterm{def:g10:energy-conservation:efficiency}{eficiências}: um poder plantr
($0.38$) alimentando a rede ($0.94$) alimentando um carregador ($0.85$)
entrega $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.30$. Longo
correntes são canos com vazamento: cada flecha sobrecarrega o fluxo.
\end{remark}

\section{Potência, o watt e o quilowatt-hora}

\begin{definition}[Poder]\label{def:g10:energy-conservation:power}
\emph{Poder}\index{Poder} é a taxa na qual energia fluxos,
$ P = E/t $, medido em \emph{watts}\index{watts} (\unit{W}), com
$\qty{1}{W} = \qty{1}{J/s}$. \omterm{def:g10:energy-conservation:energy}{Energia} é uma quantidade, alimenta um fluxo ---
uma banheira versus sua torneira.
\end{definition}

\begin{example}[Chaleira versus humano]\label{ex:g10:energy-conservation:power}
UM \qty{2200}{W} chaleira funcionando \qty{150}{s} usa
$ E = 2200 \times 150 = \qty{3.3e5}{J}$. Um humano corre sobre
\qty{10}{MJ} de comida por dia, em média
$\num{1.0e7}/\num{86400} \approx \qty{116}{W}$: você fica ocioso como um
lâmpada velha e brilhante.
\end{example}

\begin{definition}[O quilowatt-hora]\label{def:g10:energy-conservation:kwh}
O \emph{quilowatt-hora}\index{quilowatt-hora} (\unit{kW.h}) é o
energia de um \qty{1}{kW} fluxo funcionando por uma hora:
$\qty{1}{kW.h} = 1000 \times 3600\,\unit{J} = \qty{3.6}{MJ}$ ---
o \omterm{def:g10:orders-of-magnitude:unit}{unidade} contagem de \omterm{def:g10:orders-of-magnitude:unit}{medidores} de eletricidade: poder em \unit{kW} vezes horas.
\end{definition}

\begin{example}[Lendo a conta]\label{ex:g10:energy-conservation:bill}
Uma nota é uma subtração e uma multiplicação: nova leitura do \omterm{def:g10:orders-of-magnitude:unit}{medidor}
menos velho dá o \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora}; vezes o \omterm{def:g10:orders-of-magnitude:unit}{unidade} preço (cerca de
$0.25$ euros por \unit{kW.h}), mais uma assinatura fixa, dá a
total. UM \qty{2}{kW} aquecedor funcionando \qty{2.5}{h} usos noturnos
\qty{5}{kW.h} um dia: $38$ euros por mês por conta própria.
\end{example}

\begin{example}[Ordens de magnitude]\label{ex:g10:energy-conservation:orders}
Vale a pena saber de cor: levantando uma maçã \omterm{def:g10:orders-of-magnitude:unit}{medidor}, sobre \qty{1}{J};
uma battery de telefone, $\qty{10}{W.h} \approx \qty{4e4}{J}$; um cereal
tigela, \qty{1.5}{MJ}; sua comida por um dia, \qty{10}{MJ}; um \omterm{def:g10:orders-of-magnitude:unit}{quilograma}
de gasolina, \qty{45}{MJ} --- quase exatamente o diário de uma família
eletricidade ($\approx \qty{12}{kW.h}$); um raio, \qty{5}{GJ}:
cem dias domésticos, entregues num piscar de olhos.
\end{example}

\section{Problema: fornecer energia a uma casa por um dia}

\begin{problem}[{Problema de fim de semana --- da tigela de cereal ao telhado
painéis: uma auditoria energética completa de um dia normal em casa, e que
custos de joule dependendo de quem o vende}]
\label{pb:g10:energy-conservation:1}
Numa terça-feira comum, uma família sai em busca de onde está
eletricidade vai, armado com este capítulo, o \omterm{def:g10:orders-of-magnitude:unit}{medidor} do corredor e
o verso da conta. O veredicto desta noite: o dia deles é igual a um
\omterm{def:g10:orders-of-magnitude:unit}{quilograma} de gasolina, dezesseis quadrados metros de sol --- ou cinco
ciclistas pedalando o tempo todo.

\medskip
\noindent\textbf{Part I --- The audit.}
Inventário do dia: geladeira \qty{100}{W} para \qty{24}{h}; água
aquecedor \qty{2400}{W} para \qty{2.5}{h}; forno \qty{2000}{W} para
\qty{1.0}{h}; máquina de lavar \qty{500}{W} para \qty{2.0}{h}; iluminação
\qty{60}{W} para \qty{5.0}{h}; telas \qty{150}{W} para \qty{6.0}{h}.
\begin{enumerate}
  \item Calcule o valor de cada aparelho energia para o dia, em \unit{kW.h}.
  \item Totalize o dia em \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora}, depois em megajoules;
        compare com um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de gasolina (\qty{45}{MJ}).
  \item Calcule a média da casa poder sobre o \qty{24}{h};
        que fração de uma chaleira funcionando é essa?
  \item Qual único aparelho domina? Recalcular o total diário
        se o aquecedor funcionasse apenas \qty{1.25}{h}.
  \item As luzes de espera desenham uma imagem permanente \qty{15}{W}: diário energia,
        parcela do custo total anual em $0.25$ euros por \unit{kW.h}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The bill.}
\begin{enumerate}[resume]
  \item Em 30 dias o \omterm{def:g10:orders-of-magnitude:unit}{medidor} passou de \qty{56214}{kW.h} para
        \qty{56604}{kW.h}: consumo e média diária verificada
        contra a auditoria?
  \item Calcule a conta: $0.25$ euros por \unit{kW.h} mais um
        fixo $14$ assinatura em euros.
  \item Preço de um megajoule da rede? Da gasolina, em
        $1.80$ euros por litro de armazenamento \qty{34}{MJ}?
  \item UM \qty{1.5}{MJ} tigela de cereal custa cerca de $0.50$ euros: é
        preço em megajoules? Classifique os três vendedores de \omterm{def:g10:energy-conservation:energy}{joules}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The rooftop.}
\begin{enumerate}[resume]
  \item A luz solar plena proporciona cerca de \qty{1000}{W} por quadrado
        \omterm{def:g10:orders-of-magnitude:unit}{medidor}; painéis convertem-no com $\eta = 0.20$: elétrico
        poder de \qty{1}{m^2} em pleno sol?
  \item A região tem em média o equivalente a \qty{4.0}{h} de cheio
        sol por dia: rendimento diário de \qty{1}{m^2}?
  \item Que área do painel cobre a família \qty{12.6}{kW.h} dia?
        O telhado sul oferece \qty{20}{m^2}: cabe?
  \item Em dezembro o equivalente cai para \qty{1.5}{h}: o
        produção de painéis, sua participação na necessidade e o que preenche
        a lacuna?
  \item A fabricação dos painéis custou cerca de \qty{700}{kW.h} por
        quadrado \omterm{def:g10:orders-of-magnitude:unit}{medidor}: energia tempo de retorno, usando a pergunta 11?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The cereal bowl.}
\begin{enumerate}[resume]
  \item Uma bicicleta geradora produz \qty{100}{W} de eletricidade:
        quantas horas de pedalada faz \qty{12.6}{kW.h} --- quantos
        pessoas pedalando \qty{24}{h} sem parar?
  \item Os músculos correm em $\eta \approx 0.25$: a comida energia atrás
        aqueles \qty{12.6}{kW.h}, em megajoules e em tigelas de cereal?
  \item Custo do dia pedalado (tigelas em $0.50$ euros), contra
        quanto a rede cobra no mesmo dia?
  \item UM \qty{5}{GJ} raio equivale a quantos dias-casa?
        Dada a sua \qty{100}{\micro s} duração, por que ainda pode
        não poder a casa?
  \item Os Joules nunca se perdem, mas a rede, a gasolina e os alimentos \omterm{def:g10:energy-conservation:energy}{joules}
        vender a preços muito diferentes: pelo que você realmente paga?
  \item Final: declare o veredicto do dia em uma frase --- uma
        \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de gasolina, dezesseis quadrados metros de telhado ensolarado ou
        cinco ciclistas 24 horas por dia --- e nomeie as duas quantidades
        (um estoque, um fluxo) para nunca mais confundir.
\end{enumerate}
\end{problem}
