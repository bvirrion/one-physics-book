\chapter{Gravitação Universal e Peso}\label{ch:g10:universal-gravitation}

Deixe cair uma maçã: ela cai. A Lua, a sessenta raios da Terra acima, parece nunca fazê-lo.
A grande descoberta de Newton é que a Lua \emph{é} caindo --- sobre
1,4 milímetros a cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}, exatamente a quantidade que atrai
enfraquecendo como o quadrado da distância prevê. Uma fórmula, com um
constante universal, governa a maçã, a Lua, as marés e o
planetas. O primeiro volume descrito \omterm{def:g10:universal-gravitation:force}{gravitação} com palavras; isso
capítulo mede isso.

\section{A lei da gravitação universal}

\begin{definition}[Força gravitacional]\label{def:g10:universal-gravitation:force}
Quaisquer dois corpos se atraem, seja qual for a sua composição e por mais
distantes eles se sentam. Essa atração é a \emph{gravitacional
força}\index{gravitacional
força}, e o fenômeno é
\emph{gravitação}\index{gravitação}. Não precisa de contato, nem corda e
não há meio, e é \emph{mútuo}: cada corpo puxa o outro.
\end{definition}

\begin{theorem}[Lei da gravitação universal]\label{thm:g10:universal-gravitation:law}
Dois corpos de massas $m_1$ e $ m_2$ (em \unit{kg}), cujos centros são um
distância $ d $ à parte (em \unit{m}), atraem-se com duas forças
direcionado ao longo da linha que une os centros, cada um apontando para o
outro corpo, de igual magnitude
\[
F = G\,\frac{m_1 m_2}{d^2}
\qquad \text{(in \unit{N})},
\]
onde $ G $ é a mesma constante para cada par de corpos no universo.
\end{theorem}
\admitted

\begin{remark}[De onde vem a lei]
Newton destilou esta lei a partir do movimento observado dos planetas
(leis de Kepler); a dedução honesta é realizada no Ano 1
volume. \Cref{ch:g12:satellites-kepler} já vai colocar a lei para funcionar
ligado \omterm{def:g10:universal-gravitation:satellite}{satélites} e planetas.
\end{remark}

\begin{definition}[A constante gravitacional]\label{def:g10:universal-gravitation:constant}
A constante
\[
G = \qty{6.67e-11}{N.m^2/kg^2}
\]
é o \emph{constante gravitacional}\index{constante gravitacional}. É
valor minúsculo é a razão pela qual a gravidade entre os objetos do cotidiano é
imperceptível: somente quando pelo menos uma massa é astronômica é que
$ G m_1 m_2 / d^2$ equivale a qualquer coisa.
\end{definition}

\begin{remark}[Lendo a fórmula]\label{rem:g10:universal-gravitation:reading}
Quatro características merecem atenção.
\begin{itemize}
  \item \emph{Mútuo e igual.} A Terra puxa você com seu \omterm{def:g10:universal-gravitation:weight}{peso}
        --- e você puxa a Terra com exatamente a mesma força. (O
        A Terra apenas responde menos: trata-se de $\num{e23}$ vezes
        mais pesado.)
  \item \emph{Quadrado inverso.} Duplicação $ d $ divide a força por $4$;
        movendo-se dez vezes mais, divide-o por $100$.
  \item \emph{Distância entre centros.} Para corpos esféricos --- estrelas,
        planetas, balas de canhão --- $ d $ é medido de centro a centro, não
        superfície a superfície.
  \item \emph{Ambas as massas são importantes.} Dobrar qualquer uma das massas duplica o
        força; a fórmula é simétrica em $ m_1$ e $ m_2$.
\end{itemize}
\end{remark}

\begin{example}[Dois amigos]\label{ex:g10:universal-gravitation:friends}
Dois \qty{70}{kg} amigos ficam com seus centros \qty{1.0}{m} separado:
\[
F = \num{6.67e-11} \times \frac{70 \times 70}{1.0^2}
\approx \qty{3.3e-7}{N}.
\]
Cada amigo \omterm{def:g10:universal-gravitation:weight}{peso} é sobre \qty{6.9e2}{N} --- dois \emph{bilhão}
vezes maior. A atração mútua deles é real, mas impossível de sentir.
\end{example}

\begin{example}[A Terra segura a Lua]\label{ex:g10:universal-gravitation:earthmoon}
Com $ m_1 = \qty{5.97e24}{kg}$ (Terra), $ m_2 = \qty{7.35e22}{kg}$ (Lua)
e $ d = \qty{3.84e8}{m}$:
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{2.0e20}{N}.
\]
E a Lua puxa a Terra de volta com o mesmo \qty{2.0e20}{N} --- um
puxe a resposta dos oceanos duas vezes por dia, conforme as marés.
\end{example}

\begin{method}[Calculando uma força gravitacional]\label{met:g10:universal-gravitation:compute}
\begin{enumerate}
  \item Converta as massas em \omterm{def:g10:orders-of-magnitude:unit}{quilogramas} e o centro a centro
        distância em metros.
  \item Aplicar $ F = G m_1 m_2 / d^2$.
  \item Verifique a potência de dez separadamente dos dígitos, como em
        \cref{ch:g10:orders-of-magnitude}: forças gravitacionais alcance
        de $\num{e-7}$ newtons (duas pessoas por \omterm{def:g10:orders-of-magnitude:unit}{medidor} à parte) para $\num{e22}$ newtons
        (Sol na Terra), então um lugar perdido expoente é fácil de pegar.
\end{enumerate}
\end{method}

\section{Peso e a força da gravidade}

\begin{definition}[Peso]\label{def:g10:universal-gravitation:weight}
O \emph{peso}\index{peso} $\vect P $ de um objeto na superfície
uma estrela (um planeta, uma lua, \dots) é o força gravitacional a estrela
exerce sobre ele. Ele aponta para o centro da estrela --- a direção que
ligar \emph{vertical}\index{vertical} --- e é medido em newtons.
\end{definition}

\begin{proposition}[Peso na superfície de uma estrela]\label{prop:g10:universal-gravitation:surfaceg}
Na superfície de uma estrela esférica de massa $ M $ e raio $ R $, um objeto
de massa $ m $ tem \omterm{def:g10:universal-gravitation:weight}{peso}
\[
P = m\,g,
\qquad\text{where}\qquad
g = \frac{G M}{R^2}
\]
depende apenas da estrela, não do objeto.
\end{proposition}

\begin{proof}
O centro do objeto e o centro da estrela estão a uma distância $ R $ à parte, então
a lei da universalidade \omterm{def:g10:universal-gravitation:force}{gravitação}
(\cref{thm:g10:universal-gravitation:law}) dá
$ P = G M m / R^2 = m \times \left(G M / R^2\right)$. O colchete é o
o mesmo para todos os objetos na superfície: chame-o $ g $.
\end{proof}

\begin{definition}[Força do campo gravitacional]\label{def:g10:universal-gravitation:fieldstrength}
A quantidade $ g = G M / R^2$, em newtons por \omterm{def:g10:orders-of-magnitude:unit}{quilograma}, é o
\emph{intensidade do campo gravitacional}\index{intensidade do campo gravitacional}
na superfície da estrela: a estrela puxa cada \omterm{def:g10:orders-of-magnitude:unit}{quilograma} com $ g $ Newtons.
Na Terra, $ g = \qty{9.81}{N/kg}$.
\end{definition}

\begin{example}[Computação $ g $]\label{ex:g10:universal-gravitation:gvalues}
Para a Terra ($ M = \qty{5.97e24}{kg}$, $ R = \qty{6.37e6}{m}$):
\[
g = \frac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.37e6})^2}
= \frac{\num{3.98e14}}{\num{4.06e13}}
\approx \qty{9.81}{N/kg}.
\]
Para a Lua ($ M = \qty{7.35e22}{kg}$, $ R = \qty{1.74e6}{m}$) o mesmo
cálculo dá $ g_{\text{Moon}} \approx \qty{1.62}{N/kg}$ --- seis
vezes menos: a Lua é muito mais clara e ser menor não compensa
por isso. Para Marte, $ g_{\text{Mars}} \approx \qty{3.73}{N/kg}$.
\end{example}

\begin{remark}[Por que os astronautas saltam]\label{rem:g10:universal-gravitation:bounce}
Um astronauta de massa \qty{120}{kg} (corpo mais terno) pesa
$120 \times 9.81 \approx \qty{1180}{N}$ na Terra, mas apenas
$120 \times 1.62 \approx \qty{194}{N}$ na Lua. Músculos treinados
sob \qty{1180}{N} de repente carregar um sexto da carga: daí o
famosa marcha lunar lenta e saltitante. A massa --- e com ela o esforço
precisava \emph{parar} ou \emph{vez} --- permanece inalterado, e é por isso
os astronautas lunares caíram com tanta frequência.
\end{remark}

\section{Massa ou peso? A balança e a balança de mola}

A linguagem cotidiana diz que um saco de farinha pesa três quilos. Física
divide essa frase em duas.

\begin{remark}[Massa não é peso]\label{rem:g10:universal-gravitation:massweight}
\begin{itemize}
  \item O \emph{massa} $ m $ mede a quantidade de matéria, em
        \omterm{def:g10:orders-of-magnitude:unit}{quilogramas}. É uma propriedade apenas do objeto: o mesmo em
        Terra, na Lua ou à deriva no espaço profundo.
  \item O \emph{\omterm{def:g10:universal-gravitation:weight}{peso}} $ P = mg $ é uma força, em newtons, exercida
        \emph{por uma estrela} no objeto. Ele muda de estrela para estrela
        com $ g $ --- e desaparece longe de todas as estrelas.
\end{itemize}
Confundi-los é inofensivo no mercado e fatal no mundo da física.
problema: o \omterm{def:g10:orders-of-magnitude:unit}{quilograma} não é um \omterm{def:g10:orders-of-magnitude:unit}{unidade} de força.
\end{remark}

\begin{method}[A balança e a balança de mola]\label{met:g10:universal-gravitation:instruments}
\begin{itemize}
  \item UM \emph{equilíbrio de feixe} compara o objeto com massas padrão:
        ambas as panelas' \omterm{def:g10:universal-gravitation:weight}{pesos} escalar pelo mesmo $ g $, então a comparação
        mede o \emph{massa} --- mesmo veredicto para qualquer estrela.
  \item UM \emph{escala de mola} mede a força que estica sua mola:
        ele lê o \emph{\omterm{def:g10:universal-gravitation:weight}{peso}}, em newtons, e sua leitura muda
        de estrela em estrela. Para recuperar a massa, divida pelo local
        força do campo: $ m = P/g $.
\end{itemize}
\end{method}

\begin{example}[Farinha na Lua]\label{ex:g10:universal-gravitation:flour}
UM \qty{3.0}{kg} saco de farinha equilibra-se contra \qty{3.0}{kg} de padrão
massas na Terra \emph{e} na Lua. Uma escala de mola, no entanto, diz
$3.0 \times 9.81 \approx \qty{29}{N}$ na Terra e
$3.0 \times 1.62 \approx \qty{4.9}{N}$ na Lua. Um comerciante vendendo
``pelo newton'' na Lua distribuiria seis vezes mais farinha ---
o comércio, tal como a ciência, funciona em massa.
\end{example}

\section{Caindo ao redor da Terra: o canhão de Newton}

Jogue uma pedra horizontalmente: a gravidade curva seu caminho em um arco e ela
pousa a poucos metros de distância. Jogue mais rápido: ele cai mais longe. Newton
experimento mental leva a ideia ao seu limite. A Terra é redonda e
sua superfície cai sobre \qty{5}{m} abaixo da horizontal para cada
\qty{8}{km} viajou; um corpo em queda livre cai \qty{4.9}{m} em seu
primeiro \omterm{def:g10:orders-of-magnitude:unit}{segundo} (\cref{ch:g12:projectile-motion} estabelecerá isso). Então um
projétil roçando a Terra em \qty{8}{km/s} cai exatamente tão rápido quanto
o chão se curva: cai para sempre, sem nunca pousar.

\begin{definition}[Satélite e órbita]\label{def:g10:universal-gravitation:satellite}
UM \emph{satélite}\index{satélite} de uma estrela é um corpo que cai
ao redor da estrela sem nunca atingir sua superfície; o caminho que ele repete
é o seu \emph{órbita}\index{órbita}.
\end{definition}

\begin{remark}[A Lua é uma pedra caindo]\label{rem:g10:universal-gravitation:fallingmoon}
A Lua é o elemento natural da Terra \omterm{def:g10:universal-gravitation:satellite}{satélite}: cai em nossa direção cerca de
\qty{1.4}{mm} a cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}, e seu movimento lateral o carrega
antes que ele possa chegar mais perto
(\cref{exo:g10:universal-gravitation:15} permite refazer o próprio Newton
verificação deste número). Os astronautas de uma estação orbital estão no
mesma queda livre de seu navio --- isso, e não qualquer ausência de gravidade,
é por isso que eles flutuam. \omterm{def:g10:universal-gravitation:force}{Gravitação} molda cada trajetória, desde o
o arco da pedra até a elipse do planeta; a história quantitativa de \omterm{def:g10:universal-gravitation:satellite}{órbitas}
é contado em \cref{ch:g12:satellites-kepler}.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:universal-gravitation:1}
Dois \qty{60}{kg} parceiros de dança se abraçam com seus centros
\qty{0.80}{m} separado.
\begin{enumerate}
  \item Calcule sua atração gravitacional mútua.
  \item Compare-o com o \omterm{def:g10:universal-gravitation:weight}{peso} de um \qty{2.5}{mg} mosquito.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:universal-gravitation:2}
Um aluno tem uma massa de \qty{55}{kg}.
\begin{enumerate}
  \item Calcule ela \omterm{def:g10:universal-gravitation:weight}{peso} na Terra.
  \item Ela viaja para a Lua. Quais são a massa dela e ela \omterm{def:g10:universal-gravitation:weight}{peso}
        lá?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:universal-gravitation:3}
\begin{enumerate}
  \item A partir do cartão de dados, recalcule o \omterm{def:g10:universal-gravitation:fieldstrength}{intensidade do campo gravitacional}
        na superfície de Marte.
  \item UM \qty{900}{kg} rover é enviado para lá. Calcule seu \omterm{def:g10:universal-gravitation:weight}{peso} em Marte
        e seu \omterm{def:g10:universal-gravitation:weight}{peso} na Terra.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:universal-gravitation:4}
Usando o cartão de dados, calcule a força exercida pela Terra sobre o
Lua. Que força a Lua exerce sobre a Terra?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:universal-gravitation:5}
Dois corpos se atraem com uma força $ F_0$. O que a força
tornar-se se:
\begin{enumerate}
  \item a distância entre os centros é duplicada;
  \item a distância é reduzida pela metade;
  \item ambas as massas são duplicadas (mesma distância);
  \item uma massa é triplicada \emph{e} a distância é triplicada?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:6}
Dois superpetroleiros carregados de \qty{5.0e8}{kg} cada um está atracado com seu
centros \qty{100}{m} separado.
\begin{enumerate}
  \item Calcule sua atração gravitacional mútua.
  \item Que massa tem \omterm{def:g10:universal-gravitation:weight}{peso} igual a esta força na Terra?
  \item A força é surpreendentemente grande  mas os petroleiros não mostram
        inclinação para flutuar juntos. Calcule a aceleração
        $ a = F/m $ dá um petroleiro e comenta.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:7}
Uma loja lunar vende um \qty{3.0}{kg} saco de arroz.
\begin{enumerate}
  \item O que uma balança de feixe e uma balança de mola indicam para esta bolsa
        na Terra? Na Lua?
  \item Um cliente paga por ``\qty{3.0}{kg}''. Qual instrumento
        garante um acordo justo em qualquer mundo, e porquê?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:8}
A Estação Espacial Internacional voa a uma altitude de \qty{400}{km}.
\begin{enumerate}
  \item Calcule o da Terra \omterm{def:g10:universal-gravitation:fieldstrength}{intensidade do campo gravitacional} nisso
        altitude. (Cuidado: qual é a distância até a Terra
        \emph{centro}?)
  \item Expresse-o como uma porcentagem do valor da superfície.
  \item Os astronautas a bordo flutuam. Reconcilie isso com sua resposta.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:9}
Na superfície de Vênus, $ g = \qty{8.87}{N/kg}$, e o planeta
raio é \qty{6.05e6}{m}. Deduza a massa de Vênus e compare-a com
a da Terra.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:10}
\begin{enumerate}
  \item Calcule a força exercida pelo Sol sobre \qty{1.0}{kg} do mar
        água, então a força exercida pela Lua sobre a mesma
        \omterm{def:g10:orders-of-magnitude:unit}{quilograma}.
  \item O Sol puxa com mais força --- por que fator?
  \item No entanto, a Lua domina as marés. A maré não é impulsionada pelo
        puxe-se, mas pelo \emph{diferença} entre os puxões
        os lados próximos e distantes da Terra. Explique, sem
        computação, por que a Lua próxima pode vencer o Sol distante
        esse critério.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:universal-gravitation:11}
\begin{enumerate}
  \item Um planeta tem o dobro da massa da Terra e o dobro do seu raio.
        Expresse a intensidade do campo superficial em termos da intensidade da Terra
        $ g $, então em \unit{N/kg}.
  \item Que raio um planeta com a massa da Terra precisaria para sua
        intensidade do campo superficial a ser $2g $?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:12}
Em algum lugar na linha Terra-Lua, os dois puxam uma sonda espacial
cancelar. Deixar $ x $ seja a distância do centro da Terra até esse ponto
e $ D $ a distância Terra-Lua.
\begin{enumerate}
  \item Mostre isso naquele momento
        $\left(\dfrac{x}{D - x}\right)^{2}
        = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.
  \item Deduzir $ x $. Que fração da viagem à Lua é essa?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:13}
Canhão de Newton, com números. Um projétil roça a Terra
horizontalmente em velocidade $ v $.
\begin{enumerate}
  \item Depois $ x = \qty{8.0}{km}$ de deslocamento horizontal, o esférico
        A Terra caiu de uma altura $ h $ satisfatório
        $(R + h)^2 = R^2 + x^2$. Usando
        $(R + h)^2 \approx R^2 + 2Rh $ para pequeno $ h $, mostre que
        $ h \approx x^2 / (2R)$ e calcular $ h $.
  \item Um corpo em queda livre cai \qty{4.9}{m} em seu primeiro \omterm{def:g10:orders-of-magnitude:unit}{segundo}.
        Compare com $ h $ e deduza a velocidade com que o projétil
        nunca pousa.
  \item Calcule a duração de um período completo \omterm{def:g10:universal-gravitation:satellite}{órbita} nessa velocidade, deslizando
        a superfície.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:14}
Uma estrela de nêutrons embala $ M = \qty{2.8e30}{kg}$ (1,4 vezes a massa do Sol)
em um raio de \qty{12}{km}.
\begin{enumerate}
  \item Calcule o \omterm{def:g10:universal-gravitation:fieldstrength}{intensidade do campo gravitacional} em sua superfície.
  \item O que seria um \qty{70}{kg} \omterm{def:g10:universal-gravitation:weight}{peso} humano lá? Compare com seus
        \omterm{def:g10:universal-gravitation:weight}{peso} na Terra.
  \item Qual é o \omterm{def:g10:universal-gravitation:weight}{peso}, lá, de um \qty{1.0}{mg} grão de areia?
        Que massa tem isso \omterm{def:g10:universal-gravitation:weight}{peso} na Terra?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:universal-gravitation:15}
Teste da Lua de Newton (1666). A Lua \omterm{def:g10:universal-gravitation:satellite}{órbita} raio é cerca de $60$ Terra
raios.
\begin{enumerate}
  \item Por que fator é a atração da Terra por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} mais fraco no
        Lua do que na superfície da Terra, se a gravidade seguir a
        quadrado inverso?
  \item Perto da superfície, um corpo em queda cai \qty{4.9}{m} em seu
        primeiro \omterm{def:g10:orders-of-magnitude:unit}{segundo}. Quão longe a Lua deve cair em direção à Terra
        cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}?
  \item Verifique contra o \omterm{def:g10:universal-gravitation:satellite}{órbita}: a Lua cobre seu círculo de raio
        $ d = \qty{3.84e8}{m}$ em $ T = 27.3$ dias. Calcule sua velocidade
        $ v $, a distância $ x $ cobre em um \omterm{def:g10:orders-of-magnitude:unit}{segundo}, depois a queda
        $ h \approx x^2/(2d)$ (como em
        \cref{exo:g10:universal-gravitation:13}).
  \item Compare as respostas das questões 2 e 3 e indique o que Newton
        concluiu.
\end{enumerate}
\end{exercise}

\section{Problema: Pesando a Terra}

\begin{problem}[{Problema de fim de semana --- de duas esferas de chumbo em um galpão para
a massa da Terra e depois do Sol: que pequena constante,
medido uma vez, vale}]
\label{pb:g10:universal-gravitation:1}
Em 1798, Henry Cavendish suspendeu uma barra de luz carregando dois pequenos
esferas de um fio fino, aproximou duas grandes esferas de chumbo e
mediu a torção quase inexistente do fio. Os jornais disseram
ele tinha \emph{pesou a Terra} --- e eles estavam certos: uma vez $ G $ é
conhecido, o próprio planeta $ g $ e raio entregam sua massa, e o
Anual da Terra \omterm{def:g10:universal-gravitation:satellite}{órbita} entrega o do Sol. Este problema remonta ao
cadeia inteira. Use o cartão de dados por toda parte.

\medskip
\noindent\textbf{Part I --- The feeblest force.}
\begin{enumerate}
  \item O ferro da Torre Eiffel tem uma massa de cerca de \qty{7.3e6}{kg}. UM
        \qty{55}{kg} stands turísticos \qty{100}{m} do seu centro de
        massa. Calcule a atração da torre sobre o turista.
  \item Compare-o com o \omterm{def:g10:universal-gravitation:weight}{peso} de um \qty{1.0}{mg} grão de areia.
  \item Exercício de proporcionalidade: o que acontece com um força gravitacional
        se (a) a distância for multiplicada por $10$; (b) ambas as massas são
        multiplicado por $1000$; (c) ambas as massas \emph{e} a distância
        são multiplicados por $1000$?
  \item Calcule a atração da Terra sobre um \qty{1.00}{kg} objeto em seu
        superfície, de $ M $, $ R $ e $ G $. Qual nome e símbolo do dia a dia
        esse número carrega?
  \item A gravidade é de longe a interação mais fraca que você já conheceu ---
        e ainda assim governa o universo. Explique em uma ou duas frases
        por que a influência da Terra sobre você domina a do seu vizinho.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Cavendish's balance.}
Na balança de torção, uma grande esfera de massa $ m_1 = \qty{158}{kg}$
atrai um pequeno de massa $ m_2 = \qty{0.73}{kg}$; seus centros são
$ d = \qty{0.23}{m}$ separado.
\begin{enumerate}[resume]
  \item Com o valor moderno de $ G $, calcule a força entre
        duas esferas.
  \item Calcule o \omterm{def:g10:universal-gravitation:weight}{peso} da pequena esfera e a razão entre
        duas forças. Por que Cavendish precisava de um fio de torção em vez de
        qualquer escala comum?
  \item A torção medida de Cavendish corresponde (em moderno \omterm{def:g10:orders-of-magnitude:unit}{unidades}) para
        $ F = \qty{1.47e-7}{N}$. Deduza seu valor de $ G $ e compará-lo
        com o de hoje.
  \item Com $ G = \qty{6.67e-11}{N.m^2/kg^2}$, $ g = \qty{9.81}{N/kg}$
        e $ R = \qty{6.37e6}{m}$, deduza a massa da Terra. (Este
        é o passo que os jornais celebraram.)
  \item Deduzir a densidade média da Terra $\rho = M / V $, com
        $ V = \frac43 \pi R^3$. As rochas superficiais têm densidades em torno
        \qty{2.7e3}{kg/m^3}: o que a comparação revela sobre o
        interior profundo?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Other worlds.}
\begin{enumerate}[resume]
  \item A partir do cartão de dados, recalcule o campo da superfície da Lua
        força.
  \item Na Terra, um bom impulso eleva o seu centro de massa em
        \qty{0.50}{m}. Para o mesmo impulso, a altura alcançada é
        inversamente proporcional a $ g $ (como o capítulo sobre energia,
        \cref{ch:g10:energy-conservation}, justificará). Quão alto
        o mesmo salto leva você à Lua?
  \item Mercúrio: $ M = \qty{3.30e23}{kg}$, $ R = \qty{2.44e6}{m}$.
        Calcule a intensidade do campo superficial e compare com Marte.
        Mercúrio é muito menor que Marte --- como podem os dois valores
        estar tão perto?
  \item A que altitude acima da superfície da Terra está o campo
        resistência caiu para metade do seu valor superficial?
  \item A que distância do centro da Terra ocorre a atração da Terra
        por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} cair para o valor da superfície da Lua,
        \qty{1.62}{N/kg}? Expresse-o em raios da Terra.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Weighing the Sun.}
A Terra percorre uma trajetória quase circular \omterm{def:g10:universal-gravitation:satellite}{órbita} de raio
$ d = \qty{1.496e11}{m}$ em um ano, $ T = \qty{3.156e7}{s}$. Tome como
dado (está provado em \cref{ch:g12:satellites-kepler}) que um corpo em
um caminho circular de raio $ d $ em velocidade $ v $ precisa de um puxão de $ v^2/d $
newtons por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} em direção ao centro.
\begin{enumerate}[resume]
  \item Calcule a velocidade orbital da Terra $ v $.
  \item Deduza a atração por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} que o Sol exerce sobre a Terra.
  \item Essa atração também é $ G M_{\text{Sun}} / d^2$. Deduza a massa de
        o Sol.
  \item Quantas Terras são isso? Comparar $ M_{\text{Sun}}$ com
        $ M_{\text{Earth}}$.
  \item Final. As esferas de Cavendish, a Terra, o Sol: uma constante
        $ G$ serviu para os três. Diga em uma frase qual é a palavra
        \emph{universal} em ``universal \omterm{def:g10:universal-gravitation:force}{gravitação}'' nos comprou
        esse problema.
\end{enumerate}
\end{problem}
