\chapter{Ordens de Grandeza: Medindo o Universo}\label{ch:g10:orders-of-magnitude}

A física começa onde termina a opinião: com uma medição. Mas as coisas
medidas físicas abrangem uma faixa absurda --- um núcleo atômico é cerca de
\qty{e-15}{m} do outro lado, o universo observável sobre \qty{e27}{m}: um
fator de $10^{42}$, muito além de qualquer coisa que uma imagem mental possa conter.
Este capítulo instala o kit de ferramentas que domina a gama --- \omterm{def:g10:orders-of-magnitude:unit}{unidades},
potências de dez, \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} --- e dois hábitos usados em cada
página após esta: escreva o \omterm{def:g10:orders-of-magnitude:unit}{unidade}, e verifique o \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza}.

\section{Unidades: a gramática da medida}

\begin{definition}[Grandezas físicas e unidades SI]\label{def:g10:orders-of-magnitude:unit}
UM \emph{quantidade física}\index{quantidade física} é uma propriedade de
o mundo que pode ser medido: um comprimento, uma duração, uma massa.
Medir significa contar quantas vezes uma quantidade de referência ---
o \emph{unidade}\index{unidade} --- cabe nele. O
\emph{Sistema Internacional de Unidades}\index{Sistema Internacional de Unidades} (SI) fixa sete unidades base, das quais três executam todo este
volume: o \emph{medidor}\index{medidor} (\unit{m}) para comprimento, o
\emph{quilograma}\index{quilograma} (\unit{kg}) para massa e o
\emph{segundo}\index{segundo} (\unit{s}) por tempo. (Os outros quatro ---
ampere, kelvin, mole, candela --- se juntarão quando eletricidade, calor e
química entra na história.)
\end{definition}

\begin{remark}[Um número sem unidade não é resultado]\label{rem:g10:orders-of-magnitude:units}
``A distância é $384$'' não significa nada: \qty{384}{mm},
\qty{384}{km} e \qty{384}{Mm} são três mundos diferentes (o último
acontece ser a distância até a Lua). Em 1999, a NASA perdeu o
\$327 milhões \emph{Orbital Climático de Marte} porque um programa calculou
forças do propulsor em libras-força, enquanto outro as lê como newtons.
Regra da casa, desta linha até o final do livro: cada medição ou
número computado carrega seu \omterm{def:g10:orders-of-magnitude:unit}{unidade}, em cada linha do cálculo.
\end{remark}

\begin{definition}[Prefixos SI]\label{def:g10:orders-of-magnitude:prefixes}
UM \emph{prefixo}\index{prefixo} colado a um \omterm{def:g10:orders-of-magnitude:unit}{unidade} multiplica por um
potência fixa de dez:
\begin{center}
\begin{tabular}{llc@{\qquad\qquad}llc}
femto & f & $10^{-15}$ & centi & c & $10^{-2}$ \\
pico  & p & $10^{-12}$ & kilo  & k & $10^{3}$  \\
nano  & n & $10^{-9}$  & mega  & M & $10^{6}$  \\
micro & $\mu $ & $10^{-6}$ & giga & G & $10^{9}$ \\
milli & m & $10^{-3}$  & tera  & T & $10^{12}$ \\
\end{tabular}
\end{center}
Por isso $\qty{1}{nm} = \qty{e-9}{m}$ e $\qty{1}{km} = \qty{e3}{m}$.
\end{definition}

\begin{example}[Lendo prefixos]\label{ex:g10:orders-of-magnitude:prefixes}
A luz verde tem comprimento de onda de cerca de $\qty{500}{nm} = \qty{5.00e-7}{m}$; um
cabelo humano é sobre $\qty{70}{\micro m} = \qty{7.0e-5}{m}$ espesso; o
A Lua orbita em $\qty{384}{Mm} = \qty{3.84e8}{m}$. Uma gota de chuva tem
massa sobre $\qty{30}{mg} = \qty{3.0e-5}{kg}$ --- observe que a base SI
\omterm{def:g10:orders-of-magnitude:unit}{unidade} de massa, o único entre os sete, carrega um \omterm{def:g10:orders-of-magnitude:prefixes}{prefixo} já.
\end{example}

\section{Notação científica e algarismos significativos}

\begin{definition}[Notação científica]\label{def:g10:orders-of-magnitude:scinot}
Um número está em \emph{notação científica}\index{notação científica}
quando escrito como
\[
a \times 10^{n}, \qquad 1 \leq a < 10, \quad n \text{ an integer.}
\]
O fator $ a $ é o \emph{mantissa}\index{mantissa}, o número inteiro
$ n $ o \emph{expositor}\index{expositor}.
Por isso $\num{149600000} = \num{1.496e8}$ e
$\num{0.00052} = \num{5.2e-4}$.
\end{definition}

\begin{proposition}[Computando com potências de dez]\label{prop:g10:orders-of-magnitude:powers}
Para todos os números inteiros $ m $ e $ n $:
\[
10^{m} \times 10^{n} = 10^{m+n}, \qquad
\frac{1}{10^{n}} = 10^{-n}, \qquad
\left(10^{m}\right)^{n} = 10^{mn}.
\]
\end{proposition}

\begin{proof}[Prova parcial]
Para positivo expoentes, conte os fatores: $10^{m} \times 10^{n}$ é um
produto de $ m + n $ dezenas e $(10^m)^n $ correntes $ n $ grupos de $ m $ dezenas.
Escrita $10^{-n}$ para $1/10^{n}$ é precisamente a convenção que
mantém a primeira regra verdadeira para todos os inteiros: força
$10^{n} \times 10^{-n} = 10^{0} = 1$. A matemática companheira
volume estuda o expoente regras na íntegra.
\end{proof}

\begin{example}[Multiplicando em notação científica]\label{ex:g10:orders-of-magnitude:scinot}
Quantas Terras cabem no Sol, em massa? Com
$ m_{\text{Sun}} = \qty{1.99e30}{kg}$ e
$ m_{\text{Earth}} = \qty{5.97e24}{kg}$:
\[
\frac{\num{1.99e30}}{\num{5.97e24}}
= \frac{1.99}{5.97} \times 10^{30-24}
= 0.333 \times 10^{6}
= \num{3.33e5}.
\]
\omterm{def:g10:orders-of-magnitude:scinot}{Mantissas} e expoentes são tratados separadamente; a resposta final é
renormalizado para que o \omterm{def:g10:orders-of-magnitude:scinot}{mantissa} pousa de volta $[1, 10)$.
\end{example}

\begin{definition}[Números significativos]\label{def:g10:orders-of-magnitude:sigfig}
O \emph{algarismos significativos}\index{algarismos significativos} de um
número medido são os dígitos realmente comprovados pelo
medição: todos os dígitos escritos, exceto os zeros à esquerda. À direita
contagem de zeros --- $\qty{5.20e-3}{s}$ (três algarismos significativos)
promete mais do que $\qty{5.2e-3}{s}$ (dois). \omterm{def:g10:orders-of-magnitude:scinot}{Notação científica} mostra
a contagem de relance: o \omterm{def:g10:orders-of-magnitude:scinot}{mantissa} carrega exatamente o significativo
figuras.
\end{definition}

\begin{method}[Quantos dígitos manter]\label{met:g10:orders-of-magnitude:sigfig}
Um resultado computado não pode ser mais preciso que seus ingredientes.
\begin{enumerate}
  \item Em um produto ou quociente, mantenha tantos \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} como
        o \emph{ao menos} fator preciso.
  \item Arredonde apenas a resposta final, nunca as etapas intermediárias.
  \item Na dúvida, três \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos} é o trabalho
        precisão deste livro.
\end{enumerate}
\end{method}

\begin{example}[Circunferência da Terra]\label{ex:g10:orders-of-magnitude:sigfig}
O diâmetro da Terra é $ d = \qty{1.27e7}{m}$ (três \omterm{def:g10:orders-of-magnitude:sigfig}{algarismos significativos}),
então sua circunferência é
$\pi d = \num{3.14159} \times \qty{1.27e7}{m} = \qty{3.99e7}{m}$ ---
três algarismos, não os oito que a calculadora exibe. Escrita
$\qty{39898227}{m}$ afirmaria conhecer o planeta ao máximo \omterm{def:g10:orders-of-magnitude:unit}{medidor}.
\end{example}

\section{A escada do universo}

\begin{definition}[Ordem de magnitude]\label{def:g10:orders-of-magnitude:oom}
O \emph{ordem de grandeza}\index{ordem de grandeza} de uma quantidade
é a potência de dez mais próxima dele: escreva a quantidade como
$ a \times 10^{n}$ com $1 \leq a < 10$ e pegue $10^{n}$ se $ a < 5$,
$10^{n+1}$ se $ a \geq 5$. (A convenção de limite exato nunca
importa: uma declaração de ordem de grandeza só se destina a dentro
um fator de dez.)
\end{definition}

\begin{example}[Ordens de magnitude]\label{ex:g10:orders-of-magnitude:oom}
Uma vida humana dura cerca de $80$ anos, ou seja,\
$80 \times \qty{3.16e7}{s} \approx \qty{2.5e9}{s}$: \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza}
$\qty{e9}{s}$ --- alguns bilhões de \omterm{def:g10:orders-of-magnitude:unit}{segundos}, gaste-os bem. Um humano é
$\qty{1.7}{m}$ alto: ordem $\qty{e0}{m}$. Monte Everest,
$\qty{8848}{m}$: ordem $\qty{e4}{m}$, desde $8.848 \geq 5$.
\end{example}

Empilhar essas estimativas por expoente produz a imagem mestre de
este capítulo: uma escada em que cada degrau é um mundo dez vezes maior
do que o abaixo.

Acima do degrau da Terra, metros deixe de ser conveniente; astronomia mantém
próprio \omterm{def:g10:orders-of-magnitude:unit}{unidades}, ambos definidos a partir da própria escada.

\begin{definition}[Unidade astronômica e ano-luz]\label{def:g10:orders-of-magnitude:lightyear}
O \emph{unidade astronômica}\index{unidade astronômica}
($\qty{1}{au} = \qty{1.496e11}{m}$) é a média Terra-Sol
distância. O \emph{ano-luz}\index{ano-luz} (\unit{ly}) é o
distância que a luz percorre em um ano. A luz se move no vácuo em
\[
c = \qty{3.00e8}{m/s},
\]
a velocidade mais rápida que a natureza permite --- então o ano-luz é uma distância,
não é um momento.
\end{definition}

\begin{example}[O ano-luz em metros]\label{ex:g10:orders-of-magnitude:ly}
Um ano se mantém $365.25 \times 24 \times 3600 = \qty{3.156e7}{s}$
(agradavelmente perto de $\pi \times 10^{7}\,\unit{s}$). Por isso
\[
\qty{1}{ly} = c \times \qty{3.156e7}{s}
= \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}.
\]
Proxima Centauri, a estrela mais próxima depois do Sol, fica
$\qty{4.2}{ly} \approx \qty{4.0e16}{m}$ away: luz, que atravessa um
Distância Terra-Sol em oito minutos, precisa de mais de quatro anos para chegar
de lá.
\end{example}

\begin{method}[Verificação de sanidade por ordem de grandeza]\label{met:g10:orders-of-magnitude:sanity}
Antes de confiar em qualquer resultado computado:
\begin{enumerate}
  \item arredondar cada entrada para o seu \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza};
  \item refazer o cálculo apenas com potências de dez, usando
        \cref{prop:g10:orders-of-magnitude:powers} --- isso leva
        \omterm{def:g10:orders-of-magnitude:unit}{segundos};
  \item comparar: a resposta precisa deve concordar com a estimativa para
        dentro de um fator de dez ou mais, e seu \omterm{def:g10:orders-of-magnitude:unit}{unidade} deve ser o certo
        um. Se alguma das verificações falhar, procure o erro antes de prosseguir
        ligado.
\end{enumerate}
\end{method}

\begin{remark}[Por que se preocupar]\label{rem:g10:orders-of-magnitude:sanity}
Uma calculadora informará com prazer que uma maçã tem massa
$\qty{4e8}{kg}$ --- não pode saber melhor. Nenhuma fórmula sinaliza o
absurdo; o hábito faz. Os físicos estimam primeiro e calculam depois:
a estimativa captura pontos decimais perdidos, trocados \omterm{def:g10:orders-of-magnitude:unit}{unidades} e
frações invertidas, os três erros por trás da maioria das respostas erradas em
todos os níveis.
\end{remark}

\section{Medição e incerteza}

Nenhum instrumento lê o valor verdadeiro. Uma medição só é honesta quando
também informa a que distância pode estar.

\begin{definition}[Incerteza absoluta e relativa]\label{def:g10:orders-of-magnitude:uncertainty}
Uma medição de uma quantidade $ x $ é afirmado como
\[
x = x_{\text{m}} \pm \Delta x ,
\]
onde $ x_{\text{m}}$ é o valor medido e o \emph{absoluto
incerteza}\index{absoluto
incerteza} $\Delta x > 0$, no mesmo
\omterm{def:g10:orders-of-magnitude:unit}{unidade}, limita o erro plausível: afirma-se que o valor verdadeiro reside
entre $ x_{\text{m}} - \Delta x $ e $ x_{\text{m}} + \Delta x $. O
\emph{incerteza relativa}\index{incerteza relativa} é a proporção
$\Delta x / x_{\text{m}}$, geralmente cotado em porcentagem; é o que
palavra medidas precisas.
\end{definition}

\begin{method}[Lendo um instrumento graduado]\label{met:g10:orders-of-magnitude:reading}
Para uma régua, um cilindro de medição, um mostrador:
\begin{enumerate}
  \item leia a graduação mais próxima da marca, olho perpendicular ao
        escala (um olho inclinado muda a leitura);
  \item tome pelo menos metade da menor graduação como \omterm{def:g10:orders-of-magnitude:uncertainty}{absoluto
        incerteza} --- mais se a borda do objeto ou do líquido
        a superfície está confusa;
  \item escreva o resultado como $ x_{\text{m}} \pm \Delta x $ com seu \omterm{def:g10:orders-of-magnitude:unit}{unidade},
        e não mantenha nenhum dígito além do incerto.
\end{enumerate}
\end{method}

\begin{example}[Precisão relativa]\label{ex:g10:orders-of-magnitude:reading}
A leitura acima tem \omterm{def:g10:orders-of-magnitude:uncertainty}{incerteza relativa}
$\frac{0.5}{127} \approx 0.4\%$. A mesma régua medindo um
$\qty{12}{mm}$ parafuso dá $\frac{0.5}{12} \approx 4\%$: o idêntico
instrumento é dez vezes menos preciso no objeto pequeno. A precisão é
uma propriedade do \emph{medição}, não da ferramenta.
\end{example}

\begin{remark}[Absoluto pequeno, relativo enorme --- e inversamente]\label{rem:g10:orders-of-magnitude:precision}
Estações de laser medem a distância Terra-Lua, $\qty{3.84e8}{m}$, para
sobre $\pm\qty{3}{cm}$: um \omterm{def:g10:orders-of-magnitude:uncertainty}{incerteza relativa} de $10^{-10}$, entre
as medições mais precisas já feitas --- com um erro absoluto não
governante se vangloriaria. Sempre pergunte \emph{em relação ao que}: o par
$(\Delta x,\ \Delta x / x_{\text{m}})$ conta toda a história e
qualquer número por si só pode enganar.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:orders-of-magnitude:1}
Converter para metros, em \omterm{def:g10:orders-of-magnitude:scinot}{notação científica}:
\begin{enumerate}
  \item $\qty{750}{nm}$ (comprimento de onda da luz vermelha);
  \item $\qty{2.5}{km}$;
  \item $\qty{3.2}{\micro m}$ (uma bactéria do solo);
  \item $\qty{0.15}{Tm}$ (a distância Terra-Sol, de uma forma incomum
        traje).
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:orders-of-magnitude:2}
Escreva em \omterm{def:g10:orders-of-magnitude:scinot}{notação científica} e dê o número de significativo
figuras:
\begin{enumerate}
  \item a distância Terra-Sol, $\qty{149600000}{km}$, em metros;
  \item o diâmetro de um átomo de hidrogênio, $\qty{0.000000000106}{m}$;
  \item a população mundial, cerca de $\num{8100000000}$;
  \item uma duração de $\qty{0.00520}{s}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:orders-of-magnitude:3}
Calcule sem calculadora, fornecendo cada resultado em científico
notação:
\[
(3\times10^{8})\times(2\times10^{-5}), \qquad
\frac{8\times10^{4}}{4\times10^{-3}}, \qquad
(5\times10^{6})^{2}, \qquad
\frac{(4\times10^{-3})\times(3\times10^{-9})}{6\times10^{-7}} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:orders-of-magnitude:4}
Dê o \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza}, com seu \omterm{def:g10:orders-of-magnitude:unit}{unidade}, de: uma altura humana de
$\qty{1.7}{m}$; Monte Everest, $\qty{8848}{m}$; um \emph{E. coli}
bactéria, $\qty{2e-6}{m}$; a massa da Terra,
$\qty{5.97e24}{kg}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:orders-of-magnitude:5}
Uma régua é graduada em milímetros. Você mede um lápis em
$\qty{127}{mm}$.
\begin{enumerate}
  \item Indique o resultado com seu incerteza absoluta.
  \item Calcule o \omterm{def:g10:orders-of-magnitude:uncertainty}{incerteza relativa}.
  \item A mesma régua mede uma $\qty{12}{mm}$ apagador. Qual dos
        duas medições é mais precisa, e por qual fator?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:orders-of-magnitude:6}
Usando $ c = \qty{3.00e8}{m/s}$ e $\qty{1}{ly} = \qty{9.47e15}{m}$:
\begin{enumerate}
  \item Expressar em metros as distâncias para Proxima Centauri
        ($\qty{4.24}{ly}$) e para Sirius ($\qty{8.6}{ly}$).
  \item A sonda \emph{Viajante 1} é sobre $\qty{2.4e13}{m}$ de
        Terra. Quanto tempo dura seu sinal de rádio, viajando a $ c $, pegue
        para chegar até nós? Expresse a resposta em horas.
\end{enumerate}
\end{exercise}

\section{Problema: Do átomo a Andrômeda}

\begin{problem}[{Problema de fim de semana --- um único grão de areia dimensiona o
átomo, o Sistema Solar, a galáxia e o universo observável, e
localiza você na escada entre eles}]
\label{pb:g10:orders-of-magnitude:1}
Um grão de areia é o objeto mais humilde da física, e neste fim de semana ele
faz hora extra: primeiro contamos seus átomos, depois encolhemos o Sol para
seu tamanho e percorrer o cosmos, então deixamos a luz fazer o levantamento,
e finalmente contamos os átomos em \emph{você}. Dados usados em:
diâmetro do grão $\qty{0.5}{mm}$; diâmetro do átomo $\qty{2.5e-10}{m}$;
densidade da areia $\qty{2500}{kg/m^3}$; diâmetros: Sol
$\qty{1.39e9}{m}$, Terra $\qty{1.27e7}{m}$, Via Láctea
$\qty{9.5e20}{m}$, universo observável $\qty{8.8e26}{m}$; distâncias
de nós: Lua $\qty{3.84e8}{m}$, Sol $\qty{1.496e11}{m}$, Netuno
$\qty{4.5e12}{m}$, Próxima Centauri $\qty{4.0e16}{m}$, Andrômeda
galáxia $\qty{2.4e22}{m}$; $ c = \qty{3.00e8}{m/s}$; uma molécula de água
tem massa $\qty{3.0e-26}{kg}$ e contém três átomos.

\medskip
\noindent\textbf{Part I --- Anatomy of the grain.}
\begin{enumerate}
  \item Escreva o diâmetro do grão e o diâmetro do átomo em metros, em
        \omterm{def:g10:orders-of-magnitude:scinot}{notação científica}, e nomeie o SI \omterm{def:g10:orders-of-magnitude:prefixes}{prefixo} cada um está mais próximo
        para.
  \item Quantos átomos ficam lado a lado em um diâmetro de grão?
  \item Modele o grão como um cubo de lado $\qty{0.5}{mm}$ embalado com
        átomos. Estime o número de átomos no grão, como ordem
        de magnitude.
  \item Calcule o volume desse cubo e depois a massa do grão.
        Expresse a massa em miligramas.
  \item Verifique a sanidade do modelo: divida a massa do grão pelo seu número
        de átomos e compare com o conhecido \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza} de
        massas atômicas (alguns $10^{-26}\,\unit{kg}$). Veredicto?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Sun as a grain of sand.}
Construímos um modelo em escala do cosmos no qual o Sol é o grão:
cada tamanho de modelo é igual ao tamanho real vezes um fator $ s $.
\begin{enumerate}[resume]
  \item Calcule o fator de escala $ s $.
  \item No modelo, encontre o diâmetro da Terra e sua distância de
        o grão-Sol. Qual objeto do mundo da Parte I tem aproximadamente o
        modelo do tamanho da Terra?
  \item A que distância do grão o modelo orbita Netuno? Que pedaço de
        móveis conteriam todo o sistema planetário?
  \item A que distância está o modelo Proxima Centauri?
  \item Dois grãos de areia, $\qty{14}{km}$ à parte, e quase nada
        no meio: indique em uma ou duas frases o que o modelo
        revela sobre o quão vazia a galáxia está.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Light does the surveying.}
\begin{enumerate}[resume]
  \item Quanto tempo leva a luz para atravessar o grão de areia real?
  \item Quanto tempo leva a luz para chegar até nós vinda da Lua e de
        o Sol?
  \item Recalcular o \omterm{def:g10:orders-of-magnitude:lightyear}{ano-luz} em metros de $ c $ e o comprimento de
        o ano e verifique se a distância Proxima nos dados
        corresponde a cerca $\qty{4.2}{ly}$.
  \item Converta a distância de Andrômeda para anos-luz. Aproximadamente quando
        a luz que agora entra em seus olhos saiu de Andrômeda, e o que
        estava andando pela Terra naquela época?
  \item Quão longe a luz viaja em um nanossegundo? Um processador
        cronometrado em $\qty{3}{GHz}$ inicia uma operação a cada terço de um
        nanossegundo: explique em uma frase por que um computador rápido deve
        ser um pequeno computador.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- You, between atom and Andromeda.}
\begin{enumerate}[resume]
  \item Um humano é aproximadamente $\qty{70}{kg}$ de água. Usando os dados,
        estimar a massa média de um átomo na água, então o
        número de átomos em um ser humano.
  \item Compare essa contagem com os átomos do grão (pergunta 3)
        e com aproximadamente $2\times10^{23}$ estrelas do observável
        universo. Dê os dois fatores.
  \item Na escada, você está mais próximo do átomo ou do observável
        universo? Compare as proporções $\text{(you)}/\text{(atom)}$ e
        $\text{(universe)}/\text{(you)}$, tirando $\qty{1.7}{m}$ para você.
  \item Encontre o tamanho $ x $ sentado exatamente na metade da escada
        entre o núcleo ($\qty{e-15}{m}$) e o observável
        universo: $ x $ deve satisfazer $ x / 10^{-15} = 10^{26.9} / x $
        (usando $\qty{8.8e26}{m} \approx 10^{26.9}\,\unit{m}$). O que
        O objeto do Sistema Solar tem exatamente esse tamanho?
  \item A piada. Encolher o universo observável pela Parte II
        fator $ s$, de modo que o Sol é um grão de areia. Dê o modelo
        diâmetro do universo em metros, então em \omterm{def:g10:orders-of-magnitude:unit}{unidades} astronômicase
        concluo em uma frase: mesmo com cada estrela voltada para um
        grão de areia, o universo transborda todas as escalas humanas --- apenas
        potências de dez o seguram.
\end{enumerate}
\end{problem}
