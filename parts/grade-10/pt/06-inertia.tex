\chapter{Forças e o Princípio da Inércia}\label{ch:g10:inertia}

Pare de pedalar e a bicicleta para; pare de puxar e o
trenó fica parado. A experiência sussurra que o movimento deve ser alimentado por um
esforço --- e por dois mil anos a física concordou. Este capítulo
constrói o \omterm{def:g10:inertia:force}{vigor}, uma flecha medida em \omterm{def:g10:inertia:force}{newtons}, e afirma o primeiro
grande lei da mecânica: deixado verdadeiramente sozinho, um corpo em movimento mantém sua
movimento, reto e constante, para sempre.

\section{Modelando uma ação como uma força}

\begin{definition}[Vigor]\label{def:g10:inertia:force}
UM \emph{vigor}\index{vigor} modela uma ação mecânica (empurrar, puxar,
uma atração) exercida sobre um corpo. É caracterizado por:
\begin{itemize}
  \item isso é \emph{ponto de aplicação}\index{ponto de aplicação};
  \item sua direção --- a linha ao longo da qual atua e o caminho
        ao longo dessa linha;
  \item sua magnitude, medida em \emph{newtons}\index{newtons}
        (símbolo \unit{N}).
\end{itemize}
Uma força é desenhada como uma flecha $\vect F $ (um vetor, como na matemática
volume) a partir do seu ponto de aplicação, com comprimento proporcional ao
magnitude em uma escala declarada.
\end{definition}

\begin{example}[O peso de uma mochila escolar]\label{ex:g10:inertia:schoolbag}
Uma mochila de massa $ m = \qty{4.0}{kg}$ é puxado para baixo pela Terra
com seu \omterm{def:g10:universal-gravitation:weight}{peso} (\cref{ch:g10:universal-gravitation}):
$ P = mg = 4.0 \times 9.81 \approx \qty{39}{N}$ --- um Newton é sobre
uma maçã \omterm{def:g10:universal-gravitation:weight}{peso}. Na escala $\qty{1}{cm} \leftrightarrow
\qty{10}{N}$: uma seta \omterm{def:g10:universal-gravitation:weight}{vertical} \qty{3.9}{cm} longo, aplicado no
centro de gravidade da bolsa, apontando para baixo.
\end{example}

\section{Um inventário de forças}

\begin{definition}[Forças de contato e distância]\label{def:g10:inertia:contact}
UM \emph{força de contato}\index{força de contato} é exercido nos pontos
onde dois corpos se tocam. UM \emph{força à distância}\index{força à distância} não precisa de contato: \omterm{def:g10:universal-gravitation:weight}{peso} --- a atração gravitacional de
\cref{ch:g10:universal-gravitation} --- é o único usado neste
capítulo.
\end{definition}

\begin{definition}[As forças de contato comuns]\label{def:g10:inertia:inventory}
Quatro forças de contato cobrir a maioria das situações cotidianas:
\begin{itemize}
  \item o \emph{ocorrência normal}\index{ocorrência normal} $\vect R $ de
        um suporte, perpendicular à sua superfície, afastando-se dela;
  \item o \emph{tensão}\index{tensão} $\vect T $ de um fio, corda ou
        corrente, direcionada ao longo dela, puxando em sua direção;
  \item \emph{atrito}\index{atrito} $\vect f $ de uma superfície, o
        ar ou água, opondo-se ao deslizamento ou ao movimento;
  \item o \emph{impulso}\index{impulso} de um motor, jato ou mão,
        empurrando o esforço.
\end{itemize}
\end{definition}

\begin{example}[Inventário em um trenó rebocado]\label{ex:g10:inertia:sled}
Um trenó rebocado pela neve: \omterm{def:g10:universal-gravitation:weight}{peso} $\vect P $ (distância), normais
reação $\vect R $ da neve, \omterm{def:g10:inertia:inventory}{tensão} $\vect T $ da corda, \omterm{def:g10:inertia:inventory}{atrito}
$\vect f $ opondo-se ao deslizamento. Nada mais toca o trenó: o
o inventário está completo --- \emph{\omterm{def:g10:universal-gravitation:weight}{peso} sempre, então um \omterm{def:g10:inertia:contact}{força de contato}
por objeto tocante}.
\end{example}

\section{O princípio da inércia}

\begin{definition}[Movimento reto e uniforme]\label{def:g10:inertia:uniform}
Um corpo está em \emph{movimento reto e uniforme}\index{movimento reto e uniforme} em um quadro quando é \omterm{def:g10:relative-motion:trajectory}{trajetória} é uma linha reta percorrida em
velocidade constante: seu vetor velocidade $\vect v $ é o mesmo em todos
instantâneo. O descanso é o caso especial $\vect v = \vect 0$.
\end{definition}

\begin{definition}[Forças compensatórias]\label{def:g10:inertia:compensate}
Forças agindo sobre um corpo \emph{compensar}\index{compensar}
quando seu efeito combinado é nulo: de ponta a ponta, suas flechas retornam para
o começo. Dois forças compensar exatamente quando eles compartilham a mesma linha
de ação com magnitudes iguais e caminhos opostos.
\end{definition}

\begin{theorem}[Princípio da inércia]\label{thm:g10:inertia:principle}
\index{princípio da inércia}\index{inércia}
No \omterm{def:g10:relative-motion:frames}{estrutura de solo}, um corpo sujeito a não \omterm{def:g10:inertia:force}{vigor}, ou para forças que
\omterm{def:g10:inertia:compensate}{compensar}, permanece \omterm{def:g10:relative-motion:frame}{em repouso} ou mantém um \omterm{def:g10:inertia:uniform}{movimento reto e uniforme};
inversamente, um corpo \omterm{def:g10:relative-motion:frame}{em repouso} ou em \omterm{def:g10:inertia:uniform}{movimento reto e uniforme} está sujeito
para forças compensatórias (ou para nenhum).
\end{theorem}
\admitted

\begin{remark}
Nenhum experimento pode isolar completamente um corpo, então o princípio é um postulado
--- vislumbrado por Galileu, afirmado por Newton. Um capítulo posterior o atualiza para um
lei relativa forças para \emph{mudanças} de velocidade; o volume do Ano 1
pergunta em quais quadros ele realmente se mantém.
\end{remark}

\begin{proposition}[Lendo o movimento das forças e vice-versa]\label{prop:g10:inertia:converse}
No \omterm{def:g10:relative-motion:frames}{estrutura de solo}:
\begin{enumerate}
  \item se a velocidade de um corpo muda --- em magnitude \emph{ou} em
        direção --- o forças nisso faça \emph{não} \omterm{def:g10:inertia:compensate}{compensar};
  \item se um corpo está \omterm{def:g10:relative-motion:frame}{em repouso} ou em \omterm{def:g10:inertia:uniform}{movimento reto e uniforme}, é
        inventariado forças deve cancelar --- determinar incógnitas.
\end{enumerate}
\end{proposition}

\begin{proof}
A primeira afirmação é a contrapositiva de
\cref{thm:g10:inertia:principle}, o \omterm{def:g10:orders-of-magnitude:unit}{segundo} sua parte inversa aplicada
para um inventário concluído.
\end{proof}

\begin{example}[O disco de hóquei]\label{ex:g10:inertia:puck}
Solto pelo stick, um disco cruza \qty{30}{m} de gelo liso em
\qty{2.5}{s}, direto na constante $ v = 30/2.5 = \qty{12}{m/s}$.
Inventário: \omterm{def:g10:universal-gravitation:weight}{peso} e reação normal (\omterm{def:g10:inertia:inventory}{atrito} insignificante). Eles
\omterm{def:g10:inertia:compensate}{compensar}, e o princípio prevê exatamente o movimento observado:
nada empurra o disco para frente --- e nada precisa fazê-lo.
\end{example}

\begin{remark}[Galileu contra Aristóteles]
Aristóteles ensinou que todo movimento precisa de um motor e da vida diária ---
\omterm{def:g10:inertia:inventory}{atrito} em todos os contatos --- parece concordar. Galileu idealizou o
\omterm{def:g10:inertia:inventory}{atrito} distância: superfícies cada vez mais lisas precisam de cada vez menos empurrão para
\emph{manter} movimento. A persistência, e não o descanso, é o estado natural.
\end{remark}

\begin{remark}[O passageiro cambaleante]
Quando um ônibus freia bruscamente, os passageiros em pé inclinam-se para a frente. Não \omterm{def:g10:inertia:force}{vigor}
os arremessa: por inércia seus corpos mantêm a velocidade enquanto o ônibus
perde o seu. O cinto de segurança fornece a marcha para trás \omterm{def:g10:inertia:force}{vigor} o inventário
falta.
\end{remark}

\section{Forças equilibradas: estática de um desenho}

\begin{definition}[Diagrama de força]\label{def:g10:inertia:diagram}
O \emph{diagrama de força}\index{diagrama de força} de um corpo mostra o corpo
sozinho, com cada \omterm{def:g10:inertia:force}{vigor} atuando \emph{sobre} foi desenhado em escala a partir de seu
\omterm{def:g10:inertia:force}{ponto de aplicação}; forças exercido \emph{por} o corpo nunca aparece.
\end{definition}

\begin{method}[Analisando uma situação estática]\label{met:g10:inertia:method}
\begin{enumerate}
  \item Escolha o corpo para estudar; trabalhar no \omterm{def:g10:relative-motion:frames}{estrutura de solo}.
  \item Inventário: \omterm{def:g10:universal-gravitation:weight}{peso} sempre, um \omterm{def:g10:inertia:contact}{força de contato} por contato.
  \item Desenhe o \omterm{def:g10:inertia:diagram}{diagrama de força}.
  \item O corpo está \omterm{def:g10:relative-motion:frame}{em repouso}, então o forças \omterm{def:g10:inertia:compensate}{compensar}
        (\cref{thm:g10:inertia:principle}): leia magnitudes desconhecidas
        o desenho (comprimentos iguais em 1D, componentes em 2D).
\end{enumerate}
\end{method}

\begin{example}[O livro sobre a mesa]\label{ex:g10:inertia:book}
Um livro de massa \qty{0.50}{kg} repousa sobre uma mesa: \omterm{def:g10:universal-gravitation:weight}{peso}
$ P = 0.50 \times 9.81 \approx \qty{4.9}{N}$ e reação normal
$\vect R $. \omterm{def:g10:relative-motion:frame}{Em repouso}, $\vect R $ compensa $\vect P $: \omterm{def:g10:universal-gravitation:weight}{vertical}, para cima,
$ R = \qty{4.9}{N}$. Adicione um \omterm{def:g10:orders-of-magnitude:unit}{segundo} livro e $ R $ ajusta --- um suporte
empurra exatamente com a força necessária.
\end{example}

\begin{example}[A lâmpada em dois fios]\label{ex:g10:inertia:lamp}
Uma lâmpada de massa \qty{3.0}{kg} ($ P \approx \qty{29.4}{N}$) está pendurado em dois
fios, cada um em $45^\circ $ da \omterm{def:g10:universal-gravitation:weight}{vertical}. Por simetria o tensões
são iguais ($ T_1 = T_2 = T $) e suas componentes horizontais se cancelam; o
componentes verticais, cada $ T \cos 45^\circ $, leve o \omterm{def:g10:universal-gravitation:weight}{peso}:
\[
2\,T \cos 45^\circ = P
\qquad\text{so}\qquad
T = \frac{29.4}{2 \times 0.707} \approx \qty{21}{N}
\]
--- mais da metade \omterm{def:g10:universal-gravitation:weight}{peso} por fio: puxar em ângulo custa
\omterm{def:g10:inertia:inventory}{tensão}.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:inertia:1}
Inventariar e classificar (contato ou distância) os forças em: (a) um livro
\omterm{def:g10:relative-motion:frame}{em repouso} sobre uma mesa; (b) uma lâmpada pendurada em seu fio; (c) um disco
deslizando em gelo liso; (d) uma maçã caindo (resistência do ar
insignificante).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:2}
Calcule o \omterm{def:g10:universal-gravitation:weight}{peso} de um \qty{250}{g} livro e de um \qty{60}{kg}
pessoa. Na escala $\qty{1}{cm} \leftrightarrow \qty{1}{N}$, como
longa é a flecha do livro, e por que a escala é desesperadora para o
pessoa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:3}
Em um diagrama na escala $\qty{1}{cm} \leftrightarrow \qty{2}{N}$, um
\omterm{def:g10:inertia:force}{vigor} é uma seta horizontal \qty{3.5}{cm} longo apontando para a esquerda, aplicado
em um ponto $ A $. Dê suas três características.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:4}
Um dicionário de massa \qty{1.2}{kg} está em uma prateleira. Desenhe seu força
diagrama e indique o módulo e a direção de cada \omterm{def:g10:inertia:force}{vigor}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:inertia:5}
Verdadeiro ou falso --- corrija os falsos: (a) um corpo sempre acaba
parar a menos que um \omterm{def:g10:inertia:force}{vigor} continua empurrando; (b) \omterm{def:g10:inertia:uniform}{movimento reto e uniforme}
não requer nenhuma compensação \omterm{def:g10:inertia:force}{vigor}; (c) um corpo \omterm{def:g10:relative-motion:frame}{em repouso} não está sujeito a nenhuma
\omterm{def:g10:inertia:force}{vigor}; (d) se a velocidade de um corpo mudar, o forças nisso não
\omterm{def:g10:inertia:compensate}{compensar}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:6}
Após o lançamento, um disco cobre \qty{24}{m} de gelo em \qty{3.0}{s},
em linha reta com velocidade constante. Calcule sua velocidade; faça o forças nisso
\omterm{def:g10:inertia:compensate}{compensar}? No concreto, o mesmo disco para dentro metros: o que mudou
no inventário e o que \cref{prop:g10:inertia:converse}
concluir?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:7}
UM \qty{70}{kg} passageiro está em um elevador subindo a uma velocidade constante
\qty{1.5}{m/s}. Liste o forças no passageiro; eles \omterm{def:g10:inertia:compensate}{compensar}?
Dê a magnitude do piso \omterm{def:g10:inertia:force}{vigor}. Alguma coisa mudaria com
o elevador \omterm{def:g10:relative-motion:frame}{em repouso}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:8}
Uma lâmpada de massa \qty{1.8}{kg} está pendurado \omterm{def:g10:relative-motion:frame}{em repouso} por um único fio \omterm{def:g10:universal-gravitation:weight}{vertical}.
Calcule o \omterm{def:g10:inertia:inventory}{tensão} do fio. O fio é substituído por dois
\emph{\omterm{def:g10:universal-gravitation:weight}{vertical}} fios dividindo a carga igualmente: qual é o \omterm{def:g10:inertia:inventory}{tensão} de
cada um?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:9}
Um ônibus freia bruscamente e um passageiro em pé dá um solavanco para frente; em um

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:10}
Um pára-quedista de massa total \qty{85}{kg} desce verticalmente no

\begin{exercise}[$\star\star $]\label{exo:g10:inertia:11}
Um esquiador aquático é rebocado em linha reta com velocidade constante; a corda horizontal

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:12}
Uma lâmpada de massa \qty{2.4}{kg} está pendurado em dois fios em $40^\circ $ de

Por que o \omterm{def:g10:inertia:inventory}{tensão} crescer à medida que os fios se aproximam da horizontal, e

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:13}
Uma pedra de curling é empurrada, liberada, desliza enquanto desacelera muito

\cref{thm:g10:inertia:principle} ou \cref{prop:g10:inertia:converse}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:14}
A sonda \emph{Viajante 1} navega através do espaço interestelar em
\qty{17}{km/s}, motores desligados. Por que não precisa de combustível para manter sua
velocidade, e o que Aristóteles preveria? Qual a distância que ele percorre em um
ano, em \unit{km} então em \omterm{def:g10:orders-of-magnitude:unit}{unidades} astronômicas ($\qty{1}{au} \approx
\qty{1.496e8}{km}$)? Por que ele deve disparar um propulsor até mesmo para mudar
direção \emph{sem} mudando a velocidade?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:inertia:15}
Três cordas puxam um anel \omterm{def:g10:relative-motion:frame}{em repouso}. Em uma grade onde um quadrado é
\qty{1}{N}, dois forças ler $\vect F_1\,(3, 2)$ e
$\vect F_2\,(-1, 2)$. Encontre o terceiro vigorcomponentes, magnitude e
direção.
\end{exercise}

\section{Problema: Inércia em três atos}

\begin{problem}[{Problema de fim de semana --- a passarela do aeroporto, a pista de gelo
e o pára-quedas: velocidade constante como assinatura de compensação
forças, e por que um pára-quedas não reduz o arrasto, mas a velocidade}]
\label{pb:g10:inertia:1}
Três cenas --- uma mala em uma esteira rolante, um disco em uma pista de gelo,
um pára-quedista no ar noturno --- e uma lei falando em todos os três.

\medskip
\noindent\textbf{Part I --- The airport walkway.}
UM \qty{23}{kg} mala está em uma passarela em movimento constante
velocidade \qty{0.75}{m/s}.
\begin{enumerate}
  \item Movimento da mala no chão? Inventariar o forças nisso.
  \item Faça isso forças \omterm{def:g10:inertia:compensate}{compensar}? Justificar; calcule ambas as magnitudes.
  \item Compare seu \omterm{def:g10:inertia:diagram}{diagrama de força} com a mesma mala está no
        chão. Que declaração profunda a comparação ilustra?
  \item A passarela para abruptamente. Descreva o que a mala faz,
        sem inventar um atacante \omterm{def:g10:inertia:force}{vigor}.
  \item Seu dono anda em \qty{1.2}{m/s} em relação à passarela. Ela
        velocidade do solo? Faça o forças sobre \emph{dela} \omterm{def:g10:inertia:compensate}{compensar}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The ice rink.}
\begin{enumerate}[resume]
  \item UM \qty{160}{g} disco desliza direto na constante
        \qty{8.0}{m/s}. Desenhe seu \omterm{def:g10:inertia:diagram}{diagrama de força} com magnitudes.
  \item Aristóteles afirma que o disco precisa de um motor. O que a pista
        resposta, e por que as superfícies cotidianas parecem estar do lado dele?
  \item O disco atravessa uma fase difícil e desacelera. Conclusão sobre o
        forças lá? Qual \omterm{def:g10:inertia:force}{vigor} é o culpado?
  \item Um patinador faz uma curva suave com velocidade constante. Faça o forças ligado
        ela \omterm{def:g10:inertia:compensate}{compensar}? Cuidado: que tipo de quantidade é a velocidade?
  \item Combine o diagrama com o movimento: (a) compensado forças; (b) um extra
        \omterm{def:g10:inertia:force}{vigor} oposto $\vect v $; (c) um extra \omterm{def:g10:inertia:force}{vigor} perpendicular a
        $\vect v $ --- (i) desaceleração em linha reta; (ii) girando em constante
        velocidade; (iii) \omterm{def:g10:inertia:uniform}{movimento reto e uniforme}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The parachute.}
A massa total de um saltador, incluindo o equipamento, é \qty{80}{kg}.
\begin{enumerate}[resume]
  \item Saindo do avião, a velocidade \omterm{def:g10:universal-gravitation:weight}{vertical} e o arrasto ainda são zero.
        Calcule o forças; eles \omterm{def:g10:inertia:compensate}{compensar}? O que deve acontecer a seguir?
  \item O arrasto cresce com a velocidade. Descreva a queda enquanto arrasta $<$ \omterm{def:g10:universal-gravitation:weight}{peso}.
  \item Antes que o velame se abra, o saltador atinge uma posição estável
        \qty{50}{m/s} (``velocidade terminal''). Calcule o arrasto. O que é
        o movimento, e por que é inércia disfarçada?
  \item O velame se abre: o arrasto agora excede em muito o \omterm{def:g10:universal-gravitation:weight}{peso}. O que
        acontece com a velocidade? (O saltador se move \emph{abaixo}
        por todo.)
  \item Uma nova constante \qty{5.0}{m/s} é alcançado. Calcule o arrasto;
        compare com a questão 13.
  \item A piada: o que o pára-quedas mudou --- o terminal
        arrasto, ou a velocidade com que o arrasto atinge o \omterm{def:g10:universal-gravitation:weight}{peso}?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Verdicts.}
\begin{enumerate}[resume]
  \item Complete e justifique: ``\omterm{def:g10:relative-motion:frame}{em repouso} ou em \omterm{def:g10:inertia:uniform}{movimento reto e uniforme}
        $\iff$ \dots''. Por que o descanso não merece uma lei separada?
  \item Em qual ato a expressão o movimento precisa de um motor de Aristóteles teria
        falhou de forma mais visível, e o que Galileu apontaria?
  \item Um roteiro de exame afirma: o paraquedista cai a uma velocidade constante
        velocidade, então não \omterm{def:g10:inertia:force}{vigor} age sobre ele''. Corrija em uma linha.
  \item Finale: para cada ato, nomeie o par de forças que \omterm{def:g10:inertia:compensate}{compensar},
        e indique a única lei usada vinte vezes neste problema.
\end{enumerate}
\end{problem}
