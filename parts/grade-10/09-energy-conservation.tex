\chapter{Energy: Forms and Conservation}\label{ch:g10:energy-conservation}

A moving car, a charged phone, a hot oven and a raised hammer share
one property: each can make something happen. Physics measures that
capacity with a single number --- \omterm{def:g10:energy-conservation:energy}{energy} --- and keeps its accounts
under one iron rule: the total never changes, it only moves and
changes costume. This chapter opens the ledger: the forms \omterm{def:g10:energy-conservation:energy}{energy}
takes, the chains it travels, and what a \omterm{def:g10:energy-conservation:kwh}{kilowatt-hour} buys.

\section{Energy: the common currency}

\begin{definition}[Energy and the joule]\label{def:g10:energy-conservation:energy}
\emph{Energy}\index{energy} measures a system's capacity to produce
changes: set matter in motion, lift it, heat it, light it. Its \omterm{def:g10:orders-of-magnitude:unit}{unit}
is the \emph{joule}\index{joule} (\unit{J}). Stored, transferred,
converted --- every branch of physics counts in this one currency.
\end{definition}

\begin{definition}[Kinetic energy]\label{def:g10:energy-conservation:kinetic}
A body of mass $m$ (in \unit{kg}) moving at speed $v$ (in
\unit{m/s}) carries the \emph{kinetic energy}\index{kinetic energy}
$E_k = \tfrac12\, m v^2$: doubling the mass doubles it, doubling
the speed quadruples it.
\end{definition}

\begin{example}[Car versus pedestrian]\label{ex:g10:energy-conservation:kinetic}
A \qty{1300}{kg} car at \qty{100}{km/h} $= \qty{27.8}{m/s}$ carries
$E_k = \tfrac12 \times 1300 \times 27.8^2 \approx \qty{5.0e5}{J}$;
a \qty{70}{kg} pedestrian at \qty{1.5}{m/s},
$\tfrac12 \times 70 \times 1.5^2 \approx \qty{79}{J}$ --- six
thousand times less. Speed is expensive: the square sees to it.
\end{example}

\begin{definition}[Gravitational potential energy]\label{def:g10:energy-conservation:potential}
A body of mass $m$ at height $h$ (in \unit{m}) above a chosen
reference level stores the \emph{gravitational potential
energy}\index{potential energy!gravitational} $E_p = m g h$, with
$g = \qty{9.81}{N/kg}$ --- valid near the ground, where $g$ is
effectively constant. Only \emph{differences} of height matter: the
level where $E_p = 0$ is chosen freely.
\end{definition}

\begin{example}[Water tower]\label{ex:g10:energy-conservation:potential}
Each cubic \omterm{def:g10:orders-of-magnitude:unit}{metre} (\qty{1000}{kg}) of water stored \qty{30}{m} up a
water tower holds $1000 \times 9.81 \times 30 \approx \qty{2.9e5}{J}$
--- about six tenths of the car's \omterm{def:g10:energy-conservation:kinetic}{kinetic energy} above. Cities
stockpile \omterm{def:g10:energy-conservation:energy}{energy} overhead.
\end{example}

\begin{definition}[The other forms]\label{def:g10:energy-conservation:forms}
\omterm{def:g10:energy-conservation:energy}{Energy} also hides in less visible costumes: \emph{thermal
energy}\index{thermal energy}, the disordered agitation of a body's
molecules (the hotter, the more); \emph{chemical
energy}\index{chemical energy}, stored in molecular bonds --- food,
wood, gasoline, batteries; \emph{electrical
energy}\index{electrical energy}, carried by currents from plant to
socket; \emph{radiative energy}\index{radiative energy}, carried by
light even across empty space --- how the Sun's \omterm{def:g10:energy-conservation:energy}{energy} reaches us;
\emph{nuclear energy}\index{nuclear energy}, stored inside atomic
nuclei, released in stars (\cref{ch:g12:nuclear-energy}).
\end{definition}

\section{Transfers and conversion chains}

\begin{definition}[Transfer, conversion, chain]\label{def:g10:energy-conservation:transfer}
An \emph{energy transfer}\index{energy transfer} moves \omterm{def:g10:energy-conservation:energy}{energy}
between systems --- \emph{reservoirs}\index{reservoir (of energy)};
an \emph{energy conversion}\index{energy conversion} changes its
form. A device doing either (turbine, motor, lamp, muscle) is a
\emph{converter}\index{converter}. A \emph{conversion
chain}\index{conversion chain} pictures the journey: boxes for
reservoirs and forms, labeled arrows for transfers.
\end{definition}

\begin{method}[Drawing a conversion chain]\label{met:g10:energy-conservation:chain}
\begin{enumerate}
  \item Identify the initial reservoir and form (what is consumed?).
  \item Follow the \omterm{def:g10:energy-conservation:energy}{energy} \omterm{def:g10:energy-conservation:transfer}{converter} by \omterm{def:g10:energy-conservation:transfer}{converter}: one arrow per
        transfer, the form named in each box.
  \item Add a heat arrow at every real \omterm{def:g10:energy-conservation:transfer}{converter} --- some \omterm{def:g10:energy-conservation:energy}{energy}
        always leaks as heat (\omterm{def:g10:inertia:inventory}{friction}, wires, exhaust).
\end{enumerate}
\end{method}

\section{Conservation of energy}

\begin{theorem}[Conservation of energy]\label{thm:g10:energy-conservation:conservation}
\index{conservation of energy}
The total \omterm{def:g10:energy-conservation:energy}{energy} of an isolated system --- one exchanging nothing
with the outside --- is constant. \omterm{def:g10:energy-conservation:energy}{Energy} is never created and never
destroyed: it only changes form and place, and what one reservoir
loses, the others gain, \omterm{def:g10:energy-conservation:energy}{joule} for \omterm{def:g10:energy-conservation:energy}{joule}.
\end{theorem}
\admitted

\begin{remark}\label{rem:g10:energy-conservation:admitted}
No experiment has ever caught this principle failing; here we take
it as given and use it numerically. \cref{ch:g12:work-and-energy}
derives its mechanical part ($E_k + E_p$) from Newton's laws; the
full accounting waits for the university volumes.
\end{remark}

\begin{example}[Free fall, priced in joules]\label{ex:g10:energy-conservation:fall}
A \qty{2.0}{kg} ball drops from \qty{10}{m} (air resistance
negligible): it starts with
$E_p = 2.0 \times 9.81 \times 10 \approx \qty{196}{J}$, lands with
$E_k = \qty{196}{J}$, so $v = \sqrt{2 \times 196/2.0} =
\qty{14}{m/s} \approx \qty{50}{km/h}$. No \omterm{def:g10:inertia:force}{forces}, no stopwatch:
conservation alone prices the impact.
\end{example}

\section{Efficiency}

\begin{definition}[Efficiency]\label{def:g10:energy-conservation:efficiency}
A \omterm{def:g10:energy-conservation:transfer}{converter} consumes $E_{\text{consumed}}$ and delivers the wanted
form $E_{\text{useful}}$; its \emph{efficiency}\index{efficiency} is
\[
\eta = \frac{E_{\text{useful}}}{E_{\text{consumed}}} < 1 .
\]
Conservation forbids $\eta > 1$, and real \omterm{def:g10:energy-conservation:transfer}{converters} never reach $1$:
the missing \omterm{def:g10:energy-conservation:energy}{energy} is not destroyed --- it leaves as heat.
\end{definition}

\begin{example}[Where the gasoline goes]\label{ex:g10:energy-conservation:engine}
A \omterm{def:g10:orders-of-magnitude:unit}{kilogram} of gasoline stores \qty{45}{MJ} of \omterm{def:g10:energy-conservation:forms}{chemical energy}; a
car engine converts it with $\eta \approx 0.30$: \qty{13.5}{MJ} of
motion, \qty{31.5}{MJ} heating the exhaust, the radiator and the
street. Two thirds of every tank warms the atmosphere.
\end{example}

\begin{remark}[Efficiencies multiply]\label{rem:g10:energy-conservation:chain-eta}
Chained \omterm{def:g10:energy-conservation:transfer}{converters} multiply their efficiencies: a \omterm{def:g10:energy-conservation:power}{power} plant
($0.38$) feeding the grid ($0.94$) feeding a charger ($0.85$)
delivers $\eta = 0.38 \times 0.94 \times 0.85 \approx 0.30$. Long
chains are leaky pipes: each arrow taxes the flow.
\end{remark}

\section{Power, the watt and the kilowatt-hour}

\begin{definition}[Power]\label{def:g10:energy-conservation:power}
\emph{Power}\index{power} is the rate at which \omterm{def:g10:energy-conservation:energy}{energy} flows,
$P = E/t$, measured in \emph{watts}\index{watt} (\unit{W}), with
$\qty{1}{W} = \qty{1}{J/s}$. \omterm{def:g10:energy-conservation:energy}{Energy} is an amount, power a flow ---
a bathtub versus its tap.
\end{definition}

\begin{example}[Kettle versus human]\label{ex:g10:energy-conservation:power}
A \qty{2200}{W} kettle running \qty{150}{s} uses
$E = 2200 \times 150 = \qty{3.3e5}{J}$. A human runs on about
\qty{10}{MJ} of food per day, an average
$\num{1.0e7}/\num{86400} \approx \qty{116}{W}$: you idle like a
bright old light bulb.
\end{example}

\begin{definition}[The kilowatt-hour]\label{def:g10:energy-conservation:kwh}
The \emph{kilowatt-hour}\index{kilowatt-hour} (\unit{kW.h}) is the
\omterm{def:g10:energy-conservation:energy}{energy} of a \qty{1}{kW} flow running for one hour:
$\qty{1}{kW.h} = 1000 \times 3600\,\unit{J} = \qty{3.6}{MJ}$ ---
the \omterm{def:g10:orders-of-magnitude:unit}{unit} electricity meters count: \omterm{def:g10:energy-conservation:power}{power} in \unit{kW} times hours.
\end{definition}

\begin{example}[Reading the bill]\label{ex:g10:energy-conservation:bill}
A bill is a subtraction and a multiplication: new meter reading
minus old gives the \omterm{def:g10:energy-conservation:kwh}{kilowatt-hours}; times the \omterm{def:g10:orders-of-magnitude:unit}{unit} price (about
$0.25$ euros per \unit{kW.h}), plus a fixed subscription, gives the
total. A \qty{2}{kW} heater running \qty{2.5}{h} nightly uses
\qty{5}{kW.h} a day: $38$ euros a month on its own.
\end{example}

\begin{example}[Orders of magnitude]\label{ex:g10:energy-conservation:orders}
Worth knowing by heart: lifting an apple one \omterm{def:g10:orders-of-magnitude:unit}{metre}, about \qty{1}{J};
a phone battery, $\qty{10}{W.h} \approx \qty{4e4}{J}$; a cereal
bowl, \qty{1.5}{MJ}; your food for a day, \qty{10}{MJ}; a \omterm{def:g10:orders-of-magnitude:unit}{kilogram}
of gasoline, \qty{45}{MJ} --- almost exactly a household's daily
electricity ($\approx \qty{12}{kW.h}$); a lightning bolt, \qty{5}{GJ}:
a hundred household-days, delivered in a flash.
\end{example}

\section{Problem: Powering a house for a day}

\begin{problem}[{Weekend problem --- from cereal bowl to rooftop
panels: a full energy audit of one ordinary day at home, and what a
joule costs depending on who sells it}]
\label{pb:g10:energy-conservation:1}
One ordinary Tuesday, a family sets out to find where its
electricity goes, armed with this chapter, the hallway meter and
the back of the bill. Tonight's verdict: their day equals one
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} of gasoline, sixteen square \omterm{def:g10:orders-of-magnitude:unit}{metres} of sunshine --- or five
cyclists pedaling around the clock.

\medskip
\noindent\textbf{Part I --- The audit.}
The day's inventory: fridge \qty{100}{W} for \qty{24}{h}; water
heater \qty{2400}{W} for \qty{2.5}{h}; oven \qty{2000}{W} for
\qty{1.0}{h}; washing machine \qty{500}{W} for \qty{2.0}{h}; lighting
\qty{60}{W} for \qty{5.0}{h}; screens \qty{150}{W} for \qty{6.0}{h}.
\begin{enumerate}
  \item Compute each appliance's \omterm{def:g10:energy-conservation:energy}{energy} for the day, in \unit{kW.h}.
  \item Total the day in \omterm{def:g10:energy-conservation:kwh}{kilowatt-hours}, then in megajoules;
        compare with one \omterm{def:g10:orders-of-magnitude:unit}{kilogram} of gasoline (\qty{45}{MJ}).
  \item Compute the house's average \omterm{def:g10:energy-conservation:power}{power} over the \qty{24}{h};
        what fraction of a running kettle is that?
  \item Which single appliance dominates? Recompute the daily total
        if the heater ran only \qty{1.25}{h}.
  \item Standby lights draw a permanent \qty{15}{W}: daily \omterm{def:g10:energy-conservation:energy}{energy},
