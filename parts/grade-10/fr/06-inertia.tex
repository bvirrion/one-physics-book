\chapter{Forces et le principe d'inertie}\label{ch:g10:inertia}

Arrêtez de pédaler et le vélo finit par s'arrêter~; arrêtez de tirer et
le traîneau reste immobile. L'expérience murmure que le mouvement doit
être nourri par un effort --- et pendant deux mille ans la physique a
été d'accord. Ce chapitre construit la \omterm{def:g10:inertia:force}{force}, une flèche mesurée en
\omterm{def:g10:inertia:force}{newtons}, et énonce la première grande loi de la mécanique~: laissé
vraiment seul, un corps en mouvement garde son mouvement, droit et
régulier, pour toujours.

\section{Modéliser une action par une force}

\begin{definition}[Force]\label{def:g10:inertia:force}
Une \emph{force}\index{force} modélise une action mécanique (une
poussée, une traction, une attraction) exercée sur un corps. Elle se
caractérise par~:
\begin{itemize}
  \item son \emph{point d'application}\index{point d'application}~;
  \item sa direction --- la droite le long de laquelle elle agit et le
        sens le long de cette droite~;
  \item son intensité, mesurée en \emph{newtons}\index{newton}
        (symbole \unit{N}).
\end{itemize}
Une force se dessine comme une flèche $\vect F$ (un vecteur, comme dans
le volume de mathématiques) depuis son point d'application, de longueur
proportionnelle à l'intensité à une échelle indiquée.
\end{definition}

\begin{example}[Le poids d'un cartable]\label{ex:g10:inertia:schoolbag}
Un cartable de masse $m = \qty{4.0}{kg}$ est tiré vers le bas par la
Terre avec son \omterm{def:g10:universal-gravitation:weight}{poids}
(\cref{ch:g10:universal-gravitation})~:
$P = mg = 4.0 \times 9.81 \approx \qty{39}{N}$ --- un \omterm{def:g10:inertia:force}{newton}
est environ le \omterm{def:g10:universal-gravitation:weight}{poids} d'une pomme. À l'échelle
$\qty{1}{cm} \leftrightarrow \qty{10}{N}$~: une flèche verticale longue
de \qty{3.9}{cm}, appliquée au centre de gravité du cartable, pointant
vers le bas.
\end{example}

\section{Un inventaire des forces}

\begin{definition}[Forces de contact et à distance]\label{def:g10:inertia:contact}
Une \emph{force de contact}\index{force de contact} s'exerce aux points
où deux corps se touchent. Une \emph{force à distance}\index{force à
distance} n'a besoin d'aucun contact~: le \omterm{def:g10:universal-gravitation:weight}{poids} --- la traction
gravitationnelle du~\cref{ch:g10:universal-gravitation} --- est la seule
utilisée dans ce chapitre.
\end{definition}

\begin{definition}[Les forces de contact courantes]\label{def:g10:inertia:inventory}
Quatre \omterm{def:g10:inertia:contact}{forces de contact} couvrent la plupart des situations
quotidiennes~:
\begin{itemize}
  \item la \emph{réaction normale}\index{réaction normale} $\vect R$
        d'un support, perpendiculaire à sa surface, poussant en
        s'éloignant d'elle~;
  \item la \emph{tension}\index{tension} $\vect T$ d'un fil, d'une
        corde ou d'une chaîne, dirigée le long d'elle, tirant vers
        elle~;
  \item le \emph{frottement}\index{frottement} $\vect f$ d'une
        surface, de l'air ou de l'eau, s'opposant au glissement ou au
        mouvement~;
  \item la \emph{poussée}\index{poussée} d'un moteur, d'un jet ou d'une
        main, poussant le long de l'effort.
\end{itemize}
\end{definition}

\begin{example}[Inventaire sur un traîneau tracté]\label{ex:g10:inertia:sled}
Un traîneau tracté sur la neige~: \omterm{def:g10:universal-gravitation:weight}{poids} $\vect P$ (à distance),
\omterm{def:g10:inertia:inventory}{réaction normale} $\vect R$ de la neige, \omterm{def:g10:inertia:inventory}{tension}
$\vect T$ de la corde, \omterm{def:g10:inertia:inventory}{frottement} $\vect f$ s'opposant au
glissement. Rien d'autre ne touche le traîneau~: l'inventaire est
complet --- \emph{\omterm{def:g10:universal-gravitation:weight}{poids} toujours, puis une
\omterm{def:g10:inertia:contact}{force de contact} par objet qui touche}.
\end{example}

\section{Le principe d'inertie}

\begin{definition}[Mouvement rectiligne uniforme]\label{def:g10:inertia:uniform}
Un corps est en \emph{mouvement rectiligne uniforme}\index{mouvement
rectiligne uniforme} dans un référentiel lorsque sa
\omterm{def:g10:relative-motion:trajectory}{trajectoire} est une droite parcourue à vitesse
constante~: son vecteur vitesse $\vect v$ est le même à chaque
instant. Le repos est le cas particulier $\vect v = \vect 0$.
\end{definition}

\begin{definition}[Forces qui se compensent]\label{def:g10:inertia:compensate}
Des \omterm{def:g10:inertia:force}{forces} agissant sur un corps se
\emph{compensent}\index{forces compensées} lorsque leur effet combiné
est nul~: bout à bout, leurs flèches reviennent au départ. Deux
\omterm{def:g10:inertia:force}{forces} se compensent exactement lorsqu'elles partagent la même
droite d'action avec des intensités égales et des sens opposés.
\end{definition}

\begin{theorem}[Principe d'inertie]\label{thm:g10:inertia:principle}
\index{principe d'inertie}\index{inertie}
Dans le \omterm{def:g10:relative-motion:frames}{référentiel du sol}, un corps soumis à aucune
\omterm{def:g10:inertia:force}{force}, ou à des \omterm{def:g10:inertia:force}{forces} qui se
\omterm{def:g10:inertia:compensate}{compensent}, reste au repos ou garde un
\omterm{def:g10:inertia:uniform}{mouvement rectiligne uniforme}~; inversement, un corps au
repos ou en \omterm{def:g10:inertia:uniform}{mouvement rectiligne uniforme} est soumis à des
\omterm{def:g10:inertia:compensate}{forces compensées} (ou à aucune).
\end{theorem}
\admitted

\begin{remark}
Aucune expérience ne peut pleinement isoler un corps, donc le principe
est un postulat --- entrevu par Galilée, énoncé par Newton. Un chapitre
ultérieur le promeut en une loi reliant les \omterm{def:g10:inertia:force}{forces} aux
\emph{changements} de vitesse~; le volume de 1\textsuperscript{re}
année demande dans quels référentiels il tient réellement.
\end{remark}

\begin{proposition}[Lire le mouvement depuis les forces, et inversement]\label{prop:g10:inertia:converse}
Dans le \omterm{def:g10:relative-motion:frames}{référentiel du sol}~:
\begin{enumerate}
  \item si la vitesse d'un corps change --- en intensité \emph{ou} en
        direction --- les \omterm{def:g10:inertia:force}{forces} sur lui ne se
        \omterm{def:g10:inertia:compensate}{compensent} \emph{pas}~;
  \item si un corps est au repos ou en \omterm{def:g10:inertia:uniform}{mouvement rectiligne
        uniforme}, ses \omterm{def:g10:inertia:force}{forces} inventoriées doivent s'annuler ---
        déterminant les inconnues.
\end{enumerate}
\end{proposition}

\begin{proof}
La première affirmation est la contraposée
du~\cref{thm:g10:inertia:principle}, la seconde sa partie réciproque
appliquée à un inventaire achevé.
\end{proof}

\begin{example}[Le palet de hockey]\label{ex:g10:inertia:puck}
Lâché par le bâton, un palet traverse \qty{30}{m} de glace lisse en
\qty{2.5}{s}, en ligne droite à la vitesse constante
$v = 30/2.5 = \qty{12}{m/s}$. Inventaire~: \omterm{def:g10:universal-gravitation:weight}{poids} et
\omterm{def:g10:inertia:inventory}{réaction normale} (\omterm{def:g10:inertia:inventory}{frottement} négligeable).
Ils se \omterm{def:g10:inertia:compensate}{compensent}, et le principe prédit exactement le
mouvement observé~: rien ne pousse le palet en avant --- et rien n'a
besoin de le faire.
\end{example}

\begin{remark}[Galilée contre Aristote]
Aristote enseignait que tout mouvement a besoin d'un moteur, et la vie
quotidienne --- le \omterm{def:g10:inertia:inventory}{frottement} dans chaque contact --- semble
être d'accord. Galilée idéalisait le \omterm{def:g10:inertia:inventory}{frottement} hors de
l'image~: des surfaces de plus en plus lisses ont besoin d'une poussée
de plus en plus petite pour \emph{maintenir} le mouvement. La
persistance, non le repos, est l'état naturel.
\end{remark}

\begin{remark}[Le passager qui bascule]
Lorsqu'un bus freine brutalement, les passagers debout basculent en
avant. Aucune \omterm{def:g10:inertia:force}{force} ne les jette~: par inertie leurs corps
gardent leur vitesse tandis que le bus perd la sienne. La ceinture de
sécurité fournit la \omterm{def:g10:inertia:force}{force} vers l'arrière qui manque à
l'inventaire.
\end{remark}

\section{Forces équilibrées~: la statique hors d'un dessin}

\begin{definition}[Diagramme des forces]\label{def:g10:inertia:diagram}
Le \emph{diagramme des forces}\index{diagramme des forces} d'un corps
montre le corps seul, avec chaque \omterm{def:g10:inertia:force}{force} agissant \emph{sur}
lui dessinée à l'échelle depuis son \omterm{def:g10:inertia:force}{point d'application}~; les
\omterm{def:g10:inertia:force}{forces} exercées \emph{par} le corps n'apparaissent jamais.
\end{definition}

\begin{method}[Analyser une situation statique]\label{met:g10:inertia:method}
\begin{enumerate}
  \item Choisir le corps à étudier~; travailler dans le
        \omterm{def:g10:relative-motion:frames}{référentiel du sol}.
  \item Inventaire~: \omterm{def:g10:universal-gravitation:weight}{poids} toujours, une
        \omterm{def:g10:inertia:contact}{force de contact} par contact.
  \item Dessiner le \omterm{def:g10:inertia:diagram}{diagramme des forces}.
  \item Le corps est au repos, donc les \omterm{def:g10:inertia:force}{forces} se
        \omterm{def:g10:inertia:compensate}{compensent}
        (\cref{thm:g10:inertia:principle})~: lire les intensités
        inconnues sur le dessin (longueurs égales en 1D, composantes en
        2D).
\end{enumerate}
\end{method}

\begin{example}[Le livre sur la table]\label{ex:g10:inertia:book}
Un livre de masse \qty{0.50}{kg} repose sur une table~:
\omterm{def:g10:universal-gravitation:weight}{poids} $P = 0.50 \times 9.81 \approx \qty{4.9}{N}$ et
\omterm{def:g10:inertia:inventory}{réaction normale} $\vect R$. Au repos, $\vect R$
\omterm{def:g10:inertia:compensate}{compense} $\vect P$~: verticale, vers le haut,
$R = \qty{4.9}{N}$. Ajoutez un second livre et $R$ s'ajuste --- un
support pousse exactement aussi fort que nécessaire.
\end{example}

\begin{example}[La lampe sur deux fils]\label{ex:g10:inertia:lamp}
Une lampe de masse \qty{3.0}{kg} ($P \approx \qty{29.4}{N}$) pend à
deux fils, chacun à $45^\circ$ de la verticale. Par symétrie les
\omterm{def:g10:inertia:inventory}{tensions} sont égales ($T_1 = T_2 = T$) et leurs composantes
horizontales s'annulent~; les composantes verticales, chacune
$T \cos 45^\circ$, portent le \omterm{def:g10:universal-gravitation:weight}{poids}~:
\[
2\,T \cos 45^\circ = P
\qquad\text{donc}\qquad
T = \frac{29.4}{2 \times 0.707} \approx \qty{21}{N}
\]
--- plus de la moitié du \omterm{def:g10:universal-gravitation:weight}{poids} par fil~: tirer en biais coûte
de la \omterm{def:g10:inertia:inventory}{tension}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:inertia:1}
Inventorier et classer (contact ou à distance) les
\omterm{def:g10:inertia:force}{forces} sur~: (a) un livre au repos sur une table~; (b) une
lampe suspendue à son fil~; (c) un palet glissant sur de la glace
lisse~; (d) une pomme qui tombe (résistance de l'air négligeable).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:2}
Calculer le \omterm{def:g10:universal-gravitation:weight}{poids} d'un livre de \qty{250}{g} et d'une
personne de \qty{60}{kg}. À l'échelle
$\qty{1}{cm} \leftrightarrow \qty{1}{N}$, quelle longueur a la flèche
du livre, et pourquoi l'échelle est-elle désespérée pour la personne~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:3}
Sur un diagramme à l'échelle $\qty{1}{cm} \leftrightarrow \qty{2}{N}$,
une \omterm{def:g10:inertia:force}{force} est une flèche horizontale longue de \qty{3.5}{cm}
pointant vers la gauche, appliquée en un point $A$. Donner ses trois
caractéristiques.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:4}
Un dictionnaire de masse \qty{1.2}{kg} repose sur une étagère. Dessiner
son \omterm{def:g10:inertia:diagram}{diagramme des forces} et donner l'intensité et la
direction de chaque \omterm{def:g10:inertia:force}{force}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:5}
Vrai ou faux --- corriger les faux~: (a) un corps finit toujours par
s'arrêter à moins qu'une \omterm{def:g10:inertia:force}{force} continue de le pousser~;
(b) le \omterm{def:g10:inertia:uniform}{mouvement rectiligne uniforme} n'exige aucune
\omterm{def:g10:inertia:force}{force} non compensée~; (c) un corps au repos n'est soumis à
aucune \omterm{def:g10:inertia:force}{force}~; (d) si la vitesse d'un corps change, les
\omterm{def:g10:inertia:force}{forces} sur lui ne se \omterm{def:g10:inertia:compensate}{compensent} pas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:6}
Après le lâcher, un palet couvre \qty{24}{m} de glace en \qty{3.0}{s},
en ligne droite à vitesse constante. Calculer sa vitesse~; les
\omterm{def:g10:inertia:force}{forces} sur lui se \omterm{def:g10:inertia:compensate}{compensent}-elles~? Sur le
béton le même palet s'arrête en quelques \omterm{def:g10:orders-of-magnitude:unit}{mètres}~: qu'est-ce
qui a changé dans l'inventaire, et que conclut
le~\cref{prop:g10:inertia:converse}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:7}
Un passager de \qty{70}{kg} se tient debout dans un ascenseur montant
à la vitesse constante \qty{1.5}{m/s}. Lister les
\omterm{def:g10:inertia:force}{forces} sur le passager~; se \omterm{def:g10:inertia:compensate}{compensent}-elles~?
Donner l'intensité de la \omterm{def:g10:inertia:force}{force} du plancher. Quelque chose
changerait-il avec l'ascenseur au repos~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:8}
Une lampe de masse \qty{1.8}{kg} pend au repos à un seul fil vertical.
Calculer la \omterm{def:g10:inertia:inventory}{tension} du fil. Le fil est remplacé par deux
fils \emph{verticaux} partageant la charge également~: quelle est la
\omterm{def:g10:inertia:inventory}{tension} de chacun~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:9}
Un bus freine brutalement et un passager debout bascule en avant~; dans
un virage à vitesse constante, le même passager penche vers
l'extérieur. Expliquer les deux sans inventer de \omterm{def:g10:inertia:force}{force}
«~vers l'avant~» ou «~vers l'extérieur~».
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:10}
Un parachutiste de masse totale \qty{85}{kg} descend
\omterm{def:g10:universal-gravitation:weight}{verticalement} à la vitesse constante \qty{5.0}{m/s}. Dessiner
le \omterm{def:g10:inertia:diagram}{diagramme des forces} et calculer la résistance de l'air
sur la voile.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:11}
Un skieur nautique est tracté en ligne droite à vitesse constante~; la
corde horizontale tire avec \qty{300}{N}. Quelle est la traînée
horizontale de l'eau sur les skis (justifier)~? Quelles deux
\omterm{def:g10:inertia:force}{forces} verticales se \omterm{def:g10:inertia:compensate}{compensent} l'une l'autre~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:12}
Une lampe de masse \qty{2.4}{kg} pend à deux fils à $40^\circ$ de la
verticale ($\cos 40^\circ \approx 0.766$). Calculer chaque
\omterm{def:g10:inertia:inventory}{tension}. Pourquoi la \omterm{def:g10:inertia:inventory}{tension} croît-elle lorsque
les fils s'approchent de l'horizontale, et pourquoi deux fils
exactement horizontaux ne pourraient-ils jamais tenir la lampe~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:13}
Une pierre de curling est poussée, lâchée, glisse en ralentissant très
graduellement, puis repose. Pour chaque phase (poussée, glisse,
repos), dessiner le \omterm{def:g10:inertia:diagram}{diagramme des forces} et indiquer si les
\omterm{def:g10:inertia:force}{forces} se \omterm{def:g10:inertia:compensate}{compensent}, en citant
le~\cref{thm:g10:inertia:principle} ou
le~\cref{prop:g10:inertia:converse}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:14}
La sonde \emph{Voyager 1} côtoie l'espace interstellaire à
\qty{17}{km/s}, moteurs éteints. Pourquoi n'a-t-elle besoin d'aucun
carburant pour garder sa vitesse, et que prédirait Aristote~?
Quelle distance parcourt-elle en un an, en \unit{km} puis en
\omterm{def:g10:orders-of-magnitude:lightyear}{unités astronomiques} ($\qty{1}{au} \approx
\qty{1.496e8}{km}$)~? Pourquoi doit-elle allumer un propulseur même
pour changer de direction \emph{sans} changer de vitesse~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:15}
Trois cordes tirent sur un anneau au repos. Sur une grille où un carré
fait \qty{1}{N}, deux \omterm{def:g10:inertia:force}{forces} se lisent
$\vect F_1\,(3, 2)$ et $\vect F_2\,(-1, 2)$. Trouver les composantes,
l'intensité et la direction de la troisième \omterm{def:g10:inertia:force}{force}.
\end{exercise}

\section{Problème~: L'inertie en trois actes}

\begin{problem}[{Problème du week-end --- le tapis d'aéroport, la
patinoire et le parachute~: la vitesse constante comme signature des
forces compensées, et pourquoi un parachute ne réduit pas la traînée
mais la vitesse}]
\label{pb:g10:inertia:1}
Trois scènes --- une valise sur un tapis roulant, un palet sur une
patinoire, un parachutiste dans l'air du soir --- et une seule loi qui
parle dans les trois.

\medskip
\noindent\textbf{Partie I --- Le tapis d'aéroport.}
Une valise de \qty{23}{kg} se tient sur un tapis se déplaçant à la
vitesse constante \qty{0.75}{m/s}.
\begin{enumerate}
  \item Mouvement de la valise dans le référentiel du sol~?
        Inventorier les \omterm{def:g10:inertia:force}{forces} sur elle.
  \item Ces \omterm{def:g10:inertia:force}{forces} se \omterm{def:g10:inertia:compensate}{compensent}-elles~?
        Justifier~; calculer les deux intensités.
  \item Comparer son \omterm{def:g10:inertia:diagram}{diagramme des forces} à celui de la même
        valise sur le sol. Quelle affirmation profonde la comparaison
        illustre-t-elle~?
  \item Le tapis s'arrête d'un coup. Décrire ce que fait la valise,
        sans inventer de \omterm{def:g10:inertia:force}{force} vers l'avant.
  \item Sa propriétaire marche à \qty{1.2}{m/s} par rapport au tapis.
        Sa vitesse dans le référentiel du sol~? Les
        \omterm{def:g10:inertia:force}{forces} sur \emph{elle} se
        \omterm{def:g10:inertia:compensate}{compensent}-elles~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- La patinoire.}
\begin{enumerate}[resume]
  \item Un palet de \qty{160}{g} glisse en ligne droite à la vitesse
        constante \qty{8.0}{m/s}. Dessiner son
        \omterm{def:g10:inertia:diagram}{diagramme des forces} avec les intensités.
  \item Aristote prétend que le palet a besoin d'un moteur. Que répond
        la patinoire, et pourquoi les surfaces du quotidien semblent-elles
        se ranger de son côté~?
  \item Le palet traverse une zone rugueuse et ralentit. Conclusion sur
        les \omterm{def:g10:inertia:force}{forces} là~? Quelle \omterm{def:g10:inertia:force}{force} est la
        coupable~?
  \item Un patineur courbe doucement à vitesse constante. Les
        \omterm{def:g10:inertia:force}{forces} sur lui se \omterm{def:g10:inertia:compensate}{compensent}-elles~?
        Attention~: de quelle nature est la vitesse~?
  \item Apparier diagramme et mouvement~: (a) \omterm{def:g10:inertia:force}{forces}
        compensées~; (b) une \omterm{def:g10:inertia:force}{force} supplémentaire opposée à
        $\vect v$~; (c) une \omterm{def:g10:inertia:force}{force} supplémentaire perpendiculaire
        à $\vect v$ --- (i) ralentissement en ligne droite~; (ii) virage
        à vitesse constante~; (iii) \omterm{def:g10:inertia:uniform}{mouvement rectiligne
        uniforme}.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le parachute.}
La masse totale d'un sauteur, équipement compris, est \qty{80}{kg}.
\begin{enumerate}[resume]
  \item En quittant l'avion, la vitesse verticale et la traînée sont
        encore nulles. Calculer les \omterm{def:g10:inertia:force}{forces}~; se
        \omterm{def:g10:inertia:compensate}{compensent}-elles~? Que doit-il se passer ensuite~?
  \item La traînée croît avec la vitesse. Décrire la chute tant que la
        traînée $<$ le \omterm{def:g10:universal-gravitation:weight}{poids}.
  \item Avant l'ouverture de la voile, le sauteur atteint une vitesse
        stable de \qty{50}{m/s} («~vitesse limite~»). Calculer la
        traînée. Quel est le mouvement, et pourquoi est-ce «~l'inertie
        déguisée~»~?
  \item La voile s'ouvre~: la traînée dépasse maintenant largement le
        \omterm{def:g10:universal-gravitation:weight}{poids}. Que devient la vitesse~? (Le sauteur se
        déplace \emph{vers le bas} tout du long.)
  \item Une nouvelle vitesse stable de \qty{5.0}{m/s} est atteinte.
        Calculer la traînée~; comparer à la question~13.
  \item La chute. Qu'a changé le parachute --- la traînée limite, ou la
        vitesse à laquelle la traînée atteint le
        \omterm{def:g10:universal-gravitation:weight}{poids}~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Verdicts.}
\begin{enumerate}[resume]
  \item Compléter et justifier~: «~au repos ou en
        \omterm{def:g10:inertia:uniform}{mouvement rectiligne uniforme} $\iff$ \dots~».
        Pourquoi le repos ne mérite-t-il aucune loi séparée~?
  \item Dans quel acte le «~le mouvement a besoin d'un moteur~»
        d'Aristote aurait-il le plus visiblement échoué, et que
        montrerait Galilée~?
  \item Une copie d'examen affirme~: «~le parachutiste tombe à vitesse
        constante, donc aucune \omterm{def:g10:inertia:force}{force} n'agit sur lui~». La
        corriger en une ligne.
  \item Final~: pour chaque acte, nommer la paire de
        \omterm{def:g10:inertia:force}{forces} qui se \omterm{def:g10:inertia:compensate}{compensent}, et
        énoncer la seule loi utilisée vingt fois dans ce problème.
\end{enumerate}
\end{problem}
