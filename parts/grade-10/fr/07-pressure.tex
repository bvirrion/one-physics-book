\chapter{Pression~: du sport à la plongée}\label{ch:g10:pressure}

Un skieur glisse sur une neige dans laquelle un marcheur s'enfonce~; une
poussée de \qty{5}{N} enfonce une aiguille dans du cuir qu'un poing ne
peut entamer. Mêmes \omterm{def:g10:inertia:force}{forces}, effets sauvagement différents~: la
grandeur manquante est la \emph{\omterm{def:g10:pressure:pressure}{pression}}, la
\omterm{def:g10:inertia:force}{force} divisée par l'aire qui la reçoit. Ce chapitre la
définit, la suit au fond d'une piscine, le long d'un baromètre, dans une
seringue comprimée --- et s'achève à quarante \omterm{def:g10:orders-of-magnitude:unit}{mètres} sous la
mer, où elle devient vitale.

\section{Pression~: une force répartie sur une surface}

\begin{definition}[Pression]\label{def:g10:pressure:pressure}
Lorsqu'une \omterm{def:g10:inertia:force}{force} d'intensité $F$ (en \unit{N}) appuie
perpendiculairement sur une surface d'aire $S$ (en \unit{m^2}), la
\emph{pression}\index{pression} exercée sur cette surface est
\[
P = \frac{F}{S}.
\]
Son \omterm{def:g10:orders-of-magnitude:unit}{unité}, le \unit{N/m^2}, est le \emph{pascal}\index{pascal}
(\unit{Pa}) --- une \omterm{def:g10:orders-of-magnitude:unit}{unité} minuscule, le
\omterm{def:g10:universal-gravitation:weight}{poids} d'une pomme réparti sur un
\omterm{def:g10:orders-of-magnitude:unit}{mètre} carré, donc les \unit{kPa} et \unit{MPa} sont les
\omterm{def:g10:orders-of-magnitude:unit}{unités} de travail.
\end{definition}

\begin{example}[Talon contre raquette à neige]\label{ex:g10:pressure:snowshoe}
Une personne de \qty{60}{kg} pèse $F = 60 \times 9.81 \approx
\qty{590}{N}$. Sur un talon aiguille de
\qty{1.0}{cm^2} = \qty{1.0e-4}{m^2}~:
$P = 590/\num{1.0e-4} \approx \qty{5.9e6}{Pa} = \qty{5.9}{MPa}$. Sur
deux raquettes de \qty{0.20}{m^2} chacune~:
$P = 590/0.40 \approx \qty{1.5}{kPa}$. Le \emph{même}
\omterm{def:g10:universal-gravitation:weight}{poids} appuie $4000$~fois plus fort sous le talon~: les
talons marquent les planchers de bois, les raquettes flottent sur la
poudreuse.
\end{example}

\begin{remark}[Concentrer ou étaler]\label{rem:g10:pressure:tools}
Une aiguille à coudre poussée avec \qty{5}{N} sur une pointe de
\qty{0.01}{mm^2} = \qty{1e-8}{m^2} exerce
$\qty{5e8}{Pa} = \qty{500}{MPa}$ --- assez pour séparer les fibres du
cuir. Les outils de percement concentrent la \omterm{def:g10:inertia:force}{force}~; les
conceptions porteuses (skis, chenilles, fondations) l'étalent. Le sport
est de la gestion de \omterm{def:g10:pressure:pressure}{pression} appliquée~: les patins mordent
dans la glace, les skis glissent sur la neige.
\end{remark}

\section{Pression dans un liquide au repos}

Pourquoi l'eau presse-t-elle plus fort plus bas~? Isoler une colonne
verticale imaginaire d'eau de section $S$, de la surface à la profondeur
$h$~: volume $Sh$, masse $\rho S h$ ($\rho$ la masse volumique du
liquide, en \unit{kg/m^3}), \omterm{def:g10:universal-gravitation:weight}{poids} $\rho S h g$. Le bas de la
colonne supporte ce \omterm{def:g10:universal-gravitation:weight}{poids} \emph{plus} la poussée
atmosphérique $P_0 S$ sur le sommet. En divisant la
\omterm{def:g10:inertia:force}{force} totale par $S$~: la \omterm{def:g10:pressure:pressure}{pression} à la
profondeur $h$ devrait être $P_0 + \rho g h$ --- et elle l'est, partout
dans le liquide.

\begin{theorem}[Pression en profondeur]\label{thm:g10:pressure:depth}
Dans un liquide de masse volumique $\rho$ au repos, ouvert à l'air, la
\omterm{def:g10:pressure:pressure}{pression} à la profondeur $h$ sous la surface est
\[
P = P_0 + \rho g h,
\]
où $P_0$ est la \omterm{def:g10:pressure:atmospheric}{pression atmosphérique} à la surface. La
\omterm{def:g10:pressure:pressure}{pression} ne dépend que de la profondeur, non de la forme du
récipient, et en un point donné le liquide presse également fort dans
toutes les directions.
\end{theorem}
\admitted

\begin{remark}
L'argument de la colonne ne rend la loi que plausible~; la dérivation
honnête (toute forme, toutes directions) est dans le volume de
1\textsuperscript{re} année. Le terme propre de l'eau $\rho g h$ est la
\omterm{def:g10:pressure:pressure}{pression} \emph{relative}\index{pression relative} --- ce qu'affiche
un \omterm{thm:g10:pressure:depth}{manomètre} de plongeur.
\end{remark}

\begin{example}[Dix mètres d'eau]\label{ex:g10:pressure:tenmetres}
Dans l'eau douce ($\rho = \qty{1000}{kg/m^3}$), à $h = \qty{10}{m}$~:
$\rho g h = 1000 \times 9.81 \times 10 \approx \qty{98}{kPa}$, donc
$P = 101 + 98 \approx \qty{199}{kPa}$. Dix
\omterm{def:g10:orders-of-magnitude:unit}{mètres} d'eau ajoutent presque exactement une
atmosphère~: la règle empirique du plongeur, \emph{une atmosphère
supplémentaire tous les dix \omterm{def:g10:orders-of-magnitude:unit}{mètres}}.
\end{example}

\section{L'océan d'air}

\begin{definition}[Pression atmosphérique]\label{def:g10:pressure:atmospheric}
Nous vivons au fond d'un océan d'air, dont le
\omterm{def:g10:universal-gravitation:weight}{poids} crée la \emph{pression
atmosphérique}\index{pression atmosphérique}, au niveau de la mer
\[
P_0 \approx \qty{101}{kPa} = \qty{1.01e5}{Pa}.
\]
Sur chaque centimètre carré cela fait
$F = \num{1.01e5} \times \num{1e-4} \approx \qty{10}{N}$~: le
\omterm{def:g10:universal-gravitation:weight}{poids} d'une masse d'un \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} sur
chaque \unit{cm^2} de votre peau, de votre bureau, de tout.
\end{definition}

\begin{example}[La charge invisible]\label{ex:g10:pressure:sheet}
Une feuille A4 (\qty{21.0}{cm} $\times$ \qty{29.7}{cm}, donc
$S \approx \qty{0.0624}{m^2}$) reçoit de l'air au-dessus
$F = \num{1.01e5} \times 0.0624 \approx \qty{6.3e3}{N}$ --- le
\omterm{def:g10:universal-gravitation:weight}{poids} d'une petite voiture. Rien ne se déchire~: l'air
en dessous pousse \emph{vers le haut} tout aussi fort. Nous ne
remarquons l'atmosphère que lorsqu'un côté la perd --- une ventouse, une
paille, une cabine d'avion.
\end{example}

\begin{remark}[Le baromètre de Torricelli]\label{rem:g10:pressure:torricelli}
En 1643 Torricelli renversa un tube scellé de mercure sur un bain de
mercure. La colonne tomba à la hauteur où $\rho g h = P_0$~:
$h = \num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$, laissant
au-dessus le premier vide jamais réalisé. La hauteur suit le temps et
diminue avec l'altitude~: un baromètre pèse l'air au-dessus de vous
(l'eau aurait besoin de \qty{10.3}{m}~: \cref{exo:g10:pressure:9}).
\end{remark}

\section{Gaz~: bombardement et loi de Boyle--Mariotte}

Un gaz presse aussi sur son contenant~: c'est un essaim de molécules en
mouvement aléatoire incessant, chaque collision donne à la paroi une
petite poussée vers l'extérieur, et des milliards de milliards
d'impacts par seconde se fondent en la \omterm{def:g10:inertia:force}{force} stable que
nous mesurons comme \omterm{def:g10:pressure:pressure}{pression} du gaz. Comprimer le gaz
dans la moitié du volume~: les molécules frappent deux fois plus
souvent, donc la \omterm{def:g10:pressure:pressure}{pression} devrait doubler. Elle le fait.

\begin{proposition}[Loi de Boyle--Mariotte]\label{prop:g10:pressure:boyle}
\index{loi de Boyle--Mariotte}
Pour une quantité fixe de gaz maintenue à température constante, la
\omterm{def:g10:pressure:pressure}{pression} et le volume sont inversement proportionnels~:
\[
P \, V = \text{constante},
\qquad\text{c'est-à-dire}\qquad
P_1 V_1 = P_2 V_2 .
\]
\end{proposition}
\admitted

\begin{remark}
Un fait expérimental à ce niveau --- le volume de 1\textsuperscript{re}
année le dérive de l'image des collisions. L'expérience (une seringue
scellée d'air, un capteur de \omterm{def:g10:pressure:pressure}{pression},
\qty{20}{\degreeCelsius})~:
\begin{center}
\begin{tabular}{lccccc}
$V$ (\unit{mL}) & 60 & 50 & 40 & 30 & 20 \\
$P$ (\unit{kPa}) & 100 & 120 & 150 & 200 & 300 \\
$P \times V$ (\unit{kPa.mL}) & 6000 & 6000 & 6000 & 6000 & 6000
\end{tabular}
\end{center}
\end{remark}

\begin{example}[Une seringue comprimée]\label{ex:g10:pressure:boyle}
L'air occupe \qty{4.0}{L} à \qty{100}{kPa}~; comprimé lentement (pour
que la température reste constante) à \qty{250}{kPa}, il occupe
$V_2 = P_1 V_1 / P_2 = 400/250 = \qty{1.6}{L}$.
\end{example}

\section{Plongée~: la pression au travail}

\begin{method}[Pression absolue en profondeur]\label{met:g10:pressure:dive}
Pour un plongeur à la profondeur $h$ en mer
($\rho \approx \qty{1025}{kg/m^3}$, donc
$\rho g h \approx \qty{101}{kPa}$ par \qty{10}{m}), calculer
$P = P_0 + \rho g h$ --- et dans la \omterm{prop:g10:pressure:boyle}{loi de
Boyle--Mariotte} toujours utiliser cette \omterm{def:g10:pressure:pressure}{pression}
\emph{absolue}\index{pression absolue}, jamais le terme de
\omterm{thm:g10:pressure:depth}{manomètre} seul. Un plongeur se trouve sous $2$
atmosphères à \qty{10}{m}, $3$ à \qty{20}{m}, $5$ à \qty{40}{m}.
\end{method}

\begin{remark}[Ne jamais bloquer sa respiration]\label{rem:g10:pressure:breath}
Un détendeur livre de l'air à la \omterm{def:g10:pressure:pressure}{pression} ambiante~: les
poumons se remplissent normalement à toute profondeur. Mais remplir
\qty{6.0}{L} à \qty{20}{m} ($\qty{302}{kPa}$), bloquer cette
respiration et faire surface~: Boyle exige
$6.0 \times 302/101 \approx \qty{18}{L}$ --- trois fois ce qu'une
poitrine contient, et les poumons se déchirent bien avant, même sur
quelques \omterm{def:g10:orders-of-magnitude:unit}{mètres} de montée. D'où la première règle de la
plongée bouteille~: \emph{ne jamais bloquer sa respiration en
remontant} --- continuer de respirer et l'air qui se dilate s'échappe
simplement. Un apnéiste est en sécurité~: son air a été pris en surface,
donc en remontant il ne se redilate qu'à son volume d'origine.
\end{remark}

\begin{remark}[Décompression]\label{rem:g10:pressure:deco}
Respirer de l'air à \omterm{def:g10:pressure:pressure}{haute pression} dissout aussi de l'azote
supplémentaire dans le sang, comme le gaz dans une bouteille de soda
bouchée~; remonter trop vite et il pétille en bulles dans le corps ---
l'accident de décompression. D'où la remontée lente (environ
\qty{10}{m} par minute) et les arrêts peu profonds, qui laissent le gaz
s'échapper calmement par les poumons. La théorie attend le volume de
1\textsuperscript{re} année~; la règle de conduite appartient à tout
plongeur.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:pressure:1}
Une caisse pesant \qty{600}{N} repose sur une face de
$\qty{0.50}{m} \times \qty{0.30}{m}$, puis sur une face de
$\qty{0.30}{m} \times \qty{0.20}{m}$. Calculer les deux
\omterm{def:g10:pressure:pressure}{pressions} sur le sol. Quelle orientation est meilleure
sur du sable mou~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:2}
Exprimer \qty{0.25}{MPa} en \unit{kPa} et en \unit{Pa}. Quoi presse plus
fort~: \qty{3.0}{N} sur \qty{2.0}{cm^2}, ou \qty{60}{N} sur
\qty{500}{cm^2}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:3}
Calculer la \omterm{thm:g10:pressure:depth}{pression relative} $\rho g h$ et la
\omterm{met:g10:pressure:dive}{pression absolue} au fond d'une piscine profonde de
\qty{3.0}{m} ($\rho = \qty{1000}{kg/m^3}$, $P_0 = \qty{101}{kPa}$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:4}
L'air occupe \qty{4.0}{L} à \qty{100}{kPa}, température constante. Son
volume à \qty{250}{kPa}~? À \qty{50}{kPa}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:5}
Un patineur de \qty{65}{kg} glisse sur une lame d'aire de contact
\qty{12}{cm^2}, puis se tient sur des semelles totalisant
\qty{360}{cm^2}. Calculer les deux \omterm{def:g10:pressure:pressure}{pressions} et leur
rapport.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:6}
Un pouce pousse une punaise avec \qty{20}{N}. La tête a une aire
\qty{1.2}{cm^2}, la pointe \qty{0.010}{mm^2}. Calculer les
\omterm{def:g10:pressure:pressure}{pressions} sur le pouce et le mur~; pourquoi seul le mur
est-il percé~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:7}
Un panneau d'écoutille de sous-marin d'aire \qty{0.50}{m^2} se trouve
à \qty{30}{m} de profondeur en eau de mer
($\rho = \qty{1025}{kg/m^3}$)~; à l'intérieur, l'air est à la
\omterm{def:g10:pressure:atmospheric}{pression atmosphérique}. Calculer la
\omterm{def:g10:inertia:force}{force} nette qui le maintient fermé, et la masse dont le
\omterm{def:g10:universal-gravitation:weight}{poids} l'égale.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:8}
Un capteur à \qty{5.00}{m} de profondeur dans un liquide inconnu lit
une \omterm{thm:g10:pressure:depth}{pression relative} de \qty{66.2}{kPa}. Trouver la masse
volumique. Est-ce de l'eau douce~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:9}
Quelle hauteur aurait le baromètre de Torricelli avec de l'eau
($\rho = \qty{1000}{kg/m^3}$) au lieu de mercure
($\rho = \qty{13600}{kg/m^3}$)~? Retrouver les deux hauteurs à partir
de $\rho g h = P_0$~; pourquoi le mercure a-t-il gagné~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:10}
En utilisant les données de la seringue
de la~\cref{prop:g10:pressure:boyle} ($PV = \qty{6000}{kPa.mL}$),
prédire le volume à \qty{240}{kPa} et la \omterm{def:g10:pressure:pressure}{pression} à
\qty{75}{mL}. Quelle courbe est $P$ contre $V$~? Et $P$ contre
$1/V$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:11}
Un plongeur à \qty{20}{m} libère une bulle de volume \qty{0.50}{cm^3}.
Quel est son volume juste sous la surface~? (Température constante~;
l'eau de mer ajoute \qty{101}{kPa} par \qty{10}{m}.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:pressure:12}
Une ventouse d'aire \qty{12}{cm^2} est pressée à plat contre un plafond.
\begin{enumerate}
  \item Vide parfait à l'intérieur~: quelle \omterm{def:g10:inertia:force}{force} l'atmosphère
        exerce-t-elle sur la ventouse, et quelle masse suspendue peut-elle
        tenir~?
  \item Une ventouse réelle garde \qty{20}{kPa} d'air à l'intérieur.
        Quelle masse maintenant~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:pressure:13}
Une bouteille de plongée contient \qty{12}{L} d'air à \qty{200}{bar}
(\qty{1}{bar} = \qty{100}{kPa} $\approx 1$ atmosphère)~;
\qty{50}{bar} doivent rester en réserve. Un plongeur respire
\qty{15}{L} par minute à la \omterm{def:g10:pressure:pressure}{pression} ambiante. Combien de
litres d'air à \omterm{def:g10:pressure:pressure}{pression de surface} sont utilisables~? Combien
de temps la bouteille dure-t-elle à \qty{30}{m}
(\omterm{met:g10:pressure:dive}{pression absolue} $\approx \qty{4}{bar}$)~? Et à
\qty{10}{m}~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:pressure:14}
Les poumons d'un apnéiste contiennent \qty{6.0}{L} à la surface et ne
peuvent se contracter en dessous du volume résiduel \qty{1.5}{L}. À
quelle profondeur en mer ($\rho = \qty{1025}{kg/m^3}$) cette limite
est-elle atteinte~? (Un temps déclarée limite de l'apnée~; les records
dépassent maintenant \qty{100}{m}, le sang se déplaçant dans la
poitrine prenant le volume manquant.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:pressure:15}
Dans un vérin hydraulique, un liquide transmet la
\omterm{def:g10:pressure:pressure}{pression} inchangée d'un petit piston
($s = \qty{2.0}{cm^2}$) à un grand ($S = \qty{400}{cm^2}$) portant
une voiture de \qty{1200}{kg}.
\begin{enumerate}
  \item Quelle \omterm{def:g10:inertia:force}{force} sur le petit piston maintient la voiture~?
        Quelle \omterm{def:g10:pressure:pressure}{pression} le liquide porte-t-il~?
  \item Pour soulever la voiture de \qty{2.0}{cm}, de combien le petit
        piston doit-il se déplacer~? Comparer le travail fait sur chaque
        piston.
\end{enumerate}
\end{exercise}

\section{Problème~: Une inspiration en bas, une inspiration en haut}

\begin{problem}[{Problème du week-end --- une inspiration en bas, une
inspiration en haut~: la physique d'une plongée en apnée et d'une
remontée en bouteille, du squeeze de masque à la règle que les dix
derniers mètres sont les plus dangereux}]
\label{pb:g10:pressure:1}
Léna plonge en apnée~: une inspiration en surface, jusqu'à \qty{30}{m}
et retour. Marco plonge en bouteille à côté d'elle, respirant d'une
bouteille. Même mer ($\rho = \qty{1025}{kg/m^3}$,
$g = \qty{9.81}{N/kg}$, $P_0 = \qty{101}{kPa}$), mêmes poumons de
\qty{6.0}{L}, dangers opposés --- tous gouvernés par
$P = P_0 + \rho g h$ et $PV$ = constante (température constante tout
du long).

\medskip
\noindent\textbf{Partie I --- Le \omterm{def:g10:universal-gravitation:weight}{poids} de l'eau.}
\begin{enumerate}
  \item Calculer $\rho g$ pour l'eau de mer en \unit{kPa} par
        \omterm{def:g10:orders-of-magnitude:unit}{mètre}, et vérifier la règle empirique~: une
        atmosphère par \qty{10}{m}.
  \item Calculer la \omterm{met:g10:pressure:dive}{pression absolue} à \qty{10}{m},
        \qty{20}{m}, \qty{30}{m} et \qty{40}{m}.
  \item Exprimer chacune en multiple de la
        \omterm{def:g10:pressure:pressure}{pression} de surface.
  \item Le masque de Léna couvre \qty{150}{cm^2}. À \qty{30}{m}, si
        l'air à l'intérieur était encore à la
        \omterm{def:g10:pressure:pressure}{pression} de surface, quelle \omterm{def:g10:inertia:force}{force}
        nette l'eau exercerait-elle~? (Les plongeurs expirent dans le
        masque par le nez pour empêcher ce \emph{squeeze de masque}.)
  \item Un tympan a une aire d'environ \qty{0.60}{cm^2}. Calculer la
        \omterm{def:g10:inertia:force}{force} nette sur lui à \qty{3.0}{m} si l'oreille moyenne
        reste à la \omterm{def:g10:pressure:pressure}{pression} de surface~; pourquoi les oreilles
        font-elles mal dans une simple piscine, et que doit accomplir
        «~l'équilibrage~»~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Une inspiration en bas.}
Léna quitte la surface avec \qty{6.0}{L} d'air dans les poumons.
\begin{enumerate}[resume]
  \item En utilisant la \omterm{prop:g10:pressure:boyle}{loi de Boyle--Mariotte} avec les
        \omterm{met:g10:pressure:dive}{pressions absolues}, calculer le volume de ses
        poumons à \qty{10}{m}, \qty{20}{m} et \qty{30}{m}.
  \item À quelle profondeur son volume pulmonaire est-il divisé par
        deux~?
  \item Le volume pulmonaire est-il proportionnel à la profondeur~?
        Décrire la courbe de $V$ contre la
        \omterm{met:g10:pressure:dive}{pression absolue} $P$.
  \item Son volume résiduel est \qty{1.5}{L}~: les poumons ne peuvent
        se contracter davantage. Montrer que cette limite est atteinte
        près de \qty{30}{m}.
  \item Les records d'apnée dépassent pourtant \qty{100}{m}~: qu'est-ce
        qui remplit le volume manquant~? (Quoi d'autre peut s'écouler
        dans la poitrine~?)
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Une inspiration en haut.}
À \qty{30}{m}, le détendeur de Marco remplit ses poumons de
\qty{6.0}{L}.
\begin{enumerate}[resume]
  \item À quelle \omterm{def:g10:pressure:pressure}{pression} le détendeur livre-t-il cet air~?
        Combien de fois plus dense est-il que l'air de surface~?
  \item Marco panique, bloque sa respiration, et remonte à
        \qty{10}{m}~: quel volume son air piégé exige-t-il là~?
  \item Quel volume à la surface --- combien de fois sa capacité~?
  \item Énoncer la première règle du plongeur bouteille, et pourquoi
        respirer normalement supprime le danger.
  \item Léna bloque aussi sa respiration de \qty{30}{m} à la surface,
        pourtant ses poumons sont parfaitement en sécurité. Expliquer
        l'asymétrie.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Bulles, et les dix derniers
\omterm{def:g10:orders-of-magnitude:unit}{mètres}.}
\begin{enumerate}[resume]
  \item Marco libère une bulle de \qty{1.0}{cm^3} à \qty{40}{m}.
        Calculer son volume à \qty{30}{m}, \qty{20}{m}, \qty{10}{m} et
        juste sous la surface.
  \item Pour chaque palier de \qty{10}{m} de la montée, calculer le
        facteur par lequel la bulle croît. Où la croissance est-elle
        la plus rapide~?
  \item Les plongeurs remontent au plus à \qty{10}{m} par minute~:
        combien de temps depuis \qty{40}{m}, et dans quelle minute tout
        gaz piégé ou dissous se dilate-t-il le plus~?
  \item En profondeur le sang de Marco dissout de l'azote
        supplémentaire, comme le gaz dans une bouteille de soda
        bouchée. Que fait une remontée trop rapide, et comment la
        remontée lente et un palier de sécurité à \qty{5}{m}
        l'empêchent-ils~?
  \item Chute~: en une phrase chacune, donner le verdict de
        l'apnéiste, le verdict du plongeur bouteille, et la raison
        quantifiée pour laquelle les dix derniers
        \omterm{def:g10:orders-of-magnitude:unit}{mètres} de toute remontée méritent le plus de
        respect.
\end{enumerate}
\end{problem}
