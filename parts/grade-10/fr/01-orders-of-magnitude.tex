\chapter{Ordres de grandeur~: mesurer l'Univers}\label{ch:g10:orders-of-magnitude}

La physique commence là où l'opinion s'arrête~: avec une mesure. Mais
les grandeurs que la physique mesure couvrent un intervalle absurde
--- un noyau atomique mesure environ \qty{e-15}{m}, l'univers
observable environ \qty{e27}{m}~: un facteur $10^{42}$, bien au-delà
de ce qu'une image mentale peut contenir. Ce chapitre installe la
boîte à outils qui dompte cet intervalle --- \omterm{def:g10:orders-of-magnitude:unit}{unités}, puissances de
dix, \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs} --- et deux habitudes utilisées à
chaque page qui suit~: écrire l'\omterm{def:g10:orders-of-magnitude:unit}{unité}, et vérifier l'\omterm{def:g10:orders-of-magnitude:oom}{ordre de
grandeur}.

\section{Unités~: la grammaire de la mesure}

\begin{definition}[Grandeurs physiques et unités SI]\label{def:g10:orders-of-magnitude:unit}
Une \emph{grandeur physique}\index{grandeur physique} est une
propriété du monde que l'on peut mesurer~: une longueur, une durée,
une masse. La mesurer, c'est compter combien de fois une grandeur de
référence --- l'\emph{unité}\index{unité} --- y tient. Le
\emph{Système international d'unités}\index{Système international
d'unités (SI)} (SI) fixe sept unités de base, dont trois parcourent
tout ce volume~: le \emph{mètre}\index{mètre} (\unit{m}) pour la
longueur, le \emph{kilogramme}\index{kilogramme} (\unit{kg}) pour la
masse et la \emph{seconde}\index{seconde} (\unit{s}) pour le temps.
(Les quatre autres --- ampère, kelvin, mole, candela --- entreront en
scène lorsque l'électricité, la chaleur et la chimie rejoindront
l'histoire.)
\end{definition}

\begin{remark}[Un nombre sans unité n'est pas un résultat]\label{rem:g10:orders-of-magnitude:units}
«~La distance est $384$~» ne veut rien dire~: \qty{384}{mm},
\qty{384}{km} et \qty{384}{Mm} sont trois mondes différents (le dernier
se trouve être la distance à la Lune). En 1999, la NASA a perdu la
sonde \emph{Mars Climate Orbiter}, d'une valeur de 327~millions de
dollars, parce qu'un programme calculait les forces des propulseurs en
livres-force tandis qu'un autre les lisait en newtons. Règle de la
maison, de cette ligne jusqu'à la fin du livre~: tout nombre mesuré ou
calculé porte son \omterm{def:g10:orders-of-magnitude:unit}{unité}, à chaque ligne du calcul.
\end{remark}

\begin{definition}[Préfixes SI]\label{def:g10:orders-of-magnitude:prefixes}
Un \emph{préfixe}\index{préfixe (SI)} collé à une \omterm{def:g10:orders-of-magnitude:unit}{unité} la multiplie
par une puissance de dix fixe~:
\begin{center}
\begin{tabular}{llc@{\qquad\qquad}llc}
femto & f & $10^{-15}$ & centi & c & $10^{-2}$ \\
pico  & p & $10^{-12}$ & kilo  & k & $10^{3}$  \\
nano  & n & $10^{-9}$  & méga  & M & $10^{6}$  \\
micro & $\mu$ & $10^{-6}$ & giga & G & $10^{9}$ \\
milli & m & $10^{-3}$  & téra  & T & $10^{12}$ \\
\end{tabular}
\end{center}
Ainsi $\qty{1}{nm} = \qty{e-9}{m}$ et $\qty{1}{km} = \qty{e3}{m}$.
\end{definition}

\begin{example}[Lire les préfixes]\label{ex:g10:orders-of-magnitude:prefixes}
La lumière verte a une longueur d'onde d'environ
$\qty{500}{nm} = \qty{5.00e-7}{m}$~; un cheveu humain mesure environ
$\qty{70}{\micro m} = \qty{7.0e-5}{m}$ d'épaisseur~; la Lune orbite à
$\qty{384}{Mm} = \qty{3.84e8}{m}$. Une goutte de pluie a une masse
d'environ $\qty{30}{mg} = \qty{3.0e-5}{kg}$ --- remarquez que l'\omterm{def:g10:orders-of-magnitude:unit}{unité}
SI de base de la masse, seule parmi les sept, porte déjà un \omterm{def:g10:orders-of-magnitude:prefixes}{préfixe}.
\end{example}

\section{Notation scientifique et chiffres significatifs}

\begin{definition}[Notation scientifique]\label{def:g10:orders-of-magnitude:scinot}
Un nombre est en \emph{notation scientifique}\index{notation
scientifique} lorsqu'il s'écrit
\[
a \times 10^{n}, \qquad 1 \leq a < 10, \quad n \text{ un entier.}
\]
Le facteur $a$ est la \emph{mantisse}\index{mantisse}, l'entier $n$
l'\emph{exposant}\index{exposant}.
Ainsi $\num{149600000} = \num{1.496e8}$ et
$\num{0.00052} = \num{5.2e-4}$.
\end{definition}

\begin{proposition}[Calculer avec les puissances de dix]\label{prop:g10:orders-of-magnitude:powers}
Pour tous entiers $m$ et $n$~:
\[
10^{m} \times 10^{n} = 10^{m+n}, \qquad
\frac{1}{10^{n}} = 10^{-n}, \qquad
\left(10^{m}\right)^{n} = 10^{mn}.
\]
\end{proposition}

\begin{proof}[Démonstration partielle]
Pour des \omterm{def:g10:orders-of-magnitude:scinot}{exposants} positifs, on compte les facteurs~: $10^{m}
\times 10^{n}$ est un produit de $m + n$ dix, et $(10^m)^n$ enchaîne
$n$ groupes de $m$ dix. Écrire $10^{-n}$ pour $1/10^{n}$ est
précisément la convention qui maintient la première règle vraie pour
tous les entiers~: elle force $10^{n} \times 10^{-n} = 10^{0} = 1$.
Le volume de mathématiques compagnon étudie les règles des \omterm{def:g10:orders-of-magnitude:scinot}{exposants}
en détail.
\end{proof}

\begin{example}[Multiplier en notation scientifique]\label{ex:g10:orders-of-magnitude:scinot}
Combien de Terres tiendraient dans le Soleil, en masse~? Avec
$m_{\text{Sun}} = \qty{1.99e30}{kg}$ et
$m_{\text{Earth}} = \qty{5.97e24}{kg}$~:
\[
\frac{\num{1.99e30}}{\num{5.97e24}}
= \frac{1.99}{5.97} \times 10^{30-24}
= 0.333 \times 10^{6}
= \num{3.33e5}.
\]
Les \omterm{def:g10:orders-of-magnitude:scinot}{mantisses} et les \omterm{def:g10:orders-of-magnitude:scinot}{exposants} se traitent séparément~; la
réponse finale est renormalisée pour que la \omterm{def:g10:orders-of-magnitude:scinot}{mantisse} retombe dans
$[1, 10)$.
\end{example}

\begin{definition}[Chiffres significatifs]\label{def:g10:orders-of-magnitude:sigfig}
Les \emph{chiffres significatifs}\index{chiffres significatifs} d'un
nombre mesuré sont les chiffres réellement garantis par la mesure~:
tous les chiffres écrits sauf les zéros de tête. Les zéros de queue
comptent --- $\qty{5.20e-3}{s}$ (trois chiffres significatifs) promet
plus que $\qty{5.2e-3}{s}$ (deux). La \omterm{def:g10:orders-of-magnitude:scinot}{notation scientifique} affiche
le décompte d'un coup d'œil~: la \omterm{def:g10:orders-of-magnitude:scinot}{mantisse} porte exactement les
chiffres significatifs.
\end{definition}

\begin{method}[Combien de chiffres garder]\label{met:g10:orders-of-magnitude:sigfig}
Un résultat calculé ne peut pas être plus précis que ses ingrédients.
\begin{enumerate}
  \item Dans un produit ou un quotient, garder autant de \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres
        significatifs} que le facteur le \emph{moins} précis.
  \item N'arrondir que la réponse finale, jamais les étapes
        intermédiaires.
  \item En cas de doute, trois \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres significatifs} sont la
        précision de travail de ce livre.
\end{enumerate}
\end{method}

\begin{example}[Circonférence de la Terre]\label{ex:g10:orders-of-magnitude:sigfig}
Le diamètre de la Terre est $d = \qty{1.27e7}{m}$ (trois \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres
significatifs}), donc sa circonférence est
$\pi d = \num{3.14159} \times \qty{1.27e7}{m} = \qty{3.99e7}{m}$ ---
trois chiffres, pas les huit qu'affiche la calculatrice. Écrire
$\qty{39898227}{m}$ reviendrait à prétendre connaître la planète au
\omterm{def:g10:orders-of-magnitude:unit}{mètre} près.
\end{example}

\section{L'échelle de l'Univers}

\begin{definition}[Ordre de grandeur]\label{def:g10:orders-of-magnitude:oom}
L'\emph{ordre de grandeur}\index{ordre de grandeur} d'une grandeur
est la puissance de dix la plus proche~: écrire la grandeur sous la
forme $a \times 10^{n}$ avec $1 \leq a < 10$ et prendre $10^{n}$ si
$a < 5$, $10^{n+1}$ si $a \geq 5$. (La convention exacte de frontière
n'a jamais d'importance~: un énoncé d'ordre de grandeur n'est jamais
censé valoir mieux qu'à un facteur dix près.)
\end{definition}

\begin{example}[Ordres de grandeur]\label{ex:g10:orders-of-magnitude:oom}
Une vie humaine dure environ $80$~ans, c'est-à-dire
$80 \times \qty{3.16e7}{s} \approx \qty{2.5e9}{s}$~: \omterm{def:g10:orders-of-magnitude:oom}{ordre de
grandeur} $\qty{e9}{s}$ --- quelques milliards de secondes, à bien
employer. Un humain mesure $\qty{1.7}{m}$~: ordre $\qty{e0}{m}$. Le
mont Everest, $\qty{8848}{m}$~: ordre $\qty{e4}{m}$, puisque
$8.848 \geq 5$.
\end{example}

Empiler de telles estimations par \omterm{def:g10:orders-of-magnitude:scinot}{exposant} produit le tableau
maître de ce chapitre~: une échelle dont chaque barreau est un monde
dix fois plus large que celui d'en dessous.

Au-dessus du barreau Terre, les \omterm{def:g10:orders-of-magnitude:unit}{mètres} cessent d'être commodes~;

partir de l'échelle elle-même.

\begin{definition}[Unité astronomique et année-lumière]\label{def:g10:orders-of-magnitude:lightyear}
L'\emph{unité astronomique}\index{unité astronomique}
($\qty{1}{au} = \qty{1.496e11}{m}$) est la distance moyenne
Terre--Soleil. L'\emph{année-lumière}\index{année-lumière}
(\unit{ly}) est la distance parcourue par la lumière en une année. La
lumière se déplace dans le vide à
\[
c = \qty{3.00e8}{m/s},
\]
la plus grande vitesse que la nature autorise --- donc
l'année-lumière est une distance, pas un temps.
\end{definition}

\begin{example}[L'année-lumière en mètres]\label{ex:g10:orders-of-magnitude:ly}
Une année compte $365.25 \times 24 \times 3600 = \qty{3.156e7}{s}$
(agréablement proche de $\pi \times 10^{7}\,\unit{s}$). D'où
\[
\qty{1}{ly} = c \times \qty{3.156e7}{s}
= \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}.
\]
Proxima du Centaure, l'étoile la plus proche après le Soleil, se
trouve à $\qty{4.2}{ly} \approx \qty{4.0e16}{m}$~: la lumière, qui
traverse une distance Terre--Soleil en huit minutes, met plus de
quatre ans pour arriver de là-bas.
\end{example}

\begin{method}[Contrôle de vraisemblance par ordre de grandeur]\label{met:g10:orders-of-magnitude:sanity}
Avant de faire confiance à un résultat calculé~:
\begin{enumerate}
  \item arrondir chaque donnée d'entrée à son \omterm{def:g10:orders-of-magnitude:oom}{ordre de grandeur}~;
  \item refaire le calcul avec les seules puissances de dix, en
        utilisant le~\cref{prop:g10:orders-of-magnitude:powers} ---
        cela prend quelques secondes~;
  \item comparer~: la réponse précise doit s'accorder avec
        l'estimation à un facteur dix près environ, et son \omterm{def:g10:orders-of-magnitude:unit}{unité}
        doit être la bonne. Si l'un des deux contrôles échoue, traquer
        l'erreur avant de poursuivre.
\end{enumerate}
\end{method}

\begin{remark}[Pourquoi s'en soucier]\label{rem:g10:orders-of-magnitude:sanity}
Une calculatrice indiquera volontiers qu'une pomme a une masse
$\qty{4e8}{kg}$ --- elle ne peut pas savoir mieux. Aucune formule ne
signale l'absurdité~; l'habitude le fait. Les physiciens estiment
d'abord et calculent ensuite~: l'estimation attrape les virgules
mal placées, les \omterm{def:g10:orders-of-magnitude:unit}{unités} échangées et les fractions inversées, les
trois erreurs derrière la plupart des mauvaises réponses à tous les
niveaux.
\end{remark}

\section{Mesure et incertitude}

Aucun instrument ne lit la vraie valeur. Une mesure n'est honnête que
lorsqu'elle indique aussi de combien elle peut s'en écarter.

\begin{definition}[Incertitude absolue et relative]\label{def:g10:orders-of-magnitude:uncertainty}
Une mesure d'une grandeur $x$ s'énonce
\[
x = x_{\text{m}} \pm \Delta x ,
\]
où $x_{\text{m}}$ est la valeur mesurée et l'\emph{incertitude
absolue}\index{incertitude!absolue} $\Delta x > 0$, dans la même
\omterm{def:g10:orders-of-magnitude:unit}{unité}, borne l'erreur plausible~: la vraie valeur est censée
se trouver entre $x_{\text{m}} - \Delta x$ et
$x_{\text{m}} + \Delta x$. L'\emph{incertitude
relative}\index{incertitude!relative} est le rapport
$\Delta x / x_{\text{m}}$, habituellement exprimée en pourcent~; c'est
elle que le mot «~précis~» mesure.
\end{definition}

\begin{method}[Lire un instrument gradué]\label{met:g10:orders-of-magnitude:reading}
Pour une règle, une éprouvette graduée, un cadran~:
\begin{enumerate}
  \item lire la graduation la plus proche de la marque, l'œil
        perpendiculaire à l'échelle (un œil de biais décale la
        lecture)~;
  \item prendre au moins la moitié de la plus petite graduation comme
        \omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude absolue} --- plus si le bord de l'objet ou la
        surface du liquide est flou~;
  \item écrire le résultat sous la forme $x_{\text{m}} \pm \Delta x$
        avec son \omterm{def:g10:orders-of-magnitude:unit}{unité}, et ne garder aucun chiffre au-delà de celui
        qui est incertain.
\end{enumerate}
\end{method}

\begin{example}[Précision relative]\label{ex:g10:orders-of-magnitude:reading}
La lecture ci-dessus a une \omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude relative}
$\frac{0.5}{127} \approx 0.4\%$. La même règle mesurant une vis de
$\qty{12}{mm}$ donne $\frac{0.5}{12} \approx 4\%$~: l'instrument
identique est dix fois moins précis sur le petit objet. La précision
est une propriété de la \emph{mesure}, non de l'outil.
\end{example}

\begin{remark}[Petite absolue, énorme relative --- et inversement]\label{rem:g10:orders-of-magnitude:precision}
Des stations laser mesurent la distance Terre--Lune,
$\qty{3.84e8}{m}$, à environ $\pm\qty{3}{cm}$~: une \omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude
relative} de $10^{-10}$, parmi les mesures les plus fines jamais
réalisées --- avec une erreur absolue dont aucune règle ne se
vanterait. Toujours demander \emph{par rapport à quoi}~: le couple
$(\Delta x,\ \Delta x / x_{\text{m}})$ raconte toute l'histoire, et
chacun des deux nombres seul peut tromper.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:1}
Convertir en \omterm{def:g10:orders-of-magnitude:unit}{mètres}, en \omterm{def:g10:orders-of-magnitude:scinot}{notation scientifique}~:
\begin{enumerate}
  \item $\qty{750}{nm}$ (longueur d'onde de la lumière rouge)~;
  \item $\qty{2.5}{km}$~;
  \item $\qty{3.2}{\micro m}$ (une bactérie du sol)~;
  \item $\qty{0.15}{Tm}$ (la distance Terre--Soleil, sous un costume
        inhabituel).
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:2}
Écrire en \omterm{def:g10:orders-of-magnitude:scinot}{notation scientifique} et indiquer le nombre de \omterm{def:g10:orders-of-magnitude:sigfig}{chiffres
significatifs}~:
\begin{enumerate}
  \item la distance Terre--Soleil, $\qty{149600000}{km}$, en \omterm{def:g10:orders-of-magnitude:unit}{mètres}~;
  \item le diamètre d'un atome d'hydrogène, $\qty{0.000000000106}{m}$~;
  \item la population mondiale, environ $\num{8100000000}$~;
  \item une durée de $\qty{0.00520}{s}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:3}
Calculer sans calculatrice, en donnant chaque résultat en \omterm{def:g10:orders-of-magnitude:scinot}{notation
scientifique}~:
\[
(3\times10^{8})\times(2\times10^{-5}), \qquad
\frac{8\times10^{4}}{4\times10^{-3}}, \qquad
(5\times10^{6})^{2}, \qquad
\frac{(4\times10^{-3})\times(3\times10^{-9})}{6\times10^{-7}} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:4}
Donner l'\omterm{def:g10:orders-of-magnitude:oom}{ordre de grandeur}, avec son \omterm{def:g10:orders-of-magnitude:unit}{unité}, de~: une taille
humaine de $\qty{1.7}{m}$~; le mont Everest, $\qty{8848}{m}$~; une
bactérie \emph{E.~coli}, $\qty{2e-6}{m}$~; la masse de la Terre,
$\qty{5.97e24}{kg}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:5}

\begin{enumerate}

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:6}

  \item La sonde \emph{Voyager 1} se trouve à environ

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:7}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:8}

  \item Quelle est l'\omterm{def:g10:orders-of-magnitude:uncertainty}{incertitude relative} de cette épaisseur~?

\begin{exercise}[$\star\star\star$]\label{exo:g10:orders-of-magnitude:12}

\section{Problème~: De l'atome à Andromède}

\begin{problem}[{Problème du week-end --- un seul grain de sable
dimensionne l'atome, le Système solaire, la galaxie et l'univers
observable, et vous place sur l'échelle entre eux}]
\label{pb:g10:orders-of-magnitude:1}
Un grain de sable est l'objet le plus humble de la physique, et ce
week-end il travaille d'arrache-pied~: d'abord on compte ses atomes,
puis on réduit le Soleil à sa taille et on arpent le cosmos, puis on
laisse la lumière faire le levé, et enfin on compte les atomes
\emph{en vous}. Données, utilisées tout au long~: diamètre du grain
$\qty{0.5}{mm}$~; diamètre de l'atome $\qty{2.5e-10}{m}$~; masse
volumique du sable $\qty{2500}{kg/m^3}$~; diamètres~: Soleil
$\qty{1.39e9}{m}$, Terre $\qty{1.27e7}{m}$, Voie lactée
$\qty{9.5e20}{m}$, univers observable $\qty{8.8e26}{m}$~; distances
depuis nous~: Lune $\qty{3.84e8}{m}$, Soleil $\qty{1.496e11}{m}$,
Neptune $\qty{4.5e12}{m}$, Proxima du Centaure $\qty{4.0e16}{m}$,
galaxie d'Andromède $\qty{2.4e22}{m}$~; $c = \qty{3.00e8}{m/s}$~; une
molécule d'eau a une masse $\qty{3.0e-26}{kg}$ et contient trois
atomes.

\medskip
\noindent\textbf{Partie I --- Anatomie du grain.}
\begin{enumerate}
  \item Écrire le diamètre du grain et le diamètre de l'atome en
        \omterm{def:g10:orders-of-magnitude:unit}{mètres}, en \omterm{def:g10:orders-of-magnitude:scinot}{notation scientifique}, et nommer le \omterm{def:g10:orders-of-magnitude:prefixes}{préfixe}
        SI auquel chacun est le plus proche.
  \item Combien d'atomes se placent côte à côte sur un diamètre de
        grain~?
  \item Modéliser le grain comme un cube de côté $\qty{0.5}{mm}$
        rempli d'atomes. Estimer le nombre d'atomes dans le grain,
        comme \omterm{def:g10:orders-of-magnitude:oom}{ordre de grandeur}.
  \item Calculer le volume de ce cube, puis la masse du grain.
        Exprimer la masse en milligrammes.
  \item Contrôle de vraisemblance du modèle~: diviser la masse du
        grain par son nombre d'atomes et comparer à l'\omterm{def:g10:orders-of-magnitude:oom}{ordre de
        grandeur} connu des masses atomiques (quelques
        $10^{-26}\,\unit{kg}$). Verdict~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le Soleil comme grain de sable.}
On construit une maquette du cosmos dans laquelle le Soleil est le
grain~: chaque taille modèle égale la taille réelle multipliée par un
facteur $s$.
\begin{enumerate}[resume]
  \item Calculer le facteur d'échelle $s$.
  \item Dans la maquette, trouver le diamètre de la Terre et sa
        distance au grain-Soleil. Quel objet du monde de la Partie~I
        a à peu près la taille de la Terre modèle~?
  \item À quelle distance du grain orbite Neptune modèle~? Quel meuble
        contiendrait le système planétaire tout entier~?
  \item À quelle distance se trouve Proxima du Centaure modèle~?
  \item Deux grains de sable, distants de $\qty{14}{km}$, et presque
        rien entre eux~: énoncer en une ou deux phrases ce que la
        maquette révèle sur le vide de la galaxie.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La lumière fait le levé.}
\begin{enumerate}[resume]
  \item Combien de temps la lumière met-elle à traverser le vrai grain
        de sable~?
  \item Combien de temps la lumière met-elle à nous atteindre depuis
        la Lune, et depuis le Soleil~?
  \item Recalculer l'\omterm{def:g10:orders-of-magnitude:lightyear}{année-lumière} en \omterm{def:g10:orders-of-magnitude:unit}{mètres} à partir de $c$ et
        de la durée de l'année, et vérifier que la distance de Proxima
        dans les données correspond à environ $\qty{4.2}{ly}$.
  \item Convertir la distance d'Andromède en \omterm{def:g10:orders-of-magnitude:lightyear}{années-lumière}.
        Environ quand la lumière qui entre maintenant dans votre œil a-t-elle
        quitté Andromède, et qu'est-ce qui marchait sur la Terre à
        cette époque~?
  \item Quelle distance la lumière parcourt-elle en une nanoseconde~?
        Un processeur cadencé à $\qty{3}{GHz}$ démarre une opération
        tous les tiers de nanoseconde~: expliquer en une phrase
        pourquoi un ordinateur rapide doit être un ordinateur petit.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Vous, entre atome et Andromède.}
\begin{enumerate}[resume]
  \item Un humain est à peu près $\qty{70}{kg}$ d'eau. En utilisant
        les données, estimer la masse moyenne d'un atome dans l'eau,
        puis le nombre d'atomes dans un humain.
  \item Comparer ce décompte aux atomes du grain (question~3) et aux
        environ $2\times10^{23}$ étoiles de l'univers observable.
        Donner les deux facteurs.
  \item Sur l'échelle, êtes-vous plus près de l'atome ou de l'univers
        observable~? Comparer les rapports
        $\text{(vous)}/\text{(atome)}$ et
        $\text{(univers)}/\text{(vous)}$, en prenant $\qty{1.7}{m}$
        pour vous.
  \item Trouver la taille $x$ située exactement à mi-chemin sur
        l'échelle entre le noyau ($\qty{e-15}{m}$) et l'univers
        observable~: $x$ doit satisfaire
        $x / 10^{-15} = 10^{26.9} / x$ (en utilisant
        $\qty{8.8e26}{m} \approx 10^{26.9}\,\unit{m}$). Quel objet du
        Système solaire a justement cette taille~?
  \item La chute. Réduire l'univers observable par le facteur $s$ de
        la Partie~II, de sorte que le Soleil soit un grain de sable.
        Donner le diamètre de l'univers modèle en \omterm{def:g10:orders-of-magnitude:unit}{mètres}, puis en
        \omterm{def:g10:orders-of-magnitude:lightyear}{unités astronomiques}, et conclure en une phrase~: même
        avec chaque étoile transformée en grain de sable, l'univers
        déborde de toute échelle humaine --- seules les puissances de
        dix le contiennent.
\end{enumerate}
\end{problem}
