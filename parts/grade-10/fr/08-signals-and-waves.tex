\chapter{Signaux et ondes}\label{ch:g10:signals-and-waves}

Chaque message que vous avez jamais reçu est arrivé comme un
\omterm{def:g10:signals-and-waves:signal}{signal}~: une voix a pressé sur votre tympan, cette page vous a
atteint comme un scintillement de radio. Derrière la variété se tient
une seule idée --- une \omterm{def:g10:orders-of-magnitude:unit}{grandeur physique} variant dans le temps
--- et une seule boîte à outils~: \omterm{def:g10:signals-and-waves:period}{période},
\omterm{def:g10:signals-and-waves:frequency}{fréquence}, \omterm{def:g10:signals-and-waves:amplitude}{amplitude}, et l'art de
chronométrer un délai. Avec elle, un bateau mesure le fond de la mer et
un médecin compte les battements de cœur.

\section{Les signaux portent de l'information}

\begin{definition}[Signal]\label{def:g10:signals-and-waves:signal}
Un \emph{signal}\index{signal} est une \omterm{def:g10:orders-of-magnitude:unit}{grandeur physique} qui
varie dans le temps pour porter de l'information~: la
\omterm{def:g10:pressure:pressure}{pression} de l'air sur votre tympan, la tension aux
bornes d'un microphone, la luminosité de la lumière dans une
\omterm{def:g10:refraction:fiber}{fibre optique}. Pour étudier un signal, l'enregistrer
comme fonction du temps et lire l'information sur la courbe.
\end{definition}

Dans un appel téléphonique la même information change de porteur
plusieurs fois~: \omterm{def:g10:pressure:pressure}{pression} dans l'air, puis tension dans le
microphone, puis radio à l'antenne --- et à l'envers à l'autre bout. Un
seul vocabulaire les gère tous~; ce chapitre le construit.

\section{Signaux périodiques}

\begin{definition}[Signal périodique, période]\label{def:g10:signals-and-waves:period}
Un \omterm{def:g10:signals-and-waves:signal}{signal} est \emph{périodique}\index{signal périodique}
lorsqu'il se répète à l'identique à intervalles réguliers. La
\emph{période}\index{période} $T$ est la durée d'un cycle complet, en
secondes.
\end{definition}

\begin{definition}[Fréquence]\label{def:g10:signals-and-waves:frequency}
La \emph{fréquence}\index{fréquence} d'un \omterm{def:g10:signals-and-waves:period}{signal périodique}
est le nombre de cycles par seconde~:
\[
f = \frac{1}{T} .
\]
Son \omterm{def:g10:orders-of-magnitude:unit}{unité}, un cycle par seconde, est le \emph{hertz}\index{hertz}
(\unit{Hz}), avec les multiples habituels \unit{kHz}, \unit{MHz},
\unit{GHz}.
\end{definition}

\begin{definition}[Amplitude]\label{def:g10:signals-and-waves:amplitude}
L'\emph{amplitude}\index{amplitude} d'un \omterm{def:g10:signals-and-waves:period}{signal périodique}
est son écart maximal par rapport à la valeur de repos. Pour une tension
oscillant symétriquement autour de zéro, l'amplitude est la valeur de
crête~; de crête à creux mesure \emph{deux fois} l'amplitude (la valeur
crête à crête).
\end{definition}

\begin{example}[Secteur et la de concert]\label{ex:g10:signals-and-waves:mains}
La tension du secteur oscille à $f = \qty{50}{Hz}$~: sa
\omterm{def:g10:signals-and-waves:period}{période} est $T = 1/f = \qty{0.020}{s} = \qty{20}{ms}$.
Le la de concert d'un orchestre est un \omterm{def:g10:signals-and-waves:signal}{signal} de
\omterm{def:g10:pressure:pressure}{pression} à \qty{440}{Hz}~: un cycle dure
$T = 1/440 = \qty{2.3}{ms}$. Plus haute la
\omterm{def:g10:signals-and-waves:frequency}{fréquence}, plus court le cycle~: $f$ et $T$ sont
inverses l'un de l'autre.
\end{example}

\section{Lire un oscillogramme}

\begin{definition}[Oscilloscope]\label{def:g10:signals-and-waves:oscilloscope}
Un \emph{oscilloscope}\index{oscilloscope} trace une tension en fonction
du temps~; la courbe sur son écran quadrillé est un
\emph{oscillogramme}\index{oscillogramme}. Deux réglages convertissent
les divisions d'écran en valeurs physiques~: la \emph{base de
temps}\index{base de temps} (secondes par division horizontale) et le
\emph{gain}\index{gain} vertical (volts par division verticale).
\end{definition}

\begin{method}[De l'écran aux nombres]\label{met:g10:signals-and-waves:oscillogram}
\begin{enumerate}
  \item Compter les divisions horizontales couvertes par un cycle
        complet (crête à crête est le plus sûr)~; multiplier par la
        \omterm{def:g10:signals-and-waves:oscilloscope}{base de temps} pour obtenir $T$, puis $f = 1/T$.
  \item Compter les divisions verticales de la ligne centrale à une
        crête et multiplier par le \omterm{def:g10:signals-and-waves:oscilloscope}{gain} pour obtenir
        l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude} (crête à creux donne la valeur crête
        à crête~: la diviser par deux).
\end{enumerate}
\end{method}

\begin{example}[Une lecture complète]\label{ex:g10:signals-and-waves:reading}
Sur l'\omterm{def:g10:signals-and-waves:oscilloscope}{oscillogramme} ci-dessous, la
\omterm{def:g10:signals-and-waves:oscilloscope}{base de temps} est \qty{5}{ms} par division et le
\omterm{def:g10:signals-and-waves:oscilloscope}{gain} \qty{2}{V} par division. Un cycle couvre $4{,}0$
divisions~: $T = 4.0 \times \qty{5}{ms} = \qty{20}{ms}$ et
$f = 1/\qty{0.020}{s} = \qty{50}{Hz}$ --- le secteur encore. Une crête
se trouve $3{,}0$ divisions au-dessus de l'axe~:
l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude} est $3.0 \times \qty{2}{V} = \qty{6.0}{V}$.
\end{example}

\section{Signaux sonores et électromagnétiques}

\begin{definition}[Son]\label{def:g10:signals-and-waves:sound}
Le \emph{son}\index{son} est un \omterm{def:g10:signals-and-waves:signal}{signal} de
\omterm{def:g10:pressure:pressure}{pression}~: un objet vibrant --- corde, membrane, cordes
vocales --- pousse rythmiquement sur l'air, et les compressions se
propageant vers l'extérieur à environ \qty{340}{m/s}. Un microphone
convertit la \omterm{def:g10:pressure:pressure}{pression} arrivante en tension~; le tympan, en
impulsions nerveuses. Le son a besoin d'un milieu~: une cloche agitée
sous une cloche à vide se tait à mesure que l'air est pompé.
\end{definition}

\begin{definition}[Onde]\label{def:g10:signals-and-waves:wave}
Une perturbation qui se propage --- le motif de compression d'un
\omterm{def:g10:signals-and-waves:sound}{son}, l'ondulation d'un étang, une impulsion radio --- est une
\emph{onde}\index{onde}. Le milieu, lorsqu'il y en a un, reste en place~:
chaque portion d'eau bobine sur place tandis que l'ondulation traverse
l'étang. La géométrie des ondes --- \omterm{def:g10:light-spectra:wavelength}{longueur d'onde},
interférence, diffraction --- est reprise dans un chapitre ultérieur~;
ce chapitre n'a besoin que de leurs vitesses.
\end{definition}

\begin{definition}[Ultrason et infrason]\label{def:g10:signals-and-waves:ultrasound}
L'oreille humaine entend le \omterm{def:g10:signals-and-waves:sound}{son} entre environ \qty{20}{Hz} et
\qty{20}{kHz} (le plafond baisse avec l'âge). Le
\omterm{def:g10:signals-and-waves:sound}{son} au-dessus de \qty{20}{kHz} est
\emph{ultrason}\index{ultrason} --- les chauves-souris chassent à
\qty{50}{kHz}, les scanners médicaux imagent le corps à quelques
mégahertz. Le \omterm{def:g10:signals-and-waves:sound}{son} en dessous de \qty{20}{Hz} est
\emph{infrason}\index{infrason}, senti par les éléphants et les
sismomètres.
\end{definition}

\begin{proposition}[Signaux électromagnétiques]\label{prop:g10:signals-and-waves:em}
La radio, le Wi-Fi et la lumière sont des \emph{signaux
électromagnétiques}\index{signal électromagnétique}~: une seule famille,
tous voyageant à travers le vide --- aucun milieu nécessaire --- et tous
à une seule vitesse, la vitesse de la lumière
du~\cref{ch:g10:refraction},
\[
c = \qty{3.00e8}{m/s} \quad \text{(très presque la même dans l'air).}
\]
\end{proposition}
\admitted

\begin{remark}\label{rem:g10:signals-and-waves:honest}
Ici c'est un fait expérimental --- la lumière du Soleil traverse
$150$~millions de kilomètres d'espace vide~; elle est dérivée
honnêtement dans le volume de 2\textsuperscript{e} année des lois de
l'électricité et du magnétisme. Garder le contraste~: le
\omterm{def:g10:signals-and-waves:sound}{son} est une vibration \emph{du} milieu et meurt sans lui~; un
\omterm{prop:g10:signals-and-waves:em}{signal électromagnétique} porte son propre champ à travers le
vide, un million de fois plus vite.
\end{remark}

\section{Échos~: des distances à partir de délais}

Un \omterm{def:g10:signals-and-waves:signal}{signal} se déplaçant à la vitesse $v$ couvre $d = v\,t$ en
un temps $t$~: chaque délai est une distance déguisée. Mieux encore,
envoyer une courte impulsion vers un obstacle et chronométrer son
\emph{écho}\index{écho} --- l'aller-retour couvre $2d$.

\begin{method}[Télémétrie par écho]\label{met:g10:signals-and-waves:echo}
\begin{enumerate}
  \item Envoyer une courte impulsion~; mesurer le temps d'aller-retour
        $t$ de l'\omterm{rem:g10:signals-and-waves:honest}{écho}.
  \item Prendre la vitesse $v$ du \omterm{def:g10:signals-and-waves:signal}{signal} \emph{dans le milieu
        traversé}.
  \item L'impulsion a couvert $2d = v\,t$, donc $d = \dfrac{v\,t}{2}$.
\end{enumerate}
Des impulsions de \omterm{def:g10:signals-and-waves:sound}{son} dans l'eau font le
\emph{sonar}\index{sonar} ($v \approx \qty{1500}{m/s}$)~; des
impulsions radio dans l'air font le \emph{radar}\index{radar}
($v = c$)~; des impulsions d'\omterm{def:g10:signals-and-waves:ultrasound}{ultrason} dans le corps font
le scanner médical ($v \approx \qty{1540}{m/s}$ dans les tissus mous).
\end{method}

\begin{example}[Sonar]\label{ex:g10:signals-and-waves:sonar}
Le \omterm{met:g10:signals-and-waves:echo}{sonar} d'un navire envoie un ping et l'\omterm{rem:g10:signals-and-waves:honest}{écho}
du fond revient après $t = \qty{0.80}{s}$~: la profondeur est
$d = \frac{1500 \times 0.80}{2} = \qty{600}{m}$. Oublier le facteur $2$
double l'océan.
\end{example}

\begin{example}[Radar]\label{ex:g10:signals-and-waves:radar}
Un \omterm{met:g10:signals-and-waves:echo}{radar} d'aéroport reçoit l'\omterm{rem:g10:signals-and-waves:honest}{écho} d'un avion
$\qty{0.30}{ms}$ après l'impulsion~:
$d = \frac{\qty{3.00e8}{m/s} \times \qty{3.0e-4}{s}}{2} = \qty{45}{km}$.
Parce que $c$ est si grand, le \omterm{met:g10:signals-and-waves:echo}{radar} vit sur des horloges
à la microseconde~: chronométrer la lumière est un travail de précision.
\end{example}

\begin{example}[Le cœur comme signal]\label{ex:g10:signals-and-waves:ecg}
Chaque battement de cœur envoie une petite impulsion de tension à
travers la poitrine~; des électrodes l'enregistrent comme
l'\emph{électrocardiogramme}\index{électrocardiogramme} (ECG), un
\omterm{def:g10:signals-and-waves:period}{signal} quasi périodique. Si les pics hauts sont
distants de $T = \qty{0.75}{s}$, le cœur bat à
$f = 1/0.75 = \qty{1.33}{Hz}$, c'est-à-dire $1.33 \times 60 = 80$
battements par minute~: lire $T$ sur un tracé, c'est ainsi qu'un
cardiologue prend votre pouls.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:1}
Calculer la \omterm{def:g10:signals-and-waves:frequency}{fréquence} du \omterm{def:g10:signals-and-waves:period}{signal périodique}
de \omterm{def:g10:signals-and-waves:period}{période} (a) $T = \qty{20}{ms}$~; (b)
$T = \qty{4.0}{ms}$~; (c) $T = \qty{50}{\micro s}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:2}
Calculer la \omterm{def:g10:signals-and-waves:period}{période} de (a) le secteur à \qty{50}{Hz}~;
(b) un la de concert à \qty{440}{Hz}~; (c) un \omterm{def:g10:signals-and-waves:signal}{signal}
Wi-Fi à \qty{2.4}{GHz}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:3}
Classer en \omterm{def:g10:signals-and-waves:ultrasound}{infrason}, \omterm{def:g10:signals-and-waves:sound}{son} audible ou
\omterm{def:g10:signals-and-waves:ultrasound}{ultrason}~: \qty{12}{Hz}~; \qty{440}{Hz}~; \qty{18}{kHz}~;
\qty{40}{kHz} (un capteur de stationnement)~; \qty{5}{MHz} (une sonde
médicale).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:4}
Sur un \omterm{def:g10:signals-and-waves:oscilloscope}{oscilloscope} réglé à \qty{2}{ms} par division
et \qty{0.5}{V} par division, un cycle couvre $5{,}0$ divisions et une
crête se trouve $4{,}0$ divisions au-dessus de l'axe. Trouver $T$, $f$
et l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:5}
Vous voyez l'éclair, puis entendez le tonnerre \qty{6.0}{s} plus tard.
À quelle distance a-t-il frappé
($v_{\text{sound}} = \qty{340}{m/s}$)~? Pourquoi le temps de parcours
de la lumière peut-il être négligé~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:6}
Un \omterm{rem:g10:signals-and-waves:honest}{écho} de \omterm{met:g10:signals-and-waves:echo}{sonar} revient du fond après
\qty{0.60}{s} ($v = \qty{1500}{m/s}$ dans l'eau de mer). Pourquoi le
produit $v\,t$ doit-il être divisé par deux~? Calculer la profondeur.
Combien de temps l'\omterm{rem:g10:signals-and-waves:honest}{écho} mettrait-il sur une fosse de
\qty{1200}{m} de profondeur~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:7}
Un \omterm{met:g10:signals-and-waves:echo}{radar} d'aéroport reçoit un \omterm{rem:g10:signals-and-waves:honest}{écho} d'avion
après \qty{0.24}{ms} et, exactement \qty{1.0}{s} plus tard, un second
\omterm{rem:g10:signals-and-waves:honest}{écho} après \qty{0.238}{ms}. Calculer les deux distances,
puis la vitesse de l'avion. S'approche-t-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:8}
Un ECG est imprimé à \qty{25}{mm/s}~; les pics hauts sont distants de
\qty{20}{mm}. Trouver la \omterm{def:g10:signals-and-waves:period}{période} et la fréquence
cardiaque en battements par minute. La tachycardie signifie plus de
$100$ battements par minute~: en dessous de quel espacement de pics
apparaît-elle sur ce papier~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:9}
Vous devez afficher un \omterm{def:g10:signals-and-waves:signal}{signal} à \qty{200}{Hz}
d'\omterm{def:g10:signals-and-waves:amplitude}{amplitude} \qty{3.0}{V} sur un écran de $10$
divisions horizontales et $8$ verticales ($4$ au-dessus de la ligne
centrale). Choisir une \omterm{def:g10:signals-and-waves:oscilloscope}{base de temps} montrant environ
deux cycles complets, et un \omterm{def:g10:signals-and-waves:oscilloscope}{gain} utilisant la plupart
de l'écran sans écrêtage.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:10}
Une sonde médicale envoie une impulsion d'\omterm{def:g10:signals-and-waves:ultrasound}{ultrason} dans
l'abdomen ($v = \qty{1540}{m/s}$ dans les tissus mous) et entend des
\omterm{rem:g10:signals-and-waves:honest}{échos} après \qty{40}{\micro s} (paroi avant d'un organe) et
\qty{60}{\micro s} (paroi arrière). Trouver la profondeur de chaque
paroi et l'épaisseur de l'organe.
\end{exercise}

\section{Problème~: Le sonar, l'orage et le cardiogramme}

\begin{problem}[{Problème du week-end --- lire les signaux du monde~:
une loi, $d = v\,t$, localise un orage, cartographie un fond, compte
un battement de cœur et tire au clair comment fonctionne le GPS}]
\label{pb:g10:signals-and-waves:1}
Un bateau de pêche travaille une nuit d'orage~: éclairs à l'horizon, le
\omterm{met:g10:signals-and-waves:echo}{sonar} balayant le fond, le cœur du patron sur le papier du
médecin, un récepteur GPS écoutant des \omterm{def:g10:universal-gravitation:satellite}{satellites}.
Quatre instruments, une loi. Prendre $v_{\text{sound}} = \qty{340}{m/s}$
dans l'air, \qty{1500}{m/s} dans l'eau de mer, et $c = \qty{3.00e8}{m/s}$.

\medskip
\noindent\textbf{Partie I --- L'orage.}
\begin{enumerate}
  \item Un éclair~; le tonnerre suit \qty{9.0}{s} plus tard. À quelle
        distance est-il~?
  \item Calculer le temps de parcours de la lumière sur cette distance,
        et justifier de traiter l'éclair comme instantané.
  \item Les marins comptent les secondes entre éclair et tonnerre et
        divisent par trois pour obtenir des kilomètres. Justifier la
        règle.
  \item Cinq minutes plus tard, un éclair donne \qty{6.0}{s}. À quelle
        distance maintenant~? Trouver la vitesse moyenne d'approche de
        l'orage en \unit{m/s} et \unit{km/h}.
  \item Le tonnerre gronde au lieu de claquer~: le canal de foudre fait
        des kilomètres de long, donc ses parties sont à des distances
        différentes. Si le canal s'étend de \qty{2.0}{km} à
        \qty{5.0}{km} du bateau, combien de temps dure le grondement~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le \omterm{met:g10:signals-and-waves:echo}{sonar}.}
\begin{enumerate}[resume]
  \item Le \omterm{met:g10:signals-and-waves:echo}{sonar} envoie un ping~; le fond répond en
        \qty{0.90}{s}. Profondeur~?
  \item Sur le plateau l'\omterm{rem:g10:signals-and-waves:honest}{écho} se raccourcit à
        \qty{0.20}{s}. Profondeur~?
  \item Un ping renvoie \emph{deux} \omterm{rem:g10:signals-and-waves:honest}{échos}, à
        \qty{0.16}{s} et \qty{0.20}{s}. Les interpréter~; à quelle
        hauteur au-dessus du fond le banc de poissons nage-t-il~?
  \item Le \omterm{met:g10:signals-and-waves:echo}{sonar} envoie un ping toutes les \qty{0.50}{s}.
        Quelle est la plus grande profondeur qu'il peut mesurer sans
        confusion, et que se passe-t-il au-delà~?
  \item Dans l'air, le même délai de \qty{0.60}{s} signifierait quelle
        distance~? Morale~: que faut-il savoir avant de transformer un
        délai en distance~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le cardiogramme.}
\begin{enumerate}[resume]
  \item Les pics d'ECG du patron sont distants de \qty{0.86}{s}.
        \omterm{def:g10:signals-and-waves:frequency}{Fréquence}~? Battements par minute~?
  \item La bande avance à \qty{25}{mm/s}. Quel espacement de pics le
        médecin a-t-il mesuré~?
  \item Après avoir sorti les filets, l'espacement est \qty{15}{mm}.
        Nouveau rythme cardiaque~?
  \item Le jeune matelot, rameur entraîné, se repose à $50$ battements
        par minute. \omterm{def:g10:signals-and-waves:period}{Période}~? Espacement sur la bande~?
  \item Un ECG est-il strictement \omterm{def:g10:signals-and-waves:period}{périodique}~? Que lit
        alors le médecin sur le tracé~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Retour par la lumière.}
\begin{enumerate}[resume]
  \item Le bateau radio le port, à \qty{50}{km}. Combien de temps le
        message met-il~? Combien de temps le \omterm{def:g10:signals-and-waves:sound}{son}
        mettrait-il~?
  \item Un \omterm{def:g10:universal-gravitation:satellite}{satellite} GPS \omterm{def:g10:universal-gravitation:satellite}{orbite}
        à environ \qty{20200}{km} au-dessus. Combien de temps son
        \omterm{def:g10:signals-and-waves:signal}{signal} met-il pour atteindre le bateau~?
  \item Le GPS transforme le temps en position. Quelle erreur de
        distance une erreur d'horloge de \qty{1.0}{\micro s}
        provoque-t-elle~? Quelle précision de chronométrage une fix à
        \qty{3.0}{m} exige-t-elle~?
  \item Pourquoi aucun GPS fondé sur le son ne pourrait-il exister, même
        en principe~?
  \item Final~: lister les trois vitesses utilisées ce soir et la seule
        loi derrière les quatre instruments, puis énoncer l'habitude de
        télémétrie en une phrase.
\end{enumerate}
\end{problem}
