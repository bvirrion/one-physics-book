\chapter{Spectres de la lumière et le message de la lumière}\label{ch:g10:light-spectra}

Un faisceau de lumière stellaire est le seul échantillon d'une étoile que
nous tiendrons jamais, et il s'avère généreux. Étaler le faisceau en son
\omterm{def:g10:light-spectra:white}{spectre} --- un prisme de verre le fait, un rideau de gouttes de pluie le
fait --- et la lumière devient un message~: un fond d'arc-en-ciel
continu annonce la température de l'étoile, et un code-barres de fines
raies énonce, élément par élément, de quoi l'étoile est faite. Ce
chapitre apprend à lire les deux parties du message --- d'abord sur des
lampes de laboratoire, puis sur le Soleil.

\section{Lumière blanche et son spectre}

\begin{definition}[Décomposition de la lumière blanche]\label{def:g10:light-spectra:white}
Lorsqu'un faisceau étroit de lumière solaire traverse un prisme de
verre, il en ressort étalé en une bande continue de couleurs, du rouge
au violet. Cette bande est le \emph{spectre}\index{spectre} de la
lumière. Une lumière qui contient toutes les couleurs visibles, comme
la lumière du Soleil ou celle d'une ampoule ordinaire, s'appelle
\emph{lumière blanche}\index{lumière blanche}. Une lumière d'une seule
couleur pure, qu'aucun prisme ne peut décomposer davantage, est
\emph{monochromatique}\index{lumière monochromatique} (un faisceau
laser, par exemple)~; un mélange de plusieurs couleurs est
\emph{polychromatique}\index{lumière polychromatique}. Un instrument
qui effectue la décomposition et lit la composition d'une lumière est
un \emph{spectroscope}\index{spectroscope}.
\end{definition}

\begin{remark}[Prismes et gouttes de pluie]\label{rem:g10:light-spectra:rainbow}
Le prisme fonctionne parce que le verre courbe la lumière violette un
peu plus que la lumière rouge --- \emph{pourquoi} il le fait appartient
à l'étude de la réfraction, au~\cref{ch:g10:refraction}~; ici l'on
n'utilise que l'effet. L'eau fait le même travail~: chaque goutte d'une
averse qui passe agit comme un petit prisme, recevant la lumière solaire
blanche et la renvoyant triée par couleur. Un arc-en-ciel est le
\omterm{def:g10:light-spectra:white}{spectre} du Soleil, dessiné dans le ciel.
\end{remark}

\begin{definition}[Longueur d'onde, carte d'identité d'un rayonnement]\label{def:g10:light-spectra:wavelength}
Chaque rayonnement \omterm{def:g10:light-spectra:white}{monochromatique} est caractérisé par un nombre, sa
\emph{longueur d'onde}\index{longueur d'onde} $\lambda$, mesurée en
nanomètres ($\qty{1}{nm} = \qty{e-9}{m}$). La longueur d'onde est la
carte d'identité du rayonnement~: donner $\lambda$, c'est tout dire de
la couleur pure. L'œil répond aux longueurs d'onde entre environ
$\qty{400}{nm}$ (violet) et $\qty{800}{nm}$ (rouge) --- le
\emph{spectre visible}\index{spectre visible}. Au-delà des deux
extrémités le rayonnement continue, invisible pour nous~:
\emph{ultraviolet}\index{ultraviolet} en dessous de $\qty{400}{nm}$,
\emph{infrarouge}\index{infrarouge} au-dessus de $\qty{800}{nm}$. Ce
que la longueur d'onde mesure réellement --- la longueur d'une
ondulation d'une onde lumineuse --- est expliqué au~\cref{ch:g10:signals-and-waves}~; dans ce chapitre, c'est le
nombre lui-même qui compte.
\end{definition}

\begin{example}[Lire les longueurs d'onde]\label{ex:g10:light-spectra:classify}
Un pointeur laser vert émet à $\qty{532}{nm}$~: dans le visible, dans le
vert. Une lampe germicide au mercure émet à $\qty{254}{nm}$~:
\omterm{def:g10:light-spectra:wavelength}{ultraviolet}, invisible (et dangereuse pour l'œil précisément parce
qu'invisible). La diode d'une télécommande de télévision émet à
$\qty{940}{nm}$~: \omterm{def:g10:light-spectra:wavelength}{infrarouge} --- pointez-la vers l'appareil photo d'un
téléphone, dont le capteur voit un peu au-delà de $\qty{800}{nm}$, et
les flashs invisibles apparaissent à l'écran.
\end{example}

\section{Spectres continus et température}

\begin{definition}[Spectre continu]\label{def:g10:light-spectra:continuous}
La matière dense et chaude --- le filament d'une ampoule, l'acier en
fusion, la surface d'une étoile --- brille, et sa lumière contient
\emph{toutes} les \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} sur un large intervalle~: une bande
de couleur ininterrompue, sans rien qui manque. Un tel
\omterm{def:g10:light-spectra:white}{spectre} est un \emph{spectre continu}\index{spectre continu}. Il ne
porte aucune trace de la nature chimique du corps~: une barre de fer
blanche et une barre de cuivre blanche brillent de la même façon. Ce
qu'il encode --- entièrement --- c'est la température.
\end{definition}

\begin{definition}[Température absolue]\label{def:g10:light-spectra:kelvin}
Pour les lois du rayonnement, la \emph{température
absolue}\index{température absolue} se compte non pas à partir du point
de congélation de l'eau mais à partir de l'état le plus froid possible,
$\qty{-273}{\degreeCelsius}$. L'\omterm{def:g10:orders-of-magnitude:unit}{unité} est le
\emph{kelvin}\index{kelvin} (symbole $\unit{K}$), avec le même écart de
degré~:
\[
T \text{ (en } \unit{K}) = \theta \text{ (en } \unit{\degreeCelsius}) + 273 .
\]
Ainsi une pièce à $\qty{20}{\degreeCelsius}$ est à $\qty{293}{K}$, et
la surface du Soleil, près de $\qty{5500}{\degreeCelsius}$, est près de
$\qty{5800}{K}$.
\end{definition}

\begin{proposition}[Plus chaud signifie plus brillant et plus bleu (loi de Wien)]\label{prop:g10:light-spectra:wien}
Lorsque la température d'un corps dense lumineux s'élève,
\begin{enumerate}
  \item il rayonne davantage à \emph{toute} \omterm{def:g10:light-spectra:wavelength}{longueur d'onde}~: le
        \omterm{def:g10:light-spectra:white}{spectre} entier s'éclaircit~;
  \item la \omterm{def:g10:light-spectra:wavelength}{longueur d'onde} $\lambda_{\max}$ à laquelle il rayonne le
        plus fort glisse vers le côté des \omterm{def:g10:light-spectra:wavelength}{courtes longueurs d'onde}
        (bleu), selon
        \[
        \lambda_{\max} \times T = \qty{2.90e-3}{m.K},
        \]
        où $T$ est la \omterm{def:g10:light-spectra:kelvin}{température absolue} en \omterm{def:g10:light-spectra:kelvin}{kelvins}.
\end{enumerate}
\end{proposition}
\admitted

\begin{remark}[D'où vient cette loi]\label{rem:g10:light-spectra:wienorigin}
La loi de Wien résume des mesures soigneuses sur des fours et des
filaments à la fin du dix-neuvième siècle. La dériver exige la théorie
quantique du rayonnement --- ce \omterm{def:g10:light-spectra:white}{spectre} est, en fait, précisément ce
que la physique classique \emph{ne pouvait pas} expliquer, et l'énigme
qui força Planck à inventer le quantum. La dérivation honnête est
menée dans le volume de 3\textsuperscript{e} année.
\end{remark}

\section{Spectres de raies des gaz excités}

\begin{definition}[Spectre d'émission de raies]\label{def:g10:light-spectra:emission}
Un gaz à basse pression, excité par une décharge électrique (comme dans
un tube au néon ou une lampe de rue au sodium), brille aussi --- mais
à travers un \omterm{def:g10:light-spectra:white}{spectroscope} sa lumière n'a rien d'un arc-en-ciel.
Seules quelques \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} isolées sont présentes~: de fines
raies brillantes sur fond noir, appelées \emph{raies
spectrales}\index{raie spectrale}. Un tel \omterm{def:g10:light-spectra:white}{spectre} est un
\emph{spectre d'émission de raies}\index{spectre d'émission}\index{spectre
de raies}.
\end{definition}

\begin{example}[Hydrogène et sodium]\label{ex:g10:light-spectra:hydrogen}
L'hydrogène excité émet exactement quatre raies dans le visible~: une
raie rouge à $\qty{656}{nm}$, une raie bleu-vert à $\qty{486}{nm}$, et
deux raies violettes à $\qty{434}{nm}$ et $\qty{410}{nm}$. Le sodium
excité émet essentiellement une forte raie jaune-orange à
$\qty{589}{nm}$ --- c'est pourquoi les anciennes lampes de rue baignent
des quartiers entiers dans cette seule couleur~: leur lumière ne
contient presque rien d'autre.
\end{example}

\begin{proposition}[L'empreinte digitale des éléments]\label{prop:g10:light-spectra:fingerprint}
Chaque élément chimique émet son propre catalogue fixe de \omterm{def:g10:light-spectra:emission}{raies
spectrales}, le même dans tout laboratoire et à toute époque, et aucun
élément ne partage le même catalogue. Un \omterm{def:g10:light-spectra:emission}{spectre de raies}
identifie donc l'élément émetteur, comme une empreinte digitale
identifie une personne.
\end{proposition}
\admitted

\begin{remark}[Pourquoi des raies~?]\label{rem:g10:light-spectra:whylines}
Pourquoi un atome n'émet que certaines \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} --- et
pourquoi ces \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} trahissent l'élément --- s'explique
par la théorie quantique de l'atome, dans la dernière année de ce
volume~: chaque élément possède une échelle de niveaux d'énergie, et
chaque raie marque un saut entre barreaux. Pour l'instant on utilise
l'empreinte comme le font les détectives~: sans demander aux doigts
comment ils ont poussé.
\end{remark}

\section{Spectres d'absorption}

\begin{definition}[Spectre d'absorption]\label{def:g10:light-spectra:absorption}
Envoyer de la \omterm{def:g10:light-spectra:white}{lumière blanche} \emph{à travers} un gaz froid à basse
pression et disperser ce qui en sort~: l'arc-en-ciel continu
réapparaît, mais strié de fines raies sombres. Le gaz a retiré certaines
\omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} de la lumière qui le traverse. Un tel
\omterm{def:g10:light-spectra:white}{spectre} --- un fond continu moins un ensemble de raies --- est un
\emph{spectre d'absorption}\index{spectre d'absorption}.
\end{definition}

\begin{proposition}[Les raies d'émission et d'absorption coïncident]\label{prop:g10:light-spectra:kirchhoff}
Un gaz absorbe exactement les \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} qu'il est capable
d'émettre~: les raies sombres du \omterm{def:g10:light-spectra:absorption}{spectre d'absorption} d'un élément
siègent aux mêmes \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} que les raies brillantes de son
\omterm{def:g10:light-spectra:emission}{spectre d'émission}. L'empreinte du~\cref{prop:g10:light-spectra:fingerprint} se lit
donc dans le sombre aussi bien que dans le brillant.
\end{proposition}
\admitted

\begin{remark}[Un mécanisme, deux signes]\label{rem:g10:light-spectra:mechanism}
La coïncidence n'est pas un accident~: absorber à $\lambda$ gravit le
même barreau d'énergie que descendre en émettant à $\lambda$ --- les
chapitres quantiques à la fin de ce volume le rendent précis. Établie
par Kirchhoff et Bunsen en 1859, c'est la clé qui ouvre toute la section
suivante.
\end{remark}

\begin{example}[Une fiole de vapeur de sodium]\label{ex:g10:light-spectra:sodiumflask}
De la \omterm{def:g10:light-spectra:white}{lumière blanche} traverse une fiole contenant de la vapeur de

à cette même \omterm{def:g10:light-spectra:wavelength}{longueur d'onde}.
\end{example}

\section{Analyse spectrale~: lire une étoile}

\begin{definition}[Analyse spectrale]\label{def:g10:light-spectra:analysis}
L'\emph{analyse spectrale}\index{analyse spectrale} est l'identification
des éléments chimiques contenus dans une source (une flamme, une lampe,
une étoile) en confrontant les raies de son \omterm{def:g10:light-spectra:white}{spectre} aux catalogues de
laboratoire des éléments.
\end{definition}

\begin{method}[Lire un spectre stellaire]\label{met:g10:light-spectra:reading}
Une étoile est une boule de matière dense et chaude (la surface)
enveloppée d'une atmosphère plus mince et plus froide. Sa lumière
arrive donc comme un \omterm{def:g10:light-spectra:continuous}{spectre continu} strié de raies d'absorption
--- et chaque partie se lit séparément~:
\begin{enumerate}
  \item Localiser le pic $\lambda_{\max}$ du fond continu~; la loi de
        Wien (\cref{prop:g10:light-spectra:wien}) donne la température
        de surface $T = \qty{2.90e-3}{m.K} / \lambda_{\max}$.
  \item Lister les \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} des raies sombres.
  \item Les confronter aux catalogues~: un élément ne peut être déclaré
        présent que si \emph{toutes} ses raies fortes apparaissent.
  \item Conclure~: l'absorption a lieu dans l'atmosphère de l'étoile,
        donc les éléments identifiés sont ceux de l'atmosphère.
\end{enumerate}
\end{method}

\begin{example}[La recette du Soleil]\label{ex:g10:light-spectra:sunrecipe}
Le \omterm{def:g10:light-spectra:white}{spectre} solaire porte des milliers de raies sombres, cartographiées
par Fraunhofer en 1814. Parmi les plus fortes~: $410$, $434$, $486$ et
$\qty{656}{nm}$ --- l'empreinte complète de l'hydrogène --- ainsi que
la raie du sodium à $\qty{589}{nm}$ et une paire de calcium à $393$ et
$\qty{397}{nm}$. L'atmosphère du Soleil contient surtout de
l'hydrogène, plus des traces d'éléments familiers des roches
terrestres. Les mesures modernes affinent le titre~: le Soleil est
environ aux trois quarts de l'hydrogène en masse.
\end{example}

\begin{remark}[L'élément trouvé d'abord dans le ciel]\label{rem:g10:light-spectra:helium}
Pendant l'éclipse de 1868, une raie brillante à $\qty{587.6}{nm}$
apparut dans le \omterm{def:g10:light-spectra:white}{spectre} de l'atmosphère du Soleil --- ne correspondant
à aucun élément connu. Par le~\cref{prop:g10:light-spectra:fingerprint}, une
empreinte inconnue signifie un élément inconnu, aussi en annonça-t-on
un, nommé d'après le soleil grec, \emph{helios}~: l'hélium. Il fut
enfin isolé sur Terre en 1895 --- le seul élément chimique découvert
dans l'espace avant d'être trouvé chez soi.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:light-spectra:1}
Pour chaque rayonnement, dire s'il est \omterm{def:g10:light-spectra:wavelength}{ultraviolet}, visible ou
\omterm{def:g10:light-spectra:wavelength}{infrarouge}, et donner sa couleur s'il est visible~:
$\qty{254}{nm}$ (lampe au mercure), $\qty{480}{nm}$, $\qty{589}{nm}$,
$\qty{656}{nm}$, $\qty{1064}{nm}$ (laser de découpe).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:light-spectra:2}
Un faisceau étroit de lumière solaire tombe sur un prisme de verre.
\begin{enumerate}
  \item Décrire ce qui apparaît sur un écran placé derrière le prisme.
  \item Quelle extrémité de la bande a été le plus déviée~?
  \item La lumière solaire est remplacée par le faisceau d'un pointeur
        laser vert. Qu'apparaît-il maintenant sur l'écran, et que cela
        montre-t-il de la lumière laser~?
\end{enumerate}
\end{exercise}

\section{Problème~: Lire la lumière des étoiles}

\begin{problem}[{Problème du week-end --- lire la lumière des étoiles~:
le philosophe qui disait «~jamais~», et comment un prisme prend la
température d'une étoile et lit sa recette chimique}]
\label{pb:g10:light-spectra:1}
En 1835 le philosophe Auguste Comte avait besoin d'un exemple de
connaissance à jamais hors de portée humaine, et le choisit avec soin~:
la composition chimique des étoiles. Personne ne mettrait jamais une
étoile en bouteille. En 1859 Kirchhoff et Bunsen confrontèrent les
raies sombres du Soleil aux raies brillantes des flammes de laboratoire,
et la connaissance inatteignable arriva par retour de lumière. Ce
problème retrace tout le chemin~: empreintes de laboratoire, corps
lumineux, raies sombres du Soleil --- et enfin une étoile dont vous
déterminerez température et composition depuis votre bureau.

\medskip
\noindent\textbf{Partie I --- Empreintes au dossier.}
\begin{enumerate}
  \item Donner l'intervalle visible des \omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} en
        nanomètres et la couleur à chaque extrémité. Où se trouve
        $\qty{550}{nm}$~?
  \item Une lampe à décharge d'hydrogène montre des raies à $410$,
        $434$, $486$ et $\qty{656}{nm}$. Donner la couleur de chaque
        raie.
  \item Une lampe au sodium émet essentiellement une raie à
        $\qty{589}{nm}$. À quoi ressemble la lampe pour l'œil, et
        pourquoi est-il vain de juger la couleur d'une voiture garée
        sous un tel lampadaire~?
  \item Expliquer pourquoi un ensemble de \omterm{def:g10:light-spectra:emission}{raies spectrales}
        identifie un élément --- et pourquoi \emph{toutes} les raies
        fortes d'un élément doivent être trouvées avant de le déclarer
        présent.
  \item Prédire le \omterm{def:g10:light-spectra:white}{spectre} vu (a) lorsque de la \omterm{def:g10:light-spectra:white}{lumière blanche}
        traverse une fiole de vapeur de sodium froide, et (b) lorsque
        la même vapeur, seule dans une pièce sombre, est excitée par
        une décharge.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Corps chauds et leurs couleurs.}
\begin{enumerate}[resume]
  \item Énoncer les deux effets d'une élévation de température d'un
        corps dense lumineux sur son \omterm{def:g10:light-spectra:continuous}{spectre continu}.
  \item Un filament halogène tourne à $\qty{3400}{K}$. Calculer
        $\lambda_{\max}$ et indiquer son domaine.
  \item Le \omterm{def:g10:light-spectra:continuous}{spectre continu} du Soleil culmine à $\qty{500}{nm}$.
        Calculer sa température de surface.
  \item Ce pic se trouve dans le vert --- pourtant le Soleil ne paraît
        pas vert. Expliquer.
  \item Un tison laissé dans la forge est d'abord senti à distance sans
        être vu briller, puis brille d'un rouge sombre, puis
        orange-blanc. Expliquer toute la séquence avec la question~6.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Les raies sombres du Soleil.}
\begin{enumerate}[resume]
  \item Le \omterm{def:g10:light-spectra:white}{spectre} solaire est une bande continue traversée de
        milliers de fines raies sombres. Quelle partie du Soleil
        produit le fond continu, et quelle partie produit les raies~?
  \item De fortes raies solaires se trouvent à $393$, $397$, $410$,
        $434$, $486$, $589$ et $\qty{656}{nm}$. Catalogues~: hydrogène
        $\{410, 434, 486, 656\}\,\unit{nm}$~; sodium
        $\{589\}\,\unit{nm}$~; calcium $\{393, 397\}\,\unit{nm}$.
        Quels éléments sont présents dans l'atmosphère du Soleil~?
  \item Les raies de l'hydrogène sont de loin les plus fortes. Que cela
        suggère-t-il de la composition du Soleil --- et que disent les
        mesures modernes~?
  \item En 1868, une raie à $\qty{587.6}{nm}$ observée dans
        l'atmosphère du Soleil ne correspondait à aucun catalogue.
        Reconstituer le raisonnement qui justifia d'annoncer un nouvel
        élément, et dire comment l'histoire se termina.
  \item Pourquoi les raies solaires sont-elles sombres, alors que les
        mêmes atomes dans une décharge de laboratoire donnent des raies
        brillantes~? Pendant une éclipse totale, pendant quelques
        secondes, seule la mince atmosphère du Soleil reste visible
        contre un ciel sombre --- et les mêmes raies flambent
        \emph{brillantes}. Expliquer.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Une étoile sur le bureau.}
Le \omterm{def:g10:light-spectra:white}{spectre} de l'étoile brillante Véga est placé devant vous~: un fond
continu culminant à $\qty{290}{nm}$, strié de fortes raies sombres à
$410$, $434$, $486$ et $\qty{656}{nm}$.
\begin{enumerate}[resume]
  \item Calculer la température de surface de Véga.
  \item Le pic se trouve dans l'\omterm{def:g10:light-spectra:wavelength}{ultraviolet}. Prédire la couleur
        apparente de l'étoile, et concilier «~pic hors du visible~» avec
        «~étoile brillante~».
  \item Identifier l'élément qui domine l'atmosphère de Véga.
  \item Comparer Véga au Soleil~: rapport des températures, couleur, et
        brillance par \omterm{def:g10:orders-of-magnitude:unit}{unité} de surface rayonnante.
  \item Final --- remplir la fiche d'identité que Comte déclarait
        impossible~: température, élément dominant, couleur apparente
        de Véga, le tout lu dans un seul faisceau de lumière. Combien
        d'années le «~jamais~» a-t-il duré~?
\end{enumerate}
\end{problem}
