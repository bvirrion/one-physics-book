\chapter{Universele zwaartekracht en gewicht}\label{ch:g10:universal-gravitation}

Laat een appel vallen: hij valt. De Maan, zestig Aardstralen hoog,
lijkt dat nooit te doen. Newtons grote inzicht is dat de Maan
\emph{valt} --- ongeveer $1{,}4$ millimeter elke seconde, precies de
hoeveelheid die een aantrekking die verzwakt als het kwadraat van de
afstand voorspelt. Eén formule, met één universele constante, regeert
de appel, de Maan, de getijden en de planeten. Het eerste deel
beschreef de \omterm{def:g10:universal-gravitation:force}{zwaartekracht} met woorden; dit
hoofdstuk meet haar.

\section{De wet van de universele zwaartekracht}

\begin{definition}[Zwaartekracht]\label{def:g10:universal-gravitation:force}
Elke twee lichamen trekken elkaar aan, ongeacht waaruit ze bestaan en
hoe ver ze uit elkaar staan. Deze aantrekking is de
\emph{zwaartekracht}\index{zwaartekracht}, en het verschijnsel is de
\emph{gravitatie}\index{gravitatie}. Ze heeft geen contact, geen
touw en geen medium nodig, en ze is \emph{wederzijds}: elk lichaam
trekt aan het andere.
\end{definition}

\begin{theorem}[Wet van de universele zwaartekracht]\label{thm:g10:universal-gravitation:law}
Twee lichamen van massa's $m_1$ en $m_2$ (in \unit{kg}), waarvan de
centra een afstand $d$ uit elkaar liggen (in \unit{m}), trekken
elkaar aan met twee krachten gericht langs de lijn die de centra
verbindt, elk wijzend naar het andere lichaam, van gelijke grootte
\[
F = G\,\frac{m_1 m_2}{d^2}
\qquad \text{(in \unit{N})},
\]
waarbij $G$ dezelfde constante is voor elk paar lichamen in het
heelal.
\end{theorem}
\admitted

\begin{remark}[Waar de wet vandaan komt]
Newton distilleerde deze wet uit de waargenomen beweging van de
planeten (wetten van Kepler); de eerlijke afleiding staat in het deel
van jaar 1. \Cref{ch:g12:satellites-kepler} zal de wet al aan het
werk zetten op \omterm{def:g10:universal-gravitation:satellite}{satellieten} en planeten.
\end{remark}

\begin{definition}[De gravitatieconstante]\label{def:g10:universal-gravitation:constant}
De constante
\[
G = \qty{6.67e-11}{N.m^2/kg^2}
\]
is de \emph{gravitatieconstante}\index{gravitatieconstante}. Haar
piepkleine waarde is de reden dat zwaartekracht tussen alledaagse
objecten onmerkbaar is: alleen wanneer minstens één massa astronomisch
is, bedraagt $G m_1 m_2 / d^2$ iets.
\end{definition}

\begin{remark}[De formule lezen]\label{rem:g10:universal-gravitation:reading}
Vier kenmerken verdienen aandacht.
\begin{itemize}
  \item \emph{Wederzijds en gelijk.} De Aarde trekt aan jou met jouw
        \omterm{def:g10:universal-gravitation:weight}{gewicht} --- en jij trekt aan de Aarde met precies
        dezelfde kracht. (De Aarde reageert alleen minder: ze is
        ongeveer $\num{e23}$ keer zwaarder.)
  \item \emph{Omgekeerd kwadraat.} $d$ verdubbelen deelt de kracht
        door $4$; tien keer verder delen door $100$.
  \item \emph{Afstand tussen centra.} Voor bolvormige lichamen ---
        sterren, planeten, kanonskogels --- meet men $d$ centrum tot
        centrum, niet oppervlak tot oppervlak.
  \item \emph{Beide massa's tellen.} Een van de massa's verdubbelen
        verdubbelt de kracht; de formule is symmetrisch in $m_1$ en
        $m_2$.
\end{itemize}
\end{remark}

\begin{example}[Twee vrienden]\label{ex:g10:universal-gravitation:friends}
Twee vrienden van \qty{70}{kg} staan met hun centra \qty{1.0}{m} uit
elkaar:
\[
F = \num{6.67e-11} \times \frac{70 \times 70}{1.0^2}
\approx \qty{3.3e-7}{N}.
\]
Het \omterm{def:g10:universal-gravitation:weight}{gewicht} van elke vriend is ongeveer \qty{6.9e2}{N}
--- twee \emph{miljard} keer groter. Hun wederzijdse aantrekking is
echt, maar hopeloos te voelen.
\end{example}

\begin{example}[De Aarde houdt de Maan]\label{ex:g10:universal-gravitation:earthmoon}
Met $m_1 = \qty{5.97e24}{kg}$ (Aarde), $m_2 = \qty{7.35e22}{kg}$
(Maan) en $d = \qty{3.84e8}{m}$:
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{2.0e20}{N}.
\]
En de Maan trekt de Aarde terug met dezelfde \qty{2.0e20}{N} --- een
trek die de oceanen twee keer per dag beantwoorden, als de getijden.
\end{example}

\begin{method}[Een zwaartekracht berekenen]\label{met:g10:universal-gravitation:compute}
\begin{enumerate}
  \item Zet de massa's om naar \omterm{def:g10:orders-of-magnitude:unit}{kilogram} en de
        centrum-tot-centrum-afstand naar meters.
  \item Pas $F = G m_1 m_2 / d^2$ toe.
  \item Controleer de macht van tien apart van de cijfers, zoals in
        \cref{ch:g10:orders-of-magnitude}:
        \omterm{def:g10:universal-gravitation:force}{zwaartekrachten} lopen van $\num{e-7}$ newton
        (twee mensen een \omterm{def:g10:orders-of-magnitude:unit}{meter} uit elkaar) tot $\num{e22}$
        newton (Zon op Aarde), dus een verkeerde
        \omterm{def:g10:orders-of-magnitude:scinot}{exponent} is makkelijk te vangen.
\end{enumerate}
\end{method}

\section{Gewicht en de sterkte van de zwaartekracht}

\begin{definition}[Gewicht]\label{def:g10:universal-gravitation:weight}
Het \emph{gewicht}\index{gewicht} $\vect P$ van een object aan het
oppervlak van een hemellichaam (een planeet, een maan, \dots) is de
\omterm{def:g10:universal-gravitation:force}{zwaartekracht} die het hemellichaam erop uitoefent. Het
wijst naar het centrum van het hemellichaam --- de richting die we
\emph{verticaal}\index{verticaal} noemen --- en wordt in newton
gemeten.
\end{definition}

\begin{proposition}[Gewicht aan het oppervlak van een hemellichaam]\label{prop:g10:universal-gravitation:surfaceg}
Aan het oppervlak van een bolvormig hemellichaam van massa $M$ en
straal $R$ heeft een object van massa $m$ \omterm{def:g10:universal-gravitation:weight}{gewicht}
\[
P = m\,g,
\qquad\text{waarbij}\qquad
g = \frac{G M}{R^2}
\]
alleen van het hemellichaam afhangt, niet van het object.
\end{proposition}

\begin{proof}
Het centrum van het object en dat van het hemellichaam liggen een
afstand $R$ uit elkaar, dus de wet van de universele
\omterm{def:g10:universal-gravitation:force}{zwaartekracht}
(\cref{thm:g10:universal-gravitation:law}) geeft
$P = G M m / R^2 = m \times \left(G M / R^2\right)$. De haak is
hetzelfde voor elk object aan het oppervlak: noem hem $g$.
\end{proof}

\begin{definition}[Zwaarteveldsterkte]\label{def:g10:universal-gravitation:fieldstrength}
De grootheid $g = G M / R^2$, in newton per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram}, is de \emph{zwaarteveldsterkte}\index{zwaarteveldsterkte}
aan het oppervlak van het hemellichaam: het hemellichaam trekt aan
elk \omterm{def:g10:orders-of-magnitude:unit}{kilogram} met $g$ newton. Op Aarde is
$g = \qty{9.81}{N/kg}$.
\end{definition}

\begin{example}[$g$ berekenen]\label{ex:g10:universal-gravitation:gvalues}
Voor de Aarde ($M = \qty{5.97e24}{kg}$, $R = \qty{6.37e6}{m}$):
\[
g = \frac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.37e6})^2}
= \frac{\num{3.98e14}}{\num{4.06e13}}
\approx \qty{9.81}{N/kg}.
\]
Voor de Maan ($M = \qty{7.35e22}{kg}$, $R = \qty{1.74e6}{m}$) geeft
dezelfde berekening $g_{\text{Moon}} \approx \qty{1.62}{N/kg}$ ---
zes keer minder: de Maan is veel lichter, en kleiner zijn compenseert
dat niet. Voor Mars is $g_{\text{Mars}} \approx \qty{3.73}{N/kg}$.
\end{example}

\begin{remark}[Waarom astronauten huppelen]\label{rem:g10:universal-gravitation:bounce}
Een astronaut van massa \qty{120}{kg} (lichaam plus pak) weegt
$120 \times 9.81 \approx \qty{1180}{N}$ op Aarde maar slechts
$120 \times 1.62 \approx \qty{194}{N}$ op de Maan. Spieren getraind
onder \qty{1180}{N} dragen plotseling een zesde van de last: vandaar
de beroemde trage, huppelende maangang. De massa --- en daarmee de
inspanning om te \emph{stoppen} of te \emph{draaien} --- is
onveranderd, daarom vielen maanastronauten zo vaak om.
\end{remark}

\section{Massa of gewicht? De balans en de veerweegschaal}

De alledaagse taal zegt dat een zak meel «drie kilo weegt». De
natuurkunde splitst die zin in twee.

\begin{remark}[Massa is geen gewicht]\label{rem:g10:universal-gravitation:massweight}
\begin{itemize}
  \item De \emph{massa} $m$ meet de hoeveelheid materie, in
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram}. Ze is een eigenschap van het object
        alleen: dezelfde op Aarde, op de Maan, of drijvend in de
        diepe ruimte.
  \item Het \emph{\omterm{def:g10:universal-gravitation:weight}{gewicht}} $P = mg$ is een kracht, in
        newton, uitgeoefend \emph{door een hemellichaam} op het
        object. Het verandert van hemellichaam tot hemellichaam met
        $g$ --- en vervaagt ver van elk hemellichaam.
\end{itemize}
Ze verwarren is onschuldig op de markt en fataal in een
natuurkundeprobleem: de \omterm{def:g10:orders-of-magnitude:unit}{kilogram} is geen
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} van kracht.
\end{remark}

\begin{method}[De balans en de veerweegschaal]\label{met:g10:universal-gravitation:instruments}
\begin{itemize}
  \item Een \emph{balkbalans} vergelijkt het object met standaardmassa's:
        de \omterm{def:g10:universal-gravitation:weight}{gewichten} van beide schalen schalen met
        dezelfde $g$, dus de vergelijking meet de \emph{massa} ---
        hetzelfde oordeel op elk hemellichaam.
  \item Een \emph{veerweegschaal} meet de kracht die haar veer
        uitrekt: ze leest het \emph{\omterm{def:g10:universal-gravitation:weight}{gewicht}}, in
        newton, en haar aflezing verandert van hemellichaam tot
        hemellichaam. Om de massa te herwinnen, deel door de lokale
        veldsterkte: $m = P/g$.
\end{itemize}
\end{method}

\begin{example}[Meel op de Maan]\label{ex:g10:universal-gravitation:flour}
Een zak meel van \qty{3.0}{kg} balanceert tegen \qty{3.0}{kg}
standaardmassa's op Aarde \emph{en} op de Maan. Een veerweegschaal
echter leest $3.0 \times 9.81 \approx \qty{29}{N}$ op Aarde en
$3.0 \times 1.62 \approx \qty{4.9}{N}$ op de Maan. Een handelaar die
«per newton» op de Maan verkoopt, zou zes keer zoveel meel meegeven
--- handel, net als wetenschap, loopt op massa.
\end{example}

\section{Vallen rond de Aarde: Newtons kanon}

Gooi een steen horizontaal: de zwaartekracht buigt zijn pad tot een
boog en hij landt enkele meters verder. Gooi sneller: hij landt
verder. Newtons gedachte-experiment duwt het idee tot het uiterste.
De Aarde is rond, en haar oppervlak zakt ongeveer \qty{5}{m} onder de
horizontale per \qty{8}{km} afgelegd; een vrij vallend lichaam daalt
\qty{4.9}{m} in zijn eerste seconde
(\cref{ch:g12:projectile-motion} zal dit vaststellen). Dus een
projectiel dat de Aarde scheert met \qty{8}{km/s} valt precies zo
snel als de grond wegkromt: het valt eeuwig zonder ooit te landen.

\begin{definition}[Satelliet en baan]\label{def:g10:universal-gravitation:satellite}
Een \emph{satelliet}\index{satelliet} van een hemellichaam is een
lichaam dat rond het hemellichaam valt zonder ooit haar oppervlak te
bereiken; het pad dat het herhaalt is zijn \emph{baan}\index{baan}.
\end{definition}

\begin{remark}[De Maan is een vallende steen]\label{rem:g10:universal-gravitation:fallingmoon}
De Maan is de natuurlijke \omterm{def:g10:universal-gravitation:satellite}{satelliet} van de Aarde:
ze valt naar ons toe met ongeveer \qty{1.4}{mm} elke seconde, en haar
zijwaartse beweging voert haar rond voordat ze dichterbij kan komen
(\cref{exo:g10:universal-gravitation:15} laat je Newtons eigen check van
dit getal herdoen). De astronauten van een omloopstation zijn in
dezelfde vrije val als hun schip --- dat, en niet enige afwezigheid
van zwaartekracht, is waarom ze zweven.
\omterm{def:g10:universal-gravitation:force}{Zwaartekracht} vormt elke baan, van de boog van de
steen tot de ellips van de planeet; het kwantitatieve verhaal van
\omterm{def:g10:universal-gravitation:satellite}{banen} wordt verteld in
\cref{ch:g12:satellites-kepler}.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:1}
Twee danspartners van \qty{60}{kg} houden elkaar vast met hun centra
\qty{0.80}{m} uit elkaar.
\begin{enumerate}
  \item Bereken hun wederzijdse zwaartekrachtige aantrekking.
  \item Vergelijk die met het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een
        mug van \qty{2.5}{mg}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:2}
Een student heeft een massa van \qty{55}{kg}.
\begin{enumerate}
  \item Bereken haar \omterm{def:g10:universal-gravitation:weight}{gewicht} op Aarde.
  \item Ze reist naar de Maan. Wat zijn daar haar massa en haar
        \omterm{def:g10:universal-gravitation:weight}{gewicht}?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:3}
\begin{enumerate}
  \item Herbereken uit de gegevenskaart de
        \omterm{def:g10:universal-gravitation:fieldstrength}{zwaarteveldsterkte} aan het oppervlak van Mars.
  \item Een rover van \qty{900}{kg} wordt daarheen gestuurd. Bereken
        haar \omterm{def:g10:universal-gravitation:weight}{gewicht} op Mars en haar
        \omterm{def:g10:universal-gravitation:weight}{gewicht} op Aarde.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:4}
Bereken met de gegevenskaart de kracht die de Aarde op de Maan
uitoefent. Welke kracht oefent de Maan op de Aarde uit?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:5}
Twee lichamen trekken elkaar aan met een kracht $F_0$. Wat wordt de
kracht als:
\begin{enumerate}
  \item de afstand tussen de centra verdubbeld wordt;
  \item de afstand gehalveerd wordt;
  \item beide massa's verdubbeld worden (zelfde afstand);
  \item één massa verdrievoudigd wordt \emph{en} de afstand
        verdrievoudigd?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:6}
Twee beladen super tankers van elk \qty{5.0e8}{kg} liggen afgemeerd
met hun centra \qty{100}{m} uit elkaar.
\begin{enumerate}
  \item Bereken hun wederzijdse zwaartekrachtige aantrekking.
  \item Welke massa heeft een \omterm{def:g10:universal-gravitation:weight}{gewicht} gelijk aan
        deze kracht op Aarde?
  \item De kracht is verrassend groot --- toch tonen de tankers geen
        neiging om naar elkaar toe te drijven. Bereken de versnelling
        $a = F/m$ die ze aan één tanker geeft en becommentarieer.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:7}
Een maanwinkel verkoopt een zak rijst van \qty{3.0}{kg}.
\begin{enumerate}
  \item Wat geven een balkbalans en een veerweegschaal voor deze zak
        op Aarde? Op de Maan?
  \item Een klant betaalt voor «\qty{3.0}{kg}». Welk instrument
        garandeert een eerlijke deal op elke wereld, en waarom?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:8}
Het Internationaal Ruimtestation vliegt op een hoogte van
\qty{400}{km}.
\begin{enumerate}
  \item Bereken de \omterm{def:g10:universal-gravitation:fieldstrength}{zwaarteveldsterkte} van de Aarde
        op die hoogte. (Let op: wat is de afstand tot het
        \emph{centrum} van de Aarde?)
  \item Druk haar uit als percentage van de oppervlaktewaarde.
  \item De astronauten aan boord zweven. Verzoen dit met je antwoord.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:9}
Aan het oppervlak van Venus is $g = \qty{8.87}{N/kg}$, en de straal
van de planeet is \qty{6.05e6}{m}. Leid de massa van Venus af en
vergelijk die met die van de Aarde.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:10}
\begin{enumerate}
  \item Bereken de kracht die de Zon uitoefent op \qty{1.0}{kg}
        zeewater, daarna de kracht die de Maan uitoefent op hetzelfde
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram}.
  \item De Zon trekt harder --- met welke factor?
  \item Toch domineert de Maan de getijden. Het getij wordt niet door
        de trek zelf aangedreven maar door het \emph{verschil} tussen
        de trekken aan de nabije en verre kant van de Aarde. Leg uit,
        zonder te rekenen, waarom de nabije Maan de verre Zon op dat
        criterium kan verslaan.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:11}
\begin{enumerate}
  \item Een planeet heeft twee keer de massa van de Aarde en twee
        keer haar straal. Druk haar oppervlakteveldsterkte uit in
        termen van de $g$ van de Aarde, daarna in \unit{N/kg}.
  \item Welke straal zou een planeet van de massa van de Aarde nodig
        hebben om een oppervlakteveldsterkte van $2g$ te hebben?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:12}
Ergens op de Aarde--Maan-lijn heffen de twee trekken op een
ruimtesonde elkaar op. Laat $x$ de afstand van het centrum van de
Aarde tot dat punt zijn en $D$ de Aarde--Maan-afstand.
\begin{enumerate}
  \item Toon dat in dat punt
        $\left(\dfrac{x}{D - x}\right)^{2}
        = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.
  \item Leid $x$ af. Welke fractie van de reis naar de Maan is dat?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:13}
Newtons kanon, met cijfers. Een projectiel scheert de Aarde
horizontaal met snelheid $v$.
\begin{enumerate}
  \item Na $x = \qty{8.0}{km}$ horizontale reis is de bolvormige
        Aarde «weggezakt» met een hoogte $h$ die voldoet aan
        $(R + h)^2 = R^2 + x^2$. Met
        $(R + h)^2 \approx R^2 + 2Rh$ voor kleine $h$, toon dat
        $h \approx x^2 / (2R)$ en bereken $h$.
  \item Een vrij vallend lichaam daalt \qty{4.9}{m} in zijn eerste
        seconde. Vergelijk met $h$ en leid de snelheid af waarbij
        het projectiel nooit landt.
  \item Bereken de duur van één volledige
        \omterm{def:g10:universal-gravitation:satellite}{baan} bij die snelheid, scherend over
        het oppervlak.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:14}

van de Zon) in een straal van \qty{12}{km}.
\begin{enumerate}
  \item Bereken de \omterm{def:g10:universal-gravitation:fieldstrength}{zwaarteveldsterkte} aan haar
        oppervlak.
  \item Wat zou een mens van \qty{70}{kg} daar wegen? Vergelijk met
        hun \omterm{def:g10:universal-gravitation:weight}{gewicht} op Aarde.
  \item Wat is daar het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een
        zandkorrel van \qty{1.0}{mg}? Welke massa heeft dat
        \omterm{def:g10:universal-gravitation:weight}{gewicht} op Aarde?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:15}
Newtons Maantest (1666). De \omterm{def:g10:universal-gravitation:satellite}{baan}straal van de
Maan is ongeveer $60$ Aardstralen.
\begin{enumerate}
  \item Met welke factor is de trek van de Aarde per
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram} zwakker bij de Maan dan aan het
        oppervlak van de Aarde, als de zwaartekracht het omgekeerde
        kwadraat volgt?
  \item Nabij het oppervlak daalt een vallend lichaam \qty{4.9}{m} in
        zijn eerste seconde. Hoe ver zou de Maan elke seconde naar de
        Aarde moeten vallen?
  \item Controleer tegen de \omterm{def:g10:universal-gravitation:satellite}{baan}: de Maan dekt
        haar cirkel van straal $d = \qty{3.84e8}{m}$ in $T = 27.3$
        dagen. Bereken haar snelheid $v$, de afstand $x$ die ze in
        één seconde aflegt, daarna de val $h \approx x^2/(2d)$
        (zoals in \cref{exo:g10:universal-gravitation:13}).
  \item Vergelijk de antwoorden van vragen 2 en 3, en zeg wat Newton

\end{enumerate}
\end{exercise}

\section{Probleem: De Aarde wegen}

\begin{problem}[{Weekendprobleem --- van twee loden bollen in een
schuur tot de massa van de Aarde en dan van de Zon: wat één kleine
constante, eenmaal gemeten, waard is}]
\label{pb:g10:universal-gravitation:1}
In 1798 hing Henry Cavendish een lichte stang met twee kleine loden
bollen op aan een dunne draad, bracht twee grote loden bollen dichterbij,
en mat de bijna onbestaande torsie van de draad. De kranten zeiden dat
hij de Aarde \emph{gewogen} had --- en ze hadden gelijk: eenmaal $G$
bekend, geven de $g$ en straal van de Aarde haar massa, en de
jaarlijkse \omterm{def:g10:universal-gravitation:satellite}{baan} van de Aarde geeft die van de
Zon. Dit probleem hertekent de hele keten. Gebruik overal de
gegevenskaart.

\medskip
\noindent\textbf{Deel I --- De zwakste kracht.}
\begin{enumerate}
  \item Het ijzer van de Eiffeltoren heeft een massa van ongeveer
        \qty{7.3e6}{kg}. Een toerist van \qty{55}{kg} staat
        \qty{100}{m} van haar massamiddelpunt. Bereken de trek van
        de toren op de toerist.
  \item Vergelijk die met het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een
        zandkorrel van \qty{1.0}{mg}.
  \item Evenredigheidsdrill: wat gebeurt er met een
        \omterm{def:g10:universal-gravitation:force}{zwaartekracht} als (a) de afstand met $10$
        vermenigvuldigd wordt; (b) beide massa's met $1000$
        vermenigvuldigd worden; (c) beide massa's \emph{en} de
        afstand met $1000$ vermenigvuldigd worden?
  \item Bereken de trek van de Aarde op een object van
        \qty{1.00}{kg} aan haar oppervlak, uit $M$, $R$ en $G$.
        Welke alledaagse naam en welk symbool draagt dit getal?
  \item Zwaartekracht is verreweg de zwakste wisselwerking die je
        ontmoet hebt --- en toch runt ze het heelal. Leg in één of
        twee zinnen uit waarom de trek van de Aarde op jou die van
        je buur domineert.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De balans van Cavendish.}
In de torsiebalans trekt een grote bol van massa $m_1 = \qty{158}{kg}$
een kleine van massa $m_2 = \qty{0.73}{kg}$ aan; hun centra liggen
$d = \qty{0.23}{m}$ uit elkaar.
\begin{enumerate}[resume]
  \item Met de moderne waarde van $G$, bereken de kracht tussen de
        twee bollen.
  \item Bereken het \omterm{def:g10:universal-gravitation:weight}{gewicht} van de kleine bol en
        de verhouding van de twee krachten. Waarom had Cavendish een
        torsiedraad nodig in plaats van elke gewone weegschaal?
  \item De gemeten torsie van Cavendish komt (in moderne
        \omterm{def:g10:orders-of-magnitude:unit}{eenheden}) overeen met $F = \qty{1.47e-7}{N}$.
        Leid zijn waarde van $G$ af en vergelijk die met de huidige.
  \item Met $G = \qty{6.67e-11}{N.m^2/kg^2}$, $g = \qty{9.81}{N/kg}$
        en $R = \qty{6.37e6}{m}$, leid de massa van de Aarde af.
        (Dit is de stap die de kranten vierden.)
  \item Leid de gemiddelde dichtheid van de Aarde af
        $\rho = M / V$, met $V = \frac43 \pi R^3$. Oppervlakterotsen
        hebben dichtheden rond \qty{2.7e3}{kg/m^3}: wat onthult de
        vergelijking over het diepe binnenste?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Andere werelden.}
\begin{enumerate}[resume]
  \item Herbereken uit de gegevenskaart de oppervlakteveldsterkte
        van de Maan.
  \item Op Aarde tilt een goede afzet je massamiddelpunt met
        \qty{0.50}{m}. Voor dezelfde afzet is de bereikte hoogte
        omgekeerd evenredig met $g$ (zoals het energiehoofdstuk,
        \cref{ch:g10:energy-conservation}, zal rechtvaardigen). Hoe
        hoog voert dezelfde sprong je op de Maan?
  \item Mercurius: $M = \qty{3.30e23}{kg}$, $R = \qty{2.44e6}{m}$.
        Bereken haar oppervlakteveldsterkte en vergelijk met Mars.
        Mercurius is veel kleiner dan Mars --- hoe kunnen de twee
        waarden zo dicht bij elkaar liggen?
  \item Op welke hoogte boven het Aardoppervlak is de veldsterkte
        gedaald tot de helft van haar oppervlaktewaarde?
  \item Op welke afstand van het centrum van de Aarde daalt de trek
        van de Aarde per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} tot de
        oppervlaktewaarde van de Maan, \qty{1.62}{N/kg}? Druk uit in
        Aardstralen.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De Zon wegen.}
De Aarde reist een bijna cirkelvormige
\omterm{def:g10:universal-gravitation:satellite}{baan} van straal $d = \qty{1.496e11}{m}$ in
één jaar, $T = \qty{3.156e7}{s}$. Neem als gegeven (het is bewezen in
\cref{ch:g12:satellites-kepler}) dat een lichaam op een cirkelpad van
straal $d$ met snelheid $v$ een trek van $v^2/d$ newton per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} naar het centrum nodig heeft.
\begin{enumerate}[resume]
  \item Bereken de omloopsnelheid $v$ van de Aarde.
  \item Leid de trek per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} af die de Zon op
        de Aarde uitoefent.
  \item Die trek is ook $G M_{\text{Sun}} / d^2$. Leid de massa van
        de Zon af.
  \item Hoeveel Aardes is dat? Vergelijk $M_{\text{Sun}}$ met
        $M_{\text{Earth}}$.
  \item Finale. De bollen van Cavendish, de Aarde, de Zon: één
        constante $G$ diende voor alle drie. Zeg in één zin wat het
        woord \emph{universeel} in «universele
        \omterm{def:g10:universal-gravitation:force}{zwaartekracht}» ons in dit probleem
        opleverde.
\end{enumerate}
\end{problem}
