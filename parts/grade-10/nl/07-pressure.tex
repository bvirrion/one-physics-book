\chapter{Druk: van sport tot duiken}\label{ch:g10:pressure}

Een skiër glijdt over sneeuw waarin een wandelaar wegzakt; een duw van
\qty{5}{N} drijft een naald door leer dat een vuist niet deukt.
Dezelfde \omterm{def:g10:inertia:force}{krachten}, wild verschillende effecten: de ontbrekende
grootheid is de \emph{\omterm{def:g10:pressure:pressure}{druk}},
\omterm{def:g10:inertia:force}{kracht} gedeeld door het ontvangende oppervlak. Dit hoofdstuk
definieert haar, volgt haar naar beneden in een bad, omhoog in een
barometer, in een samengedrukte spuit --- en eindigt veertig
\omterm{def:g10:orders-of-magnitude:unit}{meters} onder de zee, waar ze vitaal wordt.

\section{Druk: kracht gespreid over een oppervlak}

\begin{definition}[Druk]\label{def:g10:pressure:pressure}
Wanneer een \omterm{def:g10:inertia:force}{kracht} van grootte $F$ (in \unit{N}) loodrecht drukt op
een oppervlak van oppervlakte $S$ (in \unit{m^2}), is de
\emph{druk}\index{druk} uitgeoefend op dat oppervlak
\[
P = \frac{F}{S}.
\]
Haar \omterm{def:g10:orders-of-magnitude:unit}{eenheid}, de \unit{N/m^2}, is de \emph{pascal}\index{pascal}
(\unit{Pa}) --- een piepkleine \omterm{def:g10:orders-of-magnitude:unit}{eenheid}, het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van een appel gespreid over een vierkante
\omterm{def:g10:orders-of-magnitude:unit}{meter}, dus \unit{kPa} en \unit{MPa} zijn de werk\-
\omterm{def:g10:orders-of-magnitude:unit}{eenheden}.
\end{definition}

\begin{example}[Hak versus sneeuwschoen]\label{ex:g10:pressure:snowshoe}
Een persoon van \qty{60}{kg} weegt $F = 60 \times 9.81 \approx \qty{590}{N}$.
Op één naaldhak van \qty{1.0}{cm^2} = \qty{1.0e-4}{m^2}:
$P = 590/\num{1.0e-4} \approx \qty{5.9e6}{Pa} = \qty{5.9}{MPa}$. Op twee
sneeuwschoenen van elk \qty{0.20}{m^2}:
$P = 590/0.40 \approx \qty{1.5}{kPa}$. Hetzelfde
\omterm{def:g10:universal-gravitation:weight}{gewicht} drukt $4000$ keer harder onder de hak: hakken
putten houten vloeren, sneeuwschoenen drijven op poedersneeuw.
\end{example}

\begin{remark}[Concentreren of spreiden]\label{rem:g10:pressure:tools}
Een naainaald geduwd met \qty{5}{N} op een punt van \qty{0.01}{mm^2} =
\qty{1e-8}{m^2} oefent $\qty{5e8}{Pa} = \qty{500}{MPa}$ uit --- genoeg
om leervezels te scheiden. Prikgereedschap concentreert
\omterm{def:g10:inertia:force}{kracht}; dragende ontwerpen (ski's, rupsbanden, funderingen)
spreiden haar. Sport is toegepast
\omterm{def:g10:pressure:pressure}{druk}management: schaatsen bijten in ijs, ski's glijden
op sneeuw.
\end{remark}

\section{Druk in een vloeistof in rust}

Waarom drukt water harder lager? Isoleer een imaginaire verticale
waterkolom van doorsnede $S$, van het oppervlak tot diepte $h$:
volume $Sh$, massa $\rho S h$ ($\rho$ de dichtheid van de vloeistof, in
\unit{kg/m^3}), \omterm{def:g10:universal-gravitation:weight}{gewicht} $\rho S h g$. De bodem van de
kolom steunt dat \omterm{def:g10:universal-gravitation:weight}{gewicht} \emph{plus} de duw $P_0 S$ van
de atmosfeer op de top. Delen van de totale
\omterm{def:g10:inertia:force}{kracht} door $S$: de \omterm{def:g10:pressure:pressure}{druk} op diepte $h$
zou $P_0 + \rho g h$ moeten zijn --- en dat is ze, overal in de
vloeistof.

\begin{theorem}[Druk op diepte]\label{thm:g10:pressure:depth}
In een vloeistof van dichtheid $\rho$ in rust, open naar de lucht, is
de \omterm{def:g10:pressure:pressure}{druk} op diepte $h$ onder het oppervlak
\[
P = P_0 + \rho g h,
\]
waarbij $P_0$ de \omterm{def:g10:pressure:atmospheric}{atmosferische druk} aan het oppervlak
is. De \omterm{def:g10:pressure:pressure}{druk} hangt alleen van de diepte af, niet van de
vorm van de container, en in een gegeven punt drukt de vloeistof even
hard in alle richtingen.
\end{theorem}
\admitted

\begin{remark}
Het kolomargument maakt de wet alleen aannemelijk; de eerlijke
afleiding (elke vorm, alle richtingen) staat in het deel van jaar 1.
De eigen term van het water $\rho g h$ is de
\emph{overdruk}\index{overdruk} \omterm{def:g10:pressure:pressure}{druk} --- wat de
\omterm{def:g10:pressure:pressure}{druk}meter van een duiker toont.
\end{remark}

\begin{example}[Tien meter water]\label{ex:g10:pressure:tenmetres}
In zoet water ($\rho = \qty{1000}{kg/m^3}$), bij $h = \qty{10}{m}$:
$\rho g h = 1000 \times 9.81 \times 10 \approx \qty{98}{kPa}$, dus
$P = 101 + 98 \approx \qty{199}{kPa}$. Tien \omterm{def:g10:orders-of-magnitude:unit}{meters}
water voegen bijna exact één atmosfeer toe: de vuistregel van de
duiker, \emph{één extra atmosfeer per tien
\omterm{def:g10:orders-of-magnitude:unit}{meters}}.
\end{example}

\section{De oceaan van lucht}

\begin{definition}[Atmosferische druk]\label{def:g10:pressure:atmospheric}
We leven op de bodem van een oceaan van lucht, waarvan het
\omterm{def:g10:universal-gravitation:weight}{gewicht} de \emph{atmosferische druk}\index{atmosferische druk}
creëert, op zeeniveau
\[
P_0 \approx \qty{101}{kPa} = \qty{1.01e5}{Pa}.
\]
Op elke vierkante centimeter is dit $F = \num{1.01e5} \times \num{1e-4}
\approx \qty{10}{N}$: het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een
\omterm{def:g10:orders-of-magnitude:unit}{één-kilogram}-massa op elke \unit{cm^2} van je huid, je
bureau, alles.
\end{definition}

\begin{example}[De onzichtbare last]\label{ex:g10:pressure:sheet}
Een A4-blad papier (\qty{21.0}{cm} $\times$ \qty{29.7}{cm}, dus
$S \approx \qty{0.0624}{m^2}$) ontvangt van de lucht erboven
$F = \num{1.01e5} \times 0.0624 \approx \qty{6.3e3}{N}$ --- het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van een kleine auto. Niets scheurt: de lucht
eronder duwt even hard \emph{omhoog}. We merken de atmosfeer alleen
wanneer één kant haar verliest --- een zuignap, een rietje, een
vliegtuigcabine.
\end{example}

\begin{remark}[De barometer van Torricelli]\label{rem:g10:pressure:torricelli}
In 1643 keerde Torricelli een gesloten kwikbuis om boven een
kwikbad. De kolom daalde tot de hoogte waarbij $\rho g h = P_0$:
$h = \num{1.01e5}/(13600 \times 9.81) \approx \qty{0.76}{m}$, en liet
erboven het eerste vacuüm ooit gemaakt. De hoogte volgt het weer en
krimpt met de hoogte: een barometer weegt de lucht boven jou (water
zou \qty{10.3}{m} nodig hebben: \cref{exo:g10:pressure:9}).
\end{remark}

\section{Gassen: bombardement en de wet van Boyle}

Een gas drukt ook op zijn container: het is een zwerm molecules in
onophoudelijke willekeurige beweging, elke botsing geeft de wand een
piepkleine buitenwaartse duw, en miljarden miljarden inslagen per
seconde vervagen tot de gestage \omterm{def:g10:inertia:force}{kracht} die we meten als
gas-\omterm{def:g10:pressure:pressure}{druk}. Pers het gas tot de helft van het volume: de
molecules slaan twee keer zo vaak, dus moet de
\omterm{def:g10:pressure:pressure}{druk} verdubbelen. Dat doet ze.

\begin{proposition}[Wet van Boyle]\label{prop:g10:pressure:boyle}
\index{wet van Boyle}
Voor een vaste hoeveelheid gas bij constante temperatuur zijn
\omterm{def:g10:pressure:pressure}{druk} en volume omgekeerd evenredig:
\[
P \, V = \text{constante},
\qquad\text{d.w.z.}\qquad
P_1 V_1 = P_2 V_2 .
\]
\end{proposition}
\admitted

\begin{remark}
Een experimenteel feit op dit niveau --- het deel van jaar 1 leidt
haar af uit het botsingsbeeld. Het experiment (een gesloten spuit met
lucht, een \omterm{def:g10:pressure:pressure}{druksensor}, \qty{20}{\degreeCelsius}):
\begin{center}
\begin{tabular}{lccccc}
$V$ (\unit{mL}) & 60 & 50 & 40 & 30 & 20 \\
$P$ (\unit{kPa}) & 100 & 120 & 150 & 200 & 300 \\
$P \times V$ (\unit{kPa.mL}) & 6000 & 6000 & 6000 & 6000 & 6000
\end{tabular}
\end{center}
\end{remark}

\begin{example}[Een samengedrukte spuit]\label{ex:g10:pressure:boyle}
Lucht beslaat \qty{4.0}{L} bij \qty{100}{kPa}; langzaam samengedrukt
(zodat de temperatuur constant blijft) tot \qty{250}{kPa}, beslaat ze
$V_2 = P_1 V_1 / P_2 = 400/250 = \qty{1.6}{L}$.
\end{example}

\section{Duiken: druk aan het werk}

\begin{method}[Absolute druk op diepte]\label{met:g10:pressure:dive}
Voor een duiker op diepte $h$ in de zee ($\rho \approx \qty{1025}{kg/m^3}$,
dus $\rho g h \approx \qty{101}{kPa}$ per \qty{10}{m}), bereken
$P = P_0 + \rho g h$ --- en in de \omterm{prop:g10:pressure:boyle}{wet van Boyle} altijd
deze \emph{absolute}\index{absolute druk} \omterm{def:g10:pressure:pressure}{druk} gebruiken,
nooit de \omterm{thm:g10:pressure:depth}{overdruk}-term alleen. Een duiker zit onder
$2$ atmosferen bij \qty{10}{m}, $3$ bij \qty{20}{m}, $5$ bij
\qty{40}{m}.
\end{method}

\begin{remark}[Houd nooit de adem in]\label{rem:g10:pressure:breath}
Een regulator levert lucht bij omgevings-\omterm{def:g10:pressure:pressure}{druk}: longen
vullen normaal op elke diepte. Maar vul \qty{6.0}{L} bij \qty{20}{m}
($\qty{302}{kPa}$), houd die adem in en stijg: Boyle eist
$6.0 \times 302/101 \approx \qty{18}{L}$ --- drie keer wat een borstkas
houdt, en longen scheuren ver daarvoor, al bij enkele
\omterm{def:g10:orders-of-magnitude:unit}{meters} opstijging. Vandaar de eerste scuba-regel:
\emph{houd nooit de adem in tijdens het opstijgen} --- blijf ademen en
de expanderende lucht stroomt er eenvoudig uit. Een vrije duiker is
veilig: haar lucht werd aan het oppervlak genomen, dus op de weg omhoog
expandeert ze alleen tot haar oorspronkelijke volume.
\end{remark}

\begin{remark}[Decompressie]\label{rem:g10:pressure:deco}
Ademen van \omterm{def:g10:pressure:pressure}{hogedruk}-lucht lost ook extra stikstof op in
het bloed, zoals gas in een gesloten frisdrankfles; te snel naar de
oppervlakte en het fizzelt tot bellen in het lichaam ---
decompressieziekte. Vandaar de trage opstijging (ongeveer \qty{10}{m}
per minuut) en ondiepe stops, die het gas rustig via de longen laten
vertrekken. De theorie wacht op het deel van jaar 1; de gedragsregel
behoort tot elke duiker.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:pressure:1}
Een krat van \qty{600}{N} rust op een zijde van
$\qty{0.50}{m} \times \qty{0.30}{m}$, daarna op een zijde van
$\qty{0.30}{m} \times \qty{0.20}{m}$. Bereken beide
\omterm{def:g10:pressure:pressure}{drukken} op de grond. Welke oriëntatie is beter op
zacht zand?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:2}
Druk \qty{0.25}{MPa} uit in \unit{kPa} en in \unit{Pa}. Wat drukt
harder: \qty{3.0}{N} op \qty{2.0}{cm^2}, of \qty{60}{N} op
\qty{500}{cm^2}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:3}
Bereken de \omterm{thm:g10:pressure:depth}{overdruk} $\rho g h$ en de
\omterm{met:g10:pressure:dive}{absolute druk} op de bodem van een \qty{3.0}{m} diep

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:4}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:pressure:5}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:6}

\begin{exercise}[$\star\star$]\label{exo:g10:pressure:7}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:pressure:14}
De longen van een vrije duiker houden \qty{6.0}{L} aan het oppervlak en

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:pressure:15}
In een hydraulische lift zendt een vloeistof
\omterm{def:g10:pressure:pressure}{druk} onveranderd over van een kleine zuiger
($s = \qty{2.0}{cm^2}$) naar een grote ($S = \qty{400}{cm^2}$) die een
auto van \qty{1200}{kg} draagt.
\begin{enumerate}
  \item Welke \omterm{def:g10:inertia:force}{kracht} op de kleine zuiger houdt de auto
        omhoog? Welke \omterm{def:g10:pressure:pressure}{druk} draagt de vloeistof?
  \item Om de auto \qty{2.0}{cm} te heffen, hoe ver moet de kleine
        zuiger reizen? Vergelijk de arbeid op elke zuiger.
\end{enumerate}
\end{exercise}

\section{Probleem: Eén adem omlaag, één adem omhoog}

\begin{problem}[{Weekendprobleem --- één adem omlaag, één adem omhoog:
de fysica van een vrije duik en een scuba-opstijging, van masker\-
samenpersing tot de regel dat de laatste tien meters het gevaarlijkst
zijn}]
\label{pb:g10:pressure:1}
Lena duikt vrij: één oppervlakte-adem, omlaag naar \qty{30}{m} en
terug. Marco scuba-duikt naast haar, ademend uit een tank. Dezelfde
zee ($\rho = \qty{1025}{kg/m^3}$, $g = \qty{9.81}{N/kg}$,
$P_0 = \qty{101}{kPa}$), dezelfde longen van \qty{6.0}{L},
tegengestelde gevaren --- allemaal geregeerd door
$P = P_0 + \rho g h$ en $PV$ = constant (temperatuur constant doorheen).

\medskip
\noindent\textbf{Deel I --- Het \omterm{def:g10:universal-gravitation:weight}{gewicht} van water.}
\begin{enumerate}
  \item Bereken $\rho g$ voor zeewater in \unit{kPa} per
        \omterm{def:g10:orders-of-magnitude:unit}{meter}, en controleer de vuistregel: één
        atmosfeer per \qty{10}{m}.
  \item Bereken de \omterm{met:g10:pressure:dive}{absolute druk} bij \qty{10}{m},
        \qty{20}{m}, \qty{30}{m} en \qty{40}{m}.
  \item Druk elk uit als meervoud van de
        oppervlakte-\omterm{def:g10:pressure:pressure}{druk}.
  \item Lena's masker bedekt \qty{150}{cm^2}. Bij \qty{30}{m}, als de
        lucht erin nog op oppervlakte-\omterm{def:g10:pressure:pressure}{druk} was, welke
        netto \omterm{def:g10:inertia:force}{kracht} zou het water uitoefenen? (Duikers
        ademen via de neus in het masker om deze \emph{maskersqueeze} te
        voorkomen.)
  \item Een trommelvlies heeft oppervlakte ongeveer \qty{0.60}{cm^2}.
        Bereken de netto \omterm{def:g10:inertia:force}{kracht} erop bij \qty{3.0}{m} als
        het middenoor op oppervlakte-\omterm{def:g10:pressure:pressure}{druk} blijft; waarom
        doen oren pijn in een louter bad, en wat moet «equaliseren»
        bereiken?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Eén adem omlaag.}
Lena verlaat het oppervlak met \qty{6.0}{L} lucht in haar longen.
\begin{enumerate}[resume]
  \item Bereken met de \omterm{prop:g10:pressure:boyle}{wet van Boyle} en
        \omterm{met:g10:pressure:dive}{absolute drukken} haar longvolume bij
        \qty{10}{m}, \qty{20}{m} en \qty{30}{m}.
  \item Op welke diepte is haar longvolume gehalveerd?
  \item Is longvolume evenredig met diepte? Beschrijf de kromme van
        $V$ tegen \omterm{met:g10:pressure:dive}{absolute druk} $P$.
  \item Haar restvolume is \qty{1.5}{L}: longen kunnen niet verder
        krimpen. Toon dat deze limiet nabij \qty{30}{m} bereikt wordt.
  \item Records van vrij duiken overschrijden desondanks
        \qty{100}{m}: wat vult het ontbrekende volume? (Wat anders kan
        de borstkas in stromen?)
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Eén adem omhoog.}
Bij \qty{30}{m} vult Marco's regulator zijn longen van \qty{6.0}{L}.
\begin{enumerate}[resume]
  \item Bij welke \omterm{def:g10:pressure:pressure}{druk} levert de regulator die lucht?
        Hoeveel keer dichter is ze dan oppervlaktelucht?
  \item Marco panikeert, houdt de adem in, en stijgt naar \qty{10}{m}:
        welk volume eist zijn gevangen lucht daar?
  \item Welk volume aan het oppervlak --- hoeveel keer zijn
        capaciteit?
  \item Noem de eerste regel van de scuba-duiker, en waarom normaal
        ademen het gevaar wegneemt.
  \item Lena houdt ook de adem in van \qty{30}{m} naar het oppervlak,
        toch zijn haar longen perfect veilig. Verklaar de asymmetrie.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Bellen, en de laatste tien
\omterm{def:g10:orders-of-magnitude:unit}{meters}.}
\begin{enumerate}[resume]
  \item Marco laat een bel van \qty{1.0}{cm^3} los bij \qty{40}{m}.
        Bereken haar volume bij \qty{30}{m}, \qty{20}{m}, \qty{10}{m}
        en net onder het oppervlak.
  \item Bereken voor elke \qty{10}{m}-etappe van de stijging de factor
        waarmee de bel groeit. Waar is de groei het snelst?
  \item Duikers stijgen hoogstens \qty{10}{m} per minuut: hoe lang van
        \qty{40}{m}, en in welke minuut expandeert elk gevangen of
        opgelost gas het meest?
  \item Op diepte lost Marco's bloed extra stikstof op, zoals gas in
        een gesloten frisdrankfles. Wat doet een te snelle opstijging,
        en hoe voorkomen de trage opstijging en een
        \qty{5}{m}-veiligheidsstop dat?
  \item Punchline: geef in één zin elk het oordeel van de vrije
        duiker, dat van de scuba-duiker, en de gekwantificeerde reden
        waarom de laatste tien \omterm{def:g10:orders-of-magnitude:unit}{meters} van elke
        opstijging de meeste eerbied verdienen.
\end{enumerate}
\end{problem}
