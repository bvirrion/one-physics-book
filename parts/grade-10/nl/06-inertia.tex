\chapter{Krachten en het traagheidsprincipe}\label{ch:g10:inertia}

Stop met trappen en de fiets rolt uit tot stilstand; stop met trekken en
de slee staat stil. Ervaring fluistert dat beweging gevoed moet worden
door een inspanning --- en tweeduizend jaar lang was de natuurkunde het
ermee eens. Dit hoofdstuk bouwt de \omterm{def:g10:inertia:force}{kracht}, een pijl gemeten in
\omterm{def:g10:inertia:force}{newton}, en stelt de eerste grote wet van de mechanica: echt
alleen gelaten houdt een bewegend lichaam zijn beweging, recht en
constant, eeuwig.

\section{Een actie modelleren als kracht}

\begin{definition}[Kracht]\label{def:g10:inertia:force}
Een \emph{kracht}\index{kracht} modelleert een mechanische actie (een
duw, een trek, een aantrekking) uitgeoefend op een lichaam. Ze wordt
gekenmerkt door:
\begin{itemize}
  \item haar \emph{aangrijpingspunt}\index{aangrijpingspunt};
  \item haar richting --- de lijn waarlangs ze werkt en de zin langs
        die lijn;
  \item haar grootte, gemeten in \emph{newton}\index{newton}
        (symbool \unit{N}).
\end{itemize}
Een kracht tekent men als een pijl $\vect F$ (een vector, zoals in het
wiskundedeel) vanaf haar aangrijpingspunt, met lengte evenredig met de
grootte op een gestelde schaal.
\end{definition}

\begin{example}[Het gewicht van een schooltas]\label{ex:g10:inertia:schoolbag}
Een schooltas van massa $m = \qty{4.0}{kg}$ wordt omlaag getrokken door
de Aarde met haar \omterm{def:g10:universal-gravitation:weight}{gewicht}
(\cref{ch:g10:universal-gravitation}):
$P = mg = 4.0 \times 9.81 \approx \qty{39}{N}$ --- één
\omterm{def:g10:inertia:force}{newton} is ongeveer het \omterm{def:g10:universal-gravitation:weight}{gewicht}
van een appel. Op de schaal $\qty{1}{cm} \leftrightarrow \qty{10}{N}$:
een verticale pijl van \qty{3.9}{cm}, aangrijpend in het zwaartepunt
van de tas, wijzend omlaag.
\end{example}

\section{Een inventaris van krachten}

\begin{definition}[Contact- en afstands\-krachten]\label{def:g10:inertia:contact}
Een \emph{contactkracht}\index{contactkracht} wordt uitgeoefend op de
punten waar twee lichamen elkaar raken. Een \emph{kracht op
afstand}\index{kracht op afstand} heeft geen contact nodig: het
\omterm{def:g10:universal-gravitation:weight}{gewicht} --- de zwaartekrachttrek van
\cref{ch:g10:universal-gravitation} --- is de enige die in dit
hoofdstuk gebruikt wordt.
\end{definition}

\begin{definition}[De gewone contactkrachten]\label{def:g10:inertia:inventory}
Vier \omterm{def:g10:inertia:contact}{contactkrachten} dekken de meeste alledaagse situaties:
\begin{itemize}
  \item de \emph{normaalkracht}\index{normaalkracht} $\vect R$ van een
        steun, loodrecht op haar oppervlak, er vanaf duwend;
  \item de \emph{spankracht}\index{spankracht} $\vect T$ van een draad,
        touw of ketting, gericht erlangs, er naartoe trekkend;
  \item \emph{wrijving}\index{wrijving} $\vect f$ van een oppervlak, de
        lucht of water, die de glijding of de beweging tegengaat;
  \item de \emph{stuwkracht}\index{stuwkracht} van een motor, jet of
        hand, duwend langs de inspanning.
\end{itemize}
\end{definition}

\begin{example}[Inventaris op een getrokken slee]\label{ex:g10:inertia:sled}
Een slee getrokken over de sneeuw: \omterm{def:g10:universal-gravitation:weight}{gewicht} $\vect P$
(afstand), \omterm{def:g10:inertia:inventory}{normaalkracht} $\vect R$ van de sneeuw,
\omterm{def:g10:inertia:inventory}{spankracht} $\vect T$ van het touw,
\omterm{def:g10:inertia:inventory}{wrijving} $\vect f$ die de glijding tegengaat. Niets anders
raakt de slee: de inventaris is compleet ---
\emph{\omterm{def:g10:universal-gravitation:weight}{gewicht} altijd, dan één
\omterm{def:g10:inertia:contact}{contactkracht} per rakend object}.
\end{example}

\section{Het traagheidsprincipe}

\begin{definition}[Rechtlijnig eenparige beweging]\label{def:g10:inertia:uniform}
Een lichaam is in \emph{rechtlijnig eenparige beweging}\index{rechtlijnig
eenparige beweging} in een stelsel wanneer zijn
\omterm{def:g10:relative-motion:trajectory}{baan} een rechte lijn is die met constante
snelheid wordt afgelegd: zijn snelheidsvector $\vect v$ is op elk
ogenblik dezelfde. Rust is het speciale geval $\vect v = \vect 0$.
\end{definition}

\begin{definition}[Compenserende krachten]\label{def:g10:inertia:compensate}
\omterm{def:g10:inertia:force}{Krachten} die op een lichaam werken
\emph{compenseren}\index{compenserende krachten} wanneer hun
gecombineerde effect nihil is: tip-aan-staart keren hun pijlen terug
naar het begin. Twee \omterm{def:g10:inertia:force}{krachten} compenseren precies
wanneer ze dezelfde werklijn delen met gelijke groottes en
tegengestelde zinnen.
\end{definition}

\begin{theorem}[Traagheidsprincipe]\label{thm:g10:inertia:principle}
\index{traagheidsprincipe}\index{traagheid}
In het \omterm{def:g10:relative-motion:frames}{grondstelsel} blijft een lichaam
onderhevig aan geen \omterm{def:g10:inertia:force}{kracht}, of aan
\omterm{def:g10:inertia:force}{krachten} die \omterm{def:g10:inertia:compensate}{compenseren},
in rust of houdt een \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging};
omgekeerd is een lichaam in rust of in
\omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging} onderhevig aan
\omterm{def:g10:inertia:compensate}{compenserende krachten} (of aan geen).
\end{theorem}
\admitted

\begin{remark}
Geen experiment kan een lichaam volledig isoleren, dus is het principe
een postulaat --- gezien door Galileo, gesteld door Newton. Een later
hoofdstuk upgrade het tot een wet die \omterm{def:g10:inertia:force}{krachten}
koppelt aan \emph{veranderingen} van snelheid; het deel van jaar 1
vraagt in welke stelsels het werkelijk geldt.
\end{remark}

\begin{proposition}[Beweging uit krachten lezen, en terug]\label{prop:g10:inertia:converse}
In het \omterm{def:g10:relative-motion:frames}{grondstelsel}:
\begin{enumerate}
  \item als de snelheid van een lichaam verandert --- in grootte
        \emph{of} in richting --- \omterm{def:g10:inertia:compensate}{compenseren} de
        \omterm{def:g10:inertia:force}{krachten} erop \emph{niet};
  \item als een lichaam in rust is of in
        \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging}, moeten zijn
        geïnventariseerde \omterm{def:g10:inertia:force}{krachten} elkaar opheffen ---
        onbekenden bepalen.
\end{enumerate}
\end{proposition}

\begin{proof}
De eerste uitspraak is de contrapositieve van
\cref{thm:g10:inertia:principle}, de tweede haar omgekeerde deel
toegepast op een voltooide inventaris.
\end{proof}

\begin{example}[De hockeypuck]\label{ex:g10:inertia:puck}
Losgelaten door de stick steekt een puck \qty{30}{m} glad ijs over in
\qty{2.5}{s}, rechtuit met de constante $v = 30/2.5 = \qty{12}{m/s}$.
Inventaris: \omterm{def:g10:universal-gravitation:weight}{gewicht} en
\omterm{def:g10:inertia:inventory}{normaalkracht} (\omterm{def:g10:inertia:inventory}{wrijving}
verwaarloosbaar). Ze \omterm{def:g10:inertia:compensate}{compenseren}, en het principe
voorspelt precies de waargenomen beweging: niets duwt de puck vooruit
--- en dat hoeft ook niet.
\end{example}

\begin{remark}[Galileo tegen Aristoteles]
Aristoteles leerde dat elke beweging een beweger nodig heeft, en het
dagelijks leven --- \omterm{def:g10:inertia:inventory}{wrijving} in elk contact --- lijkt het
ermee eens. Galileo idealiseerde de \omterm{def:g10:inertia:inventory}{wrijving} weg: steeds
gladdere oppervlakken hebben steeds minder duw nodig om beweging te
\emph{onderhouden}. Persistentie, niet rust, is de natuurlijke
toestand.
\end{remark}

\begin{remark}[De schokkend passagier]
Wanneer een bus scherp remt, schieten staande passagiers naar voren.
Geen \omterm{def:g10:inertia:force}{kracht} gooit hen: door traagheid houden hun
lichamen hun snelheid terwijl de bus de zijne verliest. De
veiligheidsgordel levert de achterwaartse \omterm{def:g10:inertia:force}{kracht}
die de inventaris mist.
\end{remark}

\section{Gebalanceerde krachten: statica vanaf een tekening}

\begin{definition}[Krachtendiagram]\label{def:g10:inertia:diagram}
Het \emph{krachtendiagram}\index{krachtendiagram} van een lichaam toont
het lichaam alleen, met elke \omterm{def:g10:inertia:force}{kracht} die \emph{erop}
werkt op schaal getekend vanaf haar
\omterm{def:g10:inertia:force}{aangrijpingspunt}; \omterm{def:g10:inertia:force}{krachten}
uitgeoefend \emph{door} het lichaam verschijnen nooit.
\end{definition}

\begin{method}[Een statische situatie analyseren]\label{met:g10:inertia:method}
\begin{enumerate}
  \item Kies het te bestuderen lichaam; werk in het
        \omterm{def:g10:relative-motion:frames}{grondstelsel}.
  \item Inventaris: \omterm{def:g10:universal-gravitation:weight}{gewicht} altijd, één
        \omterm{def:g10:inertia:contact}{contactkracht} per contact.
  \item Teken het \omterm{def:g10:inertia:diagram}{krachtendiagram}.
  \item Het lichaam is in rust, dus
        \omterm{def:g10:inertia:compensate}{compenseren} de
        \omterm{def:g10:inertia:force}{krachten}
        (\cref{thm:g10:inertia:principle}): lees onbekende groottes af
        van de tekening (gelijke lengtes in 1D, componenten in 2D).
\end{enumerate}
\end{method}

\begin{example}[Het boek op de tafel]\label{ex:g10:inertia:book}
Een boek van massa \qty{0.50}{kg} rust op een tafel:
\omterm{def:g10:universal-gravitation:weight}{gewicht} $P = 0.50 \times 9.81 \approx \qty{4.9}{N}$ en
\omterm{def:g10:inertia:inventory}{normaalkracht} $\vect R$. In rust
\omterm{def:g10:inertia:compensate}{compenseert} $\vect R$ $\vect P$: verticaal,
omhoog, $R = \qty{4.9}{N}$. Voeg een tweede boek toe en $R$ past zich
aan --- een steun duwt precies zo hard als nodig.
\end{example}

\begin{example}[De lamp aan twee draden]\label{ex:g10:inertia:lamp}
Een lamp van massa \qty{3.0}{kg} ($P \approx \qty{29.4}{N}$) hangt aan
twee draden, elk onder $45^\circ$ van de verticale. Door symmetrie
zijn de \omterm{def:g10:inertia:inventory}{spankrachten} gelijk ($T_1 = T_2 = T$) en heffen
hun horizontale componenten elkaar op; de verticale componenten, elk
$T \cos 45^\circ$, dragen het \omterm{def:g10:universal-gravitation:weight}{gewicht}:
\[
2\,T \cos 45^\circ = P
\qquad\text{dus}\qquad
T = \frac{29.4}{2 \times 0.707} \approx \qty{21}{N}
\]
--- meer dan de helft van het \omterm{def:g10:universal-gravitation:weight}{gewicht} per draad:
onder een hoek trekken kost \omterm{def:g10:inertia:inventory}{spankracht}.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:inertia:1}
Inventariseer en classificeer (contact of afstand) de
\omterm{def:g10:inertia:force}{krachten} op: (a) een boek in rust op een tafel;
(b) een lamp hangend aan haar draad; (c) een puck glijdend op glad
ijs; (d) een vallende appel (luchtweerstand verwaarloosbaar).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:2}
Bereken het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een boek van \qty{250}{g}
en van een persoon van \qty{60}{kg}. Op de schaal
$\qty{1}{cm} \leftrightarrow \qty{1}{N}$, hoe lang is de pijl van het
boek, en waarom is de schaal hopeloos voor de persoon?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:3}
Op een diagram op de schaal $\qty{1}{cm} \leftrightarrow \qty{2}{N}$
is een \omterm{def:g10:inertia:force}{kracht} een horizontale pijl van \qty{3.5}{cm}
lang die naar links wijst, aangrijpend in een punt $A$. Geef haar drie
kenmerken.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:4}
Een woordenboek van massa \qty{1.2}{kg} ligt op een plank. Teken zijn
\omterm{def:g10:inertia:diagram}{krachtendiagram} en geef de grootte en richting van
elke \omterm{def:g10:inertia:force}{kracht}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:inertia:5}
Waar of onwaar --- corrigeer de onware: (a) een lichaam stopt altijd
uiteindelijk tenzij een \omterm{def:g10:inertia:force}{kracht} blijft duwen; (b)
\omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging} vereist geen
niet-gecompenseerde \omterm{def:g10:inertia:force}{kracht}; (c) een lichaam in rust is
onderhevig aan geen \omterm{def:g10:inertia:force}{kracht}; (d) als de snelheid van een
lichaam verandert, \omterm{def:g10:inertia:compensate}{compenseren} de
\omterm{def:g10:inertia:force}{krachten} erop niet.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:6}
Na loslaten steekt een puck \qty{24}{m} ijs over in \qty{3.0}{s},
rechtuit met constante snelheid. Bereken haar snelheid;
\omterm{def:g10:inertia:compensate}{compenseren} de \omterm{def:g10:inertia:force}{krachten}
erop? Op beton stopt dezelfde puck binnen \omterm{def:g10:orders-of-magnitude:unit}{meters}:
wat veranderde in de inventaris, en wat concludeert
\cref{prop:g10:inertia:converse}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:7}
Een passagier van \qty{70}{kg} staat in een lift die met de constante
\qty{1.5}{m/s} stijgt. Noteer de \omterm{def:g10:inertia:force}{krachten} op de
passagier; \omterm{def:g10:inertia:compensate}{compenseren} ze? Geef de grootte van de
\omterm{def:g10:inertia:force}{kracht} van de vloer. Zou er iets veranderen met de lift
in rust?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:8}
Een lamp van massa \qty{1.8}{kg} hangt in rust aan één verticale
draad. Bereken de \omterm{def:g10:inertia:inventory}{spankracht} van de draad. De draad
wordt vervangen door twee \emph{verticale} draden die de last gelijk
delen: wat is de \omterm{def:g10:inertia:inventory}{spankracht} van elk?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:9}
Een bus remt scherp en een staande passagier schiet naar voren; in een
bocht met constante snelheid leunt dezelfde passagier naar buiten. Leg
beide uit zonder een «voorwaartse» of «buitenwaartse»
\omterm{def:g10:inertia:force}{kracht} uit te vinden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:10}
Een parachutist van totale massa \qty{85}{kg} daalt
\omterm{def:g10:universal-gravitation:weight}{verticaal} met de constante \qty{5.0}{m/s}. Teken
het \omterm{def:g10:inertia:diagram}{krachtendiagram} en bereken de luchtweerstand
op de kap.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:inertia:11}
Een waterskiër wordt rechtuit getrokken met constante snelheid; het
horizontale touw trekt met \qty{300}{N}. Wat is de horizontale
waterweerstand op de ski's (rechtvaardig)? Welke twee verticale
\omterm{def:g10:inertia:force}{krachten} \omterm{def:g10:inertia:compensate}{compenseren} elkaar?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:12}
Een lamp van massa \qty{2.4}{kg} hangt aan twee draden onder
$40^\circ$ van de verticale ($\cos 40^\circ \approx 0.766$). Bereken
elke \omterm{def:g10:inertia:inventory}{spankracht}. Waarom groeit de
\omterm{def:g10:inertia:inventory}{spankracht} naarmate de draden de horizontale naderen, en
waarom zouden twee exact horizontale draden de lamp nooit kunnen
houden?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:13}
Een curlingsteen wordt geduwd, losgelaten, glijdt terwijl ze zeer
geleidelijk vertraagt, en rust dan. Teken voor elke fase (duw, glijden,
rust) het \omterm{def:g10:inertia:diagram}{krachtendiagram} en zeg of de
\omterm{def:g10:inertia:force}{krachten} \omterm{def:g10:inertia:compensate}{compenseren},
onder verwijzing naar \cref{thm:g10:inertia:principle} of
\cref{prop:g10:inertia:converse}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:14}
De sonde \emph{Voyager 1} glijdt door de interstellaire ruimte met
\qty{17}{km/s}, motoren uit. Waarom heeft ze geen brandstof nodig om
haar snelheid te houden, en wat zou Aristoteles voorspellen? Hoe ver
reist ze in een jaar, in \unit{km} daarna in
\omterm{def:g10:orders-of-magnitude:lightyear}{astronomische eenheden}
($\qty{1}{au} \approx \qty{1.496e8}{km}$)? Waarom moet ze een thruster
afvuren zelfs om van richting te veranderen \emph{zonder} de snelheid te
veranderen?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:inertia:15}
Drie touwen trekken aan een ring in rust. Op een rooster waar één
vierkant \qty{1}{N} is, lezen twee \omterm{def:g10:inertia:force}{krachten}
$\vect F_1\,(3, 2)$ en $\vect F_2\,(-1, 2)$. Vind de componenten,
grootte en richting van de derde \omterm{def:g10:inertia:force}{kracht}.
\end{exercise}

\section{Probleem: Traagheid in drie bedrijven}

\begin{problem}[{Weekendprobleem --- het luchthavenrolpad, de ijsbaan
en de parachute: constante snelheid als handtekening van
gecompenseerde krachten, en waarom een parachute niet de weerstand
maar de snelheid verlaagt}]
\label{pb:g10:inertia:1}
Drie scènes --- een koffer op een rolpad, een puck op een ijsbaan, een
parachutist in de avondlucht --- en één wet die in alle drie spreekt.

\medskip
\noindent\textbf{Deel I --- Het luchthavenrolpad.}
Een koffer van \qty{23}{kg} staat op een rolpad dat beweegt met de
constante snelheid \qty{0.75}{m/s}.
\begin{enumerate}
  \item Beweging van de koffer in het grondstelsel? Inventariseer de
        \omterm{def:g10:inertia:force}{krachten} erop.
  \item \omterm{def:g10:inertia:compensate}{Compenseren} deze
        \omterm{def:g10:inertia:force}{krachten}? Rechtvaardig; bereken beide
        groottes.
  \item Vergelijk haar \omterm{def:g10:inertia:diagram}{krachtendiagram} met dat
        van dezelfde koffer op de vloer. Welke diepe uitspraak
        illustreert de vergelijking?
  \item Het rolpad stopt met een schok. Beschrijf wat de koffer doet,
        zonder een voorwaartse \omterm{def:g10:inertia:force}{kracht} uit te vinden.
  \item Haar eigenaar loopt met \qty{1.2}{m/s} ten opzichte van het
        rolpad. Haar grondsnelheid?
        \omterm{def:g10:inertia:compensate}{Compenseren} de
        \omterm{def:g10:inertia:force}{krachten} op \emph{haar}?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De ijsbaan.}
\begin{enumerate}[resume]
  \item Een puck van \qty{160}{g} glijdt rechtuit met de constante
        \qty{8.0}{m/s}. Teken haar \omterm{def:g10:inertia:diagram}{krachtendiagram}
        met groottes.
  \item Aristoteles beweert dat de puck een beweger nodig heeft. Wat
        antwoordt de ijsbaan, en waarom lijken alledaagse oppervlakken
        het met hem eens?
  \item De puck steekt een ruw stuk over en vertraagt. Conclusie over
        de \omterm{def:g10:inertia:force}{krachten} daar? Welke
        \omterm{def:g10:inertia:force}{kracht} is de schuldige?
  \item Een schaatser bocht soepel met constante snelheid.
        \omterm{def:g10:inertia:compensate}{Compenseren} de
        \omterm{def:g10:inertia:force}{krachten} op haar? Let op: wat voor
        grootheid is snelheid?
  \item Match diagram met beweging: (a) gecompenseerde
        \omterm{def:g10:inertia:force}{krachten}; (b) een extra
        \omterm{def:g10:inertia:force}{kracht} tegengesteld aan $\vect v$; (c) een
        extra \omterm{def:g10:inertia:force}{kracht} loodrecht op $\vect v$ --- (i)
        rechtuit vertragen; (ii) bochten met constante snelheid; (iii)
        \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De parachute.}
De totale massa van een springer, uitrusting inbegrepen, is
\qty{80}{kg}.
\begin{enumerate}[resume]
  \item Bij het verlaten van het vliegtuig zijn verticale snelheid en
        weerstand nog nul. Bereken de \omterm{def:g10:inertia:force}{krachten};
        \omterm{def:g10:inertia:compensate}{compenseren} ze? Wat moet daarna
        gebeuren?
  \item Weerstand groeit met de snelheid. Beschrijf de val terwijl
        weerstand $<$ \omterm{def:g10:universal-gravitation:weight}{gewicht}.
  \item Voor de kap opent, bereikt de springer een gestage
        \qty{50}{m/s} («eindsnelheid»). Bereken de weerstand. Wat is
        de beweging, en waarom is ze «traagheid in vermomming»?
  \item De kap opent: de weerstand overtreft nu ver het
        \omterm{def:g10:universal-gravitation:weight}{gewicht}. Wat gebeurt er met de snelheid?
        (De springer beweegt \emph{omlaag} doorheen.)
  \item Een nieuwe gestage \qty{5.0}{m/s} wordt bereikt. Bereken de
        weerstand; vergelijk met vraag 13.
  \item De punchline: wat veranderde de parachute --- de eindweerstand,
        of de snelheid waarbij de weerstand het
        \omterm{def:g10:universal-gravitation:weight}{gewicht} bereikt?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Oordelen.}
\begin{enumerate}[resume]
  \item Vul aan en rechtvaardig: «in rust of in
        \omterm{def:g10:inertia:uniform}{rechtlijnig eenparige beweging}
        $\iff$ \dots». Waarom verdient rust geen aparte wet?
  \item In welk bedrijf zou Aristoteles' «beweging heeft een beweger
        nodig» het zichtbaarst gefaald hebben, en waar zou Galileo naar
        wijzen?
  \item Een examenclaim: «de parachutist valt met constante snelheid,
        dus werkt er geen \omterm{def:g10:inertia:force}{kracht} op hem». Corrigeer in
        één regel.
  \item Finale: noem voor elk bedrijf het paar
        \omterm{def:g10:inertia:force}{krachten} dat
        \omterm{def:g10:inertia:compensate}{compenseert}, en stel de ene wet
        die twintig keer in dit probleem gebruikt wordt.
\end{enumerate}
\end{problem}
