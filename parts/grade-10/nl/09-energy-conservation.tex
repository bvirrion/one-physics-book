\chapter{Energie: vormen en behoud}\label{ch:g10:energy-conservation}

Een rijdende auto, een opgeladen telefoon, een hete oven en een
opgeheven hamer delen één eigenschap: elk kan iets laten gebeuren. De
natuurkunde meet die capaciteit met één getal ---
\omterm{def:g10:energy-conservation:energy}{energie} --- en houdt haar rekeningen onder
één ijzeren regel: het totaal verandert nooit, het verplaatst zich
alleen en wisselt van kostuum. Dit hoofdstuk opent het grootboek: de
vormen die \omterm{def:g10:energy-conservation:energy}{energie} aanneemt, de ketens die ze
reist, en wat een \omterm{def:g10:energy-conservation:kwh}{kilowattuur} koopt.

\section{Energie: de gemeenschappelijke munteenheid}

\begin{definition}[Energie en de joule]\label{def:g10:energy-conservation:energy}
\emph{Energie}\index{energie} meet de capaciteit van een systeem om
veranderingen te produceren: materie in beweging zetten, heffen,
verwarmen, verlichten. Haar \omterm{def:g10:orders-of-magnitude:unit}{eenheid} is de
\emph{joule}\index{joule} (\unit{J}). Opgeslagen, overgedragen,
omgezet --- elke tak van de natuurkunde rekent in deze ene
munteenheid.
\end{definition}

\begin{definition}[Kinetische energie]\label{def:g10:energy-conservation:kinetic}
Een lichaam van massa $m$ (in \unit{kg}) dat beweegt met snelheid $v$
(in \unit{m/s}) draagt de \emph{kinetische energie}\index{kinetische
energie} $E_k = \tfrac12\, m v^2$: de massa verdubbelen verdubbelt
haar, de snelheid verdubbelen verviervoudigt haar.
\end{definition}

\begin{example}[Auto versus voetganger]\label{ex:g10:energy-conservation:kinetic}
Een auto van \qty{1300}{kg} bij \qty{100}{km/h} $= \qty{27.8}{m/s}$
draagt $E_k = \tfrac12 \times 1300 \times 27.8^2 \approx \qty{5.0e5}{J}$;
een voetganger van \qty{70}{kg} bij \qty{1.5}{m/s},
$\tfrac12 \times 70 \times 1.5^2 \approx \qty{79}{J}$ --- zesduizend
keer minder. Snelheid is duur: het kwadraat zorgt daarvoor.
\end{example}

\begin{definition}[Gravitatiepotentiële energie]\label{def:g10:energy-conservation:potential}
Een lichaam van massa $m$ op hoogte $h$ (in \unit{m}) boven een
gekozen referentieniveau slaat de \emph{gravitatiepotentiële
energie}\index{potentiële energie!gravitatie} $E_p = m g h$ op, met
$g = \qty{9.81}{N/kg}$ --- geldig nabij de grond, waar $g$ effectief
constant is. Alleen \emph{verschillen} in hoogte tellen: het niveau
waar $E_p = 0$ kiest men vrij.
\end{definition}

\begin{example}[Watertoren]\label{ex:g10:energy-conservation:potential}
Elke kubieke \omterm{def:g10:orders-of-magnitude:unit}{meter} (\qty{1000}{kg}) water opgeslagen
\qty{30}{m} in een watertoren houdt $1000 \times 9.81 \times 30 \approx
\qty{2.9e5}{J}$ --- ongeveer zes tienden van de
\omterm{def:g10:energy-conservation:kinetic}{kinetische energie} van de auto hierboven. Steden
hamsteren \omterm{def:g10:energy-conservation:energy}{energie} boven hun hoofd.
\end{example}

\begin{definition}[De andere vormen]\label{def:g10:energy-conservation:forms}
\omterm{def:g10:energy-conservation:energy}{Energie} verbergt zich ook in minder zichtbare
kostuums: \emph{thermische energie}\index{thermische energie}, de
ongeordende agitatie van de molecules van een lichaam (hoe heter, hoe
meer); \emph{chemische energie}\index{chemische energie}, opgeslagen
in moleculaire bindingen --- voedsel, hout, benzine, batterijen;
\emph{elektrische energie}\index{elektrische energie}, gedragen door
stromen van centrale naar stopcontact; \emph{stralingsenergie}\index{stralingsenergie},
gedragen door licht zelfs door lege ruimte --- hoe de
\omterm{def:g10:energy-conservation:energy}{energie} van de Zon ons bereikt;
\emph{nucleaire energie}\index{nucleaire energie}, opgeslagen in
atoomkernen, vrijgelaten in sterren
(\cref{ch:g12:nuclear-energy}).
\end{definition}

\section{Overdrachten en omzettingsketens}

\begin{definition}[Overdracht, omzetting, keten]\label{def:g10:energy-conservation:transfer}
Een \emph{energieoverdracht}\index{energieoverdracht} verplaatst
\omterm{def:g10:energy-conservation:energy}{energie} tussen systemen ---
\emph{reservoirs}\index{reservoir (van energie)}; een
\emph{energieomzetting}\index{energieomzetting} verandert haar vorm.
Een apparaat dat een van beide doet (turbine, motor, lamp, spier) is
een \emph{omzetter}\index{omzetter}. Een
\emph{omzettingsketen}\index{omzettingsketen} beeldt de reis: dozen
voor reservoirs en vormen, gelabelde pijlen voor overdrachten.
\end{definition}

\begin{method}[Een omzettingsketen tekenen]\label{met:g10:energy-conservation:chain}
\begin{enumerate}
  \item Identificeer het beginreservoir en -vorm (wat wordt
        verbruikt?).
  \item Volg de \omterm{def:g10:energy-conservation:energy}{energie}
        \omterm{def:g10:energy-conservation:transfer}{omzetter} voor
        \omterm{def:g10:energy-conservation:transfer}{omzetter}: één pijl per overdracht, de
        vorm genoemd in elke doos.
  \item Voeg een warmtepijl toe bij elke echte
        \omterm{def:g10:energy-conservation:transfer}{omzetter} --- wat
        \omterm{def:g10:energy-conservation:energy}{energie} lekt altijd als warmte
        (\omterm{def:g10:inertia:inventory}{wrijving}, draden, uitlaat).
\end{enumerate}
\end{method}

potentieel aan de top van de zwaai, alles kinetisch door het
laagste punt (met $E_p = 0$ daar):

\section{Behoud van energie}

\begin{theorem}[Behoud van energie]\label{thm:g10:energy-conservation:conservation}
\index{behoud van energie}
De totale \omterm{def:g10:energy-conservation:energy}{energie} van een geïsoleerd systeem
--- één dat niets met de buitenwereld uitwisselt --- is constant.
\omterm{def:g10:energy-conservation:energy}{Energie} wordt nooit geschapen en nooit vernietigd:
ze verandert alleen van vorm en plaats, en wat het ene reservoir
verliest, winnen de andere, \omterm{def:g10:energy-conservation:energy}{joule} voor
\omterm{def:g10:energy-conservation:energy}{joule}.
\end{theorem}
\admitted

\begin{remark}\label{rem:g10:energy-conservation:admitted}
Geen experiment heeft dit principe ooit op falen betrapt; hier nemen
we het als gegeven en gebruiken het numeriek.
\cref{ch:g12:work-and-energy} leidt haar mechanische deel
($E_k + E_p$) af uit de wetten van Newton; de volledige boekhouding
wacht op de universiteitsdelen.
\end{remark}

\begin{example}[Vrije val, geprijsd in joules]\label{ex:g10:energy-conservation:fall}
Een bal van \qty{2.0}{kg} valt van \qty{10}{m} (luchtweerstand
verwaarloosbaar): ze start met
$E_p = 2.0 \times 9.81 \times 10 \approx \qty{196}{J}$, landt met
$E_k = \qty{196}{J}$, dus $v = \sqrt{2 \times 196/2.0} =
\qty{14}{m/s} \approx \qty{50}{km/h}$. Geen
\omterm{def:g10:inertia:force}{krachten}, geen stopwatch: behoud alleen prijst de
inslag.
\end{example}

\section{Rendement}

\begin{definition}[Rendement]\label{def:g10:energy-conservation:efficiency}
Een \omterm{def:g10:energy-conservation:transfer}{omzetter} verbruikt $E_{\text{consumed}}$ en
levert de gewenste vorm $E_{\text{useful}}$; haar
\emph{rendement}\index{rendement} is
\[
\eta = \frac{E_{\text{useful}}}{E_{\text{consumed}}} < 1 .
\]
Behoud verbiedt $\eta > 1$, en echte
\omterm{def:g10:energy-conservation:transfer}{omzetters} bereiken $1$ nooit: de
ontbrekende \omterm{def:g10:energy-conservation:energy}{energie} is niet vernietigd --- ze
vertrekt als warmte.
\end{definition}

\begin{example}[Waar de benzine heen gaat]\label{ex:g10:energy-conservation:engine}
Eén \omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine slaat \qty{45}{MJ}
\omterm{def:g10:energy-conservation:forms}{chemische energie} op; een automotor zet ze om
met $\eta \approx 0.30$: \qty{13.5}{MJ} beweging, \qty{31.5}{MJ} die
de uitlaat, de radiateur en de straat verwarmt. Twee derden van elke
tank warmt de atmosfeer.
\end{example}

\begin{remark}[Rendementen vermenigvuldigen]\label{rem:g10:energy-conservation:chain-eta}
Gekoppelde \omterm{def:g10:energy-conservation:transfer}{omzetters} vermenigvuldigen hun
rendementen: een \omterm{def:g10:energy-conservation:power}{energie}centrale ($0.38$) die het
net voedt ($0.94$) dat een lader voedt ($0.85$) levert
$\eta = 0.38 \times 0.94 \times 0.85 \approx 0.30$. Lange ketens
zijn lekkende pijpen: elke pijl belast de stroom.
\end{remark}

\section{Vermogen, de watt en het kilowattuur}

\begin{definition}[Vermogen]\label{def:g10:energy-conservation:power}
\emph{Vermogen}\index{vermogen} is de snelheid waarmee
\omterm{def:g10:energy-conservation:energy}{energie} stroomt, $P = E/t$, gemeten in
\emph{watt}\index{watt} (\unit{W}), met
$\qty{1}{W} = \qty{1}{J/s}$. \omterm{def:g10:energy-conservation:energy}{Energie} is een
hoeveelheid, vermogen een stroom --- een badkuip versus haar kraan.
\end{definition}

\begin{example}[Waterkoker versus mens]\label{ex:g10:energy-conservation:power}
Een waterkoker van \qty{2200}{W} die \qty{150}{s} loopt, gebruikt
$E = 2200 \times 150 = \qty{3.3e5}{J}$. Een mens loopt op ongeveer
\qty{10}{MJ} voedsel per dag, een gemiddelde
$\num{1.0e7}/\num{86400} \approx \qty{116}{W}$: jij idle't als een
heldere oude gloeilamp.
\end{example}

\begin{definition}[Het kilowattuur]\label{def:g10:energy-conservation:kwh}
Het \emph{kilowattuur}\index{kilowattuur} (\unit{kW.h}) is de
\omterm{def:g10:energy-conservation:energy}{energie} van een \qty{1}{kW}-stroom die één uur
loopt: $\qty{1}{kW.h} = 1000 \times 3600\,\unit{J} = \qty{3.6}{MJ}$
--- de \omterm{def:g10:orders-of-magnitude:unit}{eenheid} die elektriciteitsmeters tellen:
\omterm{def:g10:energy-conservation:power}{vermogen} in \unit{kW} maal uren.
\end{definition}

\begin{example}[De rekening lezen]\label{ex:g10:energy-conservation:bill}
Een rekening is een aftrekking en een vermenigvuldiging: nieuwe
meterstand minus oude geeft de
\omterm{def:g10:energy-conservation:kwh}{kilowatturen}; maal de
\omterm{def:g10:orders-of-magnitude:unit}{eenheid}sprijs (ongeveer $0.25$ euro per
\unit{kW.h}), plus een vast abonnement, geeft het totaal. Een
verwarming van \qty{2}{kW} die \qty{2.5}{h} 's nachts loopt, gebruikt
\qty{5}{kW.h} per dag: $38$ euro per maand op zichzelf.
\end{example}

\begin{example}[Ordegrootten]\label{ex:g10:energy-conservation:orders}
De moeite waard uit het hoofd: een appel één
\omterm{def:g10:orders-of-magnitude:unit}{meter} heffen, ongeveer \qty{1}{J}; een
telefoonbatterij, $\qty{10}{W.h} \approx \qty{4e4}{J}$; een kom
graan, \qty{1.5}{MJ}; jouw voedsel voor een dag, \qty{10}{MJ}; een
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine, \qty{45}{MJ} --- bijna exact de
dagelijkse elektriciteit van een huishouden ($\approx \qty{12}{kW.h}$);
een bliksemschicht, \qty{5}{GJ}: honderd huisdagen, afgeleverd in een
flits.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:energy-conservation:1}
Een auto van \qty{1200}{kg} rijdt met \qty{90}{km/h}, d.w.z.\
\qty{25}{m/s}: bereken haar \omterm{def:g10:energy-conservation:kinetic}{kinetische energie}.
Herdoe bij \qty{130}{km/h} (\qty{36.1}{m/s}): wat deed $44\%$ meer
snelheid met $E_k$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:energy-conservation:2}
Een wandelaar van massa \qty{72}{kg} klimt van hoogte \qty{1200}{m}
naar \qty{1650}{m}. Bereken de gewonnen potentiële
\omterm{def:g10:energy-conservation:energy}{energie}, zet om naar
\omterm{def:g10:energy-conservation:kwh}{kilowatturen}, en prijs bij $0.25$ euro per
\unit{kW.h}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:energy-conservation:3}
Zet om: \qty{1}{kW.h} naar \omterm{def:g10:energy-conservation:energy}{joules}; een
telefoonbatterij van \qty{12}{W.h} naar kilojoules; \qty{45}{MJ}
(één \omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine) naar
\omterm{def:g10:energy-conservation:kwh}{kilowatturen}; jouw voedseldag van
\qty{10}{MJ} naar \omterm{def:g10:energy-conservation:kwh}{kilowatturen}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:energy-conservation:4}
Een waterkoker van \qty{2200}{W} loopt \qty{150}{s}. Bereken de
gebruikte \omterm{def:g10:energy-conservation:energy}{energie}, in
\omterm{def:g10:energy-conservation:energy}{joules} en in
\omterm{def:g10:energy-conservation:kwh}{kilowatturen}. Bereken daarna het gemiddelde
\omterm{def:g10:energy-conservation:power}{vermogen} van een mens die op \qty{10}{MJ} per
dag loopt.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:energy-conservation:5}
Schrijf de \omterm{def:g10:energy-conservation:transfer}{omzettingsketen}
(\cref{met:g10:energy-conservation:chain}) van: een fietsdynamo die een
lamp voedt; een gasboiler; een zonnepaneel dat een telefoon laadt.
Markeer de warmtelekken.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:energy-conservation:6}
Een motor verbrandt \qty{2.0}{kg} benzine (\qty{45}{MJ/kg}) en levert
\qty{27}{MJ} beweging. Bereken haar
\omterm{def:g10:energy-conservation:efficiency}{rendement} en de als warmte vrijgekomen
\omterm{def:g10:energy-conservation:energy}{energie}. Waar gaat die warmte heen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:energy-conservation:7}
Een klifduiker van \qty{70}{kg} stapt van een \qty{20}{m}-klif
(luchtweerstand verwaarloosbaar). Bereken met behoud de
\omterm{def:g10:energy-conservation:kinetic}{kinetische energie} bij het water en de
intreesnelheid, in \unit{m/s} en \unit{km/h}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:energy-conservation:8}
Een meter las \qty{41236}{kW.h} op 1 maart, \qty{41562}{kW.h} 30
dagen later: gebruikte \omterm{def:g10:energy-conservation:energy}{energie}, daggemiddelde,
rekening bij $0.26$ euro per \unit{kW.h} plus $12$ euro vast, en
gemiddeld \omterm{def:g10:energy-conservation:power}{vermogen} in

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:energy-conservation:9}
Een oude gloeilamp van \qty{60}{W} zet slechts $5\%$ van haar
verbruik om in licht; een led geeft hetzelfde licht uit \qty{9}{W}.

het \omterm{def:g10:energy-conservation:efficiency}{rendement} van de led; daarna, bij

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:energy-conservation:10}
Een dam laat \qty{150}{m^3} water (\qty{1.5e5}{kg}) per seconde

\begin{exercise}[$\star\star$]\label{exo:g10:energy-conservation:11}

\end{exercise}

Een steenkoolcentrale ($\eta_1 = 0.38$), het net ($\eta_2 = 0.94$) en
een lader ($\eta_3 = 0.85$) laden een telefoonbatterij van
\qty{12}{W.h} op. Bereken het totale
\omterm{def:g10:energy-conservation:efficiency}{rendement}; de
\omterm{def:g10:energy-conservation:forms}{chemische energie} verbruikt in de centrale, in
\unit{W.h} en kilojoules; en de steenkool (\qty{25}{MJ/kg}) per
lading, daarna per jaar van dagelijkse ladingen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:energy-conservation:13}
Een bliksemschicht draagt ongeveer \qty{5}{GJ}. Zet om naar
\omterm{def:g10:energy-conservation:kwh}{kilowatturen} en naar dagen van een
huishouden van \qty{12.6}{kW.h} per dag. Ze wordt afgeleverd in
ongeveer \qty{100}{\micro s}: bereken het
\omterm{def:g10:energy-conservation:power}{vermogen}, vergelijk met het gemiddelde
elektrische \omterm{def:g10:energy-conservation:power}{vermogen} van de mensheid
(\qty{3}{TW}), en leg uit waarom niemand bliksem oogst.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:energy-conservation:14}
Een auto van \qty{1300}{kg} remt van \qty{110}{km/h}
(\qty{30.6}{m/s}) tot stilstand. Bereken de gedissipeerde
\omterm{def:g10:energy-conservation:kinetic}{kinetische energie}; welke vorm neemt ze
aan, en waar? Een elektrische auto herwint $60\%$ in haar batterij:
\omterm{def:g10:energy-conservation:energy}{energie} per stop, en stops nodig om één
\omterm{def:g10:energy-conservation:kwh}{kilowattuur} te vullen?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:energy-conservation:15}

\omterm{def:g10:energy-conservation:energy}{energie} van de klim, de voedsel\-
\omterm{def:g10:energy-conservation:energy}{energie} die ze eist, het equivalent in

\section{Probleem: Een huis één dag voeden}

\begin{problem}[{Weekendprobleem --- van kom graan tot dakpanelen:
een volledige energie-audit van één gewone dag thuis, en wat een
joule kost afhankelijk van wie haar verkoopt}]
\label{pb:g10:energy-conservation:1}
Op een gewone dinsdag zet een gezin zich ertoe om te vinden waar
haar elektriciteit heen gaat, gewapend met dit hoofdstuk, de
gangmeter en de achterkant van de rekening. Het oordeel van vanavond:
hun dag is gelijk aan één
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine, zestien vierkante
\omterm{def:g10:orders-of-magnitude:unit}{meters} zonneschijn --- of vijf fietsers die
rond de klok trappen.

\medskip
\noindent\textbf{Deel I --- De audit.}
De inventaris van de dag: koelkast \qty{100}{W} gedurende
\qty{24}{h}; boiler \qty{2400}{W} gedurende \qty{2.5}{h}; oven
\qty{2000}{W} gedurende \qty{1.0}{h}; wasmachine \qty{500}{W}
gedurende \qty{2.0}{h}; verlichting \qty{60}{W} gedurende
\qty{5.0}{h}; schermen \qty{150}{W} gedurende \qty{6.0}{h}.
\begin{enumerate}
  \item Bereken de \omterm{def:g10:energy-conservation:energy}{energie} van elk apparaat voor
        de dag, in \unit{kW.h}.
  \item Tel de dag op in
        \omterm{def:g10:energy-conservation:kwh}{kilowatturen}, daarna in megajoules;
        vergelijk met één \omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine
        (\qty{45}{MJ}).
  \item Bereken het gemiddelde
        \omterm{def:g10:energy-conservation:power}{vermogen} van het huis over de
        \qty{24}{h}; welk deel van een lopende waterkoker is dat?
  \item Welk enkel apparaat domineert? Herbereken het dagtotaal als
        de boiler slechts \qty{1.25}{h} liep.
  \item Stand-by-lampjes trekken permanent \qty{15}{W}: dagelijkse
        \omterm{def:g10:energy-conservation:energy}{energie}, aandeel van het totaal,
        jaarkosten bij $0.25$ euro per \unit{kW.h}?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De rekening.}
\begin{enumerate}[resume]
  \item In 30 dagen ging de meter van \qty{56214}{kW.h} naar
        \qty{56604}{kW.h}: verbruik, en daggemiddelde getoetst aan de
        audit?
  \item Bereken de rekening: $0.25$ euro per \unit{kW.h} plus een
        vast abonnement van $14$ euro.
  \item Prijs van een megajoule van het net? Van benzine, bij
        $1.80$ euro per liter die \qty{34}{MJ} opslaat?
  \item Een kom graan van \qty{1.5}{MJ} kost ongeveer $0.50$ euro:
        haar megajouleprijs? Rangschik de drie verkopers van
        \omterm{def:g10:energy-conservation:energy}{joules}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Het dak.}
\begin{enumerate}[resume]
  \item Vol zonlicht levert ongeveer \qty{1000}{W} per vierkante
        \omterm{def:g10:orders-of-magnitude:unit}{meter}; panelen zetten het om met
        $\eta = 0.20$: elektrisch
        \omterm{def:g10:energy-conservation:power}{vermogen} van \qty{1}{m^2} in volle
        zon?
  \item De streek gemiddeld het equivalent van \qty{4.0}{h} volle
        zon per dag: dagopbrengst van \qty{1}{m^2}?
  \item Welke paneeloppervlakte dekt de \qty{12.6}{kW.h}-dag van het
        gezin? Het zuidelijke dak biedt \qty{20}{m^2}: past het?
  \item In december daalt het equivalent tot \qty{1.5}{h}: de
        productie van de panelen, haar aandeel van de behoefte, en
        wat de kloof vult?
  \item Het fabriceren van de panelen kostte ongeveer
        \qty{700}{kW.h} per vierkante
        \omterm{def:g10:orders-of-magnitude:unit}{meter}:
        \omterm{def:g10:energy-conservation:energy}{energie}-terugverdientijd, met vraag
        11?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De kom graan.}
\begin{enumerate}[resume]
  \item Een generatorfiets levert \qty{100}{W} elektriciteit:
        hoeveel trapuren maken \qty{12.6}{kW.h} --- hoeveel mensen
        die \qty{24}{h} non-stop trappen?
  \item Spieren lopen op $\eta \approx 0.25$: de voedsel\-
        \omterm{def:g10:energy-conservation:energy}{energie} achter die
        \qty{12.6}{kW.h}, in megajoules en in kommen graan?
  \item Kosten van de getrapte dag (kommen aan $0.50$ euro), tegen
        wat het net rekent voor dezelfde dag?
  \item Een bliksemschicht van \qty{5}{GJ} is gelijk aan hoeveel
        huisdagen? Gegeven haar duur van \qty{100}{\micro s},
        waarom kan ze het huis toch niet
        \omterm{def:g10:energy-conservation:power}{voeden}?
  \item Joules gaan nooit verloren, toch verkopen net, benzine en
        voedsel-\omterm{def:g10:energy-conservation:energy}{joules} tegen zeer
        verschillende prijzen: waarvoor betaal je eigenlijk?
  \item Finale: stel het oordeel van de dag in één zin --- één
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram} benzine, zestien vierkante
        \omterm{def:g10:orders-of-magnitude:unit}{meters} zonnig dak of vijf round-the-clock
        fietsers --- en noem de twee grootheden (één voorraad, één
        stroom) die nooit meer verward mogen worden.
\end{enumerate}
\end{problem}
