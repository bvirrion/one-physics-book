\chapter{Signalen en golven}\label{ch:g10:signals-and-waves}

Elk bericht dat je ooit hebt ontvangen, arriveerde als een
\omterm{def:g10:signals-and-waves:signal}{signaal}: een stem op je trommelvlies, deze pagina die je
als radioflikkering bereikte. Achter de verscheidenheid zit één idee ---
een \omterm{def:g10:orders-of-magnitude:unit}{fysische grootheid} die in de tijd varieert --- en één
gereedschapskist: \omterm{def:g10:signals-and-waves:period}{periode},
\omterm{def:g10:signals-and-waves:frequency}{frequentie},
\omterm{def:g10:signals-and-waves:amplitude}{amplitude}, en de kunst van het timen van een
vertraging. Ermee meet een boot de zeebodem en telt een arts
hartslagen.

\section{Signalen dragen informatie}

\begin{definition}[Signaal]\label{def:g10:signals-and-waves:signal}
Een \emph{signaal}\index{signaal} is een
\omterm{def:g10:orders-of-magnitude:unit}{fysische grootheid} die in de tijd varieert om
informatie te dragen: de lucht-\omterm{def:g10:pressure:pressure}{druk} op je
trommelvlies, de spanning aan de klemmen van een microfoon, de
helderheid van het licht in een
\omterm{def:g10:refraction:fiber}{optische vezel}. Om een signaal te bestuderen,
registreer het als functie van de tijd en lees de informatie van de
kromme.
\end{definition}

In een telefoongesprek verandert dezelfde informatie herhaaldelijk van
drager: \omterm{def:g10:pressure:pressure}{druk} in lucht, dan spanning in de microfoon,
dan radio bij de antenne --- en achteruit aan de andere kant. Eén
woordenschat behandelt ze allemaal; dit hoofdstuk bouwt haar.

\section{Periodieke signalen}

\begin{definition}[Periodiek signaal, periode]\label{def:g10:signals-and-waves:period}
Een \omterm{def:g10:signals-and-waves:signal}{signaal} is \emph{periodiek}\index{periodiek
signaal} wanneer het zich identiek herhaalt met regelmatige
intervallen. De \emph{periode}\index{periode} $T$ is de duur van één
volledige cyclus, in seconden.
\end{definition}

\begin{definition}[Frequentie]\label{def:g10:signals-and-waves:frequency}
De \emph{frequentie}\index{frequentie} van een
\omterm{def:g10:signals-and-waves:period}{periodiek signaal} is het aantal cycli per seconde:
\[
f = \frac{1}{T} .
\]
Haar \omterm{def:g10:orders-of-magnitude:unit}{eenheid}, één cyclus per seconde, is de
\emph{hertz}\index{hertz} (\unit{Hz}), met de gebruikelijke
meervouden \unit{kHz}, \unit{MHz}, \unit{GHz}.
\end{definition}

\begin{definition}[Amplitude]\label{def:g10:signals-and-waves:amplitude}
De \emph{amplitude}\index{amplitude} van een
\omterm{def:g10:signals-and-waves:period}{periodiek signaal} is haar maximale afwijking van de
rustwaarde. Voor een spanning die symmetrisch om nul oscilleert, is
de amplitude de piekwaarde; van top tot dal meet men \emph{twee keer}
de amplitude (de piek-tot-piekwaarde).
\end{definition}

\begin{example}[Netspanning en kamertoon]\label{ex:g10:signals-and-waves:mains}
De netspanning oscilleert bij $f = \qty{50}{Hz}$: haar
\omterm{def:g10:signals-and-waves:period}{periode} is $T = 1/f = \qty{0.020}{s} = \qty{20}{ms}$.
De kamertoon A van een orkest is een
\omterm{def:g10:pressure:pressure}{druk}-\omterm{def:g10:signals-and-waves:signal}{signaal} bij
\qty{440}{Hz}: één cyclus duurt $T = 1/440 = \qty{2.3}{ms}$. Hogere
\omterm{def:g10:signals-and-waves:frequency}{frequentie}, kortere cyclus: $f$ en $T$ zijn elkaars
omgekeerden.
\end{example}

\section{Een oscillogram lezen}

\begin{definition}[Oscilloscoop]\label{def:g10:signals-and-waves:oscilloscope}
Een \emph{oscilloscoop}\index{oscilloscoop} plot een spanning tegen de
tijd; de kromme op haar gerasterde scherm is een
\emph{oscillogram}\index{oscillogram}. Twee instellingen zetten
schermdivisies om naar fysische waarden: de
\emph{tijdbasis}\index{tijdbasis} (seconden per horizontale divisie) en
de verticale \emph{versterking}\index{versterking} (volt per verticale
divisie).
\end{definition}

\begin{method}[Van scherm naar getallen]\label{met:g10:signals-and-waves:oscillogram}
\begin{enumerate}
  \item Tel de horizontale divisies die één volledige cyclus overspant
        (top tot top is het veiligst); vermenigvuldig met de
        \omterm{def:g10:signals-and-waves:oscilloscope}{tijdbasis} om $T$ te krijgen, dan $f = 1/T$.
  \item Tel de verticale divisies van de middellijn tot een top en
        vermenigvuldig met de
        \omterm{def:g10:signals-and-waves:oscilloscope}{versterking} om de
        \omterm{def:g10:signals-and-waves:amplitude}{amplitude} te krijgen (top tot dal geeft de
        piek-tot-piekwaarde: halveer die).
\end{enumerate}
\end{method}

\begin{example}[Een volledige aflezing]\label{ex:g10:signals-and-waves:reading}
Op het \omterm{def:g10:signals-and-waves:oscilloscope}{oscillogram} hieronder is de
\omterm{def:g10:signals-and-waves:oscilloscope}{tijdbasis} \qty{5}{ms} per divisie en de
\omterm{def:g10:signals-and-waves:oscilloscope}{versterking} \qty{2}{V} per divisie. Eén cyclus
overspant $4.0$ divisies: $T = 4.0 \times \qty{5}{ms} = \qty{20}{ms}$
en $f = 1/\qty{0.020}{s} = \qty{50}{Hz}$ --- opnieuw de netspanning. Een
top zit $3.0$ divisies boven de as: de
\omterm{def:g10:signals-and-waves:amplitude}{amplitude} is $3.0 \times \qty{2}{V} = \qty{6.0}{V}$.
\end{example}

\section{Geluid en elektromagnetische signalen}

\begin{definition}[Geluid]\label{def:g10:signals-and-waves:sound}
\emph{Geluid}\index{geluid} is een
\omterm{def:g10:pressure:pressure}{druk}-\omterm{def:g10:signals-and-waves:signal}{signaal}: een trillend
object --- snaar, membraan, stembanden --- duwt ritmisch op de lucht, en
de compressies reizen naar buiten met ongeveer \qty{340}{m/s}. Een
microfoon zet de aankomende \omterm{def:g10:pressure:pressure}{druk} om in een spanning;
het trommelvlies, in zenuwimpulsen. Geluid heeft een medium nodig: een
bel onder een vacuümklok valt stil naarmate de lucht wordt
weggepompt.
\end{definition}

\begin{definition}[Golf]\label{def:g10:signals-and-waves:wave}
Een reizende verstoring --- het compressiepatroon van een
\omterm{def:g10:signals-and-waves:sound}{geluid}, de rimpel op een vijver, een radiopuls --- is
een \emph{golf}\index{golf}. Het medium, wanneer er een is, blijft
staan: elk stukje water wiebelt ter plaatse terwijl de rimpel de vijver
oversteekt. De meetkunde van golven ---
\omterm{def:g10:light-spectra:wavelength}{golflengte}, interferentie, diffractie ---
komt in een later hoofdstuk; dit hoofdstuk heeft alleen hun snelheden
nodig.
\end{definition}

\begin{definition}[Ultrageluid en infrageluid]\label{def:g10:signals-and-waves:ultrasound}
Het menselijk oor hoort \omterm{def:g10:signals-and-waves:sound}{geluid} tussen ongeveer
\qty{20}{Hz} en \qty{20}{kHz} (het plafond daalt met de leeftijd).
\omterm{def:g10:signals-and-waves:sound}{Geluid} boven \qty{20}{kHz} is
\emph{ultrageluid}\index{ultrageluid} --- vleermuizen jagen op
\qty{50}{kHz}, medische scanners beelden het lichaam af op enkele
megahertz. \omterm{def:g10:signals-and-waves:sound}{Geluid} onder \qty{20}{Hz} is
\emph{infrageluid}\index{infrageluid}, gevoeld door olifanten en
seismometers.
\end{definition}

\begin{proposition}[Elektromagnetische signalen]\label{prop:g10:signals-and-waves:em}
Radio, wifi en licht zijn \emph{elektromagnetische
signalen}\index{elektromagnetisch signaal}: één familie, allemaal
reizend door vacuüm --- geen medium nodig --- en allemaal met één
snelheid, de lichtsnelheid van \cref{ch:g10:refraction},
\[
c = \qty{3.00e8}{m/s} \quad \text{(vrijwel hetzelfde in lucht).}
\]
\end{proposition}
\admitted

\begin{remark}\label{rem:g10:signals-and-waves:honest}
Hier is dit een experimenteel feit --- zonlicht steekt $150$ miljoen
kilometer lege ruimte over; het wordt eerlijk afgeleid in het deel
van jaar 2 uit de wetten van elektriciteit en magnetisme. Houd het
contrast: \omterm{def:g10:signals-and-waves:sound}{geluid} is een trilling \emph{van} het medium
en sterft zonder het; een \omterm{prop:g10:signals-and-waves:em}{elektromagnetisch signaal}
draagt zijn eigen veld door vacuüm, een miljoen keer sneller.
\end{remark}

\section{Echo's: afstanden uit vertragingen}

Een \omterm{def:g10:signals-and-waves:signal}{signaal} dat met snelheid $v$ beweegt, overbrugt
$d = v\,t$ in een tijd $t$: elke vertraging is een afstand in
vermomming. Beter nog, stuur een korte puls naar een obstakel en tijd
haar \emph{echo}\index{echo} --- de heen-en-terugreis overbrugt $2d$.

\begin{method}[Echopeiling]\label{met:g10:signals-and-waves:echo}
\begin{enumerate}
  \item Stuur een korte puls; meet de heen-en-terug-tijd $t$ van de
        \omterm{rem:g10:signals-and-waves:honest}{echo}.
  \item Neem de snelheid $v$ van het
        \omterm{def:g10:signals-and-waves:signal}{signaal} \emph{in het doorkruiste medium}.
  \item De puls overbrugde $2d = v\,t$, dus $d = \dfrac{v\,t}{2}$.
\end{enumerate}
\omterm{def:g10:signals-and-waves:sound}{Geluids}pulsen in water maken \emph{sonar}\index{sonar}
($v \approx \qty{1500}{m/s}$); radiopulsen in lucht maken
\emph{radar}\index{radar} ($v = c$);
\omterm{def:g10:signals-and-waves:ultrasound}{ultrageluids}pulsen in het lichaam maken de
medische scanner ($v \approx \qty{1540}{m/s}$ in weke delen).
\end{method}

\begin{example}[Sonar]\label{ex:g10:signals-and-waves:sonar}
De \omterm{met:g10:signals-and-waves:echo}{sonar} van een schip pingt en de
\omterm{rem:g10:signals-and-waves:honest}{echo} van de zeebodem keert terug na
$t = \qty{0.80}{s}$: de diepte is
$d = \frac{1500 \times 0.80}{2} = \qty{600}{m}$. De factor $2$
vergeten verdubbelt de oceaan.
\end{example}

\begin{example}[Radar]\label{ex:g10:signals-and-waves:radar}
Een luchthaven-\omterm{met:g10:signals-and-waves:echo}{radar} ontvangt de
\omterm{rem:g10:signals-and-waves:honest}{echo} van een vliegtuig $\qty{0.30}{ms}$ na de
puls:
$d = \frac{\qty{3.00e8}{m/s} \times \qty{3.0e-4}{s}}{2} = \qty{45}{km}$.
Omdat $c$ zo groot is, leeft \omterm{met:g10:signals-and-waves:echo}{radar} op
microsecondeklokken: licht timen is precisiewerk.
\end{example}

\begin{example}[Het hart als signaal]\label{ex:g10:signals-and-waves:ecg}
Elke hartslag stuurt een kleine spanningspuls over de borst; elektroden
registreren hem als het
\emph{elektrocardiogram}\index{elektrocardiogram} (ECG), een bijna
\omterm{def:g10:signals-and-waves:period}{periodiek signaal}. Als de hoge pieken $T = \qty{0.75}{s}$
uit elkaar liggen, slaat het hart met $f = 1/0.75 = \qty{1.33}{Hz}$,
d.w.z.\ $1.33 \times 60 = 80$ slagen per minuut: $T$ van een spoor
aflezen is hoe een cardioloog je pols neemt.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:1}
Bereken de \omterm{def:g10:signals-and-waves:frequency}{frequentie} van het
\omterm{def:g10:signals-and-waves:period}{periodieke signaal} van
\omterm{def:g10:signals-and-waves:period}{periode} (a) $T = \qty{20}{ms}$; (b)
$T = \qty{4.0}{ms}$; (c) $T = \qty{50}{\micro s}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:2}
Bereken de \omterm{def:g10:signals-and-waves:period}{periode} van (a) het \qty{50}{Hz}-net;
(b) een \qty{440}{Hz}-kamertoon A; (c) een
\qty{2.4}{GHz}-wifi-\omterm{def:g10:signals-and-waves:signal}{signaal}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:3}
Classificeer als \omterm{def:g10:signals-and-waves:ultrasound}{infrageluid}, hoorbaar
\omterm{def:g10:signals-and-waves:sound}{geluid} of \omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid}:
\qty{12}{Hz}; \qty{440}{Hz}; \qty{18}{kHz}; \qty{40}{kHz} (een
parkeersensor); \qty{5}{MHz} (een medische sonde).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:4}
Op een \omterm{def:g10:signals-and-waves:oscilloscope}{oscilloscoop} gezet op \qty{2}{ms} per

divisies en zit een top $4.0$ divisies boven de as. Vind $T$, $f$ en

\begin{exercise}[$\star$]\label{exo:g10:signals-and-waves:5}

sloeg hij in ($v_{\text{sound}} = \qty{340}{m/s}$)? Waarom mag de

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:6}

\omterm{rem:g10:signals-and-waves:honest}{echo} over een \qty{1200}{m} diepe geul doen?

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:7}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:8}

\begin{exercise}[$\star\star$]\label{exo:g10:signals-and-waves:10}

\end{exercise}

\section{Probleem: De sonar, de storm en het cardiogram}

\begin{problem}[{Weekendprobleem --- de signalen van de wereld lezen:
één wet, $d = v\,t$, peilt een storm, karteert een zeebodem, telt een
hartslag en peurt uit hoe GPS werkt}]
\label{pb:g10:signals-and-waves:1}
Een vissersboot werkt door een stormachtige nacht: bliksem aan de
horizon, de \omterm{met:g10:signals-and-waves:echo}{sonar} die de bodem veegt, het hart
van de schipper op het papierstrip van de dokter, een GPS-ontvanger
die naar \omterm{def:g10:universal-gravitation:satellite}{satellieten} luistert. Vier
instrumenten, één wet. Neem $v_{\text{sound}} = \qty{340}{m/s}$ in
lucht, \qty{1500}{m/s} in zeewater, en $c = \qty{3.00e8}{m/s}$.

\medskip
\noindent\textbf{Deel I --- De storm.}
\begin{enumerate}
  \item Een flits; donder volgt \qty{9.0}{s} later. Hoe ver is hij?
  \item Bereken de reistijd van het licht over die afstand, en
        rechtvaardig de flits als onmiddellijk te behandelen.
  \item Zeelieden tellen de seconden tussen flits en donder en delen
        door drie om kilometers te krijgen. Rechtvaardig de regel.
  \item Vijf minuten later geeft een flits \qty{6.0}{s}. Hoe ver nu?
        Vind de gemiddelde naderingssnelheid van de storm in
        \unit{m/s} en \unit{km/h}.
  \item Donder rommelt in plaats van te kraken: het bliksemkanaal is
        kilometers lang, dus liggen zijn delen op verschillende
        afstanden. Als het kanaal van \qty{2.0}{km} tot \qty{5.0}{km}
        van de boot strekt, hoe lang duurt de rommel?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De \omterm{met:g10:signals-and-waves:echo}{sonar}.}
\begin{enumerate}[resume]
  \item De \omterm{met:g10:signals-and-waves:echo}{sonar} pingt; de zeebodem antwoordt in
        \qty{0.90}{s}. Diepte?
  \item Over het plateau verkort de
        \omterm{rem:g10:signals-and-waves:honest}{echo} tot \qty{0.20}{s}. Diepte?
  \item Eén ping keert \emph{twee}
        \omterm{rem:g10:signals-and-waves:honest}{echo's} terug, bij \qty{0.16}{s} en
        \qty{0.20}{s}. Interpreteer ze; hoe ver boven de bodem zwemt
        de visschool?
  \item De \omterm{met:g10:signals-and-waves:echo}{sonar} pingt elke \qty{0.50}{s}. Wat is
        de grootste diepte die ze zonder verwarring kan meten, en wat
        gaat er mis voorbij die?
  \item In lucht zou dezelfde vertraging van \qty{0.60}{s} welke
        afstand betekenen? Moraal: wat moet je weten voor je een
        vertraging tot afstand maakt?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Het cardiogram.}
\begin{enumerate}[resume]
  \item De ECG-pieken van de schipper liggen \qty{0.86}{s} uit
        elkaar. \omterm{def:g10:signals-and-waves:frequency}{Frequentie}? Slagen per minuut?
  \item De strip schuift met \qty{25}{mm/s}. Welke piekafstand mat de
        dokter?
  \item Na het binnenhalen van netten is de afstand \qty{15}{mm}.
        Nieuwe hartslag?
  \item De jonge matroos, getrainde roeier, rust bij $50$ slagen per
        minuut. \omterm{def:g10:signals-and-waves:period}{Periode}? Afstand op de strip?
  \item Is een ECG strikt \omterm{def:g10:signals-and-waves:period}{periodiek}? Wat leest de
        dokter dan van het spoor?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Huiswaarts bij licht.}
\begin{enumerate}[resume]
  \item De boot radio't de haven, \qty{50}{km} verder. Hoe lang doet
        het bericht erover? Hoe lang zou
        \omterm{def:g10:signals-and-waves:sound}{geluid} erover doen?
  \item Een GPS-\omterm{def:g10:universal-gravitation:satellite}{satelliet}
        \omterm{def:g10:universal-gravitation:satellite}{omloopt} ongeveer \qty{20200}{km}
        boven. Hoe lang doet haar
        \omterm{def:g10:signals-and-waves:signal}{signaal} erover om de boot te bereiken?
  \item GPS zet tijd om in positie. Welke afstands\-fout veroorzaakt
        een klokfout van \qty{1.0}{\micro s}? Welke
        timingprecisie eist een \qty{3.0}{m}-fixatie?
  \item Waarom zou geen geluid-gebaseerde GPS kunnen bestaan, zelfs
        niet in principe?
  \item Finale: noteer de drie snelheden van vanavond en de enige wet
        achter alle vier instrumenten, en stel de peilgewoonte in één
        zin.
\end{enumerate}
\end{problem}
