\chapter{Lichtspectra en de boodschap van het licht}\label{ch:g10:light-spectra}

Een bundel sterrenlicht is het enige monster van een ster dat we ooit
zullen vasthouden, en het blijkt een gulzig monster te zijn. Spreid de
bundel uit tot zijn \omterm{def:g10:light-spectra:white}{spectrum} --- een glazen prisma doet het, een
gordijn van regendruppels doet het --- en het licht wordt een
boodschap: een continue regenbogenachtergrond kondigt de temperatuur
van de ster aan, en een streepjescode van fijne lijnen spelt, element
voor element, uit waaruit de ster bestaat. Dit hoofdstuk leert beide
delen van de boodschap te lezen --- eerst op laboratoriumlampen, daarna
op de Zon.

\section{Wit licht en zijn spectrum}

\begin{definition}[Ontbinding van wit licht]\label{def:g10:light-spectra:white}
Wanneer een smalle bundel zonlicht een glazen prisma doorkruist, treedt
hij uit gespreid tot een continue band van kleuren, van rood tot violet.
Die band is het \emph{spectrum}\index{spectrum} van het licht. Licht dat
alle zichtbare kleuren bevat, zoals zonlicht of het licht van een
gewone gloeilamp, heet \emph{wit licht}\index{wit licht}. Licht van één
zuivere kleur, die geen prisma verder kan splitsen, is
\emph{monochromatisch}\index{monochromatisch licht} (bijvoorbeeld een
laserstraal); een mengsel van meerdere kleuren is
\emph{polychromatisch}\index{polychromatisch licht}. Een instrument dat
de ontbinding uitvoert en de samenstelling van een licht afleest, is een
\emph{spectroscoop}\index{spectroscoop}.
\end{definition}

\begin{remark}[Prisma's en regendruppels]\label{rem:g10:light-spectra:rainbow}
Het prisma werkt omdat glas violet licht iets sterker afbuigt dan rood
licht --- \emph{waarom} dat zo is, hoort bij de studie van de breking, in
\cref{ch:g10:refraction}; hier gebruiken we alleen het effect. Water doet
hetzelfde: elke regendruppel van een passerende bui werkt als een
miniatuurprisma, dat wit zonlicht ontvangt en gesorteerd op kleur
terugstuurt. Een regenboog is het \omterm{def:g10:light-spectra:white}{spectrum} van de Zon, over de
hemel getekend.
\end{remark}

\begin{definition}[Golflengte, de identiteitskaart van een straling]\label{def:g10:light-spectra:wavelength}
Elke \omterm{def:g10:light-spectra:white}{monochromatische} straling wordt gekenmerkt door één getal, haar
\emph{golflengte}\index{golflengte} $\lambda$, gemeten in nanometers
($\qty{1}{nm} = \qty{e-9}{m}$). De golflengte is de identiteitskaart van
de straling: geef $\lambda$ en je hebt alles gezegd wat er over de
zuivere kleur te zeggen valt. Het oog reageert op golflengten tussen
ongeveer $\qty{400}{nm}$ (violet) en $\qty{800}{nm}$ (rood) --- het
\emph{zichtbare spectrum}\index{zichtbaar spectrum}. Voorbij de twee
uiteinden gaat de straling door, voor ons onzichtbaar:
\emph{ultraviolet}\index{ultraviolet} onder $\qty{400}{nm}$,
\emph{infrarood}\index{infrarood} boven $\qty{800}{nm}$. Wat de
golflengte feitelijk meet --- de lengte van één rimpel van een
lichtgolf --- wordt uitgelegd in \cref{ch:g10:signals-and-waves}; in dit
hoofdstuk is het getal zelf wat telt.
\end{definition}

\begin{example}[Golflengten lezen]\label{ex:g10:light-spectra:classify}
Een groene laserpointer zendt uit op $\qty{532}{nm}$: binnen het
zichtbare bereik, in het groen. Een desinfecterende kwiklamp zendt uit
op $\qty{254}{nm}$: \omterm{def:g10:light-spectra:wavelength}{ultraviolet}, onzichtbaar (en gevaarlijk voor
het oog precies omdat onzichtbaar). De diode van een
televisieafstandsbediening zendt uit op $\qty{940}{nm}$:
\omterm{def:g10:light-spectra:wavelength}{infrarood} --- richt hem op de camera van een telefoon, waarvan
de sensor een beetje voorbij $\qty{800}{nm}$ ziet, en de onzichtbare
flitsen verschijnen op het scherm.
\end{example}

\section{Continue spectra en temperatuur}

\begin{definition}[Continu spectrum]\label{def:g10:light-spectra:continuous}
Dichte, hete materie --- de gloeidraad van een lamp, gesmolten staal,
het oppervlak van een ster --- gloeit, en haar licht bevat \emph{elke}
\omterm{def:g10:light-spectra:wavelength}{golflengte} over een breed bereik: een ononderbroken kleurenband
zonder hiaten. Zo'n \omterm{def:g10:light-spectra:white}{spectrum} is een \emph{continu
spectrum}\index{continu spectrum}. Het draagt geen spoor van de
chemische aard van het lichaam: een witgloeiende ijzeren staaf en een
witgloeiende koperen staaf gloeien hetzelfde. Wat het wél codeert ---
volledig --- is de temperatuur.
\end{definition}

\begin{definition}[Absolute temperatuur]\label{def:g10:light-spectra:kelvin}
Voor stralingswetten telt men de \emph{absolute
temperatuur}\index{absolute temperatuur} niet vanaf het vriespunt van
water maar vanaf de koudst mogelijke toestand,
$\qty{-273}{\degreeCelsius}$. De \omterm{def:g10:orders-of-magnitude:unit}{eenheid} is de
\emph{kelvin}\index{kelvin} (symbool $\unit{K}$), met dezelfde
graadgrootte:
\[
T \text{ (in } \unit{K}) = \theta \text{ (in } \unit{\degreeCelsius}) + 273 .
\]
Zo is een kamer bij $\qty{20}{\degreeCelsius}$ op $\qty{293}{K}$, en het
oppervlak van de Zon, nabij $\qty{5500}{\degreeCelsius}$, nabij
$\qty{5800}{K}$.
\end{definition}

\begin{proposition}[Heter betekent helderder en blauwer (wet van Wien)]\label{prop:g10:light-spectra:wien}
Wanneer de temperatuur van een dicht gloeiend lichaam stijgt,
\begin{enumerate}
  \item straalt het meer bij \emph{elke} \omterm{def:g10:light-spectra:wavelength}{golflengte}: het hele
        \omterm{def:g10:light-spectra:white}{spectrum} wordt helderder;
  \item de \omterm{def:g10:light-spectra:wavelength}{golflengte} $\lambda_{\max}$ waarbij het het sterkst
        straalt, glijdt naar de \omterm{def:g10:light-spectra:wavelength}{kortgolvige} (blauwe) kant, volgens
        \[
        \lambda_{\max} \times T = \qty{2.90e-3}{m.K},
        \]
        waarbij $T$ de \omterm{def:g10:light-spectra:kelvin}{absolute temperatuur} in
        \omterm{def:g10:light-spectra:kelvin}{kelvin} is.
\end{enumerate}
\end{proposition}
\admitted

\begin{remark}[Waar deze wet vandaan komt]\label{rem:g10:light-spectra:wienorigin}
De wet van Wien vat zorgvuldige metingen aan ovens en gloeidraden samen,
gemaakt aan het eind van de negentiende eeuw. Haar afleiden vereist de
kwantumtheorie van de straling --- dit \omterm{def:g10:light-spectra:white}{spectrum} is precies wat de
klassieke natuurkunde \emph{niet} kon verklaren, en de puzzel die Planck
dwong het kwantum uit te vinden. De eerlijke afleiding staat in het
deel van jaar 3.
\end{remark}

\section{Lijnenspectra van aangeslagen gassen}

\begin{definition}[Emissielijnenspectrum]\label{def:g10:light-spectra:emission}
Een gas bij lage druk, aangeslagen door een elektrische ontlading (zoals
in een neonbuis of een natriumstraatverlichting), gloeit ook --- maar
door een \omterm{def:g10:light-spectra:white}{spectroscoop} lijkt haar licht niets op een regenboog.
Alleen enkele geïsoleerde \omterm{def:g10:light-spectra:wavelength}{golflengten} zijn aanwezig: dunne
heldere lijnen op een zwarte achtergrond, \emph{spectrale
lijnen}\index{spectrale lijn} genoemd. Zo'n \omterm{def:g10:light-spectra:white}{spectrum} is een
\emph{emissielijnenspectrum}\index{emissiespectrum}\index{lijnenspectrum}.
\end{definition}

\begin{example}[Waterstof en natrium]\label{ex:g10:light-spectra:hydrogen}
Aangeslagen waterstof zendt precies vier lijnen uit in het zichtbare:
een rode lijn bij $\qty{656}{nm}$, een blauwgroene lijn bij
$\qty{486}{nm}$, en twee violette lijnen bij $\qty{434}{nm}$ en
$\qty{410}{nm}$. Aangeslagen natrium zendt in wezen één sterke
geeloranje lijn uit bij $\qty{589}{nm}$ --- daarom baden oudere
straatverlichtingen hele buurten in die ene kleur: hun licht bevat
vrijwel niets anders.
\end{example}

\begin{proposition}[De vingerafdruk van de elementen]\label{prop:g10:light-spectra:fingerprint}
Elk chemisch element zendt zijn eigen vaste catalogus van
\omterm{def:g10:light-spectra:emission}{spectrale lijnen} uit, dezelfde in elk laboratorium en in elk
tijdperk, en geen twee elementen delen dezelfde catalogus. Een
\omterm{def:g10:light-spectra:emission}{lijnenspectrum} identificeert daarom het emitterende element,
zoals een vingerafdruk een persoon identificeert.
\end{proposition}
\admitted

\begin{remark}[Waarom lijnen?]\label{rem:g10:light-spectra:whylines}
Waarom een atoom alleen bepaalde \omterm{def:g10:light-spectra:wavelength}{golflengten} uitzendt --- en
waarom die \omterm{def:g10:light-spectra:wavelength}{golflengten} het element verraden --- wordt uitgelegd
door de kwantumtheorie van het atoom, in het laatste jaar van dit deel:
elk element bezit een ladder van energieniveaus, en elke lijn markeert
één sprong tussen sporten. Voorlopig gebruiken we de vingerafdruk zoals
detectives: zonder te vragen hoe de vingers groeiden.
\end{remark}

\section{Absorptiespectra}

\begin{definition}[Absorptiespectrum]\label{def:g10:light-spectra:absorption}
Stuur \omterm{def:g10:light-spectra:white}{wit licht} \emph{door} een koel gas bij lage druk en ontbind
wat eruit komt: de continue regenboog verschijnt opnieuw, maar doorboord
door dunne donkere lijnen. Het gas heeft bepaalde
\omterm{def:g10:light-spectra:wavelength}{golflengten} uit het passerende licht gehaald. Zo'n
\omterm{def:g10:light-spectra:white}{spectrum} --- een continue achtergrond minus een reeks lijnen --- is
een \emph{absorptiespectrum}\index{absorptiespectrum}.
\end{definition}

\begin{proposition}[Emissie- en absorptielijnen vallen samen]\label{prop:g10:light-spectra:kirchhoff}
Een gas absorbeert precies de \omterm{def:g10:light-spectra:wavelength}{golflengten} die het kan uitzenden: de
donkere lijnen van het \omterm{def:g10:light-spectra:absorption}{absorptiespectrum} van een element zitten
bij dezelfde \omterm{def:g10:light-spectra:wavelength}{golflengten} als de heldere lijnen van zijn
\omterm{def:g10:light-spectra:emission}{emissiespectrum}. De vingerafdruk van
\cref{prop:g10:light-spectra:fingerprint} kan dus zowel in het donker als
in het heldere gelezen worden.
\end{proposition}
\admitted

\begin{remark}[Eén mechanisme, twee tekens]\label{rem:g10:light-spectra:mechanism}
De samenval is geen toeval: absorberen bij $\lambda$ beklimt dezelfde
energiesport die emitteren bij $\lambda$ afdaalt --- de
kwantumhoofdstukken aan het eind van dit deel maken dit precies. Vastgesteld
door Kirchhoff en Bunsen in 1859, is het de sleutel die de hele volgende
sectie opent.
\end{remark}

\begin{example}[Een kolf natriumdamp]\label{ex:g10:light-spectra:sodiumflask}
\omterm{def:g10:light-spectra:white}{Wit licht} doorkruist een kolf met koele natriumdamp. Het doorgelaten

\end{example}

\section{Spectrale analyse: een ster lezen}

\begin{definition}[Spectrale analyse]\label{def:g10:light-spectra:analysis}
\emph{Spectrale analyse}\index{spectrale analyse} is de identificatie
van de chemische elementen in een bron (een vlam, een lamp, een ster)
door de lijnen van haar \omterm{def:g10:light-spectra:white}{spectrum} te matchen met de
laboratoriumcatalogi van de elementen.
\end{definition}

\begin{method}[Een sterrenspectrum lezen]\label{met:g10:light-spectra:reading}
Een ster is een bal dichte hete materie (het oppervlak) gewikkeld in
een dunnere, koelere atmosfeer. Haar licht arriveert dus als een
\omterm{def:g10:light-spectra:continuous}{continu spectrum} doorboord door absorptielijnen --- en elk
deel wordt apart gelezen:
\begin{enumerate}
  \item Lokaliseer de piek $\lambda_{\max}$ van de continue
        achtergrond; de wet van Wien (\cref{prop:g10:light-spectra:wien})
        geeft de oppervlaktetemperatuur
        $T = \qty{2.90e-3}{m.K} / \lambda_{\max}$.
  \item Noteer de \omterm{def:g10:light-spectra:wavelength}{golflengten} van de donkere lijnen.
  \item Match ze met de catalogi: een element mag pas aanwezig
        verklaard worden als \emph{al} zijn sterke lijnen verschijnen.
  \item Besluit: de absorptie gebeurt in de atmosfeer van de ster, dus
        de geïdentificeerde elementen zijn die van de atmosfeer.
\end{enumerate}
\end{method}

\begin{example}[Het recept van de Zon]\label{ex:g10:light-spectra:sunrecipe}
Het zonne-\omterm{def:g10:light-spectra:white}{spectrum} draagt duizenden donkere lijnen, in 1814 door
Fraunhofer in kaart gebracht. Onder de sterkste: $410$, $434$, $486$ en
$\qty{656}{nm}$ --- de volledige waterstofvingerafdruk --- samen met de
natriumlijn bij $\qty{589}{nm}$ en een calcium-paar bij $393$ en
$\qty{397}{nm}$. De atmosfeer van de Zon bevat vooral waterstof, plus
sporen van elementen die bekend zijn uit aardse rotsen. Moderne metingen
verfijnen de kop: de Zon is grofweg driekwart waterstof naar massa.
\end{example}

\begin{remark}[Het element dat eerst in de hemel gevonden werd]\label{rem:g10:light-spectra:helium}
Tijdens de eclips van 1868 verscheen een heldere lijn bij
$\qty{587.6}{nm}$ in het \omterm{def:g10:light-spectra:white}{spectrum} van de zonneatmosfeer --- die
matchte met geen bekend element. Volgens
\cref{prop:g10:light-spectra:fingerprint} betekent een onbekende
vingerafdruk een onbekend element, dus werd er een aangekondigd en
genoemd naar de Griekse zon, \emph{helios}: helium. Het werd pas in
1895 op Aarde geïsoleerd --- het enige chemische element dat in de
ruimte ontdekt werd voordat het thuis gevonden werd.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:light-spectra:1}
Zeg voor elke straling of ze \omterm{def:g10:light-spectra:wavelength}{ultraviolet}, zichtbaar of
\omterm{def:g10:light-spectra:wavelength}{infrarood} is, en geef haar kleur als zichtbaar:
$\qty{254}{nm}$ (kwiklamp), $\qty{480}{nm}$, $\qty{589}{nm}$,
$\qty{656}{nm}$, $\qty{1064}{nm}$ (snijlaser).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:light-spectra:2}
Een smalle bundel zonlicht valt op een glazen prisma.
\begin{enumerate}
  \item Beschrijf wat op een scherm achter het prisma verschijnt.
  \item Welk uiteinde van de band is het meest afgebogen?
  \item Het zonlicht wordt vervangen door de bundel van een groene
        laserpointer. Wat verschijnt nu op het scherm, en wat toont dit
        over laserlicht?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:light-spectra:3}
Koppel elke bron aan het type \omterm{def:g10:light-spectra:white}{spectrum} dat ze produceert

\section{Probleem: Sterrenlicht lezen}

\begin{problem}[{Weekendprobleem --- sterrenlicht lezen: de filosoof die
«nooit» zei, en hoe een prisma de temperatuur van een ster neemt en haar
chemisch recept leest}]
\label{pb:g10:light-spectra:1}
In 1835 had de filosoof Auguste Comte een voorbeeld nodig van kennis
die voor altijd buiten menselijk bereik lag, en koos het zorgvuldig: de
chemische samenstelling van de sterren. Niemand zou ooit een ster
flesje doen. In 1859 matchten Kirchhoff en Bunsen de donkere lijnen van
de Zon met de heldere lijnen van laboratoriumvlammen, en de
onbereikbare kennis kwam met retour van licht. Dit probleem hertekent
het hele pad: laboratoriumvingerafdrukken, gloeiende lichamen, de
donkere lijnen van de Zon --- en ten slotte een ster waarvan je
temperatuur en samenstelling van achter je bureau zult bepalen.

\medskip
\noindent\textbf{Deel I --- Vingerafdrukken op dossier.}
\begin{enumerate}
  \item Geef het zichtbare bereik van \omterm{def:g10:light-spectra:wavelength}{golflengten} in nanometers
        en de kleur aan elk uiteinde. Waar zit $\qty{550}{nm}$?
  \item Een waterstofontladingslamp toont lijnen bij $410$, $434$, $486$
        en $\qty{656}{nm}$. Geef de kleur van elke lijn.
  \item Een natriumlamp zendt in wezen één lijn uit bij $\qty{589}{nm}$.
        Hoe ziet de lamp er voor het oog uit, en waarom is het hopeloos
        de kleur van een geparkeerde auto onder zo'n straatlicht te
        beoordelen?
  \item Leg uit waarom een reeks \omterm{def:g10:light-spectra:emission}{spectrale lijnen} een element
        identificeert --- en waarom \emph{al} de sterke lijnen van een
        element gevonden moeten worden voor men het aanwezig verklaart.
  \item Voorspel het \omterm{def:g10:light-spectra:white}{spectrum} dat men ziet (a) wanneer
        \omterm{def:g10:light-spectra:white}{wit licht} een kolf met koele natriumdamp doorkruist, en
        (b) wanneer dezelfde damp, alleen in een donkere kamer, door een
        ontlading wordt aangeslagen.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Hete lichamen en hun kleuren.}
\begin{enumerate}[resume]
  \item Noem de twee effecten van het verhogen van de temperatuur van
        een dicht gloeiend lichaam op zijn \omterm{def:g10:light-spectra:continuous}{continu spectrum}.
  \item Een halogeengloeidraad loopt op $\qty{3400}{K}$. Bereken
        $\lambda_{\max}$ en geef haar domein.
  \item Het \omterm{def:g10:light-spectra:continuous}{continue spectrum} van de Zon piekt bij
        $\qty{500}{nm}$. Bereken haar oppervlaktetemperatuur.
  \item Die piek zit in het groen --- toch ziet de Zon er niet groen
        uit. Verklaar.
  \item Een pook in de smidse wordt eerst op afstand gevoeld zonder
        gloed te tonen, gloeit dan dof rood, dan oranje-wit. Verklaar
        de hele reeks met vraag 6.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De donkere lijnen van de Zon.}
\begin{enumerate}[resume]
  \item Het zonne-\omterm{def:g10:light-spectra:white}{spectrum} is een continue band doorkruist door
        duizenden fijne donkere lijnen. Welk deel van de Zon produceert
        de continue achtergrond, en welk deel produceert de lijnen?
  \item Sterke zonnlijnen zitten bij $393$, $397$, $410$, $434$, $486$,
        $589$ en $\qty{656}{nm}$. Catalogi: waterstof
        $\{410, 434, 486, 656\}\,\unit{nm}$; natrium
        $\{589\}\,\unit{nm}$; calcium $\{393, 397\}\,\unit{nm}$. Welke
        elementen zijn aanwezig in de atmosfeer van de Zon?
  \item De waterstoflijnen zijn veruit de sterkste. Wat suggereert dit
        over de samenstelling van de Zon --- en wat zeggen moderne
        metingen?
  \item In 1868 matchte een lijn bij $\qty{587.6}{nm}$ waargenomen in
        de atmosfeer van de Zon met geen catalogus. Reconstructeer de
        redenering die het aankondigen van een nieuw element
        rechtvaardigde, en zeg hoe het verhaal eindigde.
  \item Waarom zijn de zonnlijnen donker, terwijl dezelfde atomen in
        een laboratoriumontlading heldere lijnen geven? Tijdens een
        totale eclips, voor enkele seconden, blijft alleen de dunne
        atmosfeer van de Zon zichtbaar tegen een donkere hemel --- en
        dezelfde lijnen flitsen \emph{helder}. Verklaar.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Een ster op het bureau.}
Het \omterm{def:g10:light-spectra:white}{spectrum} van de heldere ster Vega ligt voor je: een continue
achtergrond die piekt bij $\qty{290}{nm}$, doorboord door sterke
donkere lijnen bij $410$, $434$, $486$ en $\qty{656}{nm}$.
\begin{enumerate}[resume]
  \item Bereken de oppervlaktetemperatuur van Vega.
  \item De piek ligt in het \omterm{def:g10:light-spectra:wavelength}{ultraviolet}. Voorspel de schijnbare
        kleur van de ster, en verzoen «piek buiten het zichtbare» met
        «heldere ster».
  \item Identificeer het element dat de atmosfeer van Vega domineert.
  \item Vergelijk Vega met de Zon: temperatuursverhouding, kleur, en
        helderheid per \omterm{def:g10:orders-of-magnitude:unit}{eenheid} gloeiend oppervlak.
  \item Finale --- vul het identiteitsformulier in dat Comte
        onmogelijk verklaarde: temperatuur, dominant element,
        schijnbare kleur van Vega, alles gelezen uit één bundel licht.
        Hoeveel jaar duurde «nooit»?
\end{enumerate}
\end{problem}
