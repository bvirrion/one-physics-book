\chapter{Universal Gravitation and Weight}\label{ch:g10:universal-gravitation}

Drop an apple: it falls. The Moon, sixty Earth radii up, never seems to.
Newton's great insight is that the Moon \emph{is} falling --- about
1.4 millimetres every second, exactly the amount an attraction
weakening as the square of the distance predicts. One formula, with one
universal constant, governs the apple, the Moon, the tides and the
planets. The first volume described \omterm{def:g10:universal-gravitation:force}{gravitation} with words; this
chapter measures it.

\section{The law of universal gravitation}

\begin{definition}[Gravitational force]\label{def:g10:universal-gravitation:force}
Any two bodies attract each other, whatever they are made of and however
far apart they sit. This attraction is the \emph{gravitational
force}\index{gravitational force}, and the phenomenon is
\emph{gravitation}\index{gravitation}. It needs no contact, no rope and
no medium, and it is \emph{mutual}: each body pulls the other.
\end{definition}

\begin{theorem}[Law of universal gravitation]\label{thm:g10:universal-gravitation:law}
Two bodies of masses $m_1$ and $m_2$ (in \unit{kg}), whose centers are a
distance $d$ apart (in \unit{m}), attract each other with two forces
directed along the line joining the centers, each pointing toward the
other body, of equal magnitude
\[
F = G\,\frac{m_1 m_2}{d^2}
\qquad \text{(in \unit{N})},
\]
where $G$ is the same constant for every pair of bodies in the universe.
\end{theorem}
\admitted

\begin{remark}[Where the law comes from]
Newton distilled this law from the observed motion of the planets
(Kepler's laws); the honest deduction is carried out in the Year 1
volume. \Cref{ch:g12:satellites-kepler} will already put the law to work
on \omterm{def:g10:universal-gravitation:satellite}{satellites} and planets.
\end{remark}

\begin{definition}[The gravitational constant]\label{def:g10:universal-gravitation:constant}
The constant
\[
G = \qty{6.67e-11}{N.m^2/kg^2}
\]
is the \emph{gravitational constant}\index{gravitational constant}. Its
minuscule value is the reason gravity between everyday objects is
imperceptible: only when at least one mass is astronomical does
$G m_1 m_2 / d^2$ amount to anything.
\end{definition}

\begin{remark}[Reading the formula]\label{rem:g10:universal-gravitation:reading}
Four features deserve attention.
\begin{itemize}
  \item \emph{Mutual and equal.} The Earth pulls you with your \omterm{def:g10:universal-gravitation:weight}{weight}
        --- and you pull the Earth with exactly the same force. (The
        Earth just responds less: it is about $\num{e23}$ times
        heavier.)
  \item \emph{Inverse square.} Doubling $d$ divides the force by $4$;
        moving ten times farther divides it by $100$.
  \item \emph{Distance between centers.} For spherical bodies --- stars,
        planets, cannonballs --- $d$ is measured center to center, not
        surface to surface.
  \item \emph{Both masses matter.} Doubling either mass doubles the
        force; the formula is symmetric in $m_1$ and $m_2$.
\end{itemize}
\end{remark}

\begin{example}[Two friends]\label{ex:g10:universal-gravitation:friends}
Two \qty{70}{kg} friends stand with their centers \qty{1.0}{m} apart:
\[
F = \num{6.67e-11} \times \frac{70 \times 70}{1.0^2}
\approx \qty{3.3e-7}{N}.
\]
Each friend's \omterm{def:g10:universal-gravitation:weight}{weight} is about \qty{6.9e2}{N} --- two \emph{billion}
times larger. Their mutual attraction is real, but hopeless to feel.
\end{example}

\begin{example}[The Earth holds the Moon]\label{ex:g10:universal-gravitation:earthmoon}
With $m_1 = \qty{5.97e24}{kg}$ (Earth), $m_2 = \qty{7.35e22}{kg}$ (Moon)
and $d = \qty{3.84e8}{m}$:
\[
F = \num{6.67e-11} \times
\frac{\num{5.97e24} \times \num{7.35e22}}{(\num{3.84e8})^2}
\approx \qty{2.0e20}{N}.
\]
And the Moon pulls the Earth back with the same \qty{2.0e20}{N} --- a
pull the oceans answer twice a day, as the tides.
\end{example}

\begin{method}[Computing a gravitational force]\label{met:g10:universal-gravitation:compute}
\begin{enumerate}
  \item Convert the masses to \omterm{def:g10:orders-of-magnitude:unit}{kilograms} and the center-to-center
        distance to meters.
  \item Apply $F = G m_1 m_2 / d^2$.
  \item Check the power of ten separately from the digits, as in
        \cref{ch:g10:orders-of-magnitude}: \omterm{def:g10:universal-gravitation:force}{gravitational forces} range
        from $\num{e-7}$ newtons (two people a \omterm{def:g10:orders-of-magnitude:unit}{metre} apart) to $\num{e22}$ newtons
        (Sun on Earth), so a misplaced \omterm{def:g10:orders-of-magnitude:scinot}{exponent} is easy to catch.
\end{enumerate}
\end{method}

\section{Weight and the strength of gravity}

\begin{definition}[Weight]\label{def:g10:universal-gravitation:weight}
The \emph{weight}\index{weight} $\vect P$ of an object at the surface of
a star (a planet, a moon, \dots) is the \omterm{def:g10:universal-gravitation:force}{gravitational force} the star
exerts on it. It points toward the star's center --- the direction we
call \emph{vertical}\index{vertical} --- and is measured in newtons.
\end{definition}

\begin{proposition}[Weight at the surface of a star]\label{prop:g10:universal-gravitation:surfaceg}
At the surface of a spherical star of mass $M$ and radius $R$, an object
of mass $m$ has \omterm{def:g10:universal-gravitation:weight}{weight}
\[
P = m\,g,
\qquad\text{where}\qquad
g = \frac{G M}{R^2}
\]
depends only on the star, not on the object.
\end{proposition}

\begin{proof}
The object's center and the star's center are a distance $R$ apart, so
the law of universal \omterm{def:g10:universal-gravitation:force}{gravitation}
(\cref{thm:g10:universal-gravitation:law}) gives
$P = G M m / R^2 = m \times \left(G M / R^2\right)$. The bracket is the
same for every object at the surface: call it $g$.
\end{proof}

\begin{definition}[Gravitational field strength]\label{def:g10:universal-gravitation:fieldstrength}
The quantity $g = G M / R^2$, in newtons per \omterm{def:g10:orders-of-magnitude:unit}{kilogram}, is the
\emph{gravitational field strength}\index{gravitational field strength}
at the star's surface: the star pulls on each \omterm{def:g10:orders-of-magnitude:unit}{kilogram} with $g$ newtons.
On Earth, $g = \qty{9.81}{N/kg}$.
\end{definition}

\begin{example}[Computing $g$]\label{ex:g10:universal-gravitation:gvalues}
For the Earth ($M = \qty{5.97e24}{kg}$, $R = \qty{6.37e6}{m}$):
\[
g = \frac{\num{6.67e-11} \times \num{5.97e24}}{(\num{6.37e6})^2}
= \frac{\num{3.98e14}}{\num{4.06e13}}
\approx \qty{9.81}{N/kg}.
\]
For the Moon ($M = \qty{7.35e22}{kg}$, $R = \qty{1.74e6}{m}$) the same
computation gives $g_{\text{Moon}} \approx \qty{1.62}{N/kg}$ --- six
times less: the Moon is far lighter, and being smaller does not make up
for it. For Mars, $g_{\text{Mars}} \approx \qty{3.73}{N/kg}$.
\end{example}

\begin{remark}[Why astronauts bounce]\label{rem:g10:universal-gravitation:bounce}
An astronaut of mass \qty{120}{kg} (body plus suit) weighs
$120 \times 9.81 \approx \qty{1180}{N}$ on Earth but only
$120 \times 1.62 \approx \qty{194}{N}$ on the Moon. Muscles trained
under \qty{1180}{N} suddenly carry a sixth of the load: hence the
famous slow, bounding lunar gait. The mass --- and with it the effort
needed to \emph{stop} or \emph{turn} --- is unchanged, which is why
lunar astronauts fell over so often.
\end{remark}

\section{Mass or weight? The balance and the spring scale}

Everyday language says a sack of flour ``weighs three kilos''. Physics
splits that sentence in two.

\begin{remark}[Mass is not weight]\label{rem:g10:universal-gravitation:massweight}
\begin{itemize}
  \item The \emph{mass} $m$ measures the quantity of matter, in
        \omterm{def:g10:orders-of-magnitude:unit}{kilograms}. It is a property of the object alone: the same on
        Earth, on the Moon, or adrift in deep space.
  \item The \emph{\omterm{def:g10:universal-gravitation:weight}{weight}} $P = mg$ is a force, in newtons, exerted
        \emph{by a star} on the object. It changes from star to star
        with $g$ --- and fades away far from every star.
\end{itemize}
Confusing them is harmless at the market and fatal in a physics
problem: the \omterm{def:g10:orders-of-magnitude:unit}{kilogram} is not a \omterm{def:g10:orders-of-magnitude:unit}{unit} of force.
\end{remark}

\begin{method}[The balance and the spring scale]\label{met:g10:universal-gravitation:instruments}
\begin{itemize}
  \item A \emph{beam balance} compares the object with standard masses:
        both pans' \omterm{def:g10:universal-gravitation:weight}{weights} scale by the same $g$, so the comparison
        measures the \emph{mass} --- same verdict on any star.
  \item A \emph{spring scale} measures the force stretching its spring:
        it reads the \emph{\omterm{def:g10:universal-gravitation:weight}{weight}}, in newtons, and its reading changes
        from star to star. To recover the mass, divide by the local
        field strength: $m = P/g$.
\end{itemize}
\end{method}

\begin{example}[Flour on the Moon]\label{ex:g10:universal-gravitation:flour}
A \qty{3.0}{kg} sack of flour balances against \qty{3.0}{kg} of standard
masses on Earth \emph{and} on the Moon. A spring scale, however, reads
$3.0 \times 9.81 \approx \qty{29}{N}$ on Earth and
$3.0 \times 1.62 \approx \qty{4.9}{N}$ on the Moon. A merchant selling
``by the newton'' on the Moon would hand out six times the flour ---
trade, like science, runs on mass.
\end{example}

\section{Falling around the Earth: Newton's cannon}

Throw a stone horizontally: gravity bends its path into an arc and it
lands a few meters away. Throw it faster: it lands farther. Newton's
thought experiment pushes the idea to its limit. The Earth is round, and
its surface drops about \qty{5}{m} below the horizontal for every
\qty{8}{km} traveled; a freely falling body drops \qty{4.9}{m} in its
first second (\cref{ch:g12:projectile-motion} will establish this). So a
projectile skimming the Earth at \qty{8}{km/s} falls exactly as fast as
the ground curves away: it falls forever without ever landing.

\begin{definition}[Satellite and orbit]\label{def:g10:universal-gravitation:satellite}
A \emph{satellite}\index{satellite} of a star is a body that falls
around the star without ever reaching its surface; the path it repeats
is its \emph{orbit}\index{orbit}.
\end{definition}

\begin{remark}[The Moon is a falling stone]\label{rem:g10:universal-gravitation:fallingmoon}
The Moon is the Earth's natural \omterm{def:g10:universal-gravitation:satellite}{satellite}: it falls toward us by about
\qty{1.4}{mm} every second, and its sideways motion carries it around
before it can come any closer
(\cref{exo:g10:universal-gravitation:15} lets you redo Newton's own
check of this number). The astronauts of an orbiting station are in the
same free fall as their ship --- that, and not any absence of gravity,
is why they float. \omterm{def:g10:universal-gravitation:force}{Gravitation} shapes every trajectory, from the
stone's arc to the planet's ellipse; the quantitative story of \omterm{def:g10:universal-gravitation:satellite}{orbits}
is told in \cref{ch:g12:satellites-kepler}.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:1}
Two \qty{60}{kg} dance partners hold each other with their centers
\qty{0.80}{m} apart.
\begin{enumerate}
  \item Compute their mutual gravitational attraction.
  \item Compare it with the \omterm{def:g10:universal-gravitation:weight}{weight} of a \qty{2.5}{mg} mosquito.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:2}
A student has a mass of \qty{55}{kg}.
\begin{enumerate}
  \item Compute her \omterm{def:g10:universal-gravitation:weight}{weight} on Earth.
  \item She travels to the Moon. What are her mass and her \omterm{def:g10:universal-gravitation:weight}{weight}
        there?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:3}
\begin{enumerate}
  \item From the data card, recompute the \omterm{def:g10:universal-gravitation:fieldstrength}{gravitational field strength}
        at the surface of Mars.
  \item A \qty{900}{kg} rover is sent there. Compute its \omterm{def:g10:universal-gravitation:weight}{weight} on Mars
        and its \omterm{def:g10:universal-gravitation:weight}{weight} on Earth.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:4}
Using the data card, compute the force exerted by the Earth on the
Moon. What force does the Moon exert on the Earth?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:universal-gravitation:5}
Two bodies attract each other with a force $F_0$. What does the force
become if:
\begin{enumerate}
  \item the distance between centers is doubled;
  \item the distance is halved;
  \item both masses are doubled (same distance);
  \item one mass is tripled \emph{and} the distance is tripled?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:6}
Two loaded supertankers of \qty{5.0e8}{kg} each are moored with their
centers \qty{100}{m} apart.
\begin{enumerate}
  \item Compute their mutual gravitational attraction.
  \item What mass has a \omterm{def:g10:universal-gravitation:weight}{weight} equal to this force on Earth?
  \item The force is surprisingly large --- yet the tankers show no
        inclination to drift together. Compute the acceleration
        $a = F/m$ it gives one tanker and comment.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:7}
A lunar shop sells a \qty{3.0}{kg} bag of rice.
\begin{enumerate}
  \item What do a beam balance and a spring scale indicate for this bag
        on Earth? On the Moon?
  \item A customer pays for ``\qty{3.0}{kg}''. Which instrument
        guarantees a fair deal on any world, and why?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:8}
The International Space Station flies at an altitude of \qty{400}{km}.
\begin{enumerate}
  \item Compute the Earth's \omterm{def:g10:universal-gravitation:fieldstrength}{gravitational field strength} at that
        altitude. (Careful: what is the distance to the Earth's
        \emph{center}?)
  \item Express it as a percentage of the surface value.
  \item The astronauts on board float. Reconcile this with your answer.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:9}
At the surface of Venus, $g = \qty{8.87}{N/kg}$, and the planet's
radius is \qty{6.05e6}{m}. Deduce the mass of Venus and compare it with
the Earth's.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:10}
\begin{enumerate}
  \item Compute the force exerted by the Sun on \qty{1.0}{kg} of sea
        water, then the force exerted by the Moon on the same
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram}.
  \item The Sun pulls harder --- by what factor?
  \item Yet the Moon dominates the tides. The tide is driven not by the
        pull itself but by the \emph{difference} between the pulls on
        the near and far sides of the Earth. Explain, without
        computing, why the nearby Moon can beat the distant Sun on
        that criterion.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:universal-gravitation:11}
\begin{enumerate}
  \item A planet has twice the Earth's mass and twice its radius.
        Express its surface field strength in terms of the Earth's
        $g$, then in \unit{N/kg}.
  \item What radius would a planet of the Earth's mass need for its
        surface field strength to be $2g$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:12}
Somewhere on the Earth--Moon line, the two pulls on a space probe
cancel. Let $x$ be the distance from the Earth's center to that point
and $D$ the Earth--Moon distance.
\begin{enumerate}
  \item Show that at that point
        $\left(\dfrac{x}{D - x}\right)^{2}
        = \dfrac{M_{\text{Earth}}}{M_{\text{Moon}}}$.
  \item Deduce $x$. What fraction of the trip to the Moon is that?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:13}
Newton's cannon, with numbers. A projectile skims the Earth
horizontally at speed $v$.
\begin{enumerate}
  \item After $x = \qty{8.0}{km}$ of horizontal travel, the spherical
        Earth has ``dropped away'' by a height $h$ satisfying
        $(R + h)^2 = R^2 + x^2$. Using
        $(R + h)^2 \approx R^2 + 2Rh$ for small $h$, show that
        $h \approx x^2 / (2R)$ and compute $h$.
  \item A freely falling body drops \qty{4.9}{m} in its first second.
        Compare with $h$ and deduce the speed at which the projectile
        never lands.
  \item Compute the duration of one full \omterm{def:g10:universal-gravitation:satellite}{orbit} at that speed, skimming
        the surface.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:14}
A neutron star packs $M = \qty{2.8e30}{kg}$ (1.4 times the Sun's mass)
into a radius of \qty{12}{km}.
\begin{enumerate}
  \item Compute the \omterm{def:g10:universal-gravitation:fieldstrength}{gravitational field strength} at its surface.
  \item What would a \qty{70}{kg} human weigh there? Compare with their
        \omterm{def:g10:universal-gravitation:weight}{weight} on Earth.
  \item What is the \omterm{def:g10:universal-gravitation:weight}{weight}, there, of a \qty{1.0}{mg} grain of sand?
        What mass has that \omterm{def:g10:universal-gravitation:weight}{weight} on Earth?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:universal-gravitation:15}
Newton's Moon test (1666). The Moon's \omterm{def:g10:universal-gravitation:satellite}{orbit} radius is about $60$ Earth
radii.
\begin{enumerate}
  \item By what factor is the Earth's pull per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} weaker at the
        Moon than at the Earth's surface, if gravity follows the
        inverse square?
  \item Near the surface, a falling body drops \qty{4.9}{m} in its
        first second. How far should the Moon fall toward the Earth
        each second?
  \item Check against the \omterm{def:g10:universal-gravitation:satellite}{orbit}: the Moon covers its circle of radius
        $d = \qty{3.84e8}{m}$ in $T = 27.3$ days. Compute its speed
        $v$, the distance $x$ it covers in one second, then the fall
        $h \approx x^2/(2d)$ (as in
        \cref{exo:g10:universal-gravitation:13}).
  \item Compare the answers of questions 2 and 3, and state what Newton
        concluded.
\end{enumerate}
\end{exercise}

\section{Problem: Weighing the Earth}

\begin{problem}[{Weekend problem --- from two lead spheres in a shed to
the mass of the Earth and then of the Sun: what one small constant,
measured once, is worth}]
\label{pb:g10:universal-gravitation:1}
In 1798 Henry Cavendish suspended a light rod carrying two small lead
spheres from a thin wire, brought two big lead spheres close, and
measured the almost nonexistent twist of the wire. The newspapers said
he had \emph{weighed the Earth} --- and they were right: once $G$ is
known, the Earth's own $g$ and radius hand over its mass, and the
Earth's yearly \omterm{def:g10:universal-gravitation:satellite}{orbit} hands over the Sun's. This problem retraces the
whole chain. Use the data card throughout.

\medskip
\noindent\textbf{Part I --- The feeblest force.}
\begin{enumerate}
  \item The Eiffel Tower's iron has a mass of about \qty{7.3e6}{kg}. A
        \qty{55}{kg} tourist stands \qty{100}{m} from its center of
        mass. Compute the tower's pull on the tourist.
  \item Compare it with the \omterm{def:g10:universal-gravitation:weight}{weight} of a \qty{1.0}{mg} grain of sand.
  \item Proportionality drill: what happens to a \omterm{def:g10:universal-gravitation:force}{gravitational force}
        if (a) the distance is multiplied by $10$; (b) both masses are
        multiplied by $1000$; (c) both masses \emph{and} the distance
        are multiplied by $1000$?
  \item Compute the Earth's pull on a \qty{1.00}{kg} object at its
        surface, from $M$, $R$ and $G$. What everyday name and symbol
        does this number carry?
  \item Gravity is by far the feeblest interaction you have met ---
        and yet it runs the universe. Explain in one or two sentences
        why the Earth's pull on you dominates your neighbor's.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Cavendish's balance.}
In the torsion balance, a big sphere of mass $m_1 = \qty{158}{kg}$
attracts a small one of mass $m_2 = \qty{0.73}{kg}$; their centers are
$d = \qty{0.23}{m}$ apart.
\begin{enumerate}[resume]
  \item With the modern value of $G$, compute the force between the
        two spheres.
  \item Compute the \omterm{def:g10:universal-gravitation:weight}{weight} of the small sphere and the ratio of the
        two forces. Why did Cavendish need a torsion wire rather than
        any ordinary scale?
  \item Cavendish's measured twist corresponds (in modern \omterm{def:g10:orders-of-magnitude:unit}{units}) to
        $F = \qty{1.47e-7}{N}$. Deduce his value of $G$ and compare it
        with today's.
  \item With $G = \qty{6.67e-11}{N.m^2/kg^2}$, $g = \qty{9.81}{N/kg}$
        and $R = \qty{6.37e6}{m}$, deduce the mass of the Earth. (This
        is the step the newspapers celebrated.)
  \item Deduce the Earth's average density $\rho = M / V$, with
        $V = \frac43 \pi R^3$. Surface rocks have densities around
        \qty{2.7e3}{kg/m^3}: what does the comparison reveal about the
        deep interior?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Other worlds.}
\begin{enumerate}[resume]
  \item From the data card, recompute the Moon's surface field
        strength.
  \item On Earth, a good push-off lifts your center of mass by
        \qty{0.50}{m}. For the same push-off, the height reached is
        inversely proportional to $g$ (as the energy chapter,
        \cref{ch:g10:energy-conservation}, will justify). How high
        does the same jump carry you on the Moon?
  \item Mercury: $M = \qty{3.30e23}{kg}$, $R = \qty{2.44e6}{m}$.
        Compute its surface field strength and compare with Mars.
        Mercury is much smaller than Mars --- how can the two values
        be so close?
  \item At what altitude above the Earth's surface has the field
        strength dropped to half its surface value?
  \item At what distance from the Earth's center does the Earth's pull
        per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} fall to the Moon's surface value,
        \qty{1.62}{N/kg}? Express it in Earth radii.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Weighing the Sun.}
The Earth travels a near-circular \omterm{def:g10:universal-gravitation:satellite}{orbit} of radius
$d = \qty{1.496e11}{m}$ in one year, $T = \qty{3.156e7}{s}$. Take as
given (it is proved in \cref{ch:g12:satellites-kepler}) that a body on
a circular path of radius $d$ at speed $v$ needs a pull of $v^2/d$
newtons per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} toward the center.
\begin{enumerate}[resume]
  \item Compute the Earth's orbital speed $v$.
  \item Deduce the pull per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} that the Sun exerts on the Earth.
  \item That pull is also $G M_{\text{Sun}} / d^2$. Deduce the mass of
        the Sun.
  \item How many Earths is that? Compare $M_{\text{Sun}}$ with
        $M_{\text{Earth}}$.
  \item Finale. Cavendish's spheres, the Earth, the Sun: one constant
        $G$ served for all three. State in one sentence what the word
        \emph{universal} in ``universal \omterm{def:g10:universal-gravitation:force}{gravitation}'' bought us in
        this problem.
\end{enumerate}
\end{problem}
