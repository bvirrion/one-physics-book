\chapter{Orders of Magnitude: Measuring the Universe}\label{ch:g10:orders-of-magnitude}

Physics begins where opinion ends: with a measurement. But the things
physics measures span an absurd range --- an atomic nucleus is about
\qty{e-15}{m} across, the observable universe about \qty{e26}{m}: a
factor of $10^{41}$, far beyond anything a mental picture can hold.
This chapter installs the toolkit that tames the range --- units,
powers of ten, significant figures --- and two habits used on every
page after this one: write the unit, and check the order of magnitude.

\section{Units: the grammar of measurement}

\begin{definition}[Physical quantities and SI units]\label{def:g10:orders-of-magnitude:unit}
A \emph{physical quantity}\index{physical quantity} is a property of
the world that can be measured: a length, a duration, a mass.
Measuring it means counting how many times a reference quantity ---
the \emph{unit}\index{unit} --- fits into it. The
\emph{International System of Units}\index{International System of
Units (SI)} (SI) fixes seven base units, of which three run this whole
volume: the \emph{metre}\index{metre} (\unit{m}) for length, the
\emph{kilogram}\index{kilogram} (\unit{kg}) for mass and the
\emph{second}\index{second} (\unit{s}) for time. (The other four ---
ampere, kelvin, mole, candela --- will join when electricity, heat and
chemistry enter the story.)
\end{definition}

\begin{remark}[A number without a unit is not a result]\label{rem:g10:orders-of-magnitude:units}
``The distance is $384$'' means nothing: \qty{384}{mm},
\qty{384}{km} and \qty{384}{Mm} are three different worlds (the last
happens to be the distance to the Moon). In 1999, NASA lost the
\$327-million \emph{Mars Climate Orbiter} because one program computed
thruster forces in pounds-force while another read them as newtons.
House rule, from this line to the end of the book: every measured or
computed number carries its unit, on every line of the calculation.
\end{remark}

\begin{definition}[SI prefixes]\label{def:g10:orders-of-magnitude:prefixes}
A \emph{prefix}\index{prefix (SI)} glued to a unit multiplies it by a
fixed power of ten:
\begin{center}
\begin{tabular}{llc@{\qquad\qquad}llc}
femto & f & $10^{-15}$ & centi & c & $10^{-2}$ \\
pico  & p & $10^{-12}$ & kilo  & k & $10^{3}$  \\
nano  & n & $10^{-9}$  & mega  & M & $10^{6}$  \\
micro & $\mu$ & $10^{-6}$ & giga & G & $10^{9}$ \\
milli & m & $10^{-3}$  & tera  & T & $10^{12}$ \\
\end{tabular}
\end{center}
Thus $\qty{1}{nm} = \qty{e-9}{m}$ and $\qty{1}{km} = \qty{e3}{m}$.
\end{definition}

\begin{example}[Reading prefixes]\label{ex:g10:orders-of-magnitude:prefixes}
Green light has wavelength about $\qty{500}{nm} = \qty{5.00e-7}{m}$; a
human hair is about $\qty{70}{\micro m} = \qty{7.0e-5}{m}$ thick; the
Moon orbits at $\qty{384}{Mm} = \qty{3.84e8}{m}$. One raindrop has
mass about $\qty{30}{mg} = \qty{3.0e-5}{kg}$ --- note that the SI base
unit of mass, alone among the seven, carries a prefix already.
\end{example}

\section{Scientific notation and significant figures}

\begin{definition}[Scientific notation]\label{def:g10:orders-of-magnitude:scinot}
A number is in \emph{scientific notation}\index{scientific notation}
when written as
\[
a \times 10^{n}, \qquad 1 \leq a < 10, \quad n \text{ an integer.}
\]
The factor $a$ is the \emph{mantissa}, the integer $n$ the exponent.
Thus $\num{149600000} = \num{1.496e8}$ and
$\num{0.00052} = \num{5.2e-4}$.
\end{definition}

\begin{proposition}[Computing with powers of ten]\label{prop:g10:orders-of-magnitude:powers}
For all integers $m$ and $n$:
\[
10^{m} \times 10^{n} = 10^{m+n}, \qquad
\frac{1}{10^{n}} = 10^{-n}, \qquad
\left(10^{m}\right)^{n} = 10^{mn}.
\]
\end{proposition}

\begin{proof}
For positive exponents, count the factors: $10^{m} \times 10^{n}$ is a
product of $m + n$ tens, and $(10^m)^n$ chains $n$ groups of $m$ tens.
Writing $10^{-n}$ for $1/10^{n}$ is precisely the convention that
keeps the first rule true for all integers: it forces
$10^{n} \times 10^{-n} = 10^{0} = 1$. The mathematics volume of this
grade studies the exponent rules in full.
\end{proof}

\begin{example}[Multiplying in scientific notation]\label{ex:g10:orders-of-magnitude:scinot}
How many Earths fit in the Sun, by mass? With
$m_{\text{Sun}} = \qty{1.99e30}{kg}$ and
$m_{\text{Earth}} = \qty{5.97e24}{kg}$:
\[
\frac{\num{1.99e30}}{\num{5.97e24}}
= \frac{1.99}{5.97} \times 10^{30-24}
= 0.333 \times 10^{6}
= \num{3.33e5}.
\]
Mantissas and exponents are handled separately; the final answer is
re-normalized so that the mantissa lands back in $[1, 10)$.
\end{example}

\begin{definition}[Significant figures]\label{def:g10:orders-of-magnitude:sigfig}
The \emph{significant figures}\index{significant figures} of a
measured number are the digits actually vouched for by the
measurement: all written digits except the leading zeros. Trailing
zeros count --- $\qty{5.20e-3}{s}$ (three significant figures)
promises more than $\qty{5.2e-3}{s}$ (two). Scientific notation shows
the count at a glance: the mantissa carries exactly the significant
figures.
\end{definition}

\begin{method}[How many digits to keep]\label{met:g10:orders-of-magnitude:sigfig}
A computed result cannot be more precise than its ingredients.
\begin{enumerate}
  \item In a product or quotient, keep as many significant figures as
        the \emph{least} precise factor.
  \item Round only the final answer, never the intermediate steps.
  \item When in doubt, three significant figures is the working
        precision of this book.
\end{enumerate}
\end{method}

\begin{example}[Earth's circumference]\label{ex:g10:orders-of-magnitude:sigfig}
Earth's diameter is $d = \qty{1.27e7}{m}$ (three significant figures),
so its circumference is
$\pi d = \num{3.14159} \times \qty{1.27e7}{m} = \qty{3.99e7}{m}$ ---
three figures, not the eight the calculator displays. Writing
$\qty{39898227}{m}$ would claim to know the planet to the metre.
\end{example}

\section{The ladder of the universe}

\begin{definition}[Order of magnitude]\label{def:g10:orders-of-magnitude:oom}
The \emph{order of magnitude}\index{order of magnitude} of a quantity
is the power of ten closest to it: write the quantity as
$a \times 10^{n}$ with $1 \leq a < 10$ and take $10^{n}$ if $a < 5$,
$10^{n+1}$ if $a \geq 5$. (The exact boundary convention never
matters: an order-of-magnitude statement is only ever meant to within
a factor of ten.)
\end{definition}

\begin{example}[Orders of magnitude]\label{ex:g10:orders-of-magnitude:oom}
A human life lasts about $80$ years, i.e.\
$80 \times \qty{3.16e7}{s} \approx \qty{2.5e9}{s}$: order of magnitude
$\qty{e9}{s}$ --- a few billion seconds, spend them well. A human is
$\qty{1.7}{m}$ tall: order $\qty{e0}{m}$. Mount Everest,
$\qty{8848}{m}$: order $\qty{e4}{m}$, since $8.848 \geq 5$.
\end{example}

Stacking such estimates by exponent produces the master picture of
this chapter: a ladder whose every rung is a world ten times wider
than the one below.

Above the Earth rung, metres stop being convenient; astronomy keeps
its own units, both defined from the ladder itself.

\begin{definition}[Astronomical unit and light-year]\label{def:g10:orders-of-magnitude:lightyear}
The \emph{astronomical unit}\index{astronomical unit}
($\qty{1}{au} = \qty{1.496e11}{m}$) is the average Earth--Sun
distance. The \emph{light-year}\index{light-year} (\unit{ly}) is the
distance light travels in one year. Light moves in vacuum at
\[
c = \qty{3.00e8}{m/s},
\]
the fastest speed nature allows --- so the light-year is a distance,
not a time.
\end{definition}

\begin{example}[The light-year in metres]\label{ex:g10:orders-of-magnitude:ly}
A year holds $365.25 \times 24 \times 3600 = \qty{3.156e7}{s}$
(pleasantly close to $\pi \times 10^{7}\,\unit{s}$). Hence
\[
\qty{1}{ly} = c \times \qty{3.156e7}{s}
= \qty{3.00e8}{m/s} \times \qty{3.156e7}{s}
= \qty{9.47e15}{m}.
\]
Proxima Centauri, the nearest star after the Sun, lies
$\qty{4.2}{ly} \approx \qty{4.0e16}{m}$ away: light, which crosses an
Earth--Sun distance in eight minutes, needs over four years to arrive
from there.
\end{example}

\begin{method}[Sanity check by order of magnitude]\label{met:g10:orders-of-magnitude:sanity}
Before trusting any computed result:
\begin{enumerate}
  \item round every input to its order of magnitude;
  \item redo the computation with powers of ten alone, using
        \cref{prop:g10:orders-of-magnitude:powers} --- this takes
        seconds;
  \item compare: the precise answer must agree with the estimate to
        within a factor of ten or so, and its unit must be the right
        one. If either check fails, hunt down the error before moving
        on.
\end{enumerate}
\end{method}

\begin{remark}[Why bother]\label{rem:g10:orders-of-magnitude:sanity}
A calculator will happily report that an apple has mass
$\qty{4e8}{kg}$ --- it cannot know better. No formula flags the
absurd; the habit does. Physicists estimate first and compute second:
the estimate catches misplaced decimal points, swapped units and
inverted fractions, the three errors behind most wrong answers at
every level.
\end{remark}

\section{Measurement and uncertainty}

No instrument reads the true value. A measurement is honest only when
it also reports how far off it might be.

\begin{definition}[Absolute and relative uncertainty]\label{def:g10:orders-of-magnitude:uncertainty}
A measurement of a quantity $x$ is stated as
\[
x = x_{\text{m}} \pm \Delta x ,
\]
where $x_{\text{m}}$ is the measured value and the \emph{absolute
uncertainty}\index{uncertainty!absolute} $\Delta x > 0$, in the same
unit, bounds the plausible error: the true value is claimed to lie
between $x_{\text{m}} - \Delta x$ and $x_{\text{m}} + \Delta x$. The
\emph{relative uncertainty}\index{uncertainty!relative} is the ratio
$\Delta x / x_{\text{m}}$, usually quoted in percent; it is what the
word ``precise'' measures.
\end{definition}

\begin{method}[Reading a graduated instrument]\label{met:g10:orders-of-magnitude:reading}
For a ruler, a measuring cylinder, a dial:
\begin{enumerate}
  \item read the graduation nearest the mark, eye perpendicular to the
        scale (a slanted eye shifts the reading);
  \item take at least half the smallest graduation as the absolute
        uncertainty --- more if the object's edge or the liquid's
        surface is fuzzy;
  \item write the result as $x_{\text{m}} \pm \Delta x$ with its unit,
        and keep no digit beyond the uncertain one.
\end{enumerate}
\end{method}

\begin{example}[Relative precision]\label{ex:g10:orders-of-magnitude:reading}
The reading above has relative uncertainty
$\frac{0.5}{127} \approx 0.4\%$. The same ruler measuring a
$\qty{12}{mm}$ bolt gives $\frac{0.5}{12} \approx 4\%$: the identical
instrument is ten times less precise on the small object. Precision is
a property of the \emph{measurement}, not of the tool.
\end{example}

\begin{remark}[Small absolute, huge relative --- and conversely]\label{rem:g10:orders-of-magnitude:precision}
Laser stations measure the Earth--Moon distance, $\qty{3.84e8}{m}$, to
about $\pm\qty{3}{cm}$: a relative uncertainty of $10^{-10}$, among
the sharpest measurements ever made --- with an absolute error no
ruler would boast of. Always ask \emph{relative to what}: the pair
$(\Delta x,\ \Delta x / x_{\text{m}})$ tells the whole story, and
either number alone can mislead.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:1}
Convert to metres, in scientific notation:
\begin{enumerate}
  \item $\qty{750}{nm}$ (wavelength of red light);
  \item $\qty{2.5}{km}$;
  \item $\qty{3.2}{\micro m}$ (a soil bacterium);
  \item $\qty{0.15}{Tm}$ (the Earth--Sun distance, in an unusual
        costume).
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:2}
Write in scientific notation and give the number of significant
figures:
\begin{enumerate}
  \item the Earth--Sun distance, $\qty{149600000}{km}$, in metres;
  \item the diameter of a hydrogen atom, $\qty{0.000000000106}{m}$;
  \item the world population, about $\num{8100000000}$;
  \item a duration of $\qty{0.00520}{s}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:3}
Compute without a calculator, giving each result in scientific
notation:
\[
(3\times10^{8})\times(2\times10^{-5}), \qquad
\frac{8\times10^{4}}{4\times10^{-3}}, \qquad
(5\times10^{6})^{2}, \qquad
\frac{(4\times10^{-3})\times(3\times10^{-9})}{6\times10^{-7}} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:4}
Give the order of magnitude, with its unit, of: a human height of
$\qty{1.7}{m}$; Mount Everest, $\qty{8848}{m}$; an \emph{E.~coli}
bacterium, $\qty{2e-6}{m}$; the mass of the Earth,
$\qty{5.97e24}{kg}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:orders-of-magnitude:5}
A ruler is graduated in millimetres. You measure a pencil at
$\qty{127}{mm}$.
\begin{enumerate}
  \item State the result with its absolute uncertainty.
  \item Compute the relative uncertainty.
  \item The same ruler measures a $\qty{12}{mm}$ eraser. Which of the
        two measurements is the more precise, and by what factor?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:6}
Using $c = \qty{3.00e8}{m/s}$ and $\qty{1}{ly} = \qty{9.47e15}{m}$:
\begin{enumerate}
  \item Express in metres the distances to Proxima Centauri
        ($\qty{4.24}{ly}$) and to Sirius ($\qty{8.6}{ly}$).
  \item The probe \emph{Voyager 1} is about $\qty{2.4e13}{m}$ from
        Earth. How long does its radio signal, travelling at $c$, take
        to reach us? Express the answer in hours.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:7}
How long does light take to reach Earth from the Moon
($\qty{3.84e8}{m}$), from the Sun ($\qty{1.496e11}{m}$) and from
Neptune ($\qty{4.50e12}{m}$)? Give each answer in a sensible unit, and
complete the sentence: ``When you look at Neptune, you see it as it
was\dots''
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:8}
A ream of $500$ sheets of paper, measured with a millimetre ruler, is
$\qty{52.0}{mm}$ thick, with uncertainty $\pm\qty{0.5}{mm}$.
\begin{enumerate}
  \item Deduce the thickness of one sheet, with its absolute
        uncertainty.
  \item What is the relative uncertainty of that thickness?
  \item Why is measuring the ream enormously better than measuring one
        sheet directly with the same ruler? Quantify.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:9}
Take $\qty{2e-10}{m}$ as the diameter of an atom. How many atoms, side
by side, span: a bacterium of $\qty{2.0}{\micro m}$; a hair of
$\qty{70}{\micro m}$; a human of $\qty{1.7}{m}$? Give the last answer
as an order of magnitude too.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:10}
A scale model represents the Sun (diameter $\qty{1.39e9}{m}$) by a
football of diameter $\qty{22}{cm}$.
\begin{enumerate}
  \item Compute the scale factor of the model.
  \item Find the model diameter of the Earth ($\qty{1.27e7}{m}$) and
        its model distance from the football (real distance
        $\qty{1.496e11}{m}$).
  \item How far from the model Earth is the model Moon (real distance
        $\qty{3.84e8}{m}$)? And how far away is model Neptune (real
        distance $\qty{4.50e12}{m}$)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:orders-of-magnitude:11}
Significant figures at work.
\begin{enumerate}
  \item An A4 sheet measures $\qty{21.0}{cm}$ by $\qty{29.7}{cm}$.
        Give its area with the correct number of significant figures.
  \item A sprinter covers $\qty{100.0}{m}$ in $\qty{12.3}{s}$. Give
        the average speed correctly rounded.
  \item Explain in one sentence what would be dishonest about
        announcing the area as $\qty{623.70}{cm^2}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:orders-of-magnitude:12}
A drop of oil of volume $\qty{1.0}{mm^3}$ is deposited on still water
and spreads into a circular film of diameter $\qty{1.1}{m}$ --- a film
that later chapters will show to be one molecule thick.
\begin{enumerate}
  \item Compute the area of the film, then its thickness.
  \item Deduce an order of magnitude for the size of a molecule, and
        check it against the atom rung of the ladder.
\end{enumerate}
(This back-of-the-envelope experiment, first tried by Benjamin
Franklin on a London pond, was one of humanity's first honest measures
of the molecular world.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:orders-of-magnitude:13}
Estimate the number of heartbeats in a human lifetime. State your
assumptions (resting heart rate, life span), keep the arithmetic to
powers of ten and one digit of mantissa, and give the final answer as
an order of magnitude only. Why would quoting it to four significant
figures be meaningless?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:orders-of-magnitude:14}
A sheet of paper is $\qty{0.10}{mm}$ thick, and each fold doubles the
thickness of the stack. Using $2^{10} \approx 10^{3}$:
\begin{enumerate}
  \item After how many folds would the stack overtop Mount Everest
        ($\qty{8848}{m}$)?
  \item After how many folds would it reach the Moon
        ($\qty{3.84e8}{m}$)?
  \item In practice a sheet cannot be folded more than about seven
        times. Does that invalidate the arithmetic, or only the
        experiment?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:orders-of-magnitude:15}
Laser ranging gives the Earth--Moon distance as
$\qty{3.84e8}{m} \pm \qty{3}{cm}$; a good ruler gives a table as
$\qty{1.000}{m} \pm \qty{1}{mm}$.
\begin{enumerate}
  \item Compute both relative uncertainties. Which measurement is the
        more precise, and by what factor?
  \item With what absolute uncertainty would you need to measure the
        table to match the Moon measurement's relative precision?
        Compare that length with the size of an atom, and conclude
        with due respect.
\end{enumerate}
\end{exercise}

\section{Problem: From the atom to Andromeda}

\begin{problem}[{Weekend problem --- a single grain of sand sizes the
atom, the Solar System, the galaxy and the observable universe, and
locates you on the ladder between them}]
\label{pb:g10:orders-of-magnitude:1}
A grain of sand is the humblest object in physics, and this weekend it
works overtime: first we count its atoms, then we shrink the Sun to
its size and pace out the cosmos, then we let light do the surveying,
and finally we count the atoms in \emph{you}. Data, used throughout:
grain diameter $\qty{0.5}{mm}$; atom diameter $\qty{2.5e-10}{m}$;
density of sand $\qty{2500}{kg/m^3}$; diameters: Sun
$\qty{1.39e9}{m}$, Earth $\qty{1.27e7}{m}$, Milky Way
$\qty{9.5e20}{m}$, observable universe $\qty{8.8e26}{m}$; distances
from us: Moon $\qty{3.84e8}{m}$, Sun $\qty{1.496e11}{m}$, Neptune
$\qty{4.5e12}{m}$, Proxima Centauri $\qty{4.0e16}{m}$, Andromeda
galaxy $\qty{2.4e22}{m}$; $c = \qty{3.00e8}{m/s}$; a water molecule
has mass $\qty{3.0e-26}{kg}$ and contains three atoms.

\medskip
\noindent\textbf{Part I --- Anatomy of the grain.}
\begin{enumerate}
  \item Write the grain diameter and the atom diameter in metres, in
        scientific notation, and name the SI prefix each is closest
        to.
  \item How many atoms sit side by side across one grain diameter?
  \item Model the grain as a cube of side $\qty{0.5}{mm}$ packed with
        atoms. Estimate the number of atoms in the grain, as an order
        of magnitude.
  \item Compute the volume of that cube, then the mass of the grain.
        Express the mass in milligrams.
  \item Sanity-check the model: divide the grain's mass by its number
        of atoms and compare with the known order of magnitude of
        atomic masses (a few $10^{-26}\,\unit{kg}$). Verdict?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Sun as a grain of sand.}
We build a scale model of the cosmos in which the Sun is the grain:
every model size equals the real size times a factor $s$.
\begin{enumerate}[resume]
  \item Compute the scale factor $s$.
  \item In the model, find the Earth's diameter and its distance from
        the grain-Sun. What object of Part I's world has roughly the
        model Earth's size?
  \item How far from the grain does model Neptune orbit? What piece of
        furniture would hold the entire planetary system?
  \item How far away is the model Proxima Centauri?
  \item Two grains of sand, $\qty{14}{km}$ apart, and almost nothing
        in between: state in one or two sentences what the model
        reveals about how empty the galaxy is.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Light does the surveying.}
\begin{enumerate}[resume]
  \item How long does light take to cross the real grain of sand?
  \item How long does light take to reach us from the Moon, and from
        the Sun?
  \item Recompute the light-year in metres from $c$ and the length of
        the year, and verify that the Proxima distance in the data
        corresponds to about $\qty{4.2}{ly}$.
  \item Convert the Andromeda distance to light-years. Roughly when
        did the light now entering your eye leave Andromeda, and what
        was walking the Earth at the time?
  \item How far does light travel in one nanosecond? A processor
        clocked at $\qty{3}{GHz}$ starts an operation every third of a
        nanosecond: explain in one sentence why a fast computer must
        be a small computer.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- You, between atom and Andromeda.}
\begin{enumerate}[resume]
  \item A human is roughly $\qty{70}{kg}$ of water. Using the data,
        estimate the average mass of one atom in water, then the
        number of atoms in a human.
  \item Compare that count with the atoms in the grain (question 3)
        and with the roughly $2\times10^{23}$ stars of the observable
        universe. Give both factors.
  \item On the ladder, are you nearer the atom or the observable
        universe? Compare the ratios $\text{(you)}/\text{(atom)}$ and
        $\text{(universe)}/\text{(you)}$, taking $\qty{1.7}{m}$ for
        you.
  \item Find the size $x$ sitting exactly halfway up the ladder
        between the nucleus ($\qty{e-15}{m}$) and the observable
        universe: $x$ must satisfy $x / 10^{-15} = 10^{26.9} / x$
        (using $\qty{8.8e26}{m} \approx 10^{26.9}\,\unit{m}$). What
        Solar-System object happens to have just that size?
  \item The punchline. Shrink the observable universe by the Part II
        factor $s$, so that the Sun is a grain of sand. Give the model
        universe's diameter in metres, then in astronomical units, and
        conclude in one sentence: even with every star turned to a
        sand grain, the universe overflows every human scale --- only
        powers of ten hold it.
\end{enumerate}
\end{problem}
