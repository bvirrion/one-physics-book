\chapter{Kinematics in Two Dimensions}\label{ch:g12:kinematics-2d}

Watch a car take a roundabout at a steady \qty{30}{km/h}: the needle
never moves, yet every passenger feels pulled sideways --- the motion
is changing, and it is not the speed. One number per instant no
longer suffices: motion in the plane needs arrows. This chapter
equips a moving point with three --- position, velocity, acceleration
--- chained by the derivative, and reads whole families of motion off
their geometry.

\section{The position vector}

Motion only exists relative to a stated reference frame --- a rigid
body plus a clock (\cref{ch:g10:relative-motion}). New this year is
precision: attach to the frame an origin $O$ and two perpendicular
graduated axes $Ox$, $Oy$, and record where the point is at every
instant, as a pair of functions.

\begin{definition}[Position vector]\label{def:g12:kinematics-2d:position}
In a frame equipped with an origin $O$ and axes $Ox$, $Oy$, the
\emph{position vector}\index{position vector} of a moving point $M$
at time $t$ is $\vect{OM}(t)$, of coordinates
$\bigl(x(t),\, y(t)\bigr)$: two functions of time, one per axis. The
curve traced by $M$ as $t$ runs is its trajectory.
\end{definition}

\section{The velocity vector}

\begin{definition}[Velocity vector]\label{def:g12:kinematics-2d:velocity}
Over a short interval the point moves by
$\vect{OM}(t+\Delta t) - \vect{OM}(t)$; divide by $\Delta t$ and let
the interval shrink: this is the derivative from your mathematics
course, applied to each coordinate. The \emph{velocity
vector}\index{velocity vector} of $M$ is the derivative of its
position vector, $\vect v(t) = \bigl(x'(t),\, y'(t)\bigr)$, in
\unit{m/s}. Its norm $v = \sqrt{x'^2 + y'^2}$ is the
\emph{speed}\index{speed} --- the one number the speedometer shows.
\end{definition}

\begin{example}[A drone in level flight]\label{ex:g12:kinematics-2d:drone}
A drone flies with $x(t) = 4.0\,t$ and $y(t) = 3.0\,t$ (meters,
seconds): trajectory the line $y = \tfrac34 x$, velocity
$\vect v = (4.0, 3.0)$ at all times, speed
$\sqrt{4.0^2 + 3.0^2} = \qty{5.0}{m/s}$, constant.
\end{example}

\begin{proposition}[Tangent to the trajectory]\label{prop:g12:kinematics-2d:tangent}
At every instant, $\vect v(t)$ is tangent to the trajectory at the
current position of $M$, and points in the direction of travel.
\end{proposition}

\begin{proof}
The displacement is a chord of the trajectory; as $\Delta t$ shrinks,
the chord's direction becomes the tangent's.
\end{proof}

\section{The acceleration vector}

\begin{definition}[Acceleration vector]\label{def:g12:kinematics-2d:acceleration}
The \emph{acceleration vector}\index{acceleration vector} of $M$ is
the derivative of its velocity vector,
$\vect a(t) = \bigl(v_x'(t),\, v_y'(t)\bigr) =
\bigl(x''(t),\, y''(t)\bigr)$, in \unit{m/s^2}: how fast, and in
which direction, the velocity vector is currently changing.
\end{definition}

\begin{remark}[Accelerating without speeding up]\label{rem:g12:kinematics-2d:turning}
$\vect a \neq \vect 0$ whenever the velocity \emph{vector} changes
--- in length \emph{or in direction}: a braking car accelerates
(backward); a turning car at constant speed accelerates (sideways).
This is the quantity forces control
(\cref{ch:g11:forces-and-motion,ch:g12:newtons-laws}).
\end{remark}

\begin{method}[Reading a chronophotograph]\label{met:g12:kinematics-2d:chrono}
A \emph{chronophotograph}\index{chronophotograph} records the
positions $M_0, M_1, M_2, \dots$ of a moving point at equal time
intervals $\Delta t$.
\begin{enumerate}
  \item Dot spacing reads speed: equal spacings, constant speed;
        opening, speeding up; closing, slowing down.
  \item The velocity at $M_i$ is the symmetric chord,
        $\vect v_i \approx \vect{M_{i-1}M_{i+1}}/(2\,\Delta t)$,
        drawn at $M_i$, tangent to the dotted path.
  \item For the acceleration at $M_i$: copy $\vect v_{i-1}$ and
        $\vect v_{i+1}$ to $M_i$, draw
        $\Delta\vect v = \vect v_{i+1} - \vect v_{i-1}$ tip to tip;
        then $\vect a_i \approx \Delta\vect v/(2\,\Delta t)$.
\end{enumerate}
\end{method}

\section{Rectilinear motions}

Along a straight line one axis suffices: $x(t)$, $v(t) = x'(t)$,
$a(t) = v'(t)$, all signed numbers.

\begin{definition}[Uniform rectilinear motion]\label{def:g12:kinematics-2d:uniform}
A motion is \emph{uniform rectilinear}\index{uniform rectilinear
motion} when $\vect v$ is a constant vector: straight trajectory,
constant speed, $\vect a = \vect 0$. Then $x(t) = v\,t + x_0$.
\end{definition}

\begin{proposition}[Uniformly accelerated rectilinear motion]\label{prop:g12:kinematics-2d:uarm}
\index{uniformly accelerated motion}%
If a point moves along $Ox$ with \emph{constant} acceleration $a$,
starting at $t = 0$ from position $x_0$ with velocity $v_0$, then
\[
v(t) = a\,t + v_0,
\qquad
x(t) = \tfrac12\,a\,t^2 + v_0\,t + x_0 .
\]
\end{proposition}

\begin{proof}
$v$ is an antiderivative of the constant $a$: $v(t) = a t + C$, and
$t = 0$ forces $C = v_0$. In turn $x$ is an antiderivative of
$a t + v_0$: $x(t) = \tfrac12 a t^2 + v_0 t + C'$, with $C' = x_0$.
\end{proof}

\begin{example}[Braking distance]\label{ex:g12:kinematics-2d:braking}
A car at speed $v_0$ brakes with constant deceleration $a$
(acceleration $-a$). It stops when $v(t) = -a t + v_0 = 0$, at
$t_s = v_0/a$, having covered $d = -\tfrac12 a t_s^2 + v_0 t_s =
v_0^2/(2a)$. The square is the road-safety headline: doubling the
speed \emph{quadruples} the braking distance --- with $a =
\qty{6.0}{m/s^2}$, \qty{16}{m} from \qty{50}{km/h}, \qty{64}{m} from
\qty{100}{km/h}.
\end{example}

\section{Uniform circular motion and the Frenet frame}

\begin{definition}[Uniform circular motion]\label{def:g12:kinematics-2d:ucm}
A motion is \emph{uniform circular}\index{uniform circular motion}
when the trajectory is a circle of radius $R$ and the speed $v$ is
constant. The \emph{period}\index{period!of rotation} $T$ is the
duration of one lap, and the \emph{frequency}\index{frequency!of
rotation} $f = 1/T$ (in \unit{Hz}) the number of laps per second:
$v = 2\pi R/T = 2\pi R f$.
\end{definition}

Constant speed --- yet the velocity turns: there \emph{is} an
acceleration. Which way, and how big?

\begin{theorem}[Centripetal acceleration]\label{thm:g12:kinematics-2d:centripetal}
\index{centripetal acceleration}%
In uniform circular motion of radius $R$ and speed $v$, the
acceleration --- called \emph{centripetal} --- points from the moving
point toward the center of the circle, with constant norm
\[
a = \frac{v^2}{R}.
\]
\end{theorem}

\begin{proof}
Center the axes at the circle's center and set $\omega = v/R$, so the
arc $v t$ subtends the angle $\omega t$: the position is
$x(t) = R\cos(\omega t)$, $y(t) = R\sin(\omega t)$. Differentiating,
$\vect v = (-R\omega\sin(\omega t),\, R\omega\cos(\omega t))$, of
constant norm $R\omega = v$; differentiating again,
$\vect a = -\omega^2\,\vect{OM}$: opposite to the position vector ---
toward the center --- of norm $\omega^2 R = v^2/R$.
\end{proof}

\begin{definition}[Frenet frame]\label{def:g12:kinematics-2d:frenet}
The \emph{Frenet frame}\index{Frenet frame} at the current position
of a point on a curved path is the pair of perpendicular unit
vectors $(\vect{u_t}, \vect{u_n})$: $\vect{u_t}$ tangent, along the
motion; $\vect{u_n}$ normal, toward the center of the circle of
radius $R$ that best fits the curve there.
\end{definition}

\begin{proposition}[Acceleration in the Frenet frame]\label{prop:g12:kinematics-2d:frenetacc}
Along any plane path of local radius $R$, traveled at speed $v(t)$,
\[
\vect a = a_t\,\vect{u_t} + a_n\,\vect{u_n},
\qquad a_t = \frac{\dd v}{\dd t},
\qquad a_n = \frac{v^2}{R}:
\]
the \emph{tangential acceleration}\index{tangential acceleration}
$a_t$ changes the speed; the \emph{normal
acceleration}\index{normal acceleration} $a_n$ turns the velocity.
\end{proposition}
\admitted

\begin{remark}[Where the proof lives]\label{rem:g12:kinematics-2d:frenetproof}
We proved the extremes: pure $a_t$ on a line
(\cref{prop:g12:kinematics-2d:uarm}), pure $a_n$ on a circle
(\cref{thm:g12:kinematics-2d:centripetal}); the general
decomposition is derived in the Year~1 volume.
\end{remark}

\begin{example}[Braking into a bend]\label{ex:g12:kinematics-2d:bend}
A car enters a bend of radius $R = \qty{50}{m}$ at $v = \qty{15}{m/s}$
while braking at $\dd v/\dd t = \qty{-3.0}{m/s^2}$: $a_t =
\qty{-3.0}{m/s^2}$, $a_n = 15^2/50 = \qty{4.5}{m/s^2}$, so $a =
\sqrt{3.0^2 + 4.5^2} \approx \qty{5.4}{m/s^2}$, pointing backward
\emph{and} into the bend. The tires must supply both at once ---
why racing drivers brake \emph{before} the turn.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:1}
For $x(t) = 3.0\,t$, $y(t) = 4.0\,t$ (meters, seconds), give
$\vect v$, the speed, the trajectory and the class of motion.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:2}
On a chronophotograph taken every \qty{0.10}{s}, the dots lie on a
line, evenly spaced \qty{0.35}{m} apart. Identify the motion and
compute the speed. What would opening spacings indicate?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:3}
For $x(t) = 2.0\,t^2$ along $Ox$ (meters, seconds), derive $v(t)$
and $a(t)$, name the class of motion, and compute the speed at
$t = \qty{3.0}{s}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:4}
A carousel horse rides a circle of radius \qty{4.5}{m}, one lap every
\qty{8.0}{s}: compute the frequency, the speed and the acceleration
(norm and direction).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:5}
A $v(t)$ graph climbs straight from \qty{2.0}{m/s} at $t = 0$ to
\qty{10.0}{m/s} at $t = \qty{4.0}{s}$. Read off the acceleration,
then the distance covered (area under the graph).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:6}
A high-speed train takes a curve of radius \qty{6.0}{km} at
\qty{320}{km/h}. Compute $a_n$ and compare it to $g$; why must
high-speed lines avoid tight curves, even perfectly banked ones?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:7}
A trolley on a straight rail has $x(t) = -1.2\,t^2 + 8.0\,t + 3.0$
(meters, seconds). Find $v(t)$ and $a$, the instant and position of
reversal, and describe the motion before and after.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:8}
A chronophotograph along a line, every $\Delta t = \qty{0.20}{s}$,
gives $x = 0$, $0.10$, $0.28$, $0.54$, $\qty{0.88}{m}$. Estimate the
velocity at the three interior dots
(\cref{met:g12:kinematics-2d:chrono}), then the acceleration; conclude.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:9}
The Moon orbits the Earth on a near circle of radius \qty{3.84e8}{m}
in \qty{27.3}{d}. Compute its speed and its centripetal acceleration;
compare the latter to $g$, knowing the Moon is about $60$ Earth radii
away (\cref{ch:g10:universal-gravitation}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:10}
A washing-machine drum of radius \qty{25}{cm} spins at $1200$
revolutions per minute. Compute $f$, $T$, the rim speed and the rim

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:11}
A car at \qty{90}{km/h} brakes with constant $a = \qty{7.5}{m/s^2}$:
compute the stopping time and the braking distance, and sketch $v(t)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:12}
For $x(t) = R\cos(\omega t)$, $y(t) = R\sin(\omega t)$: compute
$\vect v$; check the speed is the constant $R\omega$ and
$\vect v \perp \vect{OM}$; compute $\vect a$, show it is

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:13}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:14}
A metro accelerates from rest at \qty{1.2}{m/s^2} up to \qty{20}{m/s},
cruises for \qty{40}{s}, then brakes at \qty{1.5}{m/s^2} to a stop.

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:15}
A geostationary satellite circles at radius \qty{4.22e7}{m} from the

\end{exercise}

\section{Problem: The Accident Report}

\begin{problem}[{Weekend problem --- the accident report: a dashcam
film read frame by frame, a braking law rebuilt from skid marks, a
roundabout that testifies about grip, and a verdict in kilometers
per hour}]
\label{pb:g12:kinematics-2d:1}
A car strikes a van that pulled out of a side street; nobody is hurt,
but the insurers disagree. The investigator has overhead camera
footage (one frame every \qty{0.50}{s}), measured skid marks, and the
car's dashcam. The speed limit is \qty{50}{km/h}.

\medskip
\noindent\textbf{Part I --- Reading the film.}
The frames before any reaction give, along the straight road:
\begin{center}\small
\begin{tabular}{lccccc}
$t$ (\unit{s}) & 0.00 & 0.50 & 1.00 & 1.50 & 2.00\\
$x$ (\unit{m}) & 0.0 & 9.0 & 18.0 & 27.0 & 36.0
\end{tabular}
\end{center}
\begin{enumerate}
  \item Why must the report first name a frame and axes? Which are
        used here?
  \item Compute the average velocity on each of the four intervals.
  \item What class of motion is this? Justify from the table.
  \item Give the car's speed $v_0$ in \unit{m/s} and \unit{km/h}.
  \item What is the acceleration during these two seconds?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The skid marks.}
Tire tests on this dry asphalt give a constant braking deceleration
$a = \qty{6.0}{m/s^2}$; the skid marks are $d_b = \qty{27}{m}$ long.
Take $t = 0$, $x = 0$ at the start of braking, unknown speed $v_B$.
\begin{enumerate}[resume]
  \item By antiderivatives, write $v(t)$ and $x(t)$ during braking.
  \item Express the stopping time $t_s$; show $d_b = v_B^2/(2a)$.
  \item Deduce $v_B$ from the skid marks. Consistent with Part~I?
  \item The reaction time is \qty{1.0}{s}: compute the reaction
        distance and the total stopping distance.
  \item Explain, from question 7, why doubling the speed quadruples
        the braking distance; compute $d_b$ at exactly \qty{50}{km/h}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The roundabout testifies.}
Earlier, the dashcam shows the car rounding a roundabout lane of
radius $R = \qty{12.5}{m}$ at constant speed, a quarter turn in
\qty{2.5}{s}. The road can supply at most $a_n = \mu g$, with
$\mu = 0.70$.
\begin{enumerate}[resume]
  \item The speed is constant there --- is the velocity? Is the car
        accelerating? Explain.
  \item Compute the car's speed on the roundabout.
  \item Compute its centripetal acceleration (norm and direction).
  \item Compute the maximum no-skid speed. Was the car within it?
  \item Give the period and frequency of a full loop at that speed.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The verdict.}
The van appeared $D = \qty{41}{m}$ ahead of the car at the instant
the driver started reacting.
\begin{enumerate}[resume]
  \item By how much, in \unit{km/h} and in percent, did the speed
        before braking exceed the limit?
  \item Compute the car's speed at impact.
  \item Redo the stopping computation at exactly \qty{50}{km/h} (same
        reaction, same braking): does the car stop, with what margin?
  \item Give the general stopping-distance formula $d(v)$; why does
        speed enter twice --- once linearly, once squared?
  \item The verdict, one sentence with its numbers: what speed was the
        driver doing, and would the accident have happened at the limit?
\end{enumerate}
\end{problem}
