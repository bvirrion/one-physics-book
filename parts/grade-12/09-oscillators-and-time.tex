\chapter{Mechanical Oscillators and the Measurement of Time}\label{ch:g12:oscillators-and-time}

A grandfather clock ticks: each tick is a brass pendulum crossing the
vertical, repeated fifteen million times since New Year. A quartz
sliver in your watch flexes \num{32768} times a second; a cesium \omterm{def:g11:fundamental-interactions:ladder}{atom}
hums nine billion times faster. Everything that swings can keep time:
this chapter follows the swinging, from Huygens to the \omterm{def:g10:universal-gravitation:satellite}{satellites}.

\section{Free oscillations}

\begin{definition}[Oscillator, period, frequency, amplitude]\label{def:g12:oscillators-and-time:oscillator}
A \emph{mechanical oscillator}\index{oscillator} is a system that,
displaced from a stable \emph{equilibrium position}\index{equilibrium
position} and released, swings back and forth around it: these are
\emph{free oscillations}\index{free oscillations}. The motion repeats
after the \emph{period}\index{period} $T_0$ (\unit{s}); the
\emph{frequency}\index{frequency} $f_0 = 1/T_0$
(\unit{Hz}, \cref{ch:g12:mechanical-waves}); the
\emph{amplitude}\index{amplitude} is the maximum displacement.
\end{definition}

\begin{example}[A zoo of oscillators]\label{ex:g12:oscillators-and-time:zoo}
A playground swing: $T_0 \approx \qty{3}{s}$. A guitar's A string:
$f_0 = \qty{110}{Hz}$. A quartz crystal: \qty{32768}{Hz}. The cesium
oscillation defining the second: about \qty{9.2e9}{Hz}. Ten orders
of magnitude, one job: counting.
\end{example}

\section{The simple pendulum}

\begin{definition}[Simple pendulum]\label{def:g12:oscillators-and-time:pendulum}
A \emph{simple pendulum}\index{simple pendulum}: a small bob of mass
$m$ on an inextensible wire of length $L$ and negligible mass,
swinging in a vertical plane. Of the two \omterm{def:g10:inertia:force}{forces} on the bob
(\cref{ch:g12:newtons-laws}), the \omterm{def:g10:inertia:inventory}{tension} is perpendicular to the
motion, while the \omterm{def:g10:universal-gravitation:weight}{weight}'s component along the \omterm{def:g10:relative-motion:trajectory}{trajectory} points back
toward the lowest point: the \omterm{def:g10:universal-gravitation:weight}{weight} is the \emph{restoring \omterm{def:g10:inertia:force}{force}}%
\index{restoring force}, opposing the displacement.
\end{definition}

\begin{proposition}[Period of the simple pendulum]\label{prop:g12:oscillators-and-time:pendulum-period}
\index{simple pendulum!period}
For small \omterm{def:g12:oscillators-and-time:oscillator}{amplitudes} (below about $15^\circ$), the \omterm{def:g12:oscillators-and-time:oscillator}{period} is
\[
T_0 = 2\pi \sqrt{\frac{L}{g}}, \qquad g = \qty{9.81}{m/s^2}.
\]
It depends neither on the mass of the bob nor --- while the angle
stays small --- on the \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}: small swings are
\emph{isochronous}\index{isochronism}, all taking the same time.
\end{proposition}
\admitted

\begin{remark}[Dimensional analysis, and where the $2\pi$ lives]\label{rem:g12:oscillators-and-time:dimensional}
The formula can be guessed. From $m$ (\unit{kg}), $L$ (\unit{m}) and
$g$ (\unit{m/s^2}), the only combination with the \omterm{def:g10:orders-of-magnitude:unit}{unit} of a time is
$\sqrt{L/g}$ --- and the mass \emph{cannot} enter, no other datum
carrying a \omterm{def:g10:orders-of-magnitude:unit}{kilogram} to cancel it. Dimensions cannot give the pure
number in front: the $2\pi$ comes from solving the equation of motion,
done in the Year 1 volume. Nor is the small-angle condition optional:
at $30^\circ$ the true \omterm{def:g12:oscillators-and-time:oscillator}{period} is $2\%$ longer, and isochronism fails.
\end{remark}

\begin{example}[The seconds pendulum]\label{ex:g12:oscillators-and-time:seconds}
Which length beats the second, $T_0 = \qty{2.00}{s}$, one tick each
way? Solving, $L = g\,T_0^2/4\pi^2 = 9.81 \times 2.00^2/4\pi^2 \approx
\qty{0.994}{m}$. Nearly a meter: pendulum clocks are tall because
seconds are long.
\end{example}

\section{The spring--mass oscillator}

\begin{definition}[Restoring force of a spring]\label{def:g12:oscillators-and-time:spring}
A glider of mass $m$ slides on a horizontal frictionless rail, tied to
a spring of \emph{stiffness}\index{stiffness} $k$ (\unit{N/m}). Let
$x$ be its displacement from equilibrium. Stretched or compressed, the
spring pulls or pushes the glider back with the \omterm{def:g10:inertia:force}{force} $F = -kx$:
proportional to the displacement and opposed to it --- a \omterm{def:g12:oscillators-and-time:pendulum}{restoring
force} again, now supplied by elasticity.
\end{definition}

\begin{proposition}[Equation of motion and its solution]\label{prop:g12:oscillators-and-time:spring-period}
\index{spring--mass oscillator}
\omterm{thm:g12:newtons-laws:second}{Newton's second law} along the rail (\cref{ch:g12:newtons-laws}) gives
$m \frac{d^2x}{dt^2} = -kx$, i.e.\ $\frac{d^2x}{dt^2} =
-\frac{k}{m}\,x$. Whatever the constants $A$ and $\varphi$,
\[
x(t) = A \cos\!\left(\frac{2\pi t}{T_0} + \varphi\right),
\qquad T_0 = 2\pi\sqrt{\frac{m}{k}},
\]
satisfies this equation: oscillation with \omterm{def:g12:oscillators-and-time:oscillator}{period} $T_0$, independent
of the \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}.
\end{proposition}

\begin{proof}
Differentiating, $\frac{dx}{dt} = -\frac{2\pi}{T_0}A \sin(\frac{2\pi
t}{T_0} + \varphi)$; again, $\frac{d^2x}{dt^2} = -(\frac{2\pi}{T_0})^2
A\cos(\frac{2\pi t}{T_0} + \varphi) = -(\frac{2\pi}{T_0})^2\,x(t)$,
which is $-(k/m)\,x$ exactly when $(2\pi/T_0)^2 = k/m$.
\end{proof}

\begin{remark}[All the motions there are]\label{rem:g12:oscillators-and-time:uniqueness}
That \emph{every} motion of the glider is one of these functions is
admitted, proved in the Year 1 volume. And no $g$ in $2\pi\sqrt{m/k}$:
stiffer swings faster, heavier slower, the same on the Moon.
\end{remark}

\begin{definition}[Amplitude and phase]\label{def:g12:oscillators-and-time:phase}
In $x(t) = A\cos(2\pi t/T_0 + \varphi)$ the positive constant $A$ is
the \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} --- the extremes are $x = \pm A$ --- and $2\pi t/T_0 +
\varphi$ is the \emph{phase}\index{phase}, growing by $2\pi$ each
\omterm{def:g12:oscillators-and-time:oscillator}{period}; the \emph{initial phase}\index{initial phase} $\varphi$
records the start: released at rest from $x = A$, $\varphi = 0$. For
instance $m = \qty{0.20}{kg}$ on $k = \qty{20}{N/m}$ gives $T_0 =
2\pi\sqrt{0.20/20} = \qty{0.63}{s}$; pulled to \qty{5.0}{cm} and
released, $x(t) = 5.0\cos(2\pi t/0.63)$ in centimeters.
\end{definition}

\section{Energy, damping, resonance}

\begin{proposition}[Energy of the oscillator]\label{prop:g12:oscillators-and-time:energy}
A spring stretched by $x$ stores the elastic potential \omterm{def:g10:energy-conservation:energy}{energy} $E_p =
\tfrac12 kx^2$ (justified in \cref{ch:g12:work-and-energy}). During
frictionless oscillation the \omterm{def:g11:mechanical-energy:em}{mechanical energy}
\[
E_m = E_k + E_p = \tfrac12 m v^2 + \tfrac12 k x^2 = \tfrac12 k A^2
\]
is constant: all elastic at the extremes, all kinetic at equilibrium.
The pendulum plays the same duet between $E_k$ and gravitational
$E_p$ (\cref{ch:g11:mechanical-energy}).
\end{proposition}

\begin{proof}
With $v = \frac{dx}{dt}$, the derivative of $E_m$ is $mv\frac{dv}{dt}
+ kx\frac{dx}{dt} = v\,(m\frac{d^2x}{dt^2} + kx) = 0$ by the equation
of motion; the value $\tfrac12 kA^2$ is read at an extreme.
\end{proof}

\begin{definition}[Damped oscillations]\label{def:g12:oscillators-and-time:damping}
\omterm{def:g10:inertia:inventory}{Friction} drains \omterm{def:g11:mechanical-energy:em}{mechanical energy} from every real \omterm{def:g12:oscillators-and-time:oscillator}{oscillator}: its
\omterm{def:g12:oscillators-and-time:oscillator}{free oscillations} are \emph{damped}\index{damping}. Under light
damping the system still oscillates, with a dying \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} but a
nearly unchanged repeat time, the \emph{\omterm{def:g12:oscillators-and-time:oscillator}{pseudo-period}}%
\index{pseudo-period}, close to $T_0$; under heavy damping (honey)
it creeps back to equilibrium without ever overshooting; the \omterm{def:g10:energy-conservation:energy}{energy},
proportional to \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} squared, decays into heat.
\end{definition}

\begin{definition}[Forced oscillations and resonance]\label{def:g12:oscillators-and-time:resonance}
Push an \omterm{def:g12:oscillators-and-time:oscillator}{oscillator} \omterm{def:g10:signals-and-waves:period}{periodically} --- \emph{forced oscillations}%
\index{forced oscillations} --- and after a brief transient it
oscillates at the \emph{driving} \omterm{def:g12:oscillators-and-time:oscillator}{frequency} $f$, not its own. Its
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} peaks when $f$ nears the \emph{natural \omterm{def:g12:oscillators-and-time:oscillator}{frequency}}%
\index{natural frequency} $f_0$ of the free \omterm{def:g12:oscillators-and-time:oscillator}{oscillator}: this is
\emph{resonance}\index{resonance}, each push arriving in step with
the motion, feeding in \omterm{def:g10:energy-conservation:energy}{energy} every cycle; the lighter the \omterm{def:g12:oscillators-and-time:damping}{damping},
the taller and sharper the peak.
\end{definition}

\begin{example}[Resonance for good and ill]\label{ex:g12:oscillators-and-time:resonance-zoo}
A parent pushing a swing pushes at its own \omterm{def:g12:oscillators-and-time:oscillator}{frequency} --- \omterm{def:g12:oscillators-and-time:resonance}{resonance} put
to \omterm{def:g11:work-of-force:work}{work}. In \num{2000} the new Millennium Bridge in London swayed
alarmingly when the crowd's steps locked onto a \omterm{def:g12:oscillators-and-time:resonance}{natural frequency} near
\qty{1}{Hz}; it closed two years for dampers. Every quartz watch keeps
its crystal singing by driving it at \omterm{def:g12:oscillators-and-time:resonance}{resonance}, \qty{32768}{Hz}.
\end{example}

\section{Measuring time}

\begin{method}[Timing a period]\label{met:g12:oscillators-and-time:timing}
Never time one oscillation: start the stopwatch as the bob crosses
the vertical (fastest, easiest to judge), count $N = 50$ oscillations,
divide by $N$ --- a \qty{0.2}{s} reaction error becomes \qty{4}{ms}
on $T_0$. Repeat, and compare runs.
\end{method}

\begin{example}[Three centuries of tick]\label{ex:g12:oscillators-and-time:clocks}
Huygens built the first pendulum clock in \num{1657}: isochronism
took clocks from a quarter hour of drift per day to seconds, and
compensated pendulums to fractions of a second per day. The quartz
wristwatch (\num{1969}) swapped the \omterm{def:g11:color-light-sources:cones}{rod} for a crystal flexing at
\qty{32768}{Hz}: tenths of a second per month. Since \num{1967} the
second itself is \emph{defined} by an atomic oscillation: the
duration of \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{periods} of the microwave radiation of an
internal transition of cesium-133 (\cref{ch:g12:quantum-world}). The
best cesium clocks drift one second in a hundred million years.
\end{example}

\begin{remark}[Why chase the nanosecond]\label{rem:g12:oscillators-and-time:gps}
A GPS receiver finds its position by timing radio \omterm{def:g10:signals-and-waves:signal}{signals} from
\omterm{def:g10:universal-gravitation:satellite}{satellites}; light travels \qty{30}{cm} per nanosecond, so a clock
error of one microsecond misplaces you by \qty{300}{m}. \omterm{def:g10:universal-gravitation:satellite}{Satellites}
therefore carry atomic clocks --- and at that precision the rate of a
clock depends on how fast it moves and how high it sits, an effect we
meet in \cref{ch:g12:special-relativity}.
\end{remark}

Compare timekeepers by relative drift (error over elapsed time): a
pendulum clock losing \qty{10}{s} per day, a quartz watch
\qty{0.5}{s} per month, a cesium clock \qty{1}{s} per hundred million
years: compute the three ratios. A GPS clock wrong by
\qty{1}{\micro s}: position error? Which family does GPS need?
\end{exercise}

\section{Problem: The Clockmaker's Bench}

\begin{problem}[{Weekend problem --- the clockmaker's bench: a wall
clock brought back to time --- fifty swings timed, a brass rod that
stretches in summer, the escapement's whispered push, a quartz
challenger --- the verdict rendered in seconds per month}]
\label{pb:g12:oscillators-and-time:1}
A pendulum wall clock arrives at the bench ``running wrong''. Treat
its brass \omterm{def:g11:color-light-sources:cones}{rod} and heavy bob as a \omterm{def:g12:oscillators-and-time:pendulum}{simple pendulum} of effective length
$L$, bob mass $m = \qty{1.2}{kg}$; the gearing counts each full
oscillation as exactly \qty{2.000}{s}.

\medskip
\noindent\textbf{Part I --- Taking the clock's pulse.}
\begin{enumerate}
  \item Why time $50$ oscillations rather than one? Estimate the error
        left on $T_0$ if your reaction time is \qty{0.2}{s}.
  \item The stopwatch reads \qty{99.4}{s} for $50$ oscillations: the \omterm{def:g12:oscillators-and-time:oscillator}{period}?
  \item Deduce the present effective length of the pendulum.
  \item What length would give exactly \qty{2.000}{s}? Should the
        rating nut raise or lower the bob, and by how much?
  \item Unadjusted, is the clock fast or slow, by how many seconds per day?
  \item Sensitivity to $g$: the \omterm{def:g12:oscillators-and-time:oscillator}{period} on the Moon
        ($g = \qty{1.62}{m/s^2}$)? The daily error at an observatory
        where $g = \qty{9.80}{m/s^2}$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The summer slump.} Brass expands: a
warming $\Delta\theta$ stretches the \omterm{def:g11:color-light-sources:cones}{rod} to $L' = L\,(1 + \lambda\,
\Delta\theta)$, $\lambda = \num{1.9e-5}$ per \omterm{def:g10:light-spectra:kelvin}{kelvin}. The clock is
adjusted at \qty{20}{\celsius}.
\begin{enumerate}[resume]
  \item Compute $\Delta L$ for a summer room at \qty{25}{\celsius}.
  \item Using $\sqrt{1+x} \approx 1 + x/2$, show that
        $\Delta T_0 / T_0 = \tfrac12 \lambda\, \Delta\theta$.
  \item Compute $\Delta T_0/T_0$ and the daily summer error: fast or slow?
  \item Same questions for a winter hallway at \qty{10}{\celsius}.
  \item Old precision clocks used ``gridiron'' pendulums mixing brass
        \omterm{def:g11:color-light-sources:cones}{rods} with steel ones (which expand about half as much) in
        alternating directions. Explain how this can hold the rate.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The whispered push.} With the escapement
disengaged, the swing, started at \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} $\theta_m = 2.00^\circ$,
halves its \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} in \qty{280}{s}; in service, a tiny push from the
escapement each \omterm{def:g12:oscillators-and-time:oscillator}{period} keeps $\theta_m$ constant.
\begin{enumerate}[resume]
  \item How many \omterm{def:g12:oscillators-and-time:oscillator}{periods} does the halving take?
  \item At the extreme the bob has climbed $h = L(1 - \cos\theta_m)
        \approx L\,\theta_m^2/2$ (radians): deduce $E_m = \tfrac12
        m g L\,\theta_m^2$ and compute it
        (\cref{ch:g11:mechanical-energy}).
  \item \omterm{def:g10:energy-conservation:energy}{Energy} is proportional to \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} squared: what fraction of
        $E_m$ is lost per \omterm{def:g12:oscillators-and-time:oscillator}{period}?
  \item Deduce the \omterm{def:g10:energy-conservation:energy}{energy} of each push, and the average \omterm{def:g11:work-of-force:power}{power} needed.
  \item The drive \omterm{def:g10:universal-gravitation:weight}{weight} $M = \qty{2.5}{kg}$ descends $h =
        \qty{1.0}{m}$ per week: is that enough \omterm{def:g10:energy-conservation:energy}{energy}, at what \omterm{def:g10:energy-conservation:efficiency}{efficiency}?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The quartz challenger.} A quartz
movement oscillates at $f = \qty{32768}{Hz}$ and drifts about
\qty{0.3}{s} per month; the restored pendulum still drifts
\qty{0.5}{s} per day.
\begin{enumerate}[resume]
  \item Compute the quartz \omterm{def:g12:oscillators-and-time:oscillator}{period}. Show $\num{32768} = 2^{15}$: why
        is a \omterm{def:g11:work-of-force:power}{power} of two exactly what a watch circuit wants?
  \item Compute the relative drifts of quartz and restored pendulum,
        and each in seconds per month. Who wins, by what factor?
  \item The cesium standard --- \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{periods} make one
        second --- drifts \qty{1}{s} per hundred million years: its
        relative drift? How many orders of magnitude below quartz,
        and why does GPS demand it?
  \item The verdict, one sentence with numbers: what does the restored
        clock earn on the wall, and who now owns the second?
\end{enumerate}
\end{problem}
