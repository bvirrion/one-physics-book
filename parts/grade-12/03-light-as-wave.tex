\chapter{Light as a Wave: Diffraction and Interference}\label{ch:g12:light-as-wave}

Tilt a soap film toward a window: colors slide across it, though
nothing in soap or water is colored. Point a \omterm{def:g11:color-light-sources:lamps}{laser} at a single hair:
the wall behind shows not a \omterm{def:g11:lenses-and-eye:lens}{hair-thin} shadow but a bright streak,
barred with dark lines, thousands of times wider than the hair.
Straight-marching rays (\cref{ch:g10:refraction}) explain neither
sight. This chapter promotes light to a full \omterm{def:g10:signals-and-waves:wave}{wave} --- half a
micrometre from crest to crest --- then cashes the idea in: a \omterm{def:g11:color-light-sources:lamps}{laser}
and a bare wall will measure a hair, a \omterm{def:g12:mechanical-waves:wavelength}{wavelength}, a CD's grooves.

\section{Light is a wave}

\begin{definition}[Light waves]\label{def:g12:light-as-wave:lightwave}
\omterm{def:g10:light-spectra:white}{Monochromatic light} --- one pure color, which no prism can split
further (\cref{ch:g10:light-spectra}) --- is a \omterm{def:g10:signals-and-waves:period}{periodic} \emph{light
wave}\index{light wave}: it crosses vacuum at $c = \qty{3.00e8}{m/s}$
and is characterized by its \omterm{def:g10:signals-and-waves:frequency}{frequency} $f$ or its \emph{wavelength in
vacuum}\index{wavelength!of light} $\lambda = c/f$, the
crest-to-crest distance of \cref{ch:g12:mechanical-waves}. The eye
responds from about \qty{400}{nm} (violet) to \qty{800}{nm} (red);
\omterm{def:g10:light-spectra:white}{white light} mixes the whole \omterm{def:g11:fundamental-interactions:strong}{range}.
\end{definition}

\begin{example}[Orders of magnitude]\label{ex:g12:light-as-wave:orders}
A helium--neon \omterm{def:g11:color-light-sources:lamps}{laser} emits red light of $\lambda = \qty{633}{nm} =
\qty{6.33e-7}{m}$, so $f = c/\lambda \approx \qty{4.7e14}{Hz}$: no
electronics counts that fast --- every \omterm{def:g12:mechanical-waves:wavelength}{wavelength} in this chapter will
be measured by geometry. For scale, a hair (about \qty{70}{\micro m})
is a hundred \omterm{def:g12:mechanical-waves:wavelength}{wavelengths} across. What \omterm{def:g10:signals-and-waves:wave}{waves}? Not the air: an electric
and \omterm{def:g11:magnetic-fields:bfield}{magnetic field}, honestly described in the Year 2 volume; only the
\omterm{def:g12:mechanical-waves:wavelength}{wavelength} matters below.
\end{example}

\section{Diffraction}

\begin{definition}[Diffraction]\label{def:g12:light-as-wave:diffraction}
When a \omterm{def:g10:signals-and-waves:wave}{wave} meets an aperture or an obstacle of size $a$ comparable to
its \omterm{def:g12:mechanical-waves:wavelength}{wavelength}, it spreads into the region that rays would leave dark.
This spreading is \emph{diffraction}\index{diffraction} --- negligible
while $\lambda/a$ is tiny, dominant as $a$ shrinks toward $\lambda$.
\end{definition}

\begin{proposition}[Angular half-width of the spread]\label{prop:g12:light-as-wave:halfwidth}
A slit of width $a$, lit with \omterm{def:g12:mechanical-waves:wavelength}{wavelength} $\lambda < a$, spreads the
light about the incident direction within the angular half-width
$\theta$ (radians); a screen at distance $D$ shows a central bright
band of width $L$, twice $D \tan\theta \approx D\,\theta$ (small
angle):
\[
\theta \approx \frac{\lambda}{a}, \qquad L = \frac{2\lambda D}{a}.
\]
\end{proposition}
\admitted

\begin{remark}[Read the formula backwards]\label{rem:g12:light-as-wave:honest}
The \emph{narrower} the slit, the \emph{wider} the spread --- the
opposite of the ray prediction. The honest derivation (summing
wavelets re-emitted across the slit) lives in the Year 2 volume; here
the formula is a measuring tool.
\end{remark}

\begin{example}[A laser through a slit]\label{ex:g12:light-as-wave:slit}
Red \omterm{def:g11:color-light-sources:lamps}{laser}, $\lambda = \qty{633}{nm}$; slit $a = \qty{0.10}{mm}$;
screen at $D = \qty{2.00}{m}$. Then $\theta = \num{6.33e-7} /
\num{1.0e-4} = \qty{6.3e-3}{rad}$ --- a third of a degree, but the
central band on the wall is $L = 2 \times \num{6.33e-7} \times 2.00 /
\num{1.0e-4} \approx \qty{2.5}{cm}$ wide, flanked by dark lines and
fainter side bands. Halve the slit and the band doubles.
\end{example}

\begin{proposition}[An obstacle diffracts like a slit]\label{prop:g12:light-as-wave:hair}
A \omterm{def:g11:lenses-and-eye:lens}{thin} opaque obstacle of width $d$ --- a wire, a hair --- produces,
outside the incident beam, the same pattern as a slit of width $d$
(Babinet's principle, derived in the Year~2 volume).
\end{proposition}
\admitted

\begin{method}[Measuring a tiny width]\label{met:g12:light-as-wave:hairwidth}
\begin{enumerate}
  \item Stretch the hair across the \omterm{def:g11:color-light-sources:lamps}{laser} beam; set a screen at a
        measured distance $D$ (a few \omterm{def:g10:orders-of-magnitude:unit}{metres}).
  \item Measure the central band's width $L$, dark line to dark line.
  \item Since $L = 2\lambda D/d$: $d = 2\lambda D/L$ --- finer hair,
        larger pattern.
\end{enumerate}
\end{method}

\begin{remark}[Diffraction limits every instrument]\label{rem:g12:light-as-wave:limits}
Light enters a camera, a telescope or an eye through an aperture of
width $a$, so even a perfect \omterm{def:g11:lenses-and-eye:lens}{lens} blurs a point into a spot of
angular size about $\lambda/a$: closer details merge. A visible-light
microscope thus resolves nothing much below $\lambda \approx
\qty{0.5}{\micro m}$ (bacteria yes, viruses no); telescope mirrors
are built wide for the same reason.
\end{remark}

\section{Interference}

\begin{definition}[Coherent sources, path difference]\label{def:g12:light-as-wave:coherent}
Two \omterm{def:g10:signals-and-waves:wave}{wave} sources are \emph{coherent}\index{coherent sources} when
they have the same \omterm{def:g10:signals-and-waves:frequency}{frequency} and vibrate in step --- in practice,
when they are two copies of one \omterm{def:g10:signals-and-waves:wave}{wave}, obtained by splitting it. For a
point $M$ reached by both, the \emph{path difference}%
\index{path difference} is $\delta = S_2M - S_1M$.
\end{definition}

\begin{proposition}[Interference conditions]\label{prop:g12:light-as-wave:conditions}
Where two \omterm{def:g12:light-as-wave:coherent}{coherent} \omterm{def:g10:signals-and-waves:wave}{waves} of \omterm{def:g12:mechanical-waves:wavelength}{wavelength} $\lambda$ overlap they add ---
they \emph{interfere}\index{interference}:
\begin{itemize}
  \item $\delta = k\lambda$ ($k$ integer): in step, reinforcing ---
        \emph{constructive}\index{constructive interference} interference (bright);
  \item $\delta = (k + \tfrac12)\lambda$: crest on trough, canceling
        --- \emph{destructive}\index{destructive interference} interference (dark).
\end{itemize}
\end{proposition}

\begin{proof}
Shifting a \omterm{def:g12:mechanical-waves:periodic}{periodic wave} by whole \omterm{def:g12:mechanical-waves:wavelength}{wavelengths} changes nothing: crests
pile on crests. Half a \omterm{def:g12:mechanical-waves:wavelength}{wavelength} more puts crest on trough: equal
\omterm{def:g10:signals-and-waves:wave}{waves} cancel.
\end{proof}

\begin{example}[Young's two-slit experiment]\label{ex:g12:light-as-wave:young}
Pierce two fine slits $S_1$, $S_2$, a fraction of a millimetre apart,
in an opaque plate, and light both with one \omterm{def:g11:color-light-sources:lamps}{laser}: two \omterm{def:g12:light-as-wave:coherent}{coherent
sources}. On a screen a couple of \omterm{def:g10:orders-of-magnitude:unit}{metres} away the overlap is striped:
at the center $\delta = 0$, bright; climbing the screen, $\delta$
grows through $\lambda/2$ (dark), $\lambda$ (bright)\,\dots\ darkness
where light added to light cancels.
\end{example}

\begin{definition}[Fringes and interfringe]\label{def:g12:light-as-wave:fringes}
The pattern's alternating bright and dark bands are its
\emph{interference fringes}\index{interference fringes}; the distance
$i$ between consecutive bright centers is the \emph{interfringe}%
\index{interfringe}.
\end{definition}

\begin{proposition}[The interfringe]\label{prop:g12:light-as-wave:interfringe}
For slits a distance $a$ apart and a screen at distance $D \gg a$,
the point of the screen at distance $x$ from the center $O$ has \omterm{def:g12:light-as-wave:coherent}{path
difference} $\delta \approx a\,x/D$ (admitted geometry); by
\cref{prop:g12:light-as-wave:conditions}, bright fringes sit at
$x_k = k\,\lambda D/a$, evenly spaced with \omterm{def:g12:light-as-wave:fringes}{interfringe}
\[
i = \frac{\lambda D}{a}.
\]
With $\lambda = \qty{633}{nm}$, $a = \qty{0.20}{mm}$, $D =
\qty{2.0}{m}$: $i = \qty{6.3}{mm}$ --- the setup magnifies $\lambda$
by $D/a = 10^4$. The geometry behind it is worked out in the Year~2
volume.
\end{proposition}
\admitted

\begin{method}[Measuring a wavelength]\label{met:g12:light-as-wave:lambda}
\begin{enumerate}
  \item Send the unknown \omterm{def:g11:color-light-sources:lamps}{laser} through a double slit of known spacing
        $a$; place the screen at a measured distance $D$.
  \item Measure the span of $10$ \omterm{def:g12:light-as-wave:fringes}{interfringes}; divide by $10$: the
        \omterm{def:g12:light-as-wave:fringes}{interfringe} $i$, with the ruler error shrunk tenfold.
  \item Then $\lambda = i\,a/D$.
\end{enumerate}
\end{method}

\begin{remark}[Why two lamps never interfere]\label{rem:g12:light-as-wave:lamps}
Two desk lamps lighting one wall produce no fringes: each emits
short, uncorrelated \omterm{def:g10:signals-and-waves:wave}{wave} trains whose relative timing jumps randomly
a billion times a second, so the fringes shift just as fast and the
eye averages them into uniform light. Interference demands \omterm{def:g12:light-as-wave:coherent}{coherence}
--- one \omterm{def:g10:signals-and-waves:wave}{wave} split in two, as Young's slits do; the \omterm{def:g11:color-light-sources:lamps}{laser} is the
modern shortcut, one clean long \omterm{def:g10:signals-and-waves:wave}{wave}.
\end{remark}

\section{Colors painted by interference}

\begin{definition}[Iridescence]\label{def:g12:light-as-wave:iridescence}
Colors that shift with the viewing angle --- soap bubbles, oil slicks,
peacock feathers, the back of a CD --- are \emph{iridescence}%
\index{iridescence}: no pigment, but interference in \omterm{def:g10:light-spectra:white}{white light},
reinforcing some \omterm{def:g12:mechanical-waves:wavelength}{wavelengths} and canceling others.
\end{definition}

\begin{example}[Soap films and oil slicks]\label{ex:g12:light-as-wave:soap}
Light striking a soap film reflects partly off its front face, partly
off its back: two \omterm{def:g12:light-as-wave:coherent}{coherent} copies whose \omterm{def:g12:light-as-wave:coherent}{path difference} is set by the
thickness and the viewing angle. Each thickness returns some
\omterm{def:g12:mechanical-waves:wavelength}{wavelengths} reinforced, others canceled: the film reflects a color
that slides as it drains and \omterm{def:g11:lenses-and-eye:lens}{thins}. An oil slick on wet asphalt plays
the same trick.
\end{example}

\begin{remark}[Monochromatic versus white light]\label{rem:g12:light-as-wave:white}
In \omterm{def:g10:light-spectra:white}{white light} every \omterm{def:g12:mechanical-waves:wavelength}{wavelength} draws its own Young fringes with its
own $i = \lambda D/a$: at the center all colors are bright (one white
fringe); a few fringes out the patterns drift apart --- iridescent
edges, then blur. \omterm{def:g10:light-spectra:white}{Monochromatic light} gives crisp fringes without
end: hence the \omterm{def:g11:color-light-sources:lamps}{laser} in every sharp measurement here.
\end{remark}

\section{Problem: Measuring with Light}

\begin{problem}[{Weekend problem --- measuring with light: a laser, a
ruler and a bare wall become a micrometre workshop gauging a slit, a
human hair, an unknown wavelength and the grooves of two silver discs}]
\label{pb:g12:light-as-wave:1}
Your kit: a red \omterm{def:g11:color-light-sources:lamps}{laser} ($\lambda = \qty{633}{nm}$, from the label), a
green \omterm{def:g11:color-light-sources:lamps}{laser} of unknown \omterm{def:g12:mechanical-waves:wavelength}{wavelength}, a calibrated slit
$a = \qty{100}{\micro m}$, a double slit of spacing $a' =
\qty{0.250}{mm}$, a tape measure, two hairs, a CD and a DVD.

\medskip
\noindent\textbf{Part I --- Calibrating on the known slit.}
Red \omterm{def:g11:color-light-sources:lamps}{laser}, slit, wall at $D = \qty{3.00}{m}$.
\begin{enumerate}
  \item Compute the angular half-width $\theta$ of the diffracted beam.
  \item Show that the central band has width $L = 2\lambda D/a$;
        compute it.
  \item You measure $L = \qty{3.7}{cm}$: compute the relative
        deviation. Is the method validated?
  \item A second slit is half as wide, \qty{50}{\micro m}: predict its
        $L$. State the rule linking slit width and pattern width.
  \item Justify question 2's step $\tan\theta \approx \theta$: compare
        $\theta$ and $\tan\theta$ at this angle.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hair.} Same \omterm{def:g11:color-light-sources:lamps}{laser} and wall; a
stretched hair replaces the slit.
\begin{enumerate}[resume]
  \item Which admitted result lets you keep the slit formula for a
        hair, and what does it say?
  \item The central band is $L = \qty{5.4}{cm}$ wide: compute the
        hair's diameter $d$.
  \item Your ruler reads $L$ to $\pm\qty{0.2}{cm}$: give the resulting
        \omterm{def:g11:fundamental-interactions:strong}{range} for $d$, as $d \pm \Delta d$.
  \item Your friend's hair gives $L = \qty{7.6}{cm}$: its diameter?
