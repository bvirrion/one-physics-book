\end{solution}

\begin{solution}{exo:g12:projectile-motion:15}
पर $ t^* = D/(v_0\cos\alpha)$ डार्ट ऊंचाई पर है
$ v_0\sin\alpha\, t^* - \tfrac12 g t^{*2} = D\tan\alpha - \tfrac12 g
t^{*2} = H - \tfrac12 g t^{*2}$, बिल्कुल गिरा हुआ बंदर
$ y = H - \tfrac12 g t^{*2}$. दोनों लटक गए $\tfrac12 gt^2$ उनके नीचे
नो-ग्रेविटी स्पॉट --- वे किसी के लिए भी मिलते हैं $ v_0$.
\end{solution}

\begin{solution}{pb:g12:projectile-motion:1}
\textbf{1.} \omterm{def:g10:universal-gravitation:weight}{वज़न} केवल (हवा उपेक्षित); \omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम}:
$ m\vect a = m\vect g $, इसलिए $\vect a = (0,\,-g)$.

\textbf{2.} $ v_x = v_0\cos\alpha $, $ v_y = v_0\sin\alpha - gt $.

\textbf{3.} $ x = v_0\cos\alpha\, t $,
$ y = v_0\sin\alpha\, t - \tfrac12 gt^2$.

\textbf{4.} $ t = x/(v_0\cos\alpha)$:
$ y = x\tan\alpha - g x^2/(2v_0^2\cos^2\alpha)$.

\textbf{5.} $ v_0^2 = \dfrac{g d^2}{2\cos^2\alpha\,(d\tan\alpha -
1.05)} = \dfrac{173.0}{0.826 \times 3.96} = 52.9$:
$ v_0 = \qty{7.28}{m/s} \approx \qty{26}{km/h}$.

\textbf{6.} $ t = d/(v_0\cos\alpha) = 4.20/4.68 = \qty{0.90}{s}$.

\textbf{7.} शीर्ष पर $ t = v_0\sin\alpha/g = \qty{0.57}{s}$, ऊंचाई
$2.00 + (v_0\sin\alpha)^2/2g = 2.00 + 1.58 = \qty{3.58}{m}$;
$0.57 < 0.90$: रिम पर उतरना।

\textbf{8.} $ y(0.90) = 0.90\tan\ang{50} - 0.182 = \qty{0.89}{m}$
उपरोक्त रिलीज़, यानी\ \qty{2.89}{m} फर्श के ऊपर: \qty{0.29}{m}
उंगलियों के पोरों के ऊपर.

\textbf{9.} $ v_x = \qty{4.68}{m/s}$;
$ v_y = 5.57 - 9.81 \times 0.90 = \qty{-3.24}{m/s}$;
$ v = \qty{5.69}{m/s}$, पर $\arctan(3.24/4.68) = \ang{34.7}$ नीचे
क्षैतिज.

\textbf{10.} रिम एक क्षैतिज वलय है: गेंद को इसे पार करना होगा
ऊपर से विमान. प्रवेश द्वार जितना तीव्र होगा, वलय उतना ही बड़ा होगा
उद्घाटन वेग के साथ दिखता है --- त्रुटि के लिए अधिक मार्जिन।

\textbf{11.} $1.05/26.0 = 4\%$: नगण्य. $ y(t_f) = 0$ देता है
$ t_f\,(v_0\sin\alpha - \tfrac12 g t_f) = 0$, इसलिए
$ t_f = 2v_0\sin\alpha/g $.

\textbf{12.} $ R = v_0\cos\alpha\, t_f = v_0^2\sin(2\alpha)/g $;
अधिकतम पर $\alpha = \ang{45}$.

\textbf{13.} $ v_0 = \sqrt{gR} = \sqrt{9.81 \times 26.0} =
\qty{16.0}{m/s} \approx \qty{57}{km/h}$.

\textbf{14.} $ R_{\max} = 17.5^2/9.81 = \qty{31.2}{m} >
\qty{26.0}{m}$: संभव, साथ \qty{5}{m} का \omterm{def:g12:projectile-motion:range}{श्रेणी} बचा।

\textbf{15.} $\sin(2\alpha) = 255.1/306.3 = 0.833$:
$\alpha = \ang{28.2}$ या \ang{61.8}; \omterm{def:g12:projectile-motion:range}{उड़ान का समय}
$2v_0\sin\alpha/g = \qty{1.69}{s}$ और \qty{3.14}{s}. केवल फ्लैट
गोली मारता है \qty{2.0}{s} बजर.

\textbf{16.} $ v_0^2 = \dfrac{9.81 \times 36}{2\cos^2\ang{75}\,
(6.0\tan\ang{75} - 1.05)} = \dfrac{353.2}{2.86} = 123.5$:
$ v_0 = \qty{11.1}{m/s}$.

\textbf{17.} $ y(5.55) = 5.55\tan\ang{75} - 18.26 = \qty{2.45}{m}$
उपरोक्त रिलीज़, यानी\ \qty{4.45}{m} फर्श के ऊपर: \qty{0.50}{m}
बोर्ड के शीर्ष किनारे के ऊपर.

\textbf{18.} शीर्ष $2.00 + (11.1\sin\ang{75})^2/2g \approx
\qty{7.9}{m}$; उड़ान $6.0/(11.1\cos\ang{75}) = \qty{2.1}{s}$ ---
गेंद लटकने के साथ पूरे दो सेकंड \qty{8}{m} ऊपर।

\textbf{19.} ड्रैग प्रत्येक शॉट को छोटा कर देता है और वंश को तेज़ कर देता है; द
प्रभाव \omterm{def:g12:newtons-laws:momentum}{गति} के साथ बढ़ता है, इसलिए \qty{17.5}{m/s} पूरी अदालत में खुशी की \omterm{def:g10:signals-and-waves:wave}{लहर}
सबसे अधिक कष्ट सहता है --- यह वास्तविक है \omterm{def:g12:projectile-motion:range}{श्रेणी} से कुछ \omterm{def:g10:orders-of-magnitude:unit}{मीटर} कम पड़ता है
\qty{31}{m}, हाशिये पर खाना।

\textbf{20.} \emph{बजर-बीटर फैसला}: द \qty{26.0}{m}
गोली की मांग \qty{16.0}{m/s} और एक हाथ की आपूर्ति के बारे में
\qty{17.5}{m/s}, ए \omterm{def:g12:projectile-motion:range}{श्रेणी} का मार्जिन \qty{5}{m} ($\approx 20\%$) वह
वायु \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} कम हो जाता है लेकिन मिटता नहीं है। संभव --- जो है
वास्तव में क्यों, एक सीज़न में कुछ बार, यह अंदर जाता है।
\end{solution}
