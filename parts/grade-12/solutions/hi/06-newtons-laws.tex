\begin{solution}{exo:g12:newtons-laws:5}
\omterm{def:g10:universal-gravitation:weight}{वज़न} (पृथ्वी पुस्तक पर, नीचे): साथी = पुस्तक पृथ्वी को ऊपर खींचती है।
\omterm{def:g12:newtons-laws:friction}{सामान्य बल} (किताब पर टेबल, ऊपर): पार्टनर = किताब टेबल को दबाती है,
नीचे. \omterm{def:g10:universal-gravitation:weight}{वज़न} और पर सामान्य कार्रवाई \emph{वही} शरीर और दो से संबंधित हैं
अलग-अलग इंटरैक्शन---यहां केवल इसलिए समान हैं $\vect a = \vect 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:6}

(बी) $ a = (220.7 - 0.35 \times 490.5)/50 = \qty{0.98}{m/s^2}$.
(सी) $ F = \mu_k N = \qty{172}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:7}
(ए) स्लाइड जब $\tan\alpha > \mu_s $: $\alpha = \arctan 0.45 \approx
24^\circ $. (बी) $ a = g(\sin 30^\circ - 0.35\cos 30^\circ) =
9.81 \times 0.197 = \qty{1.9}{m/s^2}$. (सी) हाँ: एक बार फिसलने पर,
\omterm{def:g10:inertia:inventory}{टकराव} पर गिरता है $\mu_k < \mu_s $, इसलिए
$ a = g\cos\alpha\,(\tan\alpha - \mu_k) \approx \qty{0.9}{m/s^2} > 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:8}
$ a = m_2 g/(m_1 + m_2) = 19.62/5.0 = \qty{3.9}{m/s^2}$;
$ T = m_1 a = \qty{11.8}{N}$. लटकता हुआ ब्लॉक नीचे की ओर \omterm{def:g12:newtons-laws:momentum}{गति} करता है,
तो \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} इस पर नीचे इंगित करता है: $ T $ कम करना चाहिए $ m_2 g $.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:9}
$ v_{\max} = \sqrt{\mu_s g R}$: सूखा
$\sqrt{0.70 \times 9.81 \times 50} = \qty{18.5}{m/s} \approx
\qty{67}{km/h}$; गीला $\qty{14.0}{m/s} = \qty{50}{km/h}$. दोनों
$ mv^2/R $ और $\mu_s mg $ ढोना $ m $: यह रद्द हो जाता है.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:10}
\omterm{def:g10:universal-gravitation:weight}{खड़ा}: $ T\cos\theta = mg $, इसलिए
$ T = 2.943/\cos 25^\circ = \qty{3.2}{N}$. RADIUS $ R = L\sin\theta =
\qty{0.507}{m}$; रेडियल: $ T\sin\theta = mv^2/R $, इसलिए
$ v = \sqrt{gR\tan\theta} = \sqrt{9.81 \times 0.507 \times 0.466}
= \qty{1.5}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:11}
$ a_z = (N - mg)/m $ साथ $ mg = \qty{49.05}{N}$: $+\qty{1.0}{m/s^2}$,
$0$, $-\qty{1.0}{m/s^2}$. जैसे\ एक ऊर्ध्वगामी यात्रा: \omterm{def:g12:newtons-laws:momentum}{गति} बढ़ाना, परिभ्रमण करना,
ब्रेक. हाँ: पहले से ही उतरते हुए प्रवेश करना, वही रीडिंग का मतलब है ब्रेक,
क्रूज़, \omterm{def:g12:newtons-laws:momentum}{गति} फिर से नीचे की ओर --- स्केल त्वरण पढ़ता है, नहीं
वेग.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:15}
(ए) $ m_2 g > \mu_s m_1 g $: $ m_2 > 0.35 \times 4.0 = \qty{1.4}{kg}$.
(बी) $ a = (m_2 - \mu_k m_1)\,g/(m_1 + m_2) = 1.0 \times 9.81/6.0 =

\begin{solution}{pb:g12:newtons-laws:1}
\textbf{1.} सिस्टम: केबिन $+$ सवारी डिब्बा ($+$ यात्री)। पहाड़
उत्कृष्ट सटीकता के लिए जड़त्वीय, जमीन पर स्थिर है। बल:
\omterm{def:g10:universal-gravitation:weight}{वज़न} $ M\vect g $ (नीचे), \omterm{def:g12:newtons-laws:friction}{सामान्य बल} $\vect N $ ट्रैक केबल का
(इसके लंबवत), ढोना \omterm{def:g10:inertia:inventory}{तनाव} $\vect T $ (ढलान के ऊपर).

\textbf{2.} विश्राम के समय, पहला नियम: $\sum \vect F = \vect 0$. साथ में
ढलान: $ T = Mg\sin\alpha $; लंबवत: $ N = Mg\cos\alpha $.

\textbf{3.} $ T = 4000 \times 9.81 \times 0.500 = \qty{19.6}{kN}$;
$ N = 4000 \times 9.81 \times 0.866 = \qty{34.0}{kN}$.

\textbf{4.} खाली: $ T = \qty{9.8}{kN}$. प्रत्येक बल संतुलन में है
के आनुपातिक $ M $, इसलिए $ T $ भी है।

\textbf{5.} ढलान के साथ, $ Ma = Mg\sin\alpha $: $ M $ रद्द करता है,
$ a = g\sin\alpha = \qty{4.9}{m/s^2}$ ढलान के नीचे.

\textbf{6.} $ T' = M(g\sin\alpha + a) = 4000 \times 5.505 =
\qty{22.0}{kN}$.

\textbf{7.} $ a/(g\sin\alpha) = 0.60/4.905 \approx 12\%$.

\textbf{8.} $ t = v/a = \qty{10}{s}$; $ d = v^2/(2a) = \qty{30}{m}$.

\textbf{9.} का ऊर्ध्वाधर घटक $\vect a $: $ a\sin\alpha =
\qty{0.30}{m/s^2}$, इसलिए $ N = m(g + a\sin\alpha) = 80 \times 10.11 =
\qty{809}{N}$ ख़िलाफ़ $ mg = \qty{785}{N}$: $3\%$ भारी.

\textbf{10.} फर्श स्थिर है \omterm{def:g10:inertia:inventory}{टकराव}, क्षैतिज:
$ ma\cos\alpha = 80 \times 0.52 \approx \qty{42}{N}$. झुकना डालता है
संयुक्त मंजिल उसके माध्यम से धक्का \omterm{def:g11:forces-and-motion:com}{सेंटर ऑफ मास}, इसलिए यह टिप नहीं करता है
उसे.

\textbf{11.} $ a_T = dv/dt = 0$ (निरंतर \omterm{def:g12:newtons-laws:momentum}{गति});
$ a_N = v^2/R = 36/40 = \qty{0.90}{m/s^2}$, केंद्र पर लक्षित ---
सीधे शिखर पर।

\textbf{12.} सामान्य अक्ष (नीचे सकारात्मक): $ Mg - N' = Mv^2/R $, इसलिए
$ N' = M(g - v^2/R) = 4000 \times 8.91 = \qty{35.6}{kN}$.

\textbf{13.} $ N = m(g - v^2/R) = 80 \times 8.91 = \qty{713}{N}$:
द्वारा हल्का $ v^2/(gR) = 0.90/9.81 \approx 9\%$.

\textbf{14.} $ N' = 0$ पर $ v = \sqrt{gR} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$ --- तीन गुना से भी अधिक क्रूज़: गाड़ी की सवारी
एक विस्तृत मार्जिन के साथ केबल.

\textbf{15.} \omterm{thm:g12:newtons-laws:third}{न्यूटन का तीसरा नियम}: \qty{35.6}{kN}, को नीचे दबाते हुए
केबल (इसलिए टावर पर) --- \omterm{def:g10:relative-motion:frame}{आराम से} थोड़ा कम \omterm{def:g10:universal-gravitation:weight}{वज़न},
घुमावदार उड़ान से राहत मिली।

\textbf{16.} $ F = \Delta p/\Delta t = 4000 \times 6.0/1.5 =
\qty{16}{kN}$, \omterm{def:g12:newtons-laws:momentum}{गति} के विपरीत ($ a = \qty{4.0}{m/s^2}$).

\textbf{17.} $ f = ma = 80 \times 4.0 = \qty{320}{N}$; छत है
$\mu_s mg = 0.40 \times 785 = \qty{314}{N} < \qty{320}{N}$: पैर
अकेले पर्ची --- रेलिंग पकड़ो.

\textbf{18.} $0 = M_0 V + m(v_{\text{rel}} + V)$:
$ V = -80 \times 2.5/2080 = \qty{-0.096}{m/s}$; यात्री आगे बढ़ता है
$2.5 - 0.096 \approx \qty{2.4}{m/s}$ ज़मीन के ऊपर.

\textbf{19.} रॉकेट: यह \omterm{def:g10:inertia:inventory}{जोर} प्रति उत्सर्जित द्रव्यमान द्वारा तय किया जाता है
इसके सापेक्ष इजेक्शन \omterm{def:g12:newtons-laws:momentum}{गति} का \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} गुना।

\textbf{20.} सबसे वृहद \omterm{def:g10:inertia:inventory}{तनाव} मिले स्टार्टअप है
$ T' = M \times \qty{5.505}{N/kg}$ (बाकी देता है $ Mg\sin\alpha $, द
शिखा और भी कम)। कैप: $ T' \leq \qty{20}{kN}$, इसलिए
$ M \leq 20000/5.505 = \qty{3.63e3}{kg}$: पेलोड
\qty{1.63e3}{kg}, यानी\ $20$ के यात्री \qty{80}{kg} (नहीं $25$).
तख्ती: \emph{20 व्यक्ति}; तब $ M = \qty{3.6e3}{kg}$,
$ T' = \qty{19.8}{kN}$, सुरक्षा कारक $100/19.8 \approx 5.1 \geq 5$.
\end{solution}
