\section*{अध्याय \ref{ch:g12:rlc-oscillations} --- निःशुल्क विद्युत दोलन: आरएलसी}

\begin{solution}{exo:g12:rlc-oscillations:1}
(ए) $\sqrt{LC} = \qty{1.0e-4}{s}$: $ T_0 = \qty{0.63}{ms}$,
$ f_0 \approx \qty{1.6}{kHz}$. (बी) $\sqrt{LC} = \qty{3.16e-7}{s}$:
$ T_0 = \qty{2.0}{\micro s}$, $ f_0 \approx \qty{0.50}{MHz}$ --- ए

\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:2}
$ LC $: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times \unit{C/A} =
\unit{s} \times \unit{s} = \unit{s^2}$, इसलिए $\sqrt{LC}$ सेकंड में है.
$2\pi $ एक शुद्ध संख्या है --- परिवर्तन का कोई \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} नहीं।
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:3}
$ Q_0 = CU_0 = \qty{2.8}{\micro C}$; $ E = \tfrac12 CU_0^2 =
\tfrac12 \times \num{4.7e-7} \times 36 \approx \qty{8.5}{\micro J}$.
आवेश दोलन करता है $ f_0 = 1/(2\pi\sqrt{LC})$, वर्तमान एक चौथाई
\omterm{def:g12:oscillators-and-time:oscillator}{अवधि} से बाहर \omterm{def:g12:oscillators-and-time:phase}{चरण}, ऊर्जा प्लेटों के बीच स्लोशिंग और कुंडल, कुल
\omterm{prop:g12:work-and-energy:equilibria}{स्थिर}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:4}
$ Q_0 = \qty{2.0}{\micro C}$; $ f_0 = 1/T_0 = \qty{250}{Hz}$;
$ q(T_0/2) = \qty{-2.0}{\micro C}$; $ i(0) = 0$ (कोसाइन फ्लैट पर $ t=0$);
$ I_0 = 2\pi \times \num{2.0e-6}/\num{4.0e-3} \approx \qty{3.1}{mA}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:5}
सभी में \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} कब $ q = \pm Q_0$, $ i = 0$ ($ t = 0$, $ T_0/2$,
$ T_0$, \dots): पेंडुलम की चरम सीमाएँ। सभी में कुंडल कब $ q = 0$,
$ i = \pm I_0$ ($ t = T_0/4$, $3T_0/4$, \dots): पेंडुलम का सबसे निचला भाग,
सबसे तेज़ बिंदु.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:6}
$ d^2q/dt^2 = -(2\pi/T_0)^2\,Q_0\cos(2\pi t/T_0) = -(2\pi/T_0)^2 q $,
इसलिए $ L\,d^2q/dt^2 + q/C = q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$ सभी के लिए $ t $
बिल्कुल कब $(2\pi/T_0)^2 = 1/(LC)$, यानी\ $ T_0 = 2\pi\sqrt{LC}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:7}
$ L = \dfrac{1}{4\pi^2 f_0^2 C} = \dfrac{1}{39.5 \times
(\num{1.98e5})^2 \times \num{2.2e-10}} \approx \qty{2.9}{mH}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:8}
$ E = \tfrac12 CU_0^2 = \qty{5.0e-5}{J}$; $\tfrac12 LI_0^2 = E $ देता है
$ I_0 = \sqrt{2E/L} = \sqrt{\num{1.0e-2}} = \qty{0.10}{A}$, प्रथम
पर पहुंच गया $ T_0/4 = \tfrac14 \times \qty{0.63}{ms} \approx
\qty{0.16}{ms}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:9}
$ E_C \propto \cos^2(2\pi t/T_0)$, और $\cos^2$ प्रत्येक को दोहराता है
\emph{आधा} बारी: की दो भराई \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} प्रति \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}, इस तरह
$2f_0$. पर $ t = T_0/8$  \omterm{def:g12:oscillators-and-time:phase}{चरण} है $\pi/4$:
$\cos^2(\pi/4) = \tfrac12$, इसलिए $ E_C = E_L = E/2$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:10}
अनुपात $3.0/4.0 = 0.75$ प्रति \omterm{def:g12:rlc-oscillations:regimes}{छद्म काल}: छठा अधिकतम
$4.0 \times 0.75^6 \approx \qty{0.71}{V}$. ऊर्जा $\propto $ \omterm{def:g12:oscillators-and-time:oscillator}{आयाम}
चुकता: $0.75^2 \approx 56\%$ प्रत्येक जीवित रहता है \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}.
\omterm{def:g12:rlc-oscillations:regimes}{छद्म काल} रहता है $\approx T_0 = 2\pi\sqrt{LC}$: रोशनी भिगोना
को सिकोड़ता है \omterm{def:g12:oscillators-and-time:oscillator}{आयाम}, शायद ही \omterm{def:g12:newtons-laws:momentum}{गति}।
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:11}
$ T_0 = 2\pi\sqrt{0.50 \times \num{2.0e-5}} = 2\pi \times
\num{3.16e-3} \approx \qty{20}{ms}$. शब्दकोष: $ k = m/(LC) =
0.50/\num{1.0e-5} = \qty{5.0e4}{N/m}$; जाँच करना
$2\pi\sqrt{m/k} = 2\pi\sqrt{\num{1.0e-5}}$ --- जो उसी \qty{20}{ms}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:12}
$ C = \qty{500}{pF}$: $ f_0 = 1/(2\pi\sqrt{\num{1.0e-13}}) \approx
\qty{0.50}{MHz}$; $ C = \qty{50}{pF}$: $ f_0 \approx \qty{1.6}{MHz}$.
निश्चित पर $ L $, $ f_0 = (2\pi\sqrt{L})^{-1} C^{-1/2} \propto 1/\sqrt{C}$:
दस गुना $ C $, केवल $\sqrt{10} \approx 3.2$ में $ f_0$ ---जैसा पाया गया.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:13}
$0.95^n = \tfrac12$: $ n = \ln 2/\lvert\ln 0.95\rvert \approx 13.5$,
तो के बारे में $14$ \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}. \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} $\propto \sqrt{E}$: नीचे तक
$1/\sqrt{2} \approx 0.71$ शुरुआत का. पर \qty{1.0}{MHz},
$ T_0 = \qty{1.0}{\micro s}$: के बारे में \qty{14}{\micro s}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:14}
प्रति चक्र हानि $0.020 \times \qty{1.0e-9}{J} = \qty{2.0e-11}{J}$, पर
\num{32768} चक्र प्रति सेकंड: $ P = \num{32768} \times \num{2.0e-11}
\approx \qty{6.6e-7}{W}$ --- एक माइक्रो\omterm{def:g11:work-of-force:power}{वाट} के तहत, एक बटन पर वर्षों
सेल. \omterm{def:g12:oscillators-and-time:phase}{चरण} में, क्योंकि दाईं ओर केवल एक धक्का है \omterm{def:g12:oscillators-and-time:phase}{चरण} \emph{कहते हैं}
ऊर्जा (\omterm{def:g12:oscillators-and-time:resonance}{गूंज}); कदम से बाहर यह झूले को ब्रेक देगा।
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:15}
$ C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
(\num{2.5e7})^2 \times \num{1.0e-6}} \approx \qty{41}{pF}$. द
अवरोधक को लूप को अंदर धकेलना चाहिए \omterm{def:g12:rlc-oscillations:regimes}{एपेरियोडिक शासन}: आरोप
फिर ओवरशूटिंग के बिना \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} हो जाता है --- कोई दोलन नहीं, कोई घंटी नहीं।
\end{solution}

\begin{solution}{pb:g12:rlc-oscillations:1}
\textbf{1.} लूप कानून: $ u_L + u_C = 0$, साथ $ u_C = q/C $,
$ u_L = L\,di/dt $, $ i = dq/dt $: $ L\,d^2q/dt^2 + q/C = 0$.

\textbf{2.} $ d^2q/dt^2 = -(2\pi/T_0)^2 q $, तो समीकरण पढ़ता है
$ q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$, सभी के लिए सच है $ t $ बिल्कुल कब
$ T_0 = 2\pi\sqrt{LC}$.

\textbf{3.} $ LC $: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times
\unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ एक समय है.

\textbf{4.} $ LC = \qty{6.6e-14}{s^2}$: $ T_0 = 2\pi \times \num{2.57e-7}
\approx \qty{1.6}{\micro s}$, $ f_0 \approx \qty{620}{kHz}$ --- अंदर
\qtyrange{530}{1700}{kHz}.

\textbf{5.} $ C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
\num{1.0e12} \times \num{2.0e-4}} \approx \qty{127}{pF}$.

\textbf{6.} पर भी वही फार्मूला \qty{530}{kHz}: $ C \approx \qty{451}{pF}$.

\textbf{7.} पर \qty{1700}{kHz}: $ C \approx \qty{44}{pF}$.

\textbf{8.} $ C = \qty{480}{pF}$: $ f_0 \approx \qty{514}{kHz}$;
$ C = \qty{20}{pF}$: $ f_0 \approx \qty{2.5}{MHz}$ --- पूरा बैंड,
दोनों सिरों पर मार्जिन के साथ.

\textbf{9.} $ f_0 = (2\pi\sqrt{L})^{-1}C^{-1/2} \propto 1/\sqrt{C}$;
दोहरीकरण $ f_0$ विभाजन की आवश्यकता है $ C $ द्वारा $4$.

\textbf{10.} तब से $ f_0 \propto 1/\sqrt{C}$, के बराबर कदम $ C $ कदम
$ f_0$ धीरे-धीरे बड़े पैमाने पर $ C $ और छोटे स्तर पर तेज़ $ C $: एक दी गई रिक्ति
स्टेशनों के बीच फिर थोड़ी डायल यात्रा होती है --- वे भीड़ लगाते हैं
उच्च \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति} (कम-$ C $) अंत।

\textbf{11.} $ Q_0 = CU_0 = \num{1.27e-10} \times 0.020 \approx
\qty{2.5e-12}{C}$; $ E = \tfrac12 CU_0^2 \approx \qty{2.5e-14}{J}$.

\textbf{12.} $\tfrac12 LI_0^2 = E $: $ I_0 = U_0\sqrt{C/L} =
0.020\sqrt{\num{6.35e-7}} \approx \qty{16}{\micro A}$.

\textbf{13.} $0.96^n = \tfrac12$: $ n = \ln 2/\lvert\ln 0.96\rvert
\approx 17$ चक्र.

\textbf{14.} पर \qty{1.00}{MHz}, $ T_0 = \qty{1.0}{\micro s}$: अंगूठी
लगभग आधा \qty{17}{\micro s}. एक अकेली अंगूठी माइक्रोसेकंड में मर जाती है;
संगीत को लगातार लूप फीड की आवश्यकता होती है।

\textbf{15.}  \omterm{def:g10:signals-and-waves:wave}{लहर} प्रति सेकंड लाखों बार धक्का देता है; केवल जब
इसका \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति} माचिस $ f_0$ क्या पुश \omterm{def:g12:mechanical-waves:phase}{चरणबद्ध} तरीके से आते हैं और निर्माण करते हैं
झूला --- \omterm{def:g12:oscillators-and-time:resonance}{गूंज}; ऑफ-ट्यून स्टेशन अक्सर विपरीत दिशा में धक्का देते हैं
यह. ट्रांसमीटर पर, एक एम्पलीफायर रिटर्न देता है \omterm{def:g11:mechanical-energy:dissipation}{व्यस्त} ऊर्जा
प्रत्येक चक्र: एक बनाए रखा थरथरानवाला.

\textbf{16.} $ LC = \qty{2.54e-14}{s^2}$: $ k = m/(LC) =
0.20/\num{2.54e-14} \approx \qty{7.9e12}{N/m}$.

\textbf{17.} के बारे में $10^8$ कार स्प्रिंग का समय: कोई बेंच-टॉप नहीं
स्प्रिंग--द्रव्यमान प्रणाली मेगाहर्ट्ज़ तक पहुंचती है--रोजमर्रा की \omterm{def:g12:oscillators-and-time:spring}{कठोरता} और
जनता निराशाजनक रूप से धीमी है।

\textbf{18.} $ m = \dfrac{k}{(2\pi f_0)^2} = \dfrac{100}
{(\num{6.28e6})^2} \approx \qty{2.5e-12}{kg}$ --- कुछ नैनोग्राम, ए
धूल का कण.

\textbf{19.} एक क्वार्टज़ क्रिस्टल: एक पंखों वाला, अति-कठोर ज़ुल्फ़
इसके पर गाना \omterm{def:g12:rlc-oscillations:period}{प्राकृतिक आवृत्ति} --- टाइमकीपर इससे पहले मिले थे
वर्ष.

\textbf{20.} साथ $ L = \qty{0.20}{mH}$ और \qtyrange{20}{480}{pF} द
पाश घूमता रहता है \qtyrange{0.51}{2.5}{MHz}, बैंड को कवर करना;
$ C \approx \qty{127}{pF}$ उठाता है \qty{1.00}{MHz}; अंगूठी को अकेला छोड़ दिया
आधे में $\approx \qty{17}{\micro s}$, लेकिन स्टेशन का अपना \omterm{def:g10:signals-and-waves:wave}{लहर},
कदम-कदम पर गूंजता हुआ, लूप को दोबारा फीड करता है --- समाचार चलता है ऊर्जा
की \omterm{def:g10:signals-and-waves:wave}{लहर} स्वयं.
\end{solution}
