\begin{solution}{exo:g12:kinematics-2d:1}
$\vect v = (3.0, 4.0)$, स्थिर; \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $\sqrt{3.0^2 + 4.0^2} =
\qty{5.0}{m/s}$. \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}: पंक्ति $ y = \tfrac43 x $. स्थिर
वेग सदिश: एकसमान सीधीरेखीय गति.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:2}
सीधी रेखा, समान दूरी: \omterm{def:g12:kinematics-2d:uniform}{एकसमान सीधा};
$ v = 0.35/0.10 = \qty{3.5}{m/s}$. रिक्त स्थान खोलने का मतलब होगा
\omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} यह बढ़ रहा है।
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:3}

समान रूप से त्वरित सीधारेखीय। $ v(3.0) = \qty{12}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:4}
$ f = 1/8.0 = \qty{0.125}{Hz}$;
$ v = 2\pi \times 4.5/8.0 \approx \qty{3.5}{m/s}$;
$ a = v^2/R = 3.53^2/4.5 \approx \qty{2.8}{m/s^2}$, की ओर इशारा करते हुए
हिंडोला की धुरी.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:5}
ढलान: $ a = (10.0 - 2.0)/4.0 = \qty{2.0}{m/s^2}$. क्षेत्र (ट्रेपेज़ॉइड):
$ d = \tfrac12 (2.0 + 10.0) \times 4.0 = \qty{24}{m}$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:6}
$ v = \qty{88.9}{m/s}$: $ a_n = 88.9^2/6000 \approx \qty{1.3}{m/s^2}
\approx 0.13\,g $. तब से $ a_n = v^2/R $, उच्च $ v $ बहुत बड़ी मांग करता है $ R $ को
बग़ल में त्वरण रखें (और) बल ट्रैक पर और
यात्रियों) स्वीकार्य --- बैंकिंग परिवर्तन जो इसकी आपूर्ति करता है, उसकी नहीं
आकार.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:7}
$ v(t) = -2.4\,t + 8.0$; $ a = \qty{-2.4}{m/s^2}$, स्थिर। उलटा
पर $ v = 0$: $ t = 8.0/2.4 \approx \qty{3.3}{s}$, पर
$ x \approx \qty{16.3}{m}$. पहले: आगे, धीमा; बाद में: पि\omterm{def:g11:color-light-sources:cones}{छड़}ा,
गति बढ़ाना --- सर्वत्र समान रूप से त्वरित।
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:8}
$ v_1 = 0.28/0.40 = \qty{0.70}{m/s}$;
$ v_2 = (0.54 - 0.10)/0.40 = \qty{1.10}{m/s}$;
$ v_3 = (0.88 - 0.28)/0.40 = \qty{1.50}{m/s}$. वेग बढ़ता है
\qty{0.40}{m/s} प्रत्येक \qty{0.20}{s}: $ a = \qty{2.0}{m/s^2}$, स्थिर
--- समान रूप से त्वरित सीधा।
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:9}
$ T = \qty{2.36e6}{s}$: $ v = 2\pi \times \num{3.84e8}/\num{2.36e6}
\approx \qty{1.0e3}{m/s}$;
$ a_n = v^2/R \approx \qty{2.7e-3}{m/s^2}$. अनुपात:
$ g/a_n \approx 3600 = 60^2$ --- गुरुत्वाकर्षण, विपरीत के रूप में \omterm{def:g11:lenses-and-eye:lens}{पतला}
दूरी का वर्ग (\cref{ch:g10:universal-gravitation}), बिल्कुल
चंद्रमा के घूमने का कारण बनता है।
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:10}
$ f = 1200/60 = \qty{20}{Hz}$, $ T = \qty{0.050}{s}$;
$ v = 2\pi R f = 2\pi \times 0.25 \times 20 \approx \qty{31}{m/s}$;
$ a_n = v^2/R \approx \qty{3.9e3}{m/s^2} \approx 400\,g $. ढोल
दीवार कपड़े धोने की आपूर्ति करती है $ a_n $; छिद्रों के माध्यम से कुछ भी नहीं धकेलता
पानी अंदर की ओर है, इसलिए यह स्पर्शरेखा के साथ उड़ जाता है।
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:11}
$ v_0 = \qty{25}{m/s}$: $ t_s = 25/7.5 \approx \qty{3.3}{s}$;
$ d = v_0^2/(2a) = 625/15 \approx \qty{42}{m}$. ग्राफ़: एक सीधा
से खंड \qty{25}{m/s} नीचे तक $0$ पर $ t_s $.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:12}
$\vect v = (-R\omega\sin\omega t,\, R\omega\cos\omega t)$, का
स्थिर मानदंड $ R\omega $; इसके डॉट उत्पाद के साथ $\vect{OM}$ गायब हो जाता है,
इसलिए यह त्रिज्या के लंबवत है, इसलिए स्पर्शरेखा है। विभेद करना
फिर से, $\vect a = -\omega^2\vect{OM}$: की ओर $ O $, आदर्श का
$\omega^2 R = (v/R)^2 R = v^2/R $.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:13}

\qty{5.0}{m/s^2}$;
$ a = \sqrt{2.5^2 + 5.0^2} \approx \qty{5.6}{m/s^2}$; के साथ कोण

(अर्थात्\ $63^\circ $ सीधे पीछे से)। $5.6 < 7.0$: पकड़ के भीतर,
बहुत कम बचत के साथ.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:14}
शुरू करना: $ t_1 = 20/1.2 \approx \qty{17}{s}$,
$ d_1 = 20^2/(2 \times 1.2) \approx \qty{167}{m}$. समुद्र में यात्रा करना:
\qty{40}{s}, $\qty{800}{m}$. ब्रेक लगाना: $ t_3 = 20/1.5 \approx

\end{solution}

\begin{solution}{exo:g12:kinematics-2d:15}
$ v = 2\pi \times \num{4.22e7}/\num{86164} \approx \qty{3.1e3}{m/s}$;

\begin{solution}{pb:g12:kinematics-2d:1}
\textbf{1.} गति सापेक्ष है (\cref{ch:g10:relative-motion}): प्रत्येक
स्थिति और वेग को इसके फ्रेम का नाम देना चाहिए। यहाँ: ज़मीन (सड़क)
फ़्रेम, $ x $ पहले फ्रेम की स्थिति से सड़क के किनारे।

\textbf{2.} प्रत्येक अंतराल: $9.0/0.50 = \qty{18.0}{m/s}$ ---चार
समान मूल्य.

\textbf{3.} सीधी सड़क, समान समय में समान विस्थापन:
एकसमान सीधीरेखीय गति.

\textbf{4.} $ v_0 = \qty{18.0}{m/s} = \qty{64.8}{km/h}$.

\textbf{5.} $\vect a = \vect 0$.

\textbf{6.} $ v(t) = -a\,t + v_B $; $ x(t) = -\tfrac12 a t^2 + v_B\,t $.

\textbf{7.} $ v(t_s) = 0$ देता है $ t_s = v_B/a $; तब
$ d_b = -\tfrac12 a (v_B/a)^2 + v_B^2/a = v_B^2/(2a)$.

\textbf{8.} $ v_B = \sqrt{2 a d_b} = \sqrt{2 \times 6.0 \times 27} =
\sqrt{324} = \qty{18.0}{m/s}$ --- बिल्कुल फिल्माया गया \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}: द
ड्राइवर ने ब्रेक लगाने से पहले कभी गति धीमी नहीं की।

\textbf{9.} प्रतिक्रिया: $18.0 \times 1.0 = \qty{18}{m}$; कुल
$18 + 27 = \qty{45}{m}$.

\textbf{10.} $ d_b = v_B^2/(2a) \propto v_B^2$: दोहरीकरण $ v_B $
गुणा $ d_b $ द्वारा $4$. पर \qty{50}{km/h} ($\qty{13.9}{m/s}$):
$ d_b = 13.9^2/12.0 \approx \qty{16}{m}$.

\textbf{11.}  \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} स्थिर है लेकिन वेग की दिशा बदल जाती है:
$\vect v $ स्थिर नहीं है, इसलिए कार की गति बढ़ रही है।

\textbf{12.} तिमाही मोड़: $2\pi R/4 = \qty{19.6}{m}$;
$ v = 19.6/2.5 \approx \qty{7.9}{m/s}$ (\qty{28}{km/h}).

\textbf{13.} $ a_n = 7.85^2/12.5 \approx \qty{4.9}{m/s^2}$, की ओर
गोलचक्कर का केंद्र.

\textbf{14.} $ a_{\max} = \mu g = 0.70 \times 9.81 \approx
\qty{6.9}{m/s^2}$; $ v_{\max} = \sqrt{\mu g R} =
\sqrt{6.87 \times 12.5} \approx \qty{9.3}{m/s}$ (\qty{33}{km/h}).
$7.9 < 9.3$: पकड़ के भीतर --- स्वच्छ, स्किड-मुक्त के अनुरूप
लेन.

\textbf{15.} $ T = 2\pi R/v = 78.5/7.85 \approx \qty{10.0}{s};
f = \qty{0.10}{Hz}$.

\textbf{16.} $64.8 - 50 \approx \qty{15}{km/h}$ ओवर --- के बारे में
$30\%$ सीमा से ऊपर.

\textbf{17.} ब्रेक लगाना खत्म हो गया है $41 - 18 = \qty{23}{m}$:
$ v^2 = v_B^2 - 2ad = 324 - 2 \times 6.0 \times 23 = 48$, इसलिए
$ v \approx \qty{6.9}{m/s} \approx \qty{25}{km/h}$ प्रभाव में.

\textbf{18.} पर \qty{13.9}{m/s}: $13.9 \times 1.0 + 13.9^2/12.0 =
13.9 + 16.1 = \qty{30.0}{m} < \qty{41}{m}$ --- कार रुकती है
\qty{11}{m} वैन की कमी.

\textbf{19.} $ d(v) = v\,t_r + v^2/(2a)$: एक बार रैखिक रूप से (प्रतिक्रिया
कवर $ v t_r $ ब्रेक प्रभावी होने से पहले), एक बार चुकता (गतिज)।
भाग $ v^2/(2a)$) --- यही कारण है कि मध्यम अतिरिक्त लागत
अनुपातहीन रूप से.

\textbf{20.} डैशकैम और स्किड मार्क्स सहमत हैं: \qty{65}{km/h} में एक \qty{50}{km/h} क्षेत्र; कानूनी सीमा पर कार रुक गई होगी
\qty{11}{m} वैन की कमी --- दुर्घटना अतिरिक्त है
\qty{15}{km/h}.
\end{solution}
