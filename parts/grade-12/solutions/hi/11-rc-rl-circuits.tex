\begin{solution}{exo:g12:rc-rl-circuits:1}
$q = Cu = 4.7\times10^{-4} \times 12 \approx \qty{5.6}{mC}$;
$ u = q/C = 2.0\times10^{-3}/10^{-4} = \qty{20}{V}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:2}
\qty{47}{pF}: रेडियो ट्यूनर; \qty{2200}{\micro F}: चिकना; \qty{10}{F}:
सुपरकैपेसिटर $ E = \tfrac12 \times 10 \times 2.7^2 \approx \qty{36}{J}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:3}
$ i = C\,\Delta u/\Delta t = 2.2\times10^{-4} \times 5.0/0.010 =
\qty{0.11}{A}$. तब $ i = C\,du/dt = 0$: एक आरोप लगाया \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} एक खुला है
सर्किट --- कोई \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} धारा नहीं।
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:4}
$ RC = 10^4 \times 10^{-4} = \qty{1.0}{s}$;
$ L/R = 0.10/50 = \qty{2.0}{ms}$. $\unit{\ohm}\,\unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$;
$\unit{H}/\unit{\ohm} = (\unit{V.s/A})/(\unit{V/A}) = \unit{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:5}
$3.8/6.0 = 0.63$, इसलिए $\tau = \qty{2.0}{s}$; $95\%$ पर $3\tau =
\qty{6.0}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:6}
$\tau = 4700 \times 2.2\times10^{-4} \approx \qty{1.0}{s}$;
$ u(\tau) = 0.63 \times 9.0 = \qty{5.7}{V}$;
$ u(3\tau) = 0.95 \times 9.0 = \qty{8.6}{V}$;
$ I_0 = \mathcal{E}/R = 9.0/4700 \approx \qty{1.9}{mA}$ (खाली \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}:
के सभी $\mathcal{E}$ पार बैठता है $ R $).
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:7}
$ du/dt = (\mathcal{E}/RC)e^{-t/RC}$, इसलिए $ RC\,du/dt + u =
\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$.
$ u(0) = 0$: द \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} खाली शुरू होता है; $ u \to \mathcal{E}$: पूरी तरह से
चार्ज किया गया, करंट बंद हो गया है।
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:8}
$\tau = 4.7\times10^{4} \times 10^{-4} = \qty{4.7}{s}$;
$ u(\tau) = 12\,e^{-1} \approx \qty{4.4}{V}$; \qty{0.60}{V} है $5\%$,
पर पहुंच गया $3\tau \approx \qty{14}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:9}
$ u = L\,\Delta i/\Delta t = 0.50 \times 2.0/0.040 = \qty{25}{V}$.
समापन, $ i $ सर्किट की अपनी \omterm{def:g12:newtons-laws:momentum}{गति} से चढ़ता है, $ di/dt $ मामूली; उद्घाटन
\emph{बल} $ i $ लगभग तुरंत ही शून्य हो जाना, $ di/dt $ बहुत बड़ा है और $ u $
स्पार्किंग मूल्यों तक पहुँचता है।
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:10}
$ di/dt = (\mathcal{E}/L)e^{-Rt/L}$, इसलिए $ L\,di/dt + Ri =
\mathcal{E}e^{-Rt/L} + \mathcal{E}(1 - e^{-Rt/L}) = \mathcal{E}$.
$ i(\tau) = 0.63 \times 2.0 = \qty{1.3}{A}$;
$ i(3\tau) = 0.95 \times 2.0 = \qty{1.9}{A}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:11}
$ u'(t) = (\mathcal{E}/\tau)e^{-t/\tau}$, इसलिए $ u'(0) = \mathcal{E}/\tau $:
स्पर्शरेखा $ u = (\mathcal{E}/\tau)\,t $ पहुँचता है $\mathcal{E}$ पर
$ t = \tau $. $ C = \tau/R = 0.50/2000 = \qty{250}{\micro F}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:12}
supercapacitor $\tfrac12 \times 3000 \times 2.7^2 \approx \qty{1.1e4}{J}$;
चमक $\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$;
कुंडल $\tfrac12 \times 0.010 \times 10^2 = \qty{0.50}{J}$. बैटरी
($\qty{1.3e4}{J}$) $>$ supercapacitor $\gg $ चमक $\gg $ कुंडल --- लेकिन सभी
तीन अपना डंप कर सकते हैं ऊर्जा मिलीसेकंड में: \omterm{def:g11:work-of-force:power}{शक्ति}, नहीं ऊर्जा.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:13}
$\tau = \qty{5.0}{s}$, इसलिए $ R = \tau/C = 5.0/10^{-4} = \qty{50}{k\ohm}$.
$1 - e^{-t/\tau} = 0.50$ देता है $ e^{-t/\tau} = 0.50$, $ t = \tau\ln 2
\approx 0.69\,\tau = \qty{3.5}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:14}
$\Delta u = i\,\Delta t/C = 0.50 \times 0.010/2.2\times10^{-3} \approx
\qty{2.3}{V}$. ज़रूरत $ C \geq i\,\Delta t/\Delta u =
5.0\times10^{-3}/0.50 = \qty{10}{mF}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:15}
$ u \approx L\,\Delta i/\Delta t = 0.80 \times 2.0/10^{-3} =
\qty{1.6}{kV}$; $ E_L = \tfrac12 \times 0.80 \times 2.0^2 =
\qty{1.6}{J}$. धीरे धीरे काटो, $ di/dt $ छोटा रहता है, $ u $ कुछ रहता है \omterm{def:g11:circuits-and-power:voltage}{वोल्ट}:
कोई चिंगारी नहीं.
\end{solution}

\begin{solution}{pb:g12:rc-rl-circuits:1}
\textbf{1.} $ q = CU_0 = 8.0\times10^{-4} \times 300 = \qty{0.24}{C}$.

\textbf{2.} $ N = 0.24/1.60\times10^{-19} = \num{1.5e18}$.

\textbf{3.} $ E_C = \tfrac12 \times 8.0\times10^{-4} \times 300^2 =
\qty{36}{J}$.

\textbf{4.} $\num{1.3e4}/36 \approx 360$ बार.

\textbf{5.} कोशिका प्रचुर मात्रा में भंडारण करती है लेकिन वितरण धीरे-धीरे करती है; \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}
विनम्र हो जाता है ऊर्जा विशाल में \emph{\omterm{def:g11:work-of-force:power}{शक्ति}}.

\textbf{6.} $\mathcal{E} = Ri + u $ साथ $ i = C\,du/dt $:
$ RC\,du/dt + u = \mathcal{E}$.

\textbf{7.} $ du/dt = (\mathcal{E}/RC)e^{-t/RC}$:
$\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$
सभी के लिए $ t $, और $ u(0) = 0$.

\textbf{8.} $\tau = RC = 10^3 \times 8.0\times10^{-4} = \qty{0.80}{s}$.

\textbf{9.} पर $3\tau = \qty{2.4}{s}$, $ u = 0.95 \times 300 =
\qty{285}{V}$ --- दो सेकंड की कराह।

\textbf{10.} $ I_0 = \mathcal{E}/R = \qty{0.30}{A}$; जैसा $ u $ चढ़ता है,
शेष $\mathcal{E} - u $ आर-पार $ R $ सिकुड़ता है, इसलिए $ i = (\mathcal{E}-u)/R $
मर जाता है.

\textbf{11.} $ u(t) = U_0\,e^{-t/\tau'}$ साथ $\tau' = R_f C =
0.50 \times 8.0\times10^{-4} = \qty{0.40}{ms}$.

\textbf{12.} $ I_0 = U_0/R_f = 300/0.50 = \qty{600}{A}$.

\textbf{13.} $ P_0 = U_0 I_0 = 300 \times 600 = \qty{180}{kW}$ --- के बारे में
$80$ एक ही बार में केतली.

\textbf{14.} \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} $3\tau' = \qty{1.2}{ms}$; औसत \omterm{def:g11:work-of-force:power}{शक्ति}
$\approx 36/1.2\times10^{-3} = \qty{30}{kW}$.

\textbf{15.} $2.4/1.2\times10^{-3} = 2000$. केवल \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} बदल गया
(\qty{1.0}{k\ohm} में, \qty{0.50}{\ohm} बाहर): ए \omterm{def:g11:work-of-force:power}{शक्ति} लीवर.

\textbf{16.} $ C = 2E_C/U_0^2 = 400/1500^2 \approx \qty{1.8e-4}{F} =
\qty{180}{\micro F}$.

\textbf{17.} $ q = CU_0 = 1.78\times10^{-4} \times 1500 \approx
\qty{0.27}{C}$.

\textbf{18.} $\tau = 50 \times 1.78\times10^{-4} \approx \qty{8.9}{ms}$,
\omterm{def:g12:oscillators-and-time:oscillator}{अवधि} $3\tau \approx \qty{27}{ms}$ --- कुछ बीस बार फ्लैश।

\textbf{19.} $ I_0 = 1500/50 = \qty{30}{A}$;
$ P_0 = 1500 \times 30 = \qty{45}{kW}$.

\textbf{20.} $3\tau_c = \qty{6.0}{s}$ देता है $\tau_c = \qty{2.0}{s}$, इसलिए
$ R_c = \tau_c/C = 2.0/1.78\times10^{-4} \approx \qty{11}{k\ohm}$ और
$ I_0 = 1500/1.12\times10^{4} \approx \qty{0.13}{A}$ --- छह में तैयार
सेकंड, एक के दसवें हिस्से पर एम्पेयर.
\end{solution}
