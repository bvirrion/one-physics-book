\begin{solution}{exo:g12:oscillators-and-time:1}

$ f_0 = 1/T_0 = \qty{0.50}{Hz}$. (बी) $ T_0 \propto \sqrt{L}$: द
\omterm{def:g12:oscillators-and-time:oscillator}{अवधि} दोगुना.
\end{solution}

\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:3}

दोहरीकरण $ m $: $ T_0 \times \sqrt{2} \approx 1.4$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:4}
\omterm{def:g12:oscillators-and-time:damping}{छद्म काल} \qty{2.0}{s}. अनुपात $1.5/2.0 = 1.13/1.5 = 0.85/1.13
= 0.75$ प्रत्येक \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}; अगला अधिकतम $0.85 \times 0.75 \approx
\qty{0.64}{cm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:5}
पेंडुलम: $ T_0 \propto 1/\sqrt{g}$, इसलिए $\times\sqrt{9.81/1.62} =
2.46$ (और धीमा)। वसंत--द्रव्यमान: अपरिवर्तित --- नहीं $ g $ में
$2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:6}
(ए) $[L/g] = \unit{m}/(\unit{m/s^2}) = \unit{s^2}$; इसका वर्गमूल
एक समय है. (बी) द \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} में ही दिखाई देता है $ m $: और कुछ नहीं हो सकता
इसे रद्द करो, इसलिए $ m $ प्रवेश नहीं किया जा सकता। (सी) कभी नहीं: शुद्ध संख्या की तरह $2\pi $
\omterm{def:g12:oscillators-and-time:oscillator}{आयाम}ों के लिए अदृश्य हैं।
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:7}
$\frac{dx}{dt} = -\frac{2\pi}{T_0}A\sin(2\pi t/T_0 + \varphi)$, इसलिए
$\frac{d^2x}{dt^2} = -(2\pi/T_0)^2 x(t)$. के साथ समानता $-(k/m)x $
सभी के लिए $ t $ बिल्कुल कब धारण करता है $(2\pi/T_0)^2 = k/m $, अर्थात\
$ T_0 = 2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:8}
$ E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.030^2 =
\qty{1.8e-2}{J}$; $ v_{\max} = \sqrt{2E_m/m} = \sqrt{0.18} =
\qty{0.42}{m/s}$ --- पर $ x = 0$; चरम सीमा पर शून्य $ x = \pm A $.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:9}
(ए) $ L = gT_0^2/4\pi^2 = \qty{0.994}{m}$. (बी) $\Delta T_0/T_0 =
\tfrac12 \times 1.0\% = 0.5\%$: $86400 \times 0.005 \approx
\qty{4.3e2}{s}$ --- प्रतिदिन सात मिनट धीमी \omterm{def:g12:newtons-laws:momentum}{गति} से।
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:10}
(ए) $ A \propto \sqrt{E}$: $\sqrt{0.95} \approx 0.975$, ए $2.5\%$
बूँद। (बी) $0.95^n = 0.5$: $ n = \ln 2/\ln(1/0.95) \approx 14$
\omterm{def:g12:oscillators-and-time:oscillator}{अवधि}. (सी) नहीं: प्रकाश के लिए भिगोना यह a के एक अंश के भीतर रहता है
का प्रतिशत $ T_0$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:11}
(ए) पर $ f_0$ प्रत्येक धक्का \omterm{def:g12:mechanical-waves:phase}{चरणबद्ध} तरीके से आता है और जुड़ता है ऊर्जा; शिखर से बाहर,
धक्का बारी-बारी से मदद और बाधा डालता है। (बी) कदम से कदम मिलाकर चलना है
\omterm{def:g10:signals-and-waves:period}{आवधिक} गाड़ी चलाना; पुल के पास $ f_0$ यह चढ़ जाएगा \omterm{def:g12:oscillators-and-time:resonance}{गूंज}
चोटी। (सी) ड्रम फ्रेम के माध्यम से घूमता है \omterm{def:g12:oscillators-and-time:resonance}{प्राकृतिक आवृत्ति}
स्पिन-अप के दौरान: क्षणिक \omterm{def:g12:oscillators-and-time:resonance}{गूंज}, फिर चरम से आगे।
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:12}
(ए) $ T_0 \propto 1/\sqrt{g}$: धीरे, द्वारा $\Delta T_0/T_0 = \tfrac12
\times 0.03/9.81 = \num{1.5e-3}$, यानी\ $\approx \qty{132}{s}$ प्रति दिन। (बी) $ L \propto g $ निश्चित पर $ T_0$: द्वारा छोटा करें $0.994 \times
0.03/9.81 \approx \qty{3.0}{mm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:13}
$ v_{\max} = 2\pi A/T_0 = 2\pi \times 0.040/0.63 = \qty{0.40}{m/s}$;
$ a_{\max} = (2\pi/T_0)^2 A = \qty{4.0}{m/s^2}$; $ k = m(2\pi/T_0)^2
= \qty{20}{N/m}$; $ E_m = \tfrac12 kA^2 = \qty{1.6e-2}{J}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:14}
$ T_0 = 40.1/20 = \qty{2.005}{s}$; $ g = 4\pi^2 L/T_0^2 = 4\pi^2
\times 0.995/4.020 = \qty{9.77}{m/s^2}$. अनिश्चितता $\approx 0.2\%
+ 2 \times 0.5\% = 1.2\%$, यानी\ $\pm\qty{0.12}{m/s^2}$: द
अंतराल शामिल है \qty{9.81}{m/s^2} --- \omterm{def:g12:light-as-wave:coherent}{सुसंगत}।
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:15}

\approx \num{1.9e-7}$; सीज़ियम $1/\num{3.2e15} \approx \num{3e-16}$.

\begin{solution}{pb:g12:oscillators-and-time:1}
\textbf{1.}  $\approx \qty{0.2}{s}$ प्रतिक्रिया त्रुटि को विभाजित किया गया है
$50$: के बारे में \qty{4}{ms} पर $ T_0$.

\textbf{2.} $ T = 99.4/50 = \qty{1.988}{s}$.

\textbf{3.} $ L = gT^2/4\pi^2 = 9.81 \times 1.988^2/4\pi^2 \approx
\qty{0.982}{m}$.

\textbf{4.} $ L = \qty{0.994}{m}$ (सेकंड पेंडुलम): नीचे करें
बॉब, द्वारा लंबा करना $\approx \qty{1.2}{cm}$.

\textbf{5.} बहुत छोटा, इतना तेज़: तेज़, द्वारा $(2.000 -
1.988)/1.988 \times 86400 \approx \qty{5.2e2}{s} \approx 8.7$
प्रति दिन मिनट.

\textbf{6.} चंद्रमा: $\times\sqrt{9.81/1.62} = 2.46$ --- अनुपयोगी.
वेधशाला: $\Delta T_0/T_0 = \tfrac12 \times 0.01/9.81 =
\num{5.1e-4}$: हार जाता है $\approx \qty{44}{s}$ प्रति दिन।

\textbf{7.} $\Delta L = L\lambda\,\Delta\theta = 0.994 \times
\num{1.9e-5} \times 5.0 \approx \qty{9.4e-5}{m} \approx
\qty{0.1}{mm}$.

\textbf{8.} $ T' = 2\pi\sqrt{L(1 + \lambda\Delta\theta)/g} =
T_0\sqrt{1 + \lambda\Delta\theta} \approx T_0\,(1 + \tfrac12
\lambda\Delta\theta)$.

\textbf{9.} $\Delta T_0/T_0 = \tfrac12 \times \num{1.9e-5} \times
5.0 = \num{4.8e-5}$: $86400 \times \num{4.8e-5} \approx
\qty{4.1}{s}$ प्रति दिन, धीमी \omterm{def:g12:newtons-laws:momentum}{गति} से (लंबी \omterm{def:g11:color-light-sources:cones}{छड़}ी, लंबी \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}).

\textbf{10.} $\Delta\theta = \qty{-10}{K}$: $\Delta T_0/T_0 =
\num{-9.5e-5}$, लाभ $\approx \qty{8.2}{s}$ प्रति दिन।

\textbf{11.} स्टील और पीतल का विस्तार अलग-अलग होता है; बारी-बारी से लटकाया गया
दिशा-निर्देश, पीतल की \omterm{def:g11:color-light-sources:cones}{छड़}ें बॉब को नीचे करती हैं जबकि स्टील की \omterm{def:g11:color-light-sources:cones}{छड़}ें ऊपर उठाती हैं
यह. आकार इसलिए कि दो शिफ्ट रद्द हो जाएं, प्रभावी लंबाई --- और
दर --- मौसमों में जीवित रहता है।

\textbf{12.} $280/2.000 = 140$ \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}.

\textbf{13.} $\theta_m = 2.00^\circ = \qty{0.0349}{rad}$: $ E_m =
\tfrac12 \times 1.2 \times 9.81 \times 0.994 \times 0.0349^2 \approx
\qty{7.1e-3}{J}$.

\textbf{14.} प्रति \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}  \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} से सिकुड़ जाता है $2^{-1/140} \approx
0.9951$, द ऊर्जा द्वारा $0.9951^2 \approx 0.990$: के बारे में $1.0\%$ का
$ E_m $ प्रति \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}.

\textbf{15.} धकेलना $\approx 0.010 \times \num{7.1e-3} =
\qty{7.1e-5}{J}$; \omterm{def:g11:work-of-force:power}{शक्ति} $\num{7.1e-5}/2.0 \approx \qty{3.5e-5}{W}$
---दसियों माइक्रो\omterm{def:g11:work-of-force:power}{वाट}.

\textbf{16.} एक सप्ताह है $604800/2.0 = \num{3.02e5}$ \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}, जरूरत है
$\num{3.02e5} \times \num{7.1e-5} \approx \qty{21}{J}$; \omterm{def:g10:universal-gravitation:weight}{वज़न}
आपूर्ति $ Mgh = 2.5 \times 9.81 \times 1.0 = \qty{24.5}{J}$: बहुत हो गया,
पर $21/24.5 \approx 88\%$ \omterm{def:g10:energy-conservation:efficiency}{क्षमता}.

\textbf{17.} $ T = 1/\num{32768} = \qty{30.5}{\micro s}$. $\num{32768}
= 2^{15}$: पंद्रह विभाजन-दर-दो \omterm{def:g12:oscillators-and-time:phase}{चरण} क्रिस्टल को घुमाते हैं \omterm{def:g10:signals-and-waves:signal}{संकेत}
एक सटीक में \qty{1}{Hz} सही का निशान लगाना।

\textbf{18.} क्वार्ट्ज: $0.3/\num{2.6e6} \approx \num{1.2e-7}$.
पेंडुलम: $0.5/86400 \approx \num{5.8e-6}$, यानी\ $\approx
\qty{15}{s}$ प्रति महीने। क्वार्ट्ज़ एक कारक से जीतता है $\approx 50$.

\textbf{19.} $1/\num{3.2e15} \approx \num{3e-16}$: के नौ आदेश
क्वार्टज़ से कम परिमाण. जीपीएस समय सिग्नल नैनोसेकंड तक ---
\qty{30}{cm} प्रकाश की प्रत्येक यात्रा --- इसलिए केवल परमाणु घड़ियाँ ही करेंगी।

\textbf{20.} बहाल, दीवार घड़ी कायम है $\pm\qty{15}{s}$ प्रति
क्वार्ट्ज के खिलाफ महीना $\pm\qty{0.3}{s}$: इसे झंकार के लिए रखें,
ट्रेन के लिए क्वार्ट्ज पर भरोसा करें --- और \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} अब स्वयं का है
सीज़ियम को \omterm{def:g11:fundamental-interactions:ladder}{एटम}.
\end{solution}
