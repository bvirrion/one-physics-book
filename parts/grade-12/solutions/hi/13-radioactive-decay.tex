\section*{अध्याय \ref{ch:g12:radioactive-decay} --- रेडियोधर्मी क्षय}

\begin{solution}{exo:g12:radioactive-decay:1}
$T_{1/2} = \ln 2/\lambda = 0.693/\num{1.0e-6} = \qty{6.9e5}{s}
\approx 8.0$ दिन---आयोडीन-131 की घड़ी।
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:2}
$ T_{1/2} = 3.8 \times \num{86400} = \qty{3.28e5}{s}$;
$\lambda = 0.693/\num{3.28e5} = \qty{2.1e-6}{s^{-1}}$;
$\mathcal A = \lambda N = \num{2.1e-6} \times \num{1.0e10} \approx
\qty{2.1e4}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:3}
$\dd N/\dd t = N_0(-\lambda)e^{-\lambda t} = -\lambda N $ ---
समीकरण कायम है; $ N(0) = N_0 e^0 = N_0$;
$ N(T_{1/2})/N_0 = e^{-\lambda \ln 2/\lambda} = e^{-\ln 2} =
\tfrac12$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:4}
$90$ साल $= 3\,T_{1/2}$: $\mathcal A = \num{4.0e5}/2^3 =
\qty{5.0e4}{Bq}$. $300$ साल $= 10\,T_{1/2}$:
$\mathcal A = \num{4.0e5}/1024 \approx \qty{3.9e2}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:5}
ढलान $= (18.85 - 20.00)/10 = -0.115$, इसलिए
$\lambda = \qty{0.115}{h^{-1}} = 0.115/3600 = \qty{3.2e-5}{s^{-1}}$;
$ T_{1/2} = 0.693/0.115 = 6.0$ घंटे --- टेक्नेटियम-99 मी.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:6}
$ N = (1.0/226) \times \num{6.02e23} = \num{2.7e21}$;
$ T_{1/2} = 1600 \times \num{3.156e7} = \qty{5.05e10}{s}$, इसलिए
$\lambda = \qty{1.37e-11}{s^{-1}}$;
$\mathcal A = \lambda N \approx \qty{3.7e10}{Bq}$ --- एक क्यूरी, द
\omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} मैरी क्यूरी के ग्राम रेडियम का।
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:7}
दोनों को उम्मीद है कि आधा हिस्सा बचेगा: $10$, और \num{1.0e20}. उतार चढ़ाव
$\sim 1/\sqrt{n}$ गिनती का: साथ $10$ क्षय, के बारे में $30\%$ ---
$7$ या $13$ कोई आश्चर्य नहीं होगा; साथ \num{1.0e20}, के बारे में
$10^{-10}$. केवल बड़ा नमूना ही मापने योग्य कानून का पालन करता है
परिशुद्धता.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:8}
$ m = 140 \times \num{1.2e-4} = \qty{1.7e-2}{g}$;
$ N = (0.0168/40) \times \num{6.02e23} = \num{2.5e20}$.
$\lambda = 0.693/(\num{1.25e9} \times \num{3.156e7}) =
\qty{1.76e-17}{s^{-1}}$; $\mathcal A = \lambda N \approx
\qty{4.4e3}{Bq}$ --- शरीर के कुल का लगभग आधा (बाकी का अधिकांश भाग है
कार्बन-14).
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:9}
$\lambda = 0.693/6.0 = \qty{0.116}{h^{-1}}$. पर $15$ घंटे:
$\mathcal A = 500\,e^{-0.116 \times 15} = 500 \times 0.177 \approx
\qty{88}{MBq}$. नीचे \qty{1}{MBq}:
$ t = \ln(500)/0.116 \approx \qty{54}{hours}$ --- के बारे में $9$
आधा जीवन.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:10}
लकड़ी का कोयला: $3.4/13.6 = \tfrac14$, बिल्कुल $2$ आधा जीवन:
$ t = \num{1.15e4}$ साल। हड्डी:
$ t = \num{5730} \times \ln(1/0.30)/\ln 2 = \num{5730} \times 1.74
\approx \num{1.0e4}$ साल।
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:11}
$\lambda = 0.693/3.8 = \qty{0.182}{d^{-1}}$;
$ t = \ln(100)/\lambda = 4.61/0.182 \approx 25$ दिन (लगभग) $6.6$
आधा जीवन)। लेकिन आसपास के ग्रेनाइट में यूरेनियम श्रृंखला और
मिट्टी लगातार रेडॉन उत्पन्न करती है: भीतर एक बार का प्रसारण पूर्ववत हो जाता है
सप्ताह, इसलिए वेंटिलेशन स्थायी होना चाहिए।
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:12}
$ N/N_0 = e^{-\lambda t}$ देता है $\lambda t = \ln(N_0/N)$, और
$\lambda = \ln 2/T_{1/2}$ देता है $ t = T_{1/2}\ln(N_0/N)/\ln 2$. यहाँ
$ t = \num{5730} \times \ln(20)/\ln 2 = \num{5730} \times 4.32
\approx \num{2.5e4}$ साल।
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:13}
$ N_{\mathrm K} = N_0 e^{-\lambda t}$ और
$ N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, इसलिए
$ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$. अनुपात $3.0$:
$ e^{\lambda t} = 4$, इसलिए $\lambda t = 2\ln 2$ और
$ t = 2\,T_{1/2} = \num{2.5e9}$ साल।
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:14}
$\ln \mathcal A = 5.99$, $5.42$, $4.84$, $4.26$, $3.69$: की बूंदों
$0.57$--$0.58$ प्रति \qty{60}{s} --- संरेखित. ढलान:
$\lambda = 2.30/240 = \qty{9.6e-3}{s^{-1}}$;
$ T_{1/2} = 0.693/\num{9.6e-3} \approx \qty{72}{s}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:15}
(ए) $2^{-10} \approx \num{1e-3}$. (बी) $(1/2)^8 = 1/256 \approx
0.4\%$. (सी) $1000/8 = 125$ अपेक्षित---लेकिन कानून इस पर मौन है
जो: यह आबादी की सटीक भविष्यवाणी करता है और व्यक्तियों की बिल्कुल नहीं।
\end{solution}

\begin{solution}{pb:g12:radioactive-decay:1}
\textbf{1.} वातावरण के साथ आदान-प्रदान (खाना, सांस लेना) होता रहता है
वायुमंडलीय मूल्य पर अनुपात सबसे ऊपर; मौत रोक देती है
सेवन और क्षय हावी हो जाता है।

\textbf{2.} $ T_{1/2} = \num{5730} \times \num{3.156e7} =
\qty{1.81e11}{s}$, इसलिए $\lambda = 0.693/\num{1.81e11} =
\qty{3.8e-12}{s^{-1}}$.

\textbf{3.} $\dd N/\dd t = -\lambda N_0 e^{-\lambda t} = -\lambda N $,
और $ N(0) = N_0$.

\textbf{4.} $\mathcal A_0 = 13.6/60 = \qty{0.227}{Bq}$ प्रति ग्राम;
$ N = \mathcal A_0/\lambda = 0.227/\num{3.8e-12} \approx \num{5.9e10}$
परमाणुओं प्रति ग्राम.

\textbf{5.} $(1/12) \times \num{6.02e23} = \num{5.0e22}$ कार्बन
परमाणुओं; अनुपात $\num{5.9e10}/\num{5.0e22} \approx \num{1.2e-12}$
--- एक में $10^{12}$, वादे के अनुसार।

\textbf{6.} $\mathcal A = \mathcal A_0 e^{-\lambda t}$ देता है
$\lambda t = \ln(\mathcal A_0/\mathcal A)$, और $\lambda =
\ln 2/T_{1/2}$ में बदल देता है
$ t = T_{1/2}\ln(\mathcal A_0/\mathcal A)/\ln 2$.

\textbf{7.} $ t = \num{5730} \times \ln(13.6/9.1)/\ln 2 =
\num{5730} \times 0.580 \approx \num{3.3e3}$ साल।

\textbf{8.} लकड़ी की मृत्यु---पेड़ की कटाई, नहीं
नक्काशी; पुरानी लकड़ी से काटी गई मूर्ति (या प्राचीन लकड़ी से बनाई गई नकली मूर्ति)।
लकड़ी) अपनी सामग्री की तारीख से पहले या बाद की तारीख बताती है।

\textbf{9.} $ t = \num{5730} \times \ln(13.6/10.4)/\ln 2 =
\num{5730} \times 0.387 \approx \num{2.2e3}$ वर्ष---पूरी तरह से
बाईस शताब्दी पूर्व के एक लेखक के अनुरूप।

\textbf{10.} $\Delta t \approx \Delta\mathcal A/(\lambda\mathcal A)$
साथ $\lambda = \ln 2/5730 = \num{1.21e-4}$ प्रति वर्ष:
$\Delta t = (0.2/10.4)/\num{1.21e-4} \approx \num{1.6e2}$ वर्ष ---
एक या दो सदी के लिए अच्छी तारीख।

\textbf{11.} $ t = \num{5730} \times \ln(13.6/13.4)/\ln 2 \approx
120$ वर्ष --- आधुनिक के भीतर से अप्रभेद्य $\pm 160$-वर्ष
अनिश्चितता: चर्मपत्र ताज़ा है, ``प्राचीन'' नकली है।

\textbf{12.} $13.6/2^n < 0.2$ आवश्यकताओं $2^n > 68$: $ n = 7$ आधा जीवन
(तब से $2^7 = 128$), के बारे में $\num{4.0e4}$ साल।

\textbf{13.} दस अर्ध-जीवन: के बारे में \num{5.7e4} वर्ष ---
व्यावहारिक क्षितिज, लगभग साठ हजार वर्ष।

\textbf{14.} $\num{6.6e7}/\num{5730} \approx \num{11500}$ आधा जीवन.
एक ग्राम का \num{5.9e10} परमाणुओं के बाद थक गए हैं
$2^n > \num{5.9e10}$, यानी\ $ n \approx 36$ आधा जीवन
($\approx \num{2e5}$ वर्ष): डायनासोर की उम्र से बहुत पहले, एक नहीं
कार्बन-14 \omterm{def:g11:fundamental-interactions:ladder}{एटम} बाकी है--गिनने को कुछ नहीं है.

\textbf{15.} बहुत धीमी घड़ी: पोटेशियम-40 या यूरेनियम-238, के साथ
अरबों वर्षों का अर्ध-जीवन।

\textbf{16.} पिघला हुआ लावा आर्गन को बुलबुला बना देता है, इसलिए ताज़ा
ठोस क्रिस्टल में पोटेशियम होता है लेकिन शून्य आर्गन: अनुपात शुरू होता है
पर $0$ जमने पर --- घड़ी शून्य हो जाती है।

\textbf{17.} जैसा कि अभ्यास में है: $ N_{\mathrm K} = N_0 e^{-\lambda t}$,
$ N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, इसलिए
$ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.

\textbf{18.} $ e^{\lambda t} = 1.60$:
$ t = \num{1.25e9} \times \ln(1.60)/\ln 2 = \num{1.25e9} \times 0.678
\approx \num{8.5e8}$ साल।

\textbf{19.} नहीं, दो बार: पर \num{8.5e8} वर्षों तक कार्बन-14 चला
$10^5$ आधा जीवन --- मौन; पर \num{3.3e3} वर्षों में पोटैशियम-40 है
भागो \num{2.6e-6} एक का \omterm{def:g11:nucleus-radioactivity:halflife}{हाफ लाइफ} --- आर्गन अनुपात
$\approx \lambda t \sim \num{2e-6}$ बेहद छोटा है. प्रत्येक
घड़ी केवल ए के दसवें हिस्से से ही पढ़ती है \omterm{def:g11:nucleus-radioactivity:halflife}{हाफ लाइफ} लगभग दस तक.

\textbf{20.} मूर्ति $\approx \num{3.3e3}$ वर्ष, पपीरस
$\approx \num{2.2e3}$ साल, रॉक $\approx \num{8.5e8}$ वर्ष ---
तीन घड़ियाँ, एक कानून: $ N = N_0\,e^{-\lambda t}$, पीछे की ओर पढ़ें।
\end{solution}
