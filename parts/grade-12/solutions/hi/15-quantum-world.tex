\begin{solution}{pb:g12:quantum-world:1}
\textbf{1.} $ E = hc/\lambda = \num{1.99e-25}/\num{5.50e-7} =
\qty{3.62e-19}{J} = \qty{2.26}{eV}$.

\textbf{2.} $ N = \num{1.0e3}/\num{3.62e-19} \approx \num{2.8e21}$
\omterm{def:g12:quantum-world:photon}{फोटॉनों} प्रति सेकंड.

\textbf{3.} $ P = 0.20 \times \qty{1.0e3}{W} = \qty{200}{W}$;
$ E = 200 \times 5.0 = \qty{1.0}{kW.h}$ ($\qty{3.6e6}{J}$).

\textbf{4.} साथ $10^{21}$ प्रति सेकंड अनाज, सांख्यिकीय
झिलमिलाहट काल्पनिक रूप से छोटी है: दाने का औसत पूर्णता में बदल जाता है
चिकनी धारा, जैसे तेजी से डाली गई रेत पानी की तरह बहती है।

\textbf{5.} क्षति के लिए एक कण की आवश्यकता होती है \qtyrange{3}{4}{eV}: \omterm{def:g10:light-spectra:wavelength}{पराबैंगनी}
अनाज योग्य है, \omterm{def:g10:light-spectra:wavelength}{अवरक्त} अनाज ($\sim \qty{0.1}{eV}$) कभी मत करो ---
संख्याएँ प्रति दाने पंच का स्थान नहीं ले सकतीं।

\textbf{6.} $ U_s $ उलटा है \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} वह सबसे तेजी से रुकता है
इलेक्ट्रॉन: $ eU_s = E_{k,\max} = h\nu - W_0$.

\textbf{7.} $\nu = c/\lambda $: $8.22$, $7.41$, $6.88$ और $5.49$
$\times\,10^{14}\,\unit{Hz}$.

\textbf{8.} $ eU_s = h\nu - W_0$ में एफ़िन है $\nu $ ढलान के साथ $ h $.
क्रमिक ढलान: $(0.94 - 0.37)/1.39 = 0.41$ और
$(1.50 - 0.94)/1.34 = 0.42$ \unit{eV} प्रति $10^{14}\,\unit{Hz}$ ---
तीन बिंदु संरेखित हैं।

\textbf{9.} $ h = e(U_{s,1}-U_{s,4})/(\nu_1-\nu_4) = \num{1.60e-19}
\times 1.13/\num{2.72e14} = \qty{6.64e-34}{J.s}$.

\textbf{10.} पर \qty{546}{nm}: $ W_0 = h\nu - eU_s = \num{3.64e-19} -
\num{0.59e-19} = \qty{3.05e-19}{J} = \qty{1.90}{eV}$;
$\nu_0 = W_0/h = \qty{4.60e14}{Hz}$; $\lambda_0 \approx \qty{653}{nm}$.

\textbf{11.} केवल $\lambda < \qty{653}{nm}$ बाहर निकालता है: बैंगनी से लाल
दृश्यमान और \omterm{def:g10:light-spectra:wavelength}{पराबैंगनी}. संपूर्ण \omterm{def:g10:light-spectra:wavelength}{अवरक्त} पक्ष---लगभग आधा
सूर्य का ऊर्जा --- इस कैथोड के लिए बेकार है.

\textbf{12.} $(6.64 - 6.63)/6.63 \approx 0.2\%$ --- एक दो-परमिली
एक सार्वभौमिक \omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक का भौतिकी-प्रयोगशाला माप।

\textbf{13.} $ E_2 = \qty{-3.40}{eV}$, $ E_3 = \qty{-1.51}{eV}$,
$ E_4 = \qty{-0.85}{eV}$, $ E_5 = \qty{-0.54}{eV}$.

\textbf{14.} $3 \to 2$: \qty{1.89}{eV}; $4 \to 2$: \qty{2.55}{eV};
$5 \to 2$: \qty{2.86}{eV}.

\textbf{15.} $\lambda = 1240/E $: \qty{656}{nm} (लाल), \qty{486}{nm}
(नीले हरे), \qty{434}{nm} (बैंगनी)।

\textbf{16.}  एटमकी ऊर्जाएं हैं \omterm{def:g12:quantum-world:levels}{मात्रा निर्धारित}, इसलिए यह केवल उत्सर्जन कर सकता है
असतत मतभेद $ E_p - E_n $: एक सीढ़ी रेखाएँ बनाती है, एक नहीं
इंद्रधनुष.

\textbf{17.} नहीं: \qty{500}{nm} मतलब \qty{2.48}{eV}, जो संख्या से मेल खाता है
स्तर का अंतर ($ E_2$ ऊपर की ओर देता है $1.89$, $2.55$, $2.86$,
\qty{3.40}{eV}; से $ E_1$ हर चीज़ की कीमत कम से कम होती है \qty{10.2}{eV})
--- वहां गैस पारदर्शी है.

\textbf{18.} $ eU = \tfrac12 m_e v^2$: $ v = \sqrt{2 \times
\num{1.60e-19} \times \num{1.0e4}/\num{9.11e-31}} \approx
\qty{5.9e7}{m/s} \approx 0.20\,c $ --- शास्त्रीय उपचार स्वीकार्य.

\textbf{19.} $ p = m_e v = \qty{5.4e-23}{kg.m/s}$;
$\lambda = h/p = \qty{1.2e-11}{m} = \qty{12}{pm}$.

\textbf{20.} \omterm{def:g10:signals-and-waves:oscilloscope}{पाना} $= \num{5.50e-7}/\num{1.23e-11} \approx \num{4.5e4}$.
सप्ताहांत के दो नंबर: $ h = \qty{6.64e-34}{J.s}$, को मापा गया
$0.2\%$ साथ फिल्टर और एक वाल्टमीटर, और एक माइक्रोस्कोप
इलेक्ट्रॉनों के कारण पैं\omterm{def:g11:lenses-and-eye:magnifier}{ताल}ीस हजार गुना अधिक तेज \omterm{def:g10:signals-and-waves:wave}{लहर}.
\end{solution}
