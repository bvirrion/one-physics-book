\section*{अध्याय \ref{ch:g12:special-relativity} --- विशेष सापेक्षता: समय फैलाव}

\begin{solution}{exo:g12:special-relativity:1}
$\gamma = 1/\sqrt{1 - (v/c)^2}$: $1.01$ ($0.1c $); $1.25$ ($0.6c $);
$1.67$ ($0.8c $); $10.0$ ($0.995c $).
\end{solution}

\begin{solution}{exo:g12:special-relativity:2}
$\gamma = 1.25$, तो जमीन मापता है $1.25 \times 10.0 =
\qty{12.5}{s}$. जहाज \qty{10.0}{s} है \omterm{def:g12:special-relativity:proper-time}{उचित समय}: द
मेट्रोनोम (एक घड़ी) दोनों बीट्स पर मौजूद है।
\end{solution}

\begin{solution}{exo:g12:special-relativity:3}
$ v = c\sqrt{1 - 1/\gamma^2}$: $\gamma = 2$ देता है $ v = 0.866c $;
$\gamma = 10$ देता है $ v = 0.995c $.
\end{solution}

\begin{solution}{exo:g12:special-relativity:4}
$ v = \qty{36.1}{m/s}$, $ v/c = \num{1.2e-7}$;
$\gamma - 1 \approx (\num{1.2e-7})^2/2 \approx \num{7.2e-15}$. एक
के बाद \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} हारा $1/\num{7.2e-15} \approx \qty{1.4e14}{s}$ --- के बारे में
ड्राइविंग के चार मिलियन वर्ष।
\end{solution}

\begin{solution}{exo:g12:special-relativity:5}
यात्री सामने की फ्लैश की ओर बढ़ता है, इसलिए यह पहले उस तक पहुंचता है;
दूसरे अभिधारणा के अनुसार दोनों फ़्लैश एक ही \omterm{def:g12:newtons-laws:momentum}{गति} से यात्रा करते हैं $ c $ के ऊपर
समान आधी-ट्रेन दूरी \emph{उसके फ्रेम में}, इसलिए पहले आगमन
मतलब पहले की हड़\omterm{def:g11:lenses-and-eye:magnifier}{ताल}. एक साथ होना फ्रेम-निर्भर है: प्रत्येक निर्णय
अपने ढाँचे में सही है।
\end{solution}

\begin{solution}{exo:g12:special-relativity:6}
$\gamma = 1/\sqrt{1 - 0.99^2} = 7.09$; प्रयोगशाला जीवनकाल
$7.09 \times \num{2.6e-8} = \qty{1.8e-7}{s}$. क्लासिक \omterm{def:g12:projectile-motion:range}{श्रेणी}
$0.99 \times \num{3.00e8} \times \num{2.6e-8} \approx \qty{7.7}{m}$;
वास्तविक $7.09 \times 7.7 \approx \qty{55}{m}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:7}
$\gamma = 1.5$ पर $ v = c\sqrt{1 - 1/2.25} \approx 0.75c $;
$\gamma = 3$ पर $\approx 0.94c $. स्पर्शोन्मुख: $\gamma $ (और यह
ऊर्जा आगे की \omterm{def:g12:newtons-laws:momentum}{गति} की लागत) के रूप में भिन्न होती है $ v \to c $ --- कोई परिमित नहीं
प्रयास एक विशाल शरीर को प्रकाश \omterm{def:g12:newtons-laws:momentum}{गति} तक ले आता है।
\end{solution}

\begin{solution}{exo:g12:special-relativity:8}
$\Delta t_0 = 2L/c = 3.0/\num{3.00e8} = \qty{1.0e-8}{s}$; पर $0.8c $,
$\gamma = 5/3$, इसलिए $\Delta t = \qty{1.7e-8}{s}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:9}
\omterm{def:g12:special-relativity:proper-time}{उचित समय} दोनों पर मौजूद घड़ी का है घटनाएँ: (ए) द
पहनने वाला; (बी) म्यूऑन; (सी) पायलट --- दोनों तरफ केवल जहाज है
प्रस्थान और आगमन.
\end{solution}

\begin{solution}{exo:g12:special-relativity:10}
$\gamma - 1 \approx 7700^2/(2 \times \num{9.0e16}) \approx
\num{3.3e-10}$; ऊपर \qty{1.58e7}{s} अंतरिक्ष यात्री चलता रहता है
$\num{3.3e-10} \times \num{1.58e7} \approx \qty{5.2e-3}{s}$ --- के बारे में
पांच मिलीसेकेंड छोटा.
\end{solution}

\begin{solution}{exo:g12:special-relativity:11}
$ E = \num{1.0e-3} \times (\num{3.00e8})^2 = \qty{9.0e13}{J}$. ए
\qty{1.0}{GW} स्टेशन पहुंचाता है $\num{1.0e9} \times 86400 \approx
\qty{8.6e13}{J}$ प्रति दिन --- एक ग्राम इसके उत्पादन का एक दिन है, और
$\num{9.0e13}/\num{8.5e5} \approx \num{e8}$: सौ करोड़
तेज़ \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} से दौड़ती गाड़ियाँ.
\end{solution}

\begin{solution}{exo:g12:special-relativity:12}
वर्गाकार और उलटा: $\gamma^2(1 - v^2/c^2) = 1$, इसलिए
$ v^2/c^2 = 1 - 1/\gamma^2$, इस तरह $ v = c\sqrt{1 - 1/\gamma^2}$. के लिए
$\gamma = 30$: $ v = c\sqrt{1 - 1/900} = 0.99944c $.
\end{solution}

\begin{solution}{exo:g12:special-relativity:13}
अनुबंधित मोटाई $ L = \qty{15}{km}/30 = \qty{500}{m}$; स्वीप समय
$500/(0.99944 \times \num{3.00e8}) \approx \qty{1.7e-6}{s} <
\qty{2.2}{\micro s}$. दोनों फ्रेम सहमत हैं कि म्यूऑन जमीन तक पहुंचता है ---
एक लम्बे जीवनकाल को दोष देता है, \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} सिकुड़े हुए वातावरण को।
\end{solution}

\begin{solution}{exo:g12:special-relativity:14}
प्रयोगशाला: $3200/\num{3.00e8} \approx \qty{1.1e-5}{s}$ (\omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}
$\approx c $). इलेक्ट्रॉन फ़्रेम: ट्यूब से अनुबंधित
$3200/\num{1.0e5} = \qty{3.2}{cm}$, पार कर गया
$\num{1.1e-5}/\num{1.0e5} \approx \qty{1.1e-10}{s}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:15}
धरती: $4.2/0.9 \approx \qty{4.7}{years}$. साथ
$\gamma = 1/\sqrt{1 - 0.81} = 2.29$, जांच की घड़ी लॉग करती है
$4.7/2.29 \approx \qty{2.0}{years}$.
\end{solution}

\begin{solution}{pb:g12:special-relativity:1}
\textbf{1.} $ c\,\Delta t_0 = \num{3.00e8} \times \num{2.2e-6} \approx
\qty{660}{m}$.

\textbf{2.} $\num{15000}/660 \approx 23$ जीवनकाल.

\textbf{3.} $\mathrm{e}^{-23} \approx \num{e-10}$: एक म्यूऑन से कम
दस अरब में आना चाहिए, फिर भी प्रवाह प्रति एक है \unit{cm^2} प्रति
मिनट --- शास्त्रीय भौतिकी आकाश द्वारा स्पष्ट रूप से विरोधाभासी है।

\textbf{4.} ग्राउंड जीवनकाल $30 \times \qty{2.2}{\micro s} =
\qty{66}{\micro s}$; \omterm{def:g12:projectile-motion:range}{श्रेणी} $\approx \num{3.00e8} \times \num{6.6e-5}
\approx \qty{20}{km} > \qty{15}{km}$: आगमन समझाया गया है.

\textbf{5.} $ v = c\sqrt{1 - 1/900} = 0.99944c $.

\textbf{6.} $\Delta t_0 = 2L/c = 3.0/\num{3.00e8} = \qty{1.0e-8}{s}$.

\textbf{7.} क्षैतिज $ v\,\Delta t/2$; \omterm{def:g10:universal-gravitation:weight}{खड़ा} $ L $; तिरछा
आधे-अधूरे उपाय $ c\,\Delta t/2$ क्योंकि \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} अभिधारणा ठीक करता है
द फोटोनकी \omterm{def:g12:newtons-laws:momentum}{गति} पर $ c $ में \omterm{def:g10:relative-motion:frames}{ज़मीन का ढाँचा} बहुत।

\textbf{8.} $(c\,\Delta t/2)^2 = L^2 + (v\,\Delta t/2)^2$ देता है
$\Delta t^2(c^2 - v^2) = 4L^2$, इसलिए $\Delta t =
(2L/c)/\sqrt{1 - v^2/c^2} = \gamma\,\Delta t_0$.

\textbf{9.} $\gamma = 1/\sqrt{1 - 0.36} = 1.25$;
$\Delta t = \qty{1.25e-8}{s}$.

\textbf{10.} जहाज़ की: इसकी एक घड़ी दोनों टिकों पर मौजूद है, इसलिए यह
पढ़ता है \omterm{def:g12:special-relativity:proper-time}{उचित समय}; ज़मीन को दो \omterm{prop:g12:oscillators-and-time:pendulum-period}{समकालिक} घड़ियों की आवश्यकता है।

\textbf{11.} $\gamma = 1/\sqrt{1 - 0.64} = 1/0.6 = 5/3 \approx 1.67$.

\textbf{12.} $10.0/(5/3) = \qty{6.0}{years}$.

\textbf{13.} आउटबाउंड $\qty{5.0}{years}$ पर $0.8c $: $4.0$ \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष}.

\textbf{14.} $10.0 - 6.0 = \qty{4.0}{years}$: अंतरिक्ष यात्री चार हैं
अपने जुड़वां भाई से कई वर्ष छोटी।

\textbf{15.} स्थिति सममित नहीं है: केवल अंतरिक्ष यात्री ही मुड़ता है
चारों ओर --- वह तेज़ हो जाती है और बदल जाती है \omterm{def:g12:special-relativity:inertial}{जड़त्वीय ढांचा}, जबकि पृथ्वी
जुड़वां एक में रहता है --- इसलिए केवल उसकी घड़ी ही कम समय दर्ज करती है।

\textbf{16.} $ v = 2\pi \times \num{2.66e7}/\num{43200} \approx
\qty{3.9e3}{m/s}$.

\textbf{17.} $\gamma - 1 \approx (\num{3.87e3})^2/(2 \times
\num{9.0e16}) \approx \num{8.3e-11}$.

\textbf{18.} $\num{8.3e-11} \times \num{86400} \approx \qty{7.2e-6}{s}
= \qty{7.2}{\micro s}$ प्रति दिन धीमी \omterm{def:g12:newtons-laws:momentum}{गति} से.

\textbf{19.} $ c \times \num{7.2e-6} \approx \qty{2.2}{km}$ पद का
प्रति दिन त्रुटि.

\textbf{20.} शुद्ध बहाव $45 - 7.2 \approx \qty{38}{\micro s}$ प्रति दिन
तेज़; गलती $ c \times \num{38e-6} \approx \qty{11}{km}$ प्रति दिन ---
सापेक्षता बिल जिसका भुगतान प्रत्येक जीपीएस घड़ी अग्रिम रूप से करती है, को अलग कर दिया जाता है
ज़मीन इसलिए कि यह सच में टिक जाए \omterm{def:g10:universal-gravitation:satellite}{कक्षा}.
\end{solution}
