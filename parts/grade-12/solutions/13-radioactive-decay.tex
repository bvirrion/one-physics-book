\section*{Chapter \ref{ch:g12:radioactive-decay} --- Radioactive Decay}

\begin{solution}{exo:g12:radioactive-decay:1}
$T_{1/2} = \ln 2/\lambda = 0.693/\num{1.0e-6} = \qty{6.9e5}{s}
\approx 8.0$ days --- iodine-131's clock.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:2}
$T_{1/2} = 3.8 \times \num{86400} = \qty{3.28e5}{s}$;
$\lambda = 0.693/\num{3.28e5} = \qty{2.1e-6}{s^{-1}}$;
$\mathcal A = \lambda N = \num{2.1e-6} \times \num{1.0e10} \approx
\qty{2.1e4}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:3}
$\dd N/\dd t = N_0(-\lambda)e^{-\lambda t} = -\lambda N$ ---
the equation holds; $N(0) = N_0 e^0 = N_0$;
$N(T_{1/2})/N_0 = e^{-\lambda \ln 2/\lambda} = e^{-\ln 2} =
\tfrac12$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:4}
$90$ years $= 3\,T_{1/2}$: $\mathcal A = \num{4.0e5}/2^3 =
\qty{5.0e4}{Bq}$. $300$ years $= 10\,T_{1/2}$:
$\mathcal A = \num{4.0e5}/1024 \approx \qty{3.9e2}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:5}
Slope $= (18.85 - 20.00)/10 = -0.115$, so
$\lambda = \qty{0.115}{h^{-1}} = 0.115/3600 = \qty{3.2e-5}{s^{-1}}$;
$T_{1/2} = 0.693/0.115 = 6.0$ hours --- technetium-99m.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:6}
$N = (1.0/226) \times \num{6.02e23} = \num{2.7e21}$;
$T_{1/2} = 1600 \times \num{3.156e7} = \qty{5.05e10}{s}$, so
$\lambda = \qty{1.37e-11}{s^{-1}}$;
$\mathcal A = \lambda N \approx \qty{3.7e10}{Bq}$ --- one curie, the
activity of Marie Curie's gram of radium.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:7}
Both expect half to remain: $10$, and \num{1.0e20}. Fluctuations
$\sim 1/\sqrt{n}$ of the count: with $10$ decays, about $30\%$ ---
$7$ or $13$ would be no surprise; with \num{1.0e20}, about
$10^{-10}$. Only the large sample obeys the law to measurable
precision.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:8}
$m = 140 \times \num{1.2e-4} = \qty{1.7e-2}{g}$;
$N = (0.0168/40) \times \num{6.02e23} = \num{2.5e20}$.
$\lambda = 0.693/(\num{1.25e9} \times \num{3.156e7}) =
\qty{1.76e-17}{s^{-1}}$; $\mathcal A = \lambda N \approx
\qty{4.4e3}{Bq}$ --- about half the body's total (most of the rest is
carbon-14).
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:9}
$\lambda = 0.693/6.0 = \qty{0.116}{h^{-1}}$. At $15$ hours:
$\mathcal A = 500\,e^{-0.116 \times 15} = 500 \times 0.177 \approx
\qty{88}{MBq}$. Below \qty{1}{MBq}:
$t = \ln(500)/0.116 \approx \qty{54}{hours}$ --- about $9$
half-lives.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:10}
Charcoal: $3.4/13.6 = \tfrac14$, exactly $2$ half-lives:
$t = \num{1.15e4}$ years. Bone:
$t = \num{5730} \times \ln(1/0.30)/\ln 2 = \num{5730} \times 1.74
\approx \num{1.0e4}$ years.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:11}
$\lambda = 0.693/3.8 = \qty{0.182}{d^{-1}}$;
$t = \ln(100)/\lambda = 4.61/0.182 \approx 25$ days (about $6.6$
half-lives). But the uranium chain in the surrounding granite and
soil breeds radon continuously: a one-off airing is undone within
weeks, so ventilation must be permanent.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:12}
$N/N_0 = e^{-\lambda t}$ gives $\lambda t = \ln(N_0/N)$, and
$\lambda = \ln 2/T_{1/2}$ gives $t = T_{1/2}\ln(N_0/N)/\ln 2$. Here
$t = \num{5730} \times \ln(20)/\ln 2 = \num{5730} \times 4.32
\approx \num{2.5e4}$ years.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:13}
$N_{\mathrm K} = N_0 e^{-\lambda t}$ and
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, so
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$. Ratio $3.0$:
$e^{\lambda t} = 4$, so $\lambda t = 2\ln 2$ and
$t = 2\,T_{1/2} = \num{2.5e9}$ years.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:14}
$\ln \mathcal A = 5.99$, $5.42$, $4.84$, $4.26$, $3.69$: drops of
$0.57$--$0.58$ per \qty{60}{s} --- aligned. Slope:
$\lambda = 2.30/240 = \qty{9.6e-3}{s^{-1}}$;
$T_{1/2} = 0.693/\num{9.6e-3} \approx \qty{72}{s}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:15}
(a) $2^{-10} \approx \num{1e-3}$. (b) $(1/2)^8 = 1/256 \approx
0.4\%$. (c) $1000/8 = 125$ expected --- but the law is silent on
which: it predicts populations exactly and individuals not at all.
\end{solution}

\begin{solution}{pb:g12:radioactive-decay:1}
\textbf{1.} Exchange with the atmosphere (eating, breathing) keeps
the proportion topped up at the atmospheric value; death stops the
intake and decay takes over.

\textbf{2.} $T_{1/2} = \num{5730} \times \num{3.156e7} =
\qty{1.81e11}{s}$, so $\lambda = 0.693/\num{1.81e11} =
\qty{3.8e-12}{s^{-1}}$.

\textbf{3.} $\dd N/\dd t = -\lambda N_0 e^{-\lambda t} = -\lambda N$,
and $N(0) = N_0$.

\textbf{4.} $\mathcal A_0 = 13.6/60 = \qty{0.227}{Bq}$ per gram;
$N = \mathcal A_0/\lambda = 0.227/\num{3.8e-12} \approx \num{5.9e10}$
atoms per gram.

\textbf{5.} $(1/12) \times \num{6.02e23} = \num{5.0e22}$ carbon
atoms; proportion $\num{5.9e10}/\num{5.0e22} \approx \num{1.2e-12}$
--- one in $10^{12}$, as promised.

\textbf{6.} $\mathcal A = \mathcal A_0 e^{-\lambda t}$ gives
$\lambda t = \ln(\mathcal A_0/\mathcal A)$, and $\lambda =
\ln 2/T_{1/2}$ turns it into
$t = T_{1/2}\ln(\mathcal A_0/\mathcal A)/\ln 2$.

\textbf{7.} $t = \num{5730} \times \ln(13.6/9.1)/\ln 2 =
\num{5730} \times 0.580 \approx \num{3.3e3}$ years.

\textbf{8.} The death of the wood --- the tree's felling, not the
carving; a statue cut from old timber (or a fake carved from ancient
wood) predates or postdates its material's date.

\textbf{9.} $t = \num{5730} \times \ln(13.6/10.4)/\ln 2 =
\num{5730} \times 0.387 \approx \num{2.2e3}$ years --- fully
consistent with a scribe of twenty-two centuries ago.

\textbf{10.} $\Delta t \approx \Delta\mathcal A/(\lambda\mathcal A)$
with $\lambda = \ln 2/5730 = \num{1.21e-4}$ per year:
$\Delta t = (0.2/10.4)/\num{1.21e-4} \approx \num{1.6e2}$ years ---
a date good to a century or two.

\textbf{11.} $t = \num{5730} \times \ln(13.6/13.4)/\ln 2 \approx
120$ years --- indistinguishable from modern within the $\pm 160$-year
uncertainty: the parchment is recent, the ``antique'' a fake.

\textbf{12.} $13.6/2^n < 0.2$ needs $2^n > 68$: $n = 7$ half-lives
(since $2^7 = 128$), about $\num{4.0e4}$ years.

\textbf{13.} Ten half-lives: about \num{5.7e4} years --- the
practical horizon, some sixty thousand years.

\textbf{14.} $\num{6.6e7}/\num{5730} \approx \num{11500}$ half-lives.
One gram's \num{5.9e10} atoms are exhausted after
$2^n > \num{5.9e10}$, i.e.\ $n \approx 36$ half-lives
($\approx \num{2e5}$ years): long before the dinosaurs' age, not one
carbon-14 atom is left --- there is nothing to count.

\textbf{15.} A much slower clock: potassium-40 or uranium-238, with
half-lives of billions of years.

\textbf{16.} Molten lava lets argon bubble away, so the freshly
solidified crystal holds potassium but zero argon: the ratio starts
at $0$ at solidification --- the clock is zeroed.

\textbf{17.} As in the exercise: $N_{\mathrm K} = N_0 e^{-\lambda t}$,
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, so
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.

\textbf{18.} $e^{\lambda t} = 1.60$:
$t = \num{1.25e9} \times \ln(1.60)/\ln 2 = \num{1.25e9} \times 0.678
\approx \num{8.5e8}$ years.

\textbf{19.} No, twice: at \num{8.5e8} years carbon-14 has run
$10^5$ half-lives --- silence; at \num{3.3e3} years potassium-40 has
run \num{2.6e-6} of a half-life --- the argon ratio
$\approx \lambda t \sim \num{2e-6}$ is unmeasurably small. Each
clock reads only from about a tenth of a half-life to about ten.

\textbf{20.} Statue $\approx \num{3.3e3}$ years, papyrus
$\approx \num{2.2e3}$ years, rock $\approx \num{8.5e8}$ years ---
three clocks, one law: $N = N_0\,e^{-\lambda t}$, read backwards.
\end{solution}
