\begin{solution}{exo:g12:radioactive-decay:2}
$T_{1/2} = 3.8 \times \num{86400} = \qty{3.28e5}{s}$;
$\lambda = 0.693/\num{3.28e5} = \qty{2.1e-6}{s^{-1}}$;
$\mathcal A = \lambda N = \num{2.1e-6} \times \num{1.0e10} \approx
\qty{2.1e4}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:3}
$\dd N/\dd t = N_0(-\lambda)e^{-\lambda t} = -\lambda N$ ---
la ecuación se cumple; $N(0) = N_0 e^0 = N_0$;
$N(T_{1/2})/N_0 = e^{-\lambda \ln 2/\lambda} = e^{-\ln 2} =
\tfrac12$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:9}

\begin{solution}{exo:g12:radioactive-decay:11}
$\lambda = 0.693/3.8 = \qty{0.182}{d^{-1}}$;
$t = \ln(100)/\lambda = 4.61/0.182 \approx 25$ días (aproximadamente $6.6$
vidas medias). Pero la cadena de uranio en el granito circundante y

\end{solution}

\begin{solution}{exo:g12:radioactive-decay:12}
$N/N_0 = e^{-\lambda t}$ da $\lambda t = \ln(N_0/N)$, y
$\lambda = \ln 2/T_{1/2}$ da $t = T_{1/2}\ln(N_0/N)/\ln 2$. aquí
$t = \num{5730} \times \ln(20)/\ln 2 = \num{5730} \times 4.32
\approx \num{2.5e4}$ años.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:13}
$N_{\mathrm K} = N_0 e^{-\lambda t}$ y
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, entonces
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$. Relación $3.0$:
$e^{\lambda t} = 4$, entonces $\lambda t = 2\ln 2$ y
$t = 2\,T_{1/2} = \num{2.5e9}$ años.
\end{solution}

\end{solution}

\begin{solution}{pb:g12:radioactive-decay:1}
\textbf{1.} El intercambio con la atmósfera (comer, respirar) mantiene
la proporción completada con el valor atmosférico; la muerte detiene el
la ingesta y la descomposición se hacen cargo.

\textbf{2.} $T_{1/2} = \num{5730} \times \num{3.156e7} =
\qty{1.81e11}{s}$, so$\lambda = 0.693/\num{1.81e11} =
\qty{3.8e-12}{s^{-1}}$.

\textbf{3.}$\dd N/\dd t = -\lambda N_0 e^{-\lambda t} = -\lambda N$,
y $N(0) = N_0$.

\textbf{4.} $\mathcal A_0 = 13.6/60 = \qty{0.227}{Bq}$ por gramo;
$N = \mathcal A_0/\lambda = 0.227/\num{3.8e-12} \approx \num{5.9e10}$
\omterm{def:g11:fundamental-interactions:ladder}{átomos} por gramo.

\textbf{5.} $(1/12) \times \num{6.02e23} = \num{5.0e22}$ carbono
\omterm{def:g11:fundamental-interactions:ladder}{átomos}; proporción $\num{5.9e10}/\num{5.0e22} \approx \num{1.2e-12}$
--- uno en $10^{12}$, como se prometió.

\textbf{6.} $\mathcal A = \mathcal A_0 e^{-\lambda t}$ da
$\lambda t = \ln(\mathcal A_0/\mathcal A)$ y $\lambda =
\ln 2/T_{1/2}$ lo convierten en
$t = T_{1/2}\ln(\mathcal A_0/\mathcal A)/\ln 2$.

\textbf{7.} $t = \num{5730} \times \ln(13.6/9.1)/\ln 2 =
\num{5730} \times 0.580 \approx \num{3.3e3}$ años.

\textbf{8.} La muerte de la madera --- la tala del árbol, no la
talla; una estatua tallada en madera vieja (o una falsa tallada en madera antigua)
madera) es anterior o posterior a la fecha de su material.

\textbf{9.} $t = \num{5730} \times \ln(13.6/10.4)/\ln 2 =
\num{5730} \times 0.387 \approx \num{2.2e3}$ años --- completamente
consistente con un escriba de hace veintidós siglos.

\textbf{10.} $\Delta t \approx \Delta\mathcal A/(\lambda\mathcal A)$
con $\lambda = \ln 2/5730 = \num{1.21e-4}$ por año:
$\Delta t = (0.2/10.4)/\num{1.21e-4} \approx \num{1.6e2}$ años ---
una fecha buena de uno o dos siglos.

\textbf{11.} $t = \num{5730} \times \ln(13.6/13.4)/\ln 2 \approx
120$años --- indistinguible de lo moderno dentro del año$\pm 160$
Incertidumbre: el pergamino es reciente, lo "antiguo" es falso.

\textbf{12.} $13.6/2^n < 0.2$ necesita $2^n > 68$: $n = 7$ vidas medias
(desde $2^7 = 128$), aproximadamente $\num{4.0e4}$ años.

\textbf{13.} Diez vidas medias: alrededor de \num{5.7e4} años --- el
horizonte práctico, unos sesenta mil años.

\textbf{14.} $\num{6.6e7}/\num{5730} \approx \num{11500}$ vidas medias.
\num{5.9e10} \omterm{def:g11:fundamental-interactions:ladder}{átomos} de un gramo se agotan después
$2^n > \num{5.9e10}$, es decir, \ $n \approx 36$ vidas medias
($\approx \num{2e5}$ años): mucho antes de la era de los dinosaurios, ni uno
Queda carbono-14 \omterm{def:g11:fundamental-interactions:ladder}{átomo} --- no hay nada que contar.

\textbf{15.} Un reloj mucho más lento: potasio-40 o uranio-238, con
vidas medias de miles de millones de años.

\textbf{16.} La lava fundida deja que el argón burbujee, por lo que el agua recién
El cristal solidificado contiene potasio pero cero argón: la relación comienza
en $0$ en solidificación --- el reloj se pone a cero.

\textbf{17.} Como en el ejercicio: $N_{\mathrm K} = N_0 e^{-\lambda t}$,
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, entonces
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.

\textbf{18.} $e^{\lambda t} = 1.60$:
$t = \num{1.25e9} \times \ln(1.60)/\ln 2 = \num{1.25e9} \times 0.678
\approx \num{8.5e8}$ años.

\textbf{19.} No, dos veces: en \num{8.5e8} años el carbono-14 se ha ejecutado
$10^5$ vidas medias --- silencio; a \num{3.3e3} años potasio-40 ha
ejecute \num{2.6e-6} de un \omterm{def:g11:nucleus-radioactivity:halflife}{vida media} --- la proporción de argón
$\approx \lambda t \sim \num{2e-6}$ es inmensurablemente pequeño. cada uno
El reloj lee solo desde aproximadamente una décima parte de \omterm{def:g11:nucleus-radioactivity:halflife}{vida media} hasta aproximadamente diez.

\textbf{20.} Estatua $\approx \num{3.3e3}$ años, papiro
$\approx \num{2.2e3}$ años, roca $\approx \num{8.5e8}$ años ---
tres relojes, una ley: $N = N_0\,e^{-\lambda t}$, leído al revés.
\end{solution}
