\end{solution}

\end{solution}

\begin{solution}{exo:g12:newtons-laws:4}

\begin{solution}{exo:g12:newtons-laws:5}
\omterm{def:g10:universal-gravitation:weight}{Peso} (Tierra sobre libro, abajo): compañero = libro tira de la Tierra, hacia arriba.
\omterm{def:g12:newtons-laws:friction}{fuerza normal} (mesa sobre libro, arriba): compañero = libro presiona la mesa,
abajo. \omterm{def:g10:universal-gravitation:weight}{Peso} y actúan normalmente en el cuerpo \emph{mismo} y pertenecen a dos
interacciones diferentes --- iguales aquí solo porque $\vect a = \vect 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:6}

(b) $a = (220.7 - 0.35 \times 490.5)/50 = \qty{0.98}{m/s^2}$.
(c) $F = \mu_k N = \qty{172}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:7}
(a) Diapositivas cuando $\tan\alpha > \mu_s$: $\alpha = \arctan 0.45 \approx
24^\circ$. (b)$a = g(\sin 30^\circ - 0.35\cos 30^\circ) =
9.81 \times 0.197 = \qty{1.9}{m/s^2}$. (c) Sí: una vez deslizado,
\omterm{def:g10:inertia:inventory}{fricción} cae a $\mu_k < \mu_s$, por lo que
$a = g\cos\alpha\,(\tan\alpha - \mu_k) \approx \qty{0.9}{m/s^2} > 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:8}
$a = m_2 g/(m_1 + m_2) = 19.62/5.0 = \qty{3.9}{m/s^2}$;
$T = m_1 a = \qty{11.8}{N}$. El bloque colgante acelera hacia abajo,
entonces el \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} apunta hacia abajo: $T$ debe socavar $m_2 g$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:9}
$v_{\max} = \sqrt{\mu_s g R}$: seco
$\sqrt{0.70 \times 9.81 \times 50} = \qty{18.5}{m/s} \approx
\qty{67}{km/h}$; mojado$\qty{14.0}{m/s} = \qty{50}{km/h}$. ambos
$mv^2/R$ y $\mu_s mg$ llevan $m$: se cancela.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:10}
Vertical: $T\cos\theta = mg$, entonces
$T = 2.943/\cos 25^\circ = \qty{3.2}{N}$. Radio $R = L\sin\theta =
\qty{0.507}{m}$; radial:$T\sin\theta = mv^2/R$, entonces
$v = \sqrt{gR\tan\theta} = \sqrt{9.81 \times 0.507 \times 0.466}
= \qty{1.5}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:11}
$a_z = (N - mg)/m$ con $mg = \qty{49.05}{N}$: $+\qty{1.0}{m/s^2}$,
$0$, $-\qty{1.0}{m/s^2}$. Por ejemplo, un viaje ascendente: acelerar, navegar,
freno. Sí: entrando ya descendiendo, las mismas lecturas significan freno,
crucero, acelere nuevamente hacia abajo --- la escala lee aceleración, no
velocidad.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:12}
(a) $0 = 60V + 4.0 \times 8.0$: $V = \qty{-0.53}{m/s}$.
(b) Bola en $8.0 + V$ sobre el suelo: $0 = 60V + 4.0(8.0 + V)$, entonces
$V = -32/64 = \qty{-0.50}{m/s}$ y la pelota vuela hacia \qty{7.5}{m/s}.
(c) El empuje de la pelota sobre sus manos --- el compañero de tercera ley de ella
tirar.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:13}
Sin fricción: $N\cos\theta = mg$ (vertical) y $N\sin\theta =
mv^2/R$(radial); dividiendo,$\tan\theta = v^2/(gR) = 400/784.8$:

\end{solution}

\begin{solution}{exo:g12:newtons-laws:15}
(a) $m_2 g > \mu_s m_1 g$: $m_2 > 0.35 \times 4.0 = \qty{1.4}{kg}$.
(b) $a = (m_2 - \mu_k m_1)\,g/(m_1 + m_2) = 1.0 \times 9.81/6.0 =

\end{solution}

\begin{solution}{pb:g12:newtons-laws:1}
\textbf{1.} Sistema: cabina $+$ vagón ($+$ pasajeros). la montaña
Está fijado al suelo, inercial con una precisión excelente. \omterm{def:g10:inertia:force}{Efectivo}:
\omterm{def:g10:universal-gravitation:weight}{peso} $M\vect g$ (abajo), \omterm{def:g12:newtons-laws:friction}{fuerza normal} $\vect N$ del cable de vía
(perpendicular a él), arrastre \omterm{def:g10:inertia:inventory}{tensión} $\vect T$ (cuesta arriba).

\textbf{2.} En reposo, la primera ley: $\sum \vect F = \vect 0$. A lo largo
la pendiente: $T = Mg\sin\alpha$; perpendicular: $N = Mg\cos\alpha$.

\textbf{3.} $T = 4000 \times 9.81 \times 0.500 = \qty{19.6}{kN}$;
$N = 4000 \times 9.81 \times 0.866 = \qty{34.0}{kN}$.

\textbf{4.} Vacío: $T = \qty{9.8}{kN}$. Cada \omterm{def:g10:inertia:force}{fuerza} en el saldo es
proporcional a $M$, por lo que $T$ también lo es.

\textbf{5.} A lo largo de la pendiente, $Ma = Mg\sin\alpha$: $M$ cancela,
$a = g\sin\alpha = \qty{4.9}{m/s^2}$ cuesta abajo.

\textbf{6.} $T' = M(g\sin\alpha + a) = 4000 \times 5.505 =
\qty{22.0}{kN}$.

\textbf{7.} $a/(g\sin\alpha) = 0.60/4.905 \approx 12\%$.

\textbf{8.} $t = v/a = \qty{10}{s}$; $d = v^2/(2a) = \qty{30}{m}$.

\textbf{9.} Componente vertical de $\vect a$: $a\sin\alpha =
\qty{0.30}{m/s^2}$, por lo que$N = m(g + a\sin\alpha) = 80 \times 10.11 =
\qty{809}{N}$contra$mg = \qty{785}{N}$:$3\%$ es más pesado.

\textbf{10.} El piso estático \omterm{def:g10:inertia:inventory}{fricción}, horizontal:
$ma\cos\alpha = 80 \times 0.52 \approx \qty{42}{N}$. Inclinarse pone el
piso combinado empuje a través de su \omterm{def:g11:forces-and-motion:com}{centro de masa}, para que no se incline
ella.

\textbf{11.} $a_T = dv/dt = 0$ (velocidad constante);
$a_N = v^2/R = 36/40 = \qty{0.90}{m/s^2}$, dirigido al centro ---
Directamente hacia la cima.

\textbf{12.} Eje normal (hacia abajo positivo): $Mg - N' = Mv^2/R$, entonces
$N' = M(g - v^2/R) = 4000 \times 8.91 = \qty{35.6}{kN}$.

\textbf{13.}$N = m(g - v^2/R) = 80 \times 8.91 = \qty{713}{N}$:
encendedor de $v^2/(gR) = 0.90/9.81 \approx 9\%$.

\textbf{14.} $N' = 0$ en $v = \sqrt{gR} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$ --- más del triple del crucero: los paseos en carruaje
el cable con un amplio margen.

\textbf{15.} \omterm{thm:g12:newtons-laws:third}{tercera ley de newton}: \qty{35.6}{kN}, presionando hacia abajo el
cable (por lo tanto en la torre) --- ligeramente menos que el \omterm{def:g10:universal-gravitation:weight}{peso} en reposo,
aliviado por el vuelo curvo.

\textbf{16.} $F = \Delta p/\Delta t = 4000 \times 6.0/1.5 =
\qty{16}{kN}$, en oposición a la moción ($a = \qty{4.0}{m/s^2}$).

\textbf{17.}$f = ma = 80 \times 4.0 = \qty{320}{N}$; el techo es
$\mu_s mg = 0.40 \times 785 = \qty{314}{N} < \qty{320}{N}$: pies
resbalón solo --- sujetarse a la barandilla.

\textbf{18.}$0 = M_0 V + m(v_{\text{rel}} + V)$:
$V = -80 \times 2.5/2080 = \qty{-0.096}{m/s}$; el pasajero se mueve en
$2.5 - 0.096 \approx \qty{2.4}{m/s}$ sobre el suelo.

\textbf{19.} El cohete: su \omterm{def:g10:inertia:inventory}{empuje} está fijado por la masa eyectada por
segunda vez la velocidad de expulsión relativa a ella.

\textbf{20.} El \omterm{def:g10:inertia:inventory}{tensión} más grande conocido es el de puesta en marcha.
$T' = M \times \qty{5.505}{N/kg}$ (el resto da $Mg\sin\alpha$, el
cresta aún menos). Gorra: $T' \leq \qty{20}{kN}$, entonces
$M \leq 20000/5.505 = \qty{3.63e3}{kg}$: carga útil
\qty{1.63e3}{kg}, es decir\ $20$ pasajeros de \qty{80}{kg} (no $25$).
Cartel: \emph{20 personas}; entonces $M = \qty{3.6e3}{kg}$,
$T' = \qty{19.8}{kN}$, factor de seguridad $100/19.8 \approx 5.1 \geq 5$.
\end{solution}
