\begin{solution}{exo:g12:rlc-oscillations:5}
Todo en el \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} cuando $q = \pm Q_0$, $i = 0$ ($t = 0$, $T_0/2$,
$T_0$, \dots): los extremos del péndulo. Todo en el \omterm{def:g11:magnetic-fields:solenoid}{bobina} cuando $q = 0$,

\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:6}
$d^2q/dt^2 = -(2\pi/T_0)^2\,Q_0\cos(2\pi t/T_0) = -(2\pi/T_0)^2 q$,
entonces $L\,d^2q/dt^2 + q/C = q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$ para todos $t$
exactamente cuando $(2\pi/T_0)^2 = 1/(LC)$, es decir\ $T_0 = 2\pi\sqrt{LC}$.
\end{solution}

\begin{solution}{pb:g12:rlc-oscillations:1}
\textbf{1.} Ley de bucle: $u_L + u_C = 0$, con $u_C = q/C$,
$u_L = L\,di/dt$, $i = dq/dt$: $L\,d^2q/dt^2 + q/C = 0$.

\textbf{2.} $d^2q/dt^2 = -(2\pi/T_0)^2 q$, por lo que la ecuación dice
$q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$, cierto para todos los $t$ exactamente cuando
$T_0 = 2\pi\sqrt{LC}$.

\textbf{3.} $LC$: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times
\unit{C/A} = \unit{s^2}$---$\sqrt{LC}$ es un tiempo.

\textbf{4.} $LC = \qty{6.6e-14}{s^2}$: $T_0 = 2\pi \times \num{2.57e-7}
\approx \qty{1.6}{\micro s}$,$f_0 \approx \qty{620}{kHz}$ --- dentro
\qtyrange{530}{1700}{kHz}.

\textbf{5.} $C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
\num{1.0e12} \times \num{2.0e-4}} \approx \qty{127}{pF}$.

\textbf{6.} Misma fórmula en \qty{530}{kHz}: $C \approx \qty{451}{pF}$.

\textbf{7.} At \qty{1700}{kHz}: $C \approx \qty{44}{pF}$.

\textbf{8.} $C = \qty{480}{pF}$: $f_0 \approx \qty{514}{kHz}$;
$C = \qty{20}{pF}$: $f_0 \approx \qty{2.5}{MHz}$ --- toda la banda,
con margen en ambos extremos.

\textbf{9.} $f_0 = (2\pi\sqrt{L})^{-1}C^{-1/2} \propto 1/\sqrt{C}$;
duplicar $f_0$ requiere dividir $C$ por $4$.

\textbf{10.} Desde $f_0 \propto 1/\sqrt{C}$, pasos iguales de movimiento $C$
$f_0$ lentamente en grande $C$ y rápido en pequeño $C$: un espaciamiento dado
entre estaciones requiere poco recorrido por dial: se aglomeran en el
Fin \omterm{def:g12:oscillators-and-time:oscillator}{alta frecuencia} (bajo-$C$).

\textbf{11.} $Q_0 = CU_0 = \num{1.27e-10} \times 0.020 \approx
\qty{2.5e-12}{C}$;$E = \tfrac12 CU_0^2 \approx \qty{2.5e-14}{J}$.

\textbf{12.} $\tfrac12 LI_0^2 = E$: $I_0 = U_0\sqrt{C/L} =
0.020\sqrt{\num{6.35e-7}} \approx \qty{16}{\micro A}$.

\textbf{13.} $0.96^n = \tfrac12$: $n = \ln 2/\lvert\ln 0.96\rvert
\approx 17$ ciclos.

\textbf{14.} En \qty{1.00}{MHz}, $T_0 = \qty{1.0}{\micro s}$: el anillo
mitades en aproximadamente \qty{17}{\micro s}. Un anillo solitario muere en microsegundos;
la música necesita que el loop se alimente continuamente.

\textbf{15.} El \omterm{def:g10:signals-and-waves:wave}{ola} empuja millones de veces por segundo; solo cuando
su \omterm{def:g12:oscillators-and-time:oscillator}{frecuencia} coincide con $f_0$. ¿Los empujones llegan al paso y construyen el
columpio --- \omterm{def:g12:oscillators-and-time:resonance}{resonancia}; las estaciones desafinadas se oponen con tanta frecuencia como con
eso. En el transmisor, un amplificador devuelve el \omterm{def:g11:mechanical-energy:dissipation}{disipado} \omterm{def:g10:energy-conservation:energy}{energía}
cada ciclo: un \omterm{def:g12:oscillators-and-time:oscillator}{oscilador} mantenido.

\textbf{16.} $LC = \qty{2.54e-14}{s^2}$: $k = m/(LC) =
0.20/\num{2.54e-14} \approx \qty{7.9e12}{N/m}$.

\textbf{17.} Aproximadamente $10^8$ veces el resorte del automóvil: sin banco
sistema resorte-masa alcanza megahercios --- rigideces cotidianas y
las masas son irremediablemente lentas.

\textbf{18.} $m = \dfrac{k}{(2\pi f_0)^2} = \dfrac{100}
{(\num{6.28e6})^2} \approx \qty{2.5e-12}{kg}$ --- unos pocos nanogramos, un
mota de polvo.

\textbf{19.} Un cristal de cuarzo: una astilla ultrarrígida y ligera
cantando en su \omterm{def:g12:rlc-oscillations:period}{frecuencia natural} --- el cronometrador se reunió anteriormente este
año.

\textbf{20.} Con $L = \qty{0.20}{mH}$ y \qtyrange{20}{480}{pF} el
el bucle recorre \qtyrange{0.51}{2.5}{MHz}, cubriendo la banda;
$C \approx \qty{127}{pF}$ elige \qty{1.00}{MHz}; dejó solo el anillo
mitades en $\approx \qty{17}{\micro s}$, pero el propio \omterm{def:g10:signals-and-waves:wave}{ola} de la estación,
resonando al paso, vuelve a alimentar el bucle --- las noticias se reproducen en el \omterm{def:g10:energy-conservation:energy}{energía}
del propio \omterm{def:g10:signals-and-waves:wave}{ola}.
\end{solution}
