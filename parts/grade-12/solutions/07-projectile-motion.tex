\begin{solution}{exo:g12:projectile-motion:2}
No air on the Moon: both in \omterm{def:g12:projectile-motion:freefall}{free fall}, $m\vect a = m\vect g$ gives
$\vect a = \vect g$ for any mass. On Earth, drag on the feather is
comparable to its tiny \omterm{def:g10:universal-gravitation:weight}{weight}: it rides at a \omterm{def:g12:projectile-motion:terminal}{terminal speed} of about
\qty{1}{m/s} while the hammer barely notices the air.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:3}
$t = \sqrt{2h/g} = 0.45$, $1.4$, \qty{4.5}{s};
$v = gt = 4.4$, $14$, \qty{44}{m/s}. Times grow like $\sqrt h$:
$\times 100$ in height is only $\times 10$ in time.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:6}
$(\ang{15},\ang{75})$ and $(\ang{30},\ang{60})$ share their \omterm{def:g12:projectile-motion:range}{range};
\ang{45} wins. The steeper of a complementary pair flies longer:
$t_f = 2v_0\sin\alpha/g$ grows with $\alpha$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:7}
$4.905\,t^2 + 5.0\,t - 20 = 0$ gives
$t = (-5 + \sqrt{25 + 392.4})/9.81 = \qty{1.57}{s}$;
$v = 5.0 + 9.81 \times 1.57 = \qty{20.4}{m/s}$. \omterm{def:g10:energy-conservation:energy}{Energy}:
$v = \sqrt{5.0^2 + 2 \times 9.81 \times 20} = \qty{20.4}{m/s}$ ---
same.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:9}

\begin{solution}{exo:g12:projectile-motion:10}

\begin{solution}{exo:g12:projectile-motion:11}
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180}-2\alpha) =
\sin(2\alpha)$: same $R$. Here $\sin(2\alpha) = gR/v_0^2 =
9.81 \times 50/625 = 0.785$: $\alpha = \ang{25.9}$ or \ang{64.1}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:12}

$h_{\max} = v_0^2/2g = \qty{3.3}{m}$ (straight up).
$\sin(2\alpha) = 9.81 \times 5.0/64 = 0.766$: $\alpha = \ang{25.0}$

\end{solution}

\begin{solution}{exo:g12:projectile-motion:13}

\begin{solution}{exo:g12:projectile-motion:14}
(a) Drag balances \omterm{def:g10:universal-gravitation:weight}{weight}: $80 \times 9.81 \approx \qty{7.8e2}{N}$.
(b) $v(t)$ rises with ever-smaller slope and levels off at
\qty{55}{m/s}. (c) Vacuum:
$\sqrt{2 \times 9.81 \times 2000} \approx \qty{2.0e2}{m/s}$, twenty
times the real \qty{9}{m/s} --- raindrops spend nearly the whole
fall at \omterm{def:g12:projectile-motion:terminal}{terminal speed}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:15}
At $t^* = D/(v_0\cos\alpha)$ the dart is at height
$v_0\sin\alpha\, t^* - \tfrac12 g t^{*2} = D\tan\alpha - \tfrac12 g
t^{*2} = H - \tfrac12 g t^{*2}$, exactly the dropped monkey's
$y = H - \tfrac12 g t^{*2}$. Both hang $\tfrac12 gt^2$ below their
no-gravity spots --- they meet for any $v_0$.
\end{solution}

\begin{solution}{pb:g12:projectile-motion:1}
\textbf{1.} \omterm{def:g10:universal-gravitation:weight}{Weight} only (air neglected); \omterm{thm:g12:newtons-laws:second}{Newton's second law}:
$m\vect a = m\vect g$, so $\vect a = (0,\,-g)$.

\textbf{2.} $v_x = v_0\cos\alpha$, $v_y = v_0\sin\alpha - gt$.

\textbf{3.} $x = v_0\cos\alpha\, t$,
$y = v_0\sin\alpha\, t - \tfrac12 gt^2$.

\textbf{4.} $t = x/(v_0\cos\alpha)$:
$y = x\tan\alpha - g x^2/(2v_0^2\cos^2\alpha)$.

\textbf{5.} $v_0^2 = \dfrac{g d^2}{2\cos^2\alpha\,(d\tan\alpha -
1.05)} = \dfrac{173.0}{0.826 \times 3.96} = 52.9$:
$v_0 = \qty{7.28}{m/s} \approx \qty{26}{km/h}$.

\textbf{6.} $t = d/(v_0\cos\alpha) = 4.20/4.68 = \qty{0.90}{s}$.

\textbf{7.} Apex at $t = v_0\sin\alpha/g = \qty{0.57}{s}$, height
$2.00 + (v_0\sin\alpha)^2/2g = 2.00 + 1.58 = \qty{3.58}{m}$;
$0.57 < 0.90$: descending at the rim.

\textbf{8.} $y(0.90) = 0.90\tan\ang{50} - 0.182 = \qty{0.89}{m}$
above release, i.e.\ \qty{2.89}{m} above the floor: \qty{0.29}{m}
above the fingertips.

\textbf{9.} $v_x = \qty{4.68}{m/s}$;
$v_y = 5.57 - 9.81 \times 0.90 = \qty{-3.24}{m/s}$;
$v = \qty{5.69}{m/s}$, at $\arctan(3.24/4.68) = \ang{34.7}$ below
the horizontal.

\textbf{10.} The rim is a horizontal ring: the ball must cross its
plane from above. The steeper the entry, the larger the ring's
opening looks along the velocity --- more margin for error.

\textbf{11.} $1.05/26.0 = 4\%$: negligible. $y(t_f) = 0$ gives
$t_f\,(v_0\sin\alpha - \tfrac12 g t_f) = 0$, so
$t_f = 2v_0\sin\alpha/g$.

\textbf{12.} $R = v_0\cos\alpha\, t_f = v_0^2\sin(2\alpha)/g$;
maximal at $\alpha = \ang{45}$.

\textbf{13.} $v_0 = \sqrt{gR} = \sqrt{9.81 \times 26.0} =
\qty{16.0}{m/s} \approx \qty{57}{km/h}$.

\textbf{14.} $R_{\max} = 17.5^2/9.81 = \qty{31.2}{m} >
\qty{26.0}{m}$: feasible, with \qty{5}{m} of \omterm{def:g12:projectile-motion:range}{range} to spare.

\textbf{15.} $\sin(2\alpha) = 255.1/306.3 = 0.833$:
$\alpha = \ang{28.2}$ or \ang{61.8}; flight times
$2v_0\sin\alpha/g = \qty{1.69}{s}$ and \qty{3.14}{s}. Only the flat
shot beats the \qty{2.0}{s} buzzer.

\textbf{16.} $v_0^2 = \dfrac{9.81 \times 36}{2\cos^2\ang{75}\,
(6.0\tan\ang{75} - 1.05)} = \dfrac{353.2}{2.86} = 123.5$:
$v_0 = \qty{11.1}{m/s}$.

\textbf{17.} $y(5.55) = 5.55\tan\ang{75} - 18.26 = \qty{2.45}{m}$
above release, i.e.\ \qty{4.45}{m} above the floor: \qty{0.50}{m}
above the board's top edge.

\textbf{18.} Apex $2.00 + (11.1\sin\ang{75})^2/2g \approx
\qty{7.9}{m}$; flight $6.0/(11.1\cos\ang{75}) = \qty{2.1}{s}$ ---
two full seconds with the ball hanging \qty{8}{m} up.

\textbf{19.} Drag shortens every shot and steepens the descent; the
effect grows with speed, so the \qty{17.5}{m/s} full-court heave
suffers most --- its real \omterm{def:g12:projectile-motion:range}{range} falls a few meters short of
\qty{31}{m}, eating into the margin.

\textbf{20.} \emph{The buzzer-beater verdict}: the \qty{26.0}{m}
shot demands \qty{16.0}{m/s} and an arm supplies about
\qty{17.5}{m/s}, a \omterm{def:g12:projectile-motion:range}{range} margin of \qty{5}{m} ($\approx 20\%$) that
air resistance trims but does not erase. Possible --- which is
exactly why, a few times a season, it goes in.
\end{solution}
