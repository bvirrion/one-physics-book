\section*{Capítulo \ref{ch:g12:rlc-oscillations} --- Oscilações Elétricas Livres: RLC}

\begin{solution}{exo:g12:rlc-oscillations:1}
(um) $\sqrt{LC} = \qty{1.0e-4}{s}$: $ T_0 = \qty{0.63}{ms}$,
$ f_0 \approx \qty{1.6}{kHz}$. (b) $\sqrt{LC} = \qty{3.16e-7}{s}$:
$ T_0 = \qty{2.0}{\micro s}$, $ f_0 \approx \qty{0.50}{MHz}$ --- um
fator $\sqrt{10^5} \approx 316$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:2}
$ LC $: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times \unit{C/A} =
\unit{s} \times \unit{s} = \unit{s^2}$, então $\sqrt{LC}$ está em \omterm{def:g10:orders-of-magnitude:unit}{segundos}.
$2\pi $ é um número puro --- nenhuma dimensão para alterar.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:3}
$ Q_0 = CU_0 = \qty{2.8}{\micro C}$; $ E = \tfrac12 CU_0^2 =
\tfrac12 \times \num{4.7e-7} \times 36 \approx \qty{8.5}{\micro J}$.
A carga oscila em $ f_0 = 1/(2\pi\sqrt{LC})$, atual um quarto
\omterm{def:g12:oscillators-and-time:oscillator}{período} fora de \omterm{def:g12:oscillators-and-time:phase}{fase}, energia chapinhando entre pratos e bobina, total
constante.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:4}
$ Q_0 = \qty{2.0}{\micro C}$; $ f_0 = 1/T_0 = \qty{250}{Hz}$;
$ q(T_0/2) = \qty{-2.0}{\micro C}$; $ i(0) = 0$ (cosseno plano em $ t=0$);
$ I_0 = 2\pi \times \num{2.0e-6}/\num{4.0e-3} \approx \qty{3.1}{mA}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:5}
Tudo no \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} quando $ q = \pm Q_0$, $ i = 0$ ($ t = 0$, $ T_0/2$,
$ T_0$, \dots): os extremos do pêndulo. Tudo no bobina quando $ q = 0$,
$ i = \pm I_0$ ($ t = T_0/4$, $3T_0/4$, \dots): o ponto mais baixo do pêndulo,
ponto mais rápido.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:6}
$ d^2q/dt^2 = -(2\pi/T_0)^2\,Q_0\cos(2\pi t/T_0) = -(2\pi/T_0)^2 q $,
então $ L\,d^2q/dt^2 + q/C = q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$ para todos $ t $
exatamente quando $(2\pi/T_0)^2 = 1/(LC)$, ou seja\ $ T_0 = 2\pi\sqrt{LC}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:7}
$ L = \dfrac{1}{4\pi^2 f_0^2 C} = \dfrac{1}{39.5 \times
(\num{1.98e5})^2 \times \num{2.2e-10}} \approx \qty{2.9}{mH}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:8}
$ E = \tfrac12 CU_0^2 = \qty{5.0e-5}{J}$; $\tfrac12 LI_0^2 = E $ dá
$ I_0 = \sqrt{2E/L} = \sqrt{\num{1.0e-2}} = \qty{0.10}{A}$, primeiro
alcançado em $ T_0/4 = \tfrac14 \times \qty{0.63}{ms} \approx
\qty{0.16}{ms}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:9}
$ E_C \propto \cos^2(2\pi t/T_0)$, e $\cos^2$ repete cada
\emph{metade} turno: dois recheios do \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} por \omterm{def:g12:oscillators-and-time:oscillator}{período}, por isso
$2f_0$. No $ t = T_0/8$ o \omterm{def:g12:oscillators-and-time:phase}{fase} é $\pi/4$:
$\cos^2(\pi/4) = \tfrac12$, então $ E_C = E_L = E/2$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:10}
Razão $3.0/4.0 = 0.75$ por \omterm{def:g12:rlc-oscillations:regimes}{pseudoperíodo}: sexto máximo
$4.0 \times 0.75^6 \approx \qty{0.71}{V}$. \omterm{def:g10:energy-conservation:energy}{Energia} $\propto $ \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}
ao quadrado: $0.75^2 \approx 56\%$ sobrevive cada \omterm{def:g12:oscillators-and-time:oscillator}{período}. O
\omterm{def:g12:rlc-oscillations:regimes}{pseudoperíodo} fica $\approx T_0 = 2\pi\sqrt{LC}$: luz amortecimento
encolhe o \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}, dificilmente o ritmo.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:11}
$ T_0 = 2\pi\sqrt{0.50 \times \num{2.0e-5}} = 2\pi \times
\num{3.16e-3} \approx \qty{20}{ms}$. Dicionário: $ k = m/(LC) =
0.50/\num{1.0e-5} = \qty{5.0e4}{N/m}$; verificar
$2\pi\sqrt{m/k} = 2\pi\sqrt{\num{1.0e-5}}$ --- o mesmo \qty{20}{ms}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:12}
$ C = \qty{500}{pF}$: $ f_0 = 1/(2\pi\sqrt{\num{1.0e-13}}) \approx
\qty{0.50}{MHz}$; $ C = \qty{50}{pF}$: $ f_0 \approx \qty{1.6}{MHz}$.
Em fixo $ L $, $ f_0 = (2\pi\sqrt{L})^{-1} C^{-1/2} \propto 1/\sqrt{C}$:
dez vezes $ C $, apenas $\sqrt{10} \approx 3.2$ em $ f_0$ --- conforme encontrado.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:13}
$0.95^n = \tfrac12$: $ n = \ln 2/\lvert\ln 0.95\rvert \approx 13.5$,
então sobre $14$ \omterm{def:g12:oscillators-and-time:oscillator}{períodos}. \omterm{def:g12:oscillators-and-time:oscillator}{Amplitude} $\propto \sqrt{E}$: até
$1/\sqrt{2} \approx 0.71$ do começo. No \qty{1.0}{MHz},
$ T_0 = \qty{1.0}{\micro s}$: sobre \qty{14}{\micro s}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:14}
Perda por ciclo $0.020 \times \qty{1.0e-9}{J} = \qty{2.0e-11}{J}$, no
\num{32768} ciclos por \omterm{def:g10:orders-of-magnitude:unit}{segundo}: $ P = \num{32768} \times \num{2.0e-11}
\approx \qty{6.6e-7}{W}$ --- abaixo de um microwatt, anos em um botão
célula. Em passo, porque apenas um empurrão à direita \omterm{def:g12:oscillators-and-time:phase}{fase} \emph{adiciona}
energia (\omterm{def:g12:oscillators-and-time:resonance}{ressonância}); fora de sintonia, frearia o balanço.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:15}
$ C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
(\num{2.5e7})^2 \times \num{1.0e-6}} \approx \qty{41}{pF}$. O
R deve empurrar o loop para dentro do \omterm{def:g12:rlc-oscillations:regimes}{regime aperiódico}: a cobrança
então se estabiliza sem ultrapassar --- sem oscilação, sem toque.
\end{solution}

\begin{solution}{pb:g12:rlc-oscillations:1}
\textbf{1.} Lei do loop: $ u_L + u_C = 0$, com $ u_C = q/C $,
$ u_L = L\,di/dt $, $ i = dq/dt $: $ L\,d^2q/dt^2 + q/C = 0$.

\textbf{2.} $ d^2q/dt^2 = -(2\pi/T_0)^2 q $, então a equação é lida
$ q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$, verdadeiro para todos $ t $ exatamente quando
$ T_0 = 2\pi\sqrt{LC}$.

\textbf{3.} $ LC $: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times
\unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ é um tempo.

\textbf{4.} $ LC = \qty{6.6e-14}{s^2}$: $ T_0 = 2\pi \times \num{2.57e-7}
\approx \qty{1.6}{\micro s}$, $ f_0 \approx \qty{620}{kHz}$ --- dentro
\qtyrange{530}{1700}{kHz}.

\textbf{5.} $ C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
\num{1.0e12} \times \num{2.0e-4}} \approx \qty{127}{pF}$.

\textbf{6.} Mesma fórmula em \qty{530}{kHz}: $ C \approx \qty{451}{pF}$.

\textbf{7.} No \qty{1700}{kHz}: $ C \approx \qty{44}{pF}$.

\textbf{8.} $ C = \qty{480}{pF}$: $ f_0 \approx \qty{514}{kHz}$;
$ C = \qty{20}{pF}$: $ f_0 \approx \qty{2.5}{MHz}$ --- toda a banda,
com margem em ambas as extremidades.

\textbf{9.} $ f_0 = (2\pi\sqrt{L})^{-1}C^{-1/2} \propto 1/\sqrt{C}$;
duplicação $ f_0$ requer divisão $ C $ por $4$.

\textbf{10.} Desde $ f_0 \propto 1/\sqrt{C}$, passos iguais de $ C $ mover
$ f_0$ lentamente em geral $ C $ e rápido em pequenos $ C $: um determinado espaçamento
entre as estações, em seguida, faz pouca viagem de discagem - eles se aglomeram no
alta \omterm{def:g12:oscillators-and-time:oscillator}{frequência} (baixo-$ C $) fim.

\textbf{11.} $ Q_0 = CU_0 = \num{1.27e-10} \times 0.020 \approx
\qty{2.5e-12}{C}$; $ E = \tfrac12 CU_0^2 \approx \qty{2.5e-14}{J}$.

\textbf{12.} $\tfrac12 LI_0^2 = E $: $ I_0 = U_0\sqrt{C/L} =
0.020\sqrt{\num{6.35e-7}} \approx \qty{16}{\micro A}$.

\textbf{13.} $0.96^n = \tfrac12$: $ n = \ln 2/\lvert\ln 0.96\rvert
\approx 17$ ciclos.

\textbf{14.} No \qty{1.00}{MHz}, $ T_0 = \qty{1.0}{\micro s}$: o anel
metades em cerca de \qty{17}{\micro s}. Um anel solitário morre em microssegundos;
a música precisa do loop alimentado continuamente.

\textbf{15.} O \omterm{def:g10:signals-and-waves:wave}{aceno} empurra milhões de vezes por \omterm{def:g10:orders-of-magnitude:unit}{segundo}; somente quando
é freqüência partidas $ f_0$ os empurrões chegam em sintonia e constroem o
balanço --- \omterm{def:g12:oscillators-and-time:resonance}{ressonância}; estações fora de sintonia empurram tão frequentemente contra quanto com
isso. No transmissor, um amplificador retorna o \omterm{def:g11:mechanical-energy:dissipation}{dissipado} energia
cada ciclo: um mantido oscilador.

\textbf{16.} $ LC = \qty{2.54e-14}{s^2}$: $ k = m/(LC) =
0.20/\num{2.54e-14} \approx \qty{7.9e12}{N/m}$.

\textbf{17.} Sobre $10^8$ vezes a mola do carro: sem bancada
mola - sistema de massa atinge megahertz --- rigidez diária e
as massas são irremediavelmente lentas.

\textbf{18.} $ m = \dfrac{k}{(2\pi f_0)^2} = \dfrac{100}
{(\num{6.28e6})^2} \approx \qty{2.5e-12}{kg}$ --- alguns nanogramas, um
partícula de poeira.

\textbf{19.} Um cristal de quartzo: uma lasca leve e ultra-rígida
cantando em seu \omterm{def:g12:rlc-oscillations:period}{frequência natural} --- o cronometrista se encontrou anteriormente
ano.

\textbf{20.} Com $ L = \qty{0.20}{mH}$ e \qtyrange{20}{480}{pF} o
loop varre sobre \qtyrange{0.51}{2.5}{MHz}, cobrindo a banda;
$ C \approx \qty{127}{pF}$ escolhas \qty{1.00}{MHz}; deixou sozinho o anel
metades em $\approx \qty{17}{\micro s}$, mas a própria estação \omterm{def:g10:signals-and-waves:wave}{aceno},
ressoando no mesmo ritmo, realimenta o loop --- as notícias são reproduzidas no energia
do \omterm{def:g10:signals-and-waves:wave}{aceno} em si.
\end{solution}
