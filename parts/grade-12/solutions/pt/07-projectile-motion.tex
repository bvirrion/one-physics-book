\begin{solution}{exo:g12:projectile-motion:2}

comparável ao seu minúsculo \omterm{def:g10:universal-gravitation:weight}{peso}: ele anda em um velocidade terminal de cerca de

\end{solution}

\begin{solution}{exo:g12:projectile-motion:15}
No $ t^* = D/(v_0\cos\alpha)$ o dardo está na altura
$ v_0\sin\alpha\, t^* - \tfrac12 g t^{*2} = D\tan\alpha - \tfrac12 g
t^{*2} = H - \tfrac12 g t^{*2}$, exatamente o macaco caído
$ y = H - \tfrac12 g t^{*2}$. Ambos penduram $\tfrac12 gt^2$ abaixo do seu
pontos sem gravidade --- eles se encontram para qualquer $ v_0$.
\end{solution}

\begin{solution}{pb:g12:projectile-motion:1}
\textbf{1.} \omterm{def:g10:universal-gravitation:weight}{Peso} apenas (ar negligenciado); Segunda lei de Newton:
$ m\vect a = m\vect g $, então $\vect a = (0,\,-g)$.

\textbf{2.} $ v_x = v_0\cos\alpha $, $ v_y = v_0\sin\alpha - gt $.

\textbf{3.} $ x = v_0\cos\alpha\, t $,
$ y = v_0\sin\alpha\, t - \tfrac12 gt^2$.

\textbf{4.} $ t = x/(v_0\cos\alpha)$:
$ y = x\tan\alpha - g x^2/(2v_0^2\cos^2\alpha)$.

\textbf{5.} $ v_0^2 = \dfrac{g d^2}{2\cos^2\alpha\,(d\tan\alpha -
1.05)} = \dfrac{173.0}{0.826 \times 3.96} = 52.9$:
$ v_0 = \qty{7.28}{m/s} \approx \qty{26}{km/h}$.

\textbf{6.} $ t = d/(v_0\cos\alpha) = 4.20/4.68 = \qty{0.90}{s}$.

\textbf{7.} Ápice em $ t = v_0\sin\alpha/g = \qty{0.57}{s}$, altura
$2.00 + (v_0\sin\alpha)^2/2g = 2.00 + 1.58 = \qty{3.58}{m}$;
$0.57 < 0.90$: descendo na borda.

\textbf{8.} $ y(0.90) = 0.90\tan\ang{50} - 0.182 = \qty{0.89}{m}$
versão acima, ou seja,\ \qty{2.89}{m} acima do chão: \qty{0.29}{m}
acima das pontas dos dedos.

\textbf{9.} $ v_x = \qty{4.68}{m/s}$;
$ v_y = 5.57 - 9.81 \times 0.90 = \qty{-3.24}{m/s}$;
$ v = \qty{5.69}{m/s}$, no $\arctan(3.24/4.68) = \ang{34.7}$ abaixo
a horizontal.

\textbf{10.} O aro é um anel horizontal: a bola deve cruzar seu
avião de cima. Quanto mais íngreme for a entrada, maior será o anel
a abertura olha ao longo da velocidade --- mais margem de erro.

\textbf{11.} $1.05/26.0 = 4\%$: insignificante. $ y(t_f) = 0$ dá
$ t_f\,(v_0\sin\alpha - \tfrac12 g t_f) = 0$, então
$ t_f = 2v_0\sin\alpha/g $.

\textbf{12.} $ R = v_0\cos\alpha\, t_f = v_0^2\sin(2\alpha)/g $;
máximo em $\alpha = \ang{45}$.

\textbf{13.} $ v_0 = \sqrt{gR} = \sqrt{9.81 \times 26.0} =
\qty{16.0}{m/s} \approx \qty{57}{km/h}$.

\textbf{14.} $ R_{\max} = 17.5^2/9.81 = \qty{31.2}{m} >
\qty{26.0}{m}$: viável, com \qty{5}{m} de \omterm{def:g12:projectile-motion:range}{faixa} de sobra.

\textbf{15.} $\sin(2\alpha) = 255.1/306.3 = 0.833$:
$\alpha = \ang{28.2}$ ou \ang{61.8}; tempos de voo
$2v_0\sin\alpha/g = \qty{1.69}{s}$ e \qty{3.14}{s}. Somente o apartamento
tiro bate o \qty{2.0}{s} campainha.

\textbf{16.} $ v_0^2 = \dfrac{9.81 \times 36}{2\cos^2\ang{75}\,
(6.0\tan\ang{75} - 1.05)} = \dfrac{353.2}{2.86} = 123.5$:
$ v_0 = \qty{11.1}{m/s}$.

\textbf{17.} $ y(5.55) = 5.55\tan\ang{75} - 18.26 = \qty{2.45}{m}$
versão acima, ou seja,\ \qty{4.45}{m} acima do chão: \qty{0.50}{m}
acima da borda superior do tabuleiro.

\textbf{18.} Ápice $2.00 + (11.1\sin\ang{75})^2/2g \approx
\qty{7.9}{m}$; voo $6.0/(11.1\cos\ang{75}) = \qty{2.1}{s}$ ---
dois \omterm{def:g10:orders-of-magnitude:unit}{segundos} completos com a bola pendurada \qty{8}{m} acima.

\textbf{19.} Arrastar encurta cada tacada e aumenta a descida; o
efeito cresce com velocidade, então o \qty{17.5}{m/s} levantamento de quadra inteira
sofre mais --- é real \omterm{def:g12:projectile-motion:range}{faixa} cai a poucos metros de
\qty{31}{m}, comendo na margem.

\textbf{20.} \emph{O veredicto da campainha}: o \qty{26.0}{m}
demandas de tiro \qty{16.0}{m/s} e um braço fornece cerca
\qty{17.5}{m/s}, um \omterm{def:g12:projectile-motion:range}{faixa} margem de \qty{5}{m} ($\approx 20\%$) que
a \omterm{def:g11:circuits-and-power:resistance}{resistência} do ar diminui, mas não apaga. Possível --- o que é
exatamente por que, algumas vezes por temporada, ele entra.
\end{solution}
