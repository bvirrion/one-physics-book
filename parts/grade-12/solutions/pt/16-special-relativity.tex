\section*{Capítulo \ref{ch:g12:special-relativity} --- Relatividade Especial: Dilatação do Tempo}

\begin{solution}{exo:g12:special-relativity:1}
$\gamma = 1/\sqrt{1 - (v/c)^2}$: $1.01$ ($0.1c $); $1.25$ ($0.6c $);
$1.67$ ($0.8c $); $10.0$ ($0.995c $).
\end{solution}

\begin{solution}{exo:g12:special-relativity:2}
$\gamma = 1.25$, então o terreno mede $1.25 \times 10.0 =
\qty{12.5}{s}$. O navio \qty{10.0}{s} é o \omterm{def:g12:special-relativity:proper-time}{hora adequada}: o
metrônomo (um relógio) está presente em ambas as batidas.
\end{solution}

\begin{solution}{exo:g12:special-relativity:3}
$ v = c\sqrt{1 - 1/\gamma^2}$: $\gamma = 2$ dá $ v = 0.866c $;
$\gamma = 10$ dá $ v = 0.995c $.
\end{solution}

\begin{solution}{exo:g12:special-relativity:4}
$ v = \qty{36.1}{m/s}$, $ v/c = \num{1.2e-7}$;
$\gamma - 1 \approx (\num{1.2e-7})^2/2 \approx \num{7.2e-15}$. Um
segunda perdida depois $1/\num{7.2e-15} \approx \qty{1.4e14}{s}$ --- sobre
quatro milhões de anos de condução.
\end{solution}

\begin{solution}{exo:g12:special-relativity:5}
O passageiro dirige-se em direção ao flash frontal, de modo que ele o alcança primeiro;
pelo \omterm{def:g10:orders-of-magnitude:unit}{segundo} postulado, ambos os flashes viajam ao mesmo tempo $ c $ sobre o
distâncias iguais de meio trem \emph{em seu quadro}, então uma chegada mais cedo
significa um ataque anterior. A simultaneidade depende do quadro: cada veredicto
está correto em seu próprio quadro.
\end{solution}

\begin{solution}{exo:g12:special-relativity:6}
$\gamma = 1/\sqrt{1 - 0.99^2} = 7.09$; vida útil do laboratório
$7.09 \times \num{2.6e-8} = \qty{1.8e-7}{s}$. Clássico \omterm{def:g12:projectile-motion:range}{faixa}
$0.99 \times \num{3.00e8} \times \num{2.6e-8} \approx \qty{7.7}{m}$;
real $7.09 \times 7.7 \approx \qty{55}{m}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:7}
$\gamma = 1.5$ no $ v = c\sqrt{1 - 1/2.25} \approx 0.75c $;
$\gamma = 3$ no $\approx 0.94c $. A assíntota: $\gamma $ (e o
energia custo de velocidade adicional) diverge conforme $ v \to c $ --- não finito
esforço leva um corpo enorme à velocidade da luz.
\end{solution}

\begin{solution}{exo:g12:special-relativity:8}
$\Delta t_0 = 2L/c = 3.0/\num{3.00e8} = \qty{1.0e-8}{s}$; no $0.8c $,
$\gamma = 5/3$, então $\Delta t = \qty{1.7e-8}{s}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:9}
\omterm{def:g12:special-relativity:proper-time}{Hora adequada} pertence ao relógio presente em ambos \omterm{def:g12:special-relativity:proper-time}{eventos}: (a) o
usuário; (b) o múon; (c) o piloto --- apenas o navio está em ambos os
partida e a chegada.
\end{solution}

\begin{solution}{exo:g12:special-relativity:10}
$\gamma - 1 \approx 7700^2/(2 \times \num{9.0e16}) \approx
\num{3.3e-10}$; sobre \qty{1.58e7}{s} o astronauta segue por
$\num{3.3e-10} \times \num{1.58e7} \approx \qty{5.2e-3}{s}$ --- sobre
cinco milissegundos mais jovem.
\end{solution}

\begin{solution}{exo:g12:special-relativity:11}
$ E = \num{1.0e-3} \times (\num{3.00e8})^2 = \qty{9.0e13}{J}$. UM
\qty{1.0}{GW} estação entrega $\num{1.0e9} \times 86400 \approx
\qty{8.6e13}{J}$ por dia --- um grama é um dia de sua produção, e
$\num{9.0e13}/\num{8.5e5} \approx \num{e8}$: cem milhões
carros em alta velocidade.
\end{solution}

\begin{solution}{exo:g12:special-relativity:12}
Quadrar e inverter: $\gamma^2(1 - v^2/c^2) = 1$, então
$ v^2/c^2 = 1 - 1/\gamma^2$, por isso $ v = c\sqrt{1 - 1/\gamma^2}$. Para
$\gamma = 30$: $ v = c\sqrt{1 - 1/900} = 0.99944c $.
\end{solution}

\begin{solution}{exo:g12:special-relativity:13}
Espessura contraída $ L = \qty{15}{km}/30 = \qty{500}{m}$; tempo de varredura
$500/(0.99944 \times \num{3.00e8}) \approx \qty{1.7e-6}{s} <
\qty{2.2}{\micro s}$. Ambos os quadros concordam que o múon atinge o solo ---
um culpa uma vida prolongada, o outro uma atmosfera encolhida.
\end{solution}

\begin{solution}{exo:g12:special-relativity:14}
Laboratório: $3200/\num{3.00e8} \approx \qty{1.1e-5}{s}$ (velocidade
$\approx c $). Quadro de elétrons: tubo contraído para
$3200/\num{1.0e5} = \qty{3.2}{cm}$, cruzado em
$\num{1.1e-5}/\num{1.0e5} \approx \qty{1.1e-10}{s}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:15}
Terra: $4.2/0.9 \approx \qty{4.7}{years}$. Com
$\gamma = 1/\sqrt{1 - 0.81} = 2.29$, os registros do relógio do probe
$4.7/2.29 \approx \qty{2.0}{years}$.
\end{solution}

\begin{solution}{pb:g12:special-relativity:1}
\textbf{1.} $ c\,\Delta t_0 = \num{3.00e8} \times \num{2.2e-6} \approx
\qty{660}{m}$.

\textbf{2.} $\num{15000}/660 \approx 23$ vidas.

\textbf{3.} $\mathrm{e}^{-23} \approx \num{e-10}$: menos de um múon
em dez bilhões deveria chegar, mas o fluxo é de um por \unit{cm^2} por
minuto --- a física clássica é categoricamente contradita pelo céu.

\textbf{4.} Vida útil do solo $30 \times \qty{2.2}{\micro s} =
\qty{66}{\micro s}$; \omterm{def:g12:projectile-motion:range}{faixa} $\approx \num{3.00e8} \times \num{6.6e-5}
\approx \qty{20}{km} > \qty{15}{km}$: a chegada é explicada.

\textbf{5.} $ v = c\sqrt{1 - 1/900} = 0.99944c $.

\textbf{6.} $\Delta t_0 = 2L/c = 3.0/\num{3.00e8} = \qty{1.0e-8}{s}$.

\textbf{7.} Horizontal $ v\,\Delta t/2$; \omterm{def:g10:universal-gravitation:weight}{vertical} $ L $; o inclinado
medidas de meio caminho $ c\,\Delta t/2$ porque o \omterm{def:g10:orders-of-magnitude:unit}{segundo} postulado corrige
o fótona velocidade em $ c $ no \omterm{def:g10:relative-motion:frames}{estrutura de solo} também.

\textbf{8.} $(c\,\Delta t/2)^2 = L^2 + (v\,\Delta t/2)^2$ dá
$\Delta t^2(c^2 - v^2) = 4L^2$, então $\Delta t =
(2L/c)/\sqrt{1 - v^2/c^2} = \gamma\,\Delta t_0$.

\textbf{9.} $\gamma = 1/\sqrt{1 - 0.36} = 1.25$;
$\Delta t = \qty{1.25e-8}{s}$.

\textbf{10.} O do navio: seu único relógio está presente em ambos os tiques, então
lê o \omterm{def:g12:special-relativity:proper-time}{hora adequada}; o solo precisa de dois relógios sincronizados.

\textbf{11.} $\gamma = 1/\sqrt{1 - 0.64} = 1/0.6 = 5/3 \approx 1.67$.

\textbf{12.} $10.0/(5/3) = \qty{6.0}{years}$.

\textbf{13.} Saída $\qty{5.0}{years}$ no $0.8c $: $4.0$ anos-luz.

\textbf{14.} $10.0 - 6.0 = \qty{4.0}{years}$: o astronauta tem quatro anos
anos mais nova que seu irmão gêmeo.

\textbf{15.} A situação não é simétrica: só o astronauta vira
ao redor --- ela acelera e muda \omterm{def:g12:special-relativity:inertial}{quadro inercial}, enquanto a Terra
gêmeo fica em um --- então apenas o relógio dela registra o tempo mais curto.

\textbf{16.} $ v = 2\pi \times \num{2.66e7}/\num{43200} \approx
\qty{3.9e3}{m/s}$.

\textbf{17.} $\gamma - 1 \approx (\num{3.87e3})^2/(2 \times
\num{9.0e16}) \approx \num{8.3e-11}$.

\textbf{18.} $\num{8.3e-11} \times \num{86400} \approx \qty{7.2e-6}{s}
= \qty{7.2}{\micro s}$ por dia lentamente.

\textbf{19.} $ c \times \num{7.2e-6} \approx \qty{2.2}{km}$ de posição
erro por dia.

\textbf{20.} Deriva líquida $45 - 7.2 \approx \qty{38}{\micro s}$ por dia
rápido; erro $ c \times \num{38e-6} \approx \qty{11}{km}$ por dia ---
a conta da relatividade que todo relógio GPS paga antecipadamente, desafinada em
o chão para que ele funcione corretamente \omterm{def:g10:universal-gravitation:satellite}{órbita}.
\end{solution}
