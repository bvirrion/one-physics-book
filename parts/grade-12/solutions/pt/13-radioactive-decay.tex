\begin{solution}{exo:g12:radioactive-decay:1}
$T_{1/2} = \ln 2/\lambda = 0.693/\num{1.0e-6} = \qty{6.9e5}{s}

\end{solution}

\begin{solution}{exo:g12:radioactive-decay:2}
$ T_{1/2} = 3.8 \times \num{86400} = \qty{3.28e5}{s}$;
$\lambda = 0.693/\num{3.28e5} = \qty{2.1e-6}{s^{-1}}$;
$\mathcal A = \lambda N = \num{2.1e-6} \times \num{1.0e10} \approx
\qty{2.1e4}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:3}
$\dd N/\dd t = N_0(-\lambda)e^{-\lambda t} = -\lambda N $ ---
a equação é válida; $ N(0) = N_0 e^0 = N_0$;
$ N(T_{1/2})/N_0 = e^{-\lambda \ln 2/\lambda} = e^{-\ln 2} =
\tfrac12$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:4}

\qty{5.0e4}{Bq}$. $300$ anos $= 10\,T_{1/2}$:
$\mathcal A = \num{4.0e5}/1024 \approx \qty{3.9e2}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:5}

\begin{solution}{exo:g12:radioactive-decay:6}

\begin{solution}{exo:g12:radioactive-decay:7}
Ambos esperam que metade permaneça: $10$, e \num{1.0e20}. Flutuações
$\sim 1/\sqrt{n}$ da contagem: com $10$ decai, cerca $30\%$ ---
$7$ ou $13$ não seria nenhuma surpresa; com \num{1.0e20}, sobre
$10^{-10}$. Somente a grande amostra obedece à lei dos valores mensuráveis.
precisão.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:8}
$ m = 140 \times \num{1.2e-4} = \qty{1.7e-2}{g}$;
$ N = (0.0168/40) \times \num{6.02e23} = \num{2.5e20}$.
$\lambda = 0.693/(\num{1.25e9} \times \num{3.156e7}) =
\qty{1.76e-17}{s^{-1}}$; $\mathcal A = \lambda N \approx

carbono-14).
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:9}
$\lambda = 0.693/6.0 = \qty{0.116}{h^{-1}}$. No $15$ horas:
$\mathcal A = 500\,e^{-0.116 \times 15} = 500 \times 0.177 \approx
\qty{88}{MBq}$. Abaixo \qty{1}{MBq}:
$ t = \ln(500)/0.116 \approx \qty{54}{hours}$ --- sobre $9$
meias-vidas.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:10}
Carvão: $3.4/13.6 = \tfrac14$, exatamente $2$ meias-vidas:
$ t = \num{1.15e4}$ anos. Osso:
$ t = \num{5730} \times \ln(1/0.30)/\ln 2 = \num{5730} \times 1.74
\approx \num{1.0e4}$ anos.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:11}
$\lambda = 0.693/3.8 = \qty{0.182}{d^{-1}}$;
$ t = \ln(100)/\lambda = 4.61/0.182 \approx 25$ dias (cerca de $6.6$
meias-vidas). Mas a cadeia de urânio no granito circundante e
o solo produz radônio continuamente: uma aeração única é desfeita dentro
semanas, portanto a ventilação deve ser permanente.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:12}
$ N/N_0 = e^{-\lambda t}$ dá $\lambda t = \ln(N_0/N)$, e
$\lambda = \ln 2/T_{1/2}$ dá $ t = T_{1/2}\ln(N_0/N)/\ln 2$. Aqui
$ t = \num{5730} \times \ln(20)/\ln 2 = \num{5730} \times 4.32
\approx \num{2.5e4}$ anos.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:13}
$ N_{\mathrm K} = N_0 e^{-\lambda t}$ e
$ N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, então
$ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$. Razão $3.0$:
$ e^{\lambda t} = 4$, então $\lambda t = 2\ln 2$ e
$ t = 2\,T_{1/2} = \num{2.5e9}$ anos.
\end{solution}

$ T_{1/2} = 0.693/\num{9.6e-3} \approx \qty{72}{s}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:15}

\begin{solution}{pb:g12:radioactive-decay:1}
\textbf{1.} A troca com a atmosfera (comer, respirar) mantém
a proporção complementada pelo valor atmosférico; a morte interrompe o
a ingestão e a decomposição assumem o controle.

\textbf{2.} $ T_{1/2} = \num{5730} \times \num{3.156e7} =
\qty{1.81e11}{s}$, então $\lambda = 0.693/\num{1.81e11} =
\qty{3.8e-12}{s^{-1}}$.

\textbf{3.} $\dd N/\dd t = -\lambda N_0 e^{-\lambda t} = -\lambda N $,
e $ N(0) = N_0$.

\textbf{4.} $\mathcal A_0 = 13.6/60 = \qty{0.227}{Bq}$ por grama;
$ N = \mathcal A_0/\lambda = 0.227/\num{3.8e-12} \approx \num{5.9e10}$
\omterm{def:g11:fundamental-interactions:ladder}{átomos} por grama.

\textbf{5.} $(1/12) \times \num{6.02e23} = \num{5.0e22}$ carbono
\omterm{def:g11:fundamental-interactions:ladder}{átomos}; proporção $\num{5.9e10}/\num{5.0e22} \approx \num{1.2e-12}$
--- um em $10^{12}$, como prometido.

\textbf{6.} $\mathcal A = \mathcal A_0 e^{-\lambda t}$ dá
$\lambda t = \ln(\mathcal A_0/\mathcal A)$, e $\lambda =
\ln 2/T_{1/2}$ transforma isso em
$ t = T_{1/2}\ln(\mathcal A_0/\mathcal A)/\ln 2$.

\textbf{7.} $ t = \num{5730} \times \ln(13.6/9.1)/\ln 2 =
\num{5730} \times 0.580 \approx \num{3.3e3}$ anos.

\textbf{8.} A morte da madeira --- o corte da árvore, não o
escultura; uma estátua cortada em madeira velha (ou uma falsificação esculpida em madeira antiga
madeira) é anterior ou posterior à data do seu material.

\textbf{9.} $ t = \num{5730} \times \ln(13.6/10.4)/\ln 2 =
\num{5730} \times 0.387 \approx \num{2.2e3}$ anos --- totalmente
\omterm{def:g12:light-as-wave:coherent}{consistente} com um escriba de vinte e dois séculos atrás.

\textbf{10.} $\Delta t \approx \Delta\mathcal A/(\lambda\mathcal A)$
com $\lambda = \ln 2/5730 = \num{1.21e-4}$ por ano:
$\Delta t = (0.2/10.4)/\num{1.21e-4} \approx \num{1.6e2}$ anos ---
uma data válida para um ou dois séculos.

\textbf{11.} $ t = \num{5730} \times \ln(13.6/13.4)/\ln 2 \approx
120$ anos --- indistinguível do moderno dentro do $\pm 160$-ano
incerteza: o pergaminho é recente, o ``antigo'' é falso.

\textbf{12.} $13.6/2^n < 0.2$ precisa $2^n > 68$: $ n = 7$ meias-vidas
(desde $2^7 = 128$), sobre $\num{4.0e4}$ anos.

\textbf{13.} Dez meias-vidas: cerca de \num{5.7e4} anos --- o
horizonte prático, cerca de sessenta mil anos.

\textbf{14.} $\num{6.6e7}/\num{5730} \approx \num{11500}$ meias-vidas.
Um grama \num{5.9e10} \omterm{def:g11:fundamental-interactions:ladder}{átomos} estão exaustos depois
$2^n > \num{5.9e10}$, ou seja\ $ n \approx 36$ meias-vidas
($\approx \num{2e5}$ anos): muito antes da era dos dinossauros, nem um
carbono-14 \omterm{def:g11:fundamental-interactions:ladder}{átomo} resta --- não há nada para contar.

\textbf{15.} Um relógio muito mais lento: potássio-40 ou urânio-238, com
meias-vidas de bilhões de anos.

\textbf{16.} A lava derretida deixa o argônio borbulhar, então o líquido fresco
cristal solidificado contém potássio, mas zero argônio: a proporção começa
em $0$ na solidificação --- o relógio é zerado.

\textbf{17.} Como no exercício: $ N_{\mathrm K} = N_0 e^{-\lambda t}$,
$ N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, então
$ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.

\textbf{18.} $ e^{\lambda t} = 1.60$:
$ t = \num{1.25e9} \times \ln(1.60)/\ln 2 = \num{1.25e9} \times 0.678
\approx \num{8.5e8}$ anos.

\textbf{19.} Não, duas vezes: às \num{8.5e8} anos de funcionamento do carbono-14
$10^5$ meias-vidas --- silêncio; no \num{3.3e3} anos potássio-40 tem
correr \num{2.6e-6} de um \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} --- a proporção de argônio
$\approx \lambda t \sim \num{2e-6}$ é incomensuravelmente pequeno. Cada
relógio lê apenas cerca de um décimo de \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} para cerca de dez.

\textbf{20.} Estátua $\approx \num{3.3e3}$ anos, papiro
$\approx \num{2.2e3}$ anos, rocha $\approx \num{8.5e8}$ anos ---
três relógios, uma lei: $ N = N_0\,e^{-\lambda t}$, leia de trás para frente.
\end{solution}
