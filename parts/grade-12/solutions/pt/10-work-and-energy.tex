\section*{Capítulo \ref{ch:g12:work-and-energy} --- Trabalho e Energia Mecânica}

\begin{solution}{exo:g12:work-and-energy:1}
$E_p = \tfrac12 \times 200 \times 0.050^2 = \qty{0.25}{J}$. Duplicado
esticar: $\times 4$, ou seja\ \qty{1.0}{J}. O \omterm{def:g10:orders-of-magnitude:unit}{segundo} \qty{5.0}{cm}
sozinho custa $1.0 - 0.25 = \qty{0.75}{J}$ --- o \omterm{def:g10:inertia:force}{vigor} é maior lá.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:2}
$ W = mg(z_A - z_B) = 0.90 \times 9.81 \times (12 - 30) \approx
\qty{-1.6e2}{J}$, qualquer que seja a rota --- a mesma na reta
subir. Ida e volta: \qty{0}{J} (força conservadora).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:3}
(a), (b) \omterm{def:g12:work-and-energy:conservative}{conservador}: \omterm{def:g11:work-of-force:work}{trabalhar} fixado pelos pontos finais (queda de altitude;
esticar). (c), (e) \omterm{def:g12:work-and-energy:conservative}{não conservador}: $-fL $ e arraste a carga por \omterm{def:g10:orders-of-magnitude:unit}{medidor}
viajou. (d) sem trabalho: sempre perpendicular ao movimento.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:4}
$ h = L(1 - \cos\qty{60}{\degree}) = \qty{1.0}{m}$;
$ v = \sqrt{2 \times 9.81 \times 1.0} \approx \qty{4.4}{m/s}$. O
\omterm{def:g11:circuits-and-power:voltage}{tensão} é perpendicular ao movimento em cada instante: zero \omterm{def:g11:work-of-force:power}{poder},
zero \omterm{def:g11:work-of-force:work}{trabalhar}.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:5}
$ F = P/v = 250/9.0 \approx \qty{28}{N}$. No \qty{5.0}{m/s}:
$ F = 250/5.0 = \qty{50}{N}$ (principalmente o componente gravitacional da encosta).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:6}
$ E_p = \tfrac12 \times 450 \times 0.080^2 = \qty{1.44}{J}$;
$ v = \sqrt{2 \times 1.44/0.020} = \qty{12}{m/s}$;
$ h = E_p/mg = 1.44/(0.020 \times 9.81) \approx \qty{7.3}{m}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:7}
Trapézios (\qty{2.0}{cm} passos): $(0.03 + 0.10 + 0.19 + 0.30) =
\qty{0.62}{J}$. Modelo em linha reta: $ k = 18.0/0.080 =
\qty{225}{N/m}$, $\tfrac12 kx^2 = \qty{0.72}{J}$. A curva medida
afunda abaixo da linha reta em pequenas $ x $, então a área verdadeira é
menor.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:8}
$\Delta E_m = \tfrac12 \times 55 \times 9.0^2 - 55 \times 9.81 \times
8.0 = 2228 - 4316 \approx \qty{-2.1e3}{J}$;
$ f = 2089/25 \approx \qty{84}{N}$. \omterm{def:g10:universal-gravitation:weight}{Peso} \omterm{def:g12:work-and-energy:conservative}{conservador}, \omterm{def:g12:newtons-laws:friction}{força normal}
sem trabalho: só a gota e o caminho \emph{comprimento} entre, não o
forma.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:9}
$ H_{\min} = \tfrac52 R = \qty{0.50}{m}$. De \qty{60}{cm}:
$ v_{\text{top}} = \sqrt{2 \times 9.81 \times (0.60 - 0.40)} \approx
\qty{2.0}{m/s}$, e $ v_{\text{top}}^2/(gR) = 2.0$ --- o dobro
mínimo. \omterm{def:g10:inertia:inventory}{Atrito} desnatado energia o tempo todo, então \omterm{def:g12:projectile-motion:range}{faixas} reais precisam de mais
altura.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:10}
Pontos de viragem: $8.0\,x^2 = 2.0$, $ x = \pm\qty{0.50}{m}$. Máximo
velocidade em $ x = 0$: $ v = \sqrt{2 \times 2.0/0.100} \approx
\qty{6.3}{m/s}$. Uma oscilação para frente e para trás em um poço parabólico ---
a massa-mola osciladorde $ E_p $ é exatamente desta forma.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:11}
$\tfrac12 kx^2 = \tfrac12 mv^2$:
$ x = v\sqrt{m/k} = 2.0\sqrt{1200/\num{6.0e5}} \approx \qty{8.9}{cm}$.
No \qty{4.0}{m/s}: \qty{18}{cm} --- duplo, não quádruplo: o energia
quadruplica, mas $ E_p $ cresce como $ x^2$, então $ x $ apenas duplica.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:12}
(um) $ E_m = \qty{5.0}{J} < \qty{6.0}{J}$: não pode passar; ele para em
o flanco onde $ E_p = \qty{5.0}{J}$ e reverte. (b) Topo da colina:
$ E_k = \qty{1.0}{J}$, $ v = \sqrt{2 \times 1.0/0.50} = \qty{2.0}{m/s}$;
vale: $ E_k = \qty{5.0}{J}$, $ v \approx \qty{4.5}{m/s}$. (c) $ x = 0$:
instável (máximo); $ x = \qty{1.5}{m}$: \omterm{prop:g12:work-and-energy:equilibria}{estável} (mínimo).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:13}
$\tfrac12 mv^2 = mgR_E $:
$ v = \sqrt{2 \times 9.81 \times \num{6.37e6}} \approx \qty{1.12e4}{m/s}
= \qty{11.2}{km/s}$. Todo termo é proporcional a $ m $: sonda e
seixo igual. Logo abaixo dele, o lançamento ainda sobe enormemente,
encontra um \omterm{def:g12:work-and-energy:diagram}{ponto de viragem}, e recua: vinculado.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:14}
$ E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.060^2 =
\qty{7.2e-2}{J}$; $ v_{\max} = \sqrt{2E_m/m} =
\sqrt{2 \times 0.072/0.250} \approx \qty{0.76}{m/s}$;

\begin{solution}{exo:g12:work-and-energy:15}
$ h = v^2/2g = 9.5^2/19.62 \approx \qty{4.6}{m}$; bar $\approx
4.6 + 1.0 = \qty{5.6}{m}$. O recorde é maior: os volteadores somam \omterm{def:g11:work-of-force:work}{trabalhar}

a escala --- o salto de nenhum velocista alcançará \qty{10}{m}.
\end{solution}

\begin{solution}{pb:g12:work-and-energy:1}
\textbf{1.} $ v = \sqrt{2gH} = \sqrt{2 \times 9.81 \times 45} \approx
\qty{29.7}{m/s}$: \omterm{def:g10:universal-gravitation:weight}{peso} \omterm{def:g12:work-and-energy:conservative}{conservador}, \omterm{def:g12:newtons-laws:friction}{força normal} sem trabalho --- o
o perfil desaparece.

\textbf{2.} Portão: $ E_m = mgH = 70 \times 9.81 \times 45 \approx
\qty{3.09e4}{J}$. Decolar: $ E_m = \tfrac12 \times 70 \times 26^2
\approx \qty{2.37e4}{J}$.

\textbf{3.} $\Delta E_m \approx \qty{-7.2e3}{J}$; o
\omterm{def:g11:mechanical-energy:em}{energia mecânica} teorema o carrega para o forças não conservadoras:
neve \omterm{def:g10:inertia:inventory}{atrito} e arrasto aéreo.

\textbf{4.} $ f = \num{7.2e3}/90 \approx \qty{80}{N}$.

\textbf{5.} $\num{7.2e3}/\num{3.09e4} \approx 23\%$. Cera abaixa $ f $,
agachar reduz o arrasto: ambos encolhem $ fL $, o único termo negociável.

\textbf{6.} \omterm{def:g12:projectile-motion:freefall}{Queda livre} com velocidade inicial horizontal --- a
projétil movimento. $ v_x = \qty{26}{m/s}$;
$ v_y = \sqrt{2 \times 9.81 \times 40} \approx \qty{28.0}{m/s}$.

\textbf{7.} $ v = \sqrt{26^2 + 28.0^2} = \sqrt{1461} \approx
\qty{38.2}{m/s}$; energia: $ v = \sqrt{v_0^2 + 2gh} = \sqrt{676 + 785}$
--- idêntico. Sobre \qty{138}{km/h}.

\textbf{8.} $ t = v_y/g = 28.0/9.81 \approx \qty{2.9}{s}$;
$ d = v_0 t \approx \qty{74}{m}$.

\textbf{9.} Não. A sustentação é perpendicular à velocidade (sem trabalho) e
arrasto se opõe a ele (negativo \omterm{def:g11:work-of-force:work}{trabalhar}): $\Delta E_m \leq 0$, então o verdadeiro
a velocidade de pouso só pode ser \emph{menor} --- o ar estica o
vôo, não o alimenta.

\textbf{10.} $\alpha = \arctan(28.0/26) \approx \qty{47}{\degree}$
abaixo da horizontal.

\textbf{11.} Ângulo em relação à inclinação: $47 - 35 = \qty{12}{\degree}$.
$ v_\parallel = 38.2\cos\qty{12}{\degree} \approx \qty{37.4}{m/s}$;
$ v_\perp = 38.2\sin\qty{12}{\degree} \approx \qty{7.9}{m/s}$.

\textbf{12.} Absorvido: $\tfrac12 \times 70 \times 7.9^2 \approx
\qty{2.2e3}{J}$, contra $\tfrac12 \times 70 \times 38.2^2 \approx
\qty{5.1e4}{J}$ no total: cerca de $4\%$.

\textbf{13.} $ h_{\text{eq}} = 7.9^2/19.62 \approx \qty{3.2}{m}$ ---
um salto ousado de uma parede. Plano: $38.2^2/19.62 \approx \qty{74}{m}$ ---
insustentável.

\textbf{14.} Uma inclinação paralela à \omterm{def:g10:relative-motion:trajectory}{trajetória} de vôo mantém $ v_\perp $
pequeno, então as pernas absorvem metros, não toda a montanha de cinético
energia.

\textbf{15.} $ H'_0 = v_0'^2/2g = 576/19.62 \approx \qty{29.4}{m}$.

\textbf{16.} $ mgH' - f\,\dfrac{H'}{\sin\qty{35}{\degree}} =
\tfrac12 m v_0'^2$.

\textbf{17.} $ H' = \dfrac{\tfrac12 \times 70 \times 576}
{686.7 - 80/0.574} = \dfrac{\num{20160}}{547} \approx \qty{37}{m}$.

\textbf{18.} $36.8/29.4 \approx 1.25$: um $25\%$ sobretaxa.

\textbf{19.} Para \qty{60}{kg}: $ H' = \num{17280}/(588.6 - 139.5)
\approx \qty{38.5}{m}$. O \emph{isqueiro} saltador precisa do mais alto
torre: a \omterm{def:g10:inertia:inventory}{atrito} conta $ fL $ é a mesma coisa, mas ela energia orçamento
$ mgH'$ é menor.

\textbf{20.} Anuncie um \qty{37}{m} torre para \qty{24}{m/s} de
decolagem --- e confesso que um quarto disso, alguns \qty{7}{m} de
concreto, é atritosobretaxa sobre o sonho sem \omterm{def:g10:inertia:inventory}{atrito}.
\end{solution}
