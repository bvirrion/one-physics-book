\section*{Chapter \ref{ch:g12:doppler-effect} --- The Doppler Effect}

\begin{solution}{exo:g12:doppler-effect:1}
(a) $f' = 700 \times 340/315 \approx \qty{756}{Hz}$.
(b) $f' = 700 \times 340/365 \approx \qty{652}{Hz}$.
(c) \qty{700}{Hz}: driver and siren move together, their distance never changes.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:2}
The \emph{loudness} fades because the wave spreads out --- it depends on the distance.
The \emph{pitch} drops from approach to recession --- the Doppler effect, which depends
on how fast the distance changes. Two independent effects.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:3}
Moving observer: $f' = 700\,(1 + 6.0/340) \approx \qty{712}{Hz}$; riding away,
$700\,(1 - 6.0/340) \approx \qty{688}{Hz}$. The siren is at rest in the air, so the wave
pattern is undisturbed; only the observer runs through it.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:4}
Ahead: $\lambda' = 310/400 = \qty{0.775}{m}$; behind: $370/400 = \qty{0.925}{m}$; at
rest: $340/400 = \qty{0.85}{m}$. The short wavelength reaches whoever the train
approaches, the long one whoever it leaves.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:5}
$\Delta f/f \approx 5.0/340 \approx 1.5\%$: $\Delta f \approx \qty{6.5}{Hz}$, so about
\qty{446}{Hz} --- a quarter of a semitone. A musician would just notice.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:6}
The wall receives $f\,v/(v - v_b)$ (moving source); the bat, a moving observer closing on
the echo, hears $(1 + v_b/v)$ times that: $f' = f(v + v_b)/(v - v_b) = 50.0 \times
346/334 \approx \qty{51.8}{kHz}$, a shift of $+\qty{1.8}{kHz}$. It is the radar-gun
double shift with emitter and reflector roles swapped: only the closing speed matters.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:7}
$u = c\,\Delta f/(2f) = \num{3.00e8} \times 4000/(2 \times \num{24.15e9}) \approx
\qty{24.8}{m/s} \approx \qty{89}{km/h}$. At \qty{50}{km/h} ($\qty{13.9}{m/s}$):
$\Delta f = 2 \times 13.9 \times \num{24.15e9}/\num{3.00e8} \approx \qty{2.2}{kHz}$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:8}
$u = v\,\Delta f/(2f) = 1540 \times 2600/\num{1.0e7} \approx \qty{0.40}{m/s}$.
Perpendicular flow has zero radial velocity, hence no shift at all.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:9}
(a) Source: $f' = 1000 \times 340/306 \approx \qty{1111}{Hz}$. (b) Observer:
$f' = 1000 \times 1.1 = \qty{1100}{Hz}$. Slow motion predicts \qty{1100}{Hz}: exact for
the observer, \qty{11}{Hz} short for the source --- the order-$x^2$ correction, $1\%$ of
$f$ for $x = 0.1$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:10}
$f = 2 \times 465 \times 415/880 \approx \qty{439}{Hz}$;
$v_s = 340 \times 50/880 \approx \qty{19.3}{m/s} \approx \qty{70}{km/h}$. The true $f$
is the harmonic mean: approach raises the frequency more than recession lowers it, so
$f$ sits below the arithmetic average (\qty{440}{Hz}).
\end{solution}

\begin{solution}{exo:g12:doppler-effect:11}
Longer wavelength: redshift, the star recedes. $v_r = c\,\Delta\lambda/\lambda =
\num{3.00e8} \times 0.2/656.3 \approx \qty{9.1e4}{m/s} \approx \qty{91}{km/s}$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:12}
$\Delta\lambda = \lambda K/c = 500 \times 55/\num{3.00e8} \approx \qty{9.2e-5}{nm}$;
$\Delta\lambda/\lambda = 55/\num{3.00e8} \approx \num{1.8e-7}$. The whole signal is two
parts in ten million: a spectrograph drifting by more than $10^{-7}$ buries it.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:13}
$v_r = 0.024\,c \approx \qty{7.2e6}{m/s}$. Speeds proportional to distance, the same in
every direction: all distances stretching uniformly --- an expanding universe with no
privileged center. At $2.4\%$ the corrections are of order $(v_r/c)^2 \approx
\num{6e-4}$: still trustworthy.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:14}
The two edges shift opposite ways, so the disk-averaged line is \emph{broadened}, not
shifted: full width $2\lambda v/c = 2 \times 656.3 \times \num{2.0e3}/\num{3.00e8}
\approx \qty{8.8e-3}{nm}$ (about $\pm\qty{4.4e-3}{nm}$).
\end{solution}

\begin{solution}{exo:g12:doppler-effect:15}
(a) Ahead of the car the crests are spaced $(v - v_s)T$; the truck flees at $v_o$ and
meets them at relative speed $v - v_o$: $f' = (v - v_o)/\bigl((v - v_s)T\bigr) =
f(v - v_o)/(v - v_s)$. (b) $f' = 700 \times 310/300 \approx \qty{723}{Hz}$.
(c) $v_o = v_s$ gives $f' = f$: the separation is constant, and no change of distance
means no Doppler shift, by definition.
\end{solution}

\begin{solution}{pb:g12:doppler-effect:1}
\textbf{1.} Each crest expands from the point where it was emitted: crests bunch ahead
of the moving horn and stretch behind it, so the approaching phone meets them faster
($f_1$) than the receding one ($f_2$).

\textbf{2.} $f_1 = f\,\dfrac{v}{v - u}$, \quad $f_2 = f\,\dfrac{v}{v + u}$.

\textbf{3.} $f_1 - f_2 = \frac{2fv\,u}{v^2 - u^2}$ and $f_1 + f_2 =
\frac{2fv^2}{v^2 - u^2}$; dividing, $u = v\,\frac{f_1 - f_2}{f_1 + f_2} =
340 \times 60/882 \approx \qty{23}{m/s} \approx \qty{83}{km/h}$.

\textbf{4.} $\dfrac{2f_1 f_2}{f_1 + f_2} = f$ (the common factor cancels):
$f = 2 \times 471 \times 411/882 \approx \qty{439}{Hz}$ --- the harmonic mean, below
the average \qty{441}{Hz} because approach boosts $f$ more than recession cuts it.

\textbf{5.} Ahead: $(v - u)/f = 316.9/439 \approx \qty{0.72}{m}$; behind:
$363.1/439 \approx \qty{0.83}{m}$; at rest: $340/439 \approx \qty{0.77}{m}$.

\textbf{6.} Once as a moving \emph{observer} receiving the wave, once as a moving
\emph{source} re-emitting what it received.

\textbf{7.} $f' = f\,(1 + u/c)/(1 - u/c) \approx f(1 + 2u/c)$, so
$\Delta f = f' - f \approx \dfrac{2u}{c}\,f$.

\textbf{8.} $u = c\,\Delta f/(2f) = \num{3.00e8} \times 3550/(2 \times \num{24.125e9})
\approx \qty{22}{m/s}$.

\textbf{9.} $22.1 \times 3.6 \approx \qty{79}{km/h}$: just under the \qty{80}{km/h}
limit.

\textbf{10.} $\Delta f = 2f/(3.6\,c) \approx \qty{45}{Hz}$ per \unit{km/h}. Beating the
echo against the emitted wave leaves only the \emph{difference}: an audio frequency,
trivially counted, though it is a $10^{-7}$ fraction of the carrier.

\textbf{11.} The wavelengths of known spectral lines, night after night; then
$v_r = c\,\Delta\lambda/\lambda$.

\textbf{12.} $K = \qty{55}{m/s}$, $P = \qty{4.2}{days}$.

\textbf{13.} $\Delta\lambda = 550 \times 55/\num{3.00e8} \approx \qty{1.0e-4}{nm}$,
$\Delta\lambda/\lambda \approx \num{1.8e-7}$ --- one part in five million, some
$\num{4e5}$ times smaller than the train's $7\%$ shift.

\textbf{14.} Star and planet attract each other equally, so both orbit their common
center of mass: the star cannot stand still. A sinusoidal $v_r$ is uniform circular
motion seen edge-on --- a circular orbit.

\textbf{15.} Only the line-of-sight component changes the star--Earth distance, hence
the wavelength. Face-on, $v_r = 0$ at all times: no signal whatsoever.

\textbf{16.} $P = \qty{3.63e5}{s}$: $r^3 = \num{6.67e-11} \times \num{2.0e30} \times
(\num{3.63e5})^2/(4\pi^2) \approx \num{4.4e29}$, so $r \approx \qty{7.6e9}{m}$ ---
about $1/20$ of the Earth--Sun distance.

\textbf{17.} $V = 2\pi r/P = 2\pi \times \num{7.6e9}/\num{3.63e5} \approx
\qty{1.3e5}{m/s}$.

\textbf{18.} $m = M K/V = \num{2.0e30} \times 55/\num{1.3e5} \approx \qty{8.3e26}{kg}$.

\textbf{19.} About $0.4$ Jupiter masses ($\approx 140$ Earths), orbiting $8$ times
closer than Mercury: a gas giant roasting against its star --- a ``hot Jupiter''.

\textbf{20.} A tilt hides part of the motion: the measured $K$ is only the radial
share, so the true mass is \emph{larger} --- \qty{8.3e26}{kg} is a minimum. A periodic
shift of one part in five million just delivered the orbit, speed and minimum mass of a
planet nobody has ever seen.
\end{solution}
