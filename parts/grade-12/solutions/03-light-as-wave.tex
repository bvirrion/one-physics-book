\section*{Chapter \ref{ch:g12:light-as-wave} --- Light as a Wave: Diffraction and Interference}

\begin{solution}{exo:g12:light-as-wave:1}
$f = c/\lambda$: red $\num{3.00e8}/\num{7.0e-7} \approx
\qty{4.3e14}{Hz}$; violet $\approx \qty{7.5e14}{Hz}$ --- violet is
higher (shorter wavelength, same speed $c$).
\end{solution}

\begin{solution}{exo:g12:light-as-wave:2}
$\theta = \num{5.32e-7}/\num{2.0e-4} = \qty{2.7e-3}{rad}$;
$L = 2\theta D = 2 \times \num{2.66e-3} \times 1.5 \approx
\qty{8.0}{mm}$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:3}
Voice: $\lambda = 340/425 = \qty{0.80}{m}$, so $\lambda/a = 1.0$ ---
maximal diffraction, sound floods the corridor. Light:
$\lambda/a \approx \num{5.5e-7}/0.80 \approx \num{7e-7}$ ---
negligible spread: light keeps straight, so the corner hides it.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:4}
At $M$: $\delta = \qty{6.0}{cm} = 3\lambda$, constructive --- loud.
At $N$: $\delta = \qty{5.0}{cm} = (2 + \tfrac12)\lambda$, destructive
--- (nearly) silent.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:5}
$i = \lambda D/a = \num{6.33e-7} \times 1.8/\num{2.5e-4} \approx
\qty{4.6}{mm}$. Within $\pm\qty{1.0}{cm}$: $|k| \leq 10/4.6 = 2.2$,
so $k = -2, \dots, 2$: five bright fringes.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:6}
$d = 2\lambda D/L = 2 \times \num{6.5e-7} \times 1.6/0.028 \approx
\qty{74}{\micro m}$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:7}
$i = \qty{4.7}{mm}$, so $\lambda = i\,a/D = \num{4.7e-3} \times
\num{2.0e-4}/1.5 \approx \qty{6.3e-7}{m} = \qty{630}{nm}$: red.
Measuring ten interfringes divides the ruler's reading error by ten.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:8}
The lamps have (broadly) matching frequencies but are \emph{not in
step}: each emits short wave trains with randomly jumping phase, so
the sources are incoherent and the instantaneous fringes shift
billions of times per second --- the eye sees the average, uniform.
Young lights both slits from \emph{one} wave: the two copies inherit
every phase jump together and stay in step.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:9}
First to seventh $= 6i = \qty{18.0}{mm}$, so $i = \qty{3.0}{mm}$;
$\lambda = i\,a/D = \num{3.0e-3} \times \num{3.0e-4}/1.4 \approx
\qty{6.4e-7}{m} \approx \qty{640}{nm}$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:10}
$\sin\theta = k\lambda/d = 0.396\,k$: $k = 1$, $\theta = 23^\circ$;
$k = 2$, $\sin\theta = 0.79$, $\theta = 52^\circ$; $k = 3$ would need
$\sin\theta = 1.19 > 1$: impossible. Highest order $k = 2$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:11}
$\theta = \num{5.5e-7}/0.10 = \qty{5.5e-6}{rad}$. The headlights
subtend $1.5/\num{1.0e5} = \qty{1.5e-5}{rad}$, about three times the
blur: separated, just.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:12}
Front- and back-face reflections interfere; in white light each
thickness reinforces some wavelengths and cancels others, hence a
color. The film's thickness depends only on height, so equal-color
points form horizontal bands. Draining thins the film, so the
thickness that painted a given color sits ever lower: the bands
creep down.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:13}
$i = \lambda D/a$: violet $\num{4.0e-7} \times 1.5/\num{2.0e-4} =
\qty{3.0}{mm}$; red $\qty{5.6}{mm}$. Center: all colors bright ---
one white fringe. A few millimetres out the color patterns have
slid apart: iridescent fringes. Farther, maxima of all colors
overlap everywhere: uniform white blur.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:14}
$\theta = \num{5.5e-7}/2.4 = \qty{2.3e-7}{rad}$; at \qty{250}{km}
that is $\num{2.3e-7} \times \num{2.5e5} \approx \qty{6}{cm}$. It can
count cars and spot a person, but newsprint (millimetre letters) is
twenty times below the diffraction limit: the claim is myth.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:15}
The frequency is fixed by the source: unchanged. The speed drops to
$c/n$, so $\lambda' = \lambda/n$ and
$i' = i/n = 4.0/1.33 \approx \qty{3.0}{mm}$: the fringes tighten.
\end{solution}

\begin{solution}{pb:g12:light-as-wave:1}
\textbf{1.} $\theta = \lambda/a = \num{6.33e-7}/\num{1.00e-4} =
\qty{6.3e-3}{rad}$.

\textbf{2.} Half-width $D\tan\theta \approx D\theta = \lambda D/a$,
so $L = 2\lambda D/a = 2 \times \num{6.33e-7} \times
3.00/\num{1.00e-4} = \qty{3.8}{cm}$.

\textbf{3.} $(3.8 - 3.7)/3.8 \approx 3\%$: within a ruler's
precision --- validated.

\textbf{4.} $L \propto 1/a$: halving $a$ doubles $L$, so
$L = \qty{7.6}{cm}$. Narrower obstacle-or-slit, wider pattern.

\textbf{5.} $\tan(\num{6.33e-3}) = \num{6.33008e-3}$: they agree to
about $\num{1e-5}$ relative --- the approximation is far better than
any measurement here.

\textbf{6.} By \cref{prop:g12:light-as-wave:hair}: an obstacle of width
$d$ diffracts like a slit of width $d$, so $L = 2\lambda D/d$ still
holds.

\textbf{7.} $d = 2\lambda D/L = 2 \times \num{6.33e-7} \times
3.00/0.054 \approx \qty{70}{\micro m}$.

\textbf{8.} $L = \qty{5.2}{cm} \to d = \qty{73}{\micro m}$;
$L = \qty{5.6}{cm} \to d = \qty{68}{\micro m}$: so
$d = \qty{70 \pm 3}{\micro m}$.

\textbf{9.} $d = \num{3.80e-6}/0.076 = \qty{50}{\micro m}$: the
friend's is finer --- and $L \propto 1/d$, so the finer hair throws
the wider pattern.

\textbf{10.} No: a lamp is white (each wavelength paints a different
pattern, and they blur) and incoherent-and-wide (no single clean beam
to diffract). The laser is monochromatic and directional.

\textbf{11.} Only copies of one wave are coherent; two independent
sources drift out of step and the fringes wash out
(\cref{rem:g12:light-as-wave:lamps}).

\textbf{12.} At the center $\delta = 0 = 0 \times \lambda$ for every
$\lambda$: constructive regardless of wavelength.

\textbf{13.} $i = 4.26/10 = \qty{4.3}{mm}$; one span of ten is read
with the same ruler error as one interfringe, so the error on $i$ is
divided by ten.

\textbf{14.} $\lambda = i\,a'/D = \num{4.26e-3} \times
\num{2.50e-4}/2.00 = \num{5.33e-7} \approx \qty{533}{nm}$.

\textbf{15.} $\qty{533}{nm}$ lies inside $\qty{532 \pm 10}{nm}$:
consistent; that wavelength is green.

\textbf{16.} $\tan\theta = 0.43/1.00$, so $\theta = 23^\circ$. Here
$\tan\theta = 0.43$ while $\sin\theta = 0.40$: at $23^\circ$ the
small-angle identification of $\sin$, $\tan$ and $\theta$ is off by
almost $10\%$ --- use the exact functions.

\textbf{17.} $d = \lambda/\sin\theta = \num{6.33e-7}/0.396 \approx
\qty{1.6}{\micro m}$.

\textbf{18.} $\tan\theta = 1.65$, $\theta = 59^\circ$,
$\sin\theta = 0.855$: $d = \num{6.33e-7}/0.855 \approx
\qty{0.74}{\micro m}$.

\textbf{19.} $1.6/0.74 \approx 2.2$: DVD tracks are twice as dense,
and its pits are correspondingly smaller (read with a
shorter-wavelength laser) --- several times the data in the same
\qty{12}{cm} disc.

\textbf{20.} Hair \qty{70 \pm 3}{\micro m}; friend's hair
\qty{50}{\micro m}; green laser \qty{533}{nm}; CD tracks
\qty{1.6}{\micro m}; DVD tracks \qty{0.74}{\micro m} --- a known
wavelength turns every pattern's geometry into a length, so light of
half a micrometre is a ruler graduated at half a micrometre.
\end{solution}
