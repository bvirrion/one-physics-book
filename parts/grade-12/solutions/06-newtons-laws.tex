\section*{Chapter \ref{ch:g12:newtons-laws} --- Newton's Laws}

\begin{solution}{exo:g12:newtons-laws:1}
(a) $p = 70 \times 10 = \qty{7.0e2}{kg.m/s}$.
(b) $v = \qty{13.9}{m/s}$: $p \approx \qty{1.8e4}{kg.m/s}$.
(c) $p = 0.0080 \times 800 = \qty{6.4}{kg.m/s}$. Car $>$ sprinter $>$
bullet: five orders of magnitude of mass are more than speed can buy.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:2}
$N = m(g + a_z)$: (a) $70 \times 11.01 = \qty{771}{N}$;
(b) $a_z = 0$, so $N = mg = \qty{687}{N}$ --- the first law: constant
velocity, \omterm{def:g10:inertia:force}{forces} \omterm{def:g10:inertia:compensate}{compensate}; (c) $70 \times 8.61 = \qty{603}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:3}
$a = g\sin 30^\circ = \qty{4.9}{m/s^2}$;
$v = \sqrt{2 \times 4.905 \times 3.0} = \qty{5.4}{m/s}$. The mass
cancels between $ma$ and $mg\sin\alpha$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:4}
Reversal: $\Delta p = 0.058 \times (25 + 15) = \qty{2.3}{kg.m/s}$;
$F = 2.32/0.0050 = \qty{4.6e2}{N}$ --- about $800$ times the
\qty{0.57}{N} \omterm{def:g10:universal-gravitation:weight}{weight}, which is why gravity is ignored during contact.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:5}
\omterm{def:g10:universal-gravitation:weight}{Weight} (Earth on book, down): partner = book pulls the Earth, up.
\omterm{def:g12:newtons-laws:friction}{Normal force} (table on book, up): partner = book presses the table,
down. \omterm{def:g10:universal-gravitation:weight}{Weight} and normal act on the \emph{same} body and belong to two
different interactions --- equal here only because $\vect a = \vect 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:6}
$N = mg = \qty{490}{N}$. (a) $F > \mu_s N = \qty{221}{N}$.
(b) $a = (220.7 - 0.35 \times 490.5)/50 = \qty{0.98}{m/s^2}$.
(c) $F = \mu_k N = \qty{172}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:7}
(a) Slides when $\tan\alpha > \mu_s$: $\alpha = \arctan 0.45 \approx
24^\circ$. (b) $a = g(\sin 30^\circ - 0.35\cos 30^\circ) =
9.81 \times 0.197 = \qty{1.9}{m/s^2}$. (c) Yes: once sliding,
\omterm{def:g10:inertia:inventory}{friction} drops to $\mu_k < \mu_s$, so
$a = g\cos\alpha\,(\tan\alpha - \mu_k) \approx \qty{0.9}{m/s^2} > 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:8}
$a = m_2 g/(m_1 + m_2) = 19.62/5.0 = \qty{3.9}{m/s^2}$;
$T = m_1 a = \qty{11.8}{N}$. The hanging block accelerates downward,
so the \omterm{def:g11:forces-and-motion:netforce}{net force} on it points down: $T$ must undercut $m_2 g$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:9}
$v_{\max} = \sqrt{\mu_s g R}$: dry
$\sqrt{0.70 \times 9.81 \times 50} = \qty{18.5}{m/s} \approx
\qty{67}{km/h}$; wet $\qty{14.0}{m/s} = \qty{50}{km/h}$. Both
$mv^2/R$ and $\mu_s mg$ carry $m$: it cancels.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:10}
Vertical: $T\cos\theta = mg$, so
$T = 2.943/\cos 25^\circ = \qty{3.2}{N}$. Radius $R = L\sin\theta =
\qty{0.507}{m}$; radial: $T\sin\theta = mv^2/R$, so
$v = \sqrt{gR\tan\theta} = \sqrt{9.81 \times 0.507 \times 0.466}
= \qty{1.5}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:11}
$a_z = (N - mg)/m$ with $mg = \qty{49.05}{N}$: $+\qty{1.0}{m/s^2}$,
$0$, $-\qty{1.0}{m/s^2}$. E.g.\ an upward trip: speed up, cruise,
brake. Yes: entering already descending, the same readings mean brake,
cruise, speed up again downward --- the scale reads acceleration, not
velocity.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:12}
(a) $0 = 60V + 4.0 \times 8.0$: $V = \qty{-0.53}{m/s}$.
(b) Ball at $8.0 + V$ over the ground: $0 = 60V + 4.0(8.0 + V)$, so
$V = -32/64 = \qty{-0.50}{m/s}$ and the ball flies at \qty{7.5}{m/s}.
(c) The ball's push on her hands --- the third-law partner of her
throw.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:13}
Frictionless: $N\cos\theta = mg$ (vertical) and $N\sin\theta =
mv^2/R$ (radial); dividing, $\tan\theta = v^2/(gR) = 400/784.8$:
$\theta \approx 27^\circ$. Slower: the car tends to slip down the
banking, \omterm{def:g10:inertia:inventory}{friction} must point up-slope; faster: the reverse.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:14}
(a) $250 \times 2500 = \qty{6.3e5}{kg.m/s}$ each second: \omterm{def:g10:inertia:inventory}{thrust}
$\qty{6.3e5}{N}$. (b) $P = \qty{4.9e5}{N}$;
$a = (6.25 - 4.905) \times 10^5 / \num{5.0e4} = \qty{2.7}{m/s^2}$.
(c) It pushes on its own exhaust (third law), not on the air.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:15}
(a) $m_2 g > \mu_s m_1 g$: $m_2 > 0.35 \times 4.0 = \qty{1.4}{kg}$.
(b) $a = (m_2 - \mu_k m_1)\,g/(m_1 + m_2) = 1.0 \times 9.81/6.0 =
\qty{1.6}{m/s^2}$; $T = m_2(g - a) = \qty{16.4}{N}$.
(c) $\mu_k = 0$ gives $a = m_2 g/(m_1 + m_2)$: the formula of
\cref{exo:g12:newtons-laws:8}.
\end{solution}

\begin{solution}{pb:g12:newtons-laws:1}
\textbf{1.} System: cabin $+$ carriage ($+$ passengers). The mountain
is fixed to the ground, inertial to excellent accuracy. \omterm{def:g10:inertia:force}{Forces}:
\omterm{def:g10:universal-gravitation:weight}{weight} $M\vect g$ (down), \omterm{def:g12:newtons-laws:friction}{normal force} $\vect N$ of the track cable
(perpendicular to it), haul \omterm{def:g10:inertia:inventory}{tension} $\vect T$ (up the slope).

\textbf{2.} At rest, the first law: $\sum \vect F = \vect 0$. Along
the slope: $T = Mg\sin\alpha$; perpendicular: $N = Mg\cos\alpha$.

\textbf{3.} $T = 4000 \times 9.81 \times 0.500 = \qty{19.6}{kN}$;
$N = 4000 \times 9.81 \times 0.866 = \qty{34.0}{kN}$.

\textbf{4.} Empty: $T = \qty{9.8}{kN}$. Every \omterm{def:g10:inertia:force}{force} in the balance is
proportional to $M$, so $T$ is too.

\textbf{5.} Along the slope, $Ma = Mg\sin\alpha$: $M$ cancels,
$a = g\sin\alpha = \qty{4.9}{m/s^2}$ down the slope.

\textbf{6.} $T' = M(g\sin\alpha + a) = 4000 \times 5.505 =
\qty{22.0}{kN}$.

\textbf{7.} $a/(g\sin\alpha) = 0.60/4.905 \approx 12\%$.

\textbf{8.} $t = v/a = \qty{10}{s}$; $d = v^2/(2a) = \qty{30}{m}$.

\textbf{9.} Vertical component of $\vect a$: $a\sin\alpha =
\qty{0.30}{m/s^2}$, so $N = m(g + a\sin\alpha) = 80 \times 10.11 =
\qty{809}{N}$ against $mg = \qty{785}{N}$: $3\%$ heavier.

\textbf{10.} The floor's static \omterm{def:g10:inertia:inventory}{friction}, horizontal:
$ma\cos\alpha = 80 \times 0.52 \approx \qty{42}{N}$. Leaning puts the
combined floor push through her \omterm{def:g11:forces-and-motion:com}{center of mass}, so it does not tip
her.

\textbf{11.} $a_T = dv/dt = 0$ (constant speed);
$a_N = v^2/R = 36/40 = \qty{0.90}{m/s^2}$, aimed at the center ---
straight down at the crest.

\textbf{12.} Normal axis (down positive): $Mg - N' = Mv^2/R$, so
$N' = M(g - v^2/R) = 4000 \times 8.91 = \qty{35.6}{kN}$.

\textbf{13.} $N = m(g - v^2/R) = 80 \times 8.91 = \qty{713}{N}$:
lighter by $v^2/(gR) = 0.90/9.81 \approx 9\%$.

\textbf{14.} $N' = 0$ at $v = \sqrt{gR} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$ --- more than triple the cruise: the carriage rides
the cable with a wide margin.

\textbf{15.} \omterm{thm:g12:newtons-laws:third}{Newton's third law}: \qty{35.6}{kN}, pressing down on the
cable (hence on the tower) --- slightly less than the resting \omterm{def:g10:universal-gravitation:weight}{weight},
relieved by the curved flight.

\textbf{16.} $F = \Delta p/\Delta t = 4000 \times 6.0/1.5 =
\qty{16}{kN}$, opposite the motion ($a = \qty{4.0}{m/s^2}$).

\textbf{17.} $f = ma = 80 \times 4.0 = \qty{320}{N}$; the ceiling is
$\mu_s mg = 0.40 \times 785 = \qty{314}{N} < \qty{320}{N}$: feet
alone slip --- hold the handrail.

\textbf{18.} $0 = M_0 V + m(v_{\text{rel}} + V)$:
$V = -80 \times 2.5/2080 = \qty{-0.096}{m/s}$; the passenger moves at
$2.5 - 0.096 \approx \qty{2.4}{m/s}$ over the ground.

\textbf{19.} The rocket: its \omterm{def:g10:inertia:inventory}{thrust} is fixed by the mass ejected per
second times the ejection speed relative to it.

\textbf{20.} The largest \omterm{def:g10:inertia:inventory}{tension} met is the start-up
$T' = M \times \qty{5.505}{N/kg}$ (rest gives $Mg\sin\alpha$, the
crest even less). Cap: $T' \leq \qty{20}{kN}$, so
$M \leq 20000/5.505 = \qty{3.63e3}{kg}$: payload
\qty{1.63e3}{kg}, i.e.\ $20$ passengers of \qty{80}{kg} (not $25$).
Placard: \emph{20 persons}; then $M = \qty{3.6e3}{kg}$,
$T' = \qty{19.8}{kN}$, safety factor $100/19.8 \approx 5.1 \geq 5$.
\end{solution}
