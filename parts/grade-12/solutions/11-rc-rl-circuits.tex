\section*{Chapter \ref{ch:g12:rc-rl-circuits} --- RC and RL Circuits}

\begin{solution}{exo:g12:rc-rl-circuits:1}
$q = Cu = 4.7\times10^{-4} \times 12 \approx \qty{5.6}{mC}$;
$u = q/C = 2.0\times10^{-3}/10^{-4} = \qty{20}{V}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:2}
\qty{47}{pF}: radio tuner; \qty{2200}{\micro F}: smoother; \qty{10}{F}:
supercapacitor. $E = \tfrac12 \times 10 \times 2.7^2 \approx \qty{36}{J}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:3}
$i = C\,\Delta u/\Delta t = 2.2\times10^{-4} \times 5.0/0.010 =
\qty{0.11}{A}$. Then $i = C\,du/dt = 0$: a charged \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} is an open
circuit --- no steady current.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:4}
$RC = 10^4 \times 10^{-4} = \qty{1.0}{s}$;
$L/R = 0.10/50 = \qty{2.0}{ms}$. $\unit{\ohm}\,\unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$;
$\unit{H}/\unit{\ohm} = (\unit{V.s/A})/(\unit{V/A}) = \unit{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:5}
$3.8/6.0 = 0.63$, so $\tau = \qty{2.0}{s}$; $95\%$ at $3\tau =
\qty{6.0}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:6}
$\tau = 4700 \times 2.2\times10^{-4} \approx \qty{1.0}{s}$;
$u(\tau) = 0.63 \times 9.0 = \qty{5.7}{V}$;
$u(3\tau) = 0.95 \times 9.0 = \qty{8.6}{V}$;
$I_0 = \mathcal{E}/R = 9.0/4700 \approx \qty{1.9}{mA}$ (empty \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}:
all of $\mathcal{E}$ sits across $R$).
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:7}
$du/dt = (\mathcal{E}/RC)e^{-t/RC}$, so $RC\,du/dt + u =
\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$.
$u(0) = 0$: the \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} starts empty; $u \to \mathcal{E}$: fully
charged, the current has stopped.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:8}
$\tau = 4.7\times10^{4} \times 10^{-4} = \qty{4.7}{s}$;
$u(\tau) = 12\,e^{-1} \approx \qty{4.4}{V}$; \qty{0.60}{V} is $5\%$,
reached at $3\tau \approx \qty{14}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:9}
$u = L\,\Delta i/\Delta t = 0.50 \times 2.0/0.040 = \qty{25}{V}$.
Closing, $i$ climbs at the circuit's own pace, $di/dt$ modest; opening
\emph{\omterm{def:g10:inertia:force}{forces}} $i$ to zero almost instantly, $di/dt$ is huge and $u$
reaches sparking values.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:10}
$di/dt = (\mathcal{E}/L)e^{-Rt/L}$, so $L\,di/dt + Ri =
\mathcal{E}e^{-Rt/L} + \mathcal{E}(1 - e^{-Rt/L}) = \mathcal{E}$.
$i(\tau) = 0.63 \times 2.0 = \qty{1.3}{A}$;
$i(3\tau) = 0.95 \times 2.0 = \qty{1.9}{A}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:11}
$u'(t) = (\mathcal{E}/\tau)e^{-t/\tau}$, so $u'(0) = \mathcal{E}/\tau$:
the tangent $u = (\mathcal{E}/\tau)\,t$ reaches $\mathcal{E}$ at
$t = \tau$. $C = \tau/R = 0.50/2000 = \qty{250}{\micro F}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:12}
Supercapacitor $\tfrac12 \times 3000 \times 2.7^2 \approx \qty{1.1e4}{J}$;
flash $\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$;
\omterm{def:g11:magnetic-fields:solenoid}{coil} $\tfrac12 \times 0.010 \times 10^2 = \qty{0.50}{J}$. Battery
($\qty{1.3e4}{J}$) $>$ supercapacitor $\gg$ flash $\gg$ \omterm{def:g11:magnetic-fields:solenoid}{coil} --- but all
three can dump their \omterm{def:g10:energy-conservation:energy}{energy} in milliseconds: \omterm{def:g11:work-of-force:power}{power}, not \omterm{def:g10:energy-conservation:energy}{energy}.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:13}
$\tau = \qty{5.0}{s}$, so $R = \tau/C = 5.0/10^{-4} = \qty{50}{k\ohm}$.
$1 - e^{-t/\tau} = 0.50$ gives $e^{-t/\tau} = 0.50$, $t = \tau\ln 2
\approx 0.69\,\tau = \qty{3.5}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:14}
$\Delta u = i\,\Delta t/C = 0.50 \times 0.010/2.2\times10^{-3} \approx
\qty{2.3}{V}$. Need $C \geq i\,\Delta t/\Delta u =
5.0\times10^{-3}/0.50 = \qty{10}{mF}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:15}
$u \approx L\,\Delta i/\Delta t = 0.80 \times 2.0/10^{-3} =
\qty{1.6}{kV}$; $E_L = \tfrac12 \times 0.80 \times 2.0^2 =
\qty{1.6}{J}$. Cut slowly, $di/dt$ stays small, $u$ stays a few \omterm{def:g11:circuits-and-power:voltage}{volts}:
no spark.
\end{solution}

\begin{solution}{pb:g12:rc-rl-circuits:1}
\textbf{1.} $q = CU_0 = 8.0\times10^{-4} \times 300 = \qty{0.24}{C}$.

\textbf{2.} $N = 0.24/1.60\times10^{-19} = \num{1.5e18}$.

\textbf{3.} $E_C = \tfrac12 \times 8.0\times10^{-4} \times 300^2 =
\qty{36}{J}$.

\textbf{4.} $\num{1.3e4}/36 \approx 360$ times.

\textbf{5.} The cell stores plenty but delivers slowly; the \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}
turns modest \omterm{def:g10:energy-conservation:energy}{energy} into enormous \emph{\omterm{def:g11:work-of-force:power}{power}}.

\textbf{6.} $\mathcal{E} = Ri + u$ with $i = C\,du/dt$:
$RC\,du/dt + u = \mathcal{E}$.

\textbf{7.} $du/dt = (\mathcal{E}/RC)e^{-t/RC}$:
$\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$
for all $t$, and $u(0) = 0$.

\textbf{8.} $\tau = RC = 10^3 \times 8.0\times10^{-4} = \qty{0.80}{s}$.

\textbf{9.} At $3\tau = \qty{2.4}{s}$, $u = 0.95 \times 300 =
\qty{285}{V}$ --- the two-second whine.

\textbf{10.} $I_0 = \mathcal{E}/R = \qty{0.30}{A}$; as $u$ climbs, the
remainder $\mathcal{E} - u$ across $R$ shrinks, so $i = (\mathcal{E}-u)/R$
dies away.

\textbf{11.} $u(t) = U_0\,e^{-t/\tau'}$ with $\tau' = R_f C =
0.50 \times 8.0\times10^{-4} = \qty{0.40}{ms}$.

\textbf{12.} $I_0 = U_0/R_f = 300/0.50 = \qty{600}{A}$.

\textbf{13.} $P_0 = U_0 I_0 = 300 \times 600 = \qty{180}{kW}$ --- about
$80$ kettles at once.

\textbf{14.} Duration $3\tau' = \qty{1.2}{ms}$; average \omterm{def:g11:work-of-force:power}{power}
$\approx 36/1.2\times10^{-3} = \qty{30}{kW}$.

\textbf{15.} $2.4/1.2\times10^{-3} = 2000$. Only the \omterm{def:g11:circuits-and-power:resistance}{resistance} changed
(\qty{1.0}{k\ohm} in, \qty{0.50}{\ohm} out): a \omterm{def:g11:work-of-force:power}{power} lever.

\textbf{16.} $C = 2E_C/U_0^2 = 400/1500^2 \approx \qty{1.8e-4}{F} =
\qty{180}{\micro F}$.

\textbf{17.} $q = CU_0 = 1.78\times10^{-4} \times 1500 \approx
\qty{0.27}{C}$.

\textbf{18.} $\tau = 50 \times 1.78\times10^{-4} \approx \qty{8.9}{ms}$,
duration $3\tau \approx \qty{27}{ms}$ --- some twenty times the flash.

\textbf{19.} $I_0 = 1500/50 = \qty{30}{A}$;
$P_0 = 1500 \times 30 = \qty{45}{kW}$.

\textbf{20.} $3\tau_c = \qty{6.0}{s}$ gives $\tau_c = \qty{2.0}{s}$, so
$R_c = \tau_c/C = 2.0/1.78\times10^{-4} \approx \qty{11}{k\ohm}$ and
$I_0 = 1500/1.12\times10^{4} \approx \qty{0.13}{A}$ --- ready in six
seconds, on a tenth of an \omterm{def:g11:circuits-and-power:current}{ampere}.
\end{solution}
