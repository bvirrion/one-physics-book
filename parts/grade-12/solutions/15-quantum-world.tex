\begin{solution}{pb:g12:quantum-world:1}
\textbf{1.} $E = hc/\lambda = \num{1.99e-25}/\num{5.50e-7} =
\qty{3.62e-19}{J} = \qty{2.26}{eV}$.

\textbf{2.} $N = \num{1.0e3}/\num{3.62e-19} \approx \num{2.8e21}$
\omterm{def:g12:quantum-world:photon}{photons} per second.

\textbf{3.} $P = 0.20 \times \qty{1.0e3}{W} = \qty{200}{W}$;
$E = 200 \times 5.0 = \qty{1.0}{kW.h}$ ($\qty{3.6e6}{J}$).

\textbf{4.} With $10^{21}$ grains per second, the statistical
flicker is fantastically small: graininess averages into a perfectly
smooth current, as sand poured fast pours like water.

\textbf{5.} Damage needs one grain of \qtyrange{3}{4}{eV}: \omterm{def:g10:light-spectra:wavelength}{ultraviolet}
grains qualify, \omterm{def:g10:light-spectra:wavelength}{infrared} grains ($\sim \qty{0.1}{eV}$) never do ---
numbers cannot substitute for punch per grain.

\textbf{6.} $U_s$ is the reverse \omterm{def:g11:circuits-and-power:voltage}{voltage} that just stops the fastest
electrons: $eU_s = E_{k,\max} = h\nu - W_0$.

\textbf{7.} $\nu = c/\lambda$: $8.22$, $7.41$, $6.88$ and $5.49$
$\times\,10^{14}\,\unit{Hz}$.

\textbf{8.} $eU_s = h\nu - W_0$ is affine in $\nu$ with slope $h$.
Successive slopes: $(0.94 - 0.37)/1.39 = 0.41$ and
$(1.50 - 0.94)/1.34 = 0.42$ \unit{eV} per $10^{14}\,\unit{Hz}$ ---
the three points align.

\textbf{9.} $h = e(U_{s,1}-U_{s,4})/(\nu_1-\nu_4) = \num{1.60e-19}
\times 1.13/\num{2.72e14} = \qty{6.64e-34}{J.s}$.

\textbf{10.} At \qty{546}{nm}: $W_0 = h\nu - eU_s = \num{3.64e-19} -
\num{0.59e-19} = \qty{3.05e-19}{J} = \qty{1.90}{eV}$;
$\nu_0 = W_0/h = \qty{4.60e14}{Hz}$; $\lambda_0 \approx \qty{653}{nm}$.

\textbf{11.} Only $\lambda < \qty{653}{nm}$ ejects: the violet-to-red
visible and the \omterm{def:g10:light-spectra:wavelength}{ultraviolet}. The whole \omterm{def:g10:light-spectra:wavelength}{infrared} side --- roughly half
the Sun's \omterm{def:g10:energy-conservation:energy}{energy} --- is useless to this cathode.

\textbf{12.} $(6.64 - 6.63)/6.63 \approx 0.2\%$ --- a two-permille
physics-lab measurement of a universal constant.

\textbf{13.} $E_2 = \qty{-3.40}{eV}$, $E_3 = \qty{-1.51}{eV}$,
$E_4 = \qty{-0.85}{eV}$, $E_5 = \qty{-0.54}{eV}$.

\textbf{14.} $3 \to 2$: \qty{1.89}{eV}; $4 \to 2$: \qty{2.55}{eV};
$5 \to 2$: \qty{2.86}{eV}.

\textbf{15.} $\lambda = 1240/E$: \qty{656}{nm} (red), \qty{486}{nm}
(blue-green), \qty{434}{nm} (violet).

\textbf{16.} The \omterm{def:g11:fundamental-interactions:ladder}{atom}'s energies are \omterm{def:g12:quantum-world:levels}{quantized}, so it can only emit
the discrete differences $E_p - E_n$: a ladder makes lines, not a
rainbow.

\textbf{17.} No: \qty{500}{nm} means \qty{2.48}{eV}, which matches no
level difference ($E_2$ upward gives $1.89$, $2.55$, $2.86$,
\qty{3.40}{eV}; from $E_1$ everything costs at least \qty{10.2}{eV})
--- the gas is transparent there.

\textbf{18.} $eU = \tfrac12 m_e v^2$: $v = \sqrt{2 \times
\num{1.60e-19} \times \num{1.0e4}/\num{9.11e-31}} \approx
\qty{5.9e7}{m/s} \approx 0.20\,c$ --- classical treatment acceptable.

\textbf{19.} $p = m_e v = \qty{5.4e-23}{kg.m/s}$;
$\lambda = h/p = \qty{1.2e-11}{m} = \qty{12}{pm}$.

\textbf{20.} \omterm{def:g10:signals-and-waves:oscilloscope}{Gain} $= \num{5.50e-7}/\num{1.23e-11} \approx \num{4.5e4}$.
The weekend's two numbers: $h = \qty{6.64e-34}{J.s}$, measured to
$0.2\%$ with \omterm{def:g11:color-light-sources:subtractive}{filters} and a voltmeter, and a microscope
forty-five-thousand times sharper because electrons \omterm{def:g10:signals-and-waves:wave}{wave}.
\end{solution}
