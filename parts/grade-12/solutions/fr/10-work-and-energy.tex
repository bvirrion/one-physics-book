\section*{Chapitre \ref{ch:g12:work-and-energy} --- Travail et énergie mécanique}

\begin{solution}{exo:g12:work-and-energy:1}
$E_p = \tfrac12 \times 200 \times 0.050^2 = \qty{0.25}{J}$. Allongement
doublé~: $\times 4$, soit \qty{1.0}{J}. Le second \qty{5.0}{cm}
seul coûte $1.0 - 0.25 = \qty{0.75}{J}$ --- la \omterm{def:g10:inertia:force}{force} y est plus
grande.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:2}
$W = mg(z_A - z_B) = 0.90 \times 9.81 \times (12 - 30) \approx
\qty{-1.6e2}{J}$, quel que soit l'itinéraire --- le même sur la
montée droite. Aller-retour~: \qty{0}{J} (\omterm{def:g12:work-and-energy:conservative}{force
conservative}).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:3}
(a), (b) \omterm{def:g12:work-and-energy:conservative}{conservatives}~: \omterm{def:g11:work-of-force:work}{travail} fixé
par les extrémités (dénivelé~; allongement). (c), (e)
\omterm{def:g12:work-and-energy:conservative}{non conservatives}~: $-fL$ et la traînée facturent par
\omterm{def:g10:orders-of-magnitude:unit}{mètre} parcouru. (d) sans travail~: toujours
perpendiculaire au mouvement.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:4}
$h = L(1 - \cos\qty{60}{\degree}) = \qty{1.0}{m}$~;
$v = \sqrt{2 \times 9.81 \times 1.0} \approx \qty{4.4}{m/s}$. La
\omterm{def:g10:inertia:inventory}{tension} est perpendiculaire au mouvement à chaque
instant~: \omterm{def:g11:work-of-force:power}{puissance} nulle, \omterm{def:g11:work-of-force:work}{travail} nul.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:5}
$F = P/v = 250/9.0 \approx \qty{28}{N}$. À \qty{5.0}{m/s}~:
$F = 250/5.0 = \qty{50}{N}$ (surtout la composante de gravité de la
pente).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:6}
$E_p = \tfrac12 \times 450 \times 0.080^2 = \qty{1.44}{J}$~;
$v = \sqrt{2 \times 1.44/0.020} = \qty{12}{m/s}$~;
$h = E_p/mg = 1.44/(0.020 \times 9.81) \approx \qty{7.3}{m}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:7}
Trapèzes (pas de \qty{2.0}{cm})~: $(0.03 + 0.10 + 0.19 + 0.30) =
\qty{0.62}{J}$. Modèle linéaire~: $k = 18.0/0.080 =
\qty{225}{N/m}$, $\tfrac12 kx^2 = \qty{0.72}{J}$. La courbe mesurée
s'affaisse sous la droite aux petits $x$, donc la vraie aire est
plus petite.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:8}
$\Delta E_m = \tfrac12 \times 55 \times 9.0^2 - 55 \times 9.81 \times
8.0 = 2228 - 4316 \approx \qty{-2.1e3}{J}$~;
$f = 2089/25 \approx \qty{84}{N}$. \omterm{def:g10:universal-gravitation:weight}{Poids}
\omterm{def:g12:work-and-energy:conservative}{conservatif}, \omterm{def:g12:newtons-laws:friction}{force normale} sans
travail~: seuls le dénivelé et la \emph{longueur} du chemin entrent,
pas la forme.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:9}
$H_{\min} = \tfrac52 R = \qty{0.50}{m}$. Depuis \qty{60}{cm}~:
$v_{\text{top}} = \sqrt{2 \times 9.81 \times (0.60 - 0.40)} \approx
\qty{2.0}{m/s}$, et $v_{\text{top}}^2/(gR) = 2.0$ --- le double du
minimum. Le \omterm{def:g10:inertia:inventory}{frottement} écréme l'\omterm{def:g10:energy-conservation:energy}{énergie}
tout du long, donc les vraies pistes ont besoin de hauteur en plus.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:10}
\omterm{def:g12:work-and-energy:diagram}{Points de rebroussement}~: $8.0\,x^2 = 2.0$,
$x = \pm\qty{0.50}{m}$. Vitesse maximale en $x = 0$~:
$v = \sqrt{2 \times 2.0/0.100} \approx \qty{6.3}{m/s}$. Une
oscillation d'aller-retour dans un puits parabolique --- le $E_p$ de
l'\omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} masse--ressort est exactement de
cette forme.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:11}
$\tfrac12 kx^2 = \tfrac12 mv^2$~:
$x = v\sqrt{m/k} = 2.0\sqrt{1200/\num{6.0e5}} \approx \qty{8.9}{cm}$.
À \qty{4.0}{m/s}~: \qty{18}{cm} --- le double, pas le quadruple~:
l'\omterm{def:g10:energy-conservation:energy}{énergie} quadruple, mais $E_p$ croît comme $x^2$,
donc $x$ ne fait que doubler.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:12}
(a) $E_m = \qty{5.0}{J} < \qty{6.0}{J}$~: il ne peut pas passer~; il
s'arrête sur le flanc où $E_p = \qty{5.0}{J}$ et rebrousse. (b)~Sommet~:
$E_k = \qty{1.0}{J}$, $v = \sqrt{2 \times 1.0/0.50} = \qty{2.0}{m/s}$~;
vallée~: $E_k = \qty{5.0}{J}$, $v \approx \qty{4.5}{m/s}$. (c)~$x = 0$~:
\omterm{prop:g12:work-and-energy:equilibria}{instable} (maximum)~; $x = \qty{1.5}{m}$~:
\omterm{prop:g12:work-and-energy:equilibria}{stable} (minimum).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:13}
$\tfrac12 mv^2 = mgR_E$~:
$v = \sqrt{2 \times 9.81 \times \num{6.37e6}} \approx \qty{1.12e4}{m/s}
= \qty{11.2}{km/s}$. Chaque terme est proportionnel à $m$~: sonde et
caillou pareils. Juste en dessous, le lancement grimpe encore
énormément loin, rencontre un \omterm{def:g12:work-and-energy:diagram}{point de rebroussement},
et retombe~: lié.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:14}
$E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.060^2 =
\qty{7.2e-2}{J}$~; $v_{\max} = \sqrt{2E_m/m} =
\sqrt{2 \times 0.072/0.250} \approx \qty{0.76}{m/s}$~;
$E_k = E_p$ là où $\tfrac12 kx^2 = E_m/2$~: $x = \pm A/\sqrt 2 \approx
\pm\qty{4.2}{cm}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:15}
$h = v^2/2g = 9.5^2/19.62 \approx \qty{4.6}{m}$~; barre $\approx
4.6 + 1.0 = \qty{5.6}{m}$. Le record est plus haut~: les sauteurs
ajoutent du \omterm{def:g11:work-of-force:work}{travail} avec bras et jambes sur la perche
qui fléchit. Mais l'estimation d'\omterm{def:g10:energy-conservation:energy}{énergie} fixe
l'échelle --- aucun saut de sprinter n'atteindra \qty{10}{m}.
\end{solution}

\begin{solution}{pb:g12:work-and-energy:1}
\textbf{1.} $v = \sqrt{2gH} = \sqrt{2 \times 9.81 \times 45} \approx
\qty{29.7}{m/s}$~: \omterm{def:g10:universal-gravitation:weight}{poids}
\omterm{def:g12:work-and-energy:conservative}{conservatif}, \omterm{def:g12:newtons-laws:friction}{force normale} sans
travail --- le profil s'efface.

\textbf{2.} Porte~: $E_m = mgH = 70 \times 9.81 \times 45 \approx
\qty{3.09e4}{J}$. Décollage~: $E_m = \tfrac12 \times 70 \times 26^2
\approx \qty{2.37e4}{J}$.

\textbf{3.} $\Delta E_m \approx \qty{-7.2e3}{J}$~; le théorème de
l'\omterm{def:g10:energy-conservation:energy}{énergie mécanique} le facture aux
\omterm{def:g12:work-and-energy:conservative}{forces non conservatives}~: \omterm{def:g10:inertia:inventory}{frottement}
de neige et traînée de l'air.

\textbf{4.} $f = \num{7.2e3}/90 \approx \qty{80}{N}$.

\textbf{5.} $\num{7.2e3}/\num{3.09e4} \approx 23\%$. La cire abaisse
$f$, se baisser abaisse la traînée~: les deux réduisent $fL$, le
seul terme négociable.

\textbf{6.} \omterm{def:g12:projectile-motion:freefall}{Chute libre} à vitesse initiale
horizontale --- un mouvement de \omterm{def:g12:projectile-motion:projectile}{projectile}.
$v_x = \qty{26}{m/s}$~; $v_y = \sqrt{2 \times 9.81 \times 40} \approx
\qty{28.0}{m/s}$.

\textbf{7.} $v = \sqrt{26^2 + 28.0^2} = \sqrt{1461} \approx
\qty{38.2}{m/s}$~; \omterm{def:g10:energy-conservation:energy}{énergie}~:
$v = \sqrt{v_0^2 + 2gh} = \sqrt{676 + 785}$ --- identique. Environ
\qty{138}{km/h}.

\textbf{8.} $t = v_y/g = 28.0/9.81 \approx \qty{2.9}{s}$~;
$d = v_0 t \approx \qty{74}{m}$.

\textbf{9.} Non. La portance est perpendiculaire à la vitesse (sans
travail) et la traînée s'y oppose (\omterm{def:g11:work-of-force:work}{travail} négatif)~:
$\Delta E_m \leq 0$, donc la vraie vitesse d'atterrissage ne peut
être que \emph{plus petite} --- l'air allonge le vol, il ne
l'alimente pas.

\textbf{10.} $\alpha = \arctan(28.0/26) \approx \qty{47}{\degree}$
sous l'horizontale.

\textbf{11.} Angle à la pente~: $47 - 35 = \qty{12}{\degree}$.
$v_\parallel = 38.2\cos\qty{12}{\degree} \approx \qty{37.4}{m/s}$~;
$v_\perp = 38.2\sin\qty{12}{\degree} \approx \qty{7.9}{m/s}$.

\textbf{12.} Absorbée~: $\tfrac12 \times 70 \times 7.9^2 \approx
\qty{2.2e3}{J}$, contre $\tfrac12 \times 70 \times 38.2^2 \approx
\qty{5.1e4}{J}$ au total~: environ $4\%$.

\textbf{13.} $h_{\text{eq}} = 7.9^2/19.62 \approx \qty{3.2}{m}$ ---
un saut hardi depuis un mur. Plat~: $38.2^2/19.62 \approx
\qty{74}{m}$ --- insurmontable.

\textbf{14.} Une pente parallèle à la trajectoire de vol garde
$v_\perp$ petit, donc les jambes absorbent des
\omterm{def:g10:orders-of-magnitude:unit}{mètres}, pas toute la montagne d'\omterm{def:g11:mechanical-energy:kinetic}{énergie
cinétique}.

\textbf{15.} $H'_0 = v_0'^2/2g = 576/19.62 \approx \qty{29.4}{m}$.

\textbf{16.} $mgH' - f\,\dfrac{H'}{\sin\qty{35}{\degree}} =
\tfrac12 m v_0'^2$.

\textbf{17.} $H' = \dfrac{\tfrac12 \times 70 \times 576}
{686.7 - 80/0.574} = \dfrac{\num{20160}}{547} \approx \qty{37}{m}$.

\textbf{18.} $36.8/29.4 \approx 1.25$~: une majoration de $25\%$.

\textbf{19.} Pour \qty{60}{kg}~: $H' = \num{17280}/(588.6 - 139.5)
\approx \qty{38.5}{m}$. Le sauteur \emph{plus léger} a besoin de la
tour la plus haute~: la facture de \omterm{def:g10:inertia:inventory}{frottement} $fL$ est la
même, mais son budget d'\omterm{def:g10:energy-conservation:energy}{énergie} $mgH'$ est plus
petit.

\textbf{20.} Annoncer une tour de \qty{37}{m} pour \qty{24}{m/s} de
décollage --- et confesser qu'un quart d'elle, quelque \qty{7}{m} de
béton, est la majoration du \omterm{def:g10:inertia:inventory}{frottement} sur le rêve sans
frottement.
\end{solution}
