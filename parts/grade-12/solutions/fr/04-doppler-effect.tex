\section*{Chapitre \ref{ch:g12:doppler-effect} --- L'effet Doppler}

\begin{solution}{exo:g12:doppler-effect:1}
(a) $f' = 700 \times 340/315 \approx \qty{756}{Hz}$.
(b) $f' = 700 \times 340/365 \approx \qty{652}{Hz}$.
(c) \qty{700}{Hz}~: conducteur et sirène avancent ensemble, leur distance ne change jamais.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:2}
L'\emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{intensité sonore}} s'estompe parce que l'\omterm{def:g10:signals-and-waves:wave}{onde} s'étale --- elle dépend de la distance.
La \emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur}} chute de l'approche à l'éloignement --- l'\omterm{def:g12:doppler-effect:doppler}{effet Doppler}, qui dépend
de la vitesse à laquelle la distance change. Deux effets indépendants.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:3}
Observateur mobile~: $f' = 700\,(1 + 6.0/340) \approx \qty{712}{Hz}$~; en s'éloignant,
$700\,(1 - 6.0/340) \approx \qty{688}{Hz}$. La sirène est au repos dans l'air, donc le motif
d'\omterm{def:g10:signals-and-waves:wave}{onde} est intact~; seul l'observateur le traverse.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:4}
Devant~: $\lambda' = 310/400 = \qty{0.775}{m}$~; derrière~: $370/400 = \qty{0.925}{m}$~; au
repos~: $340/400 = \qty{0.85}{m}$. La \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} courte atteint qui le train
approche, la longue qui il quitte.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:5}
$\Delta f/f \approx 5.0/340 \approx 1.5\%$~: $\Delta f \approx \qty{6.5}{Hz}$, donc environ
\qty{446}{Hz} --- un quart de demi-ton. Un musicien le remarquerait tout juste.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:6}
Le mur reçoit $f\,v/(v - v_b)$ (source mobile)~; la chauve-souris, observateur mobile se
rapprochant de l'\omterm{rem:g10:signals-and-waves:honest}{écho}, entend $(1 + v_b/v)$ fois cela~: $f' = f(v + v_b)/(v - v_b) = 50.0 \times
346/334 \approx \qty{51.8}{kHz}$, un décalage de $+\qty{1.8}{kHz}$. C'est le double décalage
du pistolet radar avec les rôles émetteur et réflecteur échangés~: seule la vitesse de
rapprochement compte.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:7}
$u = c\,\Delta f/(2f) = \num{3.00e8} \times 4000/(2 \times \num{24.15e9}) \approx
\qty{24.8}{m/s} \approx \qty{89}{km/h}$. À \qty{50}{km/h} ($\qty{13.9}{m/s}$)~:
$\Delta f = 2 \times 13.9 \times \num{24.15e9}/\num{3.00e8} \approx \qty{2.2}{kHz}$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:8}
$u = v\,\Delta f/(2f) = 1540 \times 2600/\num{1.0e7} \approx \qty{0.40}{m/s}$.
Un écoulement perpendiculaire a une \omterm{def:g12:doppler-effect:radial}{vitesse radiale} nulle, donc aucun décalage.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:9}
(a) Source~: $f' = 1000 \times 340/306 \approx \qty{1111}{Hz}$. (b) Observateur~:
$f' = 1000 \times 1.1 = \qty{1100}{Hz}$. Les petits mouvements prédisent \qty{1100}{Hz}~: exact
pour l'observateur, \qty{11}{Hz} en dessous pour la source --- la correction d'ordre $x^2$,
$1\%$ de $f$ pour $x = 0.1$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:10}
$f = 2 \times 465 \times 415/880 \approx \qty{439}{Hz}$~;
$v_s = 340 \times 50/880 \approx \qty{19.3}{m/s} \approx \qty{70}{km/h}$. Le vrai $f$
est la moyenne \omterm{def:g12:sound-acoustics:harmonics}{harmonique}~: l'approche élève la \omterm{def:g10:signals-and-waves:frequency}{fréquence} plus que l'éloignement
ne l'abaisse, donc $f$ se place sous la moyenne arithmétique (\qty{440}{Hz}).
\end{solution}

\begin{solution}{exo:g12:doppler-effect:11}
\omterm{def:g12:mechanical-waves:wavelength}{Longueur d'onde} plus grande~: \omterm{def:g12:doppler-effect:redshift}{décalage vers le rouge}, l'étoile s'éloigne.
$v_r = c\,\Delta\lambda/\lambda =
\num{3.00e8} \times 0.2/656.3 \approx \qty{9.1e4}{m/s} \approx \qty{91}{km/s}$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:12}
$\Delta\lambda = \lambda K/c = 500 \times 55/\num{3.00e8} \approx \qty{9.2e-5}{nm}$~;
$\Delta\lambda/\lambda = 55/\num{3.00e8} \approx \num{1.8e-7}$. Tout le \omterm{def:g10:signals-and-waves:signal}{signal} est deux
parties sur dix millions~: un spectrographe dérivant de plus de $10^{-7}$ l'enfouit.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:13}
$v_r = 0.024\,c \approx \qty{7.2e6}{m/s}$. Vitesses proportionnelles à la distance, les
mêmes dans toutes les directions~: toutes les distances s'étirent uniformément --- un
univers en expansion sans centre privilégié. À $2.4\%$ les corrections sont d'ordre
$(v_r/c)^2 \approx \num{6e-4}$~: encore fiables.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:14}
Les deux bords se décalent en sens opposés, donc la raie moyennée sur le disque est
\emph{élargie}, non décalée~: largeur totale $2\lambda v/c = 2 \times 656.3 \times
\num{2.0e3}/\num{3.00e8} \approx \qty{8.8e-3}{nm}$ (environ $\pm\qty{4.4e-3}{nm}$).
\end{solution}

\begin{solution}{exo:g12:doppler-effect:15}
(a) Devant la voiture les crêtes sont espacées de $(v - v_s)T$~; le camion fuit à $v_o$ et
les rencontre à la vitesse relative $v - v_o$~: $f' = (v - v_o)/\bigl((v - v_s)T\bigr) =
f(v - v_o)/(v - v_s)$. (b) $f' = 700 \times 310/300 \approx \qty{723}{Hz}$.
(c) $v_o = v_s$ donne $f' = f$~: la séparation est constante, et aucun changement de
distance signifie aucun décalage Doppler, par définition.
\end{solution}

\begin{solution}{pb:g12:doppler-effect:1}
\textbf{1.} Chaque crête s'élargit depuis le point où elle a été émise~: les crêtes se
resserrent devant le klaxon mobile et s'étirent derrière, donc le téléphone qui s'approche
les rencontre plus vite ($f_1$) que celui qui s'éloigne ($f_2$).

\textbf{2.} $f_1 = f\,\dfrac{v}{v - u}$, \quad $f_2 = f\,\dfrac{v}{v + u}$.

\textbf{3.} $f_1 - f_2 = \frac{2fv\,u}{v^2 - u^2}$ et $f_1 + f_2 =
\frac{2fv^2}{v^2 - u^2}$~; en divisant, $u = v\,\frac{f_1 - f_2}{f_1 + f_2} =
340 \times 60/882 \approx \qty{23}{m/s} \approx \qty{83}{km/h}$.

\textbf{4.} $\dfrac{2f_1 f_2}{f_1 + f_2} = f$ (le facteur commun s'annule)~:
$f = 2 \times 471 \times 411/882 \approx \qty{439}{Hz}$ --- la moyenne
\omterm{def:g12:sound-acoustics:harmonics}{harmonique}, sous la moyenne \qty{441}{Hz} parce que l'approche
élève $f$ plus que l'éloignement ne le coupe.

\textbf{5.} Devant~: $(v - u)/f = 316.9/439 \approx \qty{0.72}{m}$~; derrière~:
$363.1/439 \approx \qty{0.83}{m}$~; au repos~: $340/439 \approx \qty{0.77}{m}$.

\textbf{6.} Une fois comme \emph{observateur} mobile recevant l'\omterm{def:g10:signals-and-waves:wave}{onde}, une fois comme
\emph{source} mobile réémettant ce qu'elle a reçu.

\textbf{7.} $f' = f\,(1 + u/c)/(1 - u/c) \approx f(1 + 2u/c)$, donc
$\Delta f = f' - f \approx \dfrac{2u}{c}\,f$.

\textbf{8.} $u = c\,\Delta f/(2f) = \num{3.00e8} \times 3550/(2 \times \num{24.125e9})
\approx \qty{22}{m/s}$.

\textbf{9.} $22.1 \times 3.6 \approx \qty{79}{km/h}$~: juste sous la limite de
\qty{80}{km/h}.

\textbf{10.} $\Delta f = 2f/(3.6\,c) \approx \qty{45}{Hz}$ par \unit{km/h}. Battre
l'\omterm{rem:g10:signals-and-waves:honest}{écho} contre l'\omterm{def:g10:signals-and-waves:wave}{onde} émise ne laisse que la
\emph{différence}~: une \omterm{def:g10:signals-and-waves:frequency}{fréquence} audio, trivialement comptée, bien
qu'elle soit une fraction de $10^{-7}$ de la porteuse.

\textbf{11.} Les \omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} de \omterm{def:g10:light-spectra:emission}{raies spectrales}
connues, nuit après nuit~; puis $v_r = c\,\Delta\lambda/\lambda$.

\textbf{12.} $K = \qty{55}{m/s}$, $P = \qty{4.2}{jours}$.

\textbf{13.} $\Delta\lambda = 550 \times 55/\num{3.00e8} \approx \qty{1.0e-4}{nm}$,
$\Delta\lambda/\lambda \approx \num{1.8e-7}$ --- une partie sur cinq millions, environ
$\num{4e5}$ fois plus petit que le décalage de $7\%$ du train.

\textbf{14.} Étoile et planète s'attirent mutuellement de façon égale, donc toutes deux
\omterm{def:g10:universal-gravitation:satellite}{orbitent} autour de leur \omterm{def:g11:forces-and-motion:com}{centre de masse} commun~: l'étoile ne
peut rester immobile. Une $v_r$ sinusoïdale est un \omterm{def:g11:forces-and-motion:ucm}{mouvement circulaire
uniforme} vu de profil --- une \omterm{def:g10:universal-gravitation:satellite}{orbite} circulaire.

\textbf{15.} Seule la composante le long de la ligne de visée change la distance
étoile--Terre, donc la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}. Face à nous,
$v_r = 0$ à tout instant~: aucun \omterm{def:g10:signals-and-waves:signal}{signal} du tout.

\textbf{16.} $P = \qty{3.63e5}{s}$~: $r^3 = \num{6.67e-11} \times \num{2.0e30} \times
(\num{3.63e5})^2/(4\pi^2) \approx \num{4.4e29}$, donc $r \approx \qty{7.6e9}{m}$ ---
environ $1/20$ de la distance Terre--Soleil.

\textbf{17.} $V = 2\pi r/P = 2\pi \times \num{7.6e9}/\num{3.63e5} \approx
\qty{1.3e5}{m/s}$.

\textbf{18.} $m = M K/V = \num{2.0e30} \times 55/\num{1.3e5} \approx \qty{8.3e26}{kg}$.

\textbf{19.} Environ $0.4$ masse de Jupiter ($\approx 140$ Terres), orbitant $8$ fois plus
près que Mercure~: une géante gazeuse qui grille contre son étoile --- un «~Jupiter
chaud~».

\textbf{20.} Une inclinaison cache une partie du mouvement~: le $K$ mesuré n'est que la
part radiale, donc la vraie masse est \emph{plus grande} --- \qty{8.3e26}{kg} est un
minimum. Un décalage \omterm{def:g10:signals-and-waves:period}{périodique} d'une partie sur cinq millions vient de
livrer l'\omterm{def:g10:universal-gravitation:satellite}{orbite}, la vitesse et la masse minimale d'une planète
que personne n'a jamais vue.
\end{solution}
