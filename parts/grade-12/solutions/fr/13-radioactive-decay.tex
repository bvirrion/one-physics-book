\section*{Chapitre \ref{ch:g12:radioactive-decay} --- Décroissance radioactive}

\begin{solution}{exo:g12:radioactive-decay:1}
$T_{1/2} = \ln 2/\lambda = 0.693/\num{1.0e-6} = \qty{6.9e5}{s}
\approx 8.0$~jours --- l'horloge de l'iode-131.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:2}
$T_{1/2} = 3.8 \times \num{86400} = \qty{3.28e5}{s}$~;
$\lambda = 0.693/\num{3.28e5} = \qty{2.1e-6}{s^{-1}}$~;
$\mathcal A = \lambda N = \num{2.1e-6} \times \num{1.0e10} \approx
\qty{2.1e4}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:3}
$\dd N/\dd t = N_0(-\lambda)e^{-\lambda t} = -\lambda N$ ---
l'équation tient~; $N(0) = N_0 e^0 = N_0$~;
$N(T_{1/2})/N_0 = e^{-\lambda \ln 2/\lambda} = e^{-\ln 2} =
\tfrac12$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:4}
$90$~ans $= 3\,T_{1/2}$~: $\mathcal A = \num{4.0e5}/2^3 =
\qty{5.0e4}{Bq}$. $300$~ans $= 10\,T_{1/2}$~:
$\mathcal A = \num{4.0e5}/1024 \approx \qty{3.9e2}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:5}
Pente $= (18.85 - 20.00)/10 = -0.115$, donc
$\lambda = \qty{0.115}{h^{-1}} = 0.115/3600 = \qty{3.2e-5}{s^{-1}}$~;
$T_{1/2} = 0.693/0.115 = 6.0$~heures --- technétium-99m.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:6}
$N = (1.0/226) \times \num{6.02e23} = \num{2.7e21}$~;
$T_{1/2} = 1600 \times \num{3.156e7} = \qty{5.05e10}{s}$, donc
$\lambda = \qty{1.37e-11}{s^{-1}}$~;
$\mathcal A = \lambda N \approx \qty{3.7e10}{Bq}$ --- un curie,
l'\omterm{def:g11:nucleus-radioactivity:activity}{activité} du gramme de radium de Marie Curie.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:7}
Tous deux attendent la moitié restante~: $10$, et \num{1.0e20}.
Fluctuations $\sim 1/\sqrt{n}$ du décompte~: avec $10$~désintégrations,
environ $30\%$ --- $7$ ou $13$ ne surprendraient pas~; avec
\num{1.0e20}, environ $10^{-10}$. Seul le grand échantillon obéit à
la loi à une précision mesurable.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:8}
$m = 140 \times \num{1.2e-4} = \qty{1.7e-2}{g}$~;
$N = (0.0168/40) \times \num{6.02e23} = \num{2.5e20}$.
$\lambda = 0.693/(\num{1.25e9} \times \num{3.156e7}) =
\qty{1.76e-17}{s^{-1}}$~; $\mathcal A = \lambda N \approx
\qty{4.4e3}{Bq}$ --- environ la moitié du total corporel (le reste
est surtout du carbone-14).
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:9}
$\lambda = 0.693/6.0 = \qty{0.116}{h^{-1}}$. À $15$~heures~:
$\mathcal A = 500\,e^{-0.116 \times 15} = 500 \times 0.177 \approx
\qty{88}{MBq}$. Sous \qty{1}{MBq}~:
$t = \ln(500)/0.116 \approx \qty{54}{hours}$ --- environ $9$
demi-vies.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:10}
Charbon~: $3.4/13.6 = \tfrac14$, exactement $2$~demi-vies~:
$t = \num{1.15e4}$~ans. Os~:
$t = \num{5730} \times \ln(1/0.30)/\ln 2 = \num{5730} \times 1.74
\approx \num{1.0e4}$~ans.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:11}
$\lambda = 0.693/3.8 = \qty{0.182}{d^{-1}}$~;
$t = \ln(100)/\lambda = 4.61/0.182 \approx 25$~jours (environ
$6.6$~demi-vies). Mais la chaîne de l'uranium dans le granite et le
sol environnants engendre du radon en continu~: une aération unique
est annulée en quelques semaines, donc la ventilation doit être
permanente.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:12}
$N/N_0 = e^{-\lambda t}$ donne $\lambda t = \ln(N_0/N)$, et
$\lambda = \ln 2/T_{1/2}$ donne $t = T_{1/2}\ln(N_0/N)/\ln 2$. Ici
$t = \num{5730} \times \ln(20)/\ln 2 = \num{5730} \times 4.32
\approx \num{2.5e4}$~ans.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:13}
$N_{\mathrm K} = N_0 e^{-\lambda t}$ et
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, donc
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$. Rapport $3.0$~:
$e^{\lambda t} = 4$, donc $\lambda t = 2\ln 2$ et
$t = 2\,T_{1/2} = \num{2.5e9}$~ans.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:14}
$\ln \mathcal A = 5.99$, $5.42$, $4.84$, $4.26$, $3.69$~: chutes de
$0.57$--$0.58$ par \qty{60}{s} --- alignés. Pente~:
$\lambda = 2.30/240 = \qty{9.6e-3}{s^{-1}}$~;
$T_{1/2} = 0.693/\num{9.6e-3} \approx \qty{72}{s}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:15}
(a) $2^{-10} \approx \num{1e-3}$. (b) $(1/2)^8 = 1/256 \approx
0.4\%$. (c) $1000/8 = 125$ attendus --- mais la loi se tait sur
lesquels~: elle prédit les populations exactement et les individus
pas du tout.
\end{solution}

\begin{solution}{pb:g12:radioactive-decay:1}
\textbf{1.} L'échange avec l'atmosphère (manger, respirer) maintient
la proportion au niveau atmosphérique~; la mort arrête l'apport et
la décroissance prend le relais.

\textbf{2.} $T_{1/2} = \num{5730} \times \num{3.156e7} =
\qty{1.81e11}{s}$, donc $\lambda = 0.693/\num{1.81e11} =
\qty{3.8e-12}{s^{-1}}$.

\textbf{3.} $\dd N/\dd t = -\lambda N_0 e^{-\lambda t} = -\lambda N$,
et $N(0) = N_0$.

\textbf{4.} $\mathcal A_0 = 13.6/60 = \qty{0.227}{Bq}$ par gramme~;
$N = \mathcal A_0/\lambda = 0.227/\num{3.8e-12} \approx \num{5.9e10}$
\omterm{def:g11:fundamental-interactions:ladder}{atomes} par gramme.

\textbf{5.} $(1/12) \times \num{6.02e23} = \num{5.0e22}$
\omterm{def:g11:fundamental-interactions:ladder}{atomes} de carbone~; proportion
$\num{5.9e10}/\num{5.0e22} \approx \num{1.2e-12}$ --- un sur
$10^{12}$, comme promis.

\textbf{6.} $\mathcal A = \mathcal A_0 e^{-\lambda t}$ donne
$\lambda t = \ln(\mathcal A_0/\mathcal A)$, et $\lambda =
\ln 2/T_{1/2}$ le transforme en
$t = T_{1/2}\ln(\mathcal A_0/\mathcal A)/\ln 2$.

\textbf{7.} $t = \num{5730} \times \ln(13.6/9.1)/\ln 2 =
\num{5730} \times 0.580 \approx \num{3.3e3}$~ans.

\textbf{8.} La mort du bois --- l'abattage de l'arbre, pas la
sculpture~; une statue taillée dans du vieux bois (ou un faux taillé
dans du bois antique) précède ou suit la date de son matériau.

\textbf{9.} $t = \num{5730} \times \ln(13.6/10.4)/\ln 2 =
\num{5730} \times 0.387 \approx \num{2.2e3}$~ans --- tout à fait
cohérent avec un scribe d'il y a vingt-deux siècles.

\textbf{10.} $\Delta t \approx \Delta\mathcal A/(\lambda\mathcal A)$
avec $\lambda = \ln 2/5730 = \num{1.21e-4}$ par an~:
$\Delta t = (0.2/10.4)/\num{1.21e-4} \approx \num{1.6e2}$~ans ---
une date bonne à un siècle ou deux près.

\textbf{11.} $t = \num{5730} \times \ln(13.6/13.4)/\ln 2 \approx
120$~ans --- indiscernable du moderne dans l'incertitude de
$\pm 160$~ans~: le parchemin est récent, l'«~antique~» un faux.

\textbf{12.} $13.6/2^n < 0.2$ exige $2^n > 68$~: $n = 7$~demi-vies
(puisque $2^7 = 128$), environ \num{4.0e4}~ans.

\textbf{13.} Dix demi-vies~: environ \num{5.7e4}~ans --- l'horizon
pratique, quelque soixante mille ans.

\textbf{14.} $\num{6.6e7}/\num{5730} \approx \num{11500}$~demi-vies.
Les \num{5.9e10}~\omterm{def:g11:fundamental-interactions:ladder}{atomes} d'un gramme sont
épuisés après $2^n > \num{5.9e10}$, soit $n \approx 36$~demi-vies
($\approx \num{2e5}$~ans)~: bien avant l'âge des dinosaures, il ne
reste pas un \omterm{def:g11:fundamental-interactions:ladder}{atome} de carbone-14 --- il n'y a
rien à compter.

\textbf{15.} Une horloge bien plus lente~: potassium-40 ou
uranium-238, avec des demi-vies de milliards d'années.

\textbf{16.} La lave en fusion laisse l'argon s'échapper en bulles,
donc le cristal fraîchement solidifié tient du potassium mais zéro
argon~: le rapport part de $0$ à la solidification --- l'horloge est
mise à zéro.

\textbf{17.} Comme dans l'exercice~: $N_{\mathrm K} = N_0 e^{-\lambda t}$,
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, donc
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.

\textbf{18.} $e^{\lambda t} = 1.60$~:
$t = \num{1.25e9} \times \ln(1.60)/\ln 2 = \num{1.25e9} \times 0.678
\approx \num{8.5e8}$~ans.

\textbf{19.} Non, deux fois~: à \num{8.5e8}~ans le carbone-14 a
couru $10^5$~demi-vies --- silence~; à \num{3.3e3}~ans le
potassium-40 a couru \num{2.6e-6} d'une
\omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie} --- le rapport d'argon
$\approx \lambda t \sim \num{2e-6}$ est immesurablement petit.
Chaque horloge ne lit que d'environ un dixième de
\omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie} à environ dix.

\textbf{20.} Statue $\approx \num{3.3e3}$~ans, papyrus
$\approx \num{2.2e3}$~ans, roche $\approx \num{8.5e8}$~ans ---
trois horloges, une loi~: $N = N_0\,e^{-\lambda t}$, lue à l'envers.
\end{solution}
