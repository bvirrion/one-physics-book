\begin{solution}{exo:g12:sound-acoustics:9}
Air~: $340/440 \approx \qty{0.77}{m}$~; eau~: $1500/440 \approx

\begin{solution}{exo:g12:sound-acoustics:10}
$f_1 = \qty{130}{Hz}$ (l'écart entre pics). Identique~: les positions
des barres $nf_1$ (même \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur}) et l'intensité totale~;
différent~: les hauteurs des barres --- le mélange qui fait le
\omterm{def:g12:sound-acoustics:timbre}{timbre}.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:11}
$v = 2Lf_1 = 2 \times 0.650 \times 110 = \qty{143}{m/s}$. Octave~:
$L' = v/(2 \times 220) = L/2 = \qty{32.5}{cm}$~; quinte~:
$L' = 143/330 \approx \qty{43.3}{cm} = 2L/3$.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:12}
$\lambda = 1540/\num{5.0e6} \approx \qty{0.31}{mm}$. Profondeur $= vt/2$~:
$1540 \times \num{26e-6}/2 \approx \qty{2.0}{cm}$ et
$1540 \times \num{58e-6}/2 \approx \qty{4.5}{cm}$~; épaisseur
$\approx \qty{2.5}{cm}$.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:13}
$-\qty{25}{dB}$ signifie $I \div 10^{2.5} \approx 316$, donc
$r \times \sqrt{316} \approx 17.8$~: $r \approx 3.0 \times 17.8
\approx \qty{53}{m}$. Les salles réelles diffèrent~: les réflexions du
sol et des murs ajoutent de l'intensité, tandis que l'air et la foule
en absorbent (et la scène n'est pas une source ponctuelle isotrope).
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:14}
$1/10^{-12} = 10^{12}$. Tympan $S = \qty{5e-5}{m^2}$~: \omterm{def:g11:work-of-force:power}{puissance}
reçue $\qty{5e-17}{W}$ au seuil, $\qty{5e-5}{W}$ à la douleur ---
l'oreille détecte des dizaines d'attowatts et fonctionne encore à un
billion de fois plus~: un détecteur à dynamique de douze
\omterm{def:g11:fundamental-interactions:strong}{décades}.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:15}
$2$~: \qty{83}{dB}~; $10$~: \qty{90}{dB}~; $100$~: \qty{100}{dB}. Chaque
\qty{+10}{dB} supplémentaire coûte un facteur dix en haut-parleurs~: les

\end{solution}

\begin{solution}{pb:g12:sound-acoustics:1}
\textbf{1.} $v = 2Lf_1 = 2 \times 0.650 \times 110 = \qty{143}{m/s}$.

\textbf{2.} Sur la corde $\lambda_1 = 2L = \qty{1.30}{m}$~; dans l'air
$\lambda = 340/110 \approx \qty{3.1}{m}$. Même $f$, mais
$\lambda = v/f$ et les célérités de l'\omterm{def:g10:signals-and-waves:wave}{onde} diffèrent.

\textbf{3.} $L' = v/(2 \times 220) = L/2 = \qty{32.5}{cm}$~: la
douzième case se place au milieu, à $L/2$ du chevalet.

\textbf{4.} $L' = 143/(2 \times 165) \approx \qty{43.3}{cm} = 2L/3$.

\textbf{5.} $\log(220/110)/\log 2^{1/12} = 12$ demi-tons --- douze
cases jusqu'à l'octave. Première case~: $L_1 = L/2^{1/12} = 65.0/1.059
\approx \qty{61.4}{cm}$.

\textbf{6.} \qtylist{110;220;330;440;550}{Hz}.

\textbf{7.} $\qty{440}{Hz} = 4f_1$~: le quatrième \omterm{def:g12:sound-acoustics:harmonics}{harmonique}, deux
octaves au-dessus. $\qty{330}{Hz} = 3f_1$~: le troisième
\omterm{def:g12:sound-acoustics:harmonics}{harmonique}, une octave plus une quinte.

\textbf{8.} La position du pincement fixe le mélange des modes~: près
du chevalet les \omterm{def:g12:sound-acoustics:harmonics}{harmoniques} aigus sont forts (\omterm{def:g12:sound-acoustics:timbre}{spectre}
brillant), au-dessus de la touche le \omterm{def:g12:sound-acoustics:harmonics}{fondamental} domine
(doux). Même $f_1$, donc même \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} --- \omterm{def:g12:sound-acoustics:timbre}{spectre}
différent, \omterm{def:g12:sound-acoustics:timbre}{timbre} différent.

\textbf{9.} Seuls les \omterm{def:g12:sound-acoustics:harmonics}{harmoniques} avec un \omterm{def:g12:sound-acoustics:standing-wave}{nœud} au
milieu survivent~: les pairs, \qtylist{220;440;660}{Hz}, \dots La
\omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} saute d'une octave, à \qty{220}{Hz}.

\textbf{10.} $nf_1 \le \qty{20}{kHz}$~: $n \le 20000/110 = 181.8$,
donc $181$ \omterm{def:g12:sound-acoustics:harmonics}{harmoniques}.

\textbf{11.} $I = P/(4\pi r^2) = \num{6.0e-3}/(4\pi \times 9.0)
\approx \qty{5.3e-5}{W/m^2}$.

\textbf{12.} $L = 10\log_{10}(\num{5.3e7}) \approx \qty{77}{dB}$.

\textbf{13.} $I \times 4$~: $+\qty{6}{dB} \to \qty{83}{dB}$. Les
niveaux sont des logarithmes~: les intensités s'additionnent, pas les
niveaux.

\textbf{14.} $+10\log_{10}12 \approx +\qty{10.8}{dB} \to
\qty{88}{dB}$.

\textbf{15.} Cible $I = I_0 \times 10^{8.5} \approx
\qty{3.2e-4}{W/m^2}$ avec $12P = \qty{7.2e-2}{W}$~:
$r = \sqrt{12P/(4\pi I)} \approx \sqrt{18.1} \approx \qty{4.3}{m}$.

\textbf{16.} \qty{88}{dB} est $+\qty{3}{dB}$ au-dessus de la limite
de huit heures~: le temps sûr est divisé une fois --- quatre heures. Le
récital tient.

\textbf{17.} $\lambda = 5000/\num{2.0e6} = \qty{2.5}{mm}$. Une
\omterm{def:g10:signals-and-waves:wave}{onde} ne se réfléchit que sur des défauts d'au moins environ une
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}~: le \omterm{def:g10:signals-and-waves:sound}{son} audible dans le bois a
$\lambda$ de dizaines de centimètres et diffracterait autour de tout
petit défaut.

\textbf{18.} $t = 2d/v = 2 \times 0.045/5000 = \qty{18}{\micro s}$.

\textbf{19.} $d = vt/2 = 5000 \times \num{7.2e-6}/2 = \qty{18}{mm}$.

\textbf{20.} La carte~: $v = \qty{143}{m/s}$ sur la corde~; cases à
$\qty{32.5}{cm}$ (douzième) et $\qty{61.4}{cm}$ (première)~; ensemble
sûr au-delà de $\approx \qty{4.3}{m}$~; défaut à \qty{18}{mm} de
profondeur --- quatre nombres, un chapitre.
\end{solution}
