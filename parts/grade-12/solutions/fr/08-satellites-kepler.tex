\section*{Chapitre \ref{ch:g12:satellites-kepler} --- Satellites et mouvement planétaire}

\begin{solution}{exo:g12:satellites-kepler:1}
Sur le pas de tir~: $F = GMm/R^2 = \num{3.98e17}/\num{4.06e13} \approx
\qty{9.8e3}{N}$. À $r = \qty{6.77e6}{m}$~:
$F \approx \qty{8.7e3}{N}$ --- encore $89\%$ de la valeur au sol. Les
occupants flottent parce qu'ils \emph{tombent} avec la station, non
parce que la gravité a disparu.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:2}
$v = \sqrt{GM/R} = \sqrt{\num{3.98e14}/\num{6.37e6}} \approx
\qty{7.9e3}{m/s}$~; $T = 2\pi R/v \approx \qty{5.1e3}{s} \approx
\qty{84}{min}$. Exactement les \qty{7.9}{km/s} du boulet de canon~: une
\omterm{def:g10:universal-gravitation:satellite}{orbite} rase \emph{est} le tir de Newton.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:3}
$v \propto 1/\sqrt{r}$~: divisée par $\sqrt 2 \approx 1.41$.
$T \propto r^{3/2}$~: multipliée par $2^{3/2} \approx 2.8$. Ni l'une ni
l'autre ne contient $m$~: non.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:5}
\omterm{def:g12:satellites-kepler:ellipse}{Ellipses} avec le Soleil en un foyer~; aires égales en temps
égaux~; $T^2/a^3$ identique pour toutes les planètes. Plus rapide au
\omterm{def:g12:satellites-kepler:ellipse}{périhélie}, par la deuxième loi (le court segment de balayage
doit balayer vite). Non~: la masse du \omterm{def:g10:universal-gravitation:satellite}{satellite}

\end{solution}

\begin{solution}{exo:g12:satellites-kepler:6}
$T = \qty{2.36e6}{s}$~: $M = 4\pi^2 r^3/(GT^2) =
\num{2.24e27}/371 \approx \qty{6.0e24}{kg}$ --- à $1\%$ près de
\qty{5.97e24}{kg}.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:7}
La gravité pointe vers le centre de la Terre, et en mouvement circulaire
la \omterm{def:g11:forces-and-motion:netforce}{force résultante} pointe vers le centre du cercle~: les
deux centres coïncident. Un cercle au-dessus du parallèle de Paris est
centré sur l'axe, non sur le centre de la Terre --- impossible. Le
stationnement \omterm{def:g11:work-of-force:work}{fonctionne} seulement là où le parallèle est un
grand cercle centré au centre~: au-dessus de l'\emph{équateur}.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:8}
$T = \qty{2.754e4}{s}$~: $M = 4\pi^2 r^3/(GT^2) =
\num{3.26e22}/\num{5.06e-2} \approx \qty{6.4e23}{kg}$ --- Mars de la
carte de données.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:9}
Terre~: $T^2/a^3 = \num{9.96e14}/\num{3.35e33} =
\qty{2.98e-19}{s^2/m^3}$. Mars~: $T = \qty{5.94e7}{s}$,
$\num{3.52e15}/\num{1.18e34} = \qty{2.98e-19}{s^2/m^3}$ --- égaux, comme
le demande la troisième loi. Alors $M = 4\pi^2/(G \times \num{2.98e-19})
\approx \qty{1.99e30}{kg}$.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:10}
$v = 2\pi r/T = \qty{1.02e3}{m/s}$~; $a = v^2/r =
\qty{2.72e-3}{m/s^2}$~; $g/60^2 = 9.81/3600 = \qty{2.73e-3}{m/s^2}$.
La Lune est à $60$ rayons terrestres, et sa chute est $60^2$ fois plus
faible que celle de la pomme~: exactement l'inverse du carré --- une
seule loi pour les deux.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:11}
Cérès~: $T = 2.77^{3/2} = \sqrt{21.3} \approx \qty{4.6}{yr}$. Halley~:
$T = 17.8^{3/2} \approx \qty{75}{yr}$, donc prochain
\omterm{def:g12:satellites-kepler:ellipse}{périhélie} vers $1986 + 75 \approx 2061$.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:12}
À \qty{400}{km}~: $v \approx \qty{7.67e3}{m/s}$~; à \qty{350}{km}
($r = \qty{6.72e6}{m}$)~: $v \approx \qty{7.70e3}{m/s}$. Le
\omterm{def:g10:inertia:inventory}{frottement} draine bien l'\omterm{def:g11:mechanical-energy:em}{énergie mécanique}
--- l'\omterm{def:g10:universal-gravitation:satellite}{orbite} se rétrécit~; mais en descendant le
\omterm{def:g11:work-of-force:work}{travail} de la gravité rembourse plus que la perte de
vitesse due à la traînée~: $E_m$ chute tandis que $E_k$ monte.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:13}

\begin{solution}{exo:g12:satellites-kepler:14}
$T = \qty{8.862e4}{s}$, $GM = \qty{4.28e13}{m^3/s^2}$~:
$r^3 = GMT^2/(4\pi^2) = \num{8.52e21}$, donc $r \approx
\qty{2.04e7}{m}$ et $h = r - R \approx \qty{1.7e7}{m} \approx

\end{solution}

\begin{solution}{exo:g12:satellites-kepler:15}
$v_a = r_p v_p / r_a = \num{8.77e10} \times \num{5.45e4} /
\num{5.25e12} \approx \qty{9.1e2}{m/s}$. Comme $r_a/r_p \approx 60$,
la comète est $60$ fois plus loin et $60$ fois plus lente~: elle passe
des décennies à musarder dans le noir et des semaines à sprinter devant
le Soleil.
\end{solution}

\begin{solution}{pb:g12:satellites-kepler:1}
\textbf{1.} $T = \qty{86164}{s}$~: le sol tourne avec la Terre
\emph{par rapport aux étoiles} une fois par \omterm{def:g12:satellites-kepler:geo}{jour sidéral}~; le
jour de \qty{24}{h} ajoute le degré par jour dont la Terre avance
autour du Soleil.

\textbf{2.} La \omterm{def:g11:forces-and-motion:netforce}{force résultante} du mouvement circulaire vise le
centre du cercle~; la seule \omterm{def:g10:inertia:force}{force}, la gravité, vise le
centre de la Terre~: les deux centres sont le même point.

\textbf{3.} Un cercle stationnaire au-dessus de Paris serait centré sur
l'axe au nord du centre --- exclu par 2. L'\omterm{def:g10:universal-gravitation:satellite}{orbite}
doit se trouver dans le \emph{plan équatorial}.

\textbf{4.} Composante normale de la \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton}~:
$m v^2/r = GmM/r^2$, donc $v = \sqrt{GM/r}$ ($m$ s'annule).

\textbf{5.} $T = 2\pi r/v = 2\pi\sqrt{r^3/(GM)}$~; en élevant au carré,
$T^2/r^3 = 4\pi^2/(GM)$.

\textbf{6.} $r = \left(GMT^2/4\pi^2\right)^{1/3} =
(\num{7.49e22})^{1/3} \approx \qty{4.22e7}{m}$.

\textbf{7.} $h = r - R = \num{4.22e7} - \num{6.37e6} \approx
\qty{3.58e7}{m} \approx \qty{35800}{km}$.

\textbf{8.} $v = 2\pi \times \num{4.22e7}/\num{86164} \approx
\qty{3.07e3}{m/s}$.

\textbf{9.} $\sqrt{\num{3.98e14}/\num{4.22e7}} \approx
\qty{3.07e3}{m/s}$ --- cohérent.

\textbf{10.} Rayons identiques~: la masse s'est annulée en 4. Le
\omterm{def:g10:universal-gravitation:satellite}{satellite} plus lourd ne change que la facture de
lancement --- plus de carburant pour la même
\omterm{def:g10:universal-gravitation:satellite}{orbite}.

\textbf{11.} $v = \sqrt{\num{3.98e14}/\num{6.77e6}} \approx
\qty{7.67e3}{m/s}$.

\textbf{12.} $T = 2\pi r/v \approx \qty{5.55e3}{s} \approx
\qty{92.4}{min}$.

\textbf{13.} $\num{86400}/5546 \approx 15.6$ \omterm{def:g10:universal-gravitation:satellite}{orbites}~: environ
$16$ levers de soleil par jour.

\textbf{14.} $r_{\text{geo}}/r_{\text{ISS}} = \num{4.22e7}/\num{6.77e6}
= 6.23$~; $6.23^{3/2} \approx 15.5 = \num{86164}/5546$ --- la troisième
loi de Kepler, vérifiée sur place.

\textbf{15.} La station est plus rapide (\qty{7.67}{km/s} contre
\qty{3.07}{km/s}), pourtant le \omterm{def:g12:satellites-kepler:geo}{géostationnaire} coûte plus
d'\omterm{def:g10:energy-conservation:energy}{énergie} par \omterm{def:g10:orders-of-magnitude:unit}{kilogramme}~: la montée en
\omterm{def:g10:energy-conservation:energy}{énergie} potentielle l'emporte sur la perte de vitesse.

\textbf{16.} $M_J = 4\pi^2 r^3/(G T^2)$.

\textbf{17.} $T = \qty{1.529e5}{s}$~: $M_J =
\num{2.97e27}/1.56 \approx \qty{1.90e27}{kg}$.

\textbf{18.} $\num{1.90e27}/\num{5.97e24} \approx 318$ Terres ---
et environ $1/1000$ du Soleil.

\textbf{19.} $T_J = \qty{3.576e4}{s}$~: $r^3 = GM_J T_J^2/(4\pi^2) =
\num{4.11e24}$, $r \approx \qty{1.60e8}{m}$ --- Io, à
\qty{4.22e8}{m}, vole $2.6$ fois plus haut et dérive à travers le ciel
de Jupiter.

\textbf{20.} Parquer le relais sur le cercle équatorial d'altitude
\qty{35800}{km}, où il voyage à \qty{3.07}{km/s} et ne quitte jamais
la visée de l'antenne~; et la même loi, nourrie du mois d'Io, a pesé
Jupiter à $\qty{1.90e27}{kg}$ --- $318$ Terres --- gratuitement.
\end{solution}
