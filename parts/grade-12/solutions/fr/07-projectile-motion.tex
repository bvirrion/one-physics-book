\section*{Chapitre \ref{ch:g12:projectile-motion} --- Chute libre et mouvement de projectile}

\begin{solution}{exo:g12:projectile-motion:1}
$h = \tfrac12 \times 9.81 \times 2.0^2 = \qty{19.6}{m}$~;
$v = gt = \qty{19.6}{m/s} \approx \qty{71}{km/h}$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:2}
Pas d'air sur la Lune~: tous deux en \omterm{def:g12:projectile-motion:freefall}{chute libre},
$m\vect a = m\vect g$ donne $\vect a = \vect g$ pour toute masse. Sur
Terre, la traînée sur la plume est comparable à son minuscule
\omterm{def:g10:universal-gravitation:weight}{poids}~: elle voyage à une \omterm{def:g12:projectile-motion:terminal}{vitesse limite}
d'environ \qty{1}{m/s} tandis que le marteau remarque à peine l'air.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:3}
$t = \sqrt{2h/g} = 0.45$, $1.4$, \qty{4.5}{s}~;
$v = gt = 4.4$, $14$, \qty{44}{m/s}. Les temps croissent comme
$\sqrt h$~: $\times 100$ en hauteur n'est que $\times 10$ en temps.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:4}
$h = v_0^2/2g = \qty{11.5}{m}$ en $t = v_0/g = \qty{1.53}{s}$~;
retour en \qty{3.06}{s}, à \qty{15}{m/s} encore (symétrie).
\end{solution}

\begin{solution}{exo:g12:projectile-motion:5}
$t = \sqrt{2 \times 0.80/9.81} = \qty{0.40}{s}$~;
$x = 2.5 \times 0.40 = \qty{1.0}{m}$. Non~: le temps de chute
appartient au mouvement vertical seul (indépendance).
\end{solution}

\begin{solution}{exo:g12:projectile-motion:6}
$(\ang{15},\ang{75})$ et $(\ang{30},\ang{60})$ partagent leur
\omterm{def:g12:projectile-motion:range}{portée}~; \ang{45} gagne. Le plus raide d'une paire
complémentaire vole plus longtemps~: $t_f = 2v_0\sin\alpha/g$ croît
avec $\alpha$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:7}
$4.905\,t^2 + 5.0\,t - 20 = 0$ donne
$t = (-5 + \sqrt{25 + 392.4})/9.81 = \qty{1.57}{s}$~;
$v = 5.0 + 9.81 \times 1.57 = \qty{20.4}{m/s}$. \omterm{def:g10:energy-conservation:energy}{Énergie}~:
$v = \sqrt{5.0^2 + 2 \times 9.81 \times 20} = \qty{20.4}{m/s}$ ---
même chose.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:8}
$t_f = 2 \times 40 \times \sin\ang{30}/9.81 = \qty{4.1}{s}$~;
$h = (40\sin\ang{30})^2/(2 \times 9.81) = \qty{20.4}{m}$~;
$R = 40^2 \sin\ang{60}/9.81 = \qty{1.4e2}{m}$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:9}
$R = 10.5^2\sin\ang{40}/9.81 \approx \qty{7.2}{m}$. Oublié~: le
\omterm{def:g11:forces-and-motion:com}{centre de masse} part haut (corps étendu) et atterrit bas et
loin en avant (jambes projetées), et le sauteur n'est pas un point
--- les \qty{1.7}{m} manquants vivent dans cette géométrie, non dans
la parabole.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:10}
Ensemble~: tous deux ont $v_y(0) = 0$, donc le même
$t = \sqrt{2 \times 1.2/9.81} = \qty{0.49}{s}$~; celui qui est tiré
atterrit à $300 \times 0.49 \approx \qty{1.5e2}{m}$. Cela teste
l'indépendance des mouvements horizontal et vertical.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:11}
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180}-2\alpha) =
\sin(2\alpha)$~: même $R$. Ici $\sin(2\alpha) = gR/v_0^2 =
9.81 \times 50/625 = 0.785$~: $\alpha = \ang{25.9}$ ou \ang{64.1}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:12}
$R_{\max} = v_0^2/g = \qty{6.5}{m}$ (à \ang{45})~;
$h_{\max} = v_0^2/2g = \qty{3.3}{m}$ (tout droit vers le haut).
$\sin(2\alpha) = 9.81 \times 5.0/64 = 0.766$~: $\alpha = \ang{25.0}$
ou \ang{65.0}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:13}
$y(15) = 15\tan\ang{35} - \dfrac{9.81 \times 15^2}
{2 \times 18^2 \cos^2\ang{35}} = 10.5 - 5.1 = \qty{5.4}{m}$~:
elle franchit la haie de \qty{4.0}{m} d'environ \qty{1.4}{m}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:14}
(a) La traînée équilibre le \omterm{def:g10:universal-gravitation:weight}{poids}~:
$80 \times 9.81 \approx \qty{7.8e2}{N}$.
(b) $v(t)$ monte avec une pente toujours plus faible et se stabilise
à \qty{55}{m/s}. (c) Vide~:
$\sqrt{2 \times 9.81 \times 2000} \approx \qty{2.0e2}{m/s}$, vingt
fois le vrai \qty{9}{m/s} --- les gouttes de pluie passent presque
toute la chute à la \omterm{def:g12:projectile-motion:terminal}{vitesse limite}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:15}
En $t^* = D/(v_0\cos\alpha)$ le dard est à la hauteur
$v_0\sin\alpha\, t^* - \tfrac12 g t^{*2} = D\tan\alpha - \tfrac12 g
t^{*2} = H - \tfrac12 g t^{*2}$, exactement celle du singe lâché
$y = H - \tfrac12 g t^{*2}$. Tous deux pendent de $\tfrac12 gt^2$
sous leurs positions sans gravité --- ils se rencontrent pour tout
$v_0$.
\end{solution}

\begin{solution}{pb:g12:projectile-motion:1}
\textbf{1.} \omterm{def:g10:universal-gravitation:weight}{Poids} seul (air négligé)~; \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton}~:
$m\vect a = m\vect g$, donc $\vect a = (0,\,-g)$.

\textbf{2.} $v_x = v_0\cos\alpha$, $v_y = v_0\sin\alpha - gt$.

\textbf{3.} $x = v_0\cos\alpha\, t$,
$y = v_0\sin\alpha\, t - \tfrac12 gt^2$.

\textbf{4.} $t = x/(v_0\cos\alpha)$~:
$y = x\tan\alpha - g x^2/(2v_0^2\cos^2\alpha)$.

\textbf{5.} $v_0^2 = \dfrac{g d^2}{2\cos^2\alpha\,(d\tan\alpha -
1.05)} = \dfrac{173.0}{0.826 \times 3.96} = 52.9$~:
$v_0 = \qty{7.28}{m/s} \approx \qty{26}{km/h}$.

\textbf{6.} $t = d/(v_0\cos\alpha) = 4.20/4.68 = \qty{0.90}{s}$.

\textbf{7.} Sommet en $t = v_0\sin\alpha/g = \qty{0.57}{s}$, hauteur
$2.00 + (v_0\sin\alpha)^2/2g = 2.00 + 1.58 = \qty{3.58}{m}$~;
$0.57 < 0.90$~: en descente au bord.

\textbf{8.} $y(0.90) = 0.90\tan\ang{50} - 0.182 = \qty{0.89}{m}$
au-dessus du lâcher, c.-à-d.\ \qty{2.89}{m} au-dessus du sol~:
\qty{0.29}{m} au-dessus des bouts des doigts.

\textbf{9.} $v_x = \qty{4.68}{m/s}$~;
$v_y = 5.57 - 9.81 \times 0.90 = \qty{-3.24}{m/s}$~;
$v = \qty{5.69}{m/s}$, à $\arctan(3.24/4.68) = \ang{34.7}$ sous
l'horizontale.

\textbf{10.} Le bord est un anneau horizontal~: la balle doit croiser
son plan par le dessus. Plus l'entrée est raide, plus l'ouverture de
l'anneau paraît grande le long de la vitesse --- plus de marge
d'erreur.

\textbf{11.} $1.05/26.0 = 4\%$~: négligeable. $y(t_f) = 0$ donne
$t_f\,(v_0\sin\alpha - \tfrac12 g t_f) = 0$, donc
$t_f = 2v_0\sin\alpha/g$.

\textbf{12.} $R = v_0\cos\alpha\, t_f = v_0^2\sin(2\alpha)/g$~;
maximale à $\alpha = \ang{45}$.

\textbf{13.} $v_0 = \sqrt{gR} = \sqrt{9.81 \times 26.0} =
\qty{16.0}{m/s} \approx \qty{57}{km/h}$.

\textbf{14.} $R_{\max} = 17.5^2/9.81 = \qty{31.2}{m} >
\qty{26.0}{m}$~: faisable, avec \qty{5}{m} de \omterm{def:g12:projectile-motion:range}{portée}
de marge.

\textbf{15.} $\sin(2\alpha) = 255.1/306.3 = 0.833$~:
$\alpha = \ang{28.2}$ ou \ang{61.8}~; temps de vol
$2v_0\sin\alpha/g = \qty{1.69}{s}$ et \qty{3.14}{s}. Seul le tir
plat bat le buzzer de \qty{2.0}{s}.

\textbf{16.} $v_0^2 = \dfrac{9.81 \times 36}{2\cos^2\ang{75}\,
(6.0\tan\ang{75} - 1.05)} = \dfrac{353.2}{2.86} = 123.5$~:
$v_0 = \qty{11.1}{m/s}$.

\textbf{17.} $y(5.55) = 5.55\tan\ang{75} - 18.26 = \qty{2.45}{m}$
au-dessus du lâcher, c.-à-d.\ \qty{4.45}{m} au-dessus du sol~:
\qty{0.50}{m} au-dessus du bord supérieur du panneau.

\textbf{18.} Sommet $2.00 + (11.1\sin\ang{75})^2/2g \approx
\qty{7.9}{m}$~; vol $6.0/(11.1\cos\ang{75}) = \qty{2.1}{s}$ ---
deux secondes pleines avec la balle suspendue à \qty{8}{m}.

\textbf{19.} La traînée raccourcit chaque tir et raidiit la descente~;
l'effet croît avec la vitesse, donc le lancer de \qty{17.5}{m/s} de
toute la salle souffre le plus --- sa vraie \omterm{def:g12:projectile-motion:range}{portée} tombe
de quelques mètres sous \qty{31}{m}, grignotant la marge.

\textbf{20.} \emph{Le verdict du buzzer-beater}~: le tir de
\qty{26.0}{m} demande \qty{16.0}{m/s} et un bras fournit environ
\qty{17.5}{m/s}, une marge de \omterm{def:g12:projectile-motion:range}{portée} de \qty{5}{m}
($\approx 20\%$) que la résistance de l'air rogne mais n'efface pas.
Possible --- c'est exactement pourquoi, quelques fois par saison, ça
rentre.
\end{solution}
