\begin{solution}{exo:g12:quantum-world:7}

\omterm{def:g12:quantum-world:photon}{photons}), $U_s$ inchangé (même
\omterm{def:g10:energy-conservation:energy}{énergie} par \omterm{def:g12:quantum-world:photon}{photon}).
\omterm{def:g12:oscillators-and-time:oscillator}{Fréquence} élevée~: $U_s$ monte,
$E_{k,\max} = h\nu - W_0$. Le tableau d'\omterm{def:g10:signals-and-waves:wave}{onde} lie
l'\omterm{def:g10:energy-conservation:energy}{énergie} des électrons à l'intensité, pas à la
\omterm{def:g12:oscillators-and-time:oscillator}{fréquence} --- exactement ce qui n'est
\emph{pas} observé.
\end{solution}

\begin{solution}{exo:g12:quantum-world:9}
Depuis l'\omterm{def:g12:quantum-world:levels}{état fondamental} le plus petit saut coûte
$E_2 - E_1 = \qty{10.2}{eV}$ (\qty{122}{nm})~: toute raie d'absorption
est \omterm{def:g10:light-spectra:wavelength}{ultraviolette}, aucune visible. Dans une
atmosphère stellaire chaude les collisions maintiennent certains
\omterm{def:g11:fundamental-interactions:ladder}{atomes} en $n = 2$, d'où les
\omterm{def:g12:quantum-world:photon}{photons} visibles de Balmer (\qtyrange{1.89}{3.40}{eV})
peuvent être absorbés.
\end{solution}

\begin{solution}{exo:g12:quantum-world:12}

\end{solution}

\begin{solution}{pb:g12:quantum-world:1}
\textbf{1.} $E = hc/\lambda = \num{1.99e-25}/\num{5.50e-7} =
\qty{3.62e-19}{J} = \qty{2.26}{eV}$.

\textbf{2.} $N = \num{1.0e3}/\num{3.62e-19} \approx \num{2.8e21}$
\omterm{def:g12:quantum-world:photon}{photons} par seconde.

\textbf{3.} $P = 0.20 \times \qty{1.0e3}{W} = \qty{200}{W}$~;
$E = 200 \times 5.0 = \qty{1.0}{kW.h}$ ($\qty{3.6e6}{J}$).

\textbf{4.} Avec $10^{21}$~grains par seconde, le scintillement
statistique est fantastiquement petit~: le grain s'average en un
courant parfaitement lisse, comme du sable versé vite qui coule comme
de l'eau.

\textbf{5.} Les dégâts exigent un grain de \qtyrange{3}{4}{eV}~: les
grains \omterm{def:g10:light-spectra:wavelength}{ultraviolets} qualifient, les grains
\omterm{def:g10:light-spectra:wavelength}{infrarouges} ($\sim \qty{0.1}{eV}$) jamais ---
les nombres ne remplacent pas le punch par grain.

\textbf{6.} $U_s$ est la \omterm{def:g11:circuits-and-power:voltage}{tension} inverse qui
arrête juste les électrons les plus rapides~:
$eU_s = E_{k,\max} = h\nu - W_0$.

\textbf{7.} $\nu = c/\lambda$~: $8.22$, $7.41$, $6.88$ et $5.49$
$\times\,10^{14}\,\unit{Hz}$.

\textbf{8.} $eU_s = h\nu - W_0$ est affine en $\nu$ de pente $h$.
Pentes successives~: $(0.94 - 0.37)/1.39 = 0.41$ et
$(1.50 - 0.94)/1.34 = 0.42$ \unit{eV} par $10^{14}\,\unit{Hz}$ ---
les trois points s'alignent.

\textbf{9.} $h = e(U_{s,1}-U_{s,4})/(\nu_1-\nu_4) = \num{1.60e-19}
\times 1.13/\num{2.72e14} = \qty{6.64e-34}{J.s}$.

\textbf{10.} À \qty{546}{nm}~: $W_0 = h\nu - eU_s = \num{3.64e-19} -
\num{0.59e-19} = \qty{3.05e-19}{J} = \qty{1.90}{eV}$~;
$\nu_0 = W_0/h = \qty{4.60e14}{Hz}$~; $\lambda_0 \approx \qty{653}{nm}$.

\textbf{11.} Seul $\lambda < \qty{653}{nm}$ éjecte~: le visible
violet-à-rouge et l'\omterm{def:g10:light-spectra:wavelength}{ultraviolet}. Tout le côté
\omterm{def:g10:light-spectra:wavelength}{infrarouge} --- environ la moitié de
l'\omterm{def:g10:energy-conservation:energy}{énergie} du Soleil --- est inutile à cette
cathode.

\textbf{12.} $(6.64 - 6.63)/6.63 \approx 0.2\%$ --- une mesure de
labo de physique à deux pour mille d'une constante universelle.

\textbf{13.} $E_2 = \qty{-3.40}{eV}$, $E_3 = \qty{-1.51}{eV}$,
$E_4 = \qty{-0.85}{eV}$, $E_5 = \qty{-0.54}{eV}$.

\textbf{14.} $3 \to 2$~: \qty{1.89}{eV}~; $4 \to 2$~: \qty{2.55}{eV}~;
$5 \to 2$~: \qty{2.86}{eV}.

\textbf{15.} $\lambda = 1240/E$~: \qty{656}{nm} (rouge), \qty{486}{nm}
(bleu-vert), \qty{434}{nm} (violet).

\textbf{16.} Les énergies de l'\omterm{def:g11:fundamental-interactions:ladder}{atome} sont
\omterm{def:g12:quantum-world:levels}{quantifiées}, donc il ne peut émettre que les
différences discrètes $E_p - E_n$~: une échelle fait des raies, pas un
arc-en-ciel.

\textbf{17.} Non~: \qty{500}{nm} signifie \qty{2.48}{eV}, qui n'épouse
aucune différence de niveaux ($E_2$ vers le haut donne $1.89$, $2.55$,
$2.86$, \qty{3.40}{eV}~; depuis $E_1$ tout coûte au moins
\qty{10.2}{eV}) --- le gaz y est transparent.

\textbf{18.} $eU = \tfrac12 m_e v^2$~: $v = \sqrt{2 \times
\num{1.60e-19} \times \num{1.0e4}/\num{9.11e-31}} \approx
\qty{5.9e7}{m/s} \approx 0.20\,c$ --- traitement classique acceptable.

\textbf{19.} $p = m_e v = \qty{5.4e-23}{kg.m/s}$~;
$\lambda = h/p = \qty{1.2e-11}{m} = \qty{12}{pm}$.

\textbf{20.} Gain $= \num{5.50e-7}/\num{1.23e-11} \approx \num{4.5e4}$.
Les deux nombres du week-end~: $h = \qty{6.64e-34}{J.s}$, mesuré à
$0.2\%$ avec des \omterm{def:g11:color-light-sources:subtractive}{filtres} et un voltmètre, et
un microscope quarante-cinq mille fois plus aigu parce que les
électrons \omterm{def:g10:signals-and-waves:wave}{ondulent}.
\end{solution}
