\section*{Chapitre \ref{ch:g12:oscillators-and-time} --- Oscillateurs mécaniques et mesure du temps}

\begin{solution}{exo:g12:oscillators-and-time:1}
(a) $T_0 = 2\pi\sqrt{1.00/9.81} = \qty{2.01}{s}$~;
$f_0 = 1/T_0 = \qty{0.50}{Hz}$. (b) $T_0 \propto \sqrt{L}$~: la
\omterm{def:g12:oscillators-and-time:oscillator}{période} double.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:2}
Cœur~: $f = 72/60 = \qty{1.2}{Hz}$, $T = \qty{0.83}{s}$. Aile~:
$T = 1/30 = \qty{33}{ms}$. La montée de la poitrine est
l'\emph{\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}}.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:3}
$T_0 = 2\pi\sqrt{0.10/25} = \qty{0.40}{s}$, $f_0 = \qty{2.5}{Hz}$.
Doubler $m$~: $T_0 \times \sqrt{2} \approx 1.4$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:4}
\omterm{def:g12:oscillators-and-time:damping}{Pseudo-période} \qty{2.0}{s}. Rapports $1.5/2.0 = 1.13/1.5 = 0.85/1.13
= 0.75$ chaque \omterm{def:g12:oscillators-and-time:oscillator}{période}~; prochain maximum $0.85 \times 0.75
\approx \qty{0.64}{cm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:5}
Pendule~: $T_0 \propto 1/\sqrt{g}$, donc $\times\sqrt{9.81/1.62} =
2.46$ (plus lent). Masse--ressort~: inchangé --- pas de $g$ dans
$2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:6}
(a) $[L/g] = \unit{m}/(\unit{m/s^2}) = \unit{s^2}$~; sa racine
carrée est un temps. (b) Le \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} n'apparaît que dans $m$~:
rien d'autre ne pourrait l'annuler, donc $m$ ne peut pas entrer.
(c) Jamais~: les nombres purs comme $2\pi$ sont invisibles aux
dimensions.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:7}
$\frac{dx}{dt} = -\frac{2\pi}{T_0}A\sin(2\pi t/T_0 + \varphi)$, donc
$\frac{d^2x}{dt^2} = -(2\pi/T_0)^2 x(t)$. L'égalité avec $-(k/m)x$
pour tout $t$ vaut exactement lorsque $(2\pi/T_0)^2 = k/m$,
c'est-à-dire $T_0 = 2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:8}
$E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.030^2 =
\qty{1.8e-2}{J}$~; $v_{\max} = \sqrt{2E_m/m} = \sqrt{0.18} =
\qty{0.42}{m/s}$ --- en $x = 0$~; nulle aux extrêmes $x = \pm A$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:9}
(a) $L = gT_0^2/4\pi^2 = \qty{0.994}{m}$. (b) $\Delta T_0/T_0 =
\tfrac12 \times 1.0\% = 0.5\%$~: $86400 \times 0.005 \approx
\qty{4.3e2}{s}$ --- sept minutes de retard par jour.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:10}
(a) $A \propto \sqrt{E}$~: $\sqrt{0.95} \approx 0.975$, une chute
de $2.5\%$. (b) $0.95^n = 0.5$~: $n = \ln 2/\ln(1/0.95) \approx 14$
\omterm{def:g12:oscillators-and-time:oscillator}{périodes}. (c) Non~: pour un \omterm{def:g12:oscillators-and-time:damping}{amortissement} léger elle
reste à une fraction de pour cent de $T_0$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:11}
(a) À $f_0$ chaque poussée arrive en phase et ajoute de
l'\omterm{def:g10:energy-conservation:energy}{énergie}~; hors du pic, les poussées aident et gênent
alternativement. (b) Marcher au pas est une excitation
\omterm{def:g10:signals-and-waves:period}{périodique}~; près de la $f_0$ du pont elle monterait le pic de
\omterm{def:g12:oscillators-and-time:resonance}{résonance}. (c) Le tambour balaie la
\omterm{def:g12:oscillators-and-time:resonance}{fréquence propre} du châssis pendant la montée en
régime~: \omterm{def:g12:oscillators-and-time:resonance}{résonance} momentanée, puis au-delà du pic.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:12}
(a) $T_0 \propto 1/\sqrt{g}$~: retard, de $\Delta T_0/T_0 = \tfrac12
\times 0.03/9.81 = \num{1.5e-3}$, soit $\approx \qty{132}{s}$ par
jour. (b) $L \propto g$ à $T_0$ fixé~: raccourcir de $0.994 \times
0.03/9.81 \approx \qty{3.0}{mm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:13}
$v_{\max} = 2\pi A/T_0 = 2\pi \times 0.040/0.63 = \qty{0.40}{m/s}$~;
$a_{\max} = (2\pi/T_0)^2 A = \qty{4.0}{m/s^2}$~; $k = m(2\pi/T_0)^2
= \qty{20}{N/m}$~; $E_m = \tfrac12 kA^2 = \qty{1.6e-2}{J}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:14}
$T_0 = 40.1/20 = \qty{2.005}{s}$~; $g = 4\pi^2 L/T_0^2 = 4\pi^2
\times 0.995/4.020 = \qty{9.77}{m/s^2}$. Incertitude $\approx 0.2\%
+ 2 \times 0.5\% = 1.2\%$, soit $\pm\qty{0.12}{m/s^2}$~: l'intervalle
contient \qty{9.81}{m/s^2} --- cohérent.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:15}
Pendule $10/86400 \approx \num{1.2e-4}$~; quartz $0.5/\num{2.6e6}
\approx \num{1.9e-7}$~; césium $1/\num{3.2e15} \approx \num{3e-16}$.
Une erreur de \qty{1}{\micro s} est $\num{1000} \times \qty{30}{cm}
= \qty{300}{m}$ de position~: le GPS exige la famille atomique.
\end{solution}

\begin{solution}{pb:g12:oscillators-and-time:1}
\textbf{1.} L'erreur de réaction $\approx \qty{0.2}{s}$ est divisée
par $50$~: environ \qty{4}{ms} sur $T_0$.

\textbf{2.} $T = 99.4/50 = \qty{1.988}{s}$.

\textbf{3.} $L = gT^2/4\pi^2 = 9.81 \times 1.988^2/4\pi^2 \approx
\qty{0.982}{m}$.

\textbf{4.} $L = \qty{0.994}{m}$ (le pendule à secondes)~: descendre
la masse, allonger d'$\approx \qty{1.2}{cm}$.

\textbf{5.} Trop court, donc trop rapide~: avance, de $(2.000 -
1.988)/1.988 \times 86400 \approx \qty{5.2e2}{s} \approx 8.7$
minutes par jour.

\textbf{6.} Lune~: $\times\sqrt{9.81/1.62} = 2.46$ --- inutilisable.
Observatoire~: $\Delta T_0/T_0 = \tfrac12 \times 0.01/9.81 =
\num{5.1e-4}$~: perd $\approx \qty{44}{s}$ par jour.

\textbf{7.} $\Delta L = L\lambda\,\Delta\theta = 0.994 \times
\num{1.9e-5} \times 5.0 \approx \qty{9.4e-5}{m} \approx
\qty{0.1}{mm}$.

\textbf{8.} $T' = 2\pi\sqrt{L(1 + \lambda\Delta\theta)/g} =
T_0\sqrt{1 + \lambda\Delta\theta} \approx T_0\,(1 + \tfrac12
\lambda\Delta\theta)$.

\textbf{9.} $\Delta T_0/T_0 = \tfrac12 \times \num{1.9e-5} \times
5.0 = \num{4.8e-5}$~: $86400 \times \num{4.8e-5} \approx
\qty{4.1}{s}$ par jour, retard (tige plus longue, \omterm{def:g12:oscillators-and-time:oscillator}{période}
plus longue).

\textbf{10.} $\Delta\theta = \qty{-10}{K}$~: $\Delta T_0/T_0 =
\num{-9.5e-5}$, gagne $\approx \qty{8.2}{s}$ par jour.

\textbf{11.} L'acier et le laiton se dilatent différemment~; montés
en directions alternées, les tiges de laiton abaissent la masse
tandis que celles d'acier la relèvent. Dimensionnées pour que les
deux déplacements s'annulent, la longueur effective --- et le rythme
--- survit aux saisons.

\textbf{12.} $280/2.000 = 140$ \omterm{def:g12:oscillators-and-time:oscillator}{périodes}.

\textbf{13.} $\theta_m = 2.00^\circ = \qty{0.0349}{rad}$~: $E_m =
\tfrac12 \times 1.2 \times 9.81 \times 0.994 \times 0.0349^2 \approx
\qty{7.1e-3}{J}$.

\textbf{14.} Par \omterm{def:g12:oscillators-and-time:oscillator}{période} l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}
rétrécit de $2^{-1/140} \approx 0.9951$, l'\omterm{def:g10:energy-conservation:energy}{énergie} de
$0.9951^2 \approx 0.990$~: environ $1.0\%$ de $E_m$ par
\omterm{def:g12:oscillators-and-time:oscillator}{période}.

\textbf{15.} Poussée $\approx 0.010 \times \num{7.1e-3} =
\qty{7.1e-5}{J}$~; \omterm{def:g11:work-of-force:power}{puissance} $\num{7.1e-5}/2.0 \approx
\qty{3.5e-5}{W}$ --- des dizaines de microwatts.

\textbf{16.} Une semaine fait $604800/2.0 = \num{3.02e5}$
\omterm{def:g12:oscillators-and-time:oscillator}{périodes}, demandant $\num{3.02e5} \times \num{7.1e-5}
\approx \qty{21}{J}$~; le \omterm{def:g10:universal-gravitation:weight}{poids} fournit $Mgh = 2.5
\times 9.81 \times 1.0 = \qty{24.5}{J}$~: assez, à $21/24.5 \approx
88\%$ de \omterm{def:g10:energy-conservation:efficiency}{rendement}.

\textbf{17.} $T = 1/\num{32768} = \qty{30.5}{\micro s}$. $\num{32768}
= 2^{15}$~: quinze étages de division par deux transforment le
\omterm{def:g10:signals-and-waves:signal}{signal} du cristal en un tic exact à \qty{1}{Hz}.

\textbf{18.} Quartz~: $0.3/\num{2.6e6} \approx \num{1.2e-7}$.
Pendule~: $0.5/86400 \approx \num{5.8e-6}$, soit $\approx
\qty{15}{s}$ par mois. Le quartz gagne d'un facteur $\approx 50$.

\textbf{19.} $1/\num{3.2e15} \approx \num{3e-16}$~: neuf ordres de
grandeur sous le quartz. Le GPS chronomètre les \omterm{def:g10:signals-and-waves:signal}{signaux} à la
nanoseconde --- \qty{30}{cm} de parcours de lumière chacun --- donc
seules les horloges atomiques conviennent.

\textbf{20.} Restaurée, l'horloge murale tient $\pm\qty{15}{s}$ par
mois contre $\pm\qty{0.3}{s}$ pour le quartz~: la garder pour le
carillon, faire confiance au quartz pour le train --- et la seconde
elle-même appartient désormais à l'\omterm{def:g11:fundamental-interactions:ladder}{atome} de césium.
\end{solution}
