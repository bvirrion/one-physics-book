\section*{Chapitre \ref{ch:g12:newtons-laws} --- Les lois de Newton}

\begin{solution}{exo:g12:newtons-laws:1}
(a) $p = 70 \times 10 = \qty{7.0e2}{kg.m/s}$.
(b) $v = \qty{13.9}{m/s}$~: $p \approx \qty{1.8e4}{kg.m/s}$.
(c) $p = 0.0080 \times 800 = \qty{6.4}{kg.m/s}$. Voiture $>$ sprinteur $>$
balle~: cinq ordres de grandeur de masse valent plus que la vitesse.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:2}
$N = m(g + a_z)$~: (a) $70 \times 11.01 = \qty{771}{N}$~;
(b) $a_z = 0$, donc $N = mg = \qty{687}{N}$ --- la première loi~: vitesse
constante, les \omterm{def:g10:inertia:force}{forces} \omterm{def:g10:inertia:compensate}{se compensent}~; (c) $70 \times 8.61 = \qty{603}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:3}
$a = g\sin 30^\circ = \qty{4.9}{m/s^2}$~;
$v = \sqrt{2 \times 4.905 \times 3.0} = \qty{5.4}{m/s}$. La masse
s'annule entre $ma$ et $mg\sin\alpha$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:4}
Inversion~: $\Delta p = 0.058 \times (25 + 15) = \qty{2.3}{kg.m/s}$~;
$F = 2.32/0.0050 = \qty{4.6e2}{N}$ --- environ $800$ fois le
\qty{0.57}{N} de \omterm{def:g10:universal-gravitation:weight}{poids}, d'où l'ignorance de la gravité pendant le contact.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:5}
\omterm{def:g10:universal-gravitation:weight}{Poids} (Terre sur le livre, vers le bas)~: partenaire = le livre attire
la Terre, vers le haut. \omterm{def:g12:newtons-laws:friction}{Force normale} (table sur le livre, vers le
haut)~: partenaire = le livre appuie sur la table, vers le bas.
\omterm{def:g10:universal-gravitation:weight}{Poids} et normale agissent sur le \emph{même} corps et appartiennent
à deux interactions différentes --- égaux ici seulement parce que
$\vect a = \vect 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:6}
$N = mg = \qty{490}{N}$. (a) $F > \mu_s N = \qty{221}{N}$.
(b) $a = (220.7 - 0.35 \times 490.5)/50 = \qty{0.98}{m/s^2}$.
(c) $F = \mu_k N = \qty{172}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:7}
(a) Glisse quand $\tan\alpha > \mu_s$~: $\alpha = \arctan 0.45 \approx
24^\circ$. (b) $a = g(\sin 30^\circ - 0.35\cos 30^\circ) =
9.81 \times 0.197 = \qty{1.9}{m/s^2}$. (c) Oui~: une fois en glissement,
le \omterm{def:g10:inertia:inventory}{frottement} tombe à $\mu_k < \mu_s$, donc
$a = g\cos\alpha\,(\tan\alpha - \mu_k) \approx \qty{0.9}{m/s^2} > 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:8}
$a = m_2 g/(m_1 + m_2) = 19.62/5.0 = \qty{3.9}{m/s^2}$~;
$T = m_1 a = \qty{11.8}{N}$. Le bloc suspendu accélère vers le bas,
donc la \omterm{def:g11:forces-and-motion:netforce}{force résultante} sur lui pointe vers le bas~: $T$ doit
sous-coter $m_2 g$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:9}
$v_{\max} = \sqrt{\mu_s g R}$~: sec
$\sqrt{0.70 \times 9.81 \times 50} = \qty{18.5}{m/s} \approx
\qty{67}{km/h}$~; humide $\qty{14.0}{m/s} = \qty{50}{km/h}$. $mv^2/R$
et $\mu_s mg$ portent tous deux $m$~: il s'annule.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:10}
Vertical~: $T\cos\theta = mg$, donc
$T = 2.943/\cos 25^\circ = \qty{3.2}{N}$. Rayon $R = L\sin\theta =
\qty{0.507}{m}$~; radial~: $T\sin\theta = mv^2/R$, donc
$v = \sqrt{gR\tan\theta} = \sqrt{9.81 \times 0.507 \times 0.466}
= \qty{1.5}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:11}
$a_z = (N - mg)/m$ avec $mg = \qty{49.05}{N}$~: $+\qty{1.0}{m/s^2}$,
$0$, $-\qty{1.0}{m/s^2}$. Ex.~: un trajet vers le haut~: accélérer,
croisière, freiner. Oui~: en entrant déjà en descente, les mêmes
lectures signifient freiner, croisière, réaccélérer vers le bas ---
la balance lit l'accélération, non la vitesse.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:12}
(a) $0 = 60V + 4.0 \times 8.0$~: $V = \qty{-0.53}{m/s}$.
(b) Balle à $8.0 + V$ par rapport au sol~: $0 = 60V + 4.0(8.0 + V)$,
donc $V = -32/64 = \qty{-0.50}{m/s}$ et la balle vole à \qty{7.5}{m/s}.
(c) La poussée de la balle sur ses mains --- le partenaire de la
troisième loi de son lancer.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:13}
Sans frottement~: $N\cos\theta = mg$ (vertical) et $N\sin\theta =
mv^2/R$ (radial)~; en divisant, $\tan\theta = v^2/(gR) = 400/784.8$~:
$\theta \approx 27^\circ$. Plus lent~: la voiture tend à glisser vers
le bas du dévers, le \omterm{def:g10:inertia:inventory}{frottement} doit pointer vers le haut~; plus
rapide~: l'inverse.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:14}
(a) $250 \times 2500 = \qty{6.3e5}{kg.m/s}$ chaque seconde~: \omterm{def:g10:inertia:inventory}{poussée}
$\qty{6.3e5}{N}$. (b) $P = \qty{4.9e5}{N}$~;
$a = (6.25 - 4.905) \times 10^5 / \num{5.0e4} = \qty{2.7}{m/s^2}$.
(c) Elle pousse sur ses propres gaz d'échappement (troisième loi), non
sur l'air.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:15}
(a) $m_2 g > \mu_s m_1 g$~: $m_2 > 0.35 \times 4.0 = \qty{1.4}{kg}$.
(b) $a = (m_2 - \mu_k m_1)\,g/(m_1 + m_2) = 1.0 \times 9.81/6.0 =
\qty{1.6}{m/s^2}$~; $T = m_2(g - a) = \qty{16.4}{N}$.
(c) $\mu_k = 0$ donne $a = m_2 g/(m_1 + m_2)$~: la formule de
l'\cref{exo:g12:newtons-laws:8}.
\end{solution}

\begin{solution}{pb:g12:newtons-laws:1}
\textbf{1.} Système~: cabine $+$ chariot ($+$ passagers). La montagne
est fixée au sol, inertielle à excellente précision. \omterm{def:g10:inertia:force}{Forces}~:
\omterm{def:g10:universal-gravitation:weight}{poids} $M\vect g$ (vers le bas), \omterm{def:g12:newtons-laws:friction}{force normale} $\vect N$
du câble porteur (perpendiculaire à lui), \omterm{def:g10:inertia:inventory}{tension} de traction
$\vect T$ (vers le haut de la pente).

\textbf{2.} Au repos, la première loi~: $\sum \vect F = \vect 0$. Le
long de la pente~: $T = Mg\sin\alpha$~; perpendiculairement~:
$N = Mg\cos\alpha$.

\textbf{3.} $T = 4000 \times 9.81 \times 0.500 = \qty{19.6}{kN}$~;
$N = 4000 \times 9.81 \times 0.866 = \qty{34.0}{kN}$.

\textbf{4.} Vide~: $T = \qty{9.8}{kN}$. Chaque \omterm{def:g10:inertia:force}{force} de l'équilibre
est proportionnelle à $M$, donc $T$ aussi.

\textbf{5.} Le long de la pente, $Ma = Mg\sin\alpha$~: $M$ s'annule,
$a = g\sin\alpha = \qty{4.9}{m/s^2}$ vers le bas de la pente.

\textbf{6.} $T' = M(g\sin\alpha + a) = 4000 \times 5.505 =
\qty{22.0}{kN}$.

\textbf{7.} $a/(g\sin\alpha) = 0.60/4.905 \approx 12\%$.

\textbf{8.} $t = v/a = \qty{10}{s}$~; $d = v^2/(2a) = \qty{30}{m}$.

\textbf{9.} Composante verticale de $\vect a$~: $a\sin\alpha =
\qty{0.30}{m/s^2}$, donc $N = m(g + a\sin\alpha) = 80 \times 10.11 =
\qty{809}{N}$ contre $mg = \qty{785}{N}$~: $3\%$ plus lourd.

\textbf{10.} Le \omterm{def:g10:inertia:inventory}{frottement} statique du plancher, horizontal~:
$ma\cos\alpha = 80 \times 0.52 \approx \qty{42}{N}$. Se pencher
place la poussée combinée du plancher à travers son
\omterm{def:g11:forces-and-motion:com}{centre de masse}, de sorte qu'elle ne bascule pas.

\textbf{11.} $a_T = dv/dt = 0$ (vitesse constante)~;
$a_N = v^2/R = 36/40 = \qty{0.90}{m/s^2}$, dirigée vers le centre ---
tout droit vers le bas au sommet.

\textbf{12.} Axe normal (bas positif)~: $Mg - N' = Mv^2/R$, donc
$N' = M(g - v^2/R) = 4000 \times 8.91 = \qty{35.6}{kN}$.

\textbf{13.} $N = m(g - v^2/R) = 80 \times 8.91 = \qty{713}{N}$~:
plus légère de $v^2/(gR) = 0.90/9.81 \approx 9\%$.

\textbf{14.} $N' = 0$ à $v = \sqrt{gR} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$ --- plus du triple de la croisière~: le chariot
chevauche le câble avec une large marge.

\textbf{15.} \omterm{thm:g12:newtons-laws:third}{Troisième loi de Newton}~: \qty{35.6}{kN}, pressant le
câble vers le bas (donc la tour) --- légèrement moins que le
\omterm{def:g10:universal-gravitation:weight}{poids} au repos, allégé par le vol courbe.

\textbf{16.} $F = \Delta p/\Delta t = 4000 \times 6.0/1.5 =
\qty{16}{kN}$, opposée au mouvement ($a = \qty{4.0}{m/s^2}$).

\textbf{17.} $f = ma = 80 \times 4.0 = \qty{320}{N}$~; le plafond est
$\mu_s mg = 0.40 \times 785 = \qty{314}{N} < \qty{320}{N}$~: les
pieds seuls glissent --- tenir la main courante.

\textbf{18.} $0 = M_0 V + m(v_{\text{rel}} + V)$~:
$V = -80 \times 2.5/2080 = \qty{-0.096}{m/s}$~; le passager se déplace
à $2.5 - 0.096 \approx \qty{2.4}{m/s}$ par rapport au sol.

\textbf{19.} La fusée~: sa \omterm{def:g10:inertia:inventory}{poussée} est fixée par la masse éjectée par
seconde fois la vitesse d'éjection par rapport à elle.

\textbf{20.} La plus grande \omterm{def:g10:inertia:inventory}{tension} rencontrée est le démarrage
$T' = M \times \qty{5.505}{N/kg}$ (le repos donne $Mg\sin\alpha$, le
sommet encore moins). Plafond~: $T' \leq \qty{20}{kN}$, donc
$M \leq 20000/5.505 = \qty{3.63e3}{kg}$~: charge utile
\qty{1.63e3}{kg}, c.-à-d.\ $20$ passagers de \qty{80}{kg} (pas $25$).
Plaque~: \emph{20 personnes}~; alors $M = \qty{3.6e3}{kg}$,
$T' = \qty{19.8}{kN}$, facteur de sécurité $100/19.8 \approx 5.1 \geq 5$.
\end{solution}
