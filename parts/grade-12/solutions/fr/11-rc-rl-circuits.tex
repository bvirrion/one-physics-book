\section*{Chapitre \ref{ch:g12:rc-rl-circuits} --- Circuits RC et RL}

\begin{solution}{exo:g12:rc-rl-circuits:1}
$q = Cu = 4.7\times10^{-4} \times 12 \approx \qty{5.6}{mC}$~;
$u = q/C = 2.0\times10^{-3}/10^{-4} = \qty{20}{V}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:2}
\qty{47}{pF}~: accord radio~; \qty{2200}{\micro F}~: lisseur~;
\qty{10}{F}~: supercondensateur. $E = \tfrac12 \times 10 \times 2.7^2
\approx \qty{36}{J}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:3}
$i = C\,\Delta u/\Delta t = 2.2\times10^{-4} \times 5.0/0.010 =
\qty{0.11}{A}$. Ensuite $i = C\,du/dt = 0$~: un
\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} chargé est un circuit ouvert --- pas de
courant permanent.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:4}
$RC = 10^4 \times 10^{-4} = \qty{1.0}{s}$~;
$L/R = 0.10/50 = \qty{2.0}{ms}$. $\unit{\ohm}\,\unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$~;
$\unit{H}/\unit{\ohm} = (\unit{V.s/A})/(\unit{V/A}) = \unit{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:5}
$3.8/6.0 = 0.63$, donc $\tau = \qty{2.0}{s}$~; $95\%$ à $3\tau =
\qty{6.0}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:6}
$\tau = 4700 \times 2.2\times10^{-4} \approx \qty{1.0}{s}$~;
$u(\tau) = 0.63 \times 9.0 = \qty{5.7}{V}$~;
$u(3\tau) = 0.95 \times 9.0 = \qty{8.6}{V}$~;
$I_0 = \mathcal{E}/R = 9.0/4700 \approx \qty{1.9}{mA}$
(\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} vide~: tout $\mathcal{E}$ se trouve aux
bornes de $R$).
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:7}
$du/dt = (\mathcal{E}/RC)e^{-t/RC}$, donc $RC\,du/dt + u =
\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$.
$u(0) = 0$~: le \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} part vide~;
$u \to \mathcal{E}$~: complètement chargé, le courant s'est arrêté.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:8}
$\tau = 4.7\times10^{4} \times 10^{-4} = \qty{4.7}{s}$~;
$u(\tau) = 12\,e^{-1} \approx \qty{4.4}{V}$~; \qty{0.60}{V} est
$5\%$, atteint à $3\tau \approx \qty{14}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:9}
$u = L\,\Delta i/\Delta t = 0.50 \times 2.0/0.040 = \qty{25}{V}$.
À la fermeture, $i$ grimpe au rythme propre du circuit, $di/dt$
modeste~; à l'ouverture $i$ est \emph{\omterm{def:g10:inertia:force}{forcé}} à zéro
presque instantanément, $di/dt$ est énorme et $u$ atteint des valeurs
d'étincelle.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:10}
$di/dt = (\mathcal{E}/L)e^{-Rt/L}$, donc $L\,di/dt + Ri =
\mathcal{E}e^{-Rt/L} + \mathcal{E}(1 - e^{-Rt/L}) = \mathcal{E}$.
$i(\tau) = 0.63 \times 2.0 = \qty{1.3}{A}$~;
$i(3\tau) = 0.95 \times 2.0 = \qty{1.9}{A}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:11}
$u'(t) = (\mathcal{E}/\tau)e^{-t/\tau}$, donc $u'(0) = \mathcal{E}/\tau$~:
la tangente $u = (\mathcal{E}/\tau)\,t$ atteint $\mathcal{E}$ en
$t = \tau$. $C = \tau/R = 0.50/2000 = \qty{250}{\micro F}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:12}
Supercondensateur $\tfrac12 \times 3000 \times 2.7^2 \approx
\qty{1.1e4}{J}$~; flash $\tfrac12 \times 8.0\times10^{-4} \times 300^2
= \qty{36}{J}$~; \omterm{def:g11:magnetic-fields:solenoid}{bobine} $\tfrac12 \times 0.010 \times 10^2
= \qty{0.50}{J}$. Pile ($\qty{1.3e4}{J}$) $>$ supercondensateur $\gg$
flash $\gg$ \omterm{def:g11:magnetic-fields:solenoid}{bobine} --- mais les trois peuvent déverser
leur \omterm{def:g10:energy-conservation:energy}{énergie} en millisecondes~: la
\omterm{def:g11:work-of-force:power}{puissance}, pas l'\omterm{def:g10:energy-conservation:energy}{énergie}.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:13}
$\tau = \qty{5.0}{s}$, donc $R = \tau/C = 5.0/10^{-4} = \qty{50}{k\ohm}$.
$1 - e^{-t/\tau} = 0.50$ donne $e^{-t/\tau} = 0.50$, $t = \tau\ln 2
\approx 0.69\,\tau = \qty{3.5}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:14}
$\Delta u = i\,\Delta t/C = 0.50 \times 0.010/2.2\times10^{-3} \approx
\qty{2.3}{V}$. Il faut $C \geq i\,\Delta t/\Delta u =
5.0\times10^{-3}/0.50 = \qty{10}{mF}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:15}
$u \approx L\,\Delta i/\Delta t = 0.80 \times 2.0/10^{-3} =
\qty{1.6}{kV}$~; $E_L = \tfrac12 \times 0.80 \times 2.0^2 =
\qty{1.6}{J}$. Coupée lentement, $di/dt$ reste petit, $u$ reste à
quelques \omterm{def:g11:circuits-and-power:voltage}{volts}~: pas d'étincelle.
\end{solution}

\begin{solution}{pb:g12:rc-rl-circuits:1}
\textbf{1.} $q = CU_0 = 8.0\times10^{-4} \times 300 = \qty{0.24}{C}$.

\textbf{2.} $N = 0.24/1.60\times10^{-19} = \num{1.5e18}$.

\textbf{3.} $E_C = \tfrac12 \times 8.0\times10^{-4} \times 300^2 =
\qty{36}{J}$.

\textbf{4.} $\num{1.3e4}/36 \approx 360$~fois.

\textbf{5.} La pile stocke beaucoup mais livre lentement~; le
\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} transforme une
\omterm{def:g10:energy-conservation:energy}{énergie} modeste en \emph{\omterm{def:g11:work-of-force:power}{puissance}}
énorme.

\textbf{6.} $\mathcal{E} = Ri + u$ avec $i = C\,du/dt$~:
$RC\,du/dt + u = \mathcal{E}$.

\textbf{7.} $du/dt = (\mathcal{E}/RC)e^{-t/RC}$~:
$\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$
pour tout $t$, et $u(0) = 0$.

\textbf{8.} $\tau = RC = 10^3 \times 8.0\times10^{-4} = \qty{0.80}{s}$.

\textbf{9.} À $3\tau = \qty{2.4}{s}$, $u = 0.95 \times 300 =
\qty{285}{V}$ --- le sifflement de deux secondes.

\textbf{10.} $I_0 = \mathcal{E}/R = \qty{0.30}{A}$~; à mesure que $u$
grimpe, le reste $\mathcal{E} - u$ aux bornes de $R$ se rétrécit, donc
$i = (\mathcal{E}-u)/R$ meurt.

\textbf{11.} $u(t) = U_0\,e^{-t/\tau'}$ avec $\tau' = R_f C =
0.50 \times 8.0\times10^{-4} = \qty{0.40}{ms}$.

\textbf{12.} $I_0 = U_0/R_f = 300/0.50 = \qty{600}{A}$.

\textbf{13.} $P_0 = U_0 I_0 = 300 \times 600 = \qty{180}{kW}$ ---
environ $80$~bouilloires à la fois.

\textbf{14.} Durée $3\tau' = \qty{1.2}{ms}$~; \omterm{def:g11:work-of-force:power}{puissance}
moyenne $\approx 36/1.2\times10^{-3} = \qty{30}{kW}$.

\textbf{15.} $2.4/1.2\times10^{-3} = 2000$. Seule la
\omterm{def:g11:circuits-and-power:resistance}{résistance} a changé (\qty{1.0}{k\ohm} en entrée,
\qty{0.50}{\ohm} en sortie)~: un levier de
\omterm{def:g11:work-of-force:power}{puissance}.

\textbf{16.} $C = 2E_C/U_0^2 = 400/1500^2 \approx \qty{1.8e-4}{F} =
\qty{180}{\micro F}$.

\textbf{17.} $q = CU_0 = 1.78\times10^{-4} \times 1500 \approx
\qty{0.27}{C}$.

\textbf{18.} $\tau = 50 \times 1.78\times10^{-4} \approx \qty{8.9}{ms}$,
durée $3\tau \approx \qty{27}{ms}$ --- quelque vingt fois le flash.

\textbf{19.} $I_0 = 1500/50 = \qty{30}{A}$~;
$P_0 = 1500 \times 30 = \qty{45}{kW}$.

\textbf{20.} $3\tau_c = \qty{6.0}{s}$ donne $\tau_c = \qty{2.0}{s}$,
donc $R_c = \tau_c/C = 2.0/1.78\times10^{-4} \approx \qty{11}{k\ohm}$
et $I_0 = 1500/1.12\times10^{4} \approx \qty{0.13}{A}$ --- prêt en six
secondes, sur un dixième d'\omterm{def:g11:circuits-and-power:current}{ampère}.
\end{solution}
