\section*{Chapter \ref{ch:g12:oscillators-and-time} --- Mechanical Oscillators and the Measurement of Time}

\begin{solution}{exo:g12:oscillators-and-time:1}
(a) $T_0 = 2\pi\sqrt{1.00/9.81} = \qty{2.01}{s}$;
$f_0 = 1/T_0 = \qty{0.50}{Hz}$. (b) $T_0 \propto \sqrt{L}$: the
period doubles.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:2}
Heart: $f = 72/60 = \qty{1.2}{Hz}$, $T = \qty{0.83}{s}$. Wing:
$T = 1/30 = \qty{33}{ms}$. The chest rise is the \emph{amplitude}.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:3}
$T_0 = 2\pi\sqrt{0.10/25} = \qty{0.40}{s}$, $f_0 = \qty{2.5}{Hz}$.
Doubling $m$: $T_0 \times \sqrt{2} \approx 1.4$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:4}
Pseudo-period \qty{2.0}{s}. Ratios $1.5/2.0 = 1.13/1.5 = 0.85/1.13
= 0.75$ each period; next maximum $0.85 \times 0.75 \approx
\qty{0.64}{cm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:5}
Pendulum: $T_0 \propto 1/\sqrt{g}$, so $\times\sqrt{9.81/1.62} =
2.46$ (slower). Spring--mass: unchanged --- no $g$ in
$2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:6}
(a) $[L/g] = \unit{m}/(\unit{m/s^2}) = \unit{s^2}$; its square root
is a time. (b) The kilogram appears only in $m$: nothing else could
cancel it, so $m$ cannot enter. (c) Never: pure numbers like $2\pi$
are invisible to dimensions.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:7}
$\frac{dx}{dt} = -\frac{2\pi}{T_0}A\sin(2\pi t/T_0 + \varphi)$, so
$\frac{d^2x}{dt^2} = -(2\pi/T_0)^2 x(t)$. Equality with $-(k/m)x$
for all $t$ holds exactly when $(2\pi/T_0)^2 = k/m$, i.e.\
$T_0 = 2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:8}
$E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.030^2 =
\qty{1.8e-2}{J}$; $v_{\max} = \sqrt{2E_m/m} = \sqrt{0.18} =
\qty{0.42}{m/s}$ --- at $x = 0$; zero at the extremes $x = \pm A$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:9}
(a) $L = gT_0^2/4\pi^2 = \qty{0.994}{m}$. (b) $\Delta T_0/T_0 =
\tfrac12 \times 1.0\% = 0.5\%$: $86400 \times 0.005 \approx
\qty{4.3e2}{s}$ --- seven minutes slow per day.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:10}
(a) $A \propto \sqrt{E}$: $\sqrt{0.95} \approx 0.975$, a $2.5\%$
drop. (b) $0.95^n = 0.5$: $n = \ln 2/\ln(1/0.95) \approx 14$
periods. (c) No: for light damping it stays within a fraction of a
percent of $T_0$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:11}
(a) At $f_0$ each push arrives in step and adds energy; off the peak,
pushes alternately help and hinder. (b) Marching in step is a
periodic drive; near the bridge's $f_0$ it would climb the resonance
peak. (c) The drum sweeps through the frame's natural frequency
during spin-up: momentary resonance, then past the peak.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:12}
(a) $T_0 \propto 1/\sqrt{g}$: slow, by $\Delta T_0/T_0 = \tfrac12
\times 0.03/9.81 = \num{1.5e-3}$, i.e.\ $\approx \qty{132}{s}$ per
day. (b) $L \propto g$ at fixed $T_0$: shorten by $0.994 \times
0.03/9.81 \approx \qty{3.0}{mm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:13}
$v_{\max} = 2\pi A/T_0 = 2\pi \times 0.040/0.63 = \qty{0.40}{m/s}$;
$a_{\max} = (2\pi/T_0)^2 A = \qty{4.0}{m/s^2}$; $k = m(2\pi/T_0)^2
= \qty{20}{N/m}$; $E_m = \tfrac12 kA^2 = \qty{1.6e-2}{J}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:14}
$T_0 = 40.1/20 = \qty{2.005}{s}$; $g = 4\pi^2 L/T_0^2 = 4\pi^2
\times 0.995/4.020 = \qty{9.77}{m/s^2}$. Uncertainty $\approx 0.2\%
+ 2 \times 0.5\% = 1.2\%$, i.e.\ $\pm\qty{0.12}{m/s^2}$: the
interval contains \qty{9.81}{m/s^2} --- consistent.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:15}
Pendulum $10/86400 \approx \num{1.2e-4}$; quartz $0.5/\num{2.6e6}
\approx \num{1.9e-7}$; cesium $1/\num{3.2e15} \approx \num{3e-16}$.
A \qty{1}{\micro s} error is $\num{1000} \times \qty{30}{cm} =
\qty{300}{m}$ of position: GPS needs the atomic family.
\end{solution}

\begin{solution}{pb:g12:oscillators-and-time:1}
\textbf{1.} The $\approx \qty{0.2}{s}$ reaction error is divided by
$50$: about \qty{4}{ms} on $T_0$.

\textbf{2.} $T = 99.4/50 = \qty{1.988}{s}$.

\textbf{3.} $L = gT^2/4\pi^2 = 9.81 \times 1.988^2/4\pi^2 \approx
\qty{0.982}{m}$.

\textbf{4.} $L = \qty{0.994}{m}$ (the seconds pendulum): lower the
bob, lengthening by $\approx \qty{1.2}{cm}$.

\textbf{5.} Too short, so too quick: fast, by $(2.000 -
1.988)/1.988 \times 86400 \approx \qty{5.2e2}{s} \approx 8.7$
minutes per day.

\textbf{6.} Moon: $\times\sqrt{9.81/1.62} = 2.46$ --- unusable.
Observatory: $\Delta T_0/T_0 = \tfrac12 \times 0.01/9.81 =
\num{5.1e-4}$: loses $\approx \qty{44}{s}$ per day.

\textbf{7.} $\Delta L = L\lambda\,\Delta\theta = 0.994 \times
\num{1.9e-5} \times 5.0 \approx \qty{9.4e-5}{m} \approx
\qty{0.1}{mm}$.

\textbf{8.} $T' = 2\pi\sqrt{L(1 + \lambda\Delta\theta)/g} =
T_0\sqrt{1 + \lambda\Delta\theta} \approx T_0\,(1 + \tfrac12
\lambda\Delta\theta)$.

\textbf{9.} $\Delta T_0/T_0 = \tfrac12 \times \num{1.9e-5} \times
5.0 = \num{4.8e-5}$: $86400 \times \num{4.8e-5} \approx
\qty{4.1}{s}$ per day, slow (longer rod, longer period).

\textbf{10.} $\Delta\theta = \qty{-10}{K}$: $\Delta T_0/T_0 =
\num{-9.5e-5}$, gains $\approx \qty{8.2}{s}$ per day.

\textbf{11.} Steel and brass expand differently; hung in alternating
directions, the brass rods lower the bob while the steel ones raise
it. Sized so the two shifts cancel, the effective length --- and the
rate --- survives the seasons.

\textbf{12.} $280/2.000 = 140$ periods.

\textbf{13.} $\theta_m = 2.00^\circ = \qty{0.0349}{rad}$: $E_m =
\tfrac12 \times 1.2 \times 9.81 \times 0.994 \times 0.0349^2 \approx
\qty{7.1e-3}{J}$.

\textbf{14.} Per period the amplitude shrinks by $2^{-1/140} \approx
0.9951$, the energy by $0.9951^2 \approx 0.990$: about $1.0\%$ of
$E_m$ per period.

\textbf{15.} Push $\approx 0.010 \times \num{7.1e-3} =
\qty{7.1e-5}{J}$; power $\num{7.1e-5}/2.0 \approx \qty{3.5e-5}{W}$
--- tens of microwatts.

\textbf{16.} A week is $604800/2.0 = \num{3.02e5}$ periods, needing
$\num{3.02e5} \times \num{7.1e-5} \approx \qty{21}{J}$; the weight
supplies $Mgh = 2.5 \times 9.81 \times 1.0 = \qty{24.5}{J}$: enough,
at $21/24.5 \approx 88\%$ efficiency.

\textbf{17.} $T = 1/\num{32768} = \qty{30.5}{\micro s}$. $\num{32768}
= 2^{15}$: fifteen divide-by-two stages turn the crystal's signal
into an exact \qty{1}{Hz} tick.

\textbf{18.} Quartz: $0.3/\num{2.6e6} \approx \num{1.2e-7}$.
Pendulum: $0.5/86400 \approx \num{5.8e-6}$, i.e.\ $\approx
\qty{15}{s}$ per month. Quartz wins by a factor $\approx 50$.

\textbf{19.} $1/\num{3.2e15} \approx \num{3e-16}$: nine orders of
magnitude below quartz. GPS times signals to nanoseconds ---
\qty{30}{cm} of light travel each --- so only atomic clocks will do.

\textbf{20.} Restored, the wall clock holds $\pm\qty{15}{s}$ per
month against the quartz's $\pm\qty{0.3}{s}$: keep it for the chime,
trust the quartz for the train --- and the second itself now belongs
to the cesium atom.
\end{solution}
