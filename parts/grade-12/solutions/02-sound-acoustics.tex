\begin{solution}{exo:g12:sound-acoustics:13}

$r \times \sqrt{316} \approx 17.8$: $r \approx 3.0 \times 17.8

\begin{solution}{pb:g12:sound-acoustics:1}
\textbf{1.} $v = 2Lf_1 = 2 \times 0.650 \times 110 = \qty{143}{m/s}$.

\textbf{2.} On the string $\lambda_1 = 2L = \qty{1.30}{m}$; in air
$\lambda = 340/110 \approx \qty{3.1}{m}$. Same $f$, but
$\lambda = v/f$ and the \omterm{def:g10:signals-and-waves:wave}{wave} speeds differ.

\textbf{3.} $L' = v/(2 \times 220) = L/2 = \qty{32.5}{cm}$: the
twelfth fret sits at the midpoint, $L/2$ from the bridge.

\textbf{4.} $L' = 143/(2 \times 165) \approx \qty{43.3}{cm} = 2L/3$.

\textbf{5.} $\log(220/110)/\log 2^{1/12} = 12$ semitones --- twelve
frets to the octave. First fret: $L_1 = L/2^{1/12} = 65.0/1.059
\approx \qty{61.4}{cm}$.

\textbf{6.} \qtylist{110;220;330;440;550}{Hz}.

\textbf{7.} $\qty{440}{Hz} = 4f_1$: the fourth \omterm{def:g12:sound-acoustics:harmonics}{harmonic}, two octaves
up. $\qty{330}{Hz} = 3f_1$: the third \omterm{def:g12:sound-acoustics:harmonics}{harmonic}, an octave plus a
fifth.

\textbf{8.} The pluck position sets the mode mix: near the bridge the
high \omterm{def:g12:sound-acoustics:harmonics}{harmonics} are strong (bright \omterm{def:g12:sound-acoustics:timbre}{spectrum}), over the fingerboard the
\omterm{def:g12:sound-acoustics:harmonics}{fundamental} dominates (mellow). Same $f_1$, so same \omterm{def:g12:sound-acoustics:pitch-loudness}{pitch} ---
different \omterm{def:g12:sound-acoustics:timbre}{spectrum}, different \omterm{def:g12:sound-acoustics:timbre}{timbre}.

\textbf{9.} Only \omterm{def:g12:sound-acoustics:harmonics}{harmonics} with a \omterm{def:g12:sound-acoustics:standing-wave}{node} at the midpoint survive: the
even ones, \qtylist{220;440;660}{Hz}, \dots The \omterm{def:g12:sound-acoustics:pitch-loudness}{pitch} jumps an
octave, to \qty{220}{Hz}.

\textbf{10.} $nf_1 \le \qty{20}{kHz}$: $n \le 20000/110 = 181.8$,
so $181$ \omterm{def:g12:sound-acoustics:harmonics}{harmonics}.

\textbf{11.} $I = P/(4\pi r^2) = \num{6.0e-3}/(4\pi \times 9.0)
\approx \qty{5.3e-5}{W/m^2}$.

\textbf{12.} $L = 10\log_{10}(\num{5.3e7}) \approx \qty{77}{dB}$.

\textbf{13.} $I \times 4$: $+\qty{6}{dB} \to \qty{83}{dB}$. Levels
are logarithms: intensities add, levels do not.

\textbf{14.} $+10\log_{10}12 \approx +\qty{10.8}{dB} \to
\qty{88}{dB}$.

\textbf{15.} Target $I = I_0 \times 10^{8.5} \approx
\qty{3.2e-4}{W/m^2}$ with $12P = \qty{7.2e-2}{W}$:
$r = \sqrt{12P/(4\pi I)} \approx \sqrt{18.1} \approx \qty{4.3}{m}$.

\textbf{16.} \qty{88}{dB} is $+\qty{3}{dB}$ over the eight-hour
limit: the safe time halves once --- four hours. The recital fits.

\textbf{17.} $\lambda = 5000/\num{2.0e6} = \qty{2.5}{mm}$. A \omterm{def:g10:signals-and-waves:wave}{wave}
only reflects off flaws at least about a \omterm{def:g12:mechanical-waves:wavelength}{wavelength} across: audible
\omterm{def:g10:signals-and-waves:sound}{sound} in wood has $\lambda$ of tens of centimetres and diffracts
around any small defect.

\textbf{18.} $t = 2d/v = 2 \times 0.045/5000 = \qty{18}{\micro s}$.

\textbf{19.} $d = vt/2 = 5000 \times \num{7.2e-6}/2 = \qty{18}{mm}$.

\textbf{20.} The card: $v = \qty{143}{m/s}$ on the string; frets at
$\qty{32.5}{cm}$ (twelfth) and $\qty{61.4}{cm}$ (first); ensemble
safe beyond $\approx \qty{4.3}{m}$; flaw \qty{18}{mm} deep --- four
numbers, one chapter.
\end{solution}
