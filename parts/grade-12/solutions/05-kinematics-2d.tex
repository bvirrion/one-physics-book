\section*{Chapter \ref{ch:g12:kinematics-2d} --- Kinematics in Two Dimensions}

\begin{solution}{exo:g12:kinematics-2d:1}
$\vect v = (3.0, 4.0)$, constant; \omterm{def:g12:kinematics-2d:velocity}{speed} $\sqrt{3.0^2 + 4.0^2} =
\qty{5.0}{m/s}$. \omterm{def:g10:relative-motion:trajectory}{Trajectory}: the line $y = \tfrac43 x$. Constant
\omterm{def:g12:kinematics-2d:velocity}{velocity vector}: \omterm{def:g12:kinematics-2d:uniform}{uniform rectilinear motion}.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:2}
Straight line, equal spacings: \omterm{def:g12:kinematics-2d:uniform}{uniform rectilinear};
$v = 0.35/0.10 = \qty{3.5}{m/s}$. Opening spacings would mean the
\omterm{def:g12:kinematics-2d:velocity}{speed} is increasing.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:3}
$v(t) = 4.0\,t$, $a = \qty{4.0}{m/s^2}$: constant acceleration,
uniformly accelerated rectilinear. $v(3.0) = \qty{12}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:4}
$f = 1/8.0 = \qty{0.125}{Hz}$;
$v = 2\pi \times 4.5/8.0 \approx \qty{3.5}{m/s}$;
$a = v^2/R = 3.53^2/4.5 \approx \qty{2.8}{m/s^2}$, pointing toward
the carousel's axis.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:5}
Slope: $a = (10.0 - 2.0)/4.0 = \qty{2.0}{m/s^2}$. Area (trapezoid):
$d = \tfrac12 (2.0 + 10.0) \times 4.0 = \qty{24}{m}$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:6}
$v = \qty{88.9}{m/s}$: $a_n = 88.9^2/6000 \approx \qty{1.3}{m/s^2}
\approx 0.13\,g$. Since $a_n = v^2/R$, high $v$ demands huge $R$ to
keep the sideways acceleration (and the \omterm{def:g10:inertia:force}{force} on track and
passengers) acceptable --- banking changes who supplies it, not its
size.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:7}
$v(t) = -2.4\,t + 8.0$; $a = \qty{-2.4}{m/s^2}$, constant. Reversal
at $v = 0$: $t = 8.0/2.4 \approx \qty{3.3}{s}$, at
$x \approx \qty{16.3}{m}$. Before: forward, slowing; after: backward,
speeding up --- uniformly accelerated throughout.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:8}
$v_1 = 0.28/0.40 = \qty{0.70}{m/s}$;
$v_2 = (0.54 - 0.10)/0.40 = \qty{1.10}{m/s}$;
$v_3 = (0.88 - 0.28)/0.40 = \qty{1.50}{m/s}$. Velocity gains
\qty{0.40}{m/s} every \qty{0.20}{s}: $a = \qty{2.0}{m/s^2}$, constant
--- uniformly accelerated rectilinear.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:9}
$T = \qty{2.36e6}{s}$: $v = 2\pi \times \num{3.84e8}/\num{2.36e6}
\approx \qty{1.0e3}{m/s}$;
$a_n = v^2/R \approx \qty{2.7e-3}{m/s^2}$. Ratio:
$g/a_n \approx 3600 = 60^2$ --- gravity, diluted as the inverse
square of distance (\cref{ch:g10:universal-gravitation}), exactly
accounts for the Moon's turning.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:10}
$f = 1200/60 = \qty{20}{Hz}$, $T = \qty{0.050}{s}$;
$v = 2\pi R f = 2\pi \times 0.25 \times 20 \approx \qty{31}{m/s}$;
$a_n = v^2/R \approx \qty{3.9e3}{m/s^2} \approx 400\,g$. The drum
wall supplies the laundry's $a_n$; through the holes nothing pushes
the water inward, so it flies off along the tangent.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:11}
$v_0 = \qty{25}{m/s}$: $t_s = 25/7.5 \approx \qty{3.3}{s}$;
$d = v_0^2/(2a) = 625/15 \approx \qty{42}{m}$. Graph: a straight
segment from \qty{25}{m/s} down to $0$ at $t_s$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:12}
$\vect v = (-R\omega\sin\omega t,\, R\omega\cos\omega t)$, of
constant norm $R\omega$; its dot product with $\vect{OM}$ vanishes,
so it is perpendicular to the radius, hence tangent. Differentiating
again, $\vect a = -\omega^2\vect{OM}$: toward $O$, of norm
$\omega^2 R = (v/R)^2 R = v^2/R$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:13}
$a_t = \qty{2.5}{m/s^2}$ (backward), $a_n = 20^2/80 =
\qty{5.0}{m/s^2}$;
$a = \sqrt{2.5^2 + 5.0^2} \approx \qty{5.6}{m/s^2}$; angle with the
direction of motion: $\theta = 180^\circ - \arctan(5.0/2.5) \approx 117^\circ$
(i.e.\ $63^\circ$ from straight backward). $5.6 < 7.0$: within grip,
with little to spare.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:14}
Launch: $t_1 = 20/1.2 \approx \qty{17}{s}$,
$d_1 = 20^2/(2 \times 1.2) \approx \qty{167}{m}$. Cruise:
\qty{40}{s}, $\qty{800}{m}$. Braking: $t_3 = 20/1.5 \approx
\qty{13}{s}$, $d_3 = 400/3.0 \approx \qty{133}{m}$. Total:
$\qty{1.1e3}{m}$ in \qty{70}{s}; average $1100/70 \approx
\qty{16}{m/s}$ (\qty{57}{km/h}). $v(t)$ is a trapezoid whose area,
$\tfrac12(70 + 40) \times 20 = \qty{1100}{m}$, checks the total.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:15}
$v = 2\pi \times \num{4.22e7}/\num{86164} \approx \qty{3.1e3}{m/s}$;
$a_n = v^2/R \approx \qty{0.22}{m/s^2}$. Earth's gravity at that
distance supplies it; the \omterm{def:g12:kinematics-2d:ucm}{period} equals Earth's rotation \omterm{def:g12:kinematics-2d:ucm}{period}, so
\omterm{def:g10:universal-gravitation:satellite}{satellite} and ground turn together and it hangs over one equatorial
point (\cref{ch:g12:satellites-kepler}).
\end{solution}

\begin{solution}{pb:g12:kinematics-2d:1}
\textbf{1.} Motion is relative (\cref{ch:g10:relative-motion}): every
position and velocity must name its frame. Here: the ground (road)
frame, $x$ along the road from the first frame's position.

\textbf{2.} Each interval: $9.0/0.50 = \qty{18.0}{m/s}$ --- four
equal values.

\textbf{3.} Straight road, equal displacements in equal times:
\omterm{def:g12:kinematics-2d:uniform}{uniform rectilinear motion}.

\textbf{4.} $v_0 = \qty{18.0}{m/s} = \qty{64.8}{km/h}$.

\textbf{5.} $\vect a = \vect 0$.

\textbf{6.} $v(t) = -a\,t + v_B$; $x(t) = -\tfrac12 a t^2 + v_B\,t$.

\textbf{7.} $v(t_s) = 0$ gives $t_s = v_B/a$; then
$d_b = -\tfrac12 a (v_B/a)^2 + v_B^2/a = v_B^2/(2a)$.

\textbf{8.} $v_B = \sqrt{2 a d_b} = \sqrt{2 \times 6.0 \times 27} =
\sqrt{324} = \qty{18.0}{m/s}$ --- exactly the filmed \omterm{def:g12:kinematics-2d:velocity}{speed}: the
driver never slowed before braking.

\textbf{9.} Reaction: $18.0 \times 1.0 = \qty{18}{m}$; total
$18 + 27 = \qty{45}{m}$.

\textbf{10.} $d_b = v_B^2/(2a) \propto v_B^2$: doubling $v_B$
multiplies $d_b$ by $4$. At \qty{50}{km/h} ($\qty{13.9}{m/s}$):
$d_b = 13.9^2/12.0 \approx \qty{16}{m}$.

\textbf{11.} The \omterm{def:g12:kinematics-2d:velocity}{speed} is constant but the velocity's direction turns:
$\vect v$ is not constant, so the car is accelerating.

\textbf{12.} Quarter turn: $2\pi R/4 = \qty{19.6}{m}$;
$v = 19.6/2.5 \approx \qty{7.9}{m/s}$ (\qty{28}{km/h}).

\textbf{13.} $a_n = 7.85^2/12.5 \approx \qty{4.9}{m/s^2}$, toward the
roundabout's center.

\textbf{14.} $a_{\max} = \mu g = 0.70 \times 9.81 \approx
\qty{6.9}{m/s^2}$; $v_{\max} = \sqrt{\mu g R} =
\sqrt{6.87 \times 12.5} \approx \qty{9.3}{m/s}$ (\qty{33}{km/h}).
$7.9 < 9.3$: within grip --- consistent with the clean, skid-free
lane.

\textbf{15.} $T = 2\pi R/v = 78.5/7.85 \approx \qty{10.0}{s};
f = \qty{0.10}{Hz}$.

\textbf{16.} $64.8 - 50 \approx \qty{15}{km/h}$ over --- about
$30\%$ above the limit.

\textbf{17.} Braking acts over $41 - 18 = \qty{23}{m}$:
$v^2 = v_B^2 - 2ad = 324 - 2 \times 6.0 \times 23 = 48$, so
$v \approx \qty{6.9}{m/s} \approx \qty{25}{km/h}$ at impact.

\textbf{18.} At \qty{13.9}{m/s}: $13.9 \times 1.0 + 13.9^2/12.0 =
13.9 + 16.1 = \qty{30.0}{m} < \qty{41}{m}$ --- the car stops
\qty{11}{m} short of the van.

\textbf{19.} $d(v) = v\,t_r + v^2/(2a)$: once linearly (the reaction
covers $v t_r$ before the brakes act), once squared (the kinetic
part $v^2/(2a)$) --- which is why moderate excesses cost
disproportionately.

\textbf{20.} The dashcam and the skid marks agree: \qty{65}{km/h} in
a \qty{50}{km/h} zone; at the legal limit the car would have stopped
\qty{11}{m} short of the van --- the accident is the extra
\qty{15}{km/h}.
\end{solution}
