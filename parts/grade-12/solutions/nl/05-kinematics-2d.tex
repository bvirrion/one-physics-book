\section*{Hoofdstuk \ref{ch:g12:kinematics-2d} --- Kinematica in twee dimensies}

\begin{solution}{exo:g12:kinematics-2d:1}
$\vect v = (3.0, 4.0)$, constant; \omterm{def:g12:kinematics-2d:velocity}{snelheid}
$\sqrt{3.0^2 + 4.0^2} = \qty{5.0}{m/s}$.
\omterm{def:g10:relative-motion:trajectory}{Baan}: de lijn $y = \tfrac43 x$.
Constante \omterm{def:g12:kinematics-2d:velocity}{snelheidsvector}:
\omterm{def:g12:kinematics-2d:uniform}{eenparig rechtlijnige beweging}.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:2}
Rechte lijn, gelijke afstanden: \omterm{def:g12:kinematics-2d:uniform}{eenparig
rechtlijnig}; $v = 0.35/0.10 = \qty{3.5}{m/s}$. Openende afstanden
zouden betekenen dat de \omterm{def:g12:kinematics-2d:velocity}{snelheid}
toeneemt.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:3}
$v(t) = 4.0\,t$, $a = \qty{4.0}{m/s^2}$: constante versnelling,
eenparig versneld rechtlijnig. $v(3.0) = \qty{12}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:4}
$f = 1/8.0 = \qty{0.125}{Hz}$;
$v = 2\pi \times 4.5/8.0 \approx \qty{3.5}{m/s}$;
$a = v^2/R = 3.53^2/4.5 \approx \qty{2.8}{m/s^2}$, wijzend naar de
as van de carrousel.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:5}
Helling: $a = (10.0 - 2.0)/4.0 = \qty{2.0}{m/s^2}$. Oppervlakte
(trapezium): $d = \tfrac12 (2.0 + 10.0) \times 4.0 = \qty{24}{m}$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:6}
$v = \qty{88.9}{m/s}$: $a_n = 88.9^2/6000 \approx \qty{1.3}{m/s^2}
\approx 0.13\,g$. Omdat $a_n = v^2/R$ eist hoge $v$ een enorme $R$ om
de zijwaartse versnelling (en de \omterm{def:g10:inertia:force}{kracht} op
spoor en passagiers) aanvaardbaar te houden --- overhelling verandert
wie hem levert, niet zijn grootte.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:7}
$v(t) = -2.4\,t + 8.0$; $a = \qty{-2.4}{m/s^2}$, constant. Omkering
bij $v = 0$: $t = 8.0/2.4 \approx \qty{3.3}{s}$, bij
$x \approx \qty{16.3}{m}$. Erover: voorwaarts, trager wordend; erna:
achterwaarts, sneller wordend --- doorlopend eenparig versneld.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:8}
$v_1 = 0.28/0.40 = \qty{0.70}{m/s}$;
$v_2 = (0.54 - 0.10)/0.40 = \qty{1.10}{m/s}$;
$v_3 = (0.88 - 0.28)/0.40 = \qty{1.50}{m/s}$. Snelheid wint
\qty{0.40}{m/s} elke \qty{0.20}{s}: $a = \qty{2.0}{m/s^2}$, constant
--- eenparig versneld rechtlijnig.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:9}
$T = \qty{2.36e6}{s}$: $v = 2\pi \times \num{3.84e8}/\num{2.36e6}
\approx \qty{1.0e3}{m/s}$;
$a_n = v^2/R \approx \qty{2.7e-3}{m/s^2}$. Verhouding:
$g/a_n \approx 3600 = 60^2$ --- zwaartekracht, verdund als het
omgekeerde kwadraat van de afstand
(\cref{ch:g10:universal-gravitation}), verklaart precies het draaien
van de Maan.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:10}
$f = 1200/60 = \qty{20}{Hz}$, $T = \qty{0.050}{s}$;
$v = 2\pi R f = 2\pi \times 0.25 \times 20 \approx \qty{31}{m/s}$;
$a_n = v^2/R \approx \qty{3.9e3}{m/s^2} \approx 400\,g$. De
trommelwand levert de $a_n$ van het wasgoed; door de gaten duwt
niets het water naar binnen, dus vliegt het af langs de raaklijn.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:11}
$v_0 = \qty{25}{m/s}$: $t_s = 25/7.5 \approx \qty{3.3}{s}$;
$d = v_0^2/(2a) = 625/15 \approx \qty{42}{m}$. Grafiek: een recht
segment van \qty{25}{m/s} omlaag naar $0$ bij $t_s$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:12}
$\vect v = (-R\omega\sin\omega t,\, R\omega\cos\omega t)$, van
constante norm $R\omega$; zijn inwendig product met $\vect{OM}$
verdwijnt, dus loodrecht op de straal, dus raaklijn. Nogmaals
differentiëren, $\vect a = -\omega^2\vect{OM}$: naar $O$, van norm
$\omega^2 R = (v/R)^2 R = v^2/R$.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:13}
$a_t = \qty{2.5}{m/s^2}$ (achterwaarts), $a_n = 20^2/80 =
\qty{5.0}{m/s^2}$;
$a = \sqrt{2.5^2 + 5.0^2} \approx \qty{5.6}{m/s^2}$; hoek met de
bewegingsrichting: $\theta = 180^\circ - \arctan(5.0/2.5) \approx 117^\circ$
(d.w.z.\ $63^\circ$ vanaf recht achterwaarts). $5.6 < 7.0$: binnen
grip, met weinig speling.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:14}
Start: $t_1 = 20/1.2 \approx \qty{17}{s}$,
$d_1 = 20^2/(2 \times 1.2) \approx \qty{167}{m}$. Cruise:
\qty{40}{s}, $\qty{800}{m}$. Remmen: $t_3 = 20/1.5 \approx
\qty{13}{s}$, $d_3 = 400/3.0 \approx \qty{133}{m}$. Totaal:
$\qty{1.1e3}{m}$ in \qty{70}{s}; gemiddelde $1100/70 \approx
\qty{16}{m/s}$ (\qty{57}{km/h}). $v(t)$ is een trapezium waarvan de
oppervlakte, $\tfrac12(70 + 40) \times 20 = \qty{1100}{m}$, het
totaal controleert.
\end{solution}

\begin{solution}{exo:g12:kinematics-2d:15}
$v = 2\pi \times \num{4.22e7}/\num{86164} \approx \qty{3.1e3}{m/s}$;
$a_n = v^2/R \approx \qty{0.22}{m/s^2}$. De zwaartekracht van de
Aarde op die afstand levert hem; de
\omterm{def:g12:kinematics-2d:ucm}{periode} is gelijk aan de
rotatie\omterm{def:g12:kinematics-2d:ucm}{periode} van de Aarde, dus
draaien \omterm{def:g10:universal-gravitation:satellite}{satelliet} en
grond samen en hangt hij boven één equatoriaal punt
(\cref{ch:g12:satellites-kepler}).
\end{solution}

\begin{solution}{pb:g12:kinematics-2d:1}
\textbf{1.} Beweging is relatief (\cref{ch:g10:relative-motion}):
elke positie en snelheid moet zijn stelsel noemen. Hier: het
grond(weg)stelsel, $x$ langs de weg vanaf de positie van het eerste
frame.

\textbf{2.} Elk interval: $9.0/0.50 = \qty{18.0}{m/s}$ --- vier
gelijke waarden.

\textbf{3.} Rechte weg, gelijke verplaatsingen in gelijke tijden:
\omterm{def:g12:kinematics-2d:uniform}{eenparig rechtlijnige beweging}.

\textbf{4.} $v_0 = \qty{18.0}{m/s} = \qty{64.8}{km/h}$.

\textbf{5.} $\vect a = \vect 0$.

\textbf{6.} $v(t) = -a\,t + v_B$; $x(t) = -\tfrac12 a t^2 + v_B\,t$.

\textbf{7.} $v(t_s) = 0$ geeft $t_s = v_B/a$; dan
$d_b = -\tfrac12 a (v_B/a)^2 + v_B^2/a = v_B^2/(2a)$.

\textbf{8.} $v_B = \sqrt{2 a d_b} = \sqrt{2 \times 6.0 \times 27} =
\sqrt{324} = \qty{18.0}{m/s}$ --- precies de gefilmde
\omterm{def:g12:kinematics-2d:velocity}{snelheid}: de chauffeur remde niet
vóór het remmen.

\textbf{9.} Reactie: $18.0 \times 1.0 = \qty{18}{m}$; totaal
$18 + 27 = \qty{45}{m}$.

\textbf{10.} $d_b = v_B^2/(2a) \propto v_B^2$: $v_B$ verdubbelen
vermenigvuldigt $d_b$ met $4$. Bij \qty{50}{km/h}
($\qty{13.9}{m/s}$): $d_b = 13.9^2/12.0 \approx \qty{16}{m}$.

\textbf{11.} De \omterm{def:g12:kinematics-2d:velocity}{snelheid} is constant maar de
richting van de snelheidsvector draait: $\vect v$ is niet constant,
dus versnelt de auto.

\textbf{12.} Kwart slag: $2\pi R/4 = \qty{19.6}{m}$;
$v = 19.6/2.5 \approx \qty{7.9}{m/s}$ (\qty{28}{km/h}).

\textbf{13.} $a_n = 7.85^2/12.5 \approx \qty{4.9}{m/s^2}$, naar het
centrum van de rotonde.

\textbf{14.} $a_{\max} = \mu g = 0.70 \times 9.81 \approx
\qty{6.9}{m/s^2}$; $v_{\max} = \sqrt{\mu g R} =
\sqrt{6.87 \times 12.5} \approx \qty{9.3}{m/s}$ (\qty{33}{km/h}).
$7.9 < 9.3$: binnen grip --- consistent met de schone,
slipvrije strook.

\textbf{15.} $T = 2\pi R/v = 78.5/7.85 \approx \qty{10.0}{s};
f = \qty{0.10}{Hz}$.

\textbf{16.} $64.8 - 50 \approx \qty{15}{km/h}$ te veel --- ongeveer
$30\%$ boven de limiet.

\textbf{17.} Remmen werkt over $41 - 18 = \qty{23}{m}$:
$v^2 = v_B^2 - 2ad = 324 - 2 \times 6.0 \times 23 = 48$, dus
$v \approx \qty{6.9}{m/s} \approx \qty{25}{km/h}$ bij impact.

\textbf{18.} Bij \qty{13.9}{m/s}: $13.9 \times 1.0 + 13.9^2/12.0 =
13.9 + 16.1 = \qty{30.0}{m} < \qty{41}{m}$ --- de auto stopt
\qty{11}{m} voor de bestelwagen.

\textbf{19.} $d(v) = v\,t_r + v^2/(2a)$: eens lineair (de reactie
bestrijkt $v t_r$ vóór de remmen werken), eens gekwadrateerd (het
kinetische deel $v^2/(2a)$) --- waarom matige overschrijdingen
disproportioneel kosten.

\textbf{20.} De dashcam en de slipsporen stemmen overeen:
\qty{65}{km/h} in een \qty{50}{km/h}-zone; bij de legale limiet was
de auto \qty{11}{m} voor de bestelwagen gestopt --- het ongeval is
de extra \qty{15}{km/h}.
\end{solution}
