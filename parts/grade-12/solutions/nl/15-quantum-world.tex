\begin{solution}{exo:g12:quantum-world:2}

\begin{solution}{exo:g12:quantum-world:7}
Intensiteit verdubbeld: stroom verdubbelt (tweemaal de
\omterm{def:g12:quantum-world:photon}{fotonen}), $U_s$ onveranderd (zelfde
\omterm{def:g10:energy-conservation:energy}{energie} per \omterm{def:g12:quantum-world:photon}{foton}).
\omterm{def:g12:oscillators-and-time:oscillator}{Frequentie} omhoog: $U_s$ stijgt,
$E_{k,\max} = h\nu - W_0$. Het \omterm{def:g10:signals-and-waves:wave}{golf}beeld koppelt de
\omterm{def:g10:energy-conservation:energy}{energie} van de elektronen aan de intensiteit, niet
aan de \omterm{def:g12:oscillators-and-time:oscillator}{frequentie} --- precies wat \emph{niet}
waargenomen wordt.
\end{solution}

\begin{solution}{exo:g12:quantum-world:9}

\end{solution}

\begin{solution}{exo:g12:quantum-world:11}

\begin{solution}{exo:g12:quantum-world:12}

\end{solution}

\begin{solution}{pb:g12:quantum-world:1}
\textbf{1.} $E = hc/\lambda = \num{1.99e-25}/\num{5.50e-7} =
\qty{3.62e-19}{J} = \qty{2.26}{eV}$.

\textbf{2.} $N = \num{1.0e3}/\num{3.62e-19} \approx \num{2.8e21}$
\omterm{def:g12:quantum-world:photon}{fotonen} per seconde.

\textbf{3.} $P = 0.20 \times \qty{1.0e3}{W} = \qty{200}{W}$;
$E = 200 \times 5.0 = \qty{1.0}{kW.h}$ ($\qty{3.6e6}{J}$).

\textbf{4.} Met $10^{21}$ korrels per seconde is de statistische
flikker fantastisch klein: korreligheid middelelt tot een perfect
gladde stroom, zoals snel geschud zand als water stroomt.

\textbf{5.} Schade eist één korrel van \qtyrange{3}{4}{eV}:
\omterm{def:g10:light-spectra:wavelength}{ultraviolette} korrels kwalificeren,
\omterm{def:g10:light-spectra:wavelength}{infrarode} ($\sim \qty{0.1}{eV}$) nooit ---
aantallen kunnen geen stoot per korrel vervangen.

\textbf{6.} $U_s$ is de omgekeerde \omterm{def:g11:circuits-and-power:voltage}{spanning} die
juist de snelste elektronen stopt: $eU_s = E_{k,\max} = h\nu - W_0$.

\textbf{7.} $\nu = c/\lambda$: $8.22$, $7.41$, $6.88$ en $5.49$
$\times\,10^{14}\,\unit{Hz}$.

\textbf{8.} $eU_s = h\nu - W_0$ is affien in $\nu$ met helling $h$.
Opeenvolgende hellingen: $(0.94 - 0.37)/1.39 = 0.41$ en
$(1.50 - 0.94)/1.34 = 0.42$ \unit{eV} per $10^{14}\,\unit{Hz}$ ---
de drie punten liggen op één lijn.

\textbf{9.} $h = e(U_{s,1}-U_{s,4})/(\nu_1-\nu_4) = \num{1.60e-19}
\times 1.13/\num{2.72e14} = \qty{6.64e-34}{J.s}$.

\textbf{10.} Bij \qty{546}{nm}: $W_0 = h\nu - eU_s = \num{3.64e-19} -
\num{0.59e-19} = \qty{3.05e-19}{J} = \qty{1.90}{eV}$;
$\nu_0 = W_0/h = \qty{4.60e14}{Hz}$; $\lambda_0 \approx \qty{653}{nm}$.

\textbf{11.} Alleen $\lambda < \qty{653}{nm}$ stoot uit: het
violet-tot-rode zichtbare en het
\omterm{def:g10:light-spectra:wavelength}{ultraviolet}. De hele
\omterm{def:g10:light-spectra:wavelength}{infrarode} kant --- ongeveer de helft van de
\omterm{def:g10:energy-conservation:energy}{energie} van de Zon --- is nutteloos voor deze
kathode.

\textbf{12.} $(6.64 - 6.63)/6.63 \approx 0.2\%$ --- een
twee-promille-labmeting van een universele constante.

\textbf{13.} $E_2 = \qty{-3.40}{eV}$, $E_3 = \qty{-1.51}{eV}$,
$E_4 = \qty{-0.85}{eV}$, $E_5 = \qty{-0.54}{eV}$.

\textbf{14.} $3 \to 2$: \qty{1.89}{eV}; $4 \to 2$: \qty{2.55}{eV};
$5 \to 2$: \qty{2.86}{eV}.

\textbf{15.} $\lambda = 1240/E$: \qty{656}{nm} (rood), \qty{486}{nm}
(blauwgroen), \qty{434}{nm} (violet).

\textbf{16.} De energieën van het
\omterm{def:g11:fundamental-interactions:ladder}{atoom} zijn
\omterm{def:g12:quantum-world:levels}{gekwantiseerd}, dus het kan alleen de discrete
verschillen $E_p - E_n$ uitzenden: een ladder maakt lijnen, geen
regenboog.

\textbf{17.} Nee: \qty{500}{nm} betekent \qty{2.48}{eV}, wat bij geen
niveauverschil past ($E_2$ omhoog geeft $1.89$, $2.55$, $2.86$,
\qty{3.40}{eV}; vanuit $E_1$ kost alles minstens \qty{10.2}{eV}) ---
het gas is daar doorzichtig.

\textbf{18.} $eU = \tfrac12 m_e v^2$: $v = \sqrt{2 \times
\num{1.60e-19} \times \num{1.0e4}/\num{9.11e-31}} \approx
\qty{5.9e7}{m/s} \approx 0.20\,c$ --- klassieke behandeling
aanvaardbaar.

\textbf{19.} $p = m_e v = \qty{5.4e-23}{kg.m/s}$;
$\lambda = h/p = \qty{1.2e-11}{m} = \qty{12}{pm}$.

\textbf{20.} Winst $= \num{5.50e-7}/\num{1.23e-11} \approx \num{4.5e4}$.
De twee getallen van het weekend: $h = \qty{6.64e-34}{J.s}$, gemeten
tot $0.2\%$ met \omterm{def:g11:color-light-sources:subtractive}{filters} en een
voltmeter, en een microscoop vijfenveertigduizend maal scherper omdat
elektronen \omterm{def:g10:signals-and-waves:wave}{golven}.
\end{solution}
