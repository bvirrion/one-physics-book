\section*{Hoofdstuk \ref{ch:g12:rlc-oscillations} --- Vrije elektrische oscillaties: RLC}

\begin{solution}{exo:g12:rlc-oscillations:1}
(a) $\sqrt{LC} = \qty{1.0e-4}{s}$: $T_0 = \qty{0.63}{ms}$,
$f_0 \approx \qty{1.6}{kHz}$. (b) $\sqrt{LC} = \qty{3.16e-7}{s}$:
$T_0 = \qty{2.0}{\micro s}$, $f_0 \approx \qty{0.50}{MHz}$ --- een
factor $\sqrt{10^5} \approx 316$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:2}
$LC$: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times \unit{C/A} =
\unit{s} \times \unit{s} = \unit{s^2}$, dus $\sqrt{LC}$ is in
seconden. $2\pi$ is een puur getal --- geen dimensie om te
veranderen.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:3}
$Q_0 = CU_0 = \qty{2.8}{\micro C}$; $E = \tfrac12 CU_0^2 =
\tfrac12 \times \num{4.7e-7} \times 36 \approx \qty{8.5}{\micro J}$.
De lading oscilleert op $f_0 = 1/(2\pi\sqrt{LC})$, stroom een kwart
\omterm{def:g12:oscillators-and-time:oscillator}{periode} uit de
\omterm{def:g12:oscillators-and-time:phase}{fase},
\omterm{def:g10:energy-conservation:energy}{energie} slaat over tussen platen en
\omterm{def:g11:magnetic-fields:solenoid}{spoel}, totaal constant.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:4}
$Q_0 = \qty{2.0}{\micro C}$; $f_0 = 1/T_0 = \qty{250}{Hz}$;
$q(T_0/2) = \qty{-2.0}{\micro C}$; $i(0) = 0$ (cosinus plat bij
$t=0$); $I_0 = 2\pi \times \num{2.0e-6}/\num{4.0e-3} \approx
\qty{3.1}{mA}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:5}
Alles in de \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} wanneer
$q = \pm Q_0$, $i = 0$ ($t = 0$, $T_0/2$, $T_0$, \dots): de uitersten
van de slinger. Alles in de
\omterm{def:g11:magnetic-fields:solenoid}{spoel} wanneer $q = 0$,
$i = \pm I_0$ ($t = T_0/4$, $3T_0/4$, \dots): het laagste, snelste
punt van de slinger.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:6}
$d^2q/dt^2 = -(2\pi/T_0)^2\,Q_0\cos(2\pi t/T_0) = -(2\pi/T_0)^2 q$,
dus $L\,d^2q/dt^2 + q/C = q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$ voor
alle $t$ precies wanneer $(2\pi/T_0)^2 = 1/(LC)$, d.w.z.\
$T_0 = 2\pi\sqrt{LC}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:7}
$L = \dfrac{1}{4\pi^2 f_0^2 C} = \dfrac{1}{39.5 \times
(\num{1.98e5})^2 \times \num{2.2e-10}} \approx \qty{2.9}{mH}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:8}
$E = \tfrac12 CU_0^2 = \qty{5.0e-5}{J}$; $\tfrac12 LI_0^2 = E$ geeft
$I_0 = \sqrt{2E/L} = \sqrt{\num{1.0e-2}} = \qty{0.10}{A}$, voor het
eerst bereikt bij $T_0/4 = \tfrac14 \times \qty{0.63}{ms} \approx
\qty{0.16}{ms}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:9}
$E_C \propto \cos^2(2\pi t/T_0)$, en $\cos^2$ herhaalt elke
\emph{halve} omwenteling: twee vullingen van de
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} per
\omterm{def:g12:oscillators-and-time:oscillator}{periode}, dus $2f_0$. Bij
$t = T_0/8$ is de \omterm{def:g12:oscillators-and-time:phase}{fase} $\pi/4$:
$\cos^2(\pi/4) = \tfrac12$, dus $E_C = E_L = E/2$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:10}
Verhouding $3.0/4.0 = 0.75$ per
\omterm{def:g12:rlc-oscillations:regimes}{pseudoperiode}: zesde maximum
$4.0 \times 0.75^6 \approx \qty{0.71}{V}$.
\omterm{def:g10:energy-conservation:energy}{Energie} $\propto$
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} kwadraat: $0.75^2 \approx 56\%$
overleeft elke \omterm{def:g12:oscillators-and-time:oscillator}{periode}. De
\omterm{def:g12:rlc-oscillations:regimes}{pseudoperiode} blijft
$\approx T_0 = 2\pi\sqrt{LC}$: lichte
\omterm{def:g12:oscillators-and-time:damping}{demping} krimpt de
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}, nauwelijks het tempo.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:11}
$T_0 = 2\pi\sqrt{0.50 \times \num{2.0e-5}} = 2\pi \times
\num{3.16e-3} \approx \qty{20}{ms}$. Woordenboek: $k = m/(LC) =
0.50/\num{1.0e-5} = \qty{5.0e4}{N/m}$; controle
$2\pi\sqrt{m/k} = 2\pi\sqrt{\num{1.0e-5}}$ --- dezelfde \qty{20}{ms}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:12}
$C = \qty{500}{pF}$: $f_0 = 1/(2\pi\sqrt{\num{1.0e-13}}) \approx
\qty{0.50}{MHz}$; $C = \qty{50}{pF}$: $f_0 \approx \qty{1.6}{MHz}$.
Bij vaste $L$, $f_0 = (2\pi\sqrt{L})^{-1} C^{-1/2} \propto 1/\sqrt{C}$:
tienmaal $C$, slechts $\sqrt{10} \approx 3.2$ in $f_0$ --- zoals
gevonden.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:13}
$0.95^n = \tfrac12$: $n = \ln 2/\lvert\ln 0.95\rvert \approx 13.5$,
dus ongeveer $14$ \omterm{def:g12:oscillators-and-time:oscillator}{perioden}.
\omterm{def:g12:oscillators-and-time:oscillator}{Amplitude} $\propto \sqrt{E}$: omlaag tot
$1/\sqrt{2} \approx 0.71$ van de start. Bij \qty{1.0}{MHz},
$T_0 = \qty{1.0}{\micro s}$: ongeveer \qty{14}{\micro s}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:14}
Verlies per cyclus $0.020 \times \qty{1.0e-9}{J} = \qty{2.0e-11}{J}$,
bij \num{32768} cycli per seconde: $P = \num{32768} \times
\num{2.0e-11} \approx \qty{6.6e-7}{W}$ --- onder een microwatt, jaren
op een knoopcel. In de pas, want alleen een duw in de juiste
\omterm{def:g12:oscillators-and-time:phase}{fase} \emph{voegt}
\omterm{def:g10:energy-conservation:energy}{energie} toe
(\omterm{def:g12:oscillators-and-time:resonance}{resonantie}); uit de pas zou ze de
zwaai remmen.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:15}
$C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
(\num{2.5e7})^2 \times \num{1.0e-6}} \approx \qty{41}{pF}$. De
weerstand moet de lus in het
\omterm{def:g12:rlc-oscillations:regimes}{aperiodieke regime} duwen: de lading
settelt dan zonder voorbij te schieten --- geen oscillatie, geen
rinkelen.
\end{solution}

\begin{solution}{pb:g12:rlc-oscillations:1}
\textbf{1.} Luswet: $u_L + u_C = 0$, met $u_C = q/C$,
$u_L = L\,di/dt$, $i = dq/dt$: $L\,d^2q/dt^2 + q/C = 0$.

\textbf{2.} $d^2q/dt^2 = -(2\pi/T_0)^2 q$, dus de vergelijking leest
$q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$, waar voor alle $t$ precies wanneer
$T_0 = 2\pi\sqrt{LC}$.

\textbf{3.} $LC$: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times
\unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ is een tijd.

\textbf{4.} $LC = \qty{6.6e-14}{s^2}$: $T_0 = 2\pi \times \num{2.57e-7}
\approx \qty{1.6}{\micro s}$, $f_0 \approx \qty{620}{kHz}$ --- binnen
\qtyrange{530}{1700}{kHz}.

\textbf{5.} $C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
\num{1.0e12} \times \num{2.0e-4}} \approx \qty{127}{pF}$.

\textbf{6.} Zelfde formule bij \qty{530}{kHz}: $C \approx \qty{451}{pF}$.

\textbf{7.} Bij \qty{1700}{kHz}: $C \approx \qty{44}{pF}$.

\textbf{8.} $C = \qty{480}{pF}$: $f_0 \approx \qty{514}{kHz}$;
$C = \qty{20}{pF}$: $f_0 \approx \qty{2.5}{MHz}$ --- de hele band,
met marge aan beide kanten.

\textbf{9.} $f_0 = (2\pi\sqrt{L})^{-1}C^{-1/2} \propto 1/\sqrt{C}$;
$f_0$ verdubbelen eist $C$ delen door $4$.

\textbf{10.} Omdat $f_0 \propto 1/\sqrt{C}$, verplaatsen gelijke
stappen in $C$ $f_0$ traag bij grote $C$ en snel bij kleine $C$: een
gegeven afstand tussen zenders neemt dan weinig knopweg --- ze
dringen samen aan het
\omterm{def:g12:oscillators-and-time:oscillator}{hoogfrequente} (lage-$C$)-einde.

\textbf{11.} $Q_0 = CU_0 = \num{1.27e-10} \times 0.020 \approx
\qty{2.5e-12}{C}$; $E = \tfrac12 CU_0^2 \approx \qty{2.5e-14}{J}$.

\textbf{12.} $\tfrac12 LI_0^2 = E$: $I_0 = U_0\sqrt{C/L} =
0.020\sqrt{\num{6.35e-7}} \approx \qty{16}{\micro A}$.

\textbf{13.} $0.96^n = \tfrac12$: $n = \ln 2/\lvert\ln 0.96\rvert
\approx 17$ cycli.

\textbf{14.} Bij \qty{1.00}{MHz}, $T_0 = \qty{1.0}{\micro s}$: de ring
halveert in ongeveer \qty{17}{\micro s}. Een eenzame ring sterft in
microseconden; muziek eist dat de lus continu gevoed wordt.

\textbf{15.} De \omterm{def:g10:signals-and-waves:wave}{golf} duwt miljoenen keren per
seconde; alleen wanneer zijn
\omterm{def:g12:oscillators-and-time:oscillator}{frequentie} $f_0$ matcht komen de
duwen in de pas en bouwen de zwaai ---
\omterm{def:g12:oscillators-and-time:resonance}{resonantie}; off-tune zenders duwen even
vaak tegen als mee. Bij de zender geeft een versterker de
\omterm{def:g11:mechanical-energy:dissipation}{gedissocieerde}
\omterm{def:g10:energy-conservation:energy}{energie} elke cyclus terug: een onderhouden
\omterm{def:g12:oscillators-and-time:oscillator}{oscillator}.

\textbf{16.} $LC = \qty{2.54e-14}{s^2}$: $k = m/(LC) =
0.20/\num{2.54e-14} \approx \qty{7.9e12}{N/m}$.

\textbf{17.} Ongeveer $10^8$ maal de auto-veer: geen tafel-massa--veer
bereikt megahertz --- alledaagse veerconstanten en massa's zijn hopeloos
traag.

\textbf{18.} $m = \dfrac{k}{(2\pi f_0)^2} = \dfrac{100}
{(\num{6.28e6})^2} \approx \qty{2.5e-12}{kg}$ --- enkele nanogram, een
stofdeeltje.

\textbf{19.} Een kwartskristal: een veergewicht, ultra-stug scherfje
zingend op zijn \omterm{def:g12:rlc-oscillations:period}{eigenfrequentie} --- de
tijdmeter van eerder dit jaar.

\textbf{20.} Met $L = \qty{0.20}{mH}$ en \qtyrange{20}{480}{pF} veegt
de lus ongeveer \qtyrange{0.51}{2.5}{MHz}, dekt de band;
$C \approx \qty{127}{pF}$ kiest \qty{1.00}{MHz}; alleen sterft de ring
in $\approx \qty{17}{\micro s}$, maar de eigen
\omterm{def:g10:signals-and-waves:wave}{golf} van de zender, resonerend in de pas,
voedt de lus opnieuw --- het nieuws speelt op de
\omterm{def:g10:energy-conservation:energy}{energie} van de
\omterm{def:g10:signals-and-waves:wave}{golf} zelf.
\end{solution}
