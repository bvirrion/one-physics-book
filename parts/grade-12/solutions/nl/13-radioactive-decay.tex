\section*{Hoofdstuk \ref{ch:g12:radioactive-decay} --- Radioactief verval}

\begin{solution}{exo:g12:radioactive-decay:1}
$T_{1/2} = \ln 2/\lambda = 0.693/\num{1.0e-6} = \qty{6.9e5}{s}
\approx 8.0$ dagen --- de klok van jodium-131.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:2}
$T_{1/2} = 3.8 \times \num{86400} = \qty{3.28e5}{s}$;
$\lambda = 0.693/\num{3.28e5} = \qty{2.1e-6}{s^{-1}}$;
$\mathcal A = \lambda N = \num{2.1e-6} \times \num{1.0e10} \approx
\qty{2.1e4}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:3}
$\dd N/\dd t = N_0(-\lambda)e^{-\lambda t} = -\lambda N$ --- de
vergelijking geldt; $N(0) = N_0 e^0 = N_0$;
$N(T_{1/2})/N_0 = e^{-\lambda \ln 2/\lambda} = e^{-\ln 2} =
\tfrac12$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:4}
$90$ jaar $= 3\,T_{1/2}$: $\mathcal A = \num{4.0e5}/2^3 =
\qty{5.0e4}{Bq}$. $300$ jaar $= 10\,T_{1/2}$:
$\mathcal A = \num{4.0e5}/1024 \approx \qty{3.9e2}{Bq}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:5}
Helling $= (18.85 - 20.00)/10 = -0.115$, dus
$\lambda = \qty{0.115}{h^{-1}} = 0.115/3600 = \qty{3.2e-5}{s^{-1}}$;
$T_{1/2} = 0.693/0.115 = 6.0$ uur --- technetium-99m.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:6}
$N = (1.0/226) \times \num{6.02e23} = \num{2.7e21}$;
$T_{1/2} = 1600 \times \num{3.156e7} = \qty{5.05e10}{s}$, dus
$\lambda = \qty{1.37e-11}{s^{-1}}$;
$\mathcal A = \lambda N \approx \qty{3.7e10}{Bq}$ --- één curie, de
\omterm{def:g11:nucleus-radioactivity:activity}{activiteit} van Marie Curie's gram radium.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:7}
Beide verwachten de helft over: $10$, en \num{1.0e20}. Fluctuaties
$\sim 1/\sqrt{n}$ van de telling: met $10$ vervallen, ongeveer $30\%$
--- $7$ of $13$ zou geen verrassing zijn; met \num{1.0e20}, ongeveer
$10^{-10}$. Alleen het grote monster gehoorzaamt de wet tot meetbare
precisie.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:8}
$m = 140 \times \num{1.2e-4} = \qty{1.7e-2}{g}$;
$N = (0.0168/40) \times \num{6.02e23} = \num{2.5e20}$.
$\lambda = 0.693/(\num{1.25e9} \times \num{3.156e7}) =
\qty{1.76e-17}{s^{-1}}$; $\mathcal A = \lambda N \approx
\qty{4.4e3}{Bq}$ --- ongeveer de helft van het lichaamstotaal (de rest
is vooral koolstof-14).
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:9}
$\lambda = 0.693/6.0 = \qty{0.116}{h^{-1}}$. Bij $15$ uur:
$\mathcal A = 500\,e^{-0.116 \times 15} = 500 \times 0.177 \approx
\qty{88}{MBq}$. Onder \qty{1}{MBq}:
$t = \ln(500)/0.116 \approx \qty{54}{hours}$ --- ongeveer $9$
halfwaardetijden.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:10}
Houtskool: $3.4/13.6 = \tfrac14$, precies $2$ halfwaardetijden:
$t = \num{1.15e4}$ jaar. Bot:
$t = \num{5730} \times \ln(1/0.30)/\ln 2 = \num{5730} \times 1.74
\approx \num{1.0e4}$ jaar.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:11}
$\lambda = 0.693/3.8 = \qty{0.182}{d^{-1}}$;
$t = \ln(100)/\lambda = 4.61/0.182 \approx 25$ dagen (ongeveer $6.6$
halfwaardetijden). Maar de uraniumketen in het omringende graniet en
de bodem kweekt voortdurend radon: een eenmalige ventilatie is binnen
weken ongedaan, dus ventilatie moet permanent zijn.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:12}
$N/N_0 = e^{-\lambda t}$ geeft $\lambda t = \ln(N_0/N)$, en
$\lambda = \ln 2/T_{1/2}$ geeft $t = T_{1/2}\ln(N_0/N)/\ln 2$. Hier
$t = \num{5730} \times \ln(20)/\ln 2 = \num{5730} \times 4.32
\approx \num{2.5e4}$ jaar.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:13}
$N_{\mathrm K} = N_0 e^{-\lambda t}$ en
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, dus
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$. Verhouding $3.0$:
$e^{\lambda t} = 4$, dus $\lambda t = 2\ln 2$ en
$t = 2\,T_{1/2} = \num{2.5e9}$ jaar.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:14}
$\ln \mathcal A = 5.99$, $5.42$, $4.84$, $4.26$, $3.69$: dalingen van
$0.57$--$0.58$ per \qty{60}{s} --- uitgelijnd. Helling:
$\lambda = 2.30/240 = \qty{9.6e-3}{s^{-1}}$;
$T_{1/2} = 0.693/\num{9.6e-3} \approx \qty{72}{s}$.
\end{solution}

\begin{solution}{exo:g12:radioactive-decay:15}
(a) $2^{-10} \approx \num{1e-3}$. (b) $(1/2)^8 = 1/256 \approx
0.4\%$. (c) $1000/8 = 125$ verwacht --- maar de wet zwijgt over
welke: ze voorspelt populaties exact en individuen helemaal niet.
\end{solution}

\begin{solution}{pb:g12:radioactive-decay:1}
\textbf{1.} Wissel met de atmosfeer (eten, ademen) houdt de verhouding
op peil op de atmosferische waarde; de dood stopt de inname en verval
neemt over.

\textbf{2.} $T_{1/2} = \num{5730} \times \num{3.156e7} =
\qty{1.81e11}{s}$, dus $\lambda = 0.693/\num{1.81e11} =
\qty{3.8e-12}{s^{-1}}$.

\textbf{3.} $\dd N/\dd t = -\lambda N_0 e^{-\lambda t} = -\lambda N$,
en $N(0) = N_0$.

\textbf{4.} $\mathcal A_0 = 13.6/60 = \qty{0.227}{Bq}$ per gram;
$N = \mathcal A_0/\lambda = 0.227/\num{3.8e-12} \approx \num{5.9e10}$
\omterm{def:g11:fundamental-interactions:ladder}{atomen} per gram.

\textbf{5.} $(1/12) \times \num{6.02e23} = \num{5.0e22}$
koolstof\omterm{def:g11:fundamental-interactions:ladder}{atomen}; verhouding
$\num{5.9e10}/\num{5.0e22} \approx \num{1.2e-12}$ --- één op $10^{12}$,
zoals beloofd.

\textbf{6.} $\mathcal A = \mathcal A_0 e^{-\lambda t}$ geeft
$\lambda t = \ln(\mathcal A_0/\mathcal A)$, en $\lambda =
\ln 2/T_{1/2}$ maakt er
$t = T_{1/2}\ln(\mathcal A_0/\mathcal A)/\ln 2$ van.

\textbf{7.} $t = \num{5730} \times \ln(13.6/9.1)/\ln 2 =
\num{5730} \times 0.580 \approx \num{3.3e3}$ jaar.

\textbf{8.} De dood van het hout --- het vellen van de boom, niet het
snijden; een standbeeld uit oud timmerhout (of een nep uit oud hout)
dateert voor of na het materiaal.

\textbf{9.} $t = \num{5730} \times \ln(13.6/10.4)/\ln 2 =
\num{5730} \times 0.387 \approx \num{2.2e3}$ jaar --- volledig
consistent met een schrijver van tweeëntwintig eeuwen geleden.

\textbf{10.} $\Delta t \approx \Delta\mathcal A/(\lambda\mathcal A)$
met $\lambda = \ln 2/5730 = \num{1.21e-4}$ per jaar:
$\Delta t = (0.2/10.4)/\num{1.21e-4} \approx \num{1.6e2}$ jaar ---
een datum goed tot op een of twee eeuwen.

\textbf{11.} $t = \num{5730} \times \ln(13.6/13.4)/\ln 2 \approx
120$ jaar --- ononderscheidbaar van modern binnen de
$\pm 160$-jaar-onzekerheid: het perkament is recent, de ``antique''
een nep.

\textbf{12.} $13.6/2^n < 0.2$ eist $2^n > 68$: $n = 7$
halfwaardetijden (want $2^7 = 128$), ongeveer \num{4.0e4} jaar.

\textbf{13.} Tien halfwaardetijden: ongeveer \num{5.7e4} jaar --- de
praktische horizon, zo'n zestigduizend jaar.

\textbf{14.} $\num{6.6e7}/\num{5730} \approx \num{11500}$
halfwaardetijden. De \num{5.9e10}
\omterm{def:g11:fundamental-interactions:ladder}{atomen} van één gram zijn uitgeput na
$2^n > \num{5.9e10}$, d.w.z.\ $n \approx 36$ halfwaardetijden
($\approx \num{2e5}$ jaar): lang voor de dinosauriërs blijft niet één
koolstof-14-\omterm{def:g11:fundamental-interactions:ladder}{atoom} over --- er is niets te
tellen.

\textbf{15.} Een veel tragere klok: kalium-40 of uranium-238, met
halfwaardetijden van miljarden jaren.

\textbf{16.} Gesmolten lava laat argon wegbubbelen, dus het vers
gestolde kristal houdt kalium maar nul argon: de verhouding start op
$0$ bij stolling --- de klok is op nul gezet.

\textbf{17.} Zoals in de oefening: $N_{\mathrm K} = N_0 e^{-\lambda t}$,
$N_{\mathrm{Ar}} = N_0 - N_{\mathrm K}$, dus
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.

\textbf{18.} $e^{\lambda t} = 1.60$:
$t = \num{1.25e9} \times \ln(1.60)/\ln 2 = \num{1.25e9} \times 0.678
\approx \num{8.5e8}$ jaar.

\textbf{19.} Nee, tweemaal: bij \num{8.5e8} jaar heeft koolstof-14
$10^5$ halfwaardetijden gelopen --- stilte; bij \num{3.3e3} jaar heeft
kalium-40 \num{2.6e-6} van een
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd} gelopen --- de
argonverhouding $\approx \lambda t \sim \num{2e-6}$ is onmeetbaar
klein. Elke klok leest alleen van ongeveer een tiende tot ongeveer tien
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijden}.

\textbf{20.} Standbeeld $\approx \num{3.3e3}$ jaar, papyrus
$\approx \num{2.2e3}$ jaar, rots $\approx \num{8.5e8}$ jaar --- drie
klokken, één wet: $N = N_0\,e^{-\lambda t}$, achterstevoren gelezen.
\end{solution}
