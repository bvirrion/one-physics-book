\section*{Hoofdstuk \ref{ch:g12:light-as-wave} --- Licht als golf: diffractie en interferentie}

\begin{solution}{exo:g12:light-as-wave:1}
$f = c/\lambda$: rood $\num{3.00e8}/\num{7.0e-7} \approx
\qty{4.3e14}{Hz}$; violet $\approx \qty{7.5e14}{Hz}$ --- violet is
hoger (kortere \omterm{def:g12:mechanical-waves:wavelength}{golflengte}, zelfde snelheid $c$).
\end{solution}

\begin{solution}{exo:g12:light-as-wave:2}
$\theta = \num{5.32e-7}/\num{2.0e-4} = \qty{2.7e-3}{rad}$;
$L = 2\theta D = 2 \times \num{2.66e-3} \times 1.5 \approx
\qty{8.0}{mm}$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:3}
Stem: $\lambda = 340/425 = \qty{0.80}{m}$, dus $\lambda/a = 1.0$ ---
maximale \omterm{def:g12:light-as-wave:diffraction}{diffractie}, \omterm{def:g10:signals-and-waves:sound}{geluid}
overstroomt de gang. Licht: $\lambda/a \approx \num{5.5e-7}/0.80
\approx \num{7e-7}$ --- verwaarloosbare spreiding: licht blijft
recht, dus de hoek verbergt het.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:4}
Bij $M$: $\delta = \qty{6.0}{cm} = 3\lambda$,
\omterm{prop:g12:light-as-wave:conditions}{constructief} --- luid.
Bij $N$: $\delta = \qty{5.0}{cm} = (2 + \tfrac12)\lambda$,
\omterm{prop:g12:light-as-wave:conditions}{destructief} --- (bijna) stil.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:5}
$i = \lambda D/a = \num{6.33e-7} \times 1.8/\num{2.5e-4} \approx
\qty{4.6}{mm}$. Binnen $\pm\qty{1.0}{cm}$: $|k| \leq 10/4.6 = 2.2$,
dus $k = -2, \dots, 2$: vijf heldere franjes.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:6}
$d = 2\lambda D/L = 2 \times \num{6.5e-7} \times 1.6/0.028 \approx
\qty{74}{\micro m}$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:7}
$i = \qty{4.7}{mm}$, dus $\lambda = i\,a/D = \num{4.7e-3} \times
\num{2.0e-4}/1.5 \approx \qty{6.3e-7}{m} = \qty{630}{nm}$: rood. Tien
\omterm{def:g12:light-as-wave:fringes}{interfranjes} meten deelt de leesfout van
de liniaal door tien.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:8}
De lampen hebben (ruwweg) overeenkomende frequenties maar zijn
\emph{niet in de pas}: elk zendt korte
\omterm{def:g10:signals-and-waves:wave}{golf}treinen met willekeurig springende fase
uit, dus de bronnen zijn incoherent en de momentane franjes
verschuiven miljarden keren per seconde --- het oog ziet het
gemiddelde, uniform. Young belicht beide spleten vanuit \emph{één}
\omterm{def:g10:signals-and-waves:wave}{golf}: de twee kopieën erven elke fasesprong
samen en blijven in de pas.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:9}
Eerste tot zevende $= 6i = \qty{18.0}{mm}$, dus $i = \qty{3.0}{mm}$;
$\lambda = i\,a/D = \num{3.0e-3} \times \num{3.0e-4}/1.4 \approx
\qty{6.4e-7}{m} \approx \qty{640}{nm}$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:10}
$\sin\theta = k\lambda/d = 0.396\,k$: $k = 1$, $\theta = 23^\circ$;
$k = 2$, $\sin\theta = 0.79$, $\theta = 52^\circ$; $k = 3$ zou
$\sin\theta = 1.19 > 1$ nodig hebben: onmogelijk. Hoogste orde
$k = 2$.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:11}
$\theta = \num{5.5e-7}/0.10 = \qty{5.5e-6}{rad}$. De koplampen
onderspannen $1.5/\num{1.0e5} = \qty{1.5e-5}{rad}$, ongeveer
driemaal de vervaging: gescheiden, net.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:12}
Reflecties van voor- en achtervlak
\omterm{prop:g12:light-as-wave:conditions}{interfereren}; in
\omterm{def:g10:light-spectra:white}{wit licht} versterkt elke dikte sommige
\omterm{def:g12:mechanical-waves:wavelength}{golflengten} en dooft andere uit, vandaar
een kleur. De dikte van de film hangt alleen van de hoogte af, dus
vormen punten van gelijke kleur horizontale banden. Leeglopen
\omterm{def:g11:lenses-and-eye:lens}{dunner} de film, dus de dikte die een gegeven
kleur schilderde zit steeds lager: de banden kruipen omlaag.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:13}
$i = \lambda D/a$: violet $\num{4.0e-7} \times 1.5/\num{2.0e-4} =
\qty{3.0}{mm}$; rood $\qty{5.6}{mm}$. Centrum: alle kleuren helder
--- één witte franje. Enkele millimeters eruit zijn de
kleurpatronen uit elkaar gegleden: iriserende franjes. Verder
overlappen maxima van alle kleuren overal: uniforme witte vervaging.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:14}
$\theta = \num{5.5e-7}/2.4 = \qty{2.3e-7}{rad}$; op \qty{250}{km}
is dat $\num{2.3e-7} \times \num{2.5e5} \approx \qty{6}{cm}$. Hij
kan auto's tellen en een persoon spotten, maar krantendruk
(millimeterletters) ligt twintigmaal onder de
\omterm{def:g12:light-as-wave:diffraction}{diffractie}-limiet: de claim is mythe.
\end{solution}

\begin{solution}{exo:g12:light-as-wave:15}
De \omterm{def:g10:signals-and-waves:frequency}{frequentie} wordt door de bron vastgelegd:
onveranderd. De snelheid daalt tot $c/n$, dus $\lambda' = \lambda/n$
en $i' = i/n = 4.0/1.33 \approx \qty{3.0}{mm}$: de franjes
trekken samen.
\end{solution}

\begin{solution}{pb:g12:light-as-wave:1}
\textbf{1.} $\theta = \lambda/a = \num{6.33e-7}/\num{1.00e-4} =
\qty{6.3e-3}{rad}$.

\textbf{2.} Halve breedte $D\tan\theta \approx D\theta = \lambda D/a$,
dus $L = 2\lambda D/a = 2 \times \num{6.33e-7} \times
3.00/\num{1.00e-4} = \qty{3.8}{cm}$.

\textbf{3.} $(3.8 - 3.7)/3.8 \approx 3\%$: binnen de precisie van
een liniaal --- gevalideerd.

\textbf{4.} $L \propto 1/a$: $a$ halveren verdubbelt $L$, dus
$L = \qty{7.6}{cm}$. Smaller obstakel-of-spleet, breder patroon.

\textbf{5.} $\tan(\num{6.33e-3}) = \num{6.33008e-3}$: ze stemmen
overeen tot ongeveer $\num{1e-5}$ relatief --- de benadering is veel
beter dan elke meting hier.

\textbf{6.} Volgens \cref{prop:g12:light-as-wave:hair}: een obstakel
van breedte $d$ diffracteert als een spleet van breedte $d$, dus
$L = 2\lambda D/d$ geldt nog steeds.

\textbf{7.} $d = 2\lambda D/L = 2 \times \num{6.33e-7} \times
3.00/0.054 \approx \qty{70}{\micro m}$.

\textbf{8.} $L = \qty{5.2}{cm} \to d = \qty{73}{\micro m}$;
$L = \qty{5.6}{cm} \to d = \qty{68}{\micro m}$: dus
$d = \qty{70 \pm 3}{\micro m}$.

\textbf{9.} $d = \num{3.80e-6}/0.076 = \qty{50}{\micro m}$: die
van de vriend is fijner --- en $L \propto 1/d$, dus de fijnere haar
gooit het bredere patroon.

\textbf{10.} Nee: een lamp is wit (elke
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} schildert een ander
patroon, en ze vervagen) en incoherent-en-breed (geen enkele schone
bundel om te diffracteren). De
\omterm{def:g11:color-light-sources:lamps}{laser} is
\omterm{def:g10:light-spectra:white}{monochromatisch} en directioneel.

\textbf{11.} Alleen kopieën van één \omterm{def:g10:signals-and-waves:wave}{golf} zijn
\omterm{def:g12:light-as-wave:coherent}{coherent}; twee onafhankelijke bronnen
lopen uit de pas en de franjes wassen uit
(\cref{rem:g12:light-as-wave:lamps}).

\textbf{12.} In het centrum $\delta = 0 = 0 \times \lambda$ voor
elke $\lambda$: \omterm{prop:g12:light-as-wave:conditions}{constructief}
ongeacht de \omterm{def:g12:mechanical-waves:wavelength}{golflengte}.

\textbf{13.} $i = 4.26/10 = \qty{4.3}{mm}$; één span van tien wordt
met dezelfde liniaalfout als één
\omterm{def:g12:light-as-wave:fringes}{interfranje} gelezen, dus de fout op
$i$ wordt door tien gedeeld.

\textbf{14.} $\lambda = i\,a'/D = \num{4.26e-3} \times
\num{2.50e-4}/2.00 = \num{5.33e-7} \approx \qty{533}{nm}$.

\textbf{15.} $\qty{533}{nm}$ ligt binnen $\qty{532 \pm 10}{nm}$:
consistent; die \omterm{def:g12:mechanical-waves:wavelength}{golflengte} is groen.

\textbf{16.} $\tan\theta = 0.43/1.00$, dus $\theta = 23^\circ$. Hier
$\tan\theta = 0.43$ terwijl $\sin\theta = 0.40$: bij $23^\circ$ is
de kleine-hoekidentificatie van $\sin$, $\tan$ en $\theta$ bijna
$10\%$ ernaast --- gebruik de exacte functies.

\textbf{17.} $d = \lambda/\sin\theta = \num{6.33e-7}/0.396 \approx
\qty{1.6}{\micro m}$.

\textbf{18.} $\tan\theta = 1.65$, $\theta = 59^\circ$,
$\sin\theta = 0.855$: $d = \num{6.33e-7}/0.855 \approx
\qty{0.74}{\micro m}$.

\textbf{19.} $1.6/0.74 \approx 2.2$: dvd-sporen zijn tweemaal zo
dicht, en zijn putjes zijn overeenkomstig kleiner (gelezen met een
\omterm{def:g12:mechanical-waves:wavelength}{kortere-golflengte}-
\omterm{def:g11:color-light-sources:lamps}{laser}) --- enkele malen de data
op dezelfde \qty{12}{cm}-schijf.

\textbf{20.} Haar \qty{70 \pm 3}{\micro m}; haar van de vriend
\qty{50}{\micro m}; groene \omterm{def:g11:color-light-sources:lamps}{laser}
\qty{533}{nm}; cd-sporen \qty{1.6}{\micro m}; dvd-sporen
\qty{0.74}{\micro m} --- een bekende
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} maakt van de meetkunde van
elk patroon een lengte, dus licht van een half micrometer is een
liniaal gegradueerd op een half micrometer.
\end{solution}
