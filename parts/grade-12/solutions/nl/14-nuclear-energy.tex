\section*{Hoofdstuk \ref{ch:g12:nuclear-energy} --- Nucleaire energie: splijting, fusie, $E=mc^2$}

\begin{solution}{exo:g12:nuclear-energy:1}
(a) $\qty{1}{eV} = \qty{1.60e-19}{J}$;
$\qty{1}{MeV} = \qty{1.60e-13}{J}$.
(b) $200 \times \num{1.60e-13} = \qty{3.2e-11}{J}$.
(c) $E = \num{1.66054e-27} \times \num{9.00e16} = \qty{1.49e-10}{J}
= \num{1.49e-10}/\num{1.60e-13} \approx \qty{9.3e2}{MeV}$ --- de
\qty{931.5}{MeV} van de cursus, tot op onze afgeronde $c$.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:2}
$E = \num{6.0e-3} \times \num{9.00e16} = \qty{5.4e14}{J}$;
$\num{5.4e14}/\num{3.6e7} = \num{1.5e7}$ dagen $\approx \num{4.1e4}$
jaar --- veertig millennia op één suikerklontje.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:3}

$E_b/A \approx \qty{7.68}{MeV}$ per \omterm{def:g11:fundamental-interactions:ladder}{nucleon}.
\end{solution}

(b) $A = 236 - 97 - 2 = 137$, $Z = 92 - 40 = 52$:

\end{solution}

\begin{solution}{exo:g12:nuclear-energy:5}
Ongeveer $7.1$, $7.7$, $8.8$ en \qty{7.6}{MeV} per
\omterm{def:g11:fundamental-interactions:ladder}{nucleon}; ijzer-56 is het sterkst
gebonden. Uranium kan \omterm{def:g10:energy-conservation:energy}{energie} vrijmaken door
\omterm{def:g12:nuclear-energy:fission}{splijting}, helium-4 en koolstof-12 door
\omterm{def:g12:nuclear-energy:fusion}{fusie} --- beide stappen klimmen naar ijzer; ijzer
zelf heeft nergens heen.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:7}
Voor: $\qty{236.00216}{u}$; na: $143.89220 + 88.89790 +

\end{solution}

\begin{solution}{exo:g12:nuclear-energy:8}
$\Delta m = 2 \times 2.01355 - 3.01493 - 1.00866 = \qty{0.00351}{u}$:
$E \approx \qty{3.27}{MeV}$. D--T maakt helium-4, abnormaal gebonden
(\qty{7.1}{MeV} per \omterm{def:g11:fundamental-interactions:ladder}{nucleon}); helium-3 zit
veel lager op de kromme, dus de stap levert veel minder.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:9}
(a) Het neutron is neutraal: geen
\omterm{def:g12:nuclear-energy:barrier}{Coulombbarrière},
\omterm{def:g12:nuclear-energy:fission}{splijting} vuurt koud. Fusiekernen zijn beide
positief en moeten de barrière beklimmen, dus alleen $\sim
\qty{1e8}{K}$ onrust geeft botsingen heftig genoeg. (b) De kern houdt
een astronomisch aantal protonen; de zeldzame snelste in de menigte
doen de fusie, en de Zon heeft miljarden jaren over --- een trage
brand is precies wat een ster wil. (c) Er is geen neutronenlawine:
elke \omterm{def:g12:nuclear-energy:fusion}{fusie} wordt gekocht door zijn eigen botsing, en
elk verlies van opsluiting koelt het
\omterm{def:g12:nuclear-energy:barrier}{plasma} en dooft het vuur.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:10}
$\Delta m = 4 \times 1.00728 - 4.00151 = \qty{0.02761}{u}$, dus
$E \approx \qty{25.7}{MeV}$; fractie
$0.02761/4.02912 \approx 0.7\%$ van de massa.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:11}
(a) Hij vertraagt \omterm{def:g12:nuclear-energy:fission}{splijtings}neutronen door
botsingen; trage neutronen splijten uranium-235 veel eerder, dus de
keten heeft ze nodig. (b) Ze absorberen neutronen en houden de keten
op één neutron per \omterm{def:g12:nuclear-energy:fission}{splijting} --- ingeduwd, sterft
hij. (c) Geen water, geen moderatie: de neutronen blijven snel, missen,
en de keten stikt --- het ontwerp faalt naar ``uit''.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:12}
\omterm{def:g12:nuclear-energy:fission}{Splijtingen} verdubbelen elke generatie, dus na $n$
generaties ongeveer $2^n$; $2^{n} = \num{2.56e21} \approx 2 \times
(10^{3})^{7} \approx 2^{71}$ geeft $n \approx 71$. Tijd:
$71 \times \qty{10}{ns} \approx \qty{0.7}{\micro s}$ --- een gram
brandstof, drie ton kolen waard, in onder een microseconde:
kritikaliteit is een klif, geen helling.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:13}
$\Delta m = 55.92066 - 2 \times 27.96924 = \qty{-0.01782}{u}$:
$E \approx \qty{-16.6}{MeV}$ --- de splitsing \emph{absorbeert}
\omterm{def:g10:energy-conservation:energy}{energie}. IJzer zit op de top: splijten of
fuseren gaat bergaf in binding, beide kosten
\omterm{def:g10:energy-conservation:energy}{energie}. Het is de as van elk nucleair vuur.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:14}
Fe: $\Delta m = 26 \times 1.00728 + 30 \times 1.00866 - 55.92066 =
\qty{0.52842}{u}$, $E_b \approx \qty{492}{MeV}$, $E_b/A \approx
\qty{8.79}{MeV}$. U: $\Delta m = 92 \times 1.00728 + 143 \times
1.00866 - 234.99350 = \qty{1.91464}{u}$, $E_b \approx \qty{1784}{MeV}$,
$E_b/A \approx \qty{7.59}{MeV}$. $235$
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen} herpakken bij \qty{8.5}{MeV}
in plaats van \qty{7.59}{MeV} wint $235 \times 0.9 \approx
\qty{2e2}{MeV}$ --- de $185$--$200$ van het uitgewerkte voorbeeld.
\end{solution}

\begin{solution}{exo:g12:nuclear-energy:15}
$\num{5.0e8}/\num{2.82e-12} \approx \num{1.8e20}$ reacties per seconde;
massasnelheid $\num{1.8e20} \times 5.029 \times \num{1.66054e-27}
\approx \qty{1.5e-6}{kg/s}$, d.w.z.\ ongeveer \qty{0.13}{kg} per dag ---
tegen \num{1.4e6} \omterm{def:g10:orders-of-magnitude:unit}{kilogram} kolen: tien miljoen op
één.
\end{solution}

\begin{solution}{pb:g12:nuclear-energy:1}
\textbf{1.} $P_{\text{th}} = \num{1.0e9}/0.33 \approx \qty{3.0}{GW}$.

\textbf{2.} $200 \times \num{1.60e-13} = \qty{3.2e-11}{J}$.

\textbf{3.} $\num{3.0e9}/\num{3.2e-11} \approx \num{9.4e19}$
\omterm{def:g12:nuclear-energy:fission}{splijtingen} per seconde.

\textbf{4.} $m_{\mathrm U} = 235 \times \num{1.66054e-27} =
\qty{3.90e-25}{kg}$; $\num{9.4e19} \times \num{3.90e-25} \approx
\qty{3.7e-5}{kg/s}$ --- zo'n veertig microgram per seconde.

\textbf{5.} $\num{3.7e-5} \times \num{3.16e7} \approx \qty{1.2e3}{kg}$:
ongeveer $1.2$ ton uranium-235 per jaar.

\textbf{6.} $\num{1.2e3}/0.040 \approx \qty{2.9e4}{kg}$ --- zo'n $29$
ton brandstof: één zware truck, eenmaal per jaar.

\textbf{7.} $\num{3.0e9} \times \num{3.16e7} \approx \qty{9.5e16}{J}$.

\textbf{8.} $\num{9.5e16}/\num{3.0e7} \approx \qty{3.2e9}{kg}$: $3.2$
miljoen ton kolen.

\textbf{9.} $\num{3.2e9}/\num{1.2e3} \approx \num{2.7e6}$ --- de
$\num{8.2e13}/\num{3.0e7}$ van de ladder, zoals het moet.

\textbf{10.} $\num{3.2e9}/\num{3.0e6} \approx 1060$ treinen per jaar,
ongeveer drie per dag: een kolentrein bij dageraad, middag en schemer
voor altijd, tegen één truck elke lente.

\textbf{11.} $17.6 \times \num{1.60e-13} = \qty{2.82e-12}{J}$.

\textbf{12.} Eén paar weegt $5.029 \times \num{1.66054e-27} =
\qty{8.35e-27}{kg}$: $\num{2.82e-12}/\num{8.35e-27} \approx
\qty{3.4e14}{J/kg}$ --- de \omterm{def:g12:nuclear-energy:fusion}{fusie}trede van de ladder.

\textbf{13.} $\num{9.5e16}/\num{3.4e14} \approx \qty{2.8e2}{kg}$ D--T;
deuteriumaandeel $280 \times 2.014/5.029 \approx \qty{1.1e2}{kg}$.

\textbf{14.} $\num{1.1e2}/\num{3.3e-2} \approx \qty{3.4e3}{m^3}$
zeewater --- ongeveer anderhalve olympisch bad voor het hele stadsjaar.

\textbf{15.} Niets vermenigvuldigt: geen neutron steekt de volgende
\omterm{def:g12:nuclear-energy:fusion}{fusie} aan, en verlies van opsluiting koelt het
\omterm{def:g12:nuclear-energy:barrier}{plasma} en stopt het binnen seconden. Het
moeilijke is het omgekeerde --- \qty{1.5e8}{K} lang genoeg bijeengehouden
om te branden.

\textbf{16.} $\num{3.9e26}/\num{9.00e16} \approx \qty{4.3e9}{kg/s}$:
vier miljoen ton per seconde.

\textbf{17.} $\num{3.0e9}/\num{9.00e16} \approx \qty{3.3e-8}{kg/s}$,
d.w.z.\ $\num{3.3e-8} \times \num{3.16e7} \approx \qty{1.0}{kg}$ per
jaar --- en ja: $\qty{1.2e3}{kg} \times 0.00085 \approx 1$,
$\num{3.2e9} \times \num{3.3e-10} \approx 1$, $280 \times 0.00375
\approx 1$. Elke brandstof levert hetzelfde
\omterm{def:g10:orders-of-magnitude:unit}{kilogram}; ze verschillen alleen in hoeveel
vracht het moet dragen.

\textbf{18.} $\num{4.3e9} \times \num{3.16e7} \times \num{4.5e9}
\approx \qty{6.1e26}{kg}$ --- slechts $\num{6.1e26}/\num{2.0e30}
\approx \num{3e-4}$, drie delen op tienduizend.

\textbf{19.} Omzetbare massa $0.0007 \times \num{2.0e30} =
\qty{1.4e27}{kg}$, waard $\num{1.4e27} \times \num{9.00e16} \approx
\qty{1.3e44}{J}$; bij \qty{3.9e26}{W} duurt dat $\num{1.3e44}/
\num{3.9e26} \approx \qty{3.2e17}{s} \approx \num{1.0e10}$ jaar ---
tien miljard, waarvan zo'n vijf miljard over.

\textbf{20.} Eén stadsjaar: $3.2$ miljoen ton kolen, of
$\num{1.2e3}$ \omterm{def:g10:orders-of-magnitude:unit}{kilogram} uranium-235, of
$\num{2.8e2}$ \omterm{def:g10:orders-of-magnitude:unit}{kilogram} D--T uit anderhalve
bad vol zeewater. Onder elk grootboek dezelfde post: één
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} massa, weg. $E = mc^2$ kan het niet
schelen welke brandstof hem draagt --- alleen hoe groot de truck is.
\end{solution}
