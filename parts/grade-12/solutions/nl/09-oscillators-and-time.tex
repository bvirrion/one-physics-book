\section*{Hoofdstuk \ref{ch:g12:oscillators-and-time} --- Mechanische oscillatoren en de meting van de tijd}

\begin{solution}{exo:g12:oscillators-and-time:1}
(a) $T_0 = 2\pi\sqrt{1.00/9.81} = \qty{2.01}{s}$;
$f_0 = 1/T_0 = \qty{0.50}{Hz}$. (b) $T_0 \propto \sqrt{L}$: de
\omterm{def:g12:oscillators-and-time:oscillator}{periode} verdubbelt.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:2}
Hart: $f = 72/60 = \qty{1.2}{Hz}$, $T = \qty{0.83}{s}$. Vleugel:
$T = 1/30 = \qty{33}{ms}$. De borststijging is de
\emph{\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}}.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:3}
$T_0 = 2\pi\sqrt{0.10/25} = \qty{0.40}{s}$, $f_0 = \qty{2.5}{Hz}$.
Massa verdubbelen: $T_0 \times \sqrt{2} \approx 1.4$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:4}
\omterm{def:g12:oscillators-and-time:damping}{Pseudoperiode} \qty{2.0}{s}. Verhoudingen
$1.5/2.0 = 1.13/1.5 = 0.85/1.13 = 0.75$ elke
\omterm{def:g12:oscillators-and-time:oscillator}{periode}; volgend maximum
$0.85 \times 0.75 \approx \qty{0.64}{cm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:5}
Slinger: $T_0 \propto 1/\sqrt{g}$, dus $\times\sqrt{9.81/1.62} =
2.46$ (trager). Massa--veer: onveranderd --- geen $g$ in
$2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:6}
(a) $[L/g] = \unit{m}/(\unit{m/s^2}) = \unit{s^2}$; de
vierkantswortel is een tijd. (b) Het
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} verschijnt alleen in $m$: niets anders
kan het opheffen, dus $m$ kan niet meedoen. (c) Nooit: pure getallen
als $2\pi$ zijn onzichtbaar voor dimensies.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:7}
$\frac{dx}{dt} = -\frac{2\pi}{T_0}A\sin(2\pi t/T_0 + \varphi)$, dus
$\frac{d^2x}{dt^2} = -(2\pi/T_0)^2 x(t)$. Gelijkheid met $-(k/m)x$
voor alle $t$ geldt precies wanneer $(2\pi/T_0)^2 = k/m$, d.w.z.\
$T_0 = 2\pi\sqrt{m/k}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:8}
$E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.030^2 =
\qty{1.8e-2}{J}$; $v_{\max} = \sqrt{2E_m/m} = \sqrt{0.18} =
\qty{0.42}{m/s}$ --- bij $x = 0$; nul in de uitersten $x = \pm A$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:9}
(a) $L = gT_0^2/4\pi^2 = \qty{0.994}{m}$. (b) $\Delta T_0/T_0 =
\tfrac12 \times 1.0\% = 0.5\%$: $86400 \times 0.005 \approx
\qty{4.3e2}{s}$ --- zeven minuten te traag per dag.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:10}
(a) $A \propto \sqrt{E}$: $\sqrt{0.95} \approx 0.975$, een daling
van $2.5\%$. (b) $0.95^n = 0.5$: $n = \ln 2/\ln(1/0.95) \approx 14$
\omterm{def:g12:oscillators-and-time:oscillator}{perioden}. (c) Nee: bij lichte
\omterm{def:g12:oscillators-and-time:damping}{demping} blijft ze binnen een fractie van een
procent van $T_0$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:11}
(a) Bij $f_0$ komt elke duw in de pas en voegt
\omterm{def:g10:energy-conservation:energy}{energie} toe; buiten de piek helpen en hinderen
duwen afwisselend. (b) In de pas marcheren is een
\omterm{def:g10:signals-and-waves:period}{periodieke} aandrijving; nabij de $f_0$ van de
brug zou die de \omterm{def:g12:oscillators-and-time:resonance}{resonantie}piek beklimmen.
(c) De trommel veegt door de
\omterm{def:g12:oscillators-and-time:resonance}{eigenfrequentie} van het frame tijdens
opstarten: momentane \omterm{def:g12:oscillators-and-time:resonance}{resonantie}, dan voorbij de
piek.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:12}
(a) $T_0 \propto 1/\sqrt{g}$: te traag, met $\Delta T_0/T_0 =
\tfrac12 \times 0.03/9.81 = \num{1.5e-3}$, d.w.z.\ $\approx
\qty{132}{s}$ per dag. (b) $L \propto g$ bij vaste $T_0$:
verkorten met $0.994 \times 0.03/9.81 \approx \qty{3.0}{mm}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:13}
$v_{\max} = 2\pi A/T_0 = 2\pi \times 0.040/0.63 = \qty{0.40}{m/s}$;
$a_{\max} = (2\pi/T_0)^2 A = \qty{4.0}{m/s^2}$; $k = m(2\pi/T_0)^2
= \qty{20}{N/m}$; $E_m = \tfrac12 kA^2 = \qty{1.6e-2}{J}$.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:14}
$T_0 = 40.1/20 = \qty{2.005}{s}$; $g = 4\pi^2 L/T_0^2 = 4\pi^2
\times 0.995/4.020 = \qty{9.77}{m/s^2}$. Onzekerheid $\approx 0.2\%
+ 2 \times 0.5\% = 1.2\%$, d.w.z.\ $\pm\qty{0.12}{m/s^2}$: het
interval bevat \qty{9.81}{m/s^2} --- consistent.
\end{solution}

\begin{solution}{exo:g12:oscillators-and-time:15}
Slinger $10/86400 \approx \num{1.2e-4}$; kwarts $0.5/\num{2.6e6}
\approx \num{1.9e-7}$; cesium $1/\num{3.2e15} \approx \num{3e-16}$.
Een fout van \qty{1}{\micro s} is $\num{1000} \times \qty{30}{cm}
= \qty{300}{m}$ positie: GPS eist de atomaire familie.
\end{solution}

\begin{solution}{pb:g12:oscillators-and-time:1}
\textbf{1.} De reactiefout $\approx \qty{0.2}{s}$ wordt gedeeld door
$50$: ongeveer \qty{4}{ms} op $T_0$.

\textbf{2.} $T = 99.4/50 = \qty{1.988}{s}$.

\textbf{3.} $L = gT^2/4\pi^2 = 9.81 \times 1.988^2/4\pi^2 \approx
\qty{0.982}{m}$.

\textbf{4.} $L = \qty{0.994}{m}$ (de secondenslinger): de massa
omlaag, verlengen met $\approx \qty{1.2}{cm}$.

\textbf{5.} Te kort, dus te snel: te snel, met $(2.000 -
1.988)/1.988 \times 86400 \approx \qty{5.2e2}{s} \approx 8.7$
minuten per dag.

\textbf{6.} Maan: $\times\sqrt{9.81/1.62} = 2.46$ --- onbruikbaar.
Sterrenwacht: $\Delta T_0/T_0 = \tfrac12 \times 0.01/9.81 =
\num{5.1e-4}$: verliest $\approx \qty{44}{s}$ per dag.

\textbf{7.} $\Delta L = L\lambda\,\Delta\theta = 0.994 \times
\num{1.9e-5} \times 5.0 \approx \qty{9.4e-5}{m} \approx
\qty{0.1}{mm}$.

\textbf{8.} $T' = 2\pi\sqrt{L(1 + \lambda\Delta\theta)/g} =
T_0\sqrt{1 + \lambda\Delta\theta} \approx T_0\,(1 + \tfrac12
\lambda\Delta\theta)$.

\textbf{9.} $\Delta T_0/T_0 = \tfrac12 \times \num{1.9e-5} \times
5.0 = \num{4.8e-5}$: $86400 \times \num{4.8e-5} \approx
\qty{4.1}{s}$ per dag, te traag (langere stang, langere
\omterm{def:g12:oscillators-and-time:oscillator}{periode}).

\textbf{10.} $\Delta\theta = \qty{-10}{K}$: $\Delta T_0/T_0 =
\num{-9.5e-5}$, wint $\approx \qty{8.2}{s}$ per dag.

\textbf{11.} Staal en messing zetten verschillend uit; in
afwisselende richtingen hangen de messingsstaven de massa omlaag
terwijl de stalen hem omhoog brengen. Zo bemeten dat de twee
verschuivingen elkaar opheffen, overleeft de effectieve lengte ---
en het ritme --- de seizoenen.

\textbf{12.} $280/2.000 = 140$ \omterm{def:g12:oscillators-and-time:oscillator}{perioden}.

\textbf{13.} $\theta_m = 2.00^\circ = \qty{0.0349}{rad}$: $E_m =
\tfrac12 \times 1.2 \times 9.81 \times 0.994 \times 0.0349^2 \approx
\qty{7.1e-3}{J}$.

\textbf{14.} Per \omterm{def:g12:oscillators-and-time:oscillator}{periode} krimpt de
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} met $2^{-1/140} \approx 0.9951$,
de \omterm{def:g10:energy-conservation:energy}{energie} met $0.9951^2 \approx 0.990$: ongeveer
$1.0\%$ van $E_m$ per \omterm{def:g12:oscillators-and-time:oscillator}{periode}.

\textbf{15.} Duw $\approx 0.010 \times \num{7.1e-3} =
\qty{7.1e-5}{J}$; \omterm{def:g11:work-of-force:power}{vermogen}
$\num{7.1e-5}/2.0 \approx \qty{3.5e-5}{W}$ --- tientallen
microwatts.

\textbf{16.} Een week is $604800/2.0 = \num{3.02e5}$
\omterm{def:g12:oscillators-and-time:oscillator}{perioden}, vraagt
$\num{3.02e5} \times \num{7.1e-5} \approx \qty{21}{J}$; het
\omterm{def:g10:universal-gravitation:weight}{gewicht} levert $Mgh = 2.5 \times 9.81 \times 1.0
= \qty{24.5}{J}$: genoeg, bij $21/24.5 \approx 88\%$
\omterm{def:g10:energy-conservation:efficiency}{rendement}.

\textbf{17.} $T = 1/\num{32768} = \qty{30.5}{\micro s}$. $\num{32768}
= 2^{15}$: vijftien delen-door-twee-trappen zetten het
\omterm{def:g10:signals-and-waves:signal}{signaal} van het kristal om in een exacte
\qty{1}{Hz}-tik.

\textbf{18.} Kwarts: $0.3/\num{2.6e6} \approx \num{1.2e-7}$.
Slinger: $0.5/86400 \approx \num{5.8e-6}$, d.w.z.\ $\approx
\qty{15}{s}$ per maand. Kwarts wint met een factor $\approx 50$.

\textbf{19.} $1/\num{3.2e15} \approx \num{3e-16}$: negen
ordegrootten onder kwarts. GPS timet
\omterm{def:g10:signals-and-waves:signal}{signalen} tot op nanoseconden ---
\qty{30}{cm} lichtweg elk --- dus alleen atoomklokken voldoen.

\textbf{20.} Hersteld houdt de wandklok $\pm\qty{15}{s}$ per maand
tegen $\pm\qty{0.3}{s}$ van de kwarts: bewaar hem voor de bel, vertrouw
de kwarts voor de trein --- en de seconde zelf behoort nu aan het
cesium\omterm{def:g11:fundamental-interactions:ladder}{atoom}.
\end{solution}
