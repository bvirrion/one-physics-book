\begin{solution}{exo:g12:sound-acoustics:8}

\begin{solution}{exo:g12:sound-acoustics:9}
Lucht: $340/440 \approx \qty{0.77}{m}$; water: $1500/440 \approx

\end{solution}

\begin{solution}{exo:g12:sound-acoustics:10}
$f_1 = \qty{130}{Hz}$ (de piekafstand). Hetzelfde: de staafposities
$nf_1$ (zelfde \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte}) en de totale
intensiteit; anders: de staafhoogten --- de mix die het
\omterm{def:g12:sound-acoustics:timbre}{timbre} maakt.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:11}
$v = 2Lf_1 = 2 \times 0.650 \times 110 = \qty{143}{m/s}$. Octaaf:
$L' = v/(2 \times 220) = L/2 = \qty{32.5}{cm}$; kwint:
$L' = 143/330 \approx \qty{43.3}{cm} = 2L/3$.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:12}
$\lambda = 1540/\num{5.0e6} \approx \qty{0.31}{mm}$. Diepte $= vt/2$:
$1540 \times \num{26e-6}/2 \approx \qty{2.0}{cm}$ en
$1540 \times \num{58e-6}/2 \approx \qty{4.5}{cm}$; dikte
$\approx \qty{2.5}{cm}$.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:13}
$-\qty{25}{dB}$ betekent $I \div 10^{2.5} \approx 316$, dus
$r \times \sqrt{316} \approx 17.8$: $r \approx 3.0 \times 17.8
\approx \qty{53}{m}$. Echte zalen verschillen: reflecties van grond
en wanden voegen intensiteit toe, terwijl lucht en publiek absorberen
(en het toneel is geen isotrope puntbron).
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:14}
$1/10^{-12} = 10^{12}$. Trommelvlies $S = \qty{5e-5}{m^2}$: ontvangen
\omterm{def:g11:work-of-force:power}{vermogen} $\qty{5e-17}{W}$ bij de drempel,
$\qty{5e-5}{W}$ bij pijn --- het oor detecteert tientallen attowatts
en werkt nog bij een biljoen maal meer: een detector met een
twaalf-decade dynamisch \omterm{def:g11:fundamental-interactions:strong}{bereik}.
\end{solution}

\begin{solution}{exo:g12:sound-acoustics:15}
$2$: \qty{83}{dB}; $10$: \qty{90}{dB}; $100$: \qty{100}{dB}. Elke
extra \qty{+10}{dB} kost een tienvoud aan luidsprekers:

\end{solution}

\begin{solution}{pb:g12:sound-acoustics:1}
\textbf{1.} $v = 2Lf_1 = 2 \times 0.650 \times 110 = \qty{143}{m/s}$.

\textbf{2.} Op de snaar $\lambda_1 = 2L = \qty{1.30}{m}$; in lucht
$\lambda = 340/110 \approx \qty{3.1}{m}$. Zelfde $f$, maar
$\lambda = v/f$ en de \omterm{def:g10:signals-and-waves:wave}{golf}snelheden verschillen.

\textbf{3.} $L' = v/(2 \times 220) = L/2 = \qty{32.5}{cm}$: de
twaalfde fret zit op het midden, $L/2$ van de kam.

\textbf{4.} $L' = 143/(2 \times 165) \approx \qty{43.3}{cm} = 2L/3$.

\textbf{5.} $\log(220/110)/\log 2^{1/12} = 12$ halve tonen --- twaalf
frets tot het octaaf. Eerste fret: $L_1 = L/2^{1/12} = 65.0/1.059
\approx \qty{61.4}{cm}$.

\textbf{6.} \qtylist{110;220;330;440;550}{Hz}.

\textbf{7.} $\qty{440}{Hz} = 4f_1$: de vierde
\omterm{def:g12:sound-acoustics:harmonics}{harmonische}, twee octaven omhoog.
$\qty{330}{Hz} = 3f_1$: de derde
\omterm{def:g12:sound-acoustics:harmonics}{harmonische}, een octaaf plus een kwint.

\textbf{8.} De tokkelpositie bepaalt de modusmix: dicht bij de kam
zijn de hoge \omterm{def:g12:sound-acoustics:harmonics}{harmonischen} sterk (helder
\omterm{def:g12:sound-acoustics:timbre}{spectrum}), over de toets domineert de
\omterm{def:g12:sound-acoustics:harmonics}{grondtoon} (zacht). Zelfde $f_1$, dus
zelfde \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} --- ander
\omterm{def:g12:sound-acoustics:timbre}{spectrum}, ander
\omterm{def:g12:sound-acoustics:timbre}{timbre}.

\textbf{9.} Alleen \omterm{def:g12:sound-acoustics:harmonics}{harmonischen} met een
\omterm{def:g12:sound-acoustics:standing-wave}{knoop} op het midden overleven: de even,
\qtylist{220;440;660}{Hz}, \dots De
\omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} springt een octaaf, naar
\qty{220}{Hz}.

\textbf{10.} $nf_1 \le \qty{20}{kHz}$: $n \le 20000/110 = 181.8$,
dus $181$ \omterm{def:g12:sound-acoustics:harmonics}{harmonischen}.

\textbf{11.} $I = P/(4\pi r^2) = \num{6.0e-3}/(4\pi \times 9.0)
\approx \qty{5.3e-5}{W/m^2}$.

\textbf{12.} $L = 10\log_{10}(\num{5.3e7}) \approx \qty{77}{dB}$.

\textbf{13.} $I \times 4$: $+\qty{6}{dB} \to \qty{83}{dB}$. Niveaus
zijn logaritmen: intensiteiten tellen op, niveaus niet.

\textbf{14.} $+10\log_{10}12 \approx +\qty{10.8}{dB} \to
\qty{88}{dB}$.

\textbf{15.} Doel $I = I_0 \times 10^{8.5} \approx
\qty{3.2e-4}{W/m^2}$ met $12P = \qty{7.2e-2}{W}$:
$r = \sqrt{12P/(4\pi I)} \approx \sqrt{18.1} \approx \qty{4.3}{m}$.

\textbf{16.} \qty{88}{dB} is $+\qty{3}{dB}$ over de acht-uur-limiet:
de veilige tijd halveert eenmaal --- vier uur. Het recital past.

\textbf{17.} $\lambda = 5000/\num{2.0e6} = \qty{2.5}{mm}$. Een
\omterm{def:g10:signals-and-waves:wave}{golf} reflecteert alleen van fouten die minstens
ongeveer een \omterm{def:g12:mechanical-waves:wavelength}{golflengte} overspannen: hoorbaar
\omterm{def:g10:signals-and-waves:sound}{geluid} in hout heeft $\lambda$ van tientallen
centimeters en diffracteert om elk klein defect.

\textbf{18.} $t = 2d/v = 2 \times 0.045/5000 = \qty{18}{\micro s}$.

\textbf{19.} $d = vt/2 = 5000 \times \num{7.2e-6}/2 = \qty{18}{mm}$.

\textbf{20.} De kaart: $v = \qty{143}{m/s}$ op de snaar; frets op
$\qty{32.5}{cm}$ (twaalfde) en $\qty{61.4}{cm}$ (eerste); ensemble
veilig voorbij $\approx \qty{4.3}{m}$; fout \qty{18}{mm} diep ---
vier getallen, één hoofdstuk.
\end{solution}
