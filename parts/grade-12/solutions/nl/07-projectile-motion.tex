\section*{Hoofdstuk \ref{ch:g12:projectile-motion} --- Vrije val en projectielbeweging}

\begin{solution}{exo:g12:projectile-motion:1}
$h = \tfrac12 \times 9.81 \times 2.0^2 = \qty{19.6}{m}$;
$v = gt = \qty{19.6}{m/s} \approx \qty{71}{km/h}$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:2}
Geen lucht op de Maan: beide in
\omterm{def:g12:projectile-motion:freefall}{vrije val}, $m\vect a = m\vect g$
geeft $\vect a = \vect g$ voor elke massa. Op Aarde is de sleep op
de veer vergelijkbaar met zijn piepkleine
\omterm{def:g10:universal-gravitation:weight}{gewicht}: hij rijdt op een
\omterm{def:g12:projectile-motion:terminal}{eindsnelheid} van ongeveer
\qty{1}{m/s} terwijl de hamer de lucht nauwelijks merkt.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:3}
$t = \sqrt{2h/g} = 0.45$, $1.4$, \qty{4.5}{s};
$v = gt = 4.4$, $14$, \qty{44}{m/s}. Tijden groeien als $\sqrt h$:
$\times 100$ in hoogte is slechts $\times 10$ in tijd.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:4}
$h = v_0^2/2g = \qty{11.5}{m}$ bij $t = v_0/g = \qty{1.53}{s}$;
terug bij \qty{3.06}{s}, opnieuw met \qty{15}{m/s} (symmetrie).
\end{solution}

\begin{solution}{exo:g12:projectile-motion:5}
$t = \sqrt{2 \times 0.80/9.81} = \qty{0.40}{s}$;
$x = 2.5 \times 0.40 = \qty{1.0}{m}$. Nee: de valtijd hoort bij de
verticale beweging alleen (onafhankelijkheid).
\end{solution}

\begin{solution}{exo:g12:projectile-motion:6}
$(\ang{15},\ang{75})$ en $(\ang{30},\ang{60})$ delen hun
\omterm{def:g12:projectile-motion:range}{reikwijdte}; \ang{45} wint. De
steilere van een complementair paar vliegt langer:
$t_f = 2v_0\sin\alpha/g$ groeit met $\alpha$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:7}
$4.905\,t^2 + 5.0\,t - 20 = 0$ geeft
$t = (-5 + \sqrt{25 + 392.4})/9.81 = \qty{1.57}{s}$;
$v = 5.0 + 9.81 \times 1.57 = \qty{20.4}{m/s}$.
\omterm{def:g10:energy-conservation:energy}{Energie}:
$v = \sqrt{5.0^2 + 2 \times 9.81 \times 20} = \qty{20.4}{m/s}$ ---
hetzelfde.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:8}
$t_f = 2 \times 40 \times \sin\ang{30}/9.81 = \qty{4.1}{s}$;
$h = (40\sin\ang{30})^2/(2 \times 9.81) = \qty{20.4}{m}$;
$R = 40^2 \sin\ang{60}/9.81 = \qty{1.4e2}{m}$.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:9}
$R = 10.5^2\sin\ang{40}/9.81 \approx \qty{7.2}{m}$. Weggelaten: het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} zet hoog af
(uitgestrekt lichaam) en landt laag en ver vooruit (benen vooruit
geworpen), en de springer is geen punt --- de ontbrekende
\qty{1.7}{m} leven in die meetkunde, niet in de parabool.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:10}
Samen: beide hebben $v_y(0) = 0$, dus dezelfde
$t = \sqrt{2 \times 1.2/9.81} = \qty{0.49}{s}$; de afgevuurde landt
$300 \times 0.49 \approx \qty{1.5e2}{m}$ verder. Het test de
onafhankelijkheid van horizontale en verticale bewegingen.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:11}
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180}-2\alpha) =
\sin(2\alpha)$: dezelfde $R$. Hier $\sin(2\alpha) = gR/v_0^2 =
9.81 \times 50/625 = 0.785$: $\alpha = \ang{25.9}$ of \ang{64.1}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:12}
$R_{\max} = v_0^2/g = \qty{6.5}{m}$ (bij \ang{45});
$h_{\max} = v_0^2/2g = \qty{3.3}{m}$ (recht omhoog).
$\sin(2\alpha) = 9.81 \times 5.0/64 = 0.766$: $\alpha = \ang{25.0}$
of \ang{65.0}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:13}
$y(15) = 15\tan\ang{35} - \dfrac{9.81 \times 15^2}
{2 \times 18^2 \cos^2\ang{35}} = 10.5 - 5.1 = \qty{5.4}{m}$:
hij maakt de \qty{4.0}{m}-haag vrij met ongeveer \qty{1.4}{m}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:14}
(a) Sleep balanceert \omterm{def:g10:universal-gravitation:weight}{gewicht}:
$80 \times 9.81 \approx \qty{7.8e2}{N}$.
(b) $v(t)$ stijgt met steeds kleinere helling en vlakt af bij
\qty{55}{m/s}. (c) Vacuüm:
$\sqrt{2 \times 9.81 \times 2000} \approx \qty{2.0e2}{m/s}$,
twintigmaal het echte \qty{9}{m/s} --- regendruppels brengen bijna
de hele val door op
\omterm{def:g12:projectile-motion:terminal}{eindsnelheid}.
\end{solution}

\begin{solution}{exo:g12:projectile-motion:15}
Bij $t^* = D/(v_0\cos\alpha)$ is het pijltje op hoogte
$v_0\sin\alpha\, t^* - \tfrac12 g t^{*2} = D\tan\alpha - \tfrac12 g
t^{*2} = H - \tfrac12 g t^{*2}$, precies de losgelaten aap zijn
$y = H - \tfrac12 g t^{*2}$. Beide hangen $\tfrac12 gt^2$ onder
hun geen-zwaartekracht-plekken --- ze ontmoeten elkaar voor elke
$v_0$.
\end{solution}

\begin{solution}{pb:g12:projectile-motion:1}
\textbf{1.} Alleen \omterm{def:g10:universal-gravitation:weight}{gewicht}
(lucht verwaarloosd); \omterm{thm:g12:newtons-laws:second}{Newtons tweede
wet}: $m\vect a = m\vect g$, dus $\vect a = (0,\,-g)$.

\textbf{2.} $v_x = v_0\cos\alpha$, $v_y = v_0\sin\alpha - gt$.

\textbf{3.} $x = v_0\cos\alpha\, t$,
$y = v_0\sin\alpha\, t - \tfrac12 gt^2$.

\textbf{4.} $t = x/(v_0\cos\alpha)$:
$y = x\tan\alpha - g x^2/(2v_0^2\cos^2\alpha)$.

\textbf{5.} $v_0^2 = \dfrac{g d^2}{2\cos^2\alpha\,(d\tan\alpha -
1.05)} = \dfrac{173.0}{0.826 \times 3.96} = 52.9$:
$v_0 = \qty{7.28}{m/s} \approx \qty{26}{km/h}$.

\textbf{6.} $t = d/(v_0\cos\alpha) = 4.20/4.68 = \qty{0.90}{s}$.

\textbf{7.} Top bij $t = v_0\sin\alpha/g = \qty{0.57}{s}$, hoogte
$2.00 + (v_0\sin\alpha)^2/2g = 2.00 + 1.58 = \qty{3.58}{m}$;
$0.57 < 0.90$: dalend bij de ring.

\textbf{8.} $y(0.90) = 0.90\tan\ang{50} - 0.182 = \qty{0.89}{m}$
boven loslaten, d.w.z.\ \qty{2.89}{m} boven de vloer: \qty{0.29}{m}
boven de vingertoppen.

\textbf{9.} $v_x = \qty{4.68}{m/s}$;
$v_y = 5.57 - 9.81 \times 0.90 = \qty{-3.24}{m/s}$;
$v = \qty{5.69}{m/s}$, onder $\arctan(3.24/4.68) = \ang{34.7}$ onder
de horizontaal.

\textbf{10.} De ring is een horizontale ring: de bal moet zijn
vlak van boven doorkruisen. Hoe steiler de entry, hoe groter de
opening van de ring lijkt langs de snelheid --- meer foutmarge.

\textbf{11.} $1.05/26.0 = 4\%$: verwaarloosbaar. $y(t_f) = 0$ geeft
$t_f\,(v_0\sin\alpha - \tfrac12 g t_f) = 0$, dus
$t_f = 2v_0\sin\alpha/g$.

\textbf{12.} $R = v_0\cos\alpha\, t_f = v_0^2\sin(2\alpha)/g$;
maximaal bij $\alpha = \ang{45}$.

\textbf{13.} $v_0 = \sqrt{gR} = \sqrt{9.81 \times 26.0} =
\qty{16.0}{m/s} \approx \qty{57}{km/h}$.

\textbf{14.} $R_{\max} = 17.5^2/9.81 = \qty{31.2}{m} >
\qty{26.0}{m}$: haalbaar, met \qty{5}{m}
\omterm{def:g12:projectile-motion:range}{reikwijdte} speling.

\textbf{15.} $\sin(2\alpha) = 255.1/306.3 = 0.833$:
$\alpha = \ang{28.2}$ of \ang{61.8}; vluchttijden
$2v_0\sin\alpha/g = \qty{1.69}{s}$ en \qty{3.14}{s}. Alleen het
vlakke schot wint de \qty{2.0}{s}-buzzer.

\textbf{16.} $v_0^2 = \dfrac{9.81 \times 36}{2\cos^2\ang{75}\,
(6.0\tan\ang{75} - 1.05)} = \dfrac{353.2}{2.86} = 123.5$:
$v_0 = \qty{11.1}{m/s}$.

\textbf{17.} $y(5.55) = 5.55\tan\ang{75} - 18.26 = \qty{2.45}{m}$
boven loslaten, d.w.z.\ \qty{4.45}{m} boven de vloer: \qty{0.50}{m}
boven de bovenrand van het bord.

\textbf{18.} Top $2.00 + (11.1\sin\ang{75})^2/2g \approx
\qty{7.9}{m}$; vlucht $6.0/(11.1\cos\ang{75}) = \qty{2.1}{s}$ ---
twee volle seconden met de bal \qty{8}{m} hoog hangend.

\textbf{19.} Sleep verkort elk schot en versteilt de daling; het
effect groeit met de snelheid, dus de \qty{17.5}{m/s}-full-court-heave
lijdt het meest --- zijn echte
\omterm{def:g12:projectile-motion:range}{reikwijdte} valt enkele meters
kort van \qty{31}{m}, wat in de marge bijt.

\textbf{20.} \emph{Het buzzer-beater-vonnis}: het \qty{26.0}{m}-schot
eist \qty{16.0}{m/s} en een arm levert ongeveer \qty{17.5}{m/s}, een
\omterm{def:g12:projectile-motion:range}{reikwijdte}-marge van \qty{5}{m}
($\approx 20\%$) die luchtweerstand trimt maar niet uitwist.
Mogelijk --- precies waarom, enkele keren per seizoen, hij erin
gaat.
\end{solution}
