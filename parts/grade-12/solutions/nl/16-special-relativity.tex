\section*{Hoofdstuk \ref{ch:g12:special-relativity} --- Speciale relativiteit: tijddilatatie}

\begin{solution}{exo:g12:special-relativity:1}
$\gamma = 1/\sqrt{1 - (v/c)^2}$: $1.01$ ($0.1c$); $1.25$ ($0.6c$);
$1.67$ ($0.8c$); $10.0$ ($0.995c$).
\end{solution}

\begin{solution}{exo:g12:special-relativity:2}
$\gamma = 1.25$, dus de grond meet $1.25 \times 10.0 =
\qty{12.5}{s}$. De \qty{10.0}{s} van het schip is de
\omterm{def:g12:special-relativity:proper-time}{eigentijd}: de metronoom (één klok) is
aanwezig bij beide slagen.
\end{solution}

\begin{solution}{exo:g12:special-relativity:3}
$v = c\sqrt{1 - 1/\gamma^2}$: $\gamma = 2$ geeft $v = 0.866c$;
$\gamma = 10$ geeft $v = 0.995c$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:4}
$v = \qty{36.1}{m/s}$, $v/c = \num{1.2e-7}$;
$\gamma - 1 \approx (\num{1.2e-7})^2/2 \approx \num{7.2e-15}$. Eén
seconde verloren na $1/\num{7.2e-15} \approx \qty{1.4e14}{s}$ ---
ongeveer vier miljoen jaar rijden.
\end{solution}

\begin{solution}{exo:g12:special-relativity:5}
De passagier rijdt naar de voorflits toe, dus die bereikt hem eerst;
door het tweede postulaat reizen beide flitsen met dezelfde $c$ over
de gelijke half-trein-afstanden \emph{in zijn stelsel}, dus een
eerdere aankomst betekent een eerdere inslag. Gelijktijdigheid is
stelselafhankelijk: elk vonnis is correct in zijn eigen stelsel.
\end{solution}

\begin{solution}{exo:g12:special-relativity:6}
$\gamma = 1/\sqrt{1 - 0.99^2} = 7.09$; lablevensduur
$7.09 \times \num{2.6e-8} = \qty{1.8e-7}{s}$. Klassiek
\omterm{def:g12:projectile-motion:range}{bereik} $0.99 \times \num{3.00e8} \times
\num{2.6e-8} \approx \qty{7.7}{m}$; werkelijk $7.09 \times 7.7 \approx
\qty{55}{m}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:7}
$\gamma = 1.5$ bij $v = c\sqrt{1 - 1/2.25} \approx 0.75c$;
$\gamma = 3$ bij $\approx 0.94c$. De asymptoot: $\gamma$ (en de
\omterm{def:g10:energy-conservation:energy}{energie}kost van verdere snelheid) divergeert als
$v \to c$ --- geen eindige inspanning brengt een massief lichaam tot
lichtsnelheid.
\end{solution}

\begin{solution}{exo:g12:special-relativity:8}
$\Delta t_0 = 2L/c = 3.0/\num{3.00e8} = \qty{1.0e-8}{s}$; bij $0.8c$,
$\gamma = 5/3$, dus $\Delta t = \qty{1.7e-8}{s}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:9}
\omterm{def:g12:special-relativity:proper-time}{Eigentijd} behoort aan de klok die bij beide
\omterm{def:g12:special-relativity:proper-time}{gebeurtenissen} aanwezig is: (a) de drager; (b)
het muon; (c) de piloot --- alleen het schip is bij zowel vertrek als
aankomst.
\end{solution}

\begin{solution}{exo:g12:special-relativity:10}
$\gamma - 1 \approx 7700^2/(2 \times \num{9.0e16}) \approx
\num{3.3e-10}$; over \qty{1.58e7}{s} blijft de astronaut
$\num{3.3e-10} \times \num{1.58e7} \approx \qty{5.2e-3}{s}$ achter
--- ongeveer vijf milliseconden jonger.
\end{solution}

\begin{solution}{exo:g12:special-relativity:11}
$E = \num{1.0e-3} \times (\num{3.00e8})^2 = \qty{9.0e13}{J}$. Een
station van \qty{1.0}{GW} levert $\num{1.0e9} \times 86400 \approx
\qty{8.6e13}{J}$ per dag --- één gram is één dag van zijn output, en
$\num{9.0e13}/\num{8.5e5} \approx \num{e8}$: honderd miljoen snelle
auto's.
\end{solution}

\begin{solution}{exo:g12:special-relativity:12}
Kwadrateer en inverteer: $\gamma^2(1 - v^2/c^2) = 1$, dus
$v^2/c^2 = 1 - 1/\gamma^2$, vandaar $v = c\sqrt{1 - 1/\gamma^2}$. Voor
$\gamma = 30$: $v = c\sqrt{1 - 1/900} = 0.99944c$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:13}
Gecollapte dikte $L = \qty{15}{km}/30 = \qty{500}{m}$; veegtijd
$500/(0.99944 \times \num{3.00e8}) \approx \qty{1.7e-6}{s} <
\qty{2.2}{\micro s}$. Beide stelsels stemmen in dat het muon de grond
bereikt --- het ene wijt het aan een gerekte levensduur, het andere
aan een gekrompen atmosfeer.
\end{solution}

\begin{solution}{exo:g12:special-relativity:14}
Lab: $3200/\num{3.00e8} \approx \qty{1.1e-5}{s}$ (snelheid
$\approx c$). Elektronstelsel: buis gecollapt tot
$3200/\num{1.0e5} = \qty{3.2}{cm}$, overgestoken in
$\num{1.1e-5}/\num{1.0e5} \approx \qty{1.1e-10}{s}$.
\end{solution}

\begin{solution}{exo:g12:special-relativity:15}
Aarde: $4.2/0.9 \approx \qty{4.7}{years}$. Met
$\gamma = 1/\sqrt{1 - 0.81} = 2.29$ logt de klok van de sonde
$4.7/2.29 \approx \qty{2.0}{years}$.
\end{solution}

\begin{solution}{pb:g12:special-relativity:1}
\textbf{1.} $c\,\Delta t_0 = \num{3.00e8} \times \num{2.2e-6} \approx
\qty{660}{m}$.

\textbf{2.} $\num{15000}/660 \approx 23$ levensduren.

\textbf{3.} $\mathrm{e}^{-23} \approx \num{e-10}$: minder dan één muon
op tien miljard zou moeten aankomen, toch is de flux één per
\unit{cm^2} per minuut --- de klassieke natuurkunde wordt botweg door
de hemel weersproken.

\textbf{4.} Grondlevensduur $30 \times \qty{2.2}{\micro s} =
\qty{66}{\micro s}$; \omterm{def:g12:projectile-motion:range}{bereik} $\approx
\num{3.00e8} \times \num{6.6e-5} \approx \qty{20}{km} > \qty{15}{km}$:
de aankomst is verklaard.

\textbf{5.} $v = c\sqrt{1 - 1/900} = 0.99944c$.

\textbf{6.} $\Delta t_0 = 2L/c = 3.0/\num{3.00e8} = \qty{1.0e-8}{s}$.

\textbf{7.} Horizontaal $v\,\Delta t/2$; verticaal $L$; de schuine
halfweg meet $c\,\Delta t/2$ omdat het tweede postulaat de
snelheid van het \omterm{def:g12:quantum-world:photon}{foton} op $c$ vastzet, ook in
het \omterm{def:g10:relative-motion:frames}{grondstelsel}.

\textbf{8.} $(c\,\Delta t/2)^2 = L^2 + (v\,\Delta t/2)^2$ geeft
$\Delta t^2(c^2 - v^2) = 4L^2$, dus $\Delta t =
(2L/c)/\sqrt{1 - v^2/c^2} = \gamma\,\Delta t_0$.

\textbf{9.} $\gamma = 1/\sqrt{1 - 0.36} = 1.25$;
$\Delta t = \qty{1.25e-8}{s}$.

\textbf{10.} Die van het schip: zijn ene klok is bij beide tikken
aanwezig, dus hij leest de
\omterm{def:g12:special-relativity:proper-time}{eigentijd}; de grond heeft twee
gesynchroniseerde klokken nodig.

\textbf{11.} $\gamma = 1/\sqrt{1 - 0.64} = 1/0.6 = 5/3 \approx 1.67$.

\textbf{12.} $10.0/(5/3) = \qty{6.0}{years}$.

\textbf{13.} Uitgaand $\qty{5.0}{years}$ bij $0.8c$: $4.0$
\omterm{def:g10:orders-of-magnitude:lightyear}{lichtjaar}.

\textbf{14.} $10.0 - 6.0 = \qty{4.0}{years}$: de astronaut is vier
jaar jonger dan haar tweeling.

\textbf{15.} De situatie is niet symmetrisch: alleen de astronaut
keert om --- zij versnelt en wisselt van
\omterm{def:g12:special-relativity:inertial}{inertiaalstelsel}, terwijl de
aardse tweeling in één blijft --- dus alleen haar klok logt de
kortere tijd.

\textbf{16.} $v = 2\pi \times \num{2.66e7}/\num{43200} \approx
\qty{3.9e3}{m/s}$.

\textbf{17.} $\gamma - 1 \approx (\num{3.87e3})^2/(2 \times
\num{9.0e16}) \approx \num{8.3e-11}$.

\textbf{18.} $\num{8.3e-11} \times \num{86400} \approx \qty{7.2e-6}{s}
= \qty{7.2}{\micro s}$ per dag te traag.

\textbf{19.} $c \times \num{7.2e-6} \approx \qty{2.2}{km}$
positiefout per dag.

\textbf{20.} Nettodrift $45 - 7.2 \approx \qty{38}{\micro s}$ per dag
te snel; fout $c \times \num{38e-6} \approx \qty{11}{km}$ per dag ---
de relativiteitsrekening die elke GPS-klok vooruit betaalt, op de
grond verstemd zodat hij in \omterm{def:g10:universal-gravitation:satellite}{baan} juist
tikt.
\end{solution}
