\section*{Hoofdstuk \ref{ch:g12:doppler-effect} --- Het Doppler-effect}

\begin{solution}{exo:g12:doppler-effect:1}
(a) $f' = 700 \times 340/315 \approx \qty{756}{Hz}$.
(b) $f' = 700 \times 340/365 \approx \qty{652}{Hz}$.
(c) \qty{700}{Hz}: chauffeur en sirene bewegen samen, hun afstand
verandert nooit.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:2}
De \emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{luidheid}} verzwakt omdat de
\omterm{def:g10:signals-and-waves:wave}{golf} zich uitspreidt --- hij hangt van de
afstand af. De \emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte}} daalt
van nadering naar verwijdering --- het
\omterm{def:g12:doppler-effect:doppler}{Doppler-effect}, dat afhangt van hoe
snel de afstand verandert. Twee onafhankelijke effecten.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:3}
Bewegende waarnemer: $f' = 700\,(1 + 6.0/340) \approx \qty{712}{Hz}$;
wegrijdend, $700\,(1 - 6.0/340) \approx \qty{688}{Hz}$. De sirene is
in rust in de lucht, dus het \omterm{def:g10:signals-and-waves:wave}{golf}patroon is
ongestoord; alleen de waarnemer rent erdoorheen.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:4}
Vooruit: $\lambda' = 310/400 = \qty{0.775}{m}$; achter:
$370/400 = \qty{0.925}{m}$; in rust: $340/400 = \qty{0.85}{m}$. De
korte \omterm{def:g12:mechanical-waves:wavelength}{golflengte} bereikt wie de trein
nadert, de lange wie hij verlaat.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:5}
$\Delta f/f \approx 5.0/340 \approx 1.5\%$: $\Delta f \approx
\qty{6.5}{Hz}$, dus ongeveer \qty{446}{Hz} --- een kwart van een
halve toon. Een musicus zou het net merken.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:6}
De muur ontvangt $f\,v/(v - v_b)$ (bewegende bron); de vleermuis, een
bewegende waarnemer die de \omterm{rem:g10:signals-and-waves:honest}{echo}
dicht, hoort $(1 + v_b/v)$ maal dat:
$f' = f(v + v_b)/(v - v_b) = 50.0 \times 346/334 \approx
\qty{51.8}{kHz}$, een verschuiving van $+\qty{1.8}{kHz}$. Het is de
radarpistool-dubbele verschuiving met zender- en reflectorrollen
omgewisseld: alleen de naderingssnelheid telt.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:7}
$u = c\,\Delta f/(2f) = \num{3.00e8} \times 4000/(2 \times
\num{24.15e9}) \approx \qty{24.8}{m/s} \approx \qty{89}{km/h}$. Bij
\qty{50}{km/h} ($\qty{13.9}{m/s}$):
$\Delta f = 2 \times 13.9 \times \num{24.15e9}/\num{3.00e8} \approx
\qty{2.2}{kHz}$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:8}
$u = v\,\Delta f/(2f) = 1540 \times 2600/\num{1.0e7} \approx
\qty{0.40}{m/s}$. Loodrechte stroming heeft nul
\omterm{def:g12:doppler-effect:radial}{radiale snelheid}, dus helemaal geen
verschuiving.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:9}
(a) Bron: $f' = 1000 \times 340/306 \approx \qty{1111}{Hz}$. (b)
Waarnemer: $f' = 1000 \times 1.1 = \qty{1100}{Hz}$. Trage beweging
voorspelt \qty{1100}{Hz}: exact voor de waarnemer, \qty{11}{Hz} te
kort voor de bron --- de orde-$x^2$-correctie, $1\%$ van $f$ voor
$x = 0.1$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:10}
$f = 2 \times 465 \times 415/880 \approx \qty{439}{Hz}$;
$v_s = 340 \times 50/880 \approx \qty{19.3}{m/s} \approx
\qty{70}{km/h}$. De ware $f$ is het
\omterm{def:g12:sound-acoustics:harmonics}{harmonisch} gemiddelde: nadering
verhoogt de \omterm{def:g10:signals-and-waves:frequency}{frequentie} meer dan
verwijdering hem verlaagt, dus zit $f$ onder het rekenkundig
gemiddelde (\qty{440}{Hz}).
\end{solution}

\begin{solution}{exo:g12:doppler-effect:11}
Langere \omterm{def:g12:mechanical-waves:wavelength}{golflengte}:
\omterm{def:g12:doppler-effect:redshift}{roodverschuiving}, de ster verwijdert
zich. $v_r = c\,\Delta\lambda/\lambda = \num{3.00e8} \times
0.2/656.3 \approx \qty{9.1e4}{m/s} \approx \qty{91}{km/s}$.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:12}
$\Delta\lambda = \lambda K/c = 500 \times 55/\num{3.00e8} \approx
\qty{9.2e-5}{nm}$; $\Delta\lambda/\lambda = 55/\num{3.00e8} \approx
\num{1.8e-7}$. Het hele \omterm{def:g10:signals-and-waves:signal}{signaal} is twee
delen in tien miljoen: een spectrograaf die meer dan $10^{-7}$
drifts begraaft het.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:13}
$v_r = 0.024\,c \approx \qty{7.2e6}{m/s}$. Snelheden evenredig met
de afstand, hetzelfde in elke richting: alle afstanden rekken
uniform --- een uitdijend heelal zonder bevoorrecht centrum. Bij
$2.4\%$ zijn de correcties van orde $(v_r/c)^2 \approx
\num{6e-4}$: nog steeds betrouwbaar.
\end{solution}

\begin{solution}{exo:g12:doppler-effect:14}
De twee randen verschuiven tegengesteld, dus de over de schijf
gemiddelde lijn is \emph{verbreed}, niet verschoven: volle breedte
$2\lambda v/c = 2 \times 656.3 \times \num{2.0e3}/\num{3.00e8}
\approx \qty{8.8e-3}{nm}$ (ongeveer $\pm\qty{4.4e-3}{nm}$).
\end{solution}

\begin{solution}{exo:g12:doppler-effect:15}
(a) Voor de auto zijn de kammen gespatieerd $(v - v_s)T$; de
vrachtwagen vlucht met $v_o$ en ontmoet ze met relatieve snelheid
$v - v_o$: $f' = (v - v_o)/\bigl((v - v_s)T\bigr) =
f(v - v_o)/(v - v_s)$. (b) $f' = 700 \times 310/300 \approx
\qty{723}{Hz}$. (c) $v_o = v_s$ geeft $f' = f$: de scheiding is
constant, en geen verandering van afstand betekent geen
Doppler-verschuiving, per definitie.
\end{solution}

\begin{solution}{pb:g12:doppler-effect:1}
\textbf{1.} Elke kam dijt uit vanaf het punt waar hij werd
uitgezonden: kammen bundelen voor de bewegende hoorn en rekken
erachter, zodat de naderende telefoon ze sneller ontmoet ($f_1$)
dan de verwijderende ($f_2$).

\textbf{2.} $f_1 = f\,\dfrac{v}{v - u}$, \quad $f_2 = f\,\dfrac{v}{v + u}$.

\textbf{3.} $f_1 - f_2 = \frac{2fv\,u}{v^2 - u^2}$ en $f_1 + f_2 =
\frac{2fv^2}{v^2 - u^2}$; delend, $u = v\,\frac{f_1 - f_2}{f_1 + f_2} =
340 \times 60/882 \approx \qty{23}{m/s} \approx \qty{83}{km/h}$.

\textbf{4.} $\dfrac{2f_1 f_2}{f_1 + f_2} = f$ (de gemeenschappelijke
factor valt weg): $f = 2 \times 471 \times 411/882 \approx
\qty{439}{Hz}$ --- het \omterm{def:g12:sound-acoustics:harmonics}{harmonisch}
gemiddelde, onder het gemiddelde \qty{441}{Hz} omdat nadering $f$
meer opkrikkt dan verwijdering hem snijdt.

\textbf{5.} Vooruit: $(v - u)/f = 316.9/439 \approx \qty{0.72}{m}$;
achter: $363.1/439 \approx \qty{0.83}{m}$; in rust:
$340/439 \approx \qty{0.77}{m}$.

\textbf{6.} Eens als bewegende \emph{waarnemer} die de
\omterm{def:g10:signals-and-waves:wave}{golf} ontvangt, eens als bewegende
\emph{bron} die heruitzendt wat hij ontving.

\textbf{7.} $f' = f\,(1 + u/c)/(1 - u/c) \approx f(1 + 2u/c)$, dus
$\Delta f = f' - f \approx \dfrac{2u}{c}\,f$.

\textbf{8.} $u = c\,\Delta f/(2f) = \num{3.00e8} \times 3550/(2 \times
\num{24.125e9}) \approx \qty{22}{m/s}$.

\textbf{9.} $22.1 \times 3.6 \approx \qty{79}{km/h}$: net onder de
\qty{80}{km/h}-limiet.

\textbf{10.} $\Delta f = 2f/(3.6\,c) \approx \qty{45}{Hz}$ per
\unit{km/h}. De \omterm{rem:g10:signals-and-waves:honest}{echo} tegen de
uitgezonden \omterm{def:g10:signals-and-waves:wave}{golf} slaan laat alleen het
\emph{verschil}: een audio\omterm{def:g10:signals-and-waves:frequency}{frequentie},
triviaal te tellen, hoewel het een $10^{-7}$-fractie van de drager is.

\textbf{11.} De \omterm{def:g12:mechanical-waves:wavelength}{golflengten} van
bekende \omterm{def:g10:light-spectra:emission}{spectraallijnen}, nacht na nacht;
dan $v_r = c\,\Delta\lambda/\lambda$.

\textbf{12.} $K = \qty{55}{m/s}$, $P = \qty{4.2}{dagen}$.

\textbf{13.} $\Delta\lambda = 550 \times 55/\num{3.00e8} \approx
\qty{1.0e-4}{nm}$, $\Delta\lambda/\lambda \approx \num{1.8e-7}$ ---
één deel in vijf miljoen, zo'n $\num{4e5}$ maal kleiner dan de
$7\%$-verschuiving van de trein.

\textbf{14.} Ster en planeet trekken elkaar even sterk aan, dus
beide \omterm{def:g10:universal-gravitation:satellite}{cirkelen} om hun
gemeenschappelijke \omterm{def:g11:forces-and-motion:com}{zwaartepunt}: de
ster kan niet stil staan. Een sinusoïdale $v_r$ is
\omterm{def:g11:forces-and-motion:ucm}{eenparige cirkelbeweging} van opzij
gezien --- een circulaire
\omterm{def:g10:universal-gravitation:satellite}{baan}.

\textbf{15.} Alleen de zichtlijn-component verandert de
ster--Aarde-afstand, dus de
\omterm{def:g12:mechanical-waves:wavelength}{golflengte}. Van voren is
$v_r = 0$ te allen tijde: helemaal geen
\omterm{def:g10:signals-and-waves:signal}{signaal}.

\textbf{16.} $P = \qty{3.63e5}{s}$: $r^3 = \num{6.67e-11} \times
\num{2.0e30} \times (\num{3.63e5})^2/(4\pi^2) \approx \num{4.4e29}$,
dus $r \approx \qty{7.6e9}{m}$ --- ongeveer $1/20$ van de
Aarde--Zon-afstand.

\textbf{17.} $V = 2\pi r/P = 2\pi \times \num{7.6e9}/\num{3.63e5}
\approx \qty{1.3e5}{m/s}$.

\textbf{18.} $m = M K/V = \num{2.0e30} \times 55/\num{1.3e5} \approx
\qty{8.3e26}{kg}$.

\textbf{19.} Ongeveer $0.4$ Jupitermassa's ($\approx 140$ Aardes),
cirkelend $8$ maal dichterbij dan Mercurius: een gasreus die tegen
zijn ster roostert --- een ``hete Jupiter''.

\textbf{20.} Een kanteling verbergt een deel van de beweging: de
gemeten $K$ is alleen het radiale aandeel, dus de ware massa is
\emph{groter} --- \qty{8.3e26}{kg} is een minimum. Een
\omterm{def:g10:signals-and-waves:period}{periodieke} verschuiving van één deel in
vijf miljoen leverde zojuist de
\omterm{def:g10:universal-gravitation:satellite}{baan}, snelheid en minimummassa
van een planeet die niemand ooit heeft gezien.
\end{solution}
