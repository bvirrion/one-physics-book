\section*{Hoofdstuk \ref{ch:g12:work-and-energy} --- Arbeid en mechanische energie}

\begin{solution}{exo:g12:work-and-energy:1}
$E_p = \tfrac12 \times 200 \times 0.050^2 = \qty{0.25}{J}$.
Verdubbelde rek: $\times 4$, d.w.z.\ \qty{1.0}{J}. De tweede
\qty{5.0}{cm} alleen kost $1.0 - 0.25 = \qty{0.75}{J}$ --- de
\omterm{def:g10:inertia:force}{kracht} is daar groter.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:2}
$W = mg(z_A - z_B) = 0.90 \times 9.81 \times (12 - 30) \approx
\qty{-1.6e2}{J}$, welke route ook --- hetzelfde op de rechte klim.
Rondreis: \qty{0}{J} (\omterm{def:g12:work-and-energy:conservative}{conservatieve kracht}).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:3}
(a), (b) \omterm{def:g12:work-and-energy:conservative}{conservatief}:
\omterm{def:g11:work-of-force:work}{arbeid} vastgelegd door de eindpunten
(hoogteverlies; rek). (c), (e)
\omterm{def:g12:work-and-energy:conservative}{niet-conservatief}: $-fL$ en
luchtweerstand rekenen per afgelegde
\omterm{def:g10:orders-of-magnitude:unit}{meter}. (d) arbeidloos: altijd loodrecht
op de beweging.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:4}
$h = L(1 - \cos\qty{60}{\degree}) = \qty{1.0}{m}$;
$v = \sqrt{2 \times 9.81 \times 1.0} \approx \qty{4.4}{m/s}$. De
\omterm{def:g10:inertia:inventory}{spanning} staat op elk moment loodrecht op de
beweging: nul \omterm{def:g11:work-of-force:power}{vermogen}, nul
\omterm{def:g11:work-of-force:work}{arbeid}.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:5}
$F = P/v = 250/9.0 \approx \qty{28}{N}$. Bij \qty{5.0}{m/s}:
$F = 250/5.0 = \qty{50}{N}$ (vooral de zwaartekrachtcomponent van de
helling).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:6}
$E_p = \tfrac12 \times 450 \times 0.080^2 = \qty{1.44}{J}$;
$v = \sqrt{2 \times 1.44/0.020} = \qty{12}{m/s}$;
$h = E_p/mg = 1.44/(0.020 \times 9.81) \approx \qty{7.3}{m}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:7}
Trapezia (stappen van \qty{2.0}{cm}): $(0.03 + 0.10 + 0.19 + 0.30) =
\qty{0.62}{J}$. Rechte-lijnmodel: $k = 18.0/0.080 =
\qty{225}{N/m}$, $\tfrac12 kx^2 = \qty{0.72}{J}$. De gemeten kromme
zakt onder de rechte lijn bij kleine $x$, dus de echte oppervlakte is
kleiner.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:8}
$\Delta E_m = \tfrac12 \times 55 \times 9.0^2 - 55 \times 9.81 \times
8.0 = 2228 - 4316 \approx \qty{-2.1e3}{J}$;
$f = 2089/25 \approx \qty{84}{N}$.
\omterm{def:g10:universal-gravitation:weight}{Gewicht}
\omterm{def:g12:work-and-energy:conservative}{conservatief},
\omterm{def:g12:newtons-laws:friction}{normaalkracht} arbeidloos: alleen de daling
en de baans\emph{lengte} tellen, niet de vorm.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:9}
$H_{\min} = \tfrac52 R = \qty{0.50}{m}$. Vanuit \qty{60}{cm}:
$v_{\text{top}} = \sqrt{2 \times 9.81 \times (0.60 - 0.40)} \approx
\qty{2.0}{m/s}$, en $v_{\text{top}}^2/(gR) = 2.0$ --- dubbel het
minimum. \omterm{def:g10:inertia:inventory}{Wrijving} schept
\omterm{def:g10:energy-conservation:energy}{energie} overal af, dus echte banen
hebben extra hoogte nodig.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:10}
\omterm{def:g12:work-and-energy:diagram}{Omkeerpunten}: $8.0\,x^2 = 2.0$,
$x = \pm\qty{0.50}{m}$. Maximale snelheid bij $x = 0$:
$v = \sqrt{2 \times 2.0/0.100} \approx \qty{6.3}{m/s}$. Een heen-en-weer
oscillatie in een parabolische put --- de $E_p$ van de
massa--veer-\omterm{def:g12:oscillators-and-time:oscillator}{oscillator} heeft precies deze
vorm.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:11}
$\tfrac12 kx^2 = \tfrac12 mv^2$:
$x = v\sqrt{m/k} = 2.0\sqrt{1200/\num{6.0e5}} \approx \qty{8.9}{cm}$.
Bij \qty{4.0}{m/s}: \qty{18}{cm} --- verdubbeld, niet verviervoudigd:
de \omterm{def:g10:energy-conservation:energy}{energie} verviervoudigt, maar $E_p$ groeit als
$x^2$, dus $x$ verdubbelt slechts.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:12}
(a) $E_m = \qty{5.0}{J} < \qty{6.0}{J}$: hij komt er niet over; hij
stopt op de flank waar $E_p = \qty{5.0}{J}$ en rolt terug. (b)
Heuveltop: $E_k = \qty{1.0}{J}$, $v = \sqrt{2 \times 1.0/0.50} =
\qty{2.0}{m/s}$; dal: $E_k = \qty{5.0}{J}$, $v \approx
\qty{4.5}{m/s}$. (c) $x = 0$:
\omterm{prop:g12:work-and-energy:equilibria}{instabiel} (maximum);
$x = \qty{1.5}{m}$: \omterm{prop:g12:work-and-energy:equilibria}{stabiel}
(minimum).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:13}
$\tfrac12 mv^2 = mgR_E$:
$v = \sqrt{2 \times 9.81 \times \num{6.37e6}} \approx \qty{1.12e4}{m/s}
= \qty{11.2}{km/s}$. Elke term is evenredig met $m$: sonde en kiezel
gelijk. Net daaronder klimt de lancering nog enorm ver, ontmoet een
\omterm{def:g12:work-and-energy:diagram}{omkeerpunt}, en valt terug: gebonden.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:14}
$E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.060^2 =
\qty{7.2e-2}{J}$; $v_{\max} = \sqrt{2E_m/m} =
\sqrt{2 \times 0.072/0.250} \approx \qty{0.76}{m/s}$;
$E_k = E_p$ waar $\tfrac12 kx^2 = E_m/2$: $x = \pm A/\sqrt 2 \approx
\pm\qty{4.2}{cm}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:15}
$h = v^2/2g = 9.5^2/19.62 \approx \qty{4.6}{m}$; lat $\approx
4.6 + 1.0 = \qty{5.6}{m}$. Het record is hoger: springers voegen
\omterm{def:g11:work-of-force:work}{arbeid} toe met armen en benen op de buigende
stok. Maar de \omterm{def:g10:energy-conservation:energy}{energie}schatting zet de schaal ---
geen sprinters sprong haalt \qty{10}{m}.
\end{solution}

\begin{solution}{pb:g12:work-and-energy:1}
\textbf{1.} $v = \sqrt{2gH} = \sqrt{2 \times 9.81 \times 45} \approx
\qty{29.7}{m/s}$: \omterm{def:g10:universal-gravitation:weight}{gewicht}
\omterm{def:g12:work-and-energy:conservative}{conservatief},
\omterm{def:g12:newtons-laws:friction}{normaalkracht} arbeidloos --- het profiel
valt weg.

\textbf{2.} Poort: $E_m = mgH = 70 \times 9.81 \times 45 \approx
\qty{3.09e4}{J}$. Afzet: $E_m = \tfrac12 \times 70 \times 26^2
\approx \qty{2.37e4}{J}$.

\textbf{3.} $\Delta E_m \approx \qty{-7.2e3}{J}$; de
\omterm{def:g10:energy-conservation:energy}{mechanische-energie}stelling rekent het toe
aan de \omterm{def:g12:work-and-energy:conservative}{niet-conservatieve krachten}:
sneeuw\omterm{def:g10:inertia:inventory}{wrijving} en luchtweerstand.

\textbf{4.} $f = \num{7.2e3}/90 \approx \qty{80}{N}$.

\textbf{5.} $\num{7.2e3}/\num{3.09e4} \approx 23\%$. Wax verlaagt
$f$, hurken verlaagt weerstand: beide krimpen $fL$, de enige
onderhandelbare term.

\textbf{6.} \omterm{def:g12:projectile-motion:freefall}{Vrije val} met horizontale
beginsnelheid --- een
\omterm{def:g12:projectile-motion:projectile}{projectiel}beweging.
$v_x = \qty{26}{m/s}$;
$v_y = \sqrt{2 \times 9.81 \times 40} \approx \qty{28.0}{m/s}$.

\textbf{7.} $v = \sqrt{26^2 + 28.0^2} = \sqrt{1461} \approx
\qty{38.2}{m/s}$; \omterm{def:g10:energy-conservation:energy}{energie}:
$v = \sqrt{v_0^2 + 2gh} = \sqrt{676 + 785}$ --- identiek. Ongeveer
\qty{138}{km/h}.

\textbf{8.} $t = v_y/g = 28.0/9.81 \approx \qty{2.9}{s}$;
$d = v_0 t \approx \qty{74}{m}$.

\textbf{9.} Nee. Lift staat loodrecht op de snelheid (arbeidloos) en
weerstand tegengesteld eraan (negatieve
\omterm{def:g11:work-of-force:work}{arbeid}): $\Delta E_m \leq 0$, dus de echte
landingssnelheid kan alleen \emph{kleiner} zijn --- de lucht rekt de
vlucht, ze voedt hem niet.

\textbf{10.} $\alpha = \arctan(28.0/26) \approx \qty{47}{\degree}$
onder de horizontale.

\textbf{11.} Hoek tot de helling: $47 - 35 = \qty{12}{\degree}$.
$v_\parallel = 38.2\cos\qty{12}{\degree} \approx \qty{37.4}{m/s}$;
$v_\perp = 38.2\sin\qty{12}{\degree} \approx \qty{7.9}{m/s}$.

\textbf{12.} Geabsorbeerd: $\tfrac12 \times 70 \times 7.9^2 \approx
\qty{2.2e3}{J}$, tegen $\tfrac12 \times 70 \times 38.2^2 \approx
\qty{5.1e4}{J}$ totaal: ongeveer $4\%$.

\textbf{13.} $h_{\text{eq}} = 7.9^2/19.62 \approx \qty{3.2}{m}$ ---
een stevige sprong van een muur. Vlak: $38.2^2/19.62 \approx
\qty{74}{m}$ --- onoverleefbaar.

\textbf{14.} Een helling evenwijdig aan de vluchtbaan houdt $v_\perp$
klein, zodat de benen \omterm{def:g10:orders-of-magnitude:unit}{meters} absorberen, niet de
hele berg \omterm{def:g11:mechanical-energy:kinetic}{kinetische energie}.

\textbf{15.} $H'_0 = v_0'^2/2g = 576/19.62 \approx \qty{29.4}{m}$.

\textbf{16.} $mgH' - f\,\dfrac{H'}{\sin\qty{35}{\degree}} =
\tfrac12 m v_0'^2$.

\textbf{17.} $H' = \dfrac{\tfrac12 \times 70 \times 576}
{686.7 - 80/0.574} = \dfrac{\num{20160}}{547} \approx \qty{37}{m}$.

\textbf{18.} $36.8/29.4 \approx 1.25$: een toeslag van $25\%$.

\textbf{19.} Voor \qty{60}{kg}: $H' = \num{17280}/(588.6 - 139.5)
\approx \qty{38.5}{m}$. De \emph{lichtere} springer heeft de hogere
toren nodig: de \omterm{def:g10:inertia:inventory}{wrijving}srekening $fL$ is
hetzelfde, maar haar \omterm{def:g10:energy-conservation:energy}{energie}budget $mgH'$ is
kleiner.

\textbf{20.} Kondig een toren van \qty{37}{m} aan voor
\qty{24}{m/s} afzet --- en bekent dat een kwart ervan, zo'n
\qty{7}{m} beton, de toeslag van
\omterm{def:g10:inertia:inventory}{wrijving} is op de wrijvingsloze droom.
\end{solution}
