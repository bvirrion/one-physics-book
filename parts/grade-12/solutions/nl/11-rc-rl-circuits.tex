\section*{Hoofdstuk \ref{ch:g12:rc-rl-circuits} --- RC- en RL-circuits}

\begin{solution}{exo:g12:rc-rl-circuits:1}
$q = Cu = 4.7\times10^{-4} \times 12 \approx \qty{5.6}{mC}$;
$u = q/C = 2.0\times10^{-3}/10^{-4} = \qty{20}{V}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:2}
\qty{47}{pF}: radio-afstemmer; \qty{2200}{\micro F}: afvlakker;
\qty{10}{F}: supercondensator. $E = \tfrac12 \times 10 \times 2.7^2
\approx \qty{36}{J}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:3}
$i = C\,\Delta u/\Delta t = 2.2\times10^{-4} \times 5.0/0.010 =
\qty{0.11}{A}$. Daarna $i = C\,du/dt = 0$: een geladen
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} is een open circuit --- geen
stationaire stroom.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:4}
$RC = 10^4 \times 10^{-4} = \qty{1.0}{s}$;
$L/R = 0.10/50 = \qty{2.0}{ms}$. $\unit{\ohm}\,\unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$;
$\unit{H}/\unit{\ohm} = (\unit{V.s/A})/(\unit{V/A}) = \unit{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:5}
$3.8/6.0 = 0.63$, dus $\tau = \qty{2.0}{s}$; $95\%$ bij $3\tau =
\qty{6.0}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:6}
$\tau = 4700 \times 2.2\times10^{-4} \approx \qty{1.0}{s}$;
$u(\tau) = 0.63 \times 9.0 = \qty{5.7}{V}$;
$u(3\tau) = 0.95 \times 9.0 = \qty{8.6}{V}$;
$I_0 = \mathcal{E}/R = 9.0/4700 \approx \qty{1.9}{mA}$ (lege
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator}: heel $\mathcal{E}$ staat over $R$).
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:7}
$du/dt = (\mathcal{E}/RC)e^{-t/RC}$, dus $RC\,du/dt + u =
\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$.
$u(0) = 0$: de \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} start leeg;
$u \to \mathcal{E}$: vol geladen, de stroom is gestopt.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:8}
$\tau = 4.7\times10^{4} \times 10^{-4} = \qty{4.7}{s}$;
$u(\tau) = 12\,e^{-1} \approx \qty{4.4}{V}$; \qty{0.60}{V} is $5\%$,
bereikt bij $3\tau \approx \qty{14}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:9}
$u = L\,\Delta i/\Delta t = 0.50 \times 2.0/0.040 = \qty{25}{V}$.
Bij sluiten klimt $i$ op het eigen tempo van het circuit, $di/dt$
bescheiden; openen \emph{\omterm{def:g10:inertia:force}{dwingt}} $i$ vrijwel
onmiddellijk naar nul, $di/dt$ is enorm en $u$ bereikt vonkwaarden.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:10}
$di/dt = (\mathcal{E}/L)e^{-Rt/L}$, dus $L\,di/dt + Ri =
\mathcal{E}e^{-Rt/L} + \mathcal{E}(1 - e^{-Rt/L}) = \mathcal{E}$.
$i(\tau) = 0.63 \times 2.0 = \qty{1.3}{A}$;
$i(3\tau) = 0.95 \times 2.0 = \qty{1.9}{A}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:11}
$u'(t) = (\mathcal{E}/\tau)e^{-t/\tau}$, dus $u'(0) = \mathcal{E}/\tau$:
de raaklijn $u = (\mathcal{E}/\tau)\,t$ bereikt $\mathcal{E}$ bij
$t = \tau$. $C = \tau/R = 0.50/2000 = \qty{250}{\micro F}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:12}
Supercondensator $\tfrac12 \times 3000 \times 2.7^2 \approx
\qty{1.1e4}{J}$; flits $\tfrac12 \times 8.0\times10^{-4} \times 300^2
= \qty{36}{J}$; \omterm{def:g11:magnetic-fields:solenoid}{spoel}
$\tfrac12 \times 0.010 \times 10^2 = \qty{0.50}{J}$. Batterij
($\qty{1.3e4}{J}$) $>$ supercondensator $\gg$ flits $\gg$
\omterm{def:g11:magnetic-fields:solenoid}{spoel} --- maar alle drie kunnen hun
\omterm{def:g10:energy-conservation:energy}{energie} in milliseconden dumpen:
\omterm{def:g11:work-of-force:power}{vermogen}, niet \omterm{def:g10:energy-conservation:energy}{energie}.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:13}
$\tau = \qty{5.0}{s}$, dus $R = \tau/C = 5.0/10^{-4} = \qty{50}{k\ohm}$.
$1 - e^{-t/\tau} = 0.50$ geeft $e^{-t/\tau} = 0.50$, $t = \tau\ln 2
\approx 0.69\,\tau = \qty{3.5}{s}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:14}
$\Delta u = i\,\Delta t/C = 0.50 \times 0.010/2.2\times10^{-3} \approx
\qty{2.3}{V}$. Nodig $C \geq i\,\Delta t/\Delta u =
5.0\times10^{-3}/0.50 = \qty{10}{mF}$.
\end{solution}

\begin{solution}{exo:g12:rc-rl-circuits:15}
$u \approx L\,\Delta i/\Delta t = 0.80 \times 2.0/10^{-3} =
\qty{1.6}{kV}$; $E_L = \tfrac12 \times 0.80 \times 2.0^2 =
\qty{1.6}{J}$. Langzaam afsnijden, $di/dt$ blijft klein, $u$ blijft
enkele \omterm{def:g11:circuits-and-power:voltage}{volts}: geen vonk.
\end{solution}

\begin{solution}{pb:g12:rc-rl-circuits:1}
\textbf{1.} $q = CU_0 = 8.0\times10^{-4} \times 300 = \qty{0.24}{C}$.

\textbf{2.} $N = 0.24/1.60\times10^{-19} = \num{1.5e18}$.

\textbf{3.} $E_C = \tfrac12 \times 8.0\times10^{-4} \times 300^2 =
\qty{36}{J}$.

\textbf{4.} $\num{1.3e4}/36 \approx 360$ maal.

\textbf{5.} De cel slaat volop op maar levert traag; de
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} zet bescheiden
\omterm{def:g10:energy-conservation:energy}{energie} om in enorm
\emph{\omterm{def:g11:work-of-force:power}{vermogen}}.

\textbf{6.} $\mathcal{E} = Ri + u$ met $i = C\,du/dt$:
$RC\,du/dt + u = \mathcal{E}$.

\textbf{7.} $du/dt = (\mathcal{E}/RC)e^{-t/RC}$:
$\mathcal{E}e^{-t/RC} + \mathcal{E}(1 - e^{-t/RC}) = \mathcal{E}$
voor alle $t$, en $u(0) = 0$.

\textbf{8.} $\tau = RC = 10^3 \times 8.0\times10^{-4} = \qty{0.80}{s}$.

\textbf{9.} Bij $3\tau = \qty{2.4}{s}$, $u = 0.95 \times 300 =
\qty{285}{V}$ --- het twee-seconden-gejank.

\textbf{10.} $I_0 = \mathcal{E}/R = \qty{0.30}{A}$; als $u$ klimt,
krimpt de rest $\mathcal{E} - u$ over $R$, dus
$i = (\mathcal{E}-u)/R$ sterft weg.

\textbf{11.} $u(t) = U_0\,e^{-t/\tau'}$ met $\tau' = R_f C =
0.50 \times 8.0\times10^{-4} = \qty{0.40}{ms}$.

\textbf{12.} $I_0 = U_0/R_f = 300/0.50 = \qty{600}{A}$.

\textbf{13.} $P_0 = U_0 I_0 = 300 \times 600 = \qty{180}{kW}$ ---
ongeveer $80$ waterkokers tegelijk.

\textbf{14.} Duur $3\tau' = \qty{1.2}{ms}$; gemiddeld
\omterm{def:g11:work-of-force:power}{vermogen} $\approx 36/1.2\times10^{-3} =
\qty{30}{kW}$.

\textbf{15.} $2.4/1.2\times10^{-3} = 2000$. Alleen de
\omterm{def:g11:circuits-and-power:resistance}{weerstand} veranderde (\qty{1.0}{k\ohm}
in, \qty{0.50}{\ohm} uit): een
\omterm{def:g11:work-of-force:power}{vermogen}shendel.

\textbf{16.} $C = 2E_C/U_0^2 = 400/1500^2 \approx \qty{1.8e-4}{F} =
\qty{180}{\micro F}$.

\textbf{17.} $q = CU_0 = 1.78\times10^{-4} \times 1500 \approx
\qty{0.27}{C}$.

\textbf{18.} $\tau = 50 \times 1.78\times10^{-4} \approx \qty{8.9}{ms}$,
duur $3\tau \approx \qty{27}{ms}$ --- zo'n twintig maal de flits.

\textbf{19.} $I_0 = 1500/50 = \qty{30}{A}$;
$P_0 = 1500 \times 30 = \qty{45}{kW}$.

\textbf{20.} $3\tau_c = \qty{6.0}{s}$ geeft $\tau_c = \qty{2.0}{s}$, dus
$R_c = \tau_c/C = 2.0/1.78\times10^{-4} \approx \qty{11}{k\ohm}$ en
$I_0 = 1500/1.12\times10^{4} \approx \qty{0.13}{A}$ --- klaar in zes
seconden, op een tiende \omterm{def:g11:circuits-and-power:current}{ampère}.
\end{solution}
