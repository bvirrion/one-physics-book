\section*{Hoofdstuk \ref{ch:g12:satellites-kepler} --- Satellieten en planetaire beweging}

\begin{solution}{exo:g12:satellites-kepler:1}
Op het platform: $F = GMm/R^2 = \num{3.98e17}/\num{4.06e13} \approx
\qty{9.8e3}{N}$. Op $r = \qty{6.77e6}{m}$:
$F \approx \qty{8.7e3}{N}$ --- nog $89\%$ van de grondwaarde. De
inzittenden zweven omdat ze \emph{samen} met het station vallen, niet
omdat zwaartekracht weg is.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:2}
$v = \sqrt{GM/R} = \sqrt{\num{3.98e14}/\num{6.37e6}} \approx
\qty{7.9e3}{m/s}$; $T = 2\pi R/v \approx \qty{5.1e3}{s} \approx
\qty{84}{min}$. Precies de \qty{7.9}{km/s} van de kanonskogel: een
scheurende \omterm{def:g10:universal-gravitation:satellite}{baan} \emph{is} Newtons
schot.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:3}
$v \propto 1/\sqrt{r}$: gedeeld door $\sqrt 2 \approx 1.41$.
$T \propto r^{3/2}$: vermenigvuldigd met $2^{3/2} \approx 2.8$.
Geen van beide bevat $m$: nee.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:4}
$v = \sqrt{\num{3.98e14}/\num{6.77e6}} \approx \qty{7.67e3}{m/s}$;
$T = 2\pi r/v \approx \qty{5.55e3}{s} \approx \qty{92}{min}$;
$\num{86400}/5546 \approx 15.6$
\omterm{def:g10:universal-gravitation:satellite}{banen} per dag.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:5}
\omterm{def:g12:satellites-kepler:ellipse}{Ellipsen} met de Zon in een brandpunt;
gelijke oppervlakten in gelijke tijden; $T^2/a^3$ identiek voor alle
planeten. Snelst bij \omterm{def:g12:satellites-kepler:ellipse}{perihelium}, door
de tweede wet (kort veegsegment moet snel vegen). Nee: de massa van
de \omterm{def:g10:universal-gravitation:satellite}{satelliet} valt weg --- alleen
de centrale massa komt binnen.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:6}
$T = \qty{2.36e6}{s}$: $M = 4\pi^2 r^3/(GT^2) =
\num{2.24e27}/371 \approx \qty{6.0e24}{kg}$ --- binnen $1\%$ van
\qty{5.97e24}{kg}.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:7}
Zwaartekracht wijst naar het middelpunt van de Aarde, en in
circulaire beweging wijst de
\omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} naar
het middelpunt van de cirkel: de twee middelpunten vallen samen. Een
cirkel boven de parallel van Parijs is gecentreerd op de as, niet
het middelpunt van de Aarde --- onmogelijk. Stilhangen
\omterm{def:g11:work-of-force:work}{werkt} alleen waar de parallel een
grote cirkel is gecentreerd in het middelpunt: over de
\emph{evenaar}.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:8}
$T = \qty{2.754e4}{s}$: $M = 4\pi^2 r^3/(GT^2) =
\num{3.26e22}/\num{5.06e-2} \approx \qty{6.4e23}{kg}$ --- de Mars
van de datakaart.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:9}
Aarde: $T^2/a^3 = \num{9.96e14}/\num{3.35e33} =
\qty{2.98e-19}{s^2/m^3}$. Mars: $T = \qty{5.94e7}{s}$,
$\num{3.52e15}/\num{1.18e34} = \qty{2.98e-19}{s^2/m^3}$ --- gelijk,
zoals de derde wet eist. Dan $M = 4\pi^2/(G \times \num{2.98e-19})
\approx \qty{1.99e30}{kg}$.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:10}
$v = 2\pi r/T = \qty{1.02e3}{m/s}$; $a = v^2/r =
\qty{2.72e-3}{m/s^2}$; $g/60^2 = 9.81/3600 = \qty{2.73e-3}{m/s^2}$.
De Maan zit $60$ aardstralen uit, en zijn val is $60^2$ maal
zwakker dan die van de appel: precies het omgekeerde kwadraat ---
één wet voor beide.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:11}
Ceres: $T = 2.77^{3/2} = \sqrt{21.3} \approx \qty{4.6}{yr}$. Halley:
$T = 17.8^{3/2} \approx \qty{75}{yr}$, dus volgende
\omterm{def:g12:satellites-kepler:ellipse}{perihelium} rond
$1986 + 75 \approx 2061$.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:12}
Op \qty{400}{km}: $v \approx \qty{7.67e3}{m/s}$; op \qty{350}{km}
($r = \qty{6.72e6}{m}$): $v \approx \qty{7.70e3}{m/s}$.
\omterm{def:g10:inertia:inventory}{Wrijving} drainde wel
\omterm{def:g11:mechanical-energy:em}{mechanische energie} --- de
\omterm{def:g10:universal-gravitation:satellite}{baan} krimpt; maar op de weg
omlaag betaalt het \omterm{def:g11:work-of-force:work}{werk} van de
zwaartekracht meer dan het snelheidsverlies aan sleep terug: $E_m$
daalt terwijl $E_k$ stijgt.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:13}
$T = \qty{3.024e5}{s}$: $M = 4\pi^2 r^3/(GT^2) =
\num{1.65e31}/6.10 \approx \qty{2.7e30}{kg} \approx 1.4$ Zonnen. De
massa van de \emph{planeet} is buiten bereik: hij viel weg uit de
\omterm{def:g10:universal-gravitation:satellite}{baan}vergelijkingen.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:14}
$T = \qty{8.862e4}{s}$, $GM = \qty{4.28e13}{m^3/s^2}$:
$r^3 = GMT^2/(4\pi^2) = \num{8.52e21}$, dus $r \approx
\qty{2.04e7}{m}$ en $h = r - R \approx \qty{1.7e7}{m} \approx
\qty{17000}{km}$ boven de Martiaanse grond.
\end{solution}

\begin{solution}{exo:g12:satellites-kepler:15}
$v_a = r_p v_p / r_a = \num{8.77e10} \times \num{5.45e4} /
\num{5.25e12} \approx \qty{9.1e2}{m/s}$. Omdat $r_a/r_p \approx 60$
is de komeet $60$ maal verder en $60$ maal trager: hij brengt
decennia treuzelend in het donker door en weken sprintend langs de
Zon.
\end{solution}

\begin{solution}{pb:g12:satellites-kepler:1}
\textbf{1.} $T = \qty{86164}{s}$: de grond draait met de Aarde
\emph{ten opzichte van de sterren} éénmaal per
\omterm{def:g12:satellites-kepler:geo}{siderische dag}; de
\qty{24}{h}-dag voegt de graad per dag toe die de Aarde om de Zon
opschuift.

\textbf{2.} De \omterm{def:g11:forces-and-motion:netforce}{resulterende
kracht} van circulaire beweging richt zich op het middelpunt van de
cirkel; de enige \omterm{def:g10:inertia:force}{kracht}, zwaartekracht,
richt zich op het middelpunt van de Aarde: de twee middelpunten
zijn hetzelfde punt.

\textbf{3.} Een Parijs-hangende cirkel zou op de as noord van het
middelpunt zijn gecentreerd --- uitgesloten door 2. De
\omterm{def:g10:universal-gravitation:satellite}{baan} moet in het
\emph{equatoriale vlak} liggen.

\textbf{4.} Normale component van
\omterm{thm:g12:newtons-laws:second}{Newtons tweede wet}:
$m v^2/r = GmM/r^2$, dus $v = \sqrt{GM/r}$ ($m$ valt weg).

\textbf{5.} $T = 2\pi r/v = 2\pi\sqrt{r^3/(GM)}$; kwadraterend,
$T^2/r^3 = 4\pi^2/(GM)$.

\textbf{6.} $r = \left(GMT^2/4\pi^2\right)^{1/3} =
(\num{7.49e22})^{1/3} \approx \qty{4.22e7}{m}$.

\textbf{7.} $h = r - R = \num{4.22e7} - \num{6.37e6} \approx
\qty{3.58e7}{m} \approx \qty{35800}{km}$.

\textbf{8.} $v = 2\pi \times \num{4.22e7}/\num{86164} \approx
\qty{3.07e3}{m/s}$.

\textbf{9.} $\sqrt{\num{3.98e14}/\num{4.22e7}} \approx
\qty{3.07e3}{m/s}$ --- consistent.

\textbf{10.} Identieke stralen: de massa viel weg in 4. De zwaardere
\omterm{def:g10:universal-gravitation:satellite}{satelliet} verandert alleen de
lanceerrekening --- meer brandstof voor dezelfde
\omterm{def:g10:universal-gravitation:satellite}{baan}.

\textbf{11.} $v = \sqrt{\num{3.98e14}/\num{6.77e6}} \approx
\qty{7.67e3}{m/s}$.

\textbf{12.} $T = 2\pi r/v \approx \qty{5.55e3}{s} \approx
\qty{92.4}{min}$.

\textbf{13.} $\num{86400}/5546 \approx 15.6$
\omterm{def:g10:universal-gravitation:satellite}{banen}: ongeveer $16$
zonsopgangen per dag.

\textbf{14.} $r_{\text{geo}}/r_{\text{ISS}} = \num{4.22e7}/\num{6.77e6}
= 6.23$; $6.23^{3/2} \approx 15.5 = \num{86164}/5546$ --- Keplers
derde wet, ter plaatse geverifieerd.

\textbf{15.} Het station is sneller (\qty{7.67}{km/s} tegen
\qty{3.07}{km/s}), toch kostte de
\omterm{def:g12:satellites-kepler:geo}{geostationaire} meer
\omterm{def:g10:energy-conservation:energy}{energie} per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram}: de klim in
potentiële \omterm{def:g10:energy-conservation:energy}{energie} overtreft
het snelheidsverlies.

\textbf{16.} $M_J = 4\pi^2 r^3/(G T^2)$.

\textbf{17.} $T = \qty{1.529e5}{s}$: $M_J =
\num{2.97e27}/1.56 \approx \qty{1.90e27}{kg}$.

\textbf{18.} $\num{1.90e27}/\num{5.97e24} \approx 318$ Aardes ---
en ongeveer $1/1000$ van de Zon.

\textbf{19.} $T_J = \qty{3.576e4}{s}$: $r^3 = GM_J T_J^2/(4\pi^2) =
\num{4.11e24}$, $r \approx \qty{1.60e8}{m}$ --- Io, op
\qty{4.22e8}{m}, vliegt $2.6$ maal hoger en drijft over Jupiters
hemel.

\textbf{20.} Parkeer de relais op de equatoriale cirkel van hoogte
\qty{35800}{km}, waar hij met \qty{3.07}{km/s} rijdt en de mik van
de schotel nooit verlaat; en dezelfde wet, gevoed met Io's maand,
woog Jupiter op $\qty{1.90e27}{kg}$ --- $318$ Aardes --- gratis.
\end{solution}
