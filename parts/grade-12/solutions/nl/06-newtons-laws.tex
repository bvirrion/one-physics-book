\section*{Hoofdstuk \ref{ch:g12:newtons-laws} --- De wetten van Newton}

\begin{solution}{exo:g12:newtons-laws:1}
(a) $p = 70 \times 10 = \qty{7.0e2}{kg.m/s}$.
(b) $v = \qty{13.9}{m/s}$: $p \approx \qty{1.8e4}{kg.m/s}$.
(c) $p = 0.0080 \times 800 = \qty{6.4}{kg.m/s}$. Auto $>$ sprinter $>$
kogel: vijf ordegrootten massa zijn meer dan snelheid kan kopen.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:2}
$N = m(g + a_z)$: (a) $70 \times 11.01 = \qty{771}{N}$;
(b) $a_z = 0$, dus $N = mg = \qty{687}{N}$ --- de eerste wet:
constante snelheid, \omterm{def:g10:inertia:force}{krachten}
\omterm{def:g10:inertia:compensate}{heffen op}; (c)
$70 \times 8.61 = \qty{603}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:3}
$a = g\sin 30^\circ = \qty{4.9}{m/s^2}$;
$v = \sqrt{2 \times 4.905 \times 3.0} = \qty{5.4}{m/s}$. De massa
valt weg tussen $ma$ en $mg\sin\alpha$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:4}
Omkering: $\Delta p = 0.058 \times (25 + 15) = \qty{2.3}{kg.m/s}$;
$F = 2.32/0.0050 = \qty{4.6e2}{N}$ --- ongeveer $800$ maal het
\qty{0.57}{N}-\omterm{def:g10:universal-gravitation:weight}{gewicht},
waarom zwaartekracht tijdens contact wordt genegeerd.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:5}
\omterm{def:g10:universal-gravitation:weight}{Gewicht} (Aarde op
boek, omlaag): partner = boek trekt de Aarde, omhoog.
\omterm{def:g12:newtons-laws:friction}{Normaalkracht} (tafel op
boek, omhoog): partner = boek drukt de tafel, omlaag.
\omterm{def:g10:universal-gravitation:weight}{Gewicht} en normaal
werken op het \emph{zelfde} lichaam en behoren tot twee
verschillende interacties --- hier gelijk alleen omdat
$\vect a = \vect 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:6}
$N = mg = \qty{490}{N}$. (a) $F > \mu_s N = \qty{221}{N}$.
(b) $a = (220.7 - 0.35 \times 490.5)/50 = \qty{0.98}{m/s^2}$.
(c) $F = \mu_k N = \qty{172}{N}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:7}
(a) Glijdt wanneer $\tan\alpha > \mu_s$: $\alpha = \arctan 0.45
\approx 24^\circ$. (b) $a = g(\sin 30^\circ - 0.35\cos 30^\circ) =
9.81 \times 0.197 = \qty{1.9}{m/s^2}$. (c) Ja: eenmaal glijdend
daalt \omterm{def:g10:inertia:inventory}{wrijving} tot
$\mu_k < \mu_s$, dus
$a = g\cos\alpha\,(\tan\alpha - \mu_k) \approx \qty{0.9}{m/s^2}
> 0$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:8}
$a = m_2 g/(m_1 + m_2) = 19.62/5.0 = \qty{3.9}{m/s^2}$;
$T = m_1 a = \qty{11.8}{N}$. Het hangende blok versnelt omlaag,
dus wijst de \omterm{def:g11:forces-and-motion:netforce}{resulterende
kracht} erop omlaag: $T$ moet $m_2 g$ ondermijnen.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:9}
$v_{\max} = \sqrt{\mu_s g R}$: droog
$\sqrt{0.70 \times 9.81 \times 50} = \qty{18.5}{m/s} \approx
\qty{67}{km/h}$; nat $\qty{14.0}{m/s} = \qty{50}{km/h}$. Zowel
$mv^2/R$ als $\mu_s mg$ dragen $m$: hij valt weg.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:10}
Verticaal: $T\cos\theta = mg$, dus
$T = 2.943/\cos 25^\circ = \qty{3.2}{N}$. Straal $R = L\sin\theta =
\qty{0.507}{m}$; radiaal: $T\sin\theta = mv^2/R$, dus
$v = \sqrt{gR\tan\theta} = \sqrt{9.81 \times 0.507 \times 0.466}
= \qty{1.5}{m/s}$.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:11}
$a_z = (N - mg)/m$ met $mg = \qty{49.05}{N}$: $+\qty{1.0}{m/s^2}$,
$0$, $-\qty{1.0}{m/s^2}$. Bijv.\ een omhoog-rit: versnellen,
cruisen, remmen. Ja: al dalend binnengekomen betekenen dezelfde
aflezingen remmen, cruisen, opnieuw omlaag versnellen --- de
weegschaal leest versnelling, niet snelheid.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:12}
(a) $0 = 60V + 4.0 \times 8.0$: $V = \qty{-0.53}{m/s}$.
(b) Bal bij $8.0 + V$ over de grond: $0 = 60V + 4.0(8.0 + V)$, dus
$V = -32/64 = \qty{-0.50}{m/s}$ en de bal vliegt met \qty{7.5}{m/s}.
(c) De duw van de bal op haar handen --- de derde-wet-partner van
haar worp.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:13}
Wrijvingsloos: $N\cos\theta = mg$ (verticaal) en $N\sin\theta =
mv^2/R$ (radiaal); delend, $\tan\theta = v^2/(gR) = 400/784.8$:
$\theta \approx 27^\circ$. Trager: de auto neigt de helling af te
glijden, \omterm{def:g10:inertia:inventory}{wrijving} moet helling
op wijzen; sneller: het omgekeerde.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:14}
(a) $250 \times 2500 = \qty{6.3e5}{kg.m/s}$ elke seconde:
\omterm{def:g10:inertia:inventory}{stuwkracht} $\qty{6.3e5}{N}$.
(b) $P = \qty{4.9e5}{N}$;
$a = (6.25 - 4.905) \times 10^5 / \num{5.0e4} = \qty{2.7}{m/s^2}$.
(c) Hij duwt op zijn eigen uitlaat (derde wet), niet op de lucht.
\end{solution}

\begin{solution}{exo:g12:newtons-laws:15}
(a) $m_2 g > \mu_s m_1 g$: $m_2 > 0.35 \times 4.0 = \qty{1.4}{kg}$.
(b) $a = (m_2 - \mu_k m_1)\,g/(m_1 + m_2) = 1.0 \times 9.81/6.0 =
\qty{1.6}{m/s^2}$; $T = m_2(g - a) = \qty{16.4}{N}$.
(c) $\mu_k = 0$ geeft $a = m_2 g/(m_1 + m_2)$: de formule van
\cref{exo:g12:newtons-laws:8}.
\end{solution}

\begin{solution}{pb:g12:newtons-laws:1}
\textbf{1.} Systeem: cabine $+$ wagen ($+$ passagiers). De berg is
vast aan de grond, inertiaal tot uitstekende nauwkeurigheid.
\omterm{def:g10:inertia:force}{Krachten}:
\omterm{def:g10:universal-gravitation:weight}{gewicht} $M\vect g$
(omlaag), \omterm{def:g12:newtons-laws:friction}{normaalkracht}
$\vect N$ van de baankabel (loodrecht erop), trek-
\omterm{def:g10:inertia:inventory}{spanning} $\vect T$ (de helling
op).

\textbf{2.} In rust, de eerste wet: $\sum \vect F = \vect 0$. Langs
de helling: $T = Mg\sin\alpha$; loodrecht: $N = Mg\cos\alpha$.

\textbf{3.} $T = 4000 \times 9.81 \times 0.500 = \qty{19.6}{kN}$;
$N = 4000 \times 9.81 \times 0.866 = \qty{34.0}{kN}$.

\textbf{4.} Leeg: $T = \qty{9.8}{kN}$. Elke
\omterm{def:g10:inertia:force}{kracht} in de balans is evenredig met
$M$, dus $T$ ook.

\textbf{5.} Langs de helling, $Ma = Mg\sin\alpha$: $M$ valt weg,
$a = g\sin\alpha = \qty{4.9}{m/s^2}$ de helling af.

\textbf{6.} $T' = M(g\sin\alpha + a) = 4000 \times 5.505 =
\qty{22.0}{kN}$.

\textbf{7.} $a/(g\sin\alpha) = 0.60/4.905 \approx 12\%$.

\textbf{8.} $t = v/a = \qty{10}{s}$; $d = v^2/(2a) = \qty{30}{m}$.

\textbf{9.} Verticale component van $\vect a$: $a\sin\alpha =
\qty{0.30}{m/s^2}$, dus $N = m(g + a\sin\alpha) = 80 \times 10.11 =
\qty{809}{N}$ tegen $mg = \qty{785}{N}$: $3\%$ zwaarder.

\textbf{10.} De statische
\omterm{def:g10:inertia:inventory}{wrijving} van de vloer,
horizontaal: $ma\cos\alpha = 80 \times 0.52 \approx \qty{42}{N}$.
Leunen zet de gecombineerde vloerdduw door haar
\omterm{def:g11:forces-and-motion:com}{zwaartepunt}, zodat hij haar
niet kantelt.

\textbf{11.} $a_T = dv/dt = 0$ (constante snelheid);
$a_N = v^2/R = 36/40 = \qty{0.90}{m/s^2}$, gericht op het centrum
--- recht omlaag bij de kam.

\textbf{12.} Normaals (omlaag positief): $Mg - N' = Mv^2/R$, dus
$N' = M(g - v^2/R) = 4000 \times 8.91 = \qty{35.6}{kN}$.

\textbf{13.} $N = m(g - v^2/R) = 80 \times 8.91 = \qty{713}{N}$:
lichter met $v^2/(gR) = 0.90/9.81 \approx 9\%$.

\textbf{14.} $N' = 0$ bij $v = \sqrt{gR} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$ --- meer dan driemaal de cruise: de wagen rijdt de
kabel met ruime marge.

\textbf{15.} \omterm{thm:g12:newtons-laws:third}{Derde wet van Newton}:
\qty{35.6}{kN}, omlaag drukkend op de kabel (dus op de toren) ---
iets minder dan het rustende
\omterm{def:g10:universal-gravitation:weight}{gewicht}, verlicht
door de gekromde vlucht.

\textbf{16.} $F = \Delta p/\Delta t = 4000 \times 6.0/1.5 =
\qty{16}{kN}$, tegengesteld aan de beweging
($a = \qty{4.0}{m/s^2}$).

\textbf{17.} $f = ma = 80 \times 4.0 = \qty{320}{N}$; het plafond is
$\mu_s mg = 0.40 \times 785 = \qty{314}{N} < \qty{320}{N}$: voeten
alleen glijden --- houd de leuning vast.

\textbf{18.} $0 = M_0 V + m(v_{\text{rel}} + V)$:
$V = -80 \times 2.5/2080 = \qty{-0.096}{m/s}$; de passagier beweegt
met $2.5 - 0.096 \approx \qty{2.4}{m/s}$ over de grond.

\textbf{19.} De raket: zijn
\omterm{def:g10:inertia:inventory}{stuwkracht} wordt vastgelegd door
de per seconde uitgeworpen massa maal de uitwerpsnelheid ten
opzichte ervan.

\textbf{20.} De grootste
\omterm{def:g10:inertia:inventory}{spanning} die we ontmoetten is
het opstarten $T' = M \times \qty{5.505}{N/kg}$ (rust geeft
$Mg\sin\alpha$, de kam nog minder). Cap: $T' \leq \qty{20}{kN}$,
dus $M \leq 20000/5.505 = \qty{3.63e3}{kg}$: payload
\qty{1.63e3}{kg}, d.w.z.\ $20$ passagiers van \qty{80}{kg} (niet
$25$). Bordje: \emph{20 personen}; dan $M = \qty{3.6e3}{kg}$,
$T' = \qty{19.8}{kN}$, veiligheidsfactor $100/19.8 \approx 5.1
\geq 5$.
\end{solution}
