\section*{Chapter \ref{ch:g12:work-and-energy} --- Work and Mechanical Energy}

\begin{solution}{exo:g12:work-and-energy:1}
$E_p = \tfrac12 \times 200 \times 0.050^2 = \qty{0.25}{J}$. Doubled
stretch: $\times 4$, i.e.\ \qty{1.0}{J}. The second \qty{5.0}{cm}
alone costs $1.0 - 0.25 = \qty{0.75}{J}$ --- the \omterm{def:g10:inertia:force}{force} is larger there.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:2}
$W = mg(z_A - z_B) = 0.90 \times 9.81 \times (12 - 30) \approx
\qty{-1.6e2}{J}$, whatever the route --- the same on the straight
climb. Round trip: \qty{0}{J} (\omterm{def:g12:work-and-energy:conservative}{conservative force}).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:3}
(a), (b) \omterm{def:g12:work-and-energy:conservative}{conservative}: \omterm{def:g11:work-of-force:work}{work} fixed by the endpoints (drop of altitude;
stretch). (c), (e) \omterm{def:g12:work-and-energy:conservative}{non-conservative}: $-fL$ and drag charge per \omterm{def:g10:orders-of-magnitude:unit}{metre}
travelled. (d) workless: always perpendicular to the motion.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:4}
$h = L(1 - \cos\qty{60}{\degree}) = \qty{1.0}{m}$;
$v = \sqrt{2 \times 9.81 \times 1.0} \approx \qty{4.4}{m/s}$. The
\omterm{def:g10:inertia:inventory}{tension} is perpendicular to the motion at every instant: zero \omterm{def:g11:work-of-force:power}{power},
zero \omterm{def:g11:work-of-force:work}{work}.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:5}
$F = P/v = 250/9.0 \approx \qty{28}{N}$. At \qty{5.0}{m/s}:
$F = 250/5.0 = \qty{50}{N}$ (mostly the slope's gravity component).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:6}
$E_p = \tfrac12 \times 450 \times 0.080^2 = \qty{1.44}{J}$;
$v = \sqrt{2 \times 1.44/0.020} = \qty{12}{m/s}$;
$h = E_p/mg = 1.44/(0.020 \times 9.81) \approx \qty{7.3}{m}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:7}
Trapezoids (\qty{2.0}{cm} steps): $(0.03 + 0.10 + 0.19 + 0.30) =
\qty{0.62}{J}$. Straight-line model: $k = 18.0/0.080 =
\qty{225}{N/m}$, $\tfrac12 kx^2 = \qty{0.72}{J}$. The measured curve
sags below the straight line at small $x$, so the true area is
smaller.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:8}
$\Delta E_m = \tfrac12 \times 55 \times 9.0^2 - 55 \times 9.81 \times
8.0 = 2228 - 4316 \approx \qty{-2.1e3}{J}$;
$f = 2089/25 \approx \qty{84}{N}$. \omterm{def:g10:universal-gravitation:weight}{Weight} \omterm{def:g12:work-and-energy:conservative}{conservative}, \omterm{def:g12:newtons-laws:friction}{normal force}
workless: only the drop and the path \emph{length} enter, not the
shape.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:9}
$H_{\min} = \tfrac52 R = \qty{0.50}{m}$. From \qty{60}{cm}:
$v_{\text{top}} = \sqrt{2 \times 9.81 \times (0.60 - 0.40)} \approx
\qty{2.0}{m/s}$, and $v_{\text{top}}^2/(gR) = 2.0$ --- double the
minimum. \omterm{def:g10:inertia:inventory}{Friction} skims \omterm{def:g10:energy-conservation:energy}{energy} all along, so real tracks need extra
height.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:10}
\omterm{def:g12:work-and-energy:diagram}{Turning points}: $8.0\,x^2 = 2.0$, $x = \pm\qty{0.50}{m}$. Maximum
speed at $x = 0$: $v = \sqrt{2 \times 2.0/0.100} \approx
\qty{6.3}{m/s}$. A back-and-forth oscillation in a parabolic well ---
the mass-spring \omterm{def:g12:oscillators-and-time:oscillator}{oscillator}'s $E_p$ is exactly of this form.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:11}
$\tfrac12 kx^2 = \tfrac12 mv^2$:
$x = v\sqrt{m/k} = 2.0\sqrt{1200/\num{6.0e5}} \approx \qty{8.9}{cm}$.
At \qty{4.0}{m/s}: \qty{18}{cm} --- double, not quadruple: the \omterm{def:g10:energy-conservation:energy}{energy}
quadruples, but $E_p$ grows like $x^2$, so $x$ only doubles.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:12}
(a) $E_m = \qty{5.0}{J} < \qty{6.0}{J}$: it cannot pass; it stops on
the flank where $E_p = \qty{5.0}{J}$ and rolls back. (b) Hilltop:
$E_k = \qty{1.0}{J}$, $v = \sqrt{2 \times 1.0/0.50} = \qty{2.0}{m/s}$;
valley: $E_k = \qty{5.0}{J}$, $v \approx \qty{4.5}{m/s}$. (c) $x = 0$:
\omterm{prop:g12:work-and-energy:equilibria}{unstable} (maximum); $x = \qty{1.5}{m}$: \omterm{prop:g12:work-and-energy:equilibria}{stable} (minimum).
\end{solution}

\begin{solution}{exo:g12:work-and-energy:13}
$\tfrac12 mv^2 = mgR_E$:
$v = \sqrt{2 \times 9.81 \times \num{6.37e6}} \approx \qty{1.12e4}{m/s}
= \qty{11.2}{km/s}$. Every term is proportional to $m$: probe and
pebble alike. Just below it, the launch still climbs enormously far,
meets a \omterm{def:g12:work-and-energy:diagram}{turning point}, and falls back: bound.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:14}
$E_m = \tfrac12 kA^2 = \tfrac12 \times 40 \times 0.060^2 =
\qty{7.2e-2}{J}$; $v_{\max} = \sqrt{2E_m/m} =
\sqrt{2 \times 0.072/0.250} \approx \qty{0.76}{m/s}$;
$E_k = E_p$ where $\tfrac12 kx^2 = E_m/2$: $x = \pm A/\sqrt 2 \approx
\pm\qty{4.2}{cm}$.
\end{solution}

\begin{solution}{exo:g12:work-and-energy:15}
$h = v^2/2g = 9.5^2/19.62 \approx \qty{4.6}{m}$; bar $\approx
4.6 + 1.0 = \qty{5.6}{m}$. The record is higher: vaulters add \omterm{def:g11:work-of-force:work}{work}
with arms and legs on the bending pole. But the \omterm{def:g10:energy-conservation:energy}{energy} estimate sets
the scale --- no sprinter's vault will reach \qty{10}{m}.
\end{solution}

\begin{solution}{pb:g12:work-and-energy:1}
\textbf{1.} $v = \sqrt{2gH} = \sqrt{2 \times 9.81 \times 45} \approx
\qty{29.7}{m/s}$: \omterm{def:g10:universal-gravitation:weight}{weight} \omterm{def:g12:work-and-energy:conservative}{conservative}, \omterm{def:g12:newtons-laws:friction}{normal force} workless --- the
profile drops out.

\textbf{2.} Gate: $E_m = mgH = 70 \times 9.81 \times 45 \approx
\qty{3.09e4}{J}$. Takeoff: $E_m = \tfrac12 \times 70 \times 26^2
\approx \qty{2.37e4}{J}$.

\textbf{3.} $\Delta E_m \approx \qty{-7.2e3}{J}$; the
\omterm{def:g10:energy-conservation:energy}{mechanical-energy} theorem charges it to the \omterm{def:g12:work-and-energy:conservative}{non-conservative forces}:
snow \omterm{def:g10:inertia:inventory}{friction} and air drag.

\textbf{4.} $f = \num{7.2e3}/90 \approx \qty{80}{N}$.

\textbf{5.} $\num{7.2e3}/\num{3.09e4} \approx 23\%$. Wax lowers $f$,
crouching lowers drag: both shrink $fL$, the only negotiable term.

\textbf{6.} \omterm{def:g12:projectile-motion:freefall}{Free fall} with horizontal initial velocity --- a
\omterm{def:g12:projectile-motion:projectile}{projectile} motion. $v_x = \qty{26}{m/s}$;
$v_y = \sqrt{2 \times 9.81 \times 40} \approx \qty{28.0}{m/s}$.

\textbf{7.} $v = \sqrt{26^2 + 28.0^2} = \sqrt{1461} \approx
\qty{38.2}{m/s}$; \omterm{def:g10:energy-conservation:energy}{energy}: $v = \sqrt{v_0^2 + 2gh} = \sqrt{676 + 785}$
--- identical. About \qty{138}{km/h}.

\textbf{8.} $t = v_y/g = 28.0/9.81 \approx \qty{2.9}{s}$;
$d = v_0 t \approx \qty{74}{m}$.

\textbf{9.} No. Lift is perpendicular to the velocity (workless) and
drag opposes it (negative \omterm{def:g11:work-of-force:work}{work}): $\Delta E_m \leq 0$, so the real
landing speed can only be \emph{smaller} --- the air stretches the
flight, it does not feed it.

\textbf{10.} $\alpha = \arctan(28.0/26) \approx \qty{47}{\degree}$
below the horizontal.

\textbf{11.} Angle to the slope: $47 - 35 = \qty{12}{\degree}$.
$v_\parallel = 38.2\cos\qty{12}{\degree} \approx \qty{37.4}{m/s}$;
$v_\perp = 38.2\sin\qty{12}{\degree} \approx \qty{7.9}{m/s}$.

\textbf{12.} Absorbed: $\tfrac12 \times 70 \times 7.9^2 \approx
\qty{2.2e3}{J}$, against $\tfrac12 \times 70 \times 38.2^2 \approx
\qty{5.1e4}{J}$ in total: about $4\%$.

\textbf{13.} $h_{\text{eq}} = 7.9^2/19.62 \approx \qty{3.2}{m}$ ---
a bold jump from a wall. Flat: $38.2^2/19.62 \approx \qty{74}{m}$ ---
unsurvivable.

\textbf{14.} A slope parallel to the flight path keeps $v_\perp$
small, so the legs absorb \omterm{def:g10:orders-of-magnitude:unit}{metres}, not the whole mountain of \omterm{def:g11:mechanical-energy:kinetic}{kinetic
energy}.

\textbf{15.} $H'_0 = v_0'^2/2g = 576/19.62 \approx \qty{29.4}{m}$.

\textbf{16.} $mgH' - f\,\dfrac{H'}{\sin\qty{35}{\degree}} =
\tfrac12 m v_0'^2$.

\textbf{17.} $H' = \dfrac{\tfrac12 \times 70 \times 576}
{686.7 - 80/0.574} = \dfrac{\num{20160}}{547} \approx \qty{37}{m}$.

\textbf{18.} $36.8/29.4 \approx 1.25$: a $25\%$ surcharge.

\textbf{19.} For \qty{60}{kg}: $H' = \num{17280}/(588.6 - 139.5)
\approx \qty{38.5}{m}$. The \emph{lighter} jumper needs the taller
tower: the \omterm{def:g10:inertia:inventory}{friction} bill $fL$ is the same, but her \omterm{def:g10:energy-conservation:energy}{energy} budget
$mgH'$ is smaller.

\textbf{20.} Announce a \qty{37}{m} tower for \qty{24}{m/s} of
takeoff --- and confess that a quarter of it, some \qty{7}{m} of
concrete, is \omterm{def:g10:inertia:inventory}{friction}'s surcharge on the frictionless dream.
\end{solution}
