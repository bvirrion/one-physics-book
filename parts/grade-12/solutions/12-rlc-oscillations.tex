\section*{Chapter \ref{ch:g12:rlc-oscillations} --- Free Electrical Oscillations: RLC}

\begin{solution}{exo:g12:rlc-oscillations:1}
(a) $\sqrt{LC} = \qty{1.0e-4}{s}$: $T_0 = \qty{0.63}{ms}$,
$f_0 \approx \qty{1.6}{kHz}$. (b) $\sqrt{LC} = \qty{3.16e-7}{s}$:
$T_0 = \qty{2.0}{\micro s}$, $f_0 \approx \qty{0.50}{MHz}$ --- a
factor $\sqrt{10^5} \approx 316$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:2}
$LC$: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times \unit{C/A} =
\unit{s} \times \unit{s} = \unit{s^2}$, so $\sqrt{LC}$ is in seconds.
$2\pi$ is a pure number --- no dimension to change.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:3}
$Q_0 = CU_0 = \qty{2.8}{\micro C}$; $E = \tfrac12 CU_0^2 =
\tfrac12 \times \num{4.7e-7} \times 36 \approx \qty{8.5}{\micro J}$.
The charge oscillates at $f_0 = 1/(2\pi\sqrt{LC})$, current a quarter
\omterm{def:g12:oscillators-and-time:oscillator}{period} out of \omterm{def:g12:oscillators-and-time:phase}{phase}, \omterm{def:g10:energy-conservation:energy}{energy} sloshing between plates and \omterm{def:g11:magnetic-fields:solenoid}{coil}, total
constant.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:4}
$Q_0 = \qty{2.0}{\micro C}$; $f_0 = 1/T_0 = \qty{250}{Hz}$;
$q(T_0/2) = \qty{-2.0}{\micro C}$; $i(0) = 0$ (cosine flat at $t=0$);
$I_0 = 2\pi \times \num{2.0e-6}/\num{4.0e-3} \approx \qty{3.1}{mA}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:5}
All in the \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} when $q = \pm Q_0$, $i = 0$ ($t = 0$, $T_0/2$,
$T_0$, \dots): the pendulum's extremes. All in the \omterm{def:g11:magnetic-fields:solenoid}{coil} when $q = 0$,
$i = \pm I_0$ ($t = T_0/4$, $3T_0/4$, \dots): the pendulum's lowest,
fastest point.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:6}
$d^2q/dt^2 = -(2\pi/T_0)^2\,Q_0\cos(2\pi t/T_0) = -(2\pi/T_0)^2 q$,
so $L\,d^2q/dt^2 + q/C = q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$ for all $t$
exactly when $(2\pi/T_0)^2 = 1/(LC)$, i.e.\ $T_0 = 2\pi\sqrt{LC}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:7}
$L = \dfrac{1}{4\pi^2 f_0^2 C} = \dfrac{1}{39.5 \times
(\num{1.98e5})^2 \times \num{2.2e-10}} \approx \qty{2.9}{mH}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:8}
$E = \tfrac12 CU_0^2 = \qty{5.0e-5}{J}$; $\tfrac12 LI_0^2 = E$ gives
$I_0 = \sqrt{2E/L} = \sqrt{\num{1.0e-2}} = \qty{0.10}{A}$, first
reached at $T_0/4 = \tfrac14 \times \qty{0.63}{ms} \approx
\qty{0.16}{ms}$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:9}
$E_C \propto \cos^2(2\pi t/T_0)$, and $\cos^2$ repeats every
\emph{half} turn: two fillings of the \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} per \omterm{def:g12:oscillators-and-time:oscillator}{period}, hence
$2f_0$. At $t = T_0/8$ the \omterm{def:g12:oscillators-and-time:phase}{phase} is $\pi/4$:
$\cos^2(\pi/4) = \tfrac12$, so $E_C = E_L = E/2$.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:10}
Ratio $3.0/4.0 = 0.75$ per \omterm{def:g12:rlc-oscillations:regimes}{pseudo-period}: sixth maximum
$4.0 \times 0.75^6 \approx \qty{0.71}{V}$. \omterm{def:g10:energy-conservation:energy}{Energy} $\propto$ \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}
squared: $0.75^2 \approx 56\%$ survives each \omterm{def:g12:oscillators-and-time:oscillator}{period}. The
\omterm{def:g12:rlc-oscillations:regimes}{pseudo-period} stays $\approx T_0 = 2\pi\sqrt{LC}$: light \omterm{def:g12:oscillators-and-time:damping}{damping}
shrinks the \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}, hardly the tempo.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:11}
$T_0 = 2\pi\sqrt{0.50 \times \num{2.0e-5}} = 2\pi \times
\num{3.16e-3} \approx \qty{20}{ms}$. Dictionary: $k = m/(LC) =
0.50/\num{1.0e-5} = \qty{5.0e4}{N/m}$; check
$2\pi\sqrt{m/k} = 2\pi\sqrt{\num{1.0e-5}}$ --- the same \qty{20}{ms}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:12}
$C = \qty{500}{pF}$: $f_0 = 1/(2\pi\sqrt{\num{1.0e-13}}) \approx
\qty{0.50}{MHz}$; $C = \qty{50}{pF}$: $f_0 \approx \qty{1.6}{MHz}$.
At fixed $L$, $f_0 = (2\pi\sqrt{L})^{-1} C^{-1/2} \propto 1/\sqrt{C}$:
tenfold $C$, only $\sqrt{10} \approx 3.2$ in $f_0$ --- as found.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:13}
$0.95^n = \tfrac12$: $n = \ln 2/\lvert\ln 0.95\rvert \approx 13.5$,
so about $14$ \omterm{def:g12:oscillators-and-time:oscillator}{periods}. \omterm{def:g12:oscillators-and-time:oscillator}{Amplitude} $\propto \sqrt{E}$: down to
$1/\sqrt{2} \approx 0.71$ of the start. At \qty{1.0}{MHz},
$T_0 = \qty{1.0}{\micro s}$: about \qty{14}{\micro s}.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:14}
Loss per cycle $0.020 \times \qty{1.0e-9}{J} = \qty{2.0e-11}{J}$, at
\num{32768} cycles per second: $P = \num{32768} \times \num{2.0e-11}
\approx \qty{6.6e-7}{W}$ --- under a microwatt, years on a button
cell. In step, because only a push at the right \omterm{def:g12:oscillators-and-time:phase}{phase} \emph{adds}
\omterm{def:g10:energy-conservation:energy}{energy} (\omterm{def:g12:oscillators-and-time:resonance}{resonance}); out of step it would brake the swing.
\end{solution}

\begin{solution}{exo:g12:rlc-oscillations:15}
$C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
(\num{2.5e7})^2 \times \num{1.0e-6}} \approx \qty{41}{pF}$. The
resistor should push the loop into the \omterm{def:g12:rlc-oscillations:regimes}{aperiodic regime}: the charge
then settles without overshooting --- no oscillation, no ringing.
\end{solution}

\begin{solution}{pb:g12:rlc-oscillations:1}
\textbf{1.} Loop law: $u_L + u_C = 0$, with $u_C = q/C$,
$u_L = L\,di/dt$, $i = dq/dt$: $L\,d^2q/dt^2 + q/C = 0$.

\textbf{2.} $d^2q/dt^2 = -(2\pi/T_0)^2 q$, so the equation reads
$q\,[\,1/C - L(2\pi/T_0)^2\,] = 0$, true for all $t$ exactly when
$T_0 = 2\pi\sqrt{LC}$.

\textbf{3.} $LC$: $(\unit{V.s/A})(\unit{C/V}) = \unit{s} \times
\unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ is a time.

\textbf{4.} $LC = \qty{6.6e-14}{s^2}$: $T_0 = 2\pi \times \num{2.57e-7}
\approx \qty{1.6}{\micro s}$, $f_0 \approx \qty{620}{kHz}$ --- inside
\qtyrange{530}{1700}{kHz}.

\textbf{5.} $C = \dfrac{1}{4\pi^2 f_0^2 L} = \dfrac{1}{39.5 \times
\num{1.0e12} \times \num{2.0e-4}} \approx \qty{127}{pF}$.

\textbf{6.} Same formula at \qty{530}{kHz}: $C \approx \qty{451}{pF}$.

\textbf{7.} At \qty{1700}{kHz}: $C \approx \qty{44}{pF}$.

\textbf{8.} $C = \qty{480}{pF}$: $f_0 \approx \qty{514}{kHz}$;
$C = \qty{20}{pF}$: $f_0 \approx \qty{2.5}{MHz}$ --- the whole band,
with margin at both ends.

\textbf{9.} $f_0 = (2\pi\sqrt{L})^{-1}C^{-1/2} \propto 1/\sqrt{C}$;
doubling $f_0$ requires dividing $C$ by $4$.

\textbf{10.} Since $f_0 \propto 1/\sqrt{C}$, equal steps of $C$ move
$f_0$ slowly at large $C$ and fast at small $C$: a given spacing
between stations then takes little dial travel --- they crowd at the
\omterm{def:g12:oscillators-and-time:oscillator}{high-frequency} (low-$C$) end.

\textbf{11.} $Q_0 = CU_0 = \num{1.27e-10} \times 0.020 \approx
\qty{2.5e-12}{C}$; $E = \tfrac12 CU_0^2 \approx \qty{2.5e-14}{J}$.

\textbf{12.} $\tfrac12 LI_0^2 = E$: $I_0 = U_0\sqrt{C/L} =
0.020\sqrt{\num{6.35e-7}} \approx \qty{16}{\micro A}$.

\textbf{13.} $0.96^n = \tfrac12$: $n = \ln 2/\lvert\ln 0.96\rvert
\approx 17$ cycles.

\textbf{14.} At \qty{1.00}{MHz}, $T_0 = \qty{1.0}{\micro s}$: the ring
halves in about \qty{17}{\micro s}. A lone ring dies in microseconds;
music needs the loop fed continuously.

\textbf{15.} The \omterm{def:g10:signals-and-waves:wave}{wave} pushes millions of times per second; only when
its \omterm{def:g12:oscillators-and-time:oscillator}{frequency} matches $f_0$ do the pushes arrive in step and build the
swing --- \omterm{def:g12:oscillators-and-time:resonance}{resonance}; off-tune stations push as often against as with
it. At the transmitter, an amplifier returns the \omterm{def:g11:mechanical-energy:dissipation}{dissipated} \omterm{def:g10:energy-conservation:energy}{energy}
each cycle: a maintained \omterm{def:g12:oscillators-and-time:oscillator}{oscillator}.

\textbf{16.} $LC = \qty{2.54e-14}{s^2}$: $k = m/(LC) =
0.20/\num{2.54e-14} \approx \qty{7.9e12}{N/m}$.

\textbf{17.} About $10^8$ times the car spring: no bench-top
spring--mass system reaches megahertz --- everyday stiffnesses and
masses are hopelessly slow.

\textbf{18.} $m = \dfrac{k}{(2\pi f_0)^2} = \dfrac{100}
{(\num{6.28e6})^2} \approx \qty{2.5e-12}{kg}$ --- a few nanograms, a
speck of dust.

\textbf{19.} A quartz crystal: a featherweight, ultra-stiff sliver
singing at its \omterm{def:g12:rlc-oscillations:period}{natural frequency} --- the timekeeper met earlier this
year.

\textbf{20.} With $L = \qty{0.20}{mH}$ and \qtyrange{20}{480}{pF} the
loop sweeps about \qtyrange{0.51}{2.5}{MHz}, covering the band;
$C \approx \qty{127}{pF}$ picks \qty{1.00}{MHz}; left alone the ring
halves in $\approx \qty{17}{\micro s}$, but the station's own \omterm{def:g10:signals-and-waves:wave}{wave},
resonating in step, re-feeds the loop --- the news plays on the \omterm{def:g10:energy-conservation:energy}{energy}
of the \omterm{def:g10:signals-and-waves:wave}{wave} itself.
\end{solution}
