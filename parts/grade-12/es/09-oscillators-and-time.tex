\section{Oscilaciones libres}

\begin{definition}[Oscilador, periodo, frecuencia, amplitud.]\label{def:g12:oscillators-and-time:oscillator}
Un \emph{oscilador mecanico}\index{oscilador} es un sistema que,
desplazado de un establo \emph{posición de equilibrio}\index{equilibrio
posición} y liberado, se balancea hacia adelante y hacia atrás alrededor de él: estos son
\emph{oscilaciones libres}\index{oscilaciones libres}. El movimiento se repite
después del \emph{período}\index{período} $T_0$ (\unit{s}); el
\emph{frecuencia}\index{frecuencia}$f_0 = 1/T_0$
(\unit{Hz}, \cref{ch:g12:mechanical-waves}); el
\emph{amplitud}\index{amplitud} es el desplazamiento máximo.
\end{definition}

\begin{example}[Un zoológico de osciladores]\label{ex:g12:oscillators-and-time:zoo}
Un columpio para parque infantil: $T_0 \approx \qty{3}{s}$. La cuerda La de una guitarra:
$f_0 = \qty{110}{Hz}$. Un cristal de cuarzo: \qty{32768}{Hz}. el cesio
oscilación que define la segunda: aproximadamente \qty{9.2e9}{Hz}. Diez pedidos
de magnitud, un solo trabajo: contar.
\end{example}

\begin{proposition}[Período del péndulo simple]\label{prop:g12:oscillators-and-time:pendulum-period}
\index{péndulo simple!período}
Para \omterm{def:g12:oscillators-and-time:oscillator}{amplitudes} pequeño (por debajo de $15^\circ$), el \omterm{def:g12:oscillators-and-time:oscillator}{período} es
\[
T_0 = 2\pi \sqrt{\frac{L}{g}}, \qquad g = \qty{9.81}{m/s^2}.
\]
No depende ni de la masa de la masa ni de --- mientras que el ángulo
permanece pequeño --- en el \omterm{def:g12:oscillators-and-time:oscillator}{amplitud}: pequeños cambios son
\emph{isócrono}\index{isocronismo}, todos tomando el mismo tiempo.
\end{proposition}
\admitted

\begin{remark}[Análisis dimensional, y donde vive el $2\pi$]\label{rem:g12:oscillators-and-time:dimensional}
La fórmula se puede adivinar. De $m$ (\unit{kg}), $L$ (\unit{m}) y
$g$ (\unit{m/s^2}), la única combinación con el \omterm{def:g10:orders-of-magnitude:unit}{unidad} de una hora es
$\sqrt{L/g}$ --- y se ingresa la masa \emph{no puedo}, ningún otro dato
llevando un \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} para cancelarlo. Las dimensiones no pueden dar lo puro.
número al frente: el $2\pi$ proviene de resolver la ecuación de movimiento,
hecho en el volumen del Año 1. La condición de ángulo pequeño tampoco es opcional:
en $30^\circ$ el verdadero \omterm{def:g12:oscillators-and-time:oscillator}{período} es $2\%$ más largo y el isocronismo falla.
\end{remark}

\begin{example}[El péndulo de segundos]\label{ex:g12:oscillators-and-time:seconds}
¿Qué longitud supera a la segunda, $T_0 = \qty{2.00}{s}$, un tick cada una?
manera? Resolviendo, $L = g\,T_0^2/4\pi^2 = 9.81 \times 2.00^2/4\pi^2 \approx
\qty{0.994}{m}$. Casi un metro: los relojes de péndulo son altos porque
Los segundos son largos.
\end{example}

\section{El oscilador de masa-resorte}

\begin{definition}[Fuerza recuperadora de un recurso]\label{def:g12:oscillators-and-time:spring}
Un planeador de masa $m$ se desliza sobre un riel horizontal sin fricción, atado a
un resorte de \emph{rigidez}\index{rigidez} $k$ (\unit{N/m}). dejar
$x$ sea su desplazamiento del equilibrio. Estirado o comprimido, el
el resorte tira o empuja la vela hacia atrás con el \omterm{def:g10:inertia:force}{fuerza} $F = -kx$:
proporcional al desplazamiento y opuesto a él --- nuevamente un \omterm{def:g12:oscillators-and-time:pendulum}{restaurando
fuerza}, ahora suministrado por elasticidad.
\end{definition}

\begin{proposition}[Ecuación de movimiento y su solución]\label{prop:g12:oscillators-and-time:spring-period}
\index{resorte--oscilador de masa}
\omterm{thm:g12:newtons-laws:second}{Segunda ley de Newton} a lo largo del riel (\cref{ch:g12:newtons-laws}) da
$m \frac{d^2x}{dt^2} = -kx$, es decir $\frac{d^2x}{dt^2} =
-\frac{k}{m}\,x$. Cualesquiera que sean las constantes$A$y$\varphi$,
\[
x(t) = A \cos\!\left(\frac{2\pi t}{T_0} + \varphi\right),
\qquad T_0 = 2\pi\sqrt{\frac{m}{k}},
\]
satisface esta ecuación: oscilación con \omterm{def:g12:oscillators-and-time:oscillator}{período} $T_0$, independiente
del \omterm{def:g12:oscillators-and-time:oscillator}{amplitud}.
\end{proposition}

\begin{proof}
Diferenciando, $\frac{dx}{dt} = -\frac{2\pi}{T_0}A \sin(\frac{2\pi
t}{T_0} + \varphi)$; de nuevo,$\frac{d^2x}{dt^2} = -(\frac{2\pi}{T_0})^2
A\cos(\frac{2\pi t}{T_0} + \varphi) = -(\frac{2\pi}{T_0})^2\,x(t)$,
que es $-(k/m)\,x$ exactamente cuando $(2\pi/T_0)^2 = k/m$.
\end{proof}

\begin{remark}[Todos los movimientos que hay]\label{rem:g12:oscillators-and-time:uniqueness}
Que el movimiento \emph{cada} del planeador sea una de estas funciones es
admitido, probado en el volumen del Año 1. Y no hay $g$ en $2\pi\sqrt{m/k}$:
Los cambios más rígidos son más rápidos, los más pesados más lentos, lo mismo en la Luna.
\end{remark}

\begin{definition}[amplitud y fase]\label{def:g12:oscillators-and-time:phase}
En $x(t) = A\cos(2\pi t/T_0 + \varphi)$ la constante positiva $A$ es
el \omterm{def:g12:oscillators-and-time:oscillator}{amplitud} --- los extremos son $x = \pm A$ --- y $2\pi t/T_0 +
\varphi$es el \emph{fase}\index{fase}, creciendo en$2\pi$ cada uno
\omterm{def:g12:oscillators-and-time:oscillator}{período}; el \emph{fase inicial}\index{fase inicial} $\varphi$
registra el inicio: liberado en reposo desde $x = A$, $\varphi = 0$. Para
la instancia $m = \qty{0.20}{kg}$ en $k = \qty{20}{N/m}$ da $T_0 =
2\pi\sqrt{0.20/20} = \qty{0.63}{s}$; tirado a \qty{5.0}{cm} y
liberado, $x(t) = 5.0\cos(2\pi t/0.63)$ en centímetros.
\end{definition}

\begin{proposition}[Energía del oscilador]\label{prop:g12:oscillators-and-time:energy}
Un resorte estirado por $x$ almacena el potencial elástico \omterm{def:g10:energy-conservation:energy}{energía} $E_p =
\tfrac12 kx^2$ (justificado en \cref{ch:g12:work-and-energy}). durante
oscilación sin fricción el \omterm{def:g11:mechanical-energy:em}{energía mecánica}
\[
E_m = E_k + E_p = \tfrac12 m v^2 + \tfrac12 k x^2 = \tfrac12 k A^2
\]
es constante: todo elástico en los extremos, todo cinético en el equilibrio.
El péndulo hace el mismo dúo entre $E_k$ y gravitacional.
$E_p$ (\cref{ch:g11:mechanical-energy}).
\end{proposition}

\begin{proof}
Con $v = \frac{dx}{dt}$, la derivada de $E_m$ es $mv\frac{dv}{dt}
+ kx\frac{dx}{dt} = v\,(m\frac{d^2x}{dt^2} + kx) = 0$ por la ecuación
de movimiento; el valor $\tfrac12 kA^2$ se lee en un extremo.
\end{proof}

\section{Medir el tiempo}

\begin{method}[Tiempo de un periodo]\label{met:g12:oscillators-and-time:timing}
Nunca cronometres una oscilación: inicia el cronómetro cuando la pesa cruza
la vertical (más rápida, más fácil de juzgar), cuente las oscilaciones $N = 50$,
dividir por $N$ --- un error de reacción \qty{0.2}{s} se convierte en \qty{4}{ms}
en $T_0$. Repita y compare ejecuciones.
\end{method}

\begin{example}[tres siglos de tictac]\label{ex:g12:oscillators-and-time:clocks}
Huygens construyó el primer reloj de péndulo en \num{1657}: isocronismo
llevó los relojes de un cuarto de hora de deriva por día a segundos, y

\section{Problema: el banco del relojero}

\begin{problem}[{Problema del fin de semana --- el banco del relojero: una pared
Reloj retrocedido en el tiempo: cincuenta oscilaciones cronometradas, una varilla de latón que
se extiende en verano, el susurro del escape, un cuarzo
retador --- el veredicto emitido en segundos por mes}]
\label{pb:g12:oscillators-and-time:1}
Un reloj de pared de péndulo llega al banco "funcionando mal". tratar
su varilla de latón y su peso pesado como un \omterm{def:g12:oscillators-and-time:pendulum}{péndulo simple} de longitud efectiva
$L$, masa bob $m = \qty{1.2}{kg}$; el engranaje cuenta cada uno lleno
oscilación como exactamente \qty{2.000}{s}.

\medskip
\noindent\textbf{Parte I --- Taking the clock's pulse.}
\begin{enumerate}
  \item ¿Por qué cronometrar oscilaciones $50$ en lugar de una? Estimar el error
        izquierda en $T_0$ si su tiempo de reacción es \qty{0.2}{s}.
  \item El cronómetro marca \qty{99.4}{s} para las oscilaciones $50$: ¿el \omterm{def:g12:oscillators-and-time:oscillator}{período}?
  \item Deduzca la longitud efectiva actual del péndulo.
  \item ¿Qué largo daría exactamente \qty{2.000}{s}? ¿Debería el
        ¿La tuerca de clasificación aumenta o disminuye la sacudida y en cuánto?
  \item Sin ajustar, ¿el reloj está adelantado o atrasado, en cuántos segundos por día?
  \item Sensibilidad a $g$: el \omterm{def:g12:oscillators-and-time:oscillator}{período} en la Luna
        ($g = \qty{1.62}{m/s^2}$)? El error diario en un observatorio
        donde $g = \qty{9.80}{m/s^2}$?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The summer slump.} Latón se expande: a
calentar $\Delta\theta$ estira la varilla a $L' = L\,(1 + \lambda\,
\Delta\theta)$,$\lambda = \num{1.9e-5}$ por \omterm{def:g10:light-spectra:kelvin}{kelvin}. el reloj es
ajustado en \qty{20}{\celsius}.
\begin{enumerate}[resume]
  \item Calcule $\Delta L$ para una sala de verano en \qty{25}{\celsius}.
  \item Usando $\sqrt{1+x} \approx 1 + x/2$, muestre que
        $\Delta T_0 / T_0 = \tfrac12 \lambda\, \Delta\theta$.
  \item Calcula $\Delta T_0/T_0$ y el error diario de verano: ¿rápido o lento?
  \item Las mismas preguntas para un pasillo de invierno en \qty{10}{\celsius}.
  \item Antiguos relojes de precisión utilizaban péndulos ``gridiron'' mezclando latón
        varillas con varillas de acero (que se expanden aproximadamente la mitad) en
        direcciones alternas. Explique cómo esto puede mantener la tasa.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The whispered push.} Con el escape
desenganchado, el columpio, iniciado en \omterm{def:g12:oscillators-and-time:oscillator}{amplitud} $\theta_m = 2.00^\circ$,
reduce a la mitad su \omterm{def:g12:oscillators-and-time:oscillator}{amplitud} en \qty{280}{s}; en servicio, un pequeño empujón del
escape cada \omterm{def:g12:oscillators-and-time:oscillator}{período} mantiene constante $\theta_m$.
\begin{enumerate}[resume]
  \item ¿Cuántos \omterm{def:g12:oscillators-and-time:oscillator}{periodos} se necesitan para reducir a la mitad?
  \item En el extremo la masa ha subido $h = L(1 - \cos\theta_m)
        \approx L\,\theta_m^2/2$(radianes): deduce$E_m = \tfrac12
        m g L\,\theta_m^2$ y complétalo
        (\cref{ch:g11:mechanical-energy}).
  \item \omterm{def:g10:energy-conservation:energy}{Energía} es proporcional a \omterm{def:g12:oscillators-and-time:oscillator}{amplitud} al cuadrado: ¿qué fracción de
        ¿$E_m$ se pierde por \omterm{def:g12:oscillators-and-time:oscillator}{período}?
  \item Deduce el \omterm{def:g10:energy-conservation:energy}{energía} de cada empujón y el \omterm{def:g11:work-of-force:power}{fuerza} promedio necesario.
  \item El motor \omterm{def:g10:universal-gravitation:weight}{peso} $M = \qty{2.5}{kg}$ desciende $h =
        \qty{1.0}{m}$ por semana: ¿es suficiente \omterm{def:g10:energy-conservation:energy}{energía}, a qué \omterm{def:g10:energy-conservation:efficiency}{eficiencia}?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The quartz challenger.} Un cuarzo
El movimiento oscila en $f = \qty{32768}{Hz}$ y se desplaza aproximadamente.
\qty{0.3}{s} por mes; el péndulo restaurado todavía se mueve
\qty{0.5}{s} por día.
\begin{enumerate}[resume]
  \item Calcular el cuarzo \omterm{def:g12:oscillators-and-time:oscillator}{período}. Mostrar $\num{32768} = 2^{15}$: por qué
        ¿Es un \omterm{def:g11:work-of-force:power}{fuerza} de dos exactamente lo que quiere un circuito de reloj?
  \item Calcular las derivas relativas del cuarzo y el péndulo restaurado.
        y cada uno en segundos por mes. ¿Quién gana y por qué factor?
  \item El estándar de cesio --- \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{periodos} crea uno
        segundo --- deriva \qty{1}{s} cada cien millones de años: es
        ¿deriva relativa? ¿Cuántos órdenes de magnitud por debajo del cuarzo,
        ¿Y por qué el GPS lo exige?
  \item El veredicto, una frase con números: ¿qué significa el restaurado?
        reloj gana en la pared, y ¿a quién pertenece ahora el segundo?
\end{enumerate}
\end{problem}
