\section{Intensidad del sonido y escala de decibeles.}

\begin{definition}[Intensidad del sonido]\label{def:g12:sound-acoustics:intensity}
El \emph{intensidad del sonido}\index{intensidad del sonido} $I$ es el
acústico \omterm{def:g11:work-of-force:power}{fuerza} cruzando un cuadrado \omterm{def:g10:orders-of-magnitude:unit}{metro} frente al \omterm{def:g10:signals-and-waves:wave}{ola}:
$I = P/S$, en \unit{W/m^2}. El \omterm{def:g10:signals-and-waves:sound}{sonido} audible más silencioso se trata de
$I_0 = \qty{1.0e-12}{W/m^2}$; el dolor comienza cerca de \qty{1}{W/m^2}.
\end{definition}

\begin{proposition}[Atenuación del cuadrado inverso]\label{prop:g12:sound-acoustics:inverse-square}
\index{ley del cuadrado inverso}%
Una pequeña fuente que irradia un \omterm{def:g11:work-of-force:power}{fuerza} $P$ por igual en todas las direcciones.
da, a distancia $r$, $I = P/(4\pi r^2) \propto 1/r^2$: duplicación
la distancia divide la intensidad por cuatro.
\end{proposition}

\begin{proof}
El conjunto \omterm{def:g11:work-of-force:power}{fuerza} $P$ atraviesa la esfera del área $4\pi r^2$; dividir.
\end{proof}

\begin{definition}[Nivel de sonido]\label{def:g12:sound-acoustics:level}
El oído abarca doce \omterm{def:g11:work-of-force:power}{potestades} de diez, por lo que se citan las intensidades
logarítmicamente. El \emph{nivel de sonido}\index{nivel de sonido} de $I$ es
\[ L = 10 \log_{10}(I/I_0), \qquad I_0 = \qty{1.0e-12}{W/m^2}, \]
en \emph{decibeles}\index{decibel} (\unit{dB}): umbral de audición
\qty{0}{dB}, dolor cerca de \qty{120}{dB}.
\end{definition}

\begin{proposition}[Aritmética de decibeles]\label{prop:g12:sound-acoustics:db-arithmetic}
Multiplicar $I$ por $k$ suma $10\log_{10}k$ \omterm{def:g12:sound-acoustics:level}{decibeles}. en particular
$\times 10$ agrega \qty{10}{dB}; $\times 2$ agrega
$10\log_{10}2 \approx \qty{3.0}{dB}$; $N$ incoherente idéntico
las fuentes agregan \emph{intensidades}, nunca niveles, dando
$L_1 + 10\log_{10}N$; duplicar la distancia a una fuente pequeña ($I$
dividido por cuatro) elimina aproximadamente \qty{6.0}{dB}.
\end{proposition}

\begin{proof}
$10\log_{10}(kI/I_0) = 10\log_{10}(I/I_0) + 10\log_{10}k$;
distancia: \cref{prop:g12:sound-acoustics:inverse-square}, $k = 1/4$.
\end{proof}

\begin{method}[Trabajando en decibeles]\label{met:g12:sound-acoustics:db}
\begin{enumerate}
  \item$L = 10\log_{10}(I/I_0)$; atrás: $I = I_0 \times 10^{L/10}$.
  \item Nunca agregue niveles: convierta a intensidades, agregue, vuelva a convertir
        --- o $+\qty{3}{dB}$ por duplicación, $+\qty{10}{dB}$ por diez veces.
  \item Cada duplicación de la distancia a una fuente pequeña cuesta \qty{6}{dB}.
\end{enumerate}
\end{method}

\section{Timbre y armónicos}

\begin{definition}[armónicos]\label{def:g12:sound-acoustics:harmonics}
A \omterm{def:g10:signals-and-waves:period}{periódico} \omterm{def:g10:signals-and-waves:sound}{sonido} de \omterm{def:g10:signals-and-waves:frequency}{frecuencia} $f_1$ --- el
\emph{fundamental}\index{fundamental} --- es en general una suma de
senos en las frecuencias $f_n = n f_1$, $n = 1, 2, 3, \dots$, su
\emph{armonía}\index{armónico}. Un \omterm{def:g10:signals-and-waves:sound}{sonido} reducido a su fundamental
es un \emph{tono puro}\index{tono puro}, como el de un diapasón.
\end{definition}

\begin{definition}[Espectro y timbre]\label{def:g12:sound-acoustics:timbre}
El \emph{espectro}\index{espectro} de un \omterm{def:g10:signals-and-waves:sound}{sonido} es el gráfico de barras de
es \omterm{def:g12:sound-acoustics:harmonics}{armonía}: una barra por $f_n$, la altura que \omterm{def:g12:sound-acoustics:harmonics}{armónico}
fuerza. El oído escucha $f_1$ como \omterm{def:g12:sound-acoustics:pitch-loudness}{paso} y \emph{mezcla} de
puntos fuertes como el \emph{timbre}\index{timbre} --- el color del \omterm{def:g10:signals-and-waves:sound}{sonido}:
Los instrumentos que tocan la misma nota comparten $f_1$, diferenciándose sólo en
sus espectros.
\end{definition}

\section{La cuerda vibrante}

\begin{definition}[Onda estacionaria]\label{def:g12:sound-acoustics:standing-wave}
Un \omterm{def:g10:signals-and-waves:wave}{ola} confinado entre los extremos fijos de una cuerda puede asentarse en un
\emph{onda estacionaria}\index{onda estacionaria}: una oscilación en su lugar,
puntos inmóviles --- \emph{nodos}\index{nodo} --- alternando con
puntos de oscilación máxima --- \emph{antinodos}\index{antinodo}.
Ya nada viaja; la cuerda vibra en un patrón fijo.
\end{definition}

\begin{proposition}[Modos de una cadena]\label{prop:g12:sound-acoustics:string-modes}
Una cuerda de longitud $L$, fijada en ambos extremos, que lleva \omterm{def:g10:signals-and-waves:wave}{ondas} de velocidad
$v$, vibra constantemente sólo en patrones que se ajustan a un número entero $n$
de \omterm{def:g12:mechanical-waves:wavelength}{medias longitudes de onda} entre el extremo impuesto \omterm{def:g12:sound-acoustics:standing-wave}{nodos}:
\[ \lambda_n = \frac{2L}{n}, \quad f_n = n\,\frac{v}{2L} = n f_1, \quad n = 1, 2, 3, \dots \]
El modo más bajo $f_1 = v/(2L)$ es el \omterm{def:g12:sound-acoustics:harmonics}{fundamental}; los otros son
exactamente es \omterm{def:g12:sound-acoustics:harmonics}{armonía}. \admitted
\end{proposition}

\begin{remark}[Donde vive la prueba honesta]\label{rem:g12:sound-acoustics:honest}
Que \omterm{def:g10:signals-and-waves:wave}{ondas} confinado elija exactamente estos modos se desprende de
superponiendo el \omterm{def:g10:signals-and-waves:wave}{ola} con sus reflejos, un cálculo realizado en
el volumen del año 1. La geometría convence: ambos extremos deben permanecer
aún así, debe caber un número entero de \omterm{def:g12:mechanical-waves:wavelength}{medias longitudes de onda}.
\end{remark}

\begin{example}[La cuerda La de un violín]\label{ex:g12:sound-acoustics:violin}
La cuerda La de un violín tiene $L = \qty{32.8}{cm}$ y \omterm{def:g10:signals-and-waves:sound}{sonidos}

\end{example}

Cómo se excita la cuerda (punteada, inclinada, golpeada) establece cómo
Gran parte de cada modo está presente y el cuerpo amplifica algo de \omterm{def:g12:sound-acoustics:harmonics}{armonía}.
más que otros; La excitación más el cuerpo arreglan el \omterm{def:g12:sound-acoustics:timbre}{espectro}, por lo que lo mismo

\end{remark}

\section{Problema: El taller del luthier}

\begin{problem}[{Problema de fin de semana --- el taller del lauder: una guitarra
cuerda dispuesta traste a traste, sus armónicos afinados en un timbre, un
considerando mantenido por debajo de la línea de peligro, y una búsqueda por ultrasonido de un
defecto en la madera}]
\label{pb:g12:sound-acoustics:1}
Un luthier está terminando una guitarra. Su cuerda A tiene longitud vibratoria.
$L = \qty{65.0}{cm}$ y debe \omterm{def:g10:signals-and-waves:sound}{sonido} $f_1 = \qty{110}{Hz}$. tomar
\qty{340}{m/s} para \omterm{def:g10:signals-and-waves:sound}{sonido} en el aire y admite $f_1 = v/(2L)$
(\cref{prop:g12:sound-acoustics:string-modes}), siendo $v$ el \omterm{def:g10:signals-and-waves:wave}{ola}
velocidad en la cuerda.

\medskip
\noindent\textbf{Parte I --- Laying out the fretboard.}
\begin{enumerate}
  \item Calcule la velocidad \omterm{def:g10:signals-and-waves:wave}{ola} $v$ en la cadena.
  \item Calcule el \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} \emph{en la cuerda} del \omterm{def:g12:sound-acoustics:harmonics}{fundamental},
        luego el \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} del \qty{110}{Hz} \omterm{def:g10:signals-and-waves:sound}{sonido} \emph{en el
        aire}. ¿Por qué se diferencian en el mismo \omterm{def:g10:signals-and-waves:frequency}{frecuencia}?
  \item La octava (\qty{220}{Hz}): ¿qué longitud de vibración le da?
        ¿Dónde debe ubicarse el duodécimo traste, como fracción de $L$?
  \item El quinto (\qty{165}{Hz}): ¿qué longitud de vibración le da?
  \item Cada traste eleva la nota un semitono, un factor
        $2^{1/12}$. Comprueba con logaritmos cuantos semitonos separan
        \qty{110}{Hz} de \qty{220}{Hz}; longitud en el primer traste?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- \omterm{def:g12:sound-acoustics:harmonics}{Armonía} and \omterm{def:g12:sound-acoustics:timbre}{timbre}.}
\begin{enumerate}[resume]
  \item Enumere los primeros cinco \omterm{def:g12:sound-acoustics:harmonics}{armonía} de la cadena A abierta.
  \item Un violinista A \omterm{def:g10:signals-and-waves:sound}{sonidos} \qty{440}{Hz}. ¿Cuál \omterm{def:g12:sound-acoustics:harmonics}{armónico} del
        cuerda de guitarra es eso, y que intervalo musical separa
        ¿Es del \omterm{def:g12:sound-acoustics:harmonics}{fundamental}? ¿Y \qty{330}{Hz}?
  \item El luthier puntea cerca del puente, luego sobre el
        Diapasón: mismo \omterm{def:g12:sound-acoustics:pitch-loudness}{paso}, diferente color. Explique con espectros.
  \item Un ligero toque en el punto medio mientras depilas \omterm{def:g10:inertia:force}{efectivo} a \omterm{def:g12:sound-acoustics:standing-wave}{nodo}
        allí. ¿Qué \omterm{def:g12:sound-acoustics:harmonics}{armonía} sobrevive y qué \omterm{def:g12:sound-acoustics:pitch-loudness}{paso} se escucha?
  \item ¿Cuántos \omterm{def:g12:sound-acoustics:harmonics}{armonía} de la cadena abierta se encuentran, en principio,
        dentro del audible \omterm{def:g11:fundamental-interactions:strong}{rango}?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The recital.}
En el primer concierto de guitarra, cada instrumento irradia una acústica.
\omterm{def:g11:work-of-force:power}{fuerza} $P = \qty{6.0e-3}{W}$ como fuente pequeña, igualmente en todos
direcciones.
\begin{enumerate}[resume]
  \item Calcule la intensidad en $r = \qty{3.0}{m}$ desde un instrumento.
  \item Deduzca el \omterm{def:g12:sound-acoustics:level}{nivel de sonido} allí.
  \item Toca un cuarteto: ¿qué nivel dan cuatro de estos instrumentos?
        en el mismo lugar? ¿Por qué no es cuatro veces el nivel?
  \item Tocan doce instrumentos: ¿cuál es el nivel en \qty{3.0}{m}?
  \item ¿A qué distancia debe sentarse el oyente para que caiga el conjunto?
        a \qty{85}{dB}?
  \item Regla de seguridad: \qty{85}{dB} es seguro durante ocho horas y
        cada \qty{+3}{dB} reduce a la mitad el tiempo seguro. ¿Cuánto tiempo puede
        El oyente de primera fila de la pregunta 14 ¿se queda con seguridad?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The flaw in the wood.}
Antes de barnizar, el luthier prueba un cuello grueso \qty{45}{mm}
en blanco con un probador \qty{2.0}{MHz}; en esta madera tome $v = \qty{5.0e3}{m/s}$.
\begin{enumerate}[resume]
  \item Calcular el \omterm{def:g10:signals-and-waves:ultrasound}{ultrasonido} \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} en la madera. ¿Por qué debe un
        ¿La búsqueda de defectos \omterm{def:g10:signals-and-waves:wave}{ola} es ultrasónica en lugar de audible?
  \item Calcule \omterm{rem:g10:signals-and-waves:honest}{eco} \omterm{def:g12:mechanical-waves:delay}{demora} desde la cara posterior del espacio en blanco.
  \item Llega un \omterm{rem:g10:signals-and-waves:honest}{eco} después de \qty{7.2}{\micro s}: ¿qué profundidad tiene el defecto?
  \item Escribe la tarjeta de taller: \omterm{def:g10:signals-and-waves:wave}{ola} velocidad en la cuerda, duodécima-
        y longitudes del primer traste, distancia de escucha segura, profundidad de los defectos.
\end{enumerate}
\end{problem}
