\chapter{Cinemática en dos dimensiones}\label{ch:g12:kinematics-2d}

Ver un coche tomar una rotonda a una velocidad constante \qty{30}{km/h}: la aguja
nunca se mueve, pero cada pasajero se siente arrastrado hacia los lados --- el movimiento
está cambiando y no es el \omterm{def:g12:kinematics-2d:velocity}{velocidad}. Un número por instante no
Ya es suficiente: el movimiento en el plano necesita flechas. este capitulo
equipa un punto en movimiento con tres --- posición, velocidad, aceleración
--- encadenado por la derivada, y lee familias enteras de movimiento
su geometría.

\section{El vector de posición}

El movimiento sólo existe en relación con un \omterm{def:g10:relative-motion:frame}{marco de referencia} establecido --- un rígido
cuerpo más un reloj (\cref{ch:g10:relative-motion}). Lo nuevo de este año es
precisión: adjuntar al marco un origen $O$ y dos perpendiculares
ejes graduados $Ox$, $Oy$ y registre dónde está el punto en cada
instante, como un par de funciones.

\begin{definition}[Vector de posición]\label{def:g12:kinematics-2d:position}
En un bastidor equipado con un origen $O$ y ejes $Ox$, $Oy$, el
\emph{vector de posición}\index{vector de posición} de un punto en movimiento $M$
en el momento $t$ es $\vect{OM}(t)$, de coordenadas
$\bigl(x(t),\, y(t)\bigr)$: dos funciones de tiempo, una por eje. el
La curva trazada por $M$ mientras se ejecuta $t$ es su \omterm{def:g10:relative-motion:trajectory}{trayectoria}.
\end{definition}

\section{El vector de velocidad}

\begin{definition}[Vector de velocidad]\label{def:g12:kinematics-2d:velocity}
En un corto intervalo el punto se mueve
$\vect{OM}(t+\Delta t) - \vect{OM}(t)$; dividir por $\Delta t$ y dejar
la reducción del intervalo: esta es la derivada de tus matemáticas
Por supuesto, aplicado a cada coordenada. El \emph{velocidad
vector}\index{vector de velocidad} de $M$ es el derivado de su
\omterm{def:g12:kinematics-2d:position}{vector de posición}, $\vect v(t) = \bigl(x'(t),\, y'(t)\bigr)$, en
\unit{m/s}. Su norma $v = \sqrt{x'^2 + y'^2}$ es la
\emph{velocidad}\index{velocidad} --- el número que muestra el velocímetro.
\end{definition}

\begin{example}[Un dron en vuelo nivelado]\label{ex:g12:kinematics-2d:drone}
Un dron vuela con $x(t) = 4.0\,t$ y $y(t) = 3.0\,t$ (metros,
segundos): \omterm{def:g10:relative-motion:trajectory}{trayectoria} la línea $y = \tfrac34 x$, velocidad
$\vect v = (4.0, 3.0)$ en todo momento, \omterm{def:g12:kinematics-2d:velocity}{velocidad}
$\sqrt{4.0^2 + 3.0^2} = \qty{5.0}{m/s}$, constante.
\end{example}

\begin{proposition}[Tangente a la trayectoria]\label{prop:g12:kinematics-2d:tangent}
En todo instante, $\vect v(t)$ es tangente al \omterm{def:g10:relative-motion:trajectory}{trayectoria} en el
posición actual de $M$ y apunta en la dirección de viaje.
\end{proposition}

\begin{proof}
El desplazamiento es acorde del \omterm{def:g10:relative-motion:trajectory}{trayectoria}; a medida que $\Delta t$ se encoge,
la dirección de la cuerda se convierte en la de la tangente.
\end{proof}

\section{El vector de aceleración}

\begin{definition}[Vector de aceleración]\label{def:g12:kinematics-2d:acceleration}
El \emph{vector de aceleracion}\index{vector de aceleracion} de $M$ es
la derivada de su \omterm{def:g12:kinematics-2d:velocity}{vector de velocidad},
$\vect a(t) = \bigl(v_x'(t),\, v_y'(t)\bigr) =
\bigl(x''(t),\, y''(t)\bigr)$, en \unit{m/s^2}: qué tan rápido, y en
en qué dirección está cambiando actualmente el \omterm{def:g12:kinematics-2d:velocity}{vector de velocidad}.
\end{definition}

\begin{remark}[Acelerando sin acelerar]\label{rem:g12:kinematics-2d:turning}
$\vect a \neq \vect 0$ cada vez que cambia la velocidad \emph{vector}
--- en longitud \emph{o en dirección}: un coche que frena acelera
(hacia atrás); un automóvil que gira a velocidad constante \omterm{def:g12:kinematics-2d:velocity}{velocidad} acelera (hacia los lados).
Este es el control de cantidad \omterm{def:g10:inertia:force}{efectivo}
(\cref{ch:g11:forces-and-motion,ch:g12:newtons-laws}).
\end{remark}

\begin{method}[Leyendo una cronofotografía]\label{met:g12:kinematics-2d:chrono}
Un \emph{cronofotografía}\index{cronofotografía} registra el
posiciones $M_0, M_1, M_2, \dots$ de un punto en movimiento al mismo tiempo
intervalos $\Delta t$.
\begin{enumerate}
  \item El espaciado entre puntos indica \omterm{def:g12:kinematics-2d:velocity}{velocidad}: espaciado igual, constante \omterm{def:g12:kinematics-2d:velocity}{velocidad};
        abriendo, acelerando; cerrando, desacelerando.
  \item La velocidad en $M_i$ es la cuerda simétrica,
        $\vect v_i \approx \vect{M_{i-1}M_{i+1}}/(2\,\Delta t)$,
        dibujado en $M_i$, tangente al camino punteado.
  \item Para la aceleración en $M_i$: copie $\vect v_{i-1}$ y
        $\vect v_{i+1}$ a $M_i$, sorteo
        $\Delta\vect v = \vect v_{i+1} - \vect v_{i-1}$ punta a punta;
        luego $\vect a_i \approx \Delta\vect v/(2\,\Delta t)$.
\end{enumerate}
\end{method}

\section{Movimientos rectilíneos}

En línea recta basta con un eje: $x(t)$, $v(t) = x'(t)$,
$a(t) = v'(t)$, todos los números firmados.

\begin{definition}[Movimiento rectilíneo uniforme]\label{def:g12:kinematics-2d:uniform}
Un movimiento es \emph{rectilíneo uniforme}\index{rectilíneo uniforme
movimiento} cuando $\vect v$ es un vector constante: recto \omterm{def:g10:relative-motion:trajectory}{trayectoria},
constante \omterm{def:g12:kinematics-2d:velocity}{velocidad}, $\vect a = \vect 0$. Entonces $x(t) = v\,t + x_0$.
\end{definition}

\begin{proposition}[Movimiento rectilíneo uniformemente acelerado]\label{prop:g12:kinematics-2d:uarm}
\index{movimiento uniformemente acelerado}%
Si un punto se mueve a lo largo de $Ox$ con aceleración \emph{constante} $a$,
comenzando en $t = 0$ desde la posición $x_0$ con velocidad $v_0$, luego
\[
v(t) = a\,t + v_0,
\qquad
x(t) = \tfrac12\,a\,t^2 + v_0\,t + x_0 .
\]
\end{proposition}

\begin{proof}
$v$ es una primitiva de la constante $a$: $v(t) = a t + C$, y
$t = 0$ \omterm{def:g10:inertia:force}{efectivo} $C = v_0$. A su vez $x$ es una antiderivada de
$a t + v_0$: $x(t) = \tfrac12 a t^2 + v_0 t + C'$, con $C' = x_0$.
\end{proof}

\begin{example}[Distancia de frenado]\label{ex:g12:kinematics-2d:braking}
Un automóvil en \omterm{def:g12:kinematics-2d:velocity}{velocidad} $v_0$ frena con desaceleración constante $a$
(aceleración $-a$). Se detiene cuando $v(t) = -a t + v_0 = 0$, en
$t_s = v_0/a$, habiendo cubierto $d = -\tfrac12 a t_s^2 + v_0 t_s =
v_0^2/(2a)$. La plaza es el titular de seguridad vial: duplicar la
\omterm{def:g12:kinematics-2d:velocity}{velocidad} \emph{cuadriplica} la distancia de frenado --- con $a =
\qty{6.0}{m/s^2}$, \qty{16}{m} de \qty{50}{km/h}, \qty{64}{m} de
\qty{100}{km/h}.
\end{example}

\section{Movimiento circular uniforme y marco de Frenet.}

\begin{definition}[movimiento circular uniforme]\label{def:g12:kinematics-2d:ucm}
Una moción es \emph{circular uniforme}\index{movimiento circular uniforme}
cuando el \omterm{def:g10:relative-motion:trajectory}{trayectoria} es un círculo de radio $R$ y el \omterm{def:g12:kinematics-2d:velocity}{velocidad} $v$ es
constante. El \emph{período}\index{período!de rotación} $T$ es el
duración de una vuelta, y el \emph{frecuencia}\index{frecuencia!de
rotación} $f = 1/T$ (en \unit{Hz}) el número de vueltas por segundo:
$v = 2\pi R/T = 2\pi R f$.
\end{definition}

Constante \omterm{def:g12:kinematics-2d:velocity}{velocidad} --- sin embargo la velocidad gira: hay \emph{is} un
aceleración. ¿De qué manera y de qué tamaño?

\begin{theorem}[Aceleración centrípeta]\label{thm:g12:kinematics-2d:centripetal}
\index{aceleración centrípeta}%
En \omterm{def:g12:kinematics-2d:ucm}{movimiento circular uniforme} de radio $R$ y \omterm{def:g12:kinematics-2d:velocity}{velocidad} $v$, el
aceleración --- llamada \emph{centrípeto} --- puntos del movimiento
apuntar hacia el centro del círculo, con norma constante
\[
a = \frac{v^2}{R}.
\]
\end{theorem}

\begin{proof}
Centre los ejes en el centro del círculo y configure $\omega = v/R$, de modo que el
El arco $v t$ subtiende el ángulo $\omega t$: la posición es
$x(t) = R\cos(\omega t)$, $y(t) = R\sin(\omega t)$. diferenciando,
$\vect v = (-R\omega\sin(\omega t),\, R\omega\cos(\omega t))$, de
norma constante $R\omega = v$; diferenciandose de nuevo,
$\vect a = -\omega^2\,\vect{OM}$: opuesto al \omterm{def:g12:kinematics-2d:position}{vector de posición} ---
hacia el centro --- de la norma $\omega^2 R = v^2/R$.
\end{proof}

\begin{definition}[Marco Frenet]\label{def:g12:kinematics-2d:frenet}
El \emph{Marco de frenet}\index{Marco de frenet} en la posición actual
de un punto en una trayectoria curva es el par de perpendiculares \omterm{def:g10:orders-of-magnitude:unit}{unidad}
vectores $(\vect{u_t}, \vect{u_n})$: $\vect{u_t}$ tangente, a lo largo de la
movimiento; $\vect{u_n}$ normal, hacia el centro del círculo de
radio $R$ que mejor se ajuste a la curva allí.
\end{definition}

\begin{proposition}[Aceleración en el cuadro Frenet]\label{prop:g12:kinematics-2d:frenetacc}
A lo largo de cualquier trayectoria plana de radio local $R$, recorrida en \omterm{def:g12:kinematics-2d:velocity}{velocidad} $v(t)$,
\[
\vect a = a_t\,\vect{u_t} + a_n\,\vect{u_n},
\qquad a_t = \frac{\dd v}{\dd t},
\qquad a_n = \frac{v^2}{R}:
\]
el \emph{aceleración tangencial}\index{aceleración tangencial}
$a_t$ cambia el \omterm{def:g12:kinematics-2d:velocity}{velocidad}; el \emph{normales
aceleración}\index{aceleración normal} $a_n$ gira la velocidad.
\end{proposition}
\admitted

\begin{remark}[Donde vive la prueba]\label{rem:g12:kinematics-2d:frenetproof}
Probamos los extremos: puro $a_t$ en una línea
(\cref{prop:g12:kinematics-2d:uarm}), $a_n$ puro en un círculo
(\cref{thm:g12:kinematics-2d:centripetal}); el general
La descomposición se deriva del volumen del año ~1.
\end{remark}

\begin{example}[Frenando en curva]\label{ex:g12:kinematics-2d:bend}
Un automóvil entra en una curva de radio $R = \qty{50}{m}$ en $v = \qty{15}{m/s}$
mientras frena en $\dd v/\dd t = \qty{-3.0}{m/s^2}$: $a_t =
\qty{-3.0}{m/s^2}$,$a_n = 15^2/50 = \qty{4.5}{m/s^2}$, entonces$a =
\sqrt{3.0^2 + 4.5^2} \approx \qty{5.4}{m/s^2}$, apuntando hacia atrás
\emph{y} en la curva. Los neumáticos deben abastecer ambos a la vez ---
por qué los pilotos de carreras frenan \emph{antes} la curva.
\end{example}

\section{Problema: el informe del accidente}

\begin{problem}[{Problema del fin de semana --- el informe del accidente: una cámara para salpicadero
película leída fotograma a fotograma, una ley de frenado reconstruida a partir de huellas de derrape, una
rotonda que da testimonio del agarre y un veredicto en kilómetros
por hora}]
\label{pb:g12:kinematics-2d:1}
Un coche choca contra una furgoneta que salía de una calle lateral; nadie esta herido,
pero las aseguradoras no están de acuerdo. El investigador tiene una cámara cenital.
metraje (un cuadro cada \qty{0.50}{s}), marcas de derrape medidas y el
cámara del salpicadero del coche. El límite \omterm{def:g12:kinematics-2d:velocity}{velocidad} es \qty{50}{km/h}.

\medskip
\noindent\textbf{Parte I --- Reading the film.}
Los cuadros ante cualquier reacción dan, a lo largo del camino recto:
\begin{center}\small
\begin{tabular}{lccccc}
$t$ (\unit{s}) & 0.00 & 0.50 & 1.00 & 1.50 & 2.00\\
$x$ (\unit{m}) & 0.0 & 9.0 & 18.0 & 27.0 & 36.0
\end{tabular}
\end{center}
\begin{enumerate}
  \item ¿Por qué el informe debe nombrar primero un marco y ejes? cuales son
        usado aquí?
  \item Calcule la velocidad promedio en cada uno de los cuatro intervalos.
  \item ¿Qué clase de movimiento es este? Justifica a partir de la tabla.
  \item Introduzca el \omterm{def:g12:kinematics-2d:velocity}{velocidad} $v_0$ del automóvil en \unit{m/s} y \unit{km/h}.
  \item ¿Cuál es la aceleración durante estos dos segundos?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The skid marks.}
Las pruebas de neumáticos sobre este asfalto seco dan como resultado una desaceleración de frenado constante
$a = \qty{6.0}{m/s^2}$; las marcas de derrape tienen una longitud $d_b = \qty{27}{m}$.
Tome $t = 0$, $x = 0$ al inicio de la frenada, desconocido \omterm{def:g12:kinematics-2d:velocity}{velocidad} $v_B$.
\begin{enumerate}[resume]
  \item Por antiderivadas, escribir $v(t)$ y $x(t)$ durante el frenado.
  \item Expresar el tiempo de parada $t_s$; mostrar $d_b = v_B^2/(2a)$.
  \item Deducir $v_B$ de las marcas de derrape. ¿Consistente con la Parte I?
  \item El tiempo de reacción es \qty{1.0}{s}: calcular la reacción
        distancia y la distancia total de frenado.
  \item Explique, de la pregunta 7, por qué duplicar el \omterm{def:g12:kinematics-2d:velocity}{velocidad} se cuadriplica
        la distancia de frenado; calcule $d_b$ exactamente en \qty{50}{km/h}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The roundabout testifies.}
Anteriormente, la cámara del tablero muestra el automóvil doblando un carril de rotonda de
radio $R = \qty{12.5}{m}$ a \omterm{def:g12:kinematics-2d:velocity}{velocidad} constante, un cuarto de vuelta en
\qty{2.5}{s}. La carretera puede suministrar como máximo $a_n = \mu g$, con
$\mu = 0.70$.
\begin{enumerate}[resume]
  \item El \omterm{def:g12:kinematics-2d:velocity}{velocidad} es constante allí --- ¿es la velocidad? es el auto
        acelerando? Explicar.
  \item Calcula el \omterm{def:g12:kinematics-2d:velocity}{velocidad} del coche en la rotonda.
  \item Calcula su \omterm{thm:g12:kinematics-2d:centripetal}{aceleración centrípeta} (norma y dirección).
  \item Calcular el máximo antideslizante \omterm{def:g12:kinematics-2d:velocity}{velocidad}. ¿Estaba el auto dentro de él?
  \item Proporcione \omterm{def:g12:kinematics-2d:ucm}{período} y \omterm{def:g12:kinematics-2d:ucm}{frecuencia} de un bucle completo en ese \omterm{def:g12:kinematics-2d:velocity}{velocidad}.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The verdict.}
La furgoneta apareció $D = \qty{41}{m}$ delante del coche en ese instante.
El conductor comenzó a reaccionar.
\begin{enumerate}[resume]
  \item ¿En cuánto, en \unit{km/h} y en porcentaje, salió el \omterm{def:g12:kinematics-2d:velocity}{velocidad}?
        antes de frenar excede el límite?
  \item Calcule el \omterm{def:g12:kinematics-2d:velocity}{velocidad} del automóvil en el momento del impacto.
  \item Rehaga el cálculo de detención exactamente en \qty{50}{km/h} (el mismo
        reacción, misma frenada): ¿se detiene el coche, con qué margen?
  \item Proporcione la fórmula general de distancia de parada $d(v)$; ¿Por qué?
        \omterm{def:g12:kinematics-2d:velocity}{velocidad} ingresa dos veces --- ¿una vez linealmente y otra vez al cuadrado?
  \item El veredicto, una frase con sus números: cuál fue \omterm{def:g12:kinematics-2d:velocity}{velocidad}
        conductor, y el accidente habría ocurrido al límite?
\end{enumerate}
\end{problem}
