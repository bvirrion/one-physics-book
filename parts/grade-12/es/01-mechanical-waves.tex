\chapter{Ondas Mecánicas}\label{ch:g12:mechanical-waves}

Sábado por la noche, estadio lleno: una oleada de espectadores de pie
recorre las gradas en menos de un minuto, pero nadie sale de su
asiento. El anillo de un guijarro cruza el estanque mientras el corcho sólo se balancea. esto
El capítulo estudia al viajero mismo --- el \omterm{def:g10:signals-and-waves:wave}{ola} --- y los cuatro
números que lo domestican: velocidad, \omterm{def:g12:mechanical-waves:delay}{demora}, \omterm{def:g10:signals-and-waves:period}{período}, \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} --- suficiente para

\section{La perturbación viaja, el medio permanece.}

\begin{definition}[onda mecanica]\label{def:g12:mechanical-waves:wave}
A \emph{onda mecanica}\index{onda mecanica} es una perturbación
propagándose a través de un medio material --- cuerda, agua, aire, roca ---
llevando \omterm{def:g10:energy-conservation:energy}{energía} \emph{sin transportar materia}: cada pieza del
El medio se balancea brevemente en su lugar, transmite el movimiento y se acomoda.
\end{definition}

\begin{example}[No importa el transporte]\label{ex:g12:mechanical-waves:transport}
En el estadio \omterm{def:g10:signals-and-waves:wave}{ola} el ``medio'' es la multitud: lo que viaja es el
Acto de ponerse de pie, mientras cada espectador mantiene su asiento. en el
estanque, el anillo se expande pero un corcho flotante sólo se balancea en su lugar.
\end{example}

A \omterm{def:g10:signals-and-waves:wave}{ola} es \emph{transverso}\index{onda transversal} cuando el medio
mueve \emph{perpendicular} a la propagación: una joroba corre a lo largo de un
Se mueve la cuerda mientras cada punto se mueve solo hacia arriba y hacia abajo. es
\emph{longitudinal}\index{onda longitudinal} cuando el medio se mueve
\emph{a lo largo de} la propagación: una zona de \omterm{def:g11:magnetic-fields:solenoid}{bobinas} comprimida viaja hacia abajo
resorte empujado, cada \omterm{def:g11:magnetic-fields:solenoid}{bobina} balanceándose en su lugar.
\end{definition}

\section{Velocidad y retraso}

\begin{definition}[Velocidad de propagación]\label{def:g12:mechanical-waves:speed}
El \emph{velocidad de propagación}\index{velocidad de propagación} de un \omterm{def:g10:signals-and-waves:wave}{ola} es el
distancia recorrida por la perturbación dividida por el tiempo transcurrido:
$v = d/t$. Es la velocidad del \emph{forma} que viaja, no del
medio.
\end{definition}

\begin{proposition}[El medio establece la velocidad]\label{prop:g12:mechanical-waves:medium}
Para pequeñas perturbaciones, el \omterm{def:g12:mechanical-waves:speed}{velocidad de propagación} depende sólo del
\emph{medio}, no de la forma o tamaño de la perturbación. en un
cuerda estirada, $v$ aumenta con \omterm{def:g10:inertia:inventory}{tensión} y disminuye con la masa por
\omterm{def:g10:orders-of-magnitude:unit}{metro}: tenso y ligero es rápido, flojo y pesado es lento. \admitted
\end{proposition}

\begin{remark}[Donde vive la fórmula]\label{rem:g12:mechanical-waves:ropeformula}
La fórmula exacta de la velocidad de la cuerda se obtiene en el volumen del Año 1; esto
año los hechos experimentales anteriores son suficientes.
\end{remark}

\begin{example}[Velocidades del sonido]\label{ex:g12:mechanical-waves:speeds}
\omterm{def:g10:signals-and-waves:sound}{Sonido} viaja a \qty{340}{m/s} en aire, \qty{1500}{m/s} en agua,
\qty{5900}{m/s} en acero --- cuanto más rígido, más rápido. De ahí el
Oído de película occidental en el carril: el acero anuncia el tren primero.
\end{example}

\begin{definition}[Retraso]\label{def:g12:mechanical-waves:delay}
Un \omterm{def:g10:signals-and-waves:wave}{ola} de velocidad $v$ llega a un punto a la distancia $d$ después del
\emph{demora}\index{demora} $\tau = d/v$: dos puntos a distancia $d$
aparte repetir el mismo movimiento, más rezagado $\tau$ --- el
\omterm{rem:g10:signals-and-waves:honest}{eco} rango de \cref{ch:g10:signals-and-waves}, leído al revés.
\end{definition}

\begin{proposition}[Superposición sin interacción]\label{prop:g12:mechanical-waves:superposition}
Cuando dos \omterm{def:g10:signals-and-waves:wave}{ondas} se superponen, el desplazamiento del medio es en cada
instante el \emph{suma} de los desplazamientos que cada uno provocaría por sí solo;
después del cruce, cada \omterm{def:g10:signals-and-waves:wave}{ola} continúa sin cambios --- derivado en el
Año~2 volumen a partir de la linealidad de las ecuaciones de movimiento. \admitted
\end{proposition}

\begin{example}[Cruce de pulsos]\label{ex:g12:mechanical-waves:crossing}
Envía una joroba hacia arriba desde cada extremo de una cuerda: donde se encuentran, el
la cuerda se eleva brevemente hasta la suma de las dos alturas; entonces dos intactos
Surgen jorobas, cuando dos conversaciones cruzadas llegan a sus oyentes.
\end{example}

\section{Ondas periódicas: la doble periodicidad}

\begin{definition}[onda periódica]\label{def:g12:mechanical-waves:periodic}
Cuando la fuente repite su movimiento con \omterm{def:g10:signals-and-waves:period}{período} $T$ y \omterm{def:g10:signals-and-waves:frequency}{frecuencia}
$f = 1/T$ (\cref{ch:g10:signals-and-waves}), alimenta el medio con
\emph{onda periódica}\index{onda periódica}: cada punto repite el suyo
movimiento con \omterm{def:g10:signals-and-waves:period}{período} $T$, cada uno rezagado con respecto a sus vecinos en el recorrido \omterm{def:g12:mechanical-waves:delay}{demora}.
\end{definition}

\begin{definition}[longitud de onda]\label{def:g12:mechanical-waves:wavelength}
Una instantánea de un \omterm{def:g12:mechanical-waves:periodic}{onda periódica} muestra un patrón que se repite en el espacio; es
espacial \omterm{def:g10:signals-and-waves:period}{período} --- cresta a cresta --- es el
\emph{longitud de onda}\index{longitud de onda} $\lambda$, en \omterm{def:g10:orders-of-magnitude:unit}{metros}. A \omterm{def:g12:mechanical-waves:periodic}{periódico
ola} es \emph{doblemente periódico}\index{doble periodicidad}: \omterm{def:g10:signals-and-waves:period}{período} $T$
en el tiempo en cada punto, \omterm{def:g10:signals-and-waves:period}{período} $\lambda$ en el espacio en cada instante.
\end{definition}

\begin{theorem}[La relacion de onda]\label{thm:g12:mechanical-waves:relation}
Un \omterm{def:g12:mechanical-waves:periodic}{onda periódica} de velocidad $v$, \omterm{def:g10:signals-and-waves:period}{período} $T$ y \omterm{def:g10:signals-and-waves:frequency}{frecuencia} $f$ tiene
\omterm{def:g12:mechanical-waves:wavelength}{longitud de onda}$\lambda = v\,T = v/f$.
\end{theorem}

\begin{proof}
Durante un \omterm{def:g10:signals-and-waves:period}{período} el patrón se desliza hacia adelante $v\,T$; pero la fuente
--- por lo tanto, cada punto --- vuelve a su estado inicial, por lo que el
El patrón deslizado coincide con el original: la distancia de repetición es $v\,T$.
\end{proof}

\begin{remark}[La frecuencia pertenece a la fuente.]\label{rem:g12:mechanical-waves:source}
Pasando de un medio a otro --- \omterm{def:g10:signals-and-waves:sound}{sonido} del aire al agua ---
un \omterm{def:g10:signals-and-waves:wave}{ola} mantiene su \omterm{def:g10:signals-and-waves:frequency}{frecuencia}, que dicta la fuente, mientras que su velocidad
cambios; por $\lambda = v/f$ el \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} cambia con él.
\end{remark}

\begin{example}[Concierto A, en el aire y bajo el agua]\label{ex:g12:mechanical-waves:concertA}
El concierto \qty{440}{Hz} A de \cref{ch:g10:signals-and-waves} ha
\omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} $340/440 = \qty{0.77}{m}$ en el aire y $1500/440 =
\qty{3.4}{m}$ en agua: la misma nota, estirada por el medio más rápido.
\end{example}

\begin{definition}[En fase, en oposición de fase]\label{def:g12:mechanical-waves:phase}
Dos puntos separados por un número entero de \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda}, $d = k\lambda$,
se mueven idénticamente en cada instante: son
\emph{en fase}\index{en fase}. Puntos separados por
$d = (k + \tfrac12)\lambda$ siempre se mueve en sentido opuesto --- cresta contra
canal: están en \emph{oposición de fase}\index{oposición de fase}.
\end{definition}

\begin{method}[Lectura de la doble periodicidad]\label{met:g12:mechanical-waves:reading}
\begin{enumerate}
  \item En una grabación $y(t)$ en un punto fijo, lea $T$ cresta para
        cresta; $f = 1/T$.
  \item En una instantánea $y(x)$ en un instante fijo, lea cresta $\lambda$
        a la cresta.
  \item Deducir $v = \lambda/T = \lambda f$; comparar con un directo
        sincronización $v = d/\tau$ cuando esté disponible.
  \item Nunca confundas los dos gráficos: un eje lleva segundos, el
        otro \omterm{def:g10:orders-of-magnitude:unit}{metros}.
\end{enumerate}
\end{method}

\section{Frentes de onda y atenuación}

\begin{definition}[frente de onda]\label{def:g12:mechanical-waves:wavefront}
A \emph{frente de onda}\index{frente de onda} es una línea (o superficie) de puntos
que el \omterm{def:g10:signals-and-waves:wave}{ola} alcanza en el mismo instante --- una línea de cresta en el
estanque. Desde una fuente puntual, los frentes de onda son círculos en expansión; lejos
desde cualquier fuente son casi rectos. El \omterm{def:g10:signals-and-waves:wave}{ola} se propaga
perpendicular a sus frentes de onda, con crestas sucesivas $\lambda$ separadas.
\end{definition}

\begin{definition}[atenuación]\label{def:g12:mechanical-waves:attenuation}
La disminución del \omterm{def:g10:signals-and-waves:amplitude}{amplitud} de un \omterm{def:g10:signals-and-waves:wave}{ola} a medida que se propaga es su
\emph{atenuación}\index{atenuación}. Se suman dos causas: el \omterm{def:g10:energy-conservation:energy}{energía}
de una circular \omterm{def:g12:mechanical-waves:wavefront}{frente de onda} se comparte en una cresta cada vez más larga, y la
El medio absorbe un poco de paso. Heterosexual \omterm{def:g12:mechanical-waves:wavefront}{frentes de onda} escapa del
Primera causa: cómo el oleaje de una tormenta distante puede cruzar un océano.
\end{definition}
