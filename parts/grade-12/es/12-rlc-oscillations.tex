\chapter{Oscilaciones eléctricas libres: RLC}\label{ch:g12:rlc-oscillations}

Gire el pesado dial de una radio del ático y las estaciones desfilarán más allá del
Aguja: una voz, violines, estática, una voz otra vez. Detrás del panel hay
no es \omterm{def:g11:work-of-force:sign}{motor} ni computadora --- un \omterm{def:g11:magnetic-fields:solenoid}{bobina}, un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} cuyo intercalado
placas donde gira el dial y un trozo de cable aéreo. este capitulo
conecta los dos tanques \omterm{def:g10:energy-conservation:energy}{energía} del último capítulo (\cref{ch:g12:rc-rl-circuits})
cara a cara y encuentra el péndulo más rápido jamás construido: carga oscilante
adelante y atrás, hasta millones de veces por segundo, a un ritmo que el dial
elige.

\section{Un péndulo eléctrico: el circuito LC}

Cargue un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} $C$ a un \omterm{def:g11:circuits-and-power:voltage}{Voltaje} $U_0$ --- sus placas luego mantengan
$\pm Q_0$ con $Q_0 = C U_0$ --- y en $t = 0$ conéctelo a un
\emph{ideal} \omterm{def:g11:magnetic-fields:solenoid}{bobina} $L$, de cero \omterm{def:g11:circuits-and-power:resistance}{resistencia}. No hay fuente en ninguna parte del
Bucle: pase lo que pase a continuación, el circuito lo hace por sí solo.

\begin{proposition}[La ecuación LC]\label{prop:g12:rlc-oscillations:equation}
\index{circuito LC}
En el bucle LC ideal la carga $q(t)$ de la placa superior obedece
\[
L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0,
\qquad\text{i.e.}\qquad
\frac{d^2q}{dt^2} = -\frac{1}{LC}\,q .
\]
\end{proposition}

\begin{proof}
Alrededor del bucle la suma \omterm{def:g11:circuits-and-power:voltage}{voltajes} es cero
(\cref{prop:g11:circuits-and-power:laws}): $u_L + u_C = 0$. el
\omterm{def:g12:rc-rl-circuits:capacitor}{condensador} da $u_C = q/C$ y \omterm{def:g11:magnetic-fields:solenoid}{bobina} $u_L = L\,di/dt$
(\cref{def:g12:rc-rl-circuits:inductor}), con $i = dq/dt$ contando
la carga llega a la placa superior
(\cref{prop:g12:rc-rl-circuits:cdudt}); sustituto.
\end{proof}

\begin{theorem}[Oscilaciones libres del circuito LC]\label{thm:g12:rlc-oscillations:oscillation}
\index{oscilaciones electricas}
Lanzado en $t = 0$ con $q(0) = Q_0$ y $i(0) = 0$, el circuito
oscila:
\[
q(t) = Q_0 \cos\!\left(\frac{2\pi t}{T_0}\right),
\qquad
i(t) = -I_0 \sin\!\left(\frac{2\pi t}{T_0}\right),
\qquad
T_0 = 2\pi\sqrt{LC},
\]
con corriente pico $I_0 = 2\pi Q_0/T_0 = Q_0/\sqrt{LC}$.
\end{theorem}

\begin{proof}
Verificar por sustitución. diferenciando,
$i = dq/dt = -\frac{2\pi}{T_0} Q_0 \sin(2\pi t/T_0)$ --- lo indicado
$i(t)$ --- y nuevamente $\frac{d^2q}{dt^2} = -(\frac{2\pi}{T_0})^2 q$,
que es igual a $-q/(LC)$ exactamente cuando $(2\pi/T_0)^2 = 1/(LC)$, es decir\
$T_0 = 2\pi\sqrt{LC}$. Y $q(0) = Q_0$, $i(0) = 0$. que no
La función \emph{otro} se ajusta a la ecuación y se admite el inicio,
exactamente como para la primavera
(\cref{rem:g12:oscillators-and-time:uniqueness}).
\end{proof}

\begin{definition}[Periodo natural y frecuencia]\label{def:g12:rlc-oscillations:period}
El \emph{periodo natural}\index{periodo natural!de un circuito LC} de un
\omterm{prop:g12:rlc-oscillations:equation}{circuito LC} es $T_0 = 2\pi\sqrt{LC}$; es
\emph{frecuencia natural}\index{frecuencia natural!de un circuito LC} es
\[
f_0 = \frac{1}{T_0} = \frac{1}{2\pi\sqrt{LC}} .
\]
Control dimensional: $\qty{1}{H} = \qty{1}{V.s/A}$ y
$\qty{1}{F} = \qty{1}{C/V}$, entonces $LC$ lleva
$\unit{s} \times \unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ es un tiempo,
como debe ser.
\end{definition}

\begin{example}[Primeros números]\label{ex:g12:rlc-oscillations:firstnumbers}
$L = \qty{10}{mH}$ con $C = \qty{1.0}{\micro F}$:
$\sqrt{LC} = \qty{1.0e-4}{s}$, entonces $T_0 = \qty{0.63}{ms}$ y
$f_0 \approx \qty{1.6}{kHz}$ --- un tono audible. $L = \qty{1.0}{mH}$
con $C = \qty{100}{pF}$: $T_0 = \qty{2.0}{\micro s}$,
$f_0 \approx \qty{0.50}{MHz}$ --- una radio \omterm{def:g12:oscillators-and-time:oscillator}{frecuencia}. Sólo el
\emph{producto} $LC$ establece el tempo.
\end{example}

\begin{remark}[Un cuarto de período fuera de sincronismo]\label{rem:g12:rlc-oscillations:quarter}
La corriente es una cuarta parte \omterm{def:g12:oscillators-and-time:oscillator}{período} de \omterm{def:g12:oscillators-and-time:phase}{fase} con el cargo: $i = 0$
cuando $q = \pm Q_0$ (placas llenas, flujo invertido) y $i = \pm I_0$
cuando $q = 0$ (placas vacías, flujo a máxima inclinación) --- exactamente el
péndulo, inmóvil en los extremos y más rápido en la parte inferior.
\end{remark}

\section{El vals de la energía}

\begin{proposition}[Conservación de la energía almacenada]\label{prop:g12:rlc-oscillations:energy}
\index{energía!de un circuito LC}
En el \omterm{prop:g12:rlc-oscillations:equation}{circuito LC} ideal el total almacenado \omterm{def:g10:energy-conservation:energy}{energía}
(\cref{prop:g12:rc-rl-circuits:energy})
\[
E \;=\; E_C + E_L \;=\; \frac{q^2}{2C} + \tfrac12\,L i^2
\;=\; \frac{Q_0^2}{2C} \;=\; \tfrac12\,L I_0^2
\]
es constante: completamente en \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} cuando $q = \pm Q_0$, completamente en
el \omterm{def:g11:magnetic-fields:solenoid}{bobina} cuando $i = \pm I_0$.
\end{proposition}

\begin{proof}
Diferenciar: $\frac{dE}{dt} = \frac{q}{C}\frac{dq}{dt} +
Li\frac{di}{dt} = i\left(\frac{q}{C} + L\frac{d^2q}{dt^2}\right) = 0$
por \cref{prop:g12:rlc-oscillations:equation}. Su valor se lee en
$t = 0$ (todos \omterm{def:g12:rc-rl-circuits:capacitor}{condensador}), y nuevamente en el instante actual
$q = 0$ --- de donde $\tfrac12 L I_0^2 = Q_0^2/2C$, recuperando
$I_0 = Q_0/\sqrt{LC}$ por \omterm{def:g10:energy-conservation:energy}{energía} solo.
\end{proof}

\begin{example}[El vals, medidor]\label{ex:g12:rlc-oscillations:waltz}
$C = \qty{1.0}{\micro F}$ cargado a $U_0 = \qty{10}{V}$, encendido

(\qty{0.16}{ms}) después de que se cierre el interruptor.
\end{example}

\section{Circuitos reales: oscilaciones amortiguadas.}

Cada \omterm{def:g11:magnetic-fields:solenoid}{bobina} real está enrollado con alambre \omterm{def:g11:work-of-force:sign}{resistador}: el bucle honesto es un
circuito de la serie \emph{RLC}, y el \omterm{def:g11:circuits-and-power:resistance}{resistencia} grava el vals.

\begin{definition}[Regímenes pseudoperiódicos y aperiódicos]\label{def:g12:rlc-oscillations:regimes}
\index{circuito RLC}\index{régimen pseudoperiódico}\index{régimen aperiódico}
En un circuito RLC en serie, la resistencia disipa \omterm{def:g10:energy-conservation:energy}{energía} en julios
calefacción (\cref{ch:g11:circuits-and-power}): los \omterm{def:g12:oscillators-and-time:oscillator}{oscilaciones libres} son
\emph{\omterm{def:g12:oscillators-and-time:damping}{amortiguado}}. Dos regímenes, lee un \omterm{def:g10:signals-and-waves:oscilloscope}{osciloscopio}:
\begin{itemize}
  \item \emph{\omterm{def:g10:signals-and-waves:period}{pseudoperiódico}} (pequeño $R$): el circuito aún
        oscila dentro de una envoltura que se encoge, con un tiempo de repetición ---
        el \emph{pseudoperíodo}\index{pseudoperíodo!eléctrico} ---
        cerca de $T_0 = 2\pi\sqrt{LC}$ cuando \omterm{def:g12:oscillators-and-time:damping}{mojadura} es claro;
  \item \emph{aperiódico} (grande $R$): la carga vuelve a cero
        sin sobrepasarse nunca, sin oscilación alguna.
\end{itemize}
El espejo exacto del \omterm{def:g12:oscillators-and-time:damping}{amortiguado} \omterm{def:g12:oscillators-and-time:oscillator}{oscilador mecanico}
(\cref{def:g12:oscillators-and-time:damping}).
\end{definition}

Para mantener vivo el \omterm{def:g12:oscillators-and-time:oscillator}{amplitud}, alimente el circuito exactamente con el \omterm{def:g10:energy-conservation:energy}{energía}
\omterm{def:g11:circuits-and-power:resistance}{resistencia} quema, una vez por ciclo y \emph{al paso} con el columpio: un

\section{El gemelo mecánico}

\begin{proposition}[El diccionario eléctrico-mecánico]\label{prop:g12:rlc-oscillations:analogy}
\index{analogía eléctrico-mecánica}
La ecuación LC es la ecuación resorte-masa.
(\cref{prop:g12:oscillators-and-time:spring-period}) palabra por palabra:
\begin{center}\small
\begin{tabular}{lcl}
\omterm{prop:g12:rlc-oscillations:equation}{circuito LC} & & spring--mass glider\\[2pt]
charge $q$ & $\longleftrightarrow$ & position $x$\\
current $i = dq/dt$ & $\longleftrightarrow$ & velocity $v = dx/dt$\\
\omterm{def:g12:rc-rl-circuits:inductor}{inductancia} $L$ & $\longleftrightarrow$ & mass $m$\\
inverse \omterm{def:g12:rc-rl-circuits:capacitor}{capacidad} $1/C$ & $\longleftrightarrow$ & \omterm{def:g12:oscillators-and-time:spring}{rigidez} $k$\\
\omterm{def:g11:circuits-and-power:resistance}{resistencia} $R$ & $\longleftrightarrow$ & \omterm{def:g10:inertia:inventory}{fricción}\\
$L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0$ & $\longleftrightarrow$ &
  $m\,\frac{d^2x}{dt^2} + kx = 0$\\[2pt]
$T_0 = 2\pi\sqrt{LC}$ & $\longleftrightarrow$ &
  $T_0 = 2\pi\sqrt{m/k}$\\[2pt]
$E_C = q^2/2C$ & $\longleftrightarrow$ & $E_p = \tfrac12 kx^2$\\
$E_L = \tfrac12 Li^2$ & $\longleftrightarrow$ & $E_k = \tfrac12 mv^2$
\end{tabular}
\end{center}
Misma ecuación, mismas soluciones: cada resultado sobre \omterm{def:g12:oscillators-and-time:oscillator}{oscilador}
se traduce instantáneamente al otro idioma.
\end{proposition}

\begin{proof}
Comparar \cref{prop:g12:rlc-oscillations:equation} con
$m\,d^2x/dt^2 = -kx$: sustituyendo $q \to x$, $L \to m$, $1/C \to k$
convierte uno en el otro, y $LC \to m/k$ se convierte
$2\pi\sqrt{LC}$ en $2\pi\sqrt{m/k}$.
\end{proof}

\begin{example}[Sintonizando una radio]\label{ex:g12:rlc-oscillations:tuning}
Una antena envía un susurro de la estación \emph{cada} a un bucle LC, pero
sólo el que transmite cerca de $f_0$ avanza y aumenta ---
\omterm{def:g12:oscillators-and-time:resonance}{resonancia} (\cref{def:g12:oscillators-and-time:resonance}). el dial
gira las placas de una variable \omterm{def:g12:rc-rl-circuits:capacitor}{condensador}: cambiando movimientos $C$
$f_0 = 1/(2\pi\sqrt{LC})$, y la aguja pasa el auricular.
estación a la vez. En el lado transmisor, un \omterm{def:g12:oscillators-and-time:oscillator}{oscilador} mantenido
(\cref{rem:g12:rlc-oscillations:maintain}) establece la transmisión
\omterm{def:g12:oscillators-and-time:oscillator}{frecuencia}; en un reloj, un cristal de cuarzo, un gemelo mecánico de
impresionante \omterm{def:g12:oscillators-and-time:spring}{rigidez} --- juega el mismo papel.
\end{example}
