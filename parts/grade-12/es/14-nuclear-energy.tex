\chapter{Energía nuclear: fisión, fusión, $E=mc^2$}\label{ch:g12:nuclear-energy}

Una pastilla de combustible del tamaño de la goma de un lápiz cubre la electricidad de un hogar
por un año; el mismo servicio de carbón requiere tres toneladas. Arriba, el sol
paga por la luz del día desapareciendo: cuatro millones de toneladas de sí mismo por segundo.
Este capítulo abre el libro de cuentas detrás de ambos acuerdos: la masa misma es
\omterm{def:g10:energy-conservation:energy}{energía}, y núcleos (\cref{ch:g11:nucleus-radioactivity}) son donde se cobra.

\section{la masa es energía}

\begin{theorem}[Equivalencia masa-energía]\label{thm:g12:nuclear-energy:emc2}\index{equivalencia masa-energía}
Un cuerpo de masa $m$, por el solo hecho de existir, contiene el \omterm{def:g10:energy-conservation:energy}{energía}
\[
E = m c^2, \qquad c = \qty{3.00e8}{m/s}.
\]
Cada vez que un sistema pierde masa $\Delta m$, libera el \omterm{def:g10:energy-conservation:energy}{energía}
$\Delta m\, c^2$ --- cualquiera que sea el mecanismo.
\end{theorem}
\admitted

\begin{remark}[Einstein, 1905]\label{rem:g12:nuclear-energy:einstein}
Ésta es la relación de Einstein, admitida aquí; de donde viene es el negocio
del capítulo sobre relatividad especial que cierra este año
(\cref{ch:g12:special-relativity}). Tenga en cuenta la tarifa: $c^2 = \qty{9.00e16}{J/kg}$
--- un error de redondeo de masa es una fortuna de \omterm{def:g10:energy-conservation:energy}{energía}.
\end{remark}

\begin{definition}[Unidad electrón-voltio y masa atómica]\label{def:g12:nuclear-energy:units}
Dos \omterm{def:g10:orders-of-magnitude:unit}{unidades} adaptados al \omterm{def:g11:fundamental-interactions:ladder}{núcleo}. el
\emph{electronvoltio}\index{electronvoltio}, el \omterm{def:g10:energy-conservation:energy}{energía} gana un electrón
cruce \qty{1}{V} (\cref{ch:g11:circuits-and-power}):
$\qty{1}{eV} = \qty{1.60e-19}{J}$, $\qty{1}{MeV} = \qty{1.60e-13}{J}$ ---
los enlaces químicos intercambian algunos \unit{eV}, reacciones nucleares \omterm{def:g11:circuits-and-power:voltage}{megaelectrones-voltios}.
El \emph{unidad de masa atómica unificada}\index{unidad de masa atómica unificada}, uno
duodécimo de la masa de un carbono-12 \omterm{def:g11:fundamental-interactions:ladder}{átomo}: $\qty{1}{u} = \qty{1.66054e-27}{kg}$,
de \omterm{def:g10:energy-conservation:energy}{energía} equivalente \qty{931.5}{MeV} --- $\qty{1}{u} = \qty{931.5}{MeV}/c^2$.
\end{definition}

\begin{example}[La fortuna dormida]\label{ex:g12:nuclear-energy:fortune}
Un \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de cualquier cosa, completamente convertido, es $\qty{9.00e16}{J}$ --- tres
años de producción de una gran estación (\qty{1}{GW}) \omterm{def:g11:work-of-force:power}{fuerza}. Nadie se dio cuenta antes
1905 porque la física ordinaria apenas toca la masa: quemar un \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de carbón
(\qty{3.0e7}{J}) lo aclara en un tercio de microgramo. Núcleos, ya veremos,
mueve una parte entre mil: aún pequeño, pero un millón de veces la química.
\end{example}

\section{El defecto masivo}

\begin{definition}[Defecto de masa y energía de enlace]\label{def:g12:nuclear-energy:defect}
Pesar un \omterm{def:g11:fundamental-interactions:ladder}{núcleo} ${}^{A}_{Z}\mathrm{X}$ (masa $m$), luego pesar sus protones $Z$
y neutrones $A - Z$ por separado: las partes son \emph{más pesado} que el conjunto.
la diferencia
\[
\Delta m = Z\,m_p + (A - Z)\,m_n - m > 0,
\qquad m_p = \qty{1.00728}{u},\; m_n = \qty{1.00866}{u},
\]
es el \emph{defecto masivo}\index{defecto masivo}; el \emph{vinculante
energía}\index{energía de unión} $E_b = \Delta m\,c^2$ es el \omterm{def:g10:energy-conservation:energy}{energía} necesario para
separe el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} en \omterm{def:g11:fundamental-interactions:ladder}{nucleones} libre --- igualmente, el \omterm{def:g10:energy-conservation:energy}{energía} se liberó
cuando está ensamblado. Un \omterm{def:g11:fundamental-interactions:ladder}{núcleo} encuadernado asienta \emph{abajo} sus partes.
\end{definition}

\begin{example}[Helio-4]\label{ex:g12:nuclear-energy:helium}
El ${}^{4}_{2}\mathrm{He}$ \omterm{def:g11:fundamental-interactions:ladder}{núcleo} tiene $m = \qty{4.00151}{u}$; sus partes:
$2 \times 1.00728 + 2 \times 1.00866 = \qty{4.03188}{u}$. entonces
$\Delta m = \qty{0.03037}{u}$ y $E_b = 0.03037 \times 931.5 \approx
\qty{28.3}{MeV}$: el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} pesa$0.75\%$ menos que sus piezas --- el
``una parte en mil'' de \cref{ex:g12:nuclear-energy:fortune}.
\end{example}

\begin{method}[Balance energético de una reacción nuclear.]\label{met:g12:nuclear-energy:balance}
\begin{enumerate}
  \item Equilibre la ecuación: total $A$ y $Z$ conservados (\cref{ch:g11:nucleus-radioactivity}).
  \item Suma las masas antes y después, en \unit{u} --- mantén los cinco decimales.
  \item $\Delta m = m_{\text{before}} - m_{\text{after}}$, entonces
        $E = \Delta m\,(\text{in u}) \times \qty{931.5}{MeV}$, llevado como
        \omterm{def:g11:mechanical-energy:kinetic}{energía cinética} de los productos; negativo significa que la reacción debe ser pagada.
\end{enumerate}
\end{method}

\section{La curva que recorre el universo.}

\begin{definition}[Energía de enlace por nucleón]\label{def:g12:nuclear-energy:pernucleon}
La cohesión se compara de manera justa con \emph{energía de enlace por
nucleón}\index{energía de enlace por nucleón}, $E_b/A$, el costo promedio de
extrayendo un \omterm{def:g11:fundamental-interactions:ladder}{nucleón}: $28.3/4 \approx \qty{7.1}{MeV}$ para helio-4.
\end{definition}

\begin{proposition}[El pico de hierro, y dos caminos de bajada.]\label{prop:g12:nuclear-energy:ironpeak}\index{pico de hierro}
Trazado contra $A$, el \omterm{def:g12:nuclear-energy:pernucleon}{energía de enlace por nucleón} sube abruptamente
los núcleos ligeros, alcanzan su punto máximo cerca del hierro, ${}^{56}_{26}\mathrm{Fe}$, aproximadamente
\qty{8.8}{MeV} por \omterm{def:g11:fundamental-interactions:ladder}{nucleón}, luego se desliza suavemente hasta aproximadamente \qty{7.6}{MeV} en
uranio: el hierro es el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} más fuertemente unido en la naturaleza. una reacción
lanza \omterm{def:g10:energy-conservation:energy}{energía} exactamente cuando sus productos se ubican \emph{más alto} en la curva; dos
estrategias opuestas, por lo tanto, ambos pagan y ambos caminan hacia el hierro ---
\emph{terrible} los núcleos más pesados y \emph{fusionando} el más ligero.
\end{proposition}
\admitted

\begin{remark}[¿Por qué un pico?]\label{rem:g12:nuclear-energy:whypeak}
El tira y afloja de \cref{ch:g11:fundamental-interactions}: el pegamento fuerte une
tocando sólo a los vecinos, la repulsión de Coulomb se extiende por todo el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- pequeño
los núcleos son en su mayoría de superficie, los grandes en su mayoría de repulsión, hierro el compromiso.
La contabilidad honesta vive en los volúmenes universitarios.
\end{remark}

\section{La fisión y la reacción en cadena.}

\begin{definition}[Fisión]\label{def:g12:nuclear-energy:fission}
\emph{Fisión}\index{fisión} es la división de un pesado \omterm{def:g11:fundamental-interactions:ladder}{núcleo} en dos
fragmentos de tamaño mediano. Uranio-235 es \emph{físil}\index{físil}: un lento
El neutrón, absorbido, deja al \omterm{def:g11:fundamental-interactions:ladder}{núcleo} tan agitado que se parte en dos,
escupiendo dos o tres neutrones frescos ---
${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to \text{two fragments} +
2\text{--}3\;{}^{1}_{0}\mathrm{n}$, el par varía de fisión en fisión.
\end{definition}

\begin{example}[Una fision, pesada]\label{ex:g12:nuclear-energy:fissionbalance}
Una ruta frecuente, por \cref{met:g12:nuclear-energy:balance}:
\[
{}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \longrightarrow
{}^{94}_{38}\mathrm{Sr} + {}^{140}_{54}\mathrm{Xe} + 2\,{}^{1}_{0}\mathrm{n}.
\]
Antes: $1.00866 + 234.99350 = \qty{236.00216}{u}$. Después:
$93.89450 + 139.89200 + 2 \times 1.00866 = \qty{235.80382}{u}$. Por lo tanto
$\Delta m = \qty{0.19834}{u}$ y $E \approx \qty{185}{MeV}$; contando el
desintegraciones posteriores de los fragmentos radiactivos (\cref{ch:g12:radioactive-decay}),
cada \omterm{def:g12:nuclear-energy:fission}{fisión} vale aproximadamente \qty{200}{MeV} --- unos pocos \unit{eV} compran un \omterm{def:g11:fundamental-interactions:ladder}{átomo}
de carbón.
\end{example}

\begin{definition}[Reacción en cadena y masa crítica]\label{def:g12:nuclear-energy:chain}
Cada \omterm{def:g12:nuclear-energy:fission}{fisión} se ilumina con un neutrón y libera dos o tres: en una cantidad suficientemente grande
grumo golpean otros núcleos y la reacción se alimenta a sí misma --- un
\emph{reacción en cadena}\index{reacción en cadena}. En un pequeño bulto la mayoría de los neutrones
escapar primero por la superficie; la cantidad mínima que sostiene la cadena
es el \emph{masa crítica}\index{masa crítica} --- debajo está la cadena
se apaga, por encima de él se multiplica cada generación: el principio de la bomba.
\end{definition}

Un reactor \omterm{def:g11:work-of-force:power}{fuerza} mantiene la cadena exactamente a un neutrón por \omterm{def:g12:nuclear-energy:fission}{fisión}. El combustible:
bolitas de uranio modestamente enriquecidas en uranio-235. el
\emph{moderador}\index{moderador} (agua corriente, normalmente) ralentiza el ayuno
\omterm{def:g12:nuclear-energy:fission}{fisión} neutrones por colisiones --- los neutrones lentos son mucho mejores para activar
el siguiente \omterm{def:g12:nuclear-energy:fission}{fisión}; el \emph{barras de control}\index{barra de control} (boro, cadmio)
devorar neutrones --- empujados hacia adentro, la cadena muere; prolongado, se acelera. el
La agitación de los fragmentos se convierte en calor, luego vapor y luego electricidad.
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Un grano de pimienta contra un camión]\label{ex:g12:nuclear-energy:gram}
Un gramo de uranio-235 contiene
$N = \num{1.0e-3} / (235 \times \num{1.66054e-27}) \approx \num{2.56e21}$
núcleos; en $\qty{200}{MeV} = \qty{3.2e-11}{J}$ cada uno, fisionándolos a todos
rinde alrededor de \qty{8.2e10}{J} --- el \omterm{def:g10:energy-conservation:forms}{energía química} de $\num{8.2e10} /
\num{3.0e7} \approx \qty{2.7e3}{kg}$ de carbón: un grano de pimienta por tres toneladas.
\end{example}

\section{Fusión: el fuego propio del Sol}

\begin{definition}[Fusión]\label{def:g12:nuclear-energy:fusion}
\emph{Fusión}\index{fusión} es la fusión de dos núcleos ligeros en uno más pesado,
uno más estrechamente unido. La reacción más accesible combina deuterio y
tritio, el pesado \omterm{def:g11:nucleus-radioactivity:notation}{isótopos} del hidrógeno (\cref{ch:g11:nucleus-radioactivity}):
\[
{}^{2}_{1}\mathrm{H} + {}^{3}_{1}\mathrm{H} \longrightarrow
{}^{4}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}.
\]
\end{definition}

\begin{example}[D--T, pesado]\label{ex:g12:nuclear-energy:dt}
Antes: $2.01355 + 3.01550 = \qty{5.02905}{u}$; después:
$4.00151 + 1.00866 = \qty{5.01017}{u}$; entonces $\Delta m = \qty{0.01888}{u}$ y
$E \approx \qty{17.6}{MeV}$, cuatro quintas partes en el neutrón --- menos de un
\omterm{def:g12:nuclear-energy:fission}{fisión}, pero de cinco \omterm{def:g11:fundamental-interactions:ladder}{nucleones} en lugar de $236$: alrededor de \qty{3.4e14}{J} por
\omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de combustible, cuatro veces \omterm{def:g12:nuclear-energy:fission}{fisión}. Helio-4
(\cref{ex:g12:nuclear-energy:helium}) vuelve a ser la ceniza.
\end{example}

\begin{definition}[La barrera de Coulomb]\label{def:g12:nuclear-energy:barrier}
Dos núcleos son ambos positivos: para tocarse, primero deben subir la colina de
su repulsión eléctrica (\cref{ch:g11:electric-gravitational-fields}) --- el
\emph{Barrera de Coulomb}\index{Barrera de Coulomb}. Sólo cerca de \qty{1e8}{K}
colisiones llevan el \omterm{def:g10:energy-conservation:energy}{energía} para cruzarlo; el combustible es entonces un
\emph{plasma}\index{plasma} de núcleos desnudos y electrones. \omterm{def:g12:nuclear-energy:fission}{Fisión} no paga
tal peaje: su disparador, el neutrón, es neutral y \omterm{def:g11:work-of-force:work}{obras} frío.
\end{definition}

\begin{example}[El Sol corre sobre él]\label{ex:g12:nuclear-energy:sun}
El núcleo del Sol, en $\qty{1.5e7}{K}$, \omterm{def:g11:circuits-and-power:joule}{fusibles} hidrógeno ordinario paso a paso ---
el \emph{protón - cadena de protones}, de efecto neto
$4\,{}^{1}_{1}\mathrm{H} \to {}^{4}_{2}\mathrm{He} + 2\,{}^{0}_{+1}\mathrm{e}
+ \text{radiation}$, aproximadamente \qty{26}{MeV} por helio:$0.7\%$ del
La masa del hidrógeno sale como \omterm{def:g10:energy-conservation:energy}{energía}. La luz del sol en cuyo \omterm{def:g12:sound-acoustics:timbre}{espectro} leemos
\cref{ch:g10:light-spectra} es esta masa, llegando ocho minutos tarde; cada
La estrella brilla recorriendo la curva hacia el hierro.
\end{example}

\begin{remark}[ITER: embotellando una estrella]\label{rem:g12:nuclear-energy:iter}
Ningún sólido puede contener un \omterm{def:g12:nuclear-energy:barrier}{plasma} en \qty{1.5e8}{K}; un \emph{tokamak} lo enjaula en un
\omterm{def:g11:magnetic-fields:bfield}{campo magnético} (\cref{ch:g11:magnetic-fields}), un donut de \omterm{def:g11:electric-gravitational-fields:lines}{líneas de campo} el
las partículas cargadas giran en espiral sin tocar la pared; ITER, construido para
liberar diez veces su calentamiento \omterm{def:g11:work-of-force:power}{fuerza}, es el paso actual. romper el
confinamiento y el \omterm{def:g12:nuclear-energy:barrier}{plasma} se enfría y se detiene en segundos: no \omterm{def:g12:nuclear-energy:chain}{reacción en cadena}, no
\omterm{def:g12:nuclear-energy:chain}{masa crítica} --- lo difícil no es detener el fuego sino mantenerlo encendido.
\end{remark}

\begin{example}[La escalera de energía, por kilogramo.]\label{ex:g12:nuclear-energy:ladder}
\omterm{def:g10:energy-conservation:energy}{Energía} por \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de combustible: carbón, \qty{3.0e7}{J}; uranio-235 por \omterm{def:g12:nuclear-energy:fission}{fisión},
\qty{8.2e13}{J}; deuterio-tritio por \omterm{def:g12:nuclear-energy:fusion}{fusión}, \qty{3.4e14}{J}; total
conversión ($E = mc^2$), \qty{9.0e16}{J}. Seis órdenes de magnitud separados
química del \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- la razón por la que el camión reposta combustible a un reactor
y una planta de carbón por tren.
\end{example}

\section{Problema: alimentar una ciudad}

\begin{problem}[{Problema del fin de semana --- alimentar una ciudad: un gigavatio constante por
medio millón de personas, compraron de cuatro maneras: bolitas de uranio, trenes llenos de
carbón, un charco de agua de mar, la propia sustancia del Sol --- con el mismo kilogramo
de masa desaparecida escondida debajo de los cuatro}]
\label{pb:g12:nuclear-energy:1}
Una ciudad de $500\,000$ habitantes tira de una electricidad constante \omterm{def:g11:work-of-force:power}{fuerza}
$P = \qty{1.0}{GW}$; sus plantas, sea cual sea el combustible, convierten el calor en
electricidad con \omterm{def:g10:energy-conservation:efficiency}{eficiencia} $33\%$ (\cref{ch:g10:energy-conservation}).
Datos: un uranio-235 \omterm{def:g12:nuclear-energy:fission}{fisión}, \qty{200}{MeV} en total; carbón, \qty{3.0e7}{J/kg};
un D--T \omterm{def:g12:nuclear-energy:fusion}{fusión}, \qty{17.6}{MeV} de \qty{5.029}{u} de combustible;
$\qty{1}{u} = \qty{1.66054e-27}{kg}$; $\qty{1}{eV} = \qty{1.60e-19}{J}$; uno
año $= \qty{3.16e7}{s}$; el Sol: \qty{3.9e26}{W}, \qty{2.0e30}{kg}; un cúbico
\omterm{def:g10:orders-of-magnitude:unit}{metro} de agua de mar contiene aproximadamente \qty{33}{g} de deuterio.

\medskip
\noindent\textbf{Parte I --- The \omterm{def:g12:nuclear-energy:fission}{fisión} plant.}
\begin{enumerate}
  \item ¿Qué \omterm{def:g11:work-of-force:power}{fuerza} térmica debe entregar el reactor?
  \item Expresa el \omterm{def:g10:energy-conservation:energy}{energía} de un \omterm{def:g12:nuclear-energy:fission}{fisión} en \omterm{def:g11:work-of-force:work}{julios}.
  \item ¿Cuántos \omterm{def:g12:nuclear-energy:fission}{fisiones} por segundo recorren la ciudad?
  \item La masa de un uranio-235 \omterm{def:g11:fundamental-interactions:ladder}{núcleo} ($A = 235$), luego el uranio-235
        consumido por segundo?
  \item Deducir el uranio-235 consumido en un año, en \omterm{def:g10:orders-of-magnitude:unit}{kilogramos}.
  \item El combustible se enriquece con $4.0\%$ uranio-235: ¿qué masa tiene un año?
        ¿Tomar y cabe en un camión?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The coal ledger.}
\begin{enumerate}[resume]
  \item Calcular el \omterm{def:g10:energy-conservation:forms}{energía termal} que consume la ciudad en un año.
  \item ¿Qué masa de carbón lo entrega?
  \item Calcule la relación (masa de carbón)/(masa de uranio-235) para un año, y
        compruébelo con la escalera de \cref{ex:g12:nuclear-energy:ladder}.
  \item Un tren de carga transporta \qty{3.0e6}{kg}: cuántos trenes por año, y
        por dia? Una frase: imagina las dos líneas de suministro.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The \omterm{def:g12:nuclear-energy:fusion}{fusión} dream.}
\begin{enumerate}[resume]
  \item Expresar el \omterm{def:g10:energy-conservation:energy}{energía} de un D--T \omterm{def:g12:nuclear-energy:fusion}{fusión} en \omterm{def:g11:work-of-force:work}{julios}.
  \item Calcular el \omterm{def:g10:energy-conservation:energy}{energía} liberado por \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de combustible D--T; comparar con
        la escalera.
  \item ¿Qué masa de combustible D--T cubre el año de la ciudad? ¿Qué masa es eso?
        ¿Deuterio ($2.014$ del \qty{5.029}{u})?
  \item ¿Qué volumen de agua de mar contiene ese deuterio? Comparar con un
        Piscina olímpica, aproximadamente \qty{2.5e3}{m^3}. (El tritio se obtiene a partir del litio).
  \item ¿Por qué esta planta no lleva ni \omterm{def:g12:nuclear-energy:chain}{masa crítica} ni \omterm{def:g12:nuclear-energy:chain}{reacción en cadena} a
        miedo? ¿Cuál es entonces la parte difícil?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The star that pays in mass.}
\begin{enumerate}[resume]
  \item De $E = mc^2$, ¿qué masa convierte el Sol cada segundo?
  \item ¿Qué masa por segundo y por año debe ser \emph{cualquier}?
        \qty{3.0}{GW}-conversión de fuente térmica? Compruebe que los tres combustibles
        entrega esta misma masa.
  \item ¿Qué masa total ha irradiado el Sol en su $\num{4.5e9}$?
        años, ¿y qué fracción del Sol es esa?
  \item \omterm{def:g12:nuclear-energy:fusion}{Fusión} puede aprovechar aproximadamente $0.07\%$ de la masa del Sol (la parte quemable de
        su núcleo): estima su vida total como estrella y lo que queda.
  \item Cerrar la auditoría en tres frases: un año de la ciudad en toneladas
        de carbón, \omterm{def:g10:orders-of-magnitude:unit}{kilogramos} de uranio-235 y \omterm{def:g10:orders-of-magnitude:unit}{kilogramos} de combustible D--T --- y
        el \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de masa que, en cada escenario, se convirtió en la luz de la ciudad.
\end{enumerate}
\end{problem}
