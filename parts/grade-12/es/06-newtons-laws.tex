\chapter{Leyes de Newton}\label{ch:g12:newtons-laws}

Introducir de contrabando una báscula de baño en un ascensor. A medida que la cabina se eleva, la aguja indica
tres \omterm{def:g10:orders-of-magnitude:unit}{kilogramos} pesados; cerca de la parte superior se lee luz; entre los dos, la pura verdad
--- nada a bordo cambió excepto el movimiento. Este capítulo establece las tres leyes detrás
esa aguja y los forja en el método que ejecuta todas las mecánicas: elige un
sistema, dibuje su \omterm{def:g10:inertia:force}{efectivo}, proyecto $\sum \vect F = m\vect a$.

\section{Momentum y la primera ley hecha exacta}

\begin{definition}[Impulso]\label{def:g12:newtons-laws:momentum}
El \emph{impulso}\index{impulso} de un cuerpo de masa $m$ (\unit{kg}) que se mueve con
velocidad $\vect v$ es el vector
\[
\vect p = m \vect v \qquad (\unit{kg.m/s}):
\]
la velocidad, ponderada por la masa; para un sistema, la suma vectorial de sus partes.
\end{definition}

\begin{definition}[Marco de referencia inercial]\label{def:g12:newtons-laws:inertial}
Un \emph{marco de referencia inercial}\index{marco de referencia inercial} es aquel en el que
cada cuerpo aislado --- no \omterm{def:g10:inertia:force}{efectivo}, o \omterm{def:g10:inertia:force}{efectivo} que \omterm{def:g10:inertia:compensate}{compensar} --- se mueve en línea recta
línea a velocidad constante (\cref{ch:g10:inertia}). El terreno es excelente.
precisión, y también lo es cualquier cabina que se mueva uniformemente y en línea recta sobre ella; una frenada
El tren o el carrusel no lo son: allí el equipaje suelto acelera sin empuje.
\end{definition}

\begin{theorem}[Primera ley de Newton]\label{thm:g12:newtons-laws:first}\index{La primera ley de Newton}
En un marco inercial, el \omterm{def:g12:newtons-laws:momentum}{impulso} de un sistema aislado no cambia: $\vect p$
es constante, es decir\ $d\vect p/dt = \vect 0$.
\end{theorem}
\admitted

\begin{remark}[Inercia, afilado dos veces]\label{rem:g12:newtons-laws:sharpened}
El principio de inercia de \cref{ch:g10:inertia}, actualizado dos veces: dice
\emph{dónde} sostiene --- define los marcos inerciales como aquellos donde lo hace ---
y guarda un vector, $m\vect v$, cuya contabilidad da sus frutos al final del capítulo.
\end{remark}

\section{Segunda ley de Newton}

\begin{theorem}[Segunda ley de Newton]\label{thm:g12:newtons-laws:second}\index{Segunda ley de Newton}
En un sistema inercial, el \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} de un cuerpo es igual a la tasa de cambio de su
\omterm{def:g12:newtons-laws:momentum}{impulso}; para masa constante, desde $\vect p = m\vect v$,
\[
\sum \vect F = \frac{d\vect p}{dt} = m\,\frac{d\vect v}{dt} = m \vect a .
\]
\end{theorem}
\admitted

\begin{remark}[Una ley de la naturaleza, y su unidad.]\label{rem:g12:newtons-laws:nature}
No hay pruebas: ésta es una ley de la naturaleza, comparada con experimentos durante tres
siglos. El año pasado (\cref{ch:g11:forces-and-motion}) dio su sombra,
$\Delta v \propto F\,\Delta t/m$; la derivada (\cref{ch:g12:kinematics-2d}) hace
es exacto en cada instante --- su letra pequeña espera en el volumen del Año 1. \omterm{def:g10:inertia:force}{Fuerza} obtiene un
\omterm{def:g10:orders-of-magnitude:unit}{unidad}, $\unit{N} = \unit{kg.m/s^2}$; y un cuerpo bajo su \omterm{def:g10:universal-gravitation:weight}{peso} solo obedece
$m\vect a = m\vect g$: toda masa cae con $\vect a = \vect g$, y
$g = \qty{9.81}{N/kg} = \qty{9.81}{m/s^2}$ --- las dos lecturas son una.
\end{remark}

\begin{example}[Fuerza promedio, no se necesitan detalles]\label{ex:g12:newtons-laws:average}
Un automóvil \qty{1300}{kg} pasa del reposo a \qty{100}{km/h} (\qty{27.8}{m/s}) en
\qty{8.0}{s}: cualquiera que sea la historia instante a instante, el \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} promedio es
$\Delta p/\Delta t = 1300 \times 27.8/8.0 \approx \qty{4.5e3}{N}$ --- \omterm{def:g12:newtons-laws:momentum}{impulso} vueltas
una pregunta \omterm{def:g10:inertia:force}{fuerza} sobre la contabilidad del antes y el después.
\end{example}

\section{tercera ley de newton}

\begin{theorem}[Tercera ley de Newton]\label{thm:g12:newtons-laws:third}\index{tercera ley de newton}
Si un cuerpo $A$ ejerce un \omterm{def:g10:inertia:force}{fuerza} $\vect F_{A \to B}$ sobre un cuerpo $B$, entonces $B$ ejerce sobre
$A$ o \omterm{def:g10:inertia:force}{fuerza} $\vect F_{B \to A} = -\vect F_{A \to B}$: misma línea de acción, misma
magnitud, direcciones opuestas --- en cada instante, cualquiera que sea el movimiento, contacto o
acción a distancia.
\end{theorem}
\admitted

\begin{example}[El libro, la mesa y la tierra.]\label{ex:g12:newtons-laws:book}
Un libro reposa sobre una mesa: dos interacciones, dos parejas. Gravitacional: la Tierra atrae
el libro hacia abajo ($\vect P$), el libro levanta la Tierra. Contacto: la mesa empuja
el libro hacia arriba ($\vect N$), el libro presiona la mesa hacia abajo. Caminar es lo mismo:
empujar el suelo hacia atrás; su reacción es la única \omterm{def:g10:inertia:force}{fuerza} que te impulsa hacia adelante.
\end{example}

\section{El método: elegir, dibujar, proyectar}

\begin{method}[Resolviendo un problema de dinámica]\label{met:g12:newtons-laws:method}
\begin{enumerate}
  \item \emph{Sistema}: nombrar el cuerpo (o conjunto de cuerpos) estudiado.
  \item \emph{Marco}: elige un marco inercial y ejes adaptados al movimiento ---
        ejes de pendiente, o el \omterm{def:g12:kinematics-2d:frenet}{Marco de frenet} para una curva (\cref{ch:g12:kinematics-2d}).
  \item \emph{Inventario}: enumera cada \omterm{def:g10:inertia:force}{fuerza} --- el \omterm{def:g10:universal-gravitation:weight}{peso}, luego un \omterm{def:g10:inertia:force}{fuerza} por
        contacto, nada más --- y dibuja el diagrama de cuerpo libre: un punto, un
        flecha según \omterm{def:g10:inertia:force}{fuerza}.
  \item \emph{Proyecto}: escribe $\sum \vect F = m\vect a$, proyéctalo en cada eje.
  \item \emph{resolver y comprobar}: \omterm{def:g10:orders-of-magnitude:unit}{unidades}, signos, casos limitantes ($\mu \to 0$,
        $\alpha \to 0$, \dots).
\end{enumerate}
\end{method}

\begin{definition}[Fuerza y fricción normales]\label{def:g12:newtons-laws:friction}
Una superficie actúa sobre un cuerpo a través de dos componentes: la \emph{fuerza normal}\index{normales
fuerza} $\vect N$, perpendicular a la superficie, y la \omterm{def:g10:inertia:inventory}{fricción} $\vect f$, tangencial.
El experimento da las leyes \omterm{def:g10:inertia:inventory}{fricción}: mientras el cuerpo no se desliza, estático \omterm{def:g10:inertia:inventory}{fricción}
se ajusta hasta un techo, $f_s \leq \mu_s N$; una vez que se desliza, \omterm{def:g11:forces-and-motion:friction}{fricción cinética}
es $f_k = \mu_k N$, contra el deslizamiento. El \emph{coefficients of \omterm{def:g10:inertia:inventory}{fricción}}%
\index{coeficiente de fricción} $\mu_s$ y $\mu_k$ ($\mu_k \leq \mu_s$) dependen únicamente
sobre los dos materiales.
\end{definition}

\begin{example}[La inclinacion, dos veces]\label{ex:g12:newtons-laws:incline}
Un bloque se desliza por una pendiente de ángulo $\alpha$; ejes: $x$ cuesta abajo, $y$
perpendicular. En $y$: $N = mg\cos\alpha$. En $x$, sin fricción: $ma = mg\sin\alpha$,
entonces $a = g\sin\alpha$ --- sin masa; en $\alpha = 30^\circ$, $a = \qty{4.9}{m/s^2}$.
Con \omterm{def:g11:forces-and-motion:friction}{fricción cinética}: $a = g(\sin\alpha - \mu_k\cos\alpha)$; para $\mu_k = 0.20$,
$a = 9.81\,(0.500 - 0.173) = \qty{3.2}{m/s^2}$. Y se quedó quieto en primer lugar
sólo mientras la estática necesaria \omterm{def:g10:inertia:inventory}{fricción} quepa debajo de su techo: $\tan\alpha \leq \mu_s$.
\end{example}

\begin{definition}[pesos aparentes]\label{def:g12:newtons-laws:apparent}
El \emph{peso aparente}\index{peso aparente} de un cuerpo es el \omterm{def:g12:newtons-laws:friction}{fuerza normal}
entre él y su soporte --- qué escala se lee debajo.
\end{definition}

\begin{example}[El ascensor]\label{ex:g12:newtons-laws:elevator}
Un pasajero \qty{70}{kg} se encuentra sobre una báscula en un ascensor de aceleración vertical.
$a_z$ (positivo hacia arriba): $N - mg = ma_z$, entonces $N = m(g + a_z)$. Levantándose en
$a_z = \qty{1.5}{m/s^2}$: \qty{792}{N}, pesado; frenado cerca de la parte superior: \qty{582}{N},
luz; crucero constante: $mg = \qty{687}{N}$ --- la primera ley; caída libre, $a_z = -g$:
$N = 0$, ingravidez en una báscula.
\end{example}

\begin{example}[Dos bloques y una polea]\label{ex:g12:newtons-laws:pulley}
Un bloque $m_1 = \qty{4.0}{kg}$ sobre una mesa sin fricción está atado con una cuerda liviana, sobre
una polea ideal, a un bloque colgante $m_2 = \qty{1.0}{kg}$. La cuerda transmite la
mismo \omterm{def:g10:inertia:inventory}{tensión} $T$ en ambos extremos; los bloques comparten una magnitud de aceleración $a$.
Una segunda ley cada una, a lo largo de cada movimiento: $m_1 a = T$ y $m_2 a = m_2 g - T$, por lo que
$a = m_2\,g/(m_1 + m_2) = \qty{1.96}{m/s^2}$ y $T = m_1 a = \qty{7.8}{N}$ ---
más pequeño que $m_2 g = \qty{9.8}{N}$, ya que debe serlo para que $m_2$ acelere hacia abajo.
\end{example}

\begin{example}[La curva plana]\label{ex:g12:newtons-laws:curve}
Un automóvil de masa $m$ toma una curva plana de radio $R$ con rapidez constante $v$. en el
\omterm{def:g12:kinematics-2d:frenet}{Marco de frenet} (\cref{ch:g12:kinematics-2d}) la aceleración es puramente normal,
$a_N = v^2/R$, dirigido al centro. \omterm{def:g10:universal-gravitation:weight}{verticalmente}, $N = mg$; horizontalmente, el único
El \omterm{def:g10:inertia:force}{fuerza} centrípeto que se ofrece es el \omterm{def:g10:inertia:inventory}{fricción} estático de carretera con neumáticos:
$f = mv^2/R \leq \mu_s N = \mu_s mg$, entonces $v \leq \sqrt{\mu_s\, g R}$, cualquiera que sea el
masa. Para $R = \qty{90}{m}$, carretera seca ($\mu_s = 0.70$):
$v_{\max} \approx \qty{25}{m/s}$ (\qty{89}{km/h}); sobre el hielo el techo se derrumba ---
\cref{exo:g12:newtons-laws:13} peralte el camino; \cref{exo:g12:newtons-laws:10} columpios
la misma proyección en un alambre.
\end{example}

\section{Sistemas aislados: el impulso se conserva}

\begin{proposition}[Conservación del impulso]\label{prop:g12:newtons-laws:conservation}\index{conservación del impulso}
Para dos cuerpos cuyo \omterm{def:g10:inertia:force}{efectivo} externo desaparece o \omterm{def:g10:inertia:compensate}{compensar}, el total \omterm{def:g12:newtons-laws:momentum}{impulso}
$\vect p_1 + \vect p_2$ es constante, cualquiera que sea el \omterm{def:g10:inertia:force}{efectivo} que ejerzan entre sí.
\end{proposition}

\begin{proof}
Suma las dos segundas leyes: $\frac{d}{dt}(\vect p_1 + \vect p_2) = \vect F_{2 \to 1} +
\vect F_{1 \to 2} + \vect F_{\text{ext}}$: el par interno se cancela por la tercera
ley, y $\vect F_{\text{ext}} = \vect 0$ por hipótesis.
\end{proof}

\begin{example}[retroceder]\label{ex:g12:newtons-laws:recoil}
Un patinador \qty{60}{kg} en reposo sobre hielo liso lanza un balón medicinal \qty{4.0}{kg}
horizontalmente en \qty{6.0}{m/s}. Antes: $\vect p = \vect 0$; después:
$0 = 60\,V + 4.0 \times 6.0$, entonces $V = \qty{-0.40}{m/s}$ --- ella se desplaza hacia atrás.
Nada externo la empujó: la pelota lo hizo (tercera ley), y los libros se quedaron en cero.
\end{example}

\begin{remark}[Cómo un cohete empuja sobre la nada]\label{rem:g12:newtons-laws:rocket}
Un cohete es una máquina para lanzar masa. Cada segundo sus motores lanzan una gran cantidad de gas.
hacia atrás; el gas se lleva hacia atrás \omterm{def:g12:newtons-laws:momentum}{impulso}, por lo que el cohete gana lo mismo
cantidad hacia adelante --- un \omterm{def:g10:inertia:inventory}{empuje} constante, aproximadamente la masa expulsada por segundo multiplicada por
velocidad de expulsión. El cohete empuja sobre su propio escape, no sobre el aire: \omterm{def:g11:work-of-force:work}{obras} es mejor en
espacio vacío.
\end{remark}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:1}
Calcule el \omterm{def:g12:newtons-laws:momentum}{impulso} de (a) un velocista \qty{70}{kg} en \qty{10}{m/s}; (b) un
\qty{1300}{kg} coche en \qty{50}{km/h}; (c) una bala \qty{8.0}{g} en \qty{800}{m/s}.
Clasificarlos. ¿Por qué dura el rango más rápido?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:2}
Un pasajero \qty{70}{kg} está parado sobre una báscula en un ascensor. ¿Qué se lee cuando
la cabina (a) acelera hacia arriba en \qty{1.2}{m/s^2}; (b) sube a una velocidad constante
\qty{2.0}{m/s}; (c) acelera hacia abajo en \qty{1.2}{m/s^2}? ¿Qué ley resuelve (b)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:3}
Se suelta un bloque sobre una pendiente sin fricción del ángulo $30^\circ$. Calcular su
aceleración, luego su velocidad después de deslizarse \qty{3.0}{m}. ¿Adónde se fue la misa?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:4}
Una pelota de tenis \qty{58}{g} llega a \qty{15}{m/s} y es devuelta por el mismo
línea en \qty{25}{m/s}; el contacto dura \qty{5.0}{ms}. Calcular el cambio de \omterm{def:g12:newtons-laws:momentum}{impulso}
y el promedio \omterm{def:g10:inertia:force}{fuerza} de las cadenas; compare ese \omterm{def:g10:inertia:force}{fuerza} con el \omterm{def:g10:universal-gravitation:weight}{peso} de la pelota.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:5}
Un libro descansa sobre una mesa. Nombra los dos \omterm{def:g10:inertia:force}{efectivo} que actúan sobre el libro y, para cada uno, su
compañero de acción-reacción (¿qué cuerpo? ¿en qué dirección?). ¿Por qué los \omterm{def:g10:universal-gravitation:weight}{peso} del libro
¿Y los que empujan la mesa no son socios unos de otros?

\section{Problema: el teleférico}

\begin{problem}[{Problema fin de semana --- certificación del teleférico: la tensión la pendiente
demandas en reposo, el recargo de la puesta en marcha, el relevo sobre la torre, el precio
de una parada de emergencia --- y el único número que el cartel en la cabina puede prometer}]
\label{pb:g12:newtons-laws:1}
Un teleférico de montaña --- cabina más vagón, masa vacía $M_0 = \qty{2.0e3}{kg}$ ---
rueda sobre un cable de vía inclinado en $\alpha = 30^\circ$, tirado por un cable de arrastre
paralelo a la pendiente con \omterm{def:g10:inertia:inventory}{tensión} $\vect T$. El balanceo \omterm{def:g10:inertia:inventory}{fricción} es insignificante; el
El piso de la cabina permanece horizontal en todo momento. La cabina tiene capacidad para $25$ pasajeros de tamaño medio.
masa \qty{80}{kg}; usted es el ingeniero certificador. A menos que se indique lo contrario, el
cabina está llena: $M = \qty{4.0e3}{kg}$.

\medskip
\noindent\textbf{Parte I --- At rest on the slope.}
\begin{enumerate}
  \item Elija el sistema; diga por qué el marco de montaña funcionará como marco inercial;
        Inventario de los tres \omterm{def:g10:inertia:force}{efectivo}, con instrucciones.
  \item ¿Qué ley rige la espera? Proyectarlo a lo largo y perpendicular a la pendiente.
        para expresar $T$ y el cable de vía \omterm{def:g12:newtons-laws:friction}{fuerza normal} $N$.
  \item Calcula $T$ y $N$.
  \item Vuelva a calcular $T$ para la cabina vacía. ¿Por qué $T$ es proporcional a la masa total?
  \item El cable de arrastre se rompe, frena: muestra que la aceleración desbocada es
        $g \sin\alpha$, cargado o vacío, y calcularlo.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Start-up.} El cabrestante acelera toda la cabina a
$a = \qty{0.60}{m/s^2}$ subiendo la pendiente hasta cruzar en $v = \qty{6.0}{m/s}$.
\begin{enumerate}[resume]
  \item Expresar y calcular el \omterm{def:g10:inertia:inventory}{tensión} $T'$ durante esta fase.
  \item ¿En qué fracción el arranque eleva el \omterm{def:g10:inertia:inventory}{tensión} por encima de su valor en reposo?
  \item ¿Cuánto dura la fase y a qué distancia?
  \item Un pasajero \qty{80}{kg} se encuentra en el suelo horizontal. proyectar el segundo
        ley \omterm{def:g10:universal-gravitation:weight}{verticalmente} para expresar y computar su \omterm{def:g12:newtons-laws:apparent}{peso aparente}; comparar con $mg$.
  \item ¿Cuál \omterm{def:g10:inertia:force}{fuerza} la acelera horizontalmente y en qué medida? ¿Por qué los pasajeros se inclinan?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Over the tower.} A una constante \qty{6.0}{m/s}, el
El cable de seguimiento se dobla sobre una torre de soporte a lo largo de un arco circular vertical de radio.
$R = \qty{40}{m}$; en la cresta la velocidad es horizontal.
\begin{enumerate}[resume]
  \item En el \omterm{def:g12:kinematics-2d:frenet}{Marco de frenet} en la cresta (\cref{ch:g12:kinematics-2d}), dé ambos
        componentes de la aceleración, con sus valores.
  \item Proyecta la segunda ley a lo largo de la normal para expresar y calcular el \omterm{def:g12:newtons-laws:friction}{normales
        fuerza} $N'$ del cable sobre el carro.
  \item Misma pregunta para el \omterm{def:g12:newtons-laws:apparent}{peso aparente} del pasajero en la cresta: ¿por qué?
        ¿Es ella más liviana?
  \item ¿A qué velocidad desaparecería $N'$? Comparar con la velocidad de crucero y comentar.
  \item Qué \omterm{def:g10:inertia:force}{fuerza} ejerce el carro sobre el cable que pasa por encima de la torre --- qué ley
        dice eso, ¿con qué magnitud y dirección?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Braking, a jump, and the placard.}
\begin{enumerate}[resume]
  \item En tramo de llegada horizontal, parada de emergencia ocupa toda la cabina
        de \qty{6.0}{m/s} a descansar en \qty{1.5}{s}: leer $\sum \vect F = d\vect p/dt$
        como promedio sobre la parada para calcular el frenado \omterm{def:g10:inertia:force}{fuerza}.
  \item ¿Qué \omterm{def:g10:inertia:force}{fuerza} horizontal debe detener a un pasajero \qty{80}{kg}, y puede
        zapato-piso estático \omterm{def:g10:inertia:inventory}{fricción} ($\mu_s = 0.40$) ¿lo suministra? Concluir.
  \item En el andén, frenos quitados, la cabina vacía cuelga libre y quieta; el ultimo
        pasajero (\qty{80}{kg}) salta horizontalmente en \qty{2.5}{m/s}
        \emph{relativo a la cabina}. Calcule la velocidad de retroceso de la cabina y la
        la velocidad del pasajero sobre el suelo.
  \item En una frase: ¿qué dispositivo de propulsión se basa en este principio de retroceso?
        ¿Y qué soluciona su \omterm{def:g10:inertia:inventory}{empuje}?
  \item \emph{Proceso de dar un título.} El cable de transporte se rompe en \qty{100}{kN}; el límite de las reglas
        el \omterm{def:g10:inertia:inventory}{tensión} de trabajo a una quinta parte de eso. Muestre que el peor de los casos encontrado anteriormente es
        la puesta en marcha, deducir la masa total máxima, e imprimir el cartel: ¿cuántos
        \qty{80}{kg} ¿Pueden viajar los pasajeros? Verifique el factor de seguridad.
\end{enumerate}
\end{problem}
