\chapter{Caída libre y movimiento de proyectil}\label{ch:g12:projectile-motion}

Cada chorro de manguera de jardín dibuja la misma curva en el aire. También lo hace un
baloncesto en arco hacia el aro, un saltador de longitud, una bala de cañón: cualquier cuerpo
abandonado a su \omterm{def:g10:universal-gravitation:weight}{peso}. Este capítulo gana esa curva --- una parábola ---
de \omterm{thm:g12:newtons-laws:second}{Segunda ley de Newton} en dos líneas de cálculo, lee ápice, vuelo
tiempo y \omterm{def:g12:projectile-motion:range}{rango} fuera de su ecuación, luego deja que el aire vuelva a entrar para ver qué
se estropea.

\section{caída libre}

\begin{definition}[caida libre]\label{def:g12:projectile-motion:freefall}
Un cuerpo está en \emph{caída libre}\index{caída libre} cuando el único \omterm{def:g10:inertia:force}{fuerza} que actúa sobre
es su \omterm{def:g10:universal-gravitation:weight}{peso} $\vect P = m\vect g$: sin soporte, sin cuerda, resistencia al aire
insignificante. Una piedra que se ha caído o una pelota que se ha salido de la mano es, para bien,
precisión, en caída libre --- ``caída'' incluye ascenso.
\end{definition}

\begin{theorem}[Ley de caída libre de Galileo]\label{thm:g12:projectile-motion:galileo}
\index{La ley de caída libre de Galileo}
En \omterm{def:g12:projectile-motion:freefall}{caída libre}, todos los cuerpos comparten la misma aceleración, cualquiera que sea su masa:
\[
\vect a = \vect g, \qquad g = \qty{9.81}{m/s^2},
\]
dirigido hacia abajo: la fuerza \omterm{def:g11:electric-gravitational-fields:field}{campo} \qty{9.81}{N/kg} de
\cref{ch:g10:universal-gravitation} con su otro conjunto, $\unit{N/kg} = \unit{m/s^2}$.
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{Segunda ley de Newton} (\cref{ch:g12:newtons-laws}) con el \omterm{def:g10:universal-gravitation:weight}{peso} como único
\omterm{def:g10:inertia:force}{fuerza}: $m\vect a = m\vect g$, y la masa se cancela. Martillo y pluma
difieren sólo a través del aire.
\end{proof}

\begin{proposition}[Caída vertical desde el reposo]\label{prop:g12:projectile-motion:vertical}
Caído desde el reposo, con la profundidad de caída $h$ contada hacia abajo desde el
punto de liberación, un cuerpo en \omterm{def:g12:projectile-motion:freefall}{caída libre} obedece
\[
a = g, \qquad v = g t, \qquad h = \tfrac12\, g t^2 .
\]
\end{proposition}

\begin{proof}
$v(t)$ es la antiderivada de la constante $a = g$ que desaparece en $t = 0$
(\cref{ch:g12:kinematics-2d}); $h(t)$ el de $v = gt$ desapareciendo en $t = 0$.
\end{proof}

\begin{example}[Órdenes de magnitud]\label{ex:g12:projectile-motion:orders}
Después de $1$, $2$, $3$ segundos de \omterm{def:g12:projectile-motion:freefall}{caída libre}: $v = 9.8$, $19.6$, \qty{29.4}{m/s}
($35$, $71$, \qty{106}{km/h}) y $h = 4.9$, $19.6$, \qty{44.1}{m}. De un
\qty{30}{m} azotea: $t = \sqrt{2h/g} \approx \qty{2.5}{s}$, llegando a
\qty{24}{m/s} $\approx \qty{87}{km/h}$. Los tiempos crecen como $\sqrt h$ --- y
Pasados unos segundos el aire se niega a permanecer insignificante (último tramo).
\end{example}

\begin{example}[lanzamiento vertical]\label{ex:g12:projectile-motion:throw}
Lanzado hacia arriba en $v_0$, eje $Oy$ hacia arriba desde la mano: los mismos dos
las antiderivadas dan $v = v_0 - gt$ y $y = v_0 t - \tfrac12 gt^2$. el
ápice, donde $v = 0$, llega a $t = v_0/g$, altura $h = v_0^2/2g$ --- el
\omterm{def:g10:energy-conservation:energy}{energía} responde $v_0 = \sqrt{2gh}$ de ejecución \cref{ch:g11:mechanical-energy}
hacia atrás; lanzado \emph{abajo}, $v_0$ invierte el signo. Para \qty{15}{m/s}: ápice
\qty{11.5}{m} en \qty{1.53}{s}, nuevamente en \qty{3.06}{s}, en \qty{15}{m/s}.
\end{example}

\section{El proyectil: una ley, dos antiderivadas}

\begin{definition}[Proyector]\label{def:g12:projectile-motion:projectile}
Un \emph{proyectil}\index{proyectil} es un cuerpo lanzado con una velocidad
$\vect v_0$ y luego se fue en \omterm{def:g12:projectile-motion:freefall}{caída libre}. Ejes en el punto de lanzamiento: $Ox$
horizontal en el plano vertical de $\vect v_0$, $Oy$ vertical hacia arriba; el
\emph{ángulo de lanzamiento}\index{ángulo de lanzamiento} $\alpha$ de $\vect v_0$ por encima del
horizontal fija sus coordenadas $(v_0\cos\alpha,\; v_0\sin\alpha)$.
\end{definition}

\begin{theorem}[Ecuaciones de movimiento]\label{thm:g12:projectile-motion:equations}
Un \omterm{def:g12:projectile-motion:projectile}{proyectil} lanzado desde el origen con velocidad $v_0$ y ángulo $\alpha$
tiene $\vect a = (0,\,-g)$ en cada instante, por lo tanto
\[
\vect v(t) = \bigl(v_0\cos\alpha,\; v_0\sin\alpha - g t\bigr),
\qquad
\begin{cases}
x(t) = v_0\cos\alpha\; t,\\[2pt]
y(t) = v_0\sin\alpha\; t - \tfrac12\, g t^2 .
\end{cases}
\]
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{Segunda ley de Newton} da $\vect a = \vect g = (0,-g)$. Antiderivadas
eje por eje: $\vect v(t)$ de $\vect a$ con valor inicial
$(v_0\cos\alpha,\, v_0\sin\alpha)$, luego posición desde $\vect v$, desde $(0,0)$.
\end{proof}

\begin{proposition}[Independencia de las dos mociones]\label{prop:g12:projectile-motion:independence}
El movimiento horizontal es uniforme --- $v_x$ nunca cambia --- y el
El movimiento vertical es exactamente el recorrido vertical del apartado anterior:
ninguna ecuación menciona la coordenada de la otra. En particular, una pelota
disparado horizontalmente y una pelota lanzada en el mismo instante desde el mismo
altura se mantienen nivelados entre sí --- aterrizan juntos.
\end{proposition}

\begin{proof}
En \cref{thm:g12:projectile-motion:equations}, $x(t)$ implica solo
$v_0\cos\alpha$, $y(t)$ sólo $v_0\sin\alpha$ y $g$; ambas bolas tienen
$v_y(0) = 0$, por lo que ambos obedecen a $h = \tfrac12 gt^2$.
\end{proof}

\begin{remark}[El mono que cae]\label{rem:g12:projectile-motion:monkey}
Un cuidador del zoológico apunta con un dardo \emph{directo a} a un mono, quien suelta su
rama en el tiro. Mal movimiento: dardo y mono cada gota $\tfrac12 gt^2$
debajo de sus posiciones sin gravedad --- la línea de puntería, la rama --- así que
se encuentran exactamente, sea cual sea la velocidad del dardo
(\cref{exo:g12:projectile-motion:15}).
\end{remark}

\section{La parábola, leída completa}

\begin{theorem}[Trayectoria]\label{thm:g12:projectile-motion:trajectory}
Eliminando $t$ entre $x(t)$ y $y(t)$: el \omterm{def:g10:relative-motion:trajectory}{trayectoria} de un \omterm{def:g12:projectile-motion:projectile}{proyectil}
($0 \leq \alpha < \ang{90}$) es la parábola que se abre hacia abajo
\[
y = x\tan\alpha \;-\; \frac{g}{2 v_0^2 \cos^2\alpha}\, x^2 .
\]
\end{theorem}

\begin{proof}
$t = x/(v_0\cos\alpha)$, sustituido en $y(t)$ de
\cref{thm:g12:projectile-motion:equations}.
\end{proof}

\begin{definition}[Rango]\label{def:g12:projectile-motion:range}
Para un \omterm{def:g12:projectile-motion:projectile}{proyectil} sobre terreno nivelado, el \emph{rango}\index{rango} $R$ es
la distancia horizontal recorrida cuando regresa a su altura de lanzamiento; el
\emph{tiempo de vuelo}\index{tiempo de vuelo} $t_f$ es la duración de ese viaje.
\end{definition}

\begin{proposition}[Ápice, tiempo de vuelo, alcance]\label{prop:g12:projectile-motion:range}
Sobre terreno llano, con $0 < \alpha < \ang{90}$:
\[
t_f = \frac{2 v_0 \sin\alpha}{g}, \qquad
h_{\text{apex}} = \frac{(v_0\sin\alpha)^2}{2g}
\ \ \text{reached at } t_f/2, \qquad
R = \frac{v_0^2 \sin(2\alpha)}{g}.
\]
El \omterm{def:g12:projectile-motion:range}{rango} es mayor en $\alpha = \ang{45}$, donde $R = v_0^2/g$;
Los ángulos complementarios $\alpha$ y $\ang{90} - \alpha$ dan el mismo \omterm{def:g12:projectile-motion:range}{rango}.
\end{proposition}

\begin{proof}
$y = 0$ da $t\,(v_0\sin\alpha - \tfrac12 gt) = 0$: lanzamiento y $t_f$.
El ápice, $v_y = 0$, se encuentra en $t = v_0\sin\alpha/g = t_f/2$; sustituir en
$y(t)$. Entonces $R = x(t_f) = 2v_0^2\sin\alpha\cos\alpha/g =
v_0^2\sin(2\alpha)/g$. El seno alcanza su punto máximo en$2\alpha = \ang{90}$, y
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180} - 2\alpha) =
\sin(2\alpha)$.
\end{proof}

\begin{example}[Saque de portería]\label{ex:g12:projectile-motion:kick}
Un portero golpea a $v_0 = \qty{25}{m/s}$, $\alpha = \ang{45}$:
$R = 25^2/9.81 \approx \qty{64}{m}$; autorizaciones reales aterrizan más cerca
\qty{55}{m} --- se cortó el aire.
\end{example}

\begin{method}[Contabilidad de proyectiles]\label{met:g12:projectile-motion:method}
\begin{enumerate}
  \item Ejes en el punto de lanzamiento, $Ox$ horizontal, $Oy$ arriba; coordenadas
        de $\vect v_0$.
  \item Marque \omterm{def:g12:projectile-motion:freefall}{caída libre} (solo \omterm{def:g10:universal-gravitation:weight}{peso}): luego $\vect a = (0,-g)$;
        antiderivadas dos veces, condiciones iniciales cada vez.
  \item Pregunta sobre un \emph{lugar}: eliminar $t$ (\omterm{def:g10:relative-motion:trajectory}{trayectoria}). sobre un
        \emph{instante}: mantener $t$.
  \item Diccionario: ápice $\Leftrightarrow v_y = 0$; altura de lanzamiento
        $\Leftrightarrow y = 0$; pared en $d \Leftrightarrow x = d$; piso
        $\ell$ debajo de $\Leftrightarrow y = -\ell$.
  \item Cordura: vuelo simétrico respecto al ápice; $R \leq v_0^2/g$; solo aire
        cada vez se acorta.
\end{enumerate}
\end{method}

\section{que cambia el aire}

\begin{definition}[terminal de velocidad]\label{def:g12:projectile-motion:terminal}
Air \omterm{def:g11:circuits-and-power:resistance}{resistencia} se opone a la velocidad y crece con la velocidad. un cuerpo cayendo
por lo tanto acelera sólo hasta que la resistencia equilibre \omterm{def:g10:universal-gravitation:weight}{peso}; desde allí
$\vect a = \vect 0$: cae a una constante \emph{terminales
velocidad}\index{velocidad terminal}.
\end{definition}

\begin{example}[Órdenes de magnitud terminales]\label{ex:g12:projectile-motion:airorders}
Un paracaidista a toda máquina: aproximadamente \qty{55}{m/s} (\qty{200}{km/h}). Una gota de lluvia:
aproximadamente \qty{9}{m/s} --- en el vacío, \qty{2000}{m} de caída lo entregaría en
$\sqrt{2gh} \approx \qty{2.0e2}{m/s}$. Una pluma: aproximadamente \qty{1}{m/s},
alcanza en centímetros --- el secreto de su lentitud: gasta su
todo el otoño \emph{at} \omterm{def:g12:projectile-motion:terminal}{velocidad terminal}; el martillo nunca se acerca al suyo.
\end{example}

\begin{remark}[balística real]\label{rem:g12:projectile-motion:drag}
El arrastre empuja contra la velocidad a lo largo del vuelo: el \omterm{def:g12:projectile-motion:range}{rango} se acorta,
el vértice desciende y la simetría se rompe: el descenso es más pronunciado que el
subida, tiros largos visiblemente por debajo de la parábola. El honesto, cuantitativo.
el tratamiento es una ecuación diferencial resuelta en el volumen del Año 1; esto
año, la parábola es el modelo adecuado para fenómenos densos y moderadamente rápidos.
\omterm{def:g12:projectile-motion:projectile}{proyectiles} --- una pelota de baloncesto en \qty{10}{m/s}, no una pelota de golf en
\qty{70}{m/s}.
\end{remark}

\section{Problema: el timbre del baloncesto}

\begin{problem}[{Problema del fin de semana --- el timbre del baloncesto: un programa gratuito
lanzar flecha disecada por flecha, las yemas de los dedos despejadas por centímetros, una
Un esfuerzo en toda la cancha comparado con lo que un brazo puede dar, un globo por encima del
tablero --- y un veredicto sobre la toma más teatral del deporte}]
\label{pb:g12:projectile-motion:1}
El aro de un aro de baloncesto es un anillo horizontal \qty{3.05}{m} encima del
piso; Nora lanza cada disparo en \qty{2.00}{m}, por lo que con hachas en el
punto de liberación ($Ox$ horizontal hacia el aro, $Oy$ arriba) el centro del borde
se encuentra \qty{1.05}{m} encima del origen. Se desprecia el aire excepto donde se indique lo contrario.

\medskip\noindent\textbf{Parte I --- The free throw.} Nora dispara desde
$d = \qty{4.20}{m}$ (horizontal) con \omterm{def:g12:projectile-motion:projectile}{ángulo de lanzamiento} $\alpha = \ang{50}$.
\begin{enumerate}
  \item ¿Qué acto \omterm{def:g10:inertia:force}{efectivo} una vez liberado? Indique $\vect a$, citando la ley utilizada.
  \item Derive las coordenadas de $\vect v(t)$.
  \item Derive $x(t)$ y $y(t)$.
  \item Elimina $t$ para obtener el \omterm{def:g10:relative-motion:trajectory}{trayectoria} $y(x)$.
  \item La pelota debe pasar el centro del aro, $y(d) = \qty{1.05}{m}$: resolver
        para $v_0$, en \unit{m/s} y \unit{km/h}.
\end{enumerate}

\medskip\noindent\textbf{Parte II --- Over the fingertips, into the ring.}
\begin{enumerate}[resume]
  \item ¿Cuánto tiempo tarda la pelota en llegar al aro?
  \item Encuentre el tiempo y la altura del ápice; comprueba que la pelota ya está
        descendiendo hasta el borde.
  \item Un defensor \qty{0.90}{m} frente a Nora estira las yemas de los dedos para
        \qty{2.60}{m}. ¿En cuánto los despeja la pelota?
  \item Calcule $v_x$ y $v_y$ en la llanta, la velocidad allí y el ángulo.
        de la velocidad por debajo de la horizontal.
  \item ¿Por qué debe llegar una bola \emph{descendiendo} --- y por qué empinada?
\end{enumerate}

\medskip\noindent\textbf{Parte III --- The full-court heave.} Zumbador sobre
a \omterm{def:g10:signals-and-waves:sound}{sonido}, Nora está a \qty{26.0}{m} del borde más alejado.
\begin{enumerate}[resume]
  \item El aumento de \qty{1.05}{m} es pequeño frente a \qty{26.0}{m}: justifica un
        modelo de terreno nivelado y derivar el \omterm{def:g12:projectile-motion:range}{tiempo de vuelo} de $y(t_f) = 0$.
  \item Deduce $R = v_0^2\sin(2\alpha)/g$ y el ángulo mayor \omterm{def:g12:projectile-motion:range}{rango}.
  \item Calcule el $v_0$ necesario para cubrir \qty{26.0}{m} en $\alpha = \ang{45}$.
  \item Un jugador fuerte puede lanzar una pelota de baloncesto aproximadamente a \qty{17.5}{m/s}.
        ¿Cuál es su máximo \omterm{def:g12:projectile-motion:range}{rango}? ¿Es la inyección humanamente factible?
  \item En \qty{17.5}{m/s}, encuentre los ángulos \emph{dos} para aterrizar la pelota.
        \qty{26.0}{m} de distancia, y sus horarios de vuelo: con \qty{2.0}{s} encendido
        el reloj, ¿cuál suena mejor que el timbre?
\end{enumerate}

\medskip\noindent\textbf{Parte IV --- The lob over the backboard.} Desde
detrás de la línea de fondo, globos de Nora: suelte \qty{6.0}{m} desde el centro del aro,
$\alpha = \ang{75}$; el avión del tablero cruza su tiro \qty{0.45}{m}
antes del centro del borde, el borde superior \qty{3.95}{m} por encima del suelo.
\begin{enumerate}[resume]
  \item Encuentra el $v_0$ que deja caer la pelota por el centro del aro.
  \item Calcula la altura de la bola en el plano del tablero.
        ($x = \qty{5.55}{m}$): ¿limpia el borde superior, en cuánto?
  \item Encuentre la altura del ápice y el tiempo de vuelo. ¿Por qué le encanta a la multitud?
  \item Cualitativamente, ¿qué cambia la resistencia del aire en estos tres?
        disparos, y ¿cuál amenaza más?
  \item \emph{El veredicto sobre el timbre}, dos frases: compara la velocidad
        exige el tiro en toda la cancha con lo que proporciona un brazo, déle al
        margen y regla: ¿posible o mito?
\end{enumerate}
\end{problem}
