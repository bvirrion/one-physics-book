\chapter{Desintegración radiactiva}\label{ch:g12:radioactive-decay}

En el sótano de un museo, un contador Geiger hace clic sobre una astilla de madera de un
el ataúd del faraón. Cada clic es ilegal: nada lo anuncia, nada
predice el siguiente. Sin embargo, el viernes el laboratorio imprimirá una fecha válida para un
siglo. Este capítulo resuelve la paradoja de cómo los acontecimientos son perfectamente aleatorios.
uno por uno suman, por billones, los relojes más estables que poseemos.

\section{Sin ley uno por uno, un reloj por el topo}

El año pasado (\cref{ch:g11:nucleus-radioactivity}) conocimos al \omterm{prop:g12:work-and-energy:equilibria}{inestable}
núcleos, equilibraron sus ecuaciones $\alpha$ y $\beta$, y vieron la mitad de cualquier
muestra grande desaparece según \omterm{def:g11:nucleus-radioactivity:halflife}{vida media} $T_{1/2}$. Lo que no pudimos decir fue
\emph{cuando entre las escaleras} --- ni por qué un proceso aleatorio mantiene un
horario en absoluto. Ambas preguntas tienen una respuesta.

\begin{definition}[Constante de caída]\label{def:g12:radioactive-decay:lambda}
Para cada \omterm{prop:g12:work-and-energy:equilibria}{inestable} \omterm{def:g11:nucleus-radioactivity:notation}{nucleido} existe una constante $\lambda$, su \emph{decadencia
constante}\index{constante de decaimiento} (\omterm{def:g10:orders-of-magnitude:unit}{unidad} \unit{s^{-1}}), de modo que durante cualquier
intervalo corto $\dd t$ cada \omterm{def:g11:fundamental-interactions:ladder}{núcleo} tiene probabilidad $p = \lambda\,\dd t$ de
en descomposición --- lo mismo para cada \omterm{def:g11:fundamental-interactions:ladder}{núcleo} de la especie, cualquiera que sea su edad,
y no afectado por la temperatura, \omterm{def:g10:pressure:pressure}{presión} o la química.
\end{definition}

\begin{remark}[Sin ley individualmente, exacto en multitudes]\label{rem:g12:radioactive-decay:crowds}
Un \omterm{def:g11:fundamental-interactions:ladder}{núcleo} que ha esperado diez mil años tiene exactamente la misma probabilidad de descomponerse
este segundo como uno hecho esta mañana: sin desgaste, sin aviso --- lo cual
uno va a continuación es realmente impredecible. Las multitudes son otro asunto. tirar
Monedas $100$: espere caras $50$, pero no se sorprenda con $43$ ---
fluctuaciones de orden $\sqrt n$, aquí $10\%$. Lanzar $10^{20}$: el
fluctuación relativa $1/\sqrt n$ es $10^{-10}$, el resultado seguramente
diez decimales. Un gramo de materia contiene unos núcleos $10^{22}$: anarquía
promedios en un mecanismo de relojería, una ley exacta para $N(t)$, el número de núcleos
todavía intacto en el momento $t$.
\end{remark}

\section{la ley de decadencia}

\begin{proposition}[La ecuación de la evolución.]\label{prop:g12:radioactive-decay:ode}
En una muestra grande de núcleos $N$, durante $\dd t$ el número cambia en
\[
\dd N = -\lambda N\,\dd t, \qquad\text{i.e.}\qquad \frac{\dd N}{\dd t} = -\lambda N :
\]
la población se reduce a un ritmo proporcional a ella misma.
\end{proposition}

\begin{proof}
Cada uno de los núcleos $N$ decae durante $\dd t$ con probabilidad
$\lambda\,\dd t$: decaimientos esperados $N\lambda\,\dd t$, y para grandes $N$ el
El recuento real se ajusta al esperado.
(\cref{rem:g12:radioactive-decay:crowds}); cada descomposición elimina un \omterm{def:g11:fundamental-interactions:ladder}{núcleo}.
\end{proof}

\begin{theorem}[Ley de desintegración radiactiva]\label{thm:g12:radioactive-decay:law}
Una muestra que contiene núcleos $N_0$ en $t = 0$ contiene, en el momento $t$,
\[
N(t) = N_0 \, e^{-\lambda t}.
\]
\end{theorem}

\begin{proof}
Verificar por sustitución:
$\dd N/\dd t = N_0(-\lambda)\,e^{-\lambda t} = -\lambda N(t)$, y
$N(0) = N_0 e^0 = N_0$: ecuación y empezar ambas bien. (que no
\emph{otro} se demuestra el ajuste de funciones, con integrales, en la universidad
volumen.)
\end{proof}

\begin{definition}[decaimiento exponencial]\label{def:g12:radioactive-decay:exponential}
Una cantidad siguiente a $N_0\,e^{-\lambda t}$ sufre \emph{exponencial
decadencia}\index{decaimiento exponencial}: en tiempos iguales, igual a \emph{fracciones}
están perdidos. Es exactamente la descarga \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} de
\cref{ch:g12:rc-rl-circuits}, con $\lambda$ jugando $1/(RC)$: uno
La ecuación diferencial, aprendida una vez, recorre circuitos y núcleos por igual.
\end{definition}

\begin{proposition}[vida media]\label{prop:g12:radioactive-decay:halflife}
El \omterm{def:g11:nucleus-radioactivity:halflife}{vida media} (\cref{def:g11:nucleus-radioactivity:halflife}) de un \omterm{def:g11:nucleus-radioactivity:notation}{nucleido}
está establecido por su \omterm{def:g12:radioactive-decay:lambda}{constante de decaimiento}: $T_{1/2} = \ln 2/\lambda$, y después de $n$
vidas medias $N = N_0/2^{\,n}$, $n$ completo o no.
\end{proposition}

\begin{proof}
$N(T_{1/2}) = N_0/2$ requiere $e^{-\lambda T_{1/2}} = \tfrac12$, es decir\
$\lambda T_{1/2} = \ln 2$; y $N(n\,T_{1/2}) = N_0\,e^{-n\ln 2} =
N_0/2^{\,n}$para cualquier$n$: el exponencial enhebra la escalera y llena
entre las escaleras.
\end{proof}

\section{Actividad: contar los clics}

\begin{proposition}[actividad]\label{prop:g12:radioactive-decay:activity}
El \omterm{def:g11:nucleus-radioactivity:activity}{actividad} de una muestra --- sus desintegraciones por segundo, en \omterm{def:g11:nucleus-radioactivity:activity}{becquereles}
(\cref{def:g11:nucleus-radioactivity:activity}) --- es
\[
\mathcal A = \lambda N, \qquad\text{so}\qquad \mathcal A(t) = \mathcal A_0 \, e^{-\lambda t}:
\]
proporcional a la población superviviente, e igualmente exponencial.
\end{proposition}

\begin{proof}
Núcleos $N$, cada uno con probabilidad $\lambda$ por segundo: $\lambda N$
desintegraciones por segundo; multiplicando \cref{thm:g12:radioactive-decay:law} por
$\lambda$ da $\mathcal A(t) = \mathcal A_0 e^{-\lambda t}$.
\end{proof}

\begin{example}[Enorme $N$, diminuto $\lambda$]\label{ex:g12:radioactive-decay:body}
Un plátano marca aproximadamente \qty{15}{Bq}, un cuerpo humano aproximadamente
\qty{8000}{Bq}, un \omterm{def:g10:orders-of-magnitude:unit}{metro} cúbico de granito en algún \qty{e6}{Bq}
(\cref{ch:g11:nucleus-radioactivity}). Detrás de los modestos clics se sientan
poblaciones astronómicas: el potasio-40 del cuerpo contribuye
$\mathcal A \approx \qty{4.4e3}{Bq}$ con $\lambda =
\qty{1.8e-17}{s^{-1}}$:$N = \mathcal A/\lambda \approx \num{2.5e20}$
núcleos, de los cuales apenas unos miles se disparan cada segundo.
\end{example}

\begin{method}[Midiendo la vida media]\label{met:g12:radioactive-decay:semilog}
\begin{enumerate}
  \item El conteo decae con un contador Geiger en intervalos iguales sucesivos
        intervalos: esto muestra $\mathcal A$ en momentos sucesivos.
  \item Parcela $\ln \mathcal A$ contra $t$: desde
        $\ln \mathcal A = \ln \mathcal A_0 - \lambda t$, \omterm{def:g12:radioactive-decay:exponential}{exponencial
        decadencia} se muestra como \emph{linea recta} --- datos torcidos significan
        otra ley, o una mezcla de \omterm{def:g11:nucleus-radioactivity:notation}{nucleidos}.
  \item Leer $\lambda$ como menos la pendiente; $T_{1/2} = \ln 2/\lambda$.
\end{enumerate}
\end{method}

\section{Citas: tiempo de lectura fuera de una proporción}

La ley de desintegración aplicada al revés es un reloj: mide la fracción superviviente,
y el exponencial nombra el tiempo transcurrido.

\begin{method}[Datos por radiocarbono]\label{met:g12:radioactive-decay:carbon}
\begin{enumerate}
  \item \emph{Vivo:} el tejido vivo intercambia carbono con la atmósfera
        y lleva su proporción de carbono-14 --- un \omterm{def:g11:fundamental-interactions:ladder}{átomo} en $10^{12}$,
        dando $\mathcal A_0 = 13.6$ desintegraciones por minuto por gramo de carbono.
  \item \emph{La muerte pone en marcha el reloj:} se detiene la ingesta y la proporción
        decae con $T_{1/2} = \num{5730}$ años.
  \item Mida el \omterm{def:g11:nucleus-radioactivity:activity}{actividad} $\mathcal A$ de la muestra por gramo de carbono;
        $\mathcal A = \mathcal A_0 e^{-\lambda t}$ se invierte a
        $t = \frac{1}{\lambda}\ln(\mathcal A_0/\mathcal A)
        = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}
\end{method}

\begin{example}[Un hogar en una cueva]\label{ex:g12:radioactive-decay:hearth}
El carbón de un hogar enterrado produce $5.0$ desintegraciones por minuto por gramo:
$\mathcal A_0/\mathcal A = 13.6/5.0 = 2.72$, entonces
$t = \num{5730} \times \ln(2.72)/\ln 2 \approx \num{8.3e3}$ años. el
el fuego se apagó hace ocho mil años --- la madera misma es el archivo.
\end{example}

\begin{example}[Relojes para piedras]\label{ex:g12:radioactive-decay:rocks}
El carbono 14 permanece en silencio más allá de unos \num{50000} años; la geología se mantiene más lenta
relojes. \emph{Potasio-argón}: potasio-40 ($T_{1/2} = \num{1.25e9}$
años) se descompone en argón, un gas que \omterm{rem:g12:work-and-energy:escape}{escapa} derrite la lava pero que queda atrapado una vez
la roca cristaliza --- la solidificación pone el reloj en cero. \emph{Uranio-plomo}:
el uranio-238 encabeza una cadena de desintegraciones que termina en \omterm{prop:g12:work-and-energy:equilibria}{estable} plomo-206
(\cref{ch:g11:nucleus-radioactivity}), y la relación plomo-uranio en un
el cristal lo fecha. Ambos relojes coinciden en los minerales más antiguos
($\approx \num{4.4e9}$ años) y, a través de meteoritos, en el Sistema Solar
edad.
\end{example}

\begin{remark}[Cadenas y el gas en la bodega.]\label{rem:g12:radioactive-decay:radon}
Entre el uranio y el plomo la cadena pasa por el radio, luego por el radón-222,
un \emph{gas} radiactivo ($T_{1/2} = 3.8$ días). Nacido en granito y
suelo, se filtra en los sótanos y se acumula; es emisor $\alpha$
las hijas se descomponen en los pulmones --- el \omterm{rem:g11:nucleus-radioactivity:dose}{dosis} natural más grande de la mayoría de las personas, y
Lo más barato de reducir: abrir la ventana del sótano.
\end{remark}

$10$ vidas medias; (b) todos los núcleos observados de $8$ sobreviven a un \omterm{def:g11:nucleus-radioactivity:halflife}{vida media}.
(c) De los núcleos \num{1000}, ¿cuántos se esperan después de las vidas medias de $3$? ---
¿Y la ley puede decir \emph{cual}? Concluir sobre lo que predice la ley.
\end{exercise}

\section{Problema: la edad de las cosas}

\begin{problem}[{Problema del fin de semana --- la era de las cosas: el laboratorio de un museo
calibra el reloj de carbono, fecha una estatua y un papiro, aprende por qué no
el reloj lee más allá de su esfera y empuja una roca volcánica hacia atrás mil millones de años}]
\label{pb:g12:radioactive-decay:1}
Una semana en el laboratorio de datación de un museo. Datos: muestra de tejido vivo
$\mathcal A_0 = 13.6$ el carbono-14 se desintegra por minuto por gramo de carbono;
$T_{1/2} = \num{5730}$ años (carbono-14), \num{1.25e9} años
(potasio-40); $1$ año $= \qty{3.156e7}{s}$;
$N_A = \qty{6.02e23}{mol^{-1}}$; masa molar de carbono \qty{12}{g/mol};
los contadores resuelven aproximadamente $0.2$ desintegraciones por minuto por gramo.

\medskip
\noindent\textbf{Parte I --- Monday: calibrating the clock.}
\begin{enumerate}
  \item ¿Por qué la proporción de carbono 14 es constante en el tejido vivo y
        ¿Qué cambia al morir?
  \item Calcule $\lambda$ para el carbono 14 en \unit{s^{-1}}.
  \item Verificar por sustitución que $N_0 e^{-\lambda t}$ resuelve
        $\dd N/\dd t = -\lambda N$.
  \item A partir de $\mathcal A_0$, calcule el número de carbono-14 \omterm{def:g11:fundamental-interactions:ladder}{átomos} por
        gramo de carbono en tejido vivo.
  \item ¿Cuántos \omterm{def:g11:fundamental-interactions:ladder}{átomos} de carbono contiene un gramo? Deducir el carbono-14
        proporción --- comparar con ``uno en $10^{12}$''.
  \item Demuestre que un \omterm{def:g11:nucleus-radioactivity:activity}{actividad} $\mathcal A$ medido data una muestra como
        $t = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Tuesday: the statue and the papyrus.}
\begin{enumerate}[resume]
  \item Una estatua de madera proporciona $9.1$ desintegraciones por minuto por gramo. Fecha el
        madera.
  \item ¿Qué evento, exactamente, marca esa fecha --- la talla o
        ¿algo más? ¿Qué precaución sigue?
  \item Un papiro egipcio da $10.4$ desintegraciones por minuto por gramo. Fecha
        eso. ¿Es plausible un escriba de hace veintidós siglos?
  \item El \omterm{def:g11:nucleus-radioactivity:activity}{actividad} se mide en $\pm 0.2$ decaimientos por minuto. Estimar
        la incertidumbre de la edad del papiro, utilizando
        $\Delta t \approx \Delta\mathcal A/(\lambda \mathcal A)$.
  \item El pergamino "antiguo" de un comerciante da $13.4$ desintegraciones por minuto
        por gramo. ¿Veredicto?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Thursday: the limits of the clock.}
\begin{enumerate}[resume]
  \item ¿Después de cuántas vidas medias el \omterm{def:g11:nucleus-radioactivity:activity}{actividad} de una muestra cae por debajo del
        ¿Las desintegraciones $0.2$ de los contadores por minuto por gramo? ¿Qué edad es esa?
  \item Los mejores laboratorios alcanzan unas diez vidas medias. cual es el
        horizonte práctico de la datación por radiocarbono, en años?
  \item Un hueso de dinosaurio tiene aproximadamente \num{6.6e7} años: ¿cuántos
        ¿vidas medias del carbono 14? Usando la pregunta 4, ¿después de cuántas vidas medias?
        ¿Ni siquiera queda \emph{uno} \omterm{def:g11:fundamental-interactions:ladder}{átomo} del carbono 14 del gramo?
        Concluir sobre los "dinosaurios que datan mediante carbono".
  \item ¿Qué tipo de \omterm{def:g11:nucleus-radioactivity:notation}{nucleido} podría datar la roca del dinosaurio?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Friday: the volcanic rock.}
\begin{enumerate}[resume]
  \item El argón es un gas, el potasio un fiel residente del sólido: explica
        por qué la solidificación de la lava pone a cero el reloj de potasio-argón.
  \item Suponiendo que cada potasio-40 descompuesto \omterm{def:g11:fundamental-interactions:ladder}{núcleo} deja uno atrapado
        argón \omterm{def:g11:fundamental-interactions:ladder}{átomo}, demuestre que
        $N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.
  \item La roca bajo los fósiles del museo muestra
        $N_{\mathrm{Ar}}/N_{\mathrm K} = 0.60$. Fecha la roca.
  \item ¿Podría el carbono-14 haber datado esta roca, o el potasio-argón el
        estatua? Proporcione la ventana útil de cada reloj en vidas medias.
  \item Informe del viernes, una frase: las tres edades encontradas esta semana, y
        la única ley que los lee a todos.
\end{enumerate}
\end{problem}
