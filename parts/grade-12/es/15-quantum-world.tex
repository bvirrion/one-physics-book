\section{La luz llega en granos: el fotón}

\begin{definition}[El fotón]\label{def:g12:quantum-world:photon}
Luz de \omterm{def:g12:oscillators-and-time:oscillator}{frecuencia} $\nu$ intercambia \omterm{def:g10:energy-conservation:energy}{energía} con materia sólo en cereales integrales
llamado \emph{fotones}\index{fotón}, cada uno de \omterm{def:g10:energy-conservation:energy}{energía}
\[
E = h\nu = \frac{hc}{\lambda}, \qquad h = \qty{6.63e-34}{J.s},
\]
donde $h$ es \emph{constante de planck}\index{constante de planck} (Planck,
1900; Einstein, 1905) y $\lambda$ el \omterm{def:g12:light-as-wave:lightwave}{longitud de onda en el vacío}. una viga
la intensidad mide el \emph{número} de fotones por segundo; cada grano
\omterm{def:g10:energy-conservation:energy}{energía} lo soluciona solo \omterm{def:g12:oscillators-and-time:oscillator}{frecuencia}.
\end{definition}

\begin{notation}[El electronvoltio]\label{not:g12:quantum-world:ev}
La física atómica utiliza el \emph{electronvoltio}\index{electronvoltio},
$\qty{1}{eV} = \qty{1.60e-19}{J}$: el \omterm{def:g10:energy-conservation:energy}{energía} gana el \omterm{def:g11:fundamental-interactions:elementary}{carga elemental}
cruce \qty{1}{V} (\cref{ch:g11:electric-gravitational-fields}). Práctico:
$E \,(\unit{eV}) \approx 1240/\lambda\,(\unit{nm})$.
\end{notation}

\section{El efecto fotoeléctrico}

Alumbra una placa de metal limpia en el vacío y los electrones salen --- el
\emph{efecto fotoeléctrico}\index{efecto fotoeléctrico}. Sus reglas rompieron la física clásica.

\begin{proposition}[Lo que muestra el experimento]\label{prop:g12:quantum-world:facts}
Para cada metal existe un \emph{frecuencia umbral}\index{frecuencia umbral} $\nu_0$:
\begin{enumerate}
  \item por debajo de $\nu_0$, no sale ningún electrón, por muy intensa que sea la luz;
  \item por encima de $\nu_0$, la emisión comienza sin \omterm{def:g12:mechanical-waves:delay}{demora}, aunque sea débil;
  La intensidad \item eleva el \emph{número} de electrones por segundo, no
        su máximo \omterm{def:g11:mechanical-energy:kinetic}{energía cinética}; \omterm{def:g12:oscillators-and-time:oscillator}{frecuencia} eleva $E_{k,\max}$.
\end{enumerate}
Nada de esto encaja en un \omterm{def:g10:signals-and-waves:wave}{ola} continuo, que cualquier color debería permitir cargar ---
durante meses con luz tenue (\cref{exo:g12:quantum-world:14}) --- y cuyo
La intensidad debería establecer el \omterm{def:g10:energy-conservation:energy}{energía} de los electrones: las tres predicciones fallan.
\end{proposition}
\admitted

\begin{theorem}[Relación fotoeléctrica de Einstein]\label{thm:g12:quantum-world:einstein}
\index{relación fotoeléctrica}Un electrón necesita al menos el \emph{trabajo
función}\index{función de trabajo} $W_0$ del metal para escapar de su superficie.
Un \omterm{def:g12:quantum-world:photon}{fotón} es absorbido completo por un electrón, por lo que
\[
h\nu = W_0 + E_{k,\max}, \qquad \text{hence } \nu_0 = \frac{W_0}{h}.
\]
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energía} contabilidad: $h\nu$ en, $W_0$ gastado en escape, el resto cinético ---
como máximo $h\nu - W_0$, menos para electrones que comienzan más profundamente. Por debajo de $\nu_0$
un grano no puede pagar el peaje de salida, sea cual sea su número.
\end{proof}

\begin{definition}[Potencial de parada]\label{def:g12:quantum-world:stopping}
Para medir $E_{k,\max}$, un \omterm{def:g11:circuits-and-power:voltage}{Voltaje} inverso frena los electrones
(\cref{ch:g11:electric-gravitational-fields}); el \emph{parando
potencial}\index{potencial de parada} $U_s$ es el \omterm{def:g11:circuits-and-power:voltage}{Voltaje} más pequeño cancelando
la corriente: los electrones más rápidos simplemente no logran cruzarse, por lo que $E_{k,\max} = e\,U_s$.
\end{definition}

\begin{method}[Midiendo la constante de Planck]\label{met:g12:quantum-world:measure-h}
\begin{enumerate}
  \item Iluminar una fotocélula a través de \omterm{def:g11:color-light-sources:subtractive}{filtros} de \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} conocido;
        calcular cada $\nu = c/\lambda$; registrar cada \omterm{def:g12:quantum-world:stopping}{potencial de parada}.
  \item Parcela $E_{k,\max} = eU_s$ contra $\nu$: Einstein promete el
        línea recta $h\nu - W_0$.
  \item Leer la pendiente: \emph{is} $h$; las intersecciones dan $\nu_0$ y $-W_0$.
\end{enumerate}
\end{method}

\section{Niveles de energía: por qué los átomos emiten líneas}

\begin{definition}[Niveles de energía]\label{def:g12:quantum-world:levels}
El \omterm{def:g10:energy-conservation:energy}{energía} de un electrón unido en un \omterm{def:g11:fundamental-interactions:ladder}{átomo} es
\emph{cuantificado}\index{cuantificación}: restringido a una escalera discreta de
\emph{niveles de energía}\index{nivel de energía} $E_1 < E_2 < \dots < 0$, contado
negativo del electrón libre. El peldaño más bajo es el \emph{suelo
estado}\index{estado fundamental}, los demás \emph{emocionado
estados}\index{estado excitado}; llegar a $0$ es
\emph{ionización}\index{ionización}.
\end{definition}

\begin{proposition}[La escalera del hidrogeno]\label{prop:g12:quantum-world:hydrogen}
Los \omterm{def:g12:quantum-world:levels}{niveles de energía} del hidrógeno \omterm{def:g11:fundamental-interactions:ladder}{átomo} son
\[
E_n = -\frac{\qty{13.6}{eV}}{n^2}, \qquad n = 1, 2, 3, \dots
\]
entonces $E_1 = \qty{-13.6}{eV}$, $E_2 = \qty{-3.40}{eV}$, $E_3 =
\qty{-1.51}{eV}$,$E_4 = \qty{-0.85}{eV}$, acercándose a$0$.
\end{proposition}
\admitted

\begin{remark}[De donde viene la escalera]\label{rem:g12:quantum-world:ladder-origin}
Por qué estos números son mecánica cuántica, guardados para los volúmenes universitarios ---
como lo es la fórmula de de Broglie: el \omterm{def:g12:quantum-world:debroglie}{onda de materia} del electrón
(\cref{def:g12:quantum-world:debroglie} a continuación) debe cerrarse sobre sí mismo alrededor
el \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, y solo ciertos \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} encajan --- como una cuerda de guitarra, un
\omterm{def:g11:fundamental-interactions:ladder}{átomo} tiene un conjunto discreto de notas.
\end{remark}

\begin{theorem}[Foton de una transicion]\label{thm:g12:quantum-world:transition}
Un \omterm{def:g11:fundamental-interactions:ladder}{átomo} cambia de nivel solo emitiendo o absorbiendo un \omterm{def:g12:quantum-world:photon}{fotón} que lleva
exactamente la diferencia: al pasar del nivel $p$ a un nivel inferior $n$ se emite
$h\nu = E_p - E_n$; subir de $n$ a $p$ absorbe ese mismo \omterm{def:g10:energy-conservation:energy}{energía}.
\end{theorem}

\begin{proof}
Conservación \omterm{def:g10:energy-conservation:energy}{Energía}, un grano a la vez: el \omterm{def:g12:quantum-world:photon}{fotón} es la única moneda
light acepta y los libros del \omterm{def:g11:fundamental-interactions:ladder}{átomo} deben equilibrarse.
\end{proof}

\begin{remark}[Espectros de líneas explicados]\label{rem:g12:quantum-world:lines}
Aquí está la respuesta prometida por \cref{ch:g10:light-spectra}: emite un gas caliente
sólo las frecuencias $(E_p - E_n)/h$ de su propia escalera --- un código de barras de
líneas brillantes --- y un gas frío roba exactamente esas frecuencias de \omterm{def:g10:light-spectra:white}{blanco
luz}, imprimiendo líneas de absorción oscuras coincidentes. Cada elemento tiene el suyo
escalera, de ahí su propio código de barras: cómo leemos la luz de las estrellas.
\end{remark}

\begin{example}[Las lineas visibles del hidrogeno]\label{ex:g12:quantum-world:balmer}
Las transiciones que terminan en $n = 2$ caen en lo visible. Con $E \,(\unit{eV}) =
1240/\lambda\,(\unit{nm})$:$3 \to 2$: \qty{1.89}{eV}, \qty{656}{nm} (rojo);
$4 \to 2$: \qty{2.55}{eV}, \qty{486}{nm} (azul-verde); $5 \to 2$:
\qty{2.86}{eV}, \qty{434}{nm} (violeta). Las transiciones a $n = 1$ superan
\qty{10}{eV}: todos \omterm{def:g10:light-spectra:wavelength}{ultravioleta}.
\end{example}

\begin{remark}[Láseres, de un solo aliento]\label{rem:g12:quantum-world:laser}
Un \omterm{def:g12:quantum-world:photon}{fotón} de exactamente $E_p - E_n$ pasando un \emph{entusiasmado} \omterm{def:g11:fundamental-interactions:ladder}{átomo} puede activarlo
para emitir un \omterm{def:g12:quantum-world:photon}{fotón} idéntico --- el mismo \omterm{def:g10:energy-conservation:energy}{energía}, la misma dirección, en el paso. mantener
más \omterm{def:g11:fundamental-interactions:ladder}{átomos} arriba que abajo y un \omterm{def:g12:quantum-world:photon}{fotón} se convierte en una avalancha de clones:
\emph{emisión estimulada}\index{emisión estimulada} --- y un \omterm{def:g11:color-light-sources:lamps}{láser} es poco más.
\end{remark}

\begin{definition}[Longitud de onda de De Broglie]\label{def:g12:quantum-world:debroglie}
Cada partícula de \omterm{def:g12:newtons-laws:momentum}{impulso} $p = mv$ viaja como \emph{importa
ola}\index{onda de materia} de \emph{longitud de onda de Broglie}\index{de Broglie
longitud de onda}.
\[
\lambda = \frac{h}{p} = \frac{h}{mv}.
\]
\end{definition}
\admitted

\begin{example}[Por qué nunca difractas]\label{ex:g12:quantum-world:never}
Un electrón acelerado a través de \qty{100}{V} tiene $E_k = \qty{1.60e-17}{J}$,
$p = \sqrt{2m_e E_k} = \qty{5.4e-24}{kg.m/s}$, entonces $\lambda \approx
\qty{0.12}{nm}$ --- el espaciado de \omterm{def:g11:fundamental-interactions:ladder}{átomos} en un cristal, que por lo tanto actúa
como su doble rendija. Una pelota de tenis \qty{58}{g} en \qty{25}{m/s}: $\lambda =
6.63\times10^{-34}/1.45 \approx \qty{4.6e-34}{m}$, algunas veces$10^{19}$
más pequeño que un \omterm{def:g11:fundamental-interactions:ladder}{núcleo}: ninguna hendidura mostrará flecos. $h$ es así
tan pequeño que la granulosidad cuántica emerge solo en la escala de \omterm{def:g11:fundamental-interactions:ladder}{átomo}.
\end{example}

\begin{example}[El microscopio electrónico]\label{ex:g12:quantum-world:microscope}
\omterm{def:g12:light-as-wave:diffraction}{Difracción} prohíbe resolver detalles mucho más pequeños que el \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} utilizado
(\cref{ch:g12:light-as-wave}): la luz visible se detiene cerca de \qty{0.5}{\micro m}.
Los electrones en \qty{10}{kV} tienen $\lambda \approx \qty{12}{pm}$ --- decenas de
miles de veces más cortos: los microscopios electrónicos muestran virus, células
maquinaria, incluso una sola \omterm{def:g11:fundamental-interactions:ladder}{átomos}. El \omterm{def:g12:quantum-world:debroglie}{onda de materia} le dio a la física un nuevo ojo.
\end{example}
