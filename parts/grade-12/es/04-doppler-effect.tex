\chapter{El efecto Doppler}\label{ch:g12:doppler-effect}

Una ambulancia se acerca, suena la sirena y pasa rápidamente\dots y en ese instante suena la nota.
se desliza audiblemente hacia abajo, aunque el conductor no toca nada. La sirena es honesta; el \omterm{def:g10:signals-and-waves:wave}{ola}
no lo es: el movimiento lo aprieta hacia adelante y lo estira hacia atrás. Este capítulo mide la presión,
luego lleva la misma idea de las sirenas a las pistolas \omterm{met:g10:signals-and-waves:echo}{Radar}, a la sangre en una arteria y a un
Planeta traicionado por una oscilación de la luz de las estrellas.

\section{el efecto}

\begin{definition}[Efecto Doppler]\label{def:g12:doppler-effect:doppler}
El \emph{efecto doppler}\index{efecto doppler} es el cambio del \omterm{def:g10:signals-and-waves:frequency}{frecuencia} recibido.
de un \omterm{def:g10:signals-and-waves:wave}{ola} cuando la fuente y el receptor se mueven entre sí: el \omterm{def:g10:signals-and-waves:frequency}{frecuencia} recibido
$f'$ es más alto que el \omterm{def:g10:signals-and-waves:frequency}{frecuencia} emitido $f$ mientras se acercan, más bajo mientras se acercan.
retrocede e igual a $f$ solo mientras su distancia deja de cambiar momentáneamente.
\end{definition}

\begin{remark}[Tono, sin volumen]\label{rem:g12:doppler-effect:pitch}
Dos cosas cambian cuando pasa la ambulancia. El \emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{volumen}} se hincha y se desvanece porque
el \omterm{def:g10:signals-and-waves:wave}{ola} se propaga con la distancia (\cref{ch:g12:sound-acoustics}); los turnos \emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{paso}}
debido a la velocidad de acercamiento o recesión, y cae abruptamente al pasar.
\omterm{def:g12:sound-acoustics:pitch-loudness}{Volumen} dice ``hasta dónde''; \omterm{def:g12:sound-acoustics:pitch-loudness}{paso} dice "qué rápido".
\end{remark}

\section{Una fuente en movimiento: frentes de onda abarrotados}

La imagen clave: una vez que una cresta abandona la fuente, el medio la posee. Cada cresta se expande como
un círculo a la velocidad \omterm{def:g10:signals-and-waves:wave}{ola} $v$ (\cref{ch:g12:mechanical-waves}), centrado en el punto donde
era \emph{emitido} --- el médium no sabe ni le importa que la fuente haya seguido adelante.

\begin{proposition}[fuente en movimiento]\label{prop:g12:doppler-effect:moving-source}
Una fuente emite crestas con \omterm{def:g10:signals-and-waves:period}{período} $T$ (\omterm{def:g10:signals-and-waves:frequency}{frecuencia} $f = 1/T$) mientras se mueve en línea recta.
línea a velocidad $v_s < v$ a través de un medio en el que el \omterm{def:g10:signals-and-waves:wave}{ola} viaja a velocidad $v$. un
El observador en reposo en el medio recibe
\[
\lambda' = (v \mp v_s)\,T, \qquad f' = f\,\frac{v}{v \mp v_s},
\]
las señales superiores indican que una fuente se acerca y las inferiores indican que se aleja.
\end{proposition}

\begin{proof}
Entre dos crestas la primera avanza $vT$ mientras que la fuente avanza $v_s T$. Adelante, el
por tanto se emite la segunda cresta $v_s T$ \emph{íntimamente} a la primera, $\lambda' = (v - v_s)T$;
detrás se añade el mismo $v_s T$. Luego, las crestas pasan junto a un observador en reposo en el
velocidad media $v$, uno cada $\lambda'/v$ segundos, entonces $f' = v/\lambda' = f v/(v \mp v_s)$.
\end{proof}

\begin{example}[La ambulancia, en números.]\label{ex:g12:doppler-effect:siren}
Una sirena en $f = \qty{700}{Hz}$ se acerca a $v_s = \qty{25}{m/s}$; tomar
$v = \qty{340}{m/s}$ en el aire (\cref{ch:g12:sound-acoustics}). Acercándose:
$f' = 700 \times 340/315 \approx \qty{756}{Hz}$; retrocediendo:
$f' = 700 \times 340/365 \approx \qty{652}{Hz}$. El pase baja el \omterm{def:g12:sound-acoustics:pitch-loudness}{paso} por \qty{104}{Hz}
--- una relación $\approx 1.16$, de unos dos semitonos y medio, inconfundible para cualquier oído.
\end{example}

\begin{remark}[¿Qué movimiento no cambia?]\label{rem:g12:doppler-effect:speed-unchanged}
El \omterm{def:g10:signals-and-waves:wave}{ola} sigue viajando a $v$: el medio es el único que marca la velocidad
(\cref{ch:g12:mechanical-waves}). El movimiento de la fuente aprieta \emph{\omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda}}, nunca
velocidades. Y la compresión tiene un límite: a medida que $v_s \to v$ las crestas que tenemos delante se acumulan unas sobre otras.
y $f' \to \infty$ --- el atasco de \omterm{def:g12:mechanical-waves:wavefront}{frentes de onda} un avión supersónico arrastra como susto.
\end{remark}

\section{Un observador en movimiento y el atajo en cámara lenta}

\begin{proposition}[Observador en movimiento]\label{prop:g12:doppler-effect:moving-observer}
Un observador se mueve a velocidad $v_o$ directamente hacia (o alejándose de) una fuente en reposo en el
medio. El \omterm{def:g10:signals-and-waves:frequency}{frecuencia} recibido es
\[ f' = f\left(1 \pm \frac{v_o}{v}\right) \quad (+\ \text{approaching},\ -\ \text{receding}). \]
\end{proposition}

\begin{proof}
El patrón \omterm{def:g10:signals-and-waves:wave}{ola} en sí no se altera: las crestas del espaciado $\lambda = vT$ viajan en $v$. un
El observador que corre hacia ellos encuentra crestas a la velocidad relativa $v + v_o$, una cada
$\lambda/(v + v_o)$ segundos: $f' = (v + v_o)/\lambda = f(1 + v_o/v)$. Huyendo, $v - v_o$.
\end{proof}

\begin{definition}[velocidad radial]\label{def:g12:doppler-effect:radial}
El \emph{velocidad radial}\index{velocidad radial} $v_r$ de una fuente relativa a una
observador es el componente de su velocidad relativa a lo largo de la línea que los une --- el
ritmo al que la distancia se reduce o crece. Sólo este componente desplaza el \omterm{def:g10:signals-and-waves:frequency}{frecuencia};
movimiento \emph{al otro lado de} la línea de visión no cambia en este orden.
\end{definition}

\begin{proposition}[Cámara lenta: una fórmula para todos]\label{prop:g12:doppler-effect:slow}
Cuando la velocidad radial es pequeña, $v_r \ll v$, ya no importa quién se mueva:
\[
\frac{\Delta f}{f} = \frac{f' - f}{f} \approx +\frac{v_r}{v} \ \text{(approach)},
\qquad
\frac{\Delta f}{f} \approx -\frac{v_r}{v} \ \text{(recession)}.
\]
\end{proposition}

\begin{proof}
Las fórmulas del observador ya son exactamente $1 \pm v_o/v$. Para la fuente, escriba $x = v_s/v$:
la identidad $\frac{1}{1 - x} = 1 + x + \frac{x^2}{1 - x}$ muestra que $f'/f$ difiere de
$1 + x$ por un término de orden $x^2$ --- para $x = 0.1$, una corrección $1\%$ en un turno $10\%$.
Para realizar el primer pedido, las cuatro fórmulas se colapsan en $1 \pm v_r/v$.
\end{proof}

\begin{method}[leyendo un pase]\label{met:g12:doppler-effect:pass}
Una grabación de una fuente que pasa muestra dos mesetas: $f_1$ (acercamiento) y $f_2$
(recesión). entonces
\[
f = \frac{2 f_1 f_2}{f_1 + f_2}, \qquad
v_s = v\,\frac{f_1 - f_2}{f_1 + f_2}.
\]
De hecho $f_1 + f_2 = \frac{2fv^2}{v^2 - v_s^2}$ y $f_1 - f_2 = \frac{2fv\,v_s}{v^2 - v_s^2}$:
divida para $v_s$ y calcule $2f_1 f_2/(f_1 + f_2)$ para $f$. Dos lecturas de \omterm{def:g10:signals-and-waves:frequency}{frecuencia} arrojan
tanto el verdadero \omterm{def:g12:sound-acoustics:pitch-loudness}{paso} como la velocidad: sin cronómetro ni regla.
\end{method}

\section{Ecos que miden la velocidad}

Haga rebotar un \omterm{def:g10:signals-and-waves:wave}{ola} en un objetivo en movimiento y el \omterm{def:g12:doppler-effect:doppler}{efecto doppler} golpeará dos veces.

\begin{proposition}[La reflexion duplica el turno]\label{prop:g12:doppler-effect:double}
Se envía un \omterm{def:g10:signals-and-waves:wave}{ola} de \omterm{def:g10:signals-and-waves:frequency}{frecuencia} $f$ a un reflector que se acerca de frente a la velocidad $u \ll v$.
El \omterm{rem:g10:signals-and-waves:honest}{eco} regresa con
\[ f' = f\,\frac{v + u}{v - u}, \qquad \text{hence} \qquad \Delta f \approx \frac{2u}{v}\,f . \]
\end{proposition}

\begin{proof}
Dos turnos en serie. Como \emph{observador} en movimiento, el reflector recibe $f(1 + u/v)$;
reemitiendo lo que recibe, ahora es un \emph{fuente} en movimiento, por lo que el \omterm{rem:g10:signals-and-waves:honest}{eco} llega a
$f' = f(1 + u/v)/(1 - u/v) \approx f(1 + 2u/v)$ para $u \ll v$.
\end{proof}

\begin{example}[La pistola de radar]\label{ex:g12:doppler-effect:radar}
Un tráfico \omterm{met:g10:signals-and-waves:echo}{Radar} emite microondas en $f = \qty{24.15}{GHz}$ ($v = c = \qty{3.00e8}{m/s}$)
y superpone el \omterm{rem:g10:signals-and-waves:honest}{eco} al \omterm{def:g10:signals-and-waves:wave}{ola} saliente; el ritmo (\cref{ch:g12:sound-acoustics})
directamente \omterm{def:g10:signals-and-waves:sound}{sonidos} fuera $\Delta f = 2uf/c \approx \qty{161}{Hz}$ por \unit{m/s} de la velocidad del automóvil
--- un audio \omterm{def:g10:signals-and-waves:frequency}{frecuencia} trivialmente medible extraído de un portador \qty{24}{GHz}. el
El factor $2$ no es opcional: olvidarlo favorece a la mitad a cada conductor.
\end{example}

\begin{example}[Ecografía Doppler]\label{ex:g12:doppler-effect:ultrasound}
Una sonda médica envía $f = \qty{5.0}{MHz}$ \omterm{def:g10:signals-and-waves:ultrasound}{ultrasonido} a una arteria
($v \approx \qty{1540}{m/s}$ en tejido); los glóbulos rojos lo reflejan. sangre en
$u = \qty{0.50}{m/s}$ devuelve $\Delta f = 2 \times 0.50 \times \num{5.0e6}/1540 \approx
\qty{3.2}{kHz}$ --- un latido \emph{audible}: el cardiólogo escucha literalmente la sangre
acelera con cada latido, luego mide $u$ desde $\Delta f$.
\end{example}

\section{Doppler a la luz de las estrellas}

Light es un \omterm{def:g10:signals-and-waves:wave}{ola} (\cref{ch:g12:light-as-wave}), su \omterm{def:g12:sound-acoustics:timbre}{espectro} tiene rayas con líneas nítidas en
\omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} fijado por cada elemento químico (\cref{ch:g10:light-spectra}) --- a
Regla calibrada en laboratorio impresa en cada estrella.

\begin{definition}[Desplazamiento al rojo y al azul]\label{def:g12:doppler-effect:redshift}
Cuando el \omterm{def:g10:light-spectra:emission}{líneas espectrales} de una fuente aparece en \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} \emph{más extenso} que en el
laboratorio, el \omterm{def:g12:sound-acoustics:timbre}{espectro} es \emph{corrido al rojo}\index{corrimiento al rojo}; cuando aparecen \emph{más corto},
es \emph{desplazado hacia el azul}\index{desplazamiento al azul} --- desplazado hacia el extremo rojo o azul de
el \omterm{def:g10:light-spectra:wavelength}{espectro visible}.
\end{definition}

\begin{proposition}[desplazamiento Doppler de la luz]\label{prop:g12:doppler-effect:light}
Para una fuente que se aleja a velocidad radial $v_r \ll c$, se observa cada \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda}
estirado por
\[ \frac{\Delta \lambda}{\lambda} = \frac{\lambda' - \lambda}{\lambda} \approx \frac{v_r}{c} \quad \text{(redshift)}, \]
y comprimido en la misma cantidad, $\Delta\lambda/\lambda \approx -v_r/c$, para un período próximo
fuente (\omterm{def:g12:doppler-effect:redshift}{desplazamiento al azul}).
\end{proposition}
\admitted

\begin{remark}[Donde vive la fórmula honesta]\label{rem:g12:doppler-effect:honest}
La luz no necesita medio, por lo que la derivación \omterm{def:g10:signals-and-waves:sound}{sonido} no se puede copiar --- no hay "en reposo"
en el medio''. La fórmula exacta proviene de la relatividad especial, a finales de este año.
(\cref{ch:g12:special-relativity}); las correcciones son del orden $(v_r/c)^2$, insignificantes para
aviones, estrellas y galaxias cercanas. A continuación utilizamos la fórmula de cámara lenta con la conciencia tranquila.
\end{remark}

\begin{example}[Pesando la fuga de una estrella]\label{ex:g12:doppler-effect:vega}
En el \omterm{def:g12:sound-acoustics:timbre}{espectro} de la brillante estrella Vega, la línea de hidrógeno del laboratorio \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda}
\qty{656.30}{nm} se mide en \qty{656.27}{nm}: $\Delta\lambda = \qty{-0.03}{nm}$, a
\omterm{def:g12:doppler-effect:redshift}{desplazamiento al azul}, entonces Vega se acerca en $v_r = c\,|\Delta\lambda|/\lambda \approx \qty{1.4e4}{m/s}$
--- \qty{14}{km/s}, lee un desplazamiento de una parte entre veinte mil.
\end{example}

\begin{example}[Estrellas oscilantes: binarias y exoplanetas]\label{ex:g12:doppler-effect:wobble}
Cuando dos estrellas \omterm{def:g10:universal-gravitation:satellite}{órbita} se cruzan, sus líneas oscilan \omterm{def:g10:signals-and-waves:period}{periódicamente} azul y rojo --- a
\emph{binario espectroscópico}\index{binario espectroscópico}, \omterm{def:g10:universal-gravitation:satellite}{órbitas} midieron sin resolver nunca el par. un planeta lo hace
lo mismo con su estrella, débilmente: ambas rodean su \omterm{def:g11:forces-and-motion:com}{centro de masa} común, por lo que \omterm{def:g12:doppler-effect:radial}{radiales
velocidad} de la estrella oscila unas pocas decenas de \unit{m/s} --- un desplazamiento $\Delta\lambda/\lambda \sim 10^{-7}$.
Detectándolo es como se encontró el primer planeta alrededor de una estrella similar al Sol; el problema del fin de semana
repite ese descubrimiento, con números y todo.
\end{example}

\begin{remark}[desplazamientos al rojo de galaxias]\label{rem:g12:doppler-effect:expansion}
Los espectros de galaxias distantes están todos \emph{rojo} desplazados, y cuanto más distantes, más
desplazado: las distancias del universo se están alargando. Para galaxias cercanas
$v_r = c\,\Delta\lambda/\lambda$ lee la velocidad de la recesión; para los más lejanos, turnos
superan lo que la fórmula de cámara lenta puede manejar honestamente, y se retoma la contabilidad
nuevamente desde \cref{ch:g12:special-relativity} en adelante, y en los volúmenes universitarios.
\end{remark}

\section{Problema: la búsqueda de un exoplaneta}

\begin{problem}[{Problema del fin de semana --- la búsqueda de un exoplaneta: la bocina de un tren calibra
En el oído, un radar ensaya el doble desplazamiento y, el domingo por la noche, el espectro de una estrella,
tambaleándose en una parte entre cinco millones, ha pesado un planeta que nadie ha visto nunca}]
\label{pb:g12:doppler-effect:1}
Un club de astronomía dedica un fin de semana a una idea (el desplazamiento Doppler) ensayada en la Tierra.
luego apuntó a una estrella similar al Sol. Datos: $v = \qty{340}{m/s}$ para \omterm{def:g10:signals-and-waves:sound}{sonido} en aire,
$c = \qty{3.00e8}{m/s}$, $G = \qty{6.67e-11}{N.m^2/kg^2}$; la estrella tiene masa
$M = \qty{2.0e30}{kg}$. Admitido, de la mecánica de la circular \omterm{def:g10:universal-gravitation:satellite}{órbitas}.
(\cref{ch:g12:satellites-kepler}): un planeta de masa $m$ sobre una circular \omterm{def:g10:universal-gravitation:satellite}{órbita} de radio $r$
tiene \omterm{def:g10:signals-and-waves:period}{período} y velocidad dada por $r^3 = \frac{G M P^2}{4\pi^2}$ y $V = \frac{2\pi r}{P}$,
y la estrella, que gira alrededor del común \omterm{def:g11:forces-and-motion:com}{centro de masa}, se tambalea a la velocidad $K = \frac{m}{M}\,V$.

\medskip
\noindent\textbf{Parte I --- Friday: the level crossing.}
Un tren pasa a velocidad constante, con la bocina encendida; un teléfono graba $f_1 = \qty{471}{Hz}$ mientras
se acerca, $f_2 = \qty{411}{Hz}$ después de que pasa.
\begin{enumerate}
  \item Explica, con la imagen \omterm{def:g12:mechanical-waves:wavefront}{frente de onda lleno de gente}, por qué $f_1 > f_2$ aunque la bocina nunca cambia.
  \item Escribe las dos ecuaciones que vinculan $f_1$, $f_2$ con el verdadero \omterm{def:g10:signals-and-waves:frequency}{frecuencia} $f$ y la velocidad del tren $u$.
  \item Muestre que $u = v\,\dfrac{f_1 - f_2}{f_1 + f_2}$ y calcúlelo en \unit{m/s} y \unit{km/h}.
  \item Muestre que $f = \dfrac{2 f_1 f_2}{f_1 + f_2}$ y calcúlelo. ¿Por qué $f$ es ligeramente
        \emph{abajo} el promedio de $f_1$ y $f_2$?
  \item Calcule el \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} delante y detrás del tren, y el \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} en reposo.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Saturday: the \omterm{met:g10:signals-and-waves:echo}{Radar} gun.}
Una patrulla de tráfico demuestra su \omterm{met:g10:signals-and-waves:echo}{Radar}: $f = \qty{24.125}{GHz}$, y el arma lee el
batir entre el \omterm{def:g10:signals-and-waves:wave}{ola} saliente y el \omterm{rem:g10:signals-and-waves:honest}{eco}.
\begin{enumerate}[resume]
  \item Explique por qué el \omterm{rem:g10:signals-and-waves:honest}{eco} de un automóvil en movimiento se desplaza \emph{dos veces} --- nombre el rol que
        El coche juega en cada turno.
  \item Demuestre que para un automóvil con velocidad $u \ll c$ el ritmo es $\Delta f \approx \dfrac{2u}{c}\,f$.
  \item Un automóvil devuelve $\Delta f = \qty{3.55}{kHz}$: calcule $u$.
  \item Convierta a \unit{km/h} y compárelo con el límite \qty{80}{km/h} publicado allí.
  \item ¿Cuantos \omterm{def:g10:signals-and-waves:frequency}{hercios} de beat corresponden a \qty{1}{km/h}? Comentario: ¿qué hace que un niño sea tan pequeño?
        ¿El cambio relativo de un \qty{24}{GHz} \omterm{def:g10:signals-and-waves:wave}{ola} es fácil de medir?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Sunday: the star that wobbles.}
El club descarga, noche tras noche, el \omterm{def:g12:doppler-effect:radial}{velocidad radial} medido de una estrella parecida al Sol,
trazado a continuación.

\begin{enumerate}[resume]
  \item ¿Qué se mide realmente, noche tras noche, en el \emph{\omterm{def:g12:sound-acoustics:timbre}{espectro}} de la estrella? Estado
        la relación utilizada para convertirlo en $v_r$.
  \item Lea el \omterm{def:g10:signals-and-waves:amplitude}{amplitud} $K$ y el \omterm{def:g10:signals-and-waves:period}{período} $P$ del bamboleo fuera de la curva.
  \item Calcular el swing correspondiente $\Delta\lambda$ de una línea \qty{550}{nm}, y
        $\Delta\lambda/\lambda$. Comparar con la Parte I: ¿cuánto más difícil es esta medición?
        que el del tren?
  \item ¿Por qué un planeta invisible hace que la estrella se tambalee? ¿Qué significa la sinusoidal?
        ¿La forma indica sobre el \omterm{def:g10:universal-gravitation:satellite}{órbita}?
  \item ¿Por qué este método mide sólo la parte \emph{radial} del movimiento de la estrella y
        ¿Qué se vería si el \omterm{def:g10:universal-gravitation:satellite}{órbita} estuviera de cara a nosotros?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Sunday night: weighing the invisible.}
Tome la circular \omterm{def:g10:universal-gravitation:satellite}{órbita} y de canto, por lo que $K$ es la velocidad orbital completa de la estrella.
\begin{enumerate}[resume]
  \item A partir de $P$ y la relación admitida $r^3 = G M P^2/(4\pi^2)$, calcule la
        Radio orbital $r$. Compárelo con la distancia Tierra-Sol, \qty{1.5e11}{m}.
  \item Calcula la velocidad orbital del planeta $V = 2\pi r/P$.
  \item A partir de $K = (m/M)\,V$, calcula la masa del planeta $m$.
  \item Compare $m$ con Júpiter ($\qty{1.9e27}{kg}$) y con la Tierra ($\qty{6.0e24}{kg}$), y
        $r$ al Sol: distancia de Mercurio ($\qty{5.8e10}{m}$). ¿Qué clase de mundo es este?
  \item Si el \omterm{def:g10:universal-gravitation:satellite}{órbita} estuviera inclinado en lugar de de lado, ¿la masa real sería mayor o menor?
        ¿Más pequeño que tu valor? Concluya en una oración: ¿qué hizo un cambio \omterm{def:g10:signals-and-waves:period}{periódico} de uno?
        parte en cinco millones acaba de entregar?
\end{enumerate}
\end{problem}
