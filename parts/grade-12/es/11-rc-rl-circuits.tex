\section{el condensador}

\begin{definition}[Condensador y capacitancia]\label{def:g12:rc-rl-circuits:capacitor}
Un \emph{condensador}\index{condensador} es un par de placas metálicas enfrentadas.
separados por un aislante \omterm{def:g11:lenses-and-eye:lens}{delgado}. Conectadas a una fuente, las placas toman
cargas opuestas $+q$ y $-q$ --- un \omterm{def:g11:electric-gravitational-fields:efield}{campo eléctrico} llena el vacío
(\cref{ch:g11:electric-gravitational-fields}) --- y el \omterm{def:g11:circuits-and-power:voltage}{Voltaje} $u$
a través de las placas es proporcional a la carga almacenada:
\[ q = C\,u . \]
La constante $C$ es la \emph{capacidad}\index{capacidad}, en
\emph{faradios}\index{faradio} (\unit{F}): $\qty{1}{F} = \qty{1}{C/V}$.
\end{definition}

\begin{example}[Órdenes de magnitud]\label{ex:g12:rc-rl-circuits:orders}
Un \omterm{def:g12:rc-rl-circuits:capacitor}{faradio} es enorme: un \omterm{def:g11:fundamental-interactions:charge}{culombio} estacionado por \omterm{def:g11:circuits-and-power:voltage}{voltio}. Cerámica de la radio
circuitos: \unit{pF} a \unit{nF}; electrolíticos de suministros \omterm{def:g11:work-of-force:power}{fuerza} y
parpadea: \unit{\micro F} a \unit{mF}; supercondensadores: miles de \omterm{def:g12:rc-rl-circuits:capacitor}{faradios}.
\end{example}

\begin{proposition}[Corriente en un condensador]\label{prop:g12:rc-rl-circuits:cdudt}
La corriente que llega a la placa de un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} es la velocidad a la que su
la carga crece:
\[
i = \frac{dq}{dt} = C\,\frac{du}{dt} ,
\]
Marcado en minúsculas de cantidades que varían en el tiempo ($U$, $I$ permanecen estables,
\cref{ch:g11:circuits-and-power}).
\end{proposition}

\begin{proof}
La corriente es la carga entregada por segundo.
(\cref{def:g11:circuits-and-power:current}): durante un intervalo cada vez menor,
$i$ es el derivado de $q$; y $q = Cu$ con $C$ constante da
$C\,du/dt$.
\end{proof}

\begin{remark}[Dos consecuencias]\label{rem:g12:rc-rl-circuits:continuity}
Si $u$ es constante, $i = 0$: un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} cargado no pasa corriente constante.
Y $u$ nunca puede saltar --- un paso demandaría una corriente infinita: el
\omterm{def:g11:circuits-and-power:voltage}{Voltaje} de \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} es la memoria del circuito.
\end{remark}

\begin{theorem}[Ley de carga del circuito RC]\label{thm:g12:rc-rl-circuits:charging}
\index{circuito RC}
Después de que el interruptor se cierra, el \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} \omterm{def:g11:circuits-and-power:voltage}{Voltaje} obedece, luego sigue desde
$u(0) = 0$:
\[
RC\,\frac{du}{dt} + u = \mathcal{E},
\qquad
u(t) = \mathcal{E}\left(1 - e^{-t/\tau}\right),
\qquad \tau = RC .
\]
\end{theorem}

\begin{proof}
\omterm{def:g11:circuits-and-power:voltage}{voltajes} a lo largo del bucle agrega (\cref{prop:g11:circuits-and-power:laws}):
$\mathcal{E} = u_R + u = Ri + u$ en cada instante; sustituto
$i = C\,du/dt$ (\cref{prop:g12:rc-rl-circuits:cdudt}). Ahora \emph{verificar}
el candidato: $du/dt = (\mathcal{E}/\tau)e^{-t/\tau}$, entonces
$RC\,(\mathcal{E}/RC)\,e^{-t/\tau} + \mathcal{E}(1 - e^{-t/\tau}) =
\mathcal{E}$, verdadero para todos$t$; y$u(0) = \mathcal{E}(1-1) = 0$. eso
ninguna curva \emph{otro} se ajusta a la ecuación y el comienzo se demuestra en este
curso de un año de matemáticas sobre ecuaciones diferenciales.
\end{proof}

\begin{definition}[constante de tiempo]\label{def:g12:rc-rl-circuits:tau}
El \emph{constante de tiempo}\index{constante de tiempo} de un \omterm{thm:g12:rc-rl-circuits:charging}{circuito RC} es
$\tau = RC$. \omterm{def:g10:orders-of-magnitude:unit}{Unidades}: $\unit{\ohm} \times \unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$ --- el circuito
propio ritmo.
\end{definition}

\begin{method}[Leyendo $\tau$ en la curva]\label{met:g12:rc-rl-circuits:reading}
Tres lecturas equivalentes, sacadas directamente de un gráfico:
\begin{enumerate}
  \item \emph{tangente al origen:} $u'(0) = \mathcal{E}/\tau$, entonces
        la tangente inicial llega a la asíntota en $t = \tau$;
  \item \emph{63\%:} $u(\tau) = \mathcal{E}(1 - e^{-1}) \approx
        0.63\,\mathcal{E}$;
  \item \emph{95\%:} $u(3\tau) \approx 0.95\,\mathcal{E}$ --- cargado,
        para fines prácticos ($5\tau$: $99\%$).
\end{enumerate}
\end{method}

\begin{proposition}[alta]\label{prop:g12:rc-rl-circuits:discharge}
Un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} cargado a $U_0$ y cerrado con una resistencia $R$ solo obedece
$RC\,du/dt + u = 0$ y sigue
\[
u(t) = U_0\,e^{-t/\tau}, \qquad \tau = RC
\]
--- $37\%$ se fue en $\tau$, $5\%$ en $3\tau$.
\end{proposition}

\begin{proof}
Misma ley de bucle, sin fuente; sustitución nuevamente:
$RC(-U_0/\tau)e^{-t/\tau} + U_0 e^{-t/\tau} = 0$, con $u(0) = U_0$.
\end{proof}

\begin{example}[Números de una unidad de flash]\label{ex:g12:rc-rl-circuits:flashrc}
Un flash fotográfico carga $C = \qty{800}{\micro F}$ a través
$R = \qty{1.0}{k\ohm}$: $\tau = 10^3 \times 8.0\times10^{-4} =
\qty{0.80}{s}$, entonces la luz "listo" --- conectada para dispararse en$95\%$ ---

más rápido afuera que adentro.
\end{example}

\section{La bobina: la inductancia y el circuito RL}

\begin{definition}[Inductor e inductancia]\label{def:g12:rc-rl-circuits:inductor}
Un \emph{inductor}\index{inductor} es un \omterm{def:g11:magnetic-fields:solenoid}{bobina} de cable: su corriente
enhebra sus propias vueltas con un \omterm{def:g11:magnetic-fields:bfield}{campo magnético}
(\cref{ch:g11:magnetic-fields}); siempre que sea \emph{cambios}, un \omterm{def:g11:circuits-and-power:voltage}{Voltaje}
aparece en \omterm{def:g11:magnetic-fields:solenoid}{bobina}:
\[
u = L\,\frac{di}{dt} .
\]
La constante $L$ es la \emph{inductancia}\index{inductancia}, en
\emph{enrique}\index{enrique} (\unit{H}): $\qty{1}{H} = \qty{1}{V.s/A}$.
\end{definition}
\admitted

\begin{remark}[La bobina resiste el cambio.]\label{rem:g12:rc-rl-circuits:oppose}
Por qué una corriente cambiante genera una \omterm{def:g11:circuits-and-power:voltage}{Voltaje} --- inducción --- se deriva en
los volúmenes universitarios; Aquí tomamos la ley y su carácter: la
\omterm{def:g11:magnetic-fields:solenoid}{bobina} \emph{se opone a los cambios} de su corriente. Gemelos espejo: un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador}
$u$ no puede saltar, $i$ de \omterm{def:g11:magnetic-fields:solenoid}{bobina} no puede saltar.
\end{remark}

    to[R=$R$] (4.2,2.4) -- (5.4,2.4) to[L=$L$, i=$i$, v=$u$] (5.4,0) -- (0,0);
\end{circuitikz}

{\small El \omterm{prop:g12:rc-rl-circuits:rl}{circuito RL}: después de que se cierra el $K$, el \omterm{def:g11:magnetic-fields:solenoid}{bobina} deja que el actual
sube sólo a tu propio ritmo $\tau = L/R$.}
\end{omfigure}

\begin{proposition}[Aumento de corriente en un circuito RL]\label{prop:g12:rc-rl-circuits:rl}
\index{circuito RL}
Cerrando el interruptor de una fuente $\mathcal{E}$, una resistencia $R$ y una \omterm{def:g11:magnetic-fields:solenoid}{bobina}
$L$ en serie da $\mathcal{E} = Ri + L\,di/dt$, es decir
\[
\frac{L}{R}\,\frac{di}{dt} + i = \frac{\mathcal{E}}{R},
\qquad\text{solved by}\quad
i(t) = \frac{\mathcal{E}}{R}\left(1 - e^{-t/\tau}\right),
\quad \tau = \frac{L}{R} .
\]
\end{proposition}

\begin{proof}
La ley del bucle otra vez; y es el \emph{misma ecuación} como en
\cref{thm:g12:rc-rl-circuits:charging} con $u \to i$,
$\mathcal{E} \to \mathcal{E}/R$, $RC \to L/R$ --- la misma sustitución
verifica la solución y cada lectura de
Transferencias \cref{met:g12:rc-rl-circuits:reading}.
\end{proof}

\begin{example}[Se despierta un electroimán]\label{ex:g12:rc-rl-circuits:rl}
Un relé \omterm{def:g11:magnetic-fields:solenoid}{bobina}, $L = \qty{0.30}{H}$ y $R = \qty{6.0}{\ohm}$, en un
Suministro \qty{12}{V}: corriente final $\mathcal{E}/R = \qty{2.0}{A}$, en
el ritmo $\tau = 0.30/6.0 = \qty{50}{ms}$ --- despierto en $3\tau \approx
\qty{0.15}{s}$.
\end{example}

\begin{remark}[La chispa en la apertura]\label{rem:g12:rc-rl-circuits:spark}
Abra un interruptor en un \omterm{def:g11:magnetic-fields:solenoid}{bobina} en vivo y $i$ debe fallar a cero en una fracción
de un milisegundo: $di/dt$ es enorme y $u = L\,di/dt$ llega a cientos
o miles de \omterm{def:g11:circuits-and-power:voltage}{voltios} --- una chispa salta el contacto. El encendido de un coche
\omterm{def:g11:magnetic-fields:solenoid}{bobina} lo hace a propósito, miles de veces por minuto.
\end{remark}

\section{Energía almacenada}

\begin{proposition}[Energía en un condensador, energía en una bobina]\label{prop:g12:rc-rl-circuits:energy}
\index{energía!de un condensador}\index{energía!de una bobina}
Un \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} $C$ cargado a \omterm{def:g11:circuits-and-power:voltage}{Voltaje} $u$ y un \omterm{def:g11:magnetic-fields:solenoid}{bobina} $L$ que transporta corriente
Tienda $i$
\[
E_C = \tfrac12\,C u^2 = \frac{q^2}{2C},
\qquad
E_L = \tfrac12\,L i^2 .
\]
\end{proposition}

\begin{proof}
Obtener una pequeña carga $dq$ en placas que ya cuestan \omterm{def:g11:circuits-and-power:voltage}{Voltaje} $u$
$u\,dq$ (\cref{def:g11:circuits-and-power:voltage}); apilando el
costos, el total es el área bajo la línea $u = q/C$ --- el triángulo
a continuación --- entonces $E_C = \tfrac12 qu = \tfrac12 Cu^2$. El mismo triángulo para
el \omterm{def:g11:magnetic-fields:solenoid}{bobina} (aumentar $i$ en $di$ cuesta $Li\,di$) da
$E_L = \tfrac12 Li^2$.
\end{proof}

    {$E_C$};
\end{tikzpicture}

en nivel bajo \omterm{def:g11:circuits-and-power:voltage}{Voltaje}.}
\end{omfigure}

\begin{example}[Órdenes de magnitud de energía]\label{ex:g12:rc-rl-circuits:energies}
El flash \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} de \cref{ex:g12:rc-rl-circuits:flashrc}:
$\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$. un
Supercondensador \qty{3000}{F} en \qty{2.7}{V}: \qty{1.1e4}{J} --- un AA
El valor de la battery. A \qty{10}{mH} \omterm{def:g11:magnetic-fields:solenoid}{bobina} en \qty{10}{A}: \qty{0.50}{J}.
Menos de lo que contienen las batterys, pero se puede liberar en un milisegundo.
\end{example}

\begin{remark}[Para qué sirven los dos cronómetros]\label{rem:g12:rc-rl-circuits:applications}
\index{suavizado}\index{circuito de sincronización}
Entrada lenta, salida rápida \omterm{def:g11:work-of-force:power}{potestades} del flash de la cámara y del desfibrilador. el
\omterm{def:g12:mechanical-waves:delay}{demora} $\tau = RC$ ejecuta \omterm{rem:g12:rc-rl-circuits:applications}{circuitos de sincronización}: limpiaparabrisas intermitente, parpadeando
balizas, luces de escaleras. A \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} a través de un suministro \emph{suaviza}
\omterm{def:g11:circuits-and-power:voltage}{Voltaje} ($u$ no puede saltar); una serie \omterm{def:g11:magnetic-fields:solenoid}{bobina} suaviza la corriente ($i$ no puede
saltar). Y \omterm{def:g11:magnetic-fields:solenoid}{bobina} más \omterm{def:g12:rc-rl-circuits:capacitor}{condensador} juntos forman un \omterm{def:g12:oscillators-and-time:oscillator}{oscilador}
(\cref{ch:g12:rlc-oscillations}).
\end{remark}

\section{Problema: el flash de la cámara}

\begin{problem}[{Problema del fin de semana --- el flash de la cámara: dos segundos
