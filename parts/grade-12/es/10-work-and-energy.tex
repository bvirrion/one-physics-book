\section{Trabaja por cualquier camino}

\begin{definition}[Trabajar a lo largo de un camino]\label{def:g12:work-and-energy:path}
Dejemos que \omterm{def:g10:inertia:force}{fuerza} $\vect F$ actúe sobre un cuerpo que se mueve a lo largo de cualquier camino desde $A$ hasta
$B$: corta el camino en desplazamientos $\vect{\dd\ell}$ tan cortos que
cada uno es recto y $\vect F$ apenas cambia a lo largo de él. Cada paso gana
el pequeño \omterm{def:g11:work-of-force:work}{trabajar} $\scal{F}{\dd\ell} = F\,\dd\ell\cos\theta$
(\cref{ch:g11:work-of-force}), y el
\emph{trabajar a lo largo del camino}\index{trabajo (de una fuerza)!a lo largo de un camino} es
su suma, $W_{AB}(\vect F) = \sum \scal{F}{\dd\ell}$ --- que en un
camino recto con constante \omterm{def:g10:inertia:force}{fuerza} colapsa hasta el año pasado
$F \times AB \times \cos\theta$.
\end{definition}

\begin{proposition}[Trabajo del peso, cualquier recorrido]\label{prop:g12:work-and-energy:weight}
A lo largo del trayecto \emph{cualquier} desde $A$ (altitud $z_A$) a $B$ (altitud
$z_B$), el \omterm{def:g10:universal-gravitation:weight}{peso} de una masa $m$ hace el \omterm{def:g11:work-of-force:work}{trabajar}
$W_{AB}(\vect P) = mg\,(z_A - z_B)$: entra la bajada, nunca la ruta.
\end{proposition}

\begin{proof}
Cada paso corto contribuye $mg \times (\text{small drop})$ (último
cálculo del año en segmentos rectos); el telescopio de gotas para
$z_A - z_B$. Se entrega el caso curvo, admitido entonces.
\end{proof}

\begin{definition}[Fuerzas conservadoras y no conservadoras]\label{def:g12:work-and-energy:conservative}
Un \omterm{def:g10:inertia:force}{fuerza} es \emph{conservador}\index{fuerza conservadora} si es \omterm{def:g11:work-of-force:work}{trabajar}
entre dos puntos es el mismo en todos los caminos (cero en cada ronda
viaje), \emph{no conservador}\index{fuerza no conservadora}
de lo contrario. \omterm{def:g10:universal-gravitation:weight}{Peso} y resorte \omterm{def:g10:inertia:force}{fuerza} (siguiente apartado): conservador.
Deslizar \omterm{def:g10:inertia:inventory}{fricción} no es: opuesto al movimiento, lo hace $W = -fL$ en un
trayecto de longitud $L$ --- un peaje según \omterm{def:g10:orders-of-magnitude:unit}{metro}, nunca reembolsado.
\end{definition}

\begin{example}[Dos senderos hasta la cabaña.]\label{ex:g12:work-and-energy:trails}
Un excursionista \qty{75}{kg} llega a \qty{300}{m} por un sendero empinado \qty{800}{m}
o \qty{2.4}{km} de curvas. Los cargos \omterm{def:g10:universal-gravitation:weight}{peso}
$-75 \times 9.81 \times 300 \approx \qty{-2.2e5}{J}$ en ambos; un
\qty{30}{N} \omterm{def:g10:inertia:inventory}{fricción} \omterm{def:g10:inertia:force}{fuerza}, \qty{-2.4e4}{J} en uno, \qty{-7.2e4}{J}
por el otro. El \omterm{def:g10:universal-gravitation:weight}{peso} se devuelve en el descenso; La factura de \omterm{def:g10:inertia:inventory}{fricción} es
se ha ido como calor, en ambos sentidos.
\end{example}

\section{Energía potencial}

\begin{definition}[Potencial de energía]\label{def:g12:work-and-energy:ep}
A cada \omterm{def:g12:work-and-energy:conservative}{fuerza conservadora} se le adjunta un
\emph{energía potencial}\index{energía potencial!del trabajo conservador}
$E_p$, un número que depende únicamente de la posición, definido por
$W_{AB} = E_p(A) - E_p(B)$: \omterm{def:g11:work-of-force:work}{trabajar} entregado es potencial \omterm{def:g10:energy-conservation:energy}{energía} gastado.
Al arreglar $E_p = 0$ en un punto de referencia elegido, se arregla en todas partes; el
La elección desplaza todos los valores mediante una constante que se cancela de cada
diferencia. Para el \omterm{def:g10:universal-gravitation:weight}{peso} esto devuelve el $E_p = mgz$ del año pasado,
ahora válido en cualquier camino. \omterm{def:g10:inertia:inventory}{Fricción} no puede poseer ningún $E_p$: una ruta dependiente
La factura no es diferente de los números de los terminales --- solo \omterm{def:g12:work-and-energy:conservative}{fuerzas conservadoras}
almacenar de forma recuperable, exactamente lo que conserva el nombre.
\end{definition}

\begin{proposition}[Energía potencial elástica]\label{prop:g12:work-and-energy:elastic}
Un resorte de \omterm{def:g12:oscillators-and-time:spring}{rigidez} $k$ (\unit{N/m}) estirado o comprimido por $x$
retrocede con el \omterm{def:g10:inertia:force}{fuerza} $F = kx$
(\cref{ch:g12:oscillators-and-time}); el \omterm{def:g11:work-of-force:work}{trabajar} se entrega regresando
descansar --- es
\emph{energía potencial elástica}\index{energía potencial!elástico} --- es
$E_p = \tfrac12\, k x^2$.
\end{proposition}

\begin{proof}
El \omterm{def:g12:oscillators-and-time:pendulum}{fuerza restauradora} actúa a lo largo del movimiento, por lo que $\sum F\,\dd\ell$ es
el \emph{área bajo el gráfico $F$--$x$}: un triángulo de base $x$ y
altura $kx$, área $\tfrac12 x \times kx$. El \omterm{def:g10:inertia:force}{fuerza} en cada tramo.
es el mismo de ida y vuelta: \omterm{def:g11:work-of-force:work}{trabajar} fijado por los puntos finales, resorte \omterm{def:g10:inertia:force}{fuerza}
\omterm{def:g12:work-and-energy:conservative}{conservador}, $E_p$ bien definidos.
\end{proof}

\section{Los dos teoremas de la energía.}

\begin{theorem}[Teorema de la energía cinética]\label{thm:g12:work-and-energy:ketheorem}\index{teorema de la energía cinética}
Para cualquier movimiento de un cuerpo de masa $m$ de $A$ a $B$ --- curvo,
en bucle, tridimensional ---
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
la suma que recorre todos los \omterm{def:g10:inertia:force}{efectivo} que actúan sobre el cuerpo.
\end{theorem}

\begin{proof}
Desde $v^2 = \scal{v}{v}$, la regla del producto de las matemáticas de este año
da la derivada de $v^2$ como $2\,\scal{a}{v}$, por lo que
$\dd E_k/\dd t = m\,\scal{a}{v} = \scal{F_{\text{tot}}}{v}$ por
\omterm{thm:g12:newtons-laws:second}{Segunda ley de Newton} (\cref{ch:g12:newtons-laws}). En breve $\dd t$
el cuerpo se mueve $\vect{\dd\ell} = \vect v\,\dd t$: $E_k$ gana el
pequeño \omterm{def:g11:work-of-force:work}{trabajar} de \cref{def:g12:work-and-energy:path}; sumando los pasos
suma el \omterm{def:g11:work-of-force:work}{obras} --- declaración admitida del año pasado, obtenida en su totalidad.
\end{proof}

\begin{proposition}[Potencia instantánea]\label{prop:g12:work-and-energy:power}
En cada instante un \omterm{def:g10:inertia:force}{fuerza} $\vect F$ sobre un cuerpo de velocidad $\vect v$
entrega el \omterm{def:g11:work-of-force:power}{fuerza} $P = \scal{F}{v} = Fv\cos\theta$, y
$\dd E_k/\dd t$ es la suma de los \omterm{def:g11:work-of-force:power}{potestades} de todos los \omterm{def:g10:inertia:force}{efectivo}: del año pasado
Fórmula de velocidad constante, exacta en cada instante de cualquier movimiento.
\end{proposition}

\begin{proof}
Lea la prueba anterior: cada \omterm{def:g10:inertia:force}{fuerza} alimenta
$\scal{F}{\dd\ell} = \scal{F}{v}\,\dd t$ en $E_k$ durante $\dd t$.
\end{proof}

\begin{theorem}[Teorema de la energía mecánica]\label{thm:g12:work-and-energy:emtheorem}\index{teorema de la energía mecánica}
Sea $E_m = E_k + E_p$, donde $E_p$ recoge las energías potenciales de
todo el \omterm{def:g12:work-and-energy:conservative}{fuerzas conservadoras} en \omterm{def:g11:work-of-force:work}{trabajar}. Luego, por cualquier camino,
\[
\Delta E_m = W_{AB}(\text{non-conservative forces}),
\]
y $E_m$ se conserva siempre que \omterm{def:g12:work-and-energy:conservative}{fuerzas no conservadoras} no haga \omterm{def:g11:work-of-force:work}{trabajar}.
\end{theorem}

\begin{proof}
Divida el teorema \omterm{def:g10:energy-conservation:energy}{energía-cinética}: $\Delta E_k = W_{\text{cons}} +
W_{\text{nc}} = -\Delta E_p + W_{\text{nc}}$; mueva$\Delta E_p$ a través.
\end{proof}

\begin{example}[El péndulo, sinceramente esta vez.]\label{ex:g12:work-and-energy:pendulum}
Una masa de péndulo sobre un alambre de longitud $L = \qty{2.5}{m}$ se suelta en
descanse en $\qty{45}{\degree}$ desde la vertical. El \omterm{def:g10:inertia:inventory}{tensión},
perpendicular al movimiento, es impotente; el \omterm{def:g10:universal-gravitation:weight}{peso} es \omterm{def:g12:work-and-energy:conservative}{conservador}:
Se conserva $E_m$ \emph{en el arco curvo}, que el año pasado pudo
sólo debe comprobarse. La sacudida comienza $h = L(1 - \cos\qty{45}{\degree}) =
\qty{0.73}{m}$hacia arriba, por lo que en la parte inferior$v = \sqrt{2gh} \approx
\qty{3.8}{m/s}$.
\end{example}

\begin{method}[Auditoría energética de cualquier movimiento.]\label{met:g12:work-and-energy:audit}
\begin{enumerate}
  \item Elija el sistema y enumere el \omterm{def:g10:inertia:force}{efectivo} en él.
  \item Ordenar: perpendicular al movimiento $\to$ no \omterm{def:g11:work-of-force:work}{trabajar}; \omterm{def:g12:work-and-energy:conservative}{conservador}
        $\to$ y $E_p$ en $E_m$; el resto $\to$ $W_{\text{nc}}$
        (deslizante \omterm{def:g10:inertia:inventory}{fricción}: $-fL$, $L$ = longitud del recorrido).
  \item Escriba $\Delta E_m = W_{\text{nc}}$ entre el punto de datos y
        el punto de pregunta; resolver. \omterm{def:g10:energy-conservation:energy}{Energía} responde ``qué rápido, qué tan rápido
        alto''; sólo el tiempo y la aceleración necesitan las leyes de Newton.
\end{enumerate}
\end{method}

\begin{definition}[Diagrama energético, puntos de inflexión.]\label{def:g12:work-and-energy:diagram}
Para un movimiento a lo largo de un eje $x$ con $E_m$ conservado, el
\emph{diagrama de energía}\index{diagrama de energía} traza la curva $E_p(x)$
y la línea horizontal $E_m$. Desde $E_k = E_m - E_p \geq 0$, el
el movimiento se limita al lugar donde la curva se encuentra debajo de la línea; la brecha
entre ellos está $E_k$, lea directamente. en un
\emph{momento crucial}\index{momento crucial}, donde la curva se encuentra con la línea,
$v = 0$ y el movimiento se invierte.
\end{definition}

\begin{proposition}[Fuerza del paisaje; equilibrios]\label{prop:g12:work-and-energy:equilibria}
El \omterm{def:g12:work-and-energy:conservative}{fuerza conservadora} a lo largo de $x$ es menos la pendiente de $E_p$:
$F = -\,\dd E_p/\dd x$ --- apunta \emph{cuesta abajo} en el diagrama,
y desaparece donde la tangente es horizontal: un equilibrio. un
mínimo de $E_p$ es \emph{estable}\index{equilibrio estable} --- el
el cuerpo desplazado es empujado hacia atrás, una pelota en un valle; un máximo es
\emph{inestable}\index{equilibrio inestable} --- es empujado, un
pelota en la cima de una colina.
\end{proposition}

\begin{proof}
En un pequeño escalón $\dd x$ el \omterm{def:g10:inertia:force}{fuerza} \omterm{def:g11:work-of-force:work}{obras} $F\,\dd x = E_p(x) -
E_p(x + \dd x) = -\dd E_p$; dividir por$\dd x$. Cuesta abajo a ambos lados
de un valle apunta hacia atrás; alrededor de la cima de una colina, afuera.
\end{proof}

\begin{remark}[Oscilaciones y escape]\label{rem:g12:work-and-energy:escape}
Las pequeñas oscilaciones de \cref{ch:g12:oscillators-and-time} viven en
los fondos de los valles $E_p$. Y lejos de la Tierra la gravedad se debilita: el
La curva $E_p(r)$ sube cada vez más lentamente hacia un techo finito (el
El área bajo el gráfico de fuerza de contracción es finita). Un cohete cuyo $E_m$
llega al techo nunca se encuentra con un \omterm{def:g12:work-and-energy:diagram}{momento crucial}:
\emph{escapa}\index{velocidad de escape} --- sobre \qty{11.2}{km/s} de
el suelo (\cref{exo:g12:work-and-energy:13}); debajo de eso está
atado. El límite máximo se calcula en el volumen del Año 1.
\end{remark}

\begin{example}[bucle-bucle]\label{ex:g12:work-and-energy:loop}

El problema de la montaña rusa del año pasado, ahora es nuestro. Conservación desde el reposo en $H$

\end{example}

\section{Problema: el salto de esquí}

\begin{problem}[{Problema del fin de semana --- el salto de esquí: una carrera auditada
por una pistola rápida, un vuelo en caída libre, la suave geometría de una pendiente empinada
aterrizaje y la torre que un diseñador debe comprar una vez que la fricción pasa}]
\label{pb:g12:work-and-energy:1}
Un saltador de esquí olímpico ($m = \qty{70}{kg}$, esquís incluidos) comienza en
descansar desde una puerta $H = \qty{45}{m}$ encima de la mesa de despegue, toboganes
$L = \qty{90}{m}$ de entrada curva, y sale de la mesa en forma horizontal;
En una pistola rápida se lee $v_0 = \qty{26}{m/s}$. El punto de aterrizaje se encuentra
$h = \qty{40}{m}$ más abajo, en pendiente $\qty{35}{\degree}$. Partes II
y III desprecie la resistencia del aire.

\medskip\noindent\textbf{Parte I --- The in-run.}
\begin{enumerate}
  \item Predecir la velocidad de despegue sin fricción; ¿Por qué la curva de
        ¿La carrera es irrelevante?
  \item Referencia en la tabla: calcular $E_m$ en puerta y en el despegue.
  \item Calcular $\Delta E_m$; ¿Qué teorema nombra a los culpables y quiénes?
  \item Deduzca el promedio \omterm{def:g11:work-of-force:sign}{resistador} \omterm{def:g10:inertia:force}{fuerza} (nieve más aire) durante el recorrido.
  \item ¿Qué fracción de $E_m$ se pierde? ¿Por qué encerar los esquís y agacharnos?
\end{enumerate}

\medskip\noindent\textbf{Parte II --- The flight.}
\begin{enumerate}[resume]
  \item Sólo el \omterm{def:g10:universal-gravitation:weight}{peso} actúa en vuelo: cuyo movimiento
        \cref{ch:g12:projectile-motion} es esto? Componentes de velocidad
        en el aterrizaje?
  \item Velocidad de aterrizaje, dos veces: desde componentes y desde $E_m$
        conservación. ¿En \unit{km/h}?
  \item Calcula el tiempo de vuelo y la distancia horizontal recorrida.
  \item Los saltadores reales surcan el aire como un ala, volando mucho más lejos:
        ¿Puede el aire \emph{aumentar} la velocidad de aterrizaje? discutir con el
        Teorema \omterm{def:g10:energy-conservation:energy}{energía-mecánica}.
\end{enumerate}

\medskip\noindent\textbf{Parte III --- The landing.}
\begin{enumerate}[resume]
  \item Calcule el ángulo de la velocidad de aterrizaje debajo de la horizontal.
  \item Descomponerlo en componentes paralelos y perpendiculares a
        la pendiente.
  \item Los catetos absorben sólo la parte perpendicular: calcular que
        \omterm{def:g11:mechanical-energy:kinetic}{energía cinética}; compararlo con el total.
  \item Terreno plano absorbería el lote: a qué desnivel está
        cada equivalente de aterrizaje ($h_{\text{eq}} = v_\perp^2/2g$,
        respectivamente\ $v^2/2g$)?
  \item Una frase: ¿por qué las colinas de aterrizaje son empinadas y curvas?
\end{enumerate}

\medskip\noindent\textbf{Parte IV --- The designer's tower.}
Una colina más pequeña necesita despegar sólo en $v_0' = \qty{24}{m/s}$; es
El tramo recto desciende en $\qty{35}{\degree}$, por lo que la altura de la puerta $H'$
significa longitud de vía $H'/\sin\qty{35}{\degree}$; mismo arrastre \qty{80}{N}.
\begin{enumerate}[resume]
  \item Calcule la altura de la puerta que construiría un diseñador sin fricción.
  \item Escribe el teorema \omterm{def:g10:energy-conservation:energy}{energía-mecánica}, puerta a tabla, para $H'$.
  \item Resuelva para $H'$.
  \item ¿En qué porcentaje subió \omterm{def:g10:inertia:inventory}{fricción} la torre?
  \item La masa ya no se cancela: recalcular $H'$ para un
        Jersey \qty{60}{kg}. ¿Quién necesita una torre más alta y por qué?
  \item Punchline: la torre anunciada y la parte de \omterm{def:g10:inertia:inventory}{fricción} en ella.
\end{enumerate}
\end{problem}
