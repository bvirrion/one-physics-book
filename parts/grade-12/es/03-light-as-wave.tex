\chapter{La luz como onda: difracción e interferencia}\label{ch:g12:light-as-wave}

Incline una película de jabón hacia una ventana: los colores se deslizan a través de ella
nada en el agua o el jabón tiene color. Apunte un \omterm{def:g11:color-light-sources:lamps}{láser} a un solo cabello:
la pared de atrás no muestra una sombra \omterm{def:g11:lenses-and-eye:lens}{delgada como un cabello} sino una raya brillante,
barrado de líneas oscuras, miles de veces más anchas que el cabello.
Los rayos que marchan en línea recta (\cref{ch:g10:refraction}) no explican ninguno de los dos
vista. Este capítulo promueve la luz a un \omterm{def:g10:signals-and-waves:wave}{ola} completo --- medio
micrómetro de cresta a cresta --- luego materializa la idea en: a \omterm{def:g11:color-light-sources:lamps}{láser}
y una pared desnuda medirá un cabello, un \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda}, los surcos de un CD.

\section{La luz es una onda.}

\begin{definition}[ondas de luz]\label{def:g12:light-as-wave:lightwave}
\omterm{def:g10:light-spectra:white}{luz monocromática} --- un color puro, que ningún prisma puede dividir
además (\cref{ch:g10:light-spectra}) --- es un \omterm{def:g10:signals-and-waves:period}{periódico} \emph{luz
ola}\index{onda de luz}: cruza vacío en $c = \qty{3.00e8}{m/s}$
y se caracteriza por su \omterm{def:g10:signals-and-waves:frequency}{frecuencia} $f$ o su \emph{longitud de onda en
vacío}\index{longitud de onda!de luz} $\lambda = c/f$, el
distancia de cresta a cresta de \cref{ch:g12:mechanical-waves}. el ojo
responde aproximadamente desde \qty{400}{nm} (violeta) hasta \qty{800}{nm} (rojo);
\omterm{def:g10:light-spectra:white}{luz blanca} mezcla todo el \omterm{def:g11:fundamental-interactions:strong}{rango}.
\end{definition}

\begin{example}[Órdenes de magnitud]\label{ex:g12:light-as-wave:orders}
Un helio-neón \omterm{def:g11:color-light-sources:lamps}{láser} emite luz roja de $\lambda = \qty{633}{nm} =
\qty{6.33e-7}{m}$, por lo que$f = c/\lambda \approx \qty{4.7e14}{Hz}$: no
electrónica cuenta así de rápido --- cada \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} en este capítulo
ser medido por la geometría. Para escala, un cabello (aproximadamente \qty{70}{\micro m})
tiene cien \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} de ancho. ¿Qué \omterm{def:g10:signals-and-waves:wave}{ondas}? No el aire: un eléctrico
y \omterm{def:g11:magnetic-fields:bfield}{campo magnético}, descrito honestamente en el volumen del Año 2; solo el
\omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} asuntos a continuación.
\end{example}

\section{Difracción}

\begin{definition}[Difracción]\label{def:g12:light-as-wave:diffraction}
Cuando un \omterm{def:g10:signals-and-waves:wave}{ola} encuentra una abertura u obstáculo de tamaño $a$ comparable a
su \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda}, se propaga hacia la región que los rayos dejarían oscura.
Esta propagación es \emph{difracción}\index{difracción} --- insignificante
mientras que $\lambda/a$ es pequeño y dominante a medida que $a$ se reduce hacia $\lambda$.
\end{definition}

\begin{proposition}[Medio ancho angular de la extensión]\label{prop:g12:light-as-wave:halfwidth}
Una rendija de ancho $a$, iluminada con \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} $\lambda < a$, extiende el
luz alrededor de la dirección de incidencia dentro del semiancho angular
$\theta$ (radianes); una pantalla a distancia $D$ muestra un brillo central
banda de ancho $L$, dos veces $D \tan\theta \approx D\,\theta$ (pequeña
ángulo):
\[
\theta \approx \frac{\lambda}{a}, \qquad L = \frac{2\lambda D}{a}.
\]
\end{proposition}
\admitted

\begin{remark}[Lea la fórmula al revés]\label{rem:g12:light-as-wave:honest}
El \emph{más estrecho} la rendija, el \emph{más ancho} la extensión --- el
Lo contrario de la predicción del rayo. La derivación honesta (suma
ondas reemitidas a través de la rendija) vive en el volumen del Año 2; aquí
la fórmula es una herramienta de medición.
\end{remark}

\begin{proposition}[Un obstáculo se difracta como una rendija]\label{prop:g12:light-as-wave:hair}
Un obstáculo opaco \omterm{def:g11:lenses-and-eye:lens}{delgado} de ancho $d$ --- un alambre, un cabello --- produce,
fuera del haz incidente, el mismo patrón que una hendidura de ancho $d$
(El principio de Babinet, derivado en el volumen Year~2).
\end{proposition}
\admitted

\begin{method}[Midiendo un ancho diminuto]\label{met:g12:light-as-wave:hairwidth}
\begin{enumerate}
  \item Estire el cabello a lo largo del haz \omterm{def:g11:color-light-sources:lamps}{láser}; establecer una pantalla en un
        distancia medida $D$ (algunos \omterm{def:g10:orders-of-magnitude:unit}{metros}).
  \item Mida el ancho de la banda central $L$, de línea oscura a línea oscura.
  \item Desde $L = 2\lambda D/d$: $d = 2\lambda D/L$ --- cabello más fino,
        patrón más grande.
\end{enumerate}
\end{method}

\begin{remark}[La difracción limita cada instrumento.]\label{rem:g12:light-as-wave:limits}
La luz entra en una cámara, un telescopio o un ojo a través de una abertura de
ancho $a$, por lo que incluso un \omterm{def:g11:lenses-and-eye:lens}{lente} perfecto difumina un punto en un punto de
tamaño angular sobre $\lambda/a$: los detalles más cercanos se fusionan. Una luz visible
por lo tanto, el microscopio no resuelve nada muy por debajo de $\lambda \approx
\qty{0.5}{\micro m}$ (bacterias sí, virus no); espejos telescópicos
Se construyen anchos por la misma razón.
\end{remark}

\section{Interferencia}

\begin{definition}[Fuentes coherentes, diferencia de ruta.]\label{def:g12:light-as-wave:coherent}
Dos fuentes \omterm{def:g10:signals-and-waves:wave}{ola} son \emph{coherente}\index{fuentes coherentes} cuando
tienen el mismo \omterm{def:g10:signals-and-waves:frequency}{frecuencia} y vibran al paso --- en la práctica,
cuando son dos copias de un mismo \omterm{def:g10:signals-and-waves:wave}{ola}, obtenidas al dividirlo. por un
punto $M$ alcanzado por ambos, el \emph{diferencia de camino}%
\index{diferencia de camino} es $\delta = S_2M - S_1M$.
\end{definition}

\begin{proposition}[Condiciones de interferencia]\label{prop:g12:light-as-wave:conditions}
Donde dos \omterm{def:g12:light-as-wave:coherent}{coherente} \omterm{def:g10:signals-and-waves:wave}{ondas} de \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} $\lambda$ se superponen se suman ---
ellos \emph{interferir}\index{interferencia}:
\begin{itemize}
  \item $\delta = k\lambda$ ($k$ entero): en paso, reforzando ---
        \emph{constructivo}\index{interferencia constructiva} interferencia (brillante);
  \item $\delta = (k + \tfrac12)\lambda$: cresta en valle, anulando
        --- Interferencia \emph{destructivo}\index{interferencia destructiva} (oscura).
\end{itemize}
\end{proposition}

\begin{proof}
Cambiar un \omterm{def:g12:mechanical-waves:periodic}{onda periódica} por \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} completo no cambia nada: crestas
amontonarse en crestas. Medio \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} más pone cresta en valle: igual
\omterm{def:g10:signals-and-waves:wave}{ondas} cancelar.
\end{proof}

\begin{example}[Experimento de dos rendijas de Young]\label{ex:g12:light-as-wave:young}
Perfore dos finas hendiduras $S_1$, $S_2$, separadas por una fracción de milímetro,
en una placa opaca, y iluminar ambos con un \omterm{def:g11:color-light-sources:lamps}{láser}: dos \omterm{def:g12:light-as-wave:coherent}{coherente
fuentes}. En una pantalla a un par de \omterm{def:g10:orders-of-magnitude:unit}{metros} de distancia, la superposición tiene rayas:
en el centro $\delta = 0$, brillante; subiendo la pantalla, $\delta$
crece a través de $\lambda/2$ (oscuro), $\lambda$ (brillante)\,\dots\ oscuridad
donde la luz sumada a la luz se cancela.
\end{example}

\begin{definition}[Francos e interflecos]\label{def:g12:light-as-wave:fringes}
Las bandas alternas brillantes y oscuras del patrón son su
\emph{franjas de interferencia}\index{franjas de interferencia}; la distancia
$i$ entre centros brillantes consecutivos es el \emph{interfringe}%
\index{interfringe}.
\end{definition}

\begin{proposition}[La interfringe]\label{prop:g12:light-as-wave:interfringe}
Para rendijas a una distancia $a$ y una pantalla a una distancia $D \gg a$,
el punto de la pantalla a la distancia $x$ del centro $O$ tiene \omterm{def:g12:light-as-wave:coherent}{camino
diferencia} $\delta \approx a\,x/D$ (geometría admitida); por
\cref{prop:g12:light-as-wave:conditions}, flecos brillantes sentados en
$x_k = k\,\lambda D/a$, espaciados uniformemente con \omterm{def:g12:light-as-wave:fringes}{interfringe}
\[
i = \frac{\lambda D}{a}.
\]
Con $\lambda = \qty{633}{nm}$, $a = \qty{0.20}{mm}$, $D =
\qty{2.0}{m}$:$i = \qty{6.3}{mm}$--- la configuración amplía$\lambda$
por $D/a = 10^4$. La geometría detrás de esto se resuelve en el Año~2.
volumen.
\end{proposition}
\admitted

\begin{method}[Medición de una longitud de onda]\label{met:g12:light-as-wave:lambda}
\begin{enumerate}
  \item Envía el \omterm{def:g11:color-light-sources:lamps}{láser} desconocido a través de una doble rendija de espaciado conocido
        $a$; Coloque la pantalla a una distancia medida $D$.
  \item Mida el lapso de $10$ \omterm{def:g12:light-as-wave:fringes}{interfiere}; dividir por $10$: el
        \omterm{def:g12:light-as-wave:fringes}{interfringe} $i$, con el error de la regla reducido diez veces.
  \item Luego $\lambda = i\,a/D$.
\end{enumerate}
\end{method}

\begin{remark}[¿Por qué dos lamps nunca interfieren?]\label{rem:g12:light-as-wave:lamps}
Dos lamps de escritorio que iluminan una pared no producen franjas: cada una emite
Trenes \omterm{def:g10:signals-and-waves:wave}{ola} cortos y no correlacionados cuyo tiempo relativo salta aleatoriamente.
mil millones de veces por segundo, por lo que las franjas se desplazan con la misma rapidez y la
el ojo los promedia en una luz uniforme. Interferencia exige \omterm{def:g12:light-as-wave:coherent}{coherencia}
--- un \omterm{def:g10:signals-and-waves:wave}{ola} dividido en dos, como lo hacen las rendijas de Young; el \omterm{def:g11:color-light-sources:lamps}{láser} es el
atajo moderno, un \omterm{def:g10:signals-and-waves:wave}{ola} largo y limpio.
\end{remark}

\section{Colores pintados por interferencia}

\begin{definition}[iridiscencia]\label{def:g12:light-as-wave:iridescence}
Colores que cambian con el ángulo de visión: pompas de jabón, manchas de petróleo,
plumas de pavo real, la parte posterior de un CD --- son \emph{iridiscencia}%
\index{iridiscencia}: sin pigmento, pero con interferencia en \omterm{def:g10:light-spectra:white}{luz blanca},
reforzando algunos \omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} y cancelando otros.
\end{definition}

\begin{example}[Películas de jabón y manchas de aceite.]\label{ex:g12:light-as-wave:soap}
La luz que incide sobre una película de jabón se refleja en parte en su cara frontal y en parte en
de su parte trasera: dos copias \omterm{def:g12:light-as-wave:coherent}{coherente} cuyo \omterm{def:g12:light-as-wave:coherent}{diferencia de camino} está configurado por el
espesor y el ángulo de visión. Cada espesor devuelve algo
\omterm{def:g12:mechanical-waves:wavelength}{longitudes de onda} reforzado, otros anulados: la película refleja un color
que se desliza al escurrir y \omterm{def:g11:lenses-and-eye:lens}{adelgaza}. Una mancha de petróleo sobre asfalto mojado juega
el mismo truco.
\end{example}

\begin{remark}[Luz monocromática versus luz blanca]\label{rem:g12:light-as-wave:white}
En \omterm{def:g10:light-spectra:white}{luz blanca} cada \omterm{def:g12:mechanical-waves:wavelength}{longitud de onda} dibuja sus propios flecos Jóvenes con sus
propio $i = \lambda D/a$: en el centro todos los colores son brillantes (uno blanco
franja); unos cuantos flecos los patrones se separan --- iridiscente
bordes y luego difuminados. \omterm{def:g10:light-spectra:white}{luz monocromática} da flecos nítidos sin
final: de ahí el \omterm{def:g11:color-light-sources:lamps}{láser} en cada medición nítida aquí.
\end{remark}
