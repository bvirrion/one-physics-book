\chapter{Nuclear Energy: Fission, Fusion, $E=mc^2$}\label{ch:g12:nuclear-energy}

A fuel pellet the size of a pencil eraser covers a household's electricity
for a year; the same service from coal takes three tonnes. Overhead, the Sun
pays for daylight by vanishing --- four million tonnes of itself a second.
This chapter opens the account book behind both bargains: mass itself is
energy, and nuclei (\cref{ch:g11:nucleus-radioactivity}) are where it is cashed.

\section{Mass is energy}

\begin{theorem}[Mass--energy equivalence]\label{thm:g12:nuclear-energy:emc2}\index{mass--energy equivalence}
A body of mass $m$, merely by existing, holds the energy
\[
E = m c^2, \qquad c = \qty{3.00e8}{m/s}.
\]
Whenever a system loses mass $\Delta m$, it releases the energy
$\Delta m\, c^2$ --- whatever the mechanism.
\end{theorem}
\admitted

\begin{remark}[Einstein, 1905]\label{rem:g12:nuclear-energy:einstein}
This is Einstein's relation, admitted here; where it comes from is the business
of the special-relativity chapter that closes this year
(\cref{ch:g12:special-relativity}). Note the rate: $c^2 = \qty{9.00e16}{J/kg}$
--- a rounding error of mass is a fortune of energy.
\end{remark}

\begin{definition}[Electron-volt and atomic mass unit]\label{def:g12:nuclear-energy:units}
Two units tailored to the nucleus. The
\emph{electron-volt}\index{electron-volt}, the energy an electron gains
crossing \qty{1}{V} (\cref{ch:g11:circuits-and-power}):
$\qty{1}{eV} = \qty{1.60e-19}{J}$, $\qty{1}{MeV} = \qty{1.60e-13}{J}$ ---
chemical bonds trade a few \unit{eV}, nuclear reactions megaelectron-volts.
The \emph{unified atomic mass unit}\index{unified atomic mass unit}, one
twelfth of the mass of a carbon-12 atom: $\qty{1}{u} = \qty{1.66054e-27}{kg}$,
of energy equivalent \qty{931.5}{MeV} --- $\qty{1}{u} = \qty{931.5}{MeV}/c^2$.
\end{definition}

\begin{example}[The sleeping fortune]\label{ex:g12:nuclear-energy:fortune}
One kilogram of anything, fully converted, is $\qty{9.00e16}{J}$ --- three
years of output of a large (\qty{1}{GW}) power station. Nobody noticed before
1905 because ordinary physics barely touches mass: burning a kilogram of coal
(\qty{3.0e7}{J}) lightens it by a third of a microgram. Nuclei, we shall see,
move one part in a thousand: still small, but a million times chemistry.
\end{example}

\section{The mass defect}

\begin{definition}[Mass defect and binding energy]\label{def:g12:nuclear-energy:defect}
Weigh a nucleus ${}^{A}_{Z}\mathrm{X}$ (mass $m$), then weigh its $Z$ protons
and $A - Z$ neutrons separately: the parts are \emph{heavier} than the whole.
The difference
\[
\Delta m = Z\,m_p + (A - Z)\,m_n - m > 0,
\qquad m_p = \qty{1.00728}{u},\; m_n = \qty{1.00866}{u},
\]
is the \emph{mass defect}\index{mass defect}; the \emph{binding
energy}\index{binding energy} $E_b = \Delta m\,c^2$ is the energy needed to
pull the nucleus apart into free nucleons --- equally, the energy released
when it is assembled. A bound nucleus sits \emph{below} its parts.
\end{definition}

\begin{example}[Helium-4]\label{ex:g12:nuclear-energy:helium}
The ${}^{4}_{2}\mathrm{He}$ nucleus has $m = \qty{4.00151}{u}$; its parts:
$2 \times 1.00728 + 2 \times 1.00866 = \qty{4.03188}{u}$. So
$\Delta m = \qty{0.03037}{u}$ and $E_b = 0.03037 \times 931.5 \approx
\qty{28.3}{MeV}$: the nucleus weighs $0.75\%$ less than its parts --- the
``one part in a thousand'' of \cref{ex:g12:nuclear-energy:fortune}.
\end{example}

\begin{method}[Energy balance of a nuclear reaction]\label{met:g12:nuclear-energy:balance}
\begin{enumerate}
  \item Balance the equation: total $A$ and $Z$ conserved (\cref{ch:g11:nucleus-radioactivity}).
  \item Add the masses before, then after, in \unit{u} --- keep all five decimals.
  \item $\Delta m = m_{\text{before}} - m_{\text{after}}$, then
        $E = \Delta m\,(\text{in u}) \times \qty{931.5}{MeV}$, carried off as
        kinetic energy of the products; negative means the reaction must be paid.
\end{enumerate}
\end{method}

\section{The curve that runs the universe}

\begin{definition}[Binding energy per nucleon]\label{def:g12:nuclear-energy:pernucleon}
Cohesion is compared fairly by the \emph{binding energy per
nucleon}\index{binding energy per nucleon}, $E_b/A$, the average cost of
extracting one nucleon: $28.3/4 \approx \qty{7.1}{MeV}$ for helium-4.
\end{definition}

\begin{proposition}[The iron peak, and two roads down]\label{prop:g12:nuclear-energy:ironpeak}\index{iron peak}
Plotted against $A$, the binding energy per nucleon climbs steeply through
the light nuclei, peaks near iron, ${}^{56}_{26}\mathrm{Fe}$, at about
\qty{8.8}{MeV} per nucleon, then slides gently to about \qty{7.6}{MeV} at
uranium: iron is the most tightly bound nucleus in nature. A reaction
releases energy exactly when its products sit \emph{higher} on the curve; two
opposite strategies therefore both pay, and both walk toward iron ---
\emph{splitting} the heaviest nuclei, and \emph{merging} the lightest.
\end{proposition}
\admitted

\begin{remark}[Why a peak]\label{rem:g12:nuclear-energy:whypeak}
The tug-of-war of \cref{ch:g11:fundamental-interactions}: strong glue binds
only touching neighbours, Coulomb repulsion spans the whole nucleus --- small
nuclei are mostly surface, big ones mostly repulsion, iron the compromise.
The honest accounting lives in the university volumes.
\end{remark}

\section{Fission and the chain reaction}

\begin{definition}[Fission]\label{def:g12:nuclear-energy:fission}
\emph{Fission}\index{fission} is the splitting of a heavy nucleus into two
mid-sized fragments. Uranium-235 is \emph{fissile}\index{fissile}: a slow
neutron, absorbed, leaves the nucleus so agitated that it tears in two,
spitting out two or three fresh neutrons ---
${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to \text{two fragments} +
2\text{--}3\;{}^{1}_{0}\mathrm{n}$, the pair varying from fission to fission.
\end{definition}

\begin{example}[One fission, weighed]\label{ex:g12:nuclear-energy:fissionbalance}
One frequent route, by \cref{met:g12:nuclear-energy:balance}:
\[
{}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \longrightarrow
{}^{94}_{38}\mathrm{Sr} + {}^{140}_{54}\mathrm{Xe} + 2\,{}^{1}_{0}\mathrm{n}.
\]
Before: $1.00866 + 234.99350 = \qty{236.00216}{u}$. After:
$93.89450 + 139.89200 + 2 \times 1.00866 = \qty{235.80382}{u}$. Hence
$\Delta m = \qty{0.19834}{u}$ and $E \approx \qty{185}{MeV}$; counting the
later decays of the radioactive fragments (\cref{ch:g12:radioactive-decay}),
each fission is worth about \qty{200}{MeV} --- a few \unit{eV} buys one atom
of coal.
\end{example}

\begin{definition}[Chain reaction and critical mass]\label{def:g12:nuclear-energy:chain}
Each fission is lit by one neutron and frees two or three: in a large enough
lump they strike other nuclei and the reaction feeds itself --- a
\emph{chain reaction}\index{chain reaction}. In a small lump most neutrons
escape through the surface first; the minimum quantity sustaining the chain
is the \emph{critical mass}\index{critical mass} --- below it the chain
fizzles, above it every generation multiplies: the principle of the bomb.
\end{definition}

\begin{definition}[The reactor, tamed]\label{def:g12:nuclear-energy:reactor}
A power reactor holds the chain at exactly one neutron per fission. The fuel:
uranium pellets modestly enriched in uranium-235. The
\emph{moderator}\index{moderator} (ordinary water, usually) slows the fast
fission neutrons by collisions --- slow neutrons are far better at triggering
the next fission; the \emph{control rods}\index{control rod} (boron, cadmium)
devour neutrons --- pushed in, the chain dies; drawn out, it quickens. The
fragments' agitation becomes heat, then steam, then electricity
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[A peppercorn against a truck]\label{ex:g12:nuclear-energy:gram}
One gram of uranium-235 holds
$N = \num{1.0e-3} / (235 \times \num{1.66054e-27}) \approx \num{2.56e21}$
nuclei; at $\qty{200}{MeV} = \qty{3.2e-11}{J}$ each, fissioning them all
yields about \qty{8.2e10}{J} --- the chemical energy of $\num{8.2e10} /
\num{3.0e7} \approx \qty{2.7e3}{kg}$ of coal: a peppercorn against three tonnes.
\end{example}

\section{Fusion: the Sun's own fire}

\begin{definition}[Fusion]\label{def:g12:nuclear-energy:fusion}
\emph{Fusion}\index{fusion} is the merging of two light nuclei into a heavier,
more tightly bound one. The most accessible reaction weds deuterium and
tritium, the heavy isotopes of hydrogen (\cref{ch:g11:nucleus-radioactivity}):
\[
{}^{2}_{1}\mathrm{H} + {}^{3}_{1}\mathrm{H} \longrightarrow
{}^{4}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}.
\]
\end{definition}

\begin{example}[D--T, weighed]\label{ex:g12:nuclear-energy:dt}
Before: $2.01355 + 3.01550 = \qty{5.02905}{u}$; after:
$4.00151 + 1.00866 = \qty{5.01017}{u}$; so $\Delta m = \qty{0.01888}{u}$ and
$E \approx \qty{17.6}{MeV}$, four fifths of it on the neutron --- less than a
fission, but from five nucleons instead of $236$: about \qty{3.4e14}{J} per
kilogram of fuel, four times fission. Helium-4
(\cref{ex:g12:nuclear-energy:helium}) is once again the ash.
\end{example}

\begin{definition}[The Coulomb barrier]\label{def:g12:nuclear-energy:barrier}
Two nuclei are both positive: to touch, they must first climb the hill of
their electric repulsion (\cref{ch:g11:electric-gravitational-fields}) --- the
\emph{Coulomb barrier}\index{Coulomb barrier}. Only near \qty{1e8}{K} do
collisions carry the energy to cross it; the fuel is then a
\emph{plasma}\index{plasma} of bare nuclei and electrons. Fission pays no
such toll: its trigger, the neutron, is neutral, and works cold.
\end{definition}

\begin{example}[The Sun runs on it]\label{ex:g12:nuclear-energy:sun}
The Sun's core, at $\qty{1.5e7}{K}$, fuses ordinary hydrogen step by step ---
the \emph{proton--proton chain}, of net effect
$4\,{}^{1}_{1}\mathrm{H} \to {}^{4}_{2}\mathrm{He} + 2\,{}^{0}_{+1}\mathrm{e}
+ \text{radiation}$, about \qty{26}{MeV} per helium: $0.7\%$ of the
hydrogen's mass leaves as energy. The sunlight whose spectrum we read in
\cref{ch:g10:light-spectra} is this mass, arriving eight minutes late; every
star shines by walking the curve toward iron.
\end{example}

\begin{remark}[ITER: bottling a star]\label{rem:g12:nuclear-energy:iter}
No solid can hold a plasma at \qty{1.5e8}{K}; a \emph{tokamak} cages it in a
magnetic field (\cref{ch:g11:magnetic-fields}), a doughnut of field lines the
charged particles spiral along without touching the wall; ITER, built to
release ten times its heating power, is the current step. Break the
confinement and the plasma cools and stops in seconds: no chain reaction, no
critical mass --- the hard part is not stopping the fire but keeping it lit.
\end{remark}

\begin{example}[The energy ladder, per kilogram]\label{ex:g12:nuclear-energy:ladder}
Energy per kilogram of fuel: coal, \qty{3.0e7}{J}; uranium-235 by fission,
\qty{8.2e13}{J}; deuterium--tritium by fusion, \qty{3.4e14}{J}; total
conversion ($E = mc^2$), \qty{9.0e16}{J}. Six orders of magnitude separate
chemistry from the nucleus --- the reason a reactor is refuelled by the truck
and a coal plant by the trainload.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:1}
Convert: (a) \qty{1}{eV} and \qty{1}{MeV} into joules; (b) the \qty{200}{MeV} of
one fission into joules; (c) check from $\qty{1}{u} = \qty{1.66054e-27}{kg}$
that its energy equivalent is close to \qty{931.5}{MeV}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:2}
A sugar cube has mass \qty{6.0}{g}: compute its full $E = mc^2$ value. A
household draws $\qty{10}{kW.h} = \qty{3.6e7}{J}$ a day --- for how many
years could the cube, fully converted, power it?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:3}
The carbon-12 nucleus has mass \qty{11.99671}{u}: compute its mass defect,
its binding energy in \unit{MeV}, and its binding energy per nucleon
($m_p = \qty{1.00728}{u}$, $m_n = \qty{1.00866}{u}$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:4}
Complete each fission, giving $A$, $Z$ and the element ($Z = 36$: krypton,
$Z = 52$: tellurium):
(a) ${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to
{}^{144}_{56}\mathrm{Ba} + {}^{A}_{Z}\mathrm{X} + 3\,{}^{1}_{0}\mathrm{n}$;
(b) ${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to
{}^{97}_{40}\mathrm{Zr} + {}^{A}_{Z}\mathrm{Y} + 2\,{}^{1}_{0}\mathrm{n}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:5}
Read the curve of \cref{prop:g12:nuclear-energy:ironpeak}: estimate $E_b/A$ for
helium-4, carbon-12, iron-56 and uranium-235. Which is most tightly bound?
Which could release energy by fission, and which by fusion?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:nuclear-energy:6}
The deuteron ${}^{2}_{1}\mathrm{H}$ has mass \qty{2.01355}{u}: compute its
binding energy in \unit{MeV}. A chemical bond holds a few \unit{eV} --- by
what factor does the nuclear ``bond'' beat it?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:nuclear-energy:7}
Weigh the fission ${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to
{}^{144}_{56}\mathrm{Ba} + {}^{89}_{36}\mathrm{Kr} + 3\,{}^{1}_{0}\mathrm{n}$:
masses \qty{234.99350}{u} (U), \qty{143.89220}{u} (Ba), \qty{88.89790}{u}
(Kr), \qty{1.00866}{u} (n). Energy released, in \unit{MeV} and in joules?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:nuclear-energy:8}
Another fusion: ${}^{2}_{1}\mathrm{H} + {}^{2}_{1}\mathrm{H} \to
{}^{3}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}$, with
$m({}^{3}_{2}\mathrm{He}) = \qty{3.01493}{u}$. Compute the energy released.
Why so much less than D--T? (Consider where each product sits on the curve.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:nuclear-energy:9}
Explain, a few lines each: (a) why fission needs no heating while fusion
demands about \qty{1e8}{K}; (b) how the Sun manages at ``only'' \qty{1.5e7}{K}
(the crowd's size, its fastest members); (c) why a fusion plant cannot run
away like a chain reaction.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:nuclear-energy:10}
The Sun's net reaction turns four protons ($m_p = \qty{1.00728}{u}$) into one
helium-4 nucleus (\qty{4.00151}{u}): compute the energy per helium made, in
\unit{MeV}, and the fraction of the initial mass converted.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:nuclear-energy:11}
In a water-moderated reactor: (a) what does the moderator do, and why does
the chain need it? (b) the control rods? (c) the cooling water \emph{is} the
moderator: why does losing it tend to choke the chain rather than speed it up?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:nuclear-energy:12}
An uncontrolled chain doubles at each generation. From one fission, about how
many generations until the $\num{2.56e21}$ nuclei of one gram of uranium-235
have split ($2^{10} \approx 10^{3}$)? At roughly \qty{10}{ns} per generation,
how long is that --- and why, then, is criticality guarded so carefully?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:nuclear-energy:13}
Could we mine energy by fissioning iron? Weigh
${}^{56}_{26}\mathrm{Fe} \to 2\,{}^{28}_{14}\mathrm{Si}$, with
$m(\mathrm{Fe}) = \qty{55.92066}{u}$, $m(\mathrm{Si}) = \qty{27.96924}{u}$.
Compute $\Delta m$ and conclude, in terms of the curve, why iron is nuclear
ash: fuel for nothing.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:nuclear-energy:14}
From $m({}^{56}_{26}\mathrm{Fe}) = \qty{55.92066}{u}$ and
$m({}^{235}_{92}\mathrm{U}) = \qty{234.99350}{u}$, compute $E_b/A$ for both
($m_p = \qty{1.00728}{u}$, $m_n = \qty{1.00866}{u}$). Fission fragments sit
near \qty{8.5}{MeV} per nucleon: \emph{estimate} from the two levels the energy
of one fission, and compare with \cref{ex:g12:nuclear-energy:fissionbalance}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:nuclear-energy:15}
ITER aims at \qty{500}{MW} of fusion power. At \qty{17.6}{MeV} per D--T
reaction (\qty{5.029}{u} of fuel consumed), find the reactions per second,
then the fuel burned per day. A \qty{500}{MW} coal furnace burns about
\qty{1.4e6}{kg} a day: compare.
\end{exercise}

\section{Problem: Powering a City}

\begin{problem}[{Weekend problem --- powering a city: one steady gigawatt for
half a million people, bought four ways --- uranium pellets, trainloads of
coal, a pool of seawater, the Sun's own substance --- with the same kilogram
of vanished mass hiding under all four}]
\label{pb:g12:nuclear-energy:1}
A city of $500\,000$ inhabitants draws a steady electric power
$P = \qty{1.0}{GW}$; its plants, whatever the fuel, turn heat into
electricity with efficiency $33\%$ (\cref{ch:g10:energy-conservation}).
Data: one uranium-235 fission, \qty{200}{MeV} in all; coal, \qty{3.0e7}{J/kg};
one D--T fusion, \qty{17.6}{MeV} from \qty{5.029}{u} of fuel;
$\qty{1}{u} = \qty{1.66054e-27}{kg}$; $\qty{1}{eV} = \qty{1.60e-19}{J}$; one
year $= \qty{3.16e7}{s}$; the Sun: \qty{3.9e26}{W}, \qty{2.0e30}{kg}; a cubic
metre of seawater holds about \qty{33}{g} of deuterium.

\medskip
\noindent\textbf{Part I --- The fission plant.}
\begin{enumerate}
  \item What thermal power must the reactor deliver?
  \item Express the energy of one fission in joules.
  \item How many fissions per second run the city?
  \item The mass of one uranium-235 nucleus ($A = 235$), then the uranium-235
        consumed per second?
  \item Deduce the uranium-235 consumed in one year, in kilograms.
  \item The fuel is enriched to $4.0\%$ uranium-235: what mass does a year
        take, and does it fit on one truck?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The coal ledger.}
\begin{enumerate}[resume]
  \item Compute the thermal energy the city consumes in one year.
  \item What mass of coal delivers it?
  \item Compute the ratio (coal mass)/(uranium-235 mass) for one year, and
        check it against the ladder of \cref{ex:g12:nuclear-energy:ladder}.
  \item A freight train hauls \qty{3.0e6}{kg}: how many trains per year, and
        per day? One sentence: picture the two supply lines.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The fusion dream.}
\begin{enumerate}[resume]
  \item Express the energy of one D--T fusion in joules.
  \item Compute the energy released per kilogram of D--T fuel; compare with
        the ladder.
  \item What mass of D--T fuel covers the city's year? What mass of that is
        deuterium ($2.014$ of the \qty{5.029}{u})?
  \item What volume of seawater contains that deuterium? Compare with an
        Olympic pool, about \qty{2.5e3}{m^3}. (Tritium is bred from lithium.)
  \item Why does this plant carry no critical mass and no chain reaction to
        fear? What, then, is the hard part?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The star that pays in mass.}
\begin{enumerate}[resume]
  \item From $E = mc^2$, what mass does the Sun convert each second?
  \item What mass per second, and per year, must \emph{any}
        \qty{3.0}{GW}-thermal source convert? Check that all three fuels
        surrender this same mass.
  \item What total mass has the Sun radiated away in its $\num{4.5e9}$
        years, and what fraction of the Sun is that?
  \item Fusion can tap about $0.07\%$ of the Sun's mass (the burnable share of
        its core): estimate its total lifetime as a star, and what is left.
  \item Close the audit in three sentences: one year of the city in tonnes
        of coal, kilograms of uranium-235 and kilograms of D--T fuel --- and
        the kilogram of mass that, in every scenario, became the city's light.
\end{enumerate}
\end{problem}
