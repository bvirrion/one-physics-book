\chapter{The Quantum World: Photons and Energy Levels}\label{ch:g12:quantum-world}

An \omterm{def:g10:light-spectra:wavelength}{infrared} heat lamp can bathe your skin all evening and never mark it; ten
minutes of \omterm{def:g11:lenses-and-eye:lens}{thin} mountain sunlight leaves a burn. The lamp delivers far more
\omterm{def:g10:energy-conservation:energy}{energy} --- but \omterm{def:g10:energy-conservation:energy}{energy} is not what marks skin. Light, this chapter shows,
arrives in grains, and only a grain that hits hard enough does chemistry:
one \omterm{def:g10:light-spectra:wavelength}{ultraviolet} grain snaps a bond that a billion \omterm{def:g10:light-spectra:wavelength}{infrared} grains only warm.
Follow the grains and a century opens: metals leaking electrons, \omterm{def:g11:fundamental-interactions:ladder}{atoms}
singing in \omterm{def:g10:light-spectra:emission}{spectral lines}, electrons interfering like \omterm{def:g10:signals-and-waves:wave}{waves}.

\section{Light arrives in grains: the photon}

\begin{definition}[The photon]\label{def:g12:quantum-world:photon}
Light of \omterm{def:g12:oscillators-and-time:oscillator}{frequency} $\nu$ exchanges \omterm{def:g10:energy-conservation:energy}{energy} with matter only in whole grains
called \emph{photons}\index{photon}, each of \omterm{def:g10:energy-conservation:energy}{energy}
\[
E = h\nu = \frac{hc}{\lambda}, \qquad h = \qty{6.63e-34}{J.s},
\]
where $h$ is \emph{Planck's constant}\index{Planck's constant} (Planck,
1900; Einstein, 1905) and $\lambda$ the \omterm{def:g12:light-as-wave:lightwave}{wavelength in vacuum}. A beam's
intensity measures the \emph{number} of photons per second; each grain's
\omterm{def:g10:energy-conservation:energy}{energy} is fixed by the \omterm{def:g12:oscillators-and-time:oscillator}{frequency} alone.
\end{definition}

\begin{notation}[The electronvolt]\label{not:g12:quantum-world:ev}
Atomic physics uses the \emph{electronvolt}\index{electronvolt},
$\qty{1}{eV} = \qty{1.60e-19}{J}$: the \omterm{def:g10:energy-conservation:energy}{energy} one \omterm{def:g11:fundamental-interactions:elementary}{elementary charge} \omterm{def:g10:signals-and-waves:oscilloscope}{gains}
crossing \qty{1}{V} (\cref{ch:g11:electric-gravitational-fields}). Handy:
$E \,(\unit{eV}) \approx 1240/\lambda\,(\unit{nm})$.
\end{notation}

\section{The photoelectric effect}

Shine light on a clean metal plate in vacuum and electrons pop out --- the
\emph{photoelectric effect}\index{photoelectric effect}. Its rules broke classical physics.

\begin{proposition}[What the experiment shows]\label{prop:g12:quantum-world:facts}
For each metal there is a \emph{threshold frequency}\index{threshold frequency} $\nu_0$:
\begin{enumerate}
  \item below $\nu_0$, no electron leaves, however intense the light;
  \item above $\nu_0$, emission starts without \omterm{def:g12:mechanical-waves:delay}{delay}, however faint;
  \item intensity raises the \emph{number} of electrons per second, not
        their maximum \omterm{def:g11:mechanical-energy:kinetic}{kinetic energy}; \omterm{def:g12:oscillators-and-time:oscillator}{frequency} raises $E_{k,\max}$.
\end{enumerate}
None of it fits a continuous \omterm{def:g10:signals-and-waves:wave}{wave}, which any color should let charge up ---
over months in faint light (\cref{exo:g12:quantum-world:14}) --- and whose
intensity should set the electrons' \omterm{def:g10:energy-conservation:energy}{energy}: all three predictions fail.
\end{proposition}
\admitted

\begin{theorem}[Einstein's photoelectric relation]\label{thm:g12:quantum-world:einstein}
\index{photoelectric relation}An electron needs at least the \emph{work
function}\index{work function} $W_0$ of the metal to escape its surface.
One \omterm{def:g12:quantum-world:photon}{photon} is absorbed whole by one electron, so
\[
h\nu = W_0 + E_{k,\max}, \qquad \text{hence } \nu_0 = \frac{W_0}{h}.
\]
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energy} bookkeeping: $h\nu$ in, $W_0$ spent on escape, the rest kinetic ---
at most $h\nu - W_0$, less for electrons starting deeper. Below $\nu_0$
one grain cannot pay the exit toll, whatever their number.
\end{proof}

\begin{definition}[Stopping potential]\label{def:g12:quantum-world:stopping}
To measure $E_{k,\max}$, a reverse \omterm{def:g11:circuits-and-power:voltage}{voltage} brakes the electrons
(\cref{ch:g11:electric-gravitational-fields}); the \emph{stopping
potential}\index{stopping potential} $U_s$ is the smallest \omterm{def:g11:circuits-and-power:voltage}{voltage} canceling
the current: the fastest electrons just fail to cross, so $E_{k,\max} = e\,U_s$.
\end{definition}

\begin{method}[Measuring Planck's constant]\label{met:g12:quantum-world:measure-h}
\begin{enumerate}
  \item Illuminate a photocell through \omterm{def:g11:color-light-sources:subtractive}{filters} of known \omterm{def:g12:mechanical-waves:wavelength}{wavelengths};
        compute each $\nu = c/\lambda$; record each \omterm{def:g12:quantum-world:stopping}{stopping potential}.
  \item Plot $E_{k,\max} = eU_s$ against $\nu$: Einstein promises the
        straight line $h\nu - W_0$.
  \item Read the slope: it \emph{is} $h$; the intercepts give $\nu_0$ and $-W_0$.
\end{enumerate}
\end{method}

\section{Energy levels: why atoms emit lines}

\begin{definition}[Energy levels]\label{def:g12:quantum-world:levels}
The \omterm{def:g10:energy-conservation:energy}{energy} of an electron bound in an \omterm{def:g11:fundamental-interactions:ladder}{atom} is
\emph{quantized}\index{quantization}: restricted to a discrete ladder of
\emph{energy levels}\index{energy level} $E_1 < E_2 < \dots < 0$, counted
negative from the free electron. The lowest rung is the \emph{ground
state}\index{ground state}, the others \emph{excited
states}\index{excited state}; reaching $0$ is
\emph{ionization}\index{ionization}.
\end{definition}

\begin{proposition}[The hydrogen ladder]\label{prop:g12:quantum-world:hydrogen}
The \omterm{def:g12:quantum-world:levels}{energy levels} of the hydrogen \omterm{def:g11:fundamental-interactions:ladder}{atom} are
\[
E_n = -\frac{\qty{13.6}{eV}}{n^2}, \qquad n = 1, 2, 3, \dots
\]
so $E_1 = \qty{-13.6}{eV}$, $E_2 = \qty{-3.40}{eV}$, $E_3 =
\qty{-1.51}{eV}$, $E_4 = \qty{-0.85}{eV}$, crowding toward $0$.
\end{proposition}
\admitted

\begin{remark}[Where the ladder comes from]\label{rem:g12:quantum-world:ladder-origin}
Why these numbers is quantum mechanics, kept for the university volumes ---
as is de Broglie's formula: the electron's \omterm{def:g12:quantum-world:debroglie}{matter wave}
(\cref{def:g12:quantum-world:debroglie} below) must close on itself around
the \omterm{def:g11:fundamental-interactions:ladder}{nucleus}, and only certain \omterm{def:g12:mechanical-waves:wavelength}{wavelengths} fit --- like a guitar string, an
\omterm{def:g11:fundamental-interactions:ladder}{atom} has a discrete set of notes.
\end{remark}

\begin{theorem}[Photon of a transition]\label{thm:g12:quantum-world:transition}
An \omterm{def:g11:fundamental-interactions:ladder}{atom} changes level only by emitting or absorbing one \omterm{def:g12:quantum-world:photon}{photon} carrying
exactly the difference: dropping from level $p$ to a lower level $n$ emits
$h\nu = E_p - E_n$; climbing from $n$ to $p$ absorbs that same \omterm{def:g10:energy-conservation:energy}{energy}.
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energy} conservation, one grain at a time: the \omterm{def:g12:quantum-world:photon}{photon} is the only currency
light accepts, and the \omterm{def:g11:fundamental-interactions:ladder}{atom}'s books must balance.
\end{proof}

\begin{remark}[Line spectra explained]\label{rem:g12:quantum-world:lines}
Here is the answer promised by \cref{ch:g10:light-spectra}: a hot gas emits
only the frequencies $(E_p - E_n)/h$ of its own ladder --- a barcode of
bright lines --- and a cool gas steals exactly those frequencies from \omterm{def:g10:light-spectra:white}{white
light}, printing matching dark absorption lines. Each element has its own
ladder, hence its own barcode --- how we read starlight.
\end{remark}

\begin{example}[The visible lines of hydrogen]\label{ex:g12:quantum-world:balmer}
Transitions ending on $n = 2$ fall in the visible. With $E \,(\unit{eV}) =
1240/\lambda\,(\unit{nm})$: $3 \to 2$: \qty{1.89}{eV}, \qty{656}{nm} (red);
$4 \to 2$: \qty{2.55}{eV}, \qty{486}{nm} (blue-green); $5 \to 2$:
\qty{2.86}{eV}, \qty{434}{nm} (violet). Transitions to $n = 1$ exceed
\qty{10}{eV}: all \omterm{def:g10:light-spectra:wavelength}{ultraviolet}.
\end{example}

\begin{remark}[Lasers, in one breath]\label{rem:g12:quantum-world:laser}
A \omterm{def:g12:quantum-world:photon}{photon} of exactly $E_p - E_n$ passing an \emph{excited} \omterm{def:g11:fundamental-interactions:ladder}{atom} can trigger it
to emit an identical \omterm{def:g12:quantum-world:photon}{photon} --- same \omterm{def:g10:energy-conservation:energy}{energy}, same direction, in step. Keep
more \omterm{def:g11:fundamental-interactions:ladder}{atoms} up than down and one \omterm{def:g12:quantum-world:photon}{photon} becomes an avalanche of clones:
\emph{stimulated emission}\index{stimulated emission} --- and a \omterm{def:g11:color-light-sources:lamps}{laser} is little more.
\end{remark}

\section{Matter waves}

\begin{proposition}[Electrons interfere]\label{prop:g12:quantum-world:interference}
Send electrons one at a time through a double slit
(\cref{ch:g12:light-as-wave}): each lands as a single sharp dot --- a
particle. But as thousands accumulate, the dots draw \omterm{def:g12:light-as-wave:fringes}{interference fringes}:
each electron, alone, passes as a \omterm{def:g10:signals-and-waves:wave}{wave} through \emph{both} slits. Crystals
diffract electron beams too (Davisson and Germer, 1927).
\end{proposition}
\admitted

\begin{definition}[De Broglie wavelength]\label{def:g12:quantum-world:debroglie}
Every particle of \omterm{def:g12:newtons-laws:momentum}{momentum} $p = mv$ travels as a \emph{matter
wave}\index{matter wave} of \emph{de Broglie wavelength}\index{de Broglie
wavelength}
\[
\lambda = \frac{h}{p} = \frac{h}{mv}.
\]
\end{definition}
\admitted

\begin{example}[Why you never diffract]\label{ex:g12:quantum-world:never}
An electron accelerated through \qty{100}{V} has $E_k = \qty{1.60e-17}{J}$,
$p = \sqrt{2m_e E_k} = \qty{5.4e-24}{kg.m/s}$, so $\lambda \approx
\qty{0.12}{nm}$ --- the spacing of \omterm{def:g11:fundamental-interactions:ladder}{atoms} in a crystal, which therefore acts
as its double slit. A \qty{58}{g} tennis ball at \qty{25}{m/s}: $\lambda =
6.63\times10^{-34}/1.45 \approx \qty{4.6e-34}{m}$, some $10^{19}$ times
smaller than a \omterm{def:g11:fundamental-interactions:ladder}{nucleus} --- no slit will ever show it fringing. $h$ is so
small that quantum graininess surfaces only at the \omterm{def:g11:fundamental-interactions:ladder}{atom}'s scale.
\end{example}

\begin{example}[The electron microscope]\label{ex:g12:quantum-world:microscope}
\omterm{def:g12:light-as-wave:diffraction}{Diffraction} forbids resolving detail much smaller than the \omterm{def:g12:mechanical-waves:wavelength}{wavelength} used
(\cref{ch:g12:light-as-wave}): visible light stops near \qty{0.5}{\micro m}.
Electrons at \qty{10}{kV} have $\lambda \approx \qty{12}{pm}$ --- tens of
thousands of times shorter: electron microscopes image viruses, cell
machinery, even single \omterm{def:g11:fundamental-interactions:ladder}{atoms}. The \omterm{def:g12:quantum-world:debroglie}{matter wave} handed physics a new eye.
\end{example}

\section{Problem: The Solar-Panel Physics Lab}

\begin{problem}[{Weekend problem --- the solar-panel physics lab: the
$10^{21}$ grains a rooftop drinks each second, a photocell that measures
\omterm{def:g12:quantum-world:photon}{Planck's constant}, a hydrogen lamp read line by line --- one weekend, one $h$}]
\label{pb:g12:quantum-world:1}
Your class instruments the school's rooftop solar panel for a weekend.
Data: solar irradiance at noon \qty{1.0e3}{W/m^2}; mean solar \omterm{def:g12:mechanical-waves:wavelength}{wavelength}
\qty{550}{nm}; panel area \qty{1.0}{m^2}, \omterm{def:g10:energy-conservation:efficiency}{efficiency} $20\%$; the day
delivers the equivalent of \qty{5.0}{h} of full sun.

\medskip
\noindent\textbf{Part I --- \omterm{def:g12:quantum-world:photon}{Photon} bookkeeping.}
\begin{enumerate}
  \item Compute the \omterm{def:g10:energy-conservation:energy}{energy} of one mean solar \omterm{def:g12:quantum-world:photon}{photon}, in \omterm{def:g11:work-of-force:work}{joules} and electronvolts.
  \item Deduce the number of \omterm{def:g12:quantum-world:photon}{photons} striking the panel per second at noon.
  \item Compute the panel's \omterm{prop:g11:circuits-and-power:power}{electrical power} at noon, then the \omterm{def:g10:energy-conservation:energy}{energy}
        produced over the day, in \unit{kW.h}.
  \item The panel's current looks perfectly smooth. Using question 2,
        explain why the grain-by-grain arrival is invisible.
  \item Sunlight carries more \omterm{def:g10:light-spectra:wavelength}{infrared} \omterm{def:g10:energy-conservation:energy}{energy} than \omterm{def:g10:light-spectra:wavelength}{ultraviolet}, yet only
        \omterm{def:g10:light-spectra:wavelength}{ultraviolet} burns skin (bonds $\sim \qtyrange{3}{4}{eV}$). Explain in grains.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Measuring $h$ with the kit's photocell.}
The kit's photocell has an unknown cathode. \omterm{def:g11:color-light-sources:subtractive}{Filters} give:
\begin{center}\small
\begin{tabular}{lcccc}
$\lambda$ (\unit{nm}) & 365 & 405 & 436 & 546\\
$U_s$ (\unit{V}) & 1.50 & 1.16 & 0.94 & 0.37
\end{tabular}
\end{center}
\begin{enumerate}[resume]
  \item What does the \omterm{def:g12:quantum-world:stopping}{stopping potential} measure? Relate $U_s$, $\nu$, $W_0$, $h$.
  \item Convert the four \omterm{def:g12:mechanical-waves:wavelength}{wavelengths} to frequencies (in $10^{14}\,\unit{Hz}$).
  \item Why should plotting $eU_s$ against $\nu$ give a straight line?
        Check the alignment of the points.
  \item From the extreme points, compute the slope: your measured $h$.
  \item Deduce the cathode's \omterm{thm:g12:quantum-world:einstein}{work function} (\unit{eV}), \omterm{prop:g12:quantum-world:facts}{threshold
        frequency} and threshold \omterm{def:g12:mechanical-waves:wavelength}{wavelength}.
  \item Which part of the sunlight could eject electrons from this
        cathode --- and which half of the \omterm{def:g12:sound-acoustics:timbre}{spectrum} is useless for it?
  \item Compare your $h$ to \qty{6.63e-34}{J.s}: relative error in percent?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hydrogen calibration lamp.}
The kit's \omterm{def:g12:mechanical-waves:wavelength}{wavelength} scale is checked against a hydrogen lamp,
$E_n = -13.6/n^2$ \unit{eV}.
\begin{enumerate}[resume]
  \item Compute $E_2$, $E_3$, $E_4$ and $E_5$.
  \item Compute the \omterm{def:g12:quantum-world:photon}{photon} energies of the transitions $3 \to 2$, $4 \to 2$, $5 \to 2$.
  \item Deduce the three \omterm{def:g12:mechanical-waves:wavelength}{wavelengths} and give each line's color.
  \item Why does the lamp emit \emph{lines} rather than a continuous
        rainbow? One sentence, naming the key idea.
  \item Could the lamp's hydrogen absorb light at \qty{500}{nm}? Justify with the ladder.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Inspecting the cells: the electron \omterm{def:g10:signals-and-waves:wave}{wave}.}
Micro-cracks in a cell lie far below the \qty{0.5}{\micro m} reach of light
microscopes; the lab's scanning electron microscope runs at \qty{10}{kV}.
\begin{enumerate}[resume]
  \item Compute the electrons' speed; check it stays below $0.25\,c$ (classical formula).
  \item Compute their \omterm{def:g12:newtons-laws:momentum}{momentum} and \omterm{def:g12:quantum-world:debroglie}{de Broglie wavelength}.
  \item Resolution scales with \omterm{def:g12:mechanical-waves:wavelength}{wavelength}: compute the \omterm{def:g10:signals-and-waves:oscilloscope}{gain} over green light
        (\qty{550}{nm}); conclude with the weekend's two numbers --- your $h$
        and this \omterm{def:g10:signals-and-waves:oscilloscope}{gain}.
\end{enumerate}
\end{problem}
