\chapter{Oscilações Elétricas Livres: RLC}\label{ch:g12:rlc-oscillations}

Gire o botão pesado de um rádio no sótão e as estações passam pelo
agulha: uma voz, violinos, estática, uma voz novamente. Atrás do painel lá
não é \omterm{def:g11:work-of-force:sign}{motor} e nenhum computador --- um bobina, um \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} cujo intercalado
placas que o mostrador gira e um pedaço de fio aéreo. Este capítulo
conecta os dois do último capítulo energia tanques (\cref{ch:g12:rc-rl-circuits})
cara a cara e encontra o pêndulo mais rápido já construído: carga oscilante
para frente e para trás, até milhões de vezes por \omterm{def:g10:orders-of-magnitude:unit}{segundo}, em um ritmo que o dial
escolhe.

\section{Um pêndulo elétrico: o circuito LC}

Carregue um \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} $C $ para um \omterm{def:g11:circuits-and-power:voltage}{tensão} $ U_0$ --- suas placas então seguram
$\pm Q_0$ com $ Q_0 = C U_0$ --- e em $ t = 0$ mude-o para um
\emph{ideal} bobina $ L $, de zero \omterm{def:g11:circuits-and-power:resistance}{resistência}. Nenhuma fonte em qualquer lugar do
loop: o que quer que aconteça a seguir, o circuito faz sozinho.

\end{circuitikz}

{\small O \omterm{prop:g12:rlc-oscillations:equation}{Circuito LC}: um \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} cobrado para $ Q_0$, um interruptor, um
ideal bobina. Fechando $ K $ permite que a carga se espalhe através do bobina --- e
o bobina, odiando mudanças, não vai deixar isso parar.}
\end{omfigure}

\begin{proposition}[A equação LC]\label{prop:g12:rlc-oscillations:equation}
\index{Circuito LC}
No circuito LC ideal a carga $ q(t)$ da placa superior obedece
\[
L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0,
\qquad\text{i.e.}\qquad
\frac{d^2q}{dt^2} = -\frac{1}{LC}\,q .
\]
\end{proposition}

\begin{proof}
Ao redor do loop o tensões soma a zero
(\cref{prop:g11:circuits-and-power:laws}): $ u_L + u_C = 0$. O
\omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} dá $ u_C = q/C $ e o bobina $ u_L = L\,di/dt $
(\cref{def:g12:rc-rl-circuits:inductor}), com $ i = dq/dt $ contando
a carga chegando na placa superior
(\cref{prop:g12:rc-rl-circuits:cdudt}); substituir.
\end{proof}

\begin{theorem}[Oscilações livres do circuito LC]\label{thm:g12:rlc-oscillations:oscillation}
\index{oscilações elétricas}
Lançado em $ t = 0$ com $ q(0) = Q_0$ e $ i(0) = 0$, o circuito
oscila:
\[
q(t) = Q_0 \cos\!\left(\frac{2\pi t}{T_0}\right),
\qquad
i(t) = -I_0 \sin\!\left(\frac{2\pi t}{T_0}\right),
\qquad
T_0 = 2\pi\sqrt{LC},
\]
com corrente de pico $ I_0 = 2\pi Q_0/T_0 = Q_0/\sqrt{LC}$.
\end{theorem}

\begin{proof}
Verifique por substituição. Diferenciando,
$ i = dq/dt = -\frac{2\pi}{T_0} Q_0 \sin(2\pi t/T_0)$ --- o declarado
$ i(t)$ --- e novamente $\frac{d^2q}{dt^2} = -(\frac{2\pi}{T_0})^2 q $,
que é igual $-q/(LC)$ exatamente quando $(2\pi/T_0)^2 = 1/(LC)$, ou seja\
$ T_0 = 2\pi\sqrt{LC}$. E $ q(0) = Q_0$, $ i(0) = 0$. Isso não
\emph{outro} a função se ajusta à equação e o início é admitido,
exatamente como na primavera
(\cref{rem:g12:oscillators-and-time:uniqueness}).
\end{proof}

\begin{definition}[Período natural e frequência]\label{def:g12:rlc-oscillations:period}
O \emph{período natural}\index{período natural} de um
\omterm{prop:g12:rlc-oscillations:equation}{Circuito LC} é $ T_0 = 2\pi\sqrt{LC}$; isso é
\emph{frequência natural}\index{frequência natural} é
\[
f_0 = \frac{1}{T_0} = \frac{1}{2\pi\sqrt{LC}} .
\]
Verificação dimensional: $\qty{1}{H} = \qty{1}{V.s/A}$ e
$\qty{1}{F} = \qty{1}{C/V}$, então $ LC $ carrega
$\unit{s} \times \unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ é um tempo,
como deve ser.
\end{definition}

\begin{example}[Primeiros números]\label{ex:g12:rlc-oscillations:firstnumbers}
$ L = \qty{10}{mH}$ com $ C = \qty{1.0}{\micro F}$:
$\sqrt{LC} = \qty{1.0e-4}{s}$, então $ T_0 = \qty{0.63}{ms}$ e
$ f_0 \approx \qty{1.6}{kHz}$ --- um tom audível. $ L = \qty{1.0}{mH}$
com $ C = \qty{100}{pF}$: $ T_0 = \qty{2.0}{\micro s}$,
$ f_0 \approx \qty{0.50}{MHz}$ --- um rádio freqüência. Apenas o
\emph{produto} $ LC $ define o ritmo.
\end{example}

\begin{remark}[Um quarto de período fora de compasso]\label{rem:g12:rlc-oscillations:quarter}
A corrente é um quarto \omterm{def:g12:oscillators-and-time:oscillator}{período} fora de \omterm{def:g12:oscillators-and-time:phase}{fase} com a cobrança: $ i = 0$
quando $ q = \pm Q_0$ (placas cheias, fluxo invertido) e $ i = \pm I_0$
quando $ q = 0$ (placas vazias, fluxo em plena inclinação) --- exatamente o
pêndulo, imóvel nos extremos e mais rápido na parte inferior.
\end{remark}

\section{A valsa energética}

\begin{proposition}[Conservação da energia armazenada]\label{prop:g12:rlc-oscillations:energy}
\index{energia!de um circuito LC}
No ideal \omterm{prop:g12:rlc-oscillations:equation}{Circuito LC} o total armazenado energia
(\cref{prop:g12:rc-rl-circuits:energy})
\[
E \;=\; E_C + E_L \;=\; \frac{q^2}{2C} + \tfrac12\,L i^2
\;=\; \frac{Q_0^2}{2C} \;=\; \tfrac12\,L I_0^2
\]
é constante: inteiramente no \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} quando $ q = \pm Q_0$, inteiramente em
o bobina quando $ i = \pm I_0$.
\end{proposition}

\begin{proof}
Diferenciar: $\frac{dE}{dt} = \frac{q}{C}\frac{dq}{dt} +
Li\frac{di}{dt} = i\left(\frac{q}{C} + L\frac{d^2q}{dt^2}\right) = 0$
por \cref{prop:g12:rlc-oscillations:equation}. Seu valor é lido em
$ t = 0$ (todos \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}), e novamente no instante totalmente atual
$ q = 0$ --- de onde $\tfrac12 L I_0^2 = Q_0^2/2C $, recuperando
$ I_0 = Q_0/\sqrt{LC}$ por energia sozinho.
\end{proof}

    ylabel={energy $/\,E$}, xmin=0, xmax=1, ymin=0, ymax=1.3,

  \addplot[omDef, thick, domain=0:1, samples=120] {cos(360*x)^2};

\end{axis}

e volto duas vezes por \omterm{def:g12:oscillators-and-time:oscillator}{período}, a soma fixada em $ Q_0^2/2C $ --- o mesmo
figura do \omterm{prop:g12:oscillators-and-time:spring-period}{oscilador de massa de mola}, renomeado.}
\end{omfigure}

\begin{example}[A valsa, medida]\label{ex:g12:rlc-oscillations:waltz}
$ C = \qty{1.0}{\micro F}$ cobrado para $ U_0 = \qty{10}{V}$, mudou para
$ L = \qty{10}{mH}$: energia $ E = \tfrac12 C U_0^2 = \qty{5.0e-5}{J}$,
\omterm{def:g12:oscillators-and-time:oscillator}{período} $ T_0 = \qty{0.63}{ms}$, e uma corrente de pico
$ I_0 = U_0\sqrt{C/L} = \qty{0.10}{A}$ chegou a um quarto \omterm{def:g12:oscillators-and-time:oscillator}{período}
(\qty{0.16}{ms}) após o interruptor fechar.
\end{example}

\section{Circuitos reais: oscilações amortecidas}

Cada real bobina está ferido de \omterm{def:g11:work-of-force:sign}{resistivo} fio: o laço honesto é um
série \emph{RLC} circuito, e o \omterm{def:g11:circuits-and-power:resistance}{resistência} tributa a valsa.

\begin{definition}[Regimes pseudo-periódicos e aperiódicos]\label{def:g12:rlc-oscillations:regimes}
\index{Circuito RLC}\index{regime pseudo-periódico}\index{regime aperiódico}
Em um circuito RLC em série, o R se dissipa energia por Joule
aquecimento (\cref{ch:g11:circuits-and-power}): o \omterm{def:g12:oscillators-and-time:oscillator}{oscilações livres} são
\emph{\omterm{def:g12:oscillators-and-time:damping}{amortecido}}. Dois regimes, leia um \omterm{def:g10:signals-and-waves:oscilloscope}{osciloscópio}:
\begin{itemize}
  \item \emph{\omterm{def:g10:signals-and-waves:period}{pseudo-periódico}} (pequeno $ R $): o circuito ainda
        oscila dentro de um envelope encolhido, com um tempo de repetição ---
        o \emph{pseudoperíodo}\index{pseudoperíodo} ---
        perto de $ T_0 = 2\pi\sqrt{LC}$ quando o amortecimento é leve;
  \item \emph{aperiódico} (grande $ R $): a carga volta a zero
        sem nunca ultrapassar --- nenhuma oscilação.
\end{itemize}
O espelho exato do \omterm{def:g12:oscillators-and-time:damping}{amortecido} \omterm{def:g12:oscillators-and-time:oscillator}{oscilador mecânico}
(\cref{def:g12:oscillators-and-time:damping}).
\end{definition}

\end{tikzpicture}

um envelope exponencial (tracejado) ou um deslocamento para zero sem
oscilação. Levemente \omterm{def:g12:oscillators-and-time:damping}{amortecido}, o \omterm{def:g12:rlc-oscillations:regimes}{pseudoperíodo} fica $\approx T_0$.}
\end{omfigure}

\begin{remark}[Mantendo oscilações]\label{rem:g12:rlc-oscillations:maintain}
\index{oscilações mantidas}
Para manter o \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} vivo, alimente o circuito exatamente energia o
\omterm{def:g11:circuits-and-power:resistance}{resistência} queimaduras, uma vez por ciclo e \emph{no passo} com o balanço: um
amplificador faz para o loop RLC o que o \omterm{def:g12:newtons-laws:momentum}{impulso} do escapamento faz para
o pêndulo (\cref{ch:g12:oscillators-and-time}). Cada relógio de quartzo
e todo transmissor de rádio é mantido oscilador, correndo
em seu \omterm{def:g12:rlc-oscillations:period}{frequência natural} $ f_0$.
\end{remark}

\section{O gêmeo mecânico}

\begin{proposition}[O dicionário elétrico-mecânico]\label{prop:g12:rlc-oscillations:analogy}
\index{analogia elétrica-mecânica}
A equação LC é a equação massa-mola
(\cref{prop:g12:oscillators-and-time:spring-period}) palavra por palavra:
\begin{center}\small
\begin{tabular}{lcl}
\omterm{prop:g12:rlc-oscillations:equation}{Circuito LC} & & spring--mass glider\\[2pt]
charge $ q $ & $\longleftrightarrow $ & position $ x $\\
current $ i = dq/dt $ & $\longleftrightarrow $ & velocity $ v = dx/dt $\\
\omterm{def:g12:rc-rl-circuits:inductor}{indutância} $ L $ & $\longleftrightarrow $ & mass $ m $\\
inverse \omterm{def:g12:rc-rl-circuits:capacitor}{capacitância} $1/C $ & $\longleftrightarrow $ & rigidez $ k $\\
\omterm{def:g11:circuits-and-power:resistance}{resistência} $ R $ & $\longleftrightarrow $ & \omterm{def:g10:inertia:inventory}{atrito}\\
$ L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0$ & $\longleftrightarrow $ &
  $ m\,\frac{d^2x}{dt^2} + kx = 0$\\[2pt]
$ T_0 = 2\pi\sqrt{LC}$ & $\longleftrightarrow $ &
  $ T_0 = 2\pi\sqrt{m/k}$\\[2pt]
$ E_C = q^2/2C $ & $\longleftrightarrow $ & $ E_p = \tfrac12 kx^2$\\
$ E_L = \tfrac12 Li^2$ & $\longleftrightarrow $ & $ E_k = \tfrac12 mv^2$
\end{tabular}
\end{center}
Mesma equação, mesmas soluções: todos os resultados sobre qualquer um oscilador
traduz instantaneamente para o outro idioma.
\end{proposition}

\begin{proof}
Comparar \cref{prop:g12:rlc-oscillations:equation} com
$ m\,d^2x/dt^2 = -kx $: substituindo $ q \to x $, $ L \to m $, $1/C \to k $
transforma um no outro, e $ LC \to m/k $ voltas
$2\pi\sqrt{LC}$ em $2\pi\sqrt{m/k}$.
\end{proof}

\begin{example}[Sintonizando um rádio]\label{ex:g12:rlc-oscillations:tuning}
Uma antena alimenta um sussurro de \emph{todo} estação em um loop LC, mas
apenas aquele que transmite perto $ f_0$ avança e aumenta ---
\omterm{def:g12:oscillators-and-time:resonance}{ressonância} (\cref{def:g12:oscillators-and-time:resonance}). O mostrador
gira as placas de uma variável \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}: mudando $ C $ movimentos
$ f_0 = 1/(2\pi\sqrt{LC})$, e a agulha passa para o fone de ouvido
estação por vez. No lado da transmissão, um mantido oscilador
(\cref{rem:g12:rlc-oscillations:maintain}) define a transmissão
freqüência; em um relógio, um cristal de quartzo --- um gêmeo mecânico de
de tirar o fôlego rigidez --- desempenha o mesmo papel.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:rlc-oscillations:1}
Calcular $ T_0$ e $ f_0$ para (a) $ L = \qty{10}{mH}$,
$ C = \qty{1.0}{\micro F}$; (b) $ L = \qty{1.0}{mH}$,
$ C = \qty{100}{pF}$. Por qual fator as \omterm{def:g12:oscillators-and-time:oscillator}{frequências} diferem?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rlc-oscillations:2}
Usando $\qty{1}{H} = \qty{1}{V.s/A}$ e $\qty{1}{F} = \qty{1}{C/V}$,
mostrar \omterm{def:g10:orders-of-magnitude:unit}{unidade} por \omterm{def:g10:orders-of-magnitude:unit}{unidade} que $\sqrt{LC}$ é um tempo. Por que o $2\pi $
mudar nada?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rlc-oscillations:3}
UM \qty{470}{nF} \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} é cobrado \qty{6.0}{V}. Calcular $ Q_0$
e o armazenado energia; descreva, sem cálculo, o que se segue
quando é mudado para um ideal bobina.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rlc-oscillations:4}
Um \omterm{prop:g12:rlc-oscillations:equation}{Circuito LC} segue $ q(t) = 2.0\cos(2\pi t/T_0)$ em microcoulombs,
com $ T_0 = \qty{4.0}{ms}$. Leia $ Q_0$; calcular $ f_0$, $ q(T_0/2)$,
$ i(0)$, e a corrente de pico $ I_0 = 2\pi Q_0/T_0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rlc-oscillations:5}
Em quais instantes do ciclo ocorre o energia inteiramente no
\omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}? Inteiramente no bobina? Qual é a contrapartida mecânica de
cada momento para um pêndulo?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rlc-oscillations:6}
Verifique por substituição que $ q(t) = Q_0\cos(2\pi t/T_0)$ satisfaz

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rlc-oscillations:7}
UM onda longa estação transmite em \qty{198}{kHz}. Com
$ C = \qty{220}{pF}$, o que \omterm{def:g12:rc-rl-circuits:inductor}{indutância} sintoniza isso?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rlc-oscillations:8}
$ C = \qty{1.0}{\micro F}$ cobrado para \qty{10}{V} descarrega em
$ L = \qty{10}{mH}$. Calcule o armazenado energia, a corrente de pico (por
energia conservação) e o instante em que é alcançado pela primeira vez.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rlc-oscillations:9}
Explique por que $ E_C $ e $ E_L $ oscilar em freqüência $2f_0$, não $ f_0$.
Mostre isso em $ t = T_0/8$ (começou em $ q = Q_0$) o energia está dividido
exatamente pela metade.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rlc-oscillations:10}
Em um \omterm{def:g10:signals-and-waves:oscilloscope}{oscilograma} de um leve \omterm{def:g12:oscillators-and-time:damping}{amortecido} \omterm{def:g12:rlc-oscillations:regimes}{Circuito RLC}, máximos sucessivos

constante, preveja o sexto máximo após o \qty{4.0}{V} um. O que
fração do \emph{energia} sobrevive cada \omterm{def:g12:rlc-oscillations:regimes}{pseudoperíodo}? O que pode
você diz do \omterm{def:g12:rlc-oscillations:regimes}{pseudoperíodo} em si?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rlc-oscillations:11}
Um \omterm{prop:g12:rlc-oscillations:equation}{Circuito LC} tem $ L = \qty{0.50}{H}$ e $ C = \qty{20}{\micro F}$.
Calcular $ T_0$; então, usando o dicionário
(\cref{prop:g12:rlc-oscillations:analogy}), encontre o rigidez $ k $
dando um \qty{0.50}{kg} planador o mesmo períodoe verifique.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rlc-oscillations:12}
Um sintonizador usa $ L = \qty{0.20}{mH}$ e uma variável \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} varrendo
\qtyrange{50}{500}{pF}. Calcule as duas \omterm{def:g12:oscillators-and-time:oscillator}{frequências} extremas. Mostre isso
$ f_0 \propto 1/\sqrt{C}$, então dez vezes em $ C $ dá apenas
$\sqrt{10} \approx 3.2$ em freqüência.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rlc-oscillations:13}
Um verdadeiro bobina faz o circuito perder $5.0\%$ do seu energia cada

o \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} então, como uma fração do início? No
$ f_0 = \qty{1.0}{MHz}$, quanto tempo isso dura em microssegundos?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rlc-oscillations:14}
Um relógio oscilador lojas $ E = \qty{1.0e-9}{J}$ e perde $2.0\%$ de
por ciclo em $ f_0 = \qty{32768}{Hz}$. Qual média \omterm{def:g11:work-of-force:power}{poder} deve o
mantendo a alimentação do circuito? Por que o push tem que chegar no passo

\end{exercise}

\section{Problema: Construindo um rádio de cristal}

\begin{problem}[{Problema de fim de semana --- construindo um rádio de cristal: um
bobina, um capacitor variável, um diodo e um fone de ouvido --- sem battery,
sem amplificador, e o noticiário noturno chega na energia da onda
em si}]
\label{pb:g12:rlc-oscillations:1}
Um kit do sótão dos avós: um bobina $ L = \qty{0.20}{mH}$ ferida em
um tubo de papelão, uma variável \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} cujas placas giram de
\qty{20}{pF} para \qty{480}{pF}, um diodo de cristal, um fone de ouvido e
\qty{20}{m} de fio aéreo. As estações para capturar transmissão entre
\qty{530}{kHz} e \qty{1700}{kHz}.

\medskip
\noindent\textbf{Part I --- The tuning loop.}
\begin{enumerate}
  \item O \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}, carregado, é deixado para descarregar no bobina:
        escreva a lei das malhas e transforme-a na equação diferencial
        para $ q(t)$.
  \item Verifique por substituição que $ q(t) = Q_0\cos(2\pi t/T_0)$
        resolve, desde que $ T_0 = 2\pi\sqrt{LC}$.
  \item Verifique por análise dimensional que $\sqrt{LC}$ é um tempo.
  \item Discagem definida para $ C = \qty{330}{pF}$: calcular $ T_0$ e $ f_0$. É
        o loop sintonizado dentro da banda?
  \item Encontre o \omterm{def:g12:rc-rl-circuits:capacitor}{capacitância} que sintoniza exatamente \qty{1.00}{MHz}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The variable \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}.}
\begin{enumerate}[resume]
  \item Calcule o \omterm{def:g12:rc-rl-circuits:capacitor}{capacitância} necessário para \qty{530}{kHz}.
  \item Calcule o \omterm{def:g12:rc-rl-circuits:capacitor}{capacitância} necessário para \qty{1700}{kHz}.
  \item Calcule as duas \omterm{def:g12:oscillators-and-time:oscillator}{frequências} do kit \qtyrange{20}{480}{pF}
        \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} realmente alcança. Cobre toda a banda?
  \item Mostre isso $ f_0 \propto 1/\sqrt{C}$ em fixo $ L $; pelo que
        fator deve $ C $ mudar para duplo $ f_0$?
  \item As placas abrem linearmente, então $ C $ cai continuamente conforme o mostrador
        voltas. Explique por que as estações se aglomeram em uma extremidade
        do mostrador --- e qual extremidade.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g10:energy-conservation:energy}{Energia} and amortecimento.}
Sintonizado para \qty{1.00}{MHz} ($ C = \qty{127}{pF}$), a antena momentaneamente
cobra o \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} para $ U_0 = \qty{20}{mV}$.
\begin{enumerate}[resume]
  \item Calcule a cobrança $ Q_0$ e o armazenado energia $ E $.
  \item Calcule a corrente de pico $ I_0$, usando energia conservação.
  \item O bobinade \omterm{def:g11:circuits-and-power:resistance}{resistência} dissipa $4.0\%$ do armazenado energia
        por ciclo: depois de quantos ciclos o energia dividido pela metade?
  \item Quanto tempo isso dura em microssegundos? Por que esse "toque"
        sozinho não reproduz o noticiário da noite?
  \item Explique por que a chegada \omterm{def:g10:signals-and-waves:wave}{aceno} keeps the loop swinging only
        quando sintonizado naquela estação --- e o que mantém o
        oscilação no \emph{transmissor}
        (\cref{rem:g12:rlc-oscillations:maintain}).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The mechanical twin.}
\begin{enumerate}[resume]
  \item Usando o dicionário (\cref{prop:g12:rlc-oscillations:analogy})
        com um planador de massa $ m = \qty{0.20}{kg}$, calcule o
        rigidez $ k $ da mola correspondente ao circuito sintonizado.
  \item A mola rígida de um carro tem $ k \approx \qty{5e4}{N/m}$. Comentário:
        um sistema massa-mola de bancada pode oscilar em megahertz?
  \item Mantenha um razoável $ k = \qty{100}{N/m}$ em vez disso: que massa
        oscila em \qty{1.00}{MHz}?
  \item Que objeto real é tão leve e ultrarrígido
        \omterm{def:g12:oscillators-and-time:oscillator}{oscilador mecânico}, e onde você o conheceu anteriormente
        ano (\cref{ch:g12:oscillators-and-time})?
  \item Folha de especificações, uma frase: a banda que seu rádio varre, o
        \omterm{def:g12:rc-rl-circuits:capacitor}{capacitância} que escolhe \qty{1.00}{MHz}, quanto tempo dura um anel solitário
        dura, e o que reproduz as notícias de qualquer maneira.
\end{enumerate}
\end{problem}
