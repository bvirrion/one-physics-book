\chapter{Circuitos RC e RL}\label{ch:g12:rc-rl-circuits}

Pressione o obturador da câmera até a metade: a \omterm{def:g11:lenses-and-eye:lens}{afinar} lamento sobe por dois \omterm{def:g10:orders-of-magnitude:unit}{segundos}, um
a luz fica verde --- só então o flash dispara, despejando em um milissegundo
o que levou dois \omterm{def:g10:orders-of-magnitude:unit}{segundos} coletando. Este capítulo atende aos dois componentes
que dão aos circuitos uma memória e um relógio --- \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} e bobina --- e
primeiras equações diferenciais da física, verificáveis com as derivadas deste ano.

\section{O capacitor}

\begin{definition}[Capacitor e capacitância]\label{def:g12:rc-rl-circuits:capacitor}
UM \emph{capacitor}\index{capacitor} é um par de placas de metal voltadas
separados por um \omterm{def:g11:lenses-and-eye:lens}{afinar} isolante. Conectadas a uma fonte, as placas levam
cargas opostas $+q $ e $-q $ --- um \omterm{def:g11:electric-gravitational-fields:efield}{campo elétrico} preenche a lacuna
(\cref{ch:g11:electric-gravitational-fields}) --- e o \omterm{def:g11:circuits-and-power:voltage}{tensão} $ u $
através das placas é proporcional à carga armazenada:
\[ q = C\,u . \]
A constante $ C $ é o \emph{capacitância}\index{capacitância}, em
\emph{farads}\index{farads} (\unit{F}): $\qty{1}{F} = \qty{1}{C/V}$.
\end{definition}

\begin{example}[Ordens de magnitude]\label{ex:g12:rc-rl-circuits:orders}
Um farad é enorme --- um coulomb estacionado por volt. Cerâmica de rádio
circuitos: \unit{pF} para \unit{nF}; eletrolíticos de \omterm{def:g11:work-of-force:power}{poder} suprimentos e
pisca: \unit{\micro F} para \unit{mF}; supercapacitores: milhares de \omterm{def:g12:rc-rl-circuits:capacitor}{farads}.
\end{example}

\begin{proposition}[Corrente em um capacitor]\label{prop:g12:rc-rl-circuits:cdudt}
A corrente que chega em um capacitorA placa é a taxa na qual seu
carga aumenta:
\[
i = \frac{dq}{dt} = C\,\frac{du}{dt} ,
\]
marcação em letras minúsculas de quantidades variáveis no tempo ($ U $, $ I $ fique firme,
\cref{ch:g11:circuits-and-power}).
\end{proposition}

\begin{proof}
Atual é a carga entregue por \omterm{def:g10:orders-of-magnitude:unit}{segundo}
(\cref{def:g11:circuits-and-power:current}): durante um intervalo cada vez menor,
$ i $ é a derivada de $ q $; e $ q = Cu $ com $ C $ dá constante
$ C\,du/dt $.
\end{proof}

\begin{remark}[Duas consequências]\label{rem:g12:rc-rl-circuits:continuity}
Se $ u $ é constante, $ i = 0$: um carregado \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} não passa corrente constante.
E $ u $ nunca pode pular --- um passo exigiria corrente infinita: o
capacitorde \omterm{def:g11:circuits-and-power:voltage}{tensão} é a memória do circuito.
\end{remark}

\section{Carregando um capacitor: o circuito RC}

Um loop --- fonte $\mathcal{E}$ (\cref{def:g11:circuits-and-power:emf}),
interruptor, R, vazio \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}: fechar $ K $ no $ t = 0$.

através $ R $ nas placas de $ C $.}
\end{omfigure}

\begin{theorem}[Lei de carga do circuito RC]\label{thm:g12:rc-rl-circuits:charging}
\index{Circuito RC}
Após o interruptor fechar, o \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} \omterm{def:g11:circuits-and-power:voltage}{tensão} obedece, então segue de
$ u(0) = 0$:
\[
RC\,\frac{du}{dt} + u = \mathcal{E},
\qquad
u(t) = \mathcal{E}\left(1 - e^{-t/\tau}\right),
\qquad \tau = RC .
\]
\end{theorem}

\begin{proof}
Tensões ao longo do loop adicione (\cref{prop:g11:circuits-and-power:laws}):
$\mathcal{E} = u_R + u = Ri + u $ a cada instante; substituir
$ i = C\,du/dt $ (\cref{prop:g12:rc-rl-circuits:cdudt}). Agora \emph{verificar}
o candidato: $ du/dt = (\mathcal{E}/\tau)e^{-t/\tau}$, então
$ RC\,(\mathcal{E}/RC)\,e^{-t/\tau} + \mathcal{E}(1 - e^{-t/\tau}) =
\mathcal{E}$, verdadeiro para todos $ t $; e $ u(0) = \mathcal{E}(1-1) = 0$. Isso
não \emph{outro} curva se ajusta à equação e o início é provado neste
curso de matemática do ano sobre equações diferenciais.
\end{proof}

\begin{definition}[Constante de tempo]\label{def:g12:rc-rl-circuits:tau}
O \emph{constante de tempo}\index{constante de tempo} de um \omterm{thm:g12:rc-rl-circuits:charging}{Circuito RC} é
$\tau = RC $. \omterm{def:g10:orders-of-magnitude:unit}{Unidades}: $\unit{\ohm} \times \unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$ --- o circuito
próprio ritmo.
\end{definition}

\begin{method}[Leitura $\tau $ na curva]\label{met:g12:rc-rl-circuits:reading}
Três leituras equivalentes, diretamente de um gráfico:
\begin{enumerate}
  \item \emph{tangente na origem:} $ u'(0) = \mathcal{E}/\tau $, então
        a tangente inicial atinge a assíntota em $ t = \tau $;
  \item \emph{63\%:} $ u(\tau) = \mathcal{E}(1 - e^{-1}) \approx
        0.63\,\mathcal{E}$;
  \item \emph{95\%:} $ u(3\tau) \approx 0.95\,\mathcal{E}$ --- cobrado,
        para fins práticos ($5\tau $: $99\%$).
\end{enumerate}
\end{method}

\begin{proposition}[Descarga]\label{prop:g12:rc-rl-circuits:discharge}
UM \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} cobrado para $ U_0$ e fechado em um R $ R $ sozinho obedece
$ RC\,du/dt + u = 0$ e segue
\[
u(t) = U_0\,e^{-t/\tau}, \qquad \tau = RC
\]
--- $37\%$ saiu às $\tau $, $5\%$ no $3\tau $.
\end{proposition}

\begin{proof}
Mesma lei de loop, sem fonte; substituição novamente:
$ RC(-U_0/\tau)e^{-t/\tau} + U_0 e^{-t/\tau} = 0$, com $ u(0) = U_0$.
\end{proof}

\begin{example}[Os números de uma unidade de flash]\label{ex:g12:rc-rl-circuits:flashrc}
Um flash fotográfico carrega $ C = \qty{800}{\micro F}$ através
$ R = \qty{1.0}{k\ohm}$: $\tau = 10^3 \times 8.0\times10^{-4} =
\qty{0.80}{s}$, então a luz ``pronto'' --- conectada para disparar em $95\%$ ---
espera $3\tau \approx \qty{2.4}{s}$: o gemido do gancho. Demitido
através do mero tubo de flash \qty{0.50}{\ohm}, o \emph{mesmo}
\omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} esvazia com $\tau' = \qty{0.40}{ms}$: duas mil vezes
mais rápido para fora do que para dentro.
\end{example}

\section{A bobina: indutância e o circuito RL}

\begin{definition}[Indutor e indutância]\label{def:g12:rc-rl-circuits:inductor}
Um \emph{indutor}\index{indutor} é um bobina de fio: sua corrente
enfia suas próprias voltas com um \omterm{def:g11:magnetic-fields:bfield}{campo magnético}
(\cref{ch:g11:magnetic-fields}); sempre que \emph{mudanças}, um \omterm{def:g11:circuits-and-power:voltage}{tensão}
aparece em todo o bobina:
\[
u = L\,\frac{di}{dt} .
\]
A constante $ L $ é o \emph{indutância}\index{indutância}, em
\emph{Henrique}\index{Henrique} (\unit{H}): $\qty{1}{H} = \qty{1}{V.s/A}$.
\end{definition}
\admitted

\begin{remark}[A bobina resiste à mudança]\label{rem:g12:rc-rl-circuits:oppose}
Por que uma corrente variável faz com que \omterm{def:g11:circuits-and-power:voltage}{tensão} --- indução --- é derivado em
os volumes universitários; aqui tomamos a lei e seu caráter: o
bobina \emph{se opõe às mudanças} da sua corrente. Gêmeos espelho: um capacitorde
$ u $ não pode pular, um bobinade $ i $ não pode pular.
\end{remark}

    to[R=$R$] (4.2,2.4) -- (5.4,2.4) to[L=$L$, i=$i$, v=$u$] (5.4,0) -- (0,0);
\end{circuitikz}

{\small O \omterm{prop:g12:rc-rl-circuits:rl}{Circuito RL}: depois $ K $ fecha, o bobina deixa a corrente
suba apenas no seu próprio ritmo $\tau = L/R $.}
\end{omfigure}

\begin{proposition}[Aumento de corrente em um circuito RL]\label{prop:g12:rc-rl-circuits:rl}
\index{Circuito RL}
Fechando a chave em uma fonte $\mathcal{E}$, um R $ R $ e um bobina
$ L $ em série dá $\mathcal{E} = Ri + L\,di/dt $, ou seja
\[
\frac{L}{R}\,\frac{di}{dt} + i = \frac{\mathcal{E}}{R},
\qquad\text{solved by}\quad
i(t) = \frac{\mathcal{E}}{R}\left(1 - e^{-t/\tau}\right),
\quad \tau = \frac{L}{R} .
\]
\end{proposition}

\begin{proof}
A lei do loop novamente; e é o \emph{mesma equação} como em
\cref{thm:g12:rc-rl-circuits:charging} com $ u \to i $,
$\mathcal{E} \to \mathcal{E}/R $, $ RC \to L/R $ --- a mesma substituição
verifica a solução, e cada leitura de
\cref{met:g12:rc-rl-circuits:reading} transferências.
\end{proof}

\begin{example}[Um eletroímã acorda]\label{ex:g12:rc-rl-circuits:rl}
Um relé bobina, $ L = \qty{0.30}{H}$ e $ R = \qty{6.0}{\ohm}$, em um
\qty{12}{V} alimentação: corrente final $\mathcal{E}/R = \qty{2.0}{A}$, em
o ritmo $\tau = 0.30/6.0 = \qty{50}{ms}$ --- acordado em $3\tau \approx
\qty{0.15}{s}$.
\end{example}

\begin{remark}[A faísca na abertura]\label{rem:g12:rc-rl-circuits:spark}
Abra um switch ao vivo bobina e $ i $ deve cair para zero em uma fração
de um milissegundo: $ di/dt $ é enorme e $ u = L\,di/dt $ chega a centenas
ou milhares de \omterm{def:g11:circuits-and-power:voltage}{volts} --- uma faísca salta no contato. A ignição de um carro
bobina faz isso de propósito, milhares de vezes por minuto.
\end{remark}

\section{Energia armazenada}

\begin{proposition}[Energia em um capacitor, energia em uma bobina]\label{prop:g12:rc-rl-circuits:energy}
\index{energia!de um capacitor}\index{energia!de uma bobina}
UM \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} $ C $ cobrado para \omterm{def:g11:circuits-and-power:voltage}{tensão} $ u $ e um bobina $ L $ transportando uma corrente
$ i $ loja
\[
E_C = \tfrac12\,C u^2 = \frac{q^2}{2C},
\qquad
E_L = \tfrac12\,L i^2 .
\]
\end{proposition}

\begin{proof}
Conseguindo uma pequena carga $ dq $ em pratos já em \omterm{def:g11:circuits-and-power:voltage}{tensão} $ u $ custos
$ u\,dq $ (\cref{def:g11:circuits-and-power:voltage}); empilhando o
custos, o total é a área abaixo da linha $ u = q/C $ --- o triângulo
abaixo --- então $ E_C = \tfrac12 qu = \tfrac12 Cu^2$. O mesmo triângulo para
o bobina (subindo $ i $ por $ di $ custos $ Li\,di $) dá
$ E_L = \tfrac12 Li^2$.
\end{proof}

em baixo \omterm{def:g11:circuits-and-power:voltage}{tensão}.}
\end{omfigure}

\begin{example}[Ordens de grandeza de energia]\label{ex:g12:rc-rl-circuits:energies}
O flash \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} de \cref{ex:g12:rc-rl-circuits:flashrc}:
$\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$. UM
\qty{3000}{F} supercapacitor em \qty{2.7}{V}: \qty{1.1e4}{J} --- um AA
vale a pena a battery. UM \qty{10}{mH} bobina no \qty{10}{A}: \qty{0.50}{J}.
Menos do que as batterys aguentam --- mas podem ser liberadas em um milissegundo.
\end{example}

\begin{remark}[Para que servem os dois cronometristas]\label{rem:g12:rc-rl-circuits:applications}
\index{suavização}\index{circuito de temporização}
Entrada lenta, saída rápida \omterm{def:g11:work-of-force:power}{poderes} o flash da câmera e o desfibrilador. O
\omterm{def:g12:mechanical-waves:delay}{atraso} $\tau = RC $ corre circuitos de temporização: limpadores intermitentes, piscando
faróis, luzes de escadas. UM \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} através de uma oferta \emph{suaviza}
\omterm{def:g11:circuits-and-power:voltage}{tensão} ($ u $ não consegue pular); uma série bobina suaviza a corrente ($ i $ não pode
pular). E bobina mais \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} juntos fazemos um oscilador
(\cref{ch:g12:rlc-oscillations}).
\end{remark}

\section{Problema: o flash da câmera}

\begin{problem}[{Problema de fim de semana --- o flash da câmera: dois segundos
choramingar, um milissegundo de glória e o primo montado na parede que reinicia
corações --- um exponencial comanda todos eles}]
\label{pb:g12:rc-rl-circuits:1}
Um flash compacto \omterm{def:g10:orders-of-magnitude:unit}{unidade} lojas energia em um \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}
$ C = \qty{800}{\micro F}$, cobrado através $ R = \qty{1.0}{k\ohm}$ de
um \omterm{def:g10:energy-conservation:transfer}{conversor} modelado como uma fonte ideal $\mathcal{E} = \qty{300}{V}$;
a luz ``pronta'' dispara em $95\%$ de cheio \omterm{def:g11:circuits-and-power:voltage}{tensão}. Demitido, o
\omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} vê apenas o tubo de flash, $ R_f = \qty{0.50}{\ohm}$. Dados: um
A célula AA oferece \qty{1.5}{V} e \qty{2.4}{A.h}; $ e = \qty{1.60e-19}{C}$.

\medskip
\noindent\textbf{Part I --- The reservoir.}
\begin{enumerate}
  \item Calcule a cobrança no \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} no \qty{300}{V}.
  \item Quantos cargas elementares é isso?
  \item Calcule o armazenado energia $ E_C $.
  \item A célula AA contém $ q = It \approx \qty{8.6e3}{C}$, portanto, cerca de
        \qty{1.3e4}{J}. Quantas vezes o capacitorde energia é isso?
  \item Então por que se preocupar com o \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}? Uma frase, usando ``\omterm{def:g11:work-of-force:power}{poder}''.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The wait for green.}
\begin{enumerate}[resume]
  \item Escreva a lei das malhas e transforme-a na equação diferencial
        para $ u(t)$.
  \item Verifique por substituição que $ u(t) = \mathcal{E}(1 - e^{-t/RC})$
        resolve, a partir de $ u(0) = 0$.
  \item Calcule o \omterm{def:g12:rc-rl-circuits:tau}{constante de tempo} $\tau $.
  \item Depois de quanto tempo a luz de prontidão fica verde e a que horas
        \omterm{def:g11:circuits-and-power:voltage}{tensão}? Reconheça o gemido de dois \omterm{def:g10:orders-of-magnitude:unit}{segundos}.
  \item Calcule a corrente de carga inicial; por que então morre
        longe?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The millisecond of glory.}
\begin{enumerate}[resume]
  \item Escreva a lei de quitação $ u(t)$ através do tubo e calcule a
        novo \omterm{def:g12:rc-rl-circuits:tau}{constante de tempo} $\tau'$.
  \item Calcule a corrente de pico através do tubo.
  \item Calcule o pico \omterm{def:g11:work-of-force:power}{poder}; compare-o com um \qty{2.2}{kW} chaleira.
  \item Tomando o flash como em $3\tau'$, indique sua duração e
        média \omterm{def:g11:work-of-force:power}{poder}.
  \item Forme a proporção entre o tempo de carregamento e o tempo de flash: o mesmo \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor},
        mesma cobrança --- o que mudou e o que isso torna
        \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor}: um energia tanque, ou um \omterm{def:g11:work-of-force:power}{poder} alavanca?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The cousin on the wall.}
Um desfibrilador deve fornecer $ E_C = \qty{200}{J}$ no
$ U_0 = \qty{1500}{V}$; peito e remos presentes sobre \qty{50}{\ohm}.
\begin{enumerate}[resume]
  \item Calcule o necessário \omterm{def:g12:rc-rl-circuits:capacitor}{capacitância}.
  \item Calcule a carga armazenada.
  \item Calcule o choque \omterm{def:g12:rc-rl-circuits:tau}{constante de tempo} e duração ($3\tau $);
        compare com o flash.
  \item Calcule a corrente de pico e o pico \omterm{def:g11:work-of-force:power}{poder} no paciente.
  \item O circuito de recarga deve deixar o dispositivo pronto ($95\%$) em
        \qty{6.0}{s}: encontre o carregamento necessário \omterm{def:g11:circuits-and-power:resistance}{resistência} e o
        corrente de carga inicial --- os números que um designer solicitaria.
\end{enumerate}
\end{problem}
