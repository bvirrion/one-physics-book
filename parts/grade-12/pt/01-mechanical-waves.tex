\chapter{Ondas Mecânicas}\label{ch:g12:mechanical-waves}

Sábado à noite, estádio cheio: uma onda de espectadores em pé
varre as arquibancadas em menos de um minuto, mas ninguém sai de casa
assento. O anel de uma pedra atravessa o lago enquanto a rolha apenas balança. Isto
capítulo estuda o próprio viajante --- o \omterm{def:g10:signals-and-waves:wave}{aceno} --- e os quatro
números que o domam: velocidade, \omterm{def:g12:mechanical-waves:delay}{atraso}, \omterm{def:g10:signals-and-waves:period}{período}, \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} --- o suficiente para
leia um terremoto em uma tira de papel.

\section{A perturbação viaja, o meio permanece}

\begin{definition}[Onda mecânica]\label{def:g12:mechanical-waves:wave}
UM \emph{onda mecânica}\index{onda mecânica} é um distúrbio
propagando-se através de um meio material --- corda, água, ar, rocha ---
carregando energia \emph{sem transportar matéria}: cada pedaço do
o médio balança brevemente no lugar, passa o movimento e se acomoda.
\end{definition}

\begin{example}[Não importa o transporte]\label{ex:g12:mechanical-waves:transport}
No estádio \omterm{def:g10:signals-and-waves:wave}{aceno} o meio é a multidão: o que viaja é o
ato de se levantar, enquanto cada espectador permanece sentado. No
lagoa, o anel se expande, mas uma rolha flutuante apenas balança no lugar.
\end{example}

\begin{definition}[Ondas transversais e longitudinais]\label{def:g12:mechanical-waves:transverse}
UM \omterm{def:g10:signals-and-waves:wave}{aceno} é \emph{transversal}\index{transversal} quando o meio
movimentos \emph{perpendicular} à propagação: uma corcova corre ao longo de um
corda sacudida enquanto cada ponto se move apenas para cima e para baixo. Isso é
\emph{longitudinal}\index{longitudinal} quando o meio se move
\emph{junto} a propagação: uma zona de compressão bobinas viaja por um
mola empurrada, cada bobina balançando no lugar.
\end{definition}

\begin{remark}[O som é longitudinal]\label{rem:g12:mechanical-waves:sound}
\omterm{def:g10:signals-and-waves:sound}{Som} é o onda \omterm{def:g12:mechanical-waves:transverse}{longitudinal} por excelência: ar comprimido ---
mais alto \omterm{def:g10:pressure:pressure}{pressão} (\cref{ch:g10:pressure}) --- viaja da fonte para
tímpano; \cref{ch:g12:sound-acoustics} é dedicado a isso. Terremotos
envie os dois tipos: P ~ondas comprimir a rocha, S~ondas agite-o de lado.
\end{remark}

\section{Velocidade e atraso}

\begin{definition}[Velocidade de propagação]\label{def:g12:mechanical-waves:speed}
O \emph{velocidade de propagação}\index{velocidade de propagação} de um \omterm{def:g10:signals-and-waves:wave}{aceno} é o
distância percorrida pela perturbação dividida pelo tempo gasto:
$v = d/t $. É a velocidade da viagem \emph{forma}, não do
médio.
\end{definition}

\begin{proposition}[O meio define a velocidade]\label{prop:g12:mechanical-waves:medium}
Para pequenas perturbações, o \omterm{def:g12:mechanical-waves:speed}{velocidade de propagação} depende apenas do
\emph{médio}, não na forma ou tamanho da perturbação. Em um
corda esticada, $ v $ aumenta com \omterm{def:g11:circuits-and-power:voltage}{tensão} e diminui com a massa por
\omterm{def:g10:orders-of-magnitude:unit}{medidor}: tenso e leve é rápido, frouxo e pesado é lento. \admitted
\end{proposition}

\begin{remark}[Onde mora a fórmula]\label{rem:g12:mechanical-waves:ropeformula}
A fórmula exata da velocidade da corda é derivada no volume do Ano 1; isso
ano, os fatos experimentais acima são suficientes.
\end{remark}

\begin{example}[Velocidades do som]\label{ex:g12:mechanical-waves:speeds}
\omterm{def:g10:signals-and-waves:sound}{Som} viaja em \qty{340}{m/s} no ar, \qty{1500}{m/s} na água,
\qty{5900}{m/s} em aço --- quanto mais rígido, mais rápido. Daí o
ouvido de filme de faroeste no trilho: o aço anuncia o trem primeiro.
\end{example}

\begin{definition}[Atraso]\label{def:g12:mechanical-waves:delay}
UM \omterm{def:g10:signals-and-waves:wave}{aceno} de velocidade $ v $ atinge um ponto à distância $ d $ depois do
\emph{atraso}\index{atraso} $\tau = d/v $: dois pontos de distância $ d $
separados, repita o mesmo movimento, o mais atrasado $\tau $ --- o
eco variando de \cref{ch:g10:signals-and-waves}, leia de trás para frente.
\end{definition}

\begin{proposition}[Superposição sem interação]\label{prop:g12:mechanical-waves:superposition}
Onde dois ondas sobreposição, o deslocamento do meio é a cada
instante o \emph{soma} dos deslocamentos que cada um causaria sozinho;
depois de cruzar, cada \omterm{def:g10:signals-and-waves:wave}{aceno} continua inalterado --- derivado no
Volume do ano ~ 2 da linearidade das equações de movimento. \admitted
\end{proposition}

\begin{example}[Pulsos cruzados]\label{ex:g12:mechanical-waves:crossing}
Envie uma protuberância ascendente de cada extremidade de uma corda: onde eles se encontram, o
a corda sobe brevemente até a soma das duas alturas; então dois intactos
surgem corcundas --- à medida que duas conversas cruzadas alcançam seus ouvintes.
\end{example}

\section{Ondas periódicas: a dupla periodicidade}

\begin{definition}[Onda periódica]\label{def:g12:mechanical-waves:periodic}
Quando a fonte repete seu movimento com \omterm{def:g10:signals-and-waves:period}{período} $ T $ e freqüência
$ f = 1/T $ (\cref{ch:g10:signals-and-waves}), ele alimenta o meio a
\emph{onda periódica}\index{onda periódica}: cada ponto repete o seu próprio
movimento com \omterm{def:g10:signals-and-waves:period}{período} $ T $, cada um ficando atrás de seus vizinhos no deslocamento \omterm{def:g12:mechanical-waves:delay}{atraso}.
\end{definition}

\begin{definition}[Comprimento de onda]\label{def:g12:mechanical-waves:wavelength}
Um instantâneo de um \omterm{def:g12:mechanical-waves:periodic}{onda periódica} mostra um padrão que se repete no espaço; é
espacial \omterm{def:g10:signals-and-waves:period}{período} --- crista a crista --- é o
\emph{comprimento de onda}\index{comprimento de onda} $\lambda $, em metros. UM \omterm{def:g10:signals-and-waves:period}{periódico}
onda é \emph{duplamente periódico}\index{duplamente periódico}: \omterm{def:g10:signals-and-waves:period}{período} $ T $
no tempo em cada ponto, \omterm{def:g10:signals-and-waves:period}{período} $\lambda $ no espaço a cada instante.
\end{definition}

\begin{theorem}[A relação de onda]\label{thm:g12:mechanical-waves:relation}
UM \omterm{def:g12:mechanical-waves:periodic}{onda periódica} de velocidade $ v $, \omterm{def:g10:signals-and-waves:period}{período} $ T $ e freqüência $ f $ tem
\omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} $\lambda = v\,T = v/f $.
\end{theorem}

\begin{proof}
Durante um \omterm{def:g10:signals-and-waves:period}{período} o padrão desliza para frente $ v\,T $; mas a fonte
--- portanto, cada ponto --- está de volta ao seu estado inicial, então o
padrão de deslizamento coincide com o original: a distância de repetição é $ v\,T $.
\end{proof}

\begin{remark}[A frequência pertence à fonte]\label{rem:g12:mechanical-waves:source}
Passando de um meio para outro --- som from air into water ---
um \omterm{def:g10:signals-and-waves:wave}{aceno} mantém seu freqüência, que a fonte dita, enquanto sua velocidade
mudanças; por $\lambda = v/f $ o \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} muda com isso.
\end{remark}

\begin{example}[Concerto A, no ar e debaixo d'água]\label{ex:g12:mechanical-waves:concertA}
O \qty{440}{Hz} concerto A de \cref{ch:g10:signals-and-waves} tem
\omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} $340/440 = \qty{0.77}{m}$ no ar e $1500/440 =
\qty{3.4}{m}$ na água: mesma nota, esticada pelo meio mais rápido.
\end{example}

\begin{definition}[Em fase, em oposição de fase]\label{def:g12:mechanical-waves:phase}
Dois pontos separados por um número inteiro de comprimentos de onda, $ d = k\lambda $,
movem-se de forma idêntica a cada instante: eles são
\emph{em fase}\index{em fase}. Pontos separados por
$ d = (k + \tfrac12)\lambda $ sempre se mova de forma oposta --- crista contra
calha: eles estão em \emph{oposição de fase}\index{oposição de fase}.
\end{definition}

\begin{method}[Lendo a dupla periodicidade]\label{met:g12:mechanical-waves:reading}
\begin{enumerate}
  \item Em uma gravação $ y(t)$ em um ponto fixo, leia $ T $ crista para
        crista; $ f = 1/T $.
  \item Em um instantâneo $ y(x)$ em um instante fixo, leia $\lambda $ crista
        para o cume.
  \item Deduzir $ v = \lambda/T = \lambda f $; verificar contra um direto
        tempo $ v = d/\tau $ quando disponível.
  \item Nunca confunda os dois gráficos: um eixo carrega \omterm{def:g10:orders-of-magnitude:unit}{segundos}, o
        outro metros.
\end{enumerate}
\end{method}

\section{Frentes de onda e atenuação}

\begin{definition}[Frente de onda]\label{def:g12:mechanical-waves:wavefront}
UM \emph{frente de onda}\index{frente de onda} é uma linha (ou superfície) de pontos
que o \omterm{def:g10:signals-and-waves:wave}{aceno} atinge no mesmo instante --- uma linha de crista no
lagoa. De uma fonte pontual, as frentes de onda são círculos em expansão; longe
de qualquer fonte, eles são quase diretos. O \omterm{def:g10:signals-and-waves:wave}{aceno} propaga
perpendicular às suas frentes de onda, cristas sucessivas $\lambda $ separado.
\end{definition}

\begin{definition}[Atenuação]\label{def:g12:mechanical-waves:attenuation}
A diminuição de um acenode \omterm{def:g10:signals-and-waves:amplitude}{amplitude} à medida que se propaga é o seu
\emph{atenuação}\index{atenuação}. Duas causas se somam: a energia
de uma circular \omterm{def:g12:mechanical-waves:wavefront}{frente de onda} é compartilhada por uma crista cada vez mais longa, e o
o meio absorve um pouco de passagem. Direto frentes de onda escapar do
primeira causa --- como a onda de uma tempestade distante pode cruzar um oceano.
\end{definition}

\section{Problema: Lendo um Sismograma}

\begin{problem}[{Problema de fim de semana --- lendo um sismograma: duas ondas
corrida para fora de uma falha quebrada, três círculos fixam o epicentro e um
terceira onda lenta dá um aviso ao litoral}]
\label{pb:g12:mechanical-waves:1}
Turno noturno em um observatório sismológico. Às 09:14:32 a agulha do
a estação A salta: movimentos bruscos e rápidos, depois movimentos mais lentos e mais amplos. Um
o terremoto quebrou rochas em algum lugar; dois ondas correr através do
crosta: P ondas, compressões em $ v_P = \qty{6.0}{km/s}$ e S ondas,
um tremor lateral em $ v_S = \qty{3.5}{km/s}$.

\medskip
\noindent\textbf{Part I --- Two ondas, one clock.}
\begin{enumerate}
  \item A partir das descrições acima, classifique P e S ondas como
        \omterm{def:g12:mechanical-waves:transverse}{transversal} ou \omterm{def:g12:mechanical-waves:transverse}{longitudinal}.
  \item Qual \omterm{def:g10:signals-and-waves:wave}{aceno} chega primeiro em todas as estações? Calcule a viagem
        tempos de ambos $ d = \qty{120}{km}$.
  \item Mostre que a lacuna de chegada P--S é
        $\Delta t = d\,(1/v_S - 1/v_P)$ e calcule para
        $ d = \qty{120}{km}$.
  \item Deduza a regra de alcance do sismólogo
        $ d = \dfrac{v_P\,v_S}{v_P - v_S}\,\Delta t $ e mostrar o
        coeficiente é sobre \qty{8.4}{km} por \omterm{def:g10:orders-of-magnitude:unit}{segundo} de intervalo.
  \item Por que a lacuna aumenta com a distância --- e por que isso é um presente
        para uma estação que nunca viu o terremoto começar?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Circles on the map.}
\begin{enumerate}[resume]
  \item Medidas da estação A $\Delta t_A = \qty{25}{s}$. Quão longe está o
        epicentro de A?
  \item Explique por que esta leitura coloca o epicentro em um círculo,
        não em um ponto.
  \item As estações B e C medem $\Delta t_B = \qty{15}{s}$ e
        $\Delta t_C = \qty{32}{s}$: suas distâncias até o epicentro?
  \item Explique como os três círculos juntos fixam o epicentro.
  \item O P \omterm{def:g10:signals-and-waves:wave}{aceno} alcançou A às 09:14:32. Calcule seu tempo de viagem a partir de
        o epicentro e a hora de origem do terremoto.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g10:signals-and-waves:wave}{aceno} that crosses the ocean.}
Os três círculos se encontram no mar: a falha rompeu-se no fundo do mar
e coloque a coluna de água em movimento. Longo ondas em viagens em águas profundas em um
velocidade definida pela profundidade --- cerca de \qty{200}{m/s} em mar aberto,
sobre \qty{10}{m/s} perto de uma praia (admitido aqui; a fórmula é
derivado no volume do Ano 1). Há um porto $ D = \qty{600}{km}$ de
o epicentro.
\begin{enumerate}[resume]
  \item Converter \qty{200}{m/s} para \unit{km/h} e nomeie um veículo com
        uma velocidade de cruzeiro comparável.
  \item Quanto tempo leva o tsunami para chegar ao porto?
  \item O sismógrafo do próprio porto sente o P \omterm{def:g10:signals-and-waves:wave}{aceno} primeiro. Depois de como
        muito tempo? Se o alarme sons naquele instante, quanto aviso
        tempo que a costa chega antes do tsunami?
  \item No mar o tsunami \omterm{def:g10:signals-and-waves:period}{período} é sobre \qty{15}{min}. Calcule seu
        \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} lá, e explique por que um navio não percebe isso.
  \item Perto da praia a velocidade cai para \qty{10}{m/s} enquanto o
        \omterm{def:g10:signals-and-waves:period}{período} permanece fixo. Calcule o novo \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}. O que deve
        acontecer com o \omterm{def:g10:signals-and-waves:wave}{aceno} à medida que se amontoa?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{def:g10:energy-conservation:energy}{Energia}, and the report.}
Um \omterm{def:g11:fundamental-interactions:strong}{forte} terremoto ocorre na ordem de \qty{1e15}{J}, levado por
o ondas.
\begin{enumerate}[resume]
  \item Calcule o comprimento de uma circular \omterm{def:g12:mechanical-waves:wavefront}{frente de onda} \qty{10}{km} de
        o epicentro, então \qty{210}{km}; por qual fator ele cresceu?
  \item Deduza por que o tremor enfraquece com a distância, mesmo em um ambiente sem perdas
        crosta --- e nomeie a segunda causa que uma crosta real adiciona.
  \item Um líquido não resiste ao cisalhamento. As estações do lado oposto registram o P
        \omterm{def:g10:signals-and-waves:wave}{aceno} mas não S \omterm{def:g10:signals-and-waves:wave}{aceno}: o que isso revela sobre o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} da Terra?
  \item O trem de alta velocidade do ano anterior energia capítulo
        (\cref{ch:g11:mechanical-energy}) carrega sobre \qty{1.5e9}{J}
        de \omterm{def:g11:mechanical-energy:kinetic}{energia cinética}: quantos desses trens existem \qty{1e15}{J} valor?
  \item O relatório, em duas frases: onde e quando ocorreu o terremoto,
        e quanto aviso a porta recebeu? Dê os três números.
\end{enumerate}
\end{problem}
