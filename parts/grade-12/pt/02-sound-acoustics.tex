\chapter{Som e Acústica}\label{ch:g12:sound-acoustics}

Toque a corda grave de um violão: a sala se enche de um lá grave, mas
nada chegava ao seu ouvido, a não ser o ar comprimido e esticado. Isto
capítulo segue esse padrão --- sua velocidade, como freqüência torna-se
tom e \omterm{def:g11:work-of-force:power}{poder} intensidadee por que uma nota difere em cada instrumento.

\section{O som é uma onda de pressão}

\begin{definition}[Onda sonora]\label{def:g12:sound-acoustics:sound-wave}
UM \emph{onda sonora}\index{onda sonora} é um mecânico \omterm{def:g10:pressure:pressure}{pressão} \omterm{def:g10:signals-and-waves:wave}{aceno}
(\cref{ch:g12:mechanical-waves}): uma superfície vibrante cria
viajando \emph{compressões}\index{compressões} (leve
sobrepressão) e \emph{rarefações}\index{rarefações}
(subpressão). Pacotes aéreos oscilam \emph{junto} a viagem
direção (som é \emph{longitudinal}\index{longitudinal}),
e é necessário um meio: um sino sob uma jarra de vácuo silencia
(\cref{ch:g10:signals-and-waves}).
\end{definition}

\begin{proposition}[Velocidade do som]\label{prop:g12:sound-acoustics:speed}
\index{velocidade do som}%
\omterm{def:g10:signals-and-waves:sound}{Som} viaja por volta \qty{340}{m/s} no ar em
\qty{20}{\degreeCelsius}, \qty{1500}{m/s} em água e
\qty{5000}{m/s} em aço: quanto mais rígido o meio, mais
mais rápido. \admitted
\end{proposition}

\begin{remark}[O ar quente é mais rápido]\label{rem:g12:sound-acoustics:temperature}
Estes são valores medidos; predizê-los é feito no Ano 1
volume. \omterm{def:g10:signals-and-waves:sound}{Som} acelera no ar mais quente (\qty{331}{m/s} no
\qty{0}{\degreeCelsius}) --- por que os ventos ficam desafinados à medida que o salão esquenta.
\end{remark}

\begin{example}[Os tamanhos dos sons]\label{ex:g12:sound-acoustics:wavelengths}
A relação $\lambda = v/f $ tamanhos qualquer som: no ar uma \qty{20}{Hz}
estrondo $340/20 = \qty{17}{m}$ --- um prédio --- e um
\qty{20}{kHz} o silvo se encaixa \qty{1.7}{cm}; os de um metro dobram-se
facilmente em torno de portas e cantos.
\end{example}

\section{Tom, volume e ouvido}

\begin{definition}[Tom e volume]\label{def:g12:sound-acoustics:pitch-loudness}
\emph{Tom}\index{Tom} é o percepção de freqüência: quanto mais alto
$ f $, quanto mais alta for a nota; concerto A é \qty{440}{Hz}, e duplicando
$ f $ eleva a nota uma oitava. \emph{Volume}\index{Volume} é
o percepção de \omterm{def:g10:signals-and-waves:amplitude}{amplitude}: quanto mais \omterm{def:g11:fundamental-interactions:strong}{forte} o \omterm{def:g10:pressure:pressure}{pressão} oscilação,
quanto mais alto; a amplificação altera o volume, não o tom.
\end{definition}

\begin{remark}[A janela do ouvido]\label{rem:g12:sound-acoustics:audible}
A audição se estende sobre \qty{20}{Hz} para \qty{20}{kHz}
(\cref{ch:g10:signals-and-waves}); abaixo está \omterm{def:g10:signals-and-waves:ultrasound}{infra-som}, acima de mentiras
ultrassom --- morcegos em \qty{50}{kHz}, sondas médicas em megahertz. Um
piano apenas amostra o meio: \qtyrange{27.5}{4186}{Hz}.
\end{remark}

\section{Intensidade sonora e escala de decibéis}

\begin{definition}[Intensidade sonora]\label{def:g12:sound-acoustics:intensity}
O \emph{intensidade sonora}\index{intensidade sonora} $ I $ é o
acústico \omterm{def:g11:work-of-force:power}{poder} cruzando um quadrado \omterm{def:g10:orders-of-magnitude:unit}{medidor} de frente para o \omterm{def:g10:signals-and-waves:wave}{aceno}:
$ I = P/S $, em \unit{W/m^2}. O mais silencioso audível som é sobre
$ I_0 = \qty{1.0e-12}{W/m^2}$; a dor começa perto \qty{1}{W/m^2}.
\end{definition}

\begin{proposition}[Atenuação quadrada inversa]\label{prop:g12:sound-acoustics:inverse-square}
\index{lei do inverso do quadrado}%
Uma pequena fonte irradiando \omterm{def:g11:work-of-force:power}{poder} $ P $ igualmente em todas as direções
dá, à distância $ r $, $ I = P/(4\pi r^2) \propto 1/r^2$: duplicação
a distância divide a intensidade por quatro.
\end{proposition}

\begin{proof}
O todo \omterm{def:g11:work-of-force:power}{poder} $ P $ atravessa a esfera da área $4\pi r^2$; dividir.
\end{proof}

\begin{definition}[Nível de som]\label{def:g12:sound-acoustics:level}
A orelha abrange doze \omterm{def:g11:work-of-force:power}{poderes} de dez, então as intensidades são citadas
logaritmicamente. O \emph{nível de som}\index{nível de som} de $ I $ é
\[ L = 10 \log_{10}(I/I_0), \qquad I_0 = \qty{1.0e-12}{W/m^2}, \]
em \emph{decibéis}\index{decibéis} (\unit{dB}): limiar auditivo
\qty{0}{dB}, dor perto \qty{120}{dB}.
\end{definition}

\begin{proposition}[Aritmética de decibéis]\label{prop:g12:sound-acoustics:db-arithmetic}
Multiplicando $ I $ por $ k $ adiciona $10\log_{10}k $ \omterm{def:g12:sound-acoustics:level}{decibéis}. Em particular
$\times 10$ adiciona \qty{10}{dB}; $\times 2$ adiciona
$10\log_{10}2 \approx \qty{3.0}{dB}$; $ N $ incoerente idêntico
fontes adicionar \emph{intensidades}, nunca nivela, dando
$ L_1 + 10\log_{10}N $; dobrando a distância até uma pequena fonte ($ I $
dividido por quatro) remove cerca de \qty{6.0}{dB}.
\end{proposition}

\begin{proof}
$10\log_{10}(kI/I_0) = 10\log_{10}(I/I_0) + 10\log_{10}k $;
distância: \cref{prop:g12:sound-acoustics:inverse-square}, $ k = 1/4$.
\end{proof}

\begin{method}[Trabalhando em decibéis]\label{met:g12:sound-acoustics:db}
\begin{enumerate}
  \item $ L = 10\log_{10}(I/I_0)$; voltar: $ I = I_0 \times 10^{L/10}$.
  \item Nunca adicione níveis: converta em intensidades, adicione, converta de volta
        --- ou $+\qty{3}{dB}$ por duplicação, $+\qty{10}{dB}$ por dez vezes.
  \item Cada duplicação da distância até uma pequena fonte custa \qty{6}{dB}.
\end{enumerate}
\end{method}

\begin{remark}[Danos auditivos]\label{rem:g12:sound-acoustics:damage}
A perda auditiva é um \omterm{rem:g11:nucleus-radioactivity:dose}{dose}: \qty{85}{dB} é seguro por oito horas, e
cada \qty{3}{dB} mais --- uma duplicação da intensidade --- reduz pela metade o
tempo seguro; as células ciliadas perdidas não voltam a crescer.
\end{remark}

\section{Timbre e harmônicos}

\begin{definition}[Harmônicos]\label{def:g12:sound-acoustics:harmonics}
UM \omterm{def:g10:signals-and-waves:period}{periódico} som de freqüência $ f_1$ --- o
\emph{fundamental}\index{fundamental} --- é em geral uma soma de
senos nas \omterm{def:g10:signals-and-waves:frequency}{frequências} $ f_n = n f_1$, $ n = 1, 2, 3, \dots $, isso é
\emph{harmônicos}\index{harmônicos}. UM som reduzido ao seu fundamental
é um \emph{Tom Puro}\index{Tom Puro}, como um diapasão.
\end{definition}

\begin{definition}[Espectro e timbre]\label{def:g12:sound-acoustics:timbre}
O \emph{espectro}\index{espectro} de um som é o gráfico de barras de
é \omterm{def:g12:sound-acoustics:harmonics}{harmônicos}: uma barra por $ f_n $, a altura que harmônicode
força. O ouvido ouve $ f_1$ como o tom e o \emph{mistura} de
pontos \omterm{def:g11:fundamental-interactions:strong}{fortes} como o \emph{timbre}\index{timbre} --- o somcor:
instrumentos que tocam a mesma nota compartilham $ f_1$, diferindo apenas em
seus espectros.
\end{definition}

\begin{example}[Por que você reconhece o chamador]\label{ex:g12:sound-acoustics:two-instruments}
O Lá de uma flauta é quase um tom puro; um violino curvando o mesmo
\qty{220}{Hz} adiciona \omterm{def:g11:fundamental-interactions:strong}{forte} \omterm{def:g12:sound-acoustics:harmonics}{harmônicos} no \qtylist{440;660;880}{Hz}.
Mesmo tom e intensidade, mas ninguém os confunde: uma voz também é
um \omterm{def:g12:sound-acoustics:timbre}{espectro} você sabe de cor.
\end{example}

\section{A corda vibrante}

\begin{definition}[Onda estacionária]\label{def:g12:sound-acoustics:standing-wave}
UM \omterm{def:g10:signals-and-waves:wave}{aceno} confinado entre as extremidades fixas de uma corda pode se estabelecer em um
\emph{onda estacionária}\index{onda estacionária}: uma oscilação no lugar,
pontos imóveis --- \emph{nós}\index{nós} --- alternando com
pontos de oscilação máxima --- \emph{antinodos}\index{antinodos}.
Nada mais viaja; a corda vibra em um padrão fixo.
\end{definition}

\begin{proposition}[Modos de uma string]\label{prop:g12:sound-acoustics:string-modes}
Uma sequência de comprimento $ L $, fixado em ambas as extremidades, transportando ondas de velocidade
$ v $, vibra continuamente apenas em padrões que se ajustam a um número inteiro $ n $
de meios comprimentos de onda entre o fim imposto \omterm{def:g12:sound-acoustics:standing-wave}{nós}:
\[ \lambda_n = \frac{2L}{n}, \quad f_n = n\,\frac{v}{2L} = n f_1, \quad n = 1, 2, 3, \dots \]
O modo mais baixo $ f_1 = v/(2L)$ é o \omterm{def:g12:sound-acoustics:harmonics}{fundamental}; os outros são
exatamente é \omterm{def:g12:sound-acoustics:harmonics}{harmônicos}. \admitted
\end{proposition}

\begin{remark}[Onde mora a prova honesta]\label{rem:g12:sound-acoustics:honest}
Isso confinado ondas escolha exatamente esses modos segue de
sobrepondo o \omterm{def:g10:signals-and-waves:wave}{aceno} com suas reflexões, um cálculo realizado em
o volume do Ano 1. A geometria convence: ambas as extremidades devem ficar
ainda assim, então um número inteiro de meios comprimentos de onda deve caber.
\end{remark}

\begin{example}[A corda A de um violino]\label{ex:g12:sound-acoustics:violin}
A corda A de um violino tem $ L = \qty{32.8}{cm}$ e sons
$ f_1 = \qty{440}{Hz}$: ondas corra ao longo dele em
$ v = 2Lf_1 = 2 \times 0.328 \times 440 \approx \qty{289}{m/s}$ ---
definido por \omterm{def:g11:circuits-and-power:voltage}{tensão} e massa, afinada pela cavilha. Pressionando um dedo
encurta $ L $ e levanta cada $ f_n $ de uma vez: o \omterm{def:g12:sound-acoustics:timbre}{espectro} desliza para cima.
\end{example}

\begin{remark}[Por que os instrumentos diferem]\label{rem:g12:sound-acoustics:instruments}
Como a corda é excitada --- dedilhada, curvada, percutida --- define como
grande parte de cada modo está presente, e o corpo amplifica alguns \omterm{def:g12:sound-acoustics:harmonics}{harmônicos}
mais do que outros; excitação mais corpo conserta o \omterm{def:g12:sound-acoustics:timbre}{espectro}, então o mesmo
$ f_1$ deixa um violão, um violino e um piano, cada um soando como ele mesmo.
\end{remark}

\section{Exercícios}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:sound-acoustics:15}
Cada alto-falante de festival idêntico por si só fornece \qty{80}{dB} no
mesa de mixagem. Que nível fazer $2$ dar? $10$? $100$? O que cada um
extra \qty{+10}{dB} custo --- o que isso diz sobre intensidade corridas?
\end{exercise}

\section{Problema: A Oficina do Luthier}

\begin{problem}[{Problema de fim de semana --- oficina do luthier: uma guitarra
corda disposta traste por traste, seus harmônicos afinados em um timbre, um
recital mantido abaixo da linha de perigo, e uma busca de ultrassom por um
falha na madeira}]
\label{pb:g12:sound-acoustics:1}
Um luthier está terminando um violão. Sua corda A tem comprimento vibratório
$ L = \qty{65.0}{cm}$ e deve som $ f_1 = \qty{110}{Hz}$. Pegar
\qty{340}{m/s} para som no ar, e admitir $ f_1 = v/(2L)$
(\cref{prop:g12:sound-acoustics:string-modes}), $ v $ sendo o \omterm{def:g10:signals-and-waves:wave}{aceno}
velocidade na corda.

\medskip
\noindent\textbf{Part I --- Laying out the fretboard.}
\begin{enumerate}
  \item Calcule o \omterm{def:g10:signals-and-waves:wave}{aceno} velocidade $ v $ na corda.
  \item Calcule o fundamentalde \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} \emph{na corda},
        então o \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} do \qty{110}{Hz} som \emph{no
        ar}. Por que eles diferem ao mesmo freqüência?
  \item A oitava (\qty{220}{Hz}): que comprimento vibratório lhe confere?
        Onde deve ficar a décima segunda casa, como uma fração de $ L $?
  \item O quinto (\qty{165}{Hz}): que comprimento vibratório lhe confere?
  \item Cada casa eleva a nota em um semitom, um fator
        $2^{1/12}$. Verifique com logaritmos quantos semitons separam
        \qty{110}{Hz} de \qty{220}{Hz}; comprimento na primeira casa?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- \omterm{def:g12:sound-acoustics:harmonics}{Harmônicos} and \omterm{def:g12:sound-acoustics:timbre}{timbre}.}
\begin{enumerate}[resume]
  \item Liste os cinco primeiros \omterm{def:g12:sound-acoustics:harmonics}{harmônicos} da corda A aberta.
  \item A de um violinista sons \qty{440}{Hz}. Qual harmônico do
        corda do violão é essa, e qual intervalo musical separa
        isso do \omterm{def:g12:sound-acoustics:harmonics}{fundamental}? E \qty{330}{Hz}?
  \item O luthier dedilha perto da ponte, depois sobre o
        escala: mesma tom, cor diferente. Explique com \omterm{def:g12:sound-acoustics:timbre}{espectros}.
  \item Um leve toque no ponto médio ao arrancar forças um nó
        lá. Qual \omterm{def:g12:sound-acoustics:harmonics}{harmônicos} sobreviver e o que tom é ouvido?
  \item Quantos \omterm{def:g12:sound-acoustics:harmonics}{harmônicos} da corda aberta mentem, em princípio,
        dentro do audível \omterm{def:g11:fundamental-interactions:strong}{faixa}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The recital.}
No primeiro concerto de violão, cada instrumento irradia uma acústica
\omterm{def:g11:work-of-force:power}{poder} $ P = \qty{6.0e-3}{W}$ como uma pequena fonte, igualmente em todos
direções.
\begin{enumerate}[resume]
  \item Calcule a intensidade em $ r = \qty{3.0}{m}$ de um instrumento.
  \item Deduza o \omterm{def:g12:sound-acoustics:level}{nível de som} lá.
  \item Um quarteto toca: que nível dão quatro desses instrumentos
        no mesmo local? Por que não é quatro vezes o nível?
  \item Doze instrumentos tocam: qual é o nível \qty{3.0}{m}?
  \item A que distância um ouvinte deve sentar-se para que o conjunto caia
        para \qty{85}{dB}?
  \item Regra de segurança: \qty{85}{dB} é seguro por oito horas, e
        cada \qty{+3}{dB} reduz pela metade o tempo seguro. Quanto tempo pode o
        ouvinte da primeira fila da pergunta 14 permanecerá com segurança?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The flaw in the wood.}
Antes de envernizar, o luthier sonda um \qty{45}{mm} pescoço grosso
em branco com um \qty{2.0}{MHz} testador; nesta madeira pegue $ v = \qty{5.0e3}{m/s}$.
\begin{enumerate}[resume]
  \item Calcule o ultrassom \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} na madeira. Por que um
        caça às falhas \omterm{def:g10:signals-and-waves:wave}{aceno} ser ultrassônico em vez de audível?
  \item Calcule o eco \omterm{def:g12:mechanical-waves:delay}{atraso} da face posterior do espaço em branco.
  \item Um eco chega depois \qty{7.2}{\micro s}: quão profunda é a falha?
  \item Escreva o cartão do workshop: \omterm{def:g10:signals-and-waves:wave}{aceno} velocidade na corda, décimo \omterm{def:g10:orders-of-magnitude:unit}{segundo}
        e comprimentos do primeiro traste, distância de audição segura, profundidade de falha.
\end{enumerate}
\end{problem}
