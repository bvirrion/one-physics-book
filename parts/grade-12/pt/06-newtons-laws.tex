\chapter{Leis de Newton}\label{ch:g12:newtons-laws}

Contrabandeie uma balança de banheiro para um elevador. À medida que a cabine sobe, o ponteiro indica
três \omterm{def:g10:orders-of-magnitude:unit}{quilogramas} pesado; perto do topo lê-se luz; entre os dois, a pura verdade
--- nada mudou a bordo, exceto o movimento. Este capítulo declara as três leis por trás
aquela agulha e as transforma no método que comanda toda a mecânica: escolha um
sistema, desenhe seu forças, projeto $\sum \vect F = m\vect a $.

\section{Momentum, e a primeira lei tornada exata}

\begin{definition}[Momento]\label{def:g12:newtons-laws:momentum}
O \emph{impulso}\index{impulso} de um corpo de massa $ m $ (\unit{kg}) movendo-se com
velocidade $\vect v $ é o vetor
\[
\vect p = m \vect v \qquad (\unit{kg.m/s}):
\]
a velocidade, ponderada pela massa; para um sistema, a soma vetorial de suas partes.
\end{definition}

\begin{definition}[Quadro de referência inercial]\label{def:g12:newtons-laws:inertial}
Um \emph{inercial referencial}\index{inercial referencial} é aquele em que
cada corpo isolado --- não forças, ou forças que \omterm{def:g10:inertia:compensate}{compensar} --- se move em linha reta
linha com velocidade constante (\cref{ch:g10:inertia}). O terreno é excelente
precisão, assim como qualquer cabine em movimento retilíneo uniforme sobre ela; uma travagem
trem ou carrossel giratório não: ali a bagagem solta acelera sem ser empurrada.
\end{definition}

\begin{theorem}[A primeira lei de Newton]\label{thm:g12:newtons-laws:first}\index{A primeira lei de Newton}
Em um referencial inercial, o \omterm{def:g12:newtons-laws:momentum}{impulso} de um sistema isolado não muda: $\vect p $
é constante, ou seja,\ $ d\vect p/dt = \vect 0$.
\end{theorem}
\admitted

\begin{remark}[Inércia, afiada duas vezes]\label{rem:g12:newtons-laws:sharpened}
O princípio da inércia \cref{ch:g10:inertia}, atualizado duas vezes: diz
\emph{onde} mantém --- define os referenciais inerciais como aqueles onde ocorre ---
e protege um vetor, $ m\vect v $, cuja contabilidade compensa no final do capítulo.
\end{remark}

\section{Segunda lei de Newton}

\begin{theorem}[Segunda lei de Newton]\label{thm:g12:newtons-laws:second}\index{Segunda lei de Newton}
Em um referencial inercial, o \omterm{def:g11:forces-and-motion:netforce}{força resultante} em um corpo é igual à taxa de variação de seu
\omterm{def:g12:newtons-laws:momentum}{impulso}; para massa constante, uma vez que $\vect p = m\vect v $,
\[
\sum \vect F = \frac{d\vect p}{dt} = m\,\frac{d\vect v}{dt} = m \vect a .
\]
\end{theorem}
\admitted

\begin{remark}[Uma lei da natureza e sua unidade]\label{rem:g12:newtons-laws:nature}
Não há prova: esta é uma lei da natureza, pesada contra a experiência por três
séculos. Ano passado (\cref{ch:g11:forces-and-motion}) deu sua sombra,
$\Delta v \propto F\,\Delta t/m $; a derivada (\cref{ch:g12:kinematics-2d}) faz
é exato a cada instante --- suas letras miúdas aguardam no volume do Ano 1. \omterm{def:g10:inertia:force}{Vigor} recebe um
\omterm{def:g10:orders-of-magnitude:unit}{unidade}, $\unit{N} = \unit{kg.m/s^2}$; e um corpo sob seu \omterm{def:g10:universal-gravitation:weight}{peso} sozinho obedece
$ m\vect a = m\vect g $: toda massa cai com $\vect a = \vect g $, e
$ g = \qty{9.81}{N/kg} = \qty{9.81}{m/s^2}$ --- as duas leituras são uma só.
\end{remark}

\begin{example}[Força média, sem necessidade de detalhes]\label{ex:g12:newtons-laws:average}
UM \qty{1300}{kg} carro vai do repouso para \qty{100}{km/h} (\qty{27.8}{m/s}) em
\qty{8.0}{s}: qualquer que seja a história instante a instante, a média \omterm{def:g11:forces-and-motion:netforce}{força resultante} é
$\Delta p/\Delta t = 1300 \times 27.8/8.0 \approx \qty{4.5e3}{N}$ --- \omterm{def:g12:newtons-laws:momentum}{impulso} voltas
um \omterm{def:g10:inertia:force}{vigor} questão na contabilidade antes e depois.
\end{example}

\section{Terceira lei de Newton}

\begin{theorem}[Terceira lei de Newton]\label{thm:g12:newtons-laws:third}\index{Terceira lei de Newton}
Se um corpo $ A $ exerce uma \omterm{def:g10:inertia:force}{vigor} $\vect F_{A \to B}$ em um corpo $ B $, então $ B $ exerce sobre
$ A $ o \omterm{def:g10:inertia:force}{vigor} $\vect F_{B \to A} = -\vect F_{A \to B}$: mesma linha de ação, mesma
magnitude, direções opostas --- a cada instante, qualquer que seja o movimento, contato ou
ação à distância.
\end{theorem}
\admitted

\begin{example}[O livro, a mesa e a Terra]\label{ex:g12:newtons-laws:book}
Um livro repousa sobre uma mesa: duas interações, dois pares. Gravitacional: a Terra puxa
o livro para baixo ($\vect P $), o livro puxa a Terra para cima. Contato: a mesa empurra
o livro para cima ($\vect N $), o livro pressiona a mesa para baixo. Caminhar é a mesma coisa:
empurre o chão para trás; sua reação é a única \omterm{def:g10:inertia:force}{vigor} levando você adiante.
\end{example}

\section{O método: escolher, desenhar, projetar}

\begin{method}[Resolvendo um problema de dinâmica]\label{met:g12:newtons-laws:method}
\begin{enumerate}
  \item \emph{Sistema}: nomeie o corpo (ou conjunto de corpos) estudado.
  \item \emph{Quadro}: escolha um referencial inercial e eixos adequados ao movimento ---
        eixos de inclinação, ou o \omterm{def:g12:kinematics-2d:frenet}{Quadro Frenet} para uma curva (\cref{ch:g12:kinematics-2d}).
  \item \emph{Inventário}: listar todos \omterm{def:g10:inertia:force}{vigor} --- o \omterm{def:g10:universal-gravitation:weight}{peso}, então um \omterm{def:g10:inertia:force}{vigor} por
        contato, nada mais --- e desenhe o diagrama de corpo livre: um ponto, um
        seta por \omterm{def:g10:inertia:force}{vigor}.
  \item \emph{Projeto}: escrever $\sum \vect F = m\vect a $, projete-o em cada eixo.
  \item \emph{Resolva e verifique}: \omterm{def:g10:orders-of-magnitude:unit}{unidades}, sinais, casos limites ($\mu \to 0$,
        $\alpha \to 0$, \dots).
\end{enumerate}
\end{method}

\begin{definition}[Força normal e atrito]\label{def:g12:newtons-laws:friction}
Uma superfície atua sobre um corpo através de dois componentes: o \emph{força normal}\index{força normal} $\vect N $, perpendicular à superfície, e \omterm{def:g10:inertia:inventory}{atrito} $\vect f $, tangencial.
A experiência dá a \omterm{def:g10:inertia:inventory}{atrito} leis: enquanto o corpo não desliza, estática \omterm{def:g10:inertia:inventory}{atrito}
ajusta-se até o teto, $ f_s \leq \mu_s N $; uma vez que desliza, \omterm{def:g11:forces-and-motion:friction}{atrito cinético}
é $ f_k = \mu_k N $, contra o deslizamento. O \emph{coeficientes de \omterm{def:g10:inertia:inventory}{atrito}}%
\index{coeficiente de atrito} $\mu_s $ e $\mu_k $ ($\mu_k \leq \mu_s $) dependem apenas
nos dois materiais.
\end{definition}

\begin{example}[A inclinação, duas vezes]\label{ex:g12:newtons-laws:incline}
Um bloco desliza por uma rampa de ângulo $\alpha $; eixos: $ x $ descendo a encosta, $ y $
perpendicular. Sobre $ y $: $ N = mg\cos\alpha $. Sobre $ x $, sem \omterm{def:g10:inertia:inventory}{atrito}: $ ma = mg\sin\alpha $,
então $ a = g\sin\alpha $ --- sem massa; no $\alpha = 30^\circ $, $ a = \qty{4.9}{m/s^2}$.
Com \omterm{def:g11:forces-and-motion:friction}{atrito cinético}: $ a = g(\sin\alpha - \mu_k\cos\alpha)$; para $\mu_k = 0.20$,
$ a = 9.81\,(0.500 - 0.173) = \qty{3.2}{m/s^2}$. E ficou parado em primeiro lugar
somente enquanto a estática necessária \omterm{def:g10:inertia:inventory}{atrito} caber sob seu teto: $\tan\alpha \leq \mu_s $.
\end{example}

\begin{definition}[Peso aparente]\label{def:g12:newtons-laws:apparent}
O \emph{peso aparente}\index{peso aparente} de um corpo é o \omterm{def:g12:newtons-laws:friction}{força normal}
entre ele e seu suporte --- que escala abaixo dele se lê.
\end{definition}

\begin{example}[O elevador]\label{ex:g12:newtons-laws:elevator}
UM \qty{70}{kg} passageiro fica em uma balança em um elevador de aceleração \omterm{def:g10:universal-gravitation:weight}{vertical}
$ a_z $ (para cima positivo): $ N - mg = ma_z $, então $ N = m(g + a_z)$. Subindo em
$ a_z = \qty{1.5}{m/s^2}$: \qty{792}{N}, pesado; freando perto do topo: \qty{582}{N},
luz; cruzeiro constante: $ mg = \qty{687}{N}$ --- a primeira lei; queda livre, $ a_z = -g $:
$ N = 0$, ausência de \omterm{def:g10:universal-gravitation:weight}{peso} em uma escala.
\end{example}

\begin{example}[Dois blocos e uma polia]\label{ex:g12:newtons-laws:pulley}
Um bloco $ m_1 = \qty{4.0}{kg}$ sobre uma mesa sem \omterm{def:g10:inertia:inventory}{atrito} é amarrado por uma corda leve, sobre
uma polia ideal, para um bloco suspenso $ m_2 = \qty{1.0}{kg}$. A corda transmite o
mesmo \omterm{def:g11:circuits-and-power:voltage}{tensão} $ T $ em ambas as extremidades; os blocos compartilham uma magnitude de aceleração $ a $.
Uma segunda lei cada, ao longo de cada movimento: $ m_1 a = T $ e $ m_2 a = m_2 g - T $, então
$ a = m_2\,g/(m_1 + m_2) = \qty{1.96}{m/s^2}$ e $ T = m_1 a = \qty{7.8}{N}$ ---
menor que $ m_2 g = \qty{9.8}{N}$, como deve ser para $ m_2$ para acelerar para baixo.
\end{example}

\begin{example}[A curva plana]\label{ex:g12:newtons-laws:curve}
Um carro de massa $ m $ arredonda uma curva plana de raio $ R $ em velocidade constante $ v $. No
\omterm{def:g12:kinematics-2d:frenet}{Quadro Frenet} (\cref{ch:g12:kinematics-2d}) a aceleração é puramente normal,
$ a_N = v^2/R $, voltado para o centro. Verticalmente, $ N = mg $; horizontalmente, o único
centrípeto \omterm{def:g10:inertia:force}{vigor} em oferta é a estática \omterm{def:g10:inertia:inventory}{atrito} de estrada com pneus:
$ f = mv^2/R \leq \mu_s N = \mu_s mg $, então $ v \leq \sqrt{\mu_s\, g R}$, seja qual for o
massa. Para $ R = \qty{90}{m}$, estrada seca ($\mu_s = 0.70$):
$ v_{\max} \approx \qty{25}{m/s}$ (\qty{89}{km/h}); no gelo o teto desaba ---
\cref{exo:g12:newtons-laws:13} margeia a estrada; \cref{exo:g12:newtons-laws:10} balanços
a mesma projeção em um fio.
\end{example}

\section{Sistemas isolados: o momento é conservado}

\begin{proposition}[Conservação do momento]\label{prop:g12:newtons-laws:conservation}\index{conservação do momento}
Para dois corpos cujo exterior forças desaparecer ou \omterm{def:g10:inertia:compensate}{compensar}, o total \omterm{def:g12:newtons-laws:momentum}{impulso}
$\vect p_1 + \vect p_2$ é constante, seja qual for o forças eles exercem um sobre o outro.
\end{proposition}

\begin{proof}
Adicione as duas segundas leis: $\frac{d}{dt}(\vect p_1 + \vect p_2) = \vect F_{2 \to 1} +
\vect F_{1 \to 2} + \vect F_{\text{ext}}$: o par interno é cancelado no terceiro
lei, e $\vect F_{\text{ext}} = \vect 0$ por hipótese.
\end{proof}

\begin{example}[Recuo]\label{ex:g12:newtons-laws:recoil}
UM \qty{60}{kg} patinador \omterm{def:g10:relative-motion:frame}{em repouso} no gelo liso lança um \qty{4.0}{kg} bola medicinal
horizontalmente em \qty{6.0}{m/s}. Antes: $\vect p = \vect 0$; depois:
$0 = 60\,V + 4.0 \times 6.0$, então $ V = \qty{-0.40}{m/s}$ --- ela flutua para trás.
Nada externo a empurrou: a bola fez (terceira lei), e os livros ficam no zero.
\end{example}

\begin{remark}[Como um foguete não empurra nada]\label{rem:g12:newtons-laws:rocket}
Um foguete é uma máquina para lançar massa. A cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}, seus \omterm{def:g11:work-of-force:sign}{motores} lançam um lote de gasolina
para trás; o gás leva para trás \omterm{def:g12:newtons-laws:momentum}{impulso}, então o foguete ganha o mesmo
valor a prazo --- um valor constante \omterm{def:g12:newtons-laws:momentum}{impulso}, aproximadamente a massa expelida por \omterm{def:g10:orders-of-magnitude:unit}{segundo} vezes o
velocidade de ejeção. O foguete empurra seu próprio escapamento, não o ar: ele funciona melhor em
espaço vazio.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:newtons-laws:1}
Calcule o \omterm{def:g12:newtons-laws:momentum}{impulso} de (a) um \qty{70}{kg} velocista em \qty{10}{m/s}; (b) um
\qty{1300}{kg} carro em \qty{50}{km/h}; (pode \qty{8.0}{g} bala em \qty{800}{m/s}.
Classifique-os. Por que a classificação mais rápida dura?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:newtons-laws:2}
UM \qty{70}{kg} passageiro está em uma balança em um elevador. O que lê quando
a cabine (a) acelera para cima a \qty{1.2}{m/s^2}; (b) sobe a uma constante
\qty{2.0}{m/s}; (c) acelera para baixo em \qty{1.2}{m/s^2}? Qual lei estabelece (b)?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:newtons-laws:3}
Um bloco é solto em um declive sem \omterm{def:g10:inertia:inventory}{atrito} de ângulo $30^\circ $. Calcule seu
aceleração, então sua velocidade após deslizar \qty{3.0}{m}. Para onde foi a massa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:newtons-laws:4}
UM \qty{58}{g} bola de tênis chega em \qty{15}{m/s} e é retornado junto com o mesmo
linha em \qty{25}{m/s}; contato dura \qty{5.0}{ms}. Calcule a mudança de \omterm{def:g12:newtons-laws:momentum}{impulso}
e a média \omterm{def:g10:inertia:force}{vigor} das cordas; compare isso \omterm{def:g10:inertia:force}{vigor} com a bola \omterm{def:g10:universal-gravitation:weight}{peso}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:newtons-laws:5}
Um livro repousa sobre uma mesa. Nomeie os dois forças agindo sobre o livro e, para cada um, seu
parceiro de ação-reação (qual corpo? qual direção?). Por que os livros \omterm{def:g10:universal-gravitation:weight}{peso}
e o \omterm{def:g12:newtons-laws:momentum}{impulso} da mesa não é parceiro um do outro?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:newtons-laws:6}
UM \qty{50}{kg} a caixa fica em um chão horizontal, $\mu_s = 0.45$, $\mu_k = 0.35$.

\section{Problema: o teleférico}

\begin{problem}[{Problema de fim de semana --- certificando o teleférico: a tensão na encosta
demandas em repouso, a sobretaxa do start-up, o alívio sobre a torre, o preço
de uma parada de emergência --- e o único número que o cartaz na cabine pode prometer}]
\label{pb:g12:newtons-laws:1}
Um teleférico de montanha --- cabine mais carruagem, massa vazia $ M_0 = \qty{2.0e3}{kg}$ ---
rola sobre um cabo de trilho inclinado em $\alpha = 30^\circ $, puxado por um cabo de transporte
paralelo à encosta com \omterm{def:g11:circuits-and-power:voltage}{tensão} $\vect T $. Rolando \omterm{def:g10:inertia:inventory}{atrito} é insignificante; o
o piso da cabine permanece totalmente horizontal. Os assentos da cabine $25$ passageiros de média
massa \qty{80}{kg}; você é o engenheiro certificador. Salvo indicação em contrário, o
cabine está cheia: $ M = \qty{4.0e3}{kg}$.

\medskip
\noindent\textbf{Part I --- At rest on the slope.}
\begin{enumerate}
  \item Escolha o sistema; diga por que a estrutura da montanha funcionará como uma estrutura inercial;
        inventário dos três forças, com instruções.
  \item Qual lei rege a espera? Projete-o ao longo e perpendicularmente à inclinação
        expressar $ T $ e o cabo da trilha \omterm{def:g12:newtons-laws:friction}{força normal} $ N $.
  \item Calcular $ T $ e $ N $.
  \item Recalcular $ T $ para a cabine vazia. Por que é $ T $ proporcional à massa total?
  \item O cabo de transporte se rompe e freia: mostre que a aceleração descontrolada é
        $ g \sin\alpha $, carregado ou vazio e calcule-o.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Start-up.} O guincho acelera toda a cabine em
$ a = \qty{0.60}{m/s^2}$ subindo a encosta até cruzar $ v = \qty{6.0}{m/s}$.
\begin{enumerate}[resume]
  \item Expresse e calcule o \omterm{def:g11:circuits-and-power:voltage}{tensão} $ T'$ durante esta fase.
  \item Em que fração o início aumenta o \omterm{def:g11:circuits-and-power:voltage}{tensão} acima do seu valor \omterm{def:g10:relative-motion:frame}{em repouso}?
  \item Quanto tempo dura a fase e a que distância?
  \item Um \qty{80}{kg} passageiro fica no chão horizontal. Projete o \omterm{def:g10:orders-of-magnitude:unit}{segundo}
        lei verticalmente para expressá-la e computá-la \omterm{def:g12:newtons-laws:apparent}{peso aparente}; comparar com $ mg $.
  \item Qual \omterm{def:g10:inertia:force}{vigor} acelera horizontalmente, e qual o tamanho? Por que os passageiros se inclinam?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Over the tower.} Em uma constante \qty{6.0}{m/s}, o
O cabo da trilha se curva sobre uma torre de suporte ao longo de um arco circular \omterm{def:g10:universal-gravitation:weight}{vertical} de raio
$ R = \qty{40}{m}$; na crista a velocidade é horizontal.
\begin{enumerate}[resume]
  \item No \omterm{def:g12:kinematics-2d:frenet}{Quadro Frenet} na crista (\cref{ch:g12:kinematics-2d}), dê ambos
        componentes da aceleração, com seus valores.
  \item Projete a segunda lei ao longo da normal para expressar e calcular a normais
        força $ N'$ do cabo no carro.
  \item A mesma pergunta para o passageiro \omterm{def:g12:newtons-laws:apparent}{peso aparente} no topo: por que
        fração ela é mais leve?
  \item A que velocidade seria $ N'$ desaparecer? Compare com a velocidade de cruzeiro e comente.
  \item O que \omterm{def:g10:inertia:force}{vigor} o carro exerce sobre o cabo sobre a torre --- qual lei
        diz isso, com que magnitude e direção?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Braking, a jump, and the placard.}
\begin{enumerate}[resume]
  \item No trecho horizontal de chegada, parada de emergência ocupa cabine cheia
        de \qty{6.0}{m/s} descansar em \qty{1.5}{s}: ler $\sum \vect F = d\vect p/dt $
        como uma média sobre a parada para calcular a frenagem \omterm{def:g10:inertia:force}{vigor}.
  \item Que horizontal \omterm{def:g10:inertia:force}{vigor} deve parar um \qty{80}{kg} passageiro, e pode
        sapato-chão estático \omterm{def:g10:inertia:inventory}{atrito} ($\mu_s = 0.40$) fornecê-lo? Concluir.
  \item Na plataforma, os freios são acionados, a cabine vazia fica suspensa, livre e imóvel; o último
        passageiro (\qty{80}{kg}) salta horizontalmente em \qty{2.5}{m/s}
        \emph{em relação à cabine}. Calcule a velocidade de recuo da cabine e o
        velocidade do passageiro no solo.
  \item Em uma frase: qual dispositivo de propulsão vive desse princípio de recuo,
        e o que corrige seu \omterm{def:g12:newtons-laws:momentum}{impulso}?
  \item \emph{Certificação.} O cabo de transporte se rompe em \qty{100}{kN}; o limite de regras
        o trabalho \omterm{def:g11:circuits-and-power:voltage}{tensão} em um quinto disso. Mostre que o pior caso encontrado acima é
        a partida, deduza a massa total máxima e imprima o cartaz: quantos
        \qty{80}{kg} os passageiros podem viajar? Verifique o fator de segurança.
\end{enumerate}
\end{problem}
