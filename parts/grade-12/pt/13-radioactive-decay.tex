\chapter{Decadência Radioativa}\label{ch:g12:radioactive-decay}

No porão de um museu, um contador Geiger clica sobre uma lasca de madeira de um
caixão do faraó. Cada clique é ilegal: nada o anunciou, nada
prevê o próximo. Mesmo assim, na sexta-feira, o laboratório imprimirá uma data válida para um
século. Este capítulo resolve o paradoxo --- como eventos perfeitamente aleatórios
um por um, somam-se, aos trilhões, aos relógios mais constantes que possuímos.

\section{Sem lei, um por um, como um relógio pela toupeira}

Ano passado (\cref{ch:g11:nucleus-radioactivity}) conhecemos o instável
\omterm{def:g11:fundamental-interactions:ladder}{núcleos}, equilibraram seus $\alpha $ e $\beta $ equações, e vi metade de qualquer
grande amostra desaparece por \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} $ T_{1/2}$. O que não podíamos dizer era
\emph{quando entre as escadas} --- nem por que um processo aleatório mantém um
agendar nada. Ambas as perguntas têm uma resposta.

\begin{definition}[Constante de decaimento]\label{def:g12:radioactive-decay:lambda}
Para cada instável \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo} há uma constante $\lambda $, isso é \emph{decadência
constante}\index{decadência
constante} (\omterm{def:g10:orders-of-magnitude:unit}{unidade} \unit{s^{-1}}), de modo que durante qualquer
intervalo curto $\dd t $ cada \omterm{def:g11:fundamental-interactions:ladder}{núcleo} tem probabilidade $ p = \lambda\,\dd t $ de
decadente --- o mesmo para todos \omterm{def:g11:fundamental-interactions:ladder}{núcleo} da espécie, qualquer que seja a sua idade,
e intocado pela temperatura, \omterm{def:g10:pressure:pressure}{pressão} ou química.
\end{definition}

\begin{remark}[Sem lei sozinho, exato em multidões]\label{rem:g12:radioactive-decay:crowds}
UM \omterm{def:g11:fundamental-interactions:ladder}{núcleo} que esperou dez mil anos tem exatamente a mesma probabilidade de decair
este \omterm{def:g10:orders-of-magnitude:unit}{segundo} como foi feito esta manhã: sem desgaste, sem aviso --- que
o próximo é genuinamente imprevisível. Multidões são outra questão. Jogue
$100$ moedas: esperar $50$ cabeças, mas não se surpreenda com $43$ ---
flutuações de ordem $\sqrt n $, aqui $10\%$. Jogue $10^{20}$: o
flutuação relativa $1/\sqrt n $ é $10^{-10}$, o resultado certo para
dez casas decimais. Um grama de matéria contém algum $10^{22}$ \omterm{def:g11:fundamental-interactions:ladder}{núcleos}: ilegalidade
médias em um relógio, uma lei exata para $ N(t)$, o número de \omterm{def:g11:fundamental-interactions:ladder}{núcleos}
ainda intacto no momento $ t $.
\end{remark}

\section{A lei da decadência}

\begin{proposition}[A equação da evolução]\label{prop:g12:radioactive-decay:ode}
Em uma grande amostra de $ N $ \omterm{def:g11:fundamental-interactions:ladder}{núcleos}, durante $\dd t $ o número muda em
\[
\dd N = -\lambda N\,\dd t, \qquad\text{i.e.}\qquad \frac{\dd N}{\dd t} = -\lambda N :
\]
a população diminui a uma taxa proporcional a si mesma.
\end{proposition}

\begin{proof}
Cada um dos $ N $ \omterm{def:g11:fundamental-interactions:ladder}{núcleos} decaem durante $\dd t $ com probabilidade
$\lambda\,\dd t $: decaimentos esperados $ N\lambda\,\dd t $, e para grandes $ N $ o
a contagem real segue a esperada
(\cref{rem:g12:radioactive-decay:crowds}); cada decaimento remove um \omterm{def:g11:fundamental-interactions:ladder}{núcleo}.
\end{proof}

\begin{theorem}[Lei do decaimento radioativo]\label{thm:g12:radioactive-decay:law}
Uma amostra contendo $ N_0$ \omterm{def:g11:fundamental-interactions:ladder}{núcleos} em $ t = 0$ detém, na hora $ t $,
\[
N(t) = N_0 \, e^{-\lambda t}.
\]
\end{theorem}

\begin{proof}
Verifique por substituição:
$\dd N/\dd t = N_0(-\lambda)\,e^{-\lambda t} = -\lambda N(t)$, e
$ N(0) = N_0 e^0 = N_0$: equação e comece ambos à direita. (Isso não
\emph{outro} ajustes de função são comprovados, com integrais, na universidade
volume.)
\end{proof}

\begin{definition}[Decadência exponencial]\label{def:g12:radioactive-decay:exponential}
Uma quantidade seguinte $ N_0\,e^{-\lambda t}$ sofre \emph{exponencial
decadência}\index{exponencial
decadência}: em tempos iguais, igual \emph{frações}
estão perdidos. É exatamente a descarga \omterm{def:g12:rc-rl-circuits:capacitor}{capacitor} de
\cref{ch:g12:rc-rl-circuits}, com $\lambda $ jogando $1/(RC)$: um
a equação diferencial, aprendida uma vez, opera circuitos e \omterm{def:g11:fundamental-interactions:ladder}{núcleos} da mesma forma.
\end{definition}

\begin{proposition}[Meia-vida]\label{prop:g12:radioactive-decay:halflife}
O \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} (\cref{def:g11:nucleus-radioactivity:halflife}) de um \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo}
é definido por seu constante de decaimento: $ T_{1/2} = \ln 2/\lambda $, e depois $ n $
meias-vidas $ N = N_0/2^{\,n}$, todo $ n $ ou não.
\end{proposition}

\begin{proof}
$ N(T_{1/2}) = N_0/2$ requer $ e^{-\lambda T_{1/2}} = \tfrac12$, ou seja\
$\lambda T_{1/2} = \ln 2$; e $ N(n\,T_{1/2}) = N_0\,e^{-n\ln 2} =
N_0/2^{\,n}$ para qualquer $ n $: a exponencial atravessa a escada e preenche
entre as escadas.
\end{proof}

\section{Atividade: contando os cliques}

\begin{proposition}[Atividade]\label{prop:g12:radioactive-decay:activity}
O \omterm{def:g11:nucleus-radioactivity:activity}{atividade} de uma amostra --- seu decaimento por \omterm{def:g10:orders-of-magnitude:unit}{segundo}, em \omterm{def:g11:nucleus-radioactivity:activity}{bequerels}
(\cref{def:g11:nucleus-radioactivity:activity}) --- é
\[
\mathcal A = \lambda N, \qquad\text{so}\qquad \mathcal A(t) = \mathcal A_0 \, e^{-\lambda t}:
\]
proporcional à população sobrevivente e igualmente exponencial.
\end{proposition}

\begin{proof}
$ N $ \omterm{def:g11:fundamental-interactions:ladder}{núcleos}, cada um com probabilidade $\lambda $ por \omterm{def:g10:orders-of-magnitude:unit}{segundo}: $\lambda N $
decai por \omterm{def:g10:orders-of-magnitude:unit}{segundo}; multiplicando \cref{thm:g12:radioactive-decay:law} por
$\lambda $ dá $\mathcal A(t) = \mathcal A_0 e^{-\lambda t}$.
\end{proof}

\begin{example}[Enorme $ N $, pequeno $\lambda $]\label{ex:g12:radioactive-decay:body}
Uma banana marca cerca de \qty{15}{Bq}, um corpo humano em cerca
\qty{8000}{Bq}, uma cúbica \omterm{def:g10:orders-of-magnitude:unit}{medidor} de granito em alguns \qty{e6}{Bq}
(\cref{ch:g11:nucleus-radioactivity}). Atrás dos cliques modestos, sente-se
populações astronômicas: o potássio-40 do corpo contribui
$\mathcal A \approx \qty{4.4e3}{Bq}$ com $\lambda =
\qty{1.8e-17}{s^{-1}}$: $ N = \mathcal A/\lambda \approx \num{2.5e20}$
\omterm{def:g11:fundamental-interactions:ladder}{núcleos}, dos quais apenas alguns milhares disparam a cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}.
\end{example}

\begin{method}[Medindo meia-vida]\label{met:g12:radioactive-decay:semilog}
\begin{enumerate}
  \item A contagem decai com um contador Geiger em sucessivos iguais
        intervalos: esta amostra $\mathcal A $ em momentos sucessivos.
  \item Trama $\ln \mathcal A $ contra $ t $: desde
        $\ln \mathcal A = \ln \mathcal A_0 - \lambda t $, \omterm{def:g12:radioactive-decay:exponential}{exponencial
        decadência} mostra como um \emph{linha reta} --- dados tortos significam
        outra lei, ou uma mistura de \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeos}.
  \item Ler $\lambda $ desligado como menos a inclinação; $ T_{1/2} = \ln 2/\lambda $.
\end{enumerate}
\end{method}

\begin{remark}[Becquerels não fazem mal]\label{rem:g12:radioactive-decay:dose}
O bequerel conta decaimentos, não danos. O dano é monitorado pelo \omterm{rem:g11:nucleus-radioactivity:dose}{dose} em
\omterm{rem:g11:nucleus-radioactivity:dose}{sieverts} (\cref{ch:g11:nucleus-radioactivity}), pesando o energia
depositado por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de tecido pela destrutividade de cada radiação: um
\qty{5e8}{Bq} a varredura com tecnécio é rotineira, muito menos \omterm{def:g11:nucleus-radioactivity:activity}{bequerels} de inalado
$\alpha $ emissor não são; convertendo \unit{Bq} para \unit{Sv} é
material universitário.
\end{remark}

\section{Namoro: tempo de leitura fora de uma proporção}

A lei de decaimento retroativa é um relógio: meça a fração sobrevivente,
e o exponencial nomeia o tempo decorrido.

\begin{method}[Datação por radiocarbono]\label{met:g12:radioactive-decay:carbon}
\begin{enumerate}
  \item \emph{Vivo:} tecido vivo troca carbono com a atmosfera
        e carrega sua proporção de carbono-14 --- um \omterm{def:g11:fundamental-interactions:ladder}{átomo} em $10^{12}$,
        dando $\mathcal A_0 = 13.6$ decai por minuto por grama de carbono.
  \item \emph{A morte inicia o relógio:} a ingestão é interrompida e a proporção
        decai com $ T_{1/2} = \num{5730}$ anos.
  \item Meça a amostra \omterm{def:g11:nucleus-radioactivity:activity}{atividade} $\mathcal A $ por grama de carbono;
        $\mathcal A = \mathcal A_0 e^{-\lambda t}$ inverte para
        $ t = \frac{1}{\lambda}\ln(\mathcal A_0/\mathcal A)
        = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}
\end{method}

\begin{example}[Uma lareira em uma caverna]\label{ex:g12:radioactive-decay:hearth}
O carvão de uma lareira enterrada dá $5.0$ decai por minuto por grama:
$\mathcal A_0/\mathcal A = 13.6/5.0 = 2.72$, então
$ t = \num{5730} \times \ln(2.72)/\ln 2 \approx \num{8.3e3}$ anos. O
o fogo se apagou há oito mil anos  a própria madeira é o arquivo.
\end{example}

\begin{example}[Relógios para pedras]\label{ex:g12:radioactive-decay:rocks}
Carbono-14 fica em silêncio além de alguns \num{50000} anos; geologia fica mais lenta
relógios. \emph{Potássio  argônio}: potássio-40 ($ T_{1/2} = \num{1.25e9}$
anos) decai em argônio, um gás que escapa lava derretida, mas fica presa uma vez
a rocha cristaliza --- a solidificação zera o relógio. \emph{Urânio  chumbo}:
o urânio-238 lidera uma cadeia de decaimentos que termina em \omterm{prop:g12:work-and-energy:equilibria}{estável} chumbo-206
(\cref{ch:g11:nucleus-radioactivity}), e a proporção de chumbo/urânio em um
cristal data isso. Ambos os relógios concordam com os minerais mais antigos
($\approx \num{4.4e9}$ anos) e, através de meteoritos, no Sistema Solar
idade.
\end{example}

\begin{remark}[Correntes e o gás na adega]\label{rem:g12:radioactive-decay:radon}
Entre o urânio e o chumbo a cadeia passa pelo rádio, depois pelo radônio-222,
um radioativo \emph{gás} ($ T_{1/2} = 3.8$ dias). Nascido em granito e
solo, penetra nos porões e se acumula; isso é $\alpha $-emitindo
filhas deterioram-se nos pulmões --- o maior natural da maioria das pessoas dosee
o mais barato para reduzir: abra a janela da adega.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:radioactive-decay:1}
O iodo-131 tem $\lambda = \qty{1.0e-6}{s^{-1}}$. Calcule seu \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} em
\omterm{def:g10:orders-of-magnitude:unit}{segundos}, depois em dias.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:radioactive-decay:2}
Radon-222 tem $ T_{1/2} = 3.8$ dias. Calcular $\lambda $ em \unit{s^{-1}},
então o \omterm{def:g11:nucleus-radioactivity:activity}{atividade} de uma amostra de \num{1.0e10} \omterm{def:g11:fundamental-interactions:ladder}{núcleos}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:radioactive-decay:3}
Verifique por substituição que $ N(t) = N_0\,e^{-\lambda t}$ satisfaz
$\dd N/\dd t = -\lambda N $ e $ N(0) = N_0$; então calcule $ N(T_{1/2})/N_0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:radioactive-decay:4}
Uma fonte de césio-137 ($ T_{1/2} = 30$ anos) tem \omterm{def:g11:nucleus-radioactivity:activity}{atividade} \qty{4.0e5}{Bq}

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:radioactive-decay:5}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:6}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:7}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:10}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:11}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:12}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:13}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:14}
Um contador Geiger dá, em $ t = 0$, $60$, $120$, $180$, $240$ \omterm{def:g10:orders-of-magnitude:unit}{segundos},
taxas de $400$, $225$, $127$, $71$, $40$ conta por \omterm{def:g10:orders-of-magnitude:unit}{segundo}. Calcular
$\ln \mathcal A $ a cada vez, verifique o alinhamento e deduza $\lambda $ e
$ T_{1/2}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:15}
Por um único \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, a probabilidade de sobreviver $ n $ \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} é
$1/2^{\,n}$. Calcule a probabilidade de que (a) um dado \omterm{def:g11:fundamental-interactions:ladder}{núcleo} sobrevive
$10$ meias-vidas; (b) todos $8$ assistiu \omterm{def:g11:fundamental-interactions:ladder}{núcleos} sobreviverem a um \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida}.
(c) De \num{1000} \omterm{def:g11:fundamental-interactions:ladder}{núcleos}, quantos são esperados após $3$ meias-vidas ---
e a lei pode dizer \emph{qual}? Conclua sobre o que a lei prevê.
\end{exercise}

\section{Problema: A idade das coisas}

\begin{problem}[{Problema de fim de semana --- a era das coisas: um laboratório de museu
calibra o relógio de carbono, data uma estátua e um papiro, descobre por que não
o relógio passa pelo mostrador e empurra uma rocha vulcânica para trás um bilhão de anos}]
\label{pb:g12:radioactive-decay:1}
Uma semana no laboratório de datação de um museu. Dados: mostra tecido vivo
$\mathcal A_0 = 13.6$ o carbono-14 decai por minuto por grama de carbono;
$ T_{1/2} = \num{5730}$ anos (carbono-14), \num{1.25e9} anos
(potássio-40); $1$ ano $= \qty{3.156e7}{s}$;
$ N_A = \qty{6.02e23}{mol^{-1}}$; massa molar de carbono \qty{12}{g/mol};
os contadores resolvem sobre $0.2$ decai por minuto por grama.

\medskip
\noindent\textbf{Part I --- Monday: calibrating the clock.}
\begin{enumerate}
  \item Por que a proporção de carbono-14 é constante no tecido vivo, e
        o que muda na morte?
  \item Calcular $\lambda $ para carbono-14 em \unit{s^{-1}}.
  \item Verifique por substituição que $ N_0 e^{-\lambda t}$ resolve
        $\dd N/\dd t = -\lambda N $.
  \item De $\mathcal A_0$, calcule o número de carbono-14 \omterm{def:g11:fundamental-interactions:ladder}{átomos} por
        grama de carbono no tecido vivo.
  \item Quantos carbono \omterm{def:g11:fundamental-interactions:ladder}{átomos} um grama aguenta? Deduza o carbono-14
        proporção --- compare com ``um em $10^{12}$''.
  \item Mostre que uma medida \omterm{def:g11:nucleus-radioactivity:activity}{atividade} $\mathcal A $ data uma amostra como
        $ t = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Tuesday: the statue and the papyrus.}
\begin{enumerate}[resume]
  \item Uma estátua de madeira dá $9.1$ decai por minuto por grama. Data o
        madeira.
  \item Que evento, exatamente, marca essa data --- a escultura ou
        mais alguma coisa? Que cuidado se segue?
  \item Um papiro egípcio dá $10.4$ decai por minuto por grama. Data
        isso. É plausível um escriba de vinte e dois séculos atrás?
  \item O \omterm{def:g11:nucleus-radioactivity:activity}{atividade} é medido para $\pm 0.2$ decai por minuto. Estimativa
        a incerteza da idade do papiro, usando
        $\Delta t \approx \Delta\mathcal A/(\lambda \mathcal A)$.
  \item O pergaminho antigo de um negociante dá $13.4$ decai por minuto
        por grama. Veredicto?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Thursday: the limits of the clock.}
\begin{enumerate}[resume]
  \item Depois de quantas meias-vidas uma amostra \omterm{def:g11:nucleus-radioactivity:activity}{atividade} cair abaixo do
        contadores' $0.2$ decai por minuto por grama? Que idade é essa?
  \item Os melhores laboratórios atingem cerca de dez meias-vidas. Qual é o
        horizonte prático da datação por radiocarbono, em anos?
  \item Um osso de dinossauro é sobre \num{6.6e7} anos: quantos
        \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} do carbono-14? Usando a questão 4, depois de quantas meias-vidas
        não é mesmo \emph{um} \omterm{def:g11:fundamental-interactions:ladder}{átomo} do carbono-14 do grama restante?
        Conclua sobre dinossauros com datação por carbono.
  \item Que tipo de \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo} poderia datar a rocha do dinossauro?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Friday: the volcanic rock.}
\begin{enumerate}[resume]
  \item O argônio é um gás, o potássio, um residente fiel do sólido: explique
        por que a solidificação da lava zera o relógio potássio-argônio.
  \item Supondo que cada potássio-40 decomposto \omterm{def:g11:fundamental-interactions:ladder}{núcleo} deixa um preso
        argônio \omterm{def:g11:fundamental-interactions:ladder}{átomo}, mostre que
        $ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.
  \item A rocha sob os fósseis do museu mostra
        $ N_{\mathrm{Ar}}/N_{\mathrm K} = 0.60$. Data a rocha.
  \item Poderia o carbono-14 ter datado esta rocha, ou o potássio - argônio, o
        estátua? Dê a janela útil de cada relógio em meias-vidas.
  \item Relatório de sexta-feira, uma frase: as três idades encontradas esta semana, e
        a única lei que lê todos eles.
\end{enumerate}
\end{problem}
