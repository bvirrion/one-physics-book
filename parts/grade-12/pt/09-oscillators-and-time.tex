\chapter{Osciladores Mecânicos e a Medição do Tempo}\label{ch:g12:oscillators-and-time}

Um relógio de pêndulo bate: cada tique é um pêndulo de latão cruzando o
\omterm{def:g10:universal-gravitation:weight}{vertical}, repetido quinze milhões de vezes desde o Ano Novo. Um quartzo
lasca em seu relógio flexiona \num{32768} vezes por \omterm{def:g10:orders-of-magnitude:unit}{segundo}; um césio \omterm{def:g11:fundamental-interactions:ladder}{átomo}
zumbe nove bilhões de vezes mais rápido. Tudo o que balança pode marcar o tempo:
este capítulo segue o balanço, de Huygens ao \omterm{def:g10:universal-gravitation:satellite}{satélites}.

\section{Oscilações livres}

\begin{definition}[Oscilador, período, frequência, amplitude]\label{def:g12:oscillators-and-time:oscillator}
UM \emph{oscilador mecânico}\index{oscilador mecânico} é um sistema que,
deslocado de um estábulo \emph{posição de equilíbrio}\index{posição de equilíbrio} e liberado, balança para frente e para trás em torno dele: estes são
\emph{oscilações livres}\index{oscilações livres}. O movimento se repete
depois do \emph{período}\index{período} $T_0$ (\unit{s}); o
\emph{frequência}\index{frequência} $ f_0 = 1/T_0$
(\unit{Hz}, \cref{ch:g12:mechanical-waves}); o
\emph{amplitude}\index{amplitude} é o deslocamento máximo.
\end{definition}

\begin{example}[Um zoológico de osciladores]\label{ex:g12:oscillators-and-time:zoo}
Um balanço de playground: $ T_0 \approx \qty{3}{s}$. A corda A de uma guitarra:
$ f_0 = \qty{110}{Hz}$. Um cristal de quartzo: \qty{32768}{Hz}. O césio
oscilação definindo o \omterm{def:g10:orders-of-magnitude:unit}{segundo}: cerca de \qty{9.2e9}{Hz}. Dez pedidos
de magnitude, uma tarefa: contar.
\end{example}

\section{O pêndulo simples}

\begin{definition}[Pêndulo simples]\label{def:g12:oscillators-and-time:pendulum}
UM \emph{pêndulo simples}\index{pêndulo simples}: um pequeno pedaço de massa
$ m $ em um fio inextensível de comprimento $ L $ e massa desprezível,
balançando em um plano \omterm{def:g10:universal-gravitation:weight}{vertical}. Dos dois forças no bob
(\cref{ch:g12:newtons-laws}), o \omterm{def:g11:circuits-and-power:voltage}{tensão} é perpendicular ao
movimento, enquanto o pesocomponente de ao longo do \omterm{def:g10:relative-motion:trajectory}{trajetória} aponta para trás
em direção ao ponto mais baixo: o \omterm{def:g10:universal-gravitation:weight}{peso} é o \emph{restaurando \omterm{def:g10:inertia:force}{vigor}}%
\index{força restauradora}, opondo-se ao deslocamento.
\end{definition}

\begin{proposition}[Período do pêndulo simples]\label{prop:g12:oscillators-and-time:pendulum-period}
\index{pêndulo simples!ponto final}
Para pequenos \omterm{def:g12:oscillators-and-time:oscillator}{amplitudes} (abaixo sobre $15^\circ $), o \omterm{def:g12:oscillators-and-time:oscillator}{período} é
\[
T_0 = 2\pi \sqrt{\frac{L}{g}}, \qquad g = \qty{9.81}{m/s^2}.
\]
Não depende nem da massa do pêndulo nem --- enquanto o ângulo
permanece pequeno --- no \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}: pequenos balanços são
\emph{isócrono}\index{isócrono}, todos demorando o mesmo tempo.
\end{proposition}
\admitted

\begin{remark}[Análise dimensional, e onde o $2\pi $ vidas]\label{rem:g12:oscillators-and-time:dimensional}
A fórmula pode ser adivinhada. De $ m $ (\unit{kg}), $ L $ (\unit{m}) e
$ g $ (\unit{m/s^2}), a única combinação com o \omterm{def:g10:orders-of-magnitude:unit}{unidade} de um tempo é
$\sqrt{L/g}$ --- e a massa \emph{não pode} entrar, nenhum outro dado
carregando um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} para cancelá-lo. As dimensões não podem dar o puro
número na frente: o $2\pi $ vem da resolução da equação do movimento,
feito no volume do Ano 1. A condição de ângulo pequeno também não é opcional:
em $30^\circ $ o verdadeiro \omterm{def:g12:oscillators-and-time:oscillator}{período} é $2\%$ por mais tempo e o isocronismo falha.
\end{remark}

\begin{example}[O pêndulo dos segundos]\label{ex:g12:oscillators-and-time:seconds}
Qual comprimento bate o \omterm{def:g10:orders-of-magnitude:unit}{segundo}, $ T_0 = \qty{2.00}{s}$, um tique cada
caminho? Resolvendo, $ L = g\,T_0^2/4\pi^2 = 9.81 \times 2.00^2/4\pi^2 \approx
\qty{0.994}{m}$. Quase um metro: os relógios de pêndulo são altos porque
\omterm{def:g10:orders-of-magnitude:unit}{segundos} são longos.
\end{example}

\section{O oscilador mola-massa}

\begin{definition}[Restaurando a força de uma mola]\label{def:g12:oscillators-and-time:spring}
Um planador de massa $ m $ desliza sobre um trilho horizontal sem \omterm{def:g10:inertia:inventory}{atrito}, amarrado a
uma primavera de \emph{doloroso}\index{doloroso} $ k $ (\unit{N/m}). Deixar
$ x $ seja o seu deslocamento do equilíbrio. Esticado ou comprimido, o
a mola puxa ou empurra o planador para trás com o \omterm{def:g10:inertia:force}{vigor} $ F = -kx $:
proporcional ao deslocamento e oposto a ele --- um restaurando
força novamente, agora fornecido pela elasticidade.
\end{definition}

\begin{proposition}[Equação de movimento e sua solução]\label{prop:g12:oscillators-and-time:spring-period}
\index{oscilador de massa de mola}
Segunda lei de Newton ao longo do trilho (\cref{ch:g12:newtons-laws}) dá
$ m \frac{d^2x}{dt^2} = -kx $, ou seja\ $\frac{d^2x}{dt^2} =
-\frac{k}{m}\,x $. Quaisquer que sejam as constantes $ A $ e $\varphi $,
\[
x(t) = A \cos\!\left(\frac{2\pi t}{T_0} + \varphi\right),
\qquad T_0 = 2\pi\sqrt{\frac{m}{k}},
\]
satisfaz esta equação: oscilação com \omterm{def:g12:oscillators-and-time:oscillator}{período} $ T_0$, independente
do \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}.
\end{proposition}

\begin{proof}
Diferenciando, $\frac{dx}{dt} = -\frac{2\pi}{T_0}A \sin(\frac{2\pi
t}{T_0} + \varphi)$; de novo, $\frac{d^2x}{dt^2} = -(\frac{2\pi}{T_0})^2
A\cos(\frac{2\pi t}{T_0} + \varphi) = -(\frac{2\pi}{T_0})^2\,x(t)$,
o que é $-(k/m)\,x $ exatamente quando $(2\pi/T_0)^2 = k/m $.
\end{proof}

\begin{remark}[Todos os movimentos que existem]\label{rem:g12:oscillators-and-time:uniqueness}
Que \emph{todo} o movimento do planador é uma dessas funções é
admitido, comprovado no volume do Ano 1. E não $ g $ em $2\pi\sqrt{m/k}$:
oscilações mais rígidas, mais rápidas, mais pesadas, mais lentas, o mesmo na Lua.
\end{remark}

\begin{definition}[Amplitude e fase]\label{def:g12:oscillators-and-time:phase}
Em $ x(t) = A\cos(2\pi t/T_0 + \varphi)$ a constante positiva $ A $ é
o \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} --- os extremos são $ x = \pm A $ --- e $2\pi t/T_0 +
\varphi $ é o \emph{fase}\index{fase}, crescendo em $2\pi $ cada
\omterm{def:g12:oscillators-and-time:oscillator}{período}; o \emph{fase inicial}\index{fase inicial} $\varphi $
registra a largada: liberado \omterm{def:g10:relative-motion:frame}{em repouso} de $ x = A $, $\varphi = 0$. Por exemplo $ m = \qty{0.20}{kg}$ sobre $ k = \qty{20}{N/m}$ dá $ T_0 =
2\pi\sqrt{0.20/20} = \qty{0.63}{s}$; puxado para \qty{5.0}{cm} e
lançado, $ x(t) = 5.0\cos(2\pi t/0.63)$ em centímetros.
\end{definition}

\section{Energia, amortecimento, ressonância}

\begin{proposition}[Energia do oscilador]\label{prop:g12:oscillators-and-time:energy}
Uma mola esticada por $ x $ armazena o potencial elástico energia $ E_p =
\tfrac12 kx^2$ (justificado em \cref{ch:g12:work-and-energy}). Durante
oscilação sem \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g11:mechanical-energy:em}{energia mecânica}
\[
E_m = E_k + E_p = \tfrac12 m v^2 + \tfrac12 k x^2 = \tfrac12 k A^2
\]
é constante: todo elástico nos extremos, todo cinético no equilíbrio.
O pêndulo faz o mesmo dueto entre $ E_k $ e gravitacional
$ E_p $ (\cref{ch:g11:mechanical-energy}).
\end{proposition}

\begin{proof}
Com $ v = \frac{dx}{dt}$, a derivada de $ E_m $ é $ mv\frac{dv}{dt}
+ kx\frac{dx}{dt} = v\,(m\frac{d^2x}{dt^2} + kx) = 0$ pela equação
de movimento; o valor $\tfrac12 kA^2$ é lido ao extremo.
\end{proof}

\begin{definition}[Oscilações amortecidas]\label{def:g12:oscillators-and-time:damping}
\omterm{def:g10:inertia:inventory}{Atrito} drenos \omterm{def:g11:mechanical-energy:em}{energia mecânica} de cada real oscilador: isso é
\omterm{def:g12:oscillators-and-time:oscillator}{oscilações livres} são \emph{amortecido}\index{amortecido}. Sob luz
amortecimento, o sistema ainda oscila, com uma morte \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} mas um
tempo de repetição quase inalterado, o \emph{pseudoperíodo}%
\index{pseudoperíodo}, perto de $ T_0$; sob \omterm{def:g11:fundamental-interactions:strong}{forte} amortecimento (querida)
ele volta ao equilíbrio sem nunca ultrapassar; o energia,
proporcional a \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} ao quadrado, decai em calor.
\end{definition}

\begin{definition}[Oscilações forçadas e ressonância]\label{def:g12:oscillators-and-time:resonance}
Empurre um oscilador periodicamente --- \emph{oscilações forçadas}%
\index{oscilações forçadas} --- e depois de um breve transiente ele
oscila no \emph{condução} freqüência $ f $, não o seu próprio. Isso é
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} atinge o pico quando $ f $ se aproxima do \emph{natural freqüência}%
\index{frequência natural} $ f_0$ do livre oscilador: isso é
\emph{ressonância}\index{ressonância}, cada push chegando em sintonia com
o movimento, alimentando-se energia cada ciclo; quanto mais leve amortecimento,
quanto mais alto e nítido for o pico.
\end{definition}

\begin{example}[Ressonância para o bem e para o mal]\label{ex:g12:oscillators-and-time:resonance-zoo}
Um pai empurrando um balanço empurra sozinho freqüência --- \omterm{def:g12:oscillators-and-time:resonance}{ressonância} colocar
para \omterm{def:g11:work-of-force:work}{trabalhar}. Em \num{2000} a nova Millennium Bridge em Londres balançou
alarmantemente quando os passos da multidão se fixaram em um \omterm{def:g12:oscillators-and-time:resonance}{frequência natural} aproximar
\qty{1}{Hz}; fechou dois anos para amortecedores. Cada relógio de quartzo mantém
seu canto de cristal dirigindo-o em \omterm{def:g12:oscillators-and-time:resonance}{ressonância}, \qty{32768}{Hz}.
\end{example}

\section{Tempo de medição}

\begin{method}[Cronometrando um período]\label{met:g12:oscillators-and-time:timing}
Nunca cronometre uma oscilação: inicie o cronômetro quando o pêndulo cruzar
a \omterm{def:g10:universal-gravitation:weight}{vertical} (mais rápida, mais fácil de julgar), conte $ N = 50$ oscilações,
dividir por $ N $ --- um \qty{0.2}{s} erro de reação torna-se \qty{4}{ms}
sobre $ T_0$. Repita e compare as execuções.
\end{method}

\begin{example}[Três séculos de carrapato]\label{ex:g12:oscillators-and-time:clocks}
Huygens construiu o primeiro relógio de pêndulo em \num{1657}: isocronismo
levou os relógios de um quarto de hora de desvio por dia para \omterm{def:g10:orders-of-magnitude:unit}{segundos}, e
pêndulos compensados em frações de \omterm{def:g10:orders-of-magnitude:unit}{segundo} por dia. O quartzo
relógio de pulso (\num{1969}) trocou a haste por uma flexão de cristal em
\qty{32768}{Hz}: décimos de \omterm{def:g10:orders-of-magnitude:unit}{segundo} por mês. Desde \num{1967} o
o \omterm{def:g10:orders-of-magnitude:unit}{segundo} em si é \emph{definido} por uma oscilação atômica: o
duração de \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{períodos} da radiação de micro-ondas de um
transição interna de césio-133 (\cref{ch:g12:quantum-world}). O
os melhores relógios de césio oscilam um \omterm{def:g10:orders-of-magnitude:unit}{segundo} em cem milhões de anos.
\end{example}

\begin{remark}[Por que perseguir o nanossegundo]\label{rem:g12:oscillators-and-time:gps}
Um receptor GPS encontra sua posição cronometrando o rádio sinais de
\omterm{def:g10:universal-gravitation:satellite}{satélites}; a luz viaja \qty{30}{cm} por nanossegundo, então um relógio
erro de um microssegundo coloca você no lugar errado \qty{300}{m}. \omterm{def:g10:universal-gravitation:satellite}{Satélites}
portanto carregam relógios atômicos --- e com essa precisão a taxa de um
relógio depende de quão rápido ele se move e quão alto ele está, um efeito que \omterm{def:g12:sound-acoustics:standing-wave}{nós}
encontrar-se em \cref{ch:g12:special-relativity}.
\end{remark}

\section{Problema: o banco do relojoeiro}

\begin{problem}[{Problema de fim de semana --- banco do relojoeiro: uma parede
relógio trazido de volta ao tempo --- cinquenta balanços cronometrados, uma haste de latão que
trechos no verão, o impulso sussurrado do escape, um quartzo
desafiante --- o veredicto apresentado em segundos por mês}]
\label{pb:g12:oscillators-and-time:1}
Um relógio de parede de pêndulo chega à bancada funcionando errado. Tratar
sua haste de latão e \omterm{def:g10:universal-gravitation:weight}{peso} pesado como um \omterm{def:g12:oscillators-and-time:pendulum}{pêndulo simples} de comprimento efetivo
$ L $, massa de bob $ m = \qty{1.2}{kg}$; a engrenagem conta cada completa
oscilação exatamente \qty{2.000}{s}.

\medskip
\noindent\textbf{Part I --- Taking the clock's pulse.}
\begin{enumerate}
  \item Por que tempo $50$ oscilações em vez de uma? Estime o erro
        deixado ligado $ T_0$ se o seu tempo de reação for \qty{0.2}{s}.
  \item O cronômetro lê \qty{99.4}{s} para $50$ oscilações: o \omterm{def:g12:oscillators-and-time:oscillator}{período}?
  \item Deduza o comprimento efetivo atual do pêndulo.
  \item Qual comprimento daria exatamente \qty{2.000}{s}? Deveria o
        a porca de classificação aumenta ou diminui o bob e em quanto?
  \item Não ajustado, o relógio está rápido ou lento, em quantos \omterm{def:g10:orders-of-magnitude:unit}{segundos} por dia?
  \item Sensibilidade a $ g $: o \omterm{def:g12:oscillators-and-time:oscillator}{período} na lua
        ($ g = \qty{1.62}{m/s^2}$)? O erro diário em um observatório
        onde $ g = \qty{9.80}{m/s^2}$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The summer slump.} O latão se expande: a
aquecimento $\Delta\theta $ estica a haste para $ L' = L\,(1 + \lambda\,
\Delta\theta)$, $\lambda = \num{1.9e-5}$ por \omterm{def:g10:light-spectra:kelvin}{Kelvin}. O relógio está
ajustado em \qty{20}{\celsius}.
\begin{enumerate}[resume]
  \item Calcular $\Delta L $ para um quarto de verão em \qty{25}{\celsius}.
  \item Usando $\sqrt{1+x} \approx 1 + x/2$, mostre que
        $\Delta T_0 / T_0 = \tfrac12 \lambda\, \Delta\theta $.
  \item Calcular $\Delta T_0/T_0$ e o erro diário de verão: rápido ou lento?
  \item As mesmas perguntas para um corredor de inverno em \qty{10}{\celsius}.
  \item Antigos relógios de precisão usavam pêndulos "gridiron" misturando latão
        hastes com hastes de aço (que se expandem cerca de metade) em
        direções alternadas. Explique como isso pode manter a taxa.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The whispered push.} Com o escape
desengatado, o balanço começou em \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} $\theta_m = 2.00^\circ $,
metade do seu \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} em \qty{280}{s}; em serviço, um pequeno empurrão do
escape cada \omterm{def:g12:oscillators-and-time:oscillator}{período} mantém $\theta_m $ constante.
\begin{enumerate}[resume]
  \item Quantos \omterm{def:g12:oscillators-and-time:oscillator}{períodos} o halving demora?
  \item No extremo o bob subiu $ h = L(1 - \cos\theta_m)
        \approx L\,\theta_m^2/2$ (radianos): deduzir $ E_m = \tfrac12
        m g L\,\theta_m^2$ e calcule-o
        (\cref{ch:g11:mechanical-energy}).
  \item \omterm{def:g10:energy-conservation:energy}{Energia} é proporcional a \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} ao quadrado: que fração de
        $ E_m $ está perdido por \omterm{def:g12:oscillators-and-time:oscillator}{período}?
  \item Deduza o energia de cada empurrão, e a média \omterm{def:g11:work-of-force:power}{poder} necessário.
  \item A \omterm{def:g10:orders-of-magnitude:unit}{unidade} \omterm{def:g10:universal-gravitation:weight}{peso} $ M = \qty{2.5}{kg}$ desce $ h =
        \qty{1.0}{m}$ por semana: isso é suficiente energia, em que \omterm{def:g10:energy-conservation:efficiency}{eficiência}?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The quartz challenger.} Um quartzo
o movimento oscila em $ f = \qty{32768}{Hz}$ e flutua sobre
\qty{0.3}{s} por mês; o pêndulo restaurado ainda flutua
\qty{0.5}{s} por dia.
\begin{enumerate}[resume]
  \item Calcule o quartzo \omterm{def:g12:oscillators-and-time:oscillator}{período}. Mostrar $\num{32768} = 2^{15}$: por que
        é um \omterm{def:g11:work-of-force:power}{poder} de dois exatamente o que um circuito de relógio deseja?
  \item Calcule os desvios relativos do quartzo e do pêndulo restaurado,
        e cada um em \omterm{def:g10:orders-of-magnitude:unit}{segundos} por mês. Quem ganha, por que fator?
  \item O padrão de césio --- \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{períodos} faça um
        \omterm{def:g10:orders-of-magnitude:unit}{segundo} --- desvios \qty{1}{s} por cem milhões de anos: é
        deriva relativa? Quantas ordens de magnitude abaixo do quartzo,
        e por que o GPS exige isso?
  \item O veredicto, uma frase com números: o que significa o restaurado
        relógio ganha na parede, e quem agora é o dono do \omterm{def:g10:orders-of-magnitude:unit}{segundo}?
\end{enumerate}
\end{problem}
