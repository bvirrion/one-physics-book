\chapter{O mundo quântico: fótons e níveis de energia}\label{ch:g12:quantum-world}

Um \omterm{def:g10:light-spectra:wavelength}{infravermelho} a lâmpada de calor pode banhar sua pele a noite toda e nunca marcá-la; dez
minutos de \omterm{def:g11:lenses-and-eye:lens}{afinar} a luz do sol da montanha deixa uma queimadura. A lâmpada oferece muito mais
energia --- mas energia não é o que marca a pele. Luz, este capítulo mostra,
chega em grãos, e só um grão que bate \omterm{def:g11:fundamental-interactions:strong}{forte} o suficiente faz química:
um \omterm{def:g10:light-spectra:wavelength}{ultravioleta} grãos rompem um vínculo que um bilhão \omterm{def:g10:light-spectra:wavelength}{infravermelho} grãos apenas quentes.
Siga os grãos e um século se abrirá: metais vazando elétrons, \omterm{def:g11:fundamental-interactions:ladder}{átomos}
cantando em linhas espectrais, elétrons interferindo como ondas.

\section{A luz chega em grãos: o fóton}

\begin{definition}[O fóton]\label{def:g12:quantum-world:photon}
Luz de freqüência $\nu $ trocas energia com matéria apenas em grãos inteiros
chamado \emph{fótons}\index{fótons}, cada um de energia
\[
E = h\nu = \frac{hc}{\lambda}, \qquad h = \qty{6.63e-34}{J.s},
\]
onde $ h $ é \emph{Constante de Planck}\index{Constante de Planck} (Planck,
1900; Einstein, 1905) e $\lambda $ o \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} no vácuo. Uma viga
intensidade mede o \emph{número} de fótons por \omterm{def:g10:orders-of-magnitude:unit}{segundo}; cada grão
energia é fixado pelo freqüência sozinho.
\end{definition}

\begin{notation}[O elétron-volt]\label{not:g12:quantum-world:ev}
A física atômica usa o \emph{elétron-volt}\index{elétron-volt},
$\qty{1}{eV} = \qty{1.60e-19}{J}$: o energia um \omterm{def:g11:fundamental-interactions:elementary}{carga elementar} \omterm{def:g10:signals-and-waves:oscilloscope}{ganhos}
travessia \qty{1}{V} (\cref{ch:g11:electric-gravitational-fields}). Prático:
$ E \,(\unit{eV}) \approx 1240/\lambda\,(\unit{nm})$.
\end{notation}

\section{O efeito fotoelétrico}

Ilumine uma placa de metal limpa no vácuo e os elétrons saltarão --- o
\emph{efeito fotoelétrico}\index{efeito fotoelétrico}. Suas regras quebraram a física clássica.

\begin{proposition}[O que o experimento mostra]\label{prop:g12:quantum-world:facts}
Para cada metal existe um \emph{limite de frequência}\index{limite de frequência} $\nu_0$:
\begin{enumerate}
  \item abaixo $\nu_0$, nenhum elétron sai, por mais intensa que seja a luz;
  \item acima $\nu_0$, a emissão começa sem \omterm{def:g12:mechanical-waves:delay}{atraso}, por mais fraco que seja;
  \item intensidade aumenta o \emph{número} de elétrons por \omterm{def:g10:orders-of-magnitude:unit}{segundo}, não
        seu máximo \omterm{def:g11:mechanical-energy:kinetic}{energia cinética}; freqüência aumenta $ E_{k,\max}$.
\end{enumerate}
Nada disso se encaixa em um contínuo \omterm{def:g10:signals-and-waves:wave}{aceno}, que qualquer cor deve deixar carregar ---
durante meses sob luz fraca (\cref{exo:g12:quantum-world:14}) --- e cujo
intensidade deve definir os elétrons energia: todas as três previsões falham.
\end{proposition}
\admitted

\begin{theorem}[Relação fotoelétrica de Einstein]\label{thm:g12:quantum-world:einstein}
\index{relação fotoelétrica}Um elétron precisa de pelo menos \emph{trabalho
função}\index{trabalho
função} $ W_0$ do metal escapar de sua superfície.
Um fóton é absorvido inteiro por um elétron, então
\[
h\nu = W_0 + E_{k,\max}, \qquad \text{hence } \nu_0 = \frac{W_0}{h}.
\]
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energia} escrituração contábil: $ h\nu $ em, $ W_0$ gasto na fuga, o resto cinético ---
no máximo $ h\nu - W_0$, menos para elétrons começando mais profundamente. Abaixo $\nu_0$
um grão não pode pagar a portagem de saída, seja qual for o seu número.
\end{proof}

\begin{definition}[Potencial de parada]\label{def:g12:quantum-world:stopping}
Para medir $ E_{k,\max}$, um reverso \omterm{def:g11:circuits-and-power:voltage}{tensão} freia os elétrons
(\cref{ch:g11:electric-gravitational-fields}); o \emph{parando
potencial}\index{parando
potencial} $ U_s $ é o menor \omterm{def:g11:circuits-and-power:voltage}{tensão} cancelando
a corrente: os elétrons mais rápidos simplesmente não conseguem cruzar, então $ E_{k,\max} = e\,U_s $.
\end{definition}

\begin{method}[Medindo a constante de Planck]\label{met:g12:quantum-world:measure-h}
\begin{enumerate}
  \item Iluminar uma fotocélula através \omterm{def:g11:color-light-sources:subtractive}{filtros} de conhecido comprimentos de onda;
        calcular cada $\nu = c/\lambda $; registre cada potencial de parada.
  \item Trama $ E_{k,\max} = eU_s $ contra $\nu $: Einstein promete o
        linha reta $ h\nu - W_0$.
  \item Leia a inclinação: é \emph{é} $ h $; as interceptações dão $\nu_0$ e $-W_0$.
\end{enumerate}
\end{method}

\section{Níveis de energia: por que os átomos emitem linhas}

\begin{definition}[Níveis de energia]\label{def:g12:quantum-world:levels}
O energia de um elétron ligado em um \omterm{def:g11:fundamental-interactions:ladder}{átomo} é
\emph{quantificado}\index{quantificado}: restrito a uma escada discreta de
\emph{níveis de energia}\index{níveis de energia} $ E_1 < E_2 < \dots < 0$, contado
negativo do elétron livre. O degrau mais baixo é o \emph{chão
estado}\index{chão
estado}, os outros \emph{animado
estados}\index{animado
estados}; alcançando $0$ é
\emph{ionização}\index{ionização}.
\end{definition}

\begin{proposition}[A escada do hidrogênio]\label{prop:g12:quantum-world:hydrogen}
O \omterm{def:g12:quantum-world:levels}{níveis de energia} do hidrogênio \omterm{def:g11:fundamental-interactions:ladder}{átomo} são
\[
E_n = -\frac{\qty{13.6}{eV}}{n^2}, \qquad n = 1, 2, 3, \dots
\]
então $ E_1 = \qty{-13.6}{eV}$, $ E_2 = \qty{-3.40}{eV}$, $ E_3 =
\qty{-1.51}{eV}$, $ E_4 = \qty{-0.85}{eV}$, aglomerando-se em direção $0$.
\end{proposition}
\admitted

\begin{remark}[De onde vem a escada]\label{rem:g12:quantum-world:ladder-origin}
Por que esses números são da mecânica quântica, mantidos para os volumes universitários ---
como é a fórmula de de Broglie: o elétron onda de matéria
(\cref{def:g12:quantum-world:debroglie} abaixo) deve fechar-se em torno
o \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, e apenas alguns comprimentos de onda caber --- como uma corda de violão, um
\omterm{def:g11:fundamental-interactions:ladder}{átomo} tem um conjunto discreto de notas.
\end{remark}

\begin{theorem}[Fóton de uma transição]\label{thm:g12:quantum-world:transition}
Um \omterm{def:g11:fundamental-interactions:ladder}{átomo} muda de nível apenas emitindo ou absorvendo um fóton carregando
exatamente a diferença: descer do nível $ p $ para um nível inferior $ n $ emite
$ h\nu = E_p - E_n $; subindo de $ n $ para $ p $ absorve o mesmo energia.
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energia} conservação, um grão de cada vez: o fóton é a única moeda
a luz aceita, e o átomoos livros devem estar equilibrados.
\end{proof}

\begin{remark}[Espectros de linha explicados]\label{rem:g12:quantum-world:lines}
Aqui está a resposta prometida por \cref{ch:g10:light-spectra}: um gás quente emite
apenas as \omterm{def:g12:oscillators-and-time:oscillator}{frequências} $(E_p - E_n)/h $ de sua própria escada --- um código de barras de
linhas brilhantes --- e um gás frio rouba exatamente essas \omterm{def:g12:oscillators-and-time:oscillator}{frequências} branco
luz, imprimindo linhas de absorção escuras correspondentes. Cada elemento tem seu próprio
escada, daí seu próprio código de barras --- como lemos a luz das estrelas.
\end{remark}

\begin{example}[As linhas visíveis de hidrogênio]\label{ex:g12:quantum-world:balmer}
Transições que terminam em $ n = 2$ cair no visível. Com $ E \,(\unit{eV}) =
1240/\lambda\,(\unit{nm})$: $3 \to 2$: \qty{1.89}{eV}, \qty{656}{nm} (vermelho);
$4 \to 2$: \qty{2.55}{eV}, \qty{486}{nm} (azul esverdeado); $5 \to 2$:
\qty{2.86}{eV}, \qty{434}{nm} (violeta). Transições para $ n = 1$ exceder
\qty{10}{eV}: todos \omterm{def:g10:light-spectra:wavelength}{ultravioleta}.
\end{example}

\begin{remark}[Lasers, de uma só vez]\label{rem:g12:quantum-world:laser}
UM fóton de exatamente $ E_p - E_n $ passando por um \emph{excitado} \omterm{def:g11:fundamental-interactions:ladder}{átomo} pode acioná-lo
emitir um idêntico fóton --- mesmo energia, mesma direção, em passo. Mantenha
mais \omterm{def:g11:fundamental-interactions:ladder}{átomos} para cima do que para baixo e um fóton torna-se uma avalanche de clones:
\emph{emissão estimulada}\index{emissão estimulada} --- e um \omterm{def:g11:color-light-sources:lamps}{laser} é pouco mais.
\end{remark}

\section{Ondas de matéria}

\begin{proposition}[Elétrons interferem]\label{prop:g12:quantum-world:interference}
Envie elétrons um de cada vez através de uma fenda dupla
(\cref{ch:g12:light-as-wave}): cada um cai como um único ponto pontiagudo --- um
partícula. Mas à medida que milhares se acumulam, os pontos se desenham \omterm{def:g12:light-as-wave:fringes}{franjas de interferência}:
cada elétron, sozinho, passa como um \omterm{def:g10:signals-and-waves:wave}{aceno} através \emph{ambos} fendas. Cristais
difratar feixes de elétrons também (Davisson e Germer, 1927).
\end{proposition}
\admitted

\begin{definition}[Comprimento de onda de De Broglie]\label{def:g12:quantum-world:debroglie}
Cada partícula de \omterm{def:g12:newtons-laws:momentum}{impulso} $ p = mv $ viaja como um \emph{importante
onda}\index{importante
onda} de \emph{comprimento de onda de de Broglie}\index{comprimento de onda de de Broglie}
\[
\lambda = \frac{h}{p} = \frac{h}{mv}.
\]
\end{definition}
\admitted

\begin{example}[Por que você nunca difrata]\label{ex:g12:quantum-world:never}
Um elétron acelerado através \qty{100}{V} tem $ E_k = \qty{1.60e-17}{J}$,
$ p = \sqrt{2m_e E_k} = \qty{5.4e-24}{kg.m/s}$, então $\lambda \approx
\qty{0.12}{nm}$ --- o espaçamento de \omterm{def:g11:fundamental-interactions:ladder}{átomos} em um cristal, que portanto atua
como sua fenda dupla. UM \qty{58}{g} bola de tênis em \qty{25}{m/s}: $\lambda =
6.63\times10^{-34}/1.45 \approx \qty{4.6e-34}{m}$, alguns $10^{19}$ vezes
menor que um \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- nenhuma fenda jamais mostrará franjas. $ h $ é tão
pequeno que a granulação quântica surge apenas no átomoescala.
\end{example}

\begin{example}[O microscópio eletrônico]\label{ex:g12:quantum-world:microscope}
\omterm{def:g12:light-as-wave:diffraction}{Difração} proíbe a resolução de detalhes muito menores do que o \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} usado
(\cref{ch:g12:light-as-wave}): a luz visível pára perto \qty{0.5}{\micro m}.
Elétrons em \qty{10}{kV} ter $\lambda \approx \qty{12}{pm}$ --- dezenas de
milhares de vezes mais curto: microscópios eletrônicos criam imagens de vírus, células
máquinas, mesmo simples \omterm{def:g11:fundamental-interactions:ladder}{átomos}. O onda de matéria deu à física um novo olhar.
\end{example}

\section{Problema: O Laboratório de Física do Painel Solar}

\begin{problem}[{Problema de fim de semana --- o laboratório de física do painel solar: o
$10^{21}$ grãos que um telhado bebe a cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}, uma fotocélula que mede
\omterm{def:g12:quantum-world:photon}{Constante de Planck}, uma lâmpada de hidrogênio lida linha por linha --- um fim de semana, um $ h $}]
\label{pb:g12:quantum-world:1}
Sua turma instrumenta o painel solar do telhado da escola durante um fim de semana.
Dados: irradiância solar ao meio-dia \qty{1.0e3}{W/m^2}; significa solar \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}
\qty{550}{nm}; área do painel \qty{1.0}{m^2}, \omterm{def:g10:energy-conservation:efficiency}{eficiência} $20\%$; o dia
entrega o equivalente a \qty{5.0}{h} de pleno sol.

\medskip
\noindent\textbf{Part I --- Fóton bookkeeping.}
\begin{enumerate}
  \item Calcule o energia de um solar médio fóton, em \omterm{def:g11:work-of-force:work}{joules} e elétronvolts.
  \item Deduza o número de \omterm{def:g12:quantum-world:photon}{fótons} atingindo o painel por \omterm{def:g10:orders-of-magnitude:unit}{segundo} ao meio-dia.
  \item Calcule o painel \omterm{prop:g11:circuits-and-power:power}{energia elétrica} ao meio-dia, então o energia
        produzido ao longo do dia, em \unit{kW.h}.
  \item A corrente do painel parece perfeitamente suave. Usando a pergunta 2,
        explique por que a chegada grão por grão é invisível.
  \item A luz solar carrega mais \omterm{def:g10:light-spectra:wavelength}{infravermelho} energia que \omterm{def:g10:light-spectra:wavelength}{ultravioleta}, ainda apenas
        \omterm{def:g10:light-spectra:wavelength}{ultravioleta} queima a pele (ligações $\sim \qtyrange{3}{4}{eV}$). Explique em grãos.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Measuring $ h $ with the kit's photocell.}
A fotocélula do kit possui cátodo desconhecido. \omterm{def:g11:color-light-sources:subtractive}{Filtros} dar:
\begin{center}\small
\begin{tabular}{lcccc}
$\lambda $ (\unit{nm}) & 365 & 405 & 436 & 546\\
$ U_s $ (\unit{V}) & 1.50 & 1.16 & 0.94 & 0.37
\end{tabular}
\end{center}
\begin{enumerate}[resume]
  \item O que o potencial de parada medir? Relacionar $ U_s $, $\nu $, $ W_0$, $ h $.
  \item Converta os quatro comprimentos de onda para \omterm{def:g12:oscillators-and-time:oscillator}{frequências} (em $10^{14}\,\unit{Hz}$).
  \item Por que a conspiração deveria $ eU_s $ contra $\nu $ dar uma linha reta?
        Verifique o alinhamento dos pontos.
  \item A partir dos pontos extremos, calcule a inclinação: sua medida $ h $.
  \item Deduza o cátodo função de trabalho (\unit{eV}), limite
        \omterm{def:g12:oscillators-and-time:oscillator}{frequência} e limite \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}.
  \item Que parte da luz solar poderia ejetar elétrons deste
        cátodo --- e qual metade do \omterm{def:g12:sound-acoustics:timbre}{espectro} é inútil para isso?
  \item Compare o seu $ h $ para \qty{6.63e-34}{J.s}: erro relativo em porcentagem?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hydrogen calibration lamp.}
O kit \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} a escala é verificada contra uma lâmpada de hidrogênio,
$ E_n = -13.6/n^2$ \unit{eV}.
\begin{enumerate}[resume]
  \item Calcular $ E_2$, $ E_3$, $ E_4$ e $ E_5$.
  \item Calcule o fóton energias das transições $3 \to 2$, $4 \to 2$, $5 \to 2$.
  \item Deduza os três comprimentos de onda e dê a cor de cada linha.
  \item Por que a lâmpada emite \emph{linhas} em vez de um contínuo
        arco-íris? Uma frase, nomeando a ideia principal.
  \item O hidrogênio da lâmpada poderia absorver luz em \qty{500}{nm}? Justifique com a escada.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Inspecting the cells: the electron \omterm{def:g10:signals-and-waves:wave}{aceno}.}
Microfissuras em uma célula ficam muito abaixo do \qty{0.5}{\micro m} alcance da luz
microscópios; o microscópio eletrônico de varredura do laboratório funciona em \qty{10}{kV}.
\begin{enumerate}[resume]
  \item Calcule a velocidade dos elétrons; verifique se fica abaixo $0.25\,c $ (fórmula clássica).
  \item Calcule seus \omterm{def:g12:newtons-laws:momentum}{impulso} e \omterm{def:g12:quantum-world:debroglie}{comprimento de onda de de Broglie}.
  \item Escalas de resolução com \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}: calcule o \omterm{def:g10:signals-and-waves:oscilloscope}{ganho} sobre a luz verde
        (\qty{550}{nm}); conclua com os dois números do fim de semana --- seu $ h$
        e esse \omterm{def:g10:signals-and-waves:oscilloscope}{ganho}.
\end{enumerate}
\end{problem}
