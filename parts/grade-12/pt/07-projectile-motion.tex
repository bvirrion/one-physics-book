\chapter{Queda Livre e Movimento de Projéteis}\label{ch:g12:projectile-motion}

Cada jato de uma mangueira de jardim desenha a mesma curva no ar. O mesmo acontece com um
basquete fazendo um arco em direção ao aro, um saltador em distância, uma bala de canhão: qualquer corpo
abandonado ao seu \omterm{def:g10:universal-gravitation:weight}{peso}. Este capítulo ganha aquela curva --- uma parábola ---
de Segunda lei de Newton em duas linhas de cálculo, lê-se ápice, voo
tempo e \omterm{def:g12:projectile-motion:range}{faixa} fora de sua equação, então deixa o ar voltar para ver o que
isso estraga.

\section{Queda livre}

\begin{definition}[Queda livre]\label{def:g12:projectile-motion:freefall}
Um corpo está em \emph{queda livre}\index{queda livre} quando o único \omterm{def:g10:inertia:force}{vigor} agindo em
é o seu \omterm{def:g10:universal-gravitation:weight}{peso} $\vect P = m\vect g $: sem suporte, sem corda, \omterm{def:g11:circuits-and-power:resistance}{resistência} do ar
insignificante. Uma pedra caída ou uma bola que saiu da mão é, muito bem
precisão, em queda livre --- ``queda'' inclui subida.
\end{definition}

\begin{theorem}[Lei da queda livre de Galileu]\label{thm:g12:projectile-motion:galileo}
\index{Lei da queda livre de Galileu}
Em \omterm{def:g12:projectile-motion:freefall}{queda livre}, todos os corpos compartilham a mesma aceleração, qualquer que seja sua massa:
\[
\vect a = \vect g, \qquad g = \qty{9.81}{m/s^2},
\]
direcionado diretamente para baixo: o campo força \qty{9.81}{N/kg} de
\cref{ch:g10:universal-gravitation} em sua outra roupa, $\unit{N/kg} = \unit{m/s^2}$.
\end{theorem}

\begin{proof}
Segunda lei de Newton (\cref{ch:g12:newtons-laws}) com o \omterm{def:g10:universal-gravitation:weight}{peso} como apenas
\omterm{def:g10:inertia:force}{vigor}: $ m\vect a = m\vect g $, e a massa é cancelada. Martelo e pena
diferem apenas pelo ar.
\end{proof}

\begin{proposition}[Queda vertical do repouso]\label{prop:g12:projectile-motion:vertical}
Caiu do repouso, com a profundidade caída $ h $ contado para baixo a partir do
ponto de liberação, um corpo em \omterm{def:g12:projectile-motion:freefall}{queda livre} obedece
\[
a = g, \qquad v = g t, \qquad h = \tfrac12\, g t^2 .
\]
\end{proposition}

\begin{proof}
$ v(t)$ é a antiderivada da constante $ a = g $ desaparecendo em $ t = 0$
(\cref{ch:g12:kinematics-2d}); $ h(t)$ o de $ v = gt $ desaparecendo em $ t = 0$.
\end{proof}

\begin{example}[Ordens de magnitude]\label{ex:g12:projectile-motion:orders}
Depois $1$, $2$, $3$ \omterm{def:g10:orders-of-magnitude:unit}{segundos} de \omterm{def:g12:projectile-motion:freefall}{queda livre}: $ v = 9.8$, $19.6$, \qty{29.4}{m/s}
($35$, $71$, \qty{106}{km/h}) e $ h = 4.9$, $19.6$, \qty{44.1}{m}. De um
\qty{30}{m} telhado: $ t = \sqrt{2h/g} \approx \qty{2.5}{s}$, chegando em
\qty{24}{m/s} $\approx \qty{87}{km/h}$. Os tempos crescem como $\sqrt h $ --- e
passados alguns \omterm{def:g10:orders-of-magnitude:unit}{segundos} o ar recusa-se a permanecer insignificante (última secção).
\end{example}

\begin{example}[Lançamento vertical]\label{ex:g12:projectile-motion:throw}
Jogado direto para cima $ v_0$, eixo $ Oy $ para cima a partir da mão: os mesmos dois
antiderivadas dão $ v = v_0 - gt $ e $ y = v_0 t - \tfrac12 gt^2$. O
ápice, onde $ v = 0$, chega em $ t = v_0/g $, altura $ h = v_0^2/2g $ --- o
energia responder $ v_0 = \sqrt{2gh}$ de \cref{ch:g11:mechanical-energy} correr
para trás; jogado \emph{abaixo}, $ v_0$ vira \omterm{def:g10:signals-and-waves:signal}{sinal}. Para \qty{15}{m/s}: ápice
\qty{11.5}{m} no \qty{1.53}{s}, de volta às \qty{3.06}{s}, no \qty{15}{m/s}.
\end{example}

\section{O projétil: uma lei, duas antiderivadas}

\begin{definition}[Projétil]\label{def:g12:projectile-motion:projectile}
UM \emph{projeto}\index{projeto} é um corpo lançado com velocidade
$\vect v_0$ e depois saiu \omterm{def:g12:projectile-motion:freefall}{queda livre}. Eixos no ponto de lançamento: $ Ox $
horizontal no plano \omterm{def:g10:universal-gravitation:weight}{vertical} de $\vect v_0$, $ Oy $ \omterm{def:g10:universal-gravitation:weight}{vertical} para cima; o
\emph{ângulo de lançamento}\index{ângulo de lançamento} $\alpha $ de $\vect v_0$ acima do
horizontal fixa suas coordenadas $(v_0\cos\alpha,\; v_0\sin\alpha)$.
\end{definition}

\begin{theorem}[Equações de movimento]\label{thm:g12:projectile-motion:equations}
UM projétil lançado da origem com velocidade $ v_0$ e ângulo $\alpha $
tem $\vect a = (0,\,-g)$ a cada instante, portanto
\[
\vect v(t) = \bigl(v_0\cos\alpha,\; v_0\sin\alpha - g t\bigr),
\qquad
\begin{cases}
x(t) = v_0\cos\alpha\; t,\\[2pt]
y(t) = v_0\sin\alpha\; t - \tfrac12\, g t^2 .
\end{cases}
\]
\end{theorem}

\begin{proof}
Segunda lei de Newton dá $\vect a = \vect g = (0,-g)$. Antiderivadas
eixo por eixo: $\vect v(t)$ de $\vect a $ com valor inicial
$(v_0\cos\alpha,\, v_0\sin\alpha)$, em seguida, posicione de $\vect v $, de $(0,0)$.
\end{proof}

\begin{proposition}[Independência dos dois movimentos]\label{prop:g12:projectile-motion:independence}
O movimento horizontal é uniforme --- $ v_x $ nunca muda --- e o
o movimento \omterm{def:g10:universal-gravitation:weight}{vertical} é exatamente o lance \omterm{def:g10:universal-gravitation:weight}{vertical} da seção anterior:
nenhuma das equações menciona a coordenada da outra. Em particular, uma bola
disparada horizontalmente e uma bola caiu no mesmo instante do mesmo
altura permanecem nivelados um com o outro --- eles pousam juntos.
\end{proposition}

\begin{proof}
Em \cref{thm:g12:projectile-motion:equations}, $ x(t)$ envolve apenas
$ v_0\cos\alpha $, $ y(t)$ apenas $ v_0\sin\alpha $ e $ g $; ambas as bolas têm
$ v_y(0) = 0$, então ambos obedecem $ h = \tfrac12 gt^2$.
\end{proof}

\begin{remark}[O macaco caindo]\label{rem:g12:projectile-motion:monkey}
Um tratador aponta um dardo \emph{direto para} um macaco, que se solta
ramo no tiro. Movimento ruim: dardo e macaco a cada queda $\tfrac12 gt^2$
abaixo de suas posições sem gravidade --- a linha de mira, o galho --- então
eles se encontram exatamente, qualquer que seja a velocidade do dardo
(\cref{exo:g12:projectile-motion:15}).
\end{remark}

\section{A parábola, leia na íntegra}

\begin{theorem}[Trajetória]\label{thm:g12:projectile-motion:trajectory}
Eliminando $ t $ entre $ x(t)$ e $ y(t)$: o \omterm{def:g10:relative-motion:trajectory}{trajetória} de um projétil
($0 \leq \alpha < \ang{90}$) é a parábola de abertura para baixo
\[
y = x\tan\alpha \;-\; \frac{g}{2 v_0^2 \cos^2\alpha}\, x^2 .
\]
\end{theorem}

\begin{proof}
$ t = x/(v_0\cos\alpha)$, substituído em $ y(t)$ de
\cref{thm:g12:projectile-motion:equations}.
\end{proof}

\begin{definition}[Faixa]\label{def:g12:projectile-motion:range}
Por um projétil sobre terreno plano, o \emph{faixa}\index{faixa} $ R $ é
a distância horizontal percorrida quando retorna à altura de lançamento; o
\emph{hora do voo}\index{hora do voo} $ t_f $ é a duração dessa viagem.
\end{definition}

\begin{proposition}[Apex, tempo de voo, alcance]\label{prop:g12:projectile-motion:range}
Sobre terreno plano, com $0 < \alpha < \ang{90}$:
\[
t_f = \frac{2 v_0 \sin\alpha}{g}, \qquad
h_{\text{apex}} = \frac{(v_0\sin\alpha)^2}{2g}
\ \ \text{reached at } t_f/2, \qquad
R = \frac{v_0^2 \sin(2\alpha)}{g}.
\]
O \omterm{def:g12:projectile-motion:range}{faixa} é maior em $\alpha = \ang{45}$, onde $ R = v_0^2/g $;
ângulos \omterm{prop:g11:color-light-sources:complementary}{complementares} $\alpha $ e $\ang{90} - \alpha $ dê o mesmo \omterm{def:g12:projectile-motion:range}{faixa}.
\end{proposition}

\begin{proof}
$ y = 0$ dá $ t\,(v_0\sin\alpha - \tfrac12 gt) = 0$: lançamento e $ t_f $.
O ápice, $ v_y = 0$, senta-se em $ t = v_0\sin\alpha/g = t_f/2$; substituir em
$ y(t)$. Então $ R = x(t_f) = 2v_0^2\sin\alpha\cos\alpha/g =
v_0^2\sin(2\alpha)/g $. O seno atinge o pico em $2\alpha = \ang{90}$, e
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180} - 2\alpha) =
\sin(2\alpha)$.
\end{proof}

\begin{example}[Chute de meta]\label{ex:g12:projectile-motion:kick}
Um goleiro ataca $ v_0 = \qty{25}{m/s}$, $\alpha = \ang{45}$:
$ R = 25^2/9.81 \approx \qty{64}{m}$; autorizações reais chegam mais perto
\qty{55}{m} --- o ar está cortado.
\end{example}

\begin{method}[Escrituração de projéteis]\label{met:g12:projectile-motion:method}
\begin{enumerate}
  \item Eixos no ponto de lançamento, $ Ox $ horizontal, $ Oy $ acima; coordenadas
        de $\vect v_0$.
  \item Verificar \omterm{def:g12:projectile-motion:freefall}{queda livre} (\omterm{def:g10:universal-gravitation:weight}{peso} somente): então $\vect a = (0,-g)$;
        primitivas duas vezes, condições iniciais de cada vez.
  \item Pergunta sobre um \emph{lugar}: eliminar $ t $ (\omterm{def:g10:relative-motion:trajectory}{trajetória}). Sobre um
        \emph{instantâneo}: manter $ t $.
  \item Dicionário: ápice $\Leftrightarrow v_y = 0$; altura de lançamento
        $\Leftrightarrow y = 0$; parede em $ d \Leftrightarrow x = d $; chão
        $\ell $ abaixo $\Leftrightarrow y = -\ell $.
  \item Sanidade: vôo simétrico em relação ao ápice; $ R \leq v_0^2/g $; apenas ar
        sempre encurta.
\end{enumerate}
\end{method}

\section{O que o ar muda}

\begin{definition}[Velocidade terminal]\label{def:g12:projectile-motion:terminal}
Ar \omterm{def:g11:circuits-and-power:resistance}{resistência} se opõe à velocidade e cresce com velocidade. Um corpo caindo
portanto, acelera apenas até que o arrasto se equilibre \omterm{def:g10:universal-gravitation:weight}{peso}; de lá
$\vect a = \vect 0$: cai em uma constante \emph{terminal
velocidade}\index{terminal
velocidade}.
\end{definition}

\begin{example}[Ordens de magnitude terminais]\label{ex:g12:projectile-motion:airorders}
Um paraquedista completo: cerca de \qty{55}{m/s} (\qty{200}{km/h}). Uma gota de chuva:
sobre \qty{9}{m/s} --- no vácuo, \qty{2000}{m} do outono iria entregá-lo em
$\sqrt{2gh} \approx \qty{2.0e2}{m/s}$. Uma pena: cerca de \qty{1}{m/s},
alcançado em centímetros --- o segredo de sua lentidão: ele gasta seu
outono inteiro \emph{no} velocidade terminal; o martelo nunca chega perto do seu.
\end{example}

\begin{remark}[Balística real]\label{rem:g12:projectile-motion:drag}
O arrasto empurra contra a velocidade durante todo o vôo: o \omterm{def:g12:projectile-motion:range}{faixa} encurta,
o ápice desce e a simetria se quebra --- a descida é mais íngreme que a
subida, planos gerais visivelmente aquém da parábola. O honesto, quantitativo
o tratamento é uma equação diferencial resolvida no volume do Ano 1; isso
ano, a parábola é o modelo certo para dados densos e moderadamente rápidos.
projéteis --- uma bola de basquete em \qty{10}{m/s}, não uma bola de golfe em
\qty{70}{m/s}.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:1}
Uma pedra caída de uma ponte atinge a água \qty{2.0}{s} mais tarde: altura de
a ponte e a velocidade de impacto em \unit{m/s} e \unit{km/h}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:2}
Na Lua, um astronauta deixou cair um martelo e uma pena: eles
pousaram juntos. Por que --- e por que a pena perde tanto na Terra?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:3}
Desprezando o ar, calcule o tempo de queda e a velocidade de chegada \qty{1.0}{m},
\qty{10}{m}, \qty{100}{m}. Como os tempos crescem com a altura?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:4}
Uma bola é lançada diretamente para cima em \qty{15}{m/s}: altura e tempo do ápice,
tempo total de volta para a mão e velocidade de retorno?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:5}
Uma bola de gude rola de uma mesa alta \qty{0.80}{m} no \qty{2.5}{m/s}. Encontrar
o tempo de queda e a distância de pouso da borda. A hora do outono
depende da velocidade da bola de gude?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:6}
Da figura de cinco ângulos do curso: quais pares de ângulos de lançamento
compartilhe seus \omterm{def:g12:projectile-motion:range}{faixa}, e qual ângulo vence? Por um \omterm{def:g12:projectile-motion:range}{faixa} abaixo do máximo,
qual dos dois ângulos possíveis voa mais? Justifique com $ t_f $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:7}
De um \qty{20}{m} varanda, uma bola é lançada em linha reta \emph{abaixo} no
\qty{5.0}{m/s}. Encontre o tempo até o solo e a velocidade do impacto; verificar
este último com um energia equilíbrio (\cref{ch:g11:mechanical-energy}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:8}
UM projétil deixa o terreno plano em \qty{40}{m/s}, $\alpha = \ang{30}$:
hora do vôo, altura do ápice e \omterm{def:g12:projectile-motion:range}{faixa}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:9}
Modele um saltador em distância como um projétil: decolagem em \qty{10.5}{m/s},
$\alpha = \ang{20}$. Calcule o \omterm{def:g12:projectile-motion:range}{faixa}; campeões saltam quase \qty{9}{m}
--- nomeie dois ingredientes que o modelo de pontos deixa de fora.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:10}
No \qty{1.2}{m} acima do solo plano, uma bala é lançada e uma bala idêntica
um disparou horizontalmente contra \qty{300}{m/s}. Qual pousa primeiro? Calcule o
tempo de queda e distância de pouso da bala disparada. Que princípio é
testado?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:11}
Mostre algebricamente que $\alpha $ e $\ang{90} - \alpha $ compartilhe seus

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:12}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:13}
Uma bola é chutada do chão em \qty{18}{m/s}, $\alpha = \ang{35}$,

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:14}

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:15}
O macaco em queda, provou: um dardo sai da origem apontado exatamente para um
macaco à distância $ D $, altura $ H $ ($\tan\alpha = H/D $), que cai em
$ t = 0$. Mostre que quando o dardo atingir $ x = D $ sua altura é igual a
macaco --- tanto faz $ v_0$.
\end{exercise}

\section{Problema: a buzina do basquete}

\begin{problem}[{Problema de fim de semana --- a campainha do basquete: um jogo grátis
lançar flecha dissecada por flecha, as pontas dos dedos afastadas em centímetros, um
levantamento de quadra inteira pesado contra o que um braço pode dar, um lance sobre o
tabela --- e um veredicto sobre a cena mais teatral do esporte}]
\label{pb:g12:projectile-motion:1}
A borda de uma cesta de basquete é um anel horizontal \qty{3.05}{m} acima do
chão; Nora libera cada tiro em \qty{2.00}{m}, então com eixos no
ponto de liberação ($ Ox $ horizontal em direção ao aro, $ Oy $ para cima) o centro do aro
senta \qty{1.05}{m} acima da origem. O ar é negligenciado, exceto onde indicado.

\medskip\noindent\textbf{Part I --- The free throw.} Nora atira de
$ d = \qty{4.20}{m}$ (horizontal) com \omterm{def:g12:projectile-motion:projectile}{ângulo de lançamento} $\alpha = \ang{50}$.
\begin{enumerate}
  \item O que forças agir uma vez liberado? Dar $\vect a $, citando a lei utilizada.
  \item Derive as coordenadas de $\vect v(t)$.
  \item Derivar $ x(t)$ e $ y(t)$.
  \item Eliminar $ t $ para obter o \omterm{def:g10:relative-motion:trajectory}{trajetória} $ y(x)$.
  \item A bola deve passar pelo centro do aro, $ y(d) = \qty{1.05}{m}$: resolver
        para $ v_0$, em \unit{m/s} e \unit{km/h}.
\end{enumerate}

\medskip\noindent\textbf{Part II --- Over the fingertips, into the ring.}
\begin{enumerate}[resume]
  \item Quanto tempo a bola leva para chegar ao aro?
  \item Encontre o tempo e a altura do ápice; verifique se a bola já está
        descendo na borda.
  \item Um defensor \qty{0.90}{m} na frente de Nora estica as pontas dos dedos para
        \qty{2.60}{m}. Por quanto a bola os ultrapassa?
  \item Calcular $ v_x $ e $ v_y $ na borda, a velocidade ali e o ângulo
        da velocidade abaixo da horizontal.
  \item Por que uma bola deve chegar \emph{descendo} --- e por que íngreme?
\end{enumerate}

\medskip\noindent\textbf{Part III --- The full-court heave.} Campainha sobre
para som,Nora é \qty{26.0}{m} da borda distante.
\begin{enumerate}[resume]
  \item O \qty{1.05}{m} aumento é pequeno contra \qty{26.0}{m}: justificar um
        modelo de terreno plano e derivar o hora do vôo de $ y(t_f) = 0$.
  \item Deduzir $ R = v_0^2\sin(2\alpha)/g $ e o ângulo de maior \omterm{def:g12:projectile-motion:range}{faixa}.
  \item Calcule o $ v_0$ necessário para cobrir \qty{26.0}{m} no $\alpha = \ang{45}$.
  \item Um jogador \omterm{def:g11:fundamental-interactions:strong}{forte} pode arremessar uma bola de basquete a cerca de \qty{17.5}{m/s}.
        Qual é o máximo dela \omterm{def:g12:projectile-motion:range}{faixa} --- o tiro é humanamente viável?
  \item No \qty{17.5}{m/s}, encontre o \emph{dois} ângulos pousando a bola
        \qty{26.0}{m} de distância e seus horários de voo: com \qty{2.0}{s} ligado
        o relógio, qual deles bate a campainha?
\end{enumerate}

\medskip\noindent\textbf{Part IV --- The lob over the backboard.} De
atrás da linha de base Nora lobs: lançamento \qty{6.0}{m} do centro da borda,
$\alpha = \ang{75}$; o plano da prancha cruza sua tacada \qty{0.45}{m}
antes do centro do aro, borda superior \qty{3.95}{m} acima do chão.
\begin{enumerate}[resume]
  \item Encontre o $ v_0$ que deixa cair a bola pelo centro do aro.
  \item Calcule a altura da bola no plano do tabuleiro
        ($ x = \qty{5.55}{m}$): limpa a borda superior, em quanto?
  \item Encontre a altura do ápice e o tempo de voo  por que as multidões adoram?
  \item Qualitativamente, o que muda a \omterm{def:g11:circuits-and-power:resistance}{resistência} do ar nesses três
        tiros, e qual deles ameaça mais?
  \item \emph{O veredicto da campainha}, duas frases: compare a velocidade
        o tiro em quadra inteira exige o que um braço fornece, dê o
        margem e regra --- possível ou mito?
\end{enumerate}
\end{problem}
