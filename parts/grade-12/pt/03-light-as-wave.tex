\chapter{Luz como onda: difração e interferência}\label{ch:g12:light-as-wave}

Incline uma película de sabão em direção a uma janela: as cores deslizam sobre ela
nada no sabão ou na água é colorido. Aponte um \omterm{def:g11:color-light-sources:lamps}{laser} em um único cabelo:
a parede atrás não mostra um cabelo fino sombra, mas uma \omterm{def:g11:fundamental-interactions:strong}{faixa} brilhante,
barrado com linhas escuras, milhares de vezes mais largas que o cabelo.
Raios de marcha reta (\cref{ch:g10:refraction}) não explique nenhum dos dois
vista. Este capítulo promove a luz para uma plena \omterm{def:g10:signals-and-waves:wave}{aceno} --- meio
micrômetro de crista a crista --- então aproveita a ideia: um \omterm{def:g11:color-light-sources:lamps}{laser}
e uma parede nua medirá um fio de cabelo, um \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}, as ranhuras de um CD.

\section{A luz é uma onda}

\begin{definition}[Ondas de luz]\label{def:g12:light-as-wave:lightwave}
Luz monocromática --- uma cor pura, que nenhum prisma pode dividir
mais adiante (\cref{ch:g10:light-spectra}) --- é um \omterm{def:g10:signals-and-waves:period}{periódico} \emph{luz
onda}\index{luz
onda}: atravessa o vácuo em $c = \qty{3.00e8}{m/s}$
e caracteriza-se pela sua freqüência $ f $ ou seu \emph{comprimento de onda em
descartável}\index{comprimento de onda em
descartável} $\lambda = c/f $, o
distância crista a crista de \cref{ch:g12:mechanical-waves}. O olho
responde de cerca de \qty{400}{nm} (violeta) para \qty{800}{nm} (vermelho);
\omterm{def:g10:light-spectra:white}{luz branca} mistura tudo \omterm{def:g11:fundamental-interactions:strong}{faixa}.
\end{definition}

\begin{example}[Ordens de magnitude]\label{ex:g12:light-as-wave:orders}
Um hélio-néon \omterm{def:g11:color-light-sources:lamps}{laser} emite luz vermelha de $\lambda = \qty{633}{nm} =
\qty{6.33e-7}{m}$, então $ f = c/\lambda \approx \qty{4.7e14}{Hz}$: não
eletrônicos contam tão rápido --- cada \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} neste capítulo iremos
ser medido pela geometria. Para escala, um fio de cabelo (cerca de \qty{70}{\micro m})
é cem comprimentos de onda entre. O que ondas? Não o ar: um elétrico
e \omterm{def:g11:magnetic-fields:bfield}{campo magnético}, descrito honestamente no volume do Ano 2; apenas o
\omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} assuntos abaixo.
\end{example}

\section{Difração}

\begin{definition}[Difração]\label{def:g12:light-as-wave:diffraction}
Quando um \omterm{def:g10:signals-and-waves:wave}{aceno} encontra uma abertura ou um obstáculo de tamanho $ a $ comparável a
é \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}, ele se espalha pela região que os raios deixariam escuro.
Essa disseminação é \emph{difração}\index{difração} --- insignificante
enquanto $\lambda/a $ é minúsculo, dominante como $ a $ encolhe em direção $\lambda $.
\end{definition}

\begin{proposition}[Meia largura angular do spread]\label{prop:g12:light-as-wave:halfwidth}
Uma fenda de largura $ a $, iluminado com \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} $\lambda < a $, espalha o
luz sobre a direção incidente dentro da meia largura angular
$\theta $ (radianos); uma tela à distância $ D $ mostra um centro brilhante
\omterm{def:g11:fundamental-interactions:strong}{faixa} de largura $ L $, duas vezes $ D \tan\theta \approx D\,\theta $ (pequeno
ângulo):
\[
\theta \approx \frac{\lambda}{a}, \qquad L = \frac{2\lambda D}{a}.
\]
\end{proposition}
\admitted

\begin{remark}[Leia a fórmula ao contrário]\label{rem:g12:light-as-wave:honest}
O \emph{mais estreito} a fenda, o \emph{mais amplo} a propagação --- o
oposto da previsão do raio. A derivação honesta (somando
wavelets reemitidas através da fenda) vive no volume do Ano 2; aqui
a fórmula é uma ferramenta de medição.
\end{remark}

\begin{example}[Um laser através de uma fenda]\label{ex:g12:light-as-wave:slit}
Vermelho \omterm{def:g11:color-light-sources:lamps}{laser}, $\lambda = \qty{633}{nm}$; fenda $ a = \qty{0.10}{mm}$;
tela em $ D = \qty{2.00}{m}$. Então $\theta = \num{6.33e-7} /
\num{1.0e-4} = \qty{6.3e-3}{rad}$ --- um terço de grau, mas o
\omterm{def:g11:fundamental-interactions:strong}{faixa} central na parede é $ L = 2 \times \num{6.33e-7} \times 2.00 /
\num{1.0e-4} \approx \qty{2.5}{cm}$ largo, flanqueado por linhas escuras e
\omterm{def:g11:fundamental-interactions:strong}{faixas} laterais mais fracas. Divida a fenda pela metade e a banda dobra.
\end{example}

\begin{proposition}[Um obstáculo difrata como uma fenda]\label{prop:g12:light-as-wave:hair}
UM \omterm{def:g11:lenses-and-eye:lens}{afinar} obstáculo opaco de largura $ d $ --- um fio, um cabelo --- produz,
fora do feixe incidente, o mesmo padrão de uma fenda de largura $ d $
(Princípio de Babinet, derivado no volume do Ano ~ 2).
\end{proposition}
\admitted

\begin{method}[Medindo uma pequena largura]\label{met:g12:light-as-wave:hairwidth}
\begin{enumerate}
  \item Estique o cabelo ao longo \omterm{def:g11:color-light-sources:lamps}{laser} feixe; definir uma tela em um
        distância medida $ D $ (um pouco metros).
  \item Meça a largura da \omterm{def:g11:fundamental-interactions:strong}{faixa} central $ L $, linha escura a linha escura.
  \item Desde $ L = 2\lambda D/d $: $ d = 2\lambda D/L $ --- cabelos mais finos,
        padrão maior.
\end{enumerate}
\end{method}

\begin{remark}[A difração limita todos os instrumentos]\label{rem:g12:light-as-wave:limits}
A luz entra numa câmera, num telescópio ou num olho através de uma abertura de
largura $ a $, então mesmo um perfeito \omterm{def:g11:lenses-and-eye:lens}{lente} desfoca um ponto em um ponto de
tamanho angular sobre $\lambda/a $: detalhes mais próximos se fundem. Uma luz visível
microscópio, portanto, não resolve nada muito abaixo $\lambda \approx
\qty{0.5}{\micro m}$ (bactérias sim, vírus não); espelhos telescópicos
são construídos amplamente pelo mesmo motivo.
\end{remark}

\section{Interferência}

\begin{definition}[Fontes coerentes, diferença de caminho]\label{def:g12:light-as-wave:coherent}
Dois \omterm{def:g10:signals-and-waves:wave}{aceno} fontes são \emph{consistente}\index{consistente} quando
eles têm o mesmo freqüência e vibrar no passo --- na prática,
quando são duas cópias de uma \omterm{def:g10:signals-and-waves:wave}{aceno}, obtido dividindo-o. Por um
ponto $ M $ alcançado por ambos, o \emph{diferença de caminho}%
\index{diferença de caminho} é $\delta = S_2M - S_1M $.
\end{definition}

\begin{proposition}[Condições de interferência]\label{prop:g12:light-as-wave:conditions}
Onde dois coerente ondas de \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} $\lambda $ sobrepõem eles adicionam ---
eles \emph{interferir}\index{interferir}:
\begin{itemize}
  \item $\delta = k\lambda $ ($ k $ inteiro): na etapa, reforçando ---
        \emph{construtivo}\index{construtivo} interferência (brilhante);
  \item $\delta = (k + \tfrac12)\lambda $: crista na calha, cancelando
        --- \emph{destrutivo}\index{destrutivo} interferência (escuro).
\end{itemize}
\end{proposition}

\begin{proof}
Mudando um \omterm{def:g12:mechanical-waves:periodic}{onda periódica} por inteiro comprimentos de onda não muda nada: cristas
empilhar em cristas. Metade de um \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} mais coloca a crista na calha: igual
ondas cancelar.
\end{proof}

\begin{example}[Experiência de duas fendas de Young]\label{ex:g12:light-as-wave:young}
Faça duas fendas finas $ S_1$, $ S_2$, com uma fração de milímetro de distância,
em uma placa opaca e acenda ambos com um \omterm{def:g11:color-light-sources:lamps}{laser}: dois coerente
fontes. Em uma tela alguns metros a sobreposição é listrada:
no centro $\delta = 0$, brilhante; subindo na tela, $\delta $
cresce através $\lambda/2$ (escuro), $\lambda $ (brilhante)\,\dots\ escuridão
onde a luz adicionada à luz é cancelada.
\end{example}

\begin{definition}[Franjas e interfranjas]\label{def:g12:light-as-wave:fringes}
As \omterm{def:g11:fundamental-interactions:strong}{faixas} claras e escuras alternadas do padrão são a sua
\emph{franjas de interferência}\index{franjas de interferência}; a distância
$ i $ entre centros brilhantes consecutivos é o \emph{interferir}%
\index{interferir}.
\end{definition}

\begin{proposition}[A interfranja]\label{prop:g12:light-as-wave:interfringe}
Para fendas à distância $ a $ separados e uma tela à distância $ D \gg a $,
o ponto da tela à distância $ x $ do centro $ O $ tem caminho
diferença $\delta \approx a\,x/D $ (geometria admitida); por
\cref{prop:g12:light-as-wave:conditions}, franjas brilhantes ficam em
$ x_k = k\,\lambda D/a $, espaçados uniformemente com \omterm{def:g12:light-as-wave:fringes}{interferir}
\[
i = \frac{\lambda D}{a}.
\]
Com $\lambda = \qty{633}{nm}$, $ a = \qty{0.20}{mm}$, $ D =
\qty{2.0}{m}$: $ i = \qty{6.3}{mm}$ --- a configuração amplia $\lambda $
por $ D/a = 10^4$. A geometria por trás disso é elaborada no Ano ~2
volume.
\end{proposition}
\admitted

\begin{method}[Medindo um comprimento de onda]\label{met:g12:light-as-wave:lambda}
\begin{enumerate}
  \item Envie o desconhecido \omterm{def:g11:color-light-sources:lamps}{laser} através de uma fenda dupla de espaçamento conhecido
        $ a $; coloque a tela a uma distância medida $ D $.
  \item Meça o intervalo de $10$ interfere; dividir por $10$: o
        \omterm{def:g12:light-as-wave:fringes}{interferir} $ i $, com o erro da régua reduzido dez vezes.
  \item Então $\lambda = i\,a/D $.
\end{enumerate}
\end{method}

\begin{remark}[Por que duas lâmpadas nunca interferem]\label{rem:g12:light-as-wave:lamps}
Duas luminárias de mesa iluminando uma parede não produzem franjas: cada uma emite
curto, não correlacionado \omterm{def:g10:signals-and-waves:wave}{aceno} trens cujo tempo relativo salta aleatoriamente
um bilhão de vezes por \omterm{def:g10:orders-of-magnitude:unit}{segundo}, então as franjas mudam com a mesma rapidez e o
o olho calcula a média deles em luz uniforme. Demandas de interferência coerência
--- um \omterm{def:g10:signals-and-waves:wave}{aceno} dividido em dois, como fazem as fendas de Young; o \omterm{def:g11:color-light-sources:lamps}{laser} é o
atalho moderno, um limpo e longo \omterm{def:g10:signals-and-waves:wave}{aceno}.
\end{remark}

\section{Cores pintadas por interferência}

\begin{definition}[Iridescência]\label{def:g12:light-as-wave:iridescence}
Cores que mudam com o ângulo de visão --- bolhas de sabão, manchas de óleo,
penas de pavão, verso de um CD --- são \emph{iridescência}%
\index{iridescência}: sem pigmento, mas interferência em \omterm{def:g10:light-spectra:white}{luz branca},
reforçando alguns comprimentos de onda e cancelando outros.
\end{definition}

\begin{example}[Filmes de sabão e manchas de óleo]\label{ex:g12:light-as-wave:soap}
A luz que atinge uma película de sabão reflete parcialmente na sua face frontal, parcialmente
de costas: dois coerente cópias cujo \omterm{def:g12:light-as-wave:coherent}{diferença de caminho} é definido pelo
espessura e o ângulo de visão. Cada espessura retorna algum
comprimentos de onda reforçados, outros cancelados: o filme reflete uma cor
que desliza enquanto drena e afina. Uma mancha de óleo no asfalto molhado toca
o mesmo truque.
\end{example}

\begin{remark}[Luz monocromática versus luz branca]\label{rem:g12:light-as-wave:white}
Em \omterm{def:g10:light-spectra:white}{luz branca} todo \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} desenha suas próprias franjas jovens com seu
próprio $ i = \lambda D/a $: no centro todas as cores são brilhantes (uma branca
franja); algumas franjas os padrões se separam --- iridescentes
bordas e depois desfocar. Luz monocromática dá franjas nítidas sem
fim: daí o \omterm{def:g11:color-light-sources:lamps}{laser} em cada medição precisa aqui.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:light-as-wave:1}
Calcule o freqüência de luz vermelha ($\lambda = \qty{700}{nm}$) e de
luz violeta (\qty{400}{nm}). Qual é mais alto?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:light-as-wave:2}
Um verde \omterm{def:g11:color-light-sources:lamps}{laser} ($\lambda = \qty{532}{nm}$) atravessa uma fenda de largura
$ a = \qty{0.20}{mm}$: calcular $\theta $, então a largura da banda central
em uma tela \qty{1.5}{m} ausente.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:light-as-wave:3}
Uma porta é \qty{0.80}{m} amplo: calcular $\lambda/a $ por um
\qty{425}{Hz} voz ($ v = \qty{340}{m/s}$) e para \qty{550}{nm}

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:light-as-wave:4}

\omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} \qty{2.0}{cm}. \omterm{def:g12:light-as-wave:coherent}{Diferença de caminho} no $ M $: \qty{6.0}{cm}; no

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:light-as-wave:5}

\begin{exercise}[$\star\star $]\label{exo:g12:light-as-wave:6}

\begin{exercise}[$\star\star $]\label{exo:g12:light-as-wave:7}

encontre o menor detalhe do terreno que ele resolve; julgar a alegação de que
``lê jornais por cima do ombro''.
\end{exercise}

\section{Problema: Medindo com Luz}

\begin{problem}[{Problema de fim de semana --- medindo com luz: um laser, um
régua e uma parede nua tornam-se uma oficina micrométrica medindo uma fenda, um
cabelo humano, um comprimento de onda desconhecido e as ranhuras de dois discos prateados}]
\label{pb:g12:light-as-wave:1}
Seu kit: um vermelho \omterm{def:g11:color-light-sources:lamps}{laser} ($\lambda = \qty{633}{nm}$, do rótulo), um
verde \omterm{def:g11:color-light-sources:lamps}{laser} de desconhecido \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}, uma fenda calibrada
$ a = \qty{100}{\micro m}$, uma fenda dupla de espaçamento $ a' =
\qty{0.250}{mm}$, uma fita métrica, dois fios de cabelo, um CD e um DVD.

\medskip
\noindent\textbf{Part I --- Calibrating on the known slit.}
Vermelho \omterm{def:g11:color-light-sources:lamps}{laser}, fenda, parede em $ D = \qty{3.00}{m}$.
\begin{enumerate}
  \item Calcule a meia largura angular $\theta $ do feixe difratado.
  \item Mostre que a banda central tem largura $ L = 2\lambda D/a $;
        calcule-o.
  \item Você mede $ L = \qty{3.7}{cm}$: calcular o relativo
        desvio. O método é validado?
  \item Uma segunda fenda tem metade da largura, \qty{50}{\micro m}: prever seu
        $ L $. Estabeleça a regra que liga a largura da fenda e a largura do padrão.
  \item Justifique o passo da pergunta 2 $\tan\theta \approx \theta $: comparar
        $\theta $ e $\tan\theta $ neste ângulo.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hair.} Mesmo \omterm{def:g11:color-light-sources:lamps}{laser} e parede; um
o cabelo esticado substitui a fenda.
\begin{enumerate}[resume]
  \item Qual resultado admitido permite manter a fórmula da fenda para um
        cabelo, e o que diz?
  \item A \omterm{def:g11:fundamental-interactions:strong}{faixa} central é $ L = \qty{5.4}{cm}$ ampla: calcule o
        diâmetro do cabelo $ d $.
  \item Sua régua lê $ L $ para $\pm\qty{0.2}{cm}$: dê o resultado
        \omterm{def:g11:fundamental-interactions:strong}{faixa} para $ d $, como $ d \pm \Delta d $.
  \item O cabelo do seu amigo dá $ L = \qty{7.6}{cm}$: seu diâmetro?
        Cujo cabelo é mais fino e por que mais fino significa mais largo?
  \item Uma luminária de mesa poderia substituir a \omterm{def:g11:color-light-sources:lamps}{laser}? Dê duas razões.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The green \omterm{def:g11:color-light-sources:lamps}{laser}'s \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}.} Verde
\omterm{def:g11:color-light-sources:lamps}{laser}, fenda dupla, parede em $ D = \qty{2.00}{m}$.
\begin{enumerate}[resume]
  \item Por que ambas as fendas devem ser iluminadas pelo \emph{mesmo} \omterm{def:g11:color-light-sources:lamps}{laser}?
  \item Por que a franja central é brilhante qualquer que seja a \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda}?
  \item Dez interfere \omterm{def:g10:signals-and-waves:period}{período} \qty{4.26}{cm}. Dar $ i $, e o motivo
        para medir dez de uma vez.
  \item Deduza o \omterm{def:g12:mechanical-waves:wavelength}{comprimento de onda} do verde \omterm{def:g11:color-light-sources:lamps}{laser}.
  \item O rótulo do fabricante diz $\qty{532 \pm 10}{nm}$: \omterm{def:g12:light-as-wave:coherent}{consistente}?
        Que cor você espera?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Reading silver discs.} O vermelho \omterm{def:g11:color-light-sources:lamps}{laser}
atinge um CD com incidência normal; seus rastros funcionam como uma grade de reflexão
obedecendo (admitido) $ d \sin\theta = k\lambda $. Na parede em $ D =
\qty{1.00}{m}$, cada feixe de primeira ordem forma um ponto em $ x $ do
central.
\begin{enumerate}[resume]
  \item Para o CD, $ x = \qty{43}{cm}$: calcular $\theta $. Por que é que
        atalho de pequeno ângulo da Parte I agora proibido?
  \item Deduza o espaçamento entre \omterm{def:g11:fundamental-interactions:strong}{faixas} do CD $ d $.
  \item O DVD envia seu spot de primeira ordem para $ x = \qty{1.65}{m}$:
        calcule seu $\theta$ e seu espaçamento entre pistas.
  \item Compare os dois espaçamentos. Por que um DVD armazena mais que um CD?
  \item Feche o caderno: liste os comprimentos medidos neste fim de semana,
        e declare em uma frase o que fez da luz um governante para todos.
\end{enumerate}
\end{problem}
