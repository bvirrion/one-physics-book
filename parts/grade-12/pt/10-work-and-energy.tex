\chapter{Trabalho e Energia Mecânica}\label{ch:g12:work-and-energy}

Coloque uma bola de gude em uma saladeira: ela mergulha, ultrapassa o fundo,
sobe até a altura inicial, hesita, volta --- e cada
a reversão pode ser prevista a partir de uma curva (o perfil da tigela) e uma
linha horizontal (a bola de gude energia), nenhuma equação de movimento resolvida.
Preço do ano passado forças em linha reta (\cref{ch:g11:work-of-force})
e cinética equilibrada contra energia potencial
(\cref{ch:g11:mechanical-energy}); este ano a contabilidade vai
por toda parte: caminhos curvos, nascentes, paisagens à primeira vista.

\section{Trabalhe em qualquer caminho}

\begin{definition}[Trabalhe ao longo de um caminho]\label{def:g12:work-and-energy:path}
Deixe um \omterm{def:g10:inertia:force}{vigor} $\vect F $ agir sobre um corpo movendo-se ao longo de qualquer caminho de $ A $ para
$ B $: corte o caminho em deslocamentos $\vect{\dd\ell}$ tão curto que
cada um é reto e $\vect F $ quase não muda ao longo dele. Cada passo ganha
o pequeno \omterm{def:g11:work-of-force:work}{trabalhar} $\scal{F}{\dd\ell} = F\,\dd\ell\cos\theta $
(\cref{ch:g11:work-of-force}) e o
\emph{trabalhar ao longo do caminho}\index{trabalhar ao longo do caminho} é
sua soma, $ W_{AB}(\vect F) = \sum \scal{F}{\dd\ell}$ --- que em um
caminho reto com constante \omterm{def:g10:inertia:force}{vigor} cai para o do ano passado
$ F \times AB \times \cos\theta $.
\end{definition}

\begin{proposition}[Trabalho do peso, qualquer caminho]\label{prop:g12:work-and-energy:weight}
Junto \emph{qualquer} caminho de $ A $ (altitude $ z_A $) para $ B $ (altitude
$ z_B $), o \omterm{def:g10:universal-gravitation:weight}{peso} de uma massa $ m $ faz o \omterm{def:g11:work-of-force:work}{trabalhar}
$ W_{AB}(\vect P) = mg\,(z_A - z_B)$: entra a gota, nunca a rota.
\end{proposition}

\begin{proof}
Cada pequeno passo contribui $ mg \times (\text{small drop})$ (último
cálculo do segmento direto do ano); o telescópio das gotas para
$ z_A - z_B $. O caso curvo, admitido então, é entregue.
\end{proof}

\begin{definition}[Forças conservadoras e não conservadoras]\label{def:g12:work-and-energy:conservative}
UM \omterm{def:g10:inertia:force}{vigor} é \emph{conservador}\index{conservador} se for \omterm{def:g11:work-of-force:work}{trabalhar}
entre dois pontos é o mesmo ao longo de todos os caminhos (zero em cada rodada
viagem), \emph{não conservador}\index{não conservador}
de outra forma. \omterm{def:g10:universal-gravitation:weight}{Peso} e primavera \omterm{def:g10:inertia:force}{vigor} (próxima seção): conservador.
Deslizando \omterm{def:g10:inertia:inventory}{atrito} não é: oposto ao movimento, ele faz $ W = -fL $ em um
caminho de comprimento $ L $ --- um pedágio por \omterm{def:g10:orders-of-magnitude:unit}{medidor}, nunca reembolsado.
\end{definition}

\begin{example}[Duas trilhas para a cabana]\label{ex:g12:work-and-energy:trails}
UM \qty{75}{kg} \omterm{def:g10:signals-and-waves:oscilloscope}{ganhos} do caminhante \qty{300}{m} por um íngreme \qty{800}{m} trilha
ou \qty{2.4}{km} de ziguezagues. O \omterm{def:g10:universal-gravitation:weight}{peso} cobranças
$-75 \times 9.81 \times 300 \approx \qty{-2.2e5}{J}$ em ambos; um
\qty{30}{N} \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g10:inertia:force}{vigor}, \qty{-2.4e4}{J} em um, \qty{-7.2e4}{J}
por outro. O \omterm{def:g10:universal-gravitation:weight}{peso} reembolsos na descida; atritoa conta é
desapareceu como calor, em ambos os sentidos.
\end{example}

\section{Energia potencial}

\begin{definition}[Energia potencial]\label{def:g12:work-and-energy:ep}
Para cada força conservadora um anexa um
\emph{potencial energético}\index{potencial energético}
$ E_p $, um número que depende apenas da posição, definido por
$ W_{AB} = E_p(A) - E_p(B)$: \omterm{def:g11:work-of-force:work}{trabalhar} entregue é potencial energia gasto.
Consertando $ E_p = 0$ em um ponto de referência escolhido, fixa-o em todos os lugares; o
escolha muda todos os valores por uma constante que cancela todos
diferença. Para o \omterm{def:g10:universal-gravitation:weight}{peso} isso devolve o ano passado $ E_p = mgz $,
agora válido ao longo de qualquer caminho. \omterm{def:g10:inertia:inventory}{Atrito} não pode possuir $ E_p $: um dependente da rota
fatura não há diferença de números de endpoint --- apenas forças conservadoras
armazenar de forma recuperável, exatamente o que o nome conserva.
\end{definition}

\begin{proposition}[Energia potencial elástica]\label{prop:g12:work-and-energy:elastic}
Uma primavera de rigidez $ k $ (\unit{N/m}) esticado ou comprimido por $ x $
recua com o \omterm{def:g10:inertia:force}{vigor} $ F = kx $
(\cref{ch:g12:oscillators-and-time}); o \omterm{def:g11:work-of-force:work}{trabalhar} entrega retornando
descansar --- é
\emph{energia potencial hidrocarboneto}\index{energia potencial hidrocarboneto} --- é
$ E_p = \tfrac12\, k x^2$.
\end{proposition}

\begin{proof}
O \omterm{def:g12:oscillators-and-time:pendulum}{força restauradora} atua ao longo do movimento, então $\sum F\,\dd\ell $ é
o \emph{área sob o $ F $--$ x $ gráfico}: um triângulo de base $ x $ e
altura $ kx $, área $\tfrac12 x \times kx $. O \omterm{def:g10:inertia:force}{vigor} em cada trecho
é o mesmo de ida e volta: \omterm{def:g11:work-of-force:work}{trabalhar} fixado pelas extremidades, mola \omterm{def:g10:inertia:force}{vigor}
\omterm{def:g12:work-and-energy:conservative}{conservador}, $ E_p $ bem definido.
\end{proof}

\section{Os dois teoremas da energia}

\begin{theorem}[Teorema da energia cinética]\label{thm:g12:work-and-energy:ketheorem}\index{teorema da energia cinética}
Para qualquer movimento de um corpo de massa $ m $ de $ A $ para $ B $ --- curvado,
em loop, tridimensional ---
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
a soma que cobre tudo forças agindo sobre o corpo.
\end{theorem}

\begin{proof}
Desde $ v^2 = \scal{v}{v}$, a regra do produto da matemática deste ano
dá a derivada de $ v^2$ como $2\,\scal{a}{v}$, então
$\dd E_k/\dd t = m\,\scal{a}{v} = \scal{F_{\text{tot}}}{v}$ por
Segunda lei de Newton (\cref{ch:g12:newtons-laws}). Em um breve $\dd t $
o corpo se move $\vect{\dd\ell} = \vect v\,\dd t $: $ E_k $ ganha o
pequeno \omterm{def:g11:work-of-force:work}{trabalhar} de \cref{def:g12:work-and-energy:path}; resumindo os passos
soma o funciona --- declaração admitida do ano passado, obtida integralmente.
\end{proof}

\begin{proposition}[Potência instantânea]\label{prop:g12:work-and-energy:power}
A cada instante um \omterm{def:g10:inertia:force}{vigor} $\vect F $ em um corpo de velocidade $\vect v $
entrega o \omterm{def:g11:work-of-force:power}{poder} $ P = \scal{F}{v} = Fv\cos\theta $, e
$\dd E_k/\dd t $ é a soma dos \omterm{def:g11:work-of-force:power}{poderes} de tudo forças: do ano passado
fórmula de velocidade constante, exata em cada instante de qualquer movimento.
\end{proposition}

\begin{proof}
Leia a prova anterior: cada \omterm{def:g10:inertia:force}{vigor} alimenta
$\scal{F}{\dd\ell} = \scal{F}{v}\,\dd t $ em $ E_k $ durante $\dd t $.
\end{proof}

\begin{theorem}[Teorema da energia mecânica]\label{thm:g12:work-and-energy:emtheorem}\index{teorema da energia mecânica}
Deixar $ E_m = E_k + E_p $, onde $ E_p $ coleta as energias potenciais de
todos os forças conservadoras no \omterm{def:g11:work-of-force:work}{trabalhar}. Então, ao longo de qualquer caminho,
\[
\Delta E_m = W_{AB}(\text{non-conservative forces}),
\]
e $ E_m $ é conservado sempre que forças não conservadoras não \omterm{def:g11:work-of-force:work}{trabalhar}.
\end{theorem}

\begin{proof}
Divida o \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} teorema: $\Delta E_k = W_{\text{cons}} +
W_{\text{nc}} = -\Delta E_p + W_{\text{nc}}$; mover $\Delta E_p $ entre.
\end{proof}

\begin{example}[O pêndulo, honestamente desta vez]\label{ex:g12:work-and-energy:pendulum}
Um pêndulo em um fio de comprimento $ L = \qty{2.5}{m}$ é lançado às
descansar em $\qty{45}{\degree}$ da \omterm{def:g10:universal-gravitation:weight}{vertical}. O \omterm{def:g11:circuits-and-power:voltage}{tensão},
perpendicular ao movimento, é impotente; o \omterm{def:g10:universal-gravitation:weight}{peso} é \omterm{def:g12:work-and-energy:conservative}{conservador}:
$ E_m $ é conservado \emph{no arco curvo}, que no ano passado poderia
apenas ser verificado. O bob começa $ h = L(1 - \cos\qty{45}{\degree}) =
\qty{0.73}{m}$ para cima, então na parte inferior $ v = \sqrt{2gh} \approx
\qty{3.8}{m/s}$.
\end{example}

\begin{method}[Auditoria energética de qualquer movimento]\label{met:g12:work-and-energy:audit}
\begin{enumerate}
  \item Escolha o sistema e liste os forças nisso.
  \item Classificar: perpendicular ao movimento $\to $ não \omterm{def:g11:work-of-force:work}{trabalhar}; \omterm{def:g12:work-and-energy:conservative}{conservador}
        $\to $ um $ E_p $ em $ E_m $; o resto $\to $ $ W_{\text{nc}}$
        (deslizando \omterm{def:g10:inertia:inventory}{atrito}: $-fL $, $ L $ = comprimento do caminho).
  \item Escrever $\Delta E_m = W_{\text{nc}}$ entre o ponto de dados e
        o ponto de interrogação; resolver. \omterm{def:g10:energy-conservation:energy}{Energia} responde ``quão rápido, quão
        alto''; apenas o tempo e a aceleração precisam das leis de Newton.
\end{enumerate}
\end{method}

\section{Diagramas de energia}

\begin{definition}[Diagrama de energia, pontos de viragem]\label{def:g12:work-and-energy:diagram}
Para um movimento ao longo de um eixo $ x $ com conservado $ E_m $, o
\emph{diagrama de energia}\index{diagrama de energia} traça a curva $ E_p(x)$
e a linha horizontal $ E_m $. Desde $ E_k = E_m - E_p \geq 0$, o
o movimento está confinado onde a curva fica abaixo da linha; a lacuna
entre eles está $ E_k $, leia diretamente. Em um
\emph{ponto de viragem}\index{ponto de viragem}, onde a curva encontra a linha,
$ v = 0$ e o movimento se inverte.
\end{definition}

\begin{proposition}[Força da paisagem; equilíbrio]\label{prop:g12:work-and-energy:equilibria}
O força conservadora junto $ x $ é menos a inclinação de $ E_p $:
$ F = -\,\dd E_p/\dd x $ --- aponta \emph{descida} no diagrama,
e desaparece onde a tangente é horizontal: um equilíbrio. Um
mínimo de $ E_p $ é \emph{estável}\index{estável} --- o
o corpo deslocado é empurrado para trás, uma bola num vale; um máximo é
\emph{assustador}\index{assustador} --- é empurrado para longe, um
bola no topo de uma colina.
\end{proposition}

\begin{proof}
Em um pequeno passo $\dd x $ o \omterm{def:g10:inertia:force}{vigor} funciona $ F\,\dd x = E_p(x) -
E_p(x + \dd x) = -\dd E_p $; dividir por $\dd x $. Descendo em ambos os lados
de um vale aponta para trás; em torno de uma colina, fora.
\end{proof}

\begin{remark}[Oscilações e fuga]\label{rem:g12:work-and-energy:escape}
As pequenas oscilações \cref{ch:g12:oscillators-and-time} morar em
os fundos de $ E_p $ vales. E longe da Terra a gravidade enfraquece: o
$ E_p(r)$ curva sobe cada vez mais lentamente em direção a um teto finito (o
a área sob o gráfico da força de contração é finita). Um foguete cujo $ E_m $
atinge o teto nunca encontra um \omterm{def:g12:work-and-energy:diagram}{ponto de viragem}: isto
\emph{escapada}\index{escapada} --- sobre \qty{11.2}{km/s} de
o chão (\cref{exo:g12:work-and-energy:13}); abaixo disso é
vinculado. O teto em si é computado no volume do Ano 1.
\end{remark}

\begin{example}[Loop-the-loop]\label{ex:g12:work-and-energy:loop}
Uma bola de gude lançada do alto $ H $ costa em um loop \omterm{def:g10:universal-gravitation:weight}{vertical} de
raio $ R $. Dinâmica circular (\cref{ch:g12:newtons-laws}) demandas
$ v_{\text{top}}^2 \geq gR $ no topo --- o dado emprestado por último
o problema da montanha-russa do ano, agora nosso. Conservação do repouso em $ H $
para cima (altura $2R $) dá $ v_{\text{top}}^2 = 2g(H - 2R)$, então
$ H \geq \tfrac52 R $: meio raio acima do topo do loop, qualquer que seja o
massa --- um pouco mais no mundo real e esfregante.
\end{example}

\section{Problema: o salto de esqui}

\begin{problem}[{Problema de fim de semana --- o salto de esqui: uma corrida auditada
por um canhão rápido, um voo em queda livre, a geometria suave de uma subida íngreme
pouso, e a torre que um projetista deve comprar quando as contas de fricção}]
\label{pb:g12:work-and-energy:1}
Um saltador de esqui olímpico ($ m = \qty{70}{kg}$, esquis incluídos) começa em
descansar de um portão $ H = \qty{45}{m}$ acima da mesa de decolagem, slides
$ L = \qty{90}{m}$ de percurso curvo, e sai da mesa horizontalmente;
uma arma de velocidade lá diz $ v_0 = \qty{26}{m/s}$. O ponto de pouso fica
$ h = \qty{40}{m}$ mais baixo, em um $\qty{35}{\degree}$ declive. Parte II
e III despreza a \omterm{def:g11:circuits-and-power:resistance}{resistência} do ar.

\medskip\noindent\textbf{Part I --- The in-run.}
\begin{enumerate}
  \item Preveja a velocidade de decolagem sem \omterm{def:g10:inertia:inventory}{atrito}; por que a curva de
        o in-run é irrelevante?
  \item Referência na tabela: computar $ E_m $ no portão e na decolagem.
  \item Calcular $\Delta E_m $; qual teorema nomeia os culpados e quem?
  \item Deduzir a média \omterm{def:g11:work-of-force:sign}{resistivo} \omterm{def:g10:inertia:force}{vigor} (neve mais ar) durante a corrida.
  \item Que fração $ E_m $ está perdido? Por que encerar os esquis e agachar-se?
\end{enumerate}

\medskip\noindent\textbf{Part II --- The flight.}
\begin{enumerate}[resume]
  \item Apenas o \omterm{def:g10:universal-gravitation:weight}{peso} atua em vôo: qual movimento de
        \cref{ch:g12:projectile-motion} é isto? Componentes de velocidade
        no pouso?
  \item Velocidade de pouso, duas vezes: de componentes e de $ E_m $
        conservação. Em \unit{km/h}?
  \item Calcule o tempo de voo e a distância horizontal voada.
  \item Os verdadeiros saltadores voam no ar como uma asa, voando muito mais longe:
        pode arejar \emph{elevação} a velocidade de pouso? Discuta com o
        \omterm{def:g11:mechanical-energy:em}{energia mecânica} teorema.
\end{enumerate}

\medskip\noindent\textbf{Part III --- The landing.}
\begin{enumerate}[resume]
  \item Calcule o ângulo da velocidade de pouso abaixo da horizontal.
  \item Decomponha-o em componentes paralelos e perpendiculares a
        a encosta.
  \item As pernas absorvem apenas a parte perpendicular: calcule isso
        \omterm{def:g11:mechanical-energy:kinetic}{energia cinética}; compare com o total.
  \item Terreno plano absorveria o lote: qual é a queda \omterm{def:g10:universal-gravitation:weight}{vertical}
        cada equivalente de desembarque ($ h_{\text{eq}} = v_\perp^2/2g $,
        resp.\ $ v^2/2g $)?
  \item Uma frase: por que as colinas de pouso são íngremes e curvas?
\end{enumerate}

\medskip\noindent\textbf{Part IV --- The designer's tower.}
Uma colina menor precisa de decolagem apenas $ v_0' = \qty{24}{m/s}$; é
em linha reta desce em $\qty{35}{\degree}$, então a altura do portão $ H'$
significa comprimento da trilha $ H'/\sin\qty{35}{\degree}$; mesmo \qty{80}{N} arrastar.
\begin{enumerate}[resume]
  \item Calcule a altura do portão que um projetista sem \omterm{def:g10:inertia:inventory}{atrito} construiria.
  \item Escreva o \omterm{def:g11:mechanical-energy:em}{energia mecânica} teorema, porta para mesa, para $ H'$.
  \item Resolva para $ H'$.
  \item Em que porcentagem \omterm{def:g10:inertia:inventory}{atrito} levantar a torre?
  \item A massa não cancela mais: recalcular $ H'$ por um
        \qty{60}{kg} saltador. Quem precisa da torre mais alta e por quê?
  \item Punchline: a torre anunciada, e atritoé a parte dele.
\end{enumerate}
\end{problem}
