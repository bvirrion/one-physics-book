\chapter{Energia Nuclear: Fissão, Fusão, $E=mc^2$}\label{ch:g12:nuclear-energy}

Uma pastilha de combustível do tamanho de uma borracha de lápis cobre a eletricidade de uma casa
por um ano; o mesmo serviço do carvão leva três toneladas. Acima, o Sol
paga pela luz do dia desaparecendo --- quatro milhões de toneladas de si mesmo por \omterm{def:g10:orders-of-magnitude:unit}{segundo}.
Este capítulo abre o livro de contas por trás de ambas as barganhas: a própria massa é
energiae \omterm{def:g11:fundamental-interactions:ladder}{núcleos} (\cref{ch:g11:nucleus-radioactivity}) é onde é descontado.

\section{Massa é energia}

\begin{theorem}[Equivalência massa-energia]\label{thm:g12:nuclear-energy:emc2}\index{equivalência massa-energia}
Um corpo de massa $ m $, pelo simples fato de existir, detém o energia
\[
E = m c^2, \qquad c = \qty{3.00e8}{m/s}.
\]
Sempre que um sistema perde massa $\Delta m $, ele libera o energia
$\Delta m\, c^2$ --- qualquer que seja o mecanismo.
\end{theorem}
\admitted

\begin{remark}[Einstein, 1905]\label{rem:g12:nuclear-energy:einstein}
Esta é a relação de Einstein, aqui admitida; de onde vem é o negócio
do capítulo da relatividade especial que encerra este ano
(\cref{ch:g12:special-relativity}). Observe a taxa: $ c^2 = \qty{9.00e16}{J/kg}$
--- um erro de arredondamento de massa é uma fortuna de energia.
\end{remark}

\begin{definition}[Unidade de elétron-volt e massa atômica]\label{def:g12:nuclear-energy:units}
Dois \omterm{def:g10:orders-of-magnitude:unit}{unidades} adaptado ao \omterm{def:g11:fundamental-interactions:ladder}{núcleo}. O
\emph{elétron-volt}\index{elétron-volt}, o energia um elétron ganha
travessia \qty{1}{V} (\cref{ch:g11:circuits-and-power}):
$\qty{1}{eV} = \qty{1.60e-19}{J}$, $\qty{1}{MeV} = \qty{1.60e-13}{J}$ ---
ligações químicas negociam alguns \unit{eV}, reações nucleares \omterm{def:g11:circuits-and-power:voltage}{megaelétron-volts}.
O \emph{unidade de massa atômica unificada}\index{unidade de massa atômica unificada}, um
décimo \omterm{def:g10:orders-of-magnitude:unit}{segundo} da massa de um carbono-12 \omterm{def:g11:fundamental-interactions:ladder}{átomo}: $\qty{1}{u} = \qty{1.66054e-27}{kg}$,
de energia equivalente \qty{931.5}{MeV} --- $\qty{1}{u} = \qty{931.5}{MeV}/c^2$.
\end{definition}

\begin{example}[A fortuna adormecida]\label{ex:g12:nuclear-energy:fortune}
Um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de qualquer coisa, totalmente convertido, é $\qty{9.00e16}{J}$ --- três
anos de produção de um grande (\qty{1}{GW}) \omterm{def:g11:work-of-force:power}{poder} estação. Ninguém percebeu antes
1905 porque a física comum mal toca a massa: queimar um \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de carvão
(\qty{3.0e7}{J}) clareia em um terço de micrograma. \omterm{def:g11:fundamental-interactions:ladder}{Núcleos}, veremos,
mova uma parte em mil: ainda pequena, mas um milhão de vezes a química.
\end{example}

\section{O defeito de massa}

\begin{definition}[Defeito de massa e energia de ligação]\label{def:g12:nuclear-energy:defect}
Pesar um \omterm{def:g11:fundamental-interactions:ladder}{núcleo} ${}^{A}_{Z}\mathrm{X}$ (massa $ m $), depois pese seu $ Z $ prótons
e $ A - Z $ nêutrons separadamente: as partes são \emph{mais pesado} do que o todo.
A diferença
\[
\Delta m = Z\,m_p + (A - Z)\,m_n - m > 0,
\qquad m_p = \qty{1.00728}{u},\; m_n = \qty{1.00866}{u},
\]
é o \emph{defeito de massa}\index{defeito de massa}; o \emph{vinculativo
energia}\index{vinculativo
energia} $ E_b = \Delta m\,c^2$ é o energia precisava
puxe o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} separado em grátis núcleons --- igualmente, o energia lançado
quando estiver montado. Um limite \omterm{def:g11:fundamental-interactions:ladder}{núcleo} senta \emph{abaixo} suas partes.
\end{definition}

\begin{example}[Hélio-4]\label{ex:g12:nuclear-energy:helium}
O ${}^{4}_{2}\mathrm{He}$ \omterm{def:g11:fundamental-interactions:ladder}{núcleo} tem $ m = \qty{4.00151}{u}$; suas partes:
$2 \times 1.00728 + 2 \times 1.00866 = \qty{4.03188}{u}$. Então
$\Delta m = \qty{0.03037}{u}$ e $ E_b = 0.03037 \times 931.5 \approx
\qty{28.3}{MeV}$: o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} pesa $0.75\%$ menos do que suas partes --- o
uma parte em mil de \cref{ex:g12:nuclear-energy:fortune}.
\end{example}

\begin{method}[Balanço energético de uma reação nuclear]\label{met:g12:nuclear-energy:balance}
\begin{enumerate}
  \item Equilibre a equação: total $ A $ e $ Z $ conservado (\cref{ch:g11:nucleus-radioactivity}).
  \item Adicione as massas antes, depois, em \unit{u} --- mantenha todas as cinco casas decimais.
  \item $\Delta m = m_{\text{before}} - m_{\text{after}}$, então
        $ E = \Delta m\,(\text{in u}) \times \qty{931.5}{MeV}$, levado como
        \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} dos produtos; negativo significa que a reação deve ser paga.
\end{enumerate}
\end{method}

\section{A curva que percorre o universo}

\begin{definition}[Energia de ligação por núcleon]\label{def:g12:nuclear-energy:pernucleon}
A coesão é comparada de forma justa pelo \emph{energia de ligação por
núcleo}\index{energia de ligação por
núcleo}, $ E_b/A $, o custo médio de
extraindo um \omterm{def:g11:fundamental-interactions:ladder}{núcleo}: $28.3/4 \approx \qty{7.1}{MeV}$ para hélio-4.
\end{definition}

\begin{proposition}[O pico de ferro e duas estradas abaixo]\label{prop:g12:nuclear-energy:ironpeak}\index{pico de ferro}
Conspirado contra $ A $, o energia de ligação por núcleon sobe abruptamente
os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} leves, picos próximos ao ferro, ${}^{56}_{26}\mathrm{Fe}$, por volta
\qty{8.8}{MeV} por \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, em seguida, desliza suavemente para cerca \qty{7.6}{MeV} em
urânio: o ferro é o mais fortemente ligado \omterm{def:g11:fundamental-interactions:ladder}{núcleo} na natureza. Uma reação
lançamentos energia exatamente quando seus produtos ficam \emph{mais alto} na curva; dois
estratégias opostas, portanto, ambos pagam, e ambos caminham em direção ao ferro ---
\emph{divisão} os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} mais pesados, e \emph{fundindo} o mais leve.
\end{proposition}
\admitted

\begin{remark}[Por que um pico]\label{rem:g12:nuclear-energy:whypeak}
O cabo de guerra de \cref{ch:g11:fundamental-interactions}: cola \omterm{def:g11:fundamental-interactions:strong}{forte} liga
tocando apenas os vizinhos, a repulsa de Coulomb abrange todo o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- pequeno
os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} são em sua maioria superficiais, os grandes são principalmente repulsivos, acerte o compromisso.
A contabilidade honesta vive nos volumes universitários.
\end{remark}

\section{Fissão e a reação em cadeia}

\begin{definition}[Fissão]\label{def:g12:nuclear-energy:fission}
\emph{Fissão}\index{Fissão} é a divisão de um pesado \omterm{def:g11:fundamental-interactions:ladder}{núcleo} em dois
fragmentos de tamanho médio. Urânio-235 é \emph{físsil}\index{físsil}: um lento
nêutron, absorvido, deixa o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} tão agitado que se parte em dois,
cuspindo dois ou três nêutrons frescos ---
${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to \text{two fragments} +
2\text{--}3\;{}^{1}_{0}\mathrm{n}$, o par variando de fissão para fissão.
\end{definition}

\begin{example}[Uma fissão, pesada]\label{ex:g12:nuclear-energy:fissionbalance}
Um percurso frequente, por \cref{met:g12:nuclear-energy:balance}:
\[
{}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \longrightarrow
{}^{94}_{38}\mathrm{Sr} + {}^{140}_{54}\mathrm{Xe} + 2\,{}^{1}_{0}\mathrm{n}.
\]
Antes: $1.00866 + 234.99350 = \qty{236.00216}{u}$. Depois:
$93.89450 + 139.89200 + 2 \times 1.00866 = \qty{235.80382}{u}$. Por isso
$\Delta m = \qty{0.19834}{u}$ e $ E \approx \qty{185}{MeV}$; contando o
decaimentos posteriores dos fragmentos radioativos (\cref{ch:g12:radioactive-decay}),
cada fissão vale cerca de \qty{200}{MeV} --- um pouco \unit{eV} compra um \omterm{def:g11:fundamental-interactions:ladder}{átomo}
de carvão.
\end{example}

\begin{definition}[Reação em cadeia e massa crítica]\label{def:g12:nuclear-energy:chain}
Cada fissão é iluminado por um nêutron e libera dois ou três: em uma quantidade grande o suficiente
caroço, eles atingem outros \omterm{def:g11:fundamental-interactions:ladder}{núcleos} e a reação se alimenta --- um
\emph{ocorrência em cadeia}\index{ocorrência em cadeia}. Em um pequeno pedaço, a maioria dos nêutrons
escape pela superfície primeiro; a quantidade mínima que sustenta a cadeia
é o \emph{massa crítica}\index{massa crítica} --- abaixo dela a corrente
fracassa, acima dela cada geração se multiplica: o princípio da bomba.
\end{definition}

\begin{definition}[O reator, domesticado]\label{def:g12:nuclear-energy:reactor}
UM \omterm{def:g11:work-of-force:power}{poder} reator mantém a cadeia em exatamente um nêutron por fissão. O combustível:
pelotas de urânio modestamente enriquecidas em urânio-235. O
\emph{moderador}\index{moderador} (água comum, geralmente) retarda o rápido
fissão nêutrons por colisões --- nêutrons lentos são muito melhores no desencadeamento
o próximo fissão; o \emph{pressa de controle}\index{pressa de controle} (boro, cádmio)
devorar nêutrons --- empurrado, a corrente morre; prolongado, ele acelera. O
a agitação dos fragmentos torna-se calor, depois vapor e depois eletricidade
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Um grão de pimenta contra um caminhão]\label{ex:g12:nuclear-energy:gram}
Um grama de urânio-235 contém
$ N = \num{1.0e-3} / (235 \times \num{1.66054e-27}) \approx \num{2.56e21}$
\omterm{def:g11:fundamental-interactions:ladder}{núcleos}; no $\qty{200}{MeV} = \qty{3.2e-11}{J}$ cada um, fissionando todos eles
rende cerca de \qty{8.2e10}{J} --- o energia química de $\num{8.2e10} /
\num{3.0e7} \approx \qty{2.7e3}{kg}$ de carvão: um grão de pimenta contra três toneladas.
\end{example}

\section{Fusão: o próprio fogo do Sol}

\begin{definition}[Fusão]\label{def:g12:nuclear-energy:fusion}
\emph{Fusão}\index{Fusão} é a fusão de dois \omterm{def:g11:fundamental-interactions:ladder}{núcleos} leves em um mais pesado,
mais firmemente amarrado. A reação mais acessível casa o deutério e
trítio, o pesado \omterm{def:g11:nucleus-radioactivity:notation}{isótopos} de hidrogênio (\cref{ch:g11:nucleus-radioactivity}):
\[
{}^{2}_{1}\mathrm{H} + {}^{3}_{1}\mathrm{H} \longrightarrow
{}^{4}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}.
\]
\end{definition}

\begin{example}[D--T, pesado]\label{ex:g12:nuclear-energy:dt}
Antes: $2.01355 + 3.01550 = \qty{5.02905}{u}$; depois:
$4.00151 + 1.00866 = \qty{5.01017}{u}$; então $\Delta m = \qty{0.01888}{u}$ e
$ E \approx \qty{17.6}{MeV}$, quatro quintos dele no nêutron --- menos de um
fissão, mas a partir de cinco núcleons em vez de $236$: sobre \qty{3.4e14}{J} por
\omterm{def:g10:orders-of-magnitude:unit}{quilograma} de combustível, quatro vezes fissão. Hélio-4
(\cref{ex:g12:nuclear-energy:helium}) é mais uma vez a cinza.
\end{example}

\begin{definition}[A barreira de Coulomb]\label{def:g12:nuclear-energy:barrier}
Dois \omterm{def:g11:fundamental-interactions:ladder}{núcleos} são ambos positivos: para se tocarem, devem primeiro subir a colina de
sua repulsão elétrica (\cref{ch:g11:electric-gravitational-fields}) --- o
\emph{Barreira de Coulomb}\index{Barreira de Coulomb}. Apenas perto \qty{1e8}{K} fazer
colisões carregam o energia atravessá-lo; o combustível é então um
\emph{plasma}\index{plasma} de \omterm{def:g11:fundamental-interactions:ladder}{núcleos} nus e elétrons. \omterm{def:g12:nuclear-energy:fission}{Fissão} não paga
tal preço: seu gatilho, o nêutron, é neutro, e funciona frio.
\end{definition}

\begin{example}[O Sol corre sobre ele]\label{ex:g12:nuclear-energy:sun}
O \omterm{def:g11:fundamental-interactions:ladder}{núcleo} do Sol, em $\qty{1.5e7}{K}$, fusíveis hidrogênio comum passo a passo ---
o \emph{próton - cadeia de prótons}, de efeito líquido
$4\,{}^{1}_{1}\mathrm{H} \to {}^{4}_{2}\mathrm{He} + 2\,{}^{0}_{+1}\mathrm{e}
+ \text{radiation}$, sobre \qty{26}{MeV} por hélio: $0.7\%$ do
a massa do hidrogênio sai como energia. A luz solar cuja \omterm{def:g12:sound-acoustics:timbre}{espectro} \omterm{def:g12:sound-acoustics:standing-wave}{nós} lemos em
\cref{ch:g10:light-spectra} é esta missa, chegando oito minutos atrasado; cada
estrela brilha caminhando pela curva em direção ao ferro.
\end{example}

\begin{remark}[ITER: engarrafando uma estrela]\label{rem:g12:nuclear-energy:iter}
Nenhum sólido pode conter um \omterm{def:g12:nuclear-energy:barrier}{plasma} no \qty{1.5e8}{K}; um \emph{tokamak} prende-o em uma
\omterm{def:g11:magnetic-fields:bfield}{campo magnético} (\cref{ch:g11:magnetic-fields}), um donut de linhas de campo o
partículas carregadas espiralam sem tocar a parede; ITER, construído para
liberar dez vezes o seu aquecimento \omterm{def:g11:work-of-force:power}{poder}, é a etapa atual. Quebre o
confinamento e o \omterm{def:g12:nuclear-energy:barrier}{plasma} esfria e para em \omterm{def:g10:orders-of-magnitude:unit}{segundos}: não reação em cadeia, não
\omterm{def:g12:nuclear-energy:chain}{massa crítica} --- o difícil não é apagar o fogo, mas mantê-lo aceso.
\end{remark}

\begin{example}[A escada de energia, por quilograma]\label{ex:g12:nuclear-energy:ladder}
\omterm{def:g10:energy-conservation:energy}{Energia} por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de combustível: carvão, \qty{3.0e7}{J}; urânio-235 por fissão,
\qty{8.2e13}{J}; deutério  trítio por fusão, \qty{3.4e14}{J}; total
conversão ($ E = mc^2$), \qty{9.0e16}{J}. Seis ordens de magnitude separadas
química do \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- a razão pela qual um reator é reabastecido pelo caminhão
e uma usina de carvão por trem.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:nuclear-energy:1}
Converter: (a) \qty{1}{eV} e \qty{1}{MeV} em \omterm{def:g11:work-of-force:work}{joules}; (b) o \qty{200}{MeV} de
um fissão em \omterm{def:g11:work-of-force:work}{joules}; (c) cheque de $\qty{1}{u} = \qty{1.66054e-27}{kg}$
que é energia equivalente é próximo de \qty{931.5}{MeV}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:nuclear-energy:2}

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:nuclear-energy:3}

\section{Problema: energizando uma cidade}

\begin{problem}[{Problema de fim de semana --- abastecendo uma cidade: um gigawatt constante para
meio milhão de pessoas, comprou de quatro maneiras --- pelotas de urânio, cargas de trem de
carvão, uma piscina de água do mar, a própria substância do Sol --- com o mesmo quilograma
de massa desaparecida escondida sob todos os quatro}]
\label{pb:g12:nuclear-energy:1}
Uma cidade de $500\,000$ habitantes atraem uma \omterm{prop:g11:circuits-and-power:power}{energia elétrica} constante \omterm{def:g11:work-of-force:power}{poder}
$ P = \qty{1.0}{GW}$; suas plants, qualquer que seja o combustível, transformam calor em
eletricidade com \omterm{def:g10:energy-conservation:efficiency}{eficiência} $33\%$ (\cref{ch:g10:energy-conservation}).
Dados: um urânio-235 fissão, \qty{200}{MeV} em tudo; carvão, \qty{3.0e7}{J/kg};
um D--T fusão, \qty{17.6}{MeV} de \qty{5.029}{u} de combustível;
$\qty{1}{u} = \qty{1.66054e-27}{kg}$; $\qty{1}{eV} = \qty{1.60e-19}{J}$; um
ano $= \qty{3.16e7}{s}$; o Sol: \qty{3.9e26}{W}, \qty{2.0e30}{kg}; um cúbico
\omterm{def:g10:orders-of-magnitude:unit}{medidor} da água do mar detém cerca de \qty{33}{g} de deutério.

\medskip
\noindent\textbf{Part I --- The fissão plant.}
\begin{enumerate}
  \item Que térmica \omterm{def:g11:work-of-force:power}{poder} o reator deve entregar?
  \item Expresse o energia de um fissão em \omterm{def:g11:work-of-force:work}{joules}.
  \item Quantos fissões por \omterm{def:g10:orders-of-magnitude:unit}{segundo} governar a cidade?
  \item A massa de um urânio-235 \omterm{def:g11:fundamental-interactions:ladder}{núcleo} ($ A = 235$), então o urânio-235
        consumido por \omterm{def:g10:orders-of-magnitude:unit}{segundo}?
  \item Deduza o urânio-235 consumido em um ano, em \omterm{def:g10:orders-of-magnitude:unit}{quilogramas}.
  \item O combustível é enriquecido para $4.0\%$ urânio-235: que massa faz um ano
        pegue, e cabe em um caminhão?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The coal ledger.}
\begin{enumerate}[resume]
  \item Calcule o energia térmica a cidade consome em um ano.
  \item Que massa de carvão o entrega?
  \item Calcule a razão (massa de carvão)/(massa de urânio-235) para um ano e
        verifique-o contra a escada de \cref{ex:g12:nuclear-energy:ladder}.
  \item Um trem de carga transporta \qty{3.0e6}{kg}: quantos trens por ano, e
        por dia? Uma frase: imagine as duas linhas de abastecimento.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The fusão dream.}
\begin{enumerate}[resume]
  \item Expresse o energia de um D--T fusão em \omterm{def:g11:work-of-force:work}{joules}.
  \item Calcule o energia liberado por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de combustível D--T; comparar com
        a escada.
  \item Que massa de combustível D--T cobre o ano da cidade? Que massa disso é
        deutério ($2.014$ do \qty{5.029}{u})?
  \item Que volume de água do mar contém esse deutério? Compare com um
        Piscina olímpica, cerca \qty{2.5e3}{m^3}. (O trítio é derivado do lítio.)
  \item Por que esta plant não carrega \omterm{def:g12:nuclear-energy:chain}{massa crítica} e não reação em cadeia para
        medo? Qual é, então, a parte difícil?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The star that pays in mass.}
\begin{enumerate}[resume]
  \item De $ E = mc^2$, que massa o Sol converte a cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}?
  \item Que massa por \omterm{def:g10:orders-of-magnitude:unit}{segundo} e por ano deve \emph{qualquer}
        \qty{3.0}{GW}-fonte térmica converter? Verifique se todos os três combustíveis
        entregar esta mesma massa.
  \item Que massa total o Sol irradiou em seu $\num{4.5e9}$
        anos, e que fração do Sol é essa?
  \item \omterm{def:g12:nuclear-energy:fusion}{Fusão} pode tocar sobre $0.07\%$ da massa do Sol (a parcela incinerável da
        seu \omterm{def:g11:fundamental-interactions:ladder}{núcleo}): estime seu tempo de vida total como estrela e o que resta.
  \item Encerre a auditoria em três frases: um ano da cidade em toneladas
        de carvão, \omterm{def:g10:orders-of-magnitude:unit}{quilogramas} de urânio-235 e \omterm{def:g10:orders-of-magnitude:unit}{quilogramas} de combustível D--T --- e
        o \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de massa que, em todos os cenários, se tornou a luz da cidade.
\end{enumerate}
\end{problem}
