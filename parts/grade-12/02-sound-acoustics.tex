\chapter{Sound and Acoustics}\label{ch:g12:sound-acoustics}

Pluck a guitar's low string: the room fills with a low A, yet
nothing traveled to your ear but squeezed and stretched air. This
chapter follows that pattern --- its speed, how frequency becomes
pitch and power loudness, and why one note differs on every instrument.

\section{Sound is a pressure wave}

\begin{definition}[Sound wave]\label{def:g12:sound-acoustics:sound-wave}
A \emph{sound wave}\index{sound wave} is a mechanical pressure wave
(\cref{ch:g12:mechanical-waves}): a vibrating surface creates
traveling \emph{compressions}\index{compression} (slight
overpressure) and \emph{rarefactions}\index{rarefaction}
(underpressure). Air parcels oscillate \emph{along} the travel
direction (sound is \emph{longitudinal}\index{longitudinal wave}),
and a medium is needed: a bell under a vacuum jar falls silent
(\cref{ch:g10:signals-and-waves}).
\end{definition}

\begin{proposition}[Speed of sound]\label{prop:g12:sound-acoustics:speed}
\index{speed of sound}%
Sound travels at about \qty{340}{m/s} in air at
\qty{20}{\degreeCelsius}, \qty{1500}{m/s} in water and
\qty{5000}{m/s} in steel: the stiffer the medium, the
faster. \admitted
\end{proposition}

\begin{remark}[Warm air is faster]\label{rem:g12:sound-acoustics:temperature}
These are measured values; predicting them is done in the Year 1
volume. Sound speeds up in warmer air (\qty{331}{m/s} at
\qty{0}{\degreeCelsius}) --- why winds drift out of tune as the hall warms.
\end{remark}

\begin{example}[The sizes of sounds]\label{ex:g12:sound-acoustics:wavelengths}
The relation $\lambda = v/f$ sizes any sound: in air a \qty{20}{Hz}
rumble spans $340/20 = \qty{17}{m}$ --- a building --- and a
\qty{20}{kHz} hiss fits in \qty{1.7}{cm}; the metre-sized ones bend
easily around doors and corners.
\end{example}

\section{Pitch, loudness and the ear}

\begin{definition}[Pitch and loudness]\label{def:g12:sound-acoustics:pitch-loudness}
\emph{Pitch}\index{pitch} is the perception of frequency: the higher
$f$, the higher the note; concert A is \qty{440}{Hz}, and doubling
$f$ raises the note one octave. \emph{Loudness}\index{loudness} is
the perception of amplitude: the stronger the pressure oscillation,
the louder; amplification changes loudness, not pitch.
\end{definition}

\begin{remark}[The ear's window]\label{rem:g12:sound-acoustics:audible}
Hearing spans about \qty{20}{Hz} to \qty{20}{kHz}
(\cref{ch:g10:signals-and-waves}); below lies infrasound, above lies
ultrasound --- bats at \qty{50}{kHz}, medical probes at megahertz. A
piano only samples the middle: \qtyrange{27.5}{4186}{Hz}.
\end{remark}

\section{Sound intensity and the decibel scale}

\begin{definition}[Sound intensity]\label{def:g12:sound-acoustics:intensity}
The \emph{sound intensity}\index{sound intensity} $I$ is the
acoustic power crossing one square metre facing the wave:
$I = P/S$, in \unit{W/m^2}. The quietest audible sound is about
$I_0 = \qty{1.0e-12}{W/m^2}$; pain begins near \qty{1}{W/m^2}.
\end{definition}

\begin{proposition}[Inverse-square attenuation]\label{prop:g12:sound-acoustics:inverse-square}
\index{inverse-square law}%
A small source radiating a power $P$ equally in all directions
gives, at distance $r$, $I = P/(4\pi r^2) \propto 1/r^2$: doubling
the distance divides the intensity by four.
\end{proposition}

\begin{proof}
The whole power $P$ crosses the sphere of area $4\pi r^2$; divide.
\end{proof}

\begin{definition}[Sound level]\label{def:g12:sound-acoustics:level}
The ear spans twelve powers of ten, so intensities are quoted
logarithmically. The \emph{sound level}\index{sound level} of $I$ is
\[ L = 10 \log_{10}(I/I_0), \qquad I_0 = \qty{1.0e-12}{W/m^2}, \]
in \emph{decibels}\index{decibel} (\unit{dB}): hearing threshold
\qty{0}{dB}, pain near \qty{120}{dB}.
\end{definition}

\begin{proposition}[Decibel arithmetic]\label{prop:g12:sound-acoustics:db-arithmetic}
Multiplying $I$ by $k$ adds $10\log_{10}k$ decibels. In particular
$\times 10$ adds \qty{10}{dB}; $\times 2$ adds
$10\log_{10}2 \approx \qty{3.0}{dB}$; $N$ incoherent identical
sources add \emph{intensities}, never levels, giving
$L_1 + 10\log_{10}N$; doubling the distance to a small source ($I$
divided by four) removes about \qty{6.0}{dB}.
\end{proposition}

\begin{proof}
$10\log_{10}(kI/I_0) = 10\log_{10}(I/I_0) + 10\log_{10}k$;
distance: \cref{prop:g12:sound-acoustics:inverse-square}, $k = 1/4$.
\end{proof}

\begin{method}[Working in decibels]\label{met:g12:sound-acoustics:db}
\begin{enumerate}
  \item $L = 10\log_{10}(I/I_0)$; back: $I = I_0 \times 10^{L/10}$.
  \item Never add levels: convert to intensities, add, convert back
        --- or $+\qty{3}{dB}$ per doubling, $+\qty{10}{dB}$ per tenfold.
  \item Each doubling of distance to a small source costs \qty{6}{dB}.
\end{enumerate}
\end{method}

\begin{remark}[Hearing damage]\label{rem:g12:sound-acoustics:damage}
Hearing loss is a dose: \qty{85}{dB} is safe for eight hours, and
every \qty{3}{dB} more --- a doubling of intensity --- halves the
safe time; lost hair cells do not grow back.
\end{remark}

\section{Timbre and harmonics}

\begin{definition}[Harmonics]\label{def:g12:sound-acoustics:harmonics}
A periodic sound of frequency $f_1$ --- the
\emph{fundamental}\index{fundamental} --- is in general a sum of
sines at the frequencies $f_n = n f_1$, $n = 1, 2, 3, \dots$, its
\emph{harmonics}\index{harmonic}. A sound reduced to its fundamental
is a \emph{pure tone}\index{pure tone}, like a tuning fork's.
\end{definition}

\begin{definition}[Spectrum and timbre]\label{def:g12:sound-acoustics:timbre}
The \emph{spectrum}\index{spectrum} of a sound is the bar chart of
its harmonics: one bar per $f_n$, the height that harmonic's
strength. The ear hears $f_1$ as the pitch and the \emph{mix} of
strengths as the \emph{timbre}\index{timbre} --- the sound's color:
instruments playing the same note share $f_1$, differing only in
their spectra.
\end{definition}

\begin{example}[Why you recognize the caller]\label{ex:g12:sound-acoustics:two-instruments}
A flute's A is nearly a pure tone; a violin bowing the same
\qty{220}{Hz} adds strong harmonics at \qtylist{440;660;880}{Hz}.
Same pitch and loudness, yet no one confuses them: a voice, too, is
a spectrum you know by heart.
\end{example}

\section{The vibrating string}

\begin{definition}[Standing wave]\label{def:g12:sound-acoustics:standing-wave}
A wave confined between the fixed ends of a string can settle into a
\emph{standing wave}\index{standing wave}: an oscillation in place,
motionless points --- \emph{nodes}\index{node} --- alternating with
points of maximal swing --- \emph{antinodes}\index{antinode}.
Nothing travels any more; the string vibrates in a fixed pattern.
\end{definition}

\begin{proposition}[Modes of a string]\label{prop:g12:sound-acoustics:string-modes}
A string of length $L$, fixed at both ends, carrying waves of speed
$v$, vibrates steadily only in patterns fitting a whole number $n$
of half-wavelengths between the imposed end nodes:
\[ \lambda_n = \frac{2L}{n}, \quad f_n = n\,\frac{v}{2L} = n f_1, \quad n = 1, 2, 3, \dots \]
The lowest mode $f_1 = v/(2L)$ is the fundamental; the others are
exactly its harmonics. \admitted
\end{proposition}

\begin{remark}[Where the honest proof lives]\label{rem:g12:sound-acoustics:honest}
That confined waves pick exactly these modes follows from
superposing the wave with its reflections, a calculus carried out in
the Year 1 volume. The geometry convinces: both ends must stay
still, so a whole number of half-wavelengths must fit.
\end{remark}

\begin{example}[A violin's A string]\label{ex:g12:sound-acoustics:guitar}
A violin's A string has $L = \qty{32.8}{cm}$ and sounds
$f_1 = \qty{440}{Hz}$: waves run along it at
$v = 2Lf_1 = 2 \times 0.328 \times 440 \approx \qty{289}{m/s}$ ---
set by tension and mass, tuned by the peg. Pressing a finger
shortens $L$ and raises every $f_n$ at once: the spectrum slides up.
\end{example}

\begin{remark}[Why instruments differ]\label{rem:g12:sound-acoustics:instruments}
How the string is excited --- plucked, bowed, struck --- sets how
much of each mode is present, and the body amplifies some harmonics
more than others; excitation plus body fix the spectrum, so the same
$f_1$ leaves a guitar, a violin and a piano each sounding like itself.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:1}
In air ($v = \qty{340}{m/s}$), find the wavelength of (a) a
\qty{55}{Hz} bass note; (b) a \qty{1.0}{kHz} beep; (c) a
\qty{15}{kHz} whine. Compare the largest to the smallest.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:2}
Classify as infrasound, audible sound or ultrasound: \qty{8}{Hz};
\qty{100}{Hz}; \qty{15}{kHz}; \qty{40}{kHz}; \qty{3}{MHz}. Of the
audible ones, which sounds lower in pitch?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:3}
A worker strikes a rail \qty{1.0}{km} away. Compute the travel times
in the steel and in the air. What does a listener with one ear
pressed to the rail hear?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:4}
(a) The sound level for $I = \qty{1.0e-5}{W/m^2}$? (b) The
intensity at \qty{60}{dB}? (c) Name an everyday sound near each level.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:5}
A violin G string sounds $f_1 = \qty{196}{Hz}$: give its next three
harmonics. Why does a \qty{12}{kHz} whistle, however rich at the
source, reach the ear as a pure tone?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:6}
A siren radiates $P = \qty{0.50}{W}$ equally in all directions.
Compute the intensity and the sound level \qty{5.0}{m} away.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:7}
One violin gives \qty{68}{dB} at your seat. What level do two give?
Four? How many violins would it take to reach \qty{78}{dB}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:8}
The level is \qty{95}{dB} at \qty{2.0}{m} from a small loudspeaker.
Predict the level at \qty{8.0}{m}: below the \qty{85}{dB} mark?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:9}
A \qty{440}{Hz} note passes from air into water. Compute its
wavelength in each medium. The diver hears the same pitch: which
quantity is fixed by the source, and which one adjusts?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:10}
A spectrum shows peaks at \qtylist{130;260;390;520}{Hz} of
decreasing heights: what is the fundamental? Another instrument
plays the same note at the same loudness: what is the same in its
spectrum, what differs?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:11}
A guitar string of length \qty{65.0}{cm} sounds
$f_1 = \qty{110}{Hz}$. Find the wave speed, then the vibrating
lengths giving the octave (\qty{220}{Hz}) and the fifth (\qty{165}{Hz}).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:12}
A medical probe sends \qty{5.0}{MHz} ultrasound into soft tissue
($v = \qty{1540}{m/s}$): the wavelength? Echoes return after
\qty{26}{\micro s} and \qty{58}{\micro s} from an organ's two walls:
find their depths and the organ's thickness.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:13}
An outdoor stage produces \qty{110}{dB} at \qty{3.0}{m}. Treating it
as a small source, at what distance does the level fall to
\qty{85}{dB}? Give two reasons the real distance differs.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:14}
Find the ratio of the pain threshold (\qty{1}{W/m^2}) to $I_0$.
Taking the eardrum as \qty{0.5}{cm^2}, compute the power received
at each threshold; comment on the ear as a detector.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:15}
Each identical festival loudspeaker alone gives \qty{80}{dB} at the
mixing desk. What level do $2$ give? $10$? $100$? What does each
extra \qty{+10}{dB} cost --- what does this say of loudness races?
\end{exercise}

\section{Problem: The Luthier's Workshop}

\begin{problem}[{Weekend problem --- the luthier's workshop: a guitar
string laid out fret by fret, its harmonics tuned into a timbre, a
recital kept below the danger line, and an ultrasound hunt for a
flaw in the wood}]
\label{pb:g12:sound-acoustics:1}
A luthier is finishing a guitar. Its A string has vibrating length
$L = \qty{65.0}{cm}$ and must sound $f_1 = \qty{110}{Hz}$. Take
\qty{340}{m/s} for sound in air, and admit $f_1 = v/(2L)$
(\cref{prop:g12:sound-acoustics:string-modes}), $v$ being the wave
speed on the string.

\medskip
\noindent\textbf{Part I --- Laying out the fretboard.}
\begin{enumerate}
  \item Compute the wave speed $v$ on the string.
  \item Compute the fundamental's wavelength \emph{on the string},
        then the wavelength of the \qty{110}{Hz} sound \emph{in the
        air}. Why do they differ at the same frequency?
  \item The octave (\qty{220}{Hz}): what vibrating length gives it?
        Where must the twelfth fret sit, as a fraction of $L$?
  \item The fifth (\qty{165}{Hz}): what vibrating length gives it?
  \item Each fret raises the note one semitone, a factor
        $2^{1/12}$. Check with logarithms how many semitones separate
        \qty{110}{Hz} from \qty{220}{Hz}; length at the first fret?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Harmonics and timbre.}
\begin{enumerate}[resume]
  \item List the first five harmonics of the open A string.
  \item A violinist's A sounds \qty{440}{Hz}. Which harmonic of the
        guitar string is that, and what musical interval separates
        it from the fundamental? And \qty{330}{Hz}?
  \item The luthier plucks near the bridge, then over the
        fingerboard: same pitch, different color. Explain with spectra.
  \item A light touch at the midpoint while plucking forces a node
        there. Which harmonics survive, and what pitch is heard?
  \item How many harmonics of the open string lie, in principle,
        within the audible range?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The recital.}
At the guitar's first concert, each instrument radiates an acoustic
power $P = \qty{6.0e-3}{W}$ as a small source, equally in all
directions.
\begin{enumerate}[resume]
  \item Compute the intensity at $r = \qty{3.0}{m}$ from one instrument.
  \item Deduce the sound level there.
  \item A quartet plays: what level do four such instruments give
        at the same spot? Why is it not four times the level?
  \item Twelve instruments play: what is the level at \qty{3.0}{m}?
  \item How far back must a listener sit for the ensemble to fall
        to \qty{85}{dB}?
  \item Safety rule: \qty{85}{dB} is safe for eight hours, and
        every \qty{+3}{dB} halves the safe time. How long may the
        front-row listener of question 14 safely stay?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The flaw in the wood.}
Before varnishing, the luthier probes a \qty{45}{mm} thick neck
blank with a \qty{2.0}{MHz} tester; in this wood take $v = \qty{5.0e3}{m/s}$.
\begin{enumerate}[resume]
  \item Compute the ultrasound wavelength in the wood. Why must a
        flaw-hunting wave be ultrasonic rather than audible?
  \item Compute the echo delay from the back face of the blank.
  \item An echo arrives after \qty{7.2}{\micro s}: how deep is the flaw?
  \item Write the workshop card: wave speed on the string, twelfth-
        and first-fret lengths, safe listening distance, flaw depth.
\end{enumerate}
\end{problem}
