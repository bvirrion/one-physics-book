\chapter{RC and RL Circuits}\label{ch:g12:rc-rl-circuits}

Press a camera's shutter halfway: a thin whine climbs for two seconds, a
light turns green --- only then will the flash fire, dumping in a millisecond
what it spent two seconds gathering. This chapter meets the two components
that give circuits a memory and a clock --- capacitor and coil --- and
physics' first differential equations, checkable with this year's derivatives.

\section{The capacitor}

\begin{definition}[Capacitor and capacitance]\label{def:g12:rc-rl-circuits:capacitor}
A \emph{capacitor}\index{capacitor} is a pair of facing metal plates
separated by a thin insulator. Wired to a source, the plates take
opposite charges $+q$ and $-q$ --- an electric field fills the gap
(\cref{ch:g11:electric-gravitational-fields}) --- and the voltage $u$
across the plates is proportional to the stored charge:
\[ q = C\,u . \]
The constant $C$ is the \emph{capacitance}\index{capacitance}, in
\emph{farads}\index{farad} (\unit{F}): $\qty{1}{F} = \qty{1}{C/V}$.
\end{definition}

\begin{example}[Orders of magnitude]\label{ex:g12:rc-rl-circuits:orders}
One farad is enormous --- one coulomb parked per volt. Ceramics of radio
circuits: \unit{pF} to \unit{nF}; electrolytics of power supplies and
flashes: \unit{\micro F} to \unit{mF}; supercapacitors: thousands of farads.
\end{example}

\begin{proposition}[Current into a capacitor]\label{prop:g12:rc-rl-circuits:cdudt}
The current arriving on a capacitor's plate is the rate at which its
charge grows:
\[
i = \frac{dq}{dt} = C\,\frac{du}{dt} ,
\]
lowercase marking time-varying quantities ($U$, $I$ stay steady,
\cref{ch:g11:circuits-and-power}).
\end{proposition}

\begin{proof}
Current is charge delivered per second
(\cref{def:g11:circuits-and-power:current}): over a shrinking interval,
$i$ is the derivative of $q$; and $q = Cu$ with $C$ constant gives
$C\,du/dt$.
\end{proof}

\begin{remark}[Two consequences]\label{rem:g12:rc-rl-circuits:continuity}
If $u$ is constant, $i = 0$: a charged capacitor passes no steady current.
And $u$ can never jump --- a step would demand infinite current: the
capacitor's voltage is the circuit's memory.
\end{remark}

\section{Charging a capacitor: the RC circuit}

One loop --- source $\mathcal{E}$ (\cref{def:g11:circuits-and-power:emf}),
switch, resistor, empty capacitor: close $K$ at $t = 0$.

\begin{theorem}[Charging law of the RC circuit]\label{thm:g12:rc-rl-circuits:charging}
\index{RC circuit}
After the switch closes, the capacitor voltage obeys, then follows from
$u(0) = 0$:
\[
RC\,\frac{du}{dt} + u = \mathcal{E},
\qquad
u(t) = \mathcal{E}\left(1 - e^{-t/\tau}\right),
\qquad \tau = RC .
\]
\end{theorem}

\begin{proof}
Voltages along the loop add (\cref{prop:g11:circuits-and-power:laws}):
$\mathcal{E} = u_R + u = Ri + u$ at every instant; substitute
$i = C\,du/dt$ (\cref{prop:g12:rc-rl-circuits:cdudt}). Now \emph{verify}
the candidate: $du/dt = (\mathcal{E}/\tau)e^{-t/\tau}$, so
$RC\,(\mathcal{E}/RC)\,e^{-t/\tau} + \mathcal{E}(1 - e^{-t/\tau}) =
\mathcal{E}$, true for all $t$; and $u(0) = \mathcal{E}(1-1) = 0$. That
no \emph{other} curve fits the equation and the start is proved in this
year's mathematics course on differential equations.
\end{proof}

\begin{definition}[Time constant]\label{def:g12:rc-rl-circuits:tau}
The \emph{time constant}\index{time constant} of an RC circuit is
$\tau = RC$. Units: $\unit{\ohm} \times \unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$ --- the circuit's
own tempo.
\end{definition}

\begin{method}[Reading $\tau$ on the curve]\label{met:g12:rc-rl-circuits:reading}
Three equivalent readings, straight off a graph:
\begin{enumerate}
  \item \emph{tangent at the origin:} $u'(0) = \mathcal{E}/\tau$, so
        the initial tangent reaches the asymptote at $t = \tau$;
  \item \emph{63\%:} $u(\tau) = \mathcal{E}(1 - e^{-1}) \approx
        0.63\,\mathcal{E}$;
  \item \emph{95\%:} $u(3\tau) \approx 0.95\,\mathcal{E}$ --- charged,
        for practical purposes ($5\tau$: $99\%$).
\end{enumerate}
\end{method}

\begin{proposition}[Discharge]\label{prop:g12:rc-rl-circuits:discharge}
A capacitor charged to $U_0$ and closed on a resistor $R$ alone obeys
$RC\,du/dt + u = 0$ and follows
\[
u(t) = U_0\,e^{-t/\tau}, \qquad \tau = RC
\]
--- $37\%$ left at $\tau$, $5\%$ at $3\tau$.
\end{proposition}

\begin{proof}
Same loop law, no source; substitution again:
$RC(-U_0/\tau)e^{-t/\tau} + U_0 e^{-t/\tau} = 0$, with $u(0) = U_0$.
\end{proof}

\begin{example}[A flash unit's numbers]\label{ex:g12:rc-rl-circuits:flashrc}
A photo flash charges $C = \qty{800}{\micro F}$ through
$R = \qty{1.0}{k\ohm}$: $\tau = 10^3 \times 8.0\times10^{-4} =
\qty{0.80}{s}$, so the ``ready'' light --- wired to trip at $95\%$ ---
waits $3\tau \approx \qty{2.4}{s}$: the whine of the hook. Fired
through the flash tube's mere \qty{0.50}{\ohm}, the \emph{same}
capacitor empties with $\tau' = \qty{0.40}{ms}$: two thousand times
faster out than in.
\end{example}

\section{The coil: inductance and the RL circuit}

\begin{definition}[Inductor and inductance]\label{def:g12:rc-rl-circuits:inductor}
An \emph{inductor}\index{inductor} is a coil of wire: its current
threads its own turns with a magnetic field
(\cref{ch:g11:magnetic-fields}); whenever it \emph{changes}, a voltage
appears across the coil:
\[
u = L\,\frac{di}{dt} .
\]
The constant $L$ is the \emph{inductance}\index{inductance}, in
\emph{henries}\index{henry} (\unit{H}): $\qty{1}{H} = \qty{1}{V.s/A}$.
\end{definition}
\admitted

\begin{remark}[The coil resists change]\label{rem:g12:rc-rl-circuits:oppose}
Why a changing current makes a voltage --- induction --- is derived in
the university volumes; here we take the law and its character: the
coil \emph{opposes changes} of its current. Mirror twins: a capacitor's
$u$ cannot jump, a coil's $i$ cannot jump.
\end{remark}

    to[R=$R$] (4.2,2.4) -- (5.4,2.4) to[L=$L$, i=$i$, v=$u$] (5.4,0) -- (0,0);
\end{circuitikz}

{\small The RL circuit: after $K$ closes, the coil lets the current
climb only at its own pace $\tau = L/R$.}
\end{omfigure}

\begin{proposition}[Current rise in an RL circuit]\label{prop:g12:rc-rl-circuits:rl}
\index{RL circuit}
Closing the switch on a source $\mathcal{E}$, a resistor $R$ and a coil
$L$ in series gives $\mathcal{E} = Ri + L\,di/dt$, i.e.
\[
\frac{L}{R}\,\frac{di}{dt} + i = \frac{\mathcal{E}}{R},
\qquad\text{solved by}\quad
i(t) = \frac{\mathcal{E}}{R}\left(1 - e^{-t/\tau}\right),
\quad \tau = \frac{L}{R} .
\]
\end{proposition}

\begin{proof}
The loop law again; and it is the \emph{same equation} as in
\cref{thm:g12:rc-rl-circuits:charging} with $u \to i$,
$\mathcal{E} \to \mathcal{E}/R$, $RC \to L/R$ --- the same substitution
verifies the solution, and every reading of
\cref{met:g12:rc-rl-circuits:reading} transfers.
\end{proof}

\begin{example}[An electromagnet wakes up]\label{ex:g12:rc-rl-circuits:rl}
A relay coil, $L = \qty{0.30}{H}$ and $R = \qty{6.0}{\ohm}$, on a
\qty{12}{V} supply: final current $\mathcal{E}/R = \qty{2.0}{A}$, at
the pace $\tau = 0.30/6.0 = \qty{50}{ms}$ --- awake in $3\tau \approx
\qty{0.15}{s}$.
\end{example}

\begin{remark}[The spark at opening]\label{rem:g12:rc-rl-circuits:spark}
Open a switch on a live coil and $i$ must crash to zero in a fraction
of a millisecond: $di/dt$ is huge and $u = L\,di/dt$ reaches hundreds
or thousands of volts --- a spark jumps the contact. A car's ignition
coil does it on purpose, thousands of times a minute.
\end{remark}

\section{Stored energy}

\begin{proposition}[Energy in a capacitor, energy in a coil]\label{prop:g12:rc-rl-circuits:energy}
\index{energy!of a capacitor}\index{energy!of a coil}
A capacitor $C$ charged to voltage $u$ and a coil $L$ carrying a current
$i$ store
\[
E_C = \tfrac12\,C u^2 = \frac{q^2}{2C},
\qquad
E_L = \tfrac12\,L i^2 .
\]
\end{proposition}

\begin{proof}
Landing a small charge $dq$ on plates already at voltage $u$ costs
$u\,dq$ (\cref{def:g11:circuits-and-power:voltage}); stacking the
costs, the total is the area under the line $u = q/C$ --- the triangle
below --- so $E_C = \tfrac12 qu = \tfrac12 Cu^2$. The same triangle for
the coil (raising $i$ by $di$ costs $Li\,di$) gives
$E_L = \tfrac12 Li^2$.
\end{proof}

\begin{example}[Energy orders of magnitude]\label{ex:g12:rc-rl-circuits:energies}
The flash capacitor of \cref{ex:g12:rc-rl-circuits:flashrc}:
$\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$. A
\qty{3000}{F} supercapacitor at \qty{2.7}{V}: \qty{1.1e4}{J} --- an AA
battery's worth. A \qty{10}{mH} coil at \qty{10}{A}: \qty{0.50}{J}.
Less than batteries hold --- but releasable in a millisecond.
\end{example}

\begin{remark}[What the two time-keepers are for]\label{rem:g12:rc-rl-circuits:applications}
\index{smoothing}\index{timing circuit}
Slow-in, fast-out powers the camera flash and the defibrillator. The
delay $\tau = RC$ runs timing circuits: intermittent wipers, blinking
beacons, stairwell lights. A capacitor across a supply \emph{smooths}
voltage ($u$ cannot jump); a series coil smooths current ($i$ cannot
jump). And coil plus capacitor together make an oscillator
(\cref{ch:g12:rlc-oscillations}).
\end{remark}

\end{exercise}

Compute $\tau$ for $R = \qty{10}{k\ohm}$, $C = \qty{100}{\micro F}$, then

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:rc-rl-circuits:5}

\begin{exercise}[$\star\star$]\label{exo:g12:rc-rl-circuits:6}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rc-rl-circuits:8}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rc-rl-circuits:9}
The current in a \qty{0.50}{H} coil climbs steadily from $0$ to

\begin{exercise}[$\star\star$]\label{exo:g12:rc-rl-circuits:10}

\begin{exercise}[$\star\star$]\label{exo:g12:rc-rl-circuits:11}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rc-rl-circuits:12}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rc-rl-circuits:13}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rc-rl-circuits:14}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rc-rl-circuits:15}

\end{exercise}

\section{Problem: The camera flash}

\begin{problem}[{Weekend problem --- the camera flash: a two-second
whine, a millisecond of glory, and the wall-mounted cousin that restarts
hearts --- one exponential runs them all}]
\label{pb:g12:rc-rl-circuits:1}
A compact flash unit stores energy in a capacitor
$C = \qty{800}{\micro F}$, charged through $R = \qty{1.0}{k\ohm}$ from
a converter modeled as an ideal source $\mathcal{E} = \qty{300}{V}$;
the ``ready'' light trips at $95\%$ of full voltage. Fired, the
capacitor sees only the flash tube, $R_f = \qty{0.50}{\ohm}$. Data: an
AA cell delivers \qty{1.5}{V} and \qty{2.4}{A.h}; $e = \qty{1.60e-19}{C}$.

\medskip
\noindent\textbf{Part I --- The reservoir.}
\begin{enumerate}
  \item Compute the charge on the capacitor at \qty{300}{V}.
  \item How many elementary charges is that?
  \item Compute the stored energy $E_C$.
  \item The AA cell holds $q = It \approx \qty{8.6e3}{C}$, hence about
        \qty{1.3e4}{J}. How many times the capacitor's energy is that?
  \item So why bother with the capacitor? One sentence, using ``power''.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The wait for green.}
\begin{enumerate}[resume]
  \item Write the loop law and turn it into the differential equation
        for $u(t)$.
  \item Verify by substitution that $u(t) = \mathcal{E}(1 - e^{-t/RC})$
        solves it, starting from $u(0) = 0$.
  \item Compute the time constant $\tau$.
  \item After what time does the ready light turn green, and at what
        voltage? Recognize the two-second whine.
  \item Compute the initial charging current; why does it then die
        away?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The millisecond of glory.}
\begin{enumerate}[resume]
  \item Write the discharge law $u(t)$ through the tube and compute the
        new time constant $\tau'$.
  \item Compute the peak current through the tube.
  \item Compute the peak power; compare it with a \qty{2.2}{kW} kettle.
  \item Taking the flash as over at $3\tau'$, give its duration and
        average power.
  \item Form the ratio of charging time to flash time: same capacitor,
        same charge --- what changed, and what does that make a
        capacitor: an energy tank, or a power lever?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The cousin on the wall.}
A defibrillator must deliver $E_C = \qty{200}{J}$ at
$U_0 = \qty{1500}{V}$; chest and paddles present about \qty{50}{\ohm}.
\begin{enumerate}[resume]
  \item Compute the required capacitance.
  \item Compute the stored charge.
  \item Compute the shock's time constant and duration ($3\tau$);
        compare with the flash.
  \item Compute the peak current and peak power in the patient.
  \item The recharge circuit must make the device ready ($95\%$) in
        \qty{6.0}{s}: find the required charging resistance and the
        initial charging current --- the numbers a designer would order.
\end{enumerate}
\end{problem}
