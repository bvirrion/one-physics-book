\chapter{Énergie nucléaire~: fission, fusion, $E=mc^2$}\label{ch:g12:nuclear-energy}

Une pastille de combustible de la taille d'une gomme à crayon couvre
l'électricité d'un foyer pour un an~; le même service au charbon
prend trois tonnes. Au-dessus, le Soleil paie le jour en disparaissant
--- quatre millions de tonnes de lui-même par seconde. Ce chapitre
ouvre le grand livre derrière les deux marchés~: la masse elle-même
est de l'\omterm{def:g10:energy-conservation:energy}{énergie}, et les noyaux
(\cref{ch:g11:nucleus-radioactivity}) sont là où on l'encaisse.

\section{La masse est de l'énergie}

\begin{theorem}[Équivalence masse--énergie]\label{thm:g12:nuclear-energy:emc2}\index{équivalence masse--énergie}
Un corps de masse $m$, par le seul fait d'exister, détient
l'\omterm{def:g10:energy-conservation:energy}{énergie}
\[
E = m c^2, \qquad c = \qty{3.00e8}{m/s}.
\]
Chaque fois qu'un système perd une masse $\Delta m$, il libère
l'\omterm{def:g10:energy-conservation:energy}{énergie} $\Delta m\, c^2$ --- quel que soit le
mécanisme.
\end{theorem}
\admitted

\begin{remark}[Einstein, 1905]\label{rem:g12:nuclear-energy:einstein}
C'est la relation d'Einstein, admise ici~; d'où elle vient est
l'affaire du chapitre de relativité restreinte qui clôt cette année
(\cref{ch:g12:special-relativity}). Noter le taux~:
$c^2 = \qty{9.00e16}{J/kg}$ --- une erreur d'arrondi de masse est une
fortune d'\omterm{def:g10:energy-conservation:energy}{énergie}.
\end{remark}

\begin{definition}[Électronvolt et unité de masse atomique]\label{def:g12:nuclear-energy:units}
Deux \omterm{def:g10:orders-of-magnitude:unit}{unités} taillées pour le
\omterm{def:g11:fundamental-interactions:ladder}{noyau}. L'\emph{électronvolt}\index{électronvolt},
l'\omterm{def:g10:energy-conservation:energy}{énergie} qu'un électron gagne en traversant
\qty{1}{V} (\cref{ch:g11:circuits-and-power})~:
$\qty{1}{eV} = \qty{1.60e-19}{J}$, $\qty{1}{MeV} = \qty{1.60e-13}{J}$
--- les liaisons chimiques échangent quelques \unit{eV}, les réactions
nucléaires des \omterm{def:g11:circuits-and-power:voltage}{mégaélectronvolts}. L'\emph{unité de
masse atomique unifiée}\index{unité de masse atomique unifiée}, un
douzième de la masse d'un \omterm{def:g11:fundamental-interactions:ladder}{atome} de
carbone-12~: $\qty{1}{u} = \qty{1.66054e-27}{kg}$, d'équivalent
d'\omterm{def:g10:energy-conservation:energy}{énergie} \qty{931.5}{MeV} ---
$\qty{1}{u} = \qty{931.5}{MeV}/c^2$.
\end{definition}

\begin{example}[La fortune endormie]\label{ex:g12:nuclear-energy:fortune}
Un \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de n'importe quoi, entièrement converti,
vaut $\qty{9.00e16}{J}$ --- trois ans de production d'une grande
centrale de \omterm{def:g11:work-of-force:power}{puissance} (\qty{1}{GW}). Personne ne l'avait
remarqué avant 1905 parce que la physique ordinaire touche à peine la
masse~: brûler un \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de charbon
(\qty{3.0e7}{J}) l'allège d'un tiers de microgramme. Les noyaux, on le
verra, bougent d'une partie pour mille~: encore petit, mais un million
de fois la chimie.
\end{example}

\section{Le défaut de masse}

\begin{definition}[Défaut de masse et énergie de liaison]\label{def:g12:nuclear-energy:defect}
Peser un \omterm{def:g11:fundamental-interactions:ladder}{noyau} ${}^{A}_{Z}\mathrm{X}$ (masse
$m$), puis peser ses $Z$~protons et $A - Z$~neutrons séparément~: les
parties sont \emph{plus lourdes} que le tout. La différence
\[
\Delta m = Z\,m_p + (A - Z)\,m_n - m > 0,
\qquad m_p = \qty{1.00728}{u},\; m_n = \qty{1.00866}{u},
\]
est le \emph{défaut de masse}\index{défaut de masse}~;
l'\emph{énergie de liaison}\index{énergie de liaison}
$E_b = \Delta m\,c^2$ est l'\omterm{def:g10:energy-conservation:energy}{énergie} nécessaire pour
écarter le \omterm{def:g11:fundamental-interactions:ladder}{noyau} en
\omterm{def:g11:fundamental-interactions:ladder}{nucléons} libres --- également
l'\omterm{def:g10:energy-conservation:energy}{énergie} libérée lorsqu'on l'assemble. Un
\omterm{def:g11:fundamental-interactions:ladder}{noyau} lié se place \emph{sous} ses parties.
\end{definition}

\omterm{def:g11:fundamental-interactions:ladder}{nucléons} séparés --- l'assembler libère $E_b$,

\begin{example}[Hélium-4]\label{ex:g12:nuclear-energy:helium}
Le \omterm{def:g11:fundamental-interactions:ladder}{noyau} ${}^{4}_{2}\mathrm{He}$ a
$m = \qty{4.00151}{u}$~; ses parties~:
$2 \times 1.00728 + 2 \times 1.00866 = \qty{4.03188}{u}$. Donc
$\Delta m = \qty{0.03037}{u}$ et $E_b = 0.03037 \times 931.5 \approx
\qty{28.3}{MeV}$~: le \omterm{def:g11:fundamental-interactions:ladder}{noyau} pèse $0.75\%$ de
moins que ses parties --- le «~une partie pour mille~» de
l'\cref{ex:g12:nuclear-energy:fortune}.
\end{example}

\begin{method}[Bilan d'énergie d'une réaction nucléaire]\label{met:g12:nuclear-energy:balance}
\begin{enumerate}
  \item Équilibrer l'équation~: $A$ et $Z$ totaux conservés
        (\cref{ch:g11:nucleus-radioactivity}).
  \item Ajouter les masses avant, puis après, en \unit{u} --- garder
        les cinq décimales.
  \item $\Delta m = m_{\text{avant}} - m_{\text{après}}$, puis
        $E = \Delta m\,(\text{en u}) \times \qty{931.5}{MeV}$, emporté
        comme \omterm{def:g11:mechanical-energy:kinetic}{énergie cinétique} des produits~;
        négatif signifie que la réaction doit être payée.
\end{enumerate}
\end{method}

\section{La courbe qui mène l'univers}

\begin{definition}[Énergie de liaison par nucléon]\label{def:g12:nuclear-energy:pernucleon}
La cohésion se compare équitablement par l'\emph{énergie de liaison
par nucléon}\index{énergie de liaison par nucléon}, $E_b/A$, le coût
moyen d'extraire un \omterm{def:g11:fundamental-interactions:ladder}{nucléon}~:
$28.3/4 \approx \qty{7.1}{MeV}$ pour l'hélium-4.
\end{definition}

\begin{proposition}[Le pic du fer, et deux routes vers le bas]\label{prop:g12:nuclear-energy:ironpeak}\index{pic du fer}
Portée en fonction de $A$, l'\omterm{def:g12:nuclear-energy:pernucleon}{énergie de liaison par
nucléon} grimpe raide à travers les noyaux légers, culmine près du
fer, ${}^{56}_{26}\mathrm{Fe}$, à environ \qty{8.8}{MeV} par
\omterm{def:g11:fundamental-interactions:ladder}{nucléon}, puis glisse doucement vers environ
\qty{7.6}{MeV} à l'uranium~: le fer est le
\omterm{def:g11:fundamental-interactions:ladder}{noyau} le plus solidement lié de la nature. Une
réaction libère de l'\omterm{def:g10:energy-conservation:energy}{énergie} exactement lorsque ses
produits se placent \emph{plus haut} sur la courbe~; deux stratégies
opposées paient donc toutes deux, et toutes deux marchent vers le fer
--- \emph{fendre} les noyaux les plus lourds, et \emph{fusionner} les
plus légers.
\end{proposition}
\admitted

\begin{remark}[Pourquoi un pic]\label{rem:g12:nuclear-energy:whypeak}
Le bras de fer du \cref{ch:g11:fundamental-interactions}~: la colle
forte ne lie que les voisins qui se touchent, la répulsion de Coulomb
spanne tout le \omterm{def:g11:fundamental-interactions:ladder}{noyau} --- les petits noyaux
sont surtout surface, les grands surtout répulsion, le fer le
compromis. La comptabilité honnête vit dans les volumes
universitaires.
\end{remark}

\section{Fission et réaction en chaîne}

\begin{definition}[Fission]\label{def:g12:nuclear-energy:fission}
La \emph{fission}\index{fission} est la scission d'un
\omterm{def:g11:fundamental-interactions:ladder}{noyau} lourd en deux fragments de taille
moyenne. L'uranium-235 est \emph{fissile}\index{fissile}~: un neutron
lent, absorbé, laisse le \omterm{def:g11:fundamental-interactions:ladder}{noyau} si agité qu'il
se déchire en deux, crachant deux ou trois neutrons frais ---
${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to \text{deux fragments} +
2\text{--}3\;{}^{1}_{0}\mathrm{n}$, la paire variant d'une fission à
l'autre.
\end{definition}

\begin{example}[Une fission, pesée]\label{ex:g12:nuclear-energy:fissionbalance}
Une route fréquente, par la \cref{met:g12:nuclear-energy:balance}~:
\[
{}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \longrightarrow
{}^{94}_{38}\mathrm{Sr} + {}^{140}_{54}\mathrm{Xe} + 2\,{}^{1}_{0}\mathrm{n}.
\]
Avant~: $1.00866 + 234.99350 = \qty{236.00216}{u}$. Après~:
$93.89450 + 139.89200 + 2 \times 1.00866 = \qty{235.80382}{u}$. D'où
$\Delta m = \qty{0.19834}{u}$ et $E \approx \qty{185}{MeV}$~; en
comptant les désintégrations ultérieures des fragments radioactifs
(\cref{ch:g12:radioactive-decay}), chaque
\omterm{def:g12:nuclear-energy:fission}{fission} vaut environ \qty{200}{MeV} --- quelques
\unit{eV} achètent un \omterm{def:g11:fundamental-interactions:ladder}{atome} de charbon.
\end{example}

\begin{definition}[Réaction en chaîne et masse critique]\label{def:g12:nuclear-energy:chain}
Chaque \omterm{def:g12:nuclear-energy:fission}{fission} est allumée par un neutron et en libère
deux ou trois~: dans un bloc assez grand ils frappent d'autres noyaux
et la réaction se nourrit elle-même --- une \emph{réaction en
chaîne}\index{réaction en chaîne}. Dans un petit bloc la plupart des
neutrons s'échappent d'abord par la surface~; la quantité minimale
soutenant la chaîne est la \emph{masse critique}\index{masse
critique} --- en dessous la chaîne s'éteint, au-dessus chaque
génération multiplie~: le principe de la bombe.
\end{definition}

\begin{definition}[Le réacteur, dompté]\label{def:g12:nuclear-energy:reactor}
Un réacteur de \omterm{def:g11:work-of-force:power}{puissance} tient la chaîne à exactement un
neutron par \omterm{def:g12:nuclear-energy:fission}{fission}. Le combustible~: des pastilles
d'uranium modestement enrichies en uranium-235. Le
\emph{modérateur}\index{modérateur} (eau ordinaire, d'ordinaire)
ralentit les neutrons rapides de \omterm{def:g12:nuclear-energy:fission}{fission} par
collisions --- les neutrons lents sont bien meilleurs pour déclencher
la \omterm{def:g12:nuclear-energy:fission}{fission} suivante~; les \emph{barres de
contrôle}\index{barre de contrôle} (bore, cadmium) dévorent les
neutrons --- enfoncées, la chaîne meurt~; tirées, elle s'accélère.
L'agitation des fragments devient chaleur, puis vapeur, puis
électricité (\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Un grain de poivre contre un camion]\label{ex:g12:nuclear-energy:gram}
Un gramme d'uranium-235 tient
$N = \num{1.0e-3} / (235 \times \num{1.66054e-27}) \approx \num{2.56e21}$
noyaux~; à $\qty{200}{MeV} = \qty{3.2e-11}{J}$ chacun, les fissionner
tous donne environ \qty{8.2e10}{J} --- l'\omterm{def:g10:energy-conservation:forms}{énergie
chimique} de $\num{8.2e10} / \num{3.0e7} \approx \qty{2.7e3}{kg}$ de
charbon~: un grain de poivre contre trois tonnes.
\end{example}

\section{Fusion~: le feu propre du Soleil}

\begin{definition}[Fusion]\label{def:g12:nuclear-energy:fusion}
La \emph{fusion}\index{fusion} est la fusion de deux noyaux légers en
un plus lourd, plus solidement lié. La réaction la plus accessible
marie le deutérium et le tritium, les
\omterm{def:g11:nucleus-radioactivity:notation}{isotopes} lourds de l'hydrogène
(\cref{ch:g11:nucleus-radioactivity})~:
\[
{}^{2}_{1}\mathrm{H} + {}^{3}_{1}\mathrm{H} \longrightarrow
{}^{4}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}.
\]
\end{definition}

\begin{example}[D--T, pesé]\label{ex:g12:nuclear-energy:dt}
Avant~: $2.01355 + 3.01550 = \qty{5.02905}{u}$~; après~:
$4.00151 + 1.00866 = \qty{5.01017}{u}$~; donc $\Delta m =
\qty{0.01888}{u}$ et $E \approx \qty{17.6}{MeV}$, quatre cinquièmes
sur le neutron --- moins qu'une \omterm{def:g12:nuclear-energy:fission}{fission}, mais de
cinq \omterm{def:g11:fundamental-interactions:ladder}{nucléons} au lieu de $236$~: environ
\qty{3.4e14}{J} par \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de combustible, quatre
fois la \omterm{def:g12:nuclear-energy:fission}{fission}. L'hélium-4
(\cref{ex:g12:nuclear-energy:helium}) est une fois de plus la cendre.
\end{example}

\begin{definition}[La barrière de Coulomb]\label{def:g12:nuclear-energy:barrier}
Deux noyaux sont tous deux positifs~: pour se toucher, ils doivent
d'abord grimper la colline de leur répulsion électrique
(\cref{ch:g11:electric-gravitational-fields}) --- la \emph{barrière de
Coulomb}\index{barrière de Coulomb}. Seulement près de \qty{1e8}{K}
les collisions portent l'\omterm{def:g10:energy-conservation:energy}{énergie} pour la
franchir~; le combustible est alors un \emph{plasma}\index{plasma} de
noyaux nus et d'électrons. La \omterm{def:g12:nuclear-energy:fission}{fission} ne paie pas un
tel péage~: son déclencheur, le neutron, est neutre, et
\omterm{def:g11:work-of-force:work}{travaille} à froid.
\end{definition}

\begin{example}[Le Soleil en vit]\label{ex:g12:nuclear-energy:sun}
Le cœur du Soleil, à $\qty{1.5e7}{K}$,
\omterm{def:g11:circuits-and-power:joule}{fusionne} l'hydrogène ordinaire pas à pas --- la
\emph{chaîne proton--proton}, d'effet net
$4\,{}^{1}_{1}\mathrm{H} \to {}^{4}_{2}\mathrm{He} + 2\,{}^{0}_{+1}\mathrm{e}
+ \text{rayonnement}$, environ \qty{26}{MeV} par hélium~: $0.7\%$ de
la masse de l'hydrogène part en \omterm{def:g10:energy-conservation:energy}{énergie}. La lumière
solaire dont le \omterm{def:g12:sound-acoustics:timbre}{spectre} se lit au
\cref{ch:g10:light-spectra} est cette masse, arrivant avec huit
minutes de retard~; toute étoile brille en marchant la courbe vers le
fer.
\end{example}

\begin{remark}[ITER~: embouteiller une étoile]\label{rem:g12:nuclear-energy:iter}
Aucun solide ne peut tenir un \omterm{def:g12:nuclear-energy:barrier}{plasma} à
\qty{1.5e8}{K}~; un \emph{tokamak} l'enferme dans un
\omterm{def:g11:magnetic-fields:bfield}{champ magnétique} (\cref{ch:g11:magnetic-fields}), un
beignet de \omterm{def:g11:electric-gravitational-fields:lines}{lignes de champ} le long
desquelles les particules chargées spiralent sans toucher la paroi~;
ITER, construit pour libérer dix fois sa
\omterm{def:g11:work-of-force:power}{puissance} de chauffage, est l'étape actuelle. Briser
le confinement et le \omterm{def:g12:nuclear-energy:barrier}{plasma} refroidit et s'arrête en
secondes~: pas de \omterm{def:g12:nuclear-energy:chain}{réaction en chaîne}, pas de
\omterm{def:g12:nuclear-energy:chain}{masse critique} --- le difficile n'est pas d'éteindre
le feu mais de le garder allumé.
\end{remark}

\begin{example}[L'échelle d'énergie, par kilogramme]\label{ex:g12:nuclear-energy:ladder}
\omterm{def:g10:energy-conservation:energy}{Énergie} par \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de
combustible~: charbon, \qty{3.0e7}{J}~; uranium-235 par
\omterm{def:g12:nuclear-energy:fission}{fission}, \qty{8.2e13}{J}~; deutérium--tritium par
\omterm{def:g12:nuclear-energy:fusion}{fusion}, \qty{3.4e14}{J}~; conversion totale
($E = mc^2$), \qty{9.0e16}{J}. Six ordres de grandeur séparent la
chimie du \omterm{def:g11:fundamental-interactions:ladder}{noyau} --- la raison pour laquelle un
réacteur se recharge au camion et une centrale à charbon au train.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:1}
Convertir~: (a)~\qty{1}{eV} et \qty{1}{MeV} en
\omterm{def:g11:work-of-force:work}{joules}~; (b)~les \qty{200}{MeV} d'une
\omterm{def:g12:nuclear-energy:fission}{fission} en \omterm{def:g11:work-of-force:work}{joules}~;
(c)~vérifier à partir de $\qty{1}{u} = \qty{1.66054e-27}{kg}$ que son
équivalent d'\omterm{def:g10:energy-conservation:energy}{énergie} est proche de \qty{931.5}{MeV}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:2}
Un morceau de sucre a une masse de \qty{6.0}{g}~: calculer sa valeur
pleine $E = mc^2$. Un foyer tire $\qty{10}{kW.h} = \qty{3.6e7}{J}$ par
jour --- pendant combien d'années le morceau, entièrement converti,
pourrait-il l'\omterm{def:g11:work-of-force:power}{alimenter}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:3}
Le \omterm{def:g11:fundamental-interactions:ladder}{noyau} de carbone-12 a pour masse
\qty{11.99671}{u}~: calculer son \omterm{def:g12:nuclear-energy:defect}{défaut de masse}, son
\omterm{def:g12:nuclear-energy:defect}{énergie de liaison} en \unit{MeV}, et son
\omterm{def:g12:nuclear-energy:pernucleon}{énergie de liaison par nucléon}
($m_p = \qty{1.00728}{u}$, $m_n = \qty{1.00866}{u}$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:4}
Compléter chaque \omterm{def:g12:nuclear-energy:fission}{fission}, en donnant $A$, $Z$ et
l'élément ($Z = 36$~: krypton, $Z = 52$~: tellure)~:
(a)~${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to
{}^{144}_{56}\mathrm{Ba} + {}^{A}_{Z}\mathrm{X} + 3\,{}^{1}_{0}\mathrm{n}$~;
(b)~${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to

\section{Problème~: Alimenter une ville}

\begin{problem}[{Problème du week-end --- alimenter une ville~: un
gigawatt stable pour un demi-million de personnes, acheté de quatre
manières --- pastilles d'uranium, trains de charbon, une piscine d'eau
de mer, la substance même du Soleil --- avec le même kilogramme de
masse disparue caché sous les quatre}]
\label{pb:g12:nuclear-energy:1}
Une ville de $500\,000$~habitants tire une
\omterm{def:g11:work-of-force:power}{puissance} électrique stable $P = \qty{1.0}{GW}$~; ses
centrales, quel que soit le combustible, transforment la chaleur en
électricité avec un \omterm{def:g10:energy-conservation:efficiency}{rendement} de $33\%$
(\cref{ch:g10:energy-conservation}). Données~: une
\omterm{def:g12:nuclear-energy:fission}{fission} d'uranium-235, \qty{200}{MeV} en tout~;
charbon, \qty{3.0e7}{J/kg}~; une \omterm{def:g12:nuclear-energy:fusion}{fusion} D--T,
\qty{17.6}{MeV} issus de \qty{5.029}{u} de combustible~;
$\qty{1}{u} = \qty{1.66054e-27}{kg}$~; $\qty{1}{eV} = \qty{1.60e-19}{J}$~;
un an $= \qty{3.16e7}{s}$~; le Soleil~: \qty{3.9e26}{W},
\qty{2.0e30}{kg}~; un \omterm{def:g10:orders-of-magnitude:unit}{mètre} cube d'eau de mer
tient environ \qty{33}{g} de deutérium.

\medskip
\noindent\textbf{Partie I --- La centrale à
\omterm{def:g12:nuclear-energy:fission}{fission}.}
\begin{enumerate}
  \item Quelle \omterm{def:g11:work-of-force:power}{puissance} thermique le réacteur doit-il
        livrer~?
  \item Exprimer l'\omterm{def:g10:energy-conservation:energy}{énergie} d'une
        \omterm{def:g12:nuclear-energy:fission}{fission} en \omterm{def:g11:work-of-force:work}{joules}.
  \item Combien de \omterm{def:g12:nuclear-energy:fission}{fissions} par seconde font tourner
        la ville~?
  \item La masse d'un \omterm{def:g11:fundamental-interactions:ladder}{noyau} d'uranium-235
        ($A = 235$), puis l'uranium-235 consommé par seconde~?
  \item En déduire l'uranium-235 consommé en un an, en
        \omterm{def:g10:orders-of-magnitude:unit}{kilogrammes}.
  \item Le combustible est enrichi à $4.0\%$ d'uranium-235~: quelle
        masse prend un an, et tient-elle dans un camion~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le grand livre du charbon.}
\begin{enumerate}[resume]
  \item Calculer l'\omterm{def:g10:energy-conservation:forms}{énergie thermique} que la ville
        consomme en un an.
  \item Quelle masse de charbon la livre~?
  \item Calculer le rapport (masse de charbon)/(masse d'uranium-235)
        pour un an, et le vérifier contre l'échelle de
        l'\cref{ex:g12:nuclear-energy:ladder}.
  \item Un train de marchandises transporte \qty{3.0e6}{kg}~: combien
        de trains par an, et par jour~? Une phrase~: se représenter
        les deux lignes d'approvisionnement.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le rêve de
\omterm{def:g12:nuclear-energy:fusion}{fusion}.}
\begin{enumerate}[resume]
  \item Exprimer l'\omterm{def:g10:energy-conservation:energy}{énergie} d'une
        \omterm{def:g12:nuclear-energy:fusion}{fusion} D--T en \omterm{def:g11:work-of-force:work}{joules}.
  \item Calculer l'\omterm{def:g10:energy-conservation:energy}{énergie} libérée par
        \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de combustible D--T~; comparer
        avec l'échelle.
  \item Quelle masse de combustible D--T couvre l'année de la ville~?
        Quelle masse de cela est du deutérium ($2.014$ des
        \qty{5.029}{u})~?
  \item Quel volume d'eau de mer contient ce deutérium~? Comparer à
        une piscine olympique, environ \qty{2.5e3}{m^3}. (Le tritium
        est engendré à partir du lithium.)
  \item Pourquoi cette centrale ne porte-t-elle ni
        \omterm{def:g12:nuclear-energy:chain}{masse critique} ni
        \omterm{def:g12:nuclear-energy:chain}{réaction en chaîne} à craindre~? Quelle est
        alors la partie difficile~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- L'étoile qui paie en masse.}
\begin{enumerate}[resume]
  \item À partir de $E = mc^2$, quelle masse le Soleil convertit-il
        chaque seconde~?
  \item Quelle masse par seconde, et par an, toute source thermique
        de \qty{3.0}{GW} doit-elle convertir~? Vérifier que les trois
        combustibles rendent cette même masse.
  \item Quelle masse totale le Soleil a-t-il rayonnée en
        $\num{4.5e9}$~ans, et quelle fraction du Soleil est-ce~?
  \item La \omterm{def:g12:nuclear-energy:fusion}{fusion} peut puiser environ $0.07\%$ de la
        masse du Soleil (la part brûlable de son cœur)~: estimer sa
        durée de vie totale comme étoile, et ce qui reste.
  \item Clore l'audit en trois phrases~: une année de la ville en
        tonnes de charbon, \omterm{def:g10:orders-of-magnitude:unit}{kilogrammes} d'uranium-235
        et \omterm{def:g10:orders-of-magnitude:unit}{kilogrammes} de combustible D--T --- et le
        \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de masse qui, dans chaque
        scénario, est devenu la lumière de la ville.
\end{enumerate}
\end{problem}
