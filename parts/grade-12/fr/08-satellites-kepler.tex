\chapter{Satellites et mouvement planétaire}\label{ch:g12:satellites-kepler}

L'antenne sur le balcon est boulonnée, pourtant toute la journée elle
fixe un point fixe du ciel, à \qty{36000}{km}~; la station spatiale
traverse le ciel du soir en minutes~; une constellation d'horloges
volantes dit à votre téléphone où vous êtes. Aucune de ces machines
n'a de moteur en marche~: elles sont toutes simplement en train de
\emph{tomber}. Ce chapitre calcule à quelle vitesse et à quelle
hauteur chacune doit tomber --- puis pèse des planètes avec la même
loi.

\section{Tomber autour de la Terre}

Tout repose sur la loi mesurée plus tôt
(\cref{ch:g10:universal-gravitation})~: la Terre, de masse $M$, attire
un \omterm{def:g10:universal-gravitation:satellite}{satellite} de masse $m$ dont le centre est à la
distance $r$ du sien avec une \omterm{def:g10:inertia:force}{force} dirigée vers le centre
de la Terre, de norme
\[ F = G\,\frac{m M}{r^2}, \qquad G = \qty{6.67e-11}{N.m^2/kg^2}. \]
Attention à ce qu'est $r$~: la distance \emph{entre centres}. Un
\omterm{def:g10:universal-gravitation:satellite}{satellite} à l'altitude $h$ a $r = R + h$, avec
$R = \qty{6.37e6}{m}$ le rayon de la Terre --- oublier le $R$ est la
bévue classique de ce chapitre.

\begin{remark}[Le canon de Newton, revisité]\label{rem:g12:satellites-kepler:cannon}
Tiré horizontalement, un boulet atterrit~; plus vite, il atterrit plus
loin~; à la bonne vitesse le sol s'éloigne exactement aussi vite que
le boulet tombe, et la chute se referme sur elle-même~: une
\omterm{def:g10:universal-gravitation:satellite}{orbite}. Un \omterm{def:g10:universal-gravitation:satellite}{satellite}
n'a pas besoin de moteur --- seulement de la bonne vitesse
\emph{latérale}.
\end{remark}

\section{L'orbite circulaire~: une vitesse, une période}

\begin{theorem}[Vitesse orbitale]\label{thm:g12:satellites-kepler:speed}
Un \omterm{def:g10:universal-gravitation:satellite}{satellite} en \omterm{def:g12:kinematics-2d:ucm}{mouvement circulaire uniforme}
d'\omterm{def:g10:universal-gravitation:satellite}{orbite} de rayon $r$ autour d'un corps de masse $M$ voyage
à la vitesse
\[ v = \sqrt{\frac{G M}{r}}, \]
qui ne dépend pas de la masse du \omterm{def:g10:universal-gravitation:satellite}{satellite}.
\end{theorem}

\begin{proof}
La gravité est la seule \omterm{def:g10:inertia:force}{force}, donc la
\omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton}
(\cref{thm:g12:newtons-laws:second}) s'écrit $m\vect a = \vect F$~;
la projeter sur le \omterm{def:g12:kinematics-2d:frenet}{repère de Frenet}
(\cref{def:g12:kinematics-2d:frenet}). La \omterm{def:g10:inertia:force}{force} vise le
centre~: sa composante tangentielle est nulle, donc la vitesse est
constante~; sa composante normale est tout le $GmM/r^2$, et le
mouvement circulaire a $a_n = v^2/r$
(\cref{thm:g12:kinematics-2d:centripetal}). D'où $m v^2/r =
GmM/r^2$, et --- $m$ s'annulant --- $v^2 = GM/r$.
\end{proof}

\begin{proposition}[Période orbitale]\label{prop:g12:satellites-kepler:period}
Le \omterm{def:g10:universal-gravitation:satellite}{satellite} accomplit son
\omterm{def:g10:universal-gravitation:satellite}{orbite} en la \omterm{def:g12:kinematics-2d:ucm}{période}
\[ T = \frac{2\pi r}{v} = 2\pi \sqrt{\frac{r^3}{G M}}. \]
\end{proposition}

\begin{proof}
Un tour est la circonférence $2\pi r$ à vitesse constante~; injecter
$v = \sqrt{GM/r}$~: $T = 2\pi r\sqrt{r/(GM)} = 2\pi\sqrt{r^3/(GM)}$.
\end{proof}

Ni $v$ ni $T$ ne contiennent $m$~: une vis secouée de la station
spatiale \omterm{def:g10:universal-gravitation:satellite}{orbite} juste à côté d'elle. Et toutes deux
diminuent avec $r$~: \emph{plus l'\omterm{def:g10:universal-gravitation:satellite}{orbite} est haute,
plus le \omterm{def:g10:universal-gravitation:satellite}{satellite} est lent}, en vitesse
($v \propto 1/\sqrt r$) et en angle ($T \propto r^{3/2}$).

\begin{example}[La station spatiale]\label{ex:g12:satellites-kepler:iss}
La Station spatiale internationale vole à l'altitude
$h \approx \qty{400}{km}$, donc $r = \num{6.37e6} + \num{4.0e5} =
\qty{6.77e6}{m}$~; avec $GM = \num{6.67e-11} \times \num{5.97e24} =
\qty{3.98e14}{m^3/s^2}$, $v = \sqrt{\num{3.98e14}/\num{6.77e6}}
\approx \qty{7.7e3}{m/s}$ et $T = 2\pi r/v \approx \qty{5.5e3}{s}
\approx \qty{92}{min}$~: un tour de planète toutes les heures et demie,
environ seize levers de soleil par jour pour les astronautes.
\end{example}

\begin{example}[La Lune tombe comme la pomme]\label{ex:g12:satellites-kepler:moon}
La Lune, à $r = \qty{3.84e8}{m}$, \omterm{def:g10:universal-gravitation:satellite}{orbite} en
$T = \qty{27.3}{d} = \qty{2.36e6}{s}$, donc
$v = 2\pi r/T \approx \qty{1.02e3}{m/s}$ et
$a = v^2/r \approx \qty{2.7e-3}{m/s^2}$ --- précisément $g/60^2$, et
la Lune est à $60$ rayons terrestres~: la traction qui fait tomber la
pomme courbe la Lune, affaiblie par l'inverse du carré. Chaque seconde
la Lune «~tombe~» d'environ \qty{1.4}{mm} vers nous, son kilomètre
latéral l'emportant autour --- le contrôle que Newton fit en 1666.
\end{example}

\section{Les lois de Kepler}

Newton n'a pas deviné l'inverse du carré~: il l'a extrait de trois
régularités que Johannes Kepler distilla, entre 1609 et 1619, des vingt
années de positions planétaires à l'œil nu de Tycho Brahe --- les
meilleures données du monde pré-télescope.

\begin{definition}[Ellipse]\label{def:g12:satellites-kepler:ellipse}
Une \emph{ellipse}\index{ellipse} est l'ensemble des points dont les
distances à deux points fixes, les \emph{foyers}\index{foyer (d'une
ellipse)}, ont une somme constante~; la moitié de son plus long
diamètre est le \emph{demi-grand axe}\index{demi-grand axe} $a$, et un
cercle est le cas où les foyers fusionnent. Les points de
l'\omterm{def:g10:universal-gravitation:satellite}{orbite} d'une planète les plus proches et les plus
éloignés du Soleil sont son \emph{périhélie}\index{périhélie} et son
\emph{aphélie}\index{aphélie}.
\end{definition}

\begin{theorem}[Lois de Kepler]\label{thm:g12:satellites-kepler:kepler}
\index{lois de Kepler}%
Pour les planètes orbitant le Soleil (masse $M$)~:
\begin{enumerate}
  \item chaque \omterm{def:g10:universal-gravitation:satellite}{orbite} est une
        \omterm{def:g12:satellites-kepler:ellipse}{ellipse} avec le Soleil en un foyer~;
  \item le segment Soleil--planète balaye des aires égales en temps
        égaux~;
  \item le rapport du carré de la \omterm{def:g12:kinematics-2d:ucm}{période} au cube du
        \omterm{def:g12:satellites-kepler:ellipse}{demi-grand axe} est le même pour toutes les planètes~:
        $T^2/a^3 = 4\pi^2/(G M)$.
\end{enumerate}
Les mêmes lois gouvernent toute famille de
\omterm{def:g10:universal-gravitation:satellite}{satellites} autour de tout corps central, avec $M$ la
masse centrale. \admitted
\end{theorem}

\begin{remark}[Ce que l'on peut prouver cette année]\label{rem:g12:satellites-kepler:proof}
Pour une \omterm{def:g10:universal-gravitation:satellite}{orbite} \emph{circulaire} la troisième loi
est nôtre~: en élevant au carré le
\cref{prop:g12:satellites-kepler:period} on obtient
$T^2/r^3 = 4\pi^2/(GM)$ --- une constante pour chaque
\omterm{def:g10:universal-gravitation:satellite}{satellite} du même centre. Que les
\omterm{def:g12:satellites-kepler:ellipse}{ellipses}, les aires égales et le $a$ général
suivent de l'inverse du carré est montré, avec plus de calcul, dans le
volume de Licence~1. En attendant la deuxième loi se lit~: la planète
va le plus vite au \omterm{def:g12:satellites-kepler:ellipse}{périhélie}, où le court segment de
balayage doit se hâter.
\end{remark}

\section{Les orbites au travail}

\begin{definition}[Orbite géostationnaire]\label{def:g12:satellites-kepler:geo}
Un \omterm{def:g10:universal-gravitation:satellite}{satellite} est \emph{géostationnaire}\index{orbite
géostationnaire} lorsqu'il reste au-dessus d'un point fixe du sol. Son
\omterm{def:g10:universal-gravitation:satellite}{orbite} doit être circulaire, équatoriale, et de
\omterm{def:g12:kinematics-2d:ucm}{période} un \emph{jour sidéral}\index{jour sidéral},
$T = \qty{86164}{s}$ (\qty{23}{h} \qty{56}{min} \qty{4}{s})~: le temps
que met la Terre pour tourner une fois \emph{par rapport aux étoiles}.
(Le jour de \qty{24}{h} est relatif au Soleil, vers lequel la Terre
avance aussi d'un degré par jour.)
\end{definition}

\begin{example}[L'altitude géostationnaire]\label{ex:g12:satellites-kepler:geoalt}
La troisième loi de Kepler, lue à l'envers, dicte le rayon~:
$r = \left(GMT^2/4\pi^2\right)^{1/3} =
\left(\num{3.98e14} \times (\num{86164})^2/4\pi^2\right)^{1/3}
\approx \qty{4.22e7}{m}$ --- une altitude $h = r - R \approx
\qty{3.58e7}{m} \approx \qty{35800}{km}$, parcourue à $v = 2\pi r/T
\approx \qty{3.1}{km/s}$. Tout relais qui «~reste immobile~» siège sur
ce seul cercle au-dessus de l'équateur --- il n'y en a pas d'autre.
\end{example}

\begin{example}[La constellation de navigation]\label{ex:g12:satellites-kepler:gps}
Les systèmes de navigation par satellite font voler une trentaine de
\omterm{def:g10:universal-gravitation:satellite}{satellites} à $r \approx \qty{2.66e7}{m}$ (altitude
$\approx \qty{20200}{km}$), où le \cref{prop:g12:satellites-kepler:period}
donne $T \approx \qty{4.32e4}{s}$~: un demi-\omterm{def:g12:satellites-kepler:geo}{jour sidéral},
si bien que chaque \omterm{def:g10:universal-gravitation:satellite}{satellite} retrace sa trace au sol
quotidiennement. Votre récepteur compare les temps d'arrivée de leurs
\omterm{def:g10:signals-and-waves:signal}{signaux} d'horloge à des \omterm{def:g10:universal-gravitation:satellite}{orbites} connues
au mètre près.
\end{example}

\begin{method}[Peser un monde à partir de son satellite]\label{met:g12:satellites-kepler:weigh}
\index{peser une planète}%
Pour mesurer la masse d'une planète ou d'une étoile~: trouver n'importe
quel \omterm{def:g10:universal-gravitation:satellite}{satellite} d'elle (une lune, une sonde, une
planète), mesurer le rayon $r$ et la \omterm{def:g12:kinematics-2d:ucm}{période} $T$ de
son \omterm{def:g10:universal-gravitation:satellite}{orbite}, et inverser la troisième loi~:
\[ M = \frac{4\pi^2 r^3}{G\,T^2}. \]
Cela pèse seulement le corps \emph{central} --- la masse du
\omterm{def:g10:universal-gravitation:satellite}{satellite} s'est annulée dans le
\cref{thm:g12:satellites-kepler:speed}. Chaque masse de la carte de
données est venue ainsi~: la Terre depuis la Lune, le Soleil depuis
l'\omterm{def:g10:universal-gravitation:satellite}{orbite} propre de la Terre.
\end{method}

\begin{remark}[Plus haut est plus lent --- mais coûte plus]\label{rem:g12:satellites-kepler:energy}
Comme $v = \sqrt{GM/r}$, la Lune rampe à \qty{1}{km/s} tandis que la
station court à \qty{7.7}{km/s}. Pourtant l'\omterm{def:g10:universal-gravitation:satellite}{orbite}
plus haute est la plus \emph{chère}~: grimper contre la gravité stocke
de l'\omterm{def:g10:energy-conservation:energy}{énergie} potentielle (le compte est réglé au
\cref{ch:g12:work-and-energy}), et le gain en $E_p$ l'emporte sur la
perte en $E_k$. D'où une fusée qui brûle \emph{vers l'avant} pour
atteindre une \omterm{def:g10:universal-gravitation:satellite}{orbite} plus haute et plus lente --- et
un \omterm{def:g10:universal-gravitation:satellite}{satellite} freiné par un air \omterm{def:g11:lenses-and-eye:lens}{ténu}
spirale \emph{vers le bas} et accélère.
\end{remark}

\begin{remark}[Carte de données]\label{rem:g12:satellites-kepler:data}
Sauf mention contraire~: $G = \qty{6.67e-11}{N.m^2/kg^2}$~; Terre~:
$M = \qty{5.97e24}{kg}$, $R = \qty{6.37e6}{m}$,
\omterm{def:g12:satellites-kepler:geo}{jour sidéral} \qty{86164}{s}~; \omterm{def:g10:universal-gravitation:satellite}{orbite} de
la Lune~: $r = \qty{3.84e8}{m}$, $T = \qty{27.3}{d}$~; Soleil~:
$M = \qty{1.99e30}{kg}$~; distance Terre--Soleil \qty{1.496e11}{m}~; une
année $= \qty{3.156e7}{s}$~; Mars~: $M = \qty{6.42e23}{kg}$,
$R = \qty{3.39e6}{m}$.
\end{remark}

\section{Exercices}


\begin{exercise}[$\star$]\label{exo:g12:satellites-kepler:1}

\begin{exercise}[$\star$]\label{exo:g12:satellites-kepler:5}

\begin{exercise}[$\star\star$]\label{exo:g12:satellites-kepler:6}

\begin{exercise}[$\star\star$]\label{exo:g12:satellites-kepler:7}

\begin{exercise}[$\star\star$]\label{exo:g12:satellites-kepler:8}

\begin{exercise}[$\star\star$]\label{exo:g12:satellites-kepler:9}

\begin{exercise}[$\star\star$]\label{exo:g12:satellites-kepler:10}

\section{Problème~: Un siège au-dessus de l'équateur}

\begin{problem}[{Problème du week-end --- placer un
\omterm{def:g10:universal-gravitation:satellite}{satellite} au-dessus de l'équateur~: le jour
sidéral fixe la période, la troisième loi de Kepler fixe le rayon, la
station spatiale vérifie la loi, et le mois d'Io pèse Jupiter}]
\label{pb:g12:satellites-kepler:1}
Vous concevez un relais de télévision géostationnaire, puis vous
vérifiez les lois de Kepler sur la station spatiale et sur Io. Données
de la carte~: $G$, $M$, $R$ de la Terre~; jour sidéral
\qty{86164}{s}~; $GM = \qty{3.98e14}{m^3/s^2}$. Pour Jupiter,
prendre $r_{\text{Io}} = \qty{4.22e8}{m}$ et
$T_{\text{Io}} = \qty{1.769}{d}$.

\medskip
\noindent\textbf{Partie I --- Pourquoi le jour sidéral.}
\begin{enumerate}
  \item Pourquoi la période d'un \omterm{def:g10:universal-gravitation:satellite}{satellite}
        géostationnaire est-elle le jour sidéral, non \qty{24}{h}~?
  \item Pourquoi l'\omterm{def:g10:universal-gravitation:satellite}{orbite} doit-elle être circulaire~?
        (Indices~: force résultante du MCU~; direction de la gravité.)
  \item Pourquoi doit-elle être équatoriale~? Un «~hover~» au-dessus de
        Paris est-il possible~?
  \item De la \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton} en projection
        normale, dériver $v = \sqrt{GM/r}$.
  \item En déduire $T = 2\pi\sqrt{r^3/(GM)}$ et $T^2/r^3 = 4\pi^2/(GM)$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Dimensionner le relais.}
\begin{enumerate}[resume]
  \item Calculer le rayon $r$ de l'\omterm{def:g10:universal-gravitation:satellite}{orbite}
        géostationnaire.
  \item En déduire l'altitude $h$.
  \item Calculer la vitesse orbitale $v$.
  \item Vérifier avec $v = \sqrt{GM/r}$.
  \item Un \omterm{def:g10:universal-gravitation:satellite}{satellite} deux fois plus massif a-t-il le
        même $r$~? Qu'est-ce qui change~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La station comme contrôle.}
La station vole à $r = \qty{6.77e6}{m}$.
\begin{enumerate}[resume]
  \item Calculer sa vitesse.
  \item Calculer sa période.
  \item Combien d'\omterm{def:g10:universal-gravitation:satellite}{orbites} par jour solaire~?
  \item Vérifier que $r_{\text{geo}}/r_{\text{ISS}}$ élevé à $3/2$
        redonne le rapport des périodes (troisième loi de Kepler).
  \item Laquelle est plus rapide, et laquelle coûte plus d'\omterm{def:g10:energy-conservation:energy}{énergie}
        par kilogramme~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Peser Jupiter avec Io.}
\begin{enumerate}[resume]
  \item Écrire la formule qui donne $M_J$ à partir de $r$ et $T$ d'Io.
  \item Calculer $M_J$.
  \item Comparer à la Terre et au Soleil.
  \item Un \omterm{def:g10:universal-gravitation:satellite}{satellite} géostationnaire autour de Jupiter
        aurait quelle période (égale au jour sidéral jovien
        \qty{9.925}{h})~? Calculer son rayon et le comparer à Io.
  \item En une phrase~: où parquer le relais, à quelle vitesse, et
        quelle masse de Jupiter a livrée la même loi~?
\end{enumerate}
\end{problem}
