\chapter{Chute libre et mouvement de projectile}\label{ch:g12:projectile-motion}

Chaque jet d'un tuyau d'arrosage dessine la même courbe dans l'air.
Ainsi fait un ballon de basket en arc vers le panier, un sauteur en
longueur, un boulet de canon~: tout corps abandonné à son
\omterm{def:g10:universal-gravitation:weight}{poids}. Ce chapitre gagne cette courbe --- une parabole ---
à partir de la \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton} en deux lignes de
calcul, lit sommet, temps de vol et \omterm{def:g12:projectile-motion:range}{portée} sur son
équation, puis laisse l'air revenir pour voir ce qu'il gâche.

\section{Chute libre}

\begin{definition}[Chute libre]\label{def:g12:projectile-motion:freefall}
Un corps est en \emph{chute libre}\index{chute libre} lorsque la seule
\omterm{def:g10:inertia:force}{force} qui agit sur lui est son
\omterm{def:g10:universal-gravitation:weight}{poids} $\vect P = m\vect g$~: pas de support, pas de
fil, résistance de l'air négligeable. Une pierre lâchée ou une balle
qui a quitté la main est, à bonne approximation, en chute libre ---
«~chute~» inclut la montée.
\end{definition}

\begin{theorem}[Loi de Galilée de la chute libre]\label{thm:g12:projectile-motion:galileo}
\index{loi de Galilée de la chute libre}
En \omterm{def:g12:projectile-motion:freefall}{chute libre}, tous les corps partagent la même
accélération, quelle que soit leur masse~:
\[
\vect a = \vect g, \qquad g = \qty{9.81}{m/s^2},
\]
dirigée tout droit vers le bas~: l'intensité de
\omterm{def:g11:electric-gravitational-fields:field}{champ} \qty{9.81}{N/kg} du
\cref{ch:g10:universal-gravitation} dans son autre tenue,
$\unit{N/kg} = \unit{m/s^2}$.
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{Deuxième loi de Newton} (\cref{ch:g12:newtons-laws}) avec le
\omterm{def:g10:universal-gravitation:weight}{poids} comme seule \omterm{def:g10:inertia:force}{force}~:
$m\vect a = m\vect g$, et la masse s'annule. Marteau et plume ne
diffèrent que par l'air.
\end{proof}

\begin{proposition}[Chute verticale depuis le repos]\label{prop:g12:projectile-motion:vertical}
Lâché du repos, avec la profondeur tombée $h$ comptée vers le bas
depuis le point de lâcher, un corps en \omterm{def:g12:projectile-motion:freefall}{chute libre}
obéit à
\[
a = g, \qquad v = g t, \qquad h = \tfrac12\, g t^2 .
\]
\end{proposition}

\begin{proof}
$v(t)$ est la primitive de la constante $a = g$ s'annulant en $t = 0$
(\cref{ch:g12:kinematics-2d})~; $h(t)$ celle de $v = gt$ s'annulant en
$t = 0$.
\end{proof}

\begin{example}[Ordres de grandeur]\label{ex:g12:projectile-motion:orders}
Après $1$, $2$, $3$ secondes de \omterm{def:g12:projectile-motion:freefall}{chute libre}~:
$v = 9.8$, $19.6$, \qty{29.4}{m/s} ($35$, $71$, \qty{106}{km/h}) et
$h = 4.9$, $19.6$, \qty{44.1}{m}. Depuis un toit de \qty{30}{m}~:
$t = \sqrt{2h/g} \approx \qty{2.5}{s}$, arrivée à
\qty{24}{m/s} $\approx \qty{87}{km/h}$. Les temps croissent comme
$\sqrt h$ --- et au-delà de quelques secondes l'air refuse de rester
négligeable (dernière section).
\end{example}

\begin{example}[Lancer vertical]\label{ex:g12:projectile-motion:throw}
Lancé tout droit vers le haut à $v_0$, axe $Oy$ vers le haut depuis
la main~: les deux mêmes primitives donnent $v = v_0 - gt$ et
$y = v_0 t - \tfrac12 gt^2$. Le sommet, où $v = 0$, vient en
$t = v_0/g$, hauteur $h = v_0^2/2g$ --- la réponse
d'\omterm{def:g10:energy-conservation:energy}{énergie} $v_0 = \sqrt{2gh}$ du
\cref{ch:g11:mechanical-energy} lue à l'envers~; lancé \emph{vers le
bas}, $v_0$ change de signe. Pour \qty{15}{m/s}~: sommet \qty{11.5}{m}
en \qty{1.53}{s}, retour en \qty{3.06}{s}, à \qty{15}{m/s}.
\end{example}

\section{Le projectile~: une loi, deux primitives}

\begin{definition}[Projectile]\label{def:g12:projectile-motion:projectile}
Un \emph{projectile}\index{projectile} est un corps lancé avec une
vitesse $\vect v_0$ puis laissé en \omterm{def:g12:projectile-motion:freefall}{chute libre}. Axes
au point de lancement~: $Ox$ horizontal dans le plan vertical de
$\vect v_0$, $Oy$ vertical vers le haut~; l'\emph{angle de
lancement}\index{angle de lancement} $\alpha$ de $\vect v_0$ au-dessus
de l'horizontale fixe ses coordonnées
$(v_0\cos\alpha,\; v_0\sin\alpha)$.
\end{definition}


\begin{theorem}[Équations du mouvement]\label{thm:g12:projectile-motion:equations}
Un \omterm{def:g12:projectile-motion:projectile}{projectile} lancé depuis l'origine avec la vitesse
$v_0$ et l'angle $\alpha$ a $\vect a = (0,\,-g)$ à chaque instant, d'où
\[
\vect v(t) = \bigl(v_0\cos\alpha,\; v_0\sin\alpha - g t\bigr),
\qquad
\vect{OM}(t) = \bigl(v_0\cos\alpha\, t,\; v_0\sin\alpha\, t - \tfrac12 g t^2\bigr).
\]
\end{theorem}

\begin{proof}
Deux primitives de $\vect a = (0,-g)$~: $\vect v = (C_x,\, C_y - gt)$,
et $t = 0$ force $C_x = v_0\cos\alpha$, $C_y = v_0\sin\alpha$~; puis
$\vect{OM}$ est la primitive de $\vect v$ s'annulant en $t = 0$
(\cref{ch:g12:kinematics-2d}).
\end{proof}

\begin{proposition}[Indépendance des deux mouvements]\label{prop:g12:projectile-motion:independence}
Le mouvement horizontal est \omterm{def:g12:kinematics-2d:uniform}{rectiligne uniforme} à
$v_0\cos\alpha$~; le mouvement vertical est le même que pour un corps
lâché verticalement à $v_0\sin\alpha$. En particulier, une balle
lâchée et une balle tirée horizontalement depuis la même hauteur
touchent le sol au même instant.
\end{proposition}

\begin{proof}
Dans le \cref{thm:g12:projectile-motion:equations}, $x(t)$ n'implique
que $v_0\cos\alpha$, $y(t)$ seulement $v_0\sin\alpha$ et $g$~; les
deux balles ont $v_y(0) = 0$, donc toutes deux obéissent à
$h = \tfrac12 gt^2$.
\end{proof}

\begin{remark}[Le singe qui tombe]\label{rem:g12:projectile-motion:monkey}
Un gardien de zoo vise un dard \emph{droit sur} un singe, qui lâche sa
branche au tir. Mauvaise idée~: dard et singe tombent chacun de
$\tfrac12 gt^2$ sous leurs positions sans gravité --- la ligne de
visée, la branche --- donc ils se rencontrent exactement, quelle que
soit la vitesse du dard
(\cref{exo:g12:projectile-motion:15}).
\end{remark}

\section{La parabole, lue en entier}

\begin{theorem}[Trajectoire]\label{thm:g12:projectile-motion:trajectory}
En éliminant $t$ entre $x(t)$ et $y(t)$~: la
\omterm{def:g10:relative-motion:trajectory}{trajectoire} d'un
\omterm{def:g12:projectile-motion:projectile}{projectile} ($0 \leq \alpha < \ang{90}$) est la
parabole ouverte vers le bas
\[
y = x\tan\alpha \;-\; \frac{g}{2 v_0^2 \cos^2\alpha}\, x^2 .
\]
\end{theorem}

\begin{proof}
$t = x/(v_0\cos\alpha)$, substitué dans $y(t)$ du
\cref{thm:g12:projectile-motion:equations}.
\end{proof}

\begin{definition}[Portée]\label{def:g12:projectile-motion:range}
Pour un \omterm{def:g12:projectile-motion:projectile}{projectile} sur sol plat, la
\emph{portée}\index{portée} $R$ est la distance horizontale parcourue
lorsqu'il revient à sa hauteur de lancement~; le \emph{temps de
vol}\index{temps de vol} $t_f$ est la durée de ce trajet.
\end{definition}

\begin{proposition}[Sommet, temps de vol, portée]\label{prop:g12:projectile-motion:range}
Sur sol plat, avec $0 < \alpha < \ang{90}$~:
\[
t_f = \frac{2 v_0 \sin\alpha}{g}, \qquad
h_{\text{apex}} = \frac{(v_0\sin\alpha)^2}{2g}
\ \ \text{atteint en } t_f/2, \qquad
R = \frac{v_0^2 \sin(2\alpha)}{g}.
\]
La \omterm{def:g12:projectile-motion:range}{portée} est maximale à $\alpha = \ang{45}$, où
$R = v_0^2/g$~; les angles complémentaires $\alpha$ et
$\ang{90} - \alpha$ donnent la même \omterm{def:g12:projectile-motion:range}{portée}.
\end{proposition}

\begin{proof}
$y = 0$ donne $t\,(v_0\sin\alpha - \tfrac12 gt) = 0$~: lancement, et
$t_f$. Le sommet, $v_y = 0$, siège en $t = v_0\sin\alpha/g = t_f/2$~;
substituer dans $y(t)$. Puis $R = x(t_f) = 2v_0^2\sin\alpha\cos\alpha/g =
v_0^2\sin(2\alpha)/g$. Le sinus culmine à $2\alpha = \ang{90}$, et
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180} - 2\alpha) =
\sin(2\alpha)$.
\end{proof}

\begin{example}[Dégagement au but]\label{ex:g12:projectile-motion:kick}
Un gardien frappe à $v_0 = \qty{25}{m/s}$, $\alpha = \ang{45}$~:
$R = 25^2/9.81 \approx \qty{64}{m}$~; les vrais dégagements atterrissent
plus près de \qty{55}{m} --- la coupe de l'air.
\end{example}

\begin{method}[Comptabilité du projectile]\label{met:g12:projectile-motion:method}
\begin{enumerate}
  \item Axes au point de lancement, $Ox$ horizontal, $Oy$ vers le haut~;
        coordonnées de $\vect v_0$.
  \item Vérifier la \omterm{def:g12:projectile-motion:freefall}{chute libre}
        (\omterm{def:g10:universal-gravitation:weight}{poids} seul)~: alors $\vect a = (0,-g)$~;
        primitives deux fois, conditions initiales à chaque fois.
  \item Question sur un \emph{lieu}~: éliminer $t$
        (\omterm{def:g10:relative-motion:trajectory}{trajectoire}). Sur un \emph{instant}~: garder $t$.
  \item Dictionnaire~: sommet $\Leftrightarrow v_y = 0$~; hauteur de
        lancement $\Leftrightarrow y = 0$~; mur en $d \Leftrightarrow x = d$~;
        sol $\ell$ en dessous $\Leftrightarrow y = -\ell$.
  \item Santé~: vol symétrique par rapport au sommet~;
        $R \leq v_0^2/g$~; l'air ne fait jamais qu'écourter.
\end{enumerate}
\end{method}

\section{Ce que l'air change}

\begin{definition}[Vitesse limite]\label{def:g12:projectile-motion:terminal}
La \omterm{def:g11:circuits-and-power:resistance}{résistance} de l'air s'oppose à la vitesse et croît
avec elle. Un corps qui tombe n'accélère donc que jusqu'à ce que la
traînée équilibre le \omterm{def:g10:universal-gravitation:weight}{poids}~; dès lors
$\vect a = \vect 0$~: il tombe à une \emph{vitesse
limite}\index{vitesse limite} constante.
\end{definition}

\begin{example}[Ordres de grandeur de la vitesse limite]\label{ex:g12:projectile-motion:airorders}
Un parachutiste en chute libre~: environ \qty{55}{m/s}
(\qty{200}{km/h}). Une goutte de pluie~: environ \qty{9}{m/s} --- dans
le vide, \qty{2000}{m} de chute la livreraient à
$\sqrt{2gh} \approx \qty{2.0e2}{m/s}$. Une plume~: environ
\qty{1}{m/s}, atteinte en quelques centimètres --- le secret de sa
lenteur~: elle passe toute sa chute \emph{à} la
\omterm{def:g12:projectile-motion:terminal}{vitesse limite}~; le marteau n'approche jamais la
sienne.
\end{example}

\begin{remark}[Balistique réelle]\label{rem:g12:projectile-motion:drag}
La traînée pousse contre la vitesse tout le long du vol~: la
\omterm{def:g12:projectile-motion:range}{portée} raccourcit, le sommet s'abaisse, et la symétrie se
casse --- la descente plus raide que la montée, les longs tirs
visiblement courts de la parabole. Le traitement honnête et quantitatif
est une équation différentielle résolue dans le volume de Licence~1~;
cette année, la parabole est le bon modèle pour des
\omterm{def:g12:projectile-motion:projectile}{projectiles} denses et modérément rapides --- un ballon
de basket à \qty{10}{m/s}, non une balle de golf à \qty{70}{m/s}.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:1}
Une pierre lâchée du repos tombe pendant \qty{2.0}{s}. Calculer la
profondeur tombée et la vitesse d'arrivée, en \unit{m/s} et
\unit{km/h}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:2}
Sur la Lune, un marteau et une plume lâchés ensemble touchent le sol
ensemble. Pourquoi~? Sur Terre, la plume traîne très loin derrière~:
expliquer avec la \omterm{def:g12:projectile-motion:terminal}{vitesse limite}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:3}
Calculer le temps de chute et la vitesse d'arrivée depuis
\qty{1.0}{m}, \qty{10}{m} et \qty{100}{m}. Comment croissent les temps
avec la hauteur~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:4}
Une balle est lancée verticalement vers le haut à \qty{15}{m/s}.
Calculer la hauteur du sommet et le temps pour y parvenir~; quand et
à quelle vitesse revient-elle~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:5}
Une balle roule d'une table de \qty{0.80}{m} de haut à \qty{2.5}{m/s}
horizontalement. Calculer le temps de vol et la distance d'atterrissage.
Le temps de chute dépend-il de la vitesse horizontale~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:6}
Pour un même $v_0$, quelles paires d'angles donnent la même
\omterm{def:g12:projectile-motion:range}{portée}~? Quel angle maximise $R$~? Lequel d'une paire
complémentaire vole plus longtemps, et pourquoi~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:7}
Une pierre est jetée vers le bas à \qty{5.0}{m/s} depuis un balcon de
\qty{20}{m}. Écrire $y(t)$ (vers le bas positif), résoudre pour le
temps d'arrivée, et trouver la vitesse d'impact. Vérifier avec
l'\omterm{def:g10:energy-conservation:energy}{énergie}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:8}
Un arroseur lance de l'eau à \qty{40}{m/s} sous \ang{30}. Calculer
$t_f$, la hauteur du sommet et la \omterm{def:g12:projectile-motion:range}{portée} sur sol plat.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:9}
Un sauteur en longueur quitte le sol à \qty{10.5}{m/s} sous
\ang{20}~: prédire sa \omterm{def:g12:projectile-motion:range}{portée}. Les records mondiaux
approchent \qty{9}{m}~: qu'a oublié le modèle du point matériel~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:10}
Un fusil tire horizontalement à \qty{300}{m/s} depuis \qty{1.2}{m}
au-dessus du sol~; en même temps une balle tombe de la même hauteur.
Montrer qu'elles atterrissent ensemble, et où atterrit celle du fusil.
Que teste cette expérience~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:11}
Montrer que $\alpha$ et $\ang{90}-\alpha$ donnent la même
\omterm{def:g12:projectile-motion:range}{portée}. Pour $v_0 = \qty{25}{m/s}$ et
$R = \qty{50}{m}$, trouver les deux angles possibles.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:12}
Un arroseur de jardin a $v_0 = \qty{8.0}{m/s}$. Quelle est la
\omterm{def:g12:projectile-motion:range}{portée} maximale, et la hauteur maximale d'un jet
vertical~? Pour atteindre un point à \qty{5.0}{m} horizontalement à
hauteur nulle, quels angles fonctionnent~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:13}
Une balle est lancée à \qty{18}{m/s} sous \ang{35} depuis le sol.
Une haie de \qty{4.0}{m} se dresse à \qty{15}{m}. La balle la
franchit-elle, et de combien~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:14}
Un parachutiste de \qty{80}{kg} atteint \qty{55}{m/s} de
\omterm{def:g12:projectile-motion:terminal}{vitesse limite}. (a) Quelle force de traînée
l'équilibre alors~? (b) Esquisser $v(t)$ depuis le saut jusqu'à la

pluie tombant \qty{2000}{m} dans le vide.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:15}
Un dard est tiré à $v_0$, angle $\alpha$, droit sur un singe à la
distance horizontale $D$ et à la hauteur $H = D\tan\alpha$~; le singe
lâche au tir. Montrer qu'ils se rencontrent pour tout $v_0$ (tant que
le dard arrive avant le sol).
\end{exercise}


\section{Problème~: Le buzzer-beater de basket}

\begin{problem}[{Problème du week-end --- le buzzer-beater de basket~: un
tir libre calibré sur le bord, le sommet lu sur la parabole, un lancer
de toute la salle contre le buzzer, et le verdict de la résistance de
l'air}]
\label{pb:g12:projectile-motion:1}
Un match de basket~: le buzzer sonne dans \qty{2.0}{s}, le score est
égal, et un joueur a la balle à $d = \qty{4.20}{m}$ du panier
(ligne des lancers francs), le bord à \qty{3.05}{m} au-dessus du sol.
Il lâche depuis \qty{2.00}{m} au-dessus du sol sous $\alpha = \ang{50}$.
Prendre $g = \qty{9.81}{m/s^2}$~; négliger l'air jusqu'à la partie~IV.

\medskip
\noindent\textbf{Partie I --- Les équations du vol.}
\begin{enumerate}
  \item Quelle \omterm{def:g10:inertia:force}{force} agit pendant le vol (air négligé)~? En
        déduire $\vect a$ par la \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton}.
  \item Avec $Ox$ horizontal vers le panier et $Oy$ vers le haut
        depuis le lâcher, écrire $\vect v(t)$.
  \item Écrire $x(t)$ et $y(t)$.
  \item Éliminer $t$ pour obtenir l'équation de la
        \omterm{def:g10:relative-motion:trajectory}{trajectoire} $y(x)$.
  \item Le bord est en $x = d$, $y = 1.05\,\unit{m}$ (par rapport au
        lâcher). En déduire $v_0$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Lire le tir.}
\begin{enumerate}[resume]
  \item Calculer le temps de vol jusqu'au bord.
  \item Calculer l'instant et la hauteur du sommet~; le tir est-il
        encore en montée ou déjà en descente au bord~?
  \item De combien la balle passe-t-elle au-dessus des bouts des doigts
        d'un défenseur de \qty{2.60}{m} de hauteur d'atteinte (mains
        levées), placé sous le bord~?
  \item Calculer la vitesse d'entrée (norme et direction sous
        l'horizontale) au bord.
  \item Pourquoi une entrée plus raide est-elle plus indulgente pour le
        tireur~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Contre le buzzer, de toute la salle.}
Le joueur est maintenant à $D = \qty{26.0}{m}$ du panier, et le buzzer
sonne dans \qty{2.0}{s}. Le bord est encore à \qty{1.05}{m} au-dessus
du lâcher~; pour un premier calcul, traiter le sol comme plat au niveau
du lâcher (l'erreur est petite).
\begin{enumerate}[resume]
  \item Montrer que l'erreur de hauteur relative $1.05/D$ est petite.
        Sur sol plat, écrire $t_f$ en fonction de $v_0$ et $\alpha$.
  \item Montrer $R = v_0^2\sin(2\alpha)/g$~; à quel angle la
        \omterm{def:g12:projectile-motion:range}{portée} est-elle maximale~?
  \item Quelle $v_0$ minimale atteint \qty{26.0}{m} (à
        $\alpha = \ang{45}$)~?
  \item Un bras d'élite fournit environ \qty{17.5}{m/s}. La
        \omterm{def:g12:projectile-motion:range}{portée} maximale dépasse-t-elle \qty{26.0}{m}~?
  \item Pour $v_0 = \qty{17.5}{m/s}$ et $R = \qty{26.0}{m}$, trouver les
        deux angles possibles et leurs temps de vol. Lequel bat le
        buzzer~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Dunk aérien et l'air.}
\begin{enumerate}[resume]
  \item Un joueur dunk depuis \qty{6.0}{m} du panier sous \ang{75},
        lâcher à \qty{2.00}{m}, bord à \qty{3.05}{m}. Calculer $v_0$.
  \item La balle passe-t-elle au-dessus du bord supérieur du panneau
        (\qty{3.95}{m} au-dessus du sol) en $x = \qty{5.55}{m}$~?
  \item Calculer le sommet et le temps de vol~: commenter le «~temps
        d'arrêt~» du dunk.
  \item Qualitativement, que fait la traînée de l'air à chaque tir
        ci-dessus, et lequel souffre le plus~?
  \item Le verdict, une phrase avec ses nombres~: le tir de
        \qty{26.0}{m} est-il possible, avec quelle marge de
        \omterm{def:g12:projectile-motion:range}{portée}, et l'air l'efface-t-il~?
\end{enumerate}
\end{problem}
