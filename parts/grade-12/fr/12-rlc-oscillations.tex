\chapter{Oscillations électriques libres~: RLC}\label{ch:g12:rlc-oscillations}

Tournez le gros cadran d'une radio de grenier et les stations défilent
devant l'aiguille~: une voix, des violons, du grésillement, une voix
encore. Derrière le panneau il n'y a ni \omterm{def:g11:work-of-force:sign}{moteur} ni
ordinateur --- une \omterm{def:g11:magnetic-fields:solenoid}{bobine}, un
\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} dont les plaques entrelacées le cadran
fait tourner, et un bout de fil d'antenne. Ce chapitre relie tête-bêche
les deux réservoirs d'\omterm{def:g10:energy-conservation:energy}{énergie} du chapitre précédent
(\cref{ch:g12:rc-rl-circuits}) et trouve le pendule le plus rapide jamais
construit~: de la charge qui bascule d'avant en arrière, jusqu'à des
millions de fois par seconde, au tempo que le cadran choisit.

\section{Un pendule électrique~: le circuit LC}

Chargez un \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} $C$ à une
\omterm{def:g11:circuits-and-power:voltage}{tension} $U_0$ --- ses plaques tiennent alors
$\pm Q_0$ avec $Q_0 = C U_0$ --- et à $t = 0$ basculez-le sur une
\omterm{def:g11:magnetic-fields:solenoid}{bobine} \emph{idéale} $L$, de
\omterm{def:g11:circuits-and-power:resistance}{résistance} nulle. Aucune source nulle part dans
la boucle~: quoi qu'il arrive ensuite, le circuit le fait tout seul.

\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} chargé à $Q_0$, un interrupteur, une
\omterm{def:g11:magnetic-fields:solenoid}{bobine} idéale. Fermer $K$ laisse la charge se déverser
à travers la \omterm{def:g11:magnetic-fields:solenoid}{bobine} --- et la
\omterm{def:g11:magnetic-fields:solenoid}{bobine}, qui déteste le changement, ne la laissera pas
s'arrêter.}
\end{omfigure}

\begin{proposition}[L'équation LC]\label{prop:g12:rlc-oscillations:equation}
\index{circuit LC}
Dans la boucle LC idéale la charge $q(t)$ de la plaque supérieure
obéit à
\[
L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0,
\qquad\text{c'est-à-dire}\qquad
\frac{d^2q}{dt^2} = -\frac{1}{LC}\,q .
\]
\end{proposition}

\begin{proof}
Autour de la boucle les \omterm{def:g11:circuits-and-power:voltage}{tensions} se somment à zéro
(\cref{prop:g11:circuits-and-power:laws})~: $u_L + u_C = 0$. Le
\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} donne $u_C = q/C$ et la
\omterm{def:g11:magnetic-fields:solenoid}{bobine} $u_L = L\,di/dt$
(\cref{def:g12:rc-rl-circuits:inductor}), avec $i = dq/dt$ comptant la
charge arrivant sur la plaque supérieure
(\cref{prop:g12:rc-rl-circuits:cdudt})~; substituer.
\end{proof}

\begin{theorem}[Oscillations libres du circuit LC]\label{thm:g12:rlc-oscillations:oscillation}
\index{oscillations électriques}
Lâché à $t = 0$ avec $q(0) = Q_0$ et $i(0) = 0$, le circuit oscille~:
\[
q(t) = Q_0 \cos\!\left(\frac{2\pi t}{T_0}\right),
\qquad
i(t) = -I_0 \sin\!\left(\frac{2\pi t}{T_0}\right),
\qquad
T_0 = 2\pi\sqrt{LC},
\]
avec un courant de crête $I_0 = 2\pi Q_0/T_0 = Q_0/\sqrt{LC}$.
\end{theorem}

\begin{proof}
Vérifier par substitution. En dérivant,
$i = dq/dt = -\frac{2\pi}{T_0} Q_0 \sin(2\pi t/T_0)$ --- le $i(t)$
énoncé --- et encore $\frac{d^2q}{dt^2} = -(\frac{2\pi}{T_0})^2 q$,
qui égale $-q/(LC)$ exactement lorsque $(2\pi/T_0)^2 = 1/(LC)$,
c'est-à-dire $T_0 = 2\pi\sqrt{LC}$. Et $q(0) = Q_0$, $i(0) = 0$.
Qu'aucune \emph{autre} fonction ne s'accorde à l'équation et au départ
est admis, exactement comme pour le ressort
(\cref{rem:g12:oscillators-and-time:uniqueness}).
\end{proof}

\begin{definition}[Période et fréquence propres]\label{def:g12:rlc-oscillations:period}
La \emph{période propre}\index{période propre!d'un circuit LC} d'un
\omterm{prop:g12:rlc-oscillations:equation}{circuit LC} est $T_0 = 2\pi\sqrt{LC}$~; sa
\emph{fréquence propre}\index{fréquence propre!d'un circuit LC} est
\[
f_0 = \frac{1}{T_0} = \frac{1}{2\pi\sqrt{LC}} .
\]
Contrôle dimensionnel~: $\qty{1}{H} = \qty{1}{V.s/A}$ et
$\qty{1}{F} = \qty{1}{C/V}$, donc $LC$ porte
$\unit{s} \times \unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ est un
temps, comme il se doit.
\end{definition}

\begin{example}[Premiers nombres]\label{ex:g12:rlc-oscillations:firstnumbers}
$L = \qty{10}{mH}$ avec $C = \qty{1.0}{\micro F}$~:
$\sqrt{LC} = \qty{1.0e-4}{s}$, donc $T_0 = \qty{0.63}{ms}$ et
$f_0 \approx \qty{1.6}{kHz}$ --- un ton audible. $L = \qty{1.0}{mH}$
avec $C = \qty{100}{pF}$~: $T_0 = \qty{2.0}{\micro s}$,
$f_0 \approx \qty{0.50}{MHz}$ --- une \omterm{def:g12:oscillators-and-time:oscillator}{fréquence}
radio. Seul le \emph{produit} $LC$ fixe le tempo.
\end{example}

\begin{remark}[Un quart de période en décalage]\label{rem:g12:rlc-oscillations:quarter}
Le courant est d'un quart de \omterm{def:g12:oscillators-and-time:oscillator}{période} en décalage
de \omterm{def:g12:oscillators-and-time:phase}{phase} avec la charge~: $i = 0$ lorsque
$q = \pm Q_0$ (plaques pleines, flux qui s'inverse) et $i = \pm I_0$
lorsque $q = 0$ (plaques vides, flux à pleine vitesse) --- exactement
le pendule, immobile aux extrêmes et le plus rapide en bas.
\end{remark}

\section{La valse de l'énergie}

\begin{proposition}[Conservation de l'énergie stockée]\label{prop:g12:rlc-oscillations:energy}
\index{énergie!d'un circuit LC}
Dans le \omterm{prop:g12:rlc-oscillations:equation}{circuit LC} idéal l'\omterm{def:g10:energy-conservation:energy}{énergie}
totale stockée (\cref{prop:g12:rc-rl-circuits:energy})
\[
E \;=\; E_C + E_L \;=\; \frac{q^2}{2C} + \tfrac12\,L i^2
\;=\; \frac{Q_0^2}{2C} \;=\; \tfrac12\,L I_0^2
\]
est constante~: entièrement dans le
\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} lorsque $q = \pm Q_0$, entièrement dans
la \omterm{def:g11:magnetic-fields:solenoid}{bobine} lorsque $i = \pm I_0$.
\end{proposition}

\begin{proof}
Dériver~: $\frac{dE}{dt} = \frac{q}{C}\frac{dq}{dt} +
Li\frac{di}{dt} = i\left(\frac{q}{C} + L\frac{d^2q}{dt^2}\right) = 0$
par la \cref{prop:g12:rlc-oscillations:equation}. Sa valeur se lit à
$t = 0$ (tout \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur}), et encore à l'instant de
tout-courant $q = 0$ --- d'où $\tfrac12 L I_0^2 = Q_0^2/2C$,
retrouvant $I_0 = Q_0/\sqrt{LC}$ par
l'\omterm{def:g10:energy-conservation:energy}{énergie} seule.
\end{proof}

{\small Une \omterm{def:g12:oscillators-and-time:oscillator}{période} de la valse~:
l'\omterm{def:g10:energy-conservation:energy}{énergie} bascule des plaques à la
\omterm{def:g11:magnetic-fields:solenoid}{bobine} et retour deux fois par
\omterm{def:g12:oscillators-and-time:oscillator}{période}, la somme fixée à $Q_0^2/2C$ --- la
figure même de l'\omterm{prop:g12:oscillators-and-time:spring-period}{oscillateur masse--ressort},
relabelisée.}
\end{omfigure}

\begin{example}[La valse, mesurée]\label{ex:g12:rlc-oscillations:waltz}
$C = \qty{1.0}{\micro F}$ chargé à $U_0 = \qty{10}{V}$, basculé sur
$L = \qty{10}{mH}$~: \omterm{def:g10:energy-conservation:energy}{énergie}
$E = \tfrac12 C U_0^2 = \qty{5.0e-5}{J}$,
\omterm{def:g12:oscillators-and-time:oscillator}{période} $T_0 = \qty{0.63}{ms}$, et un courant
de crête $I_0 = U_0\sqrt{C/L} = \qty{0.10}{A}$ atteint un quart de
\omterm{def:g12:oscillators-and-time:oscillator}{période} (\qty{0.16}{ms}) après la fermeture de
l'interrupteur.
\end{example}

\section{Circuits réels~: oscillations amorties}

Toute \omterm{def:g11:magnetic-fields:solenoid}{bobine} réelle est enroulée en fil
\omterm{def:g11:work-of-force:sign}{résistif}~: la boucle honnête est un circuit
\emph{RLC} série, et la \omterm{def:g11:circuits-and-power:resistance}{résistance} taxe la valse.

\begin{definition}[Régimes pseudo-périodique et apériodique]\label{def:g12:rlc-oscillations:regimes}
\index{circuit RLC}\index{régime pseudo-périodique}\index{régime apériodique}
Dans un circuit RLC série le résistor dissipe de
l'\omterm{def:g10:energy-conservation:energy}{énergie} par effet Joule
(\cref{ch:g11:circuits-and-power})~: les
\omterm{def:g12:oscillators-and-time:oscillator}{oscillations libres} sont
\emph{\omterm{def:g12:oscillators-and-time:damping}{amorties}}. Deux régimes, lus sur un
\omterm{def:g10:signals-and-waves:oscilloscope}{oscilloscope}~:
\begin{itemize}
  \item \emph{\omterm{def:g10:signals-and-waves:period}{pseudo-périodique}} (petit $R$)~: le
        circuit oscille encore à l'intérieur d'une enveloppe qui se
        rétrécit, avec un temps de reprise --- la
        \emph{pseudo-période}\index{pseudo-période!électrique} ---
        proche de $T_0 = 2\pi\sqrt{LC}$ lorsque
        l'\omterm{def:g12:oscillators-and-time:damping}{amortissement} est léger~;
  \item \emph{apériodique} (grand $R$)~: la charge rampe vers zéro
        sans jamais le dépasser --- pas d'oscillation du tout.
\end{itemize}
Le miroir exact de l'\omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} mécanique
\omterm{def:g12:oscillators-and-time:damping}{amorti} (\cref{def:g12:oscillators-and-time:damping}).
\end{definition}

oscillations mourant dans une enveloppe exponentielle (tireté), ou
une rampe vers zéro sans oscillation. Faiblement
\omterm{def:g12:oscillators-and-time:damping}{amorti}, la

\end{omfigure}

\begin{remark}[Entretenir les oscillations]\label{rem:g12:rlc-oscillations:maintain}
\index{oscillations entretenues}
Pour garder l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} vivante, alimenter le
circuit exactement de l'\omterm{def:g10:energy-conservation:energy}{énergie} que la
\omterm{def:g11:circuits-and-power:resistance}{résistance} brûle, une fois par cycle et
\emph{en phase} avec le balancement~: un amplificateur fait pour la
boucle RLC ce que la poussée de l'échappement fait pour le pendule
(\cref{ch:g12:oscillators-and-time}). Toute montre à quartz et tout
émetteur radio est un tel \omterm{def:g12:oscillators-and-time:oscillator}{oscillateur}
entretenu, tournant à sa \omterm{def:g12:rlc-oscillations:period}{fréquence propre} $f_0$.
\end{remark}

\section{Le jumeau mécanique}

\begin{proposition}[Le dictionnaire électrique--mécanique]\label{prop:g12:rlc-oscillations:analogy}
\index{analogie électrique--mécanique}
L'équation LC est l'équation masse--ressort
(\cref{prop:g12:oscillators-and-time:spring-period}) mot pour mot~:
\begin{center}\small
\begin{tabular}{lcl}
\omterm{prop:g12:rlc-oscillations:equation}{circuit LC} & & mobile masse--ressort\\[2pt]
charge $q$ & $\longleftrightarrow$ & position $x$\\
courant $i = dq/dt$ & $\longleftrightarrow$ & vitesse $v = dx/dt$\\
\omterm{def:g12:rc-rl-circuits:inductor}{inductance} $L$ & $\longleftrightarrow$ & masse $m$\\
inverse de la \omterm{def:g12:rc-rl-circuits:capacitor}{capacité} $1/C$ & $\longleftrightarrow$ &
  \omterm{def:g12:oscillators-and-time:spring}{raideur} $k$\\
\omterm{def:g11:circuits-and-power:resistance}{résistance} $R$ & $\longleftrightarrow$ &
  \omterm{def:g10:inertia:inventory}{frottement}\\
$L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0$ & $\longleftrightarrow$ &
  $m\,\frac{d^2x}{dt^2} + kx = 0$\\[2pt]
$T_0 = 2\pi\sqrt{LC}$ & $\longleftrightarrow$ &
  $T_0 = 2\pi\sqrt{m/k}$\\[2pt]
$E_C = q^2/2C$ & $\longleftrightarrow$ & $E_p = \tfrac12 kx^2$\\
$E_L = \tfrac12 Li^2$ & $\longleftrightarrow$ & $E_k = \tfrac12 mv^2$
\end{tabular}
\end{center}
Même équation, mêmes solutions~: tout résultat sur l'un ou l'autre
\omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} se traduit instantanément dans
l'autre langue.
\end{proposition}

\begin{proof}
Comparer la \cref{prop:g12:rlc-oscillations:equation} avec
$m\,d^2x/dt^2 = -kx$~: substituer $q \to x$, $L \to m$, $1/C \to k$
transforme l'une en l'autre, et $LC \to m/k$ transforme
$2\pi\sqrt{LC}$ en $2\pi\sqrt{m/k}$.
\end{proof}

\begin{example}[Accorder une radio]\label{ex:g12:rlc-oscillations:tuning}
Une antenne envoie un murmure de \emph{chaque} station dans une
boucle LC, mais seule celle qui émet près de $f_0$ pousse en phase et
s'amplifie --- la \omterm{def:g12:oscillators-and-time:resonance}{résonance}
(\cref{def:g12:oscillators-and-time:resonance}). Le cadran fait tourner les
plaques d'un \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} variable~: changer $C$ déplace
$f_0 = 1/(2\pi\sqrt{LC})$, et l'aiguille tend à l'écouteur une
station à la fois. Côté émission, un
\omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} entretenu
(\cref{rem:g12:rlc-oscillations:maintain}) fixe la
\omterm{def:g12:oscillators-and-time:oscillator}{fréquence} de diffusion~; dans une montre, un
cristal de quartz --- un jumeau mécanique d'une
\omterm{def:g12:oscillators-and-time:spring}{raideur} époustouflante --- joue le même rôle.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:rlc-oscillations:1}
Calculer $T_0$ et $f_0$ pour (a)~$L = \qty{10}{mH}$,
$C = \qty{1.0}{\micro F}$~; (b)~$L = \qty{1.0}{mH}$,
$C = \qty{100}{pF}$. Par quel facteur les fréquences diffèrent-elles~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:rlc-oscillations:2}
En utilisant $\qty{1}{H} = \qty{1}{V.s/A}$ et $\qty{1}{F} =
\qty{1}{C/V}$, montrer \omterm{def:g10:orders-of-magnitude:unit}{unité} par
\omterm{def:g10:orders-of-magnitude:unit}{unité} que $\sqrt{LC}$ est un temps. Pourquoi le
$2\pi$ ne change-t-il rien~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:rlc-oscillations:3}
Un \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} de \qty{470}{nF} est chargé à
\qty{6.0}{V}. Calculer $Q_0$ et l'\omterm{def:g10:energy-conservation:energy}{énergie} stockée~;
décrire, sans calcul, ce qui suit lorsqu'on le bascule sur une
\omterm{def:g11:magnetic-fields:solenoid}{bobine} idéale.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:rlc-oscillations:4}
Un \omterm{prop:g12:rlc-oscillations:equation}{circuit LC} suit
$q(t) = 2.0\cos(2\pi t/T_0)$ en microcoulombs, avec
$T_0 = \qty{4.0}{ms}$. Lire $Q_0$~; calculer $f_0$, $q(T_0/2)$,
$i(0)$, et le courant de crête $I_0 = 2\pi Q_0/T_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:rlc-oscillations:5}
À quels instants du cycle l'\omterm{def:g10:energy-conservation:energy}{énergie} est-elle
entièrement dans le \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur}~? Entièrement dans la
\omterm{def:g11:magnetic-fields:solenoid}{bobine}~? Quel est le pendant mécanique de chaque
moment pour un pendule~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:6}
Vérifier par substitution que $q(t) = Q_0\cos(2\pi t/T_0)$ satisfait
$L\,d^2q/dt^2 + q/C = 0$ exactement lorsque $T_0 = 2\pi\sqrt{LC}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:7}
Une station \omterm{def:g10:signals-and-waves:wave}{grandes ondes} émet à \qty{198}{kHz}. Avec
$C = \qty{220}{pF}$, quelle \omterm{def:g12:rc-rl-circuits:inductor}{inductance} s'y accorde~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:8}
$C = \qty{1.0}{\micro F}$ chargé à \qty{10}{V} se décharge dans
$L = \qty{10}{mH}$. Calculer l'\omterm{def:g10:energy-conservation:energy}{énergie} stockée, le
courant de crête (par conservation de
l'\omterm{def:g10:energy-conservation:energy}{énergie}), et l'instant où il est atteint pour
la première fois.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:9}
Expliquer pourquoi $E_C$ et $E_L$ oscillent à la
\omterm{def:g12:oscillators-and-time:oscillator}{fréquence} $2f_0$, pas $f_0$. Montrer qu'à
$t = T_0/8$ (parti de $q = Q_0$) l'\omterm{def:g10:energy-conservation:energy}{énergie} est
exactement partagée en deux.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:10}
Sur un \omterm{def:g10:signals-and-waves:oscilloscope}{oscillogramme} d'un
\omterm{def:g12:rlc-oscillations:regimes}{circuit RLC} faiblement
\omterm{def:g12:oscillators-and-time:damping}{amorti}, les maxima successifs lisent
\qty{4.0}{V} puis \qty{3.0}{V}. En supposant le rapport constant,
prédire le sixième maximum après celui de \qty{4.0}{V}. Quelle
fraction de l'\emph{\omterm{def:g10:energy-conservation:energy}{énergie}} survit à chaque
\omterm{def:g12:rlc-oscillations:regimes}{pseudo-période}~? Que peut-on dire de la
\omterm{def:g12:rlc-oscillations:regimes}{pseudo-période} elle-même~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:11}
Un \omterm{prop:g12:rlc-oscillations:equation}{circuit LC} a $L = \qty{0.50}{H}$ et

dictionnaire (\cref{prop:g12:rlc-oscillations:analogy}), trouver la
\omterm{def:g12:oscillators-and-time:spring}{raideur} $k$ donnant à un mobile de

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:12}
Un accord utilise $L = \qty{0.20}{mH}$ et un
\omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} variable balayant
\qtyrange{50}{500}{pF}. Calculer les deux fréquences extrêmes. Montrer
que $f_0 \propto 1/\sqrt{C}$, donc un facteur dix en $C$ ne donne que
$\sqrt{10} \approx 3.2$ en \omterm{def:g12:oscillators-and-time:oscillator}{fréquence}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:13}
Une \omterm{def:g11:magnetic-fields:solenoid}{bobine} réelle fait perdre au circuit $5.0\%$ de son
\omterm{def:g10:energy-conservation:energy}{énergie} à chaque
\omterm{def:g12:rlc-oscillations:regimes}{pseudo-période}. Après combien de
\omterm{def:g12:oscillators-and-time:oscillator}{périodes} l'\omterm{def:g10:energy-conservation:energy}{énergie} a-t-elle
été divisée par deux~? Quelle est alors
l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}, en fraction du départ~? À
$f_0 = \qty{1.0}{MHz}$, combien de temps cela fait-il en microsecondes~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:14}
Un \omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} de montre stocke
$E = \qty{1.0e-9}{J}$ et en perd $2.0\%$ par cycle à
$f_0 = \qty{32768}{Hz}$. Quelle \omterm{def:g11:work-of-force:power}{puissance} moyenne le
circuit d'entretien doit-il fournir~? Pourquoi la poussée doit-elle
arriver en phase avec l'oscillation~?

Après chaque front d'un \omterm{def:g10:signals-and-waves:signal}{signal} carré, le câblage d'un

\section{Problème~: Construire une radio à galène}

\begin{problem}[{Problème du week-end --- construire une radio à
galène~: une bobine, un condensateur variable, une diode et un
écouteur --- pas de pile, pas d'amplificateur, et les nouvelles du
soir arrivent sur l'énergie de l'onde elle-même}]
\label{pb:g12:rlc-oscillations:1}
Un kit du grenier d'un grand-parent~: une
\omterm{def:g11:magnetic-fields:solenoid}{bobine} $L = \qty{0.20}{mH}$ enroulée sur un tube
de carton, un \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} variable dont les plaques
tournent de \qty{20}{pF} à \qty{480}{pF}, une diode à galène, un
écouteur, et \qty{20}{m} de fil d'antenne. Les stations à attraper
émettent entre \qty{530}{kHz} et \qty{1700}{kHz}.

\medskip
\noindent\textbf{Partie I --- La boucle d'accord.}
\begin{enumerate}
  \item Le \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur}, chargé, est laissé se décharger
        dans la \omterm{def:g11:magnetic-fields:solenoid}{bobine}~: écrire la loi de maille et
        la transformer en équation différentielle pour $q(t)$.
  \item Vérifier par substitution que $q(t) = Q_0\cos(2\pi t/T_0)$
        la résout, pourvu que $T_0 = 2\pi\sqrt{LC}$.
  \item Vérifier par analyse dimensionnelle que $\sqrt{LC}$ est un
        temps.
  \item Cadran réglé à $C = \qty{330}{pF}$~: calculer $T_0$ et $f_0$.
        La boucle est-elle accordée dans la bande~?
  \item Trouver la \omterm{def:g12:rc-rl-circuits:capacitor}{capacité} qui s'accorde exactement à
        \qty{1.00}{MHz}.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur}
variable.}
\begin{enumerate}[resume]
  \item Calculer la \omterm{def:g12:rc-rl-circuits:capacitor}{capacité} nécessaire pour
        \qty{530}{kHz}.
  \item Calculer la \omterm{def:g12:rc-rl-circuits:capacitor}{capacité} nécessaire pour
        \qty{1700}{kHz}.
  \item Calculer les deux fréquences que le
        \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} \qtyrange{20}{480}{pF} du kit
        atteint réellement. Couvre-t-il toute la bande~?
  \item Montrer que $f_0 \propto 1/\sqrt{C}$ à $L$ fixée~; par quel
        facteur $C$ doit-elle changer pour doubler $f_0$~?
  \item Les plaques s'ouvrent linéairement, donc $C$ chute
        régulièrement à mesure que le cadran tourne. Expliquer
        pourquoi les stations se serrent alors à une extrémité du
        cadran --- et laquelle.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- \omterm{def:g10:energy-conservation:energy}{Énergie} et
\omterm{def:g12:oscillators-and-time:damping}{amortissement}.}
Accordé à \qty{1.00}{MHz} ($C = \qty{127}{pF}$), l'antenne charge
momentanément le \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} à
$U_0 = \qty{20}{mV}$.
\begin{enumerate}[resume]
  \item Calculer la charge $Q_0$ et l'\omterm{def:g10:energy-conservation:energy}{énergie}
        stockée $E$.
  \item Calculer le courant de crête $I_0$, en utilisant la
        conservation de l'\omterm{def:g10:energy-conservation:energy}{énergie}.
  \item La \omterm{def:g11:circuits-and-power:resistance}{résistance} de la
        \omterm{def:g11:magnetic-fields:solenoid}{bobine} dissipe $4.0\%$ de
        l'\omterm{def:g10:energy-conservation:energy}{énergie} stockée par cycle~: après
        combien de cycles l'\omterm{def:g10:energy-conservation:energy}{énergie} a-t-elle
        été divisée par deux~?
  \item Combien de temps cela fait-il en microsecondes~? Pourquoi ce
        «~sonnement~» seul ne peut-il pas jouer les nouvelles du
        soir~?
  \item Expliquer pourquoi l'\omterm{def:g10:signals-and-waves:wave}{onde} qui arrive ne
        maintient le balancement de la boucle que lorsqu'elle est
        accordée sur cette station --- et ce qui entretient
        l'oscillation à l'\emph{émetteur}
        (\cref{rem:g12:rlc-oscillations:maintain}).
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le jumeau mécanique.}
\begin{enumerate}[resume]
  \item À l'aide du dictionnaire
        (\cref{prop:g12:rlc-oscillations:analogy}) avec un mobile de
        masse $m = \qty{0.20}{kg}$, calculer la
        \omterm{def:g12:oscillators-and-time:spring}{raideur} $k$ du ressort qui épouse le
        circuit accordé.
  \item Un ressort de voiture raide a $k \approx \qty{5e4}{N/m}$.
        Commenter~: un système masse--ressort de paillasse peut-il
        osciller à des mégahertz~?
  \item Garder une $k$ raisonnable $\qty{100}{N/m}$ à la place~:
        quelle masse oscille à \qty{1.00}{MHz}~?
  \item Quel objet réel est un tel
        \omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} mécanique ultra-léger et
        ultra-raide, et où l'a-t-on rencontré plus tôt cette année
        (\cref{ch:g12:oscillators-and-time})~?
  \item Fiche technique, une phrase~: la bande que balaye votre
        radio, la \omterm{def:g12:rc-rl-circuits:capacitor}{capacité} qui choisit
        \qty{1.00}{MHz}, combien dure un sonnement solitaire, et ce
        qui joue les nouvelles quand même.
\end{enumerate}
\end{problem}
