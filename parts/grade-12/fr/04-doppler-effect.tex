\chapter{L'effet Doppler}\label{ch:g12:doppler-effect}

Une ambulance s'approche, sirène hurlante, et passe --- et à cet
instant la note glisse audiblement vers le bas, bien que le conducteur
n'ait rien touché. La sirène est honnête~; l'\omterm{def:g10:signals-and-waves:wave}{onde} ne l'est pas~: le
mouvement la comprime devant et l'étire derrière. Ce chapitre mesure
cette compression, puis porte la même idée des sirènes aux pistolets
\omterm{met:g10:signals-and-waves:echo}{radar}, au sang dans une artère, et jusqu'à une planète trahie par
un tremblement dans la lumière stellaire.

\section{L'effet}

\begin{definition}[Effet Doppler]\label{def:g12:doppler-effect:doppler}
L'\emph{effet Doppler}\index{effet Doppler} est le changement de la
\omterm{def:g10:signals-and-waves:frequency}{fréquence} reçue d'une \omterm{def:g10:signals-and-waves:wave}{onde} lorsque source et
récepteur se déplacent l'un par rapport à l'autre~: la
\omterm{def:g10:signals-and-waves:frequency}{fréquence} reçue $f'$ est plus élevée que la
\omterm{def:g10:signals-and-waves:frequency}{fréquence} émise $f$ tant qu'ils se rapprochent, plus basse
tant qu'ils s'éloignent, et égale à $f$ seulement tant que leur
distance cesse momentanément de changer.
\end{definition}

\begin{remark}[Hauteur, non intensité sonore]\label{rem:g12:doppler-effect:pitch}
Deux choses changent au passage de l'ambulance. L'\emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{intensité sonore}}
enfle et s'estompe parce que l'\omterm{def:g10:signals-and-waves:wave}{onde} s'étale avec la distance
(\cref{ch:g12:sound-acoustics})~; la \emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur}} se décale
à cause de la vitesse de rapprochement ou d'éloignement, et chute
abruptement au passage. L'\omterm{def:g12:sound-acoustics:pitch-loudness}{intensité sonore} dit «~à quelle
distance~»~; la \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} dit «~à quelle vitesse~».
\end{remark}

\section{Une source en mouvement~: fronts d'onde resserrés}

L'image clé~: une fois qu'une crête quitte la source, le milieu la
possède. Chaque crête s'élargit en un cercle à la célérité $v$ de
l'\omterm{def:g10:signals-and-waves:wave}{onde} (\cref{ch:g12:mechanical-waves}), centré sur le point où
elle a été \emph{émise} --- le milieu ne sait ni ne se soucie que la
source ait poursuivi son chemin.

\begin{proposition}[Source en mouvement]\label{prop:g12:doppler-effect:moving-source}
Une source émet des crêtes de \omterm{def:g10:signals-and-waves:period}{période} $T$
(\omterm{def:g10:signals-and-waves:frequency}{fréquence} $f = 1/T$) tout en se déplaçant en ligne droite
à la vitesse $v_s < v$ dans un milieu où l'\omterm{def:g10:signals-and-waves:wave}{onde} voyage à la
célérité $v$. Un observateur au repos dans le milieu reçoit
\[
\lambda' = (v \mp v_s)\,T, \qquad f' = f\,\frac{v}{v \mp v_s},
\]
les signes supérieurs pour une source qui s'approche, les inférieurs
pour une qui s'éloigne.
\end{proposition}

\begin{proof}
Entre deux crêtes, la première avance de $vT$ tandis que la source
avance de $v_s T$. Devant, la seconde crête est donc émise $v_s T$
\emph{plus près} de la première, $\lambda' = (v - v_s)T$~; derrière,
le même $v_s T$ s'ajoute. Les crêtes balayent ensuite un observateur
au repos à la célérité $v$ du milieu, une toutes les $\lambda'/v$
secondes, d'où $f' = v/\lambda' = f v/(v \mp v_s)$.
\end{proof}

\begin{example}[L'ambulance, en chiffres]\label{ex:g12:doppler-effect:siren}
Une sirène à $f = \qty{700}{Hz}$ s'approche à $v_s = \qty{25}{m/s}$~;
prendre $v = \qty{340}{m/s}$ dans l'air (\cref{ch:g12:sound-acoustics}).
En approche~: $f' = 700 \times 340/315 \approx \qty{756}{Hz}$~; en
éloignement~: $f' = 700 \times 340/365 \approx \qty{652}{Hz}$. Le
passage abaisse la \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} de \qty{104}{Hz} --- un
rapport $\approx 1.16$, environ deux demi-tons et demi, indubitable
pour toute oreille.
\end{example}

\begin{remark}[Ce que le mouvement ne change pas]\label{rem:g12:doppler-effect:speed-unchanged}
L'\omterm{def:g10:signals-and-waves:wave}{onde} voyage toujours à $v$~: le milieu seul fixe la célérité
(\cref{ch:g12:mechanical-waves}). Le mouvement de la source comprime
les \emph{\omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde}}, jamais les célérités. Et la
compression a une limite~: lorsque $v_s \to v$ les crêtes devant
s'empilent les unes sur les autres et $f' \to \infty$ --- l'embouteillage
de \omterm{def:g12:mechanical-waves:wavefront}{fronts d'onde} qu'un appareil supersonique traîne comme un
choc.
\end{remark}

\section{Un observateur en mouvement, et le raccourci des petits mouvements}

\begin{proposition}[Observateur en mouvement]\label{prop:g12:doppler-effect:moving-observer}
Un observateur se déplace à la vitesse $v_o$ en ligne droite vers
(ou loin d') une source au repos dans le milieu. La
\omterm{def:g10:signals-and-waves:frequency}{fréquence} reçue est
\[ f' = f\left(1 \pm \frac{v_o}{v}\right) \quad (+\ \text{approche},\ -\ \text{éloignement}). \]
\end{proposition}

\begin{proof}
Le motif d'\omterm{def:g10:signals-and-waves:wave}{onde} lui-même est inchangé~: des crêtes d'écartement
$\lambda = vT$ voyagent à $v$. Un observateur qui court vers elles
rencontre les crêtes à la vitesse relative $v + v_o$, une toutes les
$\lambda/(v + v_o)$ secondes~: $f' = (v + v_o)/\lambda = f(1 + v_o/v)$.
En s'enfuyant, $v - v_o$.
\end{proof}

\begin{definition}[Vitesse radiale]\label{def:g12:doppler-effect:radial}
La \emph{vitesse radiale}\index{vitesse radiale} $v_r$ d'une source
par rapport à un observateur est la composante de leur vitesse
relative le long de la droite qui les joint --- le taux auquel la
distance diminue ou croît. Seule cette composante décale la
\omterm{def:g10:signals-and-waves:frequency}{fréquence}~; un mouvement \emph{en travers} de la ligne de
visée ne donne aucun décalage à cet ordre.
\end{definition}

\begin{proposition}[Petits mouvements~: une formule pour tous]\label{prop:g12:doppler-effect:slow}
Lorsque la vitesse radiale est petite, $v_r \ll v$, peu importe qui
bouge~:
\[
\frac{\Delta f}{f} = \frac{f' - f}{f} \approx +\frac{v_r}{v} \ \text{(approche)},
\qquad
\frac{\Delta f}{f} \approx -\frac{v_r}{v} \ \text{(éloignement)}.
\]
\end{proposition}

\begin{proof}
Les formules d'observateur sont déjà exactement $1 \pm v_o/v$. Pour la
source, écrire $x = v_s/v$~: l'identité
$\frac{1}{1 - x} = 1 + x + \frac{x^2}{1 - x}$ montre que $f'/f$
diffère de $1 + x$ d'un terme d'ordre $x^2$ --- pour $x = 0.1$, une
correction de $1\%$ sur un décalage de $10\%$. Au premier ordre, les
quatre formules se réduisent à $1 \pm v_r/v$.
\end{proof}

\begin{method}[Lire un passage]\label{met:g12:doppler-effect:pass}
L'enregistrement d'une source qui passe montre deux plateaux~: $f_1$
(approche) et $f_2$ (éloignement). Alors
\[
f = \frac{2 f_1 f_2}{f_1 + f_2}, \qquad
v_s = v\,\frac{f_1 - f_2}{f_1 + f_2}.
\]
En effet $f_1 + f_2 = \frac{2fv^2}{v^2 - v_s^2}$ et
$f_1 - f_2 = \frac{2fv\,v_s}{v^2 - v_s^2}$~: diviser pour $v_s$, et
calculer $2f_1 f_2/(f_1 + f_2)$ pour $f$. Deux lectures de
\omterm{def:g10:signals-and-waves:frequency}{fréquence} donnent à la fois la vraie
\omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} et la vitesse --- sans chronomètre ni règle.
\end{method}

\section{Des échos qui mesurent la vitesse}

Renvoyez une \omterm{def:g10:signals-and-waves:wave}{onde} sur une cible mobile et l'\omterm{def:g12:doppler-effect:doppler}{effet Doppler}
frappe deux fois.

\begin{proposition}[La réflexion double le décalage]\label{prop:g12:doppler-effect:double}
Une \omterm{def:g10:signals-and-waves:wave}{onde} de \omterm{def:g10:signals-and-waves:frequency}{fréquence} $f$ est envoyée sur un
réflecteur qui s'approche de face à la vitesse $u \ll v$.
L'\omterm{rem:g10:signals-and-waves:honest}{écho} revient avec
\[ f' = f\,\frac{v + u}{v - u}, \qquad \text{d'où} \qquad \Delta f \approx \frac{2u}{v}\,f . \]
\end{proposition}

\begin{proof}
Deux décalages en série. Comme \emph{observateur} mobile le réflecteur
reçoit $f(1 + u/v)$~; réémettant ce qu'il reçoit, il est maintenant
une \emph{source} mobile, donc l'\omterm{rem:g10:signals-and-waves:honest}{écho} arrive à
$f' = f(1 + u/v)/(1 - u/v) \approx f(1 + 2u/v)$ pour $u \ll v$.
\end{proof}

\begin{example}[Le pistolet radar]\label{ex:g12:doppler-effect:radar}
Un \omterm{met:g10:signals-and-waves:echo}{radar} de circulation émet des micro-ondes à
$f = \qty{24.15}{GHz}$ ($v = c = \qty{3.00e8}{m/s}$) et superpose
l'\omterm{rem:g10:signals-and-waves:honest}{écho} à l'\omterm{def:g10:signals-and-waves:wave}{onde} sortante~; le battement
(\cref{ch:g12:sound-acoustics}) \omterm{def:g10:signals-and-waves:sound}{sonne} directement
$\Delta f = 2uf/c \approx \qty{161}{Hz}$ par \unit{m/s} de vitesse
de voiture --- une \omterm{def:g10:signals-and-waves:frequency}{fréquence} audio trivialement mesurable
taillée dans une porteuse à \qty{24}{GHz}. Le facteur $2$ n'est pas
optionnel~: l'oublier flatte chaque conducteur de moitié.
\end{example}

\begin{example}[Échographie Doppler]\label{ex:g12:doppler-effect:ultrasound}
Une sonde médicale envoie des \omterm{def:g10:signals-and-waves:ultrasound}{ultrasons} à
$f = \qty{5.0}{MHz}$ dans une artère ($v \approx \qty{1540}{m/s}$
dans les tissus)~; les globules rouges les réfléchissent. Du sang à
$u = \qty{0.50}{m/s}$ renvoie
$\Delta f = 2 \times 0.50 \times \num{5.0e6}/1540 \approx
\qty{3.2}{kHz}$ --- un battement \emph{audible}~: le cardiologue
entend littéralement le sang accélérer à chaque battement, puis
mesure $u$ à partir de $\Delta f$.
\end{example}

\section{Doppler dans la lumière stellaire}

La lumière est une \omterm{def:g10:signals-and-waves:wave}{onde} (\cref{ch:g12:light-as-wave}), son
\omterm{def:g12:sound-acoustics:timbre}{spectre} rayé de raies nettes à des
\omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} fixées par chaque élément chimique
(\cref{ch:g10:light-spectra}) --- une règle calibrée en laboratoire
imprimée sur chaque étoile.

\begin{definition}[Décalage vers le rouge et vers le bleu]\label{def:g12:doppler-effect:redshift}
Lorsque les \omterm{def:g10:light-spectra:emission}{raies spectrales} d'une source apparaissent à des
\omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} \emph{plus grandes} qu'en laboratoire, le
\omterm{def:g12:sound-acoustics:timbre}{spectre} est \emph{décalé vers le rouge}\index{décalage vers le
rouge}~; lorsqu'elles apparaissent \emph{plus courtes}, il est
\emph{décalé vers le bleu}\index{décalage vers le bleu} --- déplacé
vers l'extrémité rouge ou bleue du \omterm{def:g10:light-spectra:wavelength}{spectre visible}.
\end{definition}

\begin{proposition}[Décalage Doppler de la lumière]\label{prop:g12:doppler-effect:light}
Pour une source s'éloignant à la vitesse radiale $v_r \ll c$, chaque
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} est observée étirée de
\[ \frac{\Delta \lambda}{\lambda} = \frac{\lambda' - \lambda}{\lambda} \approx \frac{v_r}{c} \quad \text{(décalage vers le rouge)}, \]
et comprimée de la même quantité, $\Delta\lambda/\lambda \approx -v_r/c$,
pour une source qui s'approche (\omterm{def:g12:doppler-effect:redshift}{décalage vers le bleu}).
\end{proposition}
\admitted

\begin{remark}[Où vit la formule honnête]\label{rem:g12:doppler-effect:honest}
La lumière n'a pas besoin de milieu, donc la dérivation du
\omterm{def:g10:signals-and-waves:sound}{son} ne peut être copiée --- il n'y a pas de «~au repos dans le
milieu~». La formule exacte vient de la relativité restreinte, plus
tard cette année (\cref{ch:g12:special-relativity})~; les corrections
sont d'ordre $(v_r/c)^2$, négligeables pour les avions, les étoiles
et les galaxies proches. Ci-dessous on utilise la formule des petits
mouvements en toute conscience.
\end{remark}

\begin{example}[Peser la fuite d'une étoile]\label{ex:g12:doppler-effect:vega}
Dans le \omterm{def:g12:sound-acoustics:timbre}{spectre} de la brillante étoile Véga, la raie
d'hydrogène de \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} de laboratoire
\qty{656.30}{nm} est mesurée à \qty{656.27}{nm}~:
$\Delta\lambda = \qty{-0.03}{nm}$, un \omterm{def:g12:doppler-effect:redshift}{décalage vers le bleu},
donc Véga s'approche à
$v_r = c\,|\Delta\lambda|/\lambda \approx \qty{1.4e4}{m/s}$ ---
\qty{14}{km/s}, lus sur un décalage d'une partie sur vingt mille.
\end{example}

\begin{example}[Étoiles qui vacillent~: binaires et exoplanètes]\label{ex:g12:doppler-effect:wobble}
Lorsque deux étoiles \omterm{def:g10:universal-gravitation:satellite}{orbitent} l'une autour de l'autre,
leurs raies oscillent \omterm{def:g10:signals-and-waves:period}{périodiquement} du bleu au rouge --- une
\emph{binaire spectroscopique}\index{binaire spectroscopique}, des
\omterm{def:g10:universal-gravitation:satellite}{orbites} mesurées sans jamais résoudre la paire. Une planète
fait de même à son étoile, faiblement~: toutes deux tournent autour
de leur \omterm{def:g11:forces-and-motion:com}{centre de masse} commun, si bien que la
\omterm{def:g12:doppler-effect:radial}{vitesse radiale} de l'étoile oscille de quelques dizaines
de \unit{m/s} --- un décalage $\Delta\lambda/\lambda \sim 10^{-7}$.
Le détecter est la façon dont la première planète autour d'une étoile
de type solaire a été trouvée~; le problème du week-end rejoue cette
découverte, chiffres compris.
\end{example}

\begin{remark}[Décalages vers le rouge des galaxies]\label{rem:g12:doppler-effect:expansion}
Les spectres des galaxies lointaines sont tous \emph{décalés vers le
rouge}, et plus la galaxie est lointaine, plus le décalage est fort~:
les distances de l'univers s'étirent. Pour les galaxies proches
$v_r = c\,\Delta\lambda/\lambda$ lit la vitesse d'éloignement~; pour
les plus lointaines, les décalages dépassent ce que la formule des
petits mouvements peut honnêtement traiter, et le compte est repris
à partir du \cref{ch:g12:special-relativity}, puis dans les volumes
universitaires.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:1}
La sirène d'un camion de pompiers émet à \qty{700}{Hz} et roule à
\qty{25}{m/s} ($v = \qty{340}{m/s}$). Calculer la
\omterm{def:g10:signals-and-waves:frequency}{fréquence} entendue par un piéton (a) devant lui~; (b)
derrière lui. (c) Que entend le conducteur~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:2}
Un ami affirme~: «~la \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} chute quand l'ambulance s'en
va parce que le \omterm{def:g10:signals-and-waves:sound}{son} s'affaiblit avec la distance.~» Démêler
les deux effets confondus, et dire de quoi dépend chacun.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:3}
Un cycliste roule à \qty{6.0}{m/s} droit vers une sirène stationnaire
émettant à \qty{700}{Hz}. Quelle \omterm{def:g10:signals-and-waves:frequency}{fréquence} entend-il~? Et
en s'éloignant droit~? Quelle formule s'applique, et pourquoi pas
celle de la source en mouvement~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:4}
Le klaxon d'un train émet à \qty{400}{Hz} tandis que le train roule
à \qty{30}{m/s}. Calculer la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} devant le
train, derrière lui, et au repos. Quel observateur reçoit laquelle~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:5}
Un téléphone jouant un ton à \qty{440}{Hz} est porté par un coureur à
\qty{5.0}{m/s}. Utiliser la formule des petits mouvements pour
estimer le décalage entendu par quelqu'un que le coureur approche.
Comparer à un demi-ton (environ $6\%$ en \omterm{def:g10:signals-and-waves:frequency}{fréquence})~: un
musicien le remarquerait-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:6}
Une chauve-souris vole à \qty{6.0}{m/s} droit vers un mur, émettant à
\qty{50.0}{kHz}. Montrer que l'\omterm{rem:g10:signals-and-waves:honest}{écho} qu'elle entend revient à
$f' = f(v + v_b)/(v - v_b)$, puis calculer $f'$ et le décalage.
Pourquoi est-ce la formule du pistolet radar déguisée~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:7}
Un pistolet \omterm{met:g10:signals-and-waves:echo}{radar} à \qty{24.15}{GHz} mesure un battement de
\qty{4.0}{kHz} d'une voiture qui s'approche. Trouver la vitesse de la
voiture en \unit{m/s} et \unit{km/h}. Quel battement donnerait une
voiture à exactement \qty{50}{km/h}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:8}
Une sonde Doppler à \qty{5.0}{MHz} d'\omterm{def:g10:signals-and-waves:ultrasound}{ultrasons} dirigée le
long d'une artère ($v = \qty{1540}{m/s}$ dans les tissus) entend un
battement de \qty{2.6}{kHz}. Trouver la vitesse du sang. Pourquoi la
sonde doit-elle être orientée \emph{le long} de l'écoulement plutôt
que perpendiculairement~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:9}
Prendre $f = \qty{1000}{Hz}$, $v = \qty{340}{m/s}$ et une vitesse de
rapprochement de \qty{34}{m/s}. Calculer $f'$ exactement lorsque
(a) la source bouge, (b) l'observateur bouge. Comparer les deux à la
prédiction des petits mouvements $f(1 + v_r/v)$ et expliquer la taille
du désaccord.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:10}
L'enregistrement d'un scooter passant un micro montre des plateaux à
\qty{465}{Hz} et \qty{415}{Hz}. En utilisant la
\cref{met:g12:doppler-effect:pass}, trouver la vraie
\omterm{def:g10:signals-and-waves:frequency}{fréquence} du klaxon et la vitesse du scooter en
\unit{km/h}. Pourquoi la vraie \omterm{def:g10:signals-and-waves:frequency}{fréquence} n'est-elle
\emph{pas} leur moyenne~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:11}
Dans le \omterm{def:g12:sound-acoustics:timbre}{spectre} d'une étoile, la raie d'hydrogène de
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} de laboratoire \qty{656.3}{nm} est observée
à \qty{656.5}{nm}. \omterm{def:g12:doppler-effect:redshift}{Décalage vers le rouge} ou
\omterm{def:g12:doppler-effect:redshift}{vers le bleu}~? Calculer la \omterm{def:g12:doppler-effect:radial}{vitesse radiale}
de l'étoile et sa direction.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:12}
Une étoile vacille à \qty{55}{m/s} à cause d'une planète invisible.
Calculer l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude} de l'oscillation en
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} d'une raie à \qty{500}{nm}, comme
longueur et comme fraction de $\lambda$. \omterm{def:g10:orders-of-magnitude:oom}{Ordre de
grandeur}~: pourquoi la chasse aux planètes a-t-elle dû attendre des
spectrographes stables à une partie pour $10^{7}$~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:13}
Chaque raie du \omterm{def:g12:sound-acoustics:timbre}{spectre} d'une galaxie est trouvée étirée de
$2.4\%$. Calculer sa vitesse d'éloignement. Les astronomes trouvent
que de telles vitesses croissent proportionnellement à la distance,
dans toutes les directions du ciel~: quelle image cela suggère-t-il~?
La formule des petits mouvements est-elle encore fiable ici~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:14}
Le Soleil tourne~: un bord de son disque s'approche de nous à environ
\qty{2.0}{km/s} tandis que l'autre s'éloigne aussi vite. Que fait cela

décale ou l'élargit~? Calculer l'effet en \unit{nm}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:15}
Une voiture de police à \qty{40}{m/s}, sirène à \qty{700}{Hz}, poursuit
un camion roulant à \qty{30}{m/s} dans la même direction. (a) Justifier

$f'$. (c) Montrer que si le camion égalait la vitesse de la voiture le
décalage s'annulerait, et dire pourquoi c'est comme il se doit.
\end{exercise}

\section{Problème~: La chasse à une exoplanète}

\begin{problem}[{Problème du week-end --- la chasse à une exoplanète~: un
klaxon de train calibre l'oreille, un pistolet radar répète le double
décalage, et dimanche soir le spectre d'une étoile, vacillant d'une
partie sur cinq millions, a pesé une planète que personne n'a jamais
vue}]
\label{pb:g12:doppler-effect:1}
Un club d'astronomie passe un week-end sur une seule idée --- le
décalage Doppler --- répétée sur Terre, puis pointée vers une étoile
de type solaire. Données~: $v = \qty{340}{m/s}$ pour le
\omterm{def:g10:signals-and-waves:sound}{son} dans l'air, $c = \qty{3.00e8}{m/s}$,
$G = \qty{6.67e-11}{N.m^2/kg^2}$~; l'étoile a pour masse
$M = \qty{2.0e30}{kg}$. Admis, de la mécanique des
\omterm{def:g10:universal-gravitation:satellite}{orbites} circulaires
(\cref{ch:g12:satellites-kepler})~: une planète de masse $m$ sur une
\omterm{def:g10:universal-gravitation:satellite}{orbite} circulaire de rayon $r$ a pour
\omterm{def:g10:signals-and-waves:period}{période} et vitesse $r^3 = \frac{G M P^2}{4\pi^2}$ et
$V = \frac{2\pi r}{P}$, et l'étoile, tournant autour du
\omterm{def:g11:forces-and-motion:com}{centre de masse} commun, vacille à la vitesse
$K = \frac{m}{M}\,V$.

\medskip
\noindent\textbf{Partie I --- Vendredi~: le passage à niveau.}
Un train passe à vitesse constante, klaxon allumé~; un téléphone
enregistre $f_1 = \qty{471}{Hz}$ pendant l'approche,
$f_2 = \qty{411}{Hz}$ après le passage.
\begin{enumerate}
  \item Expliquer, avec l'image des \omterm{def:g12:mechanical-waves:wavefront}{fronts d'onde resserrés},
        pourquoi $f_1 > f_2$ bien que le klaxon ne change jamais.
  \item Écrire les deux équations liant $f_1$, $f_2$ à la vraie
        \omterm{def:g10:signals-and-waves:frequency}{fréquence} $f$ et à la vitesse $u$ du train.
  \item Montrer que $u = v\,\dfrac{f_1 - f_2}{f_1 + f_2}$ et la calculer
        en \unit{m/s} et \unit{km/h}.
  \item Montrer que $f = \dfrac{2 f_1 f_2}{f_1 + f_2}$ et la calculer.
        Pourquoi $f$ est-elle légèrement \emph{sous} la moyenne de
        $f_1$ et $f_2$~?
  \item Calculer les \omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} devant et derrière le
        train, et la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} au repos.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Samedi~: le pistolet \omterm{met:g10:signals-and-waves:echo}{radar}.}
Une patrouille de circulation démontre son \omterm{met:g10:signals-and-waves:echo}{radar}~:
$f = \qty{24.125}{GHz}$, et le pistolet lit le battement entre
l'\omterm{def:g10:signals-and-waves:wave}{onde} sortante et l'\omterm{rem:g10:signals-and-waves:honest}{écho}.
\begin{enumerate}[resume]
  \item Expliquer pourquoi l'\omterm{rem:g10:signals-and-waves:honest}{écho} d'une voiture en mouvement est
        décalé \emph{deux fois} --- nommer le rôle que joue la voiture
        dans chaque décalage.
  \item Montrer que pour une voiture à la vitesse $u \ll c$ le battement
        est $\Delta f \approx \dfrac{2u}{c}\,f$.
  \item Une voiture renvoie $\Delta f = \qty{3.55}{kHz}$~: calculer $u$.
  \item Convertir en \unit{km/h} et comparer à la limite de
        \qty{80}{km/h} affichée là.
  \item Combien d'\omterm{def:g10:signals-and-waves:frequency}{hertz} de battement correspondent à
        \qty{1}{km/h}~? Commenter~: qu'est-ce qui rend un tel décalage
        relatif minuscule d'une \omterm{def:g10:signals-and-waves:wave}{onde} à \qty{24}{GHz} facile à
        mesurer~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Dimanche~: l'étoile qui vacille.}
Le club télécharge la \omterm{def:g12:doppler-effect:radial}{vitesse radiale} mesurée d'une étoile
de type solaire, nuit après nuit, tracée ci-dessous.

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis, width=8.8cm, height=4.6cm,
    xlabel={$t$ (jours)}, ylabel={$v_r$ (\unit{m/s})},
    xmin=0, xmax=8.4, ymin=-72, ymax=72,
    xtick={0,2.1,4.2,6.3,8.4},
    xticklabels={$0$,$2.1$,$4.2$,$6.3$,$8.4$}, ytick={-55,0,55}]
  \addplot[dashed, omExo, domain=0:8.4, samples=2] {55};
  \addplot[dashed, omExo, domain=0:8.4, samples=2] {-55};
  \addplot[omDef, thick, domain=0:8.4, samples=140] {55*sin(360*x/4.2)};
\end{axis}
\end{tikzpicture}

{\small La \omterm{def:g12:doppler-effect:radial}{vitesse radiale} de l'étoile sur douze nuits~: une
sinusoïde propre --- la signature d'un corps sur une
\omterm{def:g10:universal-gravitation:satellite}{orbite} circulaire, vue de profil.}
\end{omfigure}

\begin{enumerate}[resume]
  \item Que mesure-t-on réellement, nuit après nuit, dans le
        \emph{\omterm{def:g12:sound-acoustics:timbre}{spectre}} de l'étoile~? Énoncer la relation
        utilisée pour en faire $v_r$.
  \item Lire l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude} $K$ et la
        \omterm{def:g10:signals-and-waves:period}{période} $P$ du vacillement sur la courbe.
  \item Calculer l'oscillation correspondante $\Delta\lambda$ d'une raie
        à \qty{550}{nm}, et $\Delta\lambda/\lambda$. Comparer avec la
        partie~I~: de combien cette mesure est-elle plus difficile que
        celle du train~?
  \item Pourquoi une planète invisible fait-elle vaciller l'étoile~?
        Que indique la forme sinusoïdale sur l'\omterm{def:g10:universal-gravitation:satellite}{orbite}~?
  \item Pourquoi cette méthode ne mesure-t-elle que la partie
        \emph{radiale} du mouvement de l'étoile, et que verrait-on si
        l'\omterm{def:g10:universal-gravitation:satellite}{orbite} était face à nous~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Dimanche soir~: peser l'invisible.}
Prendre l'\omterm{def:g10:universal-gravitation:satellite}{orbite} circulaire et de profil, de sorte que $K$
est la pleine vitesse orbitale de l'étoile.
\begin{enumerate}[resume]
  \item À partir de $P$ et de la relation admise
        $r^3 = G M P^2/(4\pi^2)$, calculer le rayon orbital $r$ de la
        planète. Le comparer à la distance Terre--Soleil,
        \qty{1.5e11}{m}.
  \item Calculer la vitesse orbitale $V = 2\pi r/P$ de la planète.
  \item À partir de $K = (m/M)\,V$, calculer la masse $m$ de la
        planète.
  \item Comparer $m$ à Jupiter (\qty{1.9e27}{kg}) et à la Terre
        (\qty{6.0e24}{kg}), et $r$ à la distance Soleil--Mercure
        (\qty{5.8e10}{m}). Quel genre de monde est-ce~?
  \item Si l'\omterm{def:g10:universal-gravitation:satellite}{orbite} était inclinée plutôt que de profil, la
        vraie masse serait-elle plus grande ou plus petite que votre
        valeur~? Conclure en une phrase~: qu'a livré un décalage
        \omterm{def:g10:signals-and-waves:period}{périodique} d'une partie sur cinq millions~?
\end{enumerate}
\end{problem}
