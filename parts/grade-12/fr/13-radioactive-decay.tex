\chapter{Décroissance radioactive}\label{ch:g12:radioactive-decay}

Dans le sous-sol d'un musée, un compteur Geiger cliquette au-dessus
d'un lambeau de bois d'un sarcophage de pharaon. Chaque clic est
sans loi~: rien ne l'annonçait, rien ne prédit le suivant. Pourtant
vendredi le laboratoire imprimera une date bonne à un siècle près.
Ce chapitre résout le paradoxe --- comment des événements parfaitement
aléatoires un par un s'additionnent, par milliers de milliards, pour
devenir les horloges les plus régulières que l'on possède.

\section{Sans loi un par un, mécaniques par la mole}

L'an dernier (\cref{ch:g11:nucleus-radioactivity}) on a rencontré les
noyaux \omterm{prop:g12:work-and-energy:equilibria}{instables}, équilibré leurs équations $\alpha$ et
$\beta$, et vu la moitié de tout grand échantillon disparaître par
\omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie} $T_{1/2}$. Ce qu'on ne pouvait pas dire,
c'est \emph{quand entre les marches} --- ni pourquoi un processus
aléatoire tient un emploi du temps. Les deux questions n'ont qu'une
réponse.

\begin{definition}[Constante de désintégration]\label{def:g12:radioactive-decay:lambda}
Pour chaque \omterm{def:g11:nucleus-radioactivity:notation}{nucléide}
\omterm{prop:g12:work-and-energy:equilibria}{instable} il existe une constante $\lambda$, sa
\emph{constante de désintégration}\index{constante de
désintégration} (\omterm{def:g10:orders-of-magnitude:unit}{unité} \unit{s^{-1}}), telle que
pendant tout court intervalle $\dd t$ chaque
\omterm{def:g11:fundamental-interactions:ladder}{noyau} a la probabilité $p = \lambda\,\dd t$ de
se désintégrer --- la même pour tout \omterm{def:g11:fundamental-interactions:ladder}{noyau} de
l'espèce, quel que soit son âge, et intouchée par la température, la
\omterm{def:g10:pressure:pressure}{pression} ou la chimie.
\end{definition}

\begin{remark}[Sans loi un à un, exact en foule]\label{rem:g12:radioactive-decay:crowds}
Un \omterm{def:g11:fundamental-interactions:ladder}{noyau} qui a attendu dix mille ans a
exactement la même chance de se désintégrer cette seconde qu'un
fabriqué ce matin~: pas d'usure, pas d'avertissement --- lequel part
ensuite est vraiment imprévisible. Les foules sont une autre affaire.
Lancez $100$~pièces~: attendez $50$~faces, mais ne soyez pas surpris
par $43$ --- des fluctuations d'ordre $\sqrt n$, ici $10\%$. Lancez
$10^{20}$~: la fluctuation relative $1/\sqrt n$ est $10^{-10}$, le
résultat certain à dix décimales. Un gramme de matière tient quelque
$10^{22}$~noyaux~: le sans-loi s'average en mécanique d'horloge, une
loi exacte pour $N(t)$, le nombre de noyaux encore intacts à l'instant
$t$.
\end{remark}

\section{La loi de décroissance}

\begin{proposition}[L'équation d'évolution]\label{prop:g12:radioactive-decay:ode}
Dans un grand échantillon de $N$~noyaux, pendant $\dd t$ le nombre
change de
\[
\dd N = -\lambda N\,\dd t, \qquad\text{c'est-à-dire}\qquad \frac{\dd N}{\dd t} = -\lambda N :
\]
la population se rétrécit à un rythme proportionnel à elle-même.
\end{proposition}

\begin{proof}
Chacun des $N$~noyaux se désintègre pendant $\dd t$ avec la
probabilité $\lambda\,\dd t$~: désintégrations attendues
$N\lambda\,\dd t$, et pour $N$ grand le décompte réel s'accroche à
l'attendu (\cref{rem:g12:radioactive-decay:crowds})~; chaque
désintégration retire un \omterm{def:g11:fundamental-interactions:ladder}{noyau}.
\end{proof}

\begin{theorem}[Loi de décroissance radioactive]\label{thm:g12:radioactive-decay:law}
Un échantillon tenant $N_0$~noyaux à $t = 0$ en tient, à l'instant
$t$,
\[
N(t) = N_0 \, e^{-\lambda t}.
\]
\end{theorem}

\begin{proof}
Vérifier par substitution~:
$\dd N/\dd t = N_0(-\lambda)\,e^{-\lambda t} = -\lambda N(t)$, et
$N(0) = N_0 e^0 = N_0$~: équation et départ tous deux justes. (Qu'aucune
\emph{autre} fonction ne s'accorde est prouvé, avec des intégrales,
dans le volume universitaire.)
\end{proof}

\begin{definition}[Décroissance exponentielle]\label{def:g12:radioactive-decay:exponential}
Une quantité suivant $N_0\,e^{-\lambda t}$ subit une
\emph{décroissance exponentielle}\index{décroissance exponentielle}~:
en des temps égaux, des \emph{fractions} égales sont perdues. C'est
exactement le \omterm{def:g12:rc-rl-circuits:capacitor}{condensateur} qui se décharge du
\cref{ch:g12:rc-rl-circuits}, avec $\lambda$ jouant $1/(RC)$~: une
équation différentielle, apprise une fois, fait tourner circuits et
noyaux pareillement.
\end{definition}

\begin{proposition}[Demi-vie]\label{prop:g12:radioactive-decay:halflife}
La \omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie}
(\cref{def:g11:nucleus-radioactivity:halflife}) d'un
\omterm{def:g11:nucleus-radioactivity:notation}{nucléide} est fixée par sa
\omterm{def:g12:radioactive-decay:lambda}{constante de désintégration}~:
$T_{1/2} = \ln 2/\lambda$, et après $n$~demi-vies $N = N_0/2^{\,n}$,
$n$ entier ou non.
\end{proposition}

\begin{proof}
$N(T_{1/2}) = N_0/2$ exige $e^{-\lambda T_{1/2}} = \tfrac12$,
c'est-à-dire $\lambda T_{1/2} = \ln 2$~; et
$N(n\,T_{1/2}) = N_0\,e^{-n\ln 2} = N_0/2^{\,n}$ pour tout $n$~:
l'exponentielle enfile l'escalier et remplit entre les marches.
\end{proof}

\section{Activité~: compter les clics}

\begin{proposition}[Activité]\label{prop:g12:radioactive-decay:activity}
L'\omterm{def:g11:nucleus-radioactivity:activity}{activité} d'un échantillon --- ses
désintégrations par seconde, en \omterm{def:g11:nucleus-radioactivity:activity}{becquerels}
(\cref{def:g11:nucleus-radioactivity:activity}) --- est
\[
\mathcal A = \lambda N, \qquad\text{donc}\qquad \mathcal A(t) = \mathcal A_0 \, e^{-\lambda t}:
\]
proportionnelle à la population survivante, et également exponentielle.
\end{proposition}

\begin{proof}
$N$~noyaux, chacun avec la probabilité $\lambda$ par seconde~:
$\lambda N$~désintégrations par seconde~; multiplier le
\cref{thm:g12:radioactive-decay:law} par $\lambda$ donne
$\mathcal A(t) = \mathcal A_0 e^{-\lambda t}$.
\end{proof}

\begin{example}[Énorme $N$, minuscule $\lambda$]\label{ex:g12:radioactive-decay:body}
Une banane tic-tac à environ \qty{15}{Bq}, un corps humain à environ
\qty{8000}{Bq}, un \omterm{def:g10:orders-of-magnitude:unit}{mètre} cube de granite à quelque
\qty{e6}{Bq} (\cref{ch:g11:nucleus-radioactivity}). Derrière les clics
modestes se tiennent des populations astronomiques~: le potassium-40
du corps contribue $\mathcal A \approx \qty{4.4e3}{Bq}$ avec
$\lambda = \qty{1.8e-17}{s^{-1}}$~: $N = \mathcal A/\lambda \approx
\num{2.5e20}$~noyaux, dont à peine quelques milliers s'allument chaque
seconde.
\end{example}

\begin{method}[Mesurer une demi-vie]\label{met:g12:radioactive-decay:semilog}
\begin{enumerate}
  \item Compter les désintégrations avec un compteur Geiger sur des
        intervalles égaux successifs~: cela échantillonne
        $\mathcal A$ à des instants successifs.
  \item Tracer $\ln \mathcal A$ en fonction de $t$~: puisque
        $\ln \mathcal A = \ln \mathcal A_0 - \lambda t$, la
        \omterm{def:g12:radioactive-decay:exponential}{décroissance exponentielle} se montre
        comme une \emph{droite} --- des données tordues signifient
        une autre loi, ou un mélange de
        \omterm{def:g11:nucleus-radioactivity:notation}{nucléides}.
  \item Lire $\lambda$ comme moins la pente~;
        $T_{1/2} = \ln 2/\lambda$.
\end{enumerate}
\end{method}

\begin{remark}[Les becquerels ne sont pas le mal]\label{rem:g12:radioactive-decay:dose}
Le \omterm{def:g11:nucleus-radioactivity:activity}{becquerel} compte les désintégrations, pas
les dégâts. Le mal se suit par la \omterm{rem:g11:nucleus-radioactivity:dose}{dose} en
\omterm{rem:g11:nucleus-radioactivity:dose}{sieverts} (\cref{ch:g11:nucleus-radioactivity}),
pesant l'\omterm{def:g10:energy-conservation:energy}{énergie} déposée par
\omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de tissu selon le pouvoir destructeur
de chaque rayonnement~: un scan au technétium de \qty{5e8}{Bq} est
courant, bien moins de \omterm{def:g11:nucleus-radioactivity:activity}{becquerels} d'émetteur
$\alpha$ inhalé ne le sont pas~; convertir \unit{Bq} en \unit{Sv} est
matière universitaire.
\end{remark}

\section{Datation~: lire le temps sur un rapport}

La loi de décroissance à l'envers est une horloge~: mesurer la
fraction survivante, et l'exponentielle nomme le temps écoulé.

\begin{method}[Datation au radiocarbone]\label{met:g12:radioactive-decay:carbon}
\begin{enumerate}
  \item \emph{Vivant~:} le tissu vivant échange du carbone avec
        l'atmosphère et porte sa proportion de carbone-14 --- un
        \omterm{def:g11:fundamental-interactions:ladder}{atome} sur $10^{12}$, donnant
        $\mathcal A_0 = 13.6$~désintégrations par minute et par gramme
        de carbone.
  \item \emph{La mort démarre l'horloge~:} l'apport s'arrête, et la
        proportion décroît avec $T_{1/2} = \num{5730}$~ans.
  \item Mesurer l'\omterm{def:g11:nucleus-radioactivity:activity}{activité} $\mathcal A$ de
        l'échantillon par gramme de carbone~;
        $\mathcal A = \mathcal A_0 e^{-\lambda t}$ s'inverse en
        $t = \frac{1}{\lambda}\ln(\mathcal A_0/\mathcal A)
        = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}
\end{method}

\begin{example}[Un foyer dans une grotte]\label{ex:g12:radioactive-decay:hearth}
Du charbon d'un foyer enfoui donne $5.0$~désintégrations par minute
et par gramme~: $\mathcal A_0/\mathcal A = 13.6/5.0 = 2.72$, donc
$t = \num{5730} \times \ln(2.72)/\ln 2 \approx \num{8.3e3}$~ans. Le
feu s'est éteint il y a huit mille ans --- le bois lui-même est
l'archive.
\end{example}

\begin{example}[Horloges pour les roches]\label{ex:g12:radioactive-decay:rocks}
Le carbone-14 se tait au-delà de quelque \num{50000}~ans~; la géologie
garde des horloges plus lentes. \emph{Potassium--argon}~: le
potassium-40 ($T_{1/2} = \num{1.25e9}$~ans) se désintègre en argon, un
gaz qui \omterm{rem:g12:work-and-energy:escape}{s'échappe} de la lave en fusion mais est
piégé une fois la roche cristallisée --- la solidification met
l'horloge à zéro. \emph{Uranium--plomb}~: l'uranium-238 mène une chaîne
de désintégrations se terminant au plomb-206
\omterm{prop:g12:work-and-energy:equilibria}{stable} (\cref{ch:g11:nucleus-radioactivity}), et le
rapport plomb-sur-uranium dans un cristal le date. Les deux horloges
s'accordent sur les minéraux les plus anciens
($\approx \num{4.4e9}$~ans) et, via les météorites, sur l'âge du
Système solaire.
\end{example}

\begin{remark}[Chaînes, et le gaz dans la cave]\label{rem:g12:radioactive-decay:radon}
Entre l'uranium et le plomb la chaîne passe par le radium, puis le
radon-222, un \emph{gaz} radioactif ($T_{1/2} = 3.8$~jours). Né dans
le granite et le sol, il s'infiltre dans les caves et s'y accumule~;
ses filles émettrices $\alpha$ se désintègrent dans les poumons ---
la plus grande \omterm{rem:g11:nucleus-radioactivity:dose}{dose} naturelle de la plupart des
gens, et la moins chère à réduire~: ouvrir la fenêtre de la cave.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:1}
L'iode-131 a $\lambda = \qty{1.0e-6}{s^{-1}}$. Calculer sa
\omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie} en secondes, puis en jours.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:2}
Le radon-222 a $T_{1/2} = 3.8$~jours. Calculer $\lambda$ en
\unit{s^{-1}}, puis l'\omterm{def:g11:nucleus-radioactivity:activity}{activité} d'un
échantillon de \num{1.0e10}~noyaux.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:3}
Vérifier par substitution que $N(t) = N_0\,e^{-\lambda t}$ satisfait
$\dd N/\dd t = -\lambda N$ et $N(0) = N_0$~; puis calculer
$N(T_{1/2})/N_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:4}
Une source de césium-137 ($T_{1/2} = 30$~ans) a une
\omterm{def:g11:nucleus-radioactivity:activity}{activité} de \qty{4.0e5}{Bq} aujourd'hui.
Calculer son \omterm{def:g11:nucleus-radioactivity:activity}{activité} dans $90$~ans, puis
dans $300$~ans.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:5}
Un tracé semi-logarithmique de $\ln \mathcal A$ en fonction de $t$
pour un échantillon de technétium est une droite passant par
$(0, 20.00)$ et $(\qty{10}{hours}, 18.85)$. Trouver $\lambda$ (en
\unit{h^{-1}} et \unit{s^{-1}}) et $T_{1/2}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:radioactive-decay:6}
Un gramme de radium-226 ($T_{1/2} = 1600$~ans~; masse molaire
\qty{226}{g/mol}~; $N_A = \qty{6.02e23}{mol^{-1}}$)~: calculer le
nombre de noyaux, puis $\lambda$, puis
l'\omterm{def:g11:nucleus-radioactivity:activity}{activité} --- la première
\omterm{def:g10:orders-of-magnitude:unit}{unité} d'\omterm{def:g11:nucleus-radioactivity:activity}{activité}
de l'histoire, le curie.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:radioactive-decay:7}
Deux élèves scellent des échantillons de $20$ et \num{2.0e20}~noyaux
d'un même \omterm{def:g11:nucleus-radioactivity:notation}{nucléide}. Que chacun attend-il
après une \omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie}~? La prédiction de qui
est fiable~? (Estimer la fluctuation relative $1/\sqrt n$ pour
chacun.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:radioactive-decay:8}
Un corps tient environ \qty{140}{g} de potassium, dont $0.012\%$ est
du potassium-40 ($T_{1/2} = \num{1.25e9}$~ans, masse molaire
\qty{40}{g/mol}). Calculer le nombre de noyaux de potassium-40, puis
leur \omterm{def:g11:nucleus-radioactivity:activity}{activité}~; comparer au total corporel de
\qty{8000}{Bq}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:radioactive-decay:9}
Un patient reçoit \qty{500}{MBq} de technétium-99m
($T_{1/2} = 6.0$~heures). Calculer
l'\omterm{def:g11:nucleus-radioactivity:activity}{activité} $15$~heures plus tard, puis le
temps pour qu'elle tombe sous \qty{1}{MBq}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:radioactive-decay:10}
Du charbon d'une fosse montre $3.4$~désintégrations par minute et par
gramme de carbone~; un os d'une autre en montre $30\%$ du taux vivant
($\mathcal A_0 = 13.6$ par minute et par gramme,
$T_{1/2} = \num{5730}$~ans). Dater les deux échantillons.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:radioactive-decay:11}
Une cave est scellée avec un lot de radon-222 à l'intérieur
($T_{1/2} = 3.8$~jours). Au bout de combien de temps
l'\omterm{def:g11:nucleus-radioactivity:activity}{activité} du lot est-elle tombée à
$1\%$~? Les vraies caves restent radioactives pendant des décennies~:
d'où vient le radon frais, et pourquoi la ventilation doit-elle être
permanente plutôt qu'une seule fois~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:radioactive-decay:12}
Montrer à partir de $N = N_0 e^{-\lambda t}$ que
$t = T_{1/2}\ln(N_0/N)/\ln 2$. Un os retient $5.0\%$ de son
carbone-14 vivant~: le dater.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:radioactive-decay:13}
Dans un cristal qui n'a piégé aucun argon à la solidification, chaque
\omterm{def:g11:fundamental-interactions:ladder}{noyau} de potassium-40 désintégré laisse un
\omterm{def:g11:fundamental-interactions:ladder}{atome} d'argon en place. Montrer que
$N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$~; une roche
montre un rapport de $3.0$~: trouver son âge
($T_{1/2} = \num{1.25e9}$~ans).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:radioactive-decay:14}
Un compteur Geiger donne, à $t = 0$, $60$, $120$, $180$, $240$~secondes,
des taux de $400$, $225$, $127$, $71$, $40$~comptes par seconde.
Calculer $\ln \mathcal A$ à chaque instant, vérifier l'alignement, et
déduire $\lambda$ et $T_{1/2}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:radioactive-decay:15}
Pour un seul \omterm{def:g11:fundamental-interactions:ladder}{noyau}, la probabilité de
survivre à $n$~demi-vies est $1/2^{\,n}$. Calculer la probabilité que
(a)~un \omterm{def:g11:fundamental-interactions:ladder}{noyau} donné survive à
$10$~demi-vies~; (b)~tous les $8$~noyaux observés survivent à une
\omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie}. (c)~Sur \num{1000}~noyaux, combien
sont attendus après $3$~demi-vies --- et la loi peut-elle dire
\emph{lesquels}~? Conclure sur ce que la loi prédit.
\end{exercise}

\section{Problème~: L'âge des choses}

\begin{problem}[{Problème du week-end --- l'âge des choses~: un
laboratoire de musée calibre l'horloge au carbone, date une statue et
un papyrus, apprend pourquoi aucune horloge ne lit au-delà de son
cadran, et repousse une roche volcanique d'un milliard d'années}]
\label{pb:g12:radioactive-decay:1}
Une semaine dans un laboratoire de datation de musée. Données~: le
tissu vivant montre $\mathcal A_0 = 13.6$~désintégrations de
carbone-14 par minute et par gramme de carbone~;
$T_{1/2} = \num{5730}$~ans (carbone-14), \num{1.25e9}~ans
(potassium-40)~; $1$~an $= \qty{3.156e7}{s}$~;
$N_A = \qty{6.02e23}{mol^{-1}}$~; masse molaire du carbone
\qty{12}{g/mol}~; les compteurs résolvent environ
$0.2$~désintégration par minute et par gramme.

\medskip
\noindent\textbf{Partie I --- Lundi~: calibrer l'horloge.}
\begin{enumerate}
  \item Pourquoi la proportion de carbone-14 est-elle constante dans
        le tissu vivant, et qu'est-ce qui change à la mort~?
  \item Calculer $\lambda$ pour le carbone-14 en \unit{s^{-1}}.
  \item Vérifier par substitution que $N_0 e^{-\lambda t}$ résout
        $\dd N/\dd t = -\lambda N$.
  \item À partir de $\mathcal A_0$, calculer le nombre
        d'\omterm{def:g11:fundamental-interactions:ladder}{atomes} de carbone-14 par gramme de
        carbone dans le tissu vivant.
  \item Combien d'\omterm{def:g11:fundamental-interactions:ladder}{atomes} de carbone un gramme
        en tient-il~? En déduire la proportion de carbone-14 ---
        comparer avec «~un sur $10^{12}$~».
  \item Montrer qu'une \omterm{def:g11:nucleus-radioactivity:activity}{activité} $\mathcal A$
        mesurée date un échantillon comme
        $t = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Mardi~: la statue et le papyrus.}
\begin{enumerate}[resume]
  \item Une statue de bois donne $9.1$~désintégrations par minute et
        par gramme. Dater le bois.
  \item Quel événement, exactement, cette date marque-t-elle --- la
        sculpture ou autre chose~? Quelle prudence s'ensuit~?
  \item Un papyrus égyptien donne $10.4$~désintégrations par minute et
        par gramme. Le dater. Un scribe d'il y a vingt-deux siècles
        est-il plausible~?
  \item L'\omterm{def:g11:nucleus-radioactivity:activity}{activité} est mesurée à $\pm 0.2$
        désintégration par minute. Estimer l'incertitude d'âge du
        papyrus, en utilisant
        $\Delta t \approx \Delta\mathcal A/(\lambda \mathcal A)$.
  \item Un parchemin «~antique~» d'un marchand donne
        $13.4$~désintégrations par minute et par gramme. Verdict~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Jeudi~: les limites de l'horloge.}
\begin{enumerate}[resume]
  \item Après combien de demi-vies l'\omterm{def:g11:nucleus-radioactivity:activity}{activité}
        d'un échantillon tombe-t-elle sous les $0.2$~désintégrations
        par minute et par gramme des compteurs~? Quel âge est-ce~?
  \item Les meilleurs laboratoires atteignent environ dix demi-vies.
        Quel est l'horizon pratique de la datation au radiocarbone,
        en années~?
  \item Un os de dinosaure a environ \num{6.6e7}~ans~: combien de
        demi-vies de carbone-14~? À l'aide de la question~4, après
        combien de demi-vies ne reste-t-il même pas \emph{un}
        \omterm{def:g11:fundamental-interactions:ladder}{atome} du carbone-14 du gramme~?
        Conclure sur le «~datage au carbone des dinosaures~».
  \item Quel genre de \omterm{def:g11:nucleus-radioactivity:notation}{nucléide} pourrait dater
        la roche du dinosaure à la place~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Vendredi~: la roche volcanique.}
\begin{enumerate}[resume]
  \item L'argon est un gaz, le potassium un résident fidèle du
        solide~: expliquer pourquoi la solidification de la lave met
        l'horloge potassium--argon à zéro.
  \item En supposant que chaque
        \omterm{def:g11:fundamental-interactions:ladder}{noyau} de potassium-40 désintégré
        laisse un \omterm{def:g11:fundamental-interactions:ladder}{atome} d'argon piégé, montrer
        que $N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.
  \item La roche sous le fossile du musée montre
        $N_{\mathrm{Ar}}/N_{\mathrm K} = 0.60$. Dater la roche.
  \item Le carbone-14 aurait-il pu dater cette roche, ou le
        potassium--argon la statue~? Donner la fenêtre utile de chaque
        horloge en demi-vies.
  \item Rapport du vendredi, une phrase~: les trois âges trouvés
        cette semaine, et la seule loi qui les a tous lus.
\end{enumerate}
\end{problem}
