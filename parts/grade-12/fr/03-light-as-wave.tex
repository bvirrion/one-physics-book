\chapter{La lumière comme onde~: diffraction et interférences}\label{ch:g12:light-as-wave}

Inclinez un film de savon vers une fenêtre~: des couleurs y glissent,
bien que rien dans le savon ni dans l'eau ne soit coloré. Pointez un
\omterm{def:g11:color-light-sources:lamps}{laser} sur un seul cheveu~: le mur derrière ne montre pas une ombre
\omterm{def:g11:lenses-and-eye:lens}{fine comme un cheveu} mais une traînée lumineuse, barrée de lignes
sombres, des milliers de fois plus large que le cheveu. Des rayons en
ligne droite (\cref{ch:g10:refraction}) n'expliquent ni l'un ni l'autre.
Ce chapitre promeut la lumière au rang d'\omterm{def:g10:signals-and-waves:wave}{onde} complète --- un
demi-micromètre de crête à crête --- puis encaisse l'idée~: un
\omterm{def:g11:color-light-sources:lamps}{laser} et un mur nu mesureront un cheveu, une
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}, les sillons d'un CD.

\section{La lumière est une onde}

\begin{definition}[Ondes lumineuses]\label{def:g12:light-as-wave:lightwave}
La \omterm{def:g10:light-spectra:white}{lumière monochromatique} --- une seule couleur pure, qu'aucun
prisme ne peut scinder davantage (\cref{ch:g10:light-spectra}) --- est
une \emph{onde lumineuse}\index{onde lumineuse} \omterm{def:g10:signals-and-waves:period}{périodique}~: elle
traverse le vide à $c = \qty{3.00e8}{m/s}$ et se caractérise par sa
\omterm{def:g10:signals-and-waves:frequency}{fréquence} $f$ ou sa \emph{longueur d'onde dans le
vide}\index{longueur d'onde!de la lumière} $\lambda = c/f$, la
distance crête à crête du \cref{ch:g12:mechanical-waves}. L'œil
répond d'environ \qty{400}{nm} (violet) à \qty{800}{nm} (rouge)~; la
\omterm{def:g10:light-spectra:white}{lumière blanche} mélange tout le \omterm{def:g11:fundamental-interactions:strong}{domaine}.
\end{definition}

\begin{example}[Ordres de grandeur]\label{ex:g12:light-as-wave:orders}
Un \omterm{def:g11:color-light-sources:lamps}{laser} hélium--néon émet de la lumière rouge de
$\lambda = \qty{633}{nm} = \qty{6.33e-7}{m}$, donc
$f = c/\lambda \approx \qty{4.7e14}{Hz}$~: aucune électronique ne
compte si vite --- chaque \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} de ce chapitre sera
mesurée par la géométrie. Pour l'échelle, un cheveu (environ
\qty{70}{\micro m}) fait une centaine de \omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} de
largeur. Quelles \omterm{def:g10:signals-and-waves:wave}{ondes}~? Pas l'air~: un champ électrique et
\omterm{def:g11:magnetic-fields:bfield}{magnétique}, décrit honnêtement dans le volume de Licence~2~;
seule la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} importe ci-dessous.
\end{example}

\section{Diffraction}

\begin{definition}[Diffraction]\label{def:g12:light-as-wave:diffraction}
Lorsqu'une \omterm{def:g10:signals-and-waves:wave}{onde} rencontre une ouverture ou un obstacle de taille
$a$ comparable à sa \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}, elle s'étale dans la
région que les rayons laisseraient sombre. Cet étalement est la
\emph{diffraction}\index{diffraction} --- négligeable tant que
$\lambda/a$ est minuscule, dominante lorsque $a$ se rapproche de
$\lambda$.
\end{definition}

\begin{proposition}[Demi-largeur angulaire de l'étalement]\label{prop:g12:light-as-wave:halfwidth}
Une fente de largeur $a$, éclairée avec une \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}
$\lambda < a$, étale la lumière autour de la direction incidente dans
la demi-largeur angulaire $\theta$ (radians)~; un écran à la distance
$D$ montre une bande centrale brillante de largeur $L$, deux fois
$D \tan\theta \approx D\,\theta$ (petit angle)~:
\[
\theta \approx \frac{\lambda}{a}, \qquad L = \frac{2\lambda D}{a}.
\]
\end{proposition}
\admitted

\begin{remark}[Lire la formule à l'envers]\label{rem:g12:light-as-wave:honest}
Plus la fente est \emph{étroite}, plus l'étalement est \emph{large}
--- le contraire de la prédiction des rayons. La dérivation honnête
(sommer les ondelettes réémises à travers la fente) vit dans le volume
de Licence~2~; ici la formule est un outil de mesure.
\end{remark}

\begin{example}[Un laser à travers une fente]\label{ex:g12:light-as-wave:slit}
\omterm{def:g11:color-light-sources:lamps}{Laser} rouge, $\lambda = \qty{633}{nm}$~; fente
$a = \qty{0.10}{mm}$~; écran à $D = \qty{2.00}{m}$. Alors
$\theta = \num{6.33e-7} / \num{1.0e-4} = \qty{6.3e-3}{rad}$ --- un
tiers de degré, mais la bande centrale sur le mur a pour largeur
$L = 2 \times \num{6.33e-7} \times 2.00 / \num{1.0e-4} \approx
\qty{2.5}{cm}$, flanquée de lignes sombres et de bandes latérales plus
faibles. Divisez la fente par deux et la bande double.
\end{example}

\begin{proposition}[Un obstacle diffracte comme une fente]\label{prop:g12:light-as-wave:hair}
Un obstacle opaque \omterm{def:g11:lenses-and-eye:lens}{mince} de largeur $d$ --- un fil, un cheveu ---
produit, hors du faisceau incident, le même motif qu'une fente de
largeur $d$ (principe de Babinet, établi dans le volume de Licence~2).
\end{proposition}
\admitted

\begin{method}[Mesurer une largeur minuscule]\label{met:g12:light-as-wave:hairwidth}
\begin{enumerate}
  \item Étirez le cheveu en travers du faisceau du
        \omterm{def:g11:color-light-sources:lamps}{laser}~; placez un écran à une distance $D$ mesurée
        (quelques \omterm{def:g10:orders-of-magnitude:unit}{mètres}).
  \item Mesurez la largeur $L$ de la bande centrale, de ligne sombre à
        ligne sombre.
  \item Puisque $L = 2\lambda D/d$~: $d = 2\lambda D/L$ --- cheveu plus
        fin, motif plus large.
\end{enumerate}
\end{method}

\begin{remark}[La diffraction limite tout instrument]\label{rem:g12:light-as-wave:limits}
La lumière entre dans un appareil photo, un télescope ou un œil par
une ouverture de largeur $a$, si bien qu'une \omterm{def:g11:lenses-and-eye:lens}{lentille} parfaite
estompe encore un point en une tache de taille angulaire environ
$\lambda/a$~: des détails plus proches fusionnent. Un microscope en
lumière visible ne résout donc rien de bien plus fin que
$\lambda \approx \qty{0.5}{\micro m}$ (bactéries oui, virus non)~; les
miroirs de télescope sont construits larges pour la même raison.
\end{remark}

\section{Interférences}

\begin{definition}[Sources cohérentes, différence de marche]\label{def:g12:light-as-wave:coherent}
Deux sources d'\omterm{def:g10:signals-and-waves:wave}{ondes} sont \emph{cohérentes}\index{sources cohérentes}
lorsqu'elles ont la même \omterm{def:g10:signals-and-waves:frequency}{fréquence} et vibrent en phase --- en
pratique, lorsqu'elles sont deux copies d'une même
\omterm{def:g10:signals-and-waves:wave}{onde}, obtenues en la scindant. Pour un point $M$ atteint par
les deux, la \emph{différence de marche}\index{différence de marche}
est $\delta = S_2M - S_1M$.
\end{definition}

\begin{proposition}[Conditions d'interférence]\label{prop:g12:light-as-wave:conditions}
Là où deux \omterm{def:g10:signals-and-waves:wave}{ondes} \omterm{def:g12:light-as-wave:coherent}{cohérentes} de
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} $\lambda$ se chevauchent, elles
s'additionnent --- elles \emph{interfèrent}\index{interférence}~:
\begin{itemize}
  \item $\delta = k\lambda$ ($k$ entier)~: en phase, se renforçant ---
        interférence \emph{constructive}\index{interférence constructive}
        (brillante)~;
  \item $\delta = (k + \tfrac12)\lambda$~: crête sur creux, s'annulant
        --- interférence \emph{destructive}\index{interférence destructive}
        (sombre).
\end{itemize}
\end{proposition}

\begin{proof}
Décaler une \omterm{def:g12:mechanical-waves:periodic}{onde périodique} d'un nombre entier de
\omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} ne change rien~: les crêtes s'empilent sur
les crêtes. Une demi-\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} de plus place crête sur
creux~: des \omterm{def:g10:signals-and-waves:wave}{ondes} égales s'annulent.
\end{proof}

\begin{example}[L'expérience des deux fentes de Young]\label{ex:g12:light-as-wave:young}
Percez deux fentes fines $S_1$, $S_2$, distantes d'une fraction de
millimètre, dans une plaque opaque, et éclairez-les toutes deux avec
un seul \omterm{def:g11:color-light-sources:lamps}{laser}~: deux \omterm{def:g12:light-as-wave:coherent}{sources cohérentes}. Sur un
écran à quelques \omterm{def:g10:orders-of-magnitude:unit}{mètres}, le chevauchement est rayé~: au centre
$\delta = 0$, brillant~; en montant l'écran, $\delta$ croît en
passant par $\lambda/2$ (sombre), $\lambda$ (brillant)\ldots\ obscurité
là où de la lumière ajoutée à de la lumière s'annule.
\end{example}

\begin{definition}[Franges et interfrange]\label{def:g12:light-as-wave:fringes}
Les bandes alternativement brillantes et sombres du motif sont ses
\emph{franges d'interférence}\index{franges d'interférence}~; la
distance $i$ entre centres brillants consécutifs est
l'\emph{interfrange}\index{interfrange}.
\end{definition}

\begin{proposition}[L'interfrange]\label{prop:g12:light-as-wave:interfringe}
Pour des fentes distantes de $a$ et un écran à la distance $D \gg a$,
le point de l'écran à la distance $x$ du centre $O$ a pour
\omterm{def:g12:light-as-wave:coherent}{différence de marche} $\delta \approx a\,x/D$ (géométrie
admise)~; par le \cref{prop:g12:light-as-wave:conditions}, les franges
brillantes siègent en $x_k = k\,\lambda D/a$, espacées régulièrement
de l'\omterm{def:g12:light-as-wave:fringes}{interfrange}
\[
i = \frac{\lambda D}{a}.
\]
Avec $\lambda = \qty{633}{nm}$, $a = \qty{0.20}{mm}$, $D =
\qty{2.0}{m}$~: $i = \qty{6.3}{mm}$ --- le montage grossit $\lambda$
d'un facteur $D/a = 10^4$. La géométrie sous-jacente est établie dans
le volume de Licence~2.
\end{proposition}
\admitted

\begin{method}[Mesurer une longueur d'onde]\label{met:g12:light-as-wave:lambda}
\begin{enumerate}
  \item Envoyez le \omterm{def:g11:color-light-sources:lamps}{laser} inconnu à travers une double fente
        d'écartement $a$ connu~; placez l'écran à une distance $D$
        mesurée.
  \item Mesurez l'empan de $10$ \omterm{def:g12:light-as-wave:fringes}{interfranges}~; divisez par
        $10$~: l'\omterm{def:g12:light-as-wave:fringes}{interfrange} $i$, l'erreur de la règle étant
        divisée par dix.
  \item Alors $\lambda = i\,a/D$.
\end{enumerate}
\end{method}

\begin{remark}[Pourquoi deux lampes n'interfèrent jamais]\label{rem:g12:light-as-wave:lamps}
Deux lampes de bureau éclairant un même mur ne produisent aucune
frange~: chacune émet de courts trains d'\omterm{def:g10:signals-and-waves:wave}{ondes} non corrélés dont
le calage relatif saute aléatoirement un milliard de fois par seconde,
si bien que les franges se déplacent tout aussi vite et que l'œil les
moyenne en une lumière uniforme. L'interférence exige la
\omterm{def:g12:light-as-wave:coherent}{cohérence} --- une \omterm{def:g10:signals-and-waves:wave}{onde} scindée en deux, comme le
font les fentes de Young~; le \omterm{def:g11:color-light-sources:lamps}{laser} est le raccourci moderne,
une longue \omterm{def:g10:signals-and-waves:wave}{onde} propre.
\end{remark}

\section{Couleurs peintes par interférence}

\begin{definition}[Iridescence]\label{def:g12:light-as-wave:iridescence}
Les couleurs qui glissent avec l'angle de vue --- bulles de savon,
irisations d'huile, plumes de paon, dos d'un CD --- sont
l'\emph{iridescence}\index{iridescence}~: pas de pigment, mais
l'interférence dans la \omterm{def:g10:light-spectra:white}{lumière blanche}, renforçant certaines
\omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} et en annulant d'autres.
\end{definition}

\begin{example}[Films de savon et irisations d'huile]\label{ex:g12:light-as-wave:soap}
La lumière frappant un film de savon se réfléchit en partie sur sa
face avant, en partie sur sa face arrière~: deux copies
\omterm{def:g12:light-as-wave:coherent}{cohérentes} dont la \omterm{def:g12:light-as-wave:coherent}{différence de marche} est
fixée par l'épaisseur et l'angle de vue. Chaque épaisseur renvoie
certaines \omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde} renforcées, d'autres annulées~: le
film réfléchit une couleur qui glisse à mesure qu'il s'écoule et
s'\omterm{def:g11:lenses-and-eye:lens}{amincit}. Une irisation d'huile sur asphalte humide joue le même
tour.
\end{example}

\begin{remark}[Lumière monochromatique contre lumière blanche]\label{rem:g12:light-as-wave:white}
En \omterm{def:g10:light-spectra:white}{lumière blanche}, chaque \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} trace ses
propres franges de Young avec son propre $i = \lambda D/a$~: au centre
toutes les couleurs sont brillantes (une frange blanche)~; à quelques
franges de là les motifs se désolidarisent --- bords irisés, puis
flou. La \omterm{def:g10:light-spectra:white}{lumière monochromatique} donne des franges nettes sans
fin~: d'où le \omterm{def:g11:color-light-sources:lamps}{laser} dans chaque mesure précise ici.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:light-as-wave:1}
Calculer la \omterm{def:g10:signals-and-waves:frequency}{fréquence} de la lumière rouge
($\lambda = \qty{700}{nm}$) et de la lumière violette (\qty{400}{nm}).
Laquelle est plus élevée~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:light-as-wave:2}
Un \omterm{def:g11:color-light-sources:lamps}{laser} vert ($\lambda = \qty{532}{nm}$) traverse une fente
de largeur $a = \qty{0.20}{mm}$~: calculer $\theta$, puis la largeur
de la bande centrale sur un écran à \qty{1.5}{m}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:light-as-wave:3}
Une porte a \qty{0.80}{m} de large~: calculer $\lambda/a$ pour une
voix à \qty{425}{Hz} ($v = \qty{340}{m/s}$) et pour de la lumière à
\qty{550}{nm}. Pourquoi entendez-vous, mais ne voyez-vous pas, au coin
de la rue~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:light-as-wave:4}
Deux haut-parleurs pilotés par un même générateur émettent, en phase,
un \omterm{def:g10:signals-and-waves:sound}{son} de \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} \qty{2.0}{cm}.
\omterm{def:g12:light-as-wave:coherent}{Différence de marche} en $M$~: \qty{6.0}{cm}~; en $N$~:
\qty{5.0}{cm}. Fort ou silencieux en chaque point~? Justifier.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:light-as-wave:5}
Fentes de Young~: $\lambda = \qty{633}{nm}$, $a = \qty{0.25}{mm}$,
$D = \qty{1.8}{m}$. Calculer l'\omterm{def:g12:light-as-wave:fringes}{interfrange}. Combien de
franges brillantes tiennent dans les \qty{2.0}{cm} centraux de
l'écran~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:light-as-wave:6}
Un cheveu dans un faisceau de $\lambda = \qty{650}{nm}$, écran à
\qty{1.6}{m}, donne une bande centrale de \qty{2.8}{cm} de large.
Calculer le diamètre du cheveu.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:light-as-wave:7}
Un \omterm{def:g11:color-light-sources:lamps}{laser} inconnu à travers les fentes de Young
($a = \qty{0.20}{mm}$, $D = \qty{1.50}{m}$)~: dix
\omterm{def:g12:light-as-wave:fringes}{interfranges} s'étendent sur \qty{4.7}{cm}. Trouver
$\lambda$ et la couleur. Pourquoi mesurer dix plutôt qu'une~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:light-as-wave:8}
Deux lampes de bureau identiques éclairent le même mur~; les faisceaux
se chevauchent, pourtant aucune frange n'apparaît jamais. Quelle
condition de \omterm{def:g12:light-as-wave:coherent}{cohérence}
(\cref{def:g12:light-as-wave:coherent}) échoue, et comment le montage de
Young la répare-t-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:light-as-wave:9}
Sur un motif à deux fentes ($a = \qty{0.30}{mm}$, $D = \qty{1.4}{m}$),
les centres de la première et de la septième frange brillante sont
distants de \qty{18.0}{mm}~: trouver l'\omterm{def:g12:light-as-wave:fringes}{interfrange}, puis

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:light-as-wave:10}
Les pistes d'un CD forment un réseau d'écartement
$d = \qty{1.6}{\micro m}$. En admettant $d \sin\theta = k\lambda$

\end{exercise}

\section{Problème~: Mesurer avec la lumière}

\begin{problem}[{Problème du week-end --- mesurer avec la lumière~: un
laser, une règle et un mur nu deviennent un atelier micrométrique
jaugeant une fente, un cheveu humain, une longueur d'onde inconnue et
les sillons de deux disques d'argent}]
\label{pb:g12:light-as-wave:1}
Votre kit~: un \omterm{def:g11:color-light-sources:lamps}{laser} rouge ($\lambda = \qty{633}{nm}$,
d'après l'étiquette), un \omterm{def:g11:color-light-sources:lamps}{laser} vert de
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} inconnue, une fente calibrée
$a = \qty{100}{\micro m}$, une double fente d'écartement
$a' = \qty{0.250}{mm}$, un mètre ruban, deux cheveux, un CD et un DVD.

\medskip
\noindent\textbf{Partie I --- Calibrer sur la fente connue.}
\omterm{def:g11:color-light-sources:lamps}{Laser} rouge, fente, mur à $D = \qty{3.00}{m}$.
\begin{enumerate}
  \item Calculer la demi-largeur angulaire $\theta$ du faisceau
        diffracté.
  \item Montrer que la bande centrale a pour largeur $L = 2\lambda D/a$~;
        la calculer.
  \item Vous mesurez $L = \qty{3.7}{cm}$~: calculer l'écart relatif. La
        méthode est-elle validée~?
  \item Une seconde fente est deux fois plus étroite, \qty{50}{\micro m}~:
        prédire son $L$. Énoncer la règle liant largeur de fente et
        largeur du motif.
  \item Justifier l'étape $\tan\theta \approx \theta$ de la question~2~:
        comparer $\theta$ et $\tan\theta$ à cet angle.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le cheveu.} Même
\omterm{def:g11:color-light-sources:lamps}{laser} et même mur~; un cheveu tendu remplace la fente.
\begin{enumerate}[resume]
  \item Quel résultat admis permet de garder la formule de la fente pour
        un cheveu, et que dit-il~?
  \item La bande centrale a $L = \qty{5.4}{cm}$ de large~: calculer le
        diamètre $d$ du cheveu.
  \item Votre règle lit $L$ à $\pm\qty{0.2}{cm}$~: donner le
        \omterm{def:g11:fundamental-interactions:strong}{domaine} résultant pour $d$, sous la forme
        $d \pm \Delta d$.
  \item Le cheveu de votre ami donne $L = \qty{7.6}{cm}$~: son diamètre~?
        Lequel est le plus fin, et pourquoi plus fin signifie plus large~?
  \item Une lampe de bureau pourrait-elle remplacer le
        \omterm{def:g11:color-light-sources:lamps}{laser}~? Donner deux raisons.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} du
\omterm{def:g11:color-light-sources:lamps}{laser} vert.} \omterm{def:g11:color-light-sources:lamps}{Laser} vert, double fente, mur à
$D = \qty{2.00}{m}$.
\begin{enumerate}[resume]
  \item Pourquoi les deux fentes doivent-elles être éclairées par le
        \emph{même} \omterm{def:g11:color-light-sources:lamps}{laser}~?
  \item Pourquoi la frange centrale est-elle brillante quelle que soit
        la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}~?
  \item Dix \omterm{def:g12:light-as-wave:fringes}{interfranges} s'étendent sur \qty{4.26}{cm}. Donner
        $i$, et la raison de mesurer dix d'un coup.
  \item En déduire la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} du
        \omterm{def:g11:color-light-sources:lamps}{laser} vert.
  \item L'étiquette du fabricant indique $\qty{532 \pm 10}{nm}$~:
        cohérent~? Quelle couleur attendez-vous~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Lire des disques d'argent.} Le
\omterm{def:g11:color-light-sources:lamps}{laser} rouge frappe un CD en incidence normale~; ses pistes
agissent comme un réseau par réflexion obéissant (admis) à
$d \sin\theta = k\lambda$. Sur le mur à $D = \qty{1.00}{m}$, chaque
faisceau d'ordre un fait une tache en $x$ de la centrale.
\begin{enumerate}[resume]
  \item Pour le CD, $x = \qty{43}{cm}$~: calculer $\theta$. Pourquoi le
        raccourci des petits angles de la partie~I est-il maintenant
        interdit~?
  \item En déduire l'écartement $d$ des pistes du CD.
  \item Le DVD envoie sa tache d'ordre un en $x = \qty{1.65}{m}$~:
        calculer son $\theta$ et l'écartement de ses pistes.
  \item Comparer les deux écartements. Pourquoi un DVD stocke-t-il plus
        qu'un CD~?
  \item Fermer le carnet~: lister les longueurs mesurées ce week-end,
        et énoncer en une phrase ce qui a fait de la lumière une règle
        pour toutes.
\end{enumerate}
\end{problem}
