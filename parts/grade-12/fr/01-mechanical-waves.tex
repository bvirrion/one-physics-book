\chapter{Ondes mécaniques}\label{ch:g12:mechanical-waves}

Samedi soir, un stade plein~: une vague de spectateurs debout
balaye les tribunes en moins d'une minute, sans que personne ne quitte
son siège. L'anneau d'un caillou traverse l'étang tandis que le bouchon
ne fait que bobiner. Ce chapitre étudie le voyageur lui-même ---
l'\omterm{def:g10:signals-and-waves:wave}{onde} --- et les quatre nombres qui la maîtrisent~: célérité,
\omterm{def:g12:mechanical-waves:delay}{retard}, \omterm{def:g10:signals-and-waves:period}{période}, \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} --- de quoi lire un séisme
sur une bande de papier.

\section{La perturbation voyage, le milieu reste}

\begin{definition}[Onde mécanique]\label{def:g12:mechanical-waves:wave}
Une \emph{onde mécanique}\index{onde mécanique} est une perturbation
qui se propage dans un milieu matériel --- corde, eau, air, roche ---
en transportant de l'\omterm{def:g10:energy-conservation:energy}{énergie} \emph{sans transporter de matière}~: chaque
portion du milieu oscille brièvement sur place, transmet le mouvement,
et se repose.
\end{definition}

\begin{example}[Pas de transport de matière]\label{ex:g12:mechanical-waves:transport}
Dans la \omterm{def:g10:signals-and-waves:wave}{vague} du stade, le «~milieu~» est la foule~: ce qui voyage
est le geste de se lever, tandis que chaque spectateur garde son siège.
Sur l'étang, l'anneau s'élargit mais un bouchon flottant ne fait que
bobiner sur place.
\end{example}

\begin{definition}[Ondes transversales et longitudinales]\label{def:g12:mechanical-waves:transverse}
Une \omterm{def:g10:signals-and-waves:wave}{onde} est \emph{transversale}\index{onde transversale} lorsque le milieu
se déplace \emph{perpendiculairement} à la propagation~: une bosse
court le long d'une corde secouée tandis que chaque point ne fait que
monter et descendre. Elle est \emph{longitudinale}\index{onde
longitudinale} lorsque le milieu se déplace \emph{le long} de la
propagation~: une zone de \omterm{def:g11:magnetic-fields:solenoid}{spires} comprimées parcourt un
ressort poussé, chaque \omterm{def:g11:magnetic-fields:solenoid}{spire} oscillant sur place.
\end{definition}

\begin{remark}[Le son est longitudinal]\label{rem:g12:mechanical-waves:sound}
Le \omterm{def:g10:signals-and-waves:sound}{son} est l'\omterm{def:g12:mechanical-waves:transverse}{onde longitudinale} par excellence~: de l'air
comprimé --- \omterm{def:g10:pressure:pressure}{pression} plus élevée (\cref{ch:g10:pressure}) --- voyage
de la source au tympan~; le \cref{ch:g12:sound-acoustics} lui est
consacré. Les séismes envoient les deux types~: les \omterm{def:g10:signals-and-waves:wave}{ondes}~P
compressent la roche, les \omterm{def:g10:signals-and-waves:wave}{ondes}~S la secouent latéralement.
\end{remark}

\section{Célérité et retard}

\begin{definition}[Célérité]\label{def:g12:mechanical-waves:speed}
La \emph{célérité}\index{célérité} d'une \omterm{def:g10:signals-and-waves:wave}{onde} est la distance
parcourue par la perturbation divisée par le temps mis~:
$v = d/t$. C'est la vitesse de la \emph{forme} qui voyage, non celle
du milieu.
\end{definition}

avancé de \qty{2.4}{m} en \qty{0.30}{s}, d'où $v = \qty{8.0}{m/s}$ ---
la forme voyage, la corde reste.}
\end{omfigure}

\begin{proposition}[Le milieu fixe la célérité]\label{prop:g12:mechanical-waves:medium}
Pour de petites perturbations, la \omterm{def:g12:mechanical-waves:speed}{célérité} ne dépend que du
\emph{milieu}, non de la forme ni de la taille de la perturbation. Sur
une corde tendue, $v$ croît avec la \omterm{def:g10:inertia:inventory}{tension} et décroît avec la masse
par \omterm{def:g10:orders-of-magnitude:unit}{mètre}~: tendue et légère va vite, lâche et lourde va lentement.
\admitted
\end{proposition}

\begin{remark}[Où vit la formule]\label{rem:g12:mechanical-waves:ropeformula}
La formule exacte de la célérité sur une corde est établie dans le
volume de Licence~1~; cette année, les faits expérimentaux ci-dessus
suffisent.
\end{remark}

\begin{example}[Célérités du son]\label{ex:g12:mechanical-waves:speeds}
Le \omterm{def:g10:signals-and-waves:sound}{son} se propage à \qty{340}{m/s} dans l'air, \qty{1500}{m/s}
dans l'eau, \qty{5900}{m/s} dans l'acier --- plus le milieu est
rigide, plus c'est rapide. D'où l'oreille collée au rail des westerns~:
l'acier annonce le train le premier.
\end{example}

\begin{definition}[Retard]\label{def:g12:mechanical-waves:delay}
Une \omterm{def:g10:signals-and-waves:wave}{onde} de célérité $v$ atteint un point à la distance $d$ après
le \emph{retard}\index{retard} $\tau = d/v$~: deux points distants de $d$
reproduisent le même mouvement, le plus éloigné retardé de $\tau$ ---
la télémétrie par \omterm{rem:g10:signals-and-waves:honest}{écho} du \cref{ch:g10:signals-and-waves}, lue
à l'envers.
\end{definition}

\begin{proposition}[Superposition sans interaction]\label{prop:g12:mechanical-waves:superposition}
Là où deux \omterm{def:g10:signals-and-waves:wave}{ondes} se chevauchent, le déplacement du milieu est à
chaque instant la \emph{somme} des déplacements que chacune produirait
seule~; après le croisement, chaque \omterm{def:g10:signals-and-waves:wave}{onde} poursuit inchangée ---
résultat établi dans le volume de Licence~2 à partir de la linéarité des
équations du mouvement. \admitted
\end{proposition}

\begin{example}[Impulsions qui se croisent]\label{ex:g12:mechanical-waves:crossing}
Envoyez une bosse vers le haut depuis chaque extrémité d'une corde~:
là où elles se rencontrent, la corde monte brièvement à la somme des
deux hauteurs~; puis deux bosses intactes émergent --- comme deux
conversations qui se croisent et atteignent leurs auditeurs.
\end{example}

\section{Ondes périodiques~: la double périodicité}

\begin{definition}[Onde périodique]\label{def:g12:mechanical-waves:periodic}
Lorsque la source répète son mouvement avec la \omterm{def:g10:signals-and-waves:period}{période} $T$ et la
\omterm{def:g10:signals-and-waves:frequency}{fréquence} $f = 1/T$ (\cref{ch:g10:signals-and-waves}), elle alimente le
milieu d'une \emph{onde périodique}\index{onde périodique}~: chaque point
répète son propre mouvement avec la \omterm{def:g10:signals-and-waves:period}{période} $T$, chacun retardé
par rapport à ses voisins du \omterm{def:g12:mechanical-waves:delay}{retard} de parcours.
\end{definition}

\begin{definition}[Longueur d'onde]\label{def:g12:mechanical-waves:wavelength}
Une photographie d'une \omterm{def:g12:mechanical-waves:periodic}{onde périodique} montre un motif qui se
répète dans l'espace~; sa \omterm{def:g10:signals-and-waves:period}{période} spatiale --- crête à crête ---
est la \emph{longueur d'onde}\index{longueur d'onde} $\lambda$, en
\omterm{def:g10:orders-of-magnitude:unit}{mètres}. Une \omterm{def:g12:mechanical-waves:periodic}{onde périodique} est
\emph{doublement périodique}\index{double périodicité}~: \omterm{def:g10:signals-and-waves:period}{période} $T$
en temps en chaque point, \omterm{def:g10:signals-and-waves:period}{période} $\lambda$ en espace à chaque
instant.
\end{definition}

\begin{theorem}[La relation de l'onde]\label{thm:g12:mechanical-waves:relation}
Une \omterm{def:g12:mechanical-waves:periodic}{onde périodique} de célérité $v$, de \omterm{def:g10:signals-and-waves:period}{période} $T$ et de
\omterm{def:g10:signals-and-waves:frequency}{fréquence} $f$ a pour \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}
$\lambda = v\,T = v/f$.
\end{theorem}

\begin{proof}
Pendant une \omterm{def:g10:signals-and-waves:period}{période}, le motif glisse de $v\,T$~; mais la source
--- donc chaque point --- est alors revenue à son état de départ, de
sorte que le motif glissé coïncide avec l'original~: la distance de
répétition est $v\,T$.
\end{proof}

\begin{remark}[La fréquence appartient à la source]\label{rem:g12:mechanical-waves:source}
En passant d'un milieu à un autre --- \omterm{def:g10:signals-and-waves:sound}{son} de l'air à l'eau ---
une \omterm{def:g10:signals-and-waves:wave}{onde} conserve sa \omterm{def:g10:signals-and-waves:frequency}{fréquence}, dictée par la source, tandis
que sa célérité change~; par $\lambda = v/f$, la
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} change avec elle.
\end{remark}

\begin{example}[Le la du concert, dans l'air et sous l'eau]\label{ex:g12:mechanical-waves:concertA}
Le la à \qty{440}{Hz} du \cref{ch:g10:signals-and-waves} a pour
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} $340/440 = \qty{0.77}{m}$ dans l'air et
$1500/440 = \qty{3.4}{m}$ dans l'eau~: même note, étirée par le milieu
plus rapide.
\end{example}

\begin{definition}[En phase, en opposition de phase]\label{def:g12:mechanical-waves:phase}
Deux points séparés d'un nombre entier de \omterm{def:g12:mechanical-waves:wavelength}{longueurs d'onde},
$d = k\lambda$, se déplacent identiquement à chaque instant~: ils sont
\emph{en phase}\index{en phase}. Des points séparés de
$d = (k + \tfrac12)\lambda$ se déplacent toujours en sens opposés ---
crête contre creux~: ils sont en \emph{opposition de phase}\index{opposition de phase}.
\end{definition}

\begin{method}[Lire la double périodicité]\label{met:g12:mechanical-waves:reading}
\begin{enumerate}
  \item Sur un enregistrement $y(t)$ en un point fixe, lire $T$ de crête
        à crête~; $f = 1/T$.
  \item Sur un instantané $y(x)$ à un instant fixe, lire $\lambda$ de
        crête à crête.
  \item En déduire $v = \lambda/T = \lambda f$~; vérifier par un
        chronométrage direct $v = d/\tau$ quand c'est possible.
  \item Ne jamais confondre les deux graphiques~: un axe porte des
        secondes, l'autre des \omterm{def:g10:orders-of-magnitude:unit}{mètres}.
\end{enumerate}
\end{method}

\section{Fronts d'onde et atténuation}

\begin{definition}[Front d'onde]\label{def:g12:mechanical-waves:wavefront}
Un \emph{front d'onde}\index{front d'onde} est une ligne (ou une
surface) de points que l'\omterm{def:g10:signals-and-waves:wave}{onde} atteint au même instant --- une
ligne de crête sur l'étang. Depuis une source ponctuelle, les fronts
sont des cercles qui s'élargissent~; loin de toute source, ils sont
presque droits. L'\omterm{def:g10:signals-and-waves:wave}{onde} se propage perpendiculairement à ses
fronts, les crêtes successives étant distantes de $\lambda$.
\end{definition}

\begin{definition}[Atténuation]\label{def:g12:mechanical-waves:attenuation}
La diminution de l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude} d'une \omterm{def:g10:signals-and-waves:wave}{onde} au cours de sa
propagation est son \emph{atténuation}\index{atténuation}. Deux causes
s'additionnent~: l'\omterm{def:g10:energy-conservation:energy}{énergie} d'un \omterm{def:g12:mechanical-waves:wavefront}{front} circulaire est
partagée sur une crête toujours plus longue, et le milieu en absorbe un
peu au passage. Les \omterm{def:g12:mechanical-waves:wavefront}{fronts} droits échappent à la première
cause --- comment la houle d'une tempête lointaine peut traverser un
océan.
\end{definition}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:mechanical-waves:1}
Une \omterm{def:g10:signals-and-waves:wave}{vague} de stade parcourt l'anneau de \qty{320}{m} des tribunes
en \qty{40}{s}.
(a) Calculer sa célérité. (b) Un spectateur parcourt-il l'anneau~?
Qu'est-ce qui voyage~? (c) Chaque spectateur se lève \qty{1.5}{s} après
son voisin situé à \qty{12}{m}~: vérifier la cohérence.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:mechanical-waves:2}
\omterm{def:g12:mechanical-waves:transverse}{Transversale} ou \omterm{def:g12:mechanical-waves:transverse}{longitudinale}~? (a) une impulsion sur une corde
secouée de haut en bas~; (b) une compression envoyée le long d'un
ressort~; (c) le \omterm{def:g10:signals-and-waves:sound}{son} dans l'air~; (d) une ride d'étang (la surface
bobine \omterm{def:g10:universal-gravitation:weight}{verticalement}, l'anneau voyage horizontalement)~; (e)
ondes sismiques~P (compression le long du parcours) et~S (secousse
latérale).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:mechanical-waves:3}
Le tonnerre vous parvient \qty{4.0}{s} après l'éclair. (a) À quelle
distance est la foudre ($v = \qty{340}{m/s}$)~? (b) Un plongeur entend
le même \omterm{def:g12:mechanical-waves:delay}{retard} entre un flash et un bang sous-marin
($v = \qty{1500}{m/s}$)~: à quelle distance est cette source~?
(c) Que faut-il connaître avant de convertir un \omterm{def:g12:mechanical-waves:delay}{retard} en distance~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:mechanical-waves:4}
Calculer la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} d'un \omterm{def:g10:signals-and-waves:sound}{son} à \qty{440}{Hz} dans
l'air et dans l'eau. Lorsque le \omterm{def:g10:signals-and-waves:sound}{son} passe de l'air à l'eau, lesquels
de $f$, $v$, $\lambda$ changent, et lequel reste fixe~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:mechanical-waves:5}

\qty{12}{cm} et les anneaux avancent à \qty{0.30}{m/s}. Trouver la

\omterm{def:g10:signals-and-waves:frequency}{fréquence} de l'\omterm{def:g10:signals-and-waves:wave}{onde}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:mechanical-waves:6}

\begin{exercise}[$\star\star$]\label{exo:g12:mechanical-waves:7}

\begin{exercise}[$\star\star$]\label{exo:g12:mechanical-waves:8}

\begin{exercise}[$\star\star$]\label{exo:g12:mechanical-waves:10}

\begin{exercise}[$\star\star$]\label{exo:g12:mechanical-waves:11}
Un séisme envoie des ondes~P à \qty{6.0}{km/s} et des ondes~S à

\begin{exercise}[$\star\star\star$]\label{exo:g12:mechanical-waves:12}
Un enregistrement de la hauteur d'eau à une bouée montre des crêtes
toutes les \qty{0.50}{s}~; une photo aérienne montre des crêtes
distantes de \qty{0.75}{m}.
(a) Trouver $T$ et $f$. (b) Trouver $\lambda$. (c) En déduire la
célérité. (d) La bouée chevauche une crête en $t = 0$~: quand la
prochaine~? (e) Un point distant de \qty{1.875}{m} est-il
\omterm{def:g12:mechanical-waves:phase}{en phase} avec la bouée, en \omterm{def:g12:mechanical-waves:phase}{opposition de phase},
ou ni l'un ni l'autre~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:mechanical-waves:13}
Une bouée ancrée accomplit $10$ oscillations complètes en \qty{80}{s}

(a) Trouver la \omterm{def:g10:signals-and-waves:period}{période} et la \omterm{def:g10:signals-and-waves:frequency}{fréquence}.
(b) Trouver la célérité de la houle, en \unit{m/s} et en \unit{km/h}.
(c) Combien de temps une crête maintenant à la bouée met-elle pour

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:mechanical-waves:14}
Deux impulsions identiques, l'une vers le haut et l'autre vers le bas,
voyagent l'une vers l'autre à \qty{3.0}{m/s} chacune, centres distants
de \qty{6.0}{m}. (a) Quand leurs centres coïncident-ils, et à quoi

(d) Pourquoi «~momentanément plate~» ne signifie-t-il pas «~rien là~»~?
\end{exercise}

que devient l'\omterm{def:g10:energy-conservation:energy}{énergie} par \omterm{def:g10:orders-of-magnitude:unit}{mètre}, et l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude}~?
(c) Pourquoi une houle à front droit s'atténue-t-elle bien moins~?
(d) Et la \omterm{def:g10:signals-and-waves:wave}{vague} du stade, pas du tout~?

\section{Problème~: Lire un sismogramme}

\begin{problem}[{Problème du week-end --- lire un sismogramme~: deux ondes
s'élancent d'une faille rompue, trois cercles fixent l'épicentre, et une
troisième onde lente donne à une côte son alerte}]
\label{pb:g12:mechanical-waves:1}
Nuit de garde dans un observatoire sismologique. À 09:14:32 l'aiguille
de la station~A sursaute~: des zigzags rapides et nets, puis des
oscillations plus lentes et plus amples. Un séisme a rompu la roche
quelque part~; deux \omterm{def:g10:signals-and-waves:wave}{ondes} s'élancent à travers la croûte~: les
ondes~P, compressions à $v_P = \qty{6.0}{km/s}$, et les ondes~S, une
secousse latérale à $v_S = \qty{3.5}{km/s}$.

\medskip
\noindent\textbf{Partie I --- Deux \omterm{def:g10:signals-and-waves:wave}{ondes}, une horloge.}
\begin{enumerate}
  \item D'après les descriptions ci-dessus, classer les ondes~P et~S
        comme \omterm{def:g12:mechanical-waves:transverse}{transversales} ou
        \omterm{def:g12:mechanical-waves:transverse}{longitudinales}.
  \item Quelle \omterm{def:g10:signals-and-waves:wave}{onde} arrive la première à chaque station~? Calculer
        les temps de parcours des deux sur $d = \qty{120}{km}$.
  \item Montrer que l'écart d'arrivée P--S est
        $\Delta t = d\,(1/v_S - 1/v_P)$ et le calculer pour
        $d = \qty{120}{km}$.
  \item En déduire la règle de télémétrie du sismologue
        $d = \dfrac{v_P\,v_S}{v_P - v_S}\,\Delta t$ et montrer que le
        coefficient vaut environ \qty{8.4}{km} par seconde d'écart.
  \item Pourquoi l'écart croît-il avec la distance --- et pourquoi est-ce
        un cadeau pour une station qui n'a jamais vu le début du séisme~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Cercles sur la carte.}
\begin{enumerate}[resume]
  \item La station~A mesure $\Delta t_A = \qty{25}{s}$. À quelle
        distance de~A se trouve l'épicentre~?
  \item Expliquer pourquoi cette seule lecture place l'épicentre sur un
        cercle, non en un point.
  \item Les stations~B et~C mesurent $\Delta t_B = \qty{15}{s}$ et
        $\Delta t_C = \qty{32}{s}$~: leurs distances à l'épicentre~?
  \item Expliquer comment les trois cercles fixent ensemble l'épicentre.
  \item L'onde~P a atteint~A à 09:14:32. Calculer son temps de parcours
        depuis l'épicentre, et l'heure d'origine du séisme.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- L'\omterm{def:g10:signals-and-waves:wave}{onde} qui traverse l'océan.}
Les trois cercles se rencontrent au large~: la faille s'est rompue sous
le fond marin et a mis en mouvement la colonne d'eau. De longues
\omterm{def:g10:signals-and-waves:wave}{ondes} en eau profonde voyagent à une célérité fixée par la
profondeur --- environ \qty{200}{m/s} en pleine mer, environ
\qty{10}{m/s} près d'une plage (admis ici~; la formule est établie dans
le volume de Licence~1). Un port se trouve à $D = \qty{600}{km}$ de
l'épicentre.
\begin{enumerate}[resume]
  \item Convertir \qty{200}{m/s} en \unit{km/h} et nommer un véhicule de
        vitesse de croisière comparable.
  \item Combien de temps le tsunami met-il pour atteindre le port~?
  \item Le sismomètre propre du port sent d'abord l'onde~P. Au bout de
        combien de temps~? Si l'alarme \omterm{def:g10:signals-and-waves:sound}{sonne} à cet instant, de
        combien de temps d'alerte la côte dispose-t-elle avant le tsunami~?
  \item En mer, la \omterm{def:g10:signals-and-waves:period}{période} du tsunami est d'environ \qty{15}{min}.
        Calculer sa \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} là-bas, et expliquer pourquoi un
        navire ne le remarque pas.
  \item Près de la plage, la célérité tombe à \qty{10}{m/s} tandis que
        la \omterm{def:g10:signals-and-waves:period}{période} reste fixe. Calculer la nouvelle
        \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}. Que doit-il arriver à l'\omterm{def:g10:signals-and-waves:wave}{onde}
        lorsqu'elle se tasse~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- \omterm{def:g10:energy-conservation:energy}{Énergie}, et le rapport.}
Un fort séisme libère de l'ordre de \qty{1e15}{J}, emportés par les
\omterm{def:g10:signals-and-waves:wave}{ondes}.
\begin{enumerate}[resume]
  \item Calculer la longueur d'un \omterm{def:g12:mechanical-waves:wavefront}{front d'onde} circulaire à
        \qty{10}{km} de l'épicentre, puis à \qty{210}{km}~; de quel
        facteur a-t-elle grandi~?
  \item En déduire pourquoi les secousses s'affaiblissent avec la
        distance même dans une croûte sans pertes --- et nommer la
        seconde cause qu'ajoute une croûte réelle.
  \item Un liquide ne résiste pas au cisaillement. Des stations
        antipodes enregistrent l'onde~P mais pas l'onde~S~: que révèle
        cela sur le noyau de la Terre~?
  \item Le train à grande vitesse du chapitre d'\omterm{def:g10:energy-conservation:energy}{énergie} de l'année
        précédente (\cref{ch:g11:mechanical-energy}) transporte environ
        \qty{1.5e9}{J} d'\omterm{def:g11:mechanical-energy:kinetic}{énergie cinétique}~: à combien de
        tels trains correspondent \qty{1e15}{J}~?
  \item Le rapport, en deux phrases~: où et quand le séisme a-t-il eu
        lieu, et de combien de temps d'alerte le port a-t-il disposé~?
        Donner les trois nombres.
\end{enumerate}
\end{problem}
