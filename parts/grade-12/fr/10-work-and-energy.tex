\chapter{Travail et énergie mécanique}\label{ch:g12:work-and-energy}

Faites basculer une bille dans un saladier~: elle plonge, dépasse le
fond, remonte à sa hauteur de départ, hésite, rebrousse chemin --- et
chaque retournement se prédit à partir d'une courbe (le profil du
bol) et d'une droite horizontale (l'\omterm{def:g10:energy-conservation:energy}{énergie} de la bille), sans
résoudre d'équation du mouvement. L'an dernier tarifait les
\omterm{def:g10:inertia:force}{forces} sur des droites (\cref{ch:g11:work-of-force}) et
équilibrait le cinétique contre l'\omterm{def:g12:work-and-energy:ep}{énergie potentielle}
(\cref{ch:g11:mechanical-energy})~; cette année la comptabilité va
partout~: chemins courbes, ressorts, paysages d'un coup d'œil.

\section{Travail le long d'un chemin quelconque}

\begin{definition}[Travail le long d'un chemin]\label{def:g12:work-and-energy:path}
Soit une \omterm{def:g10:inertia:force}{force} $\vect F$ agissant sur un corps qui se déplace le
long d'un chemin quelconque de $A$ à $B$~: découper le chemin en
déplacements $\vect{\dd\ell}$ assez courts pour que chacun soit droit
et que $\vect F$ y change à peine. Chaque pas gagne le petit
\omterm{def:g11:work-of-force:work}{travail} $\scal{F}{\dd\ell} = F\,\dd\ell\cos\theta$
(\cref{ch:g11:work-of-force}), et le
\emph{travail le long du chemin}\index{travail (d'une force)!le long
d'un chemin} est leur somme, $W_{AB}(\vect F) = \sum
\scal{F}{\dd\ell}$ --- qui, sur un chemin droit à \omterm{def:g10:inertia:force}{force}
constante, se réduit au $F \times AB \times \cos\theta$ de l'an
dernier.
\end{definition}

\begin{proposition}[Travail du poids, chemin quelconque]\label{prop:g12:work-and-energy:weight}
Le long de \emph{n'importe quel} chemin de $A$ (altitude $z_A$) à $B$
(altitude $z_B$), le \omterm{def:g10:universal-gravitation:weight}{poids} d'une masse $m$ effectue le
\omterm{def:g11:work-of-force:work}{travail} $W_{AB}(\vect P) = mg\,(z_A - z_B)$~: la dénivellation
entre, jamais l'itinéraire.
\end{proposition}

\begin{proof}
Chaque petit pas contribue $mg \times (\text{petite chute})$
(calcul du segment droit de l'an dernier)~; les chutes se
télescopent en $z_A - z_B$. Le cas courbe, admis alors, est livré.
\end{proof}

\begin{definition}[Forces conservatives et non conservatives]\label{def:g12:work-and-energy:conservative}
Une \omterm{def:g10:inertia:force}{force} est \emph{conservative}\index{force conservative} si son
\omterm{def:g11:work-of-force:work}{travail} entre deux points est le même le long de tout chemin
(nul sur tout aller-retour), \emph{non conservative}%
\index{force non conservative} sinon. \omterm{def:g10:universal-gravitation:weight}{Poids} et
\omterm{def:g10:inertia:force}{force} de ressort (section suivante)~: conservatives. Le
\omterm{def:g10:inertia:inventory}{frottement} de glissement ne l'est pas~: opposé au mouvement, il
effectue $W = -fL$ sur un chemin de longueur $L$ --- un péage par
\omterm{def:g10:orders-of-magnitude:unit}{mètre}, jamais remboursé.
\end{definition}

\begin{example}[Deux sentiers vers le refuge]\label{ex:g12:work-and-energy:trails}
Un randonneur de \qty{75}{kg} gagne \qty{300}{m} par un sentier raide
de \qty{800}{m} ou \qty{2.4}{km} de lacets. Le \omterm{def:g10:universal-gravitation:weight}{poids} facture
$-75 \times 9.81 \times 300 \approx \qty{-2.2e5}{J}$ sur les deux~;
une \omterm{def:g10:inertia:force}{force} de \omterm{def:g10:inertia:inventory}{frottement} de \qty{30}{N},
\qty{-2.4e4}{J} sur l'un, \qty{-7.2e4}{J} sur l'autre. Le
\omterm{def:g10:universal-gravitation:weight}{poids} rembourse à la descente~; la facture du
\omterm{def:g10:inertia:inventory}{frottement} est partie en chaleur, dans les deux sens.
\end{example}

\section{Énergie potentielle}

\begin{definition}[Énergie potentielle]\label{def:g12:work-and-energy:ep}
À toute \omterm{def:g12:work-and-energy:conservative}{force conservative} on attache une
\emph{énergie potentielle}\index{énergie potentielle!issue du
travail conservatif} $E_p$, un nombre ne dépendant que de la
position, définie par $W_{AB} = E_p(A) - E_p(B)$~: le
\omterm{def:g11:work-of-force:work}{travail} livré est l'\omterm{def:g10:energy-conservation:energy}{énergie} potentielle dépensée.
Fixer $E_p = 0$ en un point de référence choisi la fixe partout~; le
choix décale toutes les valeurs d'une constante qui s'annule de
toute différence. Pour le \omterm{def:g10:universal-gravitation:weight}{poids} cela redonne le
$E_p = mgz$ de l'an dernier, valable désormais le long de tout
chemin. Le \omterm{def:g10:inertia:inventory}{frottement} ne peut posséder de $E_p$~: une facture
dépendant de l'itinéraire n'est pas une différence de nombres aux
extrémités --- seules les \omterm{def:g12:work-and-energy:conservative}{forces conservatives} stockent
de façon récupérable, exactement ce que le nom conserve.
\end{definition}

\begin{proposition}[Énergie potentielle élastique]\label{prop:g12:work-and-energy:elastic}
Un ressort de \omterm{def:g12:oscillators-and-time:spring}{raideur} $k$ (\unit{N/m}) étiré ou comprimé
de $x$ tire en arrière avec la \omterm{def:g10:inertia:force}{force} $F = kx$
(\cref{ch:g12:oscillators-and-time})~; le \omterm{def:g11:work-of-force:work}{travail} qu'il livre
en revenant au repos --- son
\emph{énergie potentielle élastique}\index{énergie
potentielle!élastique} --- est $E_p = \tfrac12\, k x^2$.
\end{proposition}

\begin{proof}
La \omterm{def:g12:oscillators-and-time:pendulum}{force de rappel} agit le long du mouvement, donc
$\sum F\,\dd\ell$ est l'\emph{aire sous le graphe $F$--$x$}~: un
triangle de base $x$ et de hauteur $kx$, aire $\tfrac12 x \times
kx$. La \omterm{def:g10:inertia:force}{force} à chaque allongement est la même à l'aller et
au retour~: \omterm{def:g11:work-of-force:work}{travail} fixé par les extrémités,
\omterm{def:g10:inertia:force}{force} de ressort \omterm{def:g12:work-and-energy:conservative}{conservative}, $E_p$ bien
définie.
\end{proof}

\section{Les deux théorèmes de l'énergie}

\begin{theorem}[Théorème de l'énergie cinétique]\label{thm:g12:work-and-energy:ketheorem}\index{théorème de l'énergie cinétique}
Pour tout mouvement d'un corps de masse $m$ de $A$ à $B$ --- courbe,
bouclé, tridimensionnel ---
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
la somme courant sur toutes les \omterm{def:g10:inertia:force}{forces} agissant sur le corps.
\end{theorem}

\begin{proof}
Puisque $v^2 = \scal{v}{v}$, la règle du produit des mathématiques
de cette année donne la dérivée de $v^2$ comme $2\,\scal{a}{v}$,
donc $\dd E_k/\dd t = m\,\scal{a}{v} = \scal{F_{\text{tot}}}{v}$ par
la \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton}
(\cref{ch:g12:newtons-laws}). En un court $\dd t$ le corps se déplace
de $\vect{\dd\ell} = \vect v\,\dd t$~: $E_k$ gagne le petit
\omterm{def:g11:work-of-force:work}{travail} de la \cref{def:g12:work-and-energy:path}~; sommer les
pas somme les \omterm{def:g11:work-of-force:work}{travaux} --- l'énoncé admis de l'an dernier,
gagné en plein.
\end{proof}

\begin{proposition}[Puissance instantanée]\label{prop:g12:work-and-energy:power}
À chaque instant une \omterm{def:g10:inertia:force}{force} $\vect F$ sur un corps de vitesse
$\vect v$ livre la \omterm{def:g11:work-of-force:power}{puissance} $P = \scal{F}{v} = Fv\cos\theta$,
et $\dd E_k/\dd t$ est la somme des \omterm{def:g11:work-of-force:power}{puissances} de toutes les
\omterm{def:g10:inertia:force}{forces}~: la formule à vitesse constante de l'an dernier,
exacte à chaque instant de tout mouvement.
\end{proposition}

\begin{proof}
Lire sur la preuve précédente~: chaque \omterm{def:g10:inertia:force}{force} alimente
$\scal{F}{\dd\ell} = \scal{F}{v}\,\dd t$ dans $E_k$ pendant $\dd t$.
\end{proof}

\begin{theorem}[Théorème de l'énergie mécanique]\label{thm:g12:work-and-energy:emtheorem}\index{théorème de l'énergie mécanique}
Soit $E_m = E_k + E_p$, où $E_p$ regroupe les énergies potentielles
de toutes les \omterm{def:g12:work-and-energy:conservative}{forces conservatives} au
\omterm{def:g11:work-of-force:work}{travail}. Alors, le long de tout chemin,
\[
\Delta E_m = W_{AB}(\text{forces non conservatives}),
\]
et $E_m$ est conservée dès que les \omterm{def:g12:work-and-energy:conservative}{forces non
conservatives} n'effectuent aucun \omterm{def:g11:work-of-force:work}{travail}.
\end{theorem}

\begin{proof}
Scinder le théorème de l'\omterm{def:g10:energy-conservation:energy}{énergie cinétique}~: $\Delta E_k =
W_{\text{cons}} + W_{\text{nc}} = -\Delta E_p + W_{\text{nc}}$~;
passer $\Delta E_p$ de l'autre côté.
\end{proof}

\begin{example}[Le pendule, honnêtement cette fois]\label{ex:g12:work-and-energy:pendulum}
Une masse de pendule sur un fil de longueur $L = \qty{2.5}{m}$ est
lâchée au repos à $\qty{45}{\degree}$ de la verticale. La
\omterm{def:g10:inertia:inventory}{tension}, perpendiculaire au mouvement, est sans puissance~;
le \omterm{def:g10:universal-gravitation:weight}{poids} est \omterm{def:g12:work-and-energy:conservative}{conservatif}~: $E_m$ est
conservée \emph{sur l'arc courbe}, ce que l'an dernier ne pouvait
que vérifier. La masse part de $h = L(1 - \cos\qty{45}{\degree}) =
\qty{0.73}{m}$ plus haut, donc au bas $v = \sqrt{2gh} \approx
\qty{3.8}{m/s}$.
\end{example}

\begin{method}[Audit énergétique d'un mouvement quelconque]\label{met:g12:work-and-energy:audit}
\begin{enumerate}
  \item Choisir le système et lister les \omterm{def:g10:inertia:force}{forces} sur lui.
  \item Trier~: perpendiculaire au mouvement $\to$ pas de
        \omterm{def:g11:work-of-force:work}{travail}~; \omterm{def:g12:work-and-energy:conservative}{conservative} $\to$ une $E_p$
        dans $E_m$~; le reste $\to$ $W_{\text{nc}}$
        (\omterm{def:g10:inertia:inventory}{frottement} de glissement~: $-fL$, $L$ = longueur du
        chemin).
  \item Écrire $\Delta E_m = W_{\text{nc}}$ entre le point de données
        et le point de la question~; résoudre.
        L'\omterm{def:g10:energy-conservation:energy}{énergie} répond «~à quelle vitesse, à quelle
        hauteur~»~; seuls le temps et l'accélération exigent les
        lois de Newton.
\end{enumerate}
\end{method}

\section{Diagrammes d'énergie}

\begin{definition}[Diagramme d'énergie, points de rebroussement]\label{def:g12:work-and-energy:diagram}
Pour un mouvement le long d'un axe $x$ avec $E_m$ conservée, le
\emph{diagramme d'énergie}\index{diagramme d'énergie} trace la
courbe $E_p(x)$ et la droite horizontale $E_m$. Puisque
$E_k = E_m - E_p \geq 0$, le mouvement est confiné là où la courbe
est sous la droite~; l'écart entre elles est $E_k$, lu
directement. En un \emph{point de rebroussement}\index{point de
rebroussement}, où courbe et droite se rencontrent, $v = 0$ et le
mouvement s'inverse.
\end{definition}

\begin{proposition}[Force issue du paysage~; équilibres]\label{prop:g12:work-and-energy:equilibria}
La \omterm{def:g12:work-and-energy:conservative}{force conservative} le long de $x$ est moins la
pente de $E_p$~: $F = -\,\dd E_p/\dd x$ --- elle pointe
\emph{vers le bas} sur le diagramme, et s'annule là où la tangente
est horizontale~: un équilibre. Un minimum de $E_p$ est
\emph{stable}\index{équilibre stable} --- le corps déplacé est
repoussé, une bille dans une vallée~; un maximum est
\emph{instable}\index{équilibre instable} --- il est chassé, une
bille sur un sommet.
\end{proposition}

\begin{proof}
Sur un petit pas $\dd x$ la \omterm{def:g10:inertia:force}{force} \omterm{def:g11:work-of-force:work}{travaille}
$F\,\dd x = E_p(x) - E_p(x + \dd x) = -\dd E_p$~; diviser par
$\dd x$. Vers le bas de chaque côté d'une vallée pointe vers
l'intérieur~; autour d'un sommet, vers l'extérieur.
\end{proof}

\begin{remark}[Oscillations et échappement]\label{rem:g12:work-and-energy:escape}
Les petites oscillations du \cref{ch:g12:oscillators-and-time} vivent
au fond des vallées de $E_p$. Et loin de la Terre la gravité
s'affaiblit~: la courbe $E_p(r)$ grimpe de plus en plus lentement
vers un plafond fini (l'aire sous le graphe de force qui se rétrécit
est finie). Une fusée dont $E_m$ atteint le plafond ne rencontre
jamais de \omterm{def:g12:work-and-energy:diagram}{point de rebroussement}~: elle
\emph{s'échappe}\index{vitesse de libération} --- environ
\qty{11.2}{km/s} depuis le sol (\cref{exo:g12:work-and-energy:13})~;
en dessous elle est liée. Le plafond lui-même se calcule dans le
volume de Licence~1.
\end{remark}

\begin{example}[Looping]\label{ex:g12:work-and-energy:loop}
Une bille lâchée d'une hauteur $H$ glisse dans une boucle verticale
de rayon $R$. La dynamique circulaire (\cref{ch:g12:newtons-laws})
exige $v_{\text{top}}^2 \geq gR$ au sommet --- la donnée empruntée
par le problème de montagnes russes de l'an dernier, désormais
nôtre. La conservation depuis le repos en $H$ jusqu'au sommet
(hauteur $2R$) donne $v_{\text{top}}^2 = 2g(H - 2R)$, donc
$H \geq \tfrac52 R$~: un demi-rayon au-dessus du sommet de la boucle,
quelle que soit la masse --- un peu plus dans le monde réel, qui
frotte.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:1}
Un ressort de \omterm{def:g12:oscillators-and-time:spring}{raideur} $k = \qty{200}{N/m}$ est étiré de
\qty{5.0}{cm}. Calculer l'\omterm{def:g10:energy-conservation:energy}{énergie} stockée~; que devient-elle
si l'allongement double~? Combien coûte le \emph{second}
\qty{5.0}{cm}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:2}
Un drone de \qty{0.90}{kg} vole sur un itinéraire errant d'une
terrasse à $z = \qty{12}{m}$ à un balcon à $z = \qty{30}{m}$.
\omterm{def:g11:work-of-force:work}{Travail} de son \omterm{def:g10:universal-gravitation:weight}{poids}~? Le long de la
montée droite~? Sur l'aller-retour~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:3}
\omterm{def:g12:work-and-energy:conservative}{Conservative}, \omterm{def:g12:work-and-energy:conservative}{non
conservative}, ou sans travail~? Justifier à partir des chemins~:
(a)~le \omterm{def:g10:universal-gravitation:weight}{poids}~; (b)~la \omterm{def:g10:inertia:force}{force} de ressort~;
(c)~le \omterm{def:g10:inertia:inventory}{frottement} de glissement~; (d)~la
\omterm{def:g12:newtons-laws:friction}{force normale} sur une caisse qui glisse~; (e)~la traînée
de l'air.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:4}
Un pendule de longueur \qty{2.0}{m} est lâché au repos à
$\qty{60}{\degree}$ de la verticale. Trouver sa vitesse au point le
plus bas. Pourquoi la \omterm{def:g10:inertia:inventory}{tension} du fil n'entre-t-elle jamais
dans le bilan~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:5}
Une cycliste roule à \qty{9.0}{m/s} sur le plat, livrant
\qty{250}{W}~: quelle \omterm{def:g10:inertia:force}{force} \omterm{def:g11:work-of-force:sign}{résistante}
totale combat-elle~? En montée à \qty{5.0}{m/s} avec la même

\section{Problème~: Le tremplin de saut à skis}

\begin{problem}[{Problème du week-end --- le tremplin de saut à skis~:
une piste d'élan auditée par un radar, un vol en chute libre, la
géométrie douce d'un atterrissage raide, et la tour qu'un concepteur
doit acheter une fois les factures de frottement payées}]
\label{pb:g12:work-and-energy:1}
Un sauteur olympique ($m = \qty{70}{kg}$, skis inclus) part du repos
d'une porte $H = \qty{45}{m}$ au-dessus de la table de décollage,
glisse sur $L = \qty{90}{m}$ de piste d'élan courbe, et quitte la
table horizontalement~; un radar y lit $v_0 = \qty{26}{m/s}$. Le
point d'atterrissage se trouve $h = \qty{40}{m}$ plus bas, sur une
pente de $\qty{35}{\degree}$. Les parties~II et~III négligent la
résistance de l'air.

\medskip\noindent\textbf{Partie I --- La piste d'élan.}
\begin{enumerate}
  \item Prédire la vitesse de décollage sans frottement~; pourquoi la
        courbe de la piste d'élan est-elle hors de propos~?
  \item Référence à la table~: calculer $E_m$ à la porte et au
        décollage.
  \item Calculer $\Delta E_m$~; quel théorème nomme les coupables, et
        qui sont-ils~?
  \item En déduire la \omterm{def:g10:inertia:force}{force} \omterm{def:g11:work-of-force:sign}{résistante}
        moyenne (neige plus air) sur la piste.
  \item Quelle fraction de $E_m$ est perdue~? Pourquoi cirer les skis
        et se baisser~?
\end{enumerate}

\medskip\noindent\textbf{Partie II --- Le vol.}
\begin{enumerate}[resume]
  \item Seul le \omterm{def:g10:universal-gravitation:weight}{poids} agit en vol~: quel mouvement du
        \cref{ch:g12:projectile-motion} est-ce~? Composantes de la
        vitesse à l'atterrissage~?
  \item Vitesse d'atterrissage, deux fois~: à partir des composantes,
        et par conservation de $E_m$. En \unit{km/h}~?
  \item Calculer le temps de vol et la distance horizontale parcourue.
  \item Les vrais sauteurs chevauchent l'air comme une aile, volant
        beaucoup plus loin~: l'air peut-il \emph{augmenter} la
        vitesse d'atterrissage~? Argumenter avec le théorème de
        l'\omterm{def:g10:energy-conservation:energy}{énergie mécanique}.
\end{enumerate}

\medskip\noindent\textbf{Partie III --- L'atterrissage.}
\begin{enumerate}[resume]
  \item Calculer l'angle de la vitesse d'atterrissage sous
        l'horizontale.
  \item La décomposer en composantes parallèle et perpendiculaire à
        la pente.
  \item Les jambes n'absorbent que la partie perpendiculaire~:
        calculer cette \omterm{def:g11:mechanical-energy:kinetic}{énergie cinétique}~; la
        comparer au total.
  \item Un sol plat absorberait le tout~: à quelle chute verticale
        chaque atterrissage est-il équivalent
        ($h_{\text{eq}} = v_\perp^2/2g$, resp.\ $v^2/2g$)~?
  \item Une phrase~: pourquoi les collines d'atterrissage sont-elles
        raides et courbes~?
\end{enumerate}

\medskip\noindent\textbf{Partie IV --- La tour du concepteur.}
Une colline plus petite n'a besoin que de $v_0' = \qty{24}{m/s}$ au
décollage~; sa piste d'élan droite descend à $\qty{35}{\degree}$,
donc la hauteur de porte $H'$ signifie une longueur de piste
$H'/\sin\qty{35}{\degree}$~; même traînée de \qty{80}{N}.
\begin{enumerate}[resume]
  \item Calculer la hauteur de porte qu'un concepteur sans frottement
        construirait.
  \item Écrire le théorème de l'\omterm{def:g10:energy-conservation:energy}{énergie mécanique}, porte
        à table, pour $H'$.
  \item Résoudre pour $H'$.
  \item De quel pourcentage le \omterm{def:g10:inertia:inventory}{frottement} a-t-il relevé la
        tour~?
  \item La masse ne s'annule plus~: recalculer $H'$ pour un sauteur
        de \qty{60}{kg}. Qui a besoin de la tour la plus haute, et
        pourquoi~?
  \item Chute~: la tour annoncée, et la part du
        \omterm{def:g10:inertia:inventory}{frottement} dedans.
\end{enumerate}
\end{problem}
