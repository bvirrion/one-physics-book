\chapter{Cinématique à deux dimensions}\label{ch:g12:kinematics-2d}

Regardez une voiture prendre un rond-point à un \qty{30}{km/h} constant~:
l'aiguille ne bouge pas, pourtant chaque passager se sent tiré de côté
--- le mouvement change, et ce n'est pas la \omterm{def:g12:kinematics-2d:velocity}{vitesse}. Un nombre
par instant ne suffit plus~: le mouvement dans le plan a besoin de
flèches. Ce chapitre équipe un point mobile de trois --- position,
vitesse, accélération --- enchaînées par la dérivée, et lit des familles
entières de mouvements sur leur géométrie.

\section{Le vecteur position}

Le mouvement n'existe que par rapport à un \omterm{def:g10:relative-motion:frame}{référentiel} déclaré
--- un corps rigide plus une horloge (\cref{ch:g10:relative-motion}).
Nouveau cette année~: la précision. Attacher au référentiel une origine
$O$ et deux axes gradués perpendiculaires $Ox$, $Oy$, et enregistrer où
se trouve le point à chaque instant, comme une paire de fonctions.

\begin{definition}[Vecteur position]\label{def:g12:kinematics-2d:position}
Dans un référentiel muni d'une origine $O$ et d'axes $Ox$, $Oy$, le
\emph{vecteur position}\index{vecteur position} d'un point mobile $M$
à l'instant $t$ est $\vect{OM}(t)$, de coordonnées
$\bigl(x(t),\, y(t)\bigr)$~: deux fonctions du temps, une par axe. La
courbe tracée par $M$ lorsque $t$ varie est sa
\omterm{def:g10:relative-motion:trajectory}{trajectoire}.
\end{definition}

\section{Le vecteur vitesse}

\begin{definition}[Vecteur vitesse]\label{def:g12:kinematics-2d:velocity}
Sur un court intervalle le point se déplace de
$\vect{OM}(t+\Delta t) - \vect{OM}(t)$~; diviser par $\Delta t$ et
laisser l'intervalle se rétrécir~: c'est la dérivée de votre cours de
mathématiques, appliquée à chaque coordonnée. Le \emph{vecteur
vitesse}\index{vecteur vitesse} de $M$ est la dérivée de son
\omterm{def:g12:kinematics-2d:position}{vecteur position}, $\vect v(t) = \bigl(x'(t),\, y'(t)\bigr)$,
en \unit{m/s}. Sa norme $v = \sqrt{x'^2 + y'^2}$ est la
\emph{vitesse}\index{vitesse (scalaire)} --- le seul nombre que montre
le compteur.
\end{definition}

\begin{example}[Un drone en vol horizontal]\label{ex:g12:kinematics-2d:drone}
Un drone vole avec $x(t) = 4.0\,t$ et $y(t) = 3.0\,t$ (mètres,
secondes)~: \omterm{def:g10:relative-motion:trajectory}{trajectoire} la droite $y = \tfrac34 x$, vitesse
$\vect v = (4.0, 3.0)$ à tout instant, \omterm{def:g12:kinematics-2d:velocity}{vitesse}
$\sqrt{4.0^2 + 3.0^2} = \qty{5.0}{m/s}$, constante.
\end{example}

\begin{proposition}[Tangente à la trajectoire]\label{prop:g12:kinematics-2d:tangent}
À chaque instant, $\vect v(t)$ est tangent à la
\omterm{def:g10:relative-motion:trajectory}{trajectoire} à la position actuelle de $M$, et pointe dans
le sens du parcours.
\end{proposition}

\begin{proof}
Le déplacement est une corde de la \omterm{def:g10:relative-motion:trajectory}{trajectoire}~; lorsque
$\Delta t$ se rétrécit, la direction de la corde devient celle de la
tangente.
\end{proof}

\section{Le vecteur accélération}

\begin{definition}[Vecteur accélération]\label{def:g12:kinematics-2d:acceleration}
Le \emph{vecteur accélération}\index{vecteur accélération} de $M$ est
la dérivée de son \omterm{def:g12:kinematics-2d:velocity}{vecteur vitesse},
$\vect a(t) = \bigl(v_x'(t),\, v_y'(t)\bigr) =
\bigl(x''(t),\, y''(t)\bigr)$, en \unit{m/s^2}~: à quelle vitesse, et
dans quelle direction, le \omterm{def:g12:kinematics-2d:velocity}{vecteur vitesse} change actuellement.
\end{definition}

\begin{remark}[Accélérer sans accélérer]\label{rem:g12:kinematics-2d:turning}
$\vect a \neq \vect 0$ dès que le vecteur vitesse change --- en
longueur \emph{ou en direction}~: une voiture qui freine accélère
(vers l'arrière)~; une voiture qui tourne à \omterm{def:g12:kinematics-2d:velocity}{vitesse} constante
accélère (latéralement). C'est la grandeur que contrôlent les
\omterm{def:g10:inertia:force}{forces}
(\cref{ch:g11:forces-and-motion,ch:g12:newtons-laws}).
\end{remark}

\begin{method}[Lire une chronophotographie]\label{met:g12:kinematics-2d:chrono}
Une \emph{chronophotographie}\index{chronophotographie} enregistre les
positions $M_0, M_1, M_2, \dots$ d'un point mobile à intervalles de
temps égaux $\Delta t$.
\begin{enumerate}
  \item L'espacement des points lit la \omterm{def:g12:kinematics-2d:velocity}{vitesse}~: espacements
        égaux, \omterm{def:g12:kinematics-2d:velocity}{vitesse} constante~; qui s'ouvrent, accélération~;
        qui se resserrent, ralentissement.
  \item La vitesse en $M_i$ est la corde symétrique,
        $\vect v_i \approx \vect{M_{i-1}M_{i+1}}/(2\,\Delta t)$,
        tracée en $M_i$, tangente au chemin pointillé.
  \item Pour l'accélération en $M_i$~: reporter $\vect v_{i-1}$ et
        $\vect v_{i+1}$ en $M_i$, tracer
        $\Delta\vect v = \vect v_{i+1} - \vect v_{i-1}$ bout à bout~;
        alors $\vect a_i \approx \Delta\vect v/(2\,\Delta t)$.
\end{enumerate}
\end{method}

\section{Mouvements rectilignes}

Le long d'une droite un seul axe suffit~: $x(t)$, $v(t) = x'(t)$,
$a(t) = v'(t)$, tous nombres signés.

\begin{definition}[Mouvement rectiligne uniforme]\label{def:g12:kinematics-2d:uniform}
Un mouvement est \emph{rectiligne uniforme}\index{mouvement rectiligne
uniforme} lorsque $\vect v$ est un vecteur constant~:
\omterm{def:g10:relative-motion:trajectory}{trajectoire} droite, \omterm{def:g12:kinematics-2d:velocity}{vitesse} constante,
$\vect a = \vect 0$. Alors $x(t) = v\,t + x_0$.
\end{definition}

\begin{proposition}[Mouvement rectiligne uniformément accéléré]\label{prop:g12:kinematics-2d:uarm}
\index{mouvement uniformément accéléré}%
Si un point se déplace le long de $Ox$ avec une accélération $a$
\emph{constante}, partant en $t = 0$ de la position $x_0$ avec la
vitesse $v_0$, alors
\[
v(t) = a\,t + v_0,
\qquad
x(t) = \tfrac12\,a\,t^2 + v_0\,t + x_0 .
\]
\end{proposition}

\begin{proof}
$v$ est une primitive de la constante $a$~: $v(t) = a t + C$, et
$t = 0$ \omterm{def:g10:inertia:force}{force} $C = v_0$. À son tour $x$ est une primitive de
$a t + v_0$~: $x(t) = \tfrac12 a t^2 + v_0 t + C'$, avec $C' = x_0$.
\end{proof}

\begin{example}[Distance de freinage]\label{ex:g12:kinematics-2d:braking}
Une voiture à la \omterm{def:g12:kinematics-2d:velocity}{vitesse} $v_0$ freine avec une décélération
constante $a$ (accélération $-a$). Elle s'arrête lorsque
$v(t) = -a t + v_0 = 0$, en $t_s = v_0/a$, ayant parcouru
$d = -\tfrac12 a t_s^2 + v_0 t_s = v_0^2/(2a)$. Le carré est le titre
de la sécurité routière~: doubler la \omterm{def:g12:kinematics-2d:velocity}{vitesse} \emph{quadruplé}
la distance de freinage --- avec $a = \qty{6.0}{m/s^2}$, \qty{16}{m}
depuis \qty{50}{km/h}, \qty{64}{m} depuis \qty{100}{km/h}.
\end{example}

\section{Mouvement circulaire uniforme et repère de Frenet}

\begin{definition}[Mouvement circulaire uniforme]\label{def:g12:kinematics-2d:ucm}
Un mouvement est \emph{circulaire uniforme}\index{mouvement circulaire
uniforme} lorsque la \omterm{def:g10:relative-motion:trajectory}{trajectoire} est un cercle de rayon
$R$ et la \omterm{def:g12:kinematics-2d:velocity}{vitesse} $v$ est constante. La
\emph{période}\index{période!de rotation} $T$ est la durée d'un tour, et
la \emph{fréquence}\index{fréquence!de rotation} $f = 1/T$ (en
\unit{Hz}) le nombre de tours par seconde~:
$v = 2\pi R/T = 2\pi R f$.
\end{definition}

\omterm{def:g12:kinematics-2d:velocity}{Vitesse} constante --- pourtant la vitesse tourne~: il
\emph{y a} une accélération. Dans quelle direction, et de quelle
taille~?

\begin{theorem}[Accélération centripète]\label{thm:g12:kinematics-2d:centripetal}
\index{accélération centripète}%
En \omterm{def:g12:kinematics-2d:ucm}{mouvement circulaire uniforme} de rayon $R$ et de
\omterm{def:g12:kinematics-2d:velocity}{vitesse} $v$, l'accélération --- dite \emph{centripète} ---
pointe du point mobile vers le centre du cercle, de norme constante
\[
a = \frac{v^2}{R}.
\]
\end{theorem}

\begin{proof}
Centrer les axes au centre du cercle et poser $\omega = v/R$, de sorte
que l'arc $v t$ sous-tend l'angle $\omega t$~: la position est
$x(t) = R\cos(\omega t)$, $y(t) = R\sin(\omega t)$. En dérivant,
$\vect v = (-R\omega\sin(\omega t),\, R\omega\cos(\omega t))$, de
norme constante $R\omega = v$~; en dérivant encore,
$\vect a = -\omega^2\,\vect{OM}$~: opposé au \omterm{def:g12:kinematics-2d:position}{vecteur position}
--- vers le centre --- de norme $\omega^2 R = v^2/R$.
\end{proof}

\begin{definition}[Repère de Frenet]\label{def:g12:kinematics-2d:frenet}
Le \emph{repère de Frenet}\index{repère de Frenet} à la position
actuelle d'un point sur un chemin courbe est le couple de vecteurs
\omterm{def:g10:orders-of-magnitude:unit}{unitaires} perpendiculaires $(\vect{u_t}, \vect{u_n})$~:
$\vect{u_t}$ tangent, le long du mouvement~; $\vect{u_n}$ normal, vers
le centre du cercle de rayon $R$ qui épouse le mieux la courbe là.
\end{definition}

\begin{proposition}[Accélération dans le repère de Frenet]\label{prop:g12:kinematics-2d:frenetacc}
Le long de tout chemin plan de rayon local $R$, parcouru à la
\omterm{def:g12:kinematics-2d:velocity}{vitesse} $v(t)$,
\[
\vect a = a_t\,\vect{u_t} + a_n\,\vect{u_n},
\qquad a_t = \frac{\dd v}{\dd t},
\qquad a_n = \frac{v^2}{R}:
\]
l'\emph{accélération tangentielle}\index{accélération tangentielle}
$a_t$ change la \omterm{def:g12:kinematics-2d:velocity}{vitesse}~; l'\emph{accélération
normale}\index{accélération normale} $a_n$ tourne la vitesse.
\end{proposition}
\admitted

\begin{remark}[Où vit la preuve]\label{rem:g12:kinematics-2d:frenetproof}
Nous avons prouvé les extrêmes~: $a_t$ pure sur une droite
(\cref{prop:g12:kinematics-2d:uarm}), $a_n$ pure sur un cercle
(\cref{thm:g12:kinematics-2d:centripetal})~; la décomposition générale
est établie dans le volume de Licence~1.
\end{remark}

\begin{example}[Freiner dans un virage]\label{ex:g12:kinematics-2d:bend}
Une voiture entre dans un virage de rayon $R = \qty{50}{m}$ à
$v = \qty{15}{m/s}$ en freinant à $\dd v/\dd t = \qty{-3.0}{m/s^2}$~:
$a_t = \qty{-3.0}{m/s^2}$, $a_n = 15^2/50 = \qty{4.5}{m/s^2}$, donc
$a = \sqrt{3.0^2 + 4.5^2} \approx \qty{5.4}{m/s^2}$, pointant vers
l'arrière \emph{et} vers l'intérieur du virage. Les pneus doivent
fournir les deux à la fois --- pourquoi les pilotes freinent
\emph{avant} le virage.
\end{example}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:2}

Identifier le mouvement et calculer la \omterm{def:g12:kinematics-2d:velocity}{vitesse}. Que

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:5}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:6}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:7}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:8}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:9}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:12}

\qty{86164}{s}. Calculer sa \omterm{def:g12:kinematics-2d:velocity}{vitesse} et son
\omterm{thm:g12:kinematics-2d:centripetal}{accélération centripète}. Qui la fournit, et pourquoi le
\omterm{def:g10:universal-gravitation:satellite}{satellite} reste-t-il au-dessus d'un point équatorial fixe
(\cref{ch:g12:satellites-kepler})~?
\end{exercise}

\section{Problème~: Le rapport d'accident}

\begin{problem}[{Problème du week-end --- le rapport d'accident~: un film
de dashcam lu image par image, une loi de freinage reconstruite à partir
des traces de freinage, un rond-point qui témoigne de l'adhérence, et un
verdict en kilomètres par heure}]
\label{pb:g12:kinematics-2d:1}
Une voiture heurte une fourgonnette sortie d'une rue latérale~;
personne n'est blessé, mais les assureurs ne sont pas d'accord.
L'enquêteur dispose d'images de caméra aérienne (une image toutes les
\qty{0.50}{s}), de traces de freinage mesurées, et de la dashcam de la
voiture. La limite de \omterm{def:g12:kinematics-2d:velocity}{vitesse} est \qty{50}{km/h}.

\medskip
\noindent\textbf{Partie I --- Lire le film.}
Les images avant toute réaction donnent, le long de la route droite~:
\begin{center}\small
\begin{tabular}{lccccc}
$t$ (\unit{s}) & 0.00 & 0.50 & 1.00 & 1.50 & 2.00\\
$x$ (\unit{m}) & 0.0 & 9.0 & 18.0 & 27.0 & 36.0
\end{tabular}
\end{center}
\begin{enumerate}
  \item Pourquoi le rapport doit-il d'abord nommer un référentiel et des
        axes~? Lesquels sont utilisés ici~?
  \item Calculer la vitesse moyenne sur chacun des quatre intervalles.
  \item Quelle classe de mouvement est-ce~? Justifier à partir du
        tableau.
  \item Donner la \omterm{def:g12:kinematics-2d:velocity}{vitesse} $v_0$ de la voiture en \unit{m/s} et
        \unit{km/h}.
  \item Quelle est l'accélération pendant ces deux secondes~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Les traces de freinage.}
Des tests de pneus sur cet asphalte sec donnent une décélération de
freinage constante $a = \qty{6.0}{m/s^2}$~; les traces mesurent
$d_b = \qty{27}{m}$. Prendre $t = 0$, $x = 0$ au début du freinage,
\omterm{def:g12:kinematics-2d:velocity}{vitesse} inconnue $v_B$.
\begin{enumerate}[resume]
  \item Par primitives, écrire $v(t)$ et $x(t)$ pendant le freinage.
  \item Exprimer le temps d'arrêt $t_s$~; montrer
        $d_b = v_B^2/(2a)$.
  \item En déduire $v_B$ à partir des traces. Cohérent avec la
        partie~I~?
  \item Le temps de réaction est \qty{1.0}{s}~: calculer la distance de
        réaction et la distance d'arrêt totale.
  \item Expliquer, à partir de la question~7, pourquoi doubler la
        \omterm{def:g12:kinematics-2d:velocity}{vitesse} multiplie par quatre la distance de freinage~;
        calculer $d_b$ à exactement \qty{50}{km/h}.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le rond-point témoigne.}
Plus tôt, la dashcam montre la voiture tournant dans une voie de
rond-point de rayon $R = \qty{12.5}{m}$ à \omterm{def:g12:kinematics-2d:velocity}{vitesse}
constante, un quart de tour en \qty{2.5}{s}. La route peut fournir au
plus $a_n = \mu g$, avec $\mu = 0.70$.
\begin{enumerate}[resume]
  \item La \omterm{def:g12:kinematics-2d:velocity}{vitesse} est constante là --- la vitesse vectorielle
        l'est-elle~? La voiture accélère-t-elle~? Expliquer.
  \item Calculer la \omterm{def:g12:kinematics-2d:velocity}{vitesse} de la voiture sur le rond-point.
  \item Calculer son \omterm{thm:g12:kinematics-2d:centripetal}{accélération centripète} (norme et
        direction).
  \item Calculer la \omterm{def:g12:kinematics-2d:velocity}{vitesse} maximale sans dérapage. La voiture
        y était-elle~?
  \item Donner la \omterm{def:g12:kinematics-2d:ucm}{période} et la \omterm{def:g12:kinematics-2d:ucm}{fréquence}
        d'un tour complet à cette \omterm{def:g12:kinematics-2d:velocity}{vitesse}.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le verdict.}
La fourgonnette est apparue $D = \qty{41}{m}$ devant la voiture à
l'instant où le conducteur a commencé à réagir.
\begin{enumerate}[resume]
  \item De combien, en \unit{km/h} et en pour cent, la
        \omterm{def:g12:kinematics-2d:velocity}{vitesse} avant freinage dépassait-elle la limite~?
  \item Calculer la \omterm{def:g12:kinematics-2d:velocity}{vitesse} de la voiture à l'impact.
  \item Refaire le calcul d'arrêt à exactement \qty{50}{km/h} (même
        réaction, même freinage)~: la voiture s'arrête-t-elle, avec
        quelle marge~?
  \item Donner la formule générale de distance d'arrêt $d(v)$~;
        pourquoi la \omterm{def:g12:kinematics-2d:velocity}{vitesse} entre-t-elle deux fois --- une fois
        linéairement, une fois au carré~?
  \item Le verdict, une phrase avec ses nombres~: à quelle
        \omterm{def:g12:kinematics-2d:velocity}{vitesse} roulait le conducteur, et l'accident se
        serait-il produit à la limite~?
\end{enumerate}
\end{problem}
