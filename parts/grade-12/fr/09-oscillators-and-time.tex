\chapter{Oscillateurs mécaniques et mesure du temps}\label{ch:g12:oscillators-and-time}

Une horloge comtoise tic-tac~: chaque tic est un balancier de laiton
qui traverse la verticale, répété quinze millions de fois depuis le
Nouvel An. Un lambeau de quartz dans votre montre fléchit
\num{32768}~fois par seconde~; un \omterm{def:g11:fundamental-interactions:ladder}{atome} de césium
vibre neuf milliards de fois plus vite. Tout ce qui oscille peut
tenir le temps~: ce chapitre suit le balancement, de Huygens aux
\omterm{def:g10:universal-gravitation:satellite}{satellites}.

\section{Oscillations libres}

\begin{definition}[Oscillateur, période, fréquence, amplitude]\label{def:g12:oscillators-and-time:oscillator}
Un \emph{oscillateur mécanique}\index{oscillateur} est un système qui,
écarté d'une \emph{position d'équilibre}\index{équilibre!position
d'} stable et lâché, oscille de part et d'autre autour d'elle~: ce
sont des \emph{oscillations libres}\index{oscillations libres}. Le
mouvement se répète après la \emph{période}\index{période} $T_0$
(\unit{s})~; la \emph{fréquence}\index{fréquence} $f_0 = 1/T_0$
(\unit{Hz}, \cref{ch:g12:mechanical-waves})~; l'\emph{amplitude}%
\index{amplitude} est l'écart maximal.
\end{definition}

\begin{example}[Un zoo d'oscillateurs]\label{ex:g12:oscillators-and-time:zoo}
Une balançoire de parc~: $T_0 \approx \qty{3}{s}$. La corde de la
d'une guitare~: $f_0 = \qty{110}{Hz}$. Un cristal de quartz~:
\qty{32768}{Hz}. L'oscillation du césium qui définit la seconde~:
environ \qty{9.2e9}{Hz}. Dix ordres de grandeur, une seule tâche~:
compter.
\end{example}

\section{Le pendule simple}

\begin{definition}[Pendule simple]\label{def:g12:oscillators-and-time:pendulum}
Un \emph{pendule simple}\index{pendule simple}~: une petite masse $m$
au bout d'un fil inextensible de longueur $L$ et de masse négligeable,
qui oscille dans un plan vertical. Des deux \omterm{def:g10:inertia:force}{forces} sur la masse
(\cref{ch:g12:newtons-laws}), la \omterm{def:g10:inertia:inventory}{tension} est perpendiculaire au
mouvement, tandis que la composante du \omterm{def:g10:universal-gravitation:weight}{poids} le long de la
\omterm{def:g10:relative-motion:trajectory}{trajectoire} pointe vers le point le plus bas~: le
\omterm{def:g10:universal-gravitation:weight}{poids} est la \emph{\omterm{def:g10:inertia:force}{force} de rappel}%
\index{force de rappel}, opposée au déplacement.
\end{definition}

\begin{proposition}[Période du pendule simple]\label{prop:g12:oscillators-and-time:pendulum-period}
\index{pendule simple!période}
Pour de petites \omterm{def:g12:oscillators-and-time:oscillator}{amplitudes} (en dessous d'environ $15^\circ$), la
\omterm{def:g12:oscillators-and-time:oscillator}{période} est
\[
T_0 = 2\pi \sqrt{\frac{L}{g}}, \qquad g = \qty{9.81}{m/s^2}.
\]
Elle ne dépend ni de la masse de la bille ni --- tant que l'angle
reste petit --- de l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}~: les petites oscillations sont
\emph{isochrones}\index{isochronisme}, toutes de même durée.
\end{proposition}
\admitted

\begin{remark}[Analyse dimensionnelle, et où vit le $2\pi$]\label{rem:g12:oscillators-and-time:dimensional}
La formule se devine. À partir de $m$ (\unit{kg}), $L$ (\unit{m}) et
$g$ (\unit{m/s^2}), la seule combinaison ayant l'\omterm{def:g10:orders-of-magnitude:unit}{unité} d'un temps est
$\sqrt{L/g}$ --- et la masse \emph{ne peut pas} entrer, aucune autre
donnée ne portant un \omterm{def:g10:orders-of-magnitude:unit}{kilogramme} pour l'annuler. Les dimensions ne
donnent pas le nombre pur devant~: le $2\pi$ vient de la résolution de
l'équation du mouvement, faite dans le volume de Licence~1. La
condition de petit angle n'est pas optionnelle non plus~: à $30^\circ$
la vraie \omterm{def:g12:oscillators-and-time:oscillator}{période} est plus longue de $2\%$, et l'isochronisme
tombe.
\end{remark}

\begin{example}[Le pendule à secondes]\label{ex:g12:oscillators-and-time:seconds}
Quelle longueur bat la seconde, $T_0 = \qty{2.00}{s}$, un tic dans
chaque sens~? En résolvant, $L = g\,T_0^2/4\pi^2 = 9.81 \times
2.00^2/4\pi^2 \approx \qty{0.994}{m}$. Presque un mètre~: les
horloges à pendule sont hautes parce que les secondes sont longues.
\end{example}

\section{L'oscillateur masse--ressort}

\begin{definition}[Force de rappel d'un ressort]\label{def:g12:oscillators-and-time:spring}
Un mobile de masse $m$ glisse sur un rail horizontal sans frottement,
attaché à un ressort de \emph{raideur}\index{raideur} $k$
(\unit{N/m}). Soit $x$ son déplacement par rapport à l'équilibre.
Étiré ou comprimé, le ressort tire ou pousse le mobile en arrière
avec la \omterm{def:g10:inertia:force}{force} $F = -kx$~: proportionnelle au déplacement et
opposée à celui-ci --- une \omterm{def:g12:oscillators-and-time:pendulum}{force de rappel} encore, fournie
cette fois par l'élasticité.
\end{definition}

\begin{proposition}[Équation du mouvement et sa solution]\label{prop:g12:oscillators-and-time:spring-period}
\index{oscillateur masse--ressort}
La \omterm{thm:g12:newtons-laws:second}{deuxième loi de Newton} le long du rail
(\cref{ch:g12:newtons-laws}) donne $m \frac{d^2x}{dt^2} = -kx$,
c'est-à-dire $\frac{d^2x}{dt^2} = -\frac{k}{m}\,x$. Quelles que
soient les constantes $A$ et $\varphi$,
\[
x(t) = A \cos\!\left(\frac{2\pi t}{T_0} + \varphi\right),
\qquad T_0 = 2\pi\sqrt{\frac{m}{k}},
\]
satisfait cette équation~: oscillation de \omterm{def:g12:oscillators-and-time:oscillator}{période} $T_0$,
indépendante de l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}.
\end{proposition}

\begin{proof}
En dérivant, $\frac{dx}{dt} = -\frac{2\pi}{T_0}A \sin(\frac{2\pi
t}{T_0} + \varphi)$~; encore une fois, $\frac{d^2x}{dt^2} =
-(\frac{2\pi}{T_0})^2 A\cos(\frac{2\pi t}{T_0} + \varphi) =
-(\frac{2\pi}{T_0})^2\,x(t)$, ce qui est $-(k/m)\,x$ exactement
lorsque $(2\pi/T_0)^2 = k/m$.
\end{proof}

\begin{remark}[Tous les mouvements qui existent]\label{rem:g12:oscillators-and-time:uniqueness}
Que \emph{tout} mouvement du mobile soit l'une de ces fonctions est
admis, prouvé dans le volume de Licence~1. Et pas de $g$ dans
$2\pi\sqrt{m/k}$~: plus raide, plus vite~; plus lourd, plus lent~;
pareil sur la Lune.
\end{remark}

\begin{definition}[Amplitude et phase]\label{def:g12:oscillators-and-time:phase}
Dans $x(t) = A\cos(2\pi t/T_0 + \varphi)$ la constante positive $A$
est l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} --- les extrêmes sont $x = \pm A$ --- et
$2\pi t/T_0 + \varphi$ est la \emph{phase}\index{phase}, qui croît de
$2\pi$ à chaque \omterm{def:g12:oscillators-and-time:oscillator}{période}~; la \emph{phase initiale}%
\index{phase initiale} $\varphi$ enregistre le départ~: lâché au
repos depuis $x = A$, $\varphi = 0$. Par exemple $m = \qty{0.20}{kg}$
sur $k = \qty{20}{N/m}$ donne $T_0 = 2\pi\sqrt{0.20/20} =
\qty{0.63}{s}$~; tiré à \qty{5.0}{cm} et lâché, $x(t) =
5.0\cos(2\pi t/0.63)$ en centimètres.
\end{definition}

\section{Énergie, amortissement, résonance}

\begin{proposition}[Énergie de l'oscillateur]\label{prop:g12:oscillators-and-time:energy}
Un ressort étiré de $x$ stocke l'\omterm{def:g10:energy-conservation:energy}{énergie} potentielle élastique
$E_p = \tfrac12 kx^2$ (justifiée au \cref{ch:g12:work-and-energy}).
Pendant l'oscillation sans frottement, l'\omterm{def:g11:mechanical-energy:em}{énergie mécanique}
\[
E_m = E_k + E_p = \tfrac12 m v^2 + \tfrac12 k x^2 = \tfrac12 k A^2
\]
est constante~: toute élastique aux extrêmes, toute cinétique à
l'équilibre. Le pendule joue le même duo entre $E_k$ et $E_p$
gravitationnelle (\cref{ch:g11:mechanical-energy}).
\end{proposition}

\begin{proof}
Avec $v = \frac{dx}{dt}$, la dérivée de $E_m$ est $mv\frac{dv}{dt}
+ kx\frac{dx}{dt} = v\,(m\frac{d^2x}{dt^2} + kx) = 0$ par l'équation
du mouvement~; la valeur $\tfrac12 kA^2$ se lit à un extrême.
\end{proof}

\begin{definition}[Oscillations amorties]\label{def:g12:oscillators-and-time:damping}
Le \omterm{def:g10:inertia:inventory}{frottement} draine l'\omterm{def:g11:mechanical-energy:em}{énergie mécanique} de tout
\omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} réel~: ses \omterm{def:g12:oscillators-and-time:oscillator}{oscillations libres} sont
\emph{amorties}\index{amortissement}. Sous un amortissement léger le
système oscille encore, avec une \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} qui meurt mais un
temps de reprise presque inchangé, la \emph{\omterm{def:g12:oscillators-and-time:oscillator}{pseudo-période}}%
\index{pseudo-période}, proche de $T_0$~; sous un fort amortissement
(du miel) il rampe vers l'équilibre sans jamais le dépasser~;
l'\omterm{def:g10:energy-conservation:energy}{énergie}, proportionnelle au carré de l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude},
se dissipe en chaleur.
\end{definition}

\begin{definition}[Oscillations forcées et résonance]\label{def:g12:oscillators-and-time:resonance}
Poussez un \omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} \omterm{def:g10:signals-and-waves:period}{périodiquement} ---
\emph{oscillations forcées}\index{oscillations forcées} --- et après
un bref régime transitoire il oscille à la \omterm{def:g12:oscillators-and-time:oscillator}{fréquence}
\emph{d'excitation} $f$, pas à la sienne. Son
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} culmine lorsque $f$ se rapproche de la
\emph{\omterm{def:g12:oscillators-and-time:oscillator}{fréquence} propre}\index{fréquence propre} $f_0$ de
l'\omterm{def:g12:oscillators-and-time:oscillator}{oscillateur} libre~: c'est la
\emph{résonance}\index{résonance}, chaque poussée arrivant en phase
avec le mouvement, injectant de l'\omterm{def:g10:energy-conservation:energy}{énergie} à chaque cycle~; plus
l'\omterm{def:g12:oscillators-and-time:damping}{amortissement} est léger, plus le pic est haut et pointu.
\end{definition}

\begin{example}[La résonance pour le bien et pour le mal]\label{ex:g12:oscillators-and-time:resonance-zoo}
Un parent qui pousse une balançoire pousse à sa propre
\omterm{def:g12:oscillators-and-time:oscillator}{fréquence} --- la \omterm{def:g12:oscillators-and-time:resonance}{résonance} mise au
\omterm{def:g11:work-of-force:work}{travail}. En \num{2000}, le nouveau Millennium Bridge à Londres
a dangereusement oscillé lorsque les pas de la foule se sont calés
sur une \omterm{def:g12:oscillators-and-time:resonance}{fréquence propre} près de \qty{1}{Hz}~; il a fermé
deux ans pour des amortisseurs. Toute montre à quartz fait chanter
son cristal en l'excitant à la \omterm{def:g12:oscillators-and-time:resonance}{résonance},
\qty{32768}{Hz}.
\end{example}

\section{Mesurer le temps}

\begin{method}[Chronométrer une période]\label{met:g12:oscillators-and-time:timing}
Ne chronométrez jamais une seule oscillation~: démarrez le
chronomètre au moment où la masse traverse la verticale (le plus
vite, le plus facile à juger), comptez $N = 50$ oscillations,
divisez par $N$ --- une erreur de réaction de \qty{0.2}{s} devient
\qty{4}{ms} sur $T_0$. Répétez, et comparez les séries.
\end{method}

\begin{example}[Trois siècles de tic-tac]\label{ex:g12:oscillators-and-time:clocks}
Huygens a construit la première horloge à pendule en \num{1657}~:
l'isochronisme a fait passer les horloges d'un quart d'heure de
dérive par jour à des secondes, et les pendules compensés à des
fractions de seconde par jour. La montre à quartz (\num{1969}) a
échangé la tige contre un cristal fléchissant à \qty{32768}{Hz}~:
des dixièmes de seconde par mois. Depuis \num{1967} la seconde
elle-même est \emph{définie} par une oscillation atomique~: la durée
de \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{périodes} du rayonnement micro-onde d'une
transition interne du césium-133 (\cref{ch:g12:quantum-world}). Les
meilleures horloges à césium dérivent d'une seconde en cent millions
d'années.
\end{example}

\begin{remark}[Pourquoi chasser la nanoseconde]\label{rem:g12:oscillators-and-time:gps}
Un récepteur GPS trouve sa position en chronométrant des
\omterm{def:g10:signals-and-waves:signal}{signaux} radio issus de
\omterm{def:g10:universal-gravitation:satellite}{satellites}~; la lumière parcourt \qty{30}{cm} par
nanoseconde, donc une erreur d'horloge d'une microseconde vous
déplace de \qty{300}{m}. Les \omterm{def:g10:universal-gravitation:satellite}{satellites} portent donc des
horloges atomiques --- et à cette précision le rythme d'une horloge
dépend de sa vitesse et de sa hauteur, un effet que l'on rencontre
au \cref{ch:g12:special-relativity}.
\end{remark}

\section{Problème~: L'établi de l'horloger}

\begin{problem}[{Problème du week-end --- l'établi de l'horloger~: une
horloge murale remise à l'heure --- cinquante oscillations chronométrées,
une tige de laiton qui s'allonge en été, la poussée chuchotée de
l'échappement, un concurrent à quartz --- le verdict rendu en secondes
par mois}]
\label{pb:g12:oscillators-and-time:1}
Une horloge murale à pendule arrive à l'établi «~qui ne marche pas~».
Traiter sa tige de laiton et sa lourde masse comme un
\omterm{def:g12:oscillators-and-time:pendulum}{pendule simple} de longueur effective $L$, masse de
la bille $m = \qty{1.2}{kg}$~; l'engrenage compte chaque oscillation
complète comme exactement \qty{2.000}{s}.

\medskip
\noindent\textbf{Partie I --- Prendre le pouls de l'horloge.}
\begin{enumerate}
  \item Pourquoi chronométrer $50$~oscillations plutôt qu'une~?
        Estimer l'erreur restante sur $T_0$ si votre temps de réaction
        est \qty{0.2}{s}.
  \item Le chronomètre indique \qty{99.4}{s} pour $50$~oscillations~:
        la \omterm{def:g12:oscillators-and-time:oscillator}{période}~?
  \item En déduire la longueur effective actuelle du pendule.
  \item Quelle longueur donnerait exactement \qty{2.000}{s}~? L'écrou
        de réglage doit-il monter ou descendre la masse, et de
        combien~?
  \item Sans réglage, l'horloge avance-t-elle ou retarde-t-elle, de
        combien de secondes par jour~?
  \item Sensibilité à $g$~: la \omterm{def:g12:oscillators-and-time:oscillator}{période} sur la Lune
        ($g = \qty{1.62}{m/s^2}$)~? L'erreur quotidienne dans un
        observatoire où $g = \qty{9.80}{m/s^2}$~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- L'affaissement d'été.} Le laiton se
dilate~: un réchauffement $\Delta\theta$ allonge la tige en
$L' = L\,(1 + \lambda\, \Delta\theta)$, $\lambda = \num{1.9e-5}$ par
\omterm{def:g10:light-spectra:kelvin}{kelvin}. L'horloge est réglée à \qty{20}{\celsius}.
\begin{enumerate}[resume]
  \item Calculer $\Delta L$ pour une pièce d'été à \qty{25}{\celsius}.
  \item En utilisant $\sqrt{1+x} \approx 1 + x/2$, montrer que
        $\Delta T_0 / T_0 = \tfrac12 \lambda\, \Delta\theta$.
  \item Calculer $\Delta T_0/T_0$ et l'erreur quotidienne d'été~:
        avance ou retard~?
  \item Mêmes questions pour un couloir d'hiver à \qty{10}{\celsius}.
  \item Les vieilles horloges de précision utilisaient des pendules
        «~grille-pain~» mélangeant des tiges de laiton et d'acier
        (qui se dilatent environ deux fois moins) en directions
        alternées. Expliquer comment cela peut tenir le rythme.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La poussée chuchotée.} Avec
l'échappement désengagé, le balancement, démarré à l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}
$\theta_m = 2.00^\circ$, divise par deux son \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}
en \qty{280}{s}~; en service, une petite poussée de l'échappement à
chaque \omterm{def:g12:oscillators-and-time:oscillator}{période} maintient $\theta_m$ constant.
\begin{enumerate}[resume]
  \item Combien de \omterm{def:g12:oscillators-and-time:oscillator}{périodes} dure le demi-affaiblissement~?
  \item À l'extrême la masse a grimpé $h = L(1 - \cos\theta_m)
        \approx L\,\theta_m^2/2$ (radians)~: en déduire
        $E_m = \tfrac12 m g L\,\theta_m^2$ et la calculer
        (\cref{ch:g11:mechanical-energy}).
  \item L'\omterm{def:g10:energy-conservation:energy}{énergie} est proportionnelle au carré de
        l'\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}~: quelle fraction de $E_m$ est perdue
        par \omterm{def:g12:oscillators-and-time:oscillator}{période}~?
  \item En déduire l'\omterm{def:g10:energy-conservation:energy}{énergie} de chaque poussée, et la
        \omterm{def:g11:work-of-force:power}{puissance} moyenne nécessaire.
  \item Le \omterm{def:g10:universal-gravitation:weight}{poids} moteur $M = \qty{2.5}{kg}$ descend de
        $h = \qty{1.0}{m}$ par semaine~: est-ce assez
        d'\omterm{def:g10:energy-conservation:energy}{énergie}, à quel \omterm{def:g10:energy-conservation:efficiency}{rendement}~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le concurrent à quartz.} Un mouvement
à quartz oscille à $f = \qty{32768}{Hz}$ et dérive d'environ
\qty{0.3}{s} par mois~; le pendule restauré dérive encore de
\qty{0.5}{s} par jour.
\begin{enumerate}[resume]
  \item Calculer la \omterm{def:g12:oscillators-and-time:oscillator}{période} du quartz. Montrer
        $\num{32768} = 2^{15}$~: pourquoi une \omterm{def:g11:work-of-force:power}{puissance}
        de deux est-elle exactement ce qu'un circuit de montre veut~?
  \item Calculer les dérives relatives du quartz et du pendule
        restauré, et chacune en secondes par mois. Qui gagne, par
        quel facteur~?
  \item L'étalon de césium --- \num{9192631770}
        \omterm{def:g12:oscillators-and-time:oscillator}{périodes} font une seconde --- dérive de
        \qty{1}{s} par cent millions d'années~: sa dérive relative~?
        Combien d'ordres de grandeur sous le quartz, et pourquoi le
        GPS l'exige-t-il~?
  \item Le verdict, une phrase avec des nombres~: que gagne l'horloge
        restaurée sur le mur, et à qui appartient désormais la
        seconde~?
\end{enumerate}
\end{problem}
