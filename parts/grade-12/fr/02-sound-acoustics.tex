\chapter{Son et acoustique}\label{ch:g12:sound-acoustics}

Pincez la corde grave d'une guitare~: la pièce se remplit d'un la
grave, pourtant rien n'a voyagé jusqu'à votre oreille que de l'air
comprimé et dilaté. Ce chapitre suit ce motif --- sa célérité, comment
la \omterm{def:g10:signals-and-waves:frequency}{fréquence} devient \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} et la \omterm{def:g11:work-of-force:power}{puissance}
\omterm{def:g12:sound-acoustics:pitch-loudness}{intensité sonore}, et pourquoi une même note diffère d'un instrument
à l'autre.

\section{Le son est une onde de pression}

\begin{definition}[Onde sonore]\label{def:g12:sound-acoustics:sound-wave}
Une \emph{onde sonore}\index{onde sonore} est une \omterm{def:g10:signals-and-waves:wave}{onde} mécanique de
\omterm{def:g10:pressure:pressure}{pression} (\cref{ch:g12:mechanical-waves})~: une surface vibrante
crée des \emph{compressions}\index{compression} qui voyagent
(légère surpression) et des \emph{dilatations}\index{dilatation}
(sous-pression). Les parcelles d'air oscillent \emph{le long} de la
direction de parcours (le \omterm{def:g10:signals-and-waves:sound}{son} est \emph{longitudinal}\index{onde
longitudinale}), et un milieu est nécessaire~: une cloche sous une
cloche à vide tombe dans le silence
(\cref{ch:g10:signals-and-waves}).
\end{definition}

\begin{proposition}[Célérité du son]\label{prop:g12:sound-acoustics:speed}
\index{célérité du son}%
Le \omterm{def:g10:signals-and-waves:sound}{son} se propage à environ \qty{340}{m/s} dans l'air à
\qty{20}{\degreeCelsius}, \qty{1500}{m/s} dans l'eau et
\qty{5000}{m/s} dans l'acier~: plus le milieu est rigide, plus c'est
rapide. \admitted
\end{proposition}

\begin{remark}[L'air chaud est plus rapide]\label{rem:g12:sound-acoustics:temperature}
Ce sont des valeurs mesurées~; les prédire se fait dans le volume de
Licence~1. Le \omterm{def:g10:signals-and-waves:sound}{son} accélère dans l'air plus chaud (\qty{331}{m/s} à
\qty{0}{\degreeCelsius}) --- pourquoi les instruments dérivent du
diapason à mesure que la salle se réchauffe.
\end{remark}

\begin{example}[Les tailles des sons]\label{ex:g12:sound-acoustics:wavelengths}
La relation $\lambda = v/f$ dimensionne tout \omterm{def:g10:signals-and-waves:sound}{son}~: dans l'air un
grondement à \qty{20}{Hz} s'étend sur $340/20 = \qty{17}{m}$ --- un
bâtiment --- et un sifflement à \qty{20}{kHz} tient dans \qty{1.7}{cm}~;
ceux de taille métrique contournent facilement portes et angles.
\end{example}

\section{Hauteur, intensité sonore et l'oreille}

\begin{definition}[Hauteur et intensité sonore]\label{def:g12:sound-acoustics:pitch-loudness}
La \emph{hauteur}\index{hauteur} est la \omterm{def:g11:color-light-sources:perception}{perception} de la
\omterm{def:g10:signals-and-waves:frequency}{fréquence}~: plus $f$ est élevée, plus la note est aiguë~; le
la du concert est à \qty{440}{Hz}, et doubler $f$ élève la note d'une
octave. L'\emph{intensité sonore}\index{intensité sonore} est la
\omterm{def:g11:color-light-sources:perception}{perception} de l'\omterm{def:g10:signals-and-waves:amplitude}{amplitude}~: plus l'oscillation de
\omterm{def:g10:pressure:pressure}{pression} est forte, plus c'est fort~; l'amplification change
l'intensité sonore, non la hauteur.
\end{definition}

\begin{remark}[La fenêtre de l'oreille]\label{rem:g12:sound-acoustics:audible}
L'audition s'étend d'environ \qty{20}{Hz} à \qty{20}{kHz}
(\cref{ch:g10:signals-and-waves})~; en dessous se trouve
l'\omterm{def:g10:signals-and-waves:ultrasound}{infrason}, au-dessus l'\omterm{def:g10:signals-and-waves:ultrasound}{ultrason} --- chauves-souris à
\qty{50}{kHz}, sondes médicales en mégahertz. Un piano n'échantillonne
que le milieu~: \qtyrange{27.5}{4186}{Hz}.
\end{remark}

\section{Intensité acoustique et échelle des décibels}

\begin{definition}[Intensité acoustique]\label{def:g12:sound-acoustics:intensity}
L'\emph{intensité acoustique}\index{intensité acoustique} $I$ est la
\omterm{def:g11:work-of-force:power}{puissance} acoustique traversant un \omterm{def:g10:orders-of-magnitude:unit}{mètre} carré face à
l'\omterm{def:g10:signals-and-waves:wave}{onde}~: $I = P/S$, en \unit{W/m^2}. Le \omterm{def:g10:signals-and-waves:sound}{son} audible le plus
faible est d'environ $I_0 = \qty{1.0e-12}{W/m^2}$~; la douleur commence
près de \qty{1}{W/m^2}.
\end{definition}

\begin{proposition}[Atténuation en inverse du carré]\label{prop:g12:sound-acoustics:inverse-square}
\index{loi en inverse du carré}%
Une petite source rayonnant une \omterm{def:g11:work-of-force:power}{puissance} $P$ également dans
toutes les directions donne, à la distance $r$,
$I = P/(4\pi r^2) \propto 1/r^2$~: doubler la distance divise
l'intensité par quatre.
\end{proposition}

\begin{proof}
Toute la \omterm{def:g11:work-of-force:power}{puissance} $P$ traverse la sphère d'aire $4\pi r^2$~;
diviser.
\end{proof}

\begin{definition}[Niveau sonore]\label{def:g12:sound-acoustics:level}
L'oreille couvre douze \omterm{def:g11:work-of-force:power}{puissances} de dix, aussi les intensités
sont-elles rapportées logarithmiquement. Le \emph{niveau
sonore}\index{niveau sonore} de $I$ est
\[ L = 10 \log_{10}(I/I_0), \qquad I_0 = \qty{1.0e-12}{W/m^2}, \]
en \emph{décibels}\index{décibel} (\unit{dB})~: seuil d'audition
\qty{0}{dB}, douleur près de \qty{120}{dB}.
\end{definition}

\begin{proposition}[Arithmétique des décibels]\label{prop:g12:sound-acoustics:db-arithmetic}
Multiplier $I$ par $k$ ajoute $10\log_{10}k$ \omterm{def:g12:sound-acoustics:level}{décibels}. En
particulier $\times 10$ ajoute \qty{10}{dB}~; $\times 2$ ajoute
$10\log_{10}2 \approx \qty{3.0}{dB}$~; $N$ sources incohérentes
identiques additionnent les \emph{intensités}, jamais les niveaux,
donnant $L_1 + 10\log_{10}N$~; doubler la distance à une petite
source ($I$ divisée par quatre) retire environ \qty{6.0}{dB}.
\end{proposition}

\begin{proof}
$10\log_{10}(kI/I_0) = 10\log_{10}(I/I_0) + 10\log_{10}k$~;
distance~: \cref{prop:g12:sound-acoustics:inverse-square}, $k = 1/4$.
\end{proof}

\begin{method}[Travailler en décibels]\label{met:g12:sound-acoustics:db}
\begin{enumerate}
  \item $L = 10\log_{10}(I/I_0)$~; retour~: $I = I_0 \times 10^{L/10}$.
  \item Ne jamais additionner les niveaux~: convertir en intensités,
        additionner, reconvertir --- ou $+\qty{3}{dB}$ par doublement,
        $+\qty{10}{dB}$ par décuple.
  \item Chaque doublement de distance à une petite source coûte
        \qty{6}{dB}.
\end{enumerate}
\end{method}

\begin{remark}[Dommages auditifs]\label{rem:g12:sound-acoustics:damage}
La perte auditive est une \omterm{rem:g11:nucleus-radioactivity:dose}{dose}~: \qty{85}{dB} est sûr pendant
huit heures, et chaque \qty{3}{dB} de plus --- un doublement
d'intensité --- divise le temps sûr par deux~; les cellules ciliées
perdues ne repoussent pas.
\end{remark}

\section{Timbre et harmoniques}

\begin{definition}[Harmoniques]\label{def:g12:sound-acoustics:harmonics}
Un \omterm{def:g10:signals-and-waves:sound}{son} \omterm{def:g10:signals-and-waves:period}{périodique} de \omterm{def:g10:signals-and-waves:frequency}{fréquence}
$f_1$ --- le \emph{fondamental}\index{fondamental} --- est en
général une somme de sinus aux fréquences $f_n = n f_1$,
$n = 1, 2, 3, \dots$, ses \emph{harmoniques}\index{harmonique}. Un
\omterm{def:g10:signals-and-waves:sound}{son} réduit à son fondamental est un \emph{son pur}\index{son pur},
comme celui d'un diapason.
\end{definition}

\begin{definition}[Spectre et timbre]\label{def:g12:sound-acoustics:timbre}
Le \emph{spectre}\index{spectre} d'un \omterm{def:g10:signals-and-waves:sound}{son} est le diagramme en
barres de ses \omterm{def:g12:sound-acoustics:harmonics}{harmoniques}~: une barre par $f_n$, la hauteur
mesurant la force de cet \omterm{def:g12:sound-acoustics:harmonics}{harmonique}. L'oreille entend $f_1$
comme la \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} et le \emph{mélange} des forces comme le
\emph{timbre}\index{timbre} --- la couleur du \omterm{def:g10:signals-and-waves:sound}{son}~: des
instruments jouant la même note partagent $f_1$, ne différant que par
leurs spectres.
\end{definition}

\begin{example}[Pourquoi vous reconnaissez l'appelant]\label{ex:g12:sound-acoustics:two-instruments}
Le la d'une flûte est presque un \omterm{def:g12:sound-acoustics:harmonics}{son pur}~; un violon frottant le
même \qty{220}{Hz} ajoute de forts \omterm{def:g12:sound-acoustics:harmonics}{harmoniques} à
\qtylist{440;660;880}{Hz}. Même \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} et même
\omterm{def:g12:sound-acoustics:pitch-loudness}{intensité sonore}, pourtant personne ne les confond~: une voix,
elle aussi, est un \omterm{def:g12:sound-acoustics:timbre}{spectre} que l'on connaît par cœur.
\end{example}

\section{La corde vibrante}

\begin{definition}[Onde stationnaire]\label{def:g12:sound-acoustics:standing-wave}
Une \omterm{def:g10:signals-and-waves:wave}{onde} confinée entre les extrémités fixes d'une corde peut se
stabiliser en une \emph{onde stationnaire}\index{onde stationnaire}~:
une oscillation sur place, des points immobiles --- les
\emph{nœuds}\index{nœud} --- alternant avec des points d'amplitude
maximale --- les \emph{ventres}\index{ventre}. Plus rien ne voyage~;
la corde vibre selon un motif fixe.
\end{definition}

\begin{proposition}[Modes d'une corde]\label{prop:g12:sound-acoustics:string-modes}
Une corde de longueur $L$, fixe aux deux extrémités, portant des
\omterm{def:g10:signals-and-waves:wave}{ondes} de célérité $v$, ne vibre de façon stable que dans des
motifs logeant un nombre entier $n$ de \omterm{def:g12:mechanical-waves:wavelength}{demi-longueurs d'onde}
entre les \omterm{def:g12:sound-acoustics:standing-wave}{nœuds} imposés aux extrémités~:
\[ \lambda_n = \frac{2L}{n}, \quad f_n = n\,\frac{v}{2L} = n f_1, \quad n = 1, 2, 3, \dots \]
Le mode le plus bas $f_1 = v/(2L)$ est le \omterm{def:g12:sound-acoustics:harmonics}{fondamental}~; les
autres sont exactement ses \omterm{def:g12:sound-acoustics:harmonics}{harmoniques}. \admitted
\end{proposition}

\begin{remark}[Où vit la preuve honnête]\label{rem:g12:sound-acoustics:honest}
Que des \omterm{def:g10:signals-and-waves:wave}{ondes} confinées choisissent exactement ces modes suit de
la superposition de l'\omterm{def:g10:signals-and-waves:wave}{onde} avec ses réflexions, calcul mené dans
le volume de Licence~1. La géométrie convainc~: les deux extrémités
doivent rester immobiles, donc un nombre entier de
\omterm{def:g12:mechanical-waves:wavelength}{demi-longueurs d'onde} doit s'y loger.
\end{remark}

\begin{example}[La corde de la d'un violon]\label{ex:g12:sound-acoustics:violin}
La corde de la d'un violon a $L = \qty{32.8}{cm}$ et
\omterm{def:g10:signals-and-waves:sound}{sonne} $f_1 = \qty{440}{Hz}$~: les \omterm{def:g10:signals-and-waves:wave}{ondes} y courent à
$v = 2Lf_1 = 2 \times 0.328 \times 440 \approx \qty{289}{m/s}$ ---
fixée par la \omterm{def:g10:inertia:inventory}{tension} et la masse, accordée par la cheville.
Appuyer un doigt raccourcit $L$ et élève chaque $f_n$ d'un coup~: le
\omterm{def:g12:sound-acoustics:timbre}{spectre} glisse vers le haut.
\end{example}

\begin{remark}[Pourquoi les instruments diffèrent]\label{rem:g12:sound-acoustics:instruments}
La façon d'exciter la corde --- pincée, frottée, frappée --- fixe la
part de chaque mode, et la caisse amplifie certains
\omterm{def:g12:sound-acoustics:harmonics}{harmoniques} plus que d'autres~; excitation plus caisse fixent
le \omterm{def:g12:sound-acoustics:timbre}{spectre}, si bien qu'un même $f_1$ laisse une guitare, un
violon et un piano sonner chacun comme lui-même.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:1}
Dans l'air ($v = \qty{340}{m/s}$), trouver la
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} de (a) une note de basse à \qty{55}{Hz}~;
(b) un bip à \qty{1.0}{kHz}~; (c) un sifflement à \qty{15}{kHz}.
Comparer la plus grande à la plus petite.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:2}
Classer comme \omterm{def:g10:signals-and-waves:ultrasound}{infrason}, \omterm{def:g10:signals-and-waves:sound}{son} audible ou
\omterm{def:g10:signals-and-waves:ultrasound}{ultrason}~: \qty{8}{Hz}~; \qty{100}{Hz}~; \qty{15}{kHz}~;
\qty{40}{kHz}~; \qty{3}{MHz}. Parmi les audibles, lequel
\omterm{def:g10:signals-and-waves:sound}{sonne} plus bas en \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:3}
Un ouvrier frappe un rail à \qty{1.0}{km}. Calculer les temps de
parcours dans l'acier et dans l'air. Qu'entend un auditeur une oreille
pressée contre le rail~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:4}
(a) Le \omterm{def:g12:sound-acoustics:level}{niveau sonore} pour $I = \qty{1.0e-5}{W/m^2}$~? (b)
L'intensité à \qty{60}{dB}~? (c) Nommer un \omterm{def:g10:signals-and-waves:sound}{son} du quotidien près
de chaque niveau.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:5}
Une corde de sol de violon \omterm{def:g10:signals-and-waves:sound}{sonne} $f_1 = \qty{196}{Hz}$~: donner
ses trois prochains \omterm{def:g12:sound-acoustics:harmonics}{harmoniques}. Pourquoi un sifflement à
\qty{12}{kHz}, aussi riche soit-il à la source, parvient-il à
l'oreille comme un \omterm{def:g12:sound-acoustics:harmonics}{son pur}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:6}
Une sirène rayonne $P = \qty{0.50}{W}$ également dans toutes les
directions. Calculer l'intensité et le \omterm{def:g12:sound-acoustics:level}{niveau sonore} à
\qty{5.0}{m}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:7}
Un violon donne \qty{68}{dB} à votre place. Quel niveau en donnent
deux~? Quatre~? Combien de violons faudrait-il pour atteindre
\qty{78}{dB}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:8}
Le niveau est \qty{95}{dB} à \qty{2.0}{m} d'un petit haut-parleur.
Prédire le niveau à \qty{8.0}{m}~: sous le seuil de \qty{85}{dB}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:9}
Une note à \qty{440}{Hz} passe de l'air à l'eau. Calculer sa
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} dans chaque milieu. Le plongeur entend la
même \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur}~: quelle grandeur est fixée par la source, et
laquelle s'ajuste~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:10}
Un \omterm{def:g12:sound-acoustics:timbre}{spectre} montre des pics à \qtylist{130;260;390;520}{Hz} de
hauteurs décroissantes~: quel est le \omterm{def:g12:sound-acoustics:harmonics}{fondamental}~? Un autre
instrument joue la même note à la même
\omterm{def:g12:sound-acoustics:pitch-loudness}{intensité sonore}~: qu'y a-t-il de même dans son
\omterm{def:g12:sound-acoustics:timbre}{spectre}, qu'y a-t-il de différent~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:11}
Une corde de guitare de longueur \qty{65.0}{cm}
\omterm{def:g10:signals-and-waves:sound}{sonne} $f_1 = \qty{110}{Hz}$. Trouver la célérité de
l'\omterm{def:g10:signals-and-waves:wave}{onde}, puis les longueurs vibrantes donnant l'octave
(\qty{220}{Hz}) et la quinte (\qty{165}{Hz}).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:12}
Une sonde médicale envoie des \omterm{def:g10:signals-and-waves:ultrasound}{ultrasons} à \qty{5.0}{MHz} dans
des tissus mous ($v = \qty{1540}{m/s}$)~: la
\omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde}~? Des \omterm{rem:g10:signals-and-waves:honest}{échos} reviennent après
\qty{26}{\micro s} et \qty{58}{\micro s} des deux parois d'un organe~:
trouver leurs profondeurs et l'épaisseur de l'organe.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:13}
Une scène en plein air produit \qty{110}{dB} à \qty{3.0}{m}. En la
traitant comme une petite source, à quelle distance le niveau tombe-t-il
à \qty{85}{dB}~? Donner deux raisons pour lesquelles la distance réelle
diffère.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:14}
Trouver le rapport du seuil de douleur (\qty{1}{W/m^2}) à $I_0$. En
prenant le tympan comme \qty{0.5}{cm^2}, calculer la

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sound-acoustics:15}
Chaque haut-parleur de festival, seul, donne \qty{80}{dB} à la

\section{Problème~: L'atelier du luthier}

\begin{problem}[{Problème du week-end --- l'atelier du luthier~: une
corde de guitare disposée case par case, ses harmoniques accordées en
un timbre, un récital tenu sous le seuil de danger, et une chasse aux
ultrasons pour un défaut dans le bois}]
\label{pb:g12:sound-acoustics:1}
Un luthier termine une guitare. Sa corde de la a pour longueur
vibrante $L = \qty{65.0}{cm}$ et doit \omterm{def:g10:signals-and-waves:sound}{sonner}
$f_1 = \qty{110}{Hz}$. Prendre \qty{340}{m/s} pour le
\omterm{def:g10:signals-and-waves:sound}{son} dans l'air, et admettre $f_1 = v/(2L)$
(\cref{prop:g12:sound-acoustics:string-modes}), $v$ étant la célérité
de l'\omterm{def:g10:signals-and-waves:wave}{onde} sur la corde.

\medskip
\noindent\textbf{Partie I --- Disposer la touche.}
\begin{enumerate}
  \item Calculer la célérité $v$ de l'\omterm{def:g10:signals-and-waves:wave}{onde} sur la corde.
  \item Calculer la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} du
        \omterm{def:g12:sound-acoustics:harmonics}{fondamental} \emph{sur la corde}, puis la
        \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} du \omterm{def:g10:signals-and-waves:sound}{son} à \qty{110}{Hz}
        \emph{dans l'air}. Pourquoi diffèrent-elles à la même
        \omterm{def:g10:signals-and-waves:frequency}{fréquence}~?
  \item L'octave (\qty{220}{Hz})~: quelle longueur vibrante la donne~?
        Où doit se placer la douzième case, en fraction de $L$~?
  \item La quinte (\qty{165}{Hz})~: quelle longueur vibrante la donne~?
  \item Chaque case élève la note d'un demi-ton, un facteur
        $2^{1/12}$. Vérifier par logarithmes combien de demi-tons
        séparent \qty{110}{Hz} de \qty{220}{Hz}~; longueur à la première
        case~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- \omterm{def:g12:sound-acoustics:harmonics}{Harmoniques} et \omterm{def:g12:sound-acoustics:timbre}{timbre}.}
\begin{enumerate}[resume]
  \item Lister les cinq premiers \omterm{def:g12:sound-acoustics:harmonics}{harmoniques} de la corde de la à
        vide.
  \item Le la d'un violoniste \omterm{def:g10:signals-and-waves:sound}{sonne} \qty{440}{Hz}. Quel
        \omterm{def:g12:sound-acoustics:harmonics}{harmonique} de la corde de guitare est-ce, et quel
        intervalle musical le sépare du \omterm{def:g12:sound-acoustics:harmonics}{fondamental}~? Et
        \qty{330}{Hz}~?
  \item Le luthier pince près du chevalet, puis au-dessus de la
        touche~: même \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur}, couleur différente. Expliquer
        avec les spectres.
  \item Un léger toucher au milieu en pinçant \omterm{def:g10:inertia:force}{force} un
        \omterm{def:g12:sound-acoustics:standing-wave}{nœud} là. Quels \omterm{def:g12:sound-acoustics:harmonics}{harmoniques} survivent, et
        quelle \omterm{def:g12:sound-acoustics:pitch-loudness}{hauteur} entend-on~?
  \item Combien d'\omterm{def:g12:sound-acoustics:harmonics}{harmoniques} de la corde à vide se trouvent,
        en principe, dans le \omterm{def:g11:fundamental-interactions:strong}{domaine} audible~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le récital.}
Au premier concert de la guitare, chaque instrument rayonne une
\omterm{def:g11:work-of-force:power}{puissance} acoustique $P = \qty{6.0e-3}{W}$ comme une petite
source, également dans toutes les directions.
\begin{enumerate}[resume]
  \item Calculer l'intensité en $r = \qty{3.0}{m}$ d'un instrument.
  \item En déduire le \omterm{def:g12:sound-acoustics:level}{niveau sonore} là.
  \item Un quatuor joue~: quel niveau quatre tels instruments donnent-ils
        au même endroit~? Pourquoi n'est-ce pas quatre fois le niveau~?
  \item Douze instruments jouent~: quel est le niveau à \qty{3.0}{m}~?
  \item À quelle distance un auditeur doit-il s'asseoir pour que
        l'ensemble tombe à \qty{85}{dB}~?
  \item Règle de sécurité~: \qty{85}{dB} est sûr pendant huit heures, et
        chaque \qty{+3}{dB} divise le temps sûr par deux. Combien de
        temps l'auditeur du premier rang de la question~14 peut-il
        rester sans danger~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le défaut dans le bois.}
Avant le vernissage, le luthier sonde une ébauche de manche d'épaisseur
\qty{45}{mm} avec un testeur à \qty{2.0}{MHz}~; dans ce bois prendre
$v = \qty{5.0e3}{m/s}$.
\begin{enumerate}[resume]
  \item Calculer la \omterm{def:g12:mechanical-waves:wavelength}{longueur d'onde} des
        \omterm{def:g10:signals-and-waves:ultrasound}{ultrasons} dans le bois. Pourquoi une
        \omterm{def:g10:signals-and-waves:wave}{onde} chassant les défauts doit-elle être ultrasonore plutôt
        qu'audible~?
  \item Calculer le \omterm{def:g12:mechanical-waves:delay}{retard} de l'\omterm{rem:g10:signals-and-waves:honest}{écho} depuis la face
        arrière de l'ébauche.
  \item Un \omterm{rem:g10:signals-and-waves:honest}{écho} arrive après \qty{7.2}{\micro s}~: à quelle
        profondeur est le défaut~?
  \item Rédiger la carte d'atelier~: célérité de l'\omterm{def:g10:signals-and-waves:wave}{onde} sur la
        corde, longueurs aux douzième et première cases, distance
        d'écoute sûre, profondeur du défaut.
\end{enumerate}
\end{problem}
