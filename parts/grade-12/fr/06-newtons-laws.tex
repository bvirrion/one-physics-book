\chapter{Les lois de Newton}\label{ch:g12:newtons-laws}

Faites passer une balance de salle de bain dans un ascenseur. Quand la
cabine s'élance vers le haut, l'aiguille lit trois
\omterm{def:g10:orders-of-magnitude:unit}{kilogrammes} de trop~; près du sommet elle lit trop peu~;
entre les deux, la simple vérité --- rien à bord n'a changé sauf le
mouvement. Ce chapitre énonce les trois lois derrière cette aiguille et
les forge en la méthode qui mène toute la mécanique~: choisir un
système, dessiner ses \omterm{def:g10:inertia:force}{forces}, projeter
$\sum \vect F = m\vect a$.

\section{Quantité de mouvement, et la première loi rendue exacte}

\begin{definition}[Quantité de mouvement]\label{def:g12:newtons-laws:momentum}
La \emph{quantité de mouvement}\index{quantité de mouvement} d'un corps
de masse $m$ (\unit{kg}) se déplaçant à la vitesse $\vect v$ est le
vecteur
\[
\vect p = m \vect v \qquad (\unit{kg.m/s}):
\]
la vitesse, pondérée par la masse~; pour un système, la somme vectorielle
sur ses parties.
\end{definition}

\begin{definition}[Référentiel galiléen]\label{def:g12:newtons-laws:inertial}
Un \emph{référentiel galiléen}\index{référentiel galiléen} est un
référentiel dans lequel tout corps isolé --- sans
\omterm{def:g10:inertia:force}{forces}, ou des \omterm{def:g10:inertia:force}{forces} qui
\omterm{def:g10:inertia:compensate}{se compensent} --- se déplace en ligne droite à vitesse
constante (\cref{ch:g10:inertia}). Le sol en est un à excellente
précision, et toute cabine en mouvement rectiligne uniforme dessus~; un
train qui freine ou un manège qui tourne ne l'est pas~: là, des bagages
lâches accélèrent sans être poussés.
\end{definition}

\begin{theorem}[Première loi de Newton]\label{thm:g12:newtons-laws:first}\index{première loi de Newton}
Dans un référentiel galiléen, la \omterm{def:g12:newtons-laws:momentum}{quantité de mouvement} d'un
système isolé ne change pas~: $\vect p$ est constante, c.-à-d.\
$d\vect p/dt = \vect 0$.
\end{theorem}
\admitted

\begin{remark}[Inertie, affûtée deux fois]\label{rem:g12:newtons-laws:sharpened}
Le principe d'inertie du \cref{ch:g10:inertia}, mis à jour deux fois~: il
dit \emph{où} il vaut --- il définit les référentiels galiléens comme
ceux où il vaut --- et il protège un vecteur, $m\vect v$, dont la
comptabilité paie à la fin du chapitre.
\end{remark}

\section{Deuxième loi de Newton}

\begin{theorem}[Deuxième loi de Newton]\label{thm:g12:newtons-laws:second}\index{deuxième loi de Newton}
Dans un référentiel galiléen, la \omterm{def:g11:forces-and-motion:netforce}{force résultante} sur un
corps égale le taux de variation de sa
\omterm{def:g12:newtons-laws:momentum}{quantité de mouvement}~; pour une masse constante, comme
$\vect p = m\vect v$,
\[
\sum \vect F = \frac{d\vect p}{dt} = m\,\frac{d\vect v}{dt} = m \vect a .
\]
\end{theorem}
\admitted

\begin{remark}[Une loi de la nature, et son unité]\label{rem:g12:newtons-laws:nature}
Il n'y a pas de preuve~: c'est une loi de la nature, pesée contre
l'expérience depuis trois siècles. L'année dernière
(\cref{ch:g11:forces-and-motion}) en a donné l'ombre,
$\Delta v \propto F\,\Delta t/m$~; la dérivée
(\cref{ch:g12:kinematics-2d}) la rend exacte à chaque instant --- ses
petits caractères attendent dans le volume de Licence~1. La
\omterm{def:g10:inertia:force}{force} reçoit une \omterm{def:g10:orders-of-magnitude:unit}{unité},
$\unit{N} = \unit{kg.m/s^2}$~; et un corps sous son seul
\omterm{def:g10:universal-gravitation:weight}{poids} obéit à $m\vect a = m\vect g$~: toute masse tombe
avec $\vect a = \vect g$, et
$g = \qty{9.81}{N/kg} = \qty{9.81}{m/s^2}$ --- les deux lectures n'en
font qu'une.
\end{remark}

\begin{example}[Force moyenne, sans le détail]\label{ex:g12:newtons-laws:average}
Une voiture de \qty{1300}{kg} passe du repos à \qty{100}{km/h}
(\qty{27.8}{m/s}) en \qty{8.0}{s}~: quelle que soit l'histoire instant
par instant, la \omterm{def:g11:forces-and-motion:netforce}{force résultante} moyenne est
$\Delta p/\Delta t = 1300 \times 27.8/8.0 \approx \qty{4.5e3}{N}$ ---
la \omterm{def:g12:newtons-laws:momentum}{quantité de mouvement} transforme une question de
\omterm{def:g10:inertia:force}{force} en une comptabilité avant-après.
\end{example}

\section{Troisième loi de Newton}

\begin{theorem}[Troisième loi de Newton]\label{thm:g12:newtons-laws:third}\index{troisième loi de Newton}
Si un corps $A$ exerce une \omterm{def:g10:inertia:force}{force} $\vect F_{A \to B}$ sur un
corps $B$, alors $B$ exerce sur $A$ la \omterm{def:g10:inertia:force}{force}
$\vect F_{B \to A} = -\vect F_{A \to B}$~: même droite d'action, même
norme, directions opposées --- à chaque instant, quel que soit le
mouvement, contact ou action à distance.
\end{theorem}
\admitted

\begin{example}[Le livre, la table et la Terre]\label{ex:g12:newtons-laws:book}
Un livre repose sur une table~: deux interactions, deux paires.
Gravitationnelle~: la Terre tire le livre vers le bas ($\vect P$), le
livre tire la Terre vers le haut. Contact~: la table pousse le livre
vers le haut ($\vect N$), le livre appuie la table vers le bas. Marcher
est le même marché~: poussez le sol vers l'arrière~; sa réaction est la
seule \omterm{def:g10:inertia:force}{force} qui vous entraîne vers l'avant.
\end{example}

\section{La méthode~: choisir, dessiner, projeter}

\begin{method}[Résoudre un problème de dynamique]\label{met:g12:newtons-laws:method}
\begin{enumerate}
  \item \emph{Système}~: nommer le corps (ou l'ensemble de corps) étudié.
  \item \emph{Référentiel}~: choisir un référentiel galiléen et des axes
        adaptés au mouvement --- axes de pente, ou le
        \omterm{def:g12:kinematics-2d:frenet}{repère de Frenet} pour une courbe
        (\cref{ch:g12:kinematics-2d}).
  \item \emph{Inventaire}~: lister chaque \omterm{def:g10:inertia:force}{force} --- le
        \omterm{def:g10:universal-gravitation:weight}{poids}, puis une \omterm{def:g10:inertia:force}{force} par
        contact, rien d'autre --- et dessiner le diagramme des forces~:
        un point, une flèche par \omterm{def:g10:inertia:force}{force}.
  \item \emph{Projeter}~: écrire $\sum \vect F = m\vect a$, projeter sur
        chaque axe.
  \item \emph{Résoudre et vérifier}~: \omterm{def:g10:orders-of-magnitude:unit}{unités}, signes, cas
        limites ($\mu \to 0$, $\alpha \to 0$, \dots).
\end{enumerate}
\end{method}

\begin{definition}[Force normale et frottement]\label{def:g12:newtons-laws:friction}
Une surface agit sur un corps par deux composantes~: la \emph{force
normale}\index{force normale} $\vect N$, perpendiculaire à la surface,
et le \omterm{def:g10:inertia:inventory}{frottement} $\vect f$, tangentiel. L'expérience donne les
lois de \omterm{def:g10:inertia:inventory}{frottement}~: tant que le corps ne glisse pas, le
\omterm{def:g10:inertia:inventory}{frottement} statique s'ajuste jusqu'à un plafond,
$f_s \leq \mu_s N$~; une fois qu'il glisse, le
\omterm{def:g11:forces-and-motion:friction}{frottement cinétique} est $f_k = \mu_k N$, contre le
glissement. Les \emph{coefficients de \omterm{def:g10:inertia:inventory}{frottement}}%
\index{coefficient de frottement} $\mu_s$ et $\mu_k$
($\mu_k \leq \mu_s$) ne dépendent que des deux matériaux.
\end{definition}

\begin{example}[Le plan incliné, deux fois]\label{ex:g12:newtons-laws:incline}
Un bloc glisse le long d'une pente d'angle $\alpha$~; axes~: $x$ vers
le bas de la pente, $y$ perpendiculaire. Sur $y$~: $N = mg\cos\alpha$.
Sur $x$, sans frottement~: $ma = mg\sin\alpha$, donc
$a = g\sin\alpha$ --- pas de masse~; à $\alpha = 30^\circ$,
$a = \qty{4.9}{m/s^2}$. Avec \omterm{def:g11:forces-and-motion:friction}{frottement cinétique}~:
$a = g(\sin\alpha - \mu_k\cos\alpha)$~; pour $\mu_k = 0.20$,
$a = 9.81\,(0.500 - 0.173) = \qty{3.2}{m/s^2}$. Et il restait d'abord
immobile tant que le \omterm{def:g10:inertia:inventory}{frottement} statique nécessaire tenait
sous son plafond~: $\tan\alpha \leq \mu_s$.
\end{example}

\begin{definition}[Poids apparent]\label{def:g12:newtons-laws:apparent}
Le \emph{poids apparent}\index{poids apparent} d'un corps est la
\omterm{def:g12:newtons-laws:friction}{force normale} entre lui et son support --- ce qu'une balance
sous lui lit.
\end{definition}

\begin{example}[L'ascenseur]\label{ex:g12:newtons-laws:elevator}
Un passager de \qty{70}{kg} se tient sur une balance dans un ascenseur
d'accélération verticale $a_z$ (haut positif)~: $N - mg = ma_z$, donc
$N = m(g + a_z)$. En s'élançant vers le haut à
$a_z = \qty{1.5}{m/s^2}$~: \qty{792}{N}, lourd~; freinage près du
sommet~: \qty{582}{N}, léger~; croisière régulière~:
$mg = \qty{687}{N}$ --- la première loi~; chute libre, $a_z = -g$~:
$N = 0$, apesanteur sur une balance.
\end{example}

\begin{example}[Deux blocs et une poulie]\label{ex:g12:newtons-laws:pulley}
Un bloc $m_1 = \qty{4.0}{kg}$ sur une table sans frottement est relié
par une corde légère, sur une poulie idéale, à un bloc suspendu
$m_2 = \qty{1.0}{kg}$. La corde transmet la même
\omterm{def:g10:inertia:inventory}{tension} $T$ aux deux bouts~; les blocs partagent une même
norme d'accélération $a$. Une deuxième loi chacun, le long de chaque
mouvement~: $m_1 a = T$ et $m_2 a = m_2 g - T$, donc
$a = m_2\,g/(m_1 + m_2) = \qty{1.96}{m/s^2}$ et
$T = m_1 a = \qty{7.8}{N}$ --- plus petit que $m_2 g = \qty{9.8}{N}$,
comme il se doit pour que $m_2$ accélère vers le bas.
\end{example}

\begin{example}[Le virage plat]\label{ex:g12:newtons-laws:curve}
Une voiture de masse $m$ prend un virage plat de rayon $R$ à vitesse
constante $v$. Dans le \omterm{def:g12:kinematics-2d:frenet}{repère de Frenet}
(\cref{ch:g12:kinematics-2d}) l'accélération est purement normale,
$a_N = v^2/R$, dirigée vers le centre. \omterm{def:g10:universal-gravitation:weight}{Verticalement},
$N = mg$~; horizontalement, la seule \omterm{def:g10:inertia:force}{force} centripète
disponible est le \omterm{def:g10:inertia:inventory}{frottement} statique de la route sur les
pneus~: $f = mv^2/R \leq \mu_s N = \mu_s mg$, donc
$v \leq \sqrt{\mu_s\, g R}$, quelle que soit la masse. Pour
$R = \qty{90}{m}$, route sèche ($\mu_s = 0.70$)~:
$v_{\max} \approx \qty{25}{m/s}$ (\qty{89}{km/h})~; sur glace le
plafond s'effondre --- l'\cref{exo:g12:newtons-laws:13} déverse la
route~; l'\cref{exo:g12:newtons-laws:10} balance la même projection sur
un fil.
\end{example}

\section{Systèmes isolés~: la quantité de mouvement se conserve}

\begin{proposition}[Conservation de la quantité de mouvement]\label{prop:g12:newtons-laws:conservation}\index{conservation de la quantité de mouvement}
Pour deux corps dont les \omterm{def:g10:inertia:force}{forces} extérieures s'annulent ou
\omterm{def:g10:inertia:compensate}{se compensent}, la \omterm{def:g12:newtons-laws:momentum}{quantité de mouvement}
totale $\vect p_1 + \vect p_2$ est constante, quelles que soient les
\omterm{def:g10:inertia:force}{forces} qu'ils exercent l'un sur l'autre.
\end{proposition}

\begin{proof}
Additionner les deux deuxièmes lois~:
$\frac{d}{dt}(\vect p_1 + \vect p_2) = \vect F_{2 \to 1} +
\vect F_{1 \to 2} + \vect F_{\text{ext}}$~: la paire interne s'annule
par la troisième loi, et $\vect F_{\text{ext}} = \vect 0$ par hypothèse.
\end{proof}

\begin{example}[Recul]\label{ex:g12:newtons-laws:recoil}
Une patineuse de \qty{60}{kg} au repos sur de la glace lisse lance une
balle médicinale de \qty{4.0}{kg} horizontalement à \qty{6.0}{m/s}.
Avant~: $\vect p = \vect 0$~; après~:
$0 = 60\,V + 4.0 \times 6.0$, donc $V = \qty{-0.40}{m/s}$ --- elle
dérive vers l'arrière. Rien d'extérieur ne l'a poussée~: la balle l'a
fait (troisième loi), et les comptes restent à zéro.
\end{example}

\begin{remark}[Comment une fusée pousse sur rien]\label{rem:g12:newtons-laws:rocket}
Une fusée est une machine à jeter de la masse. Chaque seconde ses
moteurs projettent un lot de gaz vers l'arrière~; les gaz emportent de
la \omterm{def:g12:newtons-laws:momentum}{quantité de mouvement} vers l'arrière, donc la fusée
gagne la même quantité vers l'avant --- une \omterm{def:g10:inertia:inventory}{poussée}
régulière, à peu près la masse éjectée par seconde fois la vitesse
d'éjection. La fusée pousse sur ses propres gaz d'échappement, non sur
l'air~: elle \omterm{def:g11:work-of-force:work}{fonctionne} le mieux dans le vide.
\end{remark}

\section{Exercices}


\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:1}
Calculer la \omterm{def:g12:newtons-laws:momentum}{quantité de mouvement} de (a) un sprinteur de
\qty{70}{kg} à \qty{10}{m/s}~; (b) une voiture de \qty{1300}{kg} à
\qty{50}{km/h}~; (c) une balle de \qty{8.0}{g} à \qty{800}{m/s}. Les
classer. Pourquoi le plus rapide arrive-t-il en dernier~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:2}
Un passager de \qty{70}{kg} se tient sur une balance dans un ascenseur.
Que lit-elle lorsque la cabine (a) accélère vers le haut à
\qty{1.2}{m/s^2}~; (b) monte à une \qty{2.0}{m/s} constante~; (c)
accélère vers le bas à \qty{1.2}{m/s^2}~? Quelle loi tranche (b)~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:3}
Un bloc est lâché sur une pente sans frottement d'angle $30^\circ$.
Calculer son accélération, puis sa vitesse après avoir glissé
\qty{3.0}{m}. Où est passée la masse~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:4}
Une balle de tennis de \qty{58}{g} arrive à \qty{15}{m/s} et est
renvoyée le long de la même ligne à \qty{25}{m/s}~; le contact dure
\qty{5.0}{ms}. Calculer le changement de
\omterm{def:g12:newtons-laws:momentum}{quantité de mouvement} et la \omterm{def:g10:inertia:force}{force}
moyenne des cordages~; comparer cette \omterm{def:g10:inertia:force}{force} au
\omterm{def:g10:universal-gravitation:weight}{poids} de la balle.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:5}
Un livre repose sur une table. Nommer les deux \omterm{def:g10:inertia:force}{forces}
agissant sur le livre et, pour chacune, son partenaire action--réaction
(quel corps~? quelle direction~?). Pourquoi le
\omterm{def:g10:universal-gravitation:weight}{poids} du livre et la poussée de la table ne sont-ils
pas partenaires l'un de l'autre~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:6}
Une caisse de \qty{50}{kg} repose sur un sol horizontal,
$\mu_s = 0.45$, $\mu_k = 0.35$.
(a) Quelle poussée horizontale la met tout juste en marche~?
(b) Si cette poussée est maintenue, quelle est son accélération~?
(c) Quelle poussée la maintient en mouvement à vitesse constante~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:7}
La même caisse est placée sur une rampe réglable ($\mu_s = 0.45$,
$\mu_k = 0.35$). (a) À quel angle commence-t-elle à glisser~?
(b) À $30^\circ$, calculer son accélération.
(c) Une fois partie à l'angle de (a), continue-t-elle d'accélérer~?
Pourquoi~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:8}
Un bloc de \qty{3.0}{kg} sur une table sans frottement est relié par une
poulie idéale à un bloc suspendu de \qty{2.0}{kg}. Calculer
l'accélération et la \omterm{def:g10:inertia:inventory}{tension}. Pourquoi la
\omterm{def:g10:inertia:inventory}{tension} a-t-elle dû sortir plus petite que le
\omterm{def:g10:universal-gravitation:weight}{poids} suspendu~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:9}
Un virage de rayon \qty{50}{m} est plat. Calculer la vitesse maximale
d'une voiture pour $\mu_s = 0.70$ (sec) et $0.40$ (humide), en
\unit{km/h}. Pourquoi la réponse vaut-elle pour un scooter et un
camion~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:10}
Un pendule de \qty{0.30}{kg} au bout d'un fil de \qty{1.2}{m} trace un
cercle horizontal, le fil maintenant un angle constant de $25^\circ$
avec la verticale (pendule conique). Projeter la deuxième loi
\omterm{def:g10:universal-gravitation:weight}{verticalement}, puis le long du rayon~: calculer la
\omterm{def:g10:inertia:inventory}{tension} et la vitesse.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:11}
Un colis de \qty{5.0}{kg} voyage sur une balance dans un ascenseur de
fret~; les lectures~:
\begin{center}\small\begin{tabular}{lccc}
phase & \qtyrange{0}{2}{s} & \qtyrange{2}{8}{s} & \qtyrange{8}{10}{s}\\
balance (\unit{N}) & 54.0 & 49.0 & 44.1
\end{tabular}\end{center}
Calculer l'accélération de chaque phase et décrire un trajet possible.
L'ascenseur a-t-il pu se déplacer \emph{vers le bas} tout le temps~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:12}
Une patineuse de \qty{60}{kg} au repos sur de la glace lisse lance une
balle de \qty{4.0}{kg} horizontalement. (a) La balle part à
\qty{8.0}{m/s} \emph{par rapport au sol}~: sa vitesse de recul~?
(b) Les \qty{8.0}{m/s} sont maintenant mesurés \emph{par rapport à
elle}~: recalculer les deux vitesses. (c) Quelle
\omterm{def:g10:inertia:force}{force} l'a poussée vers l'arrière~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:13}
Un virage d'autoroute de rayon \qty{80}{m} est conçu pour \qty{20}{m/s}
sans aide du \omterm{def:g10:inertia:inventory}{frottement}. Montrer que la chaussée doit être
déversée de l'angle $\theta$ donné par $\tan\theta = v^2/(gR)$ et le
calculer. Que doit faire le \omterm{def:g10:inertia:inventory}{frottement} pour une voiture
prenant le virage plus lentement --- ou plus vite~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:14}
Les moteurs d'une fusée éjectent \qty{250}{kg} de gaz par seconde à

seconde --- quelle \omterm{def:g10:inertia:inventory}{poussée} les gaz renvoient-ils~?
(b) La fusée pèse \qty{5.0e4}{kg} à l'allumage~: son
\omterm{def:g10:universal-gravitation:weight}{poids}~? son accélération au décollage~? (c) Pourquoi
le moteur \omterm{def:g11:work-of-force:work}{fonctionne}-t-il aussi bien dans le vide~?
\end{exercise}

\section{Problème~: Le téléphérique}

\begin{problem}[{Problème du week-end --- certifier le téléphérique~: la
tension que la pente exige au repos, la surtaxe du démarrage, le soulagement
au-dessus de la tour, le prix d'un arrêt d'urgence --- et le seul nombre que
la plaque de la cabine peut promettre}]
\label{pb:g12:newtons-laws:1}
Un téléphérique de montagne --- cabine plus chariot, masse à vide
$M_0 = \qty{2.0e3}{kg}$ --- roule sur un câble porteur incliné à
$\alpha = 30^\circ$, tiré par un câble tracteur parallèle à la pente
avec la \omterm{def:g10:inertia:inventory}{tension} $\vect T$. Le \omterm{def:g10:inertia:inventory}{frottement}
de roulement est négligeable~; le plancher de la cabine reste horizontal
tout du long. La cabine accueille $25$ passagers de masse moyenne
\qty{80}{kg}~; vous êtes l'ingénieur certificateur. Sauf mention
contraire, la cabine est pleine~: $M = \qty{4.0e3}{kg}$.

\medskip
\noindent\textbf{Partie I --- Au repos sur la pente.}
\begin{enumerate}
  \item Choisir le système~; dire pourquoi le référentiel de la montagne
        fera office de référentiel galiléen~; inventaire des trois
        \omterm{def:g10:inertia:force}{forces}, avec directions.
  \item Quelle loi gouverne l'attente~? La projeter le long et
        perpendiculairement à la pente pour exprimer $T$ et la
        \omterm{def:g12:newtons-laws:friction}{force normale} $N$ du câble porteur.
  \item Calculer $T$ et $N$.
  \item Recalculer $T$ pour la cabine vide. Pourquoi $T$ est-elle
        proportionnelle à la masse totale~?
  \item Le câble tracteur casse, freins ouverts~: montrer que
        l'accélération de l'emballement est $g \sin\alpha$, chargée ou
        vide, et la calculer.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Démarrage.} Le treuil accélère la cabine
pleine à $a = \qty{0.60}{m/s^2}$ vers le haut de la pente jusqu'à la
croisière à $v = \qty{6.0}{m/s}$.
\begin{enumerate}[resume]
  \item Exprimer et calculer la \omterm{def:g10:inertia:inventory}{tension} $T'$ pendant
        cette phase.
  \item De quelle fraction le démarrage élève-t-il la
        \omterm{def:g10:inertia:inventory}{tension} au-dessus de sa valeur au repos~?
  \item Combien dure la phase, et sur quelle distance~?
  \item Un passager de \qty{80}{kg} se tient sur le plancher horizontal.
        Projeter la deuxième loi \omterm{def:g10:universal-gravitation:weight}{verticalement} pour exprimer
        et calculer son \omterm{def:g12:newtons-laws:apparent}{poids apparent}~; comparer à $mg$.
  \item Quelle \omterm{def:g10:inertia:force}{force} l'accélère horizontalement, et de
        quelle taille~? Pourquoi les passagers se penchent-ils~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Au-dessus de la tour.} À une
\qty{6.0}{m/s} constante, le câble porteur se courbe au-dessus d'une
tour de support le long d'un arc circulaire vertical de rayon
$R = \qty{40}{m}$~; au sommet la vitesse est horizontale.
\begin{enumerate}[resume]
  \item Dans le \omterm{def:g12:kinematics-2d:frenet}{repère de Frenet} au sommet
        (\cref{ch:g12:kinematics-2d}), donner les deux composantes de
        l'accélération, avec leurs valeurs.
  \item Projeter la deuxième loi le long de la normale pour exprimer et
        calculer la \omterm{def:g12:newtons-laws:friction}{force normale} $N'$ du câble sur le
        chariot.
  \item Même question pour le \omterm{def:g12:newtons-laws:apparent}{poids apparent} du passager
        au sommet~: de quelle fraction est-il plus léger~?
  \item À quelle vitesse $N'$ s'annulerait-elle~? Comparer à la vitesse
        de croisière et commenter.
  \item Quelle \omterm{def:g10:inertia:force}{force} le chariot exerce-t-il sur le câble
        au-dessus de la tour --- quelle loi le dit, avec quelle norme et
        direction~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Freinage, un saut, et la plaque.}
\begin{enumerate}[resume]
  \item Sur la section d'arrivée horizontale, un arrêt d'urgence ramène
        la cabine pleine de \qty{6.0}{m/s} au repos en \qty{1.5}{s}~: lire
        $\sum \vect F = d\vect p/dt$ comme une moyenne sur l'arrêt pour
        calculer la \omterm{def:g10:inertia:force}{force} de freinage.
  \item Quelle \omterm{def:g10:inertia:force}{force} horizontale doit arrêter un passager
        de \qty{80}{kg}, et le \omterm{def:g10:inertia:inventory}{frottement} statique
        chaussure--plancher ($\mu_s = 0.40$) peut-il la fournir~?
        Conclure.
  \item Sur le quai, freins ouverts, la cabine vide pend libre et
        immobile~; le dernier passager (\qty{80}{kg}) saute
        horizontalement à \qty{2.5}{m/s} \emph{par rapport à la cabine}.
        Calculer la vitesse de recul de la cabine et la vitesse du
        passager par rapport au sol.
  \item En une phrase~: quel dispositif de propulsion vit de ce principe
        de recul, et qu'est-ce qui fixe sa \omterm{def:g10:inertia:inventory}{poussée}~?
  \item \emph{Certification.} Le câble tracteur casse à \qty{100}{kN}~;
        les règles plafonnent la \omterm{def:g10:inertia:inventory}{tension} de service au
        cinquième de cela. Montrer que le pire cas rencontré ci-dessus
        est le démarrage, en déduire la masse totale maximale, et
        imprimer la plaque~: combien de passagers de \qty{80}{kg} peuvent
        monter~? Vérifier le facteur de sécurité.
\end{enumerate}
\end{problem}
