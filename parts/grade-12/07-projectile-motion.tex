\chapter{Free Fall and Projectile Motion}\label{ch:g12:projectile-motion}

Every jet of a garden hose draws the same curve in the air. So does a
basketball arcing toward the hoop, a long jumper, a cannonball: any body
abandoned to its weight. This chapter earns that curve --- a parabola ---
from Newton's second law in two lines of calculus, reads apex, flight
time and range off its equation, then lets the air back in to see what
it spoils.

\section{Free fall}

\begin{definition}[Free fall]\label{def:g12:projectile-motion:freefall}
A body is in \emph{free fall}\index{free fall} when the only force acting on
it is its weight $\vect P = m\vect g$: no support, no string, air resistance
negligible. A dropped stone or a ball that has left the hand is, to good
accuracy, in free fall --- ``fall'' includes rising.
\end{definition}

\begin{theorem}[Galileo's law of free fall]\label{thm:g12:projectile-motion:galileo}
\index{Galileo's law of free fall}
In free fall, all bodies share the same acceleration, whatever their mass:
\[
\vect a = \vect g, \qquad g = \qty{9.81}{m/s^2},
\]
directed straight down: the field strength \qty{9.81}{N/kg} of
\cref{ch:g10:universal-gravitation} in its other outfit, $\unit{N/kg} = \unit{m/s^2}$.
\end{theorem}

\begin{proof}
Newton's second law (\cref{ch:g12:newtons-laws}) with the weight as only
force: $m\vect a = m\vect g$, and the mass cancels. Hammer and feather
differ only through the air.
\end{proof}

\begin{proposition}[Vertical fall from rest]\label{prop:g12:projectile-motion:vertical}
Dropped from rest, with the fallen depth $h$ counted downward from the
release point, a body in free fall obeys
\[
a = g, \qquad v = g t, \qquad h = \tfrac12\, g t^2 .
\]
\end{proposition}

\begin{proof}
$v(t)$ is the antiderivative of the constant $a = g$ vanishing at $t = 0$
(\cref{ch:g12:kinematics-2d}); $h(t)$ that of $v = gt$ vanishing at $t = 0$.
\end{proof}

\begin{example}[Orders of magnitude]\label{ex:g12:projectile-motion:orders}
After $1$, $2$, $3$ seconds of free fall: $v = 9.8$, $19.6$, \qty{29.4}{m/s}
($35$, $71$, \qty{106}{km/h}) and $h = 4.9$, $19.6$, \qty{44.1}{m}. From a
\qty{30}{m} rooftop: $t = \sqrt{2h/g} \approx \qty{2.5}{s}$, arriving at
\qty{24}{m/s} $\approx \qty{88}{km/h}$. Times grow like $\sqrt h$ --- and
past a few seconds the air refuses to stay negligible (last section).
\end{example}

\begin{example}[Vertical throw]\label{ex:g12:projectile-motion:throw}
Thrown straight up at $v_0$, axis $Oy$ upward from the hand: the same two
antiderivatives give $v = v_0 - gt$ and $y = v_0 t - \tfrac12 gt^2$. The
apex, where $v = 0$, comes at $t = v_0/g$, height $h = v_0^2/2g$ --- the
energy answer $v_0 = \sqrt{2gh}$ of \cref{ch:g11:mechanical-energy} run
backward; thrown \emph{down}, $v_0$ flips sign. For \qty{15}{m/s}: apex
\qty{11.5}{m} at \qty{1.53}{s}, back at \qty{3.06}{s}, at \qty{15}{m/s}.
\end{example}

\section{The projectile: one law, two antiderivatives}

\begin{definition}[Projectile]\label{def:g12:projectile-motion:projectile}
A \emph{projectile}\index{projectile} is a body launched with a velocity
$\vect v_0$ and then left in free fall. Axes at the launch point: $Ox$
horizontal in the vertical plane of $\vect v_0$, $Oy$ vertical upward; the
\emph{launch angle}\index{launch angle} $\alpha$ of $\vect v_0$ above the
horizontal fixes its coordinates $(v_0\cos\alpha,\; v_0\sin\alpha)$.
\end{definition}

\begin{theorem}[Equations of motion]\label{thm:g12:projectile-motion:equations}
A projectile launched from the origin with speed $v_0$ and angle $\alpha$
has $\vect a = (0,\,-g)$ at every instant, hence
\[
\vect v(t) = \bigl(v_0\cos\alpha,\; v_0\sin\alpha - g t\bigr),
\qquad
\begin{cases}
x(t) = v_0\cos\alpha\; t,\\[2pt]
y(t) = v_0\sin\alpha\; t - \tfrac12\, g t^2 .
\end{cases}
\]
\end{theorem}

\begin{proof}
Newton's second law gives $\vect a = \vect g = (0,-g)$. Antiderivatives
axis by axis: $\vect v(t)$ from $\vect a$ with initial value
$(v_0\cos\alpha,\, v_0\sin\alpha)$, then position from $\vect v$, from $(0,0)$.
\end{proof}

\begin{proposition}[Independence of the two motions]\label{prop:g12:projectile-motion:independence}
The horizontal motion is uniform --- $v_x$ never changes --- and the
vertical motion is exactly the vertical throw of the previous section:
neither equation mentions the other's coordinate. In particular, a ball
fired horizontally and a ball dropped at the same instant from the same
height stay level with each other --- they land together.
\end{proposition}

\begin{proof}
In \cref{thm:g12:projectile-motion:equations}, $x(t)$ involves only
$v_0\cos\alpha$, $y(t)$ only $v_0\sin\alpha$ and $g$; both balls have
$v_y(0) = 0$, so both obey $h = \tfrac12 gt^2$.
\end{proof}

\begin{remark}[The falling monkey]\label{rem:g12:projectile-motion:monkey}
A zookeeper aims a dart \emph{straight at} a monkey, who lets go of its
branch at the shot. Bad move: dart and monkey each drop $\tfrac12 gt^2$
below their no-gravity positions --- the aiming line, the branch --- so
they meet exactly, whatever the dart's speed
(\cref{exo:g12:projectile-motion:15}).
\end{remark}

\section{The parabola, read in full}

\begin{theorem}[Trajectory]\label{thm:g12:projectile-motion:trajectory}
Eliminating $t$ between $x(t)$ and $y(t)$: the trajectory of a projectile
($0 \leq \alpha < \ang{90}$) is the downward-opening parabola
\[
y = x\tan\alpha \;-\; \frac{g}{2 v_0^2 \cos^2\alpha}\, x^2 .
\]
\end{theorem}

\begin{proof}
$t = x/(v_0\cos\alpha)$, substituted into $y(t)$ of
\cref{thm:g12:projectile-motion:equations}.
\end{proof}

\begin{definition}[Range]\label{def:g12:projectile-motion:range}
For a projectile over level ground, the \emph{range}\index{range} $R$ is
the horizontal distance covered when it returns to its launch height; the
\emph{time of flight}\index{time of flight} $t_f$ is that trip's duration.
\end{definition}

\begin{proposition}[Apex, flight time, range]\label{prop:g12:projectile-motion:range}
Over level ground, with $0 < \alpha < \ang{90}$:
\[
t_f = \frac{2 v_0 \sin\alpha}{g}, \qquad
h_{\text{apex}} = \frac{(v_0\sin\alpha)^2}{2g}
\ \ \text{reached at } t_f/2, \qquad
R = \frac{v_0^2 \sin(2\alpha)}{g}.
\]
The range is greatest at $\alpha = \ang{45}$, where $R = v_0^2/g$;
complementary angles $\alpha$ and $\ang{90} - \alpha$ give the same range.
\end{proposition}

\begin{proof}
$y = 0$ gives $t\,(v_0\sin\alpha - \tfrac12 gt) = 0$: launch, and $t_f$.
The apex, $v_y = 0$, sits at $t = v_0\sin\alpha/g = t_f/2$; substitute into
$y(t)$. Then $R = x(t_f) = 2v_0^2\sin\alpha\cos\alpha/g =
v_0^2\sin(2\alpha)/g$. The sine peaks at $2\alpha = \ang{90}$, and
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180} - 2\alpha) =
\sin(2\alpha)$.
\end{proof}

\begin{example}[Goal kick]\label{ex:g12:projectile-motion:kick}
A goalkeeper strikes at $v_0 = \qty{25}{m/s}$, $\alpha = \ang{45}$:
$R = 25^2/9.81 \approx \qty{64}{m}$; real clearances land nearer
\qty{55}{m} --- the air's cut.
\end{example}

\begin{method}[Projectile bookkeeping]\label{met:g12:projectile-motion:method}
\begin{enumerate}
  \item Axes at the launch point, $Ox$ horizontal, $Oy$ up; coordinates
        of $\vect v_0$.
  \item Check free fall (weight only): then $\vect a = (0,-g)$;
        antiderivatives twice, initial conditions each time.
  \item Question about a \emph{place}: eliminate $t$ (trajectory). About an
        \emph{instant}: keep $t$.
  \item Dictionary: apex $\Leftrightarrow v_y = 0$; launch height
        $\Leftrightarrow y = 0$; wall at $d \Leftrightarrow x = d$; floor
        $\ell$ below $\Leftrightarrow y = -\ell$.
  \item Sanity: flight symmetric about the apex; $R \leq v_0^2/g$; air only
        ever shortens.
\end{enumerate}
\end{method}

\section{What the air changes}

\begin{definition}[Terminal speed]\label{def:g12:projectile-motion:terminal}
Air resistance opposes the velocity and grows with speed. A falling body
therefore accelerates only until drag balances weight; from there
$\vect a = \vect 0$: it falls at a constant \emph{terminal
speed}\index{terminal speed}.
\end{definition}

\begin{example}[Terminal orders of magnitude]\label{ex:g12:projectile-motion:airorders}
A flat-out skydiver: about \qty{55}{m/s} (\qty{200}{km/h}). A raindrop:
about \qty{9}{m/s} --- in vacuum, \qty{2000}{m} of fall would deliver it at
$\sqrt{2gh} \approx \qty{2.0e2}{m/s}$. A feather: about \qty{1}{m/s},
reached within centimeters --- the secret of its slowness: it spends its
whole fall \emph{at} terminal speed; the hammer never gets near its own.
\end{example}

\begin{remark}[Real ballistics]\label{rem:g12:projectile-motion:drag}
Drag pushes against the velocity all along the flight: the range shortens,
the apex lowers, and the symmetry breaks --- the descent steeper than the
climb, long shots visibly short of the parabola. The honest, quantitative
treatment is a differential equation solved in the Year 1 volume; this
year, the parabola is the right model for dense, moderately fast
projectiles --- a basketball at \qty{10}{m/s}, not a golf ball at
\qty{70}{m/s}.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:1}
A stone dropped from a bridge hits the water \qty{2.0}{s} later: height of
the bridge, and impact speed in \unit{m/s} and \unit{km/h}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:2}
On the Moon, an astronaut dropped a hammer and a feather together: they
landed together. Why --- and why does the feather lose so badly on Earth?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:3}
Neglecting air, compute fall time and arrival speed from \qty{1.0}{m},
\qty{10}{m}, \qty{100}{m}. How do the times grow with the height?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:4}
A ball is thrown straight up at \qty{15}{m/s}: apex height and time,
total time back to the hand, and return speed?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:5}
A marble rolls off a table of height \qty{0.80}{m} at \qty{2.5}{m/s}. Find
the fall time and the landing distance from the edge. Does the fall time
depend on the marble's speed?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:6}
From the five-angle figure of the course: which pairs of launch angles
share their range, and which angle wins? For a range below the maximum,
which of the two possible angles flies longer? Justify with $t_f$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:7}
From a \qty{20}{m} balcony a ball is thrown straight \emph{down} at
\qty{5.0}{m/s}. Find the time to the ground and the impact speed; check
the latter with an energy balance (\cref{ch:g11:mechanical-energy}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:8}
A projectile leaves level ground at \qty{40}{m/s}, $\alpha = \ang{30}$:
time of flight, apex height and range?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:9}
Model a long jumper as a projectile: takeoff at \qty{10.5}{m/s},
$\alpha = \ang{20}$. Compute the range; champions jump nearly \qty{9}{m}
--- name two ingredients the point model leaves out.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:10}
At \qty{1.2}{m} above flat ground, one bullet is dropped and an identical
one fired horizontally at \qty{300}{m/s}. Which lands first? Compute the
fall time and the fired bullet's landing distance. What principle is
tested?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:11}
Show algebraically that $\alpha$ and $\ang{90} - \alpha$ share their
range, then find the two angles landing a \qty{25}{m/s} projectile
\qty{50}{m} away.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:12}
Water leaves a garden hose at \qty{8.0}{m/s}: greatest horizontal reach?
greatest vertical height? the two nozzle angles watering a flowerbed
\qty{5.0}{m} away?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:13}
A ball is kicked from the ground at \qty{18}{m/s}, $\alpha = \ang{35}$,
toward a \qty{4.0}{m} hedge \qty{15}{m} away. Using the trajectory
equation, decide whether it clears, and by how much.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:14}
An \qty{80}{kg} skydiver falls flat at terminal speed \qty{55}{m/s}.
(a) The drag force? (b) Sketch the shape of $v(t)$ from the jump. (c) A
raindrop falls from \qty{2.0}{km}: vacuum arrival speed vs.\ its real
\qty{9}{m/s}?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:15}
The falling monkey, proved: a dart leaves the origin aimed exactly at a
monkey at distance $D$, height $H$ ($\tan\alpha = H/D$), who drops at
$t = 0$. Show that when the dart reaches $x = D$ its height equals the
monkey's --- whatever $v_0$.
\end{exercise}

\section{Problem: The Basketball Buzzer-Beater}

\begin{problem}[{Weekend problem --- the basketball buzzer-beater: a free
throw dissected arrow by arrow, fingertips cleared by centimeters, a
full-court heave weighed against what an arm can give, a lob over the
backboard --- and a verdict on the most theatrical shot in sport}]
\label{pb:g12:projectile-motion:1}
The rim of a basketball hoop is a horizontal ring \qty{3.05}{m} above the
floor; Nora releases every shot at \qty{2.00}{m}, so with axes at the
release point ($Ox$ horizontal toward the hoop, $Oy$ up) the rim center
sits \qty{1.05}{m} above the origin. Air is neglected except where stated.

\medskip\noindent\textbf{Part I --- The free throw.} Nora shoots from
$d = \qty{4.20}{m}$ (horizontal) with launch angle $\alpha = \ang{50}$.
\begin{enumerate}
  \item What forces act once released? Give $\vect a$, citing the law used.
  \item Derive the coordinates of $\vect v(t)$.
  \item Derive $x(t)$ and $y(t)$.
  \item Eliminate $t$ to obtain the trajectory $y(x)$.
  \item The ball must pass the rim center, $y(d) = \qty{1.05}{m}$: solve
        for $v_0$, in \unit{m/s} and \unit{km/h}.
\end{enumerate}

\medskip\noindent\textbf{Part II --- Over the fingertips, into the ring.}
\begin{enumerate}[resume]
  \item How long does the ball take to reach the rim?
  \item Find the apex time and height; check the ball is already
        descending at the rim.
  \item A defender \qty{0.90}{m} in front of Nora stretches fingertips to
        \qty{2.60}{m}. By how much does the ball clear them?
  \item Compute $v_x$ and $v_y$ at the rim, the speed there, and the angle
        of the velocity below the horizontal.
  \item Why must a ball arrive \emph{descending} --- and why steep?
\end{enumerate}

\medskip\noindent\textbf{Part III --- The full-court heave.} Buzzer about
to sound, Nora is \qty{26.0}{m} from the far rim.
\begin{enumerate}[resume]
  \item The \qty{1.05}{m} rise is small against \qty{26.0}{m}: justify a
        level-ground model, and derive the time of flight from $y(t_f) = 0$.
  \item Deduce $R = v_0^2\sin(2\alpha)/g$ and the angle of greatest range.
  \item Compute the $v_0$ needed to cover \qty{26.0}{m} at $\alpha = \ang{45}$.
  \item A strong player can hurl a basketball at about \qty{17.5}{m/s}.
        What is her maximum range --- is the shot humanly feasible?
  \item At \qty{17.5}{m/s}, find the \emph{two} angles landing the ball
        \qty{26.0}{m} away, and their flight times: with \qty{2.0}{s} on
        the clock, which one beats the buzzer?
\end{enumerate}

\medskip\noindent\textbf{Part IV --- The lob over the backboard.} From
behind the baseline Nora lobs: release \qty{6.0}{m} from the rim center,
$\alpha = \ang{75}$; the board's plane crosses her shot \qty{0.45}{m}
before the rim center, top edge \qty{3.95}{m} above the floor.
\begin{enumerate}[resume]
  \item Find the $v_0$ that drops the ball through the rim center.
  \item Compute the ball's height at the board's plane
        ($x = \qty{5.55}{m}$): does it clear the top edge, by how much?
  \item Find the apex height and flight time --- why do crowds love it?
  \item Qualitatively, what does air resistance change on these three
        shots, and which one does it threaten most?
  \item \emph{The buzzer-beater verdict}, two sentences: compare the speed
        the full-court shot demands with what an arm supplies, give the
        margin, and rule --- possible or myth?
\end{enumerate}
\end{problem}
