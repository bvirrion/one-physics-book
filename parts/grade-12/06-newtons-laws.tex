\chapter{Newton's Laws}\label{ch:g12:newtons-laws}

Smuggle a bathroom scale into an elevator. As the cabin pulls upward the needle reads
three kilograms heavy; near the top it reads light; between the two, the plain truth
--- nothing aboard changed but the motion. This chapter states the three laws behind
that needle and forges them into the method that runs all of mechanics: choose a
system, draw its forces, project $\sum \vect F = m\vect a$.

\section{Momentum, and the first law made exact}

\begin{definition}[Momentum]\label{def:g12:newtons-laws:momentum}
The \emph{momentum}\index{momentum} of a body of mass $m$ (\unit{kg}) moving with
velocity $\vect v$ is the vector
\[
\vect p = m \vect v \qquad (\unit{kg.m/s}):
\]
the velocity, weighted by the mass; for a system, the vector sum over its parts.
\end{definition}

\begin{definition}[Inertial reference frame]\label{def:g12:newtons-laws:inertial}
An \emph{inertial reference frame}\index{inertial reference frame} is one in which
every isolated body --- no forces, or forces that compensate --- moves in a straight
line at constant velocity (\cref{ch:g10:inertia}). The ground is one to excellent
accuracy, and so is any cabin in uniform straight-line motion over it; a braking
train or turning carousel is not: there, loose luggage accelerates unpushed.
\end{definition}

\begin{theorem}[Newton's first law]\label{thm:g12:newtons-laws:first}\index{Newton's first law}
In an inertial frame, the momentum of an isolated system does not change: $\vect p$
is constant, i.e.\ $d\vect p/dt = \vect 0$.
\end{theorem}
\admitted

\begin{remark}[Inertia, sharpened twice]\label{rem:g12:newtons-laws:sharpened}
The principle of inertia of \cref{ch:g10:inertia}, upgraded twice: it says
\emph{where} it holds --- it defines the inertial frames as those where it does ---
and it guards a vector, $m\vect v$, whose bookkeeping pays off at the chapter's end.
\end{remark}

\section{Newton's second law}

\begin{theorem}[Newton's second law]\label{thm:g12:newtons-laws:second}\index{Newton's second law}
In an inertial frame, the net force on a body equals the rate of change of its
momentum; for constant mass, since $\vect p = m\vect v$,
\[
\sum \vect F = \frac{d\vect p}{dt} = m\,\frac{d\vect v}{dt} = m \vect a .
\]
\end{theorem}
\admitted

\begin{remark}[A law of nature, and its unit]\label{rem:g12:newtons-laws:nature}
There is no proof: this is a law of nature, weighed against experiment for three
centuries. Last year (\cref{ch:g11:forces-and-motion}) gave its shadow,
$\Delta v \propto F\,\Delta t/m$; the derivative (\cref{ch:g12:kinematics-2d}) makes
it exact at every instant --- its fine print waits in the Year 1 volume. Force gets a
unit, $\unit{N} = \unit{kg.m/s^2}$; and a body under its weight alone obeys
$m\vect a = m\vect g$: every mass falls with $\vect a = \vect g$, and
$g = \qty{9.81}{N/kg} = \qty{9.81}{m/s^2}$ --- the two readings are one.
\end{remark}

\begin{example}[Average force, no details needed]\label{ex:g12:newtons-laws:average}
A \qty{1300}{kg} car goes from rest to \qty{100}{km/h} (\qty{27.8}{m/s}) in
\qty{8.0}{s}: whatever the instant-by-instant story, the average net force is
$\Delta p/\Delta t = 1300 \times 27.8/8.0 \approx \qty{4.5e3}{N}$ --- momentum turns
a force question into before-and-after bookkeeping.
\end{example}

\section{Newton's third law}

\begin{theorem}[Newton's third law]\label{thm:g12:newtons-laws:third}\index{Newton's third law}
If a body $A$ exerts a force $\vect F_{A \to B}$ on a body $B$, then $B$ exerts on
$A$ the force $\vect F_{B \to A} = -\vect F_{A \to B}$: same line of action, same
magnitude, opposite directions --- at every instant, whatever the motion, contact or
action at a distance.
\end{theorem}
\admitted

\begin{example}[The book, the table and the Earth]\label{ex:g12:newtons-laws:book}
A book rests on a table: two interactions, two pairs. Gravitational: the Earth pulls
the book down ($\vect P$), the book pulls the Earth up. Contact: the table pushes
the book up ($\vect N$), the book presses the table down. Walking is the same deal:
push the ground backward; its reaction is the only force driving you forward.
\end{example}

\section{The method: choose, draw, project}

\begin{method}[Solving a dynamics problem]\label{met:g12:newtons-laws:method}
\begin{enumerate}
  \item \emph{System}: name the body (or set of bodies) studied.
  \item \emph{Frame}: choose an inertial frame and axes fitted to the motion ---
        slope axes, or the Frenet frame for a curve (\cref{ch:g12:kinematics-2d}).
  \item \emph{Inventory}: list every force --- the weight, then one force per
        contact, nothing else --- and draw the free-body diagram: one point, one
        arrow per force.
  \item \emph{Project}: write $\sum \vect F = m\vect a$, project it on each axis.
  \item \emph{Solve and check}: units, signs, limiting cases ($\mu \to 0$,
        $\alpha \to 0$, \dots).
\end{enumerate}
\end{method}

\begin{definition}[Normal force and friction]\label{def:g12:newtons-laws:friction}
A surface acts on a body through two components: the \emph{normal force}\index{normal
force} $\vect N$, perpendicular to the surface, and friction $\vect f$, tangential.
Experiment gives the friction laws: while the body does not slide, static friction
adjusts itself up to a ceiling, $f_s \leq \mu_s N$; once it slides, kinetic friction
is $f_k = \mu_k N$, against the sliding. The \emph{coefficients of friction}%
\index{coefficient of friction} $\mu_s$ and $\mu_k$ ($\mu_k \leq \mu_s$) depend only
on the two materials.
\end{definition}

\begin{example}[The incline, twice]\label{ex:g12:newtons-laws:incline}
A block slides down a slope of angle $\alpha$; axes: $x$ down the slope, $y$
perpendicular. On $y$: $N = mg\cos\alpha$. On $x$, frictionless: $ma = mg\sin\alpha$,
so $a = g\sin\alpha$ --- no mass; at $\alpha = 30^\circ$, $a = \qty{4.9}{m/s^2}$.
With kinetic friction: $a = g(\sin\alpha - \mu_k\cos\alpha)$; for $\mu_k = 0.20$,
$a = 9.81\,(0.500 - 0.173) = \qty{3.2}{m/s^2}$. And it stayed put in the first place
only while the needed static friction fit under its ceiling: $\tan\alpha \leq \mu_s$.
\end{example}

\begin{definition}[Apparent weight]\label{def:g12:newtons-laws:apparent}
The \emph{apparent weight}\index{apparent weight} of a body is the normal force
between it and its support --- what a scale under it reads.
\end{definition}

\begin{example}[The elevator]\label{ex:g12:newtons-laws:elevator}
A \qty{70}{kg} passenger stands on a scale in an elevator of vertical acceleration
$a_z$ (upward positive): $N - mg = ma_z$, so $N = m(g + a_z)$. Pulling up at
$a_z = \qty{1.5}{m/s^2}$: \qty{792}{N}, heavy; braking near the top: \qty{582}{N},
light; steady cruise: $mg = \qty{687}{N}$ --- the first law; free fall, $a_z = -g$:
$N = 0$, weightlessness on a scale.
\end{example}

\begin{example}[Two blocks and a pulley]\label{ex:g12:newtons-laws:pulley}
A block $m_1 = \qty{4.0}{kg}$ on a frictionless table is tied by a light rope, over
an ideal pulley, to a hanging block $m_2 = \qty{1.0}{kg}$. The rope transmits the
same tension $T$ at both ends; the blocks share one magnitude of acceleration $a$.
One second law each, along each motion: $m_1 a = T$ and $m_2 a = m_2 g - T$, so
$a = m_2\,g/(m_1 + m_2) = \qty{1.96}{m/s^2}$ and $T = m_1 a = \qty{7.8}{N}$ ---
smaller than $m_2 g = \qty{9.8}{N}$, as it must be for $m_2$ to accelerate downward.
\end{example}

\begin{example}[The flat curve]\label{ex:g12:newtons-laws:curve}
A car of mass $m$ rounds a flat curve of radius $R$ at constant speed $v$. In the
Frenet frame (\cref{ch:g12:kinematics-2d}) the acceleration is purely normal,
$a_N = v^2/R$, aimed at the center. Vertically, $N = mg$; horizontally, the only
centripetal force on offer is the static friction of road on tyres:
$f = mv^2/R \leq \mu_s N = \mu_s mg$, so $v \leq \sqrt{\mu_s\, g R}$, whatever the
mass. For $R = \qty{90}{m}$, dry road ($\mu_s = 0.70$):
$v_{\max} \approx \qty{25}{m/s}$ (\qty{89}{km/h}); on ice the ceiling collapses ---
\cref{exo:g12:newtons-laws:13} banks the road; \cref{exo:g12:newtons-laws:10} swings
the same projection on a wire.
\end{example}

\section{Isolated systems: momentum is conserved}

\begin{proposition}[Conservation of momentum]\label{prop:g12:newtons-laws:conservation}\index{conservation of momentum}
For two bodies whose external forces vanish or compensate, the total momentum
$\vect p_1 + \vect p_2$ is constant, whatever the forces they exert on each other.
\end{proposition}

\begin{proof}
Add the two second laws: $\frac{d}{dt}(\vect p_1 + \vect p_2) = \vect F_{2 \to 1} +
\vect F_{1 \to 2} + \vect F_{\text{ext}}$: the internal pair cancels by the third
law, and $\vect F_{\text{ext}} = \vect 0$ by hypothesis.
\end{proof}

\begin{example}[Recoil]\label{ex:g12:newtons-laws:recoil}
A \qty{60}{kg} skater at rest on smooth ice throws a \qty{4.0}{kg} medicine ball
horizontally at \qty{6.0}{m/s}. Before: $\vect p = \vect 0$; after:
$0 = 60\,V + 4.0 \times 6.0$, so $V = \qty{-0.40}{m/s}$ --- she drifts backward.
Nothing external pushed her: the ball did (third law), and the books stay at zero.
\end{example}

\begin{remark}[How a rocket pushes on nothing]\label{rem:g12:newtons-laws:rocket}
A rocket is a machine for throwing mass. Each second its engines hurl a batch of gas
backward; the gas carries away backward momentum, so the rocket gains the same
amount forward --- a steady thrust, roughly the mass expelled per second times the
ejection speed. The rocket pushes on its own exhaust, not on air: it works best in
empty space.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:1}
Compute the momentum of (a) a \qty{70}{kg} sprinter at \qty{10}{m/s}; (b) a
\qty{1300}{kg} car at \qty{50}{km/h}; (c) an \qty{8.0}{g} bullet at \qty{800}{m/s}.
Rank them. Why does the fastest rank last?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:2}
A \qty{70}{kg} passenger stands on a scale in an elevator. What does it read when
the cabin (a) accelerates upward at \qty{1.2}{m/s^2}; (b) climbs at a constant
\qty{2.0}{m/s}; (c) accelerates downward at \qty{1.2}{m/s^2}? Which law settles (b)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:3}
A block is released on a frictionless slope of angle $30^\circ$. Compute its
acceleration, then its speed after sliding \qty{3.0}{m}. Where did the mass go?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:4}
A \qty{58}{g} tennis ball arrives at \qty{15}{m/s} and is returned along the same
line at \qty{25}{m/s}; contact lasts \qty{5.0}{ms}. Compute the change of momentum
and the average force of the strings; compare that force with the ball's weight.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:5}
A book rests on a table. Name the two forces acting on the book and, for each, its
action--reaction partner (which body? which direction?). Why are the book's weight
and the table's push not partners of each other?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:6}
A \qty{50}{kg} crate sits on a horizontal floor, $\mu_s = 0.45$, $\mu_k = 0.35$.
(a) What horizontal push just starts it? (b) If that push is maintained, what is its
acceleration? (c) What push keeps it moving at constant velocity?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:7}
The same crate is set on an adjustable ramp ($\mu_s = 0.45$, $\mu_k = 0.35$). (a) At
what angle does it start to slide? (b) At $30^\circ$, compute its acceleration.
(c) Once started at the angle of (a), does it keep accelerating? Why?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:8}
A \qty{3.0}{kg} block on a frictionless table is tied over an ideal pulley to a
hanging \qty{2.0}{kg} block. Compute the acceleration and the tension. Why did the
tension have to come out smaller than the hanging weight?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:9}
A curve of radius \qty{50}{m} is flat. Compute a car's maximum speed for
$\mu_s = 0.70$ (dry) and $0.40$ (wet), in \unit{km/h}. Why does the answer hold for
a scooter and a truck alike?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:10}
A \qty{0.30}{kg} bob on a \qty{1.2}{m} wire traces a horizontal circle, the wire
holding a constant $25^\circ$ to the vertical (a conical pendulum). Project the
second law vertically, then along the radius: compute the tension and the speed.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:11}
A \qty{5.0}{kg} parcel rides a scale in a freight elevator; the readings:
\begin{center}\small\begin{tabular}{lccc}
phase & \qtyrange{0}{2}{s} & \qtyrange{2}{8}{s} & \qtyrange{8}{10}{s}\\
scale (\unit{N}) & 54.0 & 49.0 & 44.1
\end{tabular}\end{center}
Compute the acceleration in each phase and describe a possible ride. Could the

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:12}
A \qty{60}{kg} skater at rest on smooth ice throws a \qty{4.0}{kg} ball
horizontally. (a) The ball leaves at \qty{8.0}{m/s} \emph{over the ground}: her
recoil speed? (b) The \qty{8.0}{m/s} is now measured \emph{relative to her}:
recompute both speeds. (c) Which force pushed her backward?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:13}
A highway curve of radius \qty{80}{m} is designed for \qty{20}{m/s} with no help from
friction. Show that the roadway must be banked at the angle $\theta$ given by
$\tan\theta = v^2/(gR)$ and compute it. What must friction do for a car taking the
curve slower --- or faster?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:14}
A rocket's engines eject \qty{250}{kg} of gas per second at \qty{2500}{m/s}. (a) How
much momentum leaves per second --- what thrust do the gases return? (b) The rocket
masses \qty{5.0e4}{kg} at ignition: its weight? its lift-off acceleration? (c) Why
does the engine work just as well in empty space?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:15}
A \qty{4.0}{kg} block on a table ($\mu_s = 0.35$, $\mu_k = 0.25$) is tied over an

(b) For $m_2 = \qty{2.0}{kg}$, compute acceleration and tension. (c) Check that

\end{exercise}

\section{Problem: The Cable Car}

\begin{problem}[{Weekend problem --- certifying the cable car: the tension the slope
demands at rest, the surcharge of the start-up, the relief over the tower, the price
of an emergency stop --- and the one number the placard in the cabin may promise}]
\label{pb:g12:newtons-laws:1}
A mountain cable car --- cabin plus carriage, empty mass $M_0 = \qty{2.0e3}{kg}$ ---
rolls on a track cable inclined at $\alpha = 30^\circ$, pulled by a haul cable
parallel to the slope with tension $\vect T$. Rolling friction is negligible; the
cabin floor stays horizontal throughout. The cabin seats $25$ passengers of average
mass \qty{80}{kg}; you are the certifying engineer. Unless stated otherwise, the
cabin is full: $M = \qty{4.0e3}{kg}$.

\medskip
\noindent\textbf{Part I --- At rest on the slope.}
\begin{enumerate}
  \item Choose the system; say why the mountain frame will do as an inertial frame;
        inventory the three forces, with directions.
  \item Which law governs the wait? Project it along and perpendicular to the slope
        to express $T$ and the track cable's normal force $N$.
  \item Compute $T$ and $N$.
  \item Recompute $T$ for the empty cabin. Why is $T$ proportional to the total mass?
  \item The haul cable snaps, brakes off: show that the runaway acceleration is
        $g \sin\alpha$, loaded or empty, and compute it.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Start-up.} The winch accelerates the full cabin at
$a = \qty{0.60}{m/s^2}$ up the slope until it cruises at $v = \qty{6.0}{m/s}$.
\begin{enumerate}[resume]
  \item Express and compute the tension $T'$ during this phase.
  \item By what fraction does starting up raise the tension above its value at rest?
  \item How long does the phase last, and over what distance?
  \item An \qty{80}{kg} passenger stands on the horizontal floor. Project the second
        law vertically to express and compute her apparent weight; compare with $mg$.
  \item Which force accelerates her horizontally, and how big? Why do passengers lean?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Over the tower.} At a constant \qty{6.0}{m/s}, the
track cable bends over a support tower along a vertical circular arc of radius
$R = \qty{40}{m}$; at the crest the velocity is horizontal.
\begin{enumerate}[resume]
  \item In the Frenet frame at the crest (\cref{ch:g12:kinematics-2d}), give both

\noindent\textbf{Part IV --- Braking, a jump, and the placard.}
\begin{enumerate}[resume]
  \item On the horizontal arrival stretch, an emergency stop takes the full cabin
        from \qty{6.0}{m/s} to rest in \qty{1.5}{s}: read $\sum \vect F = d\vect p/dt$
        as an average over the stop to compute the braking force.
  \item What horizontal force must stop an \qty{80}{kg} passenger, and can
        shoe--floor static friction ($\mu_s = 0.40$) supply it? Conclude.
  \item At the platform, brakes off, the empty cabin hangs free and still; the last
        passenger (\qty{80}{kg}) jumps off horizontally at \qty{2.5}{m/s}
        \emph{relative to the cabin}. Compute the cabin's recoil speed and the
        passenger's speed over the ground.
  \item In one sentence: which propulsion device lives on this recoil principle,
        and what fixes its thrust?
  \item \emph{Certification.} The haul cable breaks at \qty{100}{kN}; the rules cap
        the working tension at one fifth of that. Show the worst case met above is
        the start-up, deduce the maximum total mass, and print the placard: how many
        \qty{80}{kg} passengers may ride? Check the safety factor.
\end{enumerate}
\end{problem}
