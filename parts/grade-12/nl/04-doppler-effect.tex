\chapter{Het Doppler-effect}\label{ch:g12:doppler-effect}

Een ambulance nadert, sirene loeiend, en scheert voorbij --- en op dat
moment glijdt de noot hoorbaar omlaag, hoewel de chauffeur niets
aanraakte. De sirene is eerlijk; de \omterm{def:g10:signals-and-waves:wave}{golf} niet: beweging
perst hem vooruit en rekt hem achteruit. Dit hoofdstuk meet de
persing, en rijdt hetzelfde idee van sirenes naar
\omterm{met:g10:signals-and-waves:echo}{radar}pistolen, naar bloed in een slagader, en
uit tot een planeet verraden door een wiebeling in sterlicht.

\section{Het effect}

\begin{definition}[Doppler-effect]\label{def:g12:doppler-effect:doppler}
Het \emph{Doppler-effect}\index{Doppler-effect} is de verandering van
de ontvangen \omterm{def:g10:signals-and-waves:frequency}{frequentie} van een
\omterm{def:g10:signals-and-waves:wave}{golf} wanneer bron en ontvanger ten opzichte van
elkaar bewegen: de ontvangen \omterm{def:g10:signals-and-waves:frequency}{frequentie} $f'$
is hoger dan de uitgezonden \omterm{def:g10:signals-and-waves:frequency}{frequentie} $f$
terwijl ze naderen, lager terwijl ze zich verwijderen, en gelijk aan
$f$ alleen terwijl hun afstand momentaan ophoudt te veranderen.
\end{definition}

\begin{remark}[Toonhoogte, niet luidheid]\label{rem:g12:doppler-effect:pitch}
Twee dingen veranderen als de ambulance passeert. De
\emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{luidheid}} zwelt en verzwakt omdat de
\omterm{def:g10:signals-and-waves:wave}{golf} zich met de afstand spreidt
(\cref{ch:g12:sound-acoustics}); de
\emph{\omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte}} verschuift door de
snelheid van nadering of verwijdering, en daalt abrupt bij het
passeren. \omterm{def:g12:sound-acoustics:pitch-loudness}{Luidheid} zegt ``hoe ver'';
\omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} zegt ``hoe snel''.
\end{remark}

\section{Een bewegende bron: opeengepakte golffronten}

Het sleutelbeeld: zodra een kam de bron verlaat, is het medium de
eigenaar. Elke kam dijt uit als een cirkel met de
\omterm{def:g10:signals-and-waves:wave}{golf}snelheid $v$ (\cref{ch:g12:mechanical-waves}),
gecentreerd op het punt waar hij werd \emph{uitgezonden} --- het
medium weet noch geeft erom dat de bron verder is gegaan.

\begin{proposition}[Bewegende bron]\label{prop:g12:doppler-effect:moving-source}
Een bron zendt kammen uit met \omterm{def:g10:signals-and-waves:period}{periode} $T$
(\omterm{def:g10:signals-and-waves:frequency}{frequentie} $f = 1/T$) terwijl hij in een
rechte lijn beweegt met snelheid $v_s < v$ door een medium waarin de
\omterm{def:g10:signals-and-waves:wave}{golf} reist met snelheid $v$. Een waarnemer in
rust in het medium ontvangt
\[
\lambda' = (v \mp v_s)\,T, \qquad f' = f\,\frac{v}{v \mp v_s},
\]
de bovenste tekens voor een naderende bron, de onderste voor een
zich verwijderende.
\end{proposition}

\begin{proof}
Tussen twee kammen schuift de eerste $vT$ vooruit terwijl de bron
$v_s T$ opschuift. Vooruit wordt de tweede kam daarom $v_s T$
\emph{dichter} bij de eerste uitgezonden, $\lambda' = (v - v_s)T$;
achter wordt dezelfde $v_s T$ toegevoegd. De kammen vegen dan langs
een stilstaande waarnemer met de mediaumsnelheid $v$, één elke
$\lambda'/v$ seconden, dus $f' = v/\lambda' = f v/(v \mp v_s)$.
\end{proof}

\begin{example}[De ambulance, in getallen]\label{ex:g12:doppler-effect:siren}
Een sirene bij $f = \qty{700}{Hz}$ nadert met $v_s = \qty{25}{m/s}$;
neem $v = \qty{340}{m/s}$ in lucht (\cref{ch:g12:sound-acoustics}).
Naderend: $f' = 700 \times 340/315 \approx \qty{756}{Hz}$;
verwijderend: $f' = 700 \times 340/365 \approx \qty{652}{Hz}$. Het
passeren laat de \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} met
\qty{104}{Hz} dalen --- een verhouding $\approx 1.16$, ongeveer twee
en een halve halve toon, onmiskenbaar voor elk oor.
\end{example}

\begin{remark}[Wat beweging niet verandert]\label{rem:g12:doppler-effect:speed-unchanged}
De \omterm{def:g10:signals-and-waves:wave}{golf} reist nog steeds met $v$: alleen het medium
bepaalt de snelheid (\cref{ch:g12:mechanical-waves}). Beweging van de
bron perst \emph{\omterm{def:g12:mechanical-waves:wavelength}{golflengten}}, nooit
snelheden. En de persing heeft een limiet: als $v_s \to v$ stapelen
de kammen vooruit op elkaar en $f' \to \infty$ --- de file van
\omterm{def:g12:mechanical-waves:wavefront}{golffronten} die een supersonisch
vliegtuig als schok meesleept.
\end{remark}

\section{Een bewegende waarnemer, en de slow-motion-shortcut}

\begin{proposition}[Bewegende waarnemer]\label{prop:g12:doppler-effect:moving-observer}
Een waarnemer beweegt met snelheid $v_o$ recht naar (of weg van) een
bron in rust in het medium. De ontvangen
\omterm{def:g10:signals-and-waves:frequency}{frequentie} is
\[ f' = f\left(1 \pm \frac{v_o}{v}\right) \quad (+\ \text{naderend},\ -\ \text{verwijderend}). \]
\end{proposition}

\begin{proof}
Het \omterm{def:g10:signals-and-waves:wave}{golf}patroon zelf is ongestoord: kammen met
afstand $\lambda = vT$ reizen met $v$. Een waarnemer die erop af
rent ontmoet kammen met de relatieve snelheid $v + v_o$, één elke
$\lambda/(v + v_o)$ seconden: $f' = (v + v_o)/\lambda = f(1 + v_o/v)$.
Wegrennen: $v - v_o$.
\end{proof}

\begin{definition}[Radiale snelheid]\label{def:g12:doppler-effect:radial}
De \emph{radiale snelheid}\index{radiale snelheid} $v_r$ van een bron
ten opzichte van een waarnemer is de component van hun relatieve
snelheid langs de lijn die hen verbindt --- de snelheid waarmee de
afstand krimpt of groeit. Alleen deze component verschuift de
\omterm{def:g10:signals-and-waves:frequency}{frequentie}; beweging \emph{dwars} op de
zichtlijn geeft in deze orde geen verschuiving.
\end{definition}

\begin{proposition}[Trage beweging: één formule voor iedereen]\label{prop:g12:doppler-effect:slow}
Wanneer de radiale snelheid klein is, $v_r \ll v$, doet het er niet
meer toe wie beweegt:
\[
\frac{\Delta f}{f} = \frac{f' - f}{f} \approx +\frac{v_r}{v} \ \text{(nadering)},
\qquad
\frac{\Delta f}{f} \approx -\frac{v_r}{v} \ \text{(verwijdering)}.
\]
\end{proposition}

\begin{proof}
De waarnemerformules zijn al precies $1 \pm v_o/v$. Voor de bron, schrijf
$x = v_s/v$: de identiteit $\frac{1}{1 - x} = 1 + x + \frac{x^2}{1 - x}$
toont dat $f'/f$ van $1 + x$ verschilt met een term van orde $x^2$ ---
voor $x = 0.1$ een $1\%$-correctie op een $10\%$-verschuiving. Tot
eerste orde storten alle vier formules samen op $1 \pm v_r/v$.
\end{proof}

\begin{method}[Een passage lezen]\label{met:g12:doppler-effect:pass}
Een opname van een passerende bron toont twee plateaus: $f_1$
(nadering) en $f_2$ (verwijdering). Dan
\[
f = \frac{2 f_1 f_2}{f_1 + f_2}, \qquad
v_s = v\,\frac{f_1 - f_2}{f_1 + f_2}.
\]
Inderdaad $f_1 + f_2 = \frac{2fv^2}{v^2 - v_s^2}$ en
$f_1 - f_2 = \frac{2fv\,v_s}{v^2 - v_s^2}$: deel voor $v_s$, en bereken
$2f_1 f_2/(f_1 + f_2)$ voor $f$. Twee
\omterm{def:g10:signals-and-waves:frequency}{frequentie}-aflezingen leveren zowel de ware
\omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} als de snelheid --- geen
stopwatch, geen liniaal.
\end{method}

\section{Echo's die snelheid meten}

Kaats een \omterm{def:g10:signals-and-waves:wave}{golf} terug van een bewegend doel en het
\omterm{def:g12:doppler-effect:doppler}{Doppler-effect} treft tweemaal.

\begin{proposition}[Reflectie verdubbelt de verschuiving]\label{prop:g12:doppler-effect:double}
Een \omterm{def:g10:signals-and-waves:wave}{golf} van \omterm{def:g10:signals-and-waves:frequency}{frequentie}
$f$ wordt frontaal op een reflector gestuurd die met snelheid
$u \ll v$ nadert. De \omterm{rem:g10:signals-and-waves:honest}{echo} keert terug met
\[ f' = f\,\frac{v + u}{v - u}, \qquad \text{dus} \qquad \Delta f \approx \frac{2u}{v}\,f . \]
\end{proposition}

\begin{proof}
Twee verschuivingen in serie. Als bewegende \emph{waarnemer}
ontvangt de reflector $f(1 + u/v)$; opnieuw uitzendend wat hij
ontvangt is hij nu een bewegende \emph{bron}, dus arriveert de
\omterm{rem:g10:signals-and-waves:honest}{echo} bij
$f' = f(1 + u/v)/(1 - u/v) \approx f(1 + 2u/v)$ voor $u \ll v$.
\end{proof}

\begin{example}[Het radarpistool]\label{ex:g12:doppler-effect:radar}
Een verkeers\omterm{met:g10:signals-and-waves:echo}{radar} zendt microgolven uit bij
$f = \qty{24.15}{GHz}$ ($v = c = \qty{3.00e8}{m/s}$) en superponeert
de \omterm{rem:g10:signals-and-waves:honest}{echo} op de uitgaande
\omterm{def:g10:signals-and-waves:wave}{golf}; de zweving (\cref{ch:g12:sound-acoustics})
\omterm{def:g10:signals-and-waves:sound}{klinkt} direct $\Delta f = 2uf/c \approx
\qty{161}{Hz}$ per \unit{m/s} autosnelheid --- een triviaal meetbare
audio\omterm{def:g10:signals-and-waves:frequency}{frequentie} gesneden uit een
\qty{24}{GHz}-drager. De factor $2$ is niet optioneel: hem vergeten
vleit elke bestuurder met de helft.
\end{example}

\begin{example}[Doppler-ultrageluid]\label{ex:g12:doppler-effect:ultrasound}
Een medische sonde stuurt $f = \qty{5.0}{MHz}$
\omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid} in een slagader
($v \approx \qty{1540}{m/s}$ in weefsel); rode bloedcellen
reflecteren het. Bloed bij $u = \qty{0.50}{m/s}$ retourneert
$\Delta f = 2 \times 0.50 \times \num{5.0e6}/1540 \approx
\qty{3.2}{kHz}$ --- een \emph{hoorbare} zweving: de cardioloog hoort
letterlijk het bloed versnellen bij elke hartslag, en meet $u$ uit
$\Delta f$.
\end{example}

\section{Doppler in sterlicht}

Licht is een \omterm{def:g10:signals-and-waves:wave}{golf}
(\cref{ch:g12:light-as-wave}), zijn \omterm{def:g12:sound-acoustics:timbre}{spectrum}
gestreept met scherpe lijnen bij \omterm{def:g12:mechanical-waves:wavelength}{golflengten}
vastgelegd door elk chemisch element
(\cref{ch:g10:light-spectra}) --- een laboratorium-geijkte liniaal
op elke ster gedrukt.

\begin{definition}[Roodverschuiving en blauwverschuiving]\label{def:g12:doppler-effect:redshift}
Wanneer de \omterm{def:g10:light-spectra:emission}{spectraallijnen} van een bron
verschijnen bij \omterm{def:g12:mechanical-waves:wavelength}{golflengten}
\emph{langer} dan in het laboratorium, is het
\omterm{def:g12:sound-acoustics:timbre}{spectrum}
\emph{roodverschoven}\index{roodverschuiving}; wanneer ze
\emph{korter} verschijnen, is het
\emph{blauwverschoven}\index{blauwverschuiving} --- verschoven naar
het rode of blauwe uiteinde van het
\omterm{def:g10:light-spectra:wavelength}{zichtbare spectrum}.
\end{definition}

\begin{proposition}[Doppler-verschuiving van licht]\label{prop:g12:doppler-effect:light}
Voor een bron die zich verwijdert met radiale snelheid $v_r \ll c$
wordt elke \omterm{def:g12:mechanical-waves:wavelength}{golflengte} waargenomen
uitgerekt met
\[ \frac{\Delta \lambda}{\lambda} = \frac{\lambda' - \lambda}{\lambda} \approx \frac{v_r}{c} \quad \text{(roodverschuiving)}, \]
en samengedrukt met hetzelfde bedrag, $\Delta\lambda/\lambda \approx -v_r/c$,
voor een naderende bron (\omterm{def:g12:doppler-effect:redshift}{blauwverschuiving}).
\end{proposition}
\admitted

\begin{remark}[Waar de eerlijke formule leeft]\label{rem:g12:doppler-effect:honest}
Licht heeft geen medium nodig, dus de
\omterm{def:g10:signals-and-waves:sound}{geluid}-afleiding kan niet worden gekopieerd ---
er is geen ``in rust in het medium''. De exacte formule komt uit de
speciale relativiteit, later dit jaar
(\cref{ch:g12:special-relativity}); de correcties zijn van orde
$(v_r/c)^2$, verwaarloosbaar voor vliegtuigen, sterren en nabije
sterrenstelsels. Hieronder gebruiken we de trage-bewegingformule met
een gerust geweten.
\end{remark}

\begin{example}[De ontsnapping van een ster wegen]\label{ex:g12:doppler-effect:vega}
In het \omterm{def:g12:sound-acoustics:timbre}{spectrum} van de heldere ster Vega wordt
de waterstoflijn van laboratorium\omterm{def:g12:mechanical-waves:wavelength}{golflengte}
\qty{656.30}{nm} gemeten bij \qty{656.27}{nm}:
$\Delta\lambda = \qty{-0.03}{nm}$, een
\omterm{def:g12:doppler-effect:redshift}{blauwverschuiving}, dus nadert Vega met
$v_r = c\,|\Delta\lambda|/\lambda \approx \qty{1.4e4}{m/s}$ ---
\qty{14}{km/s}, afgelezen van een verschuiving van één deel in
twintigduizend.
\end{example}

\begin{example}[Wiebelende sterren: dubbelsterren en exoplaneten]\label{ex:g12:doppler-effect:wobble}
Wanneer twee sterren om elkaar \omterm{def:g10:universal-gravitation:satellite}{cirkelen},
zwaaien hun lijnen \omterm{def:g10:signals-and-waves:period}{periodiek} blauw en rood ---
een \emph{spectroscopische dubbelster}\index{spectroscopische dubbelster},
\omterm{def:g10:universal-gravitation:satellite}{banen} gemeten zonder ooit het paar te
scheiden. Een planeet doet hetzelfde met zijn ster, zwak: beide
cirkelen om hun gemeenschappelijke
\omterm{def:g11:forces-and-motion:com}{zwaartepunt}, zodat de
\omterm{def:g12:doppler-effect:radial}{radiale snelheid} van de ster oscilleert met
enkele tientallen \unit{m/s} --- een verschuiving
$\Delta\lambda/\lambda \sim 10^{-7}$. Dat detecteren is hoe de eerste
planeet rond een zonachtige ster werd gevonden; het weekendprobleem
speelt die ontdekking opnieuw, getallen en al.
\end{example}

\begin{remark}[Roodverschuivingen van sterrenstelsels]\label{rem:g12:doppler-effect:expansion}
De spectra van verre sterrenstelsels zijn allemaal
\emph{rood}verschoven, en hoe verder, hoe meer verschoven: de
afstanden van het heelal rekken. Voor nabije sterrenstelsels leest
$v_r = c\,\Delta\lambda/\lambda$ de verwijderingssnelheid af; voor de
verste groeien verschuivingen uit wat de trage-bewegingformule
eerlijk aankan, en de boekhouding wordt vanaf
\cref{ch:g12:special-relativity} opnieuw opgenomen, en in de
universitaire volumes.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:1}
Een brandweerauto-sirene zendt uit bij \qty{700}{Hz} en rijdt met
\qty{25}{m/s} ($v = \qty{340}{m/s}$). Bereken de
\omterm{def:g10:signals-and-waves:frequency}{frequentie} gehoord door een voetganger (a)
ervoor; (b) erachter. (c) Wat hoort de chauffeur?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:2}
Een vriend beweert: ``de
\omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} daalt als de ambulance
wegrijdt omdat het \omterm{def:g10:signals-and-waves:sound}{geluid} met de afstand
zwakker wordt.'' Ontwar de twee verwarde effecten, en zeg waarvan
elk afhangt.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:3}
Een fietser rijdt met \qty{6.0}{m/s} recht op een stilstaande sirene
die bij \qty{700}{Hz} uitzendt. Welke
\omterm{def:g10:signals-and-waves:frequency}{frequentie} horen ze? En recht
wegrijdend? Welke formule geldt, en waarom niet die van de
bewegende bron?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:4}
Een treinhoorn zendt uit bij \qty{400}{Hz} terwijl de trein met
\qty{30}{m/s} rijdt. Bereken de
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} voor de trein, erachter, en
in rust. Welke waarnemer ontvangt welke?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:doppler-effect:5}
Een telefoon die een \qty{440}{Hz}-toon speelt wordt door een
hardloper met \qty{5.0}{m/s} gedragen. Gebruik de
trage-bewegingformule om de verschuiving te schatten gehoord door
iemand die de hardloper nadert. Vergelijk met een halve toon
(ongeveer $6\%$ in \omterm{def:g10:signals-and-waves:frequency}{frequentie}): zou een
musicus het merken?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:6}
Een vleermuis vliegt met \qty{6.0}{m/s} recht op een muur, uitzendend
bij \qty{50.0}{kHz}. Toon aan dat de
\omterm{rem:g10:signals-and-waves:honest}{echo} die hij hoort terugkeert bij
$f' = f(v + v_b)/(v - v_b)$, bereken dan $f'$ en de verschuiving.
Waarom is dit de radarpistool-formule in vermomming?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:7}
Een \omterm{met:g10:signals-and-waves:echo}{radar}pistool bij \qty{24.15}{GHz} meet
een zweving van \qty{4.0}{kHz} van een naderende auto. Vind de
snelheid van de auto in \unit{m/s} en \unit{km/h}. Welke zweving zou
een auto bij precies \qty{50}{km/h} geven?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:8}
Een \qty{5.0}{MHz} Doppler-\omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid}sonde
gericht langs een slagader ($v = \qty{1540}{m/s}$ in weefsel) hoort
een zweving van \qty{2.6}{kHz}. Vind de bloedsnelheid. Waarom moet de
sonde \emph{langs} de stroom worden gericht in plaats van loodrecht
eraan?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:9}
Neem $f = \qty{1000}{Hz}$, $v = \qty{340}{m/s}$ en een
naderingssnelheid van \qty{34}{m/s}. Bereken $f'$ exact wanneer (a)
de bron beweegt, (b) de waarnemer beweegt. Vergelijk beide met de
trage-bewegingvoorspelling $f(1 + v_r/v)$ en leg de grootte van de
onenigheid uit.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:10}
De opname van een scooter die een microfoon passeert toont plateaus
bij \qty{465}{Hz} en \qty{415}{Hz}. Met
\cref{met:g12:doppler-effect:pass}, vind de ware
\omterm{def:g10:signals-and-waves:frequency}{frequentie} van de claxon en de
scootersnelheid in \unit{km/h}. Waarom is de ware
\omterm{def:g10:signals-and-waves:frequency}{frequentie} \emph{niet} hun
gemiddelde?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:doppler-effect:11}
In het \omterm{def:g12:sound-acoustics:timbre}{spectrum} van een ster wordt de
waterstoflijn van laboratorium\omterm{def:g12:mechanical-waves:wavelength}{golflengte}
\qty{656.3}{nm} waargenomen bij \qty{656.5}{nm}.
\omterm{def:g12:doppler-effect:redshift}{Roodverschuiving} of
\omterm{def:g12:doppler-effect:redshift}{blauwverschuiving}? Bereken de
\omterm{def:g12:doppler-effect:radial}{radiale snelheid} van de ster en de
richting.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:12}
Een ster wiebelt met \qty{55}{m/s} door een ongezien planeet. Bereken
de \omterm{def:g10:signals-and-waves:amplitude}{amplitude} van de
\omterm{def:g12:mechanical-waves:wavelength}{golflengte}-zwaai van een
\qty{500}{nm}-lijn, als lengte en als breukdeel van $\lambda$.
\omterm{def:g10:orders-of-magnitude:oom}{Orde van grootte}: waarom moest
planeetjacht wachten op spectrograven stabiel tot één deel in
$10^{7}$?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:13}
Elke lijn in het \omterm{def:g12:sound-acoustics:timbre}{spectrum} van een
sterrenstelsel is $2.4\%$ uitgerekt. Bereken zijn
verwijderingssnelheid. Astronomen vinden dat zulke snelheden
evenredig met de afstand groeien, voor elke richting van de hemel:
welk beeld suggereert dit? Is de trage-bewegingformule hier nog
betrouwbaar?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:14}
De Zon roteert: één rand van zijn schijf nadert ons met ongeveer
\qty{2.0}{km/s} terwijl de andere even snel zich verwijdert. Wat
doet dit met de \qty{656.3}{nm}-lijn in licht van de hele schijf ---
verschuiven of verbreden? Bereken het effect in \unit{nm}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:doppler-effect:15}
Een politieauto met \qty{40}{m/s}, sirene bij \qty{700}{Hz}, achtervolgt
een vrachtwagen die met \qty{30}{m/s} in dezelfde richting rijdt.
(a) Rechtvaardig $f' = f(v - v_o)/(v - v_s)$ voor de
vrachtwagenchauffeur. (b) Bereken $f'$. (c) Toon aan dat als de
vrachtwagen de snelheid van de auto matcht de verschuiving
verdwijnt, en zeg waarom dat zo hoort.
\end{exercise}

\section{Probleem: De jacht op een exoplaneet}

\begin{problem}[{Weekendprobleem --- de jacht op een exoplaneet: een
treinhoorn ijkt het oor, een radarpistool oefent de dubbele
verschuiving, en zondagavond heeft het spectrum van een ster,
wiebelend met één deel in vijf miljoen, een planeet gewogen die
niemand ooit heeft gezien}]
\label{pb:g12:doppler-effect:1}
Een astronomieclub besteedt een weekend aan één idee --- de
Doppler-verschuiving --- geoefend op Aarde, dan gericht op een
zonachtige ster. Data: $v = \qty{340}{m/s}$ voor
\omterm{def:g10:signals-and-waves:sound}{geluid} in lucht, $c = \qty{3.00e8}{m/s}$,
$G = \qty{6.67e-11}{N.m^2/kg^2}$; de ster heeft massa
$M = \qty{2.0e30}{kg}$. Toegegeven, uit de mechanica van circulaire
\omterm{def:g10:universal-gravitation:satellite}{banen}
(\cref{ch:g12:satellites-kepler}): een planeet van massa $m$ op een
circulaire \omterm{def:g10:universal-gravitation:satellite}{baan} van straal $r$
heeft \omterm{def:g10:signals-and-waves:period}{periode} en snelheid gegeven door
$r^3 = \frac{G M P^2}{4\pi^2}$ en $V = \frac{2\pi r}{P}$, en de
ster, cirkelend om het gemeenschappelijke
\omterm{def:g11:forces-and-motion:com}{zwaartepunt}, wiebelt met snelheid
$K = \frac{m}{M}\,V$.

\medskip
\noindent\textbf{Deel I --- Vrijdag: de spoorwegovergang.}
Een trein passeert met constante snelheid, hoorn aan; een telefoon
registreert $f_1 = \qty{471}{Hz}$ terwijl hij nadert,
$f_2 = \qty{411}{Hz}$ nadat hij is gepasseerd.
\begin{enumerate}
  \item Leg uit, met het
        \omterm{def:g12:mechanical-waves:wavefront}{opeengepakte-golffront}-beeld, waarom
        $f_1 > f_2$ hoewel de hoorn nooit verandert.
  \item Schrijf de twee vergelijkingen die $f_1$, $f_2$ verbinden met
        de ware \omterm{def:g10:signals-and-waves:frequency}{frequentie} $f$ en
        treinsnelheid $u$.
  \item Toon aan dat $u = v\,\dfrac{f_1 - f_2}{f_1 + f_2}$ en bereken
        die in \unit{m/s} en \unit{km/h}.
  \item Toon aan dat $f = \dfrac{2 f_1 f_2}{f_1 + f_2}$ en bereken
        hem. Waarom ligt $f$ iets \emph{onder} het gemiddelde van
        $f_1$ en $f_2$?
  \item Bereken de \omterm{def:g12:mechanical-waves:wavelength}{golflengten} voor en
        achter de trein, en de rust\omterm{def:g12:mechanical-waves:wavelength}{golflengte}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Zaterdag: het
\omterm{met:g10:signals-and-waves:echo}{radar}pistool.}
Een verkeerspatrouille demonstreert zijn
\omterm{met:g10:signals-and-waves:echo}{radar}: $f = \qty{24.125}{GHz}$, en het
pistool leest de zweving tussen de uitgaande
\omterm{def:g10:signals-and-waves:wave}{golf} en de
\omterm{rem:g10:signals-and-waves:honest}{echo}.
\begin{enumerate}[resume]
  \item Leg uit waarom de \omterm{rem:g10:signals-and-waves:honest}{echo} van een
        bewegende auto \emph{tweemaal} wordt verschoven --- noem de
        rol die de auto in elke verschuiving speelt.
  \item Toon aan dat voor een auto met snelheid $u \ll c$ de zweving
        $\Delta f \approx \dfrac{2u}{c}\,f$ is.
  \item Een auto retourneert $\Delta f = \qty{3.55}{kHz}$: bereken $u$.
  \item Zet om naar \unit{km/h} en vergelijk met de
        \qty{80}{km/h}-limiet daar geplaatst.
  \item Hoeveel \omterm{def:g10:signals-and-waves:frequency}{hertz} zweving
        corresponderen met \qty{1}{km/h}? Commentaar: wat maakt zo'n
        piepkleine relatieve verschuiving van een
        \qty{24}{GHz}-\omterm{def:g10:signals-and-waves:wave}{golf} makkelijk te meten?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Zondag: de ster die wiebelt.}
De club downloadt de gemeten
\omterm{def:g12:doppler-effect:radial}{radiale snelheid} van een zonachtige
ster, nacht na nacht, hieronder uitgezet.

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis, width=8.8cm, height=4.6cm,
    xlabel={$t$ (dagen)}, ylabel={$v_r$ (\unit{m/s})},
    xmin=0, xmax=8.4, ymin=-72, ymax=72,
    xtick={0,2.1,4.2,6.3,8.4},
    xticklabels={$0$,$2.1$,$4.2$,$6.3$,$8.4$}, ytick={-55,0,55}]
  \addplot[dashed, omExo, domain=0:8.4, samples=2] {55};
  \addplot[dashed, omExo, domain=0:8.4, samples=2] {-55};
  \addplot[omDef, thick, domain=0:8.4, samples=140] {55*sin(360*x/4.2)};
\end{axis}
\end{tikzpicture}

{\small De \omterm{def:g12:doppler-effect:radial}{radiale snelheid} van de ster over
twaalf nachten: een schone sinus --- de handtekening van een lichaam
op een circulaire \omterm{def:g10:universal-gravitation:satellite}{baan}, van opzij
gezien.}
\end{omfigure}

\begin{enumerate}[resume]
  \item Wat wordt er eigenlijk, nacht na nacht, in het
        \emph{\omterm{def:g12:sound-acoustics:timbre}{spectrum}} van de ster gemeten?
        Geef de relatie die het omzet in $v_r$.
  \item Lees de \omterm{def:g10:signals-and-waves:amplitude}{amplitude} $K$ en de
        \omterm{def:g10:signals-and-waves:period}{periode} $P$ van de wiebeling van de
        curve af.
  \item Bereken de corresponderende zwaai $\Delta\lambda$ van een
        \qty{550}{nm}-lijn, en $\Delta\lambda/\lambda$. Vergelijk met
        Deel I: hoeveel moeilijker is deze meting dan die van de
        trein?
  \item Waarom doet een ongezien planeet de ster überhaupt wiebelen?
        Wat geeft de sinusoïdale vorm aan over de
        \omterm{def:g10:universal-gravitation:satellite}{baan}?
  \item Waarom meet deze methode alleen het \emph{radiale} deel van
        de sterbeweging, en wat zou men zien als de
        \omterm{def:g10:universal-gravitation:satellite}{baan} van voren naar ons
        lag?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Zondagavond: het onzichtbare wegen.}
Neem de \omterm{def:g10:universal-gravitation:satellite}{baan} circulair en van opzij,
zodat $K$ de volle banensnelheid van de ster is.
\begin{enumerate}[resume]
  \item Uit $P$ en de toegegeven relatie $r^3 = G M P^2/(4\pi^2)$,
        bereken de banenstraal $r$ van de planeet. Vergelijk met de
        Aarde--Zon-afstand, \qty{1.5e11}{m}.
  \item Bereken de banensnelheid $V = 2\pi r/P$ van de planeet.
  \item Uit $K = (m/M)\,V$, bereken de massa $m$ van de planeet.
  \item Vergelijk $m$ met Jupiter ($\qty{1.9e27}{kg}$) en met Aarde
        ($\qty{6.0e24}{kg}$), en $r$ met de Zon--Mercurius-afstand
        ($\qty{5.8e10}{m}$). Wat voor wereld is dit?
  \item Als de \omterm{def:g10:universal-gravitation:satellite}{baan} gekanteld was in
        plaats van van opzij, zou de ware massa groter of kleiner
        zijn dan jouw waarde? Concludeer in één zin: wat leverde een
        \omterm{def:g10:signals-and-waves:period}{periodieke} verschuiving van één deel
        in vijf miljoen zojuist?
\end{enumerate}
\end{problem}
