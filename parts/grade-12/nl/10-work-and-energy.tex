\chapter{Arbeid en mechanische energie}\label{ch:g12:work-and-energy}

Kantel een knikker in een saladekom: hij duikt, schiet voorbij de
bodem, klimt tot zijn starthoogte, aarzelt, keert om --- en elke
omkeer is te voorspellen uit één kromme (het profiel van de kom) en
één horizontale lijn (de \omterm{def:g10:energy-conservation:energy}{energie} van de knikker), zonder
bewegingsvergelijking. Vorig jaar prijzen we
\omterm{def:g10:inertia:force}{krachten} op rechte lijnen
(\cref{ch:g11:work-of-force}) en balanceerden kinetische tegen
\omterm{def:g12:work-and-energy:ep}{potentiële energie}
(\cref{ch:g11:mechanical-energy}); dit jaar gaat de boekhouding
overal heen: gekromde banen, veren, landschappen in één oogopslag.

\section{Arbeid langs een willekeurige baan}

\begin{definition}[Arbeid langs een baan]\label{def:g12:work-and-energy:path}
Laat een \omterm{def:g10:inertia:force}{kracht} $\vect F$ werken op een lichaam dat langs
een willekeurige baan van $A$ naar $B$ beweegt: knip de baan in
verplaatsingen $\vect{\dd\ell}$ zo kort dat elk recht is en $\vect F$
nauwelijks verandert. Elke stap verdient de kleine
\omterm{def:g11:work-of-force:work}{arbeid} $\scal{F}{\dd\ell} = F\,\dd\ell\cos\theta$
(\cref{ch:g11:work-of-force}), en de
\emph{arbeid langs de baan}\index{arbeid (van een kracht)!langs een baan}
is hun som, $W_{AB}(\vect F) = \sum \scal{F}{\dd\ell}$ --- die op een
rechte baan met constante \omterm{def:g10:inertia:force}{kracht} ineenstort tot de
formule van vorig jaar $F \times AB \times \cos\theta$.
\end{definition}

\begin{proposition}[Arbeid van het gewicht, willekeurige baan]\label{prop:g12:work-and-energy:weight}
Langs \emph{elke} baan van $A$ (hoogte $z_A$) naar $B$ (hoogte
$z_B$) verricht het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een massa $m$ de
\omterm{def:g11:work-of-force:work}{arbeid} $W_{AB}(\vect P) = mg\,(z_A - z_B)$: de daling
telt, nooit de route.
\end{proposition}

\begin{proof}
Elke korte stap levert $mg \times (\text{kleine daling})$ (de
rechte-segment-berekening van vorig jaar); de dalingen telescoperen
tot $z_A - z_B$. Het gekromde geval, toen toegegeven, is geleverd.
\end{proof}

\begin{definition}[Conservatieve en niet-conservatieve krachten]\label{def:g12:work-and-energy:conservative}
Een \omterm{def:g10:inertia:force}{kracht} is \emph{conservatief}\index{conservatieve kracht}
als haar \omterm{def:g11:work-of-force:work}{arbeid} tussen twee punten langs elke baan
hetzelfde is (nul op elke rondreis),
\emph{niet-conservatief}\index{niet-conservatieve kracht} anders.
\omterm{def:g10:universal-gravitation:weight}{Gewicht} en veer\omterm{def:g10:inertia:force}{kracht} (volgende
sectie): conservatief. Glijdende \omterm{def:g10:inertia:inventory}{wrijving} is het
niet: tegengesteld aan de beweging verricht ze $W = -fL$ op een baan
van lengte $L$ --- een tol per \omterm{def:g10:orders-of-magnitude:unit}{meter}, nooit
terugbetaald.
\end{definition}

\begin{example}[Twee paden naar de hut]\label{ex:g12:work-and-energy:trails}
Een wandelaar van \qty{75}{kg} wint \qty{300}{m} via een steil pad van
\qty{800}{m} of \qty{2.4}{km} haarspeldbochten. Het
\omterm{def:g10:universal-gravitation:weight}{gewicht} rekent
$-75 \times 9.81 \times 300 \approx \qty{-2.2e5}{J}$ op beide; een
\omterm{def:g10:inertia:inventory}{wrijvings}\omterm{def:g10:inertia:force}{kracht} van
\qty{30}{N} rekent \qty{-2.4e4}{J} op de ene, \qty{-7.2e4}{J} op de
andere. Het \omterm{def:g10:universal-gravitation:weight}{gewicht} betaalt terug bij de daling;
de rekening van \omterm{def:g10:inertia:inventory}{wrijving} is als warmte verdwenen, beide
kanten op.
\end{example}

\section{Potentiële energie}

\begin{definition}[Potentiële energie]\label{def:g12:work-and-energy:ep}
Aan elke \omterm{def:g12:work-and-energy:conservative}{conservatieve kracht} koppelt men een
\emph{potentiële energie}\index{potentiële energie!uit conservatieve arbeid}
$E_p$, een getal dat alleen van de positie afhangt, gedefinieerd door
$W_{AB} = E_p(A) - E_p(B)$: geleverde \omterm{def:g11:work-of-force:work}{arbeid} is
uitgegeven potentiële \omterm{def:g10:energy-conservation:energy}{energie}. $E_p = 0$ vastleggen
in een gekozen referentiepunt legt haar overal vast; de keuze
verschuift alle waarden met een constante die uit elk verschil
wegvalt. Voor het \omterm{def:g10:universal-gravitation:weight}{gewicht} levert dat de
$E_p = mgz$ van vorig jaar terug, nu geldig langs elke baan.
\omterm{def:g10:inertia:inventory}{Wrijving} kan geen $E_p$ bezitten: een
route-afhankelijke rekening is geen verschil van eindpuntgetallen ---
alleen \omterm{def:g12:work-and-energy:conservative}{conservatieve krachten} slaan terugwinbaar
op, precies wat de naam conserveert.
\end{definition}

\begin{proposition}[Elastische potentiële energie]\label{prop:g12:work-and-energy:elastic}
Een veer van \omterm{def:g12:oscillators-and-time:spring}{veerconstante} $k$ (\unit{N/m}), over $x$
uitgerekt of samengedrukt, trekt terug met de
\omterm{def:g10:inertia:force}{kracht} $F = kx$
(\cref{ch:g12:oscillators-and-time}); de
\omterm{def:g11:work-of-force:work}{arbeid} die ze levert bij terugkeer naar rust ---
haar \emph{elastische potentiële energie}\index{potentiële energie!elastisch}
--- is $E_p = \tfrac12\, k x^2$.
\end{proposition}

\begin{proof}
De \omterm{def:g12:oscillators-and-time:pendulum}{terugdrijvende kracht} werkt langs de beweging, dus
$\sum F\,\dd\ell$ is de \emph{oppervlakte onder de $F$--$x$-grafiek}:
een driehoek met basis $x$ en hoogte $kx$, oppervlakte
$\tfrac12 x \times kx$. De \omterm{def:g10:inertia:force}{kracht} bij elke rek is
heen en terug hetzelfde: \omterm{def:g11:work-of-force:work}{arbeid} vastgelegd door de
eindpunten, veer\omterm{def:g10:inertia:force}{kracht}
\omterm{def:g12:work-and-energy:conservative}{conservatief}, $E_p$ goed gedefinieerd.
\end{proof}

\section{De twee energiestellingen}

\begin{theorem}[Kinetische-energiestelling]\label{thm:g12:work-and-energy:ketheorem}\index{kinetische-energiestelling}
Voor elke beweging van een lichaam van massa $m$ van $A$ naar $B$ ---
gekromd, gelust, driedimensionaal ---
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
waarbij de som over alle \omterm{def:g10:inertia:force}{krachten} op het lichaam loopt.
\end{theorem}

\begin{proof}
Omdat $v^2 = \scal{v}{v}$ geeft de productregel van de wiskunde van
dit jaar de afgeleide van $v^2$ als $2\,\scal{a}{v}$, dus
$\dd E_k/\dd t = m\,\scal{a}{v} = \scal{F_{\text{tot}}}{v}$ door
\omterm{thm:g12:newtons-laws:second}{Newtons tweede wet}
(\cref{ch:g12:newtons-laws}). In een kort $\dd t$ beweegt het lichaam
$\vect{\dd\ell} = \vect v\,\dd t$: $E_k$ wint de kleine
\omterm{def:g11:work-of-force:work}{arbeid} van \cref{def:g12:work-and-energy:path}; de stappen
optellen is de \omterm{def:g11:work-of-force:work}{arbeiden} optellen --- de toegegeven
uitspraak van vorig jaar, nu volledig verdiend.
\end{proof}

\begin{proposition}[Momentaan vermogen]\label{prop:g12:work-and-energy:power}
Op elk moment levert een \omterm{def:g10:inertia:force}{kracht} $\vect F$ op een lichaam
met snelheid $\vect v$ het \omterm{def:g11:work-of-force:power}{vermogen}
$P = \scal{F}{v} = Fv\cos\theta$, en $\dd E_k/\dd t$ is de som van de
\omterm{def:g11:work-of-force:power}{vermogens} van alle
\omterm{def:g10:inertia:force}{krachten}: de formule bij constante snelheid van
vorig jaar, exact op elk moment van elke beweging.
\end{proposition}

\begin{proof}
Lees af uit het vorige bewijs: elke \omterm{def:g10:inertia:force}{kracht} voedt
$\scal{F}{\dd\ell} = \scal{F}{v}\,\dd t$ in $E_k$ tijdens $\dd t$.
\end{proof}

\begin{theorem}[Mechanische-energiestelling]\label{thm:g12:work-and-energy:emtheorem}\index{mechanische-energiestelling}
Zij $E_m = E_k + E_p$, waarbij $E_p$ de potentiële energieën van alle
\omterm{def:g12:work-and-energy:conservative}{conservatieve krachten} in
\omterm{def:g11:work-of-force:work}{arbeid} verzamelt. Dan, langs elke baan,
\[
\Delta E_m = W_{AB}(\text{niet-conservatieve krachten}),
\]
en $E_m$ is behouden wanneer de
\omterm{def:g12:work-and-energy:conservative}{niet-conservatieve krachten} geen
\omterm{def:g11:work-of-force:work}{arbeid} verrichten.
\end{theorem}

\begin{proof}
Splits de \omterm{def:g10:energy-conservation:energy}{kinetische-energie}stelling:
$\Delta E_k = W_{\text{cons}} + W_{\text{nc}} = -\Delta E_p +
W_{\text{nc}}$; verplaats $\Delta E_p$.
\end{proof}

\begin{example}[De slinger, deze keer eerlijk]\label{ex:g12:work-and-energy:pendulum}
Een slingerbol aan een draad van lengte $L = \qty{2.5}{m}$ wordt in
rust losgelaten op $\qty{45}{\degree}$ van de verticale. De
\omterm{def:g10:inertia:inventory}{spanning}, loodrecht op de beweging, is
vermogenloos; het \omterm{def:g10:universal-gravitation:weight}{gewicht} is
\omterm{def:g12:work-and-energy:conservative}{conservatief}: $E_m$ is behouden
\emph{op de gekromde boog}, wat vorig jaar alleen te controleren was.
De bol start $h = L(1 - \cos\qty{45}{\degree}) = \qty{0.73}{m}$ omhoog,
dus onderaan $v = \sqrt{2gh} \approx \qty{3.8}{m/s}$.
\end{example}

\begin{method}[Energie-audit van elke beweging]\label{met:g12:work-and-energy:audit}
\begin{enumerate}
  \item Kies het systeem en som de \omterm{def:g10:inertia:force}{krachten} op.
  \item Sorteer: loodrecht op de beweging $\to$ geen
        \omterm{def:g11:work-of-force:work}{arbeid};
        \omterm{def:g12:work-and-energy:conservative}{conservatief} $\to$ een $E_p$
        in $E_m$; de rest $\to$ $W_{\text{nc}}$ (glijdende
        \omterm{def:g10:inertia:inventory}{wrijving}: $-fL$, $L$ = baanslengte).
  \item Schrijf $\Delta E_m = W_{\text{nc}}$ tussen datapunt en
        vraagpunt; los op. \omterm{def:g10:energy-conservation:energy}{Energie} beantwoordt
        ``hoe snel, hoe hoog''; alleen tijd en versnelling vragen
        Newtons wetten.
\end{enumerate}
\end{method}

\section{Energiediagrammen}

\begin{definition}[Energiediagram, omkeerpunten]\label{def:g12:work-and-energy:diagram}
Voor een beweging langs één as $x$ met behouden $E_m$ tekent het
\emph{energiediagram}\index{energiediagram} de kromme $E_p(x)$ en de
horizontale lijn $E_m$. Omdat $E_k = E_m - E_p \geq 0$ is de beweging
beperkt tot waar de kromme onder de lijn ligt; de spleet ertussen is
$E_k$, direct af te lezen. In een
\emph{omkeerpunt}\index{omkeerpunt}, waar kromme en lijn samenkomen,
is $v = 0$ en keert de beweging om.
\end{definition}

\begin{proposition}[Kracht uit het landschap; evenwichten]\label{prop:g12:work-and-energy:equilibria}
De \omterm{def:g12:work-and-energy:conservative}{conservatieve kracht} langs $x$ is min de
helling van $E_p$: $F = -\,\dd E_p/\dd x$ --- ze wijst \emph{bergaf}
op het diagram, en verdwijnt waar de raaklijn horizontaal is: een
evenwicht. Een minimum van $E_p$ is
\emph{stabiel}\index{stabiel evenwicht} --- het verplaatste lichaam
wordt teruggeduwd, een bal in een dal; een maximum is
\emph{instabiel}\index{instabiel evenwicht} --- het wordt
weggeduwd, een bal op een heuveltop.
\end{proposition}

\begin{proof}
Op een kleine stap $\dd x$ verricht de
\omterm{def:g10:inertia:force}{kracht} $F\,\dd x = E_p(x) - E_p(x + \dd x) =
-\dd E_p$; deel door $\dd x$. Bergaf aan weerszijden van een dal wijst
naar binnen; rond een heuveltop, naar buiten.
\end{proof}

\begin{remark}[Oscillaties en ontsnapping]\label{rem:g12:work-and-energy:escape}
De kleine oscillaties van \cref{ch:g12:oscillators-and-time} leven op
de bodems van $E_p$-dalen. En ver van de Aarde verzwakt de zwaartekracht:
de $E_p(r)$-kromme klimt steeds trager naar een eindig plafond (de
oppervlakte onder de krimpende krachtgrafiek is eindig). Een raket
wiens $E_m$ het plafond bereikt, ontmoet nooit een
\omterm{def:g12:work-and-energy:diagram}{omkeerpunt}: ze
\emph{ontsnapt}\index{ontsnappingssnelheid} --- ongeveer
\qty{11.2}{km/s} vanaf de grond (\cref{exo:g12:work-and-energy:13});
daaronder is ze gebonden. Het plafond zelf wordt berekend in het
Bachelor-1-volume.
\end{remark}

\begin{example}[Looping]\label{ex:g12:work-and-energy:loop}
Een knikker, losgelaten van hoogte $H$, glijdt in een verticale
lus van straal $R$. Circulaire dynamica
(\cref{ch:g12:newtons-laws}) eist $v_{\text{top}}^2 \geq gR$ bovenaan
--- het gegeven dat vorig jaar de achtbaanprobleem leende, nu van
ons. Behoud vanaf rust bij $H$ tot de top (hoogte $2R$) geeft
$v_{\text{top}}^2 = 2g(H - 2R)$, dus $H \geq \tfrac52 R$: een halve
straal boven de luskruin, onafhankelijk van de massa --- iets meer in
de echte, schurende wereld.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:1}

\omterm{def:g10:energy-conservation:energy}{energie}; wat wordt ze als de rek verdubbelt?

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:2}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:3}

\begin{exercise}[$\star$]\label{exo:g12:work-and-energy:5}

\section{Probleem: De schans}

\begin{problem}[{Weekendprobleem --- de schans: een inrun geaudit door
een snelheidspistool, een vrije valvlucht, de zachte meetkunde van een
steile landing, en de toren die een ontwerper moet kopen zodra
wrijving rekeningen stuurt}]
\label{pb:g12:work-and-energy:1}
Een olympische schansspringer ($m = \qty{70}{kg}$, ski's inbegrepen)
start in rust vanuit een poort $H = \qty{45}{m}$ boven de afzettafel,
glijdt $L = \qty{90}{m}$ gekromde inrun, en verlaat de tafel
horizontaal; een snelheidspistool leest daar $v_0 = \qty{26}{m/s}$.
Het landingspunt ligt $h = \qty{40}{m}$ lager, op een helling van
$\qty{35}{\degree}$. Delen II en III verwaarlozen luchtweerstand.

\medskip\noindent\textbf{Deel I --- De inrun.}
\begin{enumerate}
  \item Voorspel de wrijvingsloze afzetsnelheid; waarom is de kromme
        van de inrun irrelevant?
  \item Referentie op de tafel: bereken $E_m$ bij de poort en bij
        afzet.
  \item Bereken $\Delta E_m$; welke stelling noemt de schuldigen, en
        wie?
  \item Leid de gemiddelde \omterm{def:g11:work-of-force:sign}{weerstands}%
        \omterm{def:g10:inertia:force}{kracht} (sneeuw plus lucht) over de baan af.
  \item Welk deel van $E_m$ gaat verloren? Waarom ski's waxen en
        hurken?
\end{enumerate}

\medskip\noindent\textbf{Deel II --- De vlucht.}
\begin{enumerate}[resume]
  \item Alleen het \omterm{def:g10:universal-gravitation:weight}{gewicht} werkt in de vlucht:
        welke beweging van \cref{ch:g12:projectile-motion} is dit?
        Snelheidscomponenten bij landing?
  \item Landingssnelheid, tweemaal: uit componenten, en uit behoud
        van $E_m$. In \unit{km/h}?
  \item Bereken de vluchttijd en de afgelegde horizontale afstand.
  \item Echte springers rijden de lucht als een vleugel, veel verder:
        kan lucht de landingssnelheid \emph{verhogen}? Argumenteer
        met de \omterm{def:g10:energy-conservation:energy}{mechanische-energie}stelling.
\end{enumerate}

\medskip\noindent\textbf{Deel III --- De landing.}
\begin{enumerate}[resume]
  \item Bereken de hoek van de landingsnelheid onder de horizontale.
  \item Ontbind in componenten evenwijdig en loodrecht op de helling.
  \item De benen absorberen alleen het loodrechte deel: bereken die
        \omterm{def:g11:mechanical-energy:kinetic}{kinetische energie}; vergelijk met het
        totaal.
  \item Vlakke grond zou alles absorberen: met welke verticale daling
        is elke landing equivalent ($h_{\text{eq}} = v_\perp^2/2g$,
        resp.\ $v^2/2g$)?
  \item Eén zin: waarom zijn landingshellingen steil en gekromd?
\end{enumerate}

\medskip\noindent\textbf{Deel IV --- De toren van de ontwerper.}
Een kleinere heuvel eist afzet bij slechts $v_0' = \qty{24}{m/s}$;
zijn rechte inrun daalt onder $\qty{35}{\degree}$, dus poorhoogte
$H'$ betekent baanslengte $H'/\sin\qty{35}{\degree}$;zelfde
\qty{80}{N} weerstand.
\begin{enumerate}[resume]
  \item Bereken de poorhoogte die een wrijvingsloze ontwerper zou
        bouwen.
  \item Schrijf de \omterm{def:g10:energy-conservation:energy}{mechanische-energie}stelling,
        poort tot tafel, voor $H'$.
  \item Los $H'$ op.
  \item Met welk percentage verhief
        \omterm{def:g10:inertia:inventory}{wrijving} de toren?
  \item De massa valt niet meer weg: herbereken $H'$ voor een
        springer van \qty{60}{kg}. Wie heeft de hogere toren nodig,
        en waarom?
  \item Punchline: de aangekondigde toren, en het aandeel van
        \omterm{def:g10:inertia:inventory}{wrijving}.
\end{enumerate}
\end{problem}
