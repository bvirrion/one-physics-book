\chapter{Mechanische golven}\label{ch:g12:mechanical-waves}

Zaterdagavond, een vol stadion: een rimpeling van rechtopstaande
toeschouwers loopt in minder dan een minuut rond de tribunes, terwijl
niemand zijn stoel verlaat. De ring van een steentje trekt over de vijver
terwijl de kurk alleen bobbelt. Dit hoofdstuk bestudeert de reiziger
zelf --- de \omterm{def:g10:signals-and-waves:wave}{golf} --- en de vier getallen die hem temmen: snelheid,
\omterm{def:g12:mechanical-waves:delay}{vertraging}, \omterm{def:g10:signals-and-waves:period}{periode}, \omterm{def:g12:mechanical-waves:wavelength}{golflengte} --- genoeg om een aardbeving
van een papieren strook af te lezen.

\section{De verstoring reist, het medium blijft}

\begin{definition}[Mechanische golf]\label{def:g12:mechanical-waves:wave}
Een \emph{mechanische golf}\index{mechanische golf} is een verstoring
die zich voortplant door een materieel medium --- touw, water, lucht,
rots --- en \omterm{def:g10:energy-conservation:energy}{energie} meeneemt \emph{zonder materie te transporteren}:
elk stukje van het medium zwaait kort op zijn plaats, geeft de beweging
door, en komt weer tot rust.
\end{definition}

\begin{example}[Geen materie-transport]\label{ex:g12:mechanical-waves:transport}
Bij de stadion\omterm{def:g10:signals-and-waves:wave}{golf} is het ``medium'' de menigte: wat reist is
het opstaan, terwijl elke toeschouwer op zijn stoel blijft. Op de
vijver groeit de ring, maar een drijvende kurk bobbelt alleen op zijn
plaats.
\end{example}

\begin{definition}[Transversale en longitudinale golven]\label{def:g12:mechanical-waves:transverse}
Een \omterm{def:g10:signals-and-waves:wave}{golf} is \emph{transversaal}\index{transversale golf} wanneer het
medium zich \emph{loodrecht} op de voortplanting beweegt: een bult
rent langs een geschud touw terwijl elk punt alleen op en neer gaat.
Hij is \emph{longitudinaal}\index{longitudinale golf} wanneer het
medium zich \emph{langs} de voortplanting beweegt: een zone van
samengedrukte \omterm{def:g11:magnetic-fields:solenoid}{windingen} reist door een geduwde veer, waarbij
elke \omterm{def:g11:magnetic-fields:solenoid}{winding} op zijn plaats zwaait.
\end{definition}

\begin{remark}[Geluid is longitudinaal]\label{rem:g12:mechanical-waves:sound}
\omterm{def:g10:signals-and-waves:sound}{Geluid} is de \omterm{def:g12:mechanical-waves:transverse}{longitudinale golf} bij uitstek: samengedrukte
lucht --- hogere \omterm{def:g10:pressure:pressure}{druk} (\cref{ch:g10:pressure}) --- reist van
bron naar trommelvlies; \cref{ch:g12:sound-acoustics} is eraan
gewijd. Aardbevingen sturen beide soorten: P-\omterm{def:g10:signals-and-waves:wave}{golven} comprimeren
de rots, S-\omterm{def:g10:signals-and-waves:wave}{golven} schudden ze zijwaarts.
\end{remark}

\section{Snelheid en vertraging}

\begin{definition}[Voortplantingssnelheid]\label{def:g12:mechanical-waves:speed}
De \emph{voortplantingssnelheid}\index{voortplantingssnelheid} van een
\omterm{def:g10:signals-and-waves:wave}{golf} is de door de verstoring afgelegde afstand gedeeld door de
genomen tijd: $v = d/t$. Het is de snelheid van de reizende
\emph{vorm}, niet van het medium.
\end{definition}

\begin{proposition}[Het medium bepaalt de snelheid]\label{prop:g12:mechanical-waves:medium}
Voor kleine verstoringen hangt de \omterm{def:g12:mechanical-waves:speed}{voortplantingssnelheid} alleen van
het \emph{medium} af, niet van de vorm of grootte van de verstoring. Op
een gespannen touw neemt $v$ toe met de \omterm{def:g10:inertia:inventory}{spanning} en af met de
massa per \omterm{def:g10:orders-of-magnitude:unit}{meter}: strak en licht is snel, slap en zwaar is
traag. \admitted
\end{proposition}

\begin{remark}[Waar de formule leeft]\label{rem:g12:mechanical-waves:ropeformula}
De exacte snelheidsformule van het touw wordt afgeleid in het volume
van jaar~1; dit jaar volstaan de experimentele feiten hierboven.
\end{remark}

\begin{example}[Geluidssnelheden]\label{ex:g12:mechanical-waves:speeds}
\omterm{def:g10:signals-and-waves:sound}{Geluid} reist met \qty{340}{m/s} in lucht, \qty{1500}{m/s} in water,
\qty{5900}{m/s} in staal --- hoe stijver, hoe sneller. Vandaar het
westernoor op de rails: staal kondigt de trein eerst aan.
\end{example}

\begin{definition}[Vertraging]\label{def:g12:mechanical-waves:delay}
Een \omterm{def:g10:signals-and-waves:wave}{golf} van snelheid $v$ bereikt een punt op afstand $d$ na de
\emph{vertraging}\index{vertraging} $\tau = d/v$: twee punten op
afstand $d$ herhalen dezelfde beweging, waarbij het verder gelegen
punt $\tau$ achterblijft --- de \omterm{rem:g10:signals-and-waves:honest}{echo}-afstandsmeting van
\cref{ch:g10:signals-and-waves}, achterstevoren gelezen.
\end{definition}

\begin{proposition}[Superpositie zonder interactie]\label{prop:g12:mechanical-waves:superposition}
Waar twee \omterm{def:g10:signals-and-waves:wave}{golven} overlappen, is de verplaatsing van het medium op
elk moment de \emph{som} van de verplaatsingen die elk alleen zou
veroorzaken; na het kruisen zet elke \omterm{def:g10:signals-and-waves:wave}{golf} onveranderd voort ---
afgeleid in het volume van jaar~2 uit de lineariteit van de
bewegingsvergelijkingen. \admitted
\end{proposition}

\begin{example}[Kruisende impulsen]\label{ex:g12:mechanical-waves:crossing}
Stuur van elk uiteinde van een touw een opwaartse bult: waar ze
elkaar ontmoeten, stijgt het touw kort tot de som van de twee
hoogten; daarna komen twee intacte bulten tevoorschijn --- zoals twee
kruisende gesprekken hun luisteraars bereiken.
\end{example}

\section{Periodieke golven: de dubbele periodiciteit}

\begin{definition}[Periodieke golf]\label{def:g12:mechanical-waves:periodic}
Wanneer de bron zijn beweging herhaalt met \omterm{def:g10:signals-and-waves:period}{periode} $T$ en
\omterm{def:g10:signals-and-waves:frequency}{frequentie} $f = 1/T$ (\cref{ch:g10:signals-and-waves}), voedt hij
het medium met een \emph{periodieke golf}\index{periodieke golf}: elk
punt herhaalt zijn eigen beweging met \omterm{def:g10:signals-and-waves:period}{periode} $T$, elk
achterblijvend op zijn buren met de reis\omterm{def:g12:mechanical-waves:delay}{vertraging}.
\end{definition}

\begin{definition}[Golflengte]\label{def:g12:mechanical-waves:wavelength}
Een momentopname van een \omterm{def:g12:mechanical-waves:periodic}{periodieke golf} toont een patroon dat
zich in de ruimte herhaalt; zijn ruimtelijke \omterm{def:g10:signals-and-waves:period}{periode} --- kam tot
kam --- is de \emph{golflengte}\index{golflengte} $\lambda$, in
\omterm{def:g10:orders-of-magnitude:unit}{meters}. Een \omterm{def:g12:mechanical-waves:periodic}{periodieke golf} is
\emph{dubbel periodiek}\index{dubbele periodiciteit}: \omterm{def:g10:signals-and-waves:period}{periode}
$T$ in de tijd op elk punt, \omterm{def:g10:signals-and-waves:period}{periode} $\lambda$ in de ruimte
op elk moment.
\end{definition}

\begin{theorem}[De golfrelatie]\label{thm:g12:mechanical-waves:relation}
Een \omterm{def:g12:mechanical-waves:periodic}{periodieke golf} van snelheid $v$, \omterm{def:g10:signals-and-waves:period}{periode} $T$ en
\omterm{def:g10:signals-and-waves:frequency}{frequentie} $f$ heeft \omterm{def:g12:mechanical-waves:wavelength}{golflengte}
$\lambda = v\,T = v/f$.
\end{theorem}

\begin{proof}
Tijdens één \omterm{def:g10:signals-and-waves:period}{periode} schuift het patroon vooruit over $v\,T$; maar
de bron --- en daarmee elk punt --- is dan weer in zijn begintoestand,
dus valt het verschoven patroon samen met het oorspronkelijke: de
herhaalafstand is $v\,T$.
\end{proof}

\begin{remark}[De frequentie hoort bij de bron]\label{rem:g12:mechanical-waves:source}
Bij overgang van het ene medium naar het andere --- \omterm{def:g10:signals-and-waves:sound}{geluid} van
lucht naar water --- behoudt een \omterm{def:g10:signals-and-waves:wave}{golf} zijn
\omterm{def:g10:signals-and-waves:frequency}{frequentie}, die de bron voorschrijft, terwijl zijn snelheid
verandert; door $\lambda = v/f$ verandert de
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} mee.
\end{remark}

\begin{example}[Concert-A, in lucht en onder water]\label{ex:g12:mechanical-waves:concertA}
De \qty{440}{Hz} concert-A van \cref{ch:g10:signals-and-waves} heeft
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} $340/440 = \qty{0.77}{m}$ in lucht en $1500/440 =
\qty{3.4}{m}$ in water: dezelfde noot, uitgerekt door het snellere
medium.
\end{example}

\begin{definition}[In fase, in faseoppositie]\label{def:g12:mechanical-waves:phase}
Twee punten gescheiden door een geheel aantal
\omterm{def:g12:mechanical-waves:wavelength}{golflengten}, $d = k\lambda$, bewegen op elk moment
identiek: ze zijn \emph{in fase}\index{in fase}. Punten gescheiden
door $d = (k + \tfrac12)\lambda$ bewegen altijd tegengesteld --- kam
tegen dal: ze zijn in \emph{faseoppositie}\index{faseoppositie}.
\end{definition}

\begin{method}[De dubbele periodiciteit lezen]\label{met:g12:mechanical-waves:reading}
\begin{enumerate}
  \item Op een opname $y(t)$ op een vast punt lees je $T$ van kam tot
        kam; $f = 1/T$.
  \item Op een momentopname $y(x)$ op een vast moment lees je $\lambda$
        van kam tot kam.
  \item Leid $v = \lambda/T = \lambda f$ af; controleer tegen een
        directe timing $v = d/\tau$ wanneer beschikbaar.
  \item Verwar de twee grafieken nooit: de ene as draagt seconden, de
        andere \omterm{def:g10:orders-of-magnitude:unit}{meters}.
\end{enumerate}
\end{method}

\section{Golffronten en demping}

\begin{definition}[Golffront]\label{def:g12:mechanical-waves:wavefront}
Een \emph{golffront}\index{golffront} is een lijn (of oppervlak) van
punten die de \omterm{def:g10:signals-and-waves:wave}{golf} op hetzelfde moment bereikt --- een kamlijn
op de vijver. Vanuit een puntbron zijn de golffronten uitdijende
cirkels; ver van elke bron zijn ze bijna recht. De
\omterm{def:g10:signals-and-waves:wave}{golf} plant zich loodrecht op zijn golffronten voort,
opeenvolgende kammen $\lambda$ uit elkaar.
\end{definition}

\begin{definition}[Demping]\label{def:g12:mechanical-waves:attenuation}
De afname van de \omterm{def:g10:signals-and-waves:amplitude}{amplitude} van een \omterm{def:g10:signals-and-waves:wave}{golf}
tijdens de voortplanting is zijn \emph{demping}\index{demping}. Twee
oorzaken tellen op: de \omterm{def:g10:energy-conservation:energy}{energie} van een cirkelvormig
\omterm{def:g12:mechanical-waves:wavefront}{golffront} wordt over een steeds langere kam verdeeld, en het
medium absorbeert onderweg een beetje. Rechte
\omterm{def:g12:mechanical-waves:wavefront}{golffronten} ontsnappen aan de eerste oorzaak --- zo kan
de deining van een verre storm een oceaan oversteken.
\end{definition}

waarom een hele wijk bijna samen op en neer gaat.
\end{exercise}

\section{Probleem: Een seismogram lezen}

\begin{problem}[{Weekendprobleem --- een seismogram lezen: twee golven
racen uit een gebroken breuk, drie cirkels pinnen het epicentrum vast,
en een trage derde golf geeft een kustlijn zijn waarschuwing}]
\label{pb:g12:mechanical-waves:1}
Nachtdienst in een seismologisch observatorium. Om 09:14:32 springt de
naald van station A: scherpe snelle kronkels, daarna tragere, bredere
zwaaien. Een aardbeving heeft ergens rots gebroken; twee
\omterm{def:g10:signals-and-waves:wave}{golven} racen door de korst: P-\omterm{def:g10:signals-and-waves:wave}{golven},
compressies bij $v_P = \qty{6.0}{km/s}$, en S-\omterm{def:g10:signals-and-waves:wave}{golven},
een zijwaarts schudden bij $v_S = \qty{3.5}{km/s}$.

\medskip
\noindent\textbf{Deel I --- Twee \omterm{def:g10:signals-and-waves:wave}{golven}, één klok.}
\begin{enumerate}
  \item Classificeer uit de beschrijvingen hierboven P- en
        S-\omterm{def:g10:signals-and-waves:wave}{golven} als \omterm{def:g12:mechanical-waves:transverse}{transversaal} of
        \omterm{def:g12:mechanical-waves:transverse}{longitudinaal}.
  \item Welke \omterm{def:g10:signals-and-waves:wave}{golf} arriveert het eerst bij elk station? Bereken
        de reistijden van beide over $d = \qty{120}{km}$.
  \item Toon aan dat de P--S-aankomstkloof
        $\Delta t = d\,(1/v_S - 1/v_P)$ is en bereken die voor
        $d = \qty{120}{km}$.
  \item Leid de afstandsregel van de seismoloog af
        $d = \dfrac{v_P\,v_S}{v_P - v_S}\,\Delta t$ en toon dat de
        coëfficiënt ongeveer \qty{8.4}{km} per seconde kloof is.
  \item Waarom groeit de kloof met de afstand --- en waarom is dat een
        geschenk voor een station dat de beving nooit heeft zien
        starten?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Cirkels op de kaart.}
\begin{enumerate}[resume]
  \item Station A meet $\Delta t_A = \qty{25}{s}$. Hoe ver is het
        epicentrum van A?
  \item Leg uit waarom deze ene meting het epicentrum op een cirkel
        plaatst, niet op een punt.
  \item Stations B en C meten $\Delta t_B = \qty{15}{s}$ en
        $\Delta t_C = \qty{32}{s}$: hun afstanden tot het epicentrum?
  \item Leg uit hoe de drie cirkels samen het epicentrum vastpinnen.
  \item De P-\omterm{def:g10:signals-and-waves:wave}{golf} bereikte A om 09:14:32. Bereken zijn
        reistijd vanaf het epicentrum, en de oorsprongstijd van de
        beving.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De \omterm{def:g10:signals-and-waves:wave}{golf} die de oceaan oversteekt.}
De drie cirkels ontmoeten elkaar buitengaats: de breuk brak onder de
zeebodem en zette de waterkolom in beweging. Lange
\omterm{def:g10:signals-and-waves:wave}{golven} op diep water reizen met een snelheid die door de
diepte wordt bepaald --- ongeveer \qty{200}{m/s} in de open oceaan,
ongeveer \qty{10}{m/s} bij een strand (hier toegegeven; de formule
wordt afgeleid in het volume van jaar~1). Een haven ligt
$D = \qty{600}{km}$ van het epicentrum.
\begin{enumerate}[resume]
  \item Zet \qty{200}{m/s} om in \unit{km/h} en noem een voertuig met
        een vergelijkbare kruissnelheid.
  \item Hoe lang doet de tsunami erover om de haven te bereiken?
  \item De eigen seismometer van de haven voelt eerst de
        P-\omterm{def:g10:signals-and-waves:wave}{golf}. Na hoe lang? Als het alarm op dat moment
        \omterm{def:g10:signals-and-waves:sound}{klinkt}, hoeveel waarschuwingstijd krijgt de kust
        vóór de tsunami?
  \item Op zee is de \omterm{def:g10:signals-and-waves:period}{periode} van de tsunami ongeveer
        \qty{15}{min}. Bereken zijn \omterm{def:g12:mechanical-waves:wavelength}{golflengte}
        daar, en leg uit waarom een schip hem niet merkt.
  \item Dicht bij het strand daalt de snelheid tot \qty{10}{m/s}
        terwijl de \omterm{def:g10:signals-and-waves:period}{periode} vast blijft. Bereken de nieuwe
        \omterm{def:g12:mechanical-waves:wavelength}{golflengte}. Wat moet er met de
        \omterm{def:g10:signals-and-waves:wave}{golf} gebeuren als hij samentrekt?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- \omterm{def:g10:energy-conservation:energy}{Energie}, en het rapport.}
Een sterke beving geeft ongeveer \qty{1e15}{J} vrij, meegenomen door
de \omterm{def:g10:signals-and-waves:wave}{golven}.
\begin{enumerate}[resume]
  \item Bereken de lengte van een cirkelvormig
        \omterm{def:g12:mechanical-waves:wavefront}{golffront} \qty{10}{km} van het epicentrum,
        daarna \qty{210}{km}; met welke factor is hij gegroeid?
  \item Leid af waarom schudden verzwakt met de afstand zelfs in een
        verliesloze korst --- en noem de tweede oorzaak die een
        echte korst toevoegt.
  \item Een vloeistof kan geen afschuiving weerstaan. Stations aan de
        verre kant registreren de P-\omterm{def:g10:signals-and-waves:wave}{golf} maar geen
        S-\omterm{def:g10:signals-and-waves:wave}{golf}: wat onthult dat over de kern van de Aarde?
  \item De hogesnelheidstrein van het \omterm{def:g10:energy-conservation:energy}{energie}-hoofdstuk
        van vorig jaar (\cref{ch:g11:mechanical-energy}) draagt ongeveer
        \qty{1.5e9}{J} aan \omterm{def:g11:mechanical-energy:kinetic}{kinetische energie}: hoeveel
        van zulke treinen is \qty{1e15}{J} waard?
  \item Het rapport, in twee zinnen: waar en wanneer was de beving, en
        hoeveel waarschuwing kreeg de haven? Geef de drie getallen.
\end{enumerate}
\end{problem}
