\chapter{Mechanische oscillatoren en de meting van de tijd}\label{ch:g12:oscillators-and-time}

Een staande klok tikt: elke tik is een messing slinger die de
verticale passeert, herhaald vijftien miljoen keer sinds Nieuwjaar.
Een scherfje kwarts in je horloge buigt \num{32768} keer per seconde;
een cesium-\omterm{def:g11:fundamental-interactions:ladder}{atoom} zoemt nog negen miljard keer sneller.
Alles wat zwaait kan de tijd bijhouden: dit hoofdstuk volgt het
zwaaien, van Huygens tot de \omterm{def:g10:universal-gravitation:satellite}{satellieten}.

\section{Vrije oscillaties}

\begin{definition}[Oscillator, periode, frequentie, amplitude]\label{def:g12:oscillators-and-time:oscillator}
Een \emph{mechanische oscillator}\index{oscillator} is een systeem
dat, uit een stabiele \emph{evenwichtspositie}\index{evenwichtspositie}
verplaatst en losgelaten, heen en weer zwaait eromheen: dat zijn
\emph{vrije oscillaties}\index{vrije oscillaties}. De beweging
herhaalt zich na de \emph{periode}\index{periode} $T_0$ (\unit{s});
de \emph{frequentie}\index{frequentie} $f_0 = 1/T_0$
(\unit{Hz}, \cref{ch:g12:mechanical-waves}); de
\emph{amplitude}\index{amplitude} is de maximale uitwijking.
\end{definition}

\begin{example}[Een dierentuin van oscillatoren]\label{ex:g12:oscillators-and-time:zoo}
Een schommel op de speelplaats: $T_0 \approx \qty{3}{s}$. De A-snaar
van een gitaar: $f_0 = \qty{110}{Hz}$. Een kwartskristal:
\qty{32768}{Hz}. De cesiumoscillatie die de seconde definieert:
ongeveer \qty{9.2e9}{Hz}. Tien ordegrootten, één taak: tellen.
\end{example}

\section{De eenvoudige slinger}

\begin{definition}[Eenvoudige slinger]\label{def:g12:oscillators-and-time:pendulum}
Een \emph{eenvoudige slinger}\index{eenvoudige slinger}: een kleine
massa $m$ aan een onrekbaar touw van lengte $L$ en verwaarloosbare
massa, zwaaiend in een verticaal vlak. Van de twee
\omterm{def:g10:inertia:force}{krachten} op de massa
(\cref{ch:g12:newtons-laws}) staat de \omterm{def:g10:inertia:inventory}{spanning}
loodrecht op de beweging, terwijl de component van het
\omterm{def:g10:universal-gravitation:weight}{gewicht} langs de
\omterm{def:g10:relative-motion:trajectory}{baan} terugwijst naar het laagste punt: het
\omterm{def:g10:universal-gravitation:weight}{gewicht} is de \emph{terugdrijvende \omterm{def:g10:inertia:force}{kracht}}%
\index{terugdrijvende kracht}, tegengesteld aan de uitwijking.
\end{definition}

\begin{proposition}[Periode van de eenvoudige slinger]\label{prop:g12:oscillators-and-time:pendulum-period}
\index{eenvoudige slinger!periode}
Voor kleine \omterm{def:g12:oscillators-and-time:oscillator}{amplitudes} (onder ongeveer $15^\circ$) is de
\omterm{def:g12:oscillators-and-time:oscillator}{periode}
\[
T_0 = 2\pi \sqrt{\frac{L}{g}}, \qquad g = \qty{9.81}{m/s^2}.
\]
Ze hangt noch van de massa van de bol af, noch --- zolang de hoek
klein blijft --- van de \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}: kleine zwaaiingen zijn
\emph{isochroon}\index{isochronisme}, allemaal even lang.
\end{proposition}
\admitted

\begin{remark}[Dimensionale analyse, en waar de $2\pi$ woont]\label{rem:g12:oscillators-and-time:dimensional}
De formule is te raden. Uit $m$ (\unit{kg}), $L$ (\unit{m}) en
$g$ (\unit{m/s^2}) is de enige combinatie met de
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} van een tijd $\sqrt{L/g}$ --- en de massa
\emph{kan niet} meedoen, geen ander gegeven draagt een
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} om hem op te heffen. Dimensies geven niet het pure
getal ervoor: de $2\pi$ komt uit het oplossen van de
bewegingsvergelijking, gedaan in het Bachelor-1-volume. De
kleinehoekvoorwaarde is evenmin optioneel: bij $30^\circ$ is de echte
\omterm{def:g12:oscillators-and-time:oscillator}{periode} $2\%$ langer, en isochronisme faalt.
\end{remark}

\begin{example}[De secondenslinger]\label{ex:g12:oscillators-and-time:seconds}
Welke lengte tikt de seconde, $T_0 = \qty{2.00}{s}$, één tik per
richting? Oplossen: $L = g\,T_0^2/4\pi^2 = 9.81 \times
2.00^2/4\pi^2 \approx \qty{0.994}{m}$. Bijna een meter: slingerklokken
zijn hoog omdat seconden lang zijn.
\end{example}

\section{De massa--veer-oscillator}

\begin{definition}[Terugdrijvende kracht van een veer]\label{def:g12:oscillators-and-time:spring}
Een glijder van massa $m$ schuift over een horizontale wrijvingsloze
rail, vastgemaakt aan een veer van \emph{veerconstante}%
\index{veerconstante} $k$ (\unit{N/m}). Zij $x$ de uitwijking uit
evenwicht. Uitgerekt of samengedrukt trekt of duwt de veer de glijder
terug met de \omterm{def:g10:inertia:force}{kracht} $F = -kx$: evenredig met de
uitwijking en tegengesteld eraan --- opnieuw een
\omterm{def:g12:oscillators-and-time:pendulum}{terugdrijvende kracht}, nu geleverd door elasticiteit.
\end{definition}

\begin{proposition}[Bewegingsvergelijking en haar oplossing]\label{prop:g12:oscillators-and-time:spring-period}
\index{massa--veer-oscillator}
\omterm{thm:g12:newtons-laws:second}{Newtons tweede wet} langs de rail
(\cref{ch:g12:newtons-laws}) geeft $m \frac{d^2x}{dt^2} = -kx$,
d.w.z.\ $\frac{d^2x}{dt^2} = -\frac{k}{m}\,x$. Welke constanten $A$
en $\varphi$ ook,
\[
x(t) = A \cos\!\left(\frac{2\pi t}{T_0} + \varphi\right),
\qquad T_0 = 2\pi\sqrt{\frac{m}{k}},
\]
voldoet aan deze vergelijking: oscillatie met
\omterm{def:g12:oscillators-and-time:oscillator}{periode} $T_0$, onafhankelijk van de
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}.
\end{proposition}

\begin{proof}
Differentiatie: $\frac{dx}{dt} = -\frac{2\pi}{T_0}A \sin(\frac{2\pi
t}{T_0} + \varphi)$; opnieuw, $\frac{d^2x}{dt^2} = -(\frac{2\pi}{T_0})^2
A\cos(\frac{2\pi t}{T_0} + \varphi) = -(\frac{2\pi}{T_0})^2\,x(t)$,
wat precies $-(k/m)\,x$ is wanneer $(2\pi/T_0)^2 = k/m$.
\end{proof}

\begin{remark}[Alle bewegingen die er zijn]\label{rem:g12:oscillators-and-time:uniqueness}
Dat \emph{elke} beweging van de glijder één van deze functies is,
wordt toegegeven, bewezen in het Bachelor-1-volume. En geen $g$ in
$2\pi\sqrt{m/k}$: stugger trilt sneller, zwaarder trager, hetzelfde
op de Maan.
\end{remark}

\begin{definition}[Amplitude en fase]\label{def:g12:oscillators-and-time:phase}
In $x(t) = A\cos(2\pi t/T_0 + \varphi)$ is de positieve constante $A$
de \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} --- de uitersten zijn $x = \pm A$ --- en
$2\pi t/T_0 + \varphi$ is de \emph{fase}\index{fase}, die met $2\pi$
toeneemt per \omterm{def:g12:oscillators-and-time:oscillator}{periode}; de \emph{beginfase}%
\index{beginfase} $\varphi$ registreert de start: in rust losgelaten
vanuit $x = A$ geldt $\varphi = 0$. Bijvoorbeeld $m = \qty{0.20}{kg}$
op $k = \qty{20}{N/m}$ geeft $T_0 = 2\pi\sqrt{0.20/20} =
\qty{0.63}{s}$; naar \qty{5.0}{cm} getrokken en losgelaten,
$x(t) = 5.0\cos(2\pi t/0.63)$ in centimeters.
\end{definition}

\section{Energie, demping, resonantie}

\begin{proposition}[Energie van de oscillator]\label{prop:g12:oscillators-and-time:energy}
Een veer die over $x$ is uitgerekt, slaat de elastische potentiële
\omterm{def:g10:energy-conservation:energy}{energie} $E_p = \tfrac12 kx^2$ op (gerechtvaardigd in
\cref{ch:g12:work-and-energy}). Tijdens wrijvingsloze oscillatie is de
\omterm{def:g11:mechanical-energy:em}{mechanische energie}
\[
E_m = E_k + E_p = \tfrac12 m v^2 + \tfrac12 k x^2 = \tfrac12 k A^2
\]
constant: geheel elastisch in de uitersten, geheel kinetisch in
evenwicht. De slinger speelt hetzelfde duet tussen $E_k$ en
gravitationele $E_p$ (\cref{ch:g11:mechanical-energy}).
\end{proposition}

\begin{proof}
Met $v = \frac{dx}{dt}$ is de afgeleide van $E_m$
$mv\frac{dv}{dt} + kx\frac{dx}{dt} = v\,(m\frac{d^2x}{dt^2} + kx)
= 0$ door de bewegingsvergelijking; de waarde $\tfrac12 kA^2$ lees je
in een uiterste.
\end{proof}

\begin{definition}[Gedempte oscillaties]\label{def:g12:oscillators-and-time:damping}
\omterm{def:g10:inertia:inventory}{Wrijving} trekt \omterm{def:g11:mechanical-energy:em}{mechanische energie} weg uit elke
echte \omterm{def:g12:oscillators-and-time:oscillator}{oscillator}: zijn \omterm{def:g12:oscillators-and-time:oscillator}{vrije
oscillaties} zijn \emph{gedempt}\index{demping}. Bij lichte demping
oscilleert het systeem nog, met een stervende
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} maar een bijna onveranderde herhaaltijd, de
\emph{\omterm{def:g12:oscillators-and-time:oscillator}{pseudoperiode}}\index{pseudoperiode}, dicht bij
$T_0$; bij zware demping (honing) kruipt het terug naar evenwicht
zonder ooit voorbij te schieten; de
\omterm{def:g10:energy-conservation:energy}{energie}, evenredig met het kwadraat van de
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude}, verdwijnt als warmte.
\end{definition}

\begin{definition}[Gedwongen oscillaties en resonantie]\label{def:g12:oscillators-and-time:resonance}
Duw een \omterm{def:g12:oscillators-and-time:oscillator}{oscillator} \omterm{def:g10:signals-and-waves:period}{periodiek} ---
\emph{gedwongen oscillaties}\index{gedwongen oscillaties} --- en na
een kort transient regime oscilleert hij op de
\emph{aandrijvings}-\omterm{def:g12:oscillators-and-time:oscillator}{frequentie} $f$, niet op de zijne.
Zijn \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} piekt wanneer $f$ de
\emph{eigen\omterm{def:g12:oscillators-and-time:oscillator}{frequentie}}\index{eigenfrequentie} $f_0$ van de
vrije \omterm{def:g12:oscillators-and-time:oscillator}{oscillator} nadert: dat is
\emph{resonantie}\index{resonantie}, elke duw die in de pas van de
beweging aankomt en elke cyclus \omterm{def:g10:energy-conservation:energy}{energie}
toevoert; hoe lichter de \omterm{def:g12:oscillators-and-time:damping}{demping}, hoe hoger en
scherper de piek.
\end{definition}

\begin{example}[Resonantie voor goed en kwaad]\label{ex:g12:oscillators-and-time:resonance-zoo}
Een ouder die een schommel duwt, duwt op diens eigen
\omterm{def:g12:oscillators-and-time:oscillator}{frequentie} --- \omterm{def:g12:oscillators-and-time:resonance}{resonantie} in
\omterm{def:g11:work-of-force:work}{arbeid}. In \num{2000} zwaaide de nieuwe Millennium Bridge
in Londen alarmerend toen de stappen van de menigte vastklonken op
een \omterm{def:g12:oscillators-and-time:resonance}{eigenfrequentie} nabij \qty{1}{Hz}; hij ging
twee jaar dicht voor dempers. Elk kwartshorloge houdt zijn kristal
zingend door het op \omterm{def:g12:oscillators-and-time:resonance}{resonantie} aan te drijven,
\qty{32768}{Hz}.
\end{example}

\section{Tijd meten}

\begin{method}[Een periode timen]\label{met:g12:oscillators-and-time:timing}
Time nooit één oscillatie: start de stopwatch wanneer de massa de
verticale passeert (snelst, makkelijkst te beoordelen), tel
$N = 50$ oscillaties, deel door $N$ --- een
reactiefout van \qty{0.2}{s} wordt \qty{4}{ms} op $T_0$. Herhaal, en
vergelijk series.
\end{method}

\begin{example}[Drie eeuwen tikken]\label{ex:g12:oscillators-and-time:clocks}
Huygens bouwde de eerste slingerklok in \num{1657}: isochronisme
bracht klokken van een kwartier drift per dag naar seconden, en
gecompenseerde slingers naar fracties van een seconde per dag. Het
kwartshorloge (\num{1969}) wisselde de stang in voor een kristal dat
op \qty{32768}{Hz} buigt: tienden van een seconde per maand. Sinds
\num{1967} is de seconde zelf \emph{gedefinieerd} door een atomaire
oscillatie: de duur van \num{9192631770}
\omterm{def:g12:oscillators-and-time:oscillator}{perioden} van de microgolfstraling van een
interne overgang van cesium-133 (\cref{ch:g12:quantum-world}). De
beste cesiumklokken drijven één seconde in honderd miljoen jaar.
\end{example}

\begin{remark}[Waarom de nanoseconde najagen]\label{rem:g12:oscillators-and-time:gps}
Een GPS-ontvanger vindt zijn positie door radio\omterm{def:g10:signals-and-waves:signal}{signalen}
van \omterm{def:g10:universal-gravitation:satellite}{satellieten} te timen; licht reist
\qty{30}{cm} per nanoseconde, dus een klokfout van één microseconde
plaatst je \qty{300}{m} ernaast. \omterm{def:g10:universal-gravitation:satellite}{Satellieten}
dragen daarom atoomklokken --- en bij die precisie hangt het ritme
van een klok af van hoe snel hij beweegt en hoe hoog hij zit, een
effect dat we ontmoeten in \cref{ch:g12:special-relativity}.
\end{remark}

\section{Probleem: De werkbank van de horlogemaker}

\begin{problem}[{Weekendprobleem --- de werkbank van de horlogemaker: een
wandklok weer op tijd --- vijftig zwaaiingen getimed, een messing stang
die in de zomer rekt, de gefluisterde duw van het échappement, een
kwartsuitdager --- het vonnis in seconden per maand}]
\label{pb:g12:oscillators-and-time:1}
Een slingerwandklok arriveert op de werkbank ``die verkeerd loopt''.
Behandel de messing stang en de zware massa als een
\omterm{def:g12:oscillators-and-time:pendulum}{eenvoudige slinger} van effectieve lengte $L$,
massabol $m = \qty{1.2}{kg}$; de overbrenging telt elke volledige
oscillatie als precies \qty{2.000}{s}.

\medskip
\noindent\textbf{Deel I --- De pols van de klok nemen.}
\begin{enumerate}
  \item Waarom $50$ oscillaties timen in plaats van één? Schat de
        fout die op $T_0$ overblijft als je reactietijd
        \qty{0.2}{s} is.
  \item De stopwatch leest \qty{99.4}{s} voor $50$ oscillaties: de
        \omterm{def:g12:oscillators-and-time:oscillator}{periode}?
  \item Leid de huidige effectieve lengte van de slinger af.
  \item Welke lengte zou precies \qty{2.000}{s} geven? Moet de
        regelmoer de massa omhoog of omlaag brengen, en met hoeveel?
  \item Onversteld: loopt de klok te snel of te traag, met hoeveel
        seconden per dag?
  \item Gevoeligheid voor $g$: de
        \omterm{def:g12:oscillators-and-time:oscillator}{periode} op de Maan
        ($g = \qty{1.62}{m/s^2}$)? De dagelijkse fout op een
        sterrenwacht waar $g = \qty{9.80}{m/s^2}$?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De zomerdip.} Messing zet uit: een
opwarming $\Delta\theta$ rekt de stang tot
$L' = L\,(1 + \lambda\, \Delta\theta)$, $\lambda = \num{1.9e-5}$ per
\omterm{def:g10:light-spectra:kelvin}{kelvin}. De klok is afgesteld bij
\qty{20}{\celsius}.
\begin{enumerate}[resume]
  \item Bereken $\Delta L$ voor een zomerkamer bij \qty{25}{\celsius}.
  \item Met $\sqrt{1+x} \approx 1 + x/2$, toon dat
        $\Delta T_0 / T_0 = \tfrac12 \lambda\, \Delta\theta$.
  \item Bereken $\Delta T_0/T_0$ en de dagelijkse zomerfout: te snel
        of te traag?
  \item Dezelfde vragen voor een wintergang bij \qty{10}{\celsius}.
  \item Oude precisieklokken gebruikten ``rooster''slingers die
        messingstaven met stalen (die ongeveer half zo veel uitzetten)
        in afwisselende richtingen combineerden. Leg uit hoe dat het
        ritme kan vasthouden.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De gefluisterde duw.} Met het
échappement ontkoppeld halveert de zwaai, gestart bij
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} $\theta_m = 2.00^\circ$, zijn
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} in \qty{280}{s}; in dienst houdt een
kleine duw van het échappement elke
\omterm{def:g12:oscillators-and-time:oscillator}{periode} $\theta_m$ constant.
\begin{enumerate}[resume]
  \item Hoeveel \omterm{def:g12:oscillators-and-time:oscillator}{perioden} duurt de
        halvering?
  \item In het uiterste is de massa gestegen met
        $h = L(1 - \cos\theta_m) \approx L\,\theta_m^2/2$
        (radialen): leid $E_m = \tfrac12 m g L\,\theta_m^2$ af en
        bereken die (\cref{ch:g11:mechanical-energy}).
  \item \omterm{def:g10:energy-conservation:energy}{Energie} is evenredig met het kwadraat van de
        \omterm{def:g12:oscillators-and-time:oscillator}{amplitude}: welk deel van $E_m$ gaat
        verloren per \omterm{def:g12:oscillators-and-time:oscillator}{periode}?
  \item Leid de \omterm{def:g10:energy-conservation:energy}{energie} van elke duw af, en het
        gemiddelde \omterm{def:g11:work-of-force:power}{vermogen} dat nodig is.
  \item Het aandrijf\omterm{def:g10:universal-gravitation:weight}{gewicht} $M = \qty{2.5}{kg}$ daalt
        $h = \qty{1.0}{m}$ per week: is dat genoeg
        \omterm{def:g10:energy-conservation:energy}{energie}, bij welk
        \omterm{def:g10:energy-conservation:efficiency}{rendement}?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De kwartsuitdager.} Een
kwartsuurwerk oscilleert op $f = \qty{32768}{Hz}$ en drijft ongeveer
\qty{0.3}{s} per maand; de herstelde slinger drijft nog
\qty{0.5}{s} per dag.
\begin{enumerate}[resume]
  \item Bereken de kwarts-\omterm{def:g12:oscillators-and-time:oscillator}{periode}. Toon
        $\num{32768} = 2^{15}$: waarom is een
        \omterm{def:g11:work-of-force:power}{macht} van twee precies wat een
        horlogecircuit wil?
  \item Bereken de relatieve drifts van kwarts en herstelde slinger,
        en elk in seconden per maand. Wie wint, met welke factor?
  \item De cesiumstandaard --- \num{9192631770}
        \omterm{def:g12:oscillators-and-time:oscillator}{perioden} maken één seconde --- drijft
        \qty{1}{s} per honderd miljoen jaar: zijn relatieve drift?
        Hoeveel ordegrootten onder kwarts, en waarom eist GPS die?
  \item Het vonnis, één zin met getallen: wat verdient de herstelde
        klok aan de muur, en wie bezit nu de seconde?
\end{enumerate}
\end{problem}
