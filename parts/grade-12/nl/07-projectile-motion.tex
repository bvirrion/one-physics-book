\chapter{Vrije val en projectielbeweging}\label{ch:g12:projectile-motion}

Elke straal van een tuinslang trekt dezelfde kromme in de lucht. Zo
doet een basketbal die naar de ring buigt, een verspringer, een
kanonskogel: elk lichaam overgelaten aan zijn
\omterm{def:g10:universal-gravitation:weight}{gewicht}. Dit hoofdstuk verdient die kromme
--- een parabool --- uit \omterm{thm:g12:newtons-laws:second}{Newtons tweede wet} in twee
regels calculus, leest top, vluchttijd en
\omterm{def:g12:projectile-motion:range}{reikwijdte} van zijn vergelijking, en laat de
lucht weer binnen om te zien wat hij bederft.

\section{Vrije val}

\begin{definition}[Vrije val]\label{def:g12:projectile-motion:freefall}
Een lichaam is in \emph{vrije val}\index{vrije val} wanneer de enige
\omterm{def:g10:inertia:force}{kracht} die erop werkt zijn
\omterm{def:g10:universal-gravitation:weight}{gewicht} $\vect P = m\vect g$ is: geen steun,
geen touw, luchtweerstand verwaarloosbaar. Een losgelaten steen of een
bal die de hand heeft verlaten is, tot goede nauwkeurigheid, in vrije
val --- ``vallen'' omvat stijgen.
\end{definition}

\begin{theorem}[Galilei's wet van de vrije val]\label{thm:g12:projectile-motion:galileo}
\index{Galilei's wet van de vrije val}
In \omterm{def:g12:projectile-motion:freefall}{vrije val} delen alle lichamen dezelfde
versnelling, wat hun massa ook is:
\[
\vect a = \vect g, \qquad g = \qty{9.81}{m/s^2},
\]
gericht recht omlaag: de \omterm{def:g11:electric-gravitational-fields:field}{veld}sterkte
\qty{9.81}{N/kg} van \cref{ch:g10:universal-gravitation} in zijn andere
uitrusting, $\unit{N/kg} = \unit{m/s^2}$.
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{Newtons tweede wet}
(\cref{ch:g12:newtons-laws}) met het
\omterm{def:g10:universal-gravitation:weight}{gewicht} als enige
\omterm{def:g10:inertia:force}{kracht}: $m\vect a = m\vect g$, en de massa valt weg.
Hamer en veer verschillen alleen door de lucht.
\end{proof}

\begin{proposition}[Verticale val uit rust]\label{prop:g12:projectile-motion:vertical}
Losgelaten uit rust, met de gevallen diepte $h$ omlaag geteld vanaf
het loslaatpunt, gehoorzaamt een lichaam in
\omterm{def:g12:projectile-motion:freefall}{vrije val}
\[
a = g, \qquad v = g t, \qquad h = \tfrac12\, g t^2 .
\]
\end{proposition}

\begin{proof}
$v(t)$ is de primitieve van de constante $a = g$ die bij $t = 0$
verdwijnt (\cref{ch:g12:kinematics-2d}); $h(t)$ die van $v = gt$ die
bij $t = 0$ verdwijnt.
\end{proof}

\begin{example}[Ordes van grootte]\label{ex:g12:projectile-motion:orders}
Na $1$, $2$, $3$ seconden \omterm{def:g12:projectile-motion:freefall}{vrije val}:
$v = 9.8$, $19.6$, \qty{29.4}{m/s} ($35$, $71$, \qty{106}{km/h}) en
$h = 4.9$, $19.6$, \qty{44.1}{m}. Vanaf een \qty{30}{m}-dak:
$t = \sqrt{2h/g} \approx \qty{2.5}{s}$, aankomend met
\qty{24}{m/s} $\approx \qty{87}{km/h}$. Tijden groeien als $\sqrt h$
--- en voorbij enkele seconden weigert de lucht verwaarloosbaar te
blijven (laatste sectie).
\end{example}

\begin{example}[Verticale worp]\label{ex:g12:projectile-motion:throw}
Recht omhoog gegooid met $v_0$, as $Oy$ omhoog vanaf de hand:
dezelfde twee primitieven geven $v = v_0 - gt$ en
$y = v_0 t - \tfrac12 gt^2$. De top, waar $v = 0$, komt bij
$t = v_0/g$, hoogte $h = v_0^2/2g$ --- het
\omterm{def:g10:energy-conservation:energy}{energie}-antwoord
$v_0 = \sqrt{2gh}$ van \cref{ch:g11:mechanical-energy} achterstevoren;
\emph{omlaag} gegooid wisselt $v_0$ van teken. Voor \qty{15}{m/s}: top
\qty{11.5}{m} bij \qty{1.53}{s}, terug bij \qty{3.06}{s}, met
\qty{15}{m/s}.
\end{example}

\section{Het projectiel: één wet, twee primitieven}

\begin{definition}[Projectiel]\label{def:g12:projectile-motion:projectile}
Een \emph{projectiel}\index{projectiel} is een lichaam gelanceerd met
een snelheid $\vect v_0$ en daarna overgelaten in
\omterm{def:g12:projectile-motion:freefall}{vrije val}. Assen bij het
lanceerpunt: $Ox$ horizontaal in het verticale vlak van $\vect v_0$,
$Oy$ verticaal omhoog; de \emph{lanceerhoek}\index{lanceerhoek}
$\alpha$ van $\vect v_0$ boven de horizontaal legt zijn coördinaten
$(v_0\cos\alpha,\; v_0\sin\alpha)$ vast.
\end{definition}

\begin{theorem}[Bewegingsvergelijkingen]\label{thm:g12:projectile-motion:equations}
Een \omterm{def:g12:projectile-motion:projectile}{projectiel} gelanceerd vanaf de
oorsprong met snelheid $v_0$ en hoek $\alpha$ heeft
$\vect a = (0,\,-g)$ op elk moment, dus
\[
\vect v(t) = \bigl(v_0\cos\alpha,\; v_0\sin\alpha - g t\bigr),
\qquad
\begin{cases}
x(t) = v_0\cos\alpha\; t,\\[2pt]
y(t) = v_0\sin\alpha\; t - \tfrac12\, g t^2 .
\end{cases}
\]
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{Newtons tweede wet} geeft
$\vect a = \vect g = (0,-g)$. Primitieven as per as: $\vect v(t)$
uit $\vect a$ met beginwaarde $(v_0\cos\alpha,\, v_0\sin\alpha)$,
dan positie uit $\vect v$, vanaf $(0,0)$.
\end{proof}

\begin{proposition}[Onafhankelijkheid van de twee bewegingen]\label{prop:g12:projectile-motion:independence}
De horizontale beweging is eenparig --- $v_x$ verandert nooit --- en
de verticale beweging is precies de verticale worp van de vorige
sectie: geen van beide vergelijkingen noemt de coördinaat van de
andere. In het bijzonder blijven een horizontaal afgevuurde bal en
een tegelijk van dezelfde hoogte losgelaten bal op gelijke hoogte
--- ze landen samen.
\end{proposition}

\begin{proof}
In \cref{thm:g12:projectile-motion:equations} betrekt $x(t)$ alleen
$v_0\cos\alpha$, $y(t)$ alleen $v_0\sin\alpha$ en $g$; beide ballen
hebben $v_y(0) = 0$, dus gehoorzamen beide $h = \tfrac12 gt^2$.
\end{proof}

\begin{remark}[De vallende aap]\label{rem:g12:projectile-motion:monkey}
Een dierenverzorger mikt een pijltje \emph{recht op} een aap, die op
het schot zijn tak loslaat. Slechte zet: pijltje en aap dalen elk
$\tfrac12 gt^2$ onder hun geen-zwaartekracht-posities --- de
richtlijn, de tak --- dus ontmoeten ze elkaar precies, wat de
snelheid van het pijltje ook is
(\cref{exo:g12:projectile-motion:15}).
\end{remark}

\section{De parabool, volledig gelezen}

\begin{theorem}[Baan]\label{thm:g12:projectile-motion:trajectory}
Eliminatie van $t$ tussen $x(t)$ en $y(t)$: de
\omterm{def:g10:relative-motion:trajectory}{baan} van een
\omterm{def:g12:projectile-motion:projectile}{projectiel}
($0 \leq \alpha < \ang{90}$) is de omlaag openende parabool
\[
y = x\tan\alpha \;-\; \frac{g}{2 v_0^2 \cos^2\alpha}\, x^2 .
\]
\end{theorem}

\begin{proof}
$t = x/(v_0\cos\alpha)$, gesubstitueerd in $y(t)$ van
\cref{thm:g12:projectile-motion:equations}.
\end{proof}

\begin{definition}[Reikwijdte]\label{def:g12:projectile-motion:range}
Voor een \omterm{def:g12:projectile-motion:projectile}{projectiel} over vlakke grond is de
\emph{reikwijdte}\index{reikwijdte}\index{dracht} $R$ de
horizontale afstand afgelegd wanneer hij terugkeert tot zijn
lanceerhoogte; de \emph{vluchttijd}\index{vluchttijd} $t_f$ is de
duur van die trip.
\end{definition}

\begin{proposition}[Top, vluchttijd, reikwijdte]\label{prop:g12:projectile-motion:range}
Over vlakke grond, met $0 < \alpha < \ang{90}$:
\[
t_f = \frac{2 v_0 \sin\alpha}{g}, \qquad
h_{\text{apex}} = \frac{(v_0\sin\alpha)^2}{2g}
\ \ \text{bereikt bij } t_f/2, \qquad
R = \frac{v_0^2 \sin(2\alpha)}{g}.
\]
De \omterm{def:g12:projectile-motion:range}{reikwijdte} is het grootst bij
$\alpha = \ang{45}$, waar $R = v_0^2/g$; complementaire hoeken
$\alpha$ en $\ang{90} - \alpha$ geven dezelfde
\omterm{def:g12:projectile-motion:range}{reikwijdte}.
\end{proposition}

\begin{proof}
$y = 0$ geeft $t\,(v_0\sin\alpha - \tfrac12 gt) = 0$: lancering, en
$t_f$. De top, $v_y = 0$, zit bij $t = v_0\sin\alpha/g = t_f/2$;
substitueer in $y(t)$. Dan $R = x(t_f) = 2v_0^2\sin\alpha\cos\alpha/g
= v_0^2\sin(2\alpha)/g$. De sinus piekt bij $2\alpha = \ang{90}$, en
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180} - 2\alpha) =
\sin(2\alpha)$.
\end{proof}

\begin{example}[Doeltrap]\label{ex:g12:projectile-motion:kick}
Een keeper schiet met $v_0 = \qty{25}{m/s}$, $\alpha = \ang{45}$:
$R = 25^2/9.81 \approx \qty{64}{m}$; echte clearances landen dichter
bij \qty{55}{m} --- de snede van de lucht.
\end{example}

\begin{method}[Projectielboekhouding]\label{met:g12:projectile-motion:method}
\begin{enumerate}
  \item Assen bij het lanceerpunt, $Ox$ horizontaal, $Oy$ omhoog;
        coördinaten van $\vect v_0$.
  \item Controleer \omterm{def:g12:projectile-motion:freefall}{vrije val}
        (\omterm{def:g10:universal-gravitation:weight}{gewicht} alleen): dan
        $\vect a = (0,-g)$; primitieven tweemaal, beginvoorwaarden
        elke keer.
  \item Vraag over een \emph{plaats}: elimineer $t$
        (\omterm{def:g10:relative-motion:trajectory}{baan}). Over een
        \emph{moment}: houd $t$.
  \item Woordenboek: top $\Leftrightarrow v_y = 0$; lanceerhoogte
        $\Leftrightarrow y = 0$; muur op $d \Leftrightarrow x = d$;
        vloer $\ell$ lager $\Leftrightarrow y = -\ell$.
  \item Gezond verstand: vlucht symmetrisch om de top;
        $R \leq v_0^2/g$; lucht verkort alleen maar.
\end{enumerate}
\end{method}

\section{Wat de lucht verandert}

\begin{definition}[Eindsnelheid]\label{def:g12:projectile-motion:terminal}
Lucht\omterm{def:g11:circuits-and-power:resistance}{weerstand} werkt de snelheid
tegen en groeit met de snelheid. Een vallend lichaam versnelt daarom
alleen tot de sleep het \omterm{def:g10:universal-gravitation:weight}{gewicht}
balanceert; vanaf daar $\vect a = \vect 0$: het valt met een constante
\emph{eindsnelheid}\index{eindsnelheid}.
\end{definition}

\begin{example}[Eind-ordes van grootte]\label{ex:g12:projectile-motion:airorders}
Een gestrekte skydiver: ongeveer \qty{55}{m/s} (\qty{200}{km/h}). Een
regendruppel: ongeveer \qty{9}{m/s} --- in vacuüm zou \qty{2000}{m}
val hem met $\sqrt{2gh} \approx \qty{2.0e2}{m/s}$ afleveren. Een
veer: ongeveer \qty{1}{m/s}, bereikt binnen centimeters --- het
geheim van zijn traagheid: hij brengt zijn hele val \emph{op}
\omterm{def:g12:projectile-motion:terminal}{eindsnelheid} door; de hamer komt
zijn eigen nooit nabij.
\end{example}

\begin{remark}[Echte ballistiek]\label{rem:g12:projectile-motion:drag}
Sleep duwt tegen de snelheid langs de hele vlucht: de
\omterm{def:g12:projectile-motion:range}{reikwijdte} verkort, de top daalt, en
de symmetrie breekt --- de daling steiler dan de klim, lange schoten
zichtbaar kort van de parabool. De eerlijke, kwantitatieve
behandeling is een differentiaalvergelijking opgelost in het volume
van jaar~1; dit jaar is de parabool het juiste model voor dichte,
matig snelle \omterm{def:g12:projectile-motion:projectile}{projectielen} --- een
basketbal bij \qty{10}{m/s}, niet een golfbal bij \qty{70}{m/s}.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:1}
Een steen losgelaten van een brug raakt het water \qty{2.0}{s} later:
hoogte van de brug, en impactsnelheid in \unit{m/s} en \unit{km/h}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:2}
Op de Maan liet een astronaut een hamer en een veer samen vallen: ze
landden samen. Waarom --- en waarom verliest de veer zo erg op Aarde?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:3}
Lucht verwaarlozend, bereken valtijd en aankomstsnelheid vanaf
\qty{1.0}{m}, \qty{10}{m}, \qty{100}{m}. Hoe groeien de tijden met de
hoogte?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:4}
Een bal wordt recht omhoog gegooid met \qty{15}{m/s}: tophoogte en
-tijd, totale tijd terug naar de hand, en retoursnelheid?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:projectile-motion:5}
Een knikker rolt van een tafel van hoogte \qty{0.80}{m} met
\qty{2.5}{m/s}. Vind de valtijd en de landingsafstand van de rand.
Hangt de valtijd van de knikkersnelheid af?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:6}
Uit de vijf-hoeken-figuur van de cursus: welke paren
\omterm{def:g12:projectile-motion:projectile}{lanceerhoeken} delen hun
\omterm{def:g12:projectile-motion:range}{reikwijdte}, en welke hoek wint? Voor
een \omterm{def:g12:projectile-motion:range}{reikwijdte} onder het maximum, welke
van de twee mogelijke hoeken vliegt langer? Motiveer met $t_f$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:7}
Vanaf een \qty{20}{m}-balkon wordt een bal recht \emph{omlaag}
gegooid met \qty{5.0}{m/s}. Vind de tijd tot de grond en de
impactsnelheid; controleer de laatste met een
\omterm{def:g10:energy-conservation:energy}{energie}-balans
(\cref{ch:g11:mechanical-energy}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:8}
Een \omterm{def:g12:projectile-motion:projectile}{projectiel} verlaat vlakke grond met
\qty{40}{m/s}, $\alpha = \ang{30}$:
\omterm{def:g12:projectile-motion:range}{vluchttijd}, tophoogte en
\omterm{def:g12:projectile-motion:range}{reikwijdte}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:9}
Modeleer een verspringer als
\omterm{def:g12:projectile-motion:projectile}{projectiel}: afzet met \qty{10.5}{m/s},
$\alpha = \ang{20}$. Bereken de
\omterm{def:g12:projectile-motion:range}{reikwijdte}; kampioenen springen bijna
\qty{9}{m} --- noem twee ingrediënten die het puntmodel weglaat.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:10}
Op \qty{1.2}{m} boven vlakke grond wordt één kogel losgelaten en een
identieke horizontaal afgevuurd met \qty{300}{m/s}. Welke landt
eerst? Bereken de valtijd en de landingsafstand van de afgevuurde.
Welk principe wordt getest?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:projectile-motion:11}
Toon algebraïsch dat $\alpha$ en $\ang{90} - \alpha$ hun
\omterm{def:g12:projectile-motion:range}{reikwijdte} delen, vind dan de twee
hoeken die een \qty{25}{m/s}-\omterm{def:g12:projectile-motion:projectile}{projectiel}
\qty{50}{m} verder laten landen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:12}
Water verlaat een tuinslang met \qty{8.0}{m/s}: grootste horizontale
reikwijdte? grootste verticale hoogte? de twee mondstukhoeken die

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:13}
Een bal wordt vanaf de grond geschopt met \qty{18}{m/s},
$\alpha = \ang{35}$, naar een \qty{4.0}{m}-haag \qty{15}{m} verder.
Met de \omterm{def:g10:relative-motion:trajectory}{baan}vergelijking, beslis of hij
vrijkomt, en met hoeveel.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:14}
Een \qty{80}{kg}-skydiver valt plat met
\omterm{def:g12:projectile-motion:terminal}{eindsnelheid} \qty{55}{m/s}.
(a) De sleep\omterm{def:g10:inertia:force}{kracht}? (b) Schets de vorm van
$v(t)$ vanaf de sprong. (c) Een regendruppel valt van \qty{2.0}{km}:
vacuüm-aankomstsnelheid vs.\ zijn echte \qty{9}{m/s}?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:projectile-motion:15}
De vallende aap, bewezen: een pijltje verlaat de oorsprong precies
gericht op een aap op afstand $D$, hoogte $H$
($\tan\alpha = H/D$), die bij $t = 0$ loslaat. Toon aan dat wanneer
het pijltje $x = D$ bereikt zijn hoogte gelijk is aan die van de aap
--- wat $v_0$ ook is.
\end{exercise}

\section{Probleem: De basketball-buzzer-beater}

\begin{problem}[{Weekendprobleem --- de basketball-buzzer-beater: een
vrije worp pijl voor pijl ontleed, vingertoppen vrijgemaakt met
centimeters, een full-court-heave gewogen tegen wat een arm kan
geven, een lob over het bord --- en een vonnis over het meest
theatrale schot in de sport}]
\label{pb:g12:projectile-motion:1}
De ring van een basketbalring is een horizontale ring \qty{3.05}{m}
boven de vloer; Nora laat elke schot los op \qty{2.00}{m}, dus met
assen bij het loslaatpunt ($Ox$ horizontaal naar de ring, $Oy$ omhoog)
zit het ringcentrum \qty{1.05}{m} boven de oorsprong. Lucht wordt
verwaarloosd behalve waar vermeld.

\medskip\noindent\textbf{Deel I --- De vrije worp.} Nora schiet vanaf
$d = \qty{4.20}{m}$ (horizontaal) met
\omterm{def:g12:projectile-motion:projectile}{lanceerhoek} $\alpha = \ang{50}$.
\begin{enumerate}
  \item Welke \omterm{def:g10:inertia:force}{krachten} werken eenmaal
        losgelaten? Geef $\vect a$, onder verwijzing naar de
        gebruikte wet.
  \item Leid de coördinaten van $\vect v(t)$ af.
  \item Leid $x(t)$ en $y(t)$ af.
  \item Elimineer $t$ om de
        \omterm{def:g10:relative-motion:trajectory}{baan} $y(x)$ te
        verkrijgen.
  \item De bal moet het ringcentrum passeren, $y(d) = \qty{1.05}{m}$:
        los op naar $v_0$, in \unit{m/s} en \unit{km/h}.
\end{enumerate}

\medskip\noindent\textbf{Deel II --- Over de vingertoppen, in de ring.}
\begin{enumerate}[resume]
  \item Hoe lang doet de bal erover om de ring te bereiken?
  \item Vind top-tijd en -hoogte; controleer dat de bal al daalt bij
        de ring.
  \item Een verdediger \qty{0.90}{m} voor Nora strekt vingertoppen tot
        \qty{2.60}{m}. Met hoeveel maakt de bal ze vrij?
  \item Bereken $v_x$ en $v_y$ bij de ring, de snelheid daar, en de
        hoek van de snelheid onder de horizontaal.
  \item Waarom moet een bal \emph{dalend} aankomen --- en waarom
        steil?
\end{enumerate}

\medskip\noindent\textbf{Deel III --- De full-court-heave.} Buzzer
op het punt om te \omterm{def:g10:signals-and-waves:sound}{klinken}, Nora is
\qty{26.0}{m} van de verre ring.
\begin{enumerate}[resume]
  \item De \qty{1.05}{m}-stijging is klein tegen \qty{26.0}{m}:
        rechtvaardig een vlakke-grondmodel, en leid de
        \omterm{def:g12:projectile-motion:range}{vluchttijd} af uit
        $y(t_f) = 0$.
  \item Leid $R = v_0^2\sin(2\alpha)/g$ af en de hoek van grootste
        \omterm{def:g12:projectile-motion:range}{reikwijdte}.
  \item Bereken de $v_0$ nodig om \qty{26.0}{m} te bestrijken bij
        $\alpha = \ang{45}$.
  \item Een sterke speler kan een basketbal met ongeveer
        \qty{17.5}{m/s} gooien. Wat is haar maximale
        \omterm{def:g12:projectile-motion:range}{reikwijdte} --- is het schot
        menselijk haalbaar?
  \item Bij \qty{17.5}{m/s}, vind de \emph{twee} hoeken die de bal
        \qty{26.0}{m} verder laten landen, en hun vluchttijden: met
        \qty{2.0}{s} op de klok, welke wint de buzzer?
\end{enumerate}

\medskip\noindent\textbf{Deel IV --- De lob over het bord.} Van
achter de baseline lobt Nora: loslaten \qty{6.0}{m} van het
ringcentrum, $\alpha = \ang{75}$; het vlak van het bord kruist haar
schot \qty{0.45}{m} vóór het ringcentrum, bovenrand \qty{3.95}{m}
boven de vloer.
\begin{enumerate}[resume]
  \item Vind de $v_0$ die de bal door het ringcentrum laat vallen.
  \item Bereken de hoogte van de bal bij het vlak van het bord
        ($x = \qty{5.55}{m}$): maakt hij de bovenrand vrij, met
        hoeveel?
  \item Vind de tophoogte en vluchttijd --- waarom houden menigten
        ervan?
  \item Kwalitatief, wat verandert luchtweerstand aan deze drie
        schoten, en welk bedreigt hij het meest?
  \item \emph{Het buzzer-beater-vonnis}, twee zinnen: vergelijk de
        snelheid die het full-court-schot eist met wat een arm
        levert, geef de marge, en oordeel --- mogelijk of mythe?
\end{enumerate}
\end{problem}
