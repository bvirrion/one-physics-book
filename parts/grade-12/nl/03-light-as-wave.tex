\chapter{Licht als golf: diffractie en interferentie}\label{ch:g12:light-as-wave}

Kantel een zeepfilm naar een raam: kleuren glijden erover, hoewel
niets in zeep of water gekleurd is. Richt een \omterm{def:g11:color-light-sources:lamps}{laser} op een
enkele haar: de muur erachter toont geen \omterm{def:g11:lenses-and-eye:lens}{haar-dunne}
schaduw maar een heldere streep, gebarst met donkere lijnen, duizenden
malen breder dan de haar. Recht-marcherende stralen
(\cref{ch:g10:refraction}) verklaren geen van beide. Dit hoofdstuk
promoveert licht tot een volwaardige \omterm{def:g10:signals-and-waves:wave}{golf} --- een half
micrometer van kam tot kam --- en verzilvert het idee: een
\omterm{def:g11:color-light-sources:lamps}{laser} en een kale muur meten een haar, een
\omterm{def:g12:mechanical-waves:wavelength}{golflengte}, de groeven van een cd.

\section{Licht is een golf}

\begin{definition}[Lichtgolven]\label{def:g12:light-as-wave:lightwave}
\omterm{def:g10:light-spectra:white}{Monochromatisch licht} --- één pure kleur, die geen prisma
verder kan splitsen (\cref{ch:g10:light-spectra}) --- is een
\omterm{def:g10:signals-and-waves:period}{periodieke} \emph{lichtgolf}\index{lichtgolf}: hij
doorkruist het vacuüm met $c = \qty{3.00e8}{m/s}$ en wordt
gekarakteriseerd door zijn \omterm{def:g10:signals-and-waves:frequency}{frequentie} $f$ of zijn
\emph{golflengte in vacuüm}\index{golflengte!van licht}
$\lambda = c/f$, de kam-tot-kam-afstand van
\cref{ch:g12:mechanical-waves}. Het oog reageert van ongeveer
\qty{400}{nm} (violet) tot \qty{800}{nm} (rood);
\omterm{def:g10:light-spectra:white}{wit licht} mengt het hele
\omterm{def:g11:fundamental-interactions:strong}{bereik}.
\end{definition}

\begin{example}[Ordes van grootte]\label{ex:g12:light-as-wave:orders}
Een helium--neon-\omterm{def:g11:color-light-sources:lamps}{laser} zendt rood licht uit van
$\lambda = \qty{633}{nm} = \qty{6.33e-7}{m}$, dus
$f = c/\lambda \approx \qty{4.7e14}{Hz}$: geen elektronica telt zo
snel --- elke \omterm{def:g12:mechanical-waves:wavelength}{golflengte} in dit hoofdstuk wordt
door meetkunde gemeten. Ter schaal is een haar (ongeveer
\qty{70}{\micro m}) honderd \omterm{def:g12:mechanical-waves:wavelength}{golflengten} breed. Wat
voor \omterm{def:g10:signals-and-waves:wave}{golven}? Niet de lucht: een elektrisch en
\omterm{def:g11:magnetic-fields:bfield}{magnetisch veld}, eerlijk beschreven in het volume
van jaar~2; alleen de \omterm{def:g12:mechanical-waves:wavelength}{golflengte} telt hieronder.
\end{example}

\section{Diffractie}

\begin{definition}[Diffractie]\label{def:g12:light-as-wave:diffraction}
Wanneer een \omterm{def:g10:signals-and-waves:wave}{golf} een opening of een obstakel van
grootte $a$ vergelijkbaar met zijn
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} ontmoet, spreidt hij zich uit in
het gebied dat stralen donker zouden laten. Die spreiding is
\emph{diffractie}\index{diffractie} --- verwaarloosbaar zolang
$\lambda/a$ piepklein is, dominant wanneer $a$ naar $\lambda$
krimpt.
\end{definition}

\begin{proposition}[Hoekhalve breedte van de spreiding]\label{prop:g12:light-as-wave:halfwidth}
Een spleet van breedte $a$, belicht met
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} $\lambda < a$, spreidt het licht
rond de invallende richting binnen de hoekhalve breedte $\theta$
(radialen); een scherm op afstand $D$ toont een centrale heldere
band van breedte $L$, tweemaal $D \tan\theta \approx D\,\theta$
(kleine hoek):
\[
\theta \approx \frac{\lambda}{a}, \qquad L = \frac{2\lambda D}{a}.
\]
\end{proposition}
\admitted

\begin{remark}[Lees de formule achterstevoren]\label{rem:g12:light-as-wave:honest}
Hoe \emph{smaller} de spleet, hoe \emph{breder} de spreiding --- het
tegengestelde van de straalvoorspelling. De eerlijke afleiding
(optellen van golfjes opnieuw uitgezonden over de spleet) leeft in
het volume van jaar~2; hier is de formule een meetinstrument.
\end{remark}

\begin{example}[Een laser door een spleet]\label{ex:g12:light-as-wave:slit}
Rode \omterm{def:g11:color-light-sources:lamps}{laser}, $\lambda = \qty{633}{nm}$; spleet
$a = \qty{0.10}{mm}$; scherm op $D = \qty{2.00}{m}$. Dan
$\theta = \num{6.33e-7} / \num{1.0e-4} = \qty{6.3e-3}{rad}$ --- een
derde graad, maar de centrale band op de muur is
$L = 2 \times \num{6.33e-7} \times 2.00 / \num{1.0e-4} \approx
\qty{2.5}{cm}$ breed, geflankeerd door donkere lijnen en zwakkere
zijbanden. Halveer de spleet en de band verdubbelt.
\end{example}

\begin{proposition}[Een obstakel diffracteert als een spleet]\label{prop:g12:light-as-wave:hair}
Een \omterm{def:g11:lenses-and-eye:lens}{dun} opaak obstakel van breedte $d$ --- een draad,
een haar --- produceert, buiten de invallende bundel, hetzelfde
patroon als een spleet van breedte $d$ (principe van Babinet,
afgeleid in het volume van jaar~2).
\end{proposition}
\admitted

\begin{method}[Een piepkleine breedte meten]\label{met:g12:light-as-wave:hairwidth}
\begin{enumerate}
  \item Span de haar over de \omterm{def:g11:color-light-sources:lamps}{laser}bundel; plaats een
        scherm op een gemeten afstand $D$ (enkele
        \omterm{def:g10:orders-of-magnitude:unit}{meters}).
  \item Meet de breedte $L$ van de centrale band, donkere lijn tot
        donkere lijn.
  \item Omdat $L = 2\lambda D/d$: $d = 2\lambda D/L$ --- fijnere haar,
        groter patroon.
\end{enumerate}
\end{method}

\begin{remark}[Diffractie beperkt elk instrument]\label{rem:g12:light-as-wave:limits}
Licht treedt een camera, een telescoop of een oog binnen door een
opening van breedte $a$, dus zelfs een perfecte
\omterm{def:g11:lenses-and-eye:lens}{lens} vervaagt een punt tot een vlek van
hoekgrootte ongeveer $\lambda/a$: dichterbij details smelten samen.
Een zichtbaar-lichtmicroscoop lost zo niets veel onder
$\lambda \approx \qty{0.5}{\micro m}$ op (bacteriën ja, virussen
nee); telescoopspiegels worden om die reden breed gebouwd.
\end{remark}

\section{Interferentie}

\begin{definition}[Coherente bronnen, weglengteverschil]\label{def:g12:light-as-wave:coherent}
Twee \omterm{def:g10:signals-and-waves:wave}{golf}bronnen zijn \emph{coherent}\index{coherente bronnen}
wanneer ze dezelfde \omterm{def:g10:signals-and-waves:frequency}{frequentie} hebben en in de
pas trillen --- in de praktijk wanneer ze twee kopieën van één
\omterm{def:g10:signals-and-waves:wave}{golf} zijn, verkregen door hem te splitsen. Voor een
punt $M$ bereikt door beide is het \emph{weglengteverschil}%
\index{weglengteverschil} $\delta = S_2M - S_1M$.
\end{definition}

\begin{proposition}[Interferentievoorwaarden]\label{prop:g12:light-as-wave:conditions}
Waar twee \omterm{def:g12:light-as-wave:coherent}{coherente} \omterm{def:g10:signals-and-waves:wave}{golven}
van \omterm{def:g12:mechanical-waves:wavelength}{golflengte} $\lambda$ overlappen, tellen ze
op --- ze \emph{interfereren}\index{interferentie}:
\begin{itemize}
  \item $\delta = k\lambda$ ($k$ geheel): in de pas, versterkend ---
        \emph{constructieve}\index{constructieve interferentie}
        interferentie (helder);
  \item $\delta = (k + \tfrac12)\lambda$: kam op dal, opheffend ---
        \emph{destructieve}\index{destructieve interferentie}
        interferentie (donker).
\end{itemize}
\end{proposition}

\begin{proof}
Een \omterm{def:g12:mechanical-waves:periodic}{periodieke golf} verschuiven over hele
\omterm{def:g12:mechanical-waves:wavelength}{golflengten} verandert niets: kammen stapelen
op kammen. Een half \omterm{def:g12:mechanical-waves:wavelength}{golflengte} meer zet kam
op dal: gelijke \omterm{def:g10:signals-and-waves:wave}{golven} heffen op.
\end{proof}

\begin{example}[Youngs tweespletenexperiment]\label{ex:g12:light-as-wave:young}
Prik twee fijne spleten $S_1$, $S_2$, een fractie van een millimeter
uit elkaar, in een opake plaat, en belicht beide met één
\omterm{def:g11:color-light-sources:lamps}{laser}: twee \omterm{def:g12:light-as-wave:coherent}{coherente
bronnen}. Op een scherm een paar \omterm{def:g10:orders-of-magnitude:unit}{meters}
verder is de overlapping gestreept: in het centrum $\delta = 0$,
helder; klimmend over het scherm groeit $\delta$ door $\lambda/2$
(donker), $\lambda$ (helder)\,\dots\ duisternis waar licht bij licht
opgeteld uitdooft.
\end{example}

\begin{definition}[Franjes en interfranje]\label{def:g12:light-as-wave:fringes}
De afwisselende heldere en donkere banden van het patroon zijn zijn
\emph{interferentiefranjes}\index{interferentiefranjes}; de afstand
$i$ tussen opeenvolgende heldere centra is de
\emph{interfranje}\index{interfranje}.
\end{definition}

\begin{proposition}[De interfranje]\label{prop:g12:light-as-wave:interfringe}
Voor spleten op afstand $a$ en een scherm op afstand $D \gg a$ heeft
het punt van het scherm op afstand $x$ van het centrum $O$
\omterm{def:g12:light-as-wave:coherent}{weglengteverschil} $\delta \approx a\,x/D$
(toegegeven meetkunde); volgens
\cref{prop:g12:light-as-wave:conditions} zitten heldere franjes op
$x_k = k\,\lambda D/a$, gelijkmatig gespatieerd met
\omterm{def:g12:light-as-wave:fringes}{interfranje}
\[
i = \frac{\lambda D}{a}.
\]
Met $\lambda = \qty{633}{nm}$, $a = \qty{0.20}{mm}$, $D =
\qty{2.0}{m}$: $i = \qty{6.3}{mm}$ --- de opstelling vergroot
$\lambda$ met $D/a = 10^4$. De meetkunde erachter wordt uitgewerkt
in het volume van jaar~2.
\end{proposition}
\admitted

\begin{method}[Een golflengte meten]\label{met:g12:light-as-wave:lambda}
\begin{enumerate}
  \item Stuur de onbekende \omterm{def:g11:color-light-sources:lamps}{laser} door een dubbele
        spleet van bekende afstand $a$; plaats het scherm op een
        gemeten afstand $D$.
  \item Meet de span van $10$ \omterm{def:g12:light-as-wave:fringes}{interfranjes}; deel door
        $10$: de \omterm{def:g12:light-as-wave:fringes}{interfranje} $i$, met de
        liniaal-fout tienvoudig verkleind.
  \item Dan $\lambda = i\,a/D$.
\end{enumerate}
\end{method}

\begin{remark}[Waarom twee lampen nooit interfereren]\label{rem:g12:light-as-wave:lamps}
Twee bureaulampen die één muur belichten produceren geen franjes:
elk zendt korte, ongecorreleerde \omterm{def:g10:signals-and-waves:wave}{golf}treinen uit
waarvan de relatieve timing willekeurig een miljard keer per seconde
springt, zodat de franjes even snel verschuiven en het oog ze tot
uniform licht midelt. Interferentie eist
\omterm{def:g12:light-as-wave:coherent}{coherentie} --- één \omterm{def:g10:signals-and-waves:wave}{golf}
in tweeën gesplitst, zoals Youngs spleten doen; de
\omterm{def:g11:color-light-sources:lamps}{laser} is de moderne shortcut, één schone
lange \omterm{def:g10:signals-and-waves:wave}{golf}.
\end{remark}

\section{Kleuren geschilderd door interferentie}

\begin{definition}[Irisatie]\label{def:g12:light-as-wave:iridescence}
Kleuren die verschuiven met de kijkhoek --- zeepbellen, olievlekken,
pauwenveren, de achterkant van een cd --- zijn
\emph{irisatie}\index{irisatie}: geen pigment, maar interferentie in
\omterm{def:g10:light-spectra:white}{wit licht}, die sommige
\omterm{def:g12:mechanical-waves:wavelength}{golflengten} versterkt en andere
uitdooft.
\end{definition}

\begin{example}[Zeepfilms en olievlekken]\label{ex:g12:light-as-wave:soap}
Licht dat een zeepfilm treft reflecteert deels van zijn voorvlak,
deels van zijn achtervlak: twee \omterm{def:g12:light-as-wave:coherent}{coherente}
kopieën waarvan het \omterm{def:g12:light-as-wave:coherent}{weglengteverschil} door de
dikte en de kijkhoek wordt bepaald. Elke dikte retourneert sommige
\omterm{def:g12:mechanical-waves:wavelength}{golflengten} versterkt, andere uitgedoofd: de
film reflecteert een kleur die glijdt naarmate hij leegloopt en
\omterm{def:g11:lenses-and-eye:lens}{dunner} wordt. Een olievlek op nat asfalt speelt
dezelfde truc.
\end{example}

\begin{remark}[Monochromatisch versus wit licht]\label{rem:g12:light-as-wave:white}
In \omterm{def:g10:light-spectra:white}{wit licht} trekt elke
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} zijn eigen Young-franjes met
zijn eigen $i = \lambda D/a$: in het centrum zijn alle kleuren
helder (één witte franje); een paar franjes verder drijven de
patronen uit elkaar --- iriserende randen, dan vervaging.
\omterm{def:g10:light-spectra:white}{Monochromatisch licht} geeft scherpe franjes
zonder einde: vandaar de \omterm{def:g11:color-light-sources:lamps}{laser} in elke
scherpe meting hier.
\end{remark}

\begin{exercise}[$\star\star\star$]\label{exo:g12:light-as-wave:12}

\begin{exercise}[$\star\star\star$]\label{exo:g12:light-as-wave:13}

\begin{exercise}[$\star\star\star$]\label{exo:g12:light-as-wave:14}

\section{Probleem: Meten met licht}

\begin{problem}[{Weekendprobleem --- meten met licht: een laser, een
liniaal en een kale muur worden een micrometeratelier dat een spleet,
een mensenhaar, een onbekende golflengte en de groeven van twee
zilveren schijven peilt}]
\label{pb:g12:light-as-wave:1}
Je kit: een rode \omterm{def:g11:color-light-sources:lamps}{laser}
($\lambda = \qty{633}{nm}$, van het etiket), een groene
\omterm{def:g11:color-light-sources:lamps}{laser} van onbekende
\omterm{def:g12:mechanical-waves:wavelength}{golflengte}, een geijkte spleet
$a = \qty{100}{\micro m}$, een dubbele spleet van afstand $a' =
\qty{0.250}{mm}$, een meetlint, twee haren, een cd en een dvd.

\medskip
\noindent\textbf{Deel I --- IJken op de bekende spleet.}
Rode \omterm{def:g11:color-light-sources:lamps}{laser}, spleet, muur op $D = \qty{3.00}{m}$.
\begin{enumerate}
  \item Bereken de hoekhalve breedte $\theta$ van de gediffracteerde
        bundel.
  \item Toon aan dat de centrale band breedte $L = 2\lambda D/a$
        heeft; bereken hem.
  \item Je meet $L = \qty{3.7}{cm}$: bereken de relatieve afwijking.
        Is de methode gevalideerd?
  \item Een tweede spleet is half zo breed, \qty{50}{\micro m}:
        voorspel zijn $L$. Formuleer de regel die spleetbreedte en
        patroonbreedte verbindt.
  \item Rechtvaardig de stap $\tan\theta \approx \theta$ van vraag 2:
        vergelijk $\theta$ en $\tan\theta$ bij deze hoek.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De haar.} Dezelfde
\omterm{def:g11:color-light-sources:lamps}{laser} en muur; een gespannen haar
vervangt de spleet.
\begin{enumerate}[resume]
  \item Welk toegegeven resultaat laat je de spleetformule voor een
        haar houden, en wat zegt het?
  \item De centrale band is $L = \qty{5.4}{cm}$ breed: bereken de
        diameter $d$ van de haar.
  \item Je liniaal leest $L$ tot $\pm\qty{0.2}{cm}$: geef het
        resulterende \omterm{def:g11:fundamental-interactions:strong}{bereik} voor $d$, als
        $d \pm \Delta d$.
  \item De haar van je vriend geeft $L = \qty{7.6}{cm}$: zijn
        diameter? Wiens haar is fijner, en waarom betekent fijner
        breder?
  \item Zou een bureaulamp de \omterm{def:g11:color-light-sources:lamps}{laser} kunnen
        vervangen? Geef twee redenen.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} van de groene
\omterm{def:g11:color-light-sources:lamps}{laser}.} Groene
\omterm{def:g11:color-light-sources:lamps}{laser}, dubbele spleet, muur op
$D = \qty{2.00}{m}$.
\begin{enumerate}[resume]
  \item Waarom moeten beide spleten door \emph{dezelfde}
        \omterm{def:g11:color-light-sources:lamps}{laser} worden belicht?
  \item Waarom is de centrale franje helder, wat de
        \omterm{def:g12:mechanical-waves:wavelength}{golflengte} ook is?
  \item Tien \omterm{def:g12:light-as-wave:fringes}{interfranjes} bestrijken
        \qty{4.26}{cm}. Geef $i$, en de reden om er tien tegelijk te
        meten.
  \item Leid de \omterm{def:g12:mechanical-waves:wavelength}{golflengte} van de groene
        \omterm{def:g11:color-light-sources:lamps}{laser} af.
  \item Het etiket van de maker zegt $\qty{532 \pm 10}{nm}$:
        consistent? Welke kleur verwacht je?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Zilveren schijven lezen.} De rode
\omterm{def:g11:color-light-sources:lamps}{laser} treft een cd onder normale
inval; zijn sporen werken als een reflectierooster dat (toegegeven)
$d \sin\theta = k\lambda$ gehoorzaamt. Op de muur op $D =
\qty{1.00}{m}$ maakt elke bundel van de eerste orde een vlek op $x$
van de centrale.
\begin{enumerate}[resume]
  \item Voor de cd, $x = \qty{43}{cm}$: bereken $\theta$. Waarom is de
        kleine-hoekshortcut van Deel I nu verboden?
  \item Leid de spoorafstand $d$ van de cd af.
  \item De dvd stuurt zijn vlek van de eerste orde naar
        $x = \qty{1.65}{m}$: bereken zijn $\theta$ en zijn
        spoorafstand.
  \item Vergelijk de twee afstanden. Waarom slaat een dvd meer op dan
        een cd?
  \item Sluit het notitieboek: lijst de deze weekend gemeten lengten,
        en stel in één zin wat licht tot een liniaal voor allen
        maakte.
\end{enumerate}
\end{problem}
