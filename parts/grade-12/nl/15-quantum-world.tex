\chapter{De kwantumwereld: fotonen en energieniveaus}\label{ch:g12:quantum-world}

Een \omterm{def:g10:light-spectra:wavelength}{infrarode} warmtelamp kan je huid de hele avond
baden en nooit markeren; tien minuten \omterm{def:g11:lenses-and-eye:lens}{dun}
bergzonlicht laat een brandwond. De lamp levert veel meer
\omterm{def:g10:energy-conservation:energy}{energie} --- maar
\omterm{def:g10:energy-conservation:energy}{energie} is niet wat huid markeert. Licht, toont
dit hoofdstuk, arriveert in korrels, en alleen een korrel die hard
genoeg raakt doet chemie: één
\omterm{def:g10:light-spectra:wavelength}{ultraviolette} korrel knapt een binding die
een miljard \omterm{def:g10:light-spectra:wavelength}{infrarode} korrels alleen warmen.
Volg de korrels en een eeuw opent: metalen die elektronen lekken,
\omterm{def:g11:fundamental-interactions:ladder}{atomen} zingend in
\omterm{def:g10:light-spectra:emission}{spectrale lijnen}, elektronen interfererend als
\omterm{def:g10:signals-and-waves:wave}{golven}.

\section{Licht arriveert in korrels: het foton}

\begin{definition}[Het foton]\label{def:g12:quantum-world:photon}
Licht van \omterm{def:g12:oscillators-and-time:oscillator}{frequentie} $\nu$ wisselt
\omterm{def:g10:energy-conservation:energy}{energie} met materie alleen in hele korrels
genaamd \emph{fotonen}\index{foton}, elk van
\omterm{def:g10:energy-conservation:energy}{energie}
\[
E = h\nu = \frac{hc}{\lambda}, \qquad h = \qty{6.63e-34}{J.s},
\]
waar $h$ de \emph{constante van Planck}\index{constante van Planck} is
(Planck, 1900; Einstein, 1905) en $\lambda$ de
\omterm{def:g12:light-as-wave:lightwave}{golflengte in vacuüm}. De intensiteit van een
bundel meet het \emph{aantal} fotonen per seconde; de
\omterm{def:g10:energy-conservation:energy}{energie} van elke korrel is vastgelegd door
de \omterm{def:g12:oscillators-and-time:oscillator}{frequentie} alleen.
\end{definition}

\begin{notation}[De elektronvolt]\label{not:g12:quantum-world:ev}
Atoomfysica gebruikt de \emph{elektronvolt}\index{elektronvolt},
$\qty{1}{eV} = \qty{1.60e-19}{J}$: de
\omterm{def:g10:energy-conservation:energy}{energie} die één
\omterm{def:g11:fundamental-interactions:elementary}{elementaire lading} wint bij het
oversteken van \qty{1}{V} (\cref{ch:g11:electric-gravitational-fields}).
Handig: $E \,(\unit{eV}) \approx 1240/\lambda\,(\unit{nm})$.
\end{notation}

\section{Het foto-elektrisch effect}

Schijn licht op een schone metalen plaat in vacuüm en elektronen
springen uit --- het \emph{foto-elektrisch effect}\index{foto-elektrisch
effect}. Zijn regels braken de klassieke natuurkunde.

\begin{proposition}[Wat het experiment toont]\label{prop:g12:quantum-world:facts}
Voor elk metaal is er een \emph{drempelfrequentie}\index{drempelfrequentie} $\nu_0$:
\begin{enumerate}
  \item onder $\nu_0$ vertrekt geen elektron, hoe intens het licht ook;
  \item boven $\nu_0$ start emissie zonder
        \omterm{def:g12:mechanical-waves:delay}{vertraging}, hoe zwak ook;
  \item intensiteit verhoogt het \emph{aantal} elektronen per seconde,
        niet hun maximale \omterm{def:g11:mechanical-energy:kinetic}{kinetische energie};
        \omterm{def:g12:oscillators-and-time:oscillator}{frequentie} verhoogt $E_{k,\max}$.
\end{enumerate}
Niets past bij een continue \omterm{def:g10:signals-and-waves:wave}{golf}, die elke kleur
lading zou laten opladen --- over maanden bij zwak licht
(\cref{exo:g12:quantum-world:14}) --- en waarvan de intensiteit de
\omterm{def:g10:energy-conservation:energy}{energie} van de elektronen zou zetten: alle drie
voorspellingen falen.
\end{proposition}
\admitted

\begin{theorem}[Einsteins foto-elektrische relatie]\label{thm:g12:quantum-world:einstein}
\index{foto-elektrische relatie}Een elektron heeft minstens de
\emph{uittreedarbeid}\index{uittreedarbeid} $W_0$ van het metaal nodig om
zijn oppervlak te ontvluchten. Eén
\omterm{def:g12:quantum-world:photon}{foton} wordt geheel door één elektron
geabsorbeerd, dus
\[
h\nu = W_0 + E_{k,\max}, \qquad \text{vandaar } \nu_0 = \frac{W_0}{h}.
\]
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energie}boekhouding: $h\nu$ in, $W_0$ uitgegeven
aan ontsnapping, de rest kinetisch --- ten hoogste $h\nu - W_0$, minder
voor elektronen die dieper starten. Onder $\nu_0$ kan één korrel de
uitgangstol niet betalen, welk hun aantal ook.
\end{proof}

\begin{definition}[Rempotentiaal]\label{def:g12:quantum-world:stopping}
Om $E_{k,\max}$ te meten remt een omgekeerde
\omterm{def:g11:circuits-and-power:voltage}{spanning} de elektronen
(\cref{ch:g11:electric-gravitational-fields}); de
\emph{rempotentiaal}\index{rempotentiaal} $U_s$ is de kleinste
\omterm{def:g11:circuits-and-power:voltage}{spanning} die de stroom uitdooft: de snelste
elektronen slagen er net niet in over te steken, dus
$E_{k,\max} = e\,U_s$.
\end{definition}

\begin{method}[De constante van Planck meten]\label{met:g12:quantum-world:measure-h}
\begin{enumerate}
  \item Belicht een fotocel door \omterm{def:g11:color-light-sources:subtractive}{filters} van
        bekende \omterm{def:g12:mechanical-waves:wavelength}{golflengten}; bereken elke
        $\nu = c/\lambda$; noteer elke
        \omterm{def:g12:quantum-world:stopping}{rempotentiaal}.
  \item Zet $E_{k,\max} = eU_s$ uit tegen $\nu$: Einstein belooft de
        rechte lijn $h\nu - W_0$.
  \item Lees de helling: ze \emph{is} $h$; de intercepten geven $\nu_0$
        en $-W_0$.
\end{enumerate}
\end{method}

\section{Energieniveaus: waarom atomen lijnen uitzenden}

\begin{definition}[Energieniveaus]\label{def:g12:quantum-world:levels}
De \omterm{def:g10:energy-conservation:energy}{energie} van een elektron gebonden in een
\omterm{def:g11:fundamental-interactions:ladder}{atoom} is
\emph{gekwantiseerd}\index{kwantisatie}: beperkt tot een discrete ladder
van \emph{energieniveaus}\index{energieniveau} $E_1 < E_2 < \dots < 0$,
negatief geteld vanaf het vrije elektron. De laagste sport is de
\emph{grondtoestand}\index{grondtoestand}, de andere
\emph{aangeslagen toestanden}\index{aangeslagen toestand}; $0$ bereiken
is \emph{ionisatie}\index{ionisatie}.
\end{definition}

\begin{proposition}[De waterstofladder]\label{prop:g12:quantum-world:hydrogen}
De \omterm{def:g12:quantum-world:levels}{energieniveaus} van het
waterstof\omterm{def:g11:fundamental-interactions:ladder}{atoom} zijn
\[
E_n = -\frac{\qty{13.6}{eV}}{n^2}, \qquad n = 1, 2, 3, \dots
\]
dus $E_1 = \qty{-13.6}{eV}$, $E_2 = \qty{-3.40}{eV}$, $E_3 =
\qty{-1.51}{eV}$, $E_4 = \qty{-0.85}{eV}$, opeengepakt naar $0$.
\end{proposition}
\admitted

\begin{remark}[Waar de ladder vandaan komt]\label{rem:g12:quantum-world:ladder-origin}
Waarom deze getallen is kwantummechanica, bewaard voor de
universiteitsvolumes --- evenals de Broglies formule: de
\omterm{def:g12:quantum-world:debroglie}{materiegolf} van het elektron
(\cref{def:g12:quantum-world:debroglie} hieronder) moet op zichzelf
sluiten rond de \omterm{def:g11:fundamental-interactions:ladder}{kern}, en alleen
bepaalde \omterm{def:g12:mechanical-waves:wavelength}{golflengten} passen --- als een
gitaarsnaar heeft een \omterm{def:g11:fundamental-interactions:ladder}{atoom} een discrete
set noten.
\end{remark}

\begin{theorem}[Foton van een overgang]\label{thm:g12:quantum-world:transition}
Een \omterm{def:g11:fundamental-interactions:ladder}{atoom} wisselt van niveau alleen door
één \omterm{def:g12:quantum-world:photon}{foton} uit te zenden of te absorberen dat
precies het verschil draagt: dalen van niveau $p$ naar een lager niveau
$n$ zendt $h\nu = E_p - E_n$ uit; klimmen van $n$ naar $p$ absorbeert
diezelfde \omterm{def:g10:energy-conservation:energy}{energie}.
\end{theorem}

\begin{proof}
\omterm{def:g10:energy-conservation:energy}{Energie}behoud, één korrel tegelijk: het
\omterm{def:g12:quantum-world:photon}{foton} is de enige munteenheid die licht
aanvaardt, en de boeken van het
\omterm{def:g11:fundamental-interactions:ladder}{atoom} moeten kloppen.
\end{proof}

\begin{remark}[Lijnspectra verklaard]\label{rem:g12:quantum-world:lines}
Hier is het antwoord beloofd door \cref{ch:g10:light-spectra}: een heet
gas zendt alleen de frequenties $(E_p - E_n)/h$ van zijn eigen ladder
uit --- een barcode van heldere lijnen --- en een koel gas steelt precies
die frequenties uit \omterm{def:g10:light-spectra:white}{wit licht}, print
overeenkomende donkere absorptielijnen. Elk element heeft zijn eigen
ladder, dus zijn eigen barcode --- hoe we sterrenlicht lezen.
\end{remark}

\begin{example}[De zichtbare lijnen van waterstof]\label{ex:g12:quantum-world:balmer}
Overgangen die eindigen op $n = 2$ vallen in het zichtbare. Met
$E \,(\unit{eV}) = 1240/\lambda\,(\unit{nm})$: $3 \to 2$: \qty{1.89}{eV},
\qty{656}{nm} (rood); $4 \to 2$: \qty{2.55}{eV}, \qty{486}{nm}
(blauwgroen); $5 \to 2$: \qty{2.86}{eV}, \qty{434}{nm} (violet).
Overgangen naar $n = 1$ overschrijden \qty{10}{eV}: alle
\omterm{def:g10:light-spectra:wavelength}{ultraviolet}.
\end{example}

\begin{remark}[Lasers, in één adem]\label{rem:g12:quantum-world:laser}
Een \omterm{def:g12:quantum-world:photon}{foton} van precies $E_p - E_n$ dat een
\emph{aangeslagen} \omterm{def:g11:fundamental-interactions:ladder}{atoom} passeert kan
het triggeren om een identiek \omterm{def:g12:quantum-world:photon}{foton} uit te zenden
--- zelfde \omterm{def:g10:energy-conservation:energy}{energie}, zelfde richting, in de
pas. Houd meer \omterm{def:g11:fundamental-interactions:ladder}{atomen} omhoog dan omlaag
en één \omterm{def:g12:quantum-world:photon}{foton} wordt een lawine van klonen:
\emph{gestimuleerde emissie}\index{gestimuleerde emissie} --- en een
\omterm{def:g11:color-light-sources:lamps}{laser} is weinig meer.
\end{remark}

\section{Materiegolven}

\begin{proposition}[Elektronen interfereren]\label{prop:g12:quantum-world:interference}
Stuur elektronen één voor één door een dubbele spleet
(\cref{ch:g12:light-as-wave}): elk landt als een enkele scherpe stip ---
een deeltje. Maar als duizenden accumuleren, tekenen de stippen
\omterm{def:g12:light-as-wave:fringes}{interferentieranden}: elk elektron, alleen, passeert
als een \omterm{def:g10:signals-and-waves:wave}{golf} door \emph{beide} spleten.
Kristallen diffracteren elektronenbundels ook (Davisson en Germer,
1927).
\end{proposition}
\admitted

\begin{definition}[De Broglie-golflengte]\label{def:g12:quantum-world:debroglie}
Elk deeltje van \omterm{def:g12:newtons-laws:momentum}{impuls} $p = mv$ reist als een
\emph{materiegolf}\index{materiegolf} van \emph{de
Broglie-golflengte}\index{de Broglie-golflengte}
\[
\lambda = \frac{h}{p} = \frac{h}{mv}.
\]
\end{definition}
\admitted

\begin{example}[Waarom jij nooit diffracteert]\label{ex:g12:quantum-world:never}
Een elektron versneld door \qty{100}{V} heeft $E_k = \qty{1.60e-17}{J}$,
$p = \sqrt{2m_e E_k} = \qty{5.4e-24}{kg.m/s}$, dus $\lambda \approx
\qty{0.12}{nm}$ --- de afstand van
\omterm{def:g11:fundamental-interactions:ladder}{atomen} in een kristal, dat dus als
zijn dubbele spleet fungeert. Een tennisbal van \qty{58}{g} bij
\qty{25}{m/s}: $\lambda = 6.63\times10^{-34}/1.45 \approx
\qty{4.6e-34}{m}$, zo'n $10^{19}$ maal kleiner dan een
\omterm{def:g11:fundamental-interactions:ladder}{kern} --- geen spleet zal ooit randen
tonen. $h$ is zo klein dat kwantumkorreligheid alleen op de schaal van
het \omterm{def:g11:fundamental-interactions:ladder}{atoom} opduikt.
\end{example}

\begin{example}[De elektronenmicroscoop]\label{ex:g12:quantum-world:microscope}
\omterm{def:g12:light-as-wave:diffraction}{Diffractie} verbiedt detail veel kleiner dan de
gebruikte \omterm{def:g12:mechanical-waves:wavelength}{golflengte} op te lossen
(\cref{ch:g12:light-as-wave}): zichtbaar licht stopt nabij
\qty{0.5}{\micro m}. Elektronen bij \qty{10}{kV} hebben
$\lambda \approx \qty{12}{pm}$ --- tienduizenden malen korter:
elektronenmicroscopen beelden virussen, celmachines, zelfs enkele
\omterm{def:g11:fundamental-interactions:ladder}{atomen}. De
\omterm{def:g12:quantum-world:debroglie}{materiegolf} gaf de natuurkunde een nieuw
oog.
\end{example}

\section{Probleem: Het zonnepaneel-fysicalab}

\begin{problem}[{Weekendprobleem --- het zonnepaneel-fysicalab: de
$10^{21}$ korrels die een dak elke seconde drinkt, een fotocel die de
constante van Planck meet, een waterstoflamp lijn voor lijn gelezen ---
één weekend, één $h$}]
\label{pb:g12:quantum-world:1}
Jullie klas instrumenteert het schoolzonnepaneel voor een weekend.
Gegevens: zonne-instraling op de middag \qty{1.0e3}{W/m^2}; gemiddelde
zonne\omterm{def:g12:mechanical-waves:wavelength}{golflengte} \qty{550}{nm};
paneeloppervlak \qty{1.0}{m^2},
\omterm{def:g10:energy-conservation:efficiency}{rendement} $20\%$; de dag levert het
equivalent van \qty{5.0}{h} volle zon.

\medskip
\noindent\textbf{Deel I --- \omterm{def:g12:quantum-world:photon}{Foton}boekhouding.}
\begin{enumerate}
  \item Bereken de \omterm{def:g10:energy-conservation:energy}{energie} van één
        gemiddeld zonne\omterm{def:g12:quantum-world:photon}{foton}, in
        \omterm{def:g11:work-of-force:work}{joule} en elektronvolt.
  \item Leid het aantal \omterm{def:g12:quantum-world:photon}{fotonen} af dat het paneel
        per seconde op de middag treft.
  \item Bereken het \omterm{prop:g11:circuits-and-power:power}{elektrische vermogen} van
        het paneel op de middag, dan de
        \omterm{def:g10:energy-conservation:energy}{energie} over de dag, in
        \unit{kW.h}.
  \item De stroom van het paneel lijkt perfect glad. Met vraag 2, leg
        uit waarom de korrel-voor-korrel-aankomst onzichtbaar is.
  \item Zonlicht draagt meer \omterm{def:g10:light-spectra:wavelength}{infrarode}
        \omterm{def:g10:energy-conservation:energy}{energie} dan
        \omterm{def:g10:light-spectra:wavelength}{ultraviolet}, toch brandt alleen
        \omterm{def:g10:light-spectra:wavelength}{ultraviolet} huid (bindingen
        $\sim \qtyrange{3}{4}{eV}$). Leg uit in korrels.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- $h$ meten met de fotocel van de kit.}
De fotocel van de kit heeft een onbekende kathode.
\omterm{def:g11:color-light-sources:subtractive}{Filters} geven:
\begin{center}\small
\begin{tabular}{lcccc}
$\lambda$ (\unit{nm}) & 365 & 405 & 436 & 546\\
$U_s$ (\unit{V}) & 1.50 & 1.16 & 0.94 & 0.37
\end{tabular}
\end{center}
\begin{enumerate}[resume]
  \item Wat meet de \omterm{def:g12:quantum-world:stopping}{rempotentiaal}? Relateer
        $U_s$, $\nu$, $W_0$, $h$.
  \item Zet de vier \omterm{def:g12:mechanical-waves:wavelength}{golflengten} om in
        frequenties (in $10^{14}\,\unit{Hz}$).
  \item Waarom moet $eU_s$ tegen $\nu$ uitzetten een rechte lijn
        geven? Controleer de uitlijning van de punten.
  \item Uit de extreme punten, bereken de helling: jullie gemeten $h$.
  \item Leid de \omterm{thm:g12:quantum-world:einstein}{uittreedarbeid} van de kathode
        af (\unit{eV}), \omterm{prop:g12:quantum-world:facts}{drempelfrequentie} en
        drempel\omterm{def:g12:mechanical-waves:wavelength}{golflengte}.
  \item Welk deel van het zonlicht kon elektronen uit deze kathode
        stoten --- en welke helft van het
        \omterm{def:g12:sound-acoustics:timbre}{spectrum} is nutteloos ervoor?
  \item Vergelijk jullie $h$ met \qty{6.63e-34}{J.s}: relatieve fout in
        procent?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De waterstofkalibratielamp.}
De \omterm{def:g12:mechanical-waves:wavelength}{golflengte}schaal van de kit wordt
gecontroleerd tegen een waterstoflamp,
$E_n = -13.6/n^2$ \unit{eV}.
\begin{enumerate}[resume]
  \item Bereken $E_2$, $E_3$, $E_4$ en $E_5$.
  \item Bereken de \omterm{def:g10:energy-conservation:energy}{energieën} van de
        \omterm{def:g12:quantum-world:photon}{fotonen} van de overgangen $3 \to 2$,
        $4 \to 2$, $5 \to 2$.
  \item Leid de drie \omterm{def:g12:mechanical-waves:wavelength}{golflengten} af en geef
        de kleur van elke lijn.
  \item Waarom zendt de lamp \emph{lijnen} uit in plaats van een
        continue regenboog? Eén zin, noem het kernidee.
  \item Kon de waterstof van de lamp licht bij \qty{500}{nm}
        absorberen? Motiveer met de ladder.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De cellen inspecteren: de elektronen\omterm{def:g10:signals-and-waves:wave}{golf}.}
Microscheuren in een cel liggen ver onder het bereik van
\qty{0.5}{\micro m} van lichtmicroscopen; de scanning-elektronenmicroscoop
van het lab draait op \qty{10}{kV}.
\begin{enumerate}[resume]
  \item Bereken de snelheid van de elektronen; controleer dat die onder
        $0.25\,c$ blijft (klassieke formule).
  \item Bereken hun \omterm{def:g12:newtons-laws:momentum}{impuls} en
        \omterm{def:g12:quantum-world:debroglie}{de Broglie-golflengte}.
  \item Resolutie schaalt met \omterm{def:g12:mechanical-waves:wavelength}{golflengte}:
        bereken de winst over groen licht (\qty{550}{nm}); concludeer
        met de twee getallen van het weekend --- jullie $h$ en deze
        winst.
\end{enumerate}
\end{problem}
