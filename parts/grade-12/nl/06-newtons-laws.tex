\chapter{De wetten van Newton}\label{ch:g12:newtons-laws}

Smoesel een badkamerweegschaal een lift in. Als de cabine omhoog
trekt, leest de naald drie \omterm{def:g10:orders-of-magnitude:unit}{kilogram} te zwaar; nabij de
top leest hij te licht; daartussen de gewone waarheid --- niets aan
boord veranderde behalve de beweging. Dit hoofdstuk stelt de drie
wetten achter die naald en smeed ze tot de methode die heel de
mechanica leidt: kies een systeem, teken zijn
\omterm{def:g10:inertia:force}{krachten}, projecteer $\sum \vect F = m\vect a$.

\section{Impuls, en de eerste wet exact gemaakt}

\begin{definition}[Impuls]\label{def:g12:newtons-laws:momentum}
De \emph{impuls}\index{impuls}\index{hoeveelheid van beweging} (of
\emph{hoeveelheid van beweging}) van een lichaam van massa $m$
(\unit{kg}) dat met snelheid $\vect v$ beweegt is de vector
\[
\vect p = m \vect v \qquad (\unit{kg.m/s}):
\]
de snelheid, gewogen met de massa; voor een systeem, de vectorsom
over zijn delen.
\end{definition}

\begin{definition}[Inertiaalstelsel]\label{def:g12:newtons-laws:inertial}
Een \emph{inertiaalstelsel}\index{inertiaalstelsel} is een stelsel
waarin elk geïsoleerd lichaam --- geen
\omterm{def:g10:inertia:force}{krachten}, of
\omterm{def:g10:inertia:force}{krachten} die
\omterm{def:g10:inertia:compensate}{elkaar opheffen} --- in een
rechte lijn beweegt met constante snelheid (\cref{ch:g10:inertia}).
De grond is er één tot uitstekende nauwkeurigheid, en zo is elke
cabine in eenparige rechtlijnige beweging erover; een remmende trein
of draaiende carrousel niet: daar versnellen losse bagages ongeduwd.
\end{definition}

\begin{theorem}[Eerste wet van Newton]\label{thm:g12:newtons-laws:first}\index{eerste wet van Newton}
In een inertiaalstelsel verandert de
\omterm{def:g12:newtons-laws:momentum}{impuls} van een geïsoleerd systeem niet:
$\vect p$ is constant, d.w.z.\ $d\vect p/dt = \vect 0$.
\end{theorem}
\admitted

\begin{remark}[Traagheid, tweemaal aangescherpt]\label{rem:g12:newtons-laws:sharpened}
Het traagheidsprincipe van \cref{ch:g10:inertia}, tweemaal
opgewaardeerd: het zegt \emph{waar} het geldt --- het definieert de
inertiaalstelsels als die waarin het geldt --- en het bewaakt een
vector, $m\vect v$, waarvan de boekhouding aan het eind van het
hoofdstuk uitbetaalt.
\end{remark}

\section{Tweede wet van Newton}

\begin{theorem}[Tweede wet van Newton]\label{thm:g12:newtons-laws:second}\index{tweede wet van Newton}
In een inertiaalstelsel is de
\omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} op een lichaam
gelijk aan de veranderingssnelheid van zijn
\omterm{def:g12:newtons-laws:momentum}{impuls}; voor constante massa, omdat
$\vect p = m\vect v$,
\[
\sum \vect F = \frac{d\vect p}{dt} = m\,\frac{d\vect v}{dt} = m \vect a .
\]
\end{theorem}
\admitted

\begin{remark}[Een natuurwet, en zijn eenheid]\label{rem:g12:newtons-laws:nature}
Er is geen bewijs: dit is een natuurwet, gewogen tegen experiment
gedurende drie eeuwen. Vorig jaar (\cref{ch:g11:forces-and-motion})
gaf zijn schaduw, $\Delta v \propto F\,\Delta t/m$; de afgeleide
(\cref{ch:g12:kinematics-2d}) maakt hem exact op elk moment --- de
kleine lettertjes wachten in het volume van jaar~1. De
\omterm{def:g10:inertia:force}{kracht} krijgt een
\omterm{def:g10:orders-of-magnitude:unit}{eenheid},
$\unit{N} = \unit{kg.m/s^2}$; en een lichaam onder alleen zijn
\omterm{def:g10:universal-gravitation:weight}{gewicht} gehoorzaamt
$m\vect a = m\vect g$: elke massa valt met $\vect a = \vect g$, en
$g = \qty{9.81}{N/kg} = \qty{9.81}{m/s^2}$ --- de twee lezingen
zijn één.
\end{remark}

\begin{example}[Gemiddelde kracht, zonder detail]\label{ex:g12:newtons-laws:average}
Een \qty{1300}{kg}-auto gaat van stilstand naar \qty{100}{km/h}
(\qty{27.8}{m/s}) in \qty{8.0}{s}: wat de moment-tot-moment-geschiedenis
ook is, de gemiddelde \omterm{def:g11:forces-and-motion:netforce}{resulterende kracht}
is $\Delta p/\Delta t = 1300 \times 27.8/8.0 \approx \qty{4.5e3}{N}$
--- \omterm{def:g12:newtons-laws:momentum}{impuls} maakt van een
\omterm{def:g10:inertia:force}{kracht}vraag voor-en-na-boekhouding.
\end{example}

\section{Derde wet van Newton}

\begin{theorem}[Derde wet van Newton]\label{thm:g12:newtons-laws:third}\index{derde wet van Newton}
Als een lichaam $A$ een \omterm{def:g10:inertia:force}{kracht}
$\vect F_{A \to B}$ uitoefent op een lichaam $B$, dan oefent $B$ op
$A$ de \omterm{def:g10:inertia:force}{kracht}
$\vect F_{B \to A} = -\vect F_{A \to B}$ uit: zelfde werklijn,
zelfde grootte, tegengestelde richtingen --- op elk moment, wat de
beweging ook is, contact of werking op afstand.
\end{theorem}
\admitted

\begin{example}[Het boek, de tafel en de Aarde]\label{ex:g12:newtons-laws:book}
Een boek rust op een tafel: twee interacties, twee paren.
Gravitationeel: de Aarde trekt het boek omlaag ($\vect P$), het
boek trekt de Aarde omhoog. Contact: de tafel duwt het boek omhoog
($\vect N$), het boek drukt de tafel omlaag. Lopen is dezelfde deal:
duw de grond achteruit; zijn reactie is de enige
\omterm{def:g10:inertia:force}{kracht} die je vooruit drijft.
\end{example}

\section{De methode: kies, teken, projecteer}

\begin{method}[Een dynamica-probleem oplossen]\label{met:g12:newtons-laws:method}
\begin{enumerate}
  \item \emph{Systeem}: noem het lichaam (of de verzameling lichamen)
        dat wordt bestudeerd.
  \item \emph{Stelsel}: kies een inertiaalstelsel en assen die bij de
        beweging passen --- hellingassen, of het
        \omterm{def:g12:kinematics-2d:frenet}{Frenet-stelsel} voor een kromme
        (\cref{ch:g12:kinematics-2d}).
  \item \emph{Inventaris}: lijst elke
        \omterm{def:g10:inertia:force}{kracht} --- het
        \omterm{def:g10:universal-gravitation:weight}{gewicht}, dan één
        \omterm{def:g10:inertia:force}{kracht} per contact, niets
        anders --- en teken het vrijlichaamdiagram: één punt, één
        pijl per \omterm{def:g10:inertia:force}{kracht}.
  \item \emph{Projecteer}: schrijf $\sum \vect F = m\vect a$,
        projecteer op elke as.
  \item \emph{Los op en controleer}:
        \omterm{def:g10:orders-of-magnitude:unit}{eenheden}, tekens,
        limietgevallen ($\mu \to 0$, $\alpha \to 0$, \dots).
\end{enumerate}
\end{method}

\begin{definition}[Normaalkracht en wrijving]\label{def:g12:newtons-laws:friction}
Een oppervlak werkt op een lichaam via twee componenten: de
\emph{normaalkracht}\index{normaalkracht} $\vect N$, loodrecht op
het oppervlak, en \omterm{def:g10:inertia:inventory}{wrijving} $\vect f$,
tangentieel. Experiment geeft de
\omterm{def:g10:inertia:inventory}{wrijvings}wetten: zolang het
lichaam niet glijdt, past statische
\omterm{def:g10:inertia:inventory}{wrijving} zichzelf aan tot een
plafond, $f_s \leq \mu_s N$; eenmaal glijdend is
\omterm{def:g11:forces-and-motion:friction}{kinetische wrijving}
$f_k = \mu_k N$, tegen het glijden. De
\emph{wrijvingscoëfficiënten}\index{wrijvingscoëfficiënt}
$\mu_s$ en $\mu_k$ ($\mu_k \leq \mu_s$) hangen alleen van de twee
materialen af.
\end{definition}

\begin{example}[De helling, tweemaal]\label{ex:g12:newtons-laws:incline}
Een blok glijdt een helling van hoek $\alpha$ af; assen: $x$ de
helling af, $y$ loodrecht. Op $y$: $N = mg\cos\alpha$. Op $x$,
wrijvingsloos: $ma = mg\sin\alpha$, dus $a = g\sin\alpha$ --- geen
massa; bij $\alpha = 30^\circ$, $a = \qty{4.9}{m/s^2}$. Met
\omterm{def:g11:forces-and-motion:friction}{kinetische wrijving}:
$a = g(\sin\alpha - \mu_k\cos\alpha)$; voor $\mu_k = 0.20$,
$a = 9.81\,(0.500 - 0.173) = \qty{3.2}{m/s^2}$. En het bleef in
eerste instantie staan zolang de benodigde statische
\omterm{def:g10:inertia:inventory}{wrijving} onder zijn plafond paste:
$\tan\alpha \leq \mu_s$.
\end{example}

\begin{definition}[Schijnbaar gewicht]\label{def:g12:newtons-laws:apparent}
Het \emph{schijnbare gewicht}\index{schijnbaar gewicht} van een
lichaam is de \omterm{def:g12:newtons-laws:friction}{normaalkracht} tussen
het en zijn steun --- wat een weegschaal eronder afleest.
\end{definition}

\begin{example}[De lift]\label{ex:g12:newtons-laws:elevator}
Een \qty{70}{kg}-passagier staat op een weegschaal in een lift van
verticale versnelling $a_z$ (omhoog positief): $N - mg = ma_z$, dus
$N = m(g + a_z)$. Omhoog trekken bij $a_z = \qty{1.5}{m/s^2}$:
\qty{792}{N}, zwaar; remmen nabij de top: \qty{582}{N}, licht;
stabiele cruise: $mg = \qty{687}{N}$ --- de eerste wet; vrije val,
$a_z = -g$: $N = 0$, gewichtloosheid op een weegschaal.
\end{example}

\begin{example}[Twee blokken en een katrol]\label{ex:g12:newtons-laws:pulley}
Een blok $m_1 = \qty{4.0}{kg}$ op een wrijvingsloze tafel is met
een licht touw, over een ideale katrol, verbonden met een hangend
blok $m_2 = \qty{1.0}{kg}$. Het touw zendt dezelfde
\omterm{def:g10:inertia:inventory}{spanning} $T$ aan beide einden over;
de blokken delen één grootte van versnelling $a$. Eén tweede wet
elk, langs elke beweging: $m_1 a = T$ en $m_2 a = m_2 g - T$, dus
$a = m_2\,g/(m_1 + m_2) = \qty{1.96}{m/s^2}$ en
$T = m_1 a = \qty{7.8}{N}$ --- kleiner dan
$m_2 g = \qty{9.8}{N}$, zoals het moet voor $m_2$ omlaag te
versnellen.
\end{example}

\begin{example}[De vlakke bocht]\label{ex:g12:newtons-laws:curve}
Een auto van massa $m$ rondt een vlakke bocht van straal $R$ bij
constante snelheid $v$. In het
\omterm{def:g12:kinematics-2d:frenet}{Frenet-stelsel}
(\cref{ch:g12:kinematics-2d}) is de versnelling puur normaal,
$a_N = v^2/R$, gericht op het centrum.
\omterm{def:g10:universal-gravitation:weight}{Verticaal}, $N = mg$;
horizontaal is de enige centripetale
\omterm{def:g10:inertia:force}{kracht} die te bieden is de
statische \omterm{def:g10:inertia:inventory}{wrijving} van weg op
banden: $f = mv^2/R \leq \mu_s N = \mu_s mg$, dus
$v \leq \sqrt{\mu_s\, g R}$, wat de massa ook is. Voor
$R = \qty{90}{m}$, droge weg ($\mu_s = 0.70$):
$v_{\max} \approx \qty{25}{m/s}$ (\qty{89}{km/h}); op ijs stort het
plafond in --- \cref{exo:g12:newtons-laws:13} helt de weg;
\cref{exo:g12:newtons-laws:10} zwaait dezelfde projectie op een
draad.
\end{example}

\section{Geïsoleerde systemen: impuls blijft behouden}

\begin{proposition}[Behoud van impuls]\label{prop:g12:newtons-laws:conservation}\index{behoud van impuls}
Voor twee lichamen waarvan de externe
\omterm{def:g10:inertia:force}{krachten} verdwijnen of
\omterm{def:g10:inertia:compensate}{elkaar opheffen}, is de totale
\omterm{def:g12:newtons-laws:momentum}{impuls} $\vect p_1 + \vect p_2$
constant, wat de \omterm{def:g10:inertia:force}{krachten} die ze
op elkaar uitoefenen ook zijn.
\end{proposition}

\begin{proof}
Tel de twee tweede wetten op: $\frac{d}{dt}(\vect p_1 + \vect p_2)
= \vect F_{2 \to 1} + \vect F_{1 \to 2} + \vect F_{\text{ext}}$: het
interne paar heft op door de derde wet, en
$\vect F_{\text{ext}} = \vect 0$ per hypothese.
\end{proof}

\begin{example}[Terugslag]\label{ex:g12:newtons-laws:recoil}
Een \qty{60}{kg}-schaatser in rust op glad ijs gooit een
\qty{4.0}{kg}-medicijnbal horizontaal met \qty{6.0}{m/s}. Voor:
$\vect p = \vect 0$; na: $0 = 60\,V + 4.0 \times 6.0$, dus
$V = \qty{-0.40}{m/s}$ --- ze drijft achteruit. Niets extern duwde
haar: de bal deed dat (derde wet), en de boeken blijven op nul.
\end{example}

\begin{remark}[Hoe een raket op niets duwt]\label{rem:g12:newtons-laws:rocket}
Een raket is een machine om massa te gooien. Elke seconde slingeren
zijn motoren een batch gas achteruit; het gas voert achterwaartse
\omterm{def:g12:newtons-laws:momentum}{impuls} weg, dus wint de raket
hetzelfde bedrag voorwaarts --- een gestage
\omterm{def:g10:inertia:inventory}{stuwkracht}, ruwweg de per seconde
uitgeworpen massa maal de uitwerpsnelheid. De raket duwt op zijn
eigen uitlaat, niet op lucht: hij \omterm{def:g11:work-of-force:work}{werkt}
het best in de lege ruimte.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:1}
Bereken de \omterm{def:g12:newtons-laws:momentum}{impuls} van (a) een
\qty{70}{kg}-sprinter bij \qty{10}{m/s}; (b) een
\qty{1300}{kg}-auto bij \qty{50}{km/h}; (c) een \qty{8.0}{g}-kogel bij
\qty{800}{m/s}. Rangschik ze. Waarom staat de snelste laatst?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:2}
Een \qty{70}{kg}-passagier staat op een weegschaal in een lift. Wat
leest die wanneer de cabine (a) omhoog versnelt met
\qty{1.2}{m/s^2}; (b) klimt met een constant \qty{2.0}{m/s}; (c)
omlaag versnelt met \qty{1.2}{m/s^2}? Welke wet beslist (b)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:3}
Een blok wordt losgelaten op een wrijvingsloze helling van hoek
$30^\circ$. Bereken zijn versnelling, dan zijn snelheid na
\qty{3.0}{m} glijden. Waar is de massa gebleven?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:4}
Een \qty{58}{g}-tennisbal arriveert met \qty{15}{m/s} en wordt langs
dezelfde lijn teruggestuurd met \qty{25}{m/s}; contact duurt
\qty{5.0}{ms}. Bereken de verandering van
\omterm{def:g12:newtons-laws:momentum}{impuls} en de gemiddelde
\omterm{def:g10:inertia:force}{kracht} van de snaren; vergelijk die
\omterm{def:g10:inertia:force}{kracht} met het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van de bal.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:newtons-laws:5}
Een boek rust op een tafel. Noem de twee
\omterm{def:g10:inertia:force}{krachten} die op het boek werken en,
voor elk, zijn actie--reactiepartner (welk lichaam? welke
richting?). Waarom zijn het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van het boek
en de duw van de tafel geen partners van elkaar?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:6}
Een \qty{50}{kg}-kist zit op een horizontale vloer, $\mu_s = 0.45$,
$\mu_k = 0.35$. (a) Welke horizontale duw start hem net? (b) Als die
duw wordt volgehouden, wat is zijn versnelling? (c) Welke duw houdt
hem bij constante snelheid in beweging?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:7}
Dezelfde kist wordt op een verstelbare helling gezet ($\mu_s = 0.45$,
$\mu_k = 0.35$). (a) Bij welke hoek begint hij te glijden? (b) Bij
$30^\circ$, bereken zijn versnelling. (c) Eenmaal gestart bij de
hoek van (a), blijft hij versnellen? Waarom?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:8}
Een \qty{3.0}{kg}-blok op een wrijvingsloze tafel is over een
ideale katrol verbonden met een hangend \qty{2.0}{kg}-blok. Bereken
de versnelling en de \omterm{def:g10:inertia:inventory}{spanning}.
Waarom moest de \omterm{def:g10:inertia:inventory}{spanning} kleiner
uitkomen dan het hangende
\omterm{def:g10:universal-gravitation:weight}{gewicht}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:9}
Een bocht van straal \qty{50}{m} is vlak. Bereken de maximale
snelheid van een auto voor $\mu_s = 0.70$ (droog) en $0.40$ (nat),
in \unit{km/h}. Waarom geldt het antwoord evenzeer voor een scooter
als voor een vrachtwagen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:10}
Een \qty{0.30}{kg}-bol aan een \qty{1.2}{m}-draad trekt een
horizontale cirkel, de draad houdt een constant $25^\circ$ met de
verticaal (een conische slinger). Projecteer de tweede wet
\omterm{def:g10:universal-gravitation:weight}{verticaal}, dan langs
de straal: bereken de \omterm{def:g10:inertia:inventory}{spanning} en

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:newtons-laws:11}
Een \qty{5.0}{kg}-pakket rijdt een weegschaal in een goederenlift;
de aflezingen:
\begin{center}\small\begin{tabular}{lccc}
fase & \qtyrange{0}{2}{s} & \qtyrange{2}{8}{s} & \qtyrange{8}{10}{s}\\
weegschaal (\unit{N}) & 54.0 & 49.0 & 44.1
\end{tabular}\end{center}
Bereken de versnelling in elke fase en beschrijf een mogelijke rit.
Kan de lift de hele tijd \emph{omlaag} hebben bewogen?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:12}
Een \qty{60}{kg}-schaatser in rust op glad ijs gooit een
\qty{4.0}{kg}-bal horizontaal. (a) De bal vertrekt met \qty{8.0}{m/s}
\emph{over de grond}: haar terugslagsnelheid? (b) De \qty{8.0}{m/s}
is nu gemeten \emph{ten opzichte van haar}: herbereken beide
snelheden. (c) Welke \omterm{def:g10:inertia:force}{kracht} duwde
haar achteruit?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:13}
Een snelwegbocht van straal \qty{80}{m} is ontworpen voor
\qty{20}{m/s} zonder hulp van
\omterm{def:g10:inertia:inventory}{wrijving}. Toon aan dat de
rijbaan moet overhellen onder de hoek $\theta$ gegeven door
$\tan\theta = v^2/(gR)$ en bereken die. Wat moet
\omterm{def:g10:inertia:inventory}{wrijving} doen voor een auto die
de bocht trager --- of sneller --- neemt?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:14}
De motoren van een raket werpen \qty{250}{kg} gas per seconde uit

\omterm{def:g10:inertia:inventory}{stuwkracht} geven de gassen terug?
(b) De raket weegt \qty{5.0e4}{kg} bij ontsteking: zijn
\omterm{def:g10:universal-gravitation:weight}{gewicht}? zijn

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:newtons-laws:15}

\section{Probleem: De kabelbaan}

\begin{problem}[{Weekendprobleem --- de kabelbaan certificeren: de
spanning die de helling in rust eist, de toeslag van het opstarten,
de verlichting over de toren, de prijs van een noodstop --- en het
ene getal dat het bordje in de cabine mag beloven}]
\label{pb:g12:newtons-laws:1}
Een bergkabelbaan --- cabine plus wagen, lege massa
$M_0 = \qty{2.0e3}{kg}$ --- rolt op een baankabel hellend onder
$\alpha = 30^\circ$, getrokken door een trekkabel evenwijdig aan de
helling met \omterm{def:g10:inertia:inventory}{spanning} $\vect T$.
Rol\omterm{def:g10:inertia:inventory}{wrijving} is verwaarloosbaar;
de cabinevloer blijft doorlopend horizontaal. De cabine zit $25$
passagiers van gemiddelde massa \qty{80}{kg}; jij bent de
certificerende ingenieur. Tenzij anders vermeld is de cabine vol:
$M = \qty{4.0e3}{kg}$.

\medskip
\noindent\textbf{Deel I --- In rust op de helling.}
\begin{enumerate}
  \item Kies het systeem; zeg waarom het bergstelsel als
        inertiaalstelsel zal dienen; inventariseer de drie
        \omterm{def:g10:inertia:force}{krachten}, met richtingen.
  \item Welke wet regeert het wachten? Projecteer hem langs en
        loodrecht op de helling om $T$ en de
        \omterm{def:g12:newtons-laws:friction}{normaalkracht} $N$ van
        de baankabel uit te drukken.
  \item Bereken $T$ en $N$.
  \item Herbereken $T$ voor de lege cabine. Waarom is $T$ evenredig
        met de totale massa?
  \item De trekkabel breekt, remmen uit: toon aan dat de
        wegloopversnelling $g \sin\alpha$ is, beladen of leeg, en
        bereken hem.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Opstarten.} De lier versnelt de volle
cabine met $a = \qty{0.60}{m/s^2}$ de helling op tot hij cruiseert
met $v = \qty{6.0}{m/s}$.
\begin{enumerate}[resume]
  \item Druk uit en bereken de
        \omterm{def:g10:inertia:inventory}{spanning} $T'$ tijdens
        deze fase.
  \item Met welke fractie heft opstarten de
        \omterm{def:g10:inertia:inventory}{spanning} boven zijn
        rustwaarde?
  \item Hoe lang duurt de fase, en over welke afstand?
  \item Een \qty{80}{kg}-passagier staat op de horizontale vloer.
        Projecteer de tweede wet
        \omterm{def:g10:universal-gravitation:weight}{verticaal} om
        haar \omterm{def:g12:newtons-laws:apparent}{schijnbare
        gewicht} uit te drukken en te berekenen; vergelijk met $mg$.
  \item Welke \omterm{def:g10:inertia:force}{kracht} versnelt haar
        horizontaal, en hoe groot? Waarom leunen passagiers?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Over de toren.} Bij constant
\qty{6.0}{m/s} buigt de baankabel over een steuntoren langs een
verticale cirkelboog van straal $R = \qty{40}{m}$; bij de kam is
de snelheid horizontaal.
\begin{enumerate}[resume]
  \item In het \omterm{def:g12:kinematics-2d:frenet}{Frenet-stelsel}
        bij de kam (\cref{ch:g12:kinematics-2d}), geef beide
        componenten van de versnelling, met hun waarden.
  \item Projecteer de tweede wet langs de normaal om de
        \omterm{def:g12:newtons-laws:friction}{normaalkracht} $N'$
        van de kabel op de wagen uit te drukken en te berekenen.
  \item Dezelfde vraag voor het
        \omterm{def:g12:newtons-laws:apparent}{schijnbare gewicht}
        van de passagier bij de kam: met welke fractie is ze
        lichter?
  \item Bij welke snelheid zou $N'$ verdwijnen? Vergelijk met de
        cruisesnelheid en commentarieer.
  \item Welke \omterm{def:g10:inertia:force}{kracht} oefent de wagen
        uit op de kabel over de toren --- welke wet zegt dat, met
        welke grootte en richting?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Remmen, een sprong, en het bordje.}
\begin{enumerate}[resume]
  \item Op de horizontale aankomststrook brengt een noodstop de
        volle cabine van \qty{6.0}{m/s} tot stilstand in
        \qty{1.5}{s}: lees $\sum \vect F = d\vect p/dt$ als
        gemiddelde over de stop om de
        rem\omterm{def:g10:inertia:force}{kracht} te berekenen.
  \item Welke horizontale \omterm{def:g10:inertia:force}{kracht}
        moet een \qty{80}{kg}-passagier stoppen, en kan
        schoen--vloer-statische
        \omterm{def:g10:inertia:inventory}{wrijving}
        ($\mu_s = 0.40$) hem leveren? Concludeer.
  \item Op het perron, remmen uit, hangt de lege cabine vrij en
        stil; de laatste passagier (\qty{80}{kg}) springt
        horizontaal af met \qty{2.5}{m/s} \emph{ten opzichte van de
        cabine}. Bereken de terugslagsnelheid van de cabine en de
        snelheid van de passagier over de grond.
  \item In één zin: welk voortstuwingsapparaat leeft op dit
        terugslagprincipe, en wat legt zijn
        \omterm{def:g10:inertia:inventory}{stuwkracht} vast?
  \item \emph{Certificering.} De trekkabel breekt bij
        \qty{100}{kN}; de regels begrenzen de werkende
        \omterm{def:g10:inertia:inventory}{spanning} op een vijfde
        daarvan. Toon aan dat het ergste geval hierboven het
        opstarten is, leid de maximale totale massa af, en druk het
        bordje: hoeveel \qty{80}{kg}-passagiers mogen meerijden?
        Controleer de veiligheidsfactor.
\end{enumerate}
\end{problem}
