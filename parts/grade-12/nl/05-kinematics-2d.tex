\chapter{Kinematica in twee dimensies}\label{ch:g12:kinematics-2d}

Kijk hoe een auto een rotonde neemt bij een constant \qty{30}{km/h}: de
naald beweegt niet, en toch voelt elke passagier zich zijwaarts
getrokken --- de beweging verandert, en het is niet de
\omterm{def:g12:kinematics-2d:velocity}{snelheid}. Eén getal per moment volstaat niet meer:
beweging in het vlak heeft pijlen nodig. Dit hoofdstuk rust een
bewegend punt uit met drie --- positie, snelheid, versnelling ---
aaneengeschakeld door de afgeleide, en leest hele families van
beweging van hun meetkunde af.

\section{De positievector}

Beweging bestaat alleen ten opzichte van een genoemd
\omterm{def:g10:relative-motion:frame}{referentiestelsel} --- een star lichaam plus een
klok (\cref{ch:g10:relative-motion}). Nieuw dit jaar is precisie:
hecht aan het stelsel een oorsprong $O$ en twee loodrechte
gegradueerde assen $Ox$, $Oy$, en registreer waar het punt op elk
moment is, als een paar functies.

\begin{definition}[Positievector]\label{def:g12:kinematics-2d:position}
In een stelsel uitgerust met een oorsprong $O$ en assen $Ox$, $Oy$ is
de \emph{positievector}\index{positievector} van een bewegend punt
$M$ op tijdstip $t$ $\vect{OM}(t)$, met coördinaten
$\bigl(x(t),\, y(t)\bigr)$: twee functies van de tijd, één per as. De
kromme die $M$ trekt terwijl $t$ loopt is zijn
\omterm{def:g10:relative-motion:trajectory}{baan}.
\end{definition}

\section{De snelheidsvector}

\begin{definition}[Snelheidsvector]\label{def:g12:kinematics-2d:velocity}
Over een kort interval verplaatst het punt zich met
$\vect{OM}(t+\Delta t) - \vect{OM}(t)$; deel door $\Delta t$ en laat
het interval krimpen: dit is de afgeleide uit je wiskundecursus,
toegepast op elke coördinaat. De
\emph{snelheidsvector}\index{snelheidsvector} van $M$ is de
afgeleide van zijn \omterm{def:g12:kinematics-2d:position}{positievector},
$\vect v(t) = \bigl(x'(t),\, y'(t)\bigr)$, in \unit{m/s}. Zijn
norm $v = \sqrt{x'^2 + y'^2}$ is de
\emph{snelheid}\index{snelheid (scalair)} --- het ene getal dat de
snelheidsmeter toont.
\end{definition}

\begin{example}[Een drone in horizontale vlucht]\label{ex:g12:kinematics-2d:drone}
Een drone vliegt met $x(t) = 4.0\,t$ en $y(t) = 3.0\,t$ (meters,
seconden): \omterm{def:g10:relative-motion:trajectory}{baan} de lijn $y = \tfrac34 x$,
snelheid $\vect v = (4.0, 3.0)$ te allen tijde,
\omterm{def:g12:kinematics-2d:velocity}{snelheid}
$\sqrt{4.0^2 + 3.0^2} = \qty{5.0}{m/s}$, constant.
\end{example}

\begin{proposition}[Raaklijn aan de baan]\label{prop:g12:kinematics-2d:tangent}
Op elk moment is $\vect v(t)$ raaklijn aan de
\omterm{def:g10:relative-motion:trajectory}{baan} bij de huidige positie van $M$, en
wijst in de reisrichting.
\end{proposition}

\begin{proof}
De verplaatsing is een koorde van de
\omterm{def:g10:relative-motion:trajectory}{baan}; als $\Delta t$ krimpt, wordt de
richting van de koorde die van de raaklijn.
\end{proof}

\section{De versnellingsvector}

\begin{definition}[Versnellingsvector]\label{def:g12:kinematics-2d:acceleration}
De \emph{versnellingsvector}\index{versnellingsvector} van $M$ is de
afgeleide van zijn \omterm{def:g12:kinematics-2d:velocity}{snelheidsvector},
$\vect a(t) = \bigl(v_x'(t),\, v_y'(t)\bigr) =
\bigl(x''(t),\, y''(t)\bigr)$, in \unit{m/s^2}: hoe snel, en in
welke richting, de \omterm{def:g12:kinematics-2d:velocity}{snelheidsvector} momenteel
verandert.
\end{definition}

\begin{remark}[Versnellen zonder sneller te gaan]\label{rem:g12:kinematics-2d:turning}
$\vect a \neq \vect 0$ wanneer de snelheids\emph{vector} verandert
--- in lengte \emph{of in richting}: een remmende auto versnelt
(achterwaarts); een afslaande auto bij constante
\omterm{def:g12:kinematics-2d:velocity}{snelheid} versnelt (zijwaarts). Dit is de
grootheid die \omterm{def:g10:inertia:force}{krachten} sturen
(\cref{ch:g11:forces-and-motion,ch:g12:newtons-laws}).
\end{remark}

\begin{method}[Een chronofotografie lezen]\label{met:g12:kinematics-2d:chrono}
Een \emph{chronofotografie}\index{chronofotografie} registreert de
posities $M_0, M_1, M_2, \dots$ van een bewegend punt op gelijke
tijdsintervallen $\Delta t$.
\begin{enumerate}
  \item Puntafstand leest \omterm{def:g12:kinematics-2d:velocity}{snelheid}: gelijke
        afstanden, constante \omterm{def:g12:kinematics-2d:velocity}{snelheid}; openend,
        sneller; sluitend, trager.
  \item De snelheid bij $M_i$ is de symmetrische koorde,
        $\vect v_i \approx \vect{M_{i-1}M_{i+1}}/(2\,\Delta t)$,
        getekend bij $M_i$, raaklijn aan het gestippelde pad.
  \item Voor de versnelling bij $M_i$: kopieer $\vect v_{i-1}$ en
        $\vect v_{i+1}$ naar $M_i$, teken
        $\Delta\vect v = \vect v_{i+1} - \vect v_{i-1}$ punt tot
        punt; dan $\vect a_i \approx \Delta\vect v/(2\,\Delta t)$.
\end{enumerate}
\end{method}

\section{Rechtlijnige bewegingen}

Langs een rechte lijn volstaat één as: $x(t)$, $v(t) = x'(t)$,
$a(t) = v'(t)$, allemaal getekende getallen.

\begin{definition}[Eenparig rechtlijnige beweging]\label{def:g12:kinematics-2d:uniform}
Een beweging is \emph{eenparig rechtlijnig}\index{eenparig rechtlijnige
beweging} wanneer $\vect v$ een constante vector is: rechte
\omterm{def:g10:relative-motion:trajectory}{baan}, constante
\omterm{def:g12:kinematics-2d:velocity}{snelheid}, $\vect a = \vect 0$. Dan
$x(t) = v\,t + x_0$.
\end{definition}

\begin{proposition}[Eenparig versnelde rechtlijnige beweging]\label{prop:g12:kinematics-2d:uarm}
\index{eenparig versnelde beweging}%
Als een punt langs $Ox$ beweegt met \emph{constante} versnelling $a$,
startend bij $t = 0$ vanuit positie $x_0$ met snelheid $v_0$, dan
\[
v(t) = a\,t + v_0,
\qquad
x(t) = \tfrac12\,a\,t^2 + v_0\,t + x_0 .
\]
\end{proposition}

\begin{proof}
$v$ is een primitieve van de constante $a$: $v(t) = a t + C$, en
$t = 0$ \omterm{def:g10:inertia:force}{forceert} $C = v_0$. Op zijn beurt is $x$ een
primitieve van $a t + v_0$: $x(t) = \tfrac12 a t^2 + v_0 t + C'$, met
$C' = x_0$.
\end{proof}

\begin{example}[Remweg]\label{ex:g12:kinematics-2d:braking}
Een auto bij \omterm{def:g12:kinematics-2d:velocity}{snelheid} $v_0$ remt met constante
vertraging $a$ (versnelling $-a$). Hij stopt wanneer
$v(t) = -a t + v_0 = 0$, bij $t_s = v_0/a$, na $d = -\tfrac12 a t_s^2
+ v_0 t_s = v_0^2/(2a)$ te hebben afgelegd. Het kwadraat is de
verkeersveiligheids-kop: de snelheid verdubbelen \emph{verviervoudigt}
de remweg --- met $a = \qty{6.0}{m/s^2}$, \qty{16}{m} vanaf
\qty{50}{km/h}, \qty{64}{m} vanaf \qty{100}{km/h}.
\end{example}

\section{Eenparige cirkelbeweging en het Frenet-stelsel}

\begin{definition}[Eenparige cirkelbeweging]\label{def:g12:kinematics-2d:ucm}
Een beweging is \emph{eenparig circulair}\index{eenparige cirkelbeweging}
wanneer de \omterm{def:g10:relative-motion:trajectory}{baan} een cirkel van straal $R$ is
en de \omterm{def:g12:kinematics-2d:velocity}{snelheid} $v$ constant is. De
\emph{periode}\index{periode!van rotatie} $T$ is de duur van één
rondje, en de \emph{frequentie}\index{frequentie!van rotatie}
$f = 1/T$ (in \unit{Hz}) het aantal rondjes per seconde:
$v = 2\pi R/T = 2\pi R f$.
\end{definition}

Constante \omterm{def:g12:kinematics-2d:velocity}{snelheid} --- en toch draait de
snelheidsvector: er \emph{is} een versnelling. Welke kant op, en hoe
groot?

\begin{theorem}[Centripetale versnelling]\label{thm:g12:kinematics-2d:centripetal}
\index{centripetale versnelling}%
In \omterm{def:g12:kinematics-2d:ucm}{eenparige cirkelbeweging} van straal $R$ en
\omterm{def:g12:kinematics-2d:velocity}{snelheid} $v$ wijst de versnelling ---
\emph{centripetaal} genoemd --- van het bewegende punt naar het
centrum van de cirkel, met constante norm
\[
a = \frac{v^2}{R}.
\]
\end{theorem}

\begin{proof}
Centreer de assen in het middelpunt van de cirkel en zet
$\omega = v/R$, zodat de boog $v t$ de hoek $\omega t$ onderspant: de
positie is $x(t) = R\cos(\omega t)$, $y(t) = R\sin(\omega t)$.
Differentiëren, $\vect v = (-R\omega\sin(\omega t),\,
R\omega\cos(\omega t))$, van constante norm $R\omega = v$; nogmaals
differentiëren, $\vect a = -\omega^2\,\vect{OM}$: tegengesteld aan de
\omterm{def:g12:kinematics-2d:position}{positievector} --- naar het centrum --- van
norm $\omega^2 R = v^2/R$.
\end{proof}

\begin{definition}[Frenet-stelsel]\label{def:g12:kinematics-2d:frenet}
Het \emph{Frenet-stelsel}\index{Frenet-stelsel} bij de huidige positie
van een punt op een gekromd pad is het paar loodrechte
\omterm{def:g10:orders-of-magnitude:unit}{eenheids}vectoren
$(\vect{u_t}, \vect{u_n})$: $\vect{u_t}$ raaklijn, langs de
beweging; $\vect{u_n}$ normaal, naar het centrum van de cirkel van
straal $R$ die de kromme daar het best past.
\end{definition}

\begin{proposition}[Versnelling in het Frenet-stelsel]\label{prop:g12:kinematics-2d:frenetacc}
Langs elk vlak pad van lokale straal $R$, afgelegd met
\omterm{def:g12:kinematics-2d:velocity}{snelheid} $v(t)$,
\[
\vect a = a_t\,\vect{u_t} + a_n\,\vect{u_n},
\qquad a_t = \frac{\dd v}{\dd t},
\qquad a_n = \frac{v^2}{R}:
\]
de \emph{tangentiële versnelling}\index{tangentiële versnelling}
$a_t$ verandert de \omterm{def:g12:kinematics-2d:velocity}{snelheid}; de
\emph{normale versnelling}\index{normale versnelling} $a_n$ draait
de snelheidsvector.
\end{proposition}
\admitted

\begin{remark}[Waar het bewijs leeft]\label{rem:g12:kinematics-2d:frenetproof}
We bewezen de extremen: pure $a_t$ op een lijn
(\cref{prop:g12:kinematics-2d:uarm}), pure $a_n$ op een cirkel
(\cref{thm:g12:kinematics-2d:centripetal}); de algemene ontbinding
wordt afgeleid in het volume van jaar~1.
\end{remark}

\begin{example}[Remmen in een bocht]\label{ex:g12:kinematics-2d:bend}
Een auto komt een bocht van straal $R = \qty{50}{m}$ binnen met
$v = \qty{15}{m/s}$ terwijl hij remt met
$\dd v/\dd t = \qty{-3.0}{m/s^2}$: $a_t = \qty{-3.0}{m/s^2}$,
$a_n = 15^2/50 = \qty{4.5}{m/s^2}$, dus
$a = \sqrt{3.0^2 + 4.5^2} \approx \qty{5.4}{m/s^2}$, wijzend
achterwaarts \emph{en} de bocht in. De banden moeten beide tegelijk
leveren --- waarom racechauffeurs \emph{vóór} de bocht remmen.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:1}
Voor $x(t) = 3.0\,t$, $y(t) = 4.0\,t$ (meters, seconden), geef
$\vect v$, de \omterm{def:g12:kinematics-2d:velocity}{snelheid}, de
\omterm{def:g10:relative-motion:trajectory}{baan} en de klasse van beweging.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:2}
Op een \omterm{met:g12:kinematics-2d:chrono}{chronofotografie} genomen elke
\qty{0.10}{s} liggen de punten op een lijn, gelijkmatig \qty{0.35}{m}
uit elkaar. Identificeer de beweging en bereken de
\omterm{def:g12:kinematics-2d:velocity}{snelheid}. Wat zouden openende
afstanden aangeven?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:3}
Voor $x(t) = 2.0\,t^2$ langs $Ox$ (meters, seconden), leid $v(t)$ en
$a(t)$ af, noem de klasse van beweging, en bereken de
\omterm{def:g12:kinematics-2d:velocity}{snelheid} bij $t = \qty{3.0}{s}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:4}
Een carrouselpaard rijdt een cirkel van straal \qty{4.5}{m}, één
rondje elke \qty{8.0}{s}: bereken de
\omterm{def:g12:kinematics-2d:ucm}{frequentie}, de
\omterm{def:g12:kinematics-2d:velocity}{snelheid} en de versnelling (norm en
richting).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:kinematics-2d:5}
Een $v(t)$-grafiek klimt recht van \qty{2.0}{m/s} bij $t = 0$ naar
\qty{10.0}{m/s} bij $t = \qty{4.0}{s}$. Lees de versnelling af, dan
de afgelegde afstand (oppervlakte onder de grafiek).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:6}
Een \omterm{def:g12:kinematics-2d:velocity}{hogesnelheids}trein neemt een bocht van
straal \qty{6.0}{km} met \qty{320}{km/h}. Bereken $a_n$ en vergelijk
met $g$; waarom moeten \omterm{def:g12:kinematics-2d:velocity}{hogesnelheids}lijnen
krappe bochten vermijden, zelfs perfect overhellende?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:7}
Een wagentje op een rechte rail heeft
$x(t) = -1.2\,t^2 + 8.0\,t + 3.0$ (meters, seconden). Vind $v(t)$ en
$a$, het moment en de positie van omkering, en beschrijf de beweging
ervoor en erna.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:8}
Een \omterm{met:g12:kinematics-2d:chrono}{chronofotografie} langs een lijn, elke
$\Delta t = \qty{0.20}{s}$, geeft $x = 0$, $0.10$, $0.28$, $0.54$,
$\qty{0.88}{m}$. Schat de snelheid bij de drie binnenpunten
(\cref{met:g12:kinematics-2d:chrono}), dan de versnelling; concludeer.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:9}
De Maan \omterm{def:g10:universal-gravitation:satellite}{cirkelt} om de Aarde op een
bijna-cirkel van straal \qty{3.84e8}{m} in \qty{27.3}{d}. Bereken
zijn \omterm{def:g12:kinematics-2d:velocity}{snelheid} en zijn
\omterm{thm:g12:kinematics-2d:centripetal}{centripetale versnelling}; vergelijk
de laatste met $g$, wetende dat de Maan ongeveer $60$ aardstralen ver
is (\cref{ch:g10:universal-gravitation}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:10}
Een wasmachinetrommel van straal \qty{25}{cm} draait met $1200$
omwentelingen per minuut. Bereken $f$, $T$, de rand-
\omterm{def:g12:kinematics-2d:velocity}{snelheid} en de randversnelling in
veelvouden van $g$. Waarom blijft het wasgoed vastgeklemd terwijl het
water weggaat?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:kinematics-2d:11}
Een auto bij \qty{90}{km/h} remt met constante $a = \qty{7.5}{m/s^2}$:
bereken de stoptijd en de remweg, en schets $v(t)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:12}
Voor $x(t) = R\cos(\omega t)$, $y(t) = R\sin(\omega t)$: bereken
$\vect v$; controleer dat de \omterm{def:g12:kinematics-2d:velocity}{snelheid} de
constante $R\omega$ is en $\vect v \perp \vect{OM}$; bereken
$\vect a$, toon dat hij $-\omega^2\,\vect{OM}$ is, en herwin
$a = v^2/R$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:13}
Een motorfiets komt een bocht van straal \qty{80}{m} binnen met
\qty{20}{m/s}, remmend met \qty{2.5}{m/s^2}. Bereken $a_t$, $a_n$, de
norm van $\vect a$ en zijn hoek met de bewegingsrichting. De banden
grijpen tot \qty{7.0}{m/s^2} in alle richtingen: blijft de rijder
binnen de limiet?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:14}
Een metro versnelt vanuit stilstand met \qty{1.2}{m/s^2} tot
\qty{20}{m/s}, cruiseert \qty{40}{s}, remt dan met \qty{1.5}{m/s^2}
tot stilstand. Bereken de duur en lengte van elke fase, de totale
afstand en de \omterm{def:g10:relative-motion:speed}{gemiddelde snelheid}. Schets

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:kinematics-2d:15}
Een geostationaire \omterm{def:g10:universal-gravitation:satellite}{satelliet}
cirkelt op straal \qty{4.22e7}{m} van het middelpunt van de Aarde met
\omterm{def:g12:kinematics-2d:ucm}{periode} \qty{86164}{s}. Bereken zijn
\omterm{def:g12:kinematics-2d:velocity}{snelheid} en zijn
\omterm{thm:g12:kinematics-2d:centripetal}{centripetale versnelling}. Wat

\omterm{def:g10:universal-gravitation:satellite}{satelliet} boven één vast

\end{exercise}

\section{Probleem: Het ongevalsrapport}

\begin{problem}[{Weekendprobleem --- het ongevalsrapport: een
dashcamfilm beeld voor beeld gelezen, een remwet herbouwd uit
slipsporen, een rotonde die getuigt over grip, en een vonnis in
kilometers per uur}]
\label{pb:g12:kinematics-2d:1}
Een auto botst op een bestelwagen die uit een zijstraat reed; niemand
is gewond, maar de verzekeraars zijn het oneens. De onderzoeker heeft
beelden van een overheadcamera (één frame elke \qty{0.50}{s}),
gemeten slipsporen, en de dashcam van de auto. De
\omterm{def:g12:kinematics-2d:velocity}{snelheids}limiet is \qty{50}{km/h}.

\medskip
\noindent\textbf{Deel I --- De film lezen.}
De frames vóór enige reactie geven, langs de rechte weg:
\begin{center}\small
\begin{tabular}{lccccc}
$t$ (\unit{s}) & 0.00 & 0.50 & 1.00 & 1.50 & 2.00\\
$x$ (\unit{m}) & 0.0 & 9.0 & 18.0 & 27.0 & 36.0
\end{tabular}
\end{center}
\begin{enumerate}
  \item Waarom moet het rapport eerst een stelsel en assen noemen?
        Welke worden hier gebruikt?
  \item Bereken de gemiddelde snelheid op elk van de vier intervallen.
  \item Welke klasse van beweging is dit? Motiveer uit de tabel.
  \item Geef de \omterm{def:g12:kinematics-2d:velocity}{snelheid} $v_0$ van de auto in
        \unit{m/s} en \unit{km/h}.
  \item Wat is de versnelling tijdens deze twee seconden?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De slipsporen.}
Bandenproeven op dit droge asfalt geven een constante
remvertraging $a = \qty{6.0}{m/s^2}$; de slipsporen zijn
$d_b = \qty{27}{m}$ lang. Neem $t = 0$, $x = 0$ bij het begin van
remmen, onbekende \omterm{def:g12:kinematics-2d:velocity}{snelheid} $v_B$.
\begin{enumerate}[resume]
  \item Via primitieven, schrijf $v(t)$ en $x(t)$ tijdens remmen.
  \item Druk de stoptijd $t_s$ uit; toon $d_b = v_B^2/(2a)$.
  \item Leid $v_B$ af uit de slipsporen. Consistent met Deel~I?
  \item De reactietijd is \qty{1.0}{s}: bereken de reactieafstand en
        de totale stopafstand.
  \item Leg uit, uit vraag 7, waarom de
        \omterm{def:g12:kinematics-2d:velocity}{snelheid} verdubbelen de remweg
        verviervoudigt; bereken $d_b$ bij precies \qty{50}{km/h}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De rotonde getuigt.}
Eerder toont de dashcam de auto die een rotondestrook van straal
$R = \qty{12.5}{m}$ rondt bij constante
\omterm{def:g12:kinematics-2d:velocity}{snelheid}, een kwart slag in
\qty{2.5}{s}. De weg kan hoogstens $a_n = \mu g$ leveren, met
$\mu = 0.70$.
\begin{enumerate}[resume]
  \item De \omterm{def:g12:kinematics-2d:velocity}{snelheid} is daar constant --- is de
        snelheidsvector dat? Versnelt de auto? Leg uit.
  \item Bereken de \omterm{def:g12:kinematics-2d:velocity}{snelheid} van de auto op de
        rotonde.
  \item Bereken zijn \omterm{thm:g12:kinematics-2d:centripetal}{centripetale
        versnelling} (norm en richting).
  \item Bereken de maximale antislip-
        \omterm{def:g12:kinematics-2d:velocity}{snelheid}. Bleef de auto daarbinnen?
  \item Geef de \omterm{def:g12:kinematics-2d:ucm}{periode} en
        \omterm{def:g12:kinematics-2d:ucm}{frequentie} van een volle lus bij die
        \omterm{def:g12:kinematics-2d:velocity}{snelheid}.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Het vonnis.}
De bestelwagen verscheen $D = \qty{41}{m}$ voor de auto op het moment
dat de chauffeur begon te reageren.
\begin{enumerate}[resume]
  \item Met hoeveel, in \unit{km/h} en in procent, overschreed de
        \omterm{def:g12:kinematics-2d:velocity}{snelheid} vóór remmen de limiet?
  \item Bereken de \omterm{def:g12:kinematics-2d:velocity}{snelheid} van de auto bij
        impact.
  \item Doe de stopberekening over bij precies \qty{50}{km/h}
        (zelfde reactie, zelfde remmen): stopt de auto, met welke
        marge?
  \item Geef de algemene stopafstandsformule $d(v)$; waarom komt
        \omterm{def:g12:kinematics-2d:velocity}{snelheid} tweemaal binnen --- eens
        lineair, eens gekwadrateerd?
  \item Het vonnis, één zin met zijn getallen: welke
        \omterm{def:g12:kinematics-2d:velocity}{snelheid} reed de chauffeur, en was
        het ongeval bij de limiet gebeurd?
\end{enumerate}
\end{problem}
