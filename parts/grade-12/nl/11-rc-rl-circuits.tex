\chapter{RC- en RL-circuits}\label{ch:g12:rc-rl-circuits}

Druk de ontspanknop van een camera half in: een
\omterm{def:g11:lenses-and-eye:lens}{dun} gejank klimt twee seconden, een lampje wordt
groen --- pas dan vuurt de flits, en stort in een milliseconde uit wat
hij twee seconden verzamelde. Dit hoofdstuk ontmoet de twee componenten
die circuits een geheugen en een klok geven ---
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} en
\omterm{def:g11:magnetic-fields:solenoid}{spoel} --- en de eerste differentiaalvergelijkingen
van de natuurkunde, te controleren met de afgeleiden van dit jaar.

\section{De condensator}

\begin{definition}[Condensator en capaciteit]\label{def:g12:rc-rl-circuits:capacitor}
Een \emph{condensator}\index{condensator} is een paar tegenoverliggende
metalen platen, gescheiden door een \omterm{def:g11:lenses-and-eye:lens}{dun} isolator.
Aangesloten op een bron nemen de platen tegengestelde ladingen $+q$ en
$-q$ op --- een \omterm{def:g11:electric-gravitational-fields:efield}{elektrisch veld} vult de spleet
(\cref{ch:g11:electric-gravitational-fields}) --- en de
\omterm{def:g11:circuits-and-power:voltage}{spanning} $u$ over de platen is evenredig met de
opgeslagen lading:
\[ q = C\,u . \]
De constante $C$ is de \emph{capaciteit}\index{capaciteit}, in
\emph{farad}\index{farad} (\unit{F}): $\qty{1}{F} = \qty{1}{C/V}$.
\end{definition}

\begin{example}[Ordegrootten]\label{ex:g12:rc-rl-circuits:orders}
Eén \omterm{def:g12:rc-rl-circuits:capacitor}{farad} is enorm --- één
\omterm{def:g11:fundamental-interactions:charge}{coulomb} geparkeerd per
\omterm{def:g11:circuits-and-power:voltage}{volt}. Keramiek van radiocircuits: \unit{pF} tot
\unit{nF}; elektrolyten van \omterm{def:g11:work-of-force:power}{voedingen} en flitsen:
\unit{\micro F} tot \unit{mF}; supercondensatoren: duizenden
\omterm{def:g12:rc-rl-circuits:capacitor}{farad}.
\end{example}

\begin{proposition}[Stroom in een condensator]\label{prop:g12:rc-rl-circuits:cdudt}
De stroom die op een plaat van een
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} aankomt is de snelheid waarmee zijn
lading groeit:
\[
i = \frac{dq}{dt} = C\,\frac{du}{dt} ,
\]
kleine letters voor tijdvariërende grootheden ($U$, $I$ blijven
stationair, \cref{ch:g11:circuits-and-power}).
\end{proposition}

\begin{proof}
Stroom is lading geleverd per seconde
(\cref{def:g11:circuits-and-power:current}): over een krimpend interval is
$i$ de afgeleide van $q$; en $q = Cu$ met $C$ constant geeft
$C\,du/dt$.
\end{proof}

\begin{remark}[Twee gevolgen]\label{rem:g12:rc-rl-circuits:continuity}
Als $u$ constant is, is $i = 0$: een geladen
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} laat geen stationaire stroom door. En
$u$ kan nooit springen --- een stap zou oneindige stroom eisen: de
\omterm{def:g11:circuits-and-power:voltage}{spanning} van de
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} is het geheugen van het circuit.
\end{remark}

\section{Een condensator laden: het RC-circuit}

Eén lus --- bron $\mathcal{E}$ (\cref{def:g11:circuits-and-power:emf}),
schakelaar, weerstand, lege
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator}: sluit $K$ bij $t = 0$.

\begin{theorem}[Laadwet van het RC-circuit]\label{thm:g12:rc-rl-circuits:charging}
\index{RC-circuit}
Na het sluiten van de schakelaar gehoorzaamt de
\omterm{def:g11:circuits-and-power:voltage}{spanning} van de
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator}, en volgt uit $u(0) = 0$:
\[
RC\,\frac{du}{dt} + u = \mathcal{E},
\qquad
u(t) = \mathcal{E}\left(1 - e^{-t/\tau}\right),
\qquad \tau = RC .
\]
\end{theorem}

\begin{proof}
\omterm{def:g11:circuits-and-power:voltage}{Spanningen} langs de lus tellen op
(\cref{prop:g11:circuits-and-power:laws}): $\mathcal{E} = u_R + u = Ri + u$
op elk moment; substitueer $i = C\,du/dt$
(\cref{prop:g12:rc-rl-circuits:cdudt}). Nu de kandidaat
\emph{verifiëren}: $du/dt = (\mathcal{E}/\tau)e^{-t/\tau}$, dus
$RC\,(\mathcal{E}/RC)\,e^{-t/\tau} + \mathcal{E}(1 - e^{-t/\tau}) =
\mathcal{E}$, waar voor alle $t$; en $u(0) = \mathcal{E}(1-1) = 0$. Dat
geen \emph{andere} kromme aan de vergelijking en de start voldoet, wordt
bewezen in de wiskundecursus van dit jaar over
differentiaalvergelijkingen.
\end{proof}

\begin{definition}[Tijdconstante]\label{def:g12:rc-rl-circuits:tau}
De \emph{tijdconstante}\index{tijdconstante} van een
\omterm{thm:g12:rc-rl-circuits:charging}{RC-circuit} is $\tau = RC$.
\omterm{def:g10:orders-of-magnitude:unit}{Eenheden}: $\unit{\ohm} \times \unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$ --- het eigen tempo
van het circuit.
\end{definition}

\begin{method}[$\tau$ aflezen op de kromme]\label{met:g12:rc-rl-circuits:reading}
Drie equivalente aflezingen, rechtstreeks op een grafiek:
\begin{enumerate}
  \item \emph{raaklijn in de oorsprong:} $u'(0) = \mathcal{E}/\tau$,
        dus de beginraaklijn bereikt de asymptoot bij $t = \tau$;
  \item \emph{63\%:} $u(\tau) = \mathcal{E}(1 - e^{-1}) \approx
        0.63\,\mathcal{E}$;
  \item \emph{95\%:} $u(3\tau) \approx 0.95\,\mathcal{E}$ --- geladen,
        voor praktische doeleinden ($5\tau$: $99\%$).
\end{enumerate}
\end{method}

\begin{proposition}[Ontlading]\label{prop:g12:rc-rl-circuits:discharge}
Een \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} geladen tot $U_0$ en alleen op een
weerstand $R$ gesloten, gehoorzaamt $RC\,du/dt + u = 0$ en volgt
\[
u(t) = U_0\,e^{-t/\tau}, \qquad \tau = RC
\]
--- $37\%$ over bij $\tau$, $5\%$ bij $3\tau$.
\end{proposition}

\begin{proof}
Dezelfde luswet, geen bron; opnieuw substitutie:
$RC(-U_0/\tau)e^{-t/\tau} + U_0 e^{-t/\tau} = 0$, met $u(0) = U_0$.
\end{proof}

\begin{example}[De getallen van een flits]\label{ex:g12:rc-rl-circuits:flashrc}
Een fotoflits laadt $C = \qty{800}{\micro F}$ door
$R = \qty{1.0}{k\ohm}$: $\tau = 10^3 \times 8.0\times10^{-4} =
\qty{0.80}{s}$, dus het ``klaar''-lampje --- bedraad om bij $95\%$ te
schakelen --- wacht $3\tau \approx \qty{2.4}{s}$: het gejank van de
inleiding. Afgevuurd door de magere \qty{0.50}{\ohm} van de
flitsbuis, leegt dezelfde
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} met $\tau' = \qty{0.40}{ms}$:
tweeduizend keer sneller uit dan in.
\end{example}

\section{De spoel: inductantie en het RL-circuit}

\begin{definition}[Spoel en inductantie]\label{def:g12:rc-rl-circuits:inductor}
Een \emph{spoel}\index{spoel}\index{inductantie (component)} is een
\omterm{def:g11:magnetic-fields:solenoid}{solenoïde} van draad: zijn stroom rijgt zijn eigen
windingen met een \omterm{def:g11:magnetic-fields:bfield}{magnetisch veld}
(\cref{ch:g11:magnetic-fields}); zodra hij \emph{verandert}, verschijnt
een \omterm{def:g11:circuits-and-power:voltage}{spanning} over de
\omterm{def:g11:magnetic-fields:solenoid}{spoel}:
\[
u = L\,\frac{di}{dt} .
\]
De constante $L$ is de \emph{inductantie}\index{inductantie}, in
\emph{henry}\index{henry} (\unit{H}): $\qty{1}{H} = \qty{1}{V.s/A}$.
\end{definition}
\admitted

\begin{remark}[De spoel weerstreeft verandering]\label{rem:g12:rc-rl-circuits:oppose}
Waarom een veranderende stroom een
\omterm{def:g11:circuits-and-power:voltage}{spanning} maakt --- inductie --- wordt in de
universiteitsvolumes afgeleid; hier nemen we de wet en haar karakter:
de \omterm{def:g11:magnetic-fields:solenoid}{spoel} \emph{weerstreeft veranderingen} van
zijn stroom. Spiegel-tweelingen: de $u$ van een
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} kan niet springen, de $i$ van een
\omterm{def:g11:magnetic-fields:solenoid}{spoel} evenmin.
\end{remark}

\begin{circuitikz}[font=\small]
  \draw (0,0) to[battery1, l=$\mathcal{E}$] (0,2.4) to[nos, l=$K$] (2.1,2.4)
    to[R=$R$] (4.2,2.4) -- (5.4,2.4) to[L=$L$, i=$i$, v=$u$] (5.4,0) -- (0,0);
\end{circuitikz}

{\small Het \omterm{prop:g12:rc-rl-circuits:rl}{RL-circuit}: na sluiten van $K$ laat de
\omterm{def:g11:magnetic-fields:solenoid}{spoel} de stroom alleen klimmen op zijn eigen tempo
$\tau = L/R$.}
\end{omfigure}

\begin{proposition}[Stroomstijging in een RL-circuit]\label{prop:g12:rc-rl-circuits:rl}
\index{RL-circuit}
De schakelaar sluiten op een bron $\mathcal{E}$, een weerstand $R$ en
een \omterm{def:g11:magnetic-fields:solenoid}{spoel} $L$ in serie geeft
$\mathcal{E} = Ri + L\,di/dt$, d.w.z.
\[
\frac{L}{R}\,\frac{di}{dt} + i = \frac{\mathcal{E}}{R},
\qquad\text{opgelost door}\quad
i(t) = \frac{\mathcal{E}}{R}\left(1 - e^{-t/\tau}\right),
\quad \tau = \frac{L}{R} .
\]
\end{proposition}

\begin{proof}
Opnieuw de luswet; en het is de \emph{zelfde vergelijking} als in
\cref{thm:g12:rc-rl-circuits:charging} met $u \to i$,
$\mathcal{E} \to \mathcal{E}/R$, $RC \to L/R$ --- dezelfde substitutie
verifieert de oplossing, en elke aflezing van
\cref{met:g12:rc-rl-circuits:reading} gaat over.
\end{proof}

\begin{example}[Een elektromagneet ontwaakt]\label{ex:g12:rc-rl-circuits:rl}
Een relais-\omterm{def:g11:magnetic-fields:solenoid}{spoel}, $L = \qty{0.30}{H}$ en
$R = \qty{6.0}{\ohm}$, op een \qty{12}{V}-voeding: eindstroom
$\mathcal{E}/R = \qty{2.0}{A}$, op tempo
$\tau = 0.30/6.0 = \qty{50}{ms}$ --- wakker in $3\tau \approx
\qty{0.15}{s}$.
\end{example}

\begin{remark}[De vonk bij openen]\label{rem:g12:rc-rl-circuits:spark}
Open een schakelaar op een live \omterm{def:g11:magnetic-fields:solenoid}{spoel} en $i$ moet
in een fractie van een milliseconde naar nul: $di/dt$ is enorm en
$u = L\,di/dt$ bereikt honderden of duizenden
\omterm{def:g11:circuits-and-power:voltage}{volts} --- een vonk springt over het contact. De
ontstekings\omterm{def:g11:magnetic-fields:solenoid}{spoel} van een auto doet het expres,
duizenden keren per minuut.
\end{remark}

\section{Opgeslagen energie}

\begin{proposition}[Energie in een condensator, energie in een spoel]\label{prop:g12:rc-rl-circuits:energy}
\index{energie!van een condensator}\index{energie!van een spoel}
Een \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} $C$ geladen tot
\omterm{def:g11:circuits-and-power:voltage}{spanning} $u$ en een
\omterm{def:g11:magnetic-fields:solenoid}{spoel} $L$ die een stroom $i$ voert, slaan op
\[
E_C = \tfrac12\,C u^2 = \frac{q^2}{2C},
\qquad
E_L = \tfrac12\,L i^2 .
\]
\end{proposition}

\begin{proof}
Een kleine lading $dq$ op platen die al op
\omterm{def:g11:circuits-and-power:voltage}{spanning} $u$ staan kost $u\,dq$
(\cref{def:g11:circuits-and-power:voltage}); de kosten stapelen, het totaal
is de oppervlakte onder de lijn $u = q/C$ --- de driehoek hieronder ---
dus $E_C = \tfrac12 qu = \tfrac12 Cu^2$. Dezelfde driehoek voor de
\omterm{def:g11:magnetic-fields:solenoid}{spoel} ($i$ verhogen met $di$ kost $Li\,di$) geeft
$E_L = \tfrac12 Li^2$.
\end{proof}

\omterm{def:g11:fundamental-interactions:charge}{coulombs} stappen in bij lage
\omterm{def:g11:circuits-and-power:voltage}{spanning}.}
\end{omfigure}

\begin{example}[Energie-ordegrootten]\label{ex:g12:rc-rl-circuits:energies}
De flits-\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} van
\cref{ex:g12:rc-rl-circuits:flashrc}:
$\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$. Een
supercondensator van \qty{3000}{F} bij \qty{2.7}{V}: \qty{1.1e4}{J} ---
de waarde van een AA-batterij. Een
\omterm{def:g11:magnetic-fields:solenoid}{spoel} van \qty{10}{mH} bij \qty{10}{A}:
\qty{0.50}{J}. Minder dan batterijen houden --- maar vrijgeefbaar in
een milliseconde.
\end{example}

\begin{remark}[Waar de twee tijdmeters voor dienen]\label{rem:g12:rc-rl-circuits:applications}
\index{afvlakking}\index{timercircuit}
Langzaam in, snel uit \omterm{def:g11:work-of-force:power}{voedt} de cameraflits en de
defibrillator. De \omterm{def:g12:mechanical-waves:delay}{vertraging} $\tau = RC$ drijft
\omterm{rem:g12:rc-rl-circuits:applications}{timercircuits}: intermitterende ruitenwissers,
knipperende bakens, trappenhuisverlichting. Een
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} over een voeding \emph{vlakt}
\omterm{def:g11:circuits-and-power:voltage}{spanning} af ($u$ kan niet springen); een
serie-\omterm{def:g11:magnetic-fields:solenoid}{spoel} vlakt stroom af ($i$ kan niet
springen). En \omterm{def:g11:magnetic-fields:solenoid}{spoel} plus
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} samen maken een
\omterm{def:g12:oscillators-and-time:oscillator}{oscillator}
(\cref{ch:g12:rlc-oscillations}).
\end{remark}

\section{Oefeningen}

\omterm{def:g11:circuits-and-power:voltage}{spanning} zou \qty{2.0}{mC} op

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:rc-rl-circuits:2}

\begin{exercise}[$\star$]\label{exo:g12:rc-rl-circuits:3}

\begin{exercise}[$\star\star$]\label{exo:g12:rc-rl-circuits:9}

\section{Probleem: De cameraflits}

\begin{problem}[{Weekendprobleem --- de cameraflits: een gejank van
twee seconden, een milliseconde glorie, en de muur-neef die harten
herstart --- één exponentieel drijft ze allemaal}]
\label{pb:g12:rc-rl-circuits:1}
Een compacte flits-\omterm{def:g10:orders-of-magnitude:unit}{eenheid} slaat
\omterm{def:g10:energy-conservation:energy}{energie} op in een
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} $C = \qty{800}{\micro F}$, geladen door
$R = \qty{1.0}{k\ohm}$ vanuit een
\omterm{def:g10:energy-conservation:transfer}{omzetter} gemodelleerd als ideale bron
$\mathcal{E} = \qty{300}{V}$; het ``klaar''-lampje schakelt bij $95\%$
van de volle \omterm{def:g11:circuits-and-power:voltage}{spanning}. Afgevuurd ziet de
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} alleen de flitsbuis,
$R_f = \qty{0.50}{\ohm}$. Gegevens: een AA-cel levert \qty{1.5}{V} en
\qty{2.4}{A.h}; $e = \qty{1.60e-19}{C}$.

\medskip
\noindent\textbf{Deel I --- Het reservoir.}
\begin{enumerate}
  \item Bereken de lading op de
        \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} bij \qty{300}{V}.
  \item Hoeveel \omterm{def:g11:fundamental-interactions:elementary}{elementaire ladingen}
        is dat?
  \item Bereken de opgeslagen \omterm{def:g10:energy-conservation:energy}{energie} $E_C$.
  \item De AA-cel houdt $q = It \approx \qty{8.6e3}{C}$, dus ongeveer
        \qty{1.3e4}{J}. Hoeveel maal de
        \omterm{def:g10:energy-conservation:energy}{energie} van de
        \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} is dat?
  \item Waarom dan de \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}? Eén zin, met
        ``\omterm{def:g11:work-of-force:power}{vermogen}''.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Het wachten op groen.}
\begin{enumerate}[resume]
  \item Schrijf de luswet en zet haar om in de
        differentiaalvergelijking voor $u(t)$.
  \item Verifieer door substitutie dat
        $u(t) = \mathcal{E}(1 - e^{-t/RC})$ haar oplost, startend van
        $u(0) = 0$.
  \item Bereken de \omterm{def:g12:rc-rl-circuits:tau}{tijdconstante} $\tau$.
  \item Na hoeveel tijd wordt het klaar-lampje groen, en bij welke
        \omterm{def:g11:circuits-and-power:voltage}{spanning}? Herken het twee-seconden-gejank.
  \item Bereken de beginlaadstroom; waarom sterft die daarna weg?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De milliseconde glorie.}
\begin{enumerate}[resume]
  \item Schrijf de ontladingswet $u(t)$ door de buis en bereken de
        nieuwe \omterm{def:g12:rc-rl-circuits:tau}{tijdconstante} $\tau'$.
  \item Bereken de piekstroom door de buis.
  \item Bereken het piek\omterm{def:g11:work-of-force:power}{vermogen}; vergelijk met een
        waterkoker van \qty{2.2}{kW}.
  \item Neem de flits als voorbij bij $3\tau'$, geef zijn duur en
        gemiddeld \omterm{def:g11:work-of-force:power}{vermogen}.
  \item Vorm de verhouding laadtijd tot flitstijd: dezelfde
        \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}, dezelfde lading --- wat
        veranderde, en wat maakt dat van een
        \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}: een
        \omterm{def:g10:energy-conservation:energy}{energie}tank, of een
        \omterm{def:g11:work-of-force:power}{vermogen}shendel?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De neef aan de muur.}
Een defibrillator moet $E_C = \qty{200}{J}$ leveren bij
$U_0 = \qty{1500}{V}$; borst en paddles bieden ongeveer
\qty{50}{\ohm}.
\begin{enumerate}[resume]
  \item Bereken de vereiste
        \omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit}.
  \item Bereken de opgeslagen lading.
  \item Bereken de \omterm{def:g12:rc-rl-circuits:tau}{tijdconstante} van de schok en de
        duur ($3\tau$); vergelijk met de flits.
  \item Bereken piekstroom en piek\omterm{def:g11:work-of-force:power}{vermogen} in de
        patiënt.
  \item Het oplaadcircuit moet het apparaat klaar ($95\%$) maken in
        \qty{6.0}{s}: vind de vereiste laad\omterm{def:g11:circuits-and-power:resistance}{weerstand}
        en de beginlaadstroom --- de getallen die een ontwerper zou
        bestellen.
\end{enumerate}
\end{problem}
