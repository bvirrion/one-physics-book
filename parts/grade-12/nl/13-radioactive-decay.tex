\chapter{Radioactief verval}\label{ch:g12:radioactive-decay}

In een museumkelder tikt een Geigerteller over een splinter hout van
een faraokist. Elke klik is wetteloos: niets kondigde hem aan, niets
voorspelt de volgende. Toch drukt het lab vrijdag een datum af die tot
op een eeuw klopt. Dit hoofdstuk lost de paradox op --- hoe perfect
willekeurige gebeurtenissen één voor één, bij biljoenen, optellen tot
de stevigste klokken die we bezitten.

\section{Wetteloos één voor één, klokwerk per mol}

Vorig jaar (\cref{ch:g11:nucleus-radioactivity}) zagen we de
\omterm{prop:g12:work-and-energy:equilibria}{instabiele} kernen, hun
$\alpha$- en $\beta$-vergelijkingen, en de helft van elk groot monster
die per \omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}
$T_{1/2}$ verdwijnt. Wat we niet konden zeggen was \emph{wanneer
tussen de treden} --- noch waarom een willekeurig proces überhaupt een
schema aanhoudt. Beide vragen hebben één antwoord.

\begin{definition}[Vervalconstante]\label{def:g12:radioactive-decay:lambda}
Voor elk \omterm{prop:g12:work-and-energy:equilibria}{instabiel}
\omterm{def:g11:nucleus-radioactivity:notation}{nuclide} bestaat een constante $\lambda$,
zijn \emph{vervalconstante}\index{vervalconstante}
(\omterm{def:g10:orders-of-magnitude:unit}{eenheid} \unit{s^{-1}}), zo dat in elk kort
interval $\dd t$ elke \omterm{def:g11:fundamental-interactions:ladder}{kern} kans
$p = \lambda\,\dd t$ heeft om te vervallen --- dezelfde voor elke
\omterm{def:g11:fundamental-interactions:ladder}{kern} van de soort, ongeacht zijn
leeftijd, en onaangetast door temperatuur,
\omterm{def:g10:pressure:pressure}{druk} of chemie.
\end{definition}

\begin{remark}[Wetteloos alleen, exact in menigten]\label{rem:g12:radioactive-decay:crowds}
Een \omterm{def:g11:fundamental-interactions:ladder}{kern} die tienduizend jaar heeft
gewacht, heeft precies dezelfde kans om deze seconde te vervallen als
één die vanochtend is gemaakt: geen slijtage, geen waarschuwing ---
welke de volgende is, is echt onvoorspelbaar. Menigten zijn een ander
verhaal. Gooi $100$ munten: verwacht $50$ koppen, maar wees niet
verrast door $43$ --- fluctuaties van orde $\sqrt n$, hier $10\%$.
Gooi $10^{20}$: de relatieve fluctuatie $1/\sqrt n$ is $10^{-10}$,
het resultaat zeker tot tien decimalen. Een gram materie houdt zo'n
$10^{22}$ kernen: wetteloosheid middelelt tot klokwerk, een exacte wet
voor $N(t)$, het aantal nog intacte kernen op tijdstip $t$.
\end{remark}

\section{De vervalwet}

\begin{proposition}[De evolutievergelijking]\label{prop:g12:radioactive-decay:ode}
In een groot monster van $N$ kernen verandert het aantal tijdens
$\dd t$ met
\[
\dd N = -\lambda N\,\dd t, \qquad\text{d.w.z.}\qquad \frac{\dd N}{\dd t} = -\lambda N :
\]
de populatie krimpt met een tempo evenredig met zichzelf.
\end{proposition}

\begin{proof}
Elk van de $N$ kernen vervalt tijdens $\dd t$ met kans
$\lambda\,\dd t$: verwachte vervallen $N\lambda\,\dd t$, en voor groot
$N$ plakt de werkelijke telling aan de verwachte
(\cref{rem:g12:radioactive-decay:crowds}); elk verval verwijdert één
\omterm{def:g11:fundamental-interactions:ladder}{kern}.
\end{proof}

\begin{theorem}[Wet van radioactief verval]\label{thm:g12:radioactive-decay:law}
Een monster met $N_0$ kernen bij $t = 0$ houdt op tijdstip $t$
\[
N(t) = N_0 \, e^{-\lambda t}.
\]
\end{theorem}

\begin{proof}
Controleer door substitutie:
$\dd N/\dd t = N_0(-\lambda)\,e^{-\lambda t} = -\lambda N(t)$, en
$N(0) = N_0 e^0 = N_0$: vergelijking en start beide juist. (Dat geen
\emph{andere} functie past, wordt met integralen bewezen in het
universiteitsvolume.)
\end{proof}

\begin{definition}[Exponentieel verval]\label{def:g12:radioactive-decay:exponential}
Een grootheid die $N_0\,e^{-\lambda t}$ volgt, ondergaat
\emph{exponentieel verval}\index{exponentieel verval}: in gelijke
tijden gaan gelijke \emph{fracties} verloren. Het is precies de
ontladende \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} van
\cref{ch:g12:rc-rl-circuits}, met $\lambda$ als $1/(RC)$: één
differentiaalvergelijking, eenmaal geleerd, drijft circuits en kernen
gelijk.
\end{definition}

\begin{proposition}[Halfwaardetijd]\label{prop:g12:radioactive-decay:halflife}
De \omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}
(\cref{def:g11:nucleus-radioactivity:halflife}) van een
\omterm{def:g11:nucleus-radioactivity:notation}{nuclide} is vastgelegd door zijn
\omterm{def:g12:radioactive-decay:lambda}{vervalconstante}:
$T_{1/2} = \ln 2/\lambda$, en na $n$ halfwaardetijden
$N = N_0/2^{\,n}$, of $n$ geheel is of niet.
\end{proposition}

\begin{proof}
$N(T_{1/2}) = N_0/2$ eist $e^{-\lambda T_{1/2}} = \tfrac12$, d.w.z.\
$\lambda T_{1/2} = \ln 2$; en $N(n\,T_{1/2}) = N_0\,e^{-n\ln 2} =
N_0/2^{\,n}$ voor elke $n$: de exponentieel rijgt de trap en vult
tussen de treden.
\end{proof}

\section{Activiteit: de klikken tellen}

\begin{proposition}[Activiteit]\label{prop:g12:radioactive-decay:activity}
De \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} van een monster --- zijn
vervallen per seconde, in
\omterm{def:g11:nucleus-radioactivity:activity}{becquerel}
(\cref{def:g11:nucleus-radioactivity:activity}) --- is
\[
\mathcal A = \lambda N, \qquad\text{dus}\qquad \mathcal A(t) = \mathcal A_0 \, e^{-\lambda t}:
\]
evenredig met de overlevende populatie, en evenzeer exponentieel.
\end{proposition}

\begin{proof}
$N$ kernen, elk met kans $\lambda$ per seconde: $\lambda N$ vervallen
per seconde; \cref{thm:g12:radioactive-decay:law} vermenigvuldigen met
$\lambda$ geeft $\mathcal A(t) = \mathcal A_0 e^{-\lambda t}$.
\end{proof}

\begin{example}[Enorm $N$, minuscule $\lambda$]\label{ex:g12:radioactive-decay:body}
Een banaan tikt op ongeveer \qty{15}{Bq}, een menselijk lichaam op
ongeveer \qty{8000}{Bq}, een kubieke
\omterm{def:g10:orders-of-magnitude:unit}{meter} graniet op zo'n \qty{e6}{Bq}
(\cref{ch:g11:nucleus-radioactivity}). Achter de bescheiden klikken
zitten astronomische populaties: het kalium-40 van het lichaam levert
$\mathcal A \approx \qty{4.4e3}{Bq}$ met $\lambda =
\qty{1.8e-17}{s^{-1}}$: $N = \mathcal A/\lambda \approx \num{2.5e20}$
kernen, waarvan amper enkele duizenden elke seconde vuren.
\end{example}

\begin{method}[Een halfwaardetijd meten]\label{met:g12:radioactive-decay:semilog}
\begin{enumerate}
  \item Tel vervallen met een Geigerteller over opeenvolgende gelijke
        intervallen: dit bemonstert $\mathcal A$ op opeenvolgende
        tijden.
  \item Zet $\ln \mathcal A$ tegen $t$ uit: omdat
        $\ln \mathcal A = \ln \mathcal A_0 - \lambda t$, toont
        \omterm{def:g12:radioactive-decay:exponential}{exponentieel verval} zich als een
        \emph{rechte lijn} --- kromme data betekent een andere wet, of
        een mengsel van \omterm{def:g11:nucleus-radioactivity:notation}{nucliden}.
  \item Lees $\lambda$ af als min de helling;
        $T_{1/2} = \ln 2/\lambda$.
\end{enumerate}
\end{method}

\begin{remark}[Becquerels zijn geen schade]\label{rem:g12:radioactive-decay:dose}
De \omterm{def:g11:nucleus-radioactivity:activity}{becquerel} telt vervallen, niet schade.
Schade wordt gevolgd door de \omterm{rem:g11:nucleus-radioactivity:dose}{dosis} in
\omterm{rem:g11:nucleus-radioactivity:dose}{sievert}
(\cref{ch:g11:nucleus-radioactivity}), die de
\omterm{def:g10:energy-conservation:energy}{energie} per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} weefsel weegt met de
destructiviteit van elke straling: een scan van \qty{5e8}{Bq}
technetium is routine, veel minder
\omterm{def:g11:nucleus-radioactivity:activity}{becquerel} ingeademde $\alpha$-zender
niet; \unit{Bq} naar \unit{Sv} omzetten is universiteitsstof.
\end{remark}

\section{Datering: tijd aflezen uit een verhouding}

De vervalwet achterstevoren is een klok: meet de overlevende fractie,
en de exponentieel noemt de verstreken tijd.

\begin{method}[Radiokoolstofdatering]\label{met:g12:radioactive-decay:carbon}
\begin{enumerate}
  \item \emph{Levend:} levend weefsel wisselt koolstof met de
        atmosfeer en draagt diens koolstof-14-verhouding --- één
        \omterm{def:g11:fundamental-interactions:ladder}{atoom} op $10^{12}$, wat
        $\mathcal A_0 = 13.6$ vervallen per minuut per gram koolstof
        geeft.
  \item \emph{Dood start de klok:} inname stopt, en de verhouding
        vervalt met $T_{1/2} = \num{5730}$ jaar.
  \item Meet de \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} $\mathcal A$ van
        het monster per gram koolstof;
        $\mathcal A = \mathcal A_0 e^{-\lambda t}$ inverteert tot
        $t = \frac{1}{\lambda}\ln(\mathcal A_0/\mathcal A)
        = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}
\end{method}

\begin{example}[Een haard in een grot]\label{ex:g12:radioactive-decay:hearth}
Houtskool uit een begraven haard geeft $5.0$ vervallen per minuut per
gram: $\mathcal A_0/\mathcal A = 13.6/5.0 = 2.72$, dus
$t = \num{5730} \times \ln(2.72)/\ln 2 \approx \num{8.3e3}$ jaar. Het
vuur doofde achtduizend jaar geleden --- het hout zelf is het archief.
\end{example}

\begin{example}[Klokken voor rotsen]\label{ex:g12:radioactive-decay:rocks}
Koolstof-14 zwijgt voorbij zo'n \num{50000} jaar; de geologie houdt
tragere klokken. \emph{Kalium--argon}: kalium-40
($T_{1/2} = \num{1.25e9}$ jaar) vervalt tot argon, een gas dat
\omterm{rem:g12:work-and-energy:escape}{ontsnapt} uit gesmolten lava maar wordt
ingesloten zodra rots kristalliseert --- stolling zet de klok op nul.
\emph{Uranium--lood}: uranium-238 leidt een keten van vervallen die
eindigt bij \omterm{prop:g12:work-and-energy:equilibria}{stabiel} lood-206
(\cref{ch:g11:nucleus-radioactivity}), en de lood-tot-uranium-verhouding
in een kristal dateert het. Beide klokken stemmen overeen over de
oudste mineralen ($\approx \num{4.4e9}$ jaar) en, via meteorieten, over
de leeftijd van het zonnestelsel.
\end{example}

\begin{remark}[Keten, en het gas in de kelder]\label{rem:g12:radioactive-decay:radon}
Tussen uranium en lood passeert de keten radium, dan radon-222, een
radioactief \emph{gas} ($T_{1/2} = 3.8$ dagen). Geboren in graniet en
bodem, sijpelt het kelders in en stapelt zich op; zijn
$\alpha$-uitzendende dochters vervallen in de longen --- de grootste
natuurlijke \omterm{rem:g11:nucleus-radioactivity:dose}{dosis} van de meesten, en de
goedkoopste te verminderen: open het kelderraam.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:1}
Jodium-131 heeft $\lambda = \qty{1.0e-6}{s^{-1}}$. Bereken zijn
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd} in seconden, dan in dagen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:2}
Radon-222 heeft $T_{1/2} = 3.8$ dagen. Bereken $\lambda$ in
\unit{s^{-1}}, dan de \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} van een
monster van \num{1.0e10} kernen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:3}
Verifieer door substitutie dat $N(t) = N_0\,e^{-\lambda t}$ voldoet

$N(T_{1/2})/N_0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:radioactive-decay:4}

\omterm{def:g11:nucleus-radioactivity:activity}{activiteit} over $90$ jaar, dan over $300$
jaar.

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:radioactive-decay:15}
Voor één enkele \omterm{def:g11:fundamental-interactions:ladder}{kern} is de kans om $n$
halfwaardetijden te overleven $1/2^{\,n}$. Bereken de kans dat (a) één
gegeven \omterm{def:g11:fundamental-interactions:ladder}{kern} $10$ halfwaardetijden
overleeft; (b) alle $8$ bewaakte kernen één
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd} overleven. (c) Van
\num{1000} kernen, hoeveel worden na $3$ halfwaardetijden verwacht ---
en kan de wet zeggen \emph{welke}? Concludeer wat de wet voorspelt.
\end{exercise}

\section{Probleem: De leeftijd van dingen}

\begin{problem}[{Weekendprobleem --- de leeftijd van dingen: een museumlab
kalibreert de koolstofklok, dateert een standbeeld en een papyrus, leert
waarom geen klok voorbij zijn wijzerplaat leest, en duwt een vulkanische
rots een miljard jaar terug}]
\label{pb:g12:radioactive-decay:1}
Een week in het dateringslaboratorium van een museum. Gegevens: levend
weefsel toont $\mathcal A_0 = 13.6$ koolstof-14-vervallen per minuut per
gram koolstof; $T_{1/2} = \num{5730}$ jaar (koolstof-14),
\num{1.25e9} jaar (kalium-40); $1$ jaar $= \qty{3.156e7}{s}$;
$N_A = \qty{6.02e23}{mol^{-1}}$; molaire massa van koolstof
\qty{12}{g/mol}; de tellers resoleren ongeveer $0.2$ vervallen per
minuut per gram.

\medskip
\noindent\textbf{Deel I --- Maandag: de klok kalibreren.}
\begin{enumerate}
  \item Waarom is de koolstof-14-verhouding constant in levend weefsel,
        en wat verandert bij de dood?
  \item Bereken $\lambda$ voor koolstof-14 in \unit{s^{-1}}.
  \item Verifieer door substitutie dat $N_0 e^{-\lambda t}$
        $\dd N/\dd t = -\lambda N$ oplost.
  \item Uit $\mathcal A_0$, bereken het aantal
        koolstof-14-\omterm{def:g11:fundamental-interactions:ladder}{atomen} per gram koolstof
        in levend weefsel.
  \item Hoeveel koolstof\omterm{def:g11:fundamental-interactions:ladder}{atomen} houdt één gram?
        Leid de koolstof-14-verhouding af --- vergelijk met ``één op
        $10^{12}$''.
  \item Toon dat een gemeten
        \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} $\mathcal A$ een monster
        dateert als $t = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Dinsdag: het standbeeld en de papyrus.}
\begin{enumerate}[resume]
  \item Een houten standbeeld geeft $9.1$ vervallen per minuut per
        gram. Dateer het hout.
  \item Welke gebeurtenis markeert die datering precies --- het snijden
        of iets anders? Welke voorzichtigheid volgt?
  \item Een Egyptische papyrus geeft $10.4$ vervallen per minuut per
        gram. Dateer hem. Is een schrijver van tweeëntwintig eeuwen
        geleden aannemelijk?
  \item De \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} is gemeten tot $\pm 0.2$
        vervallen per minuut. Schat de leeftijdsonzekerheid van de
        papyrus met $\Delta t \approx \Delta\mathcal A/(\lambda
        \mathcal A)$.
  \item Het ``antieke'' perkament van een handelaar geeft $13.4$
        vervallen per minuut per gram. Vonnis?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Donderdag: de grenzen van de klok.}
\begin{enumerate}[resume]
  \item Na hoeveel halfwaardetijden daalt de
        \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} van een monster onder de
        $0.2$ vervallen per minuut per gram van de tellers? Welke
        leeftijd is dat?
  \item De beste laboratoria bereiken ongeveer tien halfwaardetijden.
        Wat is de praktische horizon van radiokoolstofdatering, in
        jaren?
  \item Een dinosaurusbot is ongeveer \num{6.6e7} jaar oud: hoeveel
        koolstof-14-halfwaardetijden? Met vraag 4, na hoeveel
        halfwaardetijden is niet eens \emph{één}
        \omterm{def:g11:fundamental-interactions:ladder}{atoom} van de gram koolstof-14 over?
        Concludeer over ``dinosauriërs dateren met koolstof''.
  \item Welk soort \omterm{def:g11:nucleus-radioactivity:notation}{nuclide} zou de rots van de
        dinosaurus in plaats daarvan kunnen dateren?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Vrijdag: de vulkanische rots.}
\begin{enumerate}[resume]
  \item Argon is een gas, kalium een trouwe bewoner van de vaste stof:
        leg uit waarom de stolling van lava de kalium--argon-klok op
        nul zet.
  \item Aangenomen dat elke vervallen
        kalium-40-\omterm{def:g11:fundamental-interactions:ladder}{kern} één ingesloten
        argon\omterm{def:g11:fundamental-interactions:ladder}{atoom} achterlaat, toon dat
        $N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.
  \item De rots onder het fossiel van het museum toont
        $N_{\mathrm{Ar}}/N_{\mathrm K} = 0.60$. Dateer de rots.
  \item Kon koolstof-14 deze rots dateren, of kalium--argon het
        standbeeld? Geef het nuttige venster van elke klok in
        halfwaardetijden.
  \item Vrijdagrapport, één zin: de drie leeftijden van deze week, en
        de ene wet die ze alle las.
\end{enumerate}
\end{problem}
