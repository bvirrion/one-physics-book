\chapter{Nucleaire energie: splijting, fusie, $E=mc^2$}\label{ch:g12:nuclear-energy}

Een brandstofpellet ter grootte van een gummetje dekt de elektriciteit
van een huishouden voor een jaar; dezelfde dienst uit kolen kost drie
ton. Boven ons betaalt de Zon voor daglicht door te verdwijnen ---
vier miljoen ton van zichzelf per seconde. Dit hoofdstuk opent het
grootboek achter beide bargains: massa zelf is
\omterm{def:g10:energy-conservation:energy}{energie}, en kernen
(\cref{ch:g11:nucleus-radioactivity}) zijn waar ze wordt verzilverd.

\section{Massa is energie}

\begin{theorem}[Massa--energie-equivalentie]\label{thm:g12:nuclear-energy:emc2}\index{massa--energie-equivalentie}
Een lichaam van massa $m$ houdt, louter door te bestaan, de
\omterm{def:g10:energy-conservation:energy}{energie}
\[
E = m c^2, \qquad c = \qty{3.00e8}{m/s}.
\]
Wanneer een systeem massa $\Delta m$ verliest, maakt het de
\omterm{def:g10:energy-conservation:energy}{energie} $\Delta m\, c^2$ vrij --- welk
mechanisme ook.
\end{theorem}
\admitted

\begin{remark}[Einstein, 1905]\label{rem:g12:nuclear-energy:einstein}
Dit is Einsteins relatie, hier toegegeven; waar ze vandaan komt is de
zaak van het speciale-relativiteitshoofdstuk dat dit jaar afsluit
(\cref{ch:g12:special-relativity}). Let op de rate:
$c^2 = \qty{9.00e16}{J/kg}$ --- een afrondingsfout van massa is een
fortuin \omterm{def:g10:energy-conservation:energy}{energie}.
\end{remark}

\begin{definition}[Elektronvolt en atomaire massa-eenheid]\label{def:g12:nuclear-energy:units}
Twee \omterm{def:g10:orders-of-magnitude:unit}{eenheden} op maat van de
\omterm{def:g11:fundamental-interactions:ladder}{kern}. De
\emph{elektronvolt}\index{elektronvolt}, de
\omterm{def:g10:energy-conservation:energy}{energie} die een elektron wint bij het
oversteken van \qty{1}{V} (\cref{ch:g11:circuits-and-power}):
$\qty{1}{eV} = \qty{1.60e-19}{J}$, $\qty{1}{MeV} = \qty{1.60e-13}{J}$ ---
chemische bindingen handelen in enkele \unit{eV}, nucleaire reacties in
\omterm{def:g11:circuits-and-power:voltage}{mega-elektronvolt}. De \emph{verenigde atomaire
massa-eenheid}\index{verenigde atomaire massa-eenheid}, een twaalfde van
de massa van een koolstof-12-\omterm{def:g11:fundamental-interactions:ladder}{atoom}:
$\qty{1}{u} = \qty{1.66054e-27}{kg}$, van
\omterm{def:g10:energy-conservation:energy}{energie}-equivalent \qty{931.5}{MeV} ---
$\qty{1}{u} = \qty{931.5}{MeV}/c^2$.
\end{definition}

\begin{example}[Het slapende fortuin]\label{ex:g12:nuclear-energy:fortune}
Eén \omterm{def:g10:orders-of-magnitude:unit}{kilogram} van wat dan ook, volledig
omgezet, is $\qty{9.00e16}{J}$ --- drie jaar output van een grote
(\qty{1}{GW}) \omterm{def:g11:work-of-force:power}{centrale}. Niemand merkte het vóór
1905 omdat de alledaagse natuurkunde massa nauwelijks raakt: een
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} kolen verbranden (\qty{3.0e7}{J})
verlicht hem met een derde microgram. Kernen, zullen we zien, bewegen
één deel op duizend: nog klein, maar een miljoen maal chemie.
\end{example}

\section{Het massadefect}

\begin{definition}[Massadefect en bindingsenergie]\label{def:g12:nuclear-energy:defect}
Weeg een \omterm{def:g11:fundamental-interactions:ladder}{kern}
${}^{A}_{Z}\mathrm{X}$ (massa $m$), weeg dan zijn $Z$ protonen en
$A - Z$ neutronen afzonderlijk: de delen zijn \emph{zwaarder} dan het
geheel. Het verschil
\[
\Delta m = Z\,m_p + (A - Z)\,m_n - m > 0,
\qquad m_p = \qty{1.00728}{u},\; m_n = \qty{1.00866}{u},
\]
is het \emph{massadefect}\index{massadefect}; de
\emph{bindingsenergie}\index{bindingsenergie} $E_b = \Delta m\,c^2$ is
de \omterm{def:g10:energy-conservation:energy}{energie} nodig om de
\omterm{def:g11:fundamental-interactions:ladder}{kern} uiteen te trekken in vrije
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen} --- evenzeer de
\omterm{def:g10:energy-conservation:energy}{energie} vrijgemaakt bij assemblage. Een
gebonden \omterm{def:g11:fundamental-interactions:ladder}{kern} zit \emph{onder} zijn delen.
\end{definition}

{\small Een \omterm{def:g10:energy-conservation:energy}{energie}ladder: de gebonden
\omterm{def:g11:fundamental-interactions:ladder}{kern} ligt $E_b$ onder zijn gescheiden
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen} --- assemblage maakt $E_b$ vrij,
uiteentrekken kost die.}
\end{omfigure}

\begin{example}[Helium-4]\label{ex:g12:nuclear-energy:helium}
De ${}^{4}_{2}\mathrm{He}$-\omterm{def:g11:fundamental-interactions:ladder}{kern} heeft
$m = \qty{4.00151}{u}$; zijn delen:
$2 \times 1.00728 + 2 \times 1.00866 = \qty{4.03188}{u}$. Dus
$\Delta m = \qty{0.03037}{u}$ en $E_b = 0.03037 \times 931.5 \approx
\qty{28.3}{MeV}$: de \omterm{def:g11:fundamental-interactions:ladder}{kern} weegt $0.75\%$
minder dan zijn delen --- de ``één deel op duizend'' van
\cref{ex:g12:nuclear-energy:fortune}.
\end{example}

\begin{method}[Energiebalans van een nucleaire reactie]\label{met:g12:nuclear-energy:balance}
\begin{enumerate}
  \item Balanceer de vergelijking: totale $A$ en $Z$ behouden
        (\cref{ch:g11:nucleus-radioactivity}).
  \item Tel de massa's voor, dan na, in \unit{u} --- bewaar alle vijf
        decimalen.
  \item $\Delta m = m_{\text{before}} - m_{\text{after}}$, dan
        $E = \Delta m\,(\text{in u}) \times \qty{931.5}{MeV}$,
        meegenomen als \omterm{def:g11:mechanical-energy:kinetic}{kinetische energie} van de
        producten; negatief betekent dat de reactie betaald moet worden.
\end{enumerate}
\end{method}

\section{De kromme die het heelal drijft}

\begin{definition}[Bindingsenergie per nucleon]\label{def:g12:nuclear-energy:pernucleon}
Cohesie wordt eerlijk vergeleken via de \emph{bindingsenergie per
nucleon}\index{bindingsenergie per nucleon}, $E_b/A$, de gemiddelde
kost van het extraheren van één
\omterm{def:g11:fundamental-interactions:ladder}{nucleon}: $28.3/4 \approx
\qty{7.1}{MeV}$ voor helium-4.
\end{definition}

\begin{proposition}[De ijzerpiek, en twee wegen omlaag]\label{prop:g12:nuclear-energy:ironpeak}\index{ijzerpiek}
Uitgezet tegen $A$ klimt de
\omterm{def:g12:nuclear-energy:pernucleon}{bindingsenergie per nucleon} steil door de
lichte kernen, piekt nabij ijzer, ${}^{56}_{26}\mathrm{Fe}$, bij
ongeveer \qty{8.8}{MeV} per \omterm{def:g11:fundamental-interactions:ladder}{nucleon}, en
glijdt dan zacht naar ongeveer \qty{7.6}{MeV} bij uranium: ijzer is de
sterkst gebonden \omterm{def:g11:fundamental-interactions:ladder}{kern} in de natuur. Een
reactie maakt \omterm{def:g10:energy-conservation:energy}{energie} vrij precies wanneer
haar producten \emph{hoger} op de kromme zitten; twee tegengestelde
strategieën betalen dus beide, en lopen beide naar ijzer --- de
zwaarste kernen \emph{splitsen}, en de lichtste \emph{samenvoegen}.
\end{proposition}
\admitted

\begin{remark}[Waarom een piek]\label{rem:g12:nuclear-energy:whypeak}
Het touwtrekken van \cref{ch:g11:fundamental-interactions}: sterke lijm
bindt alleen rakende buren, Coulomb-afstoting overspant de hele
\omterm{def:g11:fundamental-interactions:ladder}{kern} --- kleine kernen zijn vooral
oppervlak, grote vooral afstoting, ijzer het compromis. De eerlijke
boekhouding leeft in de universiteitsvolumes.
\end{remark}

\section{Splijting en de ketenreactie}

\begin{definition}[Splijting]\label{def:g12:nuclear-energy:fission}
\emph{Splijting}\index{splijting} is het splitsen van een zware
\omterm{def:g11:fundamental-interactions:ladder}{kern} in twee middelgrote fragmenten.
Uranium-235 is \emph{splijtbaar}\index{splijtbaar}: een traag neutron,
geabsorbeerd, laat de \omterm{def:g11:fundamental-interactions:ladder}{kern} zo opgewonden
achter dat hij in tweeën scheurt en twee of drie verse neutronen
uitspuugt ---
${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to \text{twee fragmenten} +
2\text{--}3\;{}^{1}_{0}\mathrm{n}$, het paar wisselend van splijting tot
splijting.
\end{definition}

\begin{example}[Eén splijting, gewogen]\label{ex:g12:nuclear-energy:fissionbalance}
Eén frequente route, via \cref{met:g12:nuclear-energy:balance}:
\[
{}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \longrightarrow
{}^{94}_{38}\mathrm{Sr} + {}^{140}_{54}\mathrm{Xe} + 2\,{}^{1}_{0}\mathrm{n}.
\]
Voor: $1.00866 + 234.99350 = \qty{236.00216}{u}$. Na:
$93.89450 + 139.89200 + 2 \times 1.00866 = \qty{235.80382}{u}$. Dus
$\Delta m = \qty{0.19834}{u}$ en $E \approx \qty{185}{MeV}$; de latere
vervallen van de radioactieve fragmenten meegerekend
(\cref{ch:g12:radioactive-decay}), is elke
\omterm{def:g12:nuclear-energy:fission}{splijting} ongeveer \qty{200}{MeV} waard ---
enkele \unit{eV} kopen één \omterm{def:g11:fundamental-interactions:ladder}{atoom} kolen.
\end{example}

\begin{definition}[Ketenreactie en kritische massa]\label{def:g12:nuclear-energy:chain}
Elke \omterm{def:g12:nuclear-energy:fission}{splijting} wordt aangestoken door één neutron
en maakt twee of drie vrij: in een voldoende grote klomp treffen ze
andere kernen en voedt de reactie zichzelf --- een
\emph{ketenreactie}\index{ketenreactie}. In een kleine klomp ontsnappen
de meeste neutronen eerst door het oppervlak; de minimale hoeveelheid
die de keten in stand houdt is de \emph{kritische
massa}\index{kritische massa} --- daaronder sputtert de keten, daarboven
vermenigvuldigt elke generatie: het principe van de bom.
\end{definition}

\begin{definition}[De reactor, getemd]\label{def:g12:nuclear-energy:reactor}
Een \omterm{def:g11:work-of-force:power}{energie}reactor houdt de keten op precies één
neutron per \omterm{def:g12:nuclear-energy:fission}{splijting}. De brandstof: uraniumpellets
bescheiden verrijkt in uranium-235. De
\emph{moderator}\index{moderator} (gewoonlijk gewoon water) vertraagt
de snelle \omterm{def:g12:nuclear-energy:fission}{splijtings}neutronen door botsingen ---
trage neutronen triggeren de volgende
\omterm{def:g12:nuclear-energy:fission}{splijting} veel beter; de
\emph{regelstaven}\index{regelstaaf} (boor, cadmium) verslinden
neutronen --- ingeduwd, sterft de keten; uitgetrokken, snelt hij. De
opwinding van de fragmenten wordt warmte, dan stoom, dan elektriciteit
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Een peperkorrel tegen een truck]\label{ex:g12:nuclear-energy:gram}
Eén gram uranium-235 houdt
$N = \num{1.0e-3} / (235 \times \num{1.66054e-27}) \approx \num{2.56e21}$
kernen; bij $\qty{200}{MeV} = \qty{3.2e-11}{J}$ elk levert ze allemaal
splijten ongeveer \qty{8.2e10}{J} --- de
\omterm{def:g10:energy-conservation:forms}{chemische energie} van $\num{8.2e10} /
\num{3.0e7} \approx \qty{2.7e3}{kg}$ kolen: een peperkorrel tegen drie
ton.
\end{example}

\section{Fusie: het eigen vuur van de Zon}

\begin{definition}[Fusie]\label{def:g12:nuclear-energy:fusion}
\emph{Fusie}\index{fusie} is het samensmelten van twee lichte kernen
tot een zwaardere, sterker gebonden. De meest toegankelijke reactie
trouwt deuterium en tritium, de zware
\omterm{def:g11:nucleus-radioactivity:notation}{isotopen} van waterstof
(\cref{ch:g11:nucleus-radioactivity}):
\[
{}^{2}_{1}\mathrm{H} + {}^{3}_{1}\mathrm{H} \longrightarrow
{}^{4}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}.
\]
\end{definition}

\begin{example}[D--T, gewogen]\label{ex:g12:nuclear-energy:dt}
Voor: $2.01355 + 3.01550 = \qty{5.02905}{u}$; na:
$4.00151 + 1.00866 = \qty{5.01017}{u}$; dus $\Delta m = \qty{0.01888}{u}$
en $E \approx \qty{17.6}{MeV}$, vier vijfde op het neutron --- minder
dan een \omterm{def:g12:nuclear-energy:fission}{splijting}, maar uit vijf
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen} in plaats van $236$: ongeveer
\qty{3.4e14}{J} per \omterm{def:g10:orders-of-magnitude:unit}{kilogram} brandstof, viermaal
\omterm{def:g12:nuclear-energy:fission}{splijting}. Helium-4
(\cref{ex:g12:nuclear-energy:helium}) is opnieuw de as.
\end{example}

\begin{definition}[De Coulombbarrière]\label{def:g12:nuclear-energy:barrier}
Twee kernen zijn beide positief: om te raken moeten ze eerst de heuvel
van hun elektrische afstoting beklimmen
(\cref{ch:g11:electric-gravitational-fields}) --- de
\emph{Coulombbarrière}\index{Coulombbarrière}. Alleen nabij
\qty{1e8}{K} dragen botsingen de
\omterm{def:g10:energy-conservation:energy}{energie} om haar over te steken; de
brandstof is dan een \emph{plasma}\index{plasma} van bare kernen en
elektronen. \omterm{def:g12:nuclear-energy:fission}{Splijting} betaalt geen tol: haar
trekker, het neutron, is neutraal, en \omterm{def:g11:work-of-force:work}{werkt}
koud.
\end{definition}

\begin{example}[De Zon draait erop]\label{ex:g12:nuclear-energy:sun}
De kern van de Zon, bij $\qty{1.5e7}{K}$,
\omterm{def:g11:circuits-and-power:joule}{fuseert} gewone waterstof stap voor stap ---
de \emph{proton--protonketen}, met netto-effect
$4\,{}^{1}_{1}\mathrm{H} \to {}^{4}_{2}\mathrm{He} + 2\,{}^{0}_{+1}\mathrm{e}
+ \text{straling}$, ongeveer \qty{26}{MeV} per helium: $0.7\%$ van de
massa van de waterstof vertrekt als
\omterm{def:g10:energy-conservation:energy}{energie}. Het zonlicht waarvan we het
\omterm{def:g12:sound-acoustics:timbre}{spectrum} lazen in \cref{ch:g10:light-spectra} is
deze massa, acht minuten te laat; elke ster schijnt door de kromme naar
ijzer te lopen.
\end{example}

\begin{remark}[ITER: een ster bottelen]\label{rem:g12:nuclear-energy:iter}
Geen vaste stof kan een \omterm{def:g12:nuclear-energy:barrier}{plasma} bij
\qty{1.5e8}{K} houden; een \emph{tokamak} kooit hem in een
\omterm{def:g11:magnetic-fields:bfield}{magnetisch veld}
(\cref{ch:g11:magnetic-fields}), een donut van
\omterm{def:g11:electric-gravitational-fields:lines}{veldlijnen} waarlangs de geladen
deeltjes spiraalsgewijs lopen zonder de muur te raken; ITER, gebouwd om
tienmaal zijn verwarmings\omterm{def:g11:work-of-force:power}{vermogen} vrij te maken,
is de huidige stap. Breek de opsluiting en het
\omterm{def:g12:nuclear-energy:barrier}{plasma} koelt en stopt in seconden: geen
\omterm{def:g12:nuclear-energy:chain}{ketenreactie}, geen
\omterm{def:g12:nuclear-energy:chain}{kritische massa} --- het moeilijke is niet het
vuur stoppen maar het brandend houden.
\end{remark}

\begin{example}[De energieladder, per kilogram]\label{ex:g12:nuclear-energy:ladder}
\omterm{def:g10:energy-conservation:energy}{Energie} per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} brandstof: kolen,
\qty{3.0e7}{J}; uranium-235 per \omterm{def:g12:nuclear-energy:fission}{splijting},
\qty{8.2e13}{J}; deuterium--tritium per
\omterm{def:g12:nuclear-energy:fusion}{fusie}, \qty{3.4e14}{J}; totale omzetting
($E = mc^2$), \qty{9.0e16}{J}. Zes ordegrootten scheiden chemie van de
\omterm{def:g11:fundamental-interactions:ladder}{kern} --- de reden dat een reactor per
truck wordt bijgetankt en een kolencentrale per treinlading.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:1}
Zet om: (a) \qty{1}{eV} en \qty{1}{MeV} in
\omterm{def:g11:work-of-force:work}{joule}; (b) de \qty{200}{MeV} van één
\omterm{def:g12:nuclear-energy:fission}{splijting} in
\omterm{def:g11:work-of-force:work}{joule}; (c) controleer uit
$\qty{1}{u} = \qty{1.66054e-27}{kg}$ dat zijn
\omterm{def:g10:energy-conservation:energy}{energie}-equivalent dicht bij
\qty{931.5}{MeV} ligt.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:2}
Een suikerklontje heeft massa \qty{6.0}{g}: bereken zijn volle
$E = mc^2$-waarde. Een huishouden trekt
$\qty{10}{kW.h} = \qty{3.6e7}{J}$ per dag --- voor hoeveel jaren kon
het klontje, volledig omgezet, het \omterm{def:g11:work-of-force:power}{voeden}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:3}
De koolstof-12-\omterm{def:g11:fundamental-interactions:ladder}{kern} heeft massa
\qty{11.99671}{u}: bereken zijn
\omterm{def:g12:nuclear-energy:defect}{massadefect}, zijn
\omterm{def:g12:nuclear-energy:defect}{bindingsenergie} in \unit{MeV}, en zijn
\omterm{def:g12:nuclear-energy:pernucleon}{bindingsenergie per nucleon}
($m_p = \qty{1.00728}{u}$, $m_n = \qty{1.00866}{u}$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:nuclear-energy:4}
Voltooi elke \omterm{def:g12:nuclear-energy:fission}{splijting}, geef $A$, $Z$ en het

(b) ${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to

\end{exercise}

\section{Probleem: Een stad voeden}

\begin{problem}[{Weekendprobleem --- een stad voeden: één gestage gigawatt voor
een half miljoen mensen, vier manieren gekocht --- uraniumpellets,
treinladings kolen, een bad zeewater, de eigen stof van de Zon --- met
hetzelfde kilogram verdwenen massa onder alle vier}]
\label{pb:g12:nuclear-energy:1}
Een stad van $500\,000$ inwoners trekt een gestage elektrische
\omterm{def:g11:work-of-force:power}{vermogen} $P = \qty{1.0}{GW}$; haar centrales,
welke brandstof ook, zetten warmte om in elektriciteit met
\omterm{def:g10:energy-conservation:efficiency}{rendement} $33\%$
(\cref{ch:g10:energy-conservation}). Gegevens: één
uranium-235-\omterm{def:g12:nuclear-energy:fission}{splijting}, \qty{200}{MeV} totaal;
kolen, \qty{3.0e7}{J/kg}; één D--T-\omterm{def:g12:nuclear-energy:fusion}{fusie},
\qty{17.6}{MeV} uit \qty{5.029}{u} brandstof;
$\qty{1}{u} = \qty{1.66054e-27}{kg}$; $\qty{1}{eV} = \qty{1.60e-19}{J}$;
één jaar $= \qty{3.16e7}{s}$; de Zon: \qty{3.9e26}{W}, \qty{2.0e30}{kg};
een kubieke \omterm{def:g10:orders-of-magnitude:unit}{meter} zeewater houdt ongeveer
\qty{33}{g} deuterium.

\medskip
\noindent\textbf{Deel I --- De \omterm{def:g12:nuclear-energy:fission}{splijtings}centrale.}
\begin{enumerate}
  \item Welk thermisch \omterm{def:g11:work-of-force:power}{vermogen} moet de reactor
        leveren?
  \item Druk de \omterm{def:g10:energy-conservation:energy}{energie} van één
        \omterm{def:g12:nuclear-energy:fission}{splijting} uit in
        \omterm{def:g11:work-of-force:work}{joule}.
  \item Hoeveel \omterm{def:g12:nuclear-energy:fission}{splijtingen} per seconde drijven
        de stad?
  \item De massa van één uranium-235-\omterm{def:g11:fundamental-interactions:ladder}{kern}
        ($A = 235$), dan het uranium-235 verbruikt per seconde?
  \item Leid het uranium-235 verbruikt in één jaar af, in
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram}.
  \item De brandstof is verrijkt tot $4.0\%$ uranium-235: welke massa
        neemt een jaar, en past die op één truck?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Het kolengrootboek.}
\begin{enumerate}[resume]
  \item Bereken de \omterm{def:g10:energy-conservation:forms}{thermische energie} die de
        stad in één jaar verbruikt.
  \item Welke massa kolen levert die?
  \item Bereken de verhouding (kolenmassa)/(uranium-235-massa) voor één
        jaar, en controleer tegen de ladder van
        \cref{ex:g12:nuclear-energy:ladder}.
  \item Een goederentrein sleept \qty{3.0e6}{kg}: hoeveel treinen per
        jaar, en per dag? Eén zin: beeld de twee aanvoerlijnen.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De \omterm{def:g12:nuclear-energy:fusion}{fusie}droom.}
\begin{enumerate}[resume]
  \item Druk de \omterm{def:g10:energy-conservation:energy}{energie} van één
        D--T-\omterm{def:g12:nuclear-energy:fusion}{fusie} uit in
        \omterm{def:g11:work-of-force:work}{joule}.
  \item Bereken de \omterm{def:g10:energy-conservation:energy}{energie} vrijgemaakt per
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram} D--T-brandstof; vergelijk met
        de ladder.
  \item Welke massa D--T-brandstof dekt het stadsjaar? Welk deel daarvan
        is deuterium ($2.014$ van de \qty{5.029}{u})?
  \item Welk volume zeewater bevat dat deuterium? Vergelijk met een
        olympisch bad, ongeveer \qty{2.5e3}{m^3}. (Tritium wordt gekweekt
        uit lithium.)
  \item Waarom draagt deze centrale geen
        \omterm{def:g12:nuclear-energy:chain}{kritische massa} en geen
        \omterm{def:g12:nuclear-energy:chain}{ketenreactie} om te vrezen? Wat is dan het
        moeilijke deel?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De ster die in massa betaalt.}
\begin{enumerate}[resume]
  \item Uit $E = mc^2$, welke massa zet de Zon elke seconde om?
  \item Welke massa per seconde, en per jaar, moet \emph{elke}
        thermische bron van \qty{3.0}{GW} omzetten? Controleer dat alle
        drie brandstoffen deze zelfde massa leveren.
  \item Welke totale massa heeft de Zon in haar $\num{4.5e9}$ jaar
        uitgestraald, en welk deel van de Zon is dat?
  \item \omterm{def:g12:nuclear-energy:fusion}{Fusie} kan ongeveer $0.07\%$ van de massa
        van de Zon tappen (het brandbare aandeel van haar kern): schat
        haar totale levensduur als ster, en wat overblijft.
  \item Sluit de audit in drie zinnen: één stadsjaar in tonnen kolen,
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram} uranium-235 en
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram} D--T-brandstof --- en het
        \omterm{def:g10:orders-of-magnitude:unit}{kilogram} massa dat, in elk scenario, het
        licht van de stad werd.
\end{enumerate}
\end{problem}
