\chapter{Geluid en akoestiek}\label{ch:g12:sound-acoustics}

Tokkel de lage snaar van een gitaar: de kamer vult zich met een lage
A, en toch reisde er niets naar je oor dan samengedrukte en
uitgerekte lucht. Dit hoofdstuk volgt dat patroon --- zijn snelheid,
hoe \omterm{def:g10:signals-and-waves:frequency}{frequentie} \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} wordt en
\omterm{def:g11:work-of-force:power}{vermogen} \omterm{def:g12:sound-acoustics:pitch-loudness}{luidheid}, en waarom één noot op elk
instrument anders klinkt.

\section{Geluid is een drukgolf}

\begin{definition}[Geluidsgolf]\label{def:g12:sound-acoustics:sound-wave}
Een \emph{geluidsgolf}\index{geluidsgolf} is een mechanische
\omterm{def:g10:pressure:pressure}{druk}\omterm{def:g10:signals-and-waves:wave}{golf}
(\cref{ch:g12:mechanical-waves}): een trillend oppervlak creëert
reizende \emph{compressies}\index{compressie} (lichte overdruk) en
\emph{verdunningen}\index{verdunning} (onderdruk). Luchtpakketjes
oscilleren \emph{langs} de reisrichting
(\omterm{def:g10:signals-and-waves:sound}{geluid} is \emph{longitudinaal}\index{longitudinale golf}),
en er is een medium nodig: een bel onder een vacuümklok valt stil
(\cref{ch:g10:signals-and-waves}).
\end{definition}

\begin{proposition}[Geluidssnelheid]\label{prop:g12:sound-acoustics:speed}
\index{geluidssnelheid}%
\omterm{def:g10:signals-and-waves:sound}{Geluid} reist met ongeveer \qty{340}{m/s} in lucht bij
\qty{20}{\degreeCelsius}, \qty{1500}{m/s} in water en
\qty{5000}{m/s} in staal: hoe stijver het medium, hoe
sneller. \admitted
\end{proposition}

\begin{remark}[Warme lucht is sneller]\label{rem:g12:sound-acoustics:temperature}
Dit zijn gemeten waarden; ze voorspellen gebeurt in het volume van
jaar~1. \omterm{def:g10:signals-and-waves:sound}{Geluid} wordt sneller in warmere lucht (\qty{331}{m/s}
bij \qty{0}{\degreeCelsius}) --- waarom blazers ontstemmen als de
zaal opwarmt.
\end{remark}

\begin{example}[De afmetingen van geluiden]\label{ex:g12:sound-acoustics:wavelengths}
De relatie $\lambda = v/f$ bemeten elk \omterm{def:g10:signals-and-waves:sound}{geluid}: in lucht
beslaat een \qty{20}{Hz}-gedreun $340/20 = \qty{17}{m}$ --- een
gebouw --- en een \qty{20}{kHz}-sis past in \qty{1.7}{cm}; de
metergrote buigen gemakkelijk om deuren en hoeken.
\end{example}

\section{Toonhoogte, luidheid en het oor}

\begin{definition}[Toonhoogte en luidheid]\label{def:g12:sound-acoustics:pitch-loudness}
\emph{Toonhoogte}\index{toonhoogte} is de
\omterm{def:g11:color-light-sources:perception}{waarneming} van \omterm{def:g10:signals-and-waves:frequency}{frequentie}: hoe hoger
$f$, hoe hoger de noot; concert-A is \qty{440}{Hz}, en verdubbeling
van $f$ heft de noot één octaaf. \emph{Luidheid}\index{luidheid} is
de \omterm{def:g11:color-light-sources:perception}{waarneming} van \omterm{def:g10:signals-and-waves:amplitude}{amplitude}: hoe
sterker de \omterm{def:g10:pressure:pressure}{druk}oscillatie, hoe luider; versterking
verandert de luidheid, niet de toonhoogte.
\end{definition}

\begin{remark}[Het venster van het oor]\label{rem:g12:sound-acoustics:audible}
Horen bestrijkt ongeveer \qty{20}{Hz} tot \qty{20}{kHz}
(\cref{ch:g10:signals-and-waves}); daaronder ligt
\omterm{def:g10:signals-and-waves:ultrasound}{infrageluid}, daarboven \omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid}
--- vleermuizen bij \qty{50}{kHz}, medische sondes bij megahertz. Een
piano bemonstert alleen het midden: \qtyrange{27.5}{4186}{Hz}.
\end{remark}

\section{Geluidsintensiteit en de decibelschaal}

\begin{definition}[Geluidsintensiteit]\label{def:g12:sound-acoustics:intensity}
De \emph{geluidsintensiteit}\index{geluidsintensiteit} $I$ is het
akoestische \omterm{def:g11:work-of-force:power}{vermogen} dat één vierkante
\omterm{def:g10:orders-of-magnitude:unit}{meter} loodrecht op de \omterm{def:g10:signals-and-waves:wave}{golf}
doorkruist: $I = P/S$, in \unit{W/m^2}. Het stilste hoorbare
\omterm{def:g10:signals-and-waves:sound}{geluid} is ongeveer $I_0 = \qty{1.0e-12}{W/m^2}$; pijn
begint nabij \qty{1}{W/m^2}.
\end{definition}

\begin{proposition}[Omgekeerd-kwadraatdemping]\label{prop:g12:sound-acoustics:inverse-square}
\index{omgekeerd-kwadraatwet}%
Een kleine bron die een \omterm{def:g11:work-of-force:power}{vermogen} $P$ gelijkmatig in alle
richtingen uitstraalt, geeft op afstand $r$
$I = P/(4\pi r^2) \propto 1/r^2$: verdubbeling van de afstand deelt
de intensiteit door vier.
\end{proposition}

\begin{proof}
Het hele \omterm{def:g11:work-of-force:power}{vermogen} $P$ doorkruist de bol van oppervlakte
$4\pi r^2$; deel.
\end{proof}

\begin{definition}[Geluidsniveau]\label{def:g12:sound-acoustics:level}
Het oor bestrijkt twaalf \omterm{def:g11:work-of-force:power}{machten} van tien, dus
intensiteiten worden logaritmisch opgegeven. Het
\emph{geluidsniveau}\index{geluidsniveau} van $I$ is
\[ L = 10 \log_{10}(I/I_0), \qquad I_0 = \qty{1.0e-12}{W/m^2}, \]
in \emph{decibel}\index{decibel} (\unit{dB}): gehoordrempel
\qty{0}{dB}, pijn nabij \qty{120}{dB}.
\end{definition}

\begin{proposition}[Decibelrekenen]\label{prop:g12:sound-acoustics:db-arithmetic}
Vermenigvuldigen van $I$ met $k$ voegt $10\log_{10}k$
\omterm{def:g12:sound-acoustics:level}{decibel} toe. In het bijzonder voegt $\times 10$
\qty{10}{dB} toe; $\times 2$ voegt $10\log_{10}2 \approx \qty{3.0}{dB}$
toe; $N$ incoherente identieke bronnen tellen
\emph{intensiteiten} op, nooit niveaus, wat
$L_1 + 10\log_{10}N$ geeft; verdubbeling van de afstand tot een
kleine bron ($I$ gedeeld door vier) haalt ongeveer \qty{6.0}{dB}
weg.
\end{proposition}

\begin{proof}
$10\log_{10}(kI/I_0) = 10\log_{10}(I/I_0) + 10\log_{10}k$;
afstand: \cref{prop:g12:sound-acoustics:inverse-square}, $k = 1/4$.
\end{proof}

\begin{method}[Werken in decibel]\label{met:g12:sound-acoustics:db}
\begin{enumerate}
  \item $L = 10\log_{10}(I/I_0)$; terug: $I = I_0 \times 10^{L/10}$.
  \item Tel nooit niveaus op: zet om naar intensiteiten, tel op, zet
        terug --- of $+\qty{3}{dB}$ per verdubbeling, $+\qty{10}{dB}$
        per tienvoud.
  \item Elke verdubbeling van de afstand tot een kleine bron kost
        \qty{6}{dB}.
\end{enumerate}
\end{method}

\begin{remark}[Gehoorschade]\label{rem:g12:sound-acoustics:damage}
Gehoorverlies is een \omterm{rem:g11:nucleus-radioactivity:dose}{dosis}: \qty{85}{dB} is veilig
gedurende acht uur, en elke \qty{3}{dB} meer --- een verdubbeling van
de intensiteit --- halveert de veilige tijd; verloren haarcellen
groeien niet terug.
\end{remark}

\section{Timbre en harmonischen}

\begin{definition}[Harmonischen]\label{def:g12:sound-acoustics:harmonics}
Een \omterm{def:g10:signals-and-waves:period}{periodiek} \omterm{def:g10:signals-and-waves:sound}{geluid} van
\omterm{def:g10:signals-and-waves:frequency}{frequentie} $f_1$ --- de
\emph{grondtoon}\index{grondtoon} --- is in het algemeen een som van
sinussen bij de frequenties $f_n = n f_1$, $n = 1, 2, 3, \dots$, zijn
\emph{harmonischen}\index{harmonische}. Een
\omterm{def:g10:signals-and-waves:sound}{geluid} teruggebracht tot zijn grondtoon is een
\emph{zuivere toon}\index{zuivere toon}, zoals die van een stemvork.
\end{definition}

\begin{definition}[Spectrum en timbre]\label{def:g12:sound-acoustics:timbre}
Het \emph{spectrum}\index{spectrum} van een
\omterm{def:g10:signals-and-waves:sound}{geluid} is het staafdiagram van zijn
\omterm{def:g12:sound-acoustics:harmonics}{harmonischen}: één staaf per $f_n$, de hoogte de
sterkte van die \omterm{def:g12:sound-acoustics:harmonics}{harmonische}. Het oor hoort $f_1$
als de \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} en de \emph{mix} van sterktes
als het \emph{timbre}\index{timbre} --- de kleur van het
\omterm{def:g10:signals-and-waves:sound}{geluid}: instrumenten die dezelfde noot spelen delen
$f_1$ en verschillen alleen in hun spectra.
\end{definition}

\begin{example}[Waarom je de beller herkent]\label{ex:g12:sound-acoustics:two-instruments}
De A van een fluit is bijna een \omterm{def:g12:sound-acoustics:harmonics}{zuivere toon}; een
viool die dezelfde \qty{220}{Hz} strijkt voegt sterke
\omterm{def:g12:sound-acoustics:harmonics}{harmonischen} toe bij \qtylist{440;660;880}{Hz}.

\omterm{def:g12:sound-acoustics:pitch-loudness}{luidheid}, en toch verwart niemand ze: een stem
is ook een \omterm{def:g12:sound-acoustics:timbre}{spectrum} dat je uit het hoofd kent.
\end{example}

\section{De trillende snaar}

\begin{definition}[Staande golf]\label{def:g12:sound-acoustics:standing-wave}
Een \omterm{def:g10:signals-and-waves:wave}{golf} opgesloten tussen de vaste einden van een snaar
kan zich settelen tot een \emph{staande golf}\index{staande golf}:
een oscillatie op zijn plaats, bewegingsloze punten ---
\emph{knopen}\index{knoop} --- afgewisseld met punten van maximale
zwaai --- \emph{buiken}\index{buik}. Niets reist meer; de snaar
trilt in een vast patroon.
\end{definition}

\begin{proposition}[Modi van een snaar]\label{prop:g12:sound-acoustics:string-modes}
Een snaar van lengte $L$, vast aan beide einden, die
\omterm{def:g10:signals-and-waves:wave}{golven} van snelheid $v$ draagt, trilt stationair alleen
in patronen die een geheel getal $n$
\omterm{def:g12:mechanical-waves:wavelength}{halve golflengten} tussen de opgelegde
eind\omterm{def:g12:sound-acoustics:standing-wave}{knopen} passen:
\[ \lambda_n = \frac{2L}{n}, \quad f_n = n\,\frac{v}{2L} = n f_1, \quad n = 1, 2, 3, \dots \]
De laagste modus $f_1 = v/(2L)$ is de
\omterm{def:g12:sound-acoustics:harmonics}{grondtoon}; de andere zijn precies zijn
\omterm{def:g12:sound-acoustics:harmonics}{harmonischen}. \admitted
\end{proposition}

\begin{remark}[Waar het eerlijke bewijs leeft]\label{rem:g12:sound-acoustics:honest}
Dat opgesloten \omterm{def:g10:signals-and-waves:wave}{golven} precies deze modi kiezen volgt
uit superpositie van de \omterm{def:g10:signals-and-waves:wave}{golf} met zijn
reflecties, een berekening uitgevoerd in het volume van jaar~1. De
meetkunde overtuigt: beide einden moeten stil blijven, dus moet een
geheel aantal \omterm{def:g12:mechanical-waves:wavelength}{halve golflengten} passen.
\end{remark}

\begin{example}[De A-snaar van een viool]\label{ex:g12:sound-acoustics:violin}
De A-snaar van een viool heeft $L = \qty{32.8}{cm}$ en
\omterm{def:g10:signals-and-waves:sound}{klinkt} $f_1 = \qty{440}{Hz}$:
\omterm{def:g10:signals-and-waves:wave}{golven} lopen erover met
$v = 2Lf_1 = 2 \times 0.328 \times 440 \approx \qty{289}{m/s}$ ---
bepaald door \omterm{def:g10:inertia:inventory}{spanning} en massa, gestemd met de
stemknop. Een vinger indrukken verkort $L$ en heft elke $f_n$
tegelijk: het \omterm{def:g12:sound-acoustics:timbre}{spectrum} schuift omhoog.
\end{example}

\begin{remark}[Waarom instrumenten verschillen]\label{rem:g12:sound-acoustics:instruments}
Hoe de snaar wordt aangeslagen --- getokkeld, gestreken, geslagen ---
bepaalt hoeveel van elke modus aanwezig is, en de klankkast
versterkt sommige \omterm{def:g12:sound-acoustics:harmonics}{harmonischen} meer dan andere;
aanslag plus kast bepalen het \omterm{def:g12:sound-acoustics:timbre}{spectrum}, zodat
dezelfde $f_1$ een gitaar, een viool en een piano elk als zichzelf
laat klinken.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:1}
In lucht ($v = \qty{340}{m/s}$), vind de
\omterm{def:g12:mechanical-waves:wavelength}{golflengte} van (a) een \qty{55}{Hz}-basnoot;
(b) een \qty{1.0}{kHz}-piep; (c) een \qty{15}{kHz}-jank. Vergelijk
de grootste met de kleinste.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:2}
Classificeer als \omterm{def:g10:signals-and-waves:ultrasound}{infrageluid}, hoorbaar
\omterm{def:g10:signals-and-waves:sound}{geluid} of \omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid}:
\qty{8}{Hz}; \qty{100}{Hz}; \qty{15}{kHz}; \qty{40}{kHz};
\qty{3}{MHz}. Van de hoorbare: welke \omterm{def:g10:signals-and-waves:sound}{klinkt} lager
in \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:3}
Een arbeider slaat op een rail \qty{1.0}{km} verderop. Bereken de
reistijden in het staal en in de lucht. Wat hoort een luisteraar met
één oor tegen de rail?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:4}
(a) Het \omterm{def:g12:sound-acoustics:level}{geluidsniveau} voor
$I = \qty{1.0e-5}{W/m^2}$? (b) De intensiteit bij \qty{60}{dB}?
(c) Noem een alledaags \omterm{def:g10:signals-and-waves:sound}{geluid} nabij elk niveau.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sound-acoustics:5}
Een viool-G-snaar \omterm{def:g10:signals-and-waves:sound}{klinkt} $f_1 = \qty{196}{Hz}$:
geef zijn volgende drie \omterm{def:g12:sound-acoustics:harmonics}{harmonischen}. Waarom
bereikt een \qty{12}{kHz}-fluitje, hoe rijk ook bij de bron, het oor
als een \omterm{def:g12:sound-acoustics:harmonics}{zuivere toon}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:6}
Een sirene straalt $P = \qty{0.50}{W}$ gelijkmatig in alle
richtingen. Bereken de intensiteit en het
\omterm{def:g12:sound-acoustics:level}{geluidsniveau} \qty{5.0}{m} verderop.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:7}
Eén viool geeft \qty{68}{dB} op je stoel. Welk niveau geven er twee?
Vier? Hoeveel violen zouden nodig zijn om \qty{78}{dB} te bereiken?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:8}
Het niveau is \qty{95}{dB} op \qty{2.0}{m} van een kleine
luidspreker. Voorspel het niveau op \qty{8.0}{m}: onder de
\qty{85}{dB}-grens?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sound-acoustics:9}

\section{Probleem: Het atelier van de luthier}

\begin{problem}[{Weekendprobleem --- het atelier van de luthier: een
gitaarsnaar fret voor fret uitgelegd, zijn harmonischen gestemd tot
een timbre, een recital onder de gevaarlijn gehouden, en een
ultrageluidjacht op een fout in het hout}]
\label{pb:g12:sound-acoustics:1}
Een luthier werkt een gitaar af. Zijn A-snaar heeft trillende lengte
$L = \qty{65.0}{cm}$ en moet \omterm{def:g10:signals-and-waves:sound}{klinken}
$f_1 = \qty{110}{Hz}$. Neem \qty{340}{m/s} voor
\omterm{def:g10:signals-and-waves:sound}{geluid} in lucht, en geef toe dat
$f_1 = v/(2L)$ (\cref{prop:g12:sound-acoustics:string-modes}), waarbij $v$ de
\omterm{def:g10:signals-and-waves:wave}{golf}snelheid op de snaar is.

\medskip
\noindent\textbf{Deel I --- De fretboard uitleggen.}
\begin{enumerate}
  \item Bereken de \omterm{def:g10:signals-and-waves:wave}{golf}snelheid $v$ op de snaar.
  \item Bereken de \omterm{def:g12:mechanical-waves:wavelength}{golflengte} van de
        \omterm{def:g12:sound-acoustics:harmonics}{grondtoon} \emph{op de snaar}, daarna de
        \omterm{def:g12:mechanical-waves:wavelength}{golflengte} van het \qty{110}{Hz}-
        \omterm{def:g10:signals-and-waves:sound}{geluid} \emph{in de lucht}. Waarom
        verschillen ze bij dezelfde
        \omterm{def:g10:signals-and-waves:frequency}{frequentie}?
  \item Het octaaf (\qty{220}{Hz}): welke trillende lengte geeft het?
        Waar moet de twaalfde fret zitten, als breukdeel van $L$?
  \item De kwint (\qty{165}{Hz}): welke trillende lengte geeft hem?
  \item Elke fret heft de noot één halve toon, een factor
        $2^{1/12}$. Controleer met logaritmen hoeveel halve tonen
        \qty{110}{Hz} van \qty{220}{Hz} scheiden; lengte bij de
        eerste fret?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- \omterm{def:g12:sound-acoustics:harmonics}{Harmonischen} en
\omterm{def:g12:sound-acoustics:timbre}{timbre}.}
\begin{enumerate}[resume]
  \item Geef de eerste vijf \omterm{def:g12:sound-acoustics:harmonics}{harmonischen} van de
        open A-snaar.
  \item De A van een violist \omterm{def:g10:signals-and-waves:sound}{klinkt} \qty{440}{Hz}.
        Welke \omterm{def:g12:sound-acoustics:harmonics}{harmonische} van de gitaarsnaar is
        dat, en welk muzikaal interval scheidt hem van de
        \omterm{def:g12:sound-acoustics:harmonics}{grondtoon}? En \qty{330}{Hz}?
  \item De luthier tokkelt dicht bij de kam, dan over de toets: zelfde
        \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte}, andere kleur. Leg uit met
        spectra.
  \item Een lichte aanraking op het midden tijdens tokkelen
        \omterm{def:g10:inertia:force}{forceert} daar een
        \omterm{def:g12:sound-acoustics:standing-wave}{knoop}. Welke
        \omterm{def:g12:sound-acoustics:harmonics}{harmonischen} overleven, en welke
        \omterm{def:g12:sound-acoustics:pitch-loudness}{toonhoogte} wordt gehoord?
  \item Hoeveel \omterm{def:g12:sound-acoustics:harmonics}{harmonischen} van de open snaar
        liggen, in principe, binnen het hoorbare
        \omterm{def:g11:fundamental-interactions:strong}{bereik}?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Het recital.}
Bij het eerste concert van de gitaar straalt elk instrument een
akoestisch \omterm{def:g11:work-of-force:power}{vermogen} $P = \qty{6.0e-3}{W}$ uit als
kleine bron, gelijkmatig in alle richtingen.
\begin{enumerate}[resume]
  \item Bereken de intensiteit op $r = \qty{3.0}{m}$ van één
        instrument.
  \item Leid het \omterm{def:g12:sound-acoustics:level}{geluidsniveau} daar af.
  \item Een kwartet speelt: welk niveau geven vier zulke instrumenten
        op dezelfde plek? Waarom is het niet viermaal het niveau?
  \item Twaalf instrumenten spelen: wat is het niveau op
        \qty{3.0}{m}?
  \item Hoe ver achteraan moet een luisteraar zitten om het ensemble
        tot \qty{85}{dB} te laten dalen?
  \item Veiligheidsregel: \qty{85}{dB} is veilig gedurende acht uur,
        en elke \qty{+3}{dB} halveert de veilige tijd. Hoe lang mag
        de front-row-luisteraar van vraag 14 veilig blijven?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De fout in het hout.}
Voor het vernissen peilt de luthier een \qty{45}{mm} dikke
halsblank met een \qty{2.0}{MHz}-tester; in dit hout neem
$v = \qty{5.0e3}{m/s}$.
\begin{enumerate}[resume]
  \item Bereken de \omterm{def:g12:mechanical-waves:wavelength}{golflengte} van het
        \omterm{def:g10:signals-and-waves:ultrasound}{ultrageluid} in het hout. Waarom moet
        een fout-jagende \omterm{def:g10:signals-and-waves:wave}{golf} ultrasonisch zijn
        in plaats van hoorbaar?
  \item Bereken de \omterm{rem:g10:signals-and-waves:honest}{echo}\omterm{def:g12:mechanical-waves:delay}{vertraging}
        van de achterkant van de blank.
  \item Een \omterm{rem:g10:signals-and-waves:honest}{echo} arriveert na \qty{7.2}{\micro s}:
        hoe diep is de fout?
  \item Schrijf de atelierkaart: \omterm{def:g10:signals-and-waves:wave}{golf}snelheid op de
        snaar, lengten van twaalfde en eerste fret, veilige
        luisterafstand, diepte van de fout.
\end{enumerate}
\end{problem}
