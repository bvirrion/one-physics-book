\chapter{Vrije elektrische oscillaties: RLC}\label{ch:g12:rlc-oscillations}

Draai de zware knop van een zolderradio en de zenders passeren de
naald: een stem, violen, ruis, weer een stem. Achter het paneel is
geen \omterm{def:g11:work-of-force:sign}{motor} en geen computer --- een
\omterm{def:g11:magnetic-fields:solenoid}{spoel}, een \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}
waarvan de ineengrijpende platen de knop draait, en een stuk
antennedraad. Dit hoofdstuk koppelt de twee
\omterm{def:g10:energy-conservation:energy}{energie}tanks van het vorige hoofdstuk
(\cref{ch:g12:rc-rl-circuits}) kop aan kop en vindt de snelste slinger
ooit gebouwd: lading die heen en weer zwaait, tot miljoenen keren per
seconde, op een tempo dat de knop kiest.

\section{Een elektrische slinger: het LC-circuit}

Laad een \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} $C$ tot een
\omterm{def:g11:circuits-and-power:voltage}{spanning} $U_0$ --- zijn platen houden dan
$\pm Q_0$ met $Q_0 = C U_0$ --- en schakel hem bij $t = 0$ op een
\emph{ideale} \omterm{def:g11:magnetic-fields:solenoid}{spoel} $L$, van nul
\omterm{def:g11:circuits-and-power:resistance}{weerstand}. Nergens in de lus een bron: wat
vervolgens gebeurt, doet het circuit zelf.

een ideale \omterm{def:g11:magnetic-fields:solenoid}{spoel}. $K$ sluiten laat de lading
door de \omterm{def:g11:magnetic-fields:solenoid}{spoel} lopen --- en de
\omterm{def:g11:magnetic-fields:solenoid}{spoel}, die verandering haat, zal hem niet
laten stoppen.}

\begin{proposition}[De LC-vergelijking]\label{prop:g12:rlc-oscillations:equation}
\index{LC-circuit}
In de ideale LC-lus gehoorzaamt de lading $q(t)$ van de bovenste plaat
\[
L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0,
\qquad\text{d.w.z.}\qquad
\frac{d^2q}{dt^2} = -\frac{1}{LC}\,q .
\]
\end{proposition}

\begin{proof}
Rond de lus sommen de \omterm{def:g11:circuits-and-power:voltage}{spanningen} tot nul
(\cref{prop:g11:circuits-and-power:laws}): $u_L + u_C = 0$. De
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} geeft $u_C = q/C$ en de
\omterm{def:g11:magnetic-fields:solenoid}{spoel} $u_L = L\,di/dt$
(\cref{def:g12:rc-rl-circuits:inductor}), met $i = dq/dt$ die de lading
telt die op de bovenste plaat aankomt
(\cref{prop:g12:rc-rl-circuits:cdudt}); substitueer.
\end{proof}

\begin{theorem}[Vrije oscillaties van het LC-circuit]\label{thm:g12:rlc-oscillations:oscillation}
\index{elektrische oscillaties}
Losgelaten bij $t = 0$ met $q(0) = Q_0$ en $i(0) = 0$ oscilleert het
circuit:
\[
q(t) = Q_0 \cos\!\left(\frac{2\pi t}{T_0}\right),
\qquad
i(t) = -I_0 \sin\!\left(\frac{2\pi t}{T_0}\right),
\qquad
T_0 = 2\pi\sqrt{LC},
\]
met piekstroom $I_0 = 2\pi Q_0/T_0 = Q_0/\sqrt{LC}$.
\end{theorem}

\begin{proof}
Verifieer door substitutie. Differentiëren:
$i = dq/dt = -\frac{2\pi}{T_0} Q_0 \sin(2\pi t/T_0)$ --- de gestelde
$i(t)$ --- en opnieuw $\frac{d^2q}{dt^2} = -(\frac{2\pi}{T_0})^2 q$,
wat precies $-q/(LC)$ is wanneer $(2\pi/T_0)^2 = 1/(LC)$, d.w.z.\
$T_0 = 2\pi\sqrt{LC}$. En $q(0) = Q_0$, $i(0) = 0$. Dat geen
\emph{andere} functie aan de vergelijking en de start voldoet, wordt
toegegeven, precies als bij de veer
(\cref{rem:g12:oscillators-and-time:uniqueness}).
\end{proof}

\begin{definition}[Eigenperiode en eigenfrequentie]\label{def:g12:rlc-oscillations:period}
De \emph{eigenperiode}\index{eigenperiode!van een LC-circuit} van een
\omterm{prop:g12:rlc-oscillations:equation}{LC-circuit} is $T_0 = 2\pi\sqrt{LC}$; zijn
\emph{eigenfrequentie}\index{eigenfrequentie!van een LC-circuit} is
\[
f_0 = \frac{1}{T_0} = \frac{1}{2\pi\sqrt{LC}} .
\]
Dimensionele controle: $\qty{1}{H} = \qty{1}{V.s/A}$ en
$\qty{1}{F} = \qty{1}{C/V}$, dus $LC$ draagt
$\unit{s} \times \unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ is een
tijd, zoals het moet.
\end{definition}

\begin{example}[Eerste getallen]\label{ex:g12:rlc-oscillations:firstnumbers}
$L = \qty{10}{mH}$ met $C = \qty{1.0}{\micro F}$:
$\sqrt{LC} = \qty{1.0e-4}{s}$, dus $T_0 = \qty{0.63}{ms}$ en
$f_0 \approx \qty{1.6}{kHz}$ --- een hoorbare toon. $L = \qty{1.0}{mH}$
met $C = \qty{100}{pF}$: $T_0 = \qty{2.0}{\micro s}$,
$f_0 \approx \qty{0.50}{MHz}$ --- een radio\omterm{def:g12:oscillators-and-time:oscillator}{frequentie}.
Alleen het \emph{product} $LC$ zet het tempo.
\end{example}

\begin{remark}[Een kwart periode uit de pas]\label{rem:g12:rlc-oscillations:quarter}
De stroom is een kwart \omterm{def:g12:oscillators-and-time:oscillator}{periode} uit de
\omterm{def:g12:oscillators-and-time:phase}{fase} met de lading: $i = 0$ wanneer
$q = \pm Q_0$ (platen vol, stroom keert om) en $i = \pm I_0$ wanneer
$q = 0$ (platen leeg, stroom op volle kracht) --- precies de slinger,
stil in de uitersten en het snelst onderaan.
\end{remark}

\section{De energiewals}

\begin{proposition}[Behoud van de opgeslagen energie]\label{prop:g12:rlc-oscillations:energy}
\index{energie!van een LC-circuit}
In het ideale \omterm{prop:g12:rlc-oscillations:equation}{LC-circuit} is de totale
opgeslagen \omterm{def:g10:energy-conservation:energy}{energie}
(\cref{prop:g12:rc-rl-circuits:energy})
\[
E \;=\; E_C + E_L \;=\; \frac{q^2}{2C} + \tfrac12\,L i^2
\;=\; \frac{Q_0^2}{2C} \;=\; \tfrac12\,L I_0^2
\]
constant: geheel in de \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} wanneer
$q = \pm Q_0$, geheel in de \omterm{def:g11:magnetic-fields:solenoid}{spoel} wanneer
$i = \pm I_0$.
\end{proposition}

\begin{proof}
Differentiëren: $\frac{dE}{dt} = \frac{q}{C}\frac{dq}{dt} +
Li\frac{di}{dt} = i\left(\frac{q}{C} + L\frac{d^2q}{dt^2}\right) = 0$
door \cref{prop:g12:rlc-oscillations:equation}. De waarde lees je bij
$t = 0$ (alles \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}), en opnieuw op het
moment van pure stroom $q = 0$ --- waaruit
$\tfrac12 L I_0^2 = Q_0^2/2C$, en $I_0 = Q_0/\sqrt{LC}$ alleen uit
\omterm{def:g10:energy-conservation:energy}{energie}.
\end{proof}

\omterm{def:g10:energy-conservation:energy}{energie} slaat van platen naar
\omterm{def:g11:magnetic-fields:solenoid}{spoel} en terug, tweemaal per
\omterm{def:g12:oscillators-and-time:oscillator}{periode}, de som vastgezet op $Q_0^2/2C$
--- precies de figuur van de
\omterm{prop:g12:oscillators-and-time:spring-period}{massa--veer-oscillator}, herlabeld.}
\end{omfigure}

\begin{example}[De wals, gemeten]\label{ex:g12:rlc-oscillations:waltz}
$C = \qty{1.0}{\micro F}$ geladen tot $U_0 = \qty{10}{V}$, geschakeld
op $L = \qty{10}{mH}$: \omterm{def:g10:energy-conservation:energy}{energie}
$E = \tfrac12 C U_0^2 = \qty{5.0e-5}{J}$,
\omterm{def:g12:oscillators-and-time:oscillator}{periode} $T_0 = \qty{0.63}{ms}$, en een
piekstroom $I_0 = U_0\sqrt{C/L} = \qty{0.10}{A}$ bereikt een kwart
\omterm{def:g12:oscillators-and-time:oscillator}{periode} (\qty{0.16}{ms}) na het sluiten.
\end{example}

\section{Echte circuits: gedempte oscillaties}

Elke echte \omterm{def:g11:magnetic-fields:solenoid}{spoel} is gewikkeld van
\omterm{def:g11:work-of-force:sign}{weerstands}draad: de eerlijke lus is een serie\emph{RLC}-circuit,
en de \omterm{def:g11:circuits-and-power:resistance}{weerstand} belast de wals.

\begin{definition}[Pseudoperiodieke en aperiodieke regimes]\label{def:g12:rlc-oscillations:regimes}
\index{RLC-circuit}\index{pseudoperiodiek regime}\index{aperiodiek regime}
In een serie-RLC-circuit dissocieert de weerstand
\omterm{def:g10:energy-conservation:energy}{energie} door Joule-verwarming
(\cref{ch:g11:circuits-and-power}): de
\omterm{def:g12:oscillators-and-time:oscillator}{vrije oscillaties} zijn
\emph{\omterm{def:g12:oscillators-and-time:damping}{gedempt}}. Twee regimes, afgelezen op een
\omterm{def:g10:signals-and-waves:oscilloscope}{oscilloscoop}:
\begin{itemize}
  \item \emph{\omterm{def:g10:signals-and-waves:period}{pseudoperiodiek}} (kleine $R$): het
        circuit oscilleert nog binnen een krimpende envelop, met een
        herhaaltijd --- de
        \emph{pseudoperiode}\index{pseudoperiode!elektrisch} --- dicht
        bij $T_0 = 2\pi\sqrt{LC}$ wanneer de
        \omterm{def:g12:oscillators-and-time:damping}{demping} licht is;
  \item \emph{aperiodiek} (grote $R$): de lading kruipt terug naar nul
        zonder ooit voorbij te schieten --- helemaal geen oscillatie.
\end{itemize}
De exacte spiegel van de
\omterm{def:g12:oscillators-and-time:damping}{gedempte}
\omterm{def:g12:oscillators-and-time:oscillator}{mechanische oscillator}
(\cref{def:g12:oscillators-and-time:damping}).
\end{definition}

\omterm{def:g12:rlc-oscillations:regimes}{RLC-circuit}: oscillaties die sterven binnen een
exponentiële envelop (gestippeld), of een kruip naar nul zonder
oscillatie. Licht \omterm{def:g12:oscillators-and-time:damping}{gedempt} blijft de
\omterm{def:g12:rlc-oscillations:regimes}{pseudoperiode} $\approx T_0$.}
\end{omfigure}

\begin{remark}[Oscillaties in stand houden]\label{rem:g12:rlc-oscillations:maintain}
\index{onderhouden oscillaties}
Om de \omterm{def:g12:oscillators-and-time:oscillator}{amplitude} levend te houden, voed het
circuit precies de \omterm{def:g10:energy-conservation:energy}{energie} die de
\omterm{def:g11:circuits-and-power:resistance}{weerstand} verbrandt, eenmaal per cyclus en
\emph{in de pas} met de zwaai: een versterker doet voor de RLC-lus wat
de duw van het échappement voor de slinger doet
(\cref{ch:g12:oscillators-and-time}). Elk kwartshorloge en elke
radiozender is zo'n onderhouden
\omterm{def:g12:oscillators-and-time:oscillator}{oscillator}, lopend op zijn
\omterm{def:g12:rlc-oscillations:period}{eigenfrequentie} $f_0$.
\end{remark}

\section{De mechanische tweeling}

\begin{proposition}[Het elektrisch--mechanisch woordenboek]\label{prop:g12:rlc-oscillations:analogy}
\index{elektrisch--mechanische analogie}
De LC-vergelijking is de massa--veer-vergelijking
(\cref{prop:g12:oscillators-and-time:spring-period}) woord voor woord:
\begin{center}\small
\begin{tabular}{lcl}
\omterm{prop:g12:rlc-oscillations:equation}{LC-circuit} & & massa--veer-glijder\\[2pt]
lading $q$ & $\longleftrightarrow$ & positie $x$\\
stroom $i = dq/dt$ & $\longleftrightarrow$ & snelheid $v = dx/dt$\\
\omterm{def:g12:rc-rl-circuits:inductor}{inductantie} $L$ & $\longleftrightarrow$ & massa $m$\\
inverse \omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit} $1/C$ & $\longleftrightarrow$ & \omterm{def:g12:oscillators-and-time:spring}{veerconstante} $k$\\
\omterm{def:g11:circuits-and-power:resistance}{weerstand} $R$ & $\longleftrightarrow$ & \omterm{def:g10:inertia:inventory}{wrijving}\\
$L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0$ & $\longleftrightarrow$ &
  $m\,\frac{d^2x}{dt^2} + kx = 0$\\[2pt]
$T_0 = 2\pi\sqrt{LC}$ & $\longleftrightarrow$ &
  $T_0 = 2\pi\sqrt{m/k}$\\[2pt]
$E_C = q^2/2C$ & $\longleftrightarrow$ & $E_p = \tfrac12 kx^2$\\
$E_L = \tfrac12 Li^2$ & $\longleftrightarrow$ & $E_k = \tfrac12 mv^2$
\end{tabular}
\end{center}
Zelfde vergelijking,zelfde oplossingen: elk resultaat over de ene
\omterm{def:g12:oscillators-and-time:oscillator}{oscillator} vertaalt zich onmiddellijk in de
andere taal.
\end{proposition}

\begin{proof}
Vergelijk \cref{prop:g12:rlc-oscillations:equation} met
$m\,d^2x/dt^2 = -kx$: substitutie $q \to x$, $L \to m$, $1/C \to k$
maakt van de ene de andere, en $LC \to m/k$ maakt van
$2\pi\sqrt{LC}$ $2\pi\sqrt{m/k}$.
\end{proof}

\begin{example}[Een radio afstemmen]\label{ex:g12:rlc-oscillations:tuning}
Een antenne voert een fluistering van \emph{elke} zender in een LC-lus,
maar alleen die nabij $f_0$ duwt in de pas en bouwt op ---
\omterm{def:g12:oscillators-and-time:resonance}{resonantie}
(\cref{def:g12:oscillators-and-time:resonance}). De knop draait de platen van
een variabele \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}: $C$ veranderen
verschuift $f_0 = 1/(2\pi\sqrt{LC})$, en de naald overhandigt de
oortelefoon één zender tegelijk. Aan de zenderkant zet een onderhouden
\omterm{def:g12:oscillators-and-time:oscillator}{oscillator}
(\cref{rem:g12:rlc-oscillations:maintain}) de
uitzend\omterm{def:g12:oscillators-and-time:oscillator}{frequentie}; in een horloge speelt
een kwartskristal --- een mechanische tweeling van adembenemende
\omterm{def:g12:oscillators-and-time:spring}{veerconstante} --- dezelfde rol.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:rlc-oscillations:5}

\omterm{def:g12:rlc-oscillations:regimes}{pseudoperiode} zelf zeggen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:rlc-oscillations:11}
Een \omterm{prop:g12:rlc-oscillations:equation}{LC-circuit} heeft $L = \qty{0.50}{H}$ en
$C = \qty{20}{\micro F}$. Bereken $T_0$; vind dan, met het woordenboek
(\cref{prop:g12:rlc-oscillations:analogy}), de
\omterm{def:g12:oscillators-and-time:spring}{veerconstante} $k$ die een glijder van
\qty{0.50}{kg} dezelfde \omterm{def:g12:oscillators-and-time:oscillator}{periode} geeft, en
controleer.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:12}
Een afstemmer gebruikt $L = \qty{0.20}{mH}$ en een variabele
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} die
\qtyrange{50}{500}{pF} veegt. Bereken de twee extreme frequenties. Toon
dat $f_0 \propto 1/\sqrt{C}$, zodat tienmaal $C$ slechts
$\sqrt{10} \approx 3.2$ in
\omterm{def:g12:oscillators-and-time:oscillator}{frequentie} geeft.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:13}
Een echte \omterm{def:g11:magnetic-fields:solenoid}{spoel} laat het circuit $5.0\%$ van
zijn \omterm{def:g10:energy-conservation:energy}{energie} per
\omterm{def:g12:rlc-oscillations:regimes}{pseudoperiode} verliezen. Na hoeveel
\omterm{def:g12:oscillators-and-time:oscillator}{perioden} is de
\omterm{def:g10:energy-conservation:energy}{energie} gehalveerd? Wat is de
\omterm{def:g12:oscillators-and-time:oscillator}{amplitude} dan, als fractie van de start?
Bij $f_0 = \qty{1.0}{MHz}$, hoe lang is dat in microseconden?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:14}
Een horloge-\omterm{def:g12:oscillators-and-time:oscillator}{oscillator} slaat
$E = \qty{1.0e-9}{J}$ op en verliest $2.0\%$ ervan per cyclus bij
$f_0 = \qty{32768}{Hz}$. Welk gemiddeld
\omterm{def:g11:work-of-force:power}{vermogen} moet het onderhoudscircuit leveren?
Waarom moet de duw in de pas van de oscillatie aankomen?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:rlc-oscillations:15}
Na elke flank van een vierkant \omterm{def:g10:signals-and-waves:signal}{signaal}
``rinkelt'' de bedrading van een experimentator op \qty{25}{MHz}; de
parasitaire \omterm{def:g12:rc-rl-circuits:inductor}{inductantie} is ongeveer
\qty{1.0}{\micro H}. Schat de parasitaire
\omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit} die verantwoordelijk is. Welk
regime moet een kleine toegevoegde serieweerstand het circuit induwen,
en waarom geneest dat het rinkelen?
\end{exercise}

\section{Probleem: Een kristalradio bouwen}

\begin{problem}[{Weekendprobleem --- een kristalradio bouwen: één spoel,
één variabele condensator, een diode en een oortelefoon --- geen
batterij, geen versterker, en het avondnieuws arriveert op de energie
van de golf zelf}]
\label{pb:g12:rlc-oscillations:1}
Een kit uit de zolder van een grootouder: een
\omterm{def:g11:magnetic-fields:solenoid}{spoel} $L = \qty{0.20}{mH}$ op een kartonnen
buis, een variabele \omterm{def:g12:rc-rl-circuits:capacitor}{condensator} waarvan de
platen van \qty{20}{pF} tot \qty{480}{pF} draaien, een kristaldiode,
een oortelefoon, en \qty{20}{m} antennedraad. De te vangen zenders
zenden tussen \qty{530}{kHz} en \qty{1700}{kHz}.

\medskip
\noindent\textbf{Deel I --- De afstemlus.}
\begin{enumerate}
  \item De \omterm{def:g12:rc-rl-circuits:capacitor}{condensator}, geladen, mag in de
        \omterm{def:g11:magnetic-fields:solenoid}{spoel} ontladen: schrijf de luswet en
        zet haar om in de differentiaalvergelijking voor $q(t)$.
  \item Verifieer door substitutie dat $q(t) = Q_0\cos(2\pi t/T_0)$
        haar oplost, mits $T_0 = 2\pi\sqrt{LC}$.
  \item Controleer met dimensionele analyse dat $\sqrt{LC}$ een tijd
        is.
  \item Knop op $C = \qty{330}{pF}$: bereken $T_0$ en $f_0$. Is de
        lus binnen de band afgestemd?
  \item Vind de \omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit} die precies op
        \qty{1.00}{MHz} afstemt.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De variabele
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator}.}
\begin{enumerate}[resume]
  \item Bereken de \omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit} nodig voor
        \qty{530}{kHz}.
  \item Bereken de \omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit} nodig voor
        \qty{1700}{kHz}.
  \item Bereken de twee frequenties die de
        \qtyrange{20}{480}{pF}-\omterm{def:g12:rc-rl-circuits:capacitor}{condensator}
        van de kit echt bereikt. Dekt hij de hele band?
  \item Toon dat $f_0 \propto 1/\sqrt{C}$ bij vaste $L$; met welke
        factor moet $C$ veranderen om $f_0$ te verdubbelen?
  \item De platen openen lineair, dus $C$ daalt gestaag als de knop
        draait. Leg uit waarom de zenders dan samendringen aan één
        uiteinde van de schaal --- en welk uiteinde.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- \omterm{def:g10:energy-conservation:energy}{Energie} en
\omterm{def:g12:oscillators-and-time:damping}{demping}.}
Afgestemd op \qty{1.00}{MHz} ($C = \qty{127}{pF}$) laadt de antenne de
\omterm{def:g12:rc-rl-circuits:capacitor}{condensator} momentaan tot
$U_0 = \qty{20}{mV}$.
\begin{enumerate}[resume]
  \item Bereken de lading $Q_0$ en de opgeslagen
        \omterm{def:g10:energy-conservation:energy}{energie} $E$.
  \item Bereken de piekstroom $I_0$, via
        \omterm{def:g10:energy-conservation:energy}{energie}behoud.
  \item De \omterm{def:g11:circuits-and-power:resistance}{weerstand} van de
        \omterm{def:g11:magnetic-fields:solenoid}{spoel} dissocieert $4.0\%$ van de
        opgeslagen \omterm{def:g10:energy-conservation:energy}{energie} per cyclus: na
        hoeveel cycli is de \omterm{def:g10:energy-conservation:energy}{energie}
        gehalveerd?
  \item Hoe lang is dat in microseconden? Waarom kan dit ``rinkelen''
        alleen het avondnieuws niet spelen?
  \item Leg uit waarom de aankomende
        \omterm{def:g10:signals-and-waves:wave}{golf} de lus alleen laat zwaaien
        wanneer afgestemd op die zender --- en wat de oscillatie bij
        de \emph{zender} in stand houdt
        (\cref{rem:g12:rlc-oscillations:maintain}).
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De mechanische tweeling.}
\begin{enumerate}[resume]
  \item Met het woordenboek (\cref{prop:g12:rlc-oscillations:analogy})
        en een glijder van massa $m = \qty{0.20}{kg}$, bereken de
        \omterm{def:g12:oscillators-and-time:spring}{veerconstante} $k$ van de veer die bij
        het afgestemde circuit past.
  \item Een stugge auto-veer heeft $k \approx \qty{5e4}{N/m}$.
        Commentaar: kan een tafel-massa--veer-systeem op megahertz
        oscilleren?
  \item Houd in plaats daarvan een redelijke $k = \qty{100}{N/m}$:
        welke massa oscilleert op \qty{1.00}{MHz}?
  \item Welk echt object is zo'n veergewicht, ultra-stugge
        \omterm{def:g12:oscillators-and-time:oscillator}{mechanische oscillator}, en waar
        ontmoette je hem eerder dit jaar
        (\cref{ch:g12:oscillators-and-time})?
  \item Spec-sheet, één zin: de band die je radio veegt, de
        \omterm{def:g12:rc-rl-circuits:capacitor}{capaciteit} die \qty{1.00}{MHz} kiest,
        hoe lang een eenzame ring duurt, en wat het nieuws toch
        speelt.
\end{enumerate}
\end{problem}
