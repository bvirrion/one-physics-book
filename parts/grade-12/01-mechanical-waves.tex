\chapter{Mechanical Waves}\label{ch:g12:mechanical-waves}

Saturday evening, a full stadium: a ripple of standing spectators
sweeps around the stands in under a minute, yet nobody leaves their
seat. A pebble's ring crosses the pond while the cork only bobs. This
chapter studies the traveler itself --- the \omterm{def:g10:signals-and-waves:wave}{wave} --- and the four
numbers that tame it: speed, \omterm{def:g12:mechanical-waves:delay}{delay}, \omterm{def:g10:signals-and-waves:period}{period}, \omterm{def:g12:mechanical-waves:wavelength}{wavelength} --- enough to
read an earthquake off a strip of paper.

\section{The disturbance travels, the medium stays}

\begin{definition}[Mechanical wave]\label{def:g12:mechanical-waves:wave}
A \emph{mechanical wave}\index{mechanical wave} is a disturbance
propagating through a material medium --- rope, water, air, rock ---
carrying \omterm{def:g10:energy-conservation:energy}{energy} \emph{without transporting matter}: each piece of the
medium swings briefly in place, hands the motion on, and settles back.
\end{definition}

\begin{example}[No matter transport]\label{ex:g12:mechanical-waves:transport}
In the stadium \omterm{def:g10:signals-and-waves:wave}{wave} the ``medium'' is the crowd: what travels is the
act of standing up, while every spectator keeps their seat. On the
pond, the ring expands but a floating cork only bobs in place.
\end{example}

\begin{definition}[Transverse and longitudinal waves]\label{def:g12:mechanical-waves:transverse}
A \omterm{def:g10:signals-and-waves:wave}{wave} is \emph{transverse}\index{transverse wave} when the medium
moves \emph{perpendicular} to the propagation: a hump races along a
flicked rope while each point moves only up and down. It is
\emph{longitudinal}\index{longitudinal wave} when the medium moves
\emph{along} the propagation: a zone of squeezed \omterm{def:g11:magnetic-fields:solenoid}{coils} travels down a
pushed spring, each \omterm{def:g11:magnetic-fields:solenoid}{coil} swinging in place.
\end{definition}

\begin{remark}[Sound is longitudinal]\label{rem:g12:mechanical-waves:sound}
\omterm{def:g10:signals-and-waves:sound}{Sound} is the \omterm{def:g12:mechanical-waves:transverse}{longitudinal wave} par excellence: compressed air ---
higher \omterm{def:g10:pressure:pressure}{pressure} (\cref{ch:g10:pressure}) --- travels from source to
eardrum; \cref{ch:g12:sound-acoustics} is devoted to it. Earthquakes
send both kinds: P~\omterm{def:g10:signals-and-waves:wave}{waves} compress the rock, S~\omterm{def:g10:signals-and-waves:wave}{waves} shake it sideways.
\end{remark}

\section{Speed and delay}

\begin{definition}[Propagation speed]\label{def:g12:mechanical-waves:speed}
The \emph{propagation speed}\index{propagation speed} of a \omterm{def:g10:signals-and-waves:wave}{wave} is the
distance covered by the disturbance divided by the time taken:
$v = d/t$. It is the speed of the traveling \emph{shape}, not of the
medium.
\end{definition}

\begin{proposition}[The medium sets the speed]\label{prop:g12:mechanical-waves:medium}
For small disturbances, the \omterm{def:g12:mechanical-waves:speed}{propagation speed} depends only on the
\emph{medium}, not on the shape or size of the disturbance. On a
stretched rope, $v$ increases with \omterm{def:g10:inertia:inventory}{tension} and decreases with mass per
\omterm{def:g10:orders-of-magnitude:unit}{metre}: taut and light is fast, slack and heavy is slow. \admitted
\end{proposition}

\begin{remark}[Where the formula lives]\label{rem:g12:mechanical-waves:ropeformula}
The rope's exact speed formula is derived in the Year 1 volume; this
year the experimental facts above suffice.
\end{remark}

\begin{example}[Sound speeds]\label{ex:g12:mechanical-waves:speeds}
\omterm{def:g10:signals-and-waves:sound}{Sound} travels at \qty{340}{m/s} in air, \qty{1500}{m/s} in water,
\qty{5900}{m/s} in steel --- the stiffer, the faster. Hence the
western-movie ear on the rail: steel announces the train first.
\end{example}

\begin{definition}[Delay]\label{def:g12:mechanical-waves:delay}
A \omterm{def:g10:signals-and-waves:wave}{wave} of speed $v$ reaches a point at distance $d$ after the
\emph{delay}\index{delay} $\tau = d/v$: two points a distance $d$
apart repeat the same motion, the farther lagging by $\tau$ --- the
\omterm{rem:g10:signals-and-waves:honest}{echo} ranging of \cref{ch:g10:signals-and-waves}, read backwards.
\end{definition}

\begin{proposition}[Superposition without interaction]\label{prop:g12:mechanical-waves:superposition}
Where two \omterm{def:g10:signals-and-waves:wave}{waves} overlap, the displacement of the medium is at every
instant the \emph{sum} of the displacements each would cause alone;
after crossing, each \omterm{def:g10:signals-and-waves:wave}{wave} continues unchanged --- derived in the
Year~2 volume from the linearity of the equations of motion. \admitted
\end{proposition}

\begin{example}[Crossing pulses]\label{ex:g12:mechanical-waves:crossing}
Send an upward hump from each end of a rope: where they meet, the
rope briefly rises to the sum of the two heights; then two intact
humps emerge --- as two crossing conversations reach their listeners.
\end{example}

\section{Periodic waves: the double periodicity}

\begin{definition}[Periodic wave]\label{def:g12:mechanical-waves:periodic}
When the source repeats its motion with \omterm{def:g10:signals-and-waves:period}{period} $T$ and \omterm{def:g10:signals-and-waves:frequency}{frequency}
$f = 1/T$ (\cref{ch:g10:signals-and-waves}), it feeds the medium a
\emph{periodic wave}\index{periodic wave}: every point repeats its own
motion with \omterm{def:g10:signals-and-waves:period}{period} $T$, each lagging its neighbors by the travel \omterm{def:g12:mechanical-waves:delay}{delay}.
\end{definition}

\begin{definition}[Wavelength]\label{def:g12:mechanical-waves:wavelength}
A snapshot of a \omterm{def:g12:mechanical-waves:periodic}{periodic wave} shows a pattern repeating in space; its
spatial \omterm{def:g10:signals-and-waves:period}{period} --- crest to crest --- is the
\emph{wavelength}\index{wavelength} $\lambda$, in \omterm{def:g10:orders-of-magnitude:unit}{metres}. A \omterm{def:g12:mechanical-waves:periodic}{periodic
wave} is \emph{doubly periodic}\index{double periodicity}: \omterm{def:g10:signals-and-waves:period}{period} $T$
in time at each point, \omterm{def:g10:signals-and-waves:period}{period} $\lambda$ in space at each instant.
\end{definition}

\begin{theorem}[The wave relation]\label{thm:g12:mechanical-waves:relation}
A \omterm{def:g12:mechanical-waves:periodic}{periodic wave} of speed $v$, \omterm{def:g10:signals-and-waves:period}{period} $T$ and \omterm{def:g10:signals-and-waves:frequency}{frequency} $f$ has
\omterm{def:g12:mechanical-waves:wavelength}{wavelength} $\lambda = v\,T = v/f$.
\end{theorem}

\begin{proof}
During one \omterm{def:g10:signals-and-waves:period}{period} the pattern slides forward by $v\,T$; but the source
--- hence every point --- is then back in its starting state, so the
slid pattern coincides with the original: the repeat distance is $v\,T$.
\end{proof}

\begin{remark}[The frequency belongs to the source]\label{rem:g12:mechanical-waves:source}
Passing from one medium into another --- \omterm{def:g10:signals-and-waves:sound}{sound} from air into water ---
a \omterm{def:g10:signals-and-waves:wave}{wave} keeps its \omterm{def:g10:signals-and-waves:frequency}{frequency}, which the source dictates, while its speed
changes; by $\lambda = v/f$ the \omterm{def:g12:mechanical-waves:wavelength}{wavelength} changes with it.
\end{remark}

\begin{example}[Concert A, in air and underwater]\label{ex:g12:mechanical-waves:concertA}
The \qty{440}{Hz} concert A of \cref{ch:g10:signals-and-waves} has
\omterm{def:g12:mechanical-waves:wavelength}{wavelength} $340/440 = \qty{0.77}{m}$ in air and $1500/440 =
\qty{3.4}{m}$ in water: same note, stretched by the faster medium.
\end{example}

\begin{definition}[In phase, in phase opposition]\label{def:g12:mechanical-waves:phase}
Two points separated by a whole number of \omterm{def:g12:mechanical-waves:wavelength}{wavelengths}, $d = k\lambda$,
move identically at every instant: they are
\emph{in phase}\index{in phase}. Points separated by
$d = (k + \tfrac12)\lambda$ always move oppositely --- crest against
trough: they are in \emph{phase opposition}\index{phase opposition}.
\end{definition}

\begin{method}[Reading the double periodicity]\label{met:g12:mechanical-waves:reading}
\begin{enumerate}
  \item On a recording $y(t)$ at a fixed point, read $T$ crest to
        crest; $f = 1/T$.
  \item On a snapshot $y(x)$ at a fixed instant, read $\lambda$ crest
        to crest.
  \item Deduce $v = \lambda/T = \lambda f$; check against a direct
        timing $v = d/\tau$ when available.
  \item Never confuse the two graphs: one axis carries seconds, the
        other \omterm{def:g10:orders-of-magnitude:unit}{metres}.
\end{enumerate}
\end{method}

\section{Wavefronts and attenuation}

\begin{definition}[Wavefront]\label{def:g12:mechanical-waves:wavefront}
A \emph{wavefront}\index{wavefront} is a line (or surface) of points
that the \omterm{def:g10:signals-and-waves:wave}{wave} reaches at the same instant --- a crest line on the
pond. From a point source the wavefronts are expanding circles; far
from any source they are nearly straight. The \omterm{def:g10:signals-and-waves:wave}{wave} propagates
perpendicular to its wavefronts, successive crests $\lambda$ apart.
\end{definition}

\begin{definition}[Attenuation]\label{def:g12:mechanical-waves:attenuation}
The decrease of a \omterm{def:g10:signals-and-waves:wave}{wave}'s \omterm{def:g10:signals-and-waves:amplitude}{amplitude} as it propagates is its
\emph{attenuation}\index{attenuation}. Two causes add up: the \omterm{def:g10:energy-conservation:energy}{energy}
of a circular \omterm{def:g12:mechanical-waves:wavefront}{wavefront} is shared over an ever-longer crest, and the
medium absorbs a little in passing. Straight \omterm{def:g12:mechanical-waves:wavefront}{wavefronts} escape the
first cause --- how a distant storm's swell can cross an ocean.
\end{definition}

\section{Problem: Reading a Seismogram}

\begin{problem}[{Weekend problem --- reading a seismogram: two waves
race out of a broken fault, three circles pin the epicenter, and a
slow third wave gives a coastline its warning}]
\label{pb:g12:mechanical-waves:1}
Night shift at a seismological observatory. At 09:14:32 the needle of
station A jumps: sharp fast wiggles, then slower, wider swings. An
earthquake has broken rock somewhere; two \omterm{def:g10:signals-and-waves:wave}{waves} race out through the
crust: P \omterm{def:g10:signals-and-waves:wave}{waves}, compressions at $v_P = \qty{6.0}{km/s}$, and S \omterm{def:g10:signals-and-waves:wave}{waves},
a sideways shaking at $v_S = \qty{3.5}{km/s}$.

\medskip
\noindent\textbf{Part I --- Two \omterm{def:g10:signals-and-waves:wave}{waves}, one clock.}
\begin{enumerate}
  \item From the descriptions above, classify P and S \omterm{def:g10:signals-and-waves:wave}{waves} as
        \omterm{def:g12:mechanical-waves:transverse}{transverse} or \omterm{def:g12:mechanical-waves:transverse}{longitudinal}.
  \item Which \omterm{def:g10:signals-and-waves:wave}{wave} arrives first at every station? Compute the travel
        times of both over $d = \qty{120}{km}$.
  \item Show that the P--S arrival gap is
        $\Delta t = d\,(1/v_S - 1/v_P)$ and compute it for
        $d = \qty{120}{km}$.
  \item Deduce the seismologist's ranging rule
        $d = \dfrac{v_P\,v_S}{v_P - v_S}\,\Delta t$ and show the
        coefficient is about \qty{8.4}{km} per second of gap.
  \item Why does the gap grow with distance --- and why is that a gift
        to a station that never saw the quake start?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Circles on the map.}
\begin{enumerate}[resume]
  \item Station A measures $\Delta t_A = \qty{25}{s}$. How far is the
        epicenter from A?
  \item Explain why this one reading places the epicenter on a circle,
        not at a point.
  \item Stations B and C measure $\Delta t_B = \qty{15}{s}$ and
        $\Delta t_C = \qty{32}{s}$: their distances to the epicenter?
  \item Explain how the three circles together pin the epicenter down.
  \item The P \omterm{def:g10:signals-and-waves:wave}{wave} reached A at 09:14:32. Compute its travel time from
        the epicenter, and the origin time of the quake.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g10:signals-and-waves:wave}{wave} that crosses the ocean.}
The three circles meet offshore: the fault broke under the sea floor
and set the water column moving. Long \omterm{def:g10:signals-and-waves:wave}{waves} on deep water travel at a
speed set by the depth --- about \qty{200}{m/s} in the open ocean,
about \qty{10}{m/s} near a beach (admitted here; the formula is
derived in the Year 1 volume). A port lies $D = \qty{600}{km}$ from
the epicenter.
\begin{enumerate}[resume]
  \item Convert \qty{200}{m/s} to \unit{km/h} and name a vehicle with
        a comparable cruising speed.
  \item How long does the tsunami take to reach the port?
  \item The port's own seismometer feels the P \omterm{def:g10:signals-and-waves:wave}{wave} first. After how
        long? If the alarm \omterm{def:g10:signals-and-waves:sound}{sounds} at that instant, how much warning
        time does the coast get before the tsunami?
  \item At sea the tsunami's \omterm{def:g10:signals-and-waves:period}{period} is about \qty{15}{min}. Compute its
        \omterm{def:g12:mechanical-waves:wavelength}{wavelength} there, and explain why a ship does not notice it.
  \item Nearing the beach the speed falls to \qty{10}{m/s} while the
        \omterm{def:g10:signals-and-waves:period}{period} stays fixed. Compute the new \omterm{def:g12:mechanical-waves:wavelength}{wavelength}. What must
        happen to the \omterm{def:g10:signals-and-waves:wave}{wave} as it bunches up?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{def:g10:energy-conservation:energy}{Energy}, and the report.}
A strong quake releases on the order of \qty{1e15}{J}, carried away by
the \omterm{def:g10:signals-and-waves:wave}{waves}.
\begin{enumerate}[resume]
  \item Compute the length of a circular \omterm{def:g12:mechanical-waves:wavefront}{wavefront} \qty{10}{km} from
        the epicenter, then \qty{210}{km}; by what factor has it grown?
  \item Deduce why shaking weakens with distance even in a lossless
        crust --- and name the second cause a real crust adds.
  \item A liquid cannot resist shear. Far-side stations record the P
        \omterm{def:g10:signals-and-waves:wave}{wave} but no S \omterm{def:g10:signals-and-waves:wave}{wave}: what does that reveal about Earth's core?
  \item The high-speed train of the previous year's \omterm{def:g10:energy-conservation:energy}{energy} chapter
        (\cref{ch:g11:mechanical-energy}) carries about \qty{1.5e9}{J}
        of \omterm{def:g11:mechanical-energy:kinetic}{kinetic energy}: how many such trains is \qty{1e15}{J} worth?
  \item The report, in two sentences: where and when was the quake,
        and how much warning did the port get? Give the three numbers.
\end{enumerate}
\end{problem}
