\chapter{न्यूटन के नियम}\label{ch:g12:newtons-laws}

बाथरूम स्केल को लिफ्ट में ले जाएं। जैसे ही केबिन ऊपर की ओर खींचता है सुई पढ़ती है
तीन \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} भारी; शीर्ष के पास यह प्रकाश पढ़ता है; दोनों के बीच, स्पष्ट सत्य
--- \omterm{def:g12:newtons-laws:momentum}{गति} के अलावा जहाज पर कुछ भी नहीं बदला। यह अध्याय पीछे के तीन कानूनों को बताता है
वह सुई और उन्हें उस विधि में ढालती है जो सभी यांत्रिकी को चलाती है: एक चुनें
सिस्टम, उसका चित्र बनाएं बल, परियोजना $\sum \vect F = m\vect a $.

\section{संवेग, और पहला कानून सटीक बना}

\begin{definition}[गति]\label{def:g12:newtons-laws:momentum}
 \emph{गति}\index{गति} द्रव्यमान के एक पिंड का $ m $ (\unit{kg}) के साथ घूमना
वेग $\vect v $ वेक्टर है
\[
\vect p = m \vect v \qquad (\unit{kg.m/s}):
\]
द्रव्यमान द्वारा भारित वेग; किसी सिस्टम के लिए, उसके भागों पर वेक्टर योग।
\end{definition}

\begin{definition}[जड़त्वीय संदर्भ फ्रेम]\label{def:g12:newtons-laws:inertial}
एक \emph{जड़त्वीय संदर्भ फ्रेम}\index{जड़त्वीय संदर्भ फ्रेम} जिसमें से एक है
प्रत्येक पृथक शरीर --- नहीं बल, या बल वह \omterm{def:g10:inertia:compensate}{मुआवजा} ---सीधे चलता है
स्थिर वेग पर रेखा (\cref{ch:g10:inertia}). \omterm{def:g11:electric-gravitational-fields:field}{मैदान} एक से बढ़कर एक उत्कृष्ट है
सटीकता, और इसी प्रकार कोई भी केबिन इसके ऊपर एकसमान सीधी-रेखा \omterm{def:g12:newtons-laws:momentum}{गति} में है; एक ब्रेक लगाना
ट्रेन या टर्निंग हिंडोला नहीं है: वहां, ढीला सामान बिना धकेले तेजी से बढ़ता है।
\end{definition}

\begin{theorem}[न्यूटन का पहला नियम]\label{thm:g12:newtons-laws:first}\index{न्यूटन का पहला नियम}
एक जड़त्वीय फ्रेम में, \omterm{def:g12:newtons-laws:momentum}{गति} एक पृथक प्रणाली का परिवर्तन नहीं होता है: $\vect p $
स्थिर है, अर्थात\ $ d\vect p/dt = \vect 0$.
\end{theorem}
\admitted

\begin{remark}[जड़ता, दो बार तेज]\label{rem:g12:newtons-laws:sharpened}
की जड़ता का सिद्धांत \cref{ch:g10:inertia}, दो बार अपग्रेड किया गया: यह कहता है
\emph{कहाँ} यह धारण करता है --- यह जड़त्वीय फ़्रेमों को उन लोगों के रूप में परिभाषित करता है जहां यह करता है ---
और यह एक वेक्टर की रक्षा करता है, $ m\vect v $, जिसका बहीखाता अध्याय के अंत में भुगतान करता है।
\end{remark}

\section{न्यूटन का दूसरा नियम}

\begin{theorem}[न्यूटन का दूसरा नियम]\label{thm:g12:newtons-laws:second}\index{न्यूटन का दूसरा नियम}
एक जड़त्वीय फ्रेम में, \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} किसी पिंड पर उसके परिवर्तन की दर के बराबर होती है
\omterm{def:g12:newtons-laws:momentum}{गति}; निरंतर द्रव्यमान के लिए, तब से $\vect p = m\vect v $,
\[
\sum \vect F = \frac{d\vect p}{dt} = m\,\frac{d\vect v}{dt} = m \vect a .
\]
\end{theorem}
\admitted

\begin{remark}[प्रकृति का एक नियम और उसकी इकाई]\label{rem:g12:newtons-laws:nature}
इसका कोई प्रमाण नहीं है: यह प्रकृति का नियम है, जिसे तीन प्रयोगों के मुकाबले तौला जाता है
सदियों. पिछले साल (\cref{ch:g11:forces-and-motion}) ने अपनी छाया दी,
$\Delta v \propto F\,\Delta t/m $; व्युत्पन्न (\cref{ch:g12:kinematics-2d}) बनाता है
यह हर पल सटीक है --- इसका बढ़िया प्रिंट वर्ष 1 खंड में प्रतीक्षा कर रहा है। बल एक मिलता है
\omterm{def:g10:orders-of-magnitude:unit}{इकाई}, $\unit{N} = \unit{kg.m/s^2}$; और उसके नीचे एक शरीर \omterm{def:g10:universal-gravitation:weight}{वज़न} अकेले ही पालन करता है
$ m\vect a = m\vect g $: प्रत्येक द्रव्यमान के साथ गिरता है $\vect a = \vect g $, और
$ g = \qty{9.81}{N/kg} = \qty{9.81}{m/s^2}$ --- दो पाठन एक हैं।
\end{remark}

\begin{example}[औसत बल, किसी विवरण की आवश्यकता नहीं]\label{ex:g12:newtons-laws:average}
ए \qty{1300}{kg} कार \omterm{def:g10:relative-motion:frame}{आराम से} जाती है \qty{100}{km/h} (\qty{27.8}{m/s}) में
\qty{8.0}{s}: पल-पल की कहानी जो भी हो, औसत \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} है
$\Delta p/\Delta t = 1300 \times 27.8/8.0 \approx \qty{4.5e3}{N}$ --- \omterm{def:g12:newtons-laws:momentum}{गति} बदल जाता है
ए बल बहीखाता पद्धति से पहले और बाद में प्रश्न पूछें।
\end{example}

\section{न्यूटन का तीसरा नियम}

\begin{theorem}[न्यूटन का तीसरा नियम]\label{thm:g12:newtons-laws:third}\index{न्यूटन का तीसरा नियम}
यदि एक शरीर $ A $ ए लगाता है बल $\vect F_{A \to B}$ एक शरीर पर $ B $, तब $ B $ पर \omterm{def:g10:inertia:inventory}{जोर} देता है
$ A $  बल $\vect F_{B \to A} = -\vect F_{A \to B}$: कार्रवाई की एक ही दिशा, एक ही
परिमाण, विपरीत दिशाएँ --- हर क्षण, \omterm{def:g12:newtons-laws:momentum}{गति}, संपर्क या कुछ भी हो
दूरी पर कार्रवाई.
\end{theorem}
\admitted

\begin{example}[किताब, मेज़ और पृथ्वी]\label{ex:g12:newtons-laws:book}
एक किताब एक मेज पर रखी हुई है: दो बातचीत, दो जोड़े। गुरुत्वाकर्षण: पृथ्वी खींचती है
किताब नीचे ($\vect P $), किताब पृथ्वी को ऊपर खींचती है। संपर्क करें: टेबल धक्का देती है
किताब ऊपर ($\vect N $), किताब मेज को नीचे दबाती है। चलना वही सौदा है:
ज़मीन को पीछे की ओर धकेलें; इसकी प्रतिक्रिया ही एकमात्र है बल तुम्हें आगे बढ़ा रहा है.
\end{example}

\section{विधि: चुनें, ड्रा करें, प्रोजेक्ट करें}

\begin{method}[एक गतिशीलता समस्या का समाधान]\label{met:g12:newtons-laws:method}
\begin{enumerate}
  \item \emph{प्रणाली}: अध्ययन किए गए पिंड (या पिंडों के समूह) का नाम बताइए।
  \item \emph{चौखटा}: \omterm{def:g12:newtons-laws:momentum}{गति} के लिए उपयुक्त एक जड़त्वीय फ्रेम और अक्ष चुनें ---
        ढलान कुल्हाड़ियों, या \omterm{def:g12:kinematics-2d:frenet}{फ़्रेनेट फ़्रेम} एक वक्र के लिए (\cref{ch:g12:kinematics-2d}).
  \item \emph{भंडार}: प्रत्येक की सूची बनाएं बल --- द \omterm{def:g10:universal-gravitation:weight}{वज़न}, तब एक बल प्रति
        संपर्क करें, और कुछ नहीं --- और मुक्त-शरीर आरेख बनाएं: एक बिंदु, एक
        तीर प्रति बल.
  \item \emph{परियोजना}: लिखना $\sum \vect F = m\vect a $, इसे प्रत्येक अक्ष पर प्रक्षेपित करें।
  \item \emph{हल करें और जांचें}: इकाइयां, \omterm{def:g10:signals-and-waves:signal}{संकेत}, सीमित मामले ($\mu \to 0$,
        $\alpha \to 0$, \dots).
\end{enumerate}
\end{method}

\begin{definition}[सामान्य बल और घर्षण]\label{def:g12:newtons-laws:friction}
एक सतह दो घटकों के माध्यम से शरीर पर कार्य करती है: \emph{सामान्य बल}\index{सामान्य बल} $\vect N $, सतह के लंबवत, और \omterm{def:g10:inertia:inventory}{टकराव} $\vect f $, स्पर्शरेखीय.
प्रयोग देता है \omterm{def:g10:inertia:inventory}{टकराव} नियम: जब तक शरीर फिसलता नहीं, स्थिर रहता है \omterm{def:g10:inertia:inventory}{टकराव}
छत तक खुद को समायोजित कर लेता है, $ f_s \leq \mu_s N $; एक बार जब यह फिसल जाता है, \omterm{def:g11:forces-and-motion:friction}{गतिज घर्षण}
है $ f_k = \mu_k N $, फिसलन के विरुद्ध। \emph{के गुणांक \omterm{def:g10:inertia:inventory}{टकराव}}%
\index{घर्षण के गुणांक} $\mu_s $ और $\mu_k $ ($\mu_k \leq \mu_s $) केवल निर्भर है
दो सामग्रियों पर.
\end{definition}

\begin{example}[झुकाव, दो बार]\label{ex:g12:newtons-laws:incline}
एक ब्लॉक कोण की ढलान से नीचे की ओर फिसलता है $\alpha $; कुल्हाड़ियाँ: $ x $ ढलान के नीचे, $ y $
लंबवत. पर $ y $: $ N = mg\cos\alpha $. पर $ x $, घर्षण रहित: $ ma = mg\sin\alpha $,
इसलिए $ a = g\sin\alpha $ --- कोई द्रव्यमान नहीं; पर $\alpha = 30^\circ $, $ a = \qty{4.9}{m/s^2}$.
साथ \omterm{def:g11:forces-and-motion:friction}{गतिज घर्षण}: $ a = g(\sin\alpha - \mu_k\cos\alpha)$; के लिए $\mu_k = 0.20$,
$ a = 9.81\,(0.500 - 0.173) = \qty{3.2}{m/s^2}$. और यह पहले स्थान पर बना रहा
केवल जबकि आवश्यक स्थिर है \omterm{def:g10:inertia:inventory}{टकराव} इसकी छत के नीचे फिट: $\tan\alpha \leq \mu_s $.
\end{example}

\begin{definition}[स्पष्ट वजन]\label{def:g12:newtons-laws:apparent}
 \emph{स्पष्ट वजन}\index{स्पष्ट वजन} एक शरीर का है \omterm{def:g12:newtons-laws:friction}{सामान्य बल}
इसके और इसके समर्थन के बीच --- इसके नीचे का पैमाना क्या पढ़ता है।
\end{definition}

\begin{example}[लिफ्ट]\label{ex:g12:newtons-laws:elevator}
ए \qty{70}{kg} ऊर्ध्वाधर त्वरण वाली लिफ्ट में यात्री एक पैमाने पर \omterm{def:g10:universal-gravitation:weight}{खड़ा} है
$ a_z $ (ऊपर की ओर सकारात्मक): $ N - mg = ma_z $, इसलिए $ N = m(g + a_z)$. पर खींच रहा हूँ
$ a_z = \qty{1.5}{m/s^2}$: \qty{792}{N}, भारी; शीर्ष के पास ब्रेक लगाना: \qty{582}{N},
रोशनी; स्थिर परिभ्रमण: $ mg = \qty{687}{N}$ --- पहला कानून; निर्बाध गिरावट, $ a_z = -g $:
$ N = 0$, पैमाने पर भारहीनता।
\end{example}

\begin{example}[दो ब्लॉक और एक चरखी]\label{ex:g12:newtons-laws:pulley}
एक ब्लॉक $ m_1 = \qty{4.0}{kg}$ एक घर्षण रहित मेज पर एक हल्की रस्सी से बंधा हुआ है
एक लटकते हुए ब्लॉक के लिए एक आदर्श चरखी $ m_2 = \qty{1.0}{kg}$. रस्सी संचारित करती है
वही \omterm{def:g10:inertia:inventory}{तनाव} $ T $ दोनों सिरों पर; ब्लॉक त्वरण का एक परिमाण साझा करते हैं $ a $.
प्रत्येक \omterm{def:g12:newtons-laws:momentum}{गति} के साथ एक \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} नियम: $ m_1 a = T $ और $ m_2 a = m_2 g - T $, इसलिए
$ a = m_2\,g/(m_1 + m_2) = \qty{1.96}{m/s^2}$ और $ T = m_1 a = \qty{7.8}{N}$ ---
तुलना में छोटा $ m_2 g = \qty{9.8}{N}$, जैसा कि इसके लिए होना चाहिए $ m_2$ नीचे की ओर तेजी लाना।
\end{example}

\section{पृथक प्रणालियाँ: संवेग संरक्षित रहता है}

\begin{proposition}[संवेग का संरक्षण]\label{prop:g12:newtons-laws:conservation}\index{संवेग का संरक्षण}
दो शरीरों के लिए जिनका बाह्य बल गायब हो जाना या \omterm{def:g10:inertia:compensate}{मुआवजा}, संपूर्ण \omterm{def:g12:newtons-laws:momentum}{गति}
$\vect p_1 + \vect p_2$ स्थिर है, जो भी हो बल वे एक-दूसरे पर \omterm{def:g10:pressure:pressure}{दबाव} डालते हैं।
\end{proposition}

\begin{proof}
दो दूसरे कानून जोड़ें: $\frac{d}{dt}(\vect p_1 + \vect p_2) = \vect F_{2 \to 1} +
\vect F_{1 \to 2} + \vect F_{\text{ext}}$: आंतरिक जोड़ी तीसरे द्वारा रद्द हो जाती है
कानून, और $\vect F_{\text{ext}} = \vect 0$ परिकल्पना द्वारा.
\end{proof}

\begin{example}[पीछे हटना]\label{ex:g12:newtons-laws:recoil}
ए \qty{60}{kg} चिकनी बर्फ पर आराम करते समय स्केटर फेंकता है \qty{4.0}{kg} दवा की गेंद
क्षैतिज रूप से \qty{6.0}{m/s}. पहले: $\vect p = \vect 0$; बाद में:
$0 = 60\,V + 4.0 \times 6.0$, इसलिए $ V = \qty{-0.40}{m/s}$ --- वह पीछे की ओर बहती है।
किसी बाहरी चीज़ ने उसे धक्का नहीं दिया: गेंद ने धक्का दिया (तीसरा नियम), और किताबें शून्य पर रहीं।
\end{example}

\begin{remark}[कैसे एक रॉकेट किसी भी चीज़ पर दबाव नहीं डालता]\label{rem:g12:newtons-laws:rocket}
रॉकेट द्रव्यमान फेंकने की एक मशीन है। प्रत्येक सेकंड इसके इंजन गैस की एक खेप फेंकते हैं
पि\omterm{def:g11:color-light-sources:cones}{छड़}ा हुआ; गैस पीछे की ओर चली जाती है \omterm{def:g12:newtons-laws:momentum}{गति}, इसलिए रॉकेट को वही लाभ होता है
आगे की राशि --- एक स्थिर \omterm{def:g10:inertia:inventory}{जोर}, मोटे तौर पर प्रति सेकंड निष्कासित द्रव्यमान
निष्कासन \omterm{def:g12:newtons-laws:momentum}{गति}. रॉकेट अपने निकास पर \omterm{def:g10:pressure:pressure}{दबाव} डालता है, हवा पर नहीं: यह \omterm{def:g11:work-of-force:work}{काम} करता है में सर्वश्रेष्ठ
खाली जगह.
\end{remark}

\section{समस्या: केबल कार}

\begin{problem}[{सप्ताहांत समस्या --- केबल कार को प्रमाणित करना: ढलान पर तनाव
आराम की मांग, स्टार्ट-अप का अधिभार, टावर पर राहत, कीमत
एक आपातकालीन रोक --- और केबिन में प्लेकार्ड एक नंबर का वादा कर सकता है}]
\label{pb:g12:newtons-laws:1}
एक पहाड़ी केबल कार --- केबिन प्लस गाड़ी, खाली द्रव्यमान $ M_0 = \qty{2.0e3}{kg}$ ---
एक ट्रैक केबल पर झुका हुआ रोल करता है $\alpha = 30^\circ $, एक ढोना केबल द्वारा खींचा गया
ढलान के समानांतर \omterm{def:g10:inertia:inventory}{तनाव} $\vect T $. रोलिंग \omterm{def:g10:inertia:inventory}{टकराव} नगण्य है; द
केबिन का फर्श पूरे समय क्षैतिज रहता है। केबिन सीटें $25$ औसत के यात्री
द्रव्यमान \qty{80}{kg}; आप प्रमाणन इंजीनियर हैं. जब तक अन्यथा न कहा जाए,
केबिन भरा हुआ है: $ M = \qty{4.0e3}{kg}$.

\medskip
\noindent\textbf{Part I --- At rest on the slope.}
\begin{enumerate}
  \item सिस्टम चुनें; बताएं कि पर्वतीय ढाँचा एक जड़त्व ढाँचे के रूप में क्यों कार्य करेगा;
        तीन सूची बल, निर्देशों के साथ।
  \item कौन सा कानून प्रतीक्षा को नियंत्रित करता है? इसे ढलान के साथ और लंबवत प्रक्षेपित करें
        व्यक्त करना $ T $ और ट्रैक केबल \omterm{def:g12:newtons-laws:friction}{सामान्य बल} $ N $.
  \item गणना करना $ T $ और $ N $.
  \item पुन: परिकलन करें $ T $ खाली केबिन के लिए. क्यों $ T $ कुल द्रव्यमान का आनुपातिक?
  \item हॉल केबल टूट जाती है, ब्रेक बंद हो जाता है: दिखाएँ कि तीव्र त्वरण है
        $ g \sin\alpha $, भरा हुआ या खाली, और इसकी गणना करें।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Start-up.} चरखी पूरे केबिन को \omterm{def:g12:newtons-laws:momentum}{गति} देती है
$ a = \qty{0.60}{m/s^2}$ ढलान पर तब तक ऊपर चढ़ें जब तक वह आगे न बढ़ जाए $ v = \qty{6.0}{m/s}$.
\begin{enumerate}[resume]
  \item व्यक्त करें और गणना करें \omterm{def:g10:inertia:inventory}{तनाव} $ T'$ इस चरण के दौरान.
  \item स्टार्ट अप किस अंश से बढ़ता है \omterm{def:g10:inertia:inventory}{तनाव} विश्राम के समय इसके मूल्य से ऊपर?
  \item चरण कितने समय तक और कितनी दूरी तक चलता है?
  \item एक \qty{80}{kg} यात्री क्षैतिज फर्श पर \omterm{def:g10:universal-gravitation:weight}{खड़ा} है। \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} प्रोजेक्ट करें
        कानून खड़ी उसे व्यक्त करने और गणना करने के लिए \omterm{def:g12:newtons-laws:apparent}{स्पष्ट वजन}; साथ तुलना करें $ mg $.
  \item कौन बल उसे क्षैतिज रूप से तेज़ करता है, और कितना बड़ा? यात्री क्यों झुकते हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Over the tower.} एक स्थिरांक पर \qty{6.0}{m/s}, द
ट्रैक केबल त्रिज्या के ऊर्ध्वाधर गोलाकार चाप के साथ एक समर्थन टावर पर झुकता है
$ R = \qty{40}{m}$; शिखर पर वेग क्षैतिज है।
\begin{enumerate}[resume]
  \item में \omterm{def:g12:kinematics-2d:frenet}{फ़्रेनेट फ़्रेम} शिखर पर (\cref{ch:g12:kinematics-2d}), दोनों दे दो
        त्वरण के घटक, उनके मूल्यों के साथ।
  \item व्यक्त करने और गणना करने के लिए सामान्य के साथ दूसरे नियम को प्रोजेक्ट करें \omterm{def:g12:newtons-laws:friction}{सामान्य
        बल} $ N'$ गाड़ी पर केबल का.
  \item यात्री का भी यही सवाल \omterm{def:g12:newtons-laws:apparent}{स्पष्ट वजन} शिखर पर: किसके द्वारा
        अंश क्या वह हल्की है?
  \item किस \omterm{def:g12:newtons-laws:momentum}{गति} से होगा $ N'$ गायब होना? परिभ्रमण \omterm{def:g12:newtons-laws:momentum}{गति} से तुलना करें और टिप्पणी करें।
  \item क्या बल क्या गाड़ी टावर के ऊपर केबल पर \omterm{def:g10:pressure:pressure}{दबाव} डालती है --- कौन सा कानून
        ऐसा कहता है, किस परिमाण और दिशा से?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Braking, a jump, and the placard.}
\begin{enumerate}[resume]
  \item क्षैतिज आगमन खंड पर, एक आपातकालीन स्टॉप पूरे केबिन को लेता है
        से \qty{6.0}{m/s} आराम करने के लिए \qty{1.5}{s}: पढ़ना $\sum \vect F = d\vect p/dt $
        ब्रेकिंग की गणना करने के लिए स्टॉप पर औसत के रूप में बल.
  \item क्या क्षैतिज बल एक को रोकना होगा \qty{80}{kg} यात्री, और कर सकते हैं
        जूता--फर्श स्थिर \omterm{def:g10:inertia:inventory}{टकराव} ($\mu_s = 0.40$) इसकी आपूर्ति करें? निष्कर्ष.
  \item प्लेटफ़ॉर्म पर, ब्रेक बंद हो जाता है, खाली केबिन मुक्त और स्थिर लटक जाता है; आखिरी
        यात्री (\qty{80}{kg}) क्षैतिज रूप से कूदता है \qty{2.5}{m/s}
        \emph{केबिन के सापेक्ष}. केबिन की पीछे हटने की \omterm{def:g12:newtons-laws:momentum}{गति} और की गणना करें
        जमीन पर यात्री की \omterm{def:g12:newtons-laws:momentum}{गति}.
  \item एक वाक्य में: कौन सा प्रणोदन उपकरण इस पुनरावृत्ति सिद्धांत पर रहता है,
        और इसे क्या ठीक करता है \omterm{def:g10:inertia:inventory}{जोर}?
  \item \emph{प्रमाणन.} हॉल केबल टूट जाती है \qty{100}{kN}; नियम सीमा
        \omterm{def:g11:work-of-force:work}{काम} करने वाला \omterm{def:g10:inertia:inventory}{तनाव} उसके पांचवें हिस्से पर. दिखाएँ सबसे खराब मामला ऊपर मिला है
        स्टार्ट-अप, अधिकतम कुल द्रव्यमान निकालें, और प्लेकार्ड प्रिंट करें: कितने
        \qty{80}{kg} यात्री सवारी कर सकते हैं? सुरक्षा कारक की जाँच करें.
\end{enumerate}
\end{problem}
