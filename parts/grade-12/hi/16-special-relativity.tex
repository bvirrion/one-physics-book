\chapter{विशेष सापेक्षता: समय फैलाव}\label{ch:g12:special-relativity}

हर जीपीएस \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} एक परमाणु घड़ी को तीन में एक सेकंड तक ले जाता है
मिलियन वर्ष --- और लॉन्च से पहले, इंजीनियरों ने जानबूझकर इसे अलग कर दिया: बाएं
सच कहूँ तो, यह प्रतिदिन एक सेकंड का 38 मिलियनवाँ भाग प्राप्त कर लेगा, और कल तक
शाम को आपका फ़ोन आपको जहाँ आप खड़े हैं वहां से ग्यारह किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} दूर ले जाएगा।
घड़ी का कोई दोष नहीं है--- समय स्वयं चलने के लिए अलग-अलग टिक-टिक करता है
अभी भी: यह अध्याय आइंस्टीन को एक जिद्दी तथ्य से उस निष्कर्ष तक ले जाता है।

\section{एक गति जोड़ने से इंकार कर देती है}

\omterm{def:g12:newtons-laws:momentum}{गति} फ़्रेम पर निर्भर करती है और जोड़ द्वारा रचित होती है
(\cref{ch:g10:relative-motion}): एक यात्री चल रहा है \qty{4}{km/h} की ओर
एक ट्रेन के सामने कर रहे हैं \qty{300}{km/h} ट्रैकसाइड पर्यवेक्षक से गुजरता है \qty{304}{km/h}.

\begin{definition}[जड़त्वीय ढाँचा]\label{def:g12:special-relativity:inertial}
एक \emph{जड़त्वीय ढांचा}\index{जड़त्वीय ढांचा} एक है \omterm{def:g10:relative-motion:frame}{संदर्भ फ्रेम}
(\cref{ch:g10:relative-motion}) जिसमें एक बल-मुक्त शरीर एक \omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक रखता है
वेग: जमीन, काफी करीब, और हर फ्रेम एक समान
इसके सापेक्ष सीधी-रेखा \omterm{def:g12:newtons-laws:momentum}{गति}।
\end{definition}

नियम को चरम सीमा तक लागू करें: ट्रेन की हेडलाइट किरण को पार करना चाहिए
ग्राउंड ऑब्जर्वर पर $c + \qty{300}{km/h}$. 1887 में माइकलसन और मॉर्ले
पृथ्वी पर और उसके आर-पार प्रकाश की \omterm{def:g12:newtons-laws:momentum}{गति} की तुलना की \qty{30}{km/s}
कक्षीय भीड़, ऐसे मतभेदों का शिकार; उन्होंने अत्यंत सटीकता से पाया
कोई नहीं --- वही $ c $ हर दिशा में, हर मौसम में. आइंस्टीन, 1905 में,
प्रयोग पर विश्वास किया और दो अभिधारणाओं के आधार पर इसके चारों ओर समय और स्थान का पुनर्निर्माण किया
उनके परिणामों के आधार पर निर्णय लिया गया --- एक भी विफलता के बिना एक सदी से परीक्षण किया जा रहा है।

\begin{theorem}[विशेष सापेक्षता के सिद्धांत]\label{thm:g12:special-relativity:postulates}
\index{विशेष सापेक्षता}\index{सापेक्षता के सिद्धांत}
\begin{enumerate}
  \item \emph{सापेक्षता सिद्धांत}:भौतिकी के नियम वही मानते हैं
        प्रत्येक में फार्म \omterm{def:g12:special-relativity:inertial}{जड़त्वीय ढांचा}; कोई भी ``वास्तव में आराम पर'' नहीं है।
  \item \emph{की अपरिवर्तनशीलता $ c $}: प्रकाश निर्वात में यात्रा करता है
        $ c = \qty{3.00e8}{m/s}$ प्रत्येक \omterm{def:g12:special-relativity:inertial}{जड़त्वीय ढांचा}, जो भी हो
        स्रोत या पर्यवेक्षक की \omterm{def:g12:newtons-laws:momentum}{गति}.
\end{enumerate}
\end{theorem}
\admitted

\section{``उसी पल'' को विदाई}

\begin{proposition}[समकालिकता की सापेक्षता]\label{prop:g12:special-relativity:simultaneity}
\index{एक ही समय में होने की स्थिति}
दो घटनाएँ एक साथ एक में \omterm{def:g12:special-relativity:inertial}{जड़त्वीय ढांचा} सामान्यतः नहीं हैं
दूसरे में एक साथ: ``अब, हर जगह'' फ्रेम-निर्भर है।
\end{proposition}

\begin{proof}
आइंस्टीन की ट्रेन. दो बिजली के बोल्ट एक गतिशील वस्तु के दोनों सिरों पर प्रहार करते हैं
ट्रेन, ट्रैक को झुलसा रही है। मंच पर्यवेक्षक $ M $, बीच में
निशान, दो फ़्लैश एक साथ प्राप्त करता है; प्रत्येक ने समान दूरी पार की
वही \omterm{def:g12:newtons-laws:momentum}{गति} $ c $, इसलिए हड़\omterm{def:g11:lenses-and-eye:magnifier}{ताल}ें एक साथ हुईं। यात्री $ M'$,
ट्रेन के बीच में, सवारी \emph{की ओर} सामने का फ्लैश, जो पहुंचता है
वह पहले; लेकिन उसके फ्रेम में भी दोनों फ्लैश ने समान दूरी (आधी) तय की
एक रेलगाड़ी) समान \omterm{def:g12:newtons-laws:momentum}{गति} से $ c $ --- \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} अभिधारणा! --- तो सामने
सबसे पहले हड़\omterm{def:g11:lenses-and-eye:magnifier}{ताल} हुई. प्रत्येक फैसला अपने-अपने दायरे में सही है; द
असहमति, एक ट्रेन के लिए नैनोसेकंड, यही कारण है कि किसी ने ध्यान नहीं दिया।
\end{proof}

\section{प्रकाश घड़ी और $\gamma $ कारक}

\begin{definition}[घटना, उचित समय]\label{def:g12:special-relativity:proper-time}
एक \emph{आयोजन}\index{आयोजन} एक स्थान और एक क्षण में घटित हो रही घटना है
--- एक टिक, एक फ्लैश, एक क्षय। जब एक ही घड़ी दो घटनाओं में मौजूद होती है,
यह उनके बीच जो अंतराल पढ़ता है वह है
\emph{उचित समय}\index{उचित समय} $\Delta t_0$. एक फ्रेम जिसमें वह
घड़ी की चाल आम तौर पर भिन्न अंतराल को मापती है $\Delta t $.
\end{definition}

\begin{theorem}[समय का फैलाव]\label{thm:g12:special-relativity:dilation}
\index{समय का फैलाव}\index{gamma factor@$\gamma $ (gamma) factor}
एक घड़ी निरंतर \omterm{def:g12:newtons-laws:momentum}{गति} से घूम रही है $ v $ एक के माध्यम से \omterm{def:g12:special-relativity:inertial}{जड़त्वीय ढांचा} मापा जाता है
वहाँ धीमी \omterm{def:g12:newtons-laws:momentum}{गति} से चलने के लिए: दो के बीच घटनाएँ घड़ी पर,
\[
\Delta t = \gamma\, \Delta t_0, \qquad
\gamma = \frac{1}{\sqrt{1 - v^2/c^2}} \geq 1 .
\]
\end{theorem}

\begin{proof}
सबसे सरल कल्पना योग्य घड़ी बनाएं: दूरी पर दो आमने-सामने दर्पण $ L $
अलग, ए फोटोन उनके बीच उछल रहा है; एक राउंड ट्रिप एक टिक है, और पर
बाकी एक टिक रहता है $\Delta t_0 = 2L/c $. अब घड़ी को जहाज पर लगाओ
\omterm{def:g12:newtons-laws:momentum}{गति} से पार करना $ v $, \omterm{def:g12:newtons-laws:momentum}{गति} के लंबवत दर्पण, और समय एक
जमीन से टिक करें: जबकि फोटोन चढ़ता है, दर्पण बग़ल में सरकते हैं,
तो फोटोन एक तिरछे रास्ते पर उड़ता है --- एक साथ दो गतियाँ, जैसे
\omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य} का \cref{ch:g12:kinematics-2d}. अगर टिक टिक जाता है $\Delta t $,
प्रत्येक आधा टिक आगे बढ़ता है $ v\,\Delta t/2$ क्षैतिज रूप से और $ L $ खड़ी; और ---
महत्वपूर्ण कदम --- \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} अभिधारणा ठीक करता है फोटोनकी ज़मीनी \omterm{def:g12:newtons-laws:momentum}{गति} पर
$ c $, इसलिए तिरछा आधा रास्ता उपाय $ c\,\Delta t/2$. पाइथागोरस:
\[
\Bigl(\frac{c\,\Delta t}{2}\Bigr)^{\!2} = L^2 + \Bigl(\frac{v\,\Delta t}{2}\Bigr)^{\!2}
\;\Longrightarrow\;
\Delta t = \frac{2L/c}{\sqrt{1 - v^2/c^2}} = \gamma\,\Delta t_0 .
\]
और उसके बगल में चलने वाली कोई भी घड़ी - कलाई घड़ी, दिल की धड़कन, क्षय \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} ---
कदम में रहना चाहिए, अन्यथा तुलना जहाज की \omterm{def:g12:newtons-laws:momentum}{गति} को धोखा देगी और उल्लंघन करेगी
पहला अभिधारणा.
\end{proof}

\begin{remark}[दैनिक जीवन ने कभी ध्यान क्यों नहीं दिया]\label{rem:g12:special-relativity:everyday}
के लिए $ v \ll c $, लिखना $ x = v^2/c^2$: तब से $(1-x)(1+x) \approx 1$ और
$(1 + x/2)^2 \approx 1 + x $, हम पाते हैं $\gamma \approx 1 + v^2/(2c^2)$. एक कार
पर \qty{130}{km/h} है $\gamma - 1 \approx \num{7e-15}$ (एक सेकंड खो गया
प्रति चार मिलियन वर्षों की ड्राइविंग); एक जेट प्रबंधन करता है $\num{3.5e-13}$, एक
प्रति नब्बे हजार वर्ष में \omterm{def:g10:orders-of-magnitude:unit}{दूसरा}। हमारी घड़ियाँ बहुत ख़राब थीं।
\end{remark}

\section{प्रयोग का फैसला}

\begin{example}[मून्स जमीन पर पहुंच गए]\label{ex:g12:special-relativity:muons}
ब्रह्मांडीय किरणें ऊपरी वायुमंडल से टकराकर सृजन करती हैं \emph{मुओन्स} ---
उचित जीवनकाल के अल्पकालिक कण $\Delta t_0 = \qty{2.2}{\micro s}$
--- के बारे में \qty{15}{km} ऊपर, लगभग \omterm{def:g12:newtons-laws:momentum}{गति} से आगे बढ़ रहा है $ c $. सापेक्षता के बिना एक म्यूऑन
कवर $ c\,\Delta t_0 \approx \qty{660}{m}$ प्रति जीवनकाल: वंश लेता है
कुछ $23$ जीवनकाल, और बचा हुआ अंश
$\mathrm{e}^{-23} \approx \num{e-10}$ समुद्र-स्तरीय म्यूऑन बनाना चाहिए
लुप्त हो जाना दुर्लभ है। फिर भी डिटेक्टर प्रति वर्ग सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} लगभग एक की गिनती करते हैं
मिनट. \omterm{thm:g12:special-relativity:dilation}{समय का फैलाव} इसे हल करता है: साथ पहुंचना $\gamma \approx 30$, द
म्यूऑन रहते हैं $30 \times \qty{2.2}{\micro s} = \qty{66}{\micro s}$ हमारे पर
घड़ियाँ --- \omterm{def:g12:projectile-motion:range}{श्रेणी} $\approx \qty{20}{km}$, पूरा माहौल.
\end{example}

जन्म ऊंचाई से नीचे; फैला हुआ जीवनकाल वायुमंडल तक फैला हुआ है।}
\end{omfigure}

पुष्टियों का अंबार लग गया. 1971 में हाफेल और कीटिंग ने सीज़ियम परमाणु उड़ाया
घड़ियाँ (\cref{ch:g12:oscillators-and-time}) दुनिया भर में एयरलाइनरों पर;
उतरने पर वे अनुमानित अंशों द्वारा जमीनी घड़ियों से भिन्न थे
एक माइक्रोसेकंड. त्वरक प्रतिदिन अपने कणों के फैलाव पर निर्भर रहते हैं
को बढ़ावा दिया गया $\gamma $ हजारों जीवित गोदों में वे कभी नहीं बच सके
शास्त्रीय दृष्टि से पूर्ण. और जीपीएस एक मूल्य टैग के साथ चल रहा प्रयोग है।

\begin{remark}[जीपीएस का सापेक्षता बिल]\label{rem:g12:special-relativity:gps}
एक जीपीएस \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} कक्षाओं लगभग \qty{3.9}{km/s}, इसलिए \omterm{thm:g12:special-relativity:postulates}{विशेष सापेक्षता}
अपनी घड़ी को कुछ धीमा कर देता है \qty{7}{\micro s} प्रति दिन। गुरुत्वाकर्षण, कमजोर
ऊँचाई, \omterm{def:g11:work-of-force:work}{काम} करता है \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} रास्ता और जीतता है: सामान्य सापेक्षता (आइंस्टीन का)।
का सिद्धांत \omterm{def:g11:fundamental-interactions:four}{आकर्षण-शक्ति}, विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} के संस्करणों में बताया गया है) इसे लगभग \omterm{def:g12:newtons-laws:momentum}{गति} देता है
\qty{45}{\micro s} प्रति दिन; जाल, \qty{38}{\micro s} प्रति दिन उपवास.
पोजिशनिंग घड़ी की रीडिंग को परिवर्तित करती है
\omterm{def:g12:newtons-laws:momentum}{गति} से दूरियाँ $ c $: एक गलत दिन स्थिति को खराब कर देगा
$ c \times \qty{38}{\micro s} \approx \qty{11}{km}$ --- इसलिए असंबद्ध
शुरुआती पैराग्राफ की घड़ियाँ।
\end{remark}

\begin{proposition}[लंबाई संकुचन]\label{prop:g12:special-relativity:contraction}
\index{लंबाई संकुचन}
लंबाई का एक शरीर $ L_0$ इसके बाकी फ्रेम को इसके साथ छोटा मापा जाता है
एक फ्रेम में \omterm{def:g12:newtons-laws:momentum}{गति} की दिशा जहां यह \omterm{def:g12:newtons-laws:momentum}{गति} से चलती है $ v $:
$ L = L_0/\gamma $.
\end{proposition}
\admitted

\begin{remark}[म्यूऑन की अपनी कहानी]\label{rem:g12:special-relativity:muonframe}
\omterm{prop:g12:special-relativity:contraction}{लंबाई संकुचन} है समय का फैलावइसका \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} पहलू (ईमानदारी से व्युत्पन्न)
विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} खंड): म्यूऑन की घड़ी सामान्य रूप से अपने स्वयं के फ्रेम में टिक करती है, लेकिन
तेजी से बढ़ रहा वातावरण अनुबंधित है $\qty{15}{km}/30 =
\qty{500}{m}$, अच्छी तरह से भीतर पार कर गया \qty{2.2}{\micro s} --- दोनों फ्रेम
सहमत हूं कि म्यूऑन जमीन पर पहुंच गया है।
\end{remark}

\section{द्रव्यमान, ऊर्जा और शास्त्रीय दुनिया पुनः प्राप्त हो गई}

\begin{theorem}[द्रव्यमान--ऊर्जा तुल्यता]\label{thm:g12:special-relativity:emc2}
\index{द्रव्यमान--ऊर्जा तुल्यता}
द्रव्यमान का एक पिंड $ m $ अकेले अपने द्रव्यमान के कारण, के पास है ऊर्जा
\[ E = m c^2 . \]
द्रव्यमान किसका सांद्रित रूप है? ऊर्जा; ऊर्जा, इसके विपरीत, द्रव्यमान है।
\end{theorem}
\admitted

\begin{remark}[जहां आप उससे मिले हैं]\label{rem:g12:special-relativity:emc2seen}
यह पीछे का बहीखाता है \omterm{def:g10:energy-conservation:forms}{परमाणु ऊर्जा} (\cref{ch:g12:nuclear-energy}):
द \omterm{def:g12:nuclear-energy:defect}{सामूहिक दोष} एक प्रतिक्रिया का, समय $ c^2$, है ऊर्जा जारी किया। एक
द्रव्यमान का ग्राम है \qty{9.0e13}{J} --- पूरे दिन का एक बड़ा आउटपुट \omterm{def:g11:work-of-force:power}{शक्ति}
स्टेशन। ईमानदार व्युत्पत्ति विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} के संस्करणों से संबंधित है; द
रिएक्टर और तारे बिना किसी परवाह के उस पर चलते हैं।
\end{remark}

\begin{remark}[न्यूटन ग़लत नहीं था]\label{rem:g12:special-relativity:newton}
तय करना $ v \ll c $: $\gamma \to 1$, समय सहमत, लंबाई अनियंत्रित,
समकालिकता \omterm{met:g10:pressure:dive}{निरपेक्ष} हो जाती है --- पिछले तीन वर्षों की यांत्रिकी
कम \omterm{def:g12:newtons-laws:momentum}{गति} सीमा के रूप में, बरकरार, पुनः प्रकट होता है। सापेक्षता ध्वस्त नहीं हुई
\omterm{def:g10:inertia:force}{न्यूटन}; इसने उसके साम्राज्य की सीमा खींची। इस प्रकार भौतिकी बढ़ती है: द्वारा
सिद्धांतों को परिष्कृत करना, प्रत्येक नया एक सीमित मामले के रूप में पुराने को वापस सौंपना।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:special-relativity:1}
गणना करना $\gamma $, तीन तक \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग}, के लिए $ v = 0.1c $, $0.6c $, $0.8c $, $0.995c $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:special-relativity:2}
एक जहाज़ यात्रा कर रहा है $0.6c $; इसका मेट्रोनोम हर एक को मात देता है \qty{10.0}{s} जहाज के द्वारा
घड़ी. ज़मीन किस अंतराल को मापती है? वह कौन सा है \omterm{def:g12:special-relativity:proper-time}{उचित समय}, और क्यों?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:special-relativity:3}
किस \omterm{def:g12:newtons-laws:momentum}{गति} से (एक अंश के रूप में) $ c $) है $\gamma = 2$? और $\gamma = 10$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:special-relativity:4}
एक कार चलती है \qty{130}{km/h}. गणना करना $ v/c $, तब $\gamma - 1$
(\cref{rem:g12:special-relativity:everyday}); इसे कितनी देर तक चलाना होगा
यह गिरने की घड़ी है \qty{1}{s} पीछे?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:special-relativity:5}
आइंस्टीन के ट्रेन प्रयोग में प्रयुक्त अभिधारणा का हवाला देते हुए समझाइए ---
यात्री निष्कर्ष क्यों निकालता है? \emph{सामने} हड़\omterm{def:g11:lenses-and-eye:magnifier}{ताल} पहले आई, और क्यों
कोई भी पर्यवेक्षक ग़लत नहीं है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:special-relativity:6}
चार्ज किए गए शेर रहते हैं $\Delta t_0 = \qty{2.6e-8}{s}$. एक किरण निकलती है
त्वरक पर $0.99c $. गणना करना $\gamma $, प्रयोगशाला में जीवनकाल,
शास्त्रीय \omterm{def:g12:projectile-motion:range}{श्रेणी} $ v\,\Delta t_0$, और वास्तविक माध्य \omterm{def:g12:projectile-motion:range}{श्रेणी}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:special-relativity:7}
से $\gamma $ वक्र, \omterm{def:g12:newtons-laws:momentum}{गति} देते हुए पढ़ें $\gamma = 1.5$ और
$\gamma = 3$ (गणना द्वारा जांचें)। ऊर्ध्वाधर अनंतस्पर्शी क्या दर्शाता है
$ v = c $ एक विशाल शरीर को \omterm{def:g12:newtons-laws:momentum}{गति} देने के बारे में कहें?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:special-relativity:8}
एक हल्की घड़ी में दर्पण होते हैं $ L = \qty{1.5}{m}$ अलग। इसके टिक की गणना करें
आराम करो, फिर इसकी टिक को जमीन से मापा जाता है जब यह परिभ्रमण करता है $0.8c $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:special-relativity:9}
कौन मापता है \omterm{def:g12:special-relativity:proper-time}{उचित समय} बीच में: (ए) कलाई घड़ी के दो टिक ---
पहनने वाला, या राहगीर? (बी) म्यूऑन का जन्म और क्षय --- म्यूऑन, या
प्रयोगशाला? (सी) पृथ्वी छोड़कर मंगल ग्रह पर पहुंचना --- पायलट, या मिशन नियंत्रण?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:special-relativity:10}
अंतरिक्ष स्टेशन कक्षाओं पर \qty{7.7}{km/s}. गणना करना
$\gamma - 1 \approx v^2/2c^2$, तो एक अंतरिक्ष यात्री की घड़ी कितनी पीछे है
छह महीने बाद एक ग्राउंड क्लॉक (\qty{1.58e7}{s}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:special-relativity:11}
की गणना करें ऊर्जा सामग्री $ E = mc^2$ का \qty{1.0}{g} पदार्थ का. इसकी तुलना करें
ए के दैनिक आउटपुट के साथ \qty{1.0}{GW} \omterm{def:g11:work-of-force:power}{शक्ति} स्टेशन, और के साथ
\qty{8.5e5}{J} \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} एक कार का \qty{130}{km/h}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:special-relativity:12}
से दिखाएँ $\gamma = 1/\sqrt{1 - v^2/c^2}$ वह $ v = c\sqrt{1 - 1/\gamma^2}$;
की \omterm{def:g12:newtons-laws:momentum}{गति} कम करें $\gamma = 30$ मुओंस, पाँच तक \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} में
इकाइयां का $ c $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:special-relativity:13}
म्यूऑन की यात्रा को उसके अपने फ्रेम से दोबारा बताएं: वातावरण कितना घना है
पर $\gamma = 30$, यह कितनी देर से आगे बढ़ता है $0.99944c $, और वह करता है
एक उचित जीवनकाल के भीतर फिट?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:special-relativity:14}
एक रैखिक त्वरक इलेक्ट्रॉनों को धकेलता है $\gamma = \num{1.0e5}$ साथ में ए
\qty{3.2}{km} नली। प्रयोगशाला में यात्रा में कितना समय लगता है? में कब तक
इलेक्ट्रॉन का फ्रेम, और वहां ट्यूब कितनी ``लंबी'' है?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:special-relativity:15}
प्रॉक्सिमा सेंटौरी झूठ बोलती है $4.2$ \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष} दूर (अ \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष} दूरी है
एक वर्ष में प्रकाश कवर हो जाता है)। एक जांच वहां पर घूम रही है $0.9c $: कब तक करता है
यात्रा पृथ्वी के कैलेंडर और जांच की घड़ी पर करें?
\end{exercise}

\section{समस्या: मून और ट्विन}

\begin{problem}[{सप्ताहांत समस्या --- म्यूऑन की यात्रा और अंतरिक्ष यात्री की
जुड़वां: एक कण जो अपनी घड़ी से अधिक जीवित है, एक हल्की घड़ी फिर से बनी, एक यात्री
अपने जुड़वां बच्चे से छोटी, और बहीखाता जो जीपीएस को ईमानदार रखता है}]
\label{pb:g12:special-relativity:1}
डेटा: म्यूओन उचित जीवनकाल $\Delta t_0 = \qty{2.2}{\micro s}$; म्यून्स का जन्म
के बारे में \qty{15}{km} लगभग ऊपर $ c $; समुद्र-स्तर का प्रवाह लगभग एक म्यूऑन प्रति
\unit{cm^2} प्रति मिनट; $ c = \qty{3.00e8}{m/s}$.

\medskip
\noindent\textbf{Part I --- The muon's impossible arrival.}
\begin{enumerate}
  \item सापेक्षता के बिना, एक म्यूऑन एक जीवनकाल में कितनी दूर तक यात्रा करता है?
  \item कितने जीवनकाल होते हैं \qty{15}{km} उतरो तो ले लो?
  \item बाद $ n $ जीवनकाल एक अंश $\mathrm{e}^{-n}$ जीवित रहता है:
        इसका अनुमान लगाएं, मापे गए प्रवाह का सामना करें, और फैसला सुनाएं
        एक वाक्य में शास्त्रीय भौतिकी।
  \item ये म्यूऑन वास्तव में साथ आते हैं $\gamma \approx 30$. उनकी गणना करें
        ज़मीन से मापा गया जीवनकाल और परिणाम \omterm{def:g12:projectile-motion:range}{श्रेणी}.
  \item का उपयोग करके उनकी \omterm{def:g12:newtons-laws:momentum}{गति} की गणना करें $ v = c\sqrt{1 - 1/\gamma^2}$, पाँच तक
        \omterm{def:g10:orders-of-magnitude:sigfig}{महत्वपूर्ण लोग} में इकाइयां का $ c $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The light clock, rebuilt.}
एक हल्की घड़ी ($ L = \qty{1.5}{m}$) एक जहाज की सवारी करता है $ v = 0.6c $.
\begin{enumerate}[resume]
  \item टिक की गणना करें $\Delta t_0$ बोर्ड पर पढ़ें.
  \item जमीन से देखने पर एक टिक टिक जाती है $\Delta t $: क्षैतिज दें
        अग्रिम प्रति आधा टिक, फोटोनकी ऊर्ध्वाधर चढ़ाई, और --- नामकरण
        वह अभिधारणा जो इसे ठीक करती है --- तिरछी आधी पथ की लंबाई।
  \item अर्ध-टिक त्रिभुज पर पाइथागोरस लागू करें और हल करें
        $\Delta t $, वह दिखा रहा है $\Delta t = \gamma\,\Delta t_0$.
  \item गणना करना $\gamma $ और $\Delta t $ संख्यात्मक रूप से.
  \item कौन सा पर्यवेक्षक मापता है \omterm{def:g12:special-relativity:proper-time}{उचित समय} दो टिकों के बीच? औचित्य सिद्ध करें
        परिभाषा से.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The traveler and her twin.}
एक अंतरिक्ष यात्री अपने जुड़वां बच्चे को एक चक्कर के लिए पृथ्वी पर छोड़ता है $0.8c $; पृथ्वी
पूरी यात्रा के समय की घड़ियाँ \qty{10.0}{years}.
\begin{enumerate}[resume]
  \item गणना करना $\gamma $ पर $0.8c $ (सटीक अंश, फिर दशमलव)।
  \item अंतरिक्ष यात्री की घड़ी में कितना समय बीतता है?
  \item कितनी दूर (पृथ्वी फ्रेम, अंदर) \omterm{def:g10:orders-of-magnitude:lightyear}{प्रकाश वर्ष}) क्या बदलाव झूठ बोला गया?
  \item पुनर्मिलन पर उम्र का अंतर क्या है, और कौन सा जुड़वां छोटा है?
  \item ``लेकिन \omterm{def:g12:newtons-laws:momentum}{गति} सापेक्ष है --- अंतरिक्ष यात्री यह दावा कर सकता है कि पृथ्वी ने ऐसा किया था
        यात्रा!'' आपत्ति का समाधान एक वाक्य में करें।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The GPS bill.}
एक जीपीएस \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} कक्षाओं त्रिज्या पर $ r = \qty{2.66e7}{m}$, \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} $ T = \qty{12}{h}$.
\begin{enumerate}[resume]
  \item इसकी कक्षीय \omterm{def:g12:newtons-laws:momentum}{गति} की गणना करें $ v = 2\pi r/T $.
  \item गणना करना $\gamma - 1 \approx v^2/2c^2$.
  \item माइक्रोसेकंड में पता लगाएं कि कितनी दूर है \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} घड़ी पीछे छूट जाती है
        प्रति दिन ग्राउंड घड़ियाँ (\qty{86400}{s}).
  \item यदि केवल यह बहाव ही ठीक नहीं हुआ, तो स्थितियाँ ग़लत हो गईं
        $ c \times$ (बहाव): एक दिन के बाद त्रुटि की गणना करें।
  \item ऊंचाई पर कमजोर गुरुत्वाकर्षण, घड़ी को चलाता है \emph{तेज़} द्वारा
        \qty{45}{\micro s} प्रति दिन (सामान्य सापेक्षता): नेट की गणना करें
        बहाव और किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} में दैनिक स्थिति त्रुटि --- संख्या
        यह इंजीनियरों को प्रत्येक जीपीएस घड़ी को डीट्यून करने के लिए बाध्य करता है।
\end{enumerate}
\end{problem}

\noindent
यहां स्कूल की भौतिक विज्ञान समाप्त होती है --- भौतिक विज्ञान नहीं। विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} में
वॉल्यूम, मैकेनिक्स कैलकुलस से लैस होकर लौटता है; बिजली और चुंबकत्व
स्वयं को एक संरचना के रूप में प्रकट करें जिसमें प्रकाश, और सापेक्षता स्वयं,
हर समय छिपा रहा था; पिछले अध्याय में देखी गई क्वांटम दुनिया एक बन जाती है
पदार्थ का कार्य सिद्धांत. प्रत्येक सिद्धांत, सापेक्षता की तरह, इसके लिए बाध्य होगा
जिसे आप अब एक सीमित मामले के रूप में जानते हैं उसे वापस सौंपें। इसे आगे बढ़ाएँ: यह है
वह भाग जो नहीं बदलेगा.
