\chapter{एक लहर के रूप में प्रकाश: विवर्तन और हस्तक्षेप}\label{ch:g12:light-as-wave}

साबुन की फिल्म को खिड़की की ओर झुकाएँ: हालाँकि, रंग उस पर सरकते हैं
साबुन या पानी में कुछ भी रंगीन नहीं होता। बिंदु ए \omterm{def:g11:color-light-sources:lamps}{लेज़र} एक ही बाल पर:
पीछे की दीवार नहीं दिखाती बाल पतली छाया लेकिन एक उजली लकीर,
गहरी रेखाओं से वर्जित, बालों से हजारों गुना अधिक चौड़ा।
सीधी चलने वाली किरणें (\cref{ch:g10:refraction}) न तो समझाएं
दृष्टि. यह अध्याय प्रकाश को पूर्ण रूप से बढ़ावा देता है \omterm{def:g10:signals-and-waves:wave}{लहर} --- आधा ए
शिखा से शिखा तक माइक्रो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} --- फिर विचार को भुनाता है: a \omterm{def:g11:color-light-sources:lamps}{लेज़र}
और एक नंगी दीवार एक बाल मापेगी, ए \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}, एक सीडी के खांचे।

\section{प्रकाश एक तरंग है}

\begin{definition}[प्रकाश तरंगों]\label{def:g12:light-as-wave:lightwave}
एकवर्णी प्रकाश --- एक शुद्ध रंग, जिसे कोई भी प्रिज्म विभाजित नहीं कर सकता
आगे (\cref{ch:g10:light-spectra}) --- एक है \omterm{def:g10:signals-and-waves:period}{आवधिक} \emph{प्रकाश तरंग}\index{प्रकाश तरंग}: यह निर्वात को पार करता है $c = \qty{3.00e8}{m/s}$
और इसकी विशेषता है \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति} $ f $ या उसके \emph{तरंग दैर्ध्य में निर्वात}\index{तरंग दैर्ध्य में निर्वात} $\lambda = c/f $, द
शिखा से शिखा की दूरी \cref{ch:g12:mechanical-waves}. आँख
के बारे में जवाब देता है \qty{400}{nm} (बैंगनी) को \qty{800}{nm} (लाल);
\omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} पूरा मिला देता है \omterm{def:g11:fundamental-interactions:strong}{श्रेणी}.
\end{definition}

\begin{example}[परिमाण के आदेश]\label{ex:g12:light-as-wave:orders}
एक हीलियम--नियॉन \omterm{def:g11:color-light-sources:lamps}{लेज़र} की लाल रोशनी उत्सर्जित करता है $\lambda = \qty{633}{nm} =
\qty{6.33e-7}{m}$, इसलिए $ f = c/\lambda \approx \qty{4.7e14}{Hz}$: नहीं
इलेक्ट्रॉनिक्स इतनी तेजी से गिनती करता है --- प्रत्येक \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} इस अध्याय में होगा
ज्यामिति द्वारा मापा जाना चाहिए। स्केल के लिए, एक बाल (लगभग) \qty{70}{\micro m})
सौ है \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} आर-पार। क्या \omterm{def:g10:signals-and-waves:wave}{लहर}ें? हवा नहीं: बिजली
और \omterm{def:g11:magnetic-fields:bfield}{चुंबकीय क्षेत्र}, वर्ष 2 खंड में ईमानदारी से वर्णित है; केवल
\omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} नीचे मायने रखता है.
\end{example}

\section{विवर्तन}

\begin{definition}[विवर्तन]\label{def:g12:light-as-wave:diffraction}
जब ए \omterm{def:g10:signals-and-waves:wave}{लहर} किसी छिद्र या आकार की बाधा से मिलता है $ a $ से तुलनीय
इसका \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}, यह इस क्षेत्र में \omterm{def:g11:color-light-sources:objectcolor}{फैलता है} कि किरणें अंधेरा छोड़ देंगी।
यह फैलाव है \emph{विवर्तन}\index{विवर्तन} --- नगण्य
जबकि $\lambda/a $ छोटा है, प्रभावी है $ a $ की ओर सिकुड़ जाता है $\lambda $.
\end{definition}

\begin{proposition}[फैलाव की कोणीय आधी चौड़ाई]\label{prop:g12:light-as-wave:halfwidth}
चौड़ाई का एक भट्ठा $ a $, के साथ जलाया \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} $\lambda < a $, फैलाता है
कोणीय आधी-चौड़ाई के भीतर घटना की दिशा के बारे में प्रकाश
$\theta $ (रेडियन); दूरी पर एक स्क्रीन $ D $ एक केंद्रीय उज्ज्वल दिखाता है
चौड़ाई का बैंड $ L $, दो बार $ D \tan\theta \approx D\,\theta $ (छोटा
कोण):
\[
\theta \approx \frac{\lambda}{a}, \qquad L = \frac{2\lambda D}{a}.
\]
\end{proposition}
\admitted

\begin{remark}[सूत्र को पीछे की ओर पढ़ें]\label{rem:g12:light-as-wave:honest}
 \emph{संकरा} भट्ठा, \emph{व्यापक} फैलाव---
किरण भविष्यवाणी के विपरीत. ईमानदार व्युत्पत्ति (संक्षेप)
स्लिट के आर-पार पुनः उत्सर्जित तरंगिकाएँ) वर्ष 2 खंड में रहती हैं; यहाँ
सूत्र एक मापने का उपकरण है.
\end{remark}

\begin{example}[एक छेद के माध्यम से एक लेजर]\label{ex:g12:light-as-wave:slit}
लाल \omterm{def:g11:color-light-sources:lamps}{लेज़र}, $\lambda = \qty{633}{nm}$; भट्ठा $ a = \qty{0.10}{mm}$;
स्क्रीन पर $ D = \qty{2.00}{m}$. तब $\theta = \num{6.33e-7} /
\num{1.0e-4} = \qty{6.3e-3}{rad}$ --- एक तिहाई डिग्री, लेकिन
दीवार पर केंद्रीय बैंड है $ L = 2 \times \num{6.33e-7} \times 2.00 /
\num{1.0e-4} \approx \qty{2.5}{cm}$ चौड़ी, अँधेरी रेखाओं से घिरी हुई और
फीके साइड बैंड. स्लिट को आधा करें और बैंड दोगुना हो जाएगा।
\end{example}

\begin{proposition}[एक बाधा एक भट्ठा की तरह विचलित हो जाती है]\label{prop:g12:light-as-wave:hair}
ए \omterm{def:g11:lenses-and-eye:lens}{पतला} चौड़ाई की अपारदर्शी बाधा $ d $ --- एक तार, एक बाल --- पैदा करता है,
आपतित बीम के बाहर, चौड़ाई के स्लिट के समान पैटर्न $ d $
(बैबिनेट का सिद्धांत, वर्ष~2 खंड में व्युत्पन्न)।
\end{proposition}
\admitted

\begin{method}[छोटी सी चौड़ाई नापना]\label{met:g12:light-as-wave:hairwidth}
\begin{enumerate}
  \item बालों को आर-पार खींचें \omterm{def:g11:color-light-sources:lamps}{लेज़र} खुशी से उछलना; पर एक स्क्रीन सेट करें
        मापी गई दूरी $ D $ (कुछ एम-eters).
  \item केंद्रीय बैंड की चौड़ाई मापें $ L $, डार्क लाइन से डार्क लाइन।
  \item तब से $ L = 2\lambda D/d $: $ d = 2\lambda D/L $ --- पतले बाल,
        बड़ा पैटर्न.
\end{enumerate}
\end{method}

\begin{remark}[विवर्तन प्रत्येक उपकरण को सीमित करता है]\label{rem:g12:light-as-wave:limits}
प्रकाश किसी कैमरे, दूरबीन या आंख में किसके छिद्र से प्रवेश करता है
चौड़ाई $ a $, तो एक आदर्श भी \omterm{def:g11:lenses-and-eye:lens}{लेंस} एक बिंदु को एक स्थान में धुंधला कर देता है
कोणीय आकार के बारे में $\lambda/a $: निकट विवरण वि\omterm{def:g12:sound-acoustics:timbre}{लय}। एक दृश्य-प्रकाश
इस प्रकार माइक्रोस्कोप से नीचे कुछ भी अधिक हल नहीं होता है $\lambda \approx
\qty{0.5}{\micro m}$ (बैक्टीरिया हाँ, वायरस नहीं); दूरबीन दर्पण
इसी कारण से व्यापक रूप से निर्मित किये गये हैं।
\end{remark}

\section{दखल अंदाजी}

\begin{definition}[सुसंगत स्रोत, पथ अंतर]\label{def:g12:light-as-wave:coherent}
दो \omterm{def:g10:signals-and-waves:wave}{लहर} स्रोत हैं \emph{सुसंगत}\index{सुसंगत} कब
उनके पास समान है \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति} और चरण में कंपन --- व्यवहार में,
जब वे एक की दो प्रतियाँ हों \omterm{def:g10:signals-and-waves:wave}{लहर}, इसे विभाजित करके प्राप्त किया गया। एक के लिए
बिंदु $ M $ दोनों के द्वारा पहुँचे, \emph{पथ अंतर}%
\index{पथ अंतर} है $\delta = S_2M - S_1M $.
\end{definition}

\begin{proposition}[हस्तक्षेप की स्थितियाँ]\label{prop:g12:light-as-wave:conditions}
कहाँ दो \omterm{def:g12:light-as-wave:coherent}{सुसंगत} \omterm{def:g10:signals-and-waves:wave}{लहर}ें का \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} $\lambda $ ओवरलैप वे जोड़ते हैं ---
वे \emph{हस्तक्षेप}\index{हस्तक्षेप}:
\begin{itemize}
  \item $\delta = k\lambda $ ($ k $ पूर्णांक): चरण में, सुदृढ़ीकरण ---
        \emph{रचनात्मक}\index{रचनात्मक} \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} (उज्ज्वल);
  \item $\delta = (k + \tfrac12)\lambda $: गर्त पर शिखा, रद्द करना
        --- \emph{विनाशकारी}\index{विनाशकारी} \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} (अंधेरा)।
\end{itemize}
\end{proposition}

\begin{proof}
स्थानांतरण ए \omterm{def:g12:mechanical-waves:periodic}{आवधिक तरंग} संपूर्ण रूप से \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} कुछ भी नहीं बदलता: शिखर
शिखरों पर ढेर. का आधा \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} अधिक गर्त पर शिखा डालता है: बराबर
\omterm{def:g10:signals-and-waves:wave}{लहर}ें रद्द करना।
\end{proof}

\begin{example}[यंग का टू-स्लिट प्रयोग]\label{ex:g12:light-as-wave:young}
दो बारीक चीरे लगाओ $ S_1$, $ S_2$, एक मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} का एक अंश अलग,
एक अपारदर्शी प्लेट में, और दोनों को एक से जलाएं \omterm{def:g11:color-light-sources:lamps}{लेज़र}: दो \omterm{def:g12:light-as-wave:coherent}{सुसंगत}
स्रोत. एक स्क्रीन पर कुछ एम-eters दूर ओवरलैप धारीदार है:
केंद्र पर $\delta = 0$, चमकदार; स्क्रीन पर चढ़ना, $\delta $
के माध्यम से बढ़ता है $\lambda/2$ (अँधेरा), $\lambda $ (चमकदार)\,\dots\ अँधेरा
जहां प्रकाश में जोड़ा गया प्रकाश रद्द हो जाता है।
\end{example}

\begin{definition}[फ्रिंज और इंटरफ्रिंज]\label{def:g12:light-as-wave:fringes}
पैटर्न की बारी-बारी से चमकीली और गहरी पट्टियाँ हैं
\emph{हस्तक्षेप किनारे}\index{हस्तक्षेप किनारे}; दूरी
$ i $ क्रमागत उज्ज्वल केन्द्रों के बीच है \emph{\omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} करना}%
\index{हस्तक्षेप करना}.
\end{definition}

\begin{proposition}[हस्तक्षेप]\label{prop:g12:light-as-wave:interfringe}
स्लिट्स के लिए दूरी $ a $ अलग और दूरी पर एक स्क्रीन $ D \gg a $,
दूरी पर स्क्रीन का बिंदु $ x $ केंद्र से $ O $ है \omterm{def:g12:light-as-wave:coherent}{पथ
अंतर} $\delta \approx a\,x/D $ (स्वीकृत ज्यामिति); द्वारा
\cref{prop:g12:light-as-wave:conditions}, चमकदार झालरें बैठती हैं
$ x_k = k\,\lambda D/a $, समान रूप से दूरी पर \omterm{def:g12:light-as-wave:fringes}{हस्तक्षेप करना}
\[
i = \frac{\lambda D}{a}.
\]
साथ $\lambda = \qty{633}{nm}$, $ a = \qty{0.20}{mm}$, $ D =
\qty{2.0}{m}$: $ i = \qty{6.3}{mm}$ --- सेटअप बड़ा होता है $\lambda $
द्वारा $ D/a = 10^4$. इसके पीछे की ज्यामिति पर वर्ष~2 में \omterm{def:g11:work-of-force:work}{काम} किया गया है
आयतन.
\end{proposition}
\admitted

$ i = \lambda D/a $ अलग, एक योजनाबद्ध समग्र लिफाफे के नीचे।}
\end{omfigure}

\begin{method}[तरंग दैर्ध्य मापना]\label{met:g12:light-as-wave:lambda}
\begin{enumerate}
  \item अज्ञात भेजें \omterm{def:g11:color-light-sources:lamps}{लेज़र} ज्ञात रिक्ति के दोहरे स्लिट के माध्यम से
        $ a $; स्क्रीन को मापी गई दूरी पर रखें $ D $.
  \item की \omterm{def:g10:signals-and-waves:period}{अवधि} मापें $10$ \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} करता है; द्वारा विभाजित करें $10$: द
        \omterm{def:g12:light-as-wave:fringes}{हस्तक्षेप करना} $ i $, रूलर त्रुटि के साथ दस गुना कम हो गया।
  \item तब $\lambda = i\,a/D $.
\end{enumerate}
\end{method}

\begin{remark}[दो दीपक कभी हस्तक्षेप क्यों नहीं करते?]\label{rem:g12:light-as-wave:lamps}
एक दीवार को रोशन करने वाले दो डेस्क लैंप से कोई झालर नहीं बनती: प्रत्येक से उत्सर्जन होता है
संक्षिप्त, असंबंधित \omterm{def:g10:signals-and-waves:wave}{लहर} ऐसी रेलगाड़ियाँ जिनका सापेक्ष समय बेतरतीब ढंग से बढ़ता है
एक सेकंड में एक अरब बार, इसलिए फ्रिन्ज उतनी ही तेजी से बदलते हैं
आँख उन्हें एकसमान प्रकाश में औसत करती है। \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} की मांग जुटना
--- एक \omterm{def:g10:signals-and-waves:wave}{लहर} दो भागों में विभाजित, जैसा कि यंग के स्लिट्स करते हैं; \omterm{def:g11:color-light-sources:lamps}{लेज़र} है
आधुनिक शॉर्टकट, एक साफ़ लंबा \omterm{def:g10:signals-and-waves:wave}{लहर}.
\end{remark}

\section{हस्तक्षेप से रंगे रंग}

\begin{definition}[आनंददायकता]\label{def:g12:light-as-wave:iridescence}
रंग जो देखने के कोण के साथ बदलते हैं --- साबुन के बुलबुले, तेल की परतें,
मोर पंख, एक सीडी के पीछे --- हैं \emph{आनंददायकता}%
\index{आनंददायकता}: कोई वर्णक नहीं, लेकिन \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी},
कुछ को मजबूत करना \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} और दूसरों को रद्द कर रहा हूँ।
\end{definition}

\begin{example}[साबुन की परतें और तेल की परतें]\label{ex:g12:light-as-wave:soap}
साबुन की फिल्म से टकराने वाली रोशनी आंशिक रूप से उसके सामने के चेहरे से प्रतिबिंबित होती है
इसके पीछे से: दो \omterm{def:g12:light-as-wave:coherent}{सुसंगत} प्रतिलिपियां किसकी \omterm{def:g12:light-as-wave:coherent}{पथ अंतर} द्वारा निर्धारित किया गया है
मोटाई और देखने का कोण। प्रत्येक मोटाई कुछ लौटाती है
\omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} प्रबलित, अन्य रद्द: फिल्म एक रंग को दर्शाती है
जो नालियों के रूप में फिसलता है और \omterm{def:g11:lenses-and-eye:lens}{पतला}. गीले डामर पर तेल की परत खेलती है
वही चाल.
\end{example}

\begin{remark}[मोनोक्रोमैटिक बनाम सफेद रोशनी]\label{rem:g12:light-as-wave:white}
में \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} प्रत्येक \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} अपने स्वयं के युवा किनारों को अपने साथ खींचता है
अपना $ i = \lambda D/a $: केंद्र में सभी रंग चमकीले हैं (एक सफेद)।
फ्रिंज); कुछ किनारे पैटर्न से अलग हो जाते हैं --- इंद्रधनुषी
किनारे, फिर धुंधले। एकवर्णी प्रकाश बिना कुरकुरी झालरें देता है
अंत: इसलिए \omterm{def:g11:color-light-sources:lamps}{लेज़र} यहां हर तेज माप में.
\end{remark}

\section{समस्या: प्रकाश से मापना}

\begin{problem}[{सप्ताहांत समस्या --- प्रकाश से मापना: एक लेजर, ए
रूलर और एक नंगी दीवार एक स्लिट मापने वाली माइक्रोमीटर कार्यशाला बन जाती है, ए
मानव बाल, एक अज्ञात तरंग दैर्ध्य और दो चांदी की डिस्क के खांचे}]
\label{pb:g12:light-as-wave:1}
आपकी किट: एक लाल \omterm{def:g11:color-light-sources:lamps}{लेज़र} ($\lambda = \qty{633}{nm}$, लेबल से), ए
हरा \omterm{def:g11:color-light-sources:lamps}{लेज़र} अज्ञात का \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}, एक कैलिब्रेटेड स्लिट
$ a = \qty{100}{\micro m}$, रिक्ति का एक दोहरा भट्ठा $ a' =
\qty{0.250}{mm}$, एक टेप माप, दो बाल, एक सीडी और एक डीवीडी।

\medskip
\noindent\textbf{Part I --- Calibrating on the known slit.}
लाल \omterm{def:g11:color-light-sources:lamps}{लेज़र}, भट्ठा, दीवार पर $ D = \qty{3.00}{m}$.
\begin{enumerate}
  \item कोणीय आधी-चौड़ाई की गणना करें $\theta $ विवर्तित किरण का.
  \item दिखाएँ कि केंद्रीय बैंड की चौड़ाई है $ L = 2\lambda D/a $;
        इसकी गणना करें.
  \item आप मापें $ L = \qty{3.7}{cm}$: सापेक्ष की गणना करें
        विचलन. क्या विधि मान्य है?
  \item एक \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} स्लिट आधा चौड़ा है, \qty{50}{\micro m}: इसकी भविष्यवाणी करें
        $ L $. स्लिट चौड़ाई और पैटर्न चौड़ाई को जोड़ने वाला नियम बताएं।
  \item प्रश्न 2 के कदम का औचित्य सिद्ध करें $\tan\theta \approx \theta $: तुलना करना
        $\theta $ और $\tan\theta $ इस कोण पर.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hair.} वही \omterm{def:g11:color-light-sources:lamps}{लेज़र} और दीवार; ए
फैले हुए बाल स्लिट की जगह ले लेते हैं।
\begin{enumerate}[resume]
  \item कौन सा स्वीकृत परिणाम आपको स्लिट फॉर्मूला को ए के लिए रखने की सुविधा देता है
        बाल, और यह क्या कहता है?
  \item केन्द्रीय बैंड है $ L = \qty{5.4}{cm}$ विस्तृत: गणना करें
        बाल का व्यास $ d $.
  \item आपका शासक पढ़ता है $ L $ को $\pm\qty{0.2}{cm}$: परिणाम दीजिए
        \omterm{def:g11:fundamental-interactions:strong}{श्रेणी} के लिए $ d $, जैसा $ d \pm \Delta d $.
  \item आपके दोस्त के बाल देते हैं $ L = \qty{7.6}{cm}$: इसका व्यास?
        किसके बाल पतले हैं, और पतले का मतलब चौड़ा क्यों है?
  \item क्या कोई डेस्क लैंप इसकी जगह ले सकता है? \omterm{def:g11:color-light-sources:lamps}{लेज़र}? दो कारण बताइये.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The green \omterm{def:g11:color-light-sources:lamps}{लेज़र}'s \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}.} हरा
\omterm{def:g11:color-light-sources:lamps}{लेज़र}, डबल स्लिट, दीवार पर $ D = \qty{2.00}{m}$.
\begin{enumerate}[resume]
  \item दोनों स्लिटों को क्यों जलाया जाना चाहिए? \emph{वही} \omterm{def:g11:color-light-sources:lamps}{लेज़र}?
  \item जो भी हो, केंद्रीय किनारा चमकीला क्यों है? \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}?
  \item दस \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} करता है \omterm{def:g10:signals-and-waves:period}{अवधि} \qty{4.26}{cm}. देना $ i $, और कारण
        एक साथ दस मापने के लिए.
  \item निष्कर्ष निकालें \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} हरे रंग का \omterm{def:g11:color-light-sources:lamps}{लेज़र}.
  \item निर्माता का लेबल कहता है $\qty{532 \pm 10}{nm}$: \omterm{def:g12:light-as-wave:coherent}{सुसंगत}?
        आप किस रंग की अपेक्षा करते हैं?
\end{enumerate}
