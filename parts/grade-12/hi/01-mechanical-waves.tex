\chapter{यांत्रिक तरंगें}\label{ch:g12:mechanical-waves}

शनिवार की शाम, खचाखच भरा स्टेडियम: खड़े दर्शकों की \omterm{def:g10:signals-and-waves:wave}{लहर}
एक मिनट से भी कम समय में स्टैंड के चारों ओर सफ़ाई कर देता है, फिर भी कोई उन्हें नहीं छोड़ता
सीट. एक कंकड़ का घेरा \omterm{def:g11:lenses-and-eye:magnifier}{ताल}ाब को पार करता है जबकि कॉर्क केवल हिलता है। यह
अध्याय स्वयं यात्री का अध्ययन करता है --- \omterm{def:g10:signals-and-waves:wave}{लहर} --- और चार
संख्याएँ जो इसे वश में करती हैं: गति, \omterm{def:g12:mechanical-waves:delay}{देरी}, \omterm{def:g10:signals-and-waves:period}{अवधि}, \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} --- करने के लिए पर्याप्त
कागज की एक पट्टी पर भूकंप पढ़ें।

\section{अशांति यात्रा करती है, माध्यम बना रहता है}

\begin{definition}[यांत्रिक तरंग]\label{def:g12:mechanical-waves:wave}
ए \emph{यांत्रिक तरंग}\index{यांत्रिक तरंग} एक उपद्रव है
भौतिक माध्यम से प्रसार --- रस्सी, पानी, हवा, चट्टान ---
ले जाना ऊर्जा \emph{पदार्थ के परिवहन के बिना}: का प्रत्येक टुकड़ा
मीडियम थोड़ी देर के लिए अपनी जगह पर घूमता है, गति बढ़ाता है और वापस स्थिर हो जाता है।
\end{definition}

\begin{example}[कोई फर्क नहीं पड़ता परिवहन]\label{ex:g12:mechanical-waves:transport}
स्टेडियम में \omterm{def:g10:signals-and-waves:wave}{लहर} ``माध्यम'' भीड़ है: जो यात्रा करता है वह है
खड़े होने की क्रिया, जबकि प्रत्येक दर्शक अपनी सीट पर बैठा रहता है। पर
\omterm{def:g11:lenses-and-eye:magnifier}{ताल}ाब, वलय का विस्तार होता है लेकिन तैरता हुआ कॉर्क अपनी जगह पर ही हिलता है।
\end{example}

\begin{definition}[अनुप्रस्थ एवं अनुदैर्ध्य तरंगें]\label{def:g12:mechanical-waves:transverse}
ए \omterm{def:g10:signals-and-waves:wave}{लहर} है \emph{आड़ा}\index{आड़ा} जब माध्यम
चलता है \emph{सीधा} प्रसार के लिए: एक कूबड़ एक के साथ दौड़ता है
रस्सी को झटका जबकि प्रत्येक बिंदु केवल ऊपर और नीचे चलता है। यह है
\emph{अनुदैर्ध्य}\index{अनुदैर्ध्य} जब माध्यम चलता है
\emph{साथ में} प्रसार: निचोड़ा हुआ क्षेत्र , coils नीचे की ओर यात्रा करता है
धक्का दिया वसंत, प्रत्येक कुंडल जगह-जगह झूलना.
\end{definition}

\begin{remark}[ध्वनि अनुदैर्ध्य है]\label{rem:g12:mechanical-waves:sound}
\omterm{def:g10:signals-and-waves:sound}{आवाज़} है \omterm{def:g12:mechanical-waves:transverse}{अनुदैर्ध्य} \omterm{def:g10:signals-and-waves:wave}{लहर} सर्वोत्कृष्टता: संपीड़ित हवा ---

दोनों प्रकार भेजें: P~\omterm{def:g10:signals-and-waves:wave}{लहर}ें चट्टान को संपीड़ित करें, एस~\omterm{def:g10:signals-and-waves:wave}{लहर}ें इसे बग़ल में हिलाओ.
\end{remark}

\section{गति और विलंब}

\begin{definition}[प्रसार गति]\label{def:g12:mechanical-waves:speed}
 \emph{प्रसार गति}\index{प्रसार गति} एक का \omterm{def:g10:signals-and-waves:wave}{लहर} है
विक्षोभ द्वारा तय की गई दूरी को लगने वाले समय से विभाजित किया जाता है:
$v = d/t $. यह यात्रा की गति है \emph{आकार}, का नहीं
मध्यम.
\end{definition}

\begin{proposition}[माध्यम गति निर्धारित करता है]\label{prop:g12:mechanical-waves:medium}
छोटी-छोटी गड़बड़ियों के लिए, \omterm{def:g12:mechanical-waves:speed}{प्रसार गति} पर ही निर्भर करता है
\emph{मध्यम}, गड़बड़ी के आकार या आकृति पर नहीं। ए पर
तनी हुई रस्सी, $ v $ के साथ बढ़ता है \omterm{def:g10:inertia:inventory}{तनाव} और द्रव्यमान प्रति के साथ घटता जाता है
\omterm{def:g10:orders-of-magnitude:unit}{मीटर}: तना हुआ और हल्का तेज़ है, ढीला और भारी धीमा है। \admitted
\end{proposition}

\begin{remark}[जहां सूत्र रहता है]\label{rem:g12:mechanical-waves:ropeformula}
रस्सी का सटीक गति सूत्र वर्ष 1 खंड में लिया गया है; यह
वर्ष उपरोक्त प्रायोगिक तथ्य पर्याप्त हैं।
\end{remark}

\begin{example}[ध्वनि की गति]\label{ex:g12:mechanical-waves:speeds}
\omterm{def:g10:signals-and-waves:sound}{आवाज़} पर यात्रा करता है \qty{340}{m/s} हवा में, \qty{1500}{m/s} पानी में,
\qty{5900}{m/s} स्टील में ---जितना सख्त, उतना तेज़। इसलिए
वेस्टर्न-मूवी ईयर ऑन द रेल: स्टील ने सबसे पहले ट्रेन की घोषणा की।
\end{example}

\begin{definition}[देरी]\label{def:g12:mechanical-waves:delay}
ए \omterm{def:g10:signals-and-waves:wave}{लहर} गति का $ v $ की दूरी पर एक बिंदु तक पहुँच जाता है $ d $ के बाद
\emph{देरी}\index{देरी} $\tau = d/v $: दो अंक की दूरी $ d $
इसके अलावा उसी गति को दोहराएँ, जितना अधिक पीछे हो जाएँ $\tau $ --- द
गूंज का लेकर \cref{ch:g10:signals-and-waves}, पीछे की ओर पढ़ें।
\end{definition}

\begin{proposition}[अंतःक्रिया के बिना सुपरपोजिशन]\label{prop:g12:mechanical-waves:superposition}
कहाँ दो \omterm{def:g10:signals-and-waves:wave}{लहर}ें ओवरलैप, माध्यम का विस्थापन प्रत्येक पर होता है
तुरंत \emph{जोड़} प्रत्येक विस्थापन का कारण अकेले होगा;
पार करने के बाद, प्रत्येक \omterm{def:g10:signals-and-waves:wave}{लहर} अपरिवर्तित जारी है --- में व्युत्पन्न
गति के समीकरणों की रैखिकता से वर्ष~2 आयतन। \admitted
\end{proposition}

\begin{example}[क्रॉसिंग पल्स]\label{ex:g12:mechanical-waves:crossing}
रस्सी के प्रत्येक छोर से एक ऊपर की ओर कूबड़ भेजें: जहां वे मिलते हैं,
रस्सी संक्षेप में दो ऊंचाइयों के योग तक बढ़ जाती है; फिर दो अक्षुण्ण
कूबड़ उभर आती है --- जैसे दो परस्पर वार्\omterm{def:g11:lenses-and-eye:magnifier}{ताल}ाप अपने श्रोताओं तक पहुँचते हैं।
\end{example}

\section{आवधिक तरंगें: दोहरी आवधिकता}

\begin{definition}[आवधिक लहर]\label{def:g12:mechanical-waves:periodic}
जब स्रोत अपनी गति को दोहराता है \omterm{def:g10:signals-and-waves:period}{अवधि} $ T $ और \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}
$ f = 1/T $ (\cref{ch:g10:signals-and-waves}), यह माध्यम को पोषण देता है a
\emph{आवधिक तरंग}\index{आवधिक तरंग}: प्रत्येक बिंदु अपना-अपना दोहराता है
गति के साथ \omterm{def:g10:signals-and-waves:period}{अवधि} $ T $, प्रत्येक यात्रा में अपने पड़ोसियों से पीछे है \omterm{def:g12:mechanical-waves:delay}{देरी}.
\end{definition}

\begin{definition}[वेवलेंथ]\label{def:g12:mechanical-waves:wavelength}
ए का एक स्नैपशॉट \omterm{def:g12:mechanical-waves:periodic}{आवधिक तरंग} अंतरिक्ष में दोहराया जाने वाला एक पैटर्न दिखाता है; इसका
स्थानिक \omterm{def:g10:signals-and-waves:period}{अवधि} --- शिखा से शिखा --- है
\emph{तरंग दैर्ध्य}\index{तरंग दैर्ध्य} $\lambda $, में एम-eters. ए \omterm{def:g10:signals-and-waves:period}{आवधिक}
\omterm{def:g10:signals-and-waves:wave}{लहर} है \emph{दोगुना आवधिक}\index{दोगुना आवधिक}: \omterm{def:g10:signals-and-waves:period}{अवधि} $ T $
प्रत्येक बिंदु पर समय में, \omterm{def:g10:signals-and-waves:period}{अवधि} $\lambda $ प्रत्येक क्षण अंतरिक्ष में.
\end{definition}

\begin{theorem}[तरंग संबंध]\label{thm:g12:mechanical-waves:relation}
ए \omterm{def:g12:mechanical-waves:periodic}{आवधिक तरंग} गति का $ v $, \omterm{def:g10:signals-and-waves:period}{अवधि} $ T $ और \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति} $ f $ है
\omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} $\lambda = v\,T = v/f $.
\end{theorem}

\begin{proof}
एक के दौरान \omterm{def:g10:signals-and-waves:period}{अवधि} पैटर्न आगे की ओर खिसकता है $ v\,T $; लेकिन स्रोत
--- इसलिए प्रत्येक बिंदु --- फिर अपनी प्रारंभिक स्थिति में वापस आ जाता है, इसलिए
स्लाइड पैटर्न मूल के साथ मेल खाता है: दोहराव दूरी है $ v\,T $.
\end{proof}

\begin{remark}[आवृत्ति स्रोत से संबंधित है]\label{rem:g12:mechanical-waves:source}
एक माध्यम से दूसरे माध्यम में जाना --- \omterm{def:g10:signals-and-waves:sound}{आवाज़} हवा से पानी में ---
ए \omterm{def:g10:signals-and-waves:wave}{लहर} अपना रखता है \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}, जो स्रोत निर्देशित करता है, जबकि इसकी गति
परिवर्तन; द्वारा $\lambda = v/f $  \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} इसके साथ बदलता है.
\end{remark}

\begin{example}[कॉन्सर्ट ए, हवा में और पानी के नीचे]\label{ex:g12:mechanical-waves:concertA}
 \qty{440}{Hz} का कॉन्सर्ट ए \cref{ch:g10:signals-and-waves} है
\omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} $340/440 = \qty{0.77}{m}$ हवा में और $1500/440 =
\qty{3.4}{m}$ पानी में: वही स्वर, तेज़ माध्यम द्वारा फैलाया गया।
\end{example}

\begin{definition}[चरण में, चरण विपक्ष में]\label{def:g12:mechanical-waves:phase}
दो बिंदुओं को पूर्ण संख्या से अलग किया गया \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}, $ d = k\lambda $,
हर पल एक समान गति करें: वे हैं
\emph{चरणबद्ध}\index{चरणबद्ध}. द्वारा अलग किये गये अंक
$ d = (k + \tfrac12)\lambda $ हमेशा विपरीत दिशा में आगे बढ़ें --- विपरीत दिशा में आगे बढ़ें
गर्त: वे अंदर हैं \emph{चरण विरोध}\index{चरण विरोध}.
\end{definition}

\begin{method}[दोहरी आवधिकता पढ़ना]\label{met:g12:mechanical-waves:reading}
\begin{enumerate}
  \item एक रिकॉर्डिंग पर $ y(t)$ एक निश्चित बिंदु पर, पढ़ें $ T $ शिखा को
        शिखा; $ f = 1/T $.
  \item एक स्नैपशॉट पर $ y(x)$ एक निश्चित क्षण में, पढ़ें $\lambda $ शिखा
        शिखा करना.
  \item परिणाम निकालना $ v = \lambda/T = \lambda f $; प्रत्यक्ष के विरुद्ध जाँच करें
        समय $ v = d/\tau $ कब उपलब्ध है।
  \item दो ग्राफ़ों को कभी भ्रमित न करें: एक अक्ष सेकंड को वहन करता है
        अन्य एम-eters.
\end{enumerate}
\end{method}

\section{वेवफ्रंट और क्षीणन}

\begin{definition}[वेवफ़्रंट]\label{def:g12:mechanical-waves:wavefront}
ए \emph{तरंगाग्र}\index{तरंगाग्र} बिंदुओं की एक रेखा (या सतह) है
वह \omterm{def:g10:signals-and-waves:wave}{लहर} उसी क्षण पहुँच जाता है --- पर एक शिखा रेखा
\omterm{def:g11:lenses-and-eye:magnifier}{ताल}ाब. एक बिंदु स्रोत से तरंगाग्र वृत्तों का विस्तार कर रहे हैं; दूर
किसी भी स्रोत से वे लगभग सीधे हैं। \omterm{def:g10:signals-and-waves:wave}{लहर} प्रचारित करता है
इसके तरंगाग्रों के लंबवत, क्रमिक शिखर $\lambda $ अलग।
\end{definition}

\begin{definition}[क्षीणन]\label{def:g12:mechanical-waves:attenuation}
ए की कमी \omterm{def:g10:signals-and-waves:wave}{लहर}'एस \omterm{def:g10:signals-and-waves:amplitude}{आयाम} जैसा वह प्रचारित करता है वैसा ही उसका है
\emph{क्षीणन}\index{क्षीणन}. दो कारण जुड़ते हैं: द ऊर्जा
एक परिपत्र का \omterm{def:g12:mechanical-waves:wavefront}{तरंगाग्र} एक लंबे शिखर पर साझा किया जाता है, और
माध्यम गुजरने में थोड़ा सा अवशोषित करता है। सीधा वेवफ्रंट्स बच जाओ
पहला कारण--- कैसे दूर से आया तूफ़ान समुद्र को पार कर सकता है।
\end{definition}

\section{समस्या: सिस्मोग्राम पढ़ना}

\begin{problem}[{सप्ताहांत समस्या---भूकंपलेख पढ़ना: दो तरंगें
टूटे हुए भ्रंश से बाहर निकलें, तीन वृत्त उपरिकेंद्र को पिन करें, और ए
धीमी तीसरी लहर समुद्र तट को इसकी चेतावनी देती है}]
\label{pb:g12:mechanical-waves:1}
भूकंपीय वेधशाला में रात्रि पाली। 09:14:32 पर की सूई
स्टेशन ए छलांग: तेज़ तेज़ हिलियाँ, फिर धीमी, व्यापक झूलें। एक
भूकंप ने कहीं चट्टान तोड़ दी है; दो \omterm{def:g10:signals-and-waves:wave}{लहर}ें के माध्यम से बाहर दौड़ें
पपड़ी: पी \omterm{def:g10:signals-and-waves:wave}{लहर}ें, संपीड़न पर $ v_P = \qty{6.0}{km/s}$, और एस \omterm{def:g10:signals-and-waves:wave}{लहर}ें,
एक बग़ल में मिलाते हुए $ v_S = \qty{3.5}{km/s}$.

\medskip
\noindent\textbf{Part I --- Two \omterm{def:g10:signals-and-waves:wave}{लहर}ें, one clock.}
\begin{enumerate}
  \item उपरोक्त विवरण से, P और S को वर्गीकृत करें \omterm{def:g10:signals-and-waves:wave}{लहर}ें जैसा
        \omterm{def:g12:mechanical-waves:transverse}{आड़ा} या \omterm{def:g12:mechanical-waves:transverse}{अनुदैर्ध्य}.
  \item कौन \omterm{def:g10:signals-and-waves:wave}{लहर} प्रत्येक स्टेशन पर सबसे पहले आता है? यात्रा की गणना करें
        दोनों का समय ख़त्म $ d = \qty{120}{km}$.
  \item दिखाएँ कि P--S आगमन अंतर है
        $\Delta t = d\,(1/v_S - 1/v_P)$ और इसकी गणना करें
        $ d = \qty{120}{km}$.
  \item भूकंपविज्ञानी का रेंजिंग नियम निकालें
        $ d = \dfrac{v_P\,v_S}{v_P - v_S}\,\Delta t $ और दिखाओ
        गुणांक के बारे में है \qty{8.4}{km} प्रति सेकंड अंतराल.
  \item दूरी के साथ फासला क्यों बढ़ता है --- और वह उपहार क्यों है
        एक ऐसे स्टेशन पर जिसने कभी भूकंप की शुरुआत नहीं देखी?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Circles on the map.}
\begin{enumerate}[resume]
  \item स्टेशन ए उपाय $\Delta t_A = \qty{25}{s}$. कितनी दूर है
        ए से उपकेंद्र?
  \item स्पष्ट करें कि इस पाठ में भूकंप का केंद्र एक वृत्त पर क्यों रखा गया है,
        एक बिंदु पर नहीं.
  \item स्टेशन बी और सी मापते हैं $\Delta t_B = \qty{15}{s}$ और
        $\Delta t_C = \qty{32}{s}$: भूकंप के केंद्र से उनकी दूरी?
  \item बताएं कि कैसे तीन वृत्त मिलकर उपरिकेंद्र को नीचे दबाते हैं।
  \item पी \omterm{def:g10:signals-and-waves:wave}{लहर} 09:14:32 बजे ए पर पहुंच गया। से इसकी यात्रा के समय की गणना करें
        भूकंप का केंद्र और उत्पत्ति का समय।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g10:signals-and-waves:wave}{लहर} that crosses the ocean.}
तीन वृत्त अपतटीय क्षेत्र में मिलते हैं: समुद्र तल के नीचे भ्रंश टूट गया
और जल स्तंभ को गतिमान करें। लंबा \omterm{def:g10:signals-and-waves:wave}{लहर}ें गहरे पानी की यात्रा पर ए
गहराई द्वारा निर्धारित गति --- के बारे में \qty{200}{m/s} खुले सागर में,
के बारे में \qty{10}{m/s} एक समुद्र तट के पास (यहाँ स्वीकार किया गया; सूत्र है
वर्ष 1 खंड में व्युत्पन्न)। एक बंदरगाह स्थित है $ D = \qty{600}{km}$ से
उपकेंद्र.
\begin{enumerate}[resume]
  \item बदलना \qty{200}{m/s} को \unit{km/h} और एक वाहन का नाम बताएं
        एक तुलनीय परिभ्रमण गति.
  \item सुनामी को बंदरगाह तक पहुँचने में कितना समय लगता है?
  \item बंदरगाह का अपना भूकंपमापी पी महसूस करता है \omterm{def:g10:signals-and-waves:wave}{लहर} पहला। आखिर कैसे
        लंबा? अगर अलार्म \omterm{def:g10:signals-and-waves:sound}{आवाज़} उस पल में, कितनी चेतावनी
        सुनामी से पहले तट को क्या समय मिलता है?
  \item समुद्र में सुनामी है \omterm{def:g10:signals-and-waves:period}{अवधि} के बारे में है \qty{15}{min}. इसकी गणना करें
        \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} वहाँ, और समझाएँ कि एक जहाज़ इस पर ध्यान क्यों नहीं देता।
  \item समुद्र तट के निकट गति कम हो जाती है \qty{10}{m/s} जब
        \omterm{def:g10:signals-and-waves:period}{अवधि} स्थिर रहता है. नये की गणना करें \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}. क्या चाहिए
        के साथ घटित होता है \omterm{def:g10:signals-and-waves:wave}{लहर} जैसे यह गुच्छित होता है?
\end{enumerate}

\noindent\textbf{Part IV --- ऊर्जा, and the report.}
के आदेश पर एक तीव्र भूकंप जारी होता है \qty{1e15}{J}, द्वारा ले जाया गया
द \omterm{def:g10:signals-and-waves:wave}{लहर}ें.
\begin{enumerate}[resume]
  \item एक वृत्ताकार की लंबाई की गणना करें \omterm{def:g12:mechanical-waves:wavefront}{तरंगाग्र} \qty{10}{km} से
        फिर, उपरिकेंद्र \qty{210}{km}; यह किस कारक से बढ़ा है?
  \item पता लगाएं कि दोषरहित स्थिति में भी दूरी के साथ कंपन कमजोर क्यों हो जाता है
        पपड़ी --- और वास्तविक पपड़ी जुड़ने के दूसरे कारण का नाम बताइए।
  \item कोई तरल पदार्थ कतरनी का विरोध नहीं कर सकता। सुदूरवर्ती स्टेशन पी रिकॉर्ड करते हैं
        \omterm{def:g10:signals-and-waves:wave}{लहर} लेकिन कोई एस नहीं \omterm{def:g10:signals-and-waves:wave}{लहर}: इससे पृथ्वी के केंद्र के बारे में क्या पता चलता है?
  \item पिछले वर्ष की हाई-स्पीड ट्रेन ऊर्जा अध्याय
        (\cref{ch:g11:mechanical-energy}) ले जाता है \qty{1.5e9}{J}
        का \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा}:ऐसी कितनी ट्रेनें हैं \qty{1e15}{J} लायक?
  \item रिपोर्ट, दो वाक्यों में: भूकंप कहाँ और कब आया था,
        और बंदरगाह को कितनी चेतावनी मिली? तीन संख्याएं बताएं.
\end{enumerate}
\end{problem}
