\chapter{क्वांटम विश्व: फोटॉन और ऊर्जा स्तर}\label{ch:g12:quantum-world}

एक \omterm{def:g10:light-spectra:wavelength}{अवरक्त} हीट लैंप पूरी शाम आपकी त्वचा को नहला सकता है और उस पर कभी निशान नहीं डाल सकता; दस
के मिनट \omterm{def:g11:lenses-and-eye:lens}{पतला} पहाड़ की धूप जलन छोड़ती है। लैंप कहीं अधिक प्रदान करता है
ऊर्जा --- लेकिन ऊर्जा वह नहीं जो त्वचा पर निशान बनाता है। प्रकाश, यह अध्याय दिखाता है,
अनाज में आता है, और केवल एक अनाज जो काफी \omterm{def:g10:inertia:inventory}{जोर} से टकराता है वह रसायन विज्ञान करता है:
एक \omterm{def:g10:light-spectra:wavelength}{पराबैंगनी} अनाज एक अरब का बंधन तोड़ता है \omterm{def:g10:light-spectra:wavelength}{अवरक्त} अनाज केवल गर्म.
अनाज का पालन करें और एक शताब्दी खुलती है: धातुएं इलेक्ट्रॉनों का रिसाव करती हैं, परमाणुओं
में गाना वर्णक्रमीय रेखाएँ, इलेक्ट्रॉन जैसे \omterm{prop:g12:light-as-wave:conditions}{हस्तक्षेप} करते हैं \omterm{def:g10:signals-and-waves:wave}{लहर}ें.

\section{प्रकाश अनाज में आता है: फोटॉन}

\begin{definition}[फोटॉन]\label{def:g12:quantum-world:photon}
का प्रकाश \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति} $\nu $ एक्सचेंजों ऊर्जा केवल साबुत अनाज में पदार्थ के साथ
बुलाया \emph{फोटॉनों}\index{फोटॉनों}, की प्रत्येक ऊर्जा
\[
E = h\nu = \frac{hc}{\lambda}, \qquad h = \qty{6.63e-34}{J.s},
\]
कहाँ $ h $ है \emph{प्लैंक स्थिरांक}\index{प्लैंक स्थिरांक} (प्लैंक,
1900; आइंस्टीन, 1905) और $\lambda $  \omterm{def:g12:light-as-wave:lightwave}{निर्वात में तरंग दैर्ध्य}. एक किरण का
तीव्रता मापती है \emph{संख्या} प्रति सेकंड फोटॉन की; प्रत्येक दाने का
ऊर्जा द्वारा तय किया गया है \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति} अकेला।
\end{definition}

\begin{notation}[इलेक्ट्रॉनवोल्ट]\label{not:g12:quantum-world:ev}
परमाणु भौतिकी का उपयोग करता है \emph{इलेक्ट्रॉनवोल्ट}\index{इलेक्ट्रॉनवोल्ट},
$\qty{1}{eV} = \qty{1.60e-19}{J}$: द ऊर्जा एक \omterm{def:g11:fundamental-interactions:elementary}{प्राथमिक प्रभार} लाभ
पार करना \qty{1}{V} (\cref{ch:g11:electric-gravitational-fields}). सुविधाजनक:
$ E \,(\unit{eV}) \approx 1240/\lambda\,(\unit{nm})$.
\end{notation}

\section{फोटोइलेक्ट्रिक प्रभाव}

निर्वात में एक साफ धातु की प्लेट पर प्रकाश चमकाएं और इलेक्ट्रॉन बाहर निकल आएं---
\emph{फोटोइलेक्ट्रिक प्रभाव}\index{फोटोइलेक्ट्रिक प्रभाव}. इसके नियमों ने शास्त्रीय भौतिकी को तोड़ दिया।

\begin{proposition}[प्रयोग क्या दिखाता है]\label{prop:g12:quantum-world:facts}
प्रत्येक धातु के लिए एक है \emph{दहलीज आवृत्ति}\index{दहलीज आवृत्ति} $\nu_0$:
\begin{enumerate}
  \item नीचे $\nu_0$, कोई भी इलेक्ट्रॉन नहीं निकलता, चाहे प्रकाश कितना भी तीव्र क्यों न हो;
  \item ऊपर $\nu_0$, उत्सर्जन बिना शुरू होता है \omterm{def:g12:mechanical-waves:delay}{देरी}, चाहे कितना भी फीका क्यों न हो;
  \item तीव्रता बढ़ जाती है \emph{संख्या} प्रति सेकंड इलेक्ट्रॉनों की, नहीं
        उनका अधिकतम \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा}; \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति} उठाता $ E_{k,\max}$.
\end{enumerate}
इनमें से कोई भी निरंतर फिट नहीं बैठता \omterm{def:g10:signals-and-waves:wave}{लहर}, जिसे किसी भी रंग को चार्ज होने देना चाहिए ---
महीनों तक धुंधली रोशनी में (\cref{exo:g12:quantum-world:14}) --- और किसका
तीव्रता को इलेक्ट्रॉनों को निर्धारित करना चाहिए' ऊर्जा: तीनों भविष्यवाणियाँ विफल रहीं।
\end{proposition}
\admitted

\begin{theorem}[आइंस्टीन का फोटोइलेक्ट्रिक संबंध]\label{thm:g12:quantum-world:einstein}
\index{फोटोइलेक्ट्रिक संबंध}एक इलेक्ट्रॉन को कम से कम इसकी आवश्यकता होती है \emph{कार्य फलन}\index{कार्य फलन} $ W_0$ धातु की सतह से बचने के लिए।
एक फोटोन एक इलेक्ट्रॉन द्वारा पूरा अवशोषित किया जाता है, इसलिए
\[
h\nu = W_0 + E_{k,\max}, \qquad \text{hence } \nu_0 = \frac{W_0}{h}.
\]
\end{theorem}

\begin{proof}
ऊर्जा बहीखाता: $ h\nu $ में, $ W_0$ भागने पर खर्च, बाकी गतिज ---
अधिक से अधिक $ h\nu - W_0$, गहराई से शुरू होने वाले इलेक्ट्रॉनों के लिए कम। नीचे $\nu_0$
एक दाना निकास टोल का भुगतान नहीं कर सकता, चाहे उनकी संख्या कुछ भी हो।
\end{proof}

\begin{definition}[निरोधी विभव]\label{def:g12:quantum-world:stopping}
मापने के लिए $ E_{k,\max}$, एक उलटा \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} इलेक्ट्रॉनों को ब्रेक करता है
(\cref{ch:g11:electric-gravitational-fields}); \emph{निरोधी विभव}\index{निरोधी विभव} $ U_s $ सबसे छोटा है \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} रद्द करना
वर्तमान: सबसे तेज़ इलेक्ट्रॉन बस पार करने में विफल रहते हैं, इसलिए $ E_{k,\max} = e\,U_s $.
\end{definition}

\begin{method}[प्लैंक स्थिरांक को मापना]\label{met:g12:quantum-world:measure-h}
\begin{enumerate}
  \item एक फोटोसेल को रोशन करें फिल्टर ज्ञात का \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य};
        प्रत्येक की गणना करें $\nu = c/\lambda $; प्रत्येक को रिकॉर्ड करें \omterm{def:g12:quantum-world:stopping}{निरोधी विभव}.
  \item कथानक $ E_{k,\max} = eU_s $ ख़िलाफ़ $\nu $: आइंस्टीन का वादा है
        सीधी रेखा $ h\nu - W_0$.
  \item ढलान पढ़ें: यह \emph{है} $ h $; अवरोधन देते हैं $\nu_0$ और $-W_0$.
\end{enumerate}
\end{method}

\section{ऊर्जा स्तर: परमाणु रेखाएँ क्यों उत्सर्जित करते हैं}

\begin{definition}[उर्जा स्तर]\label{def:g12:quantum-world:levels}
 ऊर्जा एक में बंधे एक इलेक्ट्रॉन का \omterm{def:g11:fundamental-interactions:ladder}{एटम} है
\emph{मात्रा निर्धारित}\index{मात्रा निर्धारित}: की एक पृथक सीढ़ी तक सीमित
\emph{उर्जा स्तर}\index{उर्जा स्तर} $ E_1 < E_2 < \dots < 0$, गिना गया
मुक्त इलेक्ट्रॉन से ऋणात्मक. सबसे निचला पायदान है \emph{जमीनी राज्य}\index{जमीनी राज्य}, अन्य लोग \emph{उत्साहित राज्य}\index{उत्साहित राज्य}; तक पहुँचने $0$ है
\emph{आयनीकरण}\index{आयनीकरण}.
\end{definition}

\begin{proposition}[हाइड्रोजन सीढ़ी]\label{prop:g12:quantum-world:hydrogen}
 \omterm{def:g12:quantum-world:levels}{उर्जा स्तर} हाइड्रोजन का \omterm{def:g11:fundamental-interactions:ladder}{एटम} हैं
\[
E_n = -\frac{\qty{13.6}{eV}}{n^2}, \qquad n = 1, 2, 3, \dots
\]
इसलिए $ E_1 = \qty{-13.6}{eV}$, $ E_2 = \qty{-3.40}{eV}$, $ E_3 =
\qty{-1.51}{eV}$, $ E_4 = \qty{-0.85}{eV}$, की ओर भीड़ $0$.
\end{proposition}
\admitted

\begin{remark}[सीढ़ी कहाँ से आती है]\label{rem:g12:quantum-world:ladder-origin}
ये संख्याएँ क्वांटम यांत्रिकी क्यों हैं, विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} के संस्करणों के लिए रखी गई हैं ---
जैसा कि डी ब्रोगली का सूत्र है: इलेक्ट्रॉन का द्रव्य तरंग
(\cref{def:g12:quantum-world:debroglie} नीचे) अपने आप चारों ओर बंद होना चाहिए
द \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, और केवल निश्चित \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} फ़िट --- गिटार के तार की तरह, और
\omterm{def:g11:fundamental-interactions:ladder}{एटम} नोट्स का एक अलग सेट है।
\end{remark}

\begin{theorem}[एक संक्रमण का फोटॉन]\label{thm:g12:quantum-world:transition}
एक \omterm{def:g11:fundamental-interactions:ladder}{एटम} किसी एक को उत्सर्जित या अवशोषित करके ही स्तर बदलता है फोटोन ले जाना
बिल्कुल अंतर: स्तर से गिरना $ p $ निचले स्तर तक $ n $ का उत्सर्जन करता है
$ h\nu = E_p - E_n $; से चढ़ना $ n $ को $ p $ उसी को अवशोषित कर लेता है ऊर्जा.
\end{theorem}

\begin{proof}
\omterm{thm:g10:energy-conservation:conservation}{ऊर्जा संरक्षण}, एक समय में एक दाना: फोटोन एकमात्र मुद्रा है
प्रकाश स्वीकार करता है, और एटमकी पुस्तकें संतुलित होनी चाहिए।
\end{proof}

\begin{remark}[रेखा स्पेक्ट्रा की व्याख्या की गई]\label{rem:g12:quantum-world:lines}
यहाँ उत्तर दिया गया है जिसका वादा किया गया है \cref{ch:g10:light-spectra}: एक गर्म गैस निकलती है
केवल आवृत्तियाँ $(E_p - E_n)/h $ अपनी सीढ़ी का --- एक बारकोड
चमकीली रेखाएँ --- और एक ठंडी गैस बिल्कुल उन्हीं आवृत्तियों को चुरा लेती है \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी}, अंधेरे अवशोषण लाइनों से मेल खाते मुद्रण। प्रत्येक तत्व का अपना है
सीढ़ी, इसलिए इसका अपना बारकोड है --- हम तारों की रोशनी कैसे पढ़ते हैं।
\end{remark}

\begin{example}[हाइड्रोजन की दृश्य रेखाएँ]\label{ex:g12:quantum-world:balmer}
परिवर्तन समाप्त हो रहे हैं $ n = 2$ दृश्य में गिरना. साथ $ E \,(\unit{eV}) =
1240/\lambda\,(\unit{nm})$: $3 \to 2$: \qty{1.89}{eV}, \qty{656}{nm} (लाल);
$4 \to 2$: \qty{2.55}{eV}, \qty{486}{nm} (नीले हरे); $5 \to 2$:
\qty{2.86}{eV}, \qty{434}{nm} (बैंगनी)। में संक्रमण $ n = 1$ से अधिक
\qty{10}{eV}: सभी \omterm{def:g10:light-spectra:wavelength}{पराबैंगनी}.
\end{example}

\begin{remark}[लेज़र, एक सांस में]\label{rem:g12:quantum-world:laser}
ए फोटोन बिल्कुल $ E_p - E_n $ एक गुजर रहा है \emph{उत्साहित} \omterm{def:g11:fundamental-interactions:ladder}{एटम} इसे ट्रिगर कर सकते हैं
एक समान उत्सर्जित करने के लिए फोटोन --- वही ऊर्जा, एक ही दिशा, \omterm{def:g12:oscillators-and-time:phase}{चरण} में। रखें
अधिक परमाणुओं नीचे से ऊपर और एक फोटोन क्लोनों का एक हिमस्खलन बन जाता है:
\emph{उत्तेजित उत्सर्जन}\index{उत्तेजित उत्सर्जन} --- और ए \omterm{def:g11:color-light-sources:lamps}{लेज़र} थोड़ा ज्यादा है.
\end{remark}

\begin{proposition}[इलेक्ट्रॉन हस्तक्षेप करते हैं]\label{prop:g12:quantum-world:interference}
एक डबल स्लिट के माध्यम से एक समय में एक इलेक्ट्रॉन भेजें
(\cref{ch:g12:light-as-wave}): प्रत्येक एक तेज बिंदु के रूप में उतरता है --- ए
कण. लेकिन जैसे-जैसे हजारों जमा होते हैं, बिंदु खींचते जाते हैं \omterm{def:g12:light-as-wave:fringes}{हस्तक्षेप किनारे}:
प्रत्येक इलेक्ट्रॉन, अकेले, एक के रूप में गुजरता है \omterm{def:g10:signals-and-waves:wave}{लहर} के माध्यम से \emph{दोनों} स्लिट्स क्रिस्टल
इलेक्ट्रॉन किरणें भी विवर्तित होती हैं (डेविसन और जर्मर, 1927)।
\end{proposition}

\begin{definition}[डी ब्रोगली तरंग दैर्ध्य]\label{def:g12:quantum-world:debroglie}
का हर कण \omterm{def:g12:newtons-laws:momentum}{गति} $ p = mv $ के रूप में यात्रा करता है \emph{बात लहर}\index{बात लहर} का \emph{डी ब्रोगली तरंग दैर्ध्य}\index{डी ब्रोगली तरंग दैर्ध्य}
\[
\lambda = \frac{h}{p} = \frac{h}{mv}.
\]
\end{definition}
\admitted

\begin{example}[आप कभी विचलित क्यों नहीं होते?]\label{ex:g12:quantum-world:never}
एक इलेक्ट्रॉन त्वरित हो गया \qty{100}{V} है $ E_k = \qty{1.60e-17}{J}$,
$ p = \sqrt{2m_e E_k} = \qty{5.4e-24}{kg.m/s}$, इसलिए $\lambda \approx
\qty{0.12}{nm}$ --- का अंतर परमाणुओं एक क्रिस्टल में, जो इसलिए कार्य करता है
इसके दोहरे स्लिट के रूप में। ए \qty{58}{g} टेनिस बॉल पर \qty{25}{m/s}: $\lambda =
6.63\times10^{-34}/1.45 \approx \qty{4.6e-34}{m}$, कुछ $10^{19}$ बार
ए से छोटा \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} --- कोई भी भट्ठा इसे कभी भी झालरदार नहीं दिखाएगा। $ h $ ऐसा है
छोटा है कि क्वांटम दानेदारता केवल सतह पर है एटमका पैमाना.
\end{example}

\begin{example}[इलेक्ट्रॉन माइक्रोस्कोप]\label{ex:g12:quantum-world:microscope}
\omterm{def:g12:light-as-wave:diffraction}{विवर्तन} से बहुत छोटे विवरण को हल करने से मना करता है \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} इस्तेमाल किया गया
(\cref{ch:g12:light-as-wave}): दृश्यमान प्रकाश निकट रुक जाता है \qty{0.5}{\micro m}.
इलेक्ट्रॉनों पर \qty{10}{kV} पास होना $\lambda \approx \qty{12}{pm}$ --- दसियों
हज़ारों गुना छोटा: इलेक्ट्रॉन सूक्ष्मदर्शी छवि वायरस, कोशिका
मशीनरी, यहां तक कि एकल भी परमाणुओं. द्रव्य तरंग भौतिकी को एक नई आँख सौंपी।
\end{example}

\section{समस्या: सौर-पैनल भौतिकी प्रयोगशाला}

\begin{problem}[{सप्ताहांत समस्या --- सौर-पैनल भौतिकी प्रयोगशाला:
$10^{21}$ एक छत पर अनाज हर सेकंड पीता है, एक फोटोकेल जो मापता है
\omterm{def:g12:quantum-world:photon}{प्लैंक स्थिरांक}, एक हाइड्रोजन लैंप पंक्ति दर पंक्ति पढ़ता है --- एक सप्ताहांत, एक $ h $}]
\label{pb:g12:quantum-world:1}
आपकी \omterm{def:g10:universal-gravitation:satellite}{कक्षा} सप्ताहांत के लिए स्कूल की छत पर सौर पैनल लगाती है।
डेटा: दोपहर में सौर विकिरण \qty{1.0e3}{W/m^2}; मतलब सौर \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}
\qty{550}{nm}; पैनल क्षेत्र \qty{1.0}{m^2}, \omterm{def:g10:energy-conservation:efficiency}{क्षमता} $20\%$; दिन
के समतुल्य प्रदान करता है \qty{5.0}{h} पूर्ण सूर्य का.

\medskip
\noindent\textbf{Part I --- फोटोन bookkeeping.}
\begin{enumerate}
  \item की गणना करें ऊर्जा एक माध्य सौर का फोटोन, में \omterm{def:g11:work-of-force:work}{जूल} और इलेक्ट्रॉनवोल्ट.
  \item की संख्या निकालें \omterm{def:g12:quantum-world:photon}{फोटॉनों} दोपहर के समय प्रति सेकंड पैनल पर प्रहार करना।
  \item पैनल की गणना करें \omterm{prop:g11:circuits-and-power:power}{विद्युत शक्ति} दोपहर को, फिर ऊर्जा
        दिन भर में उत्पादित, में \unit{kW.h}.
  \item पैनल का करंट बिल्कुल स्मूथ दिखता है। प्रश्न 2 का उपयोग करते हुए,
        बताएं कि अनाज-दर-अनाज का आगमन अदृश्य क्यों है।
  \item सूर्य का प्रकाश अधिक वहन करता है \omterm{def:g10:light-spectra:wavelength}{अवरक्त} ऊर्जा बजाय \omterm{def:g10:light-spectra:wavelength}{पराबैंगनी}, फिर भी केवल
        \omterm{def:g10:light-spectra:wavelength}{पराबैंगनी} जलती हुई त्वचा (बंधन) $\sim \qtyrange{3}{4}{eV}$). अनाज में समझाओ.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Measuring $ h $ with the kit's photocell.}
किट के फोटोसेल में एक अज्ञात कैथोड है। फिल्टर देना:
\begin{center}\small
\begin{tabular}{lcccc}
$\lambda $ (\unit{nm}) & 365 & 405 & 436 & 546\\
$ U_s $ (\unit{V}) & 1.50 & 1.16 & 0.94 & 0.37
\end{tabular}

\begin{enumerate}[resume]
  \item क्या करता है \omterm{def:g12:quantum-world:stopping}{निरोधी विभव} उपाय? संबंधित $ U_s $, $\nu $, $ W_0$, $ h $.
  \item चारों को रूपांतरित करें \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} आवृत्तियों के लिए (में $10^{14}\,\unit{Hz}$).
  \item साजिश क्यों रचनी चाहिए $ eU_s $ ख़िलाफ़ $\nu $ एक सीधी रेखा बताएं?
        बिंदुओं के संरेखण की जाँच करें.
  \item चरम बिंदुओं से, ढलान की गणना करें: आपका माप $ h $.
  \item कैथोड का अनुमान लगाएं \omterm{thm:g12:quantum-world:einstein}{कार्य फलन} (\unit{eV}), \omterm{prop:g12:quantum-world:facts}{दहलीज
        आवृत्ति} और दहलीज \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}.
  \item इससे सूर्य के प्रकाश का कौन सा भाग इलेक्ट्रॉनों को बाहर निकाल सकता है
        कैथोड --- और कौन सा आधा \omterm{def:g12:sound-acoustics:timbre}{स्पेक्ट्रम} क्या यह इसके लिए बेकार है?
  \item अपनी तुलना करें $ h $ को \qty{6.63e-34}{J.s}: प्रतिशत में सापेक्ष त्रुटि?
\end{enumerate}

\noindent\textbf{Part III --- The hydrogen calibration lamp.}
किट का \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} स्केल की जाँच हाइड्रोजन लैंप से की जाती है,
$ E_n = -13.6/n^2$ \unit{eV}.
\begin{enumerate}[resume]
  \item गणना करना $ E_2$, $ E_3$, $ E_4$ और $ E_5$.
  \item की गणना करें फोटोन संक्रमण की ऊर्जा $3 \to 2$, $4 \to 2$, $5 \to 2$.
  \item तीन का अनुमान लगाएं \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} और प्रत्येक पंक्ति को रंग दें।
  \item दीपक क्यों निकलता है? \emph{पंक्तियां} निरंतर के बजाय
        इंद्रधनुष? एक वाक्य, \omterm{def:g10:refraction:fiber}{मुख्य} विचार का नामकरण।
  \item क्या लैंप का हाइड्रोजन प्रकाश को अवशोषित कर सकता है? \qty{500}{nm}? सीढ़ी से औचित्य सिद्ध करें।
\end{enumerate}

\noindent\textbf{Part IV --- Inspecting the cells: the electron \omterm{def:g10:signals-and-waves:wave}{लहर}.}
कोशिका में सूक्ष्म दरारें काफी नीचे होती हैं \qty{0.5}{\micro m} प्रकाश की पहुंच
सूक्ष्मदर्शी; लैब का स्कैनिंग इलेक्ट्रॉन माइक्रोस्कोप चलता है \qty{10}{kV}.
\begin{enumerate}[resume]
  \item इलेक्ट्रॉनों की \omterm{def:g12:newtons-laws:momentum}{गति} की गणना करें; जांचें कि यह नीचे रहता है $0.25\,c $ (शास्त्रीय सूत्र).
  \item उनकी गणना करें \omterm{def:g12:newtons-laws:momentum}{गति} और \omterm{def:g12:quantum-world:debroglie}{डी ब्रोगली तरंग दैर्ध्य}.
  \item संकल्प तराजू के साथ \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य}: हरी बत्ती पर लाभ की गणना करें
        (\qty{550}{nm}); सप्ताहांत के दो अंकों के साथ समापन करें --- आपका $ h$
        और यह लाभ.
\end{enumerate}
