\chapter{कार्य एवं यांत्रिक ऊर्जा}\label{ch:g12:work-and-energy}

सलाद के कटोरे में एक मार्बल डालें: यह गोता लगाता है, नीचे से ऊपर \omterm{rem:g12:work-and-energy:escape}{निकल जाता है},
अपनी आरंभिक ऊँचाई पर चढ़ता है, झिझकता है, पीछे मुड़ता है --- और प्रत्येक
उलटफेर की भविष्यवाणी एक वक्र (कटोरे की प्रोफ़ाइल) और एक से की जा सकती है
क्षैतिज रेखा (संगमरमर की) ऊर्जा), \omterm{def:g12:newtons-laws:momentum}{गति} का कोई समीकरण हल नहीं हुआ।
पिछले साल की कीमत बल सीधी रेखाओं पर (\cref{ch:g11:work-of-force})
और विरुद्ध संतुलित गतिज \omterm{def:g12:work-and-energy:ep}{संभावित ऊर्जा}
(\cref{ch:g11:mechanical-energy}); इस वर्ष बहीखाता चल रहा है
हर जगह: घुमावदार रास्ते, झरने, परिदृश्य एक नज़र में।

\section{किसी भी रास्ते पर काम करें}

\begin{definition}[एक पथ पर काम करें]\label{def:g12:work-and-energy:path}
चलो ए बल $\vect F $ किसी भी पथ पर चलते हुए शरीर पर कार्य करें $ A $ को
$ B $: पथ को विस्थापन में काटें $\vect{\dd\ell}$ इतना छोटा कि
प्रत्येक सीधा और है $\vect F $ इसके साथ बमुश्किल परिवर्तन होता है। हर कदम कमाता है
छोटा \omterm{def:g11:work-of-force:work}{काम} $\scal{F}{\dd\ell} = F\,\dd\ell\cos\theta $
(\cref{ch:g11:work-of-force}), और यह
\emph{रास्ते पर काम करो}\index{रास्ते पर काम करो} है
उनका योग, $ W_{AB}(\vect F) = \sum \scal{F}{\dd\ell}$ --- जो एक पर
\omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक के साथ सीधा रास्ता बल पिछले वर्ष के बराबर गिर गया
$ F \times AB \times \cos\theta $.
\end{definition}

\begin{proposition}[भार का कार्य, कोई भी पथ]\label{prop:g12:work-and-energy:weight}
साथ में \emph{कोई} से पथ $ A $ (ऊंचाई $ z_A $) को $ B $ (ऊंचाई
$ z_B $), द \omterm{def:g10:universal-gravitation:weight}{वज़न} एक जन का $ m $ करता है \omterm{def:g11:work-of-force:work}{काम}
$ W_{AB}(\vect P) = mg\,(z_A - z_B)$: बूँद प्रवेश करती है, मार्ग कभी नहीं।
\end{proposition}

\begin{proof}
प्रत्येक छोटा कदम योगदान देता है $ mg \times (\text{small drop})$ (अंतिम
वर्ष की सीधे-खंड गणना); दूरबीन को बूँदें
$ z_A - z_B $. घुमावदार मामला, तब स्वीकार किया गया, वितरित किया जाता है।
\end{proof}

\begin{definition}[रूढ़िवादी और गैर-रूढ़िवादी ताकतें]\label{def:g12:work-and-energy:conservative}
ए बल है \emph{रूढ़िवादी}\index{रूढ़िवादी} यदि यह है \omterm{def:g11:work-of-force:work}{काम}
प्रत्येक पथ पर दो बिंदुओं के बीच का मान समान है (प्रत्येक चक्कर पर शून्य)।
यात्रा), \emph{गैर रूढ़िवादी}\index{गैर रूढ़िवादी}
अन्यथा। \omterm{def:g10:universal-gravitation:weight}{वज़न} और वसंत बल (अगला भाग): रूढ़िवादी.
रपट \omterm{def:g10:inertia:inventory}{टकराव} नहीं है: \omterm{def:g12:newtons-laws:momentum}{गति} के विपरीत, यह करता है $ W = -fL $ एक पर
लंबाई का पथ $ L $ --- प्रति टोल \omterm{def:g10:orders-of-magnitude:unit}{मीटर}, कभी वापस नहीं किया गया।
\end{definition}

\begin{example}[झोपड़ी तक जाने के लिए दो रास्ते]\label{ex:g12:work-and-energy:trails}
ए \qty{75}{kg} पदयात्री लाभ \qty{300}{m} एक ढलान से \qty{800}{m} निशान
या \qty{2.4}{km} स्विचबैक का. \omterm{def:g10:universal-gravitation:weight}{वज़न} प्रभार
$-75 \times 9.81 \times 300 \approx \qty{-2.2e5}{J}$ दोनों पर; ए
\qty{30}{N} \omterm{def:g10:inertia:inventory}{टकराव} बल, \qty{-2.4e4}{J} एक पर, \qty{-7.2e4}{J}
दूसरे पर। \omterm{def:g10:universal-gravitation:weight}{वज़न} वंश पर रिफंड; टकरावका बिल है
गर्मी के रूप में चला गया, दोनों तरह से।
\end{example}

\section{संभावित ऊर्जा}

\begin{definition}[संभावित ऊर्जा]\label{def:g12:work-and-energy:ep}
प्रत्येक को \omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी} बल एक एक जोड़ता है
\emph{संभावित ऊर्जा}\index{संभावित ऊर्जा}
$ E_p $, केवल स्थिति के आधार पर एक संख्या, द्वारा परिभाषित
$ W_{AB} = E_p(A) - E_p(B)$: \omterm{def:g11:work-of-force:work}{काम} वितरित करना संभावित है ऊर्जा खर्च किया गया।
फिक्सिंग $ E_p = 0$ एक चुने हुए संदर्भ बिंदु पर इसे हर जगह ठीक करता है; द
चॉइस सभी मानों को एक \omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक द्वारा बदल देती है जो प्रत्येक से रद्द हो जाता है
अंतर. के लिए \omterm{def:g10:universal-gravitation:weight}{वज़न} यह पिछले वर्ष का परिणाम देता है $ E_p = mgz $,
अब किसी भी पथ पर मान्य है. \omterm{def:g10:inertia:inventory}{टकराव} नं. का मालिक हो सकता है $ E_p $: एक मार्ग-निर्भर
बिल में समापनबिंदु संख्याओं का कोई अंतर नहीं है --- केवल \omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी} ताकतें
पुनर्प्राप्तिपूर्वक संग्रहित करें, ठीक वही जो नाम में संरक्षित है।
\end{definition}

\begin{proposition}[लोचदार ऊर्जा क्षमता]\label{prop:g12:work-and-energy:elastic}
का एक झरना \omterm{def:g12:oscillators-and-time:spring}{कठोरता} $ k $ (\unit{N/m}) द्वारा फैलाया या संपीड़ित किया गया $ x $
के साथ वापस खींचता है बल $ F = kx $
(\cref{ch:g12:oscillators-and-time}); \omterm{def:g11:work-of-force:work}{काम} यह वापसी प्रदान करता है
आराम करने के लिए---यह
\emph{लोचदार ऊर्जा क्षमता}\index{लोचदार ऊर्जा क्षमता} --- है
$ E_p = \tfrac12\, k x^2$.
\end{proposition}

\begin{proof}
 \omterm{def:g12:oscillators-and-time:pendulum}{बहाल बल} \omterm{def:g12:newtons-laws:momentum}{गति} के साथ कार्य करता है, इसलिए $\sum F\,\dd\ell $ है
द \emph{के अंतर्गत क्षेत्र $ F $--$ x $ ग्राफ}: आधार का एक त्रिकोण $ x $ और
ऊंचाई $ kx $, क्षेत्र $\tfrac12 x \times kx $. बल प्रत्येक खिंचाव पर
बाहर और पीछे वही है: \omterm{def:g11:work-of-force:work}{काम} अंतिम बिंदु, स्प्रिंग द्वारा तय किया गया बल
\omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी}, $ E_p $ अच्छी तरह से परिभाषित.
\end{proof}

\section{दो ऊर्जा प्रमेय}

\begin{theorem}[गतिज-ऊर्जा प्रमेय]\label{thm:g12:work-and-energy:ketheorem}\index{गतिज-ऊर्जा प्रमेय}
किसी द्रव्यमान के पिंड की किसी भी \omterm{def:g12:newtons-laws:momentum}{गति} के लिए $ m $ से $ A $ को $ B $ --- घुमावदार,
लूप्ड, त्रि-आयामी ---
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
कुल योग सब से ऊपर चल रहा है बल शरीर पर क्रिया करना.
\end{theorem}

\begin{proof}
तब से $ v^2 = \scal{v}{v}$, इस वर्ष के गणित का उत्पाद नियम
का व्युत्पन्न देता है $ v^2$ जैसा $2\,\scal{a}{v}$, इसलिए
$\dd E_k/\dd t = m\,\scal{a}{v} = \scal{F_{\text{tot}}}{v}$ द्वारा
\omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम} (\cref{ch:g12:newtons-laws}). संक्षेप में $\dd t $
शरीर चलता है $\vect{\dd\ell} = \vect v\,\dd t $: $ E_k $ लाभ प्राप्त करता है
छोटा \omterm{def:g11:work-of-force:work}{काम} का \cref{def:g12:work-and-energy:path}; \omterm{def:g12:oscillators-and-time:phase}{चरण}ों का सारांश
का योग है \omterm{def:g11:work-of-force:work}{काम} करता है --- पिछले वर्ष का स्वीकृत विवरण, पूर्ण रूप से अर्जित।
\end{proof}

\begin{proposition}[तात्कालिक शक्ति]\label{prop:g12:work-and-energy:power}
हर पल ए बल $\vect F $ वेग के शरीर पर $\vect v $
पहुंचाता है \omterm{def:g11:work-of-force:power}{शक्ति} $ P = \scal{F}{v} = Fv\cos\theta $, और
$\dd E_k/\dd t $ का योग है पॉवर्स के सभी बल: पिछले साल का
स्थिर-वेग सूत्र, किसी भी \omterm{def:g12:newtons-laws:momentum}{गति} के प्रत्येक क्षण पर सटीक।
\end{proposition}

\begin{proof}
पिछले प्रमाण को पढ़ें: प्रत्येक बल फ़ीड
$\scal{F}{\dd\ell} = \scal{F}{v}\,\dd t $ में $ E_k $ दौरान $\dd t $.
\end{proof}

\begin{theorem}[यांत्रिक-ऊर्जा प्रमेय]\label{thm:g12:work-and-energy:emtheorem}\index{यांत्रिक-ऊर्जा प्रमेय}
होने देना $ E_m = E_k + E_p $, कहाँ $ E_p $ की \omterm{def:g12:work-and-energy:ep}{संभावित ऊर्जा} एकत्रित करता है
सभी \omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी} ताकतें पर \omterm{def:g11:work-of-force:work}{काम}. फिर, किसी भी रास्ते पर,
\[
\Delta E_m = W_{AB}(\text{non-conservative forces}),
\]
और $ E_m $ जब भी संरक्षित किया जाता है गै\omterm{def:g12:work-and-energy:conservative}{र-रूढ़िवादी} ताकतें मत करो \omterm{def:g11:work-of-force:work}{काम}.
\end{theorem}

\begin{proof}
विभाजित करें \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} प्रमेय: $\Delta E_k = W_{\text{cons}} +
W_{\text{nc}} = -\Delta E_p + W_{\text{nc}}$; कदम $\Delta E_p $ आर-पार।
\end{proof}

\begin{example}[पेंडुलम, ईमानदारी से इस बार]\label{ex:g12:work-and-energy:pendulum}
लंबाई के तार पर एक पेंडुलम बॉब $ L = \qty{2.5}{m}$ पर जारी किया गया है
पर आराम करो $\qty{45}{\degree}$ ऊर्ध्वाधर से. \omterm{def:g10:inertia:inventory}{तनाव},
\omterm{def:g12:newtons-laws:momentum}{गति} के लंबवत, शक्तिहीन है; \omterm{def:g10:universal-gravitation:weight}{वज़न} है \omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी}:
$ E_m $ संरक्षित है \emph{घुमावदार चाप पर}, जो पिछले साल हो सकता था
केवल जांच की जाएगी. बॉब शुरू होता है $ h = L(1 - \cos\qty{45}{\degree}) =
\qty{0.73}{m}$ ऊपर, तो नीचे $ v = \sqrt{2gh} \approx
\qty{3.8}{m/s}$.
\end{example}

\begin{method}[किसी भी गति का ऊर्जा ऑडिट]\label{met:g12:work-and-energy:audit}
\begin{enumerate}
  \item सिस्टम चुनें और सूचीबद्ध करें बल उस पर.
  \item क्रमबद्ध करें: \omterm{def:g12:newtons-laws:momentum}{गति} के लंबवत $\to $ नहीं \omterm{def:g11:work-of-force:work}{काम}; \omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी}
        $\to $ एक $ E_p $ में $ E_m $; बाकी का $\to $ $ W_{\text{nc}}$
        (फिसलना \omterm{def:g10:inertia:inventory}{टकराव}: $-fL $, $ L $ = पथ की लंबाई).
  \item लिखना $\Delta E_m = W_{\text{nc}}$ डेटा बिंदु और के बीच
        प्रश्न बिंदु; हल करना। ऊर्जा उत्तर ``कितनी तेजी से, कैसे
        उच्च''; केवल समय और त्वरण को \omterm{def:g10:inertia:force}{न्यूटन} के नियमों की आवश्यकता है।
\end{enumerate}
\end{method}

\section{ऊर्जा आरेख}

\begin{definition}[ऊर्जा आरेख, महत्वपूर्ण मोड़]\label{def:g12:work-and-energy:diagram}
एक अक्ष के अनुदिश \omterm{def:g12:newtons-laws:momentum}{गति} के लिए $ x $ संरक्षित के साथ $ E_m $, द
\emph{ऊर्जा आरेख}\index{ऊर्जा आरेख} वक्र आलेखित करता है $ E_p(x)$
और क्षैतिज रेखा $ E_m $. तब से $ E_k = E_m - E_p \geq 0$, द
\omterm{def:g12:newtons-laws:momentum}{गति} वहीं तक सीमित है जहां वक्र रेखा के नीचे स्थित है; अंतर
उनके बीच है $ E_k $, सीधे पढ़ें। ए पर
\emph{मोड़}\index{मोड़}, जहां वक्र रेखा से मिलता है,
$ v = 0$ और \omterm{def:g12:newtons-laws:momentum}{गति} उलट जाती है.
\end{definition}

\begin{proposition}[परिदृश्य से बल; संतुलनों]\label{prop:g12:work-and-energy:equilibria}
 \omterm{def:g12:work-and-energy:conservative}{रूढ़िवादी} बल साथ में $ x $ का ढलान शून्य से कम है $ E_p $:
$ F = -\,\dd E_p/\dd x $ --- यह इंगित करता है \emph{ढलान} आरेख पर,
और गायब हो जाता है जहां स्पर्श रेखा क्षैतिज होती है: एक संतुलन। ए
न्यूनतम $ E_p $ है \emph{स्थिर}\index{स्थिर} --- द
विस्थापित शरीर को घाटी में एक गेंद की तरह पीछे धकेल दिया जाता है; एक अधिकतम है
\emph{अस्थिर}\index{अस्थिर} --- इसे दूर धकेल दिया जाता है, ए
पहाड़ी की चोटी पर गेंद.
\end{proposition}

\begin{proof}
एक छोटे कदम पर $\dd x $  बल \omterm{def:g11:work-of-force:work}{काम} करता है $ F\,\dd x = E_p(x) -
E_p(x + \dd x) = -\dd E_p $; द्वारा विभाजित करें $\dd x $. दोनों तरफ ढलान
एक घाटी की ओर इशारा करता है; एक पहाड़ी की चोटी के आसपास, बाहर।
\end{proof}

\begin{remark}[दोलन और पलायन]\label{rem:g12:work-and-energy:escape}
के छोटे दोलन \cref{ch:g12:oscillators-and-time} पर रहते हैं
के तलवे $ E_p $ घाटियाँ और पृथ्वी से दूर गुरुत्वाकर्षण कमजोर हो जाता है:
$ E_p(r)$ वक्र एक परिमित छत की ओर और अधिक धीरे-धीरे चढ़ता है
सिकुड़ते बल-ग्राफ के अंतर्गत क्षेत्र सीमित है)। एक रॉकेट जिसका $ E_m $
छत तक पहुँचता है कभी नहीं मिलता \omterm{def:g12:work-and-energy:diagram}{मोड़}: यह
\emph{निकल जाता है}\index{निकल जाता है} --- के बारे में \qty{11.2}{km/s} से
ज़मीन (\cref{exo:g12:work-and-energy:13}); उसके नीचे यह है
बाध्य. अधिकतम सीमा की गणना वर्ष 1 खंड में ही की जाती है।
\end{remark}

\begin{example}[लूप-द-लूप]\label{ex:g12:work-and-energy:loop}
ऊँचाई से निकला हुआ एक संगमरमर $ H $ के एक ऊर्ध्वाधर लूप में तट
त्रिज्या $ R $. वृत्ताकार गतिकी (\cref{ch:g12:newtons-laws}) मांग करता है
$ v_{\text{top}}^2 \geq gR $ शीर्ष पर --- अंतिम द्वारा उधार लिया गया डेटा
साल भर की रोलर-कोस्टर समस्या, अब हमारी। \omterm{def:g10:relative-motion:frame}{आराम से} संरक्षण $ H $
शीर्ष तक (ऊंचाई) $2R $) देता है $ v_{\text{top}}^2 = 2g(H - 2R)$, इसलिए
$ H \geq \tfrac52 R $: लूप शीर्ष से आधा त्रिज्या ऊपर, जो भी हो
द्रव्यमान --- वास्तविक, रगड़ दुनिया में थोड़ा अधिक।
\end{example}

\section{समस्या: स्की जंप}

\begin{problem}[{सप्ताहांत समस्या --- स्की जंप: एक इन-रन ऑडिटेड
एक स्पीड गन से, एक फ्री-फ़ॉल उड़ान से, एक खड़ी ढलान की कोमल ज्यामिति से
लैंडिंग, और टॉवर एक डिजाइनर को घर्षण बिल के बाद खरीदना होगा}]
\label{pb:g12:work-and-energy:1}
एक ओलंपिक स्की जम्पर ($ m = \qty{70}{kg}$, स्की शामिल) से शुरू होता है
एक गेट से आराम करो $ H = \qty{45}{m}$ टेकऑफ़ टेबल के ऊपर, स्लाइड
$ L = \qty{90}{m}$ घुमावदार इन-रन, और \omterm{def:g11:lenses-and-eye:magnifier}{ताल}िका को क्षैतिज रूप से छोड़ देता है;
वहाँ एक स्पीड गन पढ़ती है $ v_0 = \qty{26}{m/s}$. लैंडिंग बिंदु स्थित है
$ h = \qty{40}{m}$ निचला, ए पर $\qty{35}{\degree}$ ढलान। भाग II
और III वायु \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} की उपेक्षा करते हैं।

\medskip\noindent\textbf{Part I --- The in-run.}
\begin{enumerate}
  \item घर्षण रहित टेकऑफ़ \omterm{def:g12:newtons-laws:momentum}{गति} की भविष्यवाणी करें; का वक्र क्यों है
        इन-रन अप्रासंगिक?
  \item \omterm{def:g11:lenses-and-eye:magnifier}{ताल}िका में संदर्भ: गणना करें $ E_m $ गेट पर और टेकऑफ़ पर।
  \item गणना करना $\Delta E_m $; कौन सा प्रमेय दोषियों का नाम बताता है, और कौन?
  \item औसत निकालें \omterm{def:g11:work-of-force:sign}{प्रतिरोधक} बल (बर्फ प्लस हवा) दौड़ के ऊपर।
  \item का कितना अंश $ E_m $ खो गया है? स्की पर मोम क्यों लगाएं और झुकें?
\end{enumerate}

\medskip\noindent\textbf{Part II --- The flight.}
\begin{enumerate}[resume]
  \item केवल \omterm{def:g10:universal-gravitation:weight}{वज़न} उड़ान में कार्य करता है: कौन सी \omterm{def:g12:newtons-laws:momentum}{गति}
        \cref{ch:g12:projectile-motion} क्या यह? वेग घटक
        लैंडिंग पर?
  \item लैंडिंग \omterm{def:g12:newtons-laws:momentum}{गति}, दो बार: घटकों से, और से $ E_m $
        संरक्षण। में \unit{km/h}?
  \item उड़ान के समय और उड़ान की क्षैतिज दूरी की गणना करें।
  \item \omterm{def:g11:lenses-and-eye:images}{असली} कूदने वाले पंख की तरह हवा में उड़ते हैं, बहुत दूर तक उड़ते हैं:
        प्रसारित कर सकते हैं \emph{उठाना} लैंडिंग की \omterm{def:g12:newtons-laws:momentum}{गति}? के साथ बहस करें
        \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} प्रमेय.
\end{enumerate}

\medskip\noindent\textbf{Part III --- The landing.}
\begin{enumerate}[resume]
  \item क्षैतिज के नीचे लैंडिंग वेग के कोण की गणना करें।
  \item इसे समानांतर और लंबवत घटकों में विघटित करें
        ढलान.
  \item पैर केवल लंबवत भाग को अवशोषित करते हैं: इसकी गणना करें
        \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा}; इसकी तुलना कुल से करें.
  \item समतल ज़मीन बहुत कुछ सोख लेगी: ऊर्ध्वाधर गिरावट क्या है
        प्रत्येक लैंडिंग समतुल्य ($ h_{\text{eq}} = v_\perp^2/2g $,
        सम्मान\ $ v^2/2g $)?
  \item एक वाक्य: लैंडिंग पहाड़ियाँ खड़ी और घुमावदार क्यों हैं?
\end{enumerate}

\medskip\noindent\textbf{Part IV --- The designer's tower.}
एक छोटी पहाड़ी पर केवल टेकऑफ़ की आवश्यकता होती है $ v_0' = \qty{24}{m/s}$; इसका
सीधे इन-रन पर उतरता है $\qty{35}{\degree}$, तो गेट की ऊंचाई $ H'$
मतलब ट्रैक की लंबाई $ H'/\sin\qty{35}{\degree}$; वही \qty{80}{N} खींचना।
\begin{enumerate}[resume]
  \item एक घर्षण रहित डिज़ाइनर द्वारा बनाए जाने वाले गेट की ऊंचाई की गणना करें।
  \item लिखना \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} प्रमेय, गेट टू टेबल, के लिए $ H'$.
  \item के लिए हल करें $ H'$.
  \item कितने प्रतिशत किया \omterm{def:g10:inertia:inventory}{टकराव} टावर बढ़ाएँ?
  \item द्रव्यमान अब रद्द नहीं होता: पुनःगणना करें $ H'$ एक के लिए
        \qty{60}{kg} जम्पर. ऊँचे टावर की जरूरत किसे है और क्यों?
  \item पंचलाइन: टावर की घोषणा की, और टकरावइसका हिस्सा है.
\end{enumerate}
