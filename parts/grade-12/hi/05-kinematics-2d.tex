\chapter{दो आयामों में गतिकी}\label{ch:g12:kinematics-2d}

एक कार को स्थिर गति से चक्कर लगाते हुए देखें \qty{30}{km/h}: सुई
कभी हिलता नहीं, फिर भी प्रत्येक यात्री को बग़ल में खींचा हुआ महसूस होता है --- गति
बदल रहा है, और यह नहीं है \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}. प्रति क्षण एक संख्या
अधिक समय पर्याप्त है: समतल में गति के लिए तीरों की आवश्यकता होती है। यह अध्याय
एक गतिमान बिंदु को तीन --- स्थिति, वेग, त्वरण से सुसज्जित करता है
--- व्युत्पन्न द्वारा जंजीर, और गति के पूरे परिवारों को पढ़ता है
उनकी ज्यामिति.

\section{स्थिति वेक्टर}

गति केवल बताए गए के सापेक्ष मौजूद होती है \omterm{def:g10:relative-motion:frame}{संदर्भ फ्रेम} --- एक कठोर
बॉडी प्लस एक घड़ी (\cref{ch:g10:relative-motion}). इस साल नया है
परिशुद्धता: फ्रेम के साथ एक मूल संलग्न करें $O $ और दो लंबवत
स्नातक अक्ष $ Ox $, $ Oy $, और रिकॉर्ड करें कि प्रत्येक बिंदु कहां है
तत्काल, कार्यों की एक जोड़ी के रूप में।

\begin{definition}[स्थिति वेक्टर]\label{def:g12:kinematics-2d:position}
एक मूल से सुसज्जित फ्रेम में $ O $ और कुल्हाड़ियाँ $ Ox $, $ Oy $, द
\emph{स्थिति वेक्टर}\index{स्थिति वेक्टर} एक गतिशील बिंदु का $ M $
समय पर $ t $ है $\vect{OM}(t)$, निर्देशांक का
$\bigl(x(t),\, y(t)\bigr)$: समय के दो कार्य, प्रति अक्ष एक। द
वक्र का पता लगाया गया $ M $ जैसा $ t $ रन इसका है \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}.
\end{definition}

\section{वेग सदिश}

\begin{definition}[वेग सदिश]\label{def:g12:kinematics-2d:velocity}
थोड़े अंतराल पर बिंदु गति करता है
$\vect{OM}(t+\Delta t) - \vect{OM}(t)$; द्वारा विभाजित करें $\Delta t $ और चलो
अंतराल सिकुड़न: यह आपके गणित से व्युत्पन्न है
पाठ्यक्रम, प्रत्येक समन्वय पर लागू होता है। \emph{वेग वेक्टर}\index{वेग वेक्टर} का $ M $ इसका व्युत्पन्न है
\omterm{def:g12:kinematics-2d:position}{स्थिति वेक्टर}, $\vect v(t) = \bigl(x'(t),\, y'(t)\bigr)$, में
\unit{m/s}. यह आदर्श है $ v = \sqrt{x'^2 + y'^2}$ है
\emph{रफ़्तार}\index{रफ़्तार} --- स्पीडो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} एक नंबर दिखाता है।
\end{definition}

\begin{example}[समतल उड़ान में एक ड्रोन]\label{ex:g12:kinematics-2d:drone}
एक ड्रोन उड़ता है $ x(t) = 4.0\,t $ और $ y(t) = 3.0\,t $ (\omterm{def:g10:orders-of-magnitude:unit}{मीटर},
सेकंड): \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} रेखा $ y = \tfrac34 x $, वेग
$\vect v = (4.0, 3.0)$ हर समय, \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}
$\sqrt{4.0^2 + 3.0^2} = \qty{5.0}{m/s}$, स्थिर।
\end{example}

\begin{proposition}[प्रक्षेपवक्र के स्पर्शरेखा]\label{prop:g12:kinematics-2d:tangent}
हर पल, $\vect v(t)$ के स्पर्शरेखा है \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} पर
की वर्तमान स्थिति $ M $, और यात्रा की दिशा की ओर इशारा करता है।
\end{proposition}

\begin{proof}
विस्थापन का एक राग है \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}; जैसा $\Delta t $ सिकुड़ता है,
तार की दिशा स्पर्शरेखा की दिशा बन जाती है।
\end{proof}

\section{त्वरण वेक्टर}

\begin{definition}[त्वरण वेक्टर]\label{def:g12:kinematics-2d:acceleration}
 \emph{त्वरण वेक्टर}\index{त्वरण वेक्टर} का $ M $ है
इसका व्युत्पन्न वेग सदिश,
$\vect a(t) = \bigl(v_x'(t),\, v_y'(t)\bigr) =
\bigl(x''(t),\, y''(t)\bigr)$, में \unit{m/s^2}: कितनी तेजी से, और अंदर
कौन सी दिशा, वेग सदिश वर्तमान में बदल रहा है.
\end{definition}

\begin{remark}[बिना तेज़ गति के तेज़ करना]\label{rem:g12:kinematics-2d:turning}
$\vect a \neq \vect 0$ जब भी वेग \emph{वेक्टर} परिवर्तन
--- लंबाई में \emph{या दिशा में}: ब्रेक लगाने वाली कार तेज हो जाती है
(पि\omterm{def:g11:color-light-sources:cones}{छड़}ा); लगातार घूमती हुई कार \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} तेजी लाता है (बग़ल में)।
यह मात्रा है बल नियंत्रण
(\cref{ch:g11:forces-and-motion,ch:g12:newtons-laws}).
\end{remark}

\begin{method}[एक क्रोनोफ़ोटोग्राफ़ पढ़ना]\label{met:g12:kinematics-2d:chrono}
ए \emph{क्रोनोफ़ोटोग्राफ़}\index{क्रोनोफ़ोटोग्राफ़} रिकार्ड करता है
पद $ M_0, M_1, M_2, \dots $ समान समय पर एक गतिमान बिंदु का
अंतराल $\Delta t $.
\begin{enumerate}
  \item बिंदु रिक्ति पढ़ता है \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}: समान दूरी, स्थिरांक \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार};
        खोलना, तेज़ करना; बंद करना, धीमा करना।
  \item वेग पर $ M_i $ सममित राग है,
        $\vect v_i \approx \vect{M_{i-1}M_{i+1}}/(2\,\Delta t)$,
        पर खींचा गया $ M_i $, बिंदीदार पथ के स्पर्शरेखा।
  \item त्वरण के लिए $ M_i $: प्रतिलिपि $\vect v_{i-1}$ और
        $\vect v_{i+1}$ को $ M_i $, खींचना
        $\Delta\vect v = \vect v_{i+1} - \vect v_{i-1}$ टिप से टिप;
        तब $\vect a_i \approx \Delta\vect v/(2\,\Delta t)$.
\end{enumerate}
\end{method}

\section{सीधीरेखीय गतियाँ}

एक सीधी रेखा के साथ एक अक्ष पर्याप्त है: $ x(t)$, $ v(t) = x'(t)$,
$ a(t) = v'(t)$, सभी हस्ताक्षरित संख्याएँ।

\begin{definition}[एकसमान सीधीरेखीय गति]\label{def:g12:kinematics-2d:uniform}
एक गति है \emph{एकसमान सीधा}\index{एकसमान सीधा} कब $\vect v $ एक स्थिर सदिश है: सीधा \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र},
स्थिर \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}, $\vect a = \vect 0$. तब $ x(t) = v\,t + x_0$.
\end{definition}

\begin{proposition}[समान रूप से त्वरित सीधीरेखीय गति]\label{prop:g12:kinematics-2d:uarm}
\index{समान रूप से त्वरित गति}%
यदि कोई बिंदु साथ चलता है $ Ox $ साथ \emph{स्थिर} त्वरण $ a $,
से शुरू हो रहा है $ t = 0$ स्थिति से $ x_0$ वेग के साथ $ v_0$, तब
\[
v(t) = a\,t + v_0,
\qquad
x(t) = \tfrac12\,a\,t^2 + v_0\,t + x_0 .
\]
\end{proposition}

\begin{proof}
$ v $ अचर का प्रतिअवकलन है $ a $: $ v(t) = a t + C $, और
$ t = 0$ बल $ C = v_0$. के बदले में $ x $ का प्रतिव्युत्पन्न है
$ a t + v_0$: $ x(t) = \tfrac12 a t^2 + v_0 t + C'$, साथ $ C' = x_0$.
\end{proof}

\begin{example}[ब्रेक लगाने की दूरी]\label{ex:g12:kinematics-2d:braking}
एक कार पर \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $ v_0$ निरंतर मंदी के साथ ब्रेक $ a $
(त्वरण $-a $). यह कब रुकता है $ v(t) = -a t + v_0 = 0$, पर
$ t_s = v_0/a $, कवर किया हुआ $ d = -\tfrac12 a t_s^2 + v_0 t_s =
v_0^2/(2a)$. वर्ग सड़क-सुरक्षा शीर्षक है: दोगुना करना
\omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} \emph{एक साथ जन्म लेनेवाले बच्चे} ब्रेकिंग दूरी --- साथ $ a =
\qty{6.0}{m/s^2}$, \qty{16}{m} से \qty{50}{km/h}, \qty{64}{m} से
\qty{100}{km/h}.
\end{example}

\section{एकसमान गोलाकार गति और फ्रेनेट फ्रेम}

\begin{definition}[एकसमान वृत्तीय गति]\label{def:g12:kinematics-2d:ucm}
एक गति है \emph{एकसमान गोलाकार}\index{एकसमान गोलाकार}
जब \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} त्रिज्या का एक वृत्त है $ R $ और यह \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $ v $ है
स्थिर. \emph{अवधि}\index{अवधि} $ T $ है
एक गोद की अवधि, और \emph{आवृत्ति}\index{आवृत्ति} $ f = 1/T $ (में \unit{Hz}) प्रति सेकंड लैप्स की संख्या:
$ v = 2\pi R/T = 2\pi R f $.
\end{definition}

स्थिर \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} --- फिर भी वेग मुड़ता है: वहाँ \emph{है} एक
त्वरण. कौन सा रास्ता, और कितना बड़ा?

\begin{theorem}[केन्द्राभिमुख त्वरण]\label{thm:g12:kinematics-2d:centripetal}
\index{केन्द्राभिमुख त्वरण}%
में \omterm{def:g11:forces-and-motion:ucm}{एकसमान वृत्तीय गति} त्रिज्या का $ R $ और \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $ v $, द
त्वरण --- कहा जाता है \emph{केंद्र की ओर जानेवाला} --- चलती से अंक
स्थिर मानदंड के साथ, वृत्त के केंद्र की ओर इंगित करें
\[
a = \frac{v^2}{R}.
\]
\end{theorem}

\begin{proof}
अक्षों को वृत्त के केंद्र पर केन्द्रित करें और सेट करें $\omega = v/R $, तो
चाप $ v t $ कोण घटाता है $\omega t $: स्थिति है
$ x(t) = R\cos(\omega t)$, $ y(t) = R\sin(\omega t)$. विभेद करना,
$\vect v = (-R\omega\sin(\omega t),\, R\omega\cos(\omega t))$, का
स्थिर मानदंड $ R\omega = v $; फिर से भेद करना,
$\vect a = -\omega^2\,\vect{OM}$: के विपरीत \omterm{def:g12:kinematics-2d:position}{स्थिति वेक्टर} ---
केंद्र की ओर --- आदर्श का $\omega^2 R = v^2/R $.
\end{proof}

\begin{definition}[फ़्रेनेट फ़्रेम]\label{def:g12:kinematics-2d:frenet}
 \emph{फ़्रेनेट फ़्रेम}\index{फ़्रेनेट फ़्रेम} वर्तमान स्थिति में
किसी वक्र पथ पर एक बिंदु का लंब युग्म होता है \omterm{def:g10:orders-of-magnitude:unit}{इकाई}
वैक्टर $(\vect{u_t}, \vect{u_n})$: $\vect{u_t}$ स्पर्शरेखा, साथ में
गति; $\vect{u_n}$ सामान्य, वृत्त के केंद्र की ओर
त्रिज्या $ R $ वह वहां के वक्र के लिए सबसे उपयुक्त है।
\end{definition}

\begin{proposition}[फ़्रेनेट फ़्रेम में त्वरण]\label{prop:g12:kinematics-2d:frenetacc}
स्थानीय त्रिज्या के किसी भी समतल पथ पर $ R $, पर यात्रा की \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $ v(t)$,
\[
\vect a = a_t\,\vect{u_t} + a_n\,\vect{u_n},
\qquad a_t = \frac{\dd v}{\dd t},
\qquad a_n = \frac{v^2}{R}:
\]
 \emph{स्पर्शरेखीय त्वरण}\index{स्पर्शरेखीय त्वरण}
$ a_t $ बदल देता है \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}; \emph{सामान्य त्वरण}\index{सामान्य त्वरण} $ a_n $ वेग बदल देता है.
\end{proposition}
\admitted

\begin{remark}[जहां सबूत रहता है]\label{rem:g12:kinematics-2d:frenetproof}
हमने अतियों को सिद्ध किया: शुद्ध $ a_t $ एक लाइन पर
(\cref{prop:g12:kinematics-2d:uarm}), शुद्ध $ a_n $ एक वृत्त पर
(\cref{thm:g12:kinematics-2d:centripetal}); सामान्य
अपघटन वर्ष~1 खंड में प्राप्त होता है।
\end{remark}

\begin{example}[एक मोड़ में ब्रेक लगाना]\label{ex:g12:kinematics-2d:bend}
एक कार त्रिज्या के मोड़ में प्रवेश करती है $ R = \qty{50}{m}$ पर $ v = \qty{15}{m/s}$
ब्रेक लगाते समय $\dd v/\dd t = \qty{-3.0}{m/s^2}$: $ a_t =
\qty{-3.0}{m/s^2}$, $ a_n = 15^2/50 = \qty{4.5}{m/s^2}$, इसलिए $ a =
\sqrt{3.0^2 + 4.5^2} \approx \qty{5.4}{m/s^2}$, पीछे की ओर इशारा करते हुए
\emph{और} मोड़ में. टायरों की आपूर्ति एक साथ होनी चाहिए ---
रेसिंग ड्राइवर ब्रेक क्यों लगाते हैं? \emph{पहले} बारी.
\end{example}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:kinematics-2d:9}

में \qty{27.3}{d}. इसकी गणना करें \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} और इसके \omterm{thm:g12:kinematics-2d:centripetal}{केन्द्राभिमुख त्वरण};

\end{exercise}

\end{exercise}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:kinematics-2d:12}
के लिए $ x(t) = R\cos(\omega t)$, $ y(t) = R\sin(\omega t)$: गणना करें
$\vect v $; जाँचें \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} स्थिरांक है $ R\omega $ और
$\vect v \perp \vect{OM}$; गणना $\vect a $, दिखाओ यह है
$-\omega^2\,\vect{OM}$, और ठीक हो जाओ $ a = v^2/R $.
\end{exercise}

द \omterm{def:g10:relative-motion:speed}{औसत गति}. रेखाचित्र $ v(t)$ और दूरी को एक क्षेत्र के रूप में जांचें।
\end{exercise}

\section{समस्या: दुर्घटना रिपोर्ट}

\begin{problem}[{सप्ताहांत समस्या --- दुर्घटना रिपोर्ट: एक डैशकैम
फिल्म को फ्रेम दर फ्रेम पढ़ा जाता है, स्किड मार्क्स से ब्रेकिंग कानून का पुनर्निर्माण किया जाता है, ए
गोलचक्कर जो पकड़ के बारे में गवाही देता है, और किलोमीटर में निर्णय देता है
प्रति घंटा}]
\label{pb:g12:kinematics-2d:1}
एक कार ने एक वैन को टक्कर मार दी जो बगल की सड़क से निकल रही थी; किसी को चोट नहीं पहुंची है,
लेकिन बीमाकर्ता असहमत हैं। अन्वेषक के पास ओवरहेड कैमरा है
फ़ुटेज (एक फ्रेम प्रत्येक) \qty{0.50}{s}), मापे गए स्किड निशान, और
कार का डैशकैम. \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} सीमा है \qty{50}{km/h}.

\medskip
\noindent\textbf{Part I --- Reading the film.}
किसी भी प्रतिक्रिया से पहले फ़्रेम सीधी सड़क के साथ देते हैं:
\begin{center}\small
\begin{tabular}{lccccc}
$ t $ (\unit{s}) & 0.00 & 0.50 & 1.00 & 1.50 & 2.00\\
$ x $ (\unit{m}) & 0.0 & 9.0 & 18.0 & 27.0 & 36.0
\end{tabular}
\end{center}
\begin{enumerate}
  \item रिपोर्ट में पहले फ़्रेम और अक्षों का नाम क्यों होना चाहिए? जो हैं
        यहाँ प्रयोग किया गया?
  \item चार अंतरालों में से प्रत्येक पर औसत वेग की गणना करें।
  \item यह किस वर्ग की गति है? \omterm{def:g11:lenses-and-eye:magnifier}{ताल}िका से औचित्य सिद्ध करें।
  \item कार दो \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $ v_0$ में \unit{m/s} और \unit{km/h}.
  \item इन दो सेकंड के दौरान त्वरण क्या है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The skid marks.}
इस सूखे डामर पर टायर परीक्षण से लगातार ब्रेकिंग धीमी होती है
$ a = \qty{6.0}{m/s^2}$; फिसलन के निशान हैं $ d_b = \qty{27}{m}$ लंबा।
लेना $ t = 0$, $ x = 0$ ब्रेक लगाने की शुरुआत में, अज्ञात \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} $ v_B $.
\begin{enumerate}[resume]
  \item प्रतिअवकलन द्वारा लिखिए $ v(t)$ और $ x(t)$ ब्रेक लगाने के दौरान.
  \item रुकने का समय व्यक्त करें $ t_s $; दिखाओ $ d_b = v_B^2/(2a)$.
  \item परिणाम निकालना $ v_B $ फिसलन के निशान से. भाग~I के अनुरूप?
  \item प्रतिक्रिया समय है \qty{1.0}{s}: प्रतिक्रिया की गणना करें
        दूरी और कुल रुकने की दूरी।
  \item प्रश्न 7 से स्पष्ट करें कि दोगुना क्यों हो रहा है \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} चौगुना
        ब्रेक लगाने की दूरी; गणना $ d_b $ बिल्कुल \qty{50}{km/h}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The roundabout testifies.}
इससे पहले, डैशकैम कार को एक गोल चक्कर वाली लेन में घूमता हुआ दिखाता है
त्रिज्या $ R = \qty{12.5}{m}$ लगातार \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}, एक चौथाई मोड़
\qty{2.5}{s}. सड़क अधिक से अधिक आपूर्ति कर सकती है $ a_n = \mu g $, साथ
$\mu = 0.70$.
\begin{enumerate}[resume]
  \item  \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} वहाँ स्थिर है --- वेग है? कार है
        तेज़ हो रहा है? व्याख्या करना।
  \item कार की गणना करें \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} चौराहे पर.
  \item इसकी गणना करें \omterm{thm:g12:kinematics-2d:centripetal}{केन्द्राभिमुख त्वरण} (मानदंड और दिशा).
  \item अधिकतम नो-स्किड की गणना करें \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}. क्या कार इसके भीतर थी?
  \item दे दो \omterm{def:g12:kinematics-2d:ucm}{अवधि} और \omterm{def:g12:kinematics-2d:ucm}{आवृत्ति} उस पर एक पूर्ण लूप का \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The verdict.}
वैन दिखाई दी $ D = \qty{41}{m}$ तुरंत कार से आगे
ड्राइवर ने प्रतिक्रिया देनी शुरू कर दी.
\begin{enumerate}[resume]
  \item कितने से, में \unit{km/h} और प्रतिशत में, किया \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}
        ब्रेक लगाने से पहले सीमा से अधिक?
  \item कार की गणना करें \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} प्रभाव में.
  \item बिल्कुल रुकने की गणना फिर से करें \qty{50}{km/h} (वही
        प्रतिक्रिया, वही ब्रेक लगाना): क्या कार रुकती है, किस मार्जिन से?
  \item सामान्य रोक-दूरी का सूत्र दीजिए $ d(v)$; क्यों करता है
        \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} दो बार दर्ज करें --- एक बार रैखिक रूप से, एक बार वर्गाकार?
  \item फैसला, एक वाक्य अपनी संख्याओं के साथ: क्या \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार} था
        ड्राइवर कर रहा है, और क्या सीमा पर दुर्घटना हुई होगी?
\end{enumerate}
\end{problem}
