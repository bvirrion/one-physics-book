\chapter{निःशुल्क विद्युत दोलन: आरएलसी}\label{ch:g12:rlc-oscillations}

अटारी रेडियो के भारी डायल को घुमाएँ और स्टेशन फ़ाइल को इसके आगे घुमाएँ
सुई: एक आवाज, वायलिन, \omterm{prop:g12:work-and-energy:equilibria}{स्थिर}, फिर से एक आवाज। वहाँ पैनल के पीछे
नहीं है \omterm{def:g11:work-of-force:sign}{मोटर} और कोई कंप्यूटर नहीं --- ए कुंडल, ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} जिसका अंतर्संबंध
डायल घूमने वाली प्लेटों और हवाई तार की एक लंबाई। यह अध्याय
अंतिम अध्याय के दो को जोड़ता है ऊर्जा टैंक (\cref{ch:g12:rc-rl-circuits})
आमने-सामने और अब तक का सबसे तेज़ पेंडुलम पाया गया: चार्ज स्विंगिंग
आगे और पीछे, प्रति सेकंड लाखों बार तक, एक \omterm{def:g12:newtons-laws:momentum}{गति} से डायल
चुनता है.

\section{एक विद्युत पेंडुलम: एलसी सर्किट}

आरोप ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} $C $ एक को \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} $ U_0$ --- इसकी प्लेटें फिर पकड़ती हैं
$\pm Q_0$ साथ $ Q_0 = C U_0$ --- और कम से $ t = 0$ इसे एक पर स्विच करें
\emph{आदर्श} कुंडल $ L $, शून्य का \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}. कहीं भी कोई स्रोत नहीं
लूप: आगे जो भी होता है, सर्किट अपने आप होता है।

आदर्श कुंडल. समापन $ K $ आवेश को इसके माध्यम से फैलने देता है कुंडल --- और यह कुंडल, परिवर्तन से नफरत है, इसे रुकने नहीं देंगे।}

\begin{proposition}[एलसी समीकरण]\label{prop:g12:rlc-oscillations:equation}
\index{एलसी सर्किट}
आदर्श एलसी लूप में चार्ज $ q(t)$ ऊपरी प्लेट का पालन करता है
\[
L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0,
\qquad\text{i.e.}\qquad
\frac{d^2q}{dt^2} = -\frac{1}{LC}\,q .
\]
\end{proposition}

\begin{proof}
लूप के चारों ओर \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} शून्य का योग
(\cref{prop:g11:circuits-and-power:laws}): $ u_L + u_C = 0$.
\omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} देता है $ u_C = q/C $ और यह कुंडल $ u_L = L\,di/dt $
(\cref{def:g12:rc-rl-circuits:inductor}), साथ $ i = dq/dt $ गिनती
चार्ज ऊपरी प्लेट पर आ रहा है
(\cref{prop:g12:rc-rl-circuits:cdudt}); विकल्प।
\end{proof}

\begin{theorem}[एलसी सर्किट के मुक्त दोलन]\label{thm:g12:rlc-oscillations:oscillation}
\index{विद्युत दोलन}
पर जारी किया गया $ t = 0$ साथ $ q(0) = Q_0$ और $ i(0) = 0$, सर्किट
दोलन करता है:
\[
q(t) = Q_0 \cos\!\left(\frac{2\pi t}{T_0}\right),
\qquad
i(t) = -I_0 \sin\!\left(\frac{2\pi t}{T_0}\right),
\qquad
T_0 = 2\pi\sqrt{LC},
\]
चरम धारा के साथ $ I_0 = 2\pi Q_0/T_0 = Q_0/\sqrt{LC}$.
\end{theorem}

\begin{proof}
प्रतिस्थापन द्वारा सत्यापित करें. विभेद करना,
$ i = dq/dt = -\frac{2\pi}{T_0} Q_0 \sin(2\pi t/T_0)$ --- कहा गया
$ i(t)$ --- और फिर $\frac{d^2q}{dt^2} = -(\frac{2\pi}{T_0})^2 q $,
जो बराबर है $-q/(LC)$ बिल्कुल कब $(2\pi/T_0)^2 = 1/(LC)$, अर्थात\
$ T_0 = 2\pi\sqrt{LC}$. और $ q(0) = Q_0$, $ i(0) = 0$. वो नहीं
\emph{अन्य} फ़ंक्शन समीकरण में फिट बैठता है और प्रारंभ को स्वीकार किया जाता है,
बिल्कुल वसंत के लिए के रूप में
(\cref{rem:g12:oscillators-and-time:uniqueness}).
\end{proof}

\begin{definition}[प्राकृतिक काल और आवृत्ति]\label{def:g12:rlc-oscillations:period}
 \emph{प्राकृतिक काल}\index{प्राकृतिक काल} की एक
\omterm{prop:g12:rlc-oscillations:equation}{एलसी सर्किट} है $ T_0 = 2\pi\sqrt{LC}$; इसका
\emph{प्राकृतिक आवृत्ति}\index{प्राकृतिक आवृत्ति} है
\[
f_0 = \frac{1}{T_0} = \frac{1}{2\pi\sqrt{LC}} .
\]
\omterm{def:g12:oscillators-and-time:oscillator}{आयाम}ी जांच: $\qty{1}{H} = \qty{1}{V.s/A}$ और
$\qty{1}{F} = \qty{1}{C/V}$, इसलिए $ LC $ किया जाता है
$\unit{s} \times \unit{C/A} = \unit{s^2}$ --- $\sqrt{LC}$ एक समय है,
जैसा होना चाहिए.
\end{definition}

\begin{example}[पहले नंबर]\label{ex:g12:rlc-oscillations:firstnumbers}
$ L = \qty{10}{mH}$ साथ $ C = \qty{1.0}{\micro F}$:
$\sqrt{LC} = \qty{1.0e-4}{s}$, इसलिए $ T_0 = \qty{0.63}{ms}$ और
$ f_0 \approx \qty{1.6}{kHz}$ --- एक श्रव्य स्वर. $ L = \qty{1.0}{mH}$
साथ $ C = \qty{100}{pF}$: $ T_0 = \qty{2.0}{\micro s}$,
$ f_0 \approx \qty{0.50}{MHz}$ --- एक रेडियो \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति}. केवल
\emph{उत्पाद} $ LC $ \omterm{def:g12:newtons-laws:momentum}{गति} सेट करता है.
\end{example}

\begin{remark}[एक चौथाई अवधि चरण से बाहर]\label{rem:g12:rlc-oscillations:quarter}
वर्तमान एक चौथाई है \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} से बाहर \omterm{def:g12:oscillators-and-time:phase}{चरण} आरोप के साथ: $ i = 0$
कब $ q = \pm Q_0$ (प्लेटें भरी हुई, प्रवाह उलटा) और $ i = \pm I_0$
कब $ q = 0$ (प्लेटें खाली, पूर्ण \omterm{def:g11:magnetic-fields:earth}{झुकाव} पर प्रवाह) --- बिल्कुल
पेंडुलम, चरम पर गतिहीन और तल पर सबसे तेज़।
\end{remark}

\section{ऊर्जा वाल्ट्ज}

\begin{proposition}[संग्रहित ऊर्जा का संरक्षण]\label{prop:g12:rlc-oscillations:energy}
\index{एक एलसी सर्किट की ऊर्जा}
आदर्श में \omterm{prop:g12:rlc-oscillations:equation}{एलसी सर्किट} कुल संग्रहित ऊर्जा
(\cref{prop:g12:rc-rl-circuits:energy})
\[
E \;=\; E_C + E_L \;=\; \frac{q^2}{2C} + \tfrac12\,L i^2
\;=\; \frac{Q_0^2}{2C} \;=\; \tfrac12\,L I_0^2
\]
\omterm{prop:g12:work-and-energy:equilibria}{स्थिर} है: पूरी तरह से में \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} कब $ q = \pm Q_0$, पूरी तरह से में
द कुंडल कब $ i = \pm I_0$.
\end{proposition}

\begin{proof}
अंतर करें: $\frac{dE}{dt} = \frac{q}{C}\frac{dq}{dt} +
Li\frac{di}{dt} = i\left(\frac{q}{C} + L\frac{d^2q}{dt^2}\right) = 0$
द्वारा \cref{prop:g12:rlc-oscillations:equation}. इसका मान पढ़ा जाता है
$ t = 0$ (सभी \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}), और फिर सर्व-वर्तमान क्षण में
$ q = 0$ ---कहाँ से $\tfrac12 L I_0^2 = Q_0^2/2C $, ठीक हो रहा है
$ I_0 = Q_0/\sqrt{LC}$ द्वारा ऊर्जा अकेला।
\end{proof}

और प्रति दो बार वापस \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}, राशि पर पिन किया गया $ Q_0^2/2C $ --- वही
का आंकड़ा \omterm{prop:g12:oscillators-and-time:spring-period}{स्प्रिंग--द्रव्यमान थरथरानवाला}, पुनः लेबल किया गया।}
\end{omfigure}

\begin{example}[वाल्ट्ज, मापा गया]\label{ex:g12:rlc-oscillations:waltz}
$ C = \qty{1.0}{\micro F}$ पर आरोप लगाया गया $ U_0 = \qty{10}{V}$, चालू किया गया
$ L = \qty{10}{mH}$: ऊर्जा $ E = \tfrac12 C U_0^2 = \qty{5.0e-5}{J}$,
\omterm{def:g12:oscillators-and-time:oscillator}{अवधि} $ T_0 = \qty{0.63}{ms}$, और एक चरम धारा
$ I_0 = U_0\sqrt{C/L} = \qty{0.10}{A}$ एक चौथाई तक पहुंच गया \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}
(\qty{0.16}{ms}) स्विच बंद होने के बाद।
\end{example}

\section{वास्तविक सर्किट: नम दोलन}

हर वास्तविक कुंडल से घाव है \omterm{def:g11:work-of-force:sign}{प्रतिरोधक} तार: ईमानदार पाश एक है
शृंखला \emph{आरएलसी} सर्किट, और \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} वाल्ट्ज पर कर लगाता है।

\begin{definition}[छद्म-आवधिक और एपेरियोडिक शासन]\label{def:g12:rlc-oscillations:regimes}
\index{आरएलसी सर्किट}\index{छद्म-आवधिक शासन}\index{एपेरियोडिक शासन}
एक श्रृंखला आरएलसी सर्किट में अवरोधक समाप्त हो जाता है ऊर्जा \omterm{def:g11:work-of-force:work}{जूल} द्वारा
गरम करना (\cref{ch:g11:circuits-and-power}): द \omterm{def:g12:oscillators-and-time:oscillator}{मुक्त दोलन} हैं
\emph{\omterm{def:g12:oscillators-and-time:damping}{अवमन्दित}}. दो शासन, एक पढ़ें \omterm{def:g10:signals-and-waves:oscilloscope}{आस्टसीलस्कप}:
\begin{itemize}
  \item \emph{छद्म \omterm{def:g10:signals-and-waves:period}{आवधिक}} (छोटा $ R $): सर्किट अभी भी
        एक सिकुड़ते लिफ़ाफ़े के अंदर दोलन करता है, दोहराव के समय के साथ ---
        द \emph{छद्म काल}\index{छद्म काल} ---
        के करीब $ T_0 = 2\pi\sqrt{LC}$ जब भिगोना प्रकाश है;
  \item \emph{अनावधिक} (बड़ा $ R $): चार्ज वापस शून्य हो जाता है
        कभी भी ओवरशूटिंग के बिना --- बिल्कुल भी कोई दोलन नहीं।
\end{itemize}
का सटीक दर्पण \omterm{def:g12:oscillators-and-time:damping}{अवमन्दित} \omterm{def:g12:oscillators-and-time:oscillator}{यांत्रिक थरथरानवाला}
(\cref{def:g12:oscillators-and-time:damping}).
\end{definition}

\begin{remark}[दोलन बनाए रखना]\label{rem:g12:rlc-oscillations:maintain}
\index{दोलन बनाए रखा}
रखने के लिए \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} जीवित, सर्किट को बिल्कुल फ़ीड करें ऊर्जा 
\omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} जलता है, प्रति चक्र एक बार और \emph{\omterm{def:g12:oscillators-and-time:phase}{चरण} में} झूले के साथ: एक
एम्पलीफायर आरएलसी लूप के लिए वही करता है जो एस्केपमेंट का पुश करता है
पेंडुलम (\cref{ch:g12:oscillators-and-time}). हर क्वार्ट्ज़ घड़ी
और प्रत्येक रेडियो ट्रांसमीटर का रखरखाव इसी प्रकार किया जाता है थरथरानवाला, दौड़ना
इसके पर \omterm{def:g12:rlc-oscillations:period}{प्राकृतिक आवृत्ति} $ f_0$.
\end{remark}

\section{यांत्रिक जुड़वां}

\begin{proposition}[विद्युत-यांत्रिक शब्दकोश]\label{prop:g12:rlc-oscillations:analogy}
\index{विद्युत-यांत्रिक सादृश्य}
एलसी समीकरण वसंत-द्रव्यमान समीकरण है
(\cref{prop:g12:oscillators-and-time:spring-period}) बिलकुल वैसा ही:
\begin{center}\small
\begin{tabular}{lcl}
\omterm{prop:g12:rlc-oscillations:equation}{एलसी सर्किट} & & spring--mass glider\\[2pt]
charge $ q $ & $\longleftrightarrow $ & position $ x $\\
current $ i = dq/dt $ & $\longleftrightarrow $ & velocity $ v = dx/dt $\\
\omterm{def:g12:rc-rl-circuits:inductor}{अधिष्ठापन} $ L $ & $\longleftrightarrow $ & mass $ m $\\
inverse \omterm{def:g12:rc-rl-circuits:capacitor}{समाई} $1/C $ & $\longleftrightarrow $ & \omterm{def:g12:oscillators-and-time:spring}{कठोरता} $ k $\\
\omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} $ R $ & $\longleftrightarrow $ & \omterm{def:g10:inertia:inventory}{टकराव}\\
$ L\,\frac{d^2q}{dt^2} + \frac{q}{C} = 0$ & $\longleftrightarrow $ &
  $ m\,\frac{d^2x}{dt^2} + kx = 0$\\[2pt]
$ T_0 = 2\pi\sqrt{LC}$ & $\longleftrightarrow $ &
  $ T_0 = 2\pi\sqrt{m/k}$\\[2pt]
$ E_C = q^2/2C $ & $\longleftrightarrow $ & $ E_p = \tfrac12 kx^2$\\
$ E_L = \tfrac12 Li^2$ & $\longleftrightarrow $ & $ E_k = \tfrac12 mv^2$
\end{tabular}
\end{center}
समान समीकरण, समान समाधान: प्रत्येक परिणाम किसी एक के बारे में थरथरानवाला
तुरंत दूसरी भाषा में अनुवाद करता है।
\end{proposition}

\begin{proof}
तुलना करना \cref{prop:g12:rlc-oscillations:equation} साथ
$ m\,d^2x/dt^2 = -kx $: प्रतिस्थापित करना $ q \to x $, $ L \to m $, $1/C \to k $
एक को दूसरे में बदल देता है, और $ LC \to m/k $ \omterm{def:g12:work-and-energy:diagram}{मोड़}ों
$2\pi\sqrt{LC}$ में $2\pi\sqrt{m/k}$.
\end{proof}

\begin{example}[रेडियो ट्यूनिंग]\label{ex:g12:rlc-oscillations:tuning}
एक हवाई फुसफुसाहट खिलाता है \emph{प्रत्येक} एक एलसी लूप में स्टेशन, लेकिन
केवल वही जो निकट प्रसारण कर रहा है $ f_0$ कदम बढ़ाता है और बनता है ---
\omterm{def:g12:oscillators-and-time:resonance}{गूंज} (\cref{def:g12:oscillators-and-time:resonance}). डायल
एक वेरिएबल की प्लेटों को घुमाता है \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}: बदल रहा है $ C $ चाल
$ f_0 = 1/(2\pi\sqrt{LC})$, और सुई इयरफ़ोन को हाथ लगाती है
एक समय में स्टेशन. संचारण पक्ष पर, एक बनाए रखा थरथरानवाला
(\cref{rem:g12:rlc-oscillations:maintain}) प्रसारण सेट करता है
\omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति}; एक घड़ी में, एक क्वार्ट्ज क्रिस्टल --- का एक यांत्रिक जुड़वां
लुभावनी \omterm{def:g12:oscillators-and-time:spring}{कठोरता} --- वही भूमिका निभाता है.
\end{example}

\section{समस्या: एक क्रिस्टल रेडियो का निर्माण}

\begin{problem}[{सप्ताहांत समस्या --- एक क्रिस्टल रेडियो का निर्माण: एक
कॉइल, एक वेरिएबल कैपेसिटर, एक डायोड और एक ईयरफोन --- कोई बैटरी नहीं,
कोई एम्पलीफायर नहीं, और शाम की खबर लहर की ऊर्जा पर आती है
स्वयं}]
\label{pb:g12:rlc-oscillations:1}
दादा-दादी की अटारी से एक किट: a कुंडल $ L = \qty{0.20}{mH}$ पर घाव
एक कार्डबोर्ड ट्यूब, एक चर \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} जिनकी प्लेटें घूमती हैं
\qty{20}{pF} को \qty{480}{pF}, एक क्रिस्टल डायोड, एक ईयरफोन, और
\qty{20}{m} हवाई तार का. बीच प्रसारण को पकड़ने के लिए स्टेशन
\qty{530}{kHz} और \qty{1700}{kHz}.

\medskip
\noindent\textbf{Part I --- The tuning loop.}
\begin{enumerate}
  \item  \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}, चार्ज किया गया, में डिस्चार्ज होने के लिए छोड़ दिया जाता है कुंडल:
        लूप नियम लिखें और इसे विभेदक समीकरण में बदलें
        के लिए $ q(t)$.
  \item प्रतिस्थापन द्वारा सत्यापित करें कि $ q(t) = Q_0\cos(2\pi t/T_0)$
        इसे हल करता है, बशर्ते $ T_0 = 2\pi\sqrt{LC}$.
  \item \omterm{def:g12:oscillators-and-time:oscillator}{आयाम}ी विश्लेषण द्वारा जाँच करें कि $\sqrt{LC}$ एक समय है.
  \item डायल सेट करें $ C = \qty{330}{pF}$: गणना करें $ T_0$ और $ f_0$. है
        बैंड के अंदर लूप ट्यून किया गया?
  \item खोजें \omterm{def:g12:rc-rl-circuits:capacitor}{समाई} यह बिल्कुल सटीक बैठता है \qty{1.00}{MHz}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The variable \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}.}
\begin{enumerate}[resume]
  \item की गणना करें \omterm{def:g12:rc-rl-circuits:capacitor}{समाई} के लिए आवश्यक है \qty{530}{kHz}.
  \item की गणना करें \omterm{def:g12:rc-rl-circuits:capacitor}{समाई} के लिए आवश्यक है \qty{1700}{kHz}.
  \item किट की दो आवृत्तियों की गणना करें \qtyrange{20}{480}{pF}
        \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} वास्तव में पहुंचता है. क्या यह पूरे बैंड को कवर करता है?
  \item बताते हैं कि $ f_0 \propto 1/\sqrt{C}$ निश्चित पर $ L $; किससे
        कारक अवश्य होना चाहिए $ C $ दोगुना में बदलें $ f_0$?
  \item प्लेटें रैखिक रूप से खुलती हैं, इसलिए $ C $ डायल के रूप में लगातार गिरता है
        बदल जाता है. बताएं कि फिर स्टेशनों पर एक छोर पर भीड़ क्यों उमड़ती है
        डायल का --- और कौन सा अंत।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- ऊर्जा and भिगोना.}
से ट्यून किया गया \qty{1.00}{MHz} ($ C = \qty{127}{pF}$), क्षण भर के लिए हवाई
चार्ज करता है \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} को $ U_0 = \qty{20}{mV}$.
\begin{enumerate}[resume]
  \item चार्ज की गणना करें $ Q_0$ और संग्रहीत ऊर्जा $ E $.
  \item चरम धारा की गणना करें $ I_0$, का उपयोग करना \omterm{thm:g10:energy-conservation:conservation}{ऊर्जा संरक्षण}।
  \item  कुंडल'एस \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} dissipates $4.0\%$ संग्रहित का ऊर्जा
        प्रति चक्र: कितने चक्रों के बाद है ऊर्जा आधा कर दिया?
  \item वह माइक्रोसेकंड में कितना समय है? यह ``बज'' क्यों सकता है
        अकेले शाम की खबर नहीं चलाएंगे?
  \item बताएं कि क्यों आ रहे हैं \omterm{def:g10:signals-and-waves:wave}{लहर} लूप को केवल घुमाता रहता है
        जब उस स्टेशन पर ट्यून किया जाता है --- और क्या बनाए रखता है
        पर दोलन \emph{ट्रांसमीटर}
        (\cref{rem:g12:rlc-oscillations:maintain}).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The mechanical twin.}
\begin{enumerate}[resume]
  \item शब्दकोश का उपयोग करना (\cref{prop:g12:rlc-oscillations:analogy})
        द्रव्यमान के एक ग्लाइडर के साथ $ m = \qty{0.20}{kg}$, गणना करें
        \omterm{def:g12:oscillators-and-time:spring}{कठोरता} $ k $ ट्यून किए गए सर्किट से मेल खाने वाले स्प्रिंग का।
  \item एक कठोर कार स्प्रिंग है $ k \approx \qty{5e4}{N/m}$. टिप्पणी:
        क्या एक बेंच-टॉप स्प्रिंग-मास सिस्टम मेगाहर्ट्ज़ पर दोलन कर सकता है?
  \item एक उचित रखें $ k = \qty{100}{N/m}$ इसके बजाय: क्या द्रव्यमान
        पर दोलन करता है \qty{1.00}{MHz}?
  \item कौन सी वास्तविक वस्तु इतनी पंखदार, अति-कठोर है
        \omterm{def:g12:oscillators-and-time:oscillator}{यांत्रिक थरथरानवाला}, और आप इससे पहले कहां मिले थे
        वर्ष (\cref{ch:g12:oscillators-and-time})?
  \item स्पेक शीट, एक वाक्य: वह बैंड जो आपका रेडियो स्वीप करता है, द
        \omterm{def:g12:rc-rl-circuits:capacitor}{समाई} वह चुनता है \qty{1.00}{MHz}, एक अकेली अंगूठी कितनी लंबी है
        रहता है, और वैसे भी समाचार क्या चलता है।
\end{enumerate}
\end{problem}
