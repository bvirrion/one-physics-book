\chapter{मुक्त पतन और प्रक्षेप्य गति}\label{ch:g12:projectile-motion}

बगीचे की नली का प्रत्येक जेट हवा में समान वक्र खींचता है। तो ए
घेरा की ओर उठता बास्केटबॉल, एक लंबा जम्पर, एक तोप का गोला: कोई भी शरीर
इसके लिए छोड़ दिया गया \omterm{def:g10:universal-gravitation:weight}{वज़न}. यह अध्याय उस वक्र को अर्जित करता है --- एक परवलय ---
से \omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम} कैलकुलस की दो पंक्तियों में शीर्ष, उड़ान पढ़ता है
समय और \omterm{def:g12:projectile-motion:range}{श्रेणी} इसके समीकरण को बंद करें, फिर हवा को वापस आने दें यह देखने के लिए कि क्या है
यह खराब कर देता है.

\section{निर्बाध गिरावट}

\begin{definition}[निर्बाध गिरावट]\label{def:g12:projectile-motion:freefall}
एक शरीर अंदर है \emph{निर्बाध गिरावट}\index{निर्बाध गिरावट} जब एकमात्र बल पर अभिनय
यह उसका है \omterm{def:g10:universal-gravitation:weight}{वज़न} $\vect P = m\vect g $: कोई सहारा नहीं, कोई डोरी नहीं, वायु \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}
नगण्य. गिरा हुआ पत्थर या गेंद जो हाथ से छूट गई हो, शुभ है
सटीकता, मुक्त गिरावट में --- ``गिरावट'' में उठना शामिल है।
\end{definition}

\begin{theorem}[गैलीलियो का मुक्त पतन का नियम]\label{thm:g12:projectile-motion:galileo}
\index{गैलीलियो का मुक्त पतन का नियम}
में \omterm{def:g12:projectile-motion:freefall}{निर्बाध गिरावट}, सभी पिंड समान त्वरण साझा करते हैं, चाहे उनका द्रव्यमान कुछ भी हो:
\[
\vect a = \vect g, \qquad g = \qty{9.81}{m/s^2},
\]
सीधे नीचे की ओर निर्देशित: द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} ताकत \qty{9.81}{N/kg} का
\cref{ch:g10:universal-gravitation} इसके दूसरे परिधान में, $\unit{N/kg} = \unit{m/s^2}$.
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम} (\cref{ch:g12:newtons-laws}) साथ \omterm{def:g10:universal-gravitation:weight}{वज़न} केवल
बल: $ m\vect a = m\vect g $, और सामूहिक रद्द हो जाता है। हथौड़ा और पंख
केवल हवा से भिन्न होता है।
\end{proof}

\begin{proposition}[आराम से ऊर्ध्वाधर गिरावट]\label{prop:g12:projectile-motion:vertical}
विश्राम से गिरा, गिरती गहराई के साथ $ h $ से नीचे की ओर गिना जाता है
रिलीज़ पॉइंट, एक बॉडी इन \omterm{def:g12:projectile-motion:freefall}{निर्बाध गिरावट} का अनुसरण करता है
\[
a = g, \qquad v = g t, \qquad h = \tfrac12\, g t^2 .
\]
\end{proposition}

\begin{proof}
$ v(t)$ अचर का प्रतिअवकलज है $ a = g $ पर लुप्त हो रहा है $ t = 0$
(\cref{ch:g12:kinematics-2d}); $ h(t)$ की है कि $ v = gt $ पर लुप्त हो रहा है $ t = 0$.
\end{proof}

\begin{example}[परिमाण के आदेश]\label{ex:g12:projectile-motion:orders}
बाद $1$, $2$, $3$ के सेकंड \omterm{def:g12:projectile-motion:freefall}{निर्बाध गिरावट}: $ v = 9.8$, $19.6$, \qty{29.4}{m/s}
($35$, $71$, \qty{106}{km/h}) और $ h = 4.9$, $19.6$, \qty{44.1}{m}. ए से
\qty{30}{m} छत: $ t = \sqrt{2h/g} \approx \qty{2.5}{s}$, में पहुंचने
\qty{24}{m/s} $\approx \qty{87}{km/h}$. समय जैसे बढ़ता है $\sqrt h $ --- और
कुछ सेकंड के बाद हवा नगण्य (अंतिम खंड) रहने से इंकार कर देती है।
\end{example}

\begin{example}[लंबवत फेंकना]\label{ex:g12:projectile-motion:throw}
सीधे ऊपर फेंक दिया $ v_0$, अक्ष $ Oy $ हाथ से ऊपर की ओर: वही दो
प्रतिअवकलज देते हैं $ v = v_0 - gt $ और $ y = v_0 t - \tfrac12 gt^2$. द
शीर्ष, कहाँ $ v = 0$, पर आता है $ t = v_0/g $, ऊंचाई $ h = v_0^2/2g $ --- द
ऊर्जा उत्तर $ v_0 = \sqrt{2gh}$ का \cref{ch:g11:mechanical-energy} भागो
पि\omterm{def:g11:color-light-sources:cones}{छड़}ा हुआ; फेंक दिया \emph{नीचे}, $ v_0$ फ़्लिप चिह्न. के लिए \qty{15}{m/s}: शीर्ष
\qty{11.5}{m} पर \qty{1.53}{s}, वापस पर \qty{3.06}{s}, पर \qty{15}{m/s}.
\end{example}

\section{प्रक्षेप्य: एक नियम, दो प्रतिअवकलन}

\begin{definition}[प्रक्षेप्य]\label{def:g12:projectile-motion:projectile}
ए \emph{प्रक्षेप्य}\index{प्रक्षेप्य} वेग से प्रक्षेपित एक पिंड है
$\vect v_0$ और फिर अंदर छोड़ दिया \omterm{def:g12:projectile-motion:freefall}{निर्बाध गिरावट}. प्रक्षेपण बिंदु पर अक्ष: $ Ox $
के ऊर्ध्वाधर तल में क्षैतिज $\vect v_0$, $ Oy $ ऊर्ध्वाधर ऊपर की ओर;
\emph{प्रक्षेपण कोण}\index{प्रक्षेपण कोण} $\alpha $ का $\vect v_0$ के ऊपर
क्षैतिज इसके निर्देशांक ठीक करता है $(v_0\cos\alpha,\; v_0\sin\alpha)$.
\end{definition}

\begin{theorem}[गति के समीकरण]\label{thm:g12:projectile-motion:equations}
ए \omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य} \omterm{def:g12:newtons-laws:momentum}{गति} के साथ मूल से प्रक्षेपित किया गया $ v_0$ और कोण $\alpha $
है $\vect a = (0,\,-g)$ इसलिए, हर पल
\[
\vect v(t) = \bigl(v_0\cos\alpha,\; v_0\sin\alpha - g t\bigr),
\qquad
\begin{cases}
x(t) = v_0\cos\alpha\; t,\\[2pt]
y(t) = v_0\sin\alpha\; t - \tfrac12\, g t^2 .
\end{cases}
\]
\end{theorem}

\begin{proof}
\omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम} देता है $\vect a = \vect g = (0,-g)$. प्रतिअवकलज
अक्ष दर अक्ष: $\vect v(t)$ से $\vect a $ प्रारंभिक मूल्य के साथ
$(v_0\cos\alpha,\, v_0\sin\alpha)$, फिर स्थिति से $\vect v $, से $(0,0)$.
\end{proof}

\begin{proposition}[दो गतियों की स्वतंत्रता]\label{prop:g12:projectile-motion:independence}
क्षैतिज \omterm{def:g12:newtons-laws:momentum}{गति} एक समान है --- $ v_x $ कभी नहीं बदलता --- और
ऊर्ध्वाधर \omterm{def:g12:newtons-laws:momentum}{गति} बिल्कुल पिछले अनुभाग का ऊर्ध्वाधर फेंक है:
कोई भी समीकरण दूसरे के निर्देशांक का उल्लेख नहीं करता है। विशेष रूप से, एक गेंद
क्षैतिज रूप से फायर किया गया और उसी क्षण एक गेंद उसी क्षण नीचे गिरी
ऊँचाई एक दूसरे के बराबर रहती है --- वे एक साथ उतरते हैं।
\end{proposition}

\begin{proof}
में \cref{thm:g12:projectile-motion:equations}, $ x(t)$ केवल शामिल है
$ v_0\cos\alpha $, $ y(t)$ केवल $ v_0\sin\alpha $ और $ g $; दोनों गेंदें हैं
$ v_y(0) = 0$, तो दोनों आज्ञापालन करते हैं $ h = \tfrac12 gt^2$.
\end{proof}

\begin{remark}[गिरता हुआ बंदर]\label{rem:g12:projectile-motion:monkey}
एक चिड़ियाघर संचालक एक डार्ट पर निशाना साधता है \emph{सीधे पर} एक बंदर, जो उसे जाने देता है
शॉट पर शाखा. ख़राब चाल: डार्ट और बंदर प्रत्येक बूंद $\tfrac12 gt^2$
उनकी शून्य-गुरुत्वाकर्षण स्थिति के नीचे --- लक्ष्य रेखा, शाखा --- तो
वे बिल्कुल मिलते हैं, चाहे डार्ट की \omterm{def:g12:newtons-laws:momentum}{गति} कुछ भी हो
(\cref{exo:g12:projectile-motion:15}).
\end{remark}

\section{परवलय, पूरा पढ़ें}

\begin{theorem}[प्रक्षेपवक्र]\label{thm:g12:projectile-motion:trajectory}
खत्म करना $ t $ बीच में $ x(t)$ और $ y(t)$: द \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} एक का \omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य}
($0 \leq \alpha < \ang{90}$) नीचे की ओर खुलने वाला परवलय है
\[
y = x\tan\alpha \;-\; \frac{g}{2 v_0^2 \cos^2\alpha}\, x^2 .
\]
\end{theorem}

\begin{proof}
$ t = x/(v_0\cos\alpha)$, में प्रतिस्थापित $ y(t)$ का
\cref{thm:g12:projectile-motion:equations}.
\end{proof}

\begin{definition}[श्रेणी]\label{def:g12:projectile-motion:range}
एक के लिए \omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य} समतल भूमि के ऊपर, \emph{श्रेणी}\index{श्रेणी} $ R $ है
जब यह अपनी प्रक्षेपण ऊंचाई पर लौटता है तो तय की गई क्षैतिज दूरी;
\emph{उड़ान का समय}\index{उड़ान का समय} $ t_f $ क्या उस यात्रा की \omterm{def:g12:kinematics-2d:ucm}{अवधि} है.
\end{definition}

\begin{proposition}[शीर्ष, उड़ान का समय, सीमा]\label{prop:g12:projectile-motion:range}
समतल भूमि से ऊपर, साथ में $0 < \alpha < \ang{90}$:
\[
t_f = \frac{2 v_0 \sin\alpha}{g}, \qquad
h_{\text{apex}} = \frac{(v_0\sin\alpha)^2}{2g}
\ \ \text{reached at } t_f/2, \qquad
R = \frac{v_0^2 \sin(2\alpha)}{g}.
\]
 \omterm{def:g12:projectile-motion:range}{श्रेणी} में सबसे महान है $\alpha = \ang{45}$, कहाँ $ R = v_0^2/g $;
सं\omterm{prop:g11:color-light-sources:complementary}{पूरक} कोण $\alpha $ और $\ang{90} - \alpha $ वही दे दो \omterm{def:g12:projectile-motion:range}{श्रेणी}.
\end{proposition}

\begin{proof}
$ y = 0$ देता है $ t\,(v_0\sin\alpha - \tfrac12 gt) = 0$: लॉन्च, और $ t_f $.
शीर्ष, $ v_y = 0$, बैठ बैठ गया $ t = v_0\sin\alpha/g = t_f/2$; में स्थानापन्न करें
$ y(t)$. तब $ R = x(t_f) = 2v_0^2\sin\alpha\cos\alpha/g =
v_0^2\sin(2\alpha)/g $. साइन चरम पर है $2\alpha = \ang{90}$, और
$\sin\bigl(2(\ang{90}-\alpha)\bigr) = \sin(\ang{180} - 2\alpha) =
\sin(2\alpha)$.
\end{proof}

\begin{example}[गोल किक]\label{ex:g12:projectile-motion:kick}
एक गोलकीपर हमला करता है $ v_0 = \qty{25}{m/s}$, $\alpha = \ang{45}$:
$ R = 25^2/9.81 \approx \qty{64}{m}$; वास्तविक मंजूरी निकट आ रही है
\qty{55}{m} --- हवा में कटौती.
\end{example}

\begin{method}[प्रक्षेप्य बहीखाता]\label{met:g12:projectile-motion:method}
\begin{enumerate}
  \item प्रक्षेपण बिंदु पर अक्ष, $ Ox $ क्षैतिज, $ Oy $ ऊपर; निर्देशांक
        का $\vect v_0$.
  \item जाँच करना \omterm{def:g12:projectile-motion:freefall}{निर्बाध गिरावट} (\omterm{def:g10:universal-gravitation:weight}{वज़न} केवल): फिर $\vect a = (0,-g)$;
        प्रतिअवकलज दो बार, प्रारंभिक स्थितियाँ हर बार।
  \item ए के बारे में प्रश्न \emph{जगह}: हटाना $ t $ (\omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}). एक के बारे में
        \emph{तुरंत}: रखना $ t $.
  \item शब्दकोश: शीर्ष $\Leftrightarrow v_y = 0$; लॉन्च ऊंचाई
        $\Leftrightarrow y = 0$; दीवार पर $ d \Leftrightarrow x = d $; ज़मीन
        $\ell $ नीचे $\Leftrightarrow y = -\ell $.
  \item विवेक: शीर्ष के बारे में उड़ान सममित; $ R \leq v_0^2/g $; केवल हवा
        कभी छोटा करता है.
\end{enumerate}
\end{method}

\section{हवा क्या बदलती है}

\begin{definition}[टर्मिनल गति]\label{def:g12:projectile-motion:terminal}
वायु \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} वेग का विरोध करता है और \omterm{def:g12:newtons-laws:momentum}{गति} के साथ बढ़ता है। गिरता हुआ शरीर
इसलिए ड्रैग संतुलन होने तक ही \omterm{def:g12:newtons-laws:momentum}{गति} होती है \omterm{def:g10:universal-gravitation:weight}{वज़न}; वहाँ से
$\vect a = \vect 0$: यह एक स्थिरांक पर गिरता है \emph{टर्मिनल गति}\index{टर्मिनल गति}.
\end{definition}

\begin{example}[परिमाण के अंतिम आदेश]\label{ex:g12:projectile-motion:airorders}
एक फ़्लैट-आउट स्काइडाइवर: के बारे में \qty{55}{m/s} (\qty{200}{km/h}). एक बारिश की बूंद:
के बारे में \qty{9}{m/s} --- शून्य में, \qty{2000}{m} गिरावट का यह इसे वितरित करेगा
$\sqrt{2gh} \approx \qty{2.0e2}{m/s}$. एक पंख: के बारे में \qty{1}{m/s},
सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} के भीतर पहुंच गया --- इसकी धीमी \omterm{def:g12:newtons-laws:momentum}{गति} का रहस्य: यह अपना खर्च करता है
संपूर्ण पतझड़ \emph{पर} \omterm{def:g12:projectile-motion:terminal}{टर्मिनल गति}; हथौड़ा कभी भी अपने करीब नहीं पहुँचता.
\end{example}

\begin{remark}[असली बैलिस्टिक]\label{rem:g12:projectile-motion:drag}
उड़ान के दौरान वेग के विरुद्ध खीचें धक्का: \omterm{def:g12:projectile-motion:range}{श्रेणी} छोटा करता है,
शीर्ष कम हो जाता है, और समरूपता टूट जाती है --- अवतरण की तुलना में अधिक तीव्र
चढ़ें, लंबे शॉट स्पष्ट रूप से परवलय से कम दिखें। ईमानदार, मात्रात्मक
उपचार वर्ष 1 खंड में हल किया गया एक अंतर समीकरण है; यह
वर्ष, परवलय सघन, मध्यम तेज़ के लिए सही मॉडल है
प्रोजेक्टाइल --- एक बास्केटबॉल पर \qty{10}{m/s}, गोल्फ की गेंद नहीं
\qty{70}{m/s}.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:1}
पुल से गिरा एक पत्थर पानी से टकराता है \qty{2.0}{s} बाद में: की ऊंचाई
पुल, और प्रभाव की \omterm{def:g12:newtons-laws:momentum}{गति} \unit{m/s} और \unit{km/h}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:2}
चंद्रमा पर, एक अंतरिक्ष यात्री ने एक हथौड़ा और एक पंख एक साथ गिराया: वे
एक साथ उतरे. क्यों --- और पृथ्वी पर पंख इतनी बुरी तरह से क्यों खो जाता है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:3}
हवा की उपेक्षा करते हुए, गिरने का समय और आगमन की \omterm{def:g12:newtons-laws:momentum}{गति} की गणना करें \qty{1.0}{m},
\qty{10}{m}, \qty{100}{m}. ऊंचाई के साथ समय कैसे बढ़ता है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:4}
एक गेंद सीधे ऊपर फेंकी जाती है \qty{15}{m/s}: शीर्ष ऊंचाई और समय,
कुल समय वापस हाथ में, और वापसी की \omterm{def:g12:newtons-laws:momentum}{गति}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:projectile-motion:5}
ऊँचाई की मेज से एक संगमरमर लुढ़कता है \qty{0.80}{m} पर \qty{2.5}{m/s}. खोजें
गिरने का समय और किनारे से उतरने की दूरी। क्या पतझड़ का समय है
संगमरमर की \omterm{def:g12:newtons-laws:momentum}{गति} पर निर्भर?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:6}
पाठ्यक्रम के पांच-कोण चित्र से: कौन से जोड़े \omterm{def:g12:projectile-motion:projectile}{प्रक्षेपण कोण}
उन्हें साझा करें \omterm{def:g12:projectile-motion:range}{श्रेणी}, और कौन सा कोण जीतता है? एक के लिए \omterm{def:g12:projectile-motion:range}{श्रेणी} अधिकतम से नीचे,
दो संभावित कोणों में से कौन सा कोण अधिक समय तक उड़ान भरता है? के साथ औचित्य सिद्ध करें $ t_f $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:7}
ए से \qty{20}{m} बालकनी से एक गेंद सीधी फेंकी जाती है \emph{नीचे} पर
\qty{5.0}{m/s}. जमीन पर पहुंचने का समय और प्रभाव की \omterm{def:g12:newtons-laws:momentum}{गति} ज्ञात करें; जांचें
बाद वाला एक के साथ ऊर्जा संतुलन (\cref{ch:g11:mechanical-energy}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:8}
ए \omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य} समतल भूमि पर पत्तियाँ \qty{40}{m/s}, $\alpha = \ang{30}$:
\omterm{def:g12:projectile-motion:range}{उड़ान का समय}, शीर्ष ऊंचाई और \omterm{def:g12:projectile-motion:range}{श्रेणी}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:9}
एक लंबी जम्पर को एक मॉडल के रूप में मॉडल करें \omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य}: टेकऑफ़ पर \qty{10.5}{m/s},
$\alpha = \ang{20}$. की गणना करें \omterm{def:g12:projectile-motion:range}{श्रेणी}; चैंपियन लगभग कूद पड़े \qty{9}{m}
--- बिंदु मॉडल द्वारा छोड़े गए दो अवयवों के नाम बताएं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:10}
पर \qty{1.2}{m} समतल ज़मीन के ऊपर, एक गोली गिराई जाती है और एक समान
एक ने क्षैतिज रूप से गोली चलाई \qty{300}{m/s}. सबसे पहले कौन सी भूमि? की गणना करें
गिरने का समय और चलाई गई गोली की लैंडिंग दूरी। कौन सा सिद्धांत है
परीक्षण किया गया?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:projectile-motion:11}
उसे बीजगणितीय रूप से दिखाएँ $\alpha $ और $\ang{90} - \alpha $ उन्हें साझा करें
\omterm{def:g12:projectile-motion:range}{श्रेणी}, फिर ए पर उतरने वाले दो कोण खोजें \qty{25}{m/s} \omterm{def:g12:projectile-motion:projectile}{प्रक्षेप्य}
\qty{50}{m} दूर।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:12}
पानी बगीचे की नली से निकलता है \qty{8.0}{m/s}: सबसे बड़ी क्षैतिज पहुंच?
सबसे बड़ी ऊर्ध्वाधर ऊंचाई? फूलों की क्यारी को पानी देने वाले दो नोजल कोण
\qty{5.0}{m} दूर?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:13}
एक गेंद को जमीन से लात मारी जाती है \qty{18}{m/s}, $\alpha = \ang{35}$,
ए की ओर \qty{4.0}{m} बचाव \qty{15}{m} दूर। का उपयोग \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र}
समीकरण, तय करें कि क्या यह साफ़ होता है, और कितना।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:14}
एक \qty{80}{kg} स्काइडाइवर औंधे मुंह गिर जाता है \omterm{def:g12:projectile-motion:terminal}{टर्मिनल गति} \qty{55}{m/s}.
(ए) खींचें बल? (बी) का आकार रेखाचित्र बनाएं $ v(t)$ छलांग से. (सी) ए
से वर्षा की बूंद गिरती है \qty{2.0}{km}: वैक्यूम आगमन \omterm{def:g12:newtons-laws:momentum}{गति} बनाम।\ यह \omterm{def:g11:lenses-and-eye:images}{असली} है
\qty{9}{m/s}?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:projectile-motion:15}
गिरते हुए बंदर ने साबित कर दिया: एक डार्ट ठीक उसी ओर लक्ष्य करके मूल स्थान से निकलता है
दूरी पर बंदर $ D $, ऊंचाई $ H $ ($\tan\alpha = H/D $), जो गिराता है
$ t = 0$. जब डार्ट पहुँचे तो दिखाओ $ x = D $ इसकी ऊंचाई के बराबर है
बंदर का --- जो भी हो $ v_0$.
\end{exercise}

\section{समस्या: बास्केटबॉल बजर-बीटर}

\begin{problem}[{सप्ताहांत समस्या --- बास्केटबॉल बजर-बीटर: एक निःशुल्क
विच्छेदित तीर को तीर दर तीर फेंकें, उंगलियों के सिरे सेंटीमीटर से साफ़ हों, ए
पूर्ण-अदालत में एक हाथ जो दे सकता है, उसके मुकाबले वजन कम किया जाता है, एक लोब ओवर द
बैकबोर्ड --- और खेल में सबसे नाटकीय शॉट पर फैसला}]
\label{pb:g12:projectile-motion:1}
बास्केटबॉल घेरा का रिम एक क्षैतिज वलय है \qty{3.05}{m} के ऊपर
फर्श; नोरा हर शॉट पर रिलीज करती हैं \qty{2.00}{m}, तो कुल्हाड़ियों के साथ
रिलीज़ पॉइंट ($ Ox $ घेरा की ओर क्षैतिज, $ Oy $ ऊपर) रिम केंद्र
बैठता है \qty{1.05}{m} मूल के ऊपर. जहाँ कहा गया है उसे छोड़कर वायु की उपेक्षा की जाती है।

\medskip\noindent\textbf{Part I --- The free throw.} नोरा ने शूट किया
$ d = \qty{4.20}{m}$ (क्षैतिज) के साथ \omterm{def:g12:projectile-motion:projectile}{प्रक्षेपण कोण} $\alpha = \ang{50}$.
\begin{enumerate}
  \item क्या बल एक बार जारी होने पर कार्य करें? देना $\vect a $, प्रयुक्त कानून का हवाला देते हुए।
  \item के निर्देशांक प्राप्त करें $\vect v(t)$.
  \item निकाले जाते हैं $ x(t)$ और $ y(t)$.
  \item हटाना $ t $ प्राप्त करने के लिए \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} $ y(x)$.
  \item गेंद को रिम केंद्र से गुजरना होगा, $ y(d) = \qty{1.05}{m}$: हल करें
        के लिए $ v_0$, में \unit{m/s} और \unit{km/h}.
\end{enumerate}

\medskip\noindent\textbf{Part II --- Over the fingertips, into the ring.}
\begin{enumerate}[resume]
  \item गेंद को रिम तक पहुँचने में कितना समय लगता है?
  \item शीर्ष समय और ऊंचाई ज्ञात करें; जांचें कि गेंद पहले से ही है
        रिम पर उतरना.
  \item एक रक्षक \qty{0.90}{m} नोरा के सामने उँगलियाँ फैलाती है
        \qty{2.60}{m}. गेंद उन्हें कितना साफ़ करती है?
  \item गणना करना $ v_x $ और $ v_y $ रिम पर, वहां \omterm{def:g12:newtons-laws:momentum}{गति}, और कोण
        क्षैतिज के नीचे वेग का.
  \item गेंद क्यों आनी चाहिए \emph{अवरोही} --- और खड़ी क्यों?
\end{enumerate}

\medskip\noindent\textbf{Part III --- The full-court heave.} बजर के बारे में
को \omterm{def:g10:signals-and-waves:sound}{आवाज़}, नोरा है \qty{26.0}{m} दूर किनारे से.
\begin{enumerate}[resume]
  \item  \qty{1.05}{m} वृद्धि के विरुद्ध छोटा है \qty{26.0}{m}: औचित्य सिद्ध करें
        लेवल-ग्राउंड मॉडल, और प्राप्त करें \omterm{def:g12:projectile-motion:range}{उड़ान का समय} से $ y(t_f) = 0$.
  \item परिणाम निकालना $ R = v_0^2\sin(2\alpha)/g $ और सबसे बड़ा कोण \omterm{def:g12:projectile-motion:range}{श्रेणी}.
  \item की गणना करें $ v_0$ कवर करने की जरूरत है \qty{26.0}{m} पर $\alpha = \ang{45}$.
  \item एक मजबूत खिलाड़ी बास्केटबॉल को लगभग उछाल सकता है \qty{17.5}{m/s}.
        उसका अधिकतम क्या है \omterm{def:g12:projectile-motion:range}{श्रेणी} --- क्या शॉट मानवीय रूप से संभव है?
  \item पर \qty{17.5}{m/s}, खोजें \emph{दो} गेंद के उतरने का कोण
        \qty{26.0}{m} दूर, और उनकी \omterm{def:g12:projectile-motion:range}{उड़ान का समय}: साथ \qty{2.0}{s} पर
        घड़ी, कौन सा बजर बजाता है?
\end{enumerate}

\medskip\noindent\textbf{Part IV --- The lob over the backboard.} से
बेसलाइन नोरा लॉब्स के पीछे: रिलीज़ \qty{6.0}{m} रिम केंद्र से,
$\alpha = \ang{75}$; बोर्ड का विमान उसके शॉट को पार कर जाता है \qty{0.45}{m}
रिम केंद्र से पहले, शीर्ष किनारा \qty{3.95}{m} फर्श के ऊपर.
\begin{enumerate}[resume]
  \item खोजें $ v_0$ जो गेंद को रिम केंद्र के माध्यम से गिराता है।
  \item बोर्ड के तल पर गेंद की ऊंचाई की गणना करें
        ($ x = \qty{5.55}{m}$): क्या यह ऊपरी किनारे को साफ़ करता है, कितना?
  \item शीर्ष ऊंचाई और \omterm{def:g12:projectile-motion:range}{उड़ान का समय} ज्ञात करें --- भीड़ इसे क्यों पसंद करती है?
  \item गुणात्मक दृष्टि से इन तीनों पर वायु \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} में क्या परिवर्तन होता है
        शॉट्स, और इससे किसको सबसे ज्यादा खतरा है?
  \item \emph{बजर-बीटर फैसला}, दो वाक्य: \omterm{def:g12:newtons-laws:momentum}{गति} की तुलना करें
        फुल-कोर्ट शॉट मांग करता है कि एक हाथ क्या आपूर्ति करता है, दे दो
        मार्जिन, और नियम --- संभव या मिथक?
\end{enumerate}
\end{problem}
