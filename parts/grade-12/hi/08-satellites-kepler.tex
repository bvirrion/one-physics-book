\section{पृथ्वी के चारों ओर गिरना}

सब कुछ पहले से मापे गए कानून पर निर्भर करता है
(\cref{ch:g10:universal-gravitation}): पृथ्वी, द्रव्यमान की $M $, खींचता है
ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} द्रव्यमान का $ m $ जिसका केंद्र एक दूरी है $ r $ अपने आप से
एक के साथ बल परिमाण का, पृथ्वी के केंद्र की ओर निर्देशित
\[ F = G\,\frac{m M}{r^2}, \qquad G = \qty{6.67e-11}{N.m^2/kg^2}. \]
क्या ख्याल है $ r $ है: दूरी \emph{केन्द्रों के बीच}. ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} पर
ऊंचाई $ h $ है $ r = R + h $, साथ $ R = \qty{6.37e6}{m}$ पृथ्वी का
त्रिज्या---भूलना $ R $ इस अध्याय की क्लासिक भूल है.

\begin{remark}[न्यूटन की तोप, दोबारा देखी गई]\label{rem:g12:satellites-kepler:cannon}
क्षैतिज रूप से दागे जाने पर, एक तोप का गोला ज़मीन पर गिरता है; तेजी से, यह दूर तक उतरता है;
सही \omterm{def:g12:newtons-laws:momentum}{गति} से जमीन उतनी ही तेजी से दूर मुड़ती है
गेंद गिरती है, और गिरना अपने आप बंद हो जाता है: an \omterm{def:g10:universal-gravitation:satellite}{कक्षा}. ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}
किसी इंजन की आवश्यकता नहीं है --- केवल अधिकार की \emph{बग़ल में} \omterm{def:g12:kinematics-2d:velocity}{रफ़्तार}।
\end{remark}

\section{वृत्ताकार कक्षा: एक गति, एक अवधि}

\begin{theorem}[कक्षीय गति]\label{thm:g12:satellites-kepler:speed}
ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} में \omterm{def:g12:kinematics-2d:ucm}{एकसमान गोलाकार} \omterm{def:g10:universal-gravitation:satellite}{कक्षा} त्रिज्या का $ r $ के एक शरीर के आसपास
द्रव्यमान $ M $ की \omterm{def:g12:newtons-laws:momentum}{गति} से यात्रा करता है
\[ v = \sqrt{\frac{G M}{r}}, \]
जो पर निर्भर नहीं है उपग्रहका द्रव्यमान.
\end{theorem}

\begin{proof}
गुरुत्वाकर्षण ही एकमात्र है बल, इसलिए \omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम}
(\cref{thm:g12:newtons-laws:second}) पढ़ता है $ m\vect a = \vect F $;
इसे पर प्रोजेक्ट करें \omterm{def:g12:kinematics-2d:frenet}{फ़्रेनेट फ़्रेम} (\cref{def:g12:kinematics-2d:frenet}).
 बल केंद्र पर लक्ष्य: इसका स्पर्शरेखीय घटक शून्य है, इसलिए
\omterm{def:g12:newtons-laws:momentum}{गति} स्थिर है; इसका सामान्य घटक संपूर्ण है $ GmM/r^2$,
और वृत्ताकार \omterm{def:g12:newtons-laws:momentum}{गति} होती है $ a_n = v^2/r $
(\cref{thm:g12:kinematics-2d:centripetal}). इस तरह $ m v^2/r =
GmM/r^2$, और --- $ m $ बाँटना --- $ v^2 = GM/r $.
\end{proof}

\begin{proposition}[कक्षीय अवधि]\label{prop:g12:satellites-kepler:period}
 \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} उसे पूरा करता है \omterm{def:g10:universal-gravitation:satellite}{कक्षा} में \omterm{def:g12:kinematics-2d:ucm}{अवधि}
\[ T = \frac{2\pi r}{v} = 2\pi \sqrt{\frac{r^3}{G M}}. \]
\end{proposition}

\begin{proof}
एक गोद परिधि है $2\pi r $ निरंतर \omterm{def:g12:newtons-laws:momentum}{गति} से; डालना
$ v = \sqrt{GM/r}$: $ T = 2\pi r\sqrt{r/(GM)} = 2\pi\sqrt{r^3/(GM)}$.
\end{proof}

कोई भी नहीं $ v $ और न $ T $ रोकना $ m $: एक पेंच ढीला होकर हिल गया
अंतरिक्ष स्टेशन कक्षाओं ठीक इसके बगल में. और दोनों साथ में गिर जाते हैं $ r $:
\emph{जितना अधिक होगा \omterm{def:g10:universal-gravitation:satellite}{कक्षा}, उतना ही धीमा \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}}, \omterm{def:g12:newtons-laws:momentum}{गति} में
($ v \propto 1/\sqrt r $) और कोण में ($ T \propto r^{3/2}$).

\begin{example}[अंतरिक्ष स्टेशन]\label{ex:g12:satellites-kepler:iss}
अंतर्राष्ट्रीय अंतरिक्ष स्टेशन ऊंचाई पर उड़ता है
$ h \approx \qty{400}{km}$, इसलिए $ r = \num{6.37e6} + \num{4.0e5} =
\qty{6.77e6}{m}$; साथ $ GM = \num{6.67e-11} \times \num{5.97e24} =
\qty{3.98e14}{m^3/s^2}$, $ v = \sqrt{\num{3.98e14}/\num{6.77e6}}
\approx \qty{7.7e3}{m/s}$ और $ T = 2\pi r/v \approx \qty{5.5e3}{s}
\approx \qty{92}{min}$: हर डेढ़ घंटे में ग्रह का एक चक्कर,
अंतरिक्ष यात्रियों के लिए प्रतिदिन लगभग सोलह सूर्योदय।
\end{example}

\begin{example}[चंद्रमा सेब की तरह गिरता है]\label{ex:g12:satellites-kepler:moon}
चंद्रमा, पर $ r = \qty{3.84e8}{m}$, कक्षाओं में $ T = \qty{27.3}{d} =
\qty{2.36e6}{s}$, इसलिए $ v = 2\pi r/T \approx \qty{1.02e3}{m/s}$ और
$ a = v^2/r \approx \qty{2.7e-3}{m/s^2}$ --- एकदम सही $ g/60^2$, और
चंद्रमा बैठता है $60$ पृथ्वी त्रिज्या दूर: वह खिंचाव जो सेब को गिराता है
चंद्रमा को झुका देता है, व्युत्क्रम वर्ग से कमजोर हो जाता है। प्रत्येक सेकंड
चंद्रमा ``गिरता है'' के बारे में \qty{1.4}{mm} हमारी ओर, यह बग़ल में है
इसे चारों ओर ले जाने वाला किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} --- चेक \omterm{def:g10:inertia:force}{न्यूटन} ने 1666 में बनाया था।
\end{example}

\section{केप्लर के नियम}

\omterm{def:g10:inertia:force}{न्यूटन} ने व्युत्क्रम वर्ग का अनुमान नहीं लगाया: उन्होंने इसे तीन से निकाला
जोहान्स केप्लर ने 1609 और 1619 के बीच नियमितताएँ आसवित कीं,
टाइको ब्राहे की बीस वर्षों की नग्न आंखों वाली ग्रह स्थितियों से ---
टेलीस्कोप-पूर्व दुनिया का सबसे अच्छा डेटा।

\begin{definition}[अंडाकार]\label{def:g12:satellites-kepler:ellipse}
एक \emph{अंडाकार}\index{अंडाकार} बिंदुओं का समूह है जिसका
दो निश्चित बिंदुओं की दूरी, \emph{फोकी}\index{फोकी}, एक स्थिर योग है; इसका आधा सबसे बड़ा व्यास है
\emph{अर्ध-प्रमुख अक्ष}\index{अर्ध-प्रमुख अक्ष} $ a $, और एक वृत्त है
वह मामला जहां फ़ॉसी विलीन हो जाती है। एक ग्रह का \omterm{def:g10:universal-gravitation:satellite}{कक्षा} निकटतम बिंदु और
सूर्य से सर्वाधिक दूर इसके हैं \emph{सूर्य समीपक}\index{सूर्य समीपक}
और \emph{नक्षत्र}\index{नक्षत्र}.
\end{definition}

\begin{theorem}[केप्लर के नियम]\label{thm:g12:satellites-kepler:kepler}
\index{केप्लर के नियम}%
सूर्य की परिक्रमा करने वाले ग्रहों के लिए (द्रव्यमान) $ M $):
\begin{enumerate}
  \item प्रत्येक \omterm{def:g10:universal-gravitation:satellite}{कक्षा} एक \omterm{def:g12:satellites-kepler:ellipse}{अंडाकार} एक फोकस पर सूर्य के साथ;
  \item सूर्य-ग्रह खंड समान समय में समान क्षेत्रों में घूमता है;
  \item वर्ग का अनुपात \omterm{def:g12:kinematics-2d:ucm}{अवधि} घनाकार करने के लिए \omterm{def:g12:satellites-kepler:ellipse}{अर्ध-प्रमुख अक्ष} है
        सभी ग्रहों के लिए समान: $ T^2/a^3 = 4\pi^2/(G M)$.
\end{enumerate}
किसी भी परिवार पर समान कानून लागू होते हैं \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों किसी भी केंद्रीय के आसपास
शरीर, साथ $ M $ केंद्रीय द्रव्यमान. \admitted
\end{theorem}

\begin{remark}[इस साल हम क्या साबित कर सकते हैं]\label{rem:g12:satellites-kepler:proof}
एक के लिए \emph{परिपत्र} \omterm{def:g10:universal-gravitation:satellite}{कक्षा} तीसरा नियम हमारा है: वर्ग करना
\cref{prop:g12:satellites-kepler:period} देता है
$ T^2/r^3 = 4\pi^2/(GM)$ --- प्रत्येक के लिए एक स्थिरांक \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} का
एक ही केंद्र. वह अनेक बिंदु, समान क्षेत्र और सामान्य $ a $ अनुसरण करें
व्युत्क्रम वर्ग से, वर्ष~1 में, अधिक कलन के साथ दिखाया गया है
आयतन. इस बीच \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} नियम पढ़ा जा सकता है: ग्रह चलता है
सबसे तेज \omterm{def:g12:satellites-kepler:ellipse}{सूर्य समीपक}, जहां शॉर्ट स्वीप सेगमेंट को जल्दी करनी चाहिए।
\end{remark}

\section{काम पर परिक्रमा}

\begin{definition}[भूस्थैतिक कक्षा]\label{def:g12:satellites-kepler:geo}
ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} है \emph{भू-स्थिर}\index{भू-स्थिर} कब
यह जमीन के एक निश्चित बिंदु पर लटका हुआ है। इसका \omterm{def:g10:universal-gravitation:satellite}{कक्षा} होना ही चाहिए
वृत्ताकार, विषुवतीय, और का \omterm{def:g12:kinematics-2d:ucm}{अवधि} एक \emph{नाक्षत्र दिन}\index{नाक्षत्र दिन}, $ T = \qty{86164}{s}$ (\qty{23}{h}
\qty{56}{min} \qty{4}{s}): पृथ्वी को एक बार घूमने में लगने वाला समय
\emph{सितारों के सापेक्ष}. (द \qty{24}{h} दिन सापेक्ष है
सूर्य, जिसकी ओर पृथ्वी भी प्रतिदिन एक डिग्री आगे बढ़ती है।)
\end{definition}

\begin{example}[भूस्थैतिक ऊंचाई]\label{ex:g12:satellites-kepler:geoalt}
केप्लर का तीसरा नियम, पीछे की ओर पढ़ें, त्रिज्या निर्धारित करता है:
$ r = \left(GMT^2/4\pi^2\right)^{1/3} =
\left(\num{3.98e14} \times (\num{86164})^2/4\pi^2\right)^{1/3}
\approx \qty{4.22e7}{m}$ --- एक ऊंचाई $ h = r - R \approx
\qty{3.58e7}{m} \approx \qty{35800}{km}$, पर यात्रा की $ v = 2\pi r/T
\approx \qty{3.1}{km/s}$. प्रत्येक रिले जो ``अभी भी लटकी हुई है'' पर बैठती है
भूमध्य रेखा पर यह एक वृत्त --- कोई \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} नहीं है।
\end{example}

\begin{example}[नेविगेशन नक्षत्र]\label{ex:g12:satellites-kepler:gps}
सैटेलाइट-नेविगेशन प्रणालियाँ लगभग तीस उड़ान भरती हैं \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों पर
$ r \approx \qty{2.66e7}{m}$ (ऊंचाई $\approx \qty{20200}{km}$),
कहाँ \cref{prop:g12:satellites-kepler:period} देता है $ T \approx
\qty{4.32e4}{s}$: का आधा \omterm{def:g12:satellites-kepler:ellipse}{नक्षत्र} दिवस, तो प्रत्येक \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} पुनः पता लगाता है
इसका ग्राउंड ट्रैक रोजाना है। आपका रिसीवर आगमन के समय की तुलना करता है
उनकी घड़ी सिग्नल ख़िलाफ़ कक्षाओं \omterm{def:g10:orders-of-magnitude:unit}{मीटर} से जाना जाता है.
\end{example}

\begin{method}[अपने उपग्रह से एक विश्व का वजन करना]\label{met:g12:satellites-kepler:weigh}
\index{किसी ग्रह का वजन करना}%
किसी ग्रह या तारे का द्रव्यमान मापने के लिए: कोई भी खोजें \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} इसका
(एक चंद्रमा, एक जांच, एक ग्रह), इसे मापें \omterm{def:g10:universal-gravitation:satellite}{कक्षा} RADIUS $ r $ और \omterm{def:g12:kinematics-2d:ucm}{अवधि}
$ T $, और तीसरे नियम को उलटा करें:
\[ M = \frac{4\pi^2 r^3}{G\,T^2}. \]
इसका वजन होता है \emph{केंद्रीय} केवल शरीर --- उपग्रहका द्रव्यमान
में रद्द कर दिया गया \cref{thm:g12:satellites-kepler:speed}. प्रत्येक डेटा-कार्ड
द्रव्यमान इस प्रकार आया: चंद्रमा से पृथ्वी, सूर्य से
धरती का अपना \omterm{def:g10:universal-gravitation:satellite}{कक्षा}.
\end{method}

\begin{remark}[उच्चतर धीमा है --- लेकिन लागत अधिक है]\label{rem:g12:satellites-kepler:energy}
तब से $ v = \sqrt{GM/r}$, चंद्रमा रेंगता है \qty{1}{km/s} जबकि
स्टेशन पर दौड़ \qty{7.7}{km/s}. फिर भी उच्चतर \omterm{def:g10:universal-gravitation:satellite}{कक्षा} अधिक है
\emph{महँगा}:गुरुत्वाकर्षण के विरुद्ध चढ़ने से \omterm{def:g10:energy-conservation:efficiency}{क्षमता} संग्रहित होती है ऊर्जा
(खाता तय हो गया है \cref{ch:g12:work-and-energy}), और
में लाभ $ E_p $ में नुकसान की तुलना में अधिक है $ E_k $. इसलिए एक रॉकेट जलता है
\emph{आगे} उच्चतर, धीमी \omterm{def:g12:newtons-laws:momentum}{गति} से पहुँचने के लिए \omterm{def:g10:universal-gravitation:satellite}{कक्षा} --- और ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}
द्वारा ब्रेक लगाया गया \omterm{def:g11:lenses-and-eye:lens}{पतला} वायु सर्पिल \emph{नीचे} और \omterm{def:g12:newtons-laws:momentum}{गति} बढ़ाता है.
\end{remark}

\begin{remark}[डेटा कार्ड]\label{rem:g12:satellites-kepler:data}
जब तक अन्यथा कहा न जाए: $ G = \qty{6.67e-11}{N.m^2/kg^2}$; धरती:
$ M = \qty{5.97e24}{kg}$, $ R = \qty{6.37e6}{m}$, \omterm{def:g12:satellites-kepler:ellipse}{नक्षत्र} दिवस
\qty{86164}{s}; चंद्रमा का \omterm{def:g10:universal-gravitation:satellite}{कक्षा}: $ r = \qty{3.84e8}{m}$, $ T =
\qty{27.3}{d}$; सूरज: $ M = \qty{1.99e30}{kg}$; पृथ्वी--सूर्य की दूरी
\qty{1.496e11}{m}; एक वर्ष $= \qty{3.156e7}{s}$; मंगल: $ M =
\qty{6.42e23}{kg}$, $ R = \qty{3.39e6}{m}$.
\end{remark}

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:satellites-kepler:2}

($ r = R $), कक्षीय \omterm{def:g12:newtons-laws:momentum}{गति} की गणना करें और \omterm{def:g12:kinematics-2d:ucm}{अवधि}. से तुलना करें

\begin{exercise}[$\star $]\label{exo:g12:satellites-kepler:3}

त्रिज्या $2r $. इसकी \omterm{def:g12:newtons-laws:momentum}{गति} और किस कारक से होती है \omterm{def:g12:kinematics-2d:ucm}{अवधि} परिवर्तन? करो
उत्तर उसके द्रव्यमान पर निर्भर करते हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:satellites-kepler:4}
स्टेशन के नंबर दोबारा करें: से $ r = \qty{6.77e6}{m}$, गणना करें

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:satellites-kepler:5}
केपलर के तीन नियम बताइये। जहाँ यह बहुत फैला हुआ है \omterm{def:g12:satellites-kepler:ellipse}{अंडाकार}
एक धूमकेतु सबसे तेज़ है, और कौन सा कानून ऐसा कहता है? क्या ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}'एस
\omterm{def:g12:kinematics-2d:ucm}{अवधि} अपने स्वयं के द्रव्यमान पर निर्भर?

\begin{exercise}[$\star\star $]\label{exo:g12:satellites-kepler:6}

कार्ड), पृथ्वी के द्रव्यमान की गणना करें और कार्ड के मान से तुलना करें।

\begin{exercise}[$\star\star $]\label{exo:g12:satellites-kepler:7}

हर सर्कुलर क्यों \omterm{def:g10:universal-gravitation:satellite}{कक्षा} पृथ्वी के केन्द्र पर केन्द्रित है; निष्कर्ष
प्रस्ताव असंभव है. कहाँ \emph{कर सकना} ए \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} मंडराना?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:satellites-kepler:9}

गणना करना $ T^2/a^3$ मंगल ग्रह के लिए और पृथ्वी के लिए (डेटा कार्ड), जांचें
केप्लर का तीसरा नियम, और सूर्य का द्रव्यमान ज्ञात करें।

\begin{exercise}[$\star\star $]\label{exo:g12:satellites-kepler:10}

और बताएं कि इसने \omterm{def:g10:inertia:force}{न्यूटन} को क्यों आश्वस्त किया कि सेब और चंद्रमा पर एक ही कानून का शासन है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:satellites-kepler:11}

इस पर क्षुद्रग्रह सेरेस, $ a = \qty{2.77}{au}$, और हैली धूमकेतु,
$ a = \qty{17.8}{au}$: दोनों की गणना करें \omterm{def:g12:kinematics-2d:ucm}{अवधि}. हैली आखिरी बार गुजरी
\omterm{def:g12:satellites-kepler:ellipse}{सूर्य समीपक} 1986 में --- यह कब वापस आएगा?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:satellites-kepler:12}

--- अतिरिक्त \omterm{def:g12:newtons-laws:momentum}{गति} कहाँ से आती है?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:satellites-kepler:13}
एक टेलीस्कोप एक एक्सोप्लैनेट को अपने तारे का चक्कर लगाता हुआ पाता है

\omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} और सन्स में. कौन सा द्रव्यमान --- तारे का या ग्रह का ---

\end{exercise}

मंगल पर बसने वाले लोग अपने आधार पर एक ``एरोस्टेशनरी'' रिले चाहते हैं।

की गणना करें \omterm{def:g10:universal-gravitation:satellite}{कक्षा} त्रिज्या और मंगल ग्रह से ऊपर की ऊँचाई

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:satellites-kepler:15}
हैली धूमकेतु गुजरता है \omterm{def:g12:satellites-kepler:ellipse}{सूर्य समीपक} पर $ r_p = \qty{8.77e10}{m}$ चल रहा है
पर \qty{54.5}{km/s}; \omterm{def:g12:satellites-kepler:ellipse}{नक्षत्र} पर है $ r_a = \qty{5.25e12}{m}$. पर
दोनों चरम वेग सूर्य-धूमकेतु के लंबवत हैं
खंड, और फिर केप्लर का \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} नियम बल $ r_p v_p = r_a v_a $
(क्षेत्रफल के समान स्वीप त्रिकोण $\tfrac12 r v\,\Delta t $). निष्कर्ष निकालें
\omterm{def:g12:satellites-kepler:ellipse}{नक्षत्र} \omterm{def:g12:newtons-laws:momentum}{गति} और अनुपात पर टिप्पणी.
\end{exercise}

\section{समस्या: भूमध्य रेखा के ऊपर एक सीट}

\begin{problem}[{सप्ताहांत समस्या--- एक डाल रहा हूँ \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} के ऊपर
भूमध्य रेखा: आकाश में एकमात्र पार्किंग स्थल एक वृत्त क्यों है?
\qty{35800}{km} ऊपर, इसकी तुलना नीचे दिए गए स्टेशन रेसिंग से कैसे की जाती है,
और वही नियम रास्ते में बृहस्पति को कैसे तौलता है}]
\label{pb:g12:satellites-kepler:1}
एक टेलीकॉम कंपनी एक ऐसी रिले चाहती है जिससे उसके ग्राउंड डिश बंद हो जाएं
एक बार, कभी पीछा नहीं करना पड़ेगा: a \omterm{def:g12:satellites-kepler:geo}{भू-स्थिर} \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}. आप ही हैं
मिशन विश्लेषक; डेटा कार्ड का उपयोग करें. भाग~IV के लिए: बृहस्पति का चंद्रमा Io
कक्षाओं पर $ r = \qty{4.22e8}{m}$ में $ T = \qty{1.77}{d}$, और बृहस्पति
में तारों के सापेक्ष बदल जाता है \qty{9}{h} \qty{56}{min}.

\medskip
\noindent\textbf{Part I --- The rules of hanging still.}
\begin{enumerate}
  \item  \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} एक निश्चित जमीनी बिंदु पर रहना चाहिए। क्या
        यह होना चाहिए \omterm{def:g12:kinematics-2d:ucm}{अवधि} हो --- और क्यों \omterm{def:g12:satellites-kepler:ellipse}{नक्षत्र} दिवस
        \qty{86164}{s} इसके बजाय \qty{24}{h}?
  \item गोलाकार \omterm{def:g12:newtons-laws:momentum}{गति} में \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} वृत्त पर बिंदु
        केंद्र; यहाँ केवल बल गुरुत्वाकर्षण है. यह निष्कर्ष निकालें कि
        किसी भी गोलाकार का केंद्र \omterm{def:g10:universal-gravitation:satellite}{कक्षा} पृथ्वी का केंद्र है.
  \item पेरिस के अक्षांश के ऊपर मंडराता एक वृत्त केन्द्रित होगा
        पृथ्वी की धुरी पर, केंद्र के उत्तर में। निष्कर्ष: जिसमें
        विमान अवश्य होना चाहिए भूस्थैतिक \omterm{def:g10:universal-gravitation:satellite}{कक्षा} झूठ?
  \item लिखना \omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम} में \omterm{def:g12:kinematics-2d:frenet}{फ़्रेनेट फ़्रेम} एक परिपत्र के लिए
        \omterm{def:g10:universal-gravitation:satellite}{कक्षा} त्रिज्या का $ r $ और व्युत्पन्न $ v = \sqrt{GM/r}$.
  \item परिणाम निकालना $ T = 2\pi\sqrt{r^3/(GM)}$ और इसे केप्लर के रूप में पुनः लिखें
        तीसरा नियम $ T^2/r^3 = 4\pi^2/(GM)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The one circle that \omterm{def:g11:work-of-force:work}{काम} करता है.}
\begin{enumerate}[resume]
  \item के लिए तीसरा नियम हल करें $ r $ और गणना करें \omterm{def:g12:satellites-kepler:geo}{भू-स्थिर}
        त्रिज्या.
  \item जमीन से ऊँचाई ज्ञात करें।
  \item से कक्षीय \omterm{def:g12:newtons-laws:momentum}{गति} की गणना करें $ v = 2\pi r/T $.
  \item इसके विरुद्ध जाँच करें $ v = \sqrt{GM/r}$.
  \item कंपनी की \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} द्रव्यमान है \qty{4.0e3}{kg}; एक प्रतिद्वंद्वी का
        उसका वजन दोगुना है. उनकी तुलना करें \omterm{def:g12:satellites-kepler:geo}{भू-स्थिर} त्रिज्या, और
        कहें कि प्रक्षेपण के समय अतिरिक्त द्रव्यमान में क्या परिवर्तन होता है।
\end{enumerate}

\noindent\textbf{Part III --- The station below.}
\begin{enumerate}[resume]
  \item अंतरिक्ष स्टेशन उड़ान भरता है $ r = \qty{6.77e6}{m}$: गणना करें
        इसकी \omterm{def:g12:newtons-laws:momentum}{गति}.
  \item इसकी गणना करें \omterm{def:g12:kinematics-2d:ucm}{अवधि}, सेकंड और मिनटों में।
  \item कितने कक्षाओं क्या यह पूरा होता है? \qty{24}{h} --- के बारे में
        अंतरिक्ष यात्री प्रति दिन कितने सूर्योदय देखते हैं?
  \item त्रिज्या के अनुपात की गणना करें $ r_{\text{geo}}/r_{\text{ISS}}$,
        और सत्यापित करें कि का अनुपात \omterm{def:g12:kinematics-2d:ucm}{अवधि} क्या वह संख्या है
        \omterm{def:g11:work-of-force:power}{शक्ति} $3/2$, जैसा कि केप्लर का तीसरा नियम मांग करता है।
  \item कौन \omterm{def:g10:universal-gravitation:satellite}{उपग्रह} तेजी से चलता है, और जिसने अधिक लिया ऊर्जा प्रति
        \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} स्थापित करने के लिए? दोनों को एक वाक्य में मिलाएँ।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Weighing Jupiter on the side.}
\begin{enumerate}[resume]
  \item आयो पर लागू केप्लर के तीसरे नियम से, बृहस्पति को व्यक्त करें
        द्रव्यमान $ M_J $ के अनुसार $ G $, $ r $ और $ T $.
  \item गणना करना $ M_J$.
  \item वह पृथ्वी का द्रव्यमान कितना है? का लगभग कितना अंश
        सूर्य का द्रव्यमान?
  \item एक ``जोविस्टेशनरी'' रिले को एक बिंदु पर लटका होना चाहिए
        बृहस्पति के बादल: इसकी गणना करें \omterm{def:g10:universal-gravitation:satellite}{कक्षा} त्रिज्या; तुलना करें
        आयो का.
  \item बोर्ड को रिपोर्ट करें, दो वाक्य: ऊंचाई जहां
        कंपनी को अपना रिले और बृहस्पति का द्रव्यमान पार्क करना होगा
        वही कानून मुफ़्त में दिया गया।
\end{enumerate}
\end{problem}
