\chapter{रेडियोधर्मी क्षय}\label{ch:g12:radioactive-decay}

एक संग्रहा\omterm{def:g12:sound-acoustics:timbre}{लय} के तहखाने में, एक गीगर काउंटर लकड़ी के एक टुकड़े पर क्लिक करता है
फिरौन का ताबूत. प्रत्येक क्लिक अराजक है: किसी ने भी इसकी घोषणा नहीं की, कुछ भी नहीं
अगले की भविष्यवाणी करता है. फिर भी शुक्रवार तक लैब एक अच्छी तारीख प्रिंट कर देगी
सदी. यह अध्याय इस विरोधाभास को हल करता है --- कैसे घटनाएँ पूरी तरह से यादृच्छिक होती हैं
एक-एक करके, खरबों तक, हमारी सबसे \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} घड़ियों को जोड़ें।

\section{एक-एक करके अराजक, तिल-तिल कर घड़ी}

पिछले साल (\cref{ch:g11:nucleus-radioactivity}) हम मिले \omterm{prop:g12:work-and-energy:equilibria}{अस्थिर}
\omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, उनके संतुलित $\alpha $ और $\beta $ समीकरण, और किसी का आधा देखा
प्रति बड़ा नमूना गायब हो गया \omterm{def:g11:nucleus-radioactivity:halflife}{हाफ लाइफ} $ T_{1/2}$. जो हम नहीं कह सके वो था
\emph{जब सीढ़ियों के बीच} --- न ही एक यादृच्छिक प्रक्रिया क्यों रहती है
बिल्कुल शेड्यूल करें। दोनों सवालों का एक ही जवाब है.

\begin{definition}[क्षय स्थिरांक]\label{def:g12:radioactive-decay:lambda}
प्रत्येक के लिए \omterm{prop:g12:work-and-energy:equilibria}{अस्थिर} \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड} एक \omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक है $\lambda $, इसका \emph{क्षय स्थिर}\index{क्षय स्थिर} (\omterm{def:g10:orders-of-magnitude:unit}{इकाई} \unit{s^{-1}}), जैसे कि किसी के दौरान
लघु अंतराल $\dd t $ प्रत्येक \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} संभावना है $ p = \lambda\,\dd t $ का
क्षय --- हर किसी के लिए समान \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} प्रजाति की, चाहे उसकी उम्र कुछ भी हो,
और तापमान से अछूता, \omterm{def:g10:pressure:pressure}{दबाव} या रसायन शास्त्र.
\end{definition}

\begin{remark}[अकेले अराजकता, भीड़ में सटीक]\label{rem:g12:radioactive-decay:crowds}
ए \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} जिसने दस हजार वर्षों तक प्रतीक्षा की है, उसके नष्ट होने की बिल्कुल संभावना है
यह \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} जैसा कि आज सुबह बनाया गया था: कोई घिसाव नहीं, कोई चेतावनी नहीं --- जो
कोई आगे जाता है वास्तव में अप्रत्याशित है। भीड़ दूसरी बात है. टॉस
$100$ सिक्के: उम्मीद $50$ सिर, लेकिन इससे आश्चर्यचकित न हों $43$ ---
क्रम में उतार-चढ़ाव $\sqrt n $, यहाँ $10\%$. टॉस $10^{20}$: द
सापेक्ष उतार-चढ़ाव $1/\sqrt n $ है $10^{-10}$, परिणाम निश्चित है
दस दशमलव. एक ग्राम पदार्थ में कुछ न कुछ होता है $10^{22}$ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}: अराजकता
घड़ी की कल में औसत, के लिए एक सटीक कानून $ N(t)$, \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}ों की संख्या
समय पर अभी भी बरकरार है $ t $.
\end{remark}

\section{क्षय नियम}

\begin{proposition}[विकास समीकरण]\label{prop:g12:radioactive-decay:ode}
के एक बड़े नमूने में $ N $ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, दौरान $\dd t $ संख्या बदल जाती है
\[
\dd N = -\lambda N\,\dd t, \qquad\text{i.e.}\qquad \frac{\dd N}{\dd t} = -\lambda N :
\]
जनसंख्या स्वयं के अनुपातिक दर से सिकुड़ती है।
\end{proposition}

\begin{proof}
हरेक $ N $ के दौरान \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} का क्षय होता है $\dd t $ संभाव्यता के साथ
$\lambda\,\dd t $: अपेक्षित क्षय $ N\lambda\,\dd t $, और बड़े के लिए $ N $ द
वास्तविक गिनती अपेक्षित के अनुरूप है
(\cref{rem:g12:radioactive-decay:crowds}); प्रत्येक क्षय एक को हटा देता है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}.
\end{proof}

\begin{theorem}[रेडियोधर्मी क्षय का नियम]\label{thm:g12:radioactive-decay:law}
एक नमूना धारण $ N_0$ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} पर $ t = 0$ समय पर धारण करता है $ t $,
\[
N(t) = N_0 \, e^{-\lambda t}.
\]
\end{theorem}

\begin{proof}
प्रतिस्थापन द्वारा जाँच करें:
$\dd N/\dd t = N_0(-\lambda)\,e^{-\lambda t} = -\lambda N(t)$, और
$ N(0) = N_0 e^0 = N_0$: समीकरण और दोनों सही से शुरू करें। (वो नहीं
\emph{अन्य} यूनिवर्सिटी में इंटीग्रल के साथ फंक्शन फिट साबित होता है
आयतन.)
\end{proof}

\begin{definition}[घातीय क्षय]\label{def:g12:radioactive-decay:exponential}
एक मात्रा निम्नलिखित है $ N_0\,e^{-\lambda t}$ से होकर गुजरती है \emph{घातीय क्षय}\index{घातीय क्षय}: समान समय में, बराबर \emph{अंशों}
खो गये। यह बिल्कुल डिस्चार्जिंग है \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} का
\cref{ch:g12:rc-rl-circuits}, साथ $\lambda $ खेलना $1/(RC)$: एक
एक बार सीखा गया अंतर समीकरण, सर्किट और \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} को समान रूप से चलाता है।
\end{definition}

\begin{proposition}[हाफ लाइफ]\label{prop:g12:radioactive-decay:halflife}
 \omterm{def:g11:nucleus-radioactivity:halflife}{हाफ लाइफ} (\cref{def:g11:nucleus-radioactivity:halflife}) का ए \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड}
इसके द्वारा निर्धारित किया गया है \omterm{def:g12:radioactive-decay:lambda}{क्षय स्थिर}ांक: $ T_{1/2} = \ln 2/\lambda $, और उसके बाद $ n $
आधा जीवन $ N = N_0/2^{\,n}$, साबुत $ n $ या नहीं।
\end{proposition}

\begin{proof}
$ N(T_{1/2}) = N_0/2$ आवश्यक है $ e^{-\lambda T_{1/2}} = \tfrac12$, अर्थात\
$\lambda T_{1/2} = \ln 2$; और $ N(n\,T_{1/2}) = N_0\,e^{-n\ln 2} =
N_0/2^{\,n}$ किसी के लिए $ n $: घातांक सीढ़ी को पिरोता है और भरता है
सीढ़ियों के बीच में.
\end{proof}

\section{गतिविधि: क्लिक गिनना}

\begin{proposition}[गतिविधि]\label{prop:g12:radioactive-decay:activity}
 \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} एक नमूने का --- प्रति सेकंड इसका क्षय होता है \omterm{def:g11:nucleus-radioactivity:activity}{बेक्वेरल}
(\cref{def:g11:nucleus-radioactivity:activity}) --- है
\[
\mathcal A = \lambda N, \qquad\text{so}\qquad \mathcal A(t) = \mathcal A_0 \, e^{-\lambda t}:
\]
जीवित जनसंख्या के समानुपाती और समान रूप से घातीय।
\end{proposition}

\begin{proof}
$ N $ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, प्रत्येक संभावना के साथ $\lambda $ प्रति सेकंड: $\lambda N $
प्रति सेकंड क्षय; गुणा \cref{thm:g12:radioactive-decay:law} द्वारा
$\lambda $ देता है $\mathcal A(t) = \mathcal A_0 e^{-\lambda t}$.
\end{proof}

\begin{example}[विशाल $ N $, छोटा $\lambda $]\label{ex:g12:radioactive-decay:body}
एक केला लगभग टिक जाता है \qty{15}{Bq}, लगभग एक मानव शरीर
\qty{8000}{Bq}, एक घन \omterm{def:g10:orders-of-magnitude:unit}{मीटर} कुछ पर ग्रेनाइट का \qty{e6}{Bq}
(\cref{ch:g11:nucleus-radioactivity}). मामूली क्लिक के पीछे बैठते हैं
खगोलीय जनसंख्या: शरीर में पोटेशियम-40 का योगदान होता है
$\mathcal A \approx \qty{4.4e3}{Bq}$ साथ $\lambda =
\qty{1.8e-17}{s^{-1}}$: $ N = \mathcal A/\lambda \approx \num{2.5e20}$
\omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, जिनमें से प्रत्येक सेकंड में बमुश्किल कुछ हज़ार आग उगलते हैं।
\end{example}

\begin{method}[आधा जीवन मापना]\label{met:g12:radioactive-decay:semilog}
\begin{enumerate}
  \item गीगर काउंटर के साथ क्रमिक बराबर पर गणना क्षय
        अंतराल: यह नमूने $\mathcal A $ लगातार समय पर.
  \item कथानक $\ln \mathcal A $ ख़िलाफ़ $ t $: तब से
        $\ln \mathcal A = \ln \mathcal A_0 - \lambda t $, \omterm{def:g12:radioactive-decay:exponential}{घातीय
        क्षय} के रूप में दिखाता है \emph{सरल रेखा} ---टेढ़ा डेटा का मतलब है
        कोई अन्य कानून, या का मिश्रण \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड}.
  \item पढ़ना $\lambda $ ढलान से कम के रूप में बंद; $ T_{1/2} = \ln 2/\lambda $.
\end{enumerate}
\end{method}

\begin{remark}[बेकरेल नुकसान नहीं पहुँचाते]\label{rem:g12:radioactive-decay:dose}
 Becquerel क्षय को गिनता है, क्षति को नहीं। द्वारा नुकसान का पता लगाया जाता है \omterm{rem:g11:nucleus-radioactivity:dose}{खुराक} में
\omterm{rem:g11:nucleus-radioactivity:dose}{सिवर्ट्स} (\cref{ch:g11:nucleus-radioactivity}), तौलना ऊर्जा
प्रति जमा किया गया \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} प्रत्येक विकिरण की विनाशकारीता से ऊतक का: a
\qty{5e8}{Bq} टेक्नेटियम स्कैन नियमित है, बहुत कम \omterm{def:g11:nucleus-radioactivity:activity}{बेक्वेरल} साँस लेने का
$\alpha $ उत्सर्जक नहीं हैं; परिवर्तित \unit{Bq} को \unit{Sv} है
विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} सामग्री.
\end{remark}

\section{डेटिंग: अनुपात से पढ़ने का समय}

क्षय नियम पीछे की ओर चलता है एक घड़ी है: बचे हुए अंश को मापें,
और घातांक बीते हुए समय को नाम देता है।

\begin{method}[रेडियोकार्बन डेटिंग]\label{met:g12:radioactive-decay:carbon}
\begin{enumerate}
  \item \emph{जीवित:} जीवित ऊतक वातावरण के साथ कार्बन का आदान-प्रदान करते हैं
        और इसका कार्बन-14 अनुपात --- एक होता है \omterm{def:g11:fundamental-interactions:ladder}{एटम} में $10^{12}$,
        देना $\mathcal A_0 = 13.6$ प्रति ग्राम कार्बन का प्रति मिनट क्षय होता है।
  \item \emph{मौत घड़ी शुरू करती है:} सेवन बंद हो जाता है, और अनुपात
        के साथ क्षय हो जाता है $ T_{1/2} = \num{5730}$ साल।
  \item नमूने को मापें \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} $\mathcal A $ प्रति ग्राम कार्बन;
        $\mathcal A = \mathcal A_0 e^{-\lambda t}$ को उलट देता है
        $ t = \frac{1}{\lambda}\ln(\mathcal A_0/\mathcal A)
        = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}
\end{method}

\begin{example}[एक गुफा में चूल्हा]\label{ex:g12:radioactive-decay:hearth}
दबे हुए चूल्हे से कोयला निकलता है $5.0$ प्रति मिनट प्रति ग्राम क्षय:
$\mathcal A_0/\mathcal A = 13.6/5.0 = 2.72$, इसलिए
$ t = \num{5730} \times \ln(2.72)/\ln 2 \approx \num{8.3e3}$ साल। द
आग आठ हजार साल पहले बुझ गई ---लकड़ी ही संग्रह है।
\end{example}

\begin{example}[चट्टानों के लिए घड़ियाँ]\label{ex:g12:radioactive-decay:rocks}
कार्बन-14 कुछ से आगे शांत हो जाता है \num{50000} साल; भूविज्ञान धीमा रहता है
घड़ियाँ. \emph{पोटैशियम-आर्गन}: पोटैशियम-40 ($ T_{1/2} = \num{1.25e9}$
वर्ष) आर्गन में बदल जाता है, एक गैस जो \omterm{rem:g12:work-and-energy:escape}{निकल जाता है} पिघला हुआ लावा लेकिन एक बार फंस जाता है
चट्टान क्रिस्टलीकृत हो जाती है --- ठोसीकरण घड़ी को शून्य कर देता है। \emph{यूरेनियम--सीसा}:
यूरेनियम-238 क्षय की एक शृंखला पर समाप्त होता है \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} लीड-206
(\cref{ch:g11:nucleus-radioactivity}), और सीसा-से-यूरेनियम अनुपात में
क्रिस्टल इसे दिनांकित करता है। दोनों घड़ियाँ सबसे पुराने खनिजों पर सहमत हैं
($\approx \num{4.4e9}$ वर्ष) और, उल्कापिंडों के माध्यम से, सौर मंडल पर
उम्र.
\end{example}

\begin{remark}[जंजीरें, और तहखाने में गैस]\label{rem:g12:radioactive-decay:radon}
यूरेनियम और सीसे के बीच श्रृंखला रेडियम से होकर गुजरती है, फिर रेडॉन-222,
एक रेडियोधर्मी \emph{गैस} ($ T_{1/2} = 3.8$ दिन)। ग्रेनाइट में जन्मे और
मिट्टी, यह तहखानों में रिसती है और जमा हो जाती है; इसका $\alpha $-उत्सर्जन
बेटियों के फेफड़ों में सड़न --- अधिकांश लोगों का सबसे बड़ा प्राकृतिक \omterm{rem:g11:nucleus-radioactivity:dose}{खुराक}, और
कम करने के लिए सबसे सस्ता: तहखाने की खिड़की खोलें।
\end{remark}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:7}
दो छात्रों ने नमूने सील किए $20$ और \num{2.0e20} एक ही के \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}
\omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड}. प्रत्येक एक के बाद क्या अपेक्षा करता है \omterm{def:g11:nucleus-radioactivity:halflife}{हाफ लाइफ}? जिसकी भविष्यवाणी है
विश्वसनीय? (सापेक्ष उतार-चढ़ाव का अनुमान लगाएं $1/\sqrt n $ प्रत्येक के लिए।)

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:8}
एक शरीर धारण करता है \qty{140}{g} पोटेशियम का, जिसमें से $0.012\%$ है

पोटैशियम-40 \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}ों की संख्या की गणना करें, फिर उनकी \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि}; तुलना करें

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:9}

नीचे गिरना \qty{1}{MBq}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:10}
एक गड्ढे से निकला कोयला दिखाता है $3.4$ प्रति ग्राम कार्बन का प्रति मिनट क्षय;
दूसरे शो से हड्डी $30\%$ जीवन दर का ($\mathcal A_0 = 13.6$

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:radioactive-decay:11}
एक तहखाने को अंदर रेडॉन-222 के एक बैच के साथ सील कर दिया गया है ($ T_{1/2} = 3.8$
दिन)। कितने समय बाद बैच है \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} गिर गया $1\%$? \omterm{def:g11:lenses-and-eye:images}{असली}
तहखाने दशकों तक रेडियोधर्मी रहते हैं: ताज़ा रेडॉन कहाँ से आता है,
और वेंटिलेशन एकबारगी के बजाय स्थायी क्यों होना चाहिए?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:12}
से दिखाएँ $ N = N_0 e^{-\lambda t}$ वह $ t = T_{1/2}\ln(N_0/N)/\ln 2$. ए
हड्डी बरकरार रहती है $5.0\%$ इसके जीवित कार्बन-14 की: तारीख बताएं।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:13}
एक ऐसे क्रिस्टल में जिसमें जमने पर कोई आर्गन नहीं फंसा, प्रत्येक का क्षय हो गया
पोटैशियम-40 \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} एक आर्गन छोड़ता है \omterm{def:g11:fundamental-interactions:ladder}{एटम} जगह में। बताते हैं कि
$ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$; एक चट्टान एक अनुपात दर्शाती है
का $3.0$: इसकी आयु ज्ञात करें ($ T_{1/2} = \num{1.25e9}$ साल)।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:14}
एक गीगर काउंटर देता है, पर $ t = 0$, $60$, $120$, $180$, $240$ सेकंड,
की दरें $400$, $225$, $127$, $71$, $40$ प्रति सेकंड मायने रखता है. गणना करना
$\ln \mathcal A $ हर समय, संरेखण की जाँच करें, और निष्कर्ष निकालें $\lambda $ और
$ T_{1/2}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:radioactive-decay:15}
एक के लिए \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, जीवित रहने की संभावना $ n $ आधा जीवन है
$1/2^{\,n}$. प्रायिकता की गणना करें कि (ए) एक दिया गया है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} जीवित
$10$ आधा जीवन; (बी) सभी $8$ एक \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} को जीवित रहते देखा \omterm{def:g11:nucleus-radioactivity:halflife}{हाफ लाइफ}.
(सी) का \num{1000} \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, कितने के बाद अपेक्षित हैं $3$ आधा जीवन ---
और क्या कानून कह सकता है \emph{कौन}? कानून क्या भविष्यवाणी करता है उस पर निष्कर्ष निकालें।
\end{exercise}

\section{समस्या: चीज़ों की उम्र}

\begin{problem}[{सप्ताहांत समस्या--- चीजों की उम्र: एक संग्रहालय प्रयोगशाला
कार्बन घड़ी को कैलिब्रेट करता है, एक मूर्ति और एक पपीरस की तारीख तय करता है, सीखता है कि क्यों नहीं
घड़ी अपने डायल को पढ़ती है, और एक ज्वालामुखीय चट्टान को एक अरब वर्ष पीछे धकेल देती है}]
\label{pb:g12:radioactive-decay:1}
संग्रहा\omterm{def:g12:sound-acoustics:timbre}{लय} की डेटिंग प्रयोगशाला में एक सप्ताह। डेटा: जीवित ऊतक दिखाता है
$\mathcal A_0 = 13.6$ कार्बन-14 प्रति मिनट प्रति ग्राम कार्बन का क्षय होता है;
$ T_{1/2} = \num{5730}$ वर्ष (कार्बन-14), \num{1.25e9} साल
(पोटेशियम-40); $1$ वर्ष $= \qty{3.156e7}{s}$;
$ N_A = \qty{6.02e23}{mol^{-1}}$; कार्बन का दाढ़ द्रव्यमान \qty{12}{g/mol};
काउंटर इसके बारे में निर्णय लेते हैं $0.2$ प्रति ग्राम प्रति मिनट क्षय होता है।

\medskip
\noindent\textbf{Part I --- Monday: calibrating the clock.}
\begin{enumerate}
  \item जीवित ऊतक में कार्बन-14 का अनुपात \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} क्यों है, और
        मृत्यु पर क्या परिवर्तन होता है?
  \item गणना करना $\lambda $ कार्बन-14 इंच के लिए \unit{s^{-1}}.
  \item प्रतिस्थापन द्वारा सत्यापित करें कि $ N_0 e^{-\lambda t}$ हल करती है
        $\dd N/\dd t = -\lambda N $.
  \item से $\mathcal A_0$, कार्बन-14 की संख्या की गणना करें परमाणुओं प्रति
        जीवित ऊतक में ग्राम कार्बन.
  \item कितने कार्बन परमाणुओं क्या एक ग्राम धारण करता है? कार्बन-14 ज्ञात कीजिए
        अनुपात --- ``एक इंच'' से तुलना करें $10^{12}$''.
  \item दिखाओ कि एक मापा \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} $\mathcal A $ एक नमूने को दिनांकित करता है
        $ t = T_{1/2}\,\ln(\mathcal A_0/\mathcal A)/\ln 2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Tuesday: the statue and the papyrus.}
\begin{enumerate}[resume]
  \item लकड़ी की मूर्ति देती है $9.1$ प्रति ग्राम प्रति मिनट क्षय होता है। दिनांक
        लकड़ी.
  \item वास्तव में, वह तारीख किस घटना को चिह्नित करती है --- नक्काशी या
        कुछ और? क्या सावधानी बरतें?
  \item एक मिस्री पपीरस देता है $10.4$ प्रति ग्राम प्रति मिनट क्षय होता है। दिनांक
        यह. क्या बाईस शताब्दी पहले का कोई लेखक प्रशंसनीय है?
  \item  \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} से मापा जाता है $\pm 0.2$ प्रति मिनट क्षय होता है। अनुमान
        पपीरस की उम्र की अनिश्चितता, का उपयोग करना
        $\Delta t \approx \Delta\mathcal A/(\lambda \mathcal A)$.
  \item एक व्यापारी का ``प्राचीन'' चर्मपत्र देता है $13.4$ प्रति मिनट क्षय होता है
        प्रति ग्राम. निर्णय?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Thursday: the limits of the clock.}
\begin{enumerate}[resume]
  \item कितने अर्ध-जीवन के बाद एक नमूना बनता है? \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} से नीचे गिरना
        काउंटर' $0.2$ प्रति मिनट प्रति ग्राम क्षय? वह कौन सी उम्र है?
  \item सर्वोत्तम प्रयोगशालाएँ लगभग दस अर्ध-जीवन तक पहुँचती हैं। क्या है
        रेडियोकार्बन डेटिंग का व्यावहारिक क्षितिज, वर्षों में?
  \item एक डायनासोर की हड्डी के बारे में है \num{6.6e7} वर्ष पुराना: कितने
        कार्बन-14 अर्ध-आयु? प्रश्न 4 का उपयोग करते हुए, कितने आधे जीवन के बाद
        सम नहीं है \emph{एक} \omterm{def:g11:fundamental-interactions:ladder}{एटम} ग्राम का कार्बन-14 शेष है?
        ``कार्बन-डेटिंग डायनासोर'' के बारे में निष्कर्ष निकालें।
  \item किस तरह का \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड} क्या इसके बजाय डायनासोर की चट्टान की तारीख तय की जा सकती है?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Friday: the volcanic rock.}
\begin{enumerate}[resume]
  \item आर्गन एक गैस है, पोटेशियम एक ठोस का वफादार निवासी है: समझाएं
        क्यों लावा का जमना पोटेशियम--आर्गन घड़ी को शून्य कर देता है।
  \item प्रत्येक क्षयित पोटैशियम-40 को मानते हुए \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} एक को फँसा कर छोड़ देता है
        आर्गन \omterm{def:g11:fundamental-interactions:ladder}{एटम}, बताते हैं कि
        $ N_{\mathrm{Ar}}/N_{\mathrm K} = e^{\lambda t} - 1$.
  \item संग्रहा\omterm{def:g12:sound-acoustics:timbre}{लय} के जीवाश्म के नीचे की चट्टान दिखाई देती है
        $ N_{\mathrm{Ar}}/N_{\mathrm K} = 0.60$. चट्टान को दिनांकित करें.
  \item क्या कार्बन-14 ने इस चट्टान को दिनांकित किया होगा, या पोटैशियम--आर्गन ने
        मूर्ति? प्रत्येक घड़ी की अर्ध-आयु में उपयोगी विंडो दीजिए।
  \item शुक्रवार की रिपोर्ट, एक वाक्य: इस सप्ताह पाए गए तीन युग, और
        एकमात्र कानून जो उन सभी को पढ़ता है।
\end{enumerate}
