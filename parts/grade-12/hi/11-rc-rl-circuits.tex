\chapter{आरसी और आरएल सर्किट}\label{ch:g12:rc-rl-circuits}

कैमरे का शटर आधा दबाएँ: a \omterm{def:g11:lenses-and-eye:lens}{पतला} कराहना दो सेकंड के लिए चढ़ता है, ए
लाइट हरी हो जाती है --- तभी फ़्लैश जलेगा, एक मिलीसेकंड में डंप हो जाएगा
इसे इकट्ठा करने में दो सेकंड का समय लगा। यह अध्याय दो घटकों से मिलता है
जो सर्किट को एक मेमोरी और एक घड़ी देता है --- \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} और कुंडल --- और
भौतिकी के पहले विभेदक समीकरण, इस वर्ष के डेरिवेटिव के साथ जांचे जाने योग्य।

\section{संधारित्र}

\begin{definition}[संधारित्र और धारिता]\label{def:g12:rc-rl-circuits:capacitor}
ए \emph{संधारित्र}\index{संधारित्र} फेसिंग धातु प्लेटों की एक जोड़ी है
ए द्वारा अलग किया गया \omterm{def:g11:lenses-and-eye:lens}{पतला} इन्सुलेटर. एक स्रोत से तार, प्लेटें लेती हैं
विपरीत आरोप $+q $ और $-q $ --- एक \omterm{def:g11:electric-gravitational-fields:efield}{विद्युत क्षेत्र} अंतर भरता है
(\cref{ch:g11:electric-gravitational-fields}) --- और यह \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} $ u $
प्लेटों के पार संग्रहीत चार्ज के समानुपाती होता है:
\[ q = C\,u . \]
\omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक $ C $ है \emph{समाई}\index{समाई}, में
\emph{फ़ैराड}\index{फ़ैराड} (\unit{F}): $\qty{1}{F} = \qty{1}{C/V}$.
\end{definition}

\begin{example}[परिमाण के आदेश]\label{ex:g12:rc-rl-circuits:orders}
एक बिजली की एक विशेष नाप बहुत बड़ा है --- एक कूलम्ब प्रति पार्क किया गया वाल्ट. रेडियो के सिरेमिक
सर्किट: \unit{pF} को \unit{nF}; के इलेक्ट्रोलाइटिक्स \omterm{def:g11:work-of-force:power}{शक्ति} आपूर्ति और
चमक: \unit{\micro F} को \unit{mF}; सुपरकैपेसिटर: हजारों \omterm{def:g12:rc-rl-circuits:capacitor}{फ़ैराड}.
\end{example}

\begin{proposition}[एक संधारित्र में धारा]\label{prop:g12:rc-rl-circuits:cdudt}
ए पर करंट आ रहा है संधारित्रकी प्लेट वह दर है जिस पर इसकी
चार्ज बढ़ता है:
\[
i = \frac{dq}{dt} = C\,\frac{du}{dt} ,
\]
समय-भिन्न मात्राओं को लोअरकेस अंकन ($ U $, $ I $ \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} रहो,
\cref{ch:g11:circuits-and-power}).
\end{proposition}

\begin{proof}
करंट प्रति सेकंड दिया जाने वाला चार्ज है
(\cref{def:g11:circuits-and-power:current}): एक सिकुड़ते अंतराल पर,
$ i $ का व्युत्पन्न है $ q $; और $ q = Cu $ साथ $ C $ निरंतर देता है
$ C\,du/dt $.
\end{proof}

\begin{remark}[दो परिणाम]\label{rem:g12:rc-rl-circuits:continuity}
अगर $ u $ \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} है, $ i = 0$: एक आरोप लगाया \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} कोई \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} धारा प्रवाहित नहीं करता.
और $ u $ कभी भी छलांग नहीं लगा सकता --- एक कदम अनंत धारा की मांग करेगा:
\omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}'एस \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} सर्किट की मेमोरी है.
\end{remark}

\section{संधारित्र को चार्ज करना: आरसी सर्किट}

एक लूप---स्रोत $\mathcal{E}$ (\cref{def:g11:circuits-and-power:emf}),
स्विच, अवरोधक, खाली \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}: बंद करना $ K $ पर $ t = 0$.

\end{omfigure}

\begin{theorem}[आरसी सर्किट का चार्जिंग नियम]\label{thm:g12:rc-rl-circuits:charging}
\index{आरसी सर्किट}
स्विच बंद होने के बाद, \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} आज्ञापालन करता है, फिर अनुसरण करता है
$ u(0) = 0$:
\[
RC\,\frac{du}{dt} + u = \mathcal{E},
\qquad
u(t) = \mathcal{E}\left(1 - e^{-t/\tau}\right),
\qquad \tau = RC .
\]
\end{theorem}

\begin{proof}
\omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} लूप के साथ जोड़ें (\cref{prop:g11:circuits-and-power:laws}):
$\mathcal{E} = u_R + u = Ri + u $ हर पल; विकल्प
$ i = C\,du/dt $ (\cref{prop:g12:rc-rl-circuits:cdudt}). अब \emph{सत्यापित करें}
उम्मीदवार: $ du/dt = (\mathcal{E}/\tau)e^{-t/\tau}$, इसलिए
$ RC\,(\mathcal{E}/RC)\,e^{-t/\tau} + \mathcal{E}(1 - e^{-t/\tau}) =
\mathcal{E}$, सभी के लिए सच है $ t $; और $ u(0) = \mathcal{E}(1-1) = 0$. वो नहीं \emph{अन्य} वक्र समीकरण में फिट बैठता है और इसमें शुरुआत साबित होती है
विभेदक समीकरणों पर वर्ष का गणित पाठ्यक्रम।
\end{proof}

\begin{definition}[स्थिर समय]\label{def:g12:rc-rl-circuits:tau}
 \emph{स्थिर समय}\index{स्थिर समय} की एक \omterm{thm:g12:rc-rl-circuits:charging}{आरसी सर्किट} है
$\tau = RC $. इकाइयों: $\unit{\ohm} \times \unit{F} =
(\unit{V/A})(\unit{C/V}) = \unit{C/A} = \unit{s}$ --- सर्किट का
खुद की \omterm{def:g12:newtons-laws:momentum}{गति}.
\end{definition}

\begin{method}[पढ़ना $\tau $ वक्र के]\label{met:g12:rc-rl-circuits:reading}
तीन समतुल्य रीडिंग, सीधे एक ग्राफ़ से:
\begin{enumerate}
  \item \emph{मूल पर स्पर्शरेखा:} $ u'(0) = \mathcal{E}/\tau $, तो
        प्रारंभिक स्पर्श रेखा अनंतस्पर्शी तक पहुँचती है $ t = \tau $;
  \item \emph{63\%:} $ u(\tau) = \mathcal{E}(1 - e^{-1}) \approx
        0.63\,\mathcal{E}$;
  \item \emph{95\%:} $ u(3\tau) \approx 0.95\,\mathcal{E}$ --- आरोपित,
        व्यावहारिक उद्देश्यों के लिए ($5\tau $: $99\%$).
\end{enumerate}
\end{method}

\begin{proposition}[स्राव होना]\label{prop:g12:rc-rl-circuits:discharge}
ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} पर आरोप लगाया गया $ U_0$ और एक अवरोधक पर बंद कर दिया गया $ R $ अकेले ही पालन करता है
$ RC\,du/dt + u = 0$ और अनुसरण करता है
\[
u(t) = U_0\,e^{-t/\tau}, \qquad \tau = RC
\]
--- $37\%$ परित्यक्त $\tau $, $5\%$ पर $3\tau $.
\end{proposition}

\begin{proof}
समान लूप कानून, कोई स्रोत नहीं; पुनः प्रतिस्थापन:
$ RC(-U_0/\tau)e^{-t/\tau} + U_0 e^{-t/\tau} = 0$, साथ $ u(0) = U_0$.
\end{proof}

\begin{example}[एक फ़्लैश इकाई की संख्या]\label{ex:g12:rc-rl-circuits:flashrc}
एक फोटो फ्लैश चार्ज होता है $ C = \qty{800}{\micro F}$ के माध्यम से
$ R = \qty{1.0}{k\ohm}$: $\tau = 10^3 \times 8.0\times10^{-4} =
\qty{0.80}{s}$, इसलिए ``रेडी'' लाइट --- ट्रिप करने के लिए वायर्ड की गई $95\%$ ---
इंतज़ार करता है $3\tau \approx \qty{2.4}{s}$: हुक की \omterm{def:g10:signals-and-waves:sound}{आवाज़}. निकाल दिया गया
फ़्लैश ट्यूब के माध्यम से मात्र \qty{0.50}{\ohm}, द \emph{वही}
\omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} के साथ खाली हो जाता है $\tau' = \qty{0.40}{ms}$: दो हजार बार
अंदर से ज्यादा तेजी से बाहर.
\end{example}

\section{कुंडल: अधिष्ठापन और आरएल सर्किट}

\begin{definition}[प्रेरक और प्रेरकत्व]\label{def:g12:rc-rl-circuits:inductor}
एक \emph{प्रारंभ करनेवाला}\index{प्रारंभ करनेवाला} एक है कुंडल तार का: इसका करंट
ए के साथ धागे अपने आप बदल जाते हैं \omterm{def:g11:magnetic-fields:bfield}{चुंबकीय क्षेत्र}
(\cref{ch:g11:magnetic-fields}); जब भी यह \emph{परिवर्तन}, ए \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}
के पार दिखाई देता है कुंडल:
\[
u = L\,\frac{di}{dt} .
\]
\omterm{prop:g12:work-and-energy:equilibria}{स्थिर}ांक $ L $ है \emph{अधिष्ठापन}\index{अधिष्ठापन}, में
\emph{हेनरी}\index{हेनरी} (\unit{H}): $\qty{1}{H} = \qty{1}{V.s/A}$.
\end{definition}
\admitted

\begin{remark}[कुंडल परिवर्तन का विरोध करता है]\label{rem:g12:rc-rl-circuits:oppose}
बदलती धारा क्यों बनाती है? \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} --- प्रेरण --- में व्युत्पन्न होता है
विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} खंड; यहां हम कानून और उसके चरित्र को लेते हैं:
कुंडल \emph{परिवर्तन का विरोध करता है} इसके वर्तमान का. दर्पण जुड़वाँ: ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}'एस
$ u $ कूद नहीं सकता, ए कुंडल'एस $ i $ कूद नहीं सकता.
\end{remark}

\begin{circuitikz}[font=\small]
  \draw (0,0) to[battery1, l=$\mathcal{E}$] (0,2.4) to[nos, l=$K$] (2.1,2.4)
    to[R=$R$] (4.2,2.4) -- (5.4,2.4) to[L=$L$, i=$i$, v=$u$] (5.4,0) -- (0,0);
\end{circuitikz}

{\small  \omterm{prop:g12:rc-rl-circuits:rl}{आरएल सर्किट}: बाद $ K $ बंद हो जाता है, कुंडल वर्तमान देता है
केवल अपनी \omterm{def:g12:newtons-laws:momentum}{गति} से चढ़ें $\tau = L/R $.}
\end{omfigure}

\begin{proposition}[आरएल सर्किट में करंट बढ़ना]\label{prop:g12:rc-rl-circuits:rl}
\index{आरएल सर्किट}
किसी स्रोत पर स्विच बंद करना $\mathcal{E}$, एक अवरोधक $ R $ और ए कुंडल
$ L $ श्रृंखला में देता है $\mathcal{E} = Ri + L\,di/dt $, यानी
\[
\frac{L}{R}\,\frac{di}{dt} + i = \frac{\mathcal{E}}{R},
\qquad\text{solved by}\quad
i(t) = \frac{\mathcal{E}}{R}\left(1 - e^{-t/\tau}\right),
\quad \tau = \frac{L}{R} .
\]
\end{proposition}

\begin{proof}
पाश कानून फिर से; और यह है \emph{वही समीकरण} के रूप में
\cref{thm:g12:rc-rl-circuits:charging} साथ $ u \to i $,
$\mathcal{E} \to \mathcal{E}/R $, $ RC \to L/R $ --- वही प्रतिस्थापन
समाधान और प्रत्येक पाठन की पुष्टि करता है
\cref{met:g12:rc-rl-circuits:reading} स्थानान्तरण.
\end{proof}

\begin{example}[एक विद्युत चुम्बक जाग उठता है]\label{ex:g12:rc-rl-circuits:rl}
एक रिले कुंडल, $ L = \qty{0.30}{H}$ और $ R = \qty{6.0}{\ohm}$, एक पर
\qty{12}{V} आपूर्ति: अंतिम वर्तमान $\mathcal{E}/R = \qty{2.0}{A}$, पर
\omterm{def:g12:newtons-laws:momentum}{गति} $\tau = 0.30/6.0 = \qty{50}{ms}$ --- अंदर जागो $3\tau \approx
\qty{0.15}{s}$.
\end{example}

\begin{remark}[खुलने पर चिंगारी]\label{rem:g12:rc-rl-circuits:spark}
लाइव पर एक स्विच खोलें कुंडल और $ i $ एक अंश में शून्य पर क्रैश होना चाहिए
एक मिलीसेकेंड का: $ di/dt $ बहुत बड़ा है और $ u = L\,di/dt $ सैकड़ों तक पहुँच जाता है
या हजारों \omterm{def:g11:circuits-and-power:voltage}{वोल्ट} --- एक चिंगारी संपर्क को उछाल देती है। एक कार का इग्निशन
कुंडल यह जानबूझकर करता है, एक मिनट में हजारों बार।
\end{remark}

\section{संग्रहित ऊर्जा}

\begin{proposition}[संधारित्र में ऊर्जा, कुंडल में ऊर्जा]\label{prop:g12:rc-rl-circuits:energy}
\index{एक संधारित्र की ऊर्जा}\index{एक कुंडल की ऊर्जा}
ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} $ C $ पर आरोप लगाया गया \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} $ u $ और ए कुंडल $ L $ करंट ले जाना
$ i $ इकट्ठा करना
\[
E_C = \tfrac12\,C u^2 = \frac{q^2}{2C},
\qquad
E_L = \tfrac12\,L i^2 .
\]
\end{proposition}

\begin{proof}
एक छोटा सा चार्ज लैंडिंग $ dq $ पहले से ही प्लेटों पर \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} $ u $ लागत
$ u\,dq $ (\cref{def:g11:circuits-and-power:voltage}); ढेर लगाना
लागत, कुल रेखा के नीचे का क्षेत्र है $ u = q/C $ ---त्रिभुज
नीचे --- तो $ E_C = \tfrac12 qu = \tfrac12 Cu^2$. के लिए वही त्रिभुज
द कुंडल (उठाना $ i $ द्वारा $ di $ लागत $ Li\,di $) देता है
$ E_L = \tfrac12 Li^2$.
\end{proof}

त्रिभुज का क्षेत्रफल $ E_C = \tfrac12 q_0 u_0$ --- शुरुआती \omterm{def:g11:fundamental-interactions:charge}{कूलंब} बोर्ड

\begin{example}[परिमाण के ऊर्जा क्रम]\label{ex:g12:rc-rl-circuits:energies}
दमक \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} का \cref{ex:g12:rc-rl-circuits:flashrc}:
$\tfrac12 \times 8.0\times10^{-4} \times 300^2 = \qty{36}{J}$. ए
\qty{3000}{F} सुपरकैपेसिटर पर \qty{2.7}{V}: \qty{1.1e4}{J} --- एक ए.ए
बैटरी का मूल्य. ए \qty{10}{mH} कुंडल पर \qty{10}{A}: \qty{0.50}{J}.
बैटरी से कम \omterm{def:g10:energy-conservation:efficiency}{क्षमता} --- लेकिन एक मिलीसेकेंड में रिलीज करने योग्य।
\end{example}

\begin{remark}[दो टाइमकीपर किस लिए हैं]\label{rem:g12:rc-rl-circuits:applications}
\index{चौरसाई}\index{टाइमिंग सर्किट}
धीमे-धीमे, तेजी से बाहर पॉवर्स कैमरा फ्लैश और डिफाइब्रिलेटर।
\omterm{def:g12:mechanical-waves:delay}{देरी} $\tau = RC $ चलता है \omterm{rem:g12:rc-rl-circuits:applications}{टाइमिंग सर्किट}: रुक-रुक कर वाइपर का झपकना
बीकन, सीढ़ी रोशनी. ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} एक आपूर्ति के पार \emph{सहज बनाता है}
\omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} ($ u $ कूद नहीं सकता); एक शृंखला कुंडल धारा को सुचारू करता है ($ i $ नहीं कर सकते
कूदो)। और कुंडल प्लस \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} मिलकर एक बनाएं थरथरानवाला
(\cref{ch:g12:rlc-oscillations}).
\end{remark}

क्या यह रुका है? क्या \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} डालूँगा \qty{2.0}{mC} पर \qty{100}{\micro F}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rc-rl-circuits:2}
गुण \qty{47}{pF}, \qty{2200}{\micro F} और \qty{10}{F} एक के बीच
बिजली-आपूर्ति सुचारू, एक रेडियो ट्यूनर और एक सुपरकैपेसिटर; अंतिम गणना करें
एक का ऊर्जा पर \qty{2.7}{V}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rc-rl-circuits:3}
 \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} पार ए \qty{220}{\micro F} \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} से लगातार बढ़ता जाता है $0$
को \qty{5.0}{V} में \qty{10}{ms}: कौन सी धारा प्रवाहित होती है? एक बार क्या करंट
द \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} अभी भी कायम है, और क्यों?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rc-rl-circuits:4}
गणना करना $\tau $ के लिए $ R = \qty{10}{k\ohm}$, $ C = \qty{100}{\micro F}$, फिर
के लिए $ L = \qty{0.10}{H}$, $ R = \qty{50}{\ohm}$. जाँच करना, \omterm{def:g10:orders-of-magnitude:unit}{इकाई} द्वारा \omterm{def:g10:orders-of-magnitude:unit}{इकाई}, वह
दोनों संयोजन सेकंड हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:rc-rl-circuits:5}
ए \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} की ओर आरोप \qty{6.0}{V}, गुजर रहा है \qty{3.8}{V} पर
$ t = \qty{2.0}{s}$. पढ़ें \omterm{def:g12:rc-rl-circuits:tau}{स्थिर समय}; चार्ज कब है $95\%$ हो गया?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rc-rl-circuits:6}
एक \omterm{thm:g12:rc-rl-circuits:charging}{आरसी सर्किट}: $\mathcal{E} = \qty{9.0}{V}$, $ R = \qty{4.7}{k\ohm}$,
$ C = \qty{220}{\micro F}$. गणना करना $\tau $, $ u(\tau)$, $ u(3\tau)$, और
स्विच बंद होते ही करंट चालू हो जाता है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rc-rl-circuits:7}
प्रतिस्थापन द्वारा सत्यापित करें कि $ u(t) = \mathcal{E}(1 - e^{-t/RC})$
संतुष्ट $ RC\,du/dt + u = \mathcal{E}$; इसका मूल्य क्या है $ t = 0$
और यह बड़ा है-$ t $ भौतिक रूप से एन्कोड सीमित करें?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rc-rl-circuits:8}
ए \qty{100}{\micro F} \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} पर आरोप लगाया गया \qty{12}{V} के माध्यम से निर्वहन करता है
\qty{47}{k\ohm}: गणना करें $\tau $, $ u(\tau)$, और इसके लिए समय $ u $ गिराना
नीचे \qty{0.60}{V}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rc-rl-circuits:9}
ए में करंट \qty{0.50}{H} कुंडल से लगातार चढ़ता है $0$ को
\qty{2.0}{A} में \qty{40}{ms}: क्या \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} इसके पार दिखाई देता है? साथ
$ u = L\,di/dt $, क्यों करता है \emph{उद्घाटन} एक स्विच ऑन ए कुंडल चिंगारी, बंद नहीं हो रही?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rc-rl-circuits:10}
का रिले \cref{ex:g12:rc-rl-circuits:rl}: प्रतिस्थापन द्वारा सत्यापित करें
$ i(t) = (\mathcal{E}/R)(1 - e^{-Rt/L})$ संतुष्ट
$ L\,di/dt + Ri = \mathcal{E}$, फिर गणना करें $ i(\tau)$ और $ i(3\tau)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:rc-rl-circuits:11}
बताते हैं कि $ u'(0) = \mathcal{E}/\tau $, और उस पर स्पर्शरेखा का निष्कर्ष निकालें
मूल अनंतस्पर्शी से मिलता है $ t = \tau $. एक दर्ज वक्र पर वह
देता है $\tau = \qty{0.50}{s}$, साथ $ R = \qty{2.0}{k\ohm}$: खोजो $ C $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rc-rl-circuits:12}
संग्रहित द्वारा रैंक ऊर्जा, प्रत्येक की गणना: a \qty{3000}{F} सुपरकैपेसिटर
पर \qty{2.7}{V}; एक फ्लैश \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}, \qty{800}{\micro F} पर
\qty{300}{V}; ए \qty{10}{mH} कुंडल पर \qty{10}{A}; एक एए बैटरी
($ E \approx \qty{1.3e4}{J}$). पहले तीन क्या प्रस्ताव देते हैं कि
बैटरी नहीं कर सकती?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rc-rl-circuits:13}
इस दौरान एक सीढ़ी टाइमर जलता है $ u < 0.63\,\mathcal{E}$ इसके पार
चार्जिंग \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}, अंधेरा हो रहा है $ t = \tau $. प्रकाश को लंबे समय तक चलने के लिए डिज़ाइन करें
\qty{5.0}{s} का उपयोग करते हुए $ C = \qty{100}{\micro F}$: क्या $ R $? एक दहलीज के लिए
$0.50\,\mathcal{E}$, दिखाएं \omterm{def:g12:mechanical-waves:delay}{देरी} है $\tau \ln 2$ और इसकी गणना करें.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rc-rl-circuits:14}
ए \qty{2200}{\micro F} \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} आपूर्ति को सुचारू करता है: रिचार्ज के बीच,
यह अकेले ही एक लोड ड्राइंग को फीड करता है \qty{0.50}{A} के लिए \qty{10}{ms}. से
$ i = C\,du/dt $, गणना करें \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} गिरना, फिर \omterm{def:g12:rc-rl-circuits:capacitor}{समाई}
इसे नीचे रखते हुए \qty{0.50}{V}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:rc-rl-circuits:15}
एक प्रज्वलन कुंडल, $ L = \qty{0.80}{H}$, ले जाता है \qty{2.0}{A} जब
ब्रेकर इसे लगभग काट देता है \qty{1.0}{ms}. अनुमान लगाएं कुंडल \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}
कट और चिंगारी के दौरान ऊर्जा; धीमी रुकावट क्यों आती है
कोई चिंगारी न दें?
\end{exercise}

\section{समस्या: कैमरा फ़्लैश}

\begin{problem}[{सप्ताहांत समस्या --- कैमरा फ्लैश: दो सेकंड
कराहना, महिमा का एक मिलीसेकंड, और दीवार पर चढ़ा हुआ चचेरा भाई जो फिर से शुरू होता है
दिल --- एक घातांक उन सभी को चलाता है}]
\label{pb:g12:rc-rl-circuits:1}
एक कॉम्पैक्ट फ़्लैश \omterm{def:g10:orders-of-magnitude:unit}{इकाई} भंडार ऊर्जा में एक \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}
$ C = \qty{800}{\micro F}$, के माध्यम से चार्ज किया गया $ R = \qty{1.0}{k\ohm}$ से
ए \omterm{def:g10:energy-conservation:transfer}{कनवर्टर} एक आदर्श स्रोत के रूप में तैयार किया गया $\mathcal{E} = \qty{300}{V}$;
``तैयार'' प्रकाश यात्राएँ $95\%$ पूर्ण का \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}. निकाल दिया,
\omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} केवल फ़्लैश ट्यूब देखता है, $ R_f = \qty{0.50}{\ohm}$. डेटा: ए
एए सेल वितरित करता है \qty{1.5}{V} और \qty{2.4}{A.h}; $ e = \qty{1.60e-19}{C}$.

\medskip
\noindent\textbf{Part I --- The reservoir.}
\begin{enumerate}
  \item पर चार्ज की गणना करें \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र} पर \qty{300}{V}.
  \item कितने प्राथमिक आरोप यह है कि?
  \item संग्रहित की गणना करें ऊर्जा $ E_C $.
  \item एए सेल रखती है $ q = It \approx \qty{8.6e3}{C}$, इसलिए के बारे में
        \qty{1.3e4}{J}. कितनी बार \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}'एस ऊर्जा यह है कि?
  \item तो क्यों परेशान हों \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}? एक वाक्य, `` का प्रयोग करते हुएशक्ति''.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The wait for green.}
\begin{enumerate}[resume]
  \item लूप नियम लिखें और इसे विभेदक समीकरण में बदलें
        के लिए $ u(t)$.
  \item प्रतिस्थापन द्वारा सत्यापित करें कि $ u(t) = \mathcal{E}(1 - e^{-t/RC})$
        से शुरू करके इसे हल करता है $ u(0) = 0$.
  \item की गणना करें \omterm{def:g12:rc-rl-circuits:tau}{स्थिर समय} $\tau $.
  \item रेडी लाइट कितने समय बाद हरी हो जाती है और कितने बजे
        \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}? दो सेकंड की चीख को पहचानो.
  \item प्रारंभिक चार्जिंग करंट की गणना करें; फिर मरता क्यों है?
        दूर?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The millisecond of glory.}
\begin{enumerate}[resume]
  \item मुक्ति नियम लिखिए $ u(t)$ ट्यूब के माध्यम से और गणना करें
        नया \omterm{def:g12:rc-rl-circuits:tau}{स्थिर समय} $\tau'$.
  \item ट्यूब के माध्यम से अधिकतम धारा की गणना करें।
  \item शिखर की गणना करें \omterm{def:g11:work-of-force:power}{शक्ति}; इसकी तुलना ए से करें \qty{2.2}{kW} केतली।
  \item फ़्लैश को ओवर के रूप में लेना $3\tau'$, इसकी \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} बताएं और
        औसत \omterm{def:g11:work-of-force:power}{शक्ति}.
  \item चार्जिंग समय और फ्लैश समय का अनुपात बनाएं: समान \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र},
        वही आरोप --- क्या बदला, और इससे क्या बनता है
        \omterm{def:g12:rc-rl-circuits:capacitor}{संधारित्र}: एक ऊर्जा टैंक, या ए \omterm{def:g11:work-of-force:power}{शक्ति} लीवर?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The cousin on the wall.}
एक डिफाइब्रिलेटर अवश्य वितरित करना होगा $ E_C = \qty{200}{J}$ पर
$ U_0 = \qty{1500}{V}$; छाती और पैडल मौजूद हैं \qty{50}{\ohm}.
\begin{enumerate}[resume]
  \item आवश्यक की गणना करें \omterm{def:g12:rc-rl-circuits:capacitor}{समाई}.
  \item संग्रहित चार्ज की गणना करें.
  \item झटके की गणना करें \omterm{def:g12:rc-rl-circuits:tau}{स्थिर समय} और \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} ($3\tau $);
        फ़्लैश से तुलना करें.
  \item पीक करंट और पीक की गणना करें \omterm{def:g11:work-of-force:power}{शक्ति} रोगी में.
  \item रिचार्ज सर्किट को डिवाइस को तैयार करना चाहिए ($95\%$) में
        \qty{6.0}{s}: आवश्यक चार्जिंग ढूंढें \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} और
        आरंभिक चार्जिंग करंट --- वे संख्याएँ जो एक डिज़ाइनर ऑर्डर करेगा।
\end{enumerate}
\end{problem}
