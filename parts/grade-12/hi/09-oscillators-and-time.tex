\chapter{यांत्रिक दोलक और समय का मापन}\label{ch:g12:oscillators-and-time}

एक दादाजी की घड़ी टिक-टिक करती है: प्रत्येक टिक-टिक एक पीतल के पेंडुलम की तरह है जो पार करती है
लंबवत, नए साल के बाद से पंद्रह मिलियन बार दोहराया गया। एक क्वार्टज़
आपकी घड़ी के फ्लेक्स में स्लिवर \num{32768} एक सेकंड में कई बार; एक सीज़ियम \omterm{def:g11:fundamental-interactions:ladder}{एटम}
नौ अरब गुना तेज गुनगुनाता है। जो कुछ भी झूलता है वह समय रख सकता है:
यह अध्याय ह्यूजेन्स से लेकर स्विंगिंग तक का अनुसरण करता है \omterm{def:g10:universal-gravitation:satellite}{उपग्रह}ों.

\section{मुक्त दोलन}

\begin{definition}[थरथरानवाला, अवधि, आवृत्ति, आयाम]\label{def:g12:oscillators-and-time:oscillator}
ए \emph{यांत्रिक थरथरानवाला}\index{यांत्रिक थरथरानवाला} एक ऐसी प्रणाली है जो,
एक अस्तबल से विस्थापित \emph{संतुलन स्थिति}\index{संतुलन स्थिति} और जारी किया गया, इसके चारों ओर आगे और पीछे झूलता है: ये हैं
\emph{मुक्त दोलन}\index{मुक्त दोलन}. \omterm{def:g12:newtons-laws:momentum}{गति} दोहराई जाती है
के बाद \emph{अवधि}\index{अवधि} $T_0$ (\unit{s});
\emph{आवृत्ति}\index{आवृत्ति} $ f_0 = 1/T_0$
(\unit{Hz}, \cref{ch:g12:mechanical-waves});
\emph{आयाम}\index{आयाम} अधिकतम विस्थापन है.
\end{definition}

\begin{example}[ऑसिलेटर्स का एक चिड़ियाघर]\label{ex:g12:oscillators-and-time:zoo}
एक खेल का \omterm{def:g11:electric-gravitational-fields:field}{मैदान} झूला: $ T_0 \approx \qty{3}{s}$. एक गिटार की एक स्ट्रिंग:
$ f_0 = \qty{110}{Hz}$. एक क्वार्टज़ क्रिस्टल: \qty{32768}{Hz}. सीज़ियम
दूसरे को परिभाषित करने वाला दोलन: के बारे में \qty{9.2e9}{Hz}. दस आदेश
परिमाण का, एक \omterm{def:g11:work-of-force:work}{काम}: गिनती।
\end{example}

\section{सरल पेंडुलम}

\begin{definition}[सरल पेंडुलम]\label{def:g12:oscillators-and-time:pendulum}
ए \emph{सरल लोलक}\index{सरल लोलक}: द्रव्यमान का एक छोटा सा गोला
$ m $ लंबाई के एक अवितान्य तार पर $ L $ और नगण्य द्रव्यमान,
ऊर्ध्वाधर तल में झूलना। दोनों में से बल बॉब पर
(\cref{ch:g12:newtons-laws}), द \omterm{def:g10:inertia:inventory}{तनाव} के लंबवत है
\omterm{def:g12:newtons-laws:momentum}{गति}, जबकि वज़नके घटक के साथ \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} वापस इंगित करता है
निम्नतम बिंदु की ओर: \omterm{def:g10:universal-gravitation:weight}{वज़न} है \emph{पुनर्स्थापित कर रहा है बल}%
\index{बहाल बल},विस्थापन का विरोध कर रहे हैं।
\end{definition}

\begin{proposition}[सरल लोलक की अवधि]\label{prop:g12:oscillators-and-time:pendulum-period}
\index{सरल लोलक!अवधि}
छोटे के लिए \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} (नीचे के बारे में $15^\circ $), द \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} है
\[
T_0 = 2\pi \sqrt{\frac{L}{g}}, \qquad g = \qty{9.81}{m/s^2}.
\]
यह न तो बॉब के द्रव्यमान पर और न ही कोण पर निर्भर करता है
छोटा रहता है --- पर \omterm{def:g12:oscillators-and-time:oscillator}{आयाम}: छोटे-छोटे झूले हैं
\emph{समकालिक}\index{समकालिक}, सभी एक ही समय ले रहे हैं।
\end{proposition}
\admitted

\begin{remark}[आयामी विश्लेषण, और कहाँ $2\pi $ ज़िंदगियाँ]\label{rem:g12:oscillators-and-time:dimensional}
सूत्र का अंदाजा लगाया जा सकता है. से $ m $ (\unit{kg}), $ L $ (\unit{m}) और
$ g $ (\unit{m/s^2}), के साथ एकमात्र संयोजन \omterm{def:g10:orders-of-magnitude:unit}{इकाई} एक समय का है
$\sqrt{L/g}$ --- और जन \emph{नही सकता} दर्ज करें, कोई अन्य डेटा नहीं
एक ले जाना \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} इसे रद्द करने के लिए. \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} शुद्ध नहीं दे सकते
सामने नंबर: द $2\pi $ \omterm{def:g12:newtons-laws:momentum}{गति} के समीकरण को हल करने से आता है,
वर्ष 1 खंड में किया गया। न ही छोटे-कोण की स्थिति वैकल्पिक है:
पर $30^\circ $ सच \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} है $2\%$ लंबा, और समकालिकता विफल हो जाती है।
\end{remark}

\begin{example}[सेकंड पेंडुलम]\label{ex:g12:oscillators-and-time:seconds}
कौन सी लंबाई दूसरे को मात देती है, $ T_0 = \qty{2.00}{s}$, प्रत्येक पर एक टिक लगाएं
रास्ता? सुलझाना, $ L = g\,T_0^2/4\pi^2 = 9.81 \times 2.00^2/4\pi^2 \approx
\qty{0.994}{m}$. लगभग एक \omterm{def:g10:orders-of-magnitude:unit}{मीटर}: पेंडुलम घड़ियाँ लंबी होती हैं क्योंकि
सेकंड लंबे हैं.
\end{example}

\section{स्प्रिंग--द्रव्यमान थरथरानवाला}

\begin{definition}[एक स्प्रिंग की पुनः शक्ति बहाल करना]\label{def:g12:oscillators-and-time:spring}
द्रव्यमान का एक ग्लाइडर $ m $ एक क्षैतिज घर्षण रहित रेल पर स्लाइड, से बंधा हुआ
का एक झरना \emph{कठोरता}\index{कठोरता} $ k $ (\unit{N/m}). होने देना
$ x $ इसका संतुलन से विस्थापन हो। फैला हुआ या संकुचित, द
स्प्रिंग ग्लाइडर को खींचती है या पीछे धकेलती है बल $ F = -kx $:
विस्थापन के आनुपातिक और इसके विपरीत --- ए \omterm{def:g12:oscillators-and-time:pendulum}{बहाल बल} फिर से, अब लोच द्वारा आपूर्ति की गई।
\end{definition}

\begin{proposition}[गति का समीकरण और उसका समाधान]\label{prop:g12:oscillators-and-time:spring-period}
\index{स्प्रिंग--द्रव्यमान थरथरानवाला}
\omterm{thm:g12:newtons-laws:second}{न्यूटन का दूसरा नियम} रेल के किनारे (\cref{ch:g12:newtons-laws}) देता है
$ m \frac{d^2x}{dt^2} = -kx $, यानी\ $\frac{d^2x}{dt^2} =
-\frac{k}{m}\,x $. स्थिरांक जो भी हो $ A $ और $\varphi $,
\[
x(t) = A \cos\!\left(\frac{2\pi t}{T_0} + \varphi\right),
\qquad T_0 = 2\pi\sqrt{\frac{m}{k}},
\]
इस समीकरण को संतुष्ट करता है: दोलन के साथ \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} $ T_0$, स्वतंत्र
का \omterm{def:g12:oscillators-and-time:oscillator}{आयाम}.
\end{proposition}

\begin{proof}
विभेद करना, $\frac{dx}{dt} = -\frac{2\pi}{T_0}A \sin(\frac{2\pi
t}{T_0} + \varphi)$; दोबारा, $\frac{d^2x}{dt^2} = -(\frac{2\pi}{T_0})^2
A\cos(\frac{2\pi t}{T_0} + \varphi) = -(\frac{2\pi}{T_0})^2\,x(t)$,
जो है $-(k/m)\,x $ बिल्कुल कब $(2\pi/T_0)^2 = k/m $.
\end{proof}

\begin{remark}[वहां जितनी भी गतियां हैं]\label{rem:g12:oscillators-and-time:uniqueness}
वह \emph{प्रत्येक} ग्लाइडर की \omterm{def:g12:newtons-laws:momentum}{गति} इन कार्यों में से एक है
स्वीकार किया गया, वर्ष 1 खंड में सिद्ध किया गया। और नहीं $ g $ में $2\pi\sqrt{m/k}$:
कठोर तेजी से झूलता है, भारी धीमा, चंद्रमा पर भी ऐसा ही होता है।
\end{remark}

\begin{definition}[आयाम और चरण]\label{def:g12:oscillators-and-time:phase}
में $ x(t) = A\cos(2\pi t/T_0 + \varphi)$ सकारात्मक स्थिरांक $ A $ है
द \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} --- चरम हैं $ x = \pm A $ --- और $2\pi t/T_0 +
\varphi $ है \emph{चरण}\index{चरण}, बढ़ रहा है $2\pi $ प्रत्येक
\omterm{def:g12:oscillators-and-time:oscillator}{अवधि}; \emph{प्रारंभिक चरण}\index{प्रारंभिक चरण} $\varphi $
प्रारंभ को रिकॉर्ड करता है: \omterm{def:g10:relative-motion:frame}{आराम से} जारी किया गया $ x = A $, $\varphi = 0$. उदाहरण के लिए $ m = \qty{0.20}{kg}$ पर $ k = \qty{20}{N/m}$ देता है $ T_0 =
2\pi\sqrt{0.20/20} = \qty{0.63}{s}$; की ओर खींचा गया \qty{5.0}{cm} और
जारी किया गया, $ x(t) = 5.0\cos(2\pi t/0.63)$ सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} में.
\end{definition}

\section{ऊर्जा, अवमंदन, प्रतिध्वनि}

\begin{proposition}[थरथरानवाला की ऊर्जा]\label{prop:g12:oscillators-and-time:energy}
एक झरना फैला हुआ है $ x $ लोचदार \omterm{def:g10:energy-conservation:efficiency}{क्षमता} को संग्रहीत करता है ऊर्जा $ E_p =
\tfrac12 kx^2$ (में उचित ठहराया गया \cref{ch:g12:work-and-energy}). दौरान
घर्षण रहित दोलन \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा}
\[
E_m = E_k + E_p = \tfrac12 m v^2 + \tfrac12 k x^2 = \tfrac12 k A^2
\]
स्थिर है: चरम पर सभी लोचदार, संतुलन पर सभी गतिज।
पेंडुलम बीच में वही युगल गीत बजाता है $ E_k $ और गुरुत्वाकर्षण
$ E_p $ (\cref{ch:g11:mechanical-energy}).
\end{proposition}

\begin{proof}
साथ $ v = \frac{dx}{dt}$, का व्युत्पन्न $ E_m $ है $ mv\frac{dv}{dt}
+ kx\frac{dx}{dt} = v\,(m\frac{d^2x}{dt^2} + kx) = 0$ समीकरण द्वारा
\omterm{def:g12:newtons-laws:momentum}{गति} का; मूल्य $\tfrac12 kA^2$ चरम पर पढ़ा जाता है.
\end{proof}

\begin{definition}[नम दोलन]\label{def:g12:oscillators-and-time:damping}
\omterm{def:g10:inertia:inventory}{टकराव} नालियों \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} हर वास्तविक से थरथरानवाला: इसका
\omterm{def:g12:oscillators-and-time:oscillator}{मुक्त दोलन} हैं \emph{अवमन्दित}\index{अवमन्दित}. प्रकाश के अंतर्गत
प्रणाली में गिरावट अभी भी दोलन कर रही है, एक समाप्ति के साथ \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} लेकिन ए
लगभग अपरिवर्तित दोहराव समय, \emph{छद्म काल}%
\index{छद्म काल}, के करीब $ T_0$; भारी नमी के तहत (शहद)
यह बिना किसी अतिरेक के संतुलन में वापस आ जाता है; ऊर्जा,
के लिए आनुपातिक \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} चुकता, गर्मी में सड़ जाता है।
\end{definition}

\begin{definition}[जबरन दोलन और प्रतिध्वनि]\label{def:g12:oscillators-and-time:resonance}
एक धक्का थरथरानवाला समय-समय --- \emph{मजबूर दोलन}%
\index{मजबूर दोलन} --- और थोड़ी देर के बाद इसे क्षणिक करें
पर दोलन करता है \emph{ड्राइविंग} \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति} $ f $, अपना नहीं. इसका
\omterm{def:g12:oscillators-and-time:oscillator}{आयाम} शिखर कब $ f $ के पास \emph{प्राकृतिक \omterm{def:g12:oscillators-and-time:oscillator}{आवृत्ति}}%
\index{प्राकृतिक आवृत्ति} $ f_0$ मुफ़्त का थरथरानवाला: यह है
\emph{गूंज}\index{गूंज}, प्रत्येक धक्का कदम से कदम मिलाकर आ रहा है
\omterm{def:g12:newtons-laws:momentum}{गति}, खिलाना ऊर्जा हर चक्र; उतना ही हल्का भिगोना,
चोटी जितनी ऊँची और नुकीली होगी।
\end{definition}

\section{समस्या: घड़ीसाज़ की बेंच}

\begin{problem}[{सप्ताहांत समस्या --- घड़ीसाज़ की बेंच: एक दीवार
घड़ी को समय पर वापस लाया गया --- पचास घुमावों का समय, एक पीतल की छड़ी
गर्मियों में फैलता है, पलायन का फुसफुसाया धक्का, एक क्वार्ट्ज
चुनौती देने वाला --- प्रति माह सेकंड में दिया गया फैसला}]
\label{pb:g12:oscillators-and-time:1}
एक पेंडुलम दीवार घड़ी बेंच पर ``गलत तरीके से चल रही है'' आती है। इलाज
इसकी पीतल की \omterm{def:g11:color-light-sources:cones}{छड़} और भारी बॉब \omterm{def:g12:oscillators-and-time:pendulum}{सरल लोलक} प्रभावी लंबाई का
$ L $, बॉब मास $ m = \qty{1.2}{kg}$; गियरिंग प्रत्येक पूर्ण को गिनती है
बिल्कुल वैसा ही दोलन \qty{2.000}{s}.

\medskip
\noindent\textbf{Part I --- Taking the clock's pulse.}
\begin{enumerate}
  \item समय क्यों? $50$ एक के बजाय दोलन? त्रुटि का अनुमान लगाएं
        पर छोड़ दिया $ T_0$ यदि आपकी प्रतिक्रिया का समय है \qty{0.2}{s}.
  \item स्टॉपवॉच पढ़ता है \qty{99.4}{s} के लिए $50$ दोलन: \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}?
  \item लोलक की वर्तमान प्रभावी लंबाई ज्ञात कीजिए।
  \item बिल्कुल कौन सी लंबाई देगी \qty{2.000}{s}? चाहिए
        रेटिंग नट बॉब को बढ़ाएँ या घटाएँ, और कितना?
  \item असमायोजित, घड़ी प्रतिदिन कितने सेकंड तेज है या धीमी?
  \item के प्रति संवेदनशीलता $ g $: द \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} चंद्रमा पर
        ($ g = \qty{1.62}{m/s^2}$)? वेधशाला में दैनिक त्रुटि
        कहाँ $ g = \qty{9.80}{m/s^2}$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The summer slump.} पीतल \omterm{def:g11:color-light-sources:objectcolor}{फैलता है}: ए
वार्मिंग $\Delta\theta $ \omterm{def:g11:color-light-sources:cones}{छड़}ी को फैलाता है $ L' = L\,(1 + \lambda\,
\Delta\theta)$, $\lambda = \num{1.9e-5}$ प्रति \omterm{def:g10:light-spectra:kelvin}{केल्विन}. घड़ी है
पर समायोजित किया गया \qty{20}{\celsius}.
\begin{enumerate}[resume]
  \item गणना करना $\Delta L $ एक ग्रीष्मकालीन कमरे के लिए \qty{25}{\celsius}.
  \item का उपयोग करते हुए $\sqrt{1+x} \approx 1 + x/2$, बताते हैं कि
        $\Delta T_0 / T_0 = \tfrac12 \lambda\, \Delta\theta $.
  \item गणना करना $\Delta T_0/T_0$ और दैनिक ग्रीष्मकालीन त्रुटि: तेज़ या धीमी?
  \item विंटर हॉलवे के लिए भी वही प्रश्न \qty{10}{\celsius}.
  \item पुरानी सटीक घड़ियों में पीतल को मिलाने वाले ``ग्रिडिरॉन'' पेंडुलम का उपयोग किया जाता था
        स्टील वाली \omterm{def:g11:color-light-sources:cones}{छड़}ें (जो लगभग आधी तक फैलती हैं)।
        वैकल्पिक दिशाएँ. बताएं कि यह दर को कैसे बनाए रख सकता है।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The whispered push.} पलायन के साथ
विच्छेदित, झूला, शुरू हुआ \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} $\theta_m = 2.00^\circ $,
इसका आधा \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} में \qty{280}{s}; सेवा में, की ओर से एक छोटा सा धक्का
प्रत्येक का पलायन \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} रखता है $\theta_m $ स्थिर।
\begin{enumerate}[resume]
  \item कितने \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} क्या आधा करने में समय लगता है?
  \item चरम पर बॉब चढ़ गया है $ h = L(1 - \cos\theta_m)
        \approx L\,\theta_m^2/2$ (रेडियन): निष्कर्ष निकालें $ E_m = \tfrac12
        m g L\,\theta_m^2$ और इसकी गणना करें
        (\cref{ch:g11:mechanical-energy}).
  \item ऊर्जा के लिए आनुपातिक है \omterm{def:g12:oscillators-and-time:oscillator}{आयाम} वर्ग: का कौन सा अंश
        $ E_m $ प्रति खो गया है \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}?
  \item निष्कर्ष निकालें ऊर्जा प्रत्येक धक्का का, और औसत \omterm{def:g11:work-of-force:power}{शक्ति} आवश्यकता है।
  \item ड्राइव \omterm{def:g10:universal-gravitation:weight}{वज़न} $ M = \qty{2.5}{kg}$ उतरता $ h =
        \qty{1.0}{m}$ प्रति सप्ताह: क्या यह पर्याप्त है ऊर्जा, किस पर \omterm{def:g10:energy-conservation:efficiency}{क्षमता}?
\end{enumerate}

\noindent\textbf{Part IV --- The quartz challenger.} एक क्वार्टज़
\omterm{def:g12:newtons-laws:momentum}{गति} दोलन करती है $ f = \qty{32768}{Hz}$ और इधर-उधर भटकता रहता है
\qty{0.3}{s} प्रति महीने; पुनर्स्थापित पेंडुलम अभी भी बहता है
\qty{0.5}{s} प्रति दिन।
\begin{enumerate}[resume]
  \item क्वार्ट्ज की गणना करें \omterm{def:g12:oscillators-and-time:oscillator}{अवधि}. दिखाओ $\num{32768} = 2^{15}$: क्यों
        एक है \omterm{def:g11:work-of-force:power}{शक्ति} दो में से वास्तव में एक घड़ी सर्किट क्या चाहता है?
  \item क्वार्ट्ज और पुनर्स्थापित पेंडुलम के सापेक्ष बहाव की गणना करें,
        और प्रत्येक प्रति माह सेकंड में। कौन जीतता है, किस कारक से?
  \item सीज़ियम मानक --- \num{9192631770} \omterm{def:g12:oscillators-and-time:oscillator}{अवधि} एक बनाओ
        दूसरा--बहाव \qty{1}{s} प्रति सौ मिलियन वर्ष: यह
        सापेक्ष बहाव? क्वार्टज़ से नीचे परिमाण के कितने ऑर्डर,
        और जीपीएस इसकी मांग क्यों करता है?
  \item फैसला, संख्याओं के साथ एक वाक्य: बहाल क्या करता है
        दीवार पर घड़ी कमाई, और अब दूसरे का मालिक कौन है?
\end{enumerate}
\end{problem}
