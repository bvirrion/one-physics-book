\chapter{ध्वनि और ध्वनिकी}\label{ch:g12:sound-acoustics}

गिटार की धीमी तार बजाएं: फिर भी कमरा कम A से भर जाता है
निचोड़ी हुई और खिंची हुई हवा के अलावा आपके कान तक कुछ भी नहीं गया। यह
अध्याय उस पैटर्न का अनुसरण करता है --- इसकी गति, कैसे \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति} बन जाता है
\omterm{def:g12:sound-acoustics:pitch-loudness}{आवाज़ का उतार-चढ़ाव} और \omterm{def:g11:work-of-force:power}{शक्ति} \omterm{def:g12:sound-acoustics:pitch-loudness}{प्रबलता}, और प्रत्येक उपकरण पर एक नोट अलग-अलग क्यों होता है।

\section{ध्वनि एक दबाव तरंग है}

\begin{definition}[ध्वनि तरंग]\label{def:g12:sound-acoustics:sound-wave}
ए \emph{ध्वनि तरंग}\index{ध्वनि तरंग} एक यांत्रिक है \omterm{def:g10:pressure:pressure}{दबाव} \omterm{def:g10:signals-and-waves:wave}{लहर}
(\cref{ch:g12:mechanical-waves}): एक कम्पायमान सतह बनाता है
यात्रा \emph{संपीड़न}\index{संपीड़न} (मामूली
अधिक \omterm{def:g10:pressure:pressure}{दबाव}) और \emph{विरलन}\index{विरलन}
(\omterm{def:g10:pressure:pressure}{दबाव} में)। हवाई पार्सल दोलन करते हैं \emph{साथ में} यात्रा
दिशा (\omterm{def:g10:signals-and-waves:sound}{आवाज़} है \emph{अनुदैर्ध्य}\index{अनुदैर्ध्य}),
और एक माध्यम की आवश्यकता होती है: वैक्यूम जार के नीचे एक घंटी शांत हो जाती है
(\cref{ch:g10:signals-and-waves}).
\end{definition}

\begin{proposition}[ध्वनि की गति]\label{prop:g12:sound-acoustics:speed}
\index{ध्वनि की गति}%
\omterm{def:g10:signals-and-waves:sound}{आवाज़} के बारे में यात्रा करता है \qty{340}{m/s} हवा में
\qty{20}{\degreeCelsius}, \qty{1500}{m/s} पानी में और
\qty{5000}{m/s} स्टील में: माध्यम जितना सख्त होगा
तेज़. \admitted
\end{proposition}

\begin{remark}[गर्म हवा तेज होती है]\label{rem:g12:sound-acoustics:temperature}
ये मापे गए मान हैं; उनकी भविष्यवाणी वर्ष 1में की जाती है
आयतन. \omterm{def:g10:signals-and-waves:sound}{आवाज़} गर्म हवा में गति बढ़ जाती है (\qty{331}{m/s} पर
\qty{0}{\degreeCelsius}) --- हॉल गर्म होने पर हवाएं बेसुरी क्यों हो जाती हैं।
\end{remark}

\begin{example}[ध्वनियों का आकार]\label{ex:g12:sound-acoustics:wavelengths}
रिश्ता $\lambda = v/f $ आकार कोई भी \omterm{def:g10:signals-and-waves:sound}{आवाज़}: हवा में ए \qty{20}{Hz}
गड़गड़ाहट फैलती है $340/20 = \qty{17}{m}$ --- एक इमारत --- और एक
\qty{20}{kHz} फुफकार फिट बैठती है \qty{1.7}{cm}; \omterm{def:g10:orders-of-magnitude:unit}{मीटर} आकार वाले झुक जाते हैं
दरवाजे और \omterm{def:g11:color-light-sources:cones}{कोन}ों के आसपास आसानी से।
\end{example}

\section{पिच, ज़ोर और कान}

\begin{definition}[पिच और ज़ोर]\label{def:g12:sound-acoustics:pitch-loudness}
\emph{आवाज़ का उतार-चढ़ाव}\index{आवाज़ का उतार-चढ़ाव} है धारणा का \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}: जितना ऊँचा
$ f $, नोट जितना ऊँचा होगा; कॉन्सर्ट ए है \qty{440}{Hz}, और दोहरीकरण
$ f $ नोट को एक सप्तक बढ़ाता है। \emph{प्रबलता}\index{प्रबलता} है
द धारणा का \omterm{def:g10:signals-and-waves:amplitude}{आयाम}: उतना ही मजबूत \omterm{def:g10:pressure:pressure}{दबाव} दोलन,
जितना \omterm{def:g10:inertia:inventory}{जोर} से; प्रवर्धन से ध्वनि की ध्वनि बदलती है, पिच नहीं।
\end{definition}

\begin{remark}[कान की खिड़की]\label{rem:g12:sound-acoustics:audible}
श्रवण चारों ओर फैला हुआ है \qty{20}{Hz} को \qty{20}{kHz}
(\cref{ch:g10:signals-and-waves}); नीचे झूठ है \omterm{def:g10:signals-and-waves:ultrasound}{इन्फ्रासाउंड}, झूठ से ऊपर
\omterm{def:g10:signals-and-waves:ultrasound}{अल्ट्रासाउंड} ---बल्लेबाजी करता है \qty{50}{kHz}, मेगाहर्ट्ज़ पर चिकित्सा जांच। ए
पियानो केवल मध्य का नमूना लेता है: \qtyrange{27.5}{4186}{Hz}.
\end{remark}

\section{ध्वनि की तीव्रता और डेसीबल पैमाना}

\begin{definition}[ध्वनि की तीव्रता]\label{def:g12:sound-acoustics:intensity}
 \emph{ध्वनि की तीव्रता}\index{ध्वनि की तीव्रता} $ I $ है
ध्वनिक \omterm{def:g11:work-of-force:power}{शक्ति} एक चौक पार करना \omterm{def:g10:orders-of-magnitude:unit}{मीटर} का सामना करना पड़ रहा है \omterm{def:g10:signals-and-waves:wave}{लहर}:
$ I = P/S $, में \unit{W/m^2}. सबसे शांत श्रव्य \omterm{def:g10:signals-and-waves:sound}{आवाज़} के बारे में है
$ I_0 = \qty{1.0e-12}{W/m^2}$; दर्द निकट शुरू होता है \qty{1}{W/m^2}.
\end{definition}

\begin{proposition}[व्युत्क्रम-वर्ग क्षीणन]\label{prop:g12:sound-acoustics:inverse-square}
\index{व्युत्क्रम-वर्ग नियम}%
विकिरण करने वाला एक छोटा सा स्रोत \omterm{def:g11:work-of-force:power}{शक्ति} $ P $ सभी दिशाओं में समान रूप से
देता है, दूरी पर $ r $, $ I = P/(4\pi r^2) \propto 1/r^2$: दोहरीकरण
दूरी तीव्रता को चार से विभाजित करती है।
\end{proposition}

\begin{proof}
संपूर्ण \omterm{def:g11:work-of-force:power}{शक्ति} $ P $ क्षेत्र के दायरे को पार करता है $4\pi r^2$; विभाजित करना।
\end{proof}

\begin{definition}[ध्वनि का स्तर]\label{def:g12:sound-acoustics:level}
कान बारह तक फैला है पॉवर्स दस में से, इसलिए तीव्रताएँ उद्धृत की गई हैं
लघुगणकीय रूप से। \emph{ध्वनि का स्तर}\index{ध्वनि का स्तर} का $ I $ है
\[ L = 10 \log_{10}(I/I_0), \qquad I_0 = \qty{1.0e-12}{W/m^2}, \]
में \emph{डेसीबल}\index{डेसीबल} (\unit{dB}): श्रवण सीमा
\qty{0}{dB}, पास दर्द \qty{120}{dB}.
\end{definition}

\begin{proposition}[डेसीबल अंकगणित]\label{prop:g12:sound-acoustics:db-arithmetic}
गुणा $ I $ द्वारा $ k $ कहते हैं $10\log_{10}k $ \omterm{def:g12:sound-acoustics:level}{डेसीबल}. विशेष रूप से
$\times 10$ कहते हैं \qty{10}{dB}; $\times 2$ कहते हैं
$10\log_{10}2 \approx \qty{3.0}{dB}$; $ N $ असंगत समान
सूत्र जोड़ते हैं \emph{तीव्रता}, कभी स्तर नहीं, देना
$ L_1 + 10\log_{10}N $; एक छोटे स्रोत की दूरी दोगुनी करना ($ I $
चार से विभाजित) के बारे में हटा देता है \qty{6.0}{dB}.
\end{proposition}

\begin{proof}
$10\log_{10}(kI/I_0) = 10\log_{10}(I/I_0) + 10\log_{10}k $;
दूरी: \cref{prop:g12:sound-acoustics:inverse-square}, $ k = 1/4$.
\end{proof}

\begin{method}[डेसीबल में कार्य करना]\label{met:g12:sound-acoustics:db}
\begin{enumerate}
  \item $ L = 10\log_{10}(I/I_0)$; पीछे: $ I = I_0 \times 10^{L/10}$.
  \item कभी भी स्तर न जोड़ें: तीव्रता में बदलें, जोड़ें, वापस बदलें
        --- या $+\qty{3}{dB}$ प्रति दोहरीकरण, $+\qty{10}{dB}$ प्रति दस गुना.
  \item एक छोटे स्रोत की दूरी को दोगुना करने में प्रत्येक लागत खर्च होती है \qty{6}{dB}.
\end{enumerate}
\end{method}

\end{remark}

\section{टिम्ब्रे और हार्मोनिक्स}

\begin{definition}[हार्मोनिक्स]\label{def:g12:sound-acoustics:harmonics}
ए \omterm{def:g10:signals-and-waves:period}{आवधिक} \omterm{def:g10:signals-and-waves:sound}{आवाज़} का \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति} $ f_1$ --- द
\emph{मौलिक}\index{मौलिक} --- सामान्यतः का योग है
आवृत्तियों पर साइन्स $ f_n = n f_1$, $ n = 1, 2, 3, \dots $, इसका
\emph{हार्मोनिक्स}\index{हार्मोनिक्स}. ए \omterm{def:g10:signals-and-waves:sound}{आवाज़} इसके मौलिक स्तर तक कम कर दिया गया है
एक है \emph{शुद्ध स्वर}\index{शुद्ध स्वर}, एक ट्यूनिंग कांटा की तरह।
\end{definition}

\begin{definition}[स्पेक्ट्रम और समय]\label{def:g12:sound-acoustics:timbre}
 \emph{स्पेक्ट्रम}\index{स्पेक्ट्रम} एक का \omterm{def:g10:signals-and-waves:sound}{आवाज़} का बार चार्ट है
इसका \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स}: एक बार प्रति $ f_n $, वह ऊंचाई लयबद्धका
ताकत. कान सुनता है $ f_1$ के रूप में \omterm{def:g12:sound-acoustics:pitch-loudness}{आवाज़ का उतार-चढ़ाव} और यह \emph{मिक्स} का
शक्तियों के रूप में \emph{लय}\index{लय} --- द आवाज़का रंग:
समान स्वर बजाने वाले यंत्र साझा करते हैं $ f_1$, केवल इसमें अंतर है
उनका स्पेक्ट्रा.
\end{definition}

\end{example}

\section{हिलती हुई डोरी}

\begin{definition}[खड़ी लहर]\label{def:g12:sound-acoustics:standing-wave}
ए \omterm{def:g10:signals-and-waves:wave}{लहर} एक डोरी के निश्चित सिरों के बीच सीमित होकर एक में व्यवस्थित हो सकता है
\emph{खड़ी लहर}\index{खड़ी लहर}: जगह में एक दोलन,
गतिहीन बिंदु --- \emph{नोड्स}\index{नोड्स} --- के साथ बारी-बारी से
अधिकतम स्विंग के बिंदु --- \emph{एंटीनोड्स}\index{एंटीनोड्स}.
अब कुछ भी यात्रा नहीं करता; स्ट्रिंग एक निश्चित पैटर्न में कंपन करती है।
\end{definition}

\begin{proposition}[एक स्ट्रिंग के मोड]\label{prop:g12:sound-acoustics:string-modes}
लम्बाई की एक डोरी $ L $, दोनों सिरों पर तय किया गया, ले जाने वाला \omterm{def:g10:signals-and-waves:wave}{लहर}ें गति का
$ v $, केवल पूर्ण संख्या में फिट होने वाले पैटर्न में ही लगातार कंपन करता है $ n $
का अर्ध-तरंगदैर्घ्य थोपे गए अंत के बीच \omterm{def:g12:sound-acoustics:standing-wave}{नोड्स}:
\[ \lambda_n = \frac{2L}{n}, \quad f_n = n\,\frac{v}{2L} = n f_1, \quad n = 1, 2, 3, \dots \]
सबसे निचला मोड $ f_1 = v/(2L)$ है \omterm{def:g12:sound-acoustics:harmonics}{मौलिक}; अन्य हैं
बिलकुल यही \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स}. \admitted
\end{proposition}

\begin{remark}[जहां ईमानदार सबूत रहते हैं]\label{rem:g12:sound-acoustics:honest}
वह सीमित \omterm{def:g10:signals-and-waves:wave}{लहर}ें बिल्कुल इन मोड्स को चुनें से अनुसरण करता है
सुपरपोज़िंग \omterm{def:g10:signals-and-waves:wave}{लहर} इसके प्रतिबिंबों के साथ, एक गणना की गई
वर्ष 1 खंड. ज्यामिति आश्वस्त करती है: दोनों छोर बने रहने चाहिए
फिर भी, तो पूरी संख्या अर्ध-तरंगदैर्घ्य फिट होना चाहिए.
\end{remark}

\begin{example}[एक वायलिन की एक स्ट्रिंग]\label{ex:g12:sound-acoustics:violin}
एक वायलिन की एक तार होती है $ L = \qty{32.8}{cm}$ और \omterm{def:g10:signals-and-waves:sound}{आवाज़}
$ f_1 = \qty{440}{Hz}$: \omterm{def:g10:signals-and-waves:wave}{लहर}ें इसके साथ-साथ दौड़ें
$ v = 2Lf_1 = 2 \times 0.328 \times 440 \approx \qty{289}{m/s}$ ---
द्वारा निर्धारित \omterm{def:g10:inertia:inventory}{तनाव} और द्रव्यमान, खूंटी द्वारा ट्यून किया गया। एक उंगली दबाना
छोटा करता है $ L $ और प्रत्येक को उठाता है $ f_n $ एक बार में: \omterm{def:g12:sound-acoustics:timbre}{स्पेक्ट्रम} ऊपर की ओर खिसकता है।
\end{example}

\begin{remark}[उपकरण अलग-अलग क्यों हैं?]\label{rem:g12:sound-acoustics:instruments}
डोरी को कैसे उत्तेजित किया जाता है --- तोड़ा जाता है, झुकाया जाता है, मारा जाता है --- कैसे सेट होता है
प्रत्येक मोड में से अधिकांश मौजूद है, और शरीर कुछ को बढ़ाता है \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स}
दूसरों से अधिक; उत्तेजना प्लस शरीर को ठीक करें \omterm{def:g12:sound-acoustics:timbre}{स्पेक्ट्रम}, तो वही
$ f_1$ एक गिटार, एक वायलिन और एक पियानो छोड़ता है, प्रत्येक की ध्वनि उसकी जैसी होती है।
\end{remark}

\section{समस्या: लूथियर की कार्यशाला}

\begin{problem}[{सप्ताहांत समस्या --- लूथियर की कार्यशाला: एक गिटार
झल्लाहट द्वारा झल्लाहट की गई स्ट्रिंग, इसके हार्मोनिक्स को एक लय में ट्यून किया गया, ए
गायन को खतरे की रेखा से नीचे रखा गया है, और एक के लिए अल्ट्रासाउंड शिकार किया गया है
लकड़ी में दोष}]
\label{pb:g12:sound-acoustics:1}
एक लूथियर गिटार ख़त्म कर रहा है। इसकी A स्ट्रिंग में कंपन लंबाई होती है
$ L = \qty{65.0}{cm}$ और अवश्य \omterm{def:g10:signals-and-waves:sound}{आवाज़} $ f_1 = \qty{110}{Hz}$. लेना
\qty{340}{m/s} के लिए \omterm{def:g10:signals-and-waves:sound}{आवाज़} हवा में, और स्वीकार करें $ f_1 = v/(2L)$
(\cref{prop:g12:sound-acoustics:string-modes}), $ v $ होने के नाते \omterm{def:g10:signals-and-waves:wave}{लहर}
स्ट्रिंग पर गति.

\medskip
\noindent\textbf{Part I --- Laying out the fretboard.}
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:signals-and-waves:wave}{लहर} रफ़्तार $ v $ स्ट्रिंग पर.
  \item की गणना करें \omterm{def:g12:sound-acoustics:harmonics}{मौलिक}'एस \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} \emph{स्ट्रिंग पर},
        फिर \omterm{def:g12:mechanical-waves:wavelength}{तरंग दैर्ध्य} की \qty{110}{Hz} \omterm{def:g10:signals-and-waves:sound}{आवाज़} \emph{में वायु}. वे एक जैसे ही भिन्न क्यों हैं? \omterm{def:g10:signals-and-waves:frequency}{आवृत्ति}?
  \item सप्तक (\qty{220}{Hz}): कौन सी कंपन लंबाई इसे देती है?
        बारहवें झल्लाहट को कहाँ बैठना चाहिए, एक अंश के रूप में $ L $?
  \item पांचवा (\qty{165}{Hz}): कौन सी कंपन लंबाई इसे देती है?
  \item प्रत्येक झल्लाहट नोट को एक सेमीटोन, एक कारक उठाती है
        $2^{1/12}$. लघुगणक से जांचें कि कितने अर्धस्वर अलग होते हैं
        \qty{110}{Hz} से \qty{220}{Hz}; पहली झल्लाहट पर लंबाई?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स} and \omterm{def:g12:sound-acoustics:timbre}{लय}.}
\begin{enumerate}[resume]
  \item पहले पांच की सूची बनाएं \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स} खुली A स्ट्रिंग का.
  \item एक वायलिन वादक का ए \omterm{def:g10:signals-and-waves:sound}{आवाज़} \qty{440}{Hz}. कौन लयबद्ध का
        गिटार का तार वह है, और जो संगीतमय अंतराल को अलग करता है
        यह से \omterm{def:g12:sound-acoustics:harmonics}{मौलिक}? और \qty{330}{Hz}?
  \item लुथिएर पुल के पास प्लक करता है, फिर पुल के ऊपर
        फ़िंगरबोर्ड: वही \omterm{def:g12:sound-acoustics:pitch-loudness}{आवाज़ का उतार-चढ़ाव}, अलग रंग. स्पेक्ट्रा सहित समझाइये।
  \item तोड़ते समय मध्यबिंदु पर हल्का सा स्पर्श बल ए नोड
        वहाँ। कौन \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स} जीवित रहो, और क्या \omterm{def:g12:sound-acoustics:pitch-loudness}{आवाज़ का उतार-चढ़ाव} सुना है कि?
  \item कितने \omterm{def:g12:sound-acoustics:harmonics}{हार्मोनिक्स} खुली स्ट्रिंग झूठ का, सिद्धांत रूप में,
        श्रव्य के भीतर \omterm{def:g11:fundamental-interactions:strong}{श्रेणी}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The recital.}
गिटार के पहले संगीत कार्यक्रम में, प्रत्येक वाद्य यंत्र एक ध्वनि उत्सर्जित करता है
\omterm{def:g11:work-of-force:power}{शक्ति} $ P = \qty{6.0e-3}{W}$ एक छोटे स्रोत के रूप में, सभी में समान रूप से
दिशानिर्देश.
\begin{enumerate}[resume]
  \item पर तीव्रता की गणना करें $ r = \qty{3.0}{m}$ एक उपकरण से.
  \item निष्कर्ष निकालें \omterm{def:g12:sound-acoustics:level}{ध्वनि का स्तर} वहाँ।
  \item एक चौकड़ी बजाती है: ऐसे चार वाद्ययंत्र क्या स्तर देते हैं?
        उसी स्थान पर? यह चार गुना स्तर का क्यों नहीं है?
  \item बारह वाद्य बजते हैं: किस स्तर पर है \qty{3.0}{m}?
  \item समूह के गिरने के लिए श्रोता को कितनी दूर बैठना चाहिए
        को \qty{85}{dB}?
  \item सुरक्षा नियम: \qty{85}{dB} आठ घंटे तक सुरक्षित है, और
        प्रत्येक \qty{+3}{dB} सुरक्षित समय आधा कर देता है. कब तक हो सकता है
        प्रश्न 14 के अग्रिम पंक्ति के श्रोता सुरक्षित रूप से रहें?
\end{enumerate}
