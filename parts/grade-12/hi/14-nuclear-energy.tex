\chapter{परमाणु ऊर्जा: विखंडन, संलयन, $E=mc^2$}\label{ch:g12:nuclear-energy}

एक पेंसिल इरेज़र के आकार का ईंधन पैलेट एक घर की बिजली को कवर करता है
एक वर्ष के लिए; कोयले से वही सेवा तीन टन लेती है। उपरि, सूर्य
दिन के उजाले की कीमत गायब होकर चुकाई जाती है --- प्रति सेकंड चार मिलियन टन।
यह अध्याय दोनों सौदेबाजी के पीछे का हिसाब-किताब खोलता है: द्रव्यमान ही है
ऊर्जा, और \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} (\cref{ch:g11:nucleus-radioactivity}) वहां हैं जहां इसे भुनाया जाता है।

\section{द्रव्यमान ऊर्जा है}

\begin{theorem}[द्रव्यमान--ऊर्जा तुल्यता]\label{thm:g12:nuclear-energy:emc2}\index{द्रव्यमान--ऊर्जा तुल्यता}
द्रव्यमान का एक पिंड $ m $, केवल विद्यमान होने से, धारण करता है ऊर्जा
\[
E = m c^2, \qquad c = \qty{3.00e8}{m/s}.
\]
जब भी कोई सिस्टम द्रव्यमान खोता है $\Delta m $, यह जारी करता है ऊर्जा
$\Delta m\, c^2$ --- तंत्र जो भी हो.
\end{theorem}
\admitted

\begin{remark}[आइंस्टीन, 1905]\label{rem:g12:nuclear-energy:einstein}
यह आइंस्टीन का संबंध है, यहाँ स्वीकार किया गया है; जहां से यह आता है वह व्यवसाय है
विशेष-सापेक्षता अध्याय का जो इस वर्ष बंद हो रहा है
(\cref{ch:g12:special-relativity}). दर नोट करें: $ c^2 = \qty{9.00e16}{J/kg}$
--- द्रव्यमान की पूर्णांक त्रुटि एक सौभाग्य है ऊर्जा.
\end{remark}

\begin{definition}[इलेक्ट्रॉन-वोल्ट और परमाणु द्रव्यमान इकाई]\label{def:g12:nuclear-energy:units}
दो इकाइयां के अनुरूप \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}.
\emph{इलेक्ट्रॉन-वोल्ट}\index{इलेक्ट्रॉन-वोल्ट}, द ऊर्जा एक इलेक्ट्रॉन प्राप्त होता है
पार करना \qty{1}{V} (\cref{ch:g11:circuits-and-power}):
$\qty{1}{eV} = \qty{1.60e-19}{J}$, $\qty{1}{MeV} = \qty{1.60e-13}{J}$ ---
रासायनिक बांड कुछ व्यापार करते हैं \unit{eV}, परमाणु प्रतिक्रियाएँ मेगाइलेक्ट्रॉ\omterm{def:g11:circuits-and-power:voltage}{न-वोल्ट}.
 \emph{एकीकृत परमाणु द्रव्यमान इकाई}\index{एकीकृत परमाणु द्रव्यमान इकाई}, एक
कार्बन-12 के द्रव्यमान का बारहवाँ भाग \omterm{def:g11:fundamental-interactions:ladder}{एटम}: $\qty{1}{u} = \qty{1.66054e-27}{kg}$,
का ऊर्जा समकक्ष \qty{931.5}{MeV} --- $\qty{1}{u} = \qty{931.5}{MeV}/c^2$.
\end{definition}

\begin{example}[सोया हुआ भाग्य]\label{ex:g12:nuclear-energy:fortune}
एक \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} किसी भी चीज़ का, पूरी तरह से परिवर्तित, है $\qty{9.00e16}{J}$ --- तीन
एक बड़े उत्पादन के वर्ष (\qty{1}{GW}) \omterm{def:g11:work-of-force:power}{शक्ति} स्टेशन। पहले किसी ने ध्यान नहीं दिया
1905 क्योंकि साधारण भौतिकी बमुश्किल द्रव्यमान को छूती है: जलना \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} कोयले का
(\qty{3.0e7}{J}) इसे एक तिहाई माइक्रोग्राम तक हल्का कर देता है। \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, हम देखेंगे,
एक हजार में एक भाग को हिलाएं: फिर भी छोटा, लेकिन रसायन विज्ञान से लाखों गुना।
\end{example}

\section{जन दोष}

\begin{definition}[द्रव्यमान दोष और बंधनकारी ऊर्जा]\label{def:g12:nuclear-energy:defect}
वजन ए \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} ${}^{A}_{Z}\mathrm{X}$ (द्रव्यमान $ m $), फिर उसका वजन करें $ Z $ प्रोटोन
और $ A - Z $ न्यूट्रॉन अलग-अलग: भाग हैं \emph{भारी} संपूर्ण की तुलना में.
के अंतर
\[
\Delta m = Z\,m_p + (A - Z)\,m_n - m > 0,
\qquad m_p = \qty{1.00728}{u},\; m_n = \qty{1.00866}{u},
\]
है \emph{सामूहिक दोष}\index{सामूहिक दोष}; \emph{बाँधने वाली ऊर्जा}\index{बाँधने वाली ऊर्जा} $ E_b = \Delta m\,c^2$ है ऊर्जा की जरूरत है
खींचो \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} मुक्त में अलग न्युक्लियोन --- समान रूप से, ऊर्जा जारी किया गया
जब इसे असेंबल किया जाता है. एक बाध्य \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} बैठता \emph{नीचे} इसके भाग.
\end{definition}

  \draw[thick] (0,2.1) -- (2.7,2.1) node[right] {$Z$ protons $+$ $(A-Z)$ neutrons, free};

  \draw[thick, omDef] (0,0) -- (2.7,0) node[right] {nucleus, mass $m$};

न्युक्लियोन --- इसे असेंबल करने से रिलीज़ होता है $ E_b $, निराकरण में लागत आती है।}
\end{omfigure}

\begin{example}[हीलियम-4]\label{ex:g12:nuclear-energy:helium}
 ${}^{4}_{2}\mathrm{He}$ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} है $ m = \qty{4.00151}{u}$; इसके भाग:
$2 \times 1.00728 + 2 \times 1.00866 = \qty{4.03188}{u}$. इसलिए
$\Delta m = \qty{0.03037}{u}$ और $ E_b = 0.03037 \times 931.5 \approx
\qty{28.3}{MeV}$: द \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} वजन का होता है $0.75\%$ इसके भागों से कम ---
``हजारों में एक भाग'' \cref{ex:g12:nuclear-energy:fortune}.
\end{example}

\begin{method}[परमाणु प्रतिक्रिया का ऊर्जा संतुलन]\label{met:g12:nuclear-energy:balance}
\begin{enumerate}
  \item समीकरण को संतुलित करें: कुल $ A $ और $ Z $ संरक्षित (\cref{ch:g11:nucleus-radioactivity}).
  \item जनता को पहले, फिर बाद में, अंदर जोड़ें \unit{u} --- सभी पांच दशमलव रखें.
  \item $\Delta m = m_{\text{before}} - m_{\text{after}}$, तब
        $ E = \Delta m\,(\text{in u}) \times \qty{931.5}{MeV}$, के रूप में ले जाया गया
        \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} उत्पादों का; नकारात्मक का मतलब है कि प्रतिक्रिया का भुगतान किया जाना चाहिए।
\end{enumerate}
\end{method}

\section{वह वक्र जो ब्रह्मांड को चलाता है}

\begin{definition}[प्रति न्यूक्लियॉन बंधन ऊर्जा]\label{def:g12:nuclear-energy:pernucleon}
सामंजस्य की तुलना निष्पक्ष रूप से की जाती है \emph{प्रति न्यूक्लियॉन बंधन ऊर्जा}\index{प्रति न्यूक्लियॉन बंधन ऊर्जा}, $ E_b/A $, की औसत लागत
एक निकाल रहा हूँ \omterm{def:g11:fundamental-interactions:ladder}{न्यूक्लियॉन}: $28.3/4 \approx \qty{7.1}{MeV}$ हीलियम-4 के लिए.
\end{definition}

\begin{proposition}[लोहे की चोटी, और नीचे दो सड़कें]\label{prop:g12:nuclear-energy:ironpeak}\index{लोहे की चोटी}
के खिलाफ साजिश रची $ A $, द प्रति \omterm{def:g11:fundamental-interactions:ladder}{न्यूक्लियॉन} बंधनकारी ऊर्जा तेजी से चढ़ता है
हल्के \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, लोहे के पास की चोटियाँ, ${}^{56}_{26}\mathrm{Fe}$, लगभग
\qty{8.8}{MeV} प्रति \omterm{def:g11:fundamental-interactions:ladder}{न्यूक्लियॉन}, फिर धीरे से सरकाता है \qty{7.6}{MeV} पर
यूरेनियम: लोहा सबसे मजबूती से बंधा होता है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} प्रकृति में. एक प्रतिक्रिया
जारी करता है ऊर्जा ठीक उसी समय जब उसके उत्पाद बैठते हैं \emph{उच्च} वक्र के; दो
विपरीत रणनीतियाँ इसलिए दोनों भुगतान करती हैं, और दोनों लोहे की ओर चलती हैं ---
\emph{बंटवारे} सबसे भारी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, और \emph{\omterm{def:g12:nuclear-energy:fusion}{विलय}} सबसे हल्का.
\end{proposition}
\admitted

\begin{remark}[शिखर क्यों?]\label{rem:g12:nuclear-energy:whypeak}
की रस्साकशी \cref{ch:g11:fundamental-interactions}: मजबूत गोंद बांधता है
केवल पड़ोसियों को छूने से, कूलम्ब का प्रतिकर्षण संपूर्ण तक फैल जाता है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} --- छोटा
\omterm{def:g11:fundamental-interactions:ladder}{नाभिक} अधिकतर सतही होते हैं, बड़े \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} अधिकतर प्रतिकर्षण वाले होते हैं, लोहा समझौता करता है।
ईमानदार लेखांकन विश्वविद्या\omterm{def:g12:sound-acoustics:timbre}{लय} के संस्करणों में रहता है।
\end{remark}

\section{विखण्डन एवं शृंखला अभिक्रिया}

\begin{definition}[विखंडन]\label{def:g12:nuclear-energy:fission}
\emph{विखंडन}\index{विखंडन} एक भारी का विभाजन है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} दो में
मध्यम आकार के टुकड़े. यूरेनियम-235 है \emph{चीरने योग्य}\index{चीरने योग्य}: एक धीमा
न्यूट्रॉन, अवशोषित, छोड़ देता है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} इतना उत्तेजित कि दो फाड़ हो गया,
दो या तीन ताजा न्यूट्रॉन बाहर थूकना ---
${}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \to \text{two fragments} +
2\text{--}3\;{}^{1}_{0}\mathrm{n}$, यह जोड़ी विखंडन से विखंडन तक बदलती रहती है।
\end{definition}

\begin{example}[एक विखंडन, तौला गया]\label{ex:g12:nuclear-energy:fissionbalance}
एक लगातार मार्ग, द्वारा \cref{met:g12:nuclear-energy:balance}:
\[
{}^{1}_{0}\mathrm{n} + {}^{235}_{92}\mathrm{U} \longrightarrow
{}^{94}_{38}\mathrm{Sr} + {}^{140}_{54}\mathrm{Xe} + 2\,{}^{1}_{0}\mathrm{n}.
\]
पहले: $1.00866 + 234.99350 = \qty{236.00216}{u}$. बाद में:
$93.89450 + 139.89200 + 2 \times 1.00866 = \qty{235.80382}{u}$. इस तरह
$\Delta m = \qty{0.19834}{u}$ और $ E \approx \qty{185}{MeV}$; की गिनती
बाद में रेडियोधर्मी टुकड़ों का क्षय (\cref{ch:g12:radioactive-decay}),
प्रत्येक \omterm{def:g12:nuclear-energy:fission}{विखंडन} के बारे में लायक है \qty{200}{MeV} --- कुछ \unit{eV} एक खरीदता है \omterm{def:g11:fundamental-interactions:ladder}{एटम}
कोयले का.
\end{example}

\begin{definition}[श्रृंखला प्रतिक्रिया और महत्वपूर्ण द्रव्यमान]\label{def:g12:nuclear-energy:chain}
प्रत्येक \omterm{def:g12:nuclear-energy:fission}{विखंडन} एक न्यूट्रॉन द्वारा जलाया जाता है और दो या तीन को मुक्त करता है: काफी बड़े पैमाने पर
गांठ बनाकर वे अन्य \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}ों पर प्रहार करते हैं और प्रतिक्रिया स्वयं ही सक्रिय हो जाती है --- ए
\emph{श्रृंखला अभिक्रिया}\index{श्रृंखला अभिक्रिया}. एक छोटी सी गांठ में अधिकांश न्यूट्रॉन होते हैं
पहले सतह से बच निकलें; श्रृंखला को बनाए रखने वाली न्यूनतम मात्रा
है \emph{क्रांतिक द्रव्यमान}\index{क्रांतिक द्रव्यमान} ---इसके नीचे चेन है
फीकी पड़ जाती है, इसके ऊपर हर पीढ़ी बढ़ती है: बम का सिद्धांत।
\end{definition}

\begin{definition}[रिएक्टर, वश में किया गया]\label{def:g12:nuclear-energy:reactor}
ए \omterm{def:g11:work-of-force:power}{शक्ति} रिएक्टर श्रृंखला को बिल्कुल एक न्यूट्रॉन प्रति पर रखता है \omterm{def:g12:nuclear-energy:fission}{विखंडन}. ईंधन:
यूरेनियम छर्रों को यूरेनियम-235 में मामूली रूप से समृद्ध किया गया।
\emph{मध्यस्थ}\index{मध्यस्थ} (साधारण पानी, आमतौर पर) उपवास को धीमा कर देता है
\omterm{def:g12:nuclear-energy:fission}{विखंडन} \omterm{def:g10:inertia:inventory}{टकराव} से न्यूट्रॉन --- धीमे न्यूट्रॉन ट्रिगर करने में कहीं बेहतर होते हैं
अगला \omterm{def:g12:nuclear-energy:fission}{विखंडन}; \emph{नियंत्रण छड़ें}\index{नियंत्रण छड़ें} (बोरान, कैडमियम)
न्यूट्रॉन को निगलें --- अंदर धकेले जाने पर, श्रृंखला मर जाती है; बाहर निकाला जाता है, यह तेज़ हो जाता है। द
टुकड़ों की हलचल गर्मी, फिर भाप, फिर बिजली बन जाती है
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[एक ट्रक के विरुद्ध एक काली मिर्च]\label{ex:g12:nuclear-energy:gram}
एक ग्राम यूरेनियम-235 होता है
$N = \num{1.0e-3} / (235 \times \num{1.66054e-27}) \approx \num{2.56e21}$
\omterm{def:g11:fundamental-interactions:ladder}{नाभिक}; पर $\qty{200}{MeV} = \qty{3.2e-11}{J}$ प्रत्येक, उन सभी को विखंडित कर रहा है
के बारे में पैदावार \qty{8.2e10}{J} --- द \omterm{def:g10:energy-conservation:forms}{रसायन ऊर्जा} का $\num{8.2e10} /
\num{3.0e7} \approx \qty{2.7e3}{kg}$ कोयले की मात्रा: तीन टन के मुकाबले एक काली मिर्च।
\end{example}

\section{संलयन: सूर्य की अपनी अग्नि}

\begin{definition}[विलय]\label{def:g12:nuclear-energy:fusion}
\emph{विलय}\index{विलय} दो हल्के \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}ों का एक भारी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} में वि\omterm{def:g12:sound-acoustics:timbre}{लय} है,
एक को और अधिक मजबूती से बांधना। सबसे सुलभ प्रतिक्रिया वेड्स ड्यूटेरियम और
ट्रिटियम, भारी \omterm{def:g11:nucleus-radioactivity:notation}{आइसोटोप} हाइड्रोजन का (\cref{ch:g11:nucleus-radioactivity}):
\[
{}^{2}_{1}\mathrm{H} + {}^{3}_{1}\mathrm{H} \longrightarrow
{}^{4}_{2}\mathrm{He} + {}^{1}_{0}\mathrm{n}.
\]
\end{definition}

\begin{example}[डी--टी, तौला गया]\label{ex:g12:nuclear-energy:dt}
पहले: $2.01355 + 3.01550 = \qty{5.02905}{u}$; बाद में:
$4.00151 + 1.00866 = \qty{5.01017}{u}$; इसलिए $\Delta m = \qty{0.01888}{u}$ और
$ E \approx \qty{17.6}{MeV}$, न्यूट्रॉन पर इसका चार-पाँचवाँ भाग---ए से कम
\omterm{def:g12:nuclear-energy:fission}{विखंडन}, लेकिन पांच से न्युक्लियोन के बजाय $236$: के बारे में \qty{3.4e14}{J} प्रति
\omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} ईंधन का, चार गुना \omterm{def:g12:nuclear-energy:fission}{विखंडन}. हीलियम-4
(\cref{ex:g12:nuclear-energy:helium}) एक बार फिर राख है।
\end{example}

\begin{definition}[कूलम्ब बाधा]\label{def:g12:nuclear-energy:barrier}
दो \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} दोनों सकारात्मक हैं: स्पर्श करने के लिए, उन्हें पहले पहाड़ी पर चढ़ना होगा
उनका \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} प्रतिकर्षण (\cref{ch:g11:electric-gravitational-fields}) --- द
\emph{कूलम्ब बाधा}\index{कूलम्ब बाधा}. केवल पास में \qty{1e8}{K} करो
\omterm{def:g10:inertia:inventory}{टकराव} ले जाते हैं ऊर्जा इसे पार करना; ईंधन तो एक है
\emph{प्लाज्मा}\index{प्लाज्मा} नंगे \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} और इलेक्ट्रॉनों की. \omterm{def:g12:nuclear-energy:fission}{विखंडन} नहीं भुगतान करता है
ऐसा टोल: इसका ट्रिगर, न्यूट्रॉन, तटस्थ है, और \omterm{def:g11:work-of-force:work}{काम} करता है ठंडा।
\end{definition}

\begin{example}[सूर्य इस पर चलता है]\label{ex:g12:nuclear-energy:sun}
सूर्य का कोर, पर $\qty{1.5e7}{K}$, फ़्यूज़ साधारण हाइड्रोजन \omterm{def:g12:oscillators-and-time:phase}{चरण} दर \omterm{def:g12:oscillators-and-time:phase}{चरण} ---
द \emph{प्रोटॉन--प्रोटॉन शृंखला}, शुद्ध प्रभाव का
$4\,{}^{1}_{1}\mathrm{H} \to {}^{4}_{2}\mathrm{He} + 2\,{}^{0}_{+1}\mathrm{e}
+ \text{radiation}$, के बारे में \qty{26}{MeV} प्रति हीलियम: $0.7\%$ का
हाइड्रोजन का द्रव्यमान इस प्रकार निकलता है ऊर्जा. सूरज की रोशनी किसकी \omterm{def:g12:sound-acoustics:timbre}{स्पेक्ट्रम} हम पढ़ते हैं
\cref{ch:g10:light-spectra} क्या यह जनसमूह आठ मिनट \omterm{def:g12:mechanical-waves:delay}{देरी} से पहुंच रहा है; प्रत्येक
तारा लोहे की ओर वक्र पर चलते हुए चमकता है।
\end{example}

\begin{remark}[आईटीईआर: एक स्टार को बोतलबंद करना]\label{rem:g12:nuclear-energy:iter}
कोई ठोस इसे धारण नहीं कर सकता \omterm{def:g12:nuclear-energy:barrier}{प्लाज्मा} पर \qty{1.5e8}{K}; ए \emph{tokamak} इसे पिंजरे में बंद कर देता है
\omterm{def:g11:magnetic-fields:bfield}{चुंबकीय क्षेत्र} (\cref{ch:g11:magnetic-fields}), का एक डोनट \omterm{def:g11:electric-gravitational-fields:lines}{फ़ील्ड लाइन}ें द
आवेशित कण दीवार को छुए बिना सर्पिल \omterm{def:g12:newtons-laws:momentum}{गति} से चलते हैं; आईटीईआर, के लिए बनाया गया
इसे दस गुना गर्म होने दें \omterm{def:g11:work-of-force:power}{शक्ति}, वर्तमान कदम है. तोड़ो
कारावास और \omterm{def:g12:nuclear-energy:barrier}{प्लाज्मा} सेकंडों में ठंडा और बंद हो जाता है: नहीं \omterm{def:g12:nuclear-energy:chain}{श्रृंखला अभिक्रिया}, नहीं
\omterm{def:g12:nuclear-energy:chain}{क्रांतिक द्रव्यमान} --- कठिन हिस्सा आग को रोकना नहीं बल्कि उसे जलाए रखना है।
\end{remark}

\begin{example}[ऊर्जा सीढ़ी, प्रति किलोग्राम]\label{ex:g12:nuclear-energy:ladder}
ऊर्जा प्रति \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} ईंधन का: कोयला, \qty{3.0e7}{J}; यूरेनियम-235 द्वारा \omterm{def:g12:nuclear-energy:fission}{विखंडन},
\qty{8.2e13}{J}; ड्यूटेरियम--ट्रिटियम द्वारा \omterm{def:g12:nuclear-energy:fusion}{विलय}, \qty{3.4e14}{J}; कुल
रूपांतरण ($ E = mc^2$), \qty{9.0e16}{J}. परिमाण के छह क्रम अलग-अलग होते हैं
रसायन विज्ञान से \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} --- ट्रक द्वारा रिएक्टर में ईंधन भरने का कारण
और ट्रेन लोड द्वारा एक कोयला संयंत्र।
\end{example}

\section{समस्या: एक शहर को शक्ति प्रदान करना}

\begin{problem}[{सप्ताहांत समस्या --- एक शहर को बिजली देना: एक स्थिर गीगावाट
पांच लाख लोगों ने, चार तरह से यूरेनियम छर्रे, ट्रेन लोड खरीदे
कोयला, समुद्री जल का एक कुंड, सूर्य का अपना पदार्थ --- एक ही किलोग्राम के साथ
चारों के नीचे छुपे हुए लुप्त द्रव्यमान का}]
\label{pb:g12:nuclear-energy:1}
का एक शहर $500\,000$ निवासी \omterm{prop:g12:work-and-energy:equilibria}{स्थिर} \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} खींचते हैं \omterm{def:g11:work-of-force:power}{शक्ति}
$ P = \qty{1.0}{GW}$; इसके पौधे, चाहे कोई भी ईंधन हो, गर्मी को बदल देते हैं
बिजली के साथ \omterm{def:g10:energy-conservation:efficiency}{क्षमता} $33\%$ (\cref{ch:g10:energy-conservation}).
डेटा: एक यूरेनियम-235 \omterm{def:g12:nuclear-energy:fission}{विखंडन}, \qty{200}{MeV} सभी में; कोयला, \qty{3.0e7}{J/kg};
एक डी--टी \omterm{def:g12:nuclear-energy:fusion}{विलय}, \qty{17.6}{MeV} से \qty{5.029}{u} ईंधन का;
$\qty{1}{u} = \qty{1.66054e-27}{kg}$; $\qty{1}{eV} = \qty{1.60e-19}{J}$; एक वर्ष $= \qty{3.16e7}{s}$; सूर्य: \qty{3.9e26}{W}, \qty{2.0e30}{kg}; एक घन
\omterm{def:g10:orders-of-magnitude:unit}{मीटर} समुद्री जल लगभग रहता है \qty{33}{g} ड्यूटेरियम का.

\medskip
\noindent\textbf{Part I --- The \omterm{def:g12:nuclear-energy:fission}{विखंडन} plant.}
\begin{enumerate}
  \item क्या थर्मल \omterm{def:g11:work-of-force:power}{शक्ति} क्या रिएक्टर को डिलीवर करना चाहिए?
  \item व्यक्त करें ऊर्जा एक का \omterm{def:g12:nuclear-energy:fission}{विखंडन} में \omterm{def:g11:work-of-force:work}{जूल}.
  \item कितने \omterm{def:g12:nuclear-energy:fission}{विखंडन} प्रति सेकंड शहर चलाते हैं?
  \item एक यूरेनियम का द्रव्यमान-235 \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} ($ A = 235$), फिर यूरेनियम-235
        प्रति सेकंड खपत?
  \item एक वर्ष में खपत यूरेनियम-235 का अनुमान लगाएं \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम}.
  \item ईंधन को समृद्ध किया जाता है $4.0\%$ यूरेनियम-235: एक वर्ष में कितना द्रव्यमान होता है
        ले लो, और क्या यह एक ट्रक पर फिट बैठता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The coal ledger.}
\begin{enumerate}[resume]
  \item की गणना करें \omterm{def:g10:energy-conservation:forms}{थर्मल ऊर्जा} शहर एक वर्ष में खपत करता है।
  \item कोयले का कितना द्रव्यमान इसे वितरित करता है?
  \item एक वर्ष के लिए अनुपात (कोयला द्रव्यमान)/(यूरेनियम-235 द्रव्यमान) की गणना करें, और
        की सीढ़ी के सामने इसकी जाँच करें \cref{ex:g12:nuclear-energy:ladder}.
  \item एक मालगाड़ी ढोती है \qty{3.0e6}{kg}: प्रति वर्ष कितनी ट्रेनें, और
        प्रति दिन? एक वाक्य: दो आपूर्ति लाइनों का चित्रण करें।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g12:nuclear-energy:fusion}{विलय} dream.}
\begin{enumerate}[resume]
  \item व्यक्त करें ऊर्जा एक D--T का \omterm{def:g12:nuclear-energy:fusion}{विलय} में \omterm{def:g11:work-of-force:work}{जूल}.
  \item की गणना करें ऊर्जा प्रति जारी किया गया \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} डी-टी ईंधन का; तुलना करें
        सीढ़ी.
  \item डी-टी ईंधन का कितना द्रव्यमान शहर के वर्ष को कवर करता है? उसका द्रव्यमान कितना है
        ड्यूटेरियम ($2.014$ की \qty{5.029}{u})?
  \item समुद्री जल की कितनी मात्रा में वह ड्यूटेरियम होता है? एक के साथ तुलना करें
        ओलिंपिक पूल, के बारे में \qty{2.5e3}{m^3}. (ट्रिटियम लिथियम से उत्पन्न होता है।)
  \item यह पौधा नंबर क्यों रखता है? \omterm{def:g12:nuclear-energy:chain}{क्रांतिक द्रव्यमान} और नहीं \omterm{def:g12:nuclear-energy:chain}{श्रृंखला अभिक्रिया} डरना? तो फिर, कठिन हिस्सा क्या है?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The star that pays in mass.}
\begin{enumerate}[resume]
  \item से $ E = mc^2$, सूर्य प्रत्येक सेकंड में किस द्रव्यमान को परिवर्तित करता है?
  \item प्रति सेकंड और प्रति वर्ष कितना द्रव्यमान होना चाहिए \emph{कोई}
        \qty{3.0}{GW}-थर्मल स्रोत कन्वर्ट? जांचें कि तीनों ईंधन
        इसी द्रव्यमान का समर्पण करो।
  \item सूर्य ने अपने अंदर कुल कितना द्रव्यमान उत्सर्जित किया है? $\num{4.5e9}$
        वर्ष, और वह सूर्य का कितना अंश है?
  \item \omterm{def:g12:nuclear-energy:fusion}{विलय} के बारे में टैप कर सकते हैं $0.07\%$ सूर्य के द्रव्यमान का (जलने योग्य हिस्सा)।
        इसका मूल): एक तारे के रूप में इसके कुल जीवनकाल का अनुमान लगाएं, और कितना बचा है।
  \item ऑडिट को तीन वाक्यों में बंद करें: शहर का एक वर्ष टन में
        कोयले का, \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} यूरेनियम-235 का और \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} डी--टी ईंधन का --- और
        द \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} जनसमूह का, जो हर परिदृश्य में, शहर की रोशनी बन गया।
\end{enumerate}
\end{problem}
