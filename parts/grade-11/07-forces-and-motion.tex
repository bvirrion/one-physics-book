\chapter{Forces and Motion}\label{ch:g11:forces-and-motion}

Throw a wrench spinning across the room: the ends wobble wildly, yet
one point inside it traces a perfect parabola. \Cref{ch:g10:inertia}
settled what happens when forces compensate --- nothing. This chapter
asks what happens when they do not: the leftover force \emph{changes
the velocity} --- speeding bodies up, slowing them down, turning them
--- and says by how much, in ratios if not yet in formulas.

\section{The center of mass}

\begin{definition}[Center of mass]\label{def:g11:forces-and-motion:com}
The \emph{center of mass}\index{center of mass} of a body or system is
its balance point --- the average position of its mass: the geometric
center of a homogeneous symmetric body; for two point masses, the point
of the joining segment at distances in the \emph{inverse} ratio of the
masses, $\frac{d_1}{d_2} = \frac{m_2}{m_1}$ --- closer to the heavier.
\end{definition}

\begin{example}[A lopsided dumbbell]\label{ex:g11:forces-and-motion:dumbbell}
Balls of \qty{2.0}{kg} and \qty{1.0}{kg} sit at the ends of a light bar
\qty{0.90}{m} long. The center of mass divides it in the inverse ratio
$d_1/d_2 = 1.0/2.0$: $\qty{0.30}{m}$ from the heavy ball,
$\qty{0.60}{m}$ from the light one --- balance the bar there.
\end{example}

\begin{remark}[The thrown wrench]\label{rem:g11:forces-and-motion:wrench}
Film a tumbling wrench: every point follows a complicated looping curve
--- except the center of mass, which traces the clean parabola a thrown
ball would. The motion of the center of mass is simple; the rest is spin
\emph{around} it --- which is why mechanics may treat a car, a planet or
a gymnast as a single point.
\end{remark}

\section{A net force changes the velocity}

\begin{definition}[Net force]\label{def:g11:forces-and-motion:netforce}
The \emph{net force}\index{net force} on a body is the vector sum of
all the forces acting on it, $\vect F = \vect F_1 + \vect F_2 + \dots$
(tip to tail, as in the mathematics volume); the forces compensate
exactly when $\vect F = \vect 0$.
\end{definition}

\begin{theorem}[Effect of a net force --- semi-quantitative]\label{thm:g11:forces-and-motion:secondlaw}
\index{second law of motion}
A net force $\vect F$ acting on a body of mass $m$ for a short duration
$\Delta t$ changes the velocity of its center of mass by a vector
$\Delta \vect v$ that has the \emph{direction} of $\vect F$, and whose
magnitude grows in proportion to $F$ and to $\Delta t$ and shrinks in
proportion to the mass:
\[
\Delta v \propto \frac{F \, \Delta t}{m} .
\]
\end{theorem}
\admitted

\begin{remark}
Double the force or its duration: twice the change; double the mass:
half. Next year sharpens this into an exact equation, and the Year 1
volume asks in which frames it holds; this year ratios are enough ---
they already run airbags, brakes and orbits.
\end{remark}

\begin{proposition}[The three canonical cases]\label{prop:g11:forces-and-motion:cases}
Let a body move with velocity $\vect v$ under a net force $\vect F$.
\begin{enumerate}
  \item $\vect F$ along $\vect v$: it \emph{speeds up} in a straight line;
  \item $\vect F$ opposite $\vect v$: it \emph{slows down} in a straight line;
  \item $\vect F$ perpendicular to $\vect v$: it \emph{turns} without
        changing speed.
\end{enumerate}
\end{proposition}

\begin{proof}
Add $\Delta\vect v$, directed along $\vect F$
(\cref{thm:g11:forces-and-motion:secondlaw}), tip to tail onto
$\vect v$: along $\vect v$ the arrows pile up (longer); opposite, they
partly cancel (shorter); perpendicular, each small $\Delta \vect v$
tilts the arrow without lengthening it --- only the direction turns.
\end{proof}

\begin{example}[A sideways kick]\label{ex:g11:forces-and-motion:kick}
A puck glides east at \qty{6.0}{m/s}; a brief sideways blow adds
$\Delta\vect v = \qty{8.0}{m/s}$ due north. The new velocity is the
vector sum: magnitude $\sqrt{6.0^2 + 8.0^2} = \qty{10.0}{m/s}$,
direction $\tan\theta = 8.0/6.0$, i.e.\ $\theta \approx 53^\circ$ north
of east --- the old velocity is not erased, the change is added to it.
\end{example}

\section{Comparing changes: force, time, mass}

\begin{method}[Ratio reasoning]\label{met:g11:forces-and-motion:ratio}
To compare two situations, never solve an equation --- form ratios: the
velocity-change ratio is the $F$ ratio, times the $\Delta t$ ratio,
divided by the $m$ ratio. Then read off which factor changed, by how much.
\end{method}

\begin{example}[The loaded cart]\label{ex:g11:forces-and-motion:cart}
The same push, applied for the same second, sends an empty
\qty{20}{kg} shopping cart off at \qty{1.2}{m/s}. Loaded to
\qty{60}{kg} --- triple mass, same $F\Delta t$ --- it leaves at a
third of that, \qty{0.40}{m/s}. Mass is the stubbornness of matter.
\end{example}

\begin{remark}[Crashes, airbags, bent knees]\label{rem:g11:forces-and-motion:airbag}
In a crash $\Delta v$ and $m$ are not negotiable --- the occupant's
speed must reach zero --- so $F\,\Delta t$ is fixed. The only freedom
left is \emph{time}: stretch $\Delta t$ and the force shrinks in the
same ratio. A head stopping on an airbag in \qty{0.08}{s} instead of on
the wheel in \qty{0.01}{s} feels an average force eight times smaller;
crumple zones, helmet foam and bent knees play the same trick.
\end{remark}

\section{Friction: the force that fights sliding}

\begin{definition}[Static and kinetic friction]\label{def:g11:forces-and-motion:friction}
Friction between touching surfaces opposes their \emph{relative
sliding}. \emph{Static friction}\index{friction!static} acts while they
do not slide, adjusting itself up to a maximum to cancel whatever tries
to start the slide; \emph{kinetic friction}\index{friction!kinetic},
once sliding has begun, is roughly constant, directed against the
sliding, and usually \emph{weaker} than the static maximum.
\end{definition}

\begin{example}[Pushing the wardrobe]\label{ex:g11:forces-and-motion:wardrobe}
Push a wardrobe gently: it does not move --- static friction cancels
your \qty{150}{N}. Push harder: at some \qty{400}{N} it lurches into
motion, then slides with less effort --- the weaker kinetic friction
has taken over.
\end{example}

\begin{remark}[Why ABS brakes work]\label{rem:g11:forces-and-motion:abs}
A rolling wheel's contact point does not slide on the road: the tyre
grips with \emph{static} friction. A locked wheel skids and gets only
the weaker \emph{kinetic} friction, which points against the skid
whatever the wheels do --- steering dies. ABS releases the brake the
instant a wheel locks: more grip, shorter stop, a driver who can steer.
\end{remark}

\section{Turning: circular motion needs an inward force}

\begin{definition}[Uniform circular motion]\label{def:g11:forces-and-motion:ucm}
A body is in \emph{uniform circular motion}\index{uniform circular
motion} when its trajectory is a circle traveled at constant speed. The
velocity, tangent to the circle, changes direction at every instant ---
so it is never constant.
\end{definition}

\begin{proposition}[The inward force]\label{prop:g11:forces-and-motion:inward}
A body in uniform circular motion is subject to a net force that is
never zero: at every instant it points from the body toward the center
of the circle.
\end{proposition}

\begin{proof}
The speed is constant, so the net force has no part along $\vect v$
(cases 1 and 2 of \cref{prop:g11:forces-and-motion:cases}): it is
perpendicular to the tangent, hence along the radius --- and inward,
since every $\Delta \vect v$ bends the path toward the center.
\end{proof}

\begin{example}[The string, the bend, the Moon]\label{ex:g11:forces-and-motion:moon}
Every steady turn hides an inward force. The whirled ball: the string's
tension. The car on a bend: the sideways static friction of road on
tyres --- gone on ice, and the car goes straight. The Moon: the Earth's
gravitational pull (\cref{ch:g10:universal-gravitation}), turning it
around in \num{27.3} days --- not held \emph{up} but pulled sideways into a
perpetual turn, falling around the Earth; next year makes this the
mechanics of satellites.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:1}
Zero or nonzero net force? Justify from the velocity: (a) a car cruising
straight at a steady \qty{90}{km/h}; (b) braking in a straight line;
(c) rounding a bend at a steady \qty{50}{km/h}; (d) a skydiver at
terminal speed.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:2}
Balls of \qty{3.0}{kg} and \qty{1.0}{kg} sit at the ends of a light bar
\qty{0.80}{m} long. Locate the center of mass; where for equal masses?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:3}
The same brief push is given to an empty \qty{15}{kg} luggage trolley
and to the same trolley loaded to \qty{45}{kg}. Compare the velocity
changes; what push would give the loaded trolley the empty one's $\Delta v$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:4}
A tram moving at \qty{10}{m/s} brakes; one second later it moves at
\qty{8}{m/s}, same direction. Give $\Delta \vect v$ and the direction of
the net force. Which case of \cref{prop:g11:forces-and-motion:cases} is this?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:5}
Static or kinetic friction --- and directed which way? (a) A crate
resisting your (too gentle) push; (b) the same crate once it slides;
(c) a book on a slightly tilted tray, not sliding.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:6}
A net force $F$ applied for $\Delta t$ to a mass $m$ produces
$\Delta v = \qty{4.0}{m/s}$. Predict $\Delta v$ for: (a) $2F, \Delta t, m$;
(b) $F, \Delta t/2, m$; (c) $F, \Delta t, 2m$; (d) $3F, 2\Delta t, 6m$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:7}
A puck slides east at \qty{8.0}{m/s}; a blow adds
$\Delta \vect v = \qty{6.0}{m/s}$ due north. Compute the new speed and
direction. Why is the speed not $8.0 + 6.0 = \qty{14}{m/s}$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:8}
A goalkeeper stops the same shot two ways: arms rigid (the ball stops in
\qty{0.02}{s}) or ``giving'' with the ball (\qty{0.08}{s}). Compare the
average forces, and state which quantities were fixed before she chose.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:9}
Which external force pushes a sprinter forward at the start? Why do
spiked shoes help --- and why can nobody accelerate, or even walk, on
perfectly frictionless ice?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:10}
In an emergency stop, why does a car with ABS (wheels kept rolling)
usually stop shorter than one with locked, skidding wheels --- and why
can the skidding car no longer be steered? Quote
\cref{def:g11:forces-and-motion:friction}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:11}
A hammer thrower whirls the ball on its wire. What force turns the
ball? Sketch its path after release. Why does the Moon need no wire,
and what would its path become without gravity?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:12}
A broom is a \qty{1.0}{kg} uniform stick (own center of mass
\qty{0.60}{m} from the top) with a \qty{2.0}{kg} head whose center of
mass is \qty{1.20}{m} from the top. Treating each part as a point mass,
locate the broom's center of mass and check the inverse-ratio rule of
\cref{def:g11:forces-and-motion:com}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:13}
A car of mass $m$ and a loaded van of mass $2m$ brake from the same
speed with the same braking force.
\begin{enumerate}
  \item Compare the times needed to stop.
  \item Each loses speed steadily, hence travels at the same
        \emph{average} speed: compare the stopping distances.
  \item The car now brakes from \emph{double} the speed, same force:
        compare time and distance with its first stop. Moral for speed limits?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:14}
The Moon orbits at $R \approx \qty{3.84e8}{m}$ in $T = \num{27.3}$ days.
\begin{enumerate}
  \item Compute its orbital speed (circumference over period).
  \item Which way does its velocity change point, and which force
        provides it?
  \item Improve, using \cref{prop:g11:forces-and-motion:inward}: ``the
        Moon does not fall because it moves fast enough''.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:15}
An egg of mass \qty{50}{g} hits the floor at \qty{4.0}{m/s}: on tile it
stops in about \qty{1}{ms}, on thick foam in about \qty{50}{ms}. Compare
the average forces. Explain, with the same reasoning, why you bend your
knees when landing a jump, and what a gymnast's crash mat is made of.
\end{exercise}

\section{Problem: The crash test}

\begin{problem}[{Weekend problem --- designing the gentle collision:
since a crash fixes the velocity change, every safety device in a car
is a machine for stretching time}]
\label{pb:g11:forces-and-motion:1}
A crash-test laboratory drives a car at \qty{50}{km/h} into a rigid
wall, with a \qty{70}{kg} adult dummy and a \qty{20}{kg} child dummy
aboard; cameras time every stop. Your job: find where the violence of a
crash lives, and design it away.

\medskip
\noindent\textbf{Part I --- The measure of a crash.}
\begin{enumerate}
  \item Convert \qty{50}{km/h} to \unit{m/s}.
  \item In any frontal stop the occupants' velocity change has the same
        magnitude. Which one? Why can no device alter it?
  \item Quote \cref{thm:g11:forces-and-motion:secondlaw}: with $\Delta v$
        and $m$ fixed, $F\,\Delta t$ is fixed. The designer's only dial?
  \item The wall stops the car in \qty{0.050}{s}; a crumple zone
        stretches this to \qty{0.150}{s}. Compare the average forces.
  \item In which direction do $\Delta \vect v$ and the net force on
        the dummies point during the stop?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crumple zone.}
\begin{enumerate}[resume]
  \item Why, in one sentence, do engineers design the front of the car
        to be destroyed?
  \item In a vintage rigid car (stop in \qty{0.050}{s}) the dummy is strapped
        rigidly to the seat. Compare its force with the modern \qty{0.150}{s} stop.
  \item The stopping car's speed falls steadily (average: half the initial).
        How far does it travel in \qty{0.150}{s}? Compare with a real crumple zone.
  \item Why not a \qty{10}{m} crumple zone? And which part of the car
        must \emph{not} crumple?
  \item Summarize Part II as a slogan: ``you cannot choose $\Delta v$,
        so you must choose \dots''.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The belt, the airbag and the child.}
\begin{enumerate}[resume]
  \item An unbelted dummy keeps its \qty{14}{m/s} (\cref{ch:g10:inertia})
        until the windshield stops it in \qty{0.005}{s}; a belted one
        stops with the car in \qty{0.150}{s}. Compare the average forces.
  \item Why is a belt that stretches slightly better than an
        unstretchable steel harness?
  \item The head would stop on the wheel in \qty{0.010}{s}; the airbag
        stretches this to \qty{0.080}{s}. Compare the forces.
  \item Adult and child undergo the same $\Delta v$ in the same
        $\Delta t$. Compare the net forces the two belts must supply.
  \item A child on a lap: the arms must change a \qty{10}{kg} baby's
        velocity by \qty{14}{m/s} in \qty{0.150}{s}, while gravity
        changes a velocity by \qty{9.8}{m/s} each second. How many times
        faster must the arms act --- holding what mass would feel that
        heavy? Conclude.
  \item Children feel smaller forces --- so why child seats? Think of
        \emph{where} the belt applies its force, and of $\Delta t$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The safety cage, and the verdict.}
\begin{enumerate}[resume]
  \item The passenger cell is rigid while both ends crumple. Why must
        the cell not deform, though rigidity means violent stops?
  \item Two identical cars collide head-on, each at \qty{50}{km/h}.
        Where does each stop? Compare each crash with the wall test.
  \item A small car meets a loaded truck head-on. The forces they exert
        on each other are equal and opposite, mutual as every interaction
        is (\cref{ch:g11:fundamental-interactions}). Whose velocity
        changes more?
  \item List the safety devices of this problem and the single quantity
        every one of them stretches.
  \item Finale: state the design law of the gentle collision in one
        sentence using only \emph{mass}, \emph{velocity change},
        \emph{force} and \emph{time}.
\end{enumerate}
\end{problem}
