\chapter{Arbeid van een kracht}\label{ch:g11:work-of-force}

Duw een kast door de kamer en je hebt gearbeid; duw de hele middag
tegen een muur en, wat je spieren ook melden, de fysica telt nul.
\omterm{def:g11:work-of-force:work}{Arbeid} is wat een
\omterm{def:g10:inertia:force}{kracht} levert \emph{langs een beweging}
--- \omterm{def:g10:energy-conservation:energy}{energie} in transit
(\cref{ch:g10:energy-conservation}), betaald in
\omterm{def:g11:work-of-force:work}{joule}. Dit hoofdstuk prijst de
\omterm{def:g10:inertia:force}{krachten} van
\cref{ch:g11:forces-and-motion} --- trekkrachten onder een hoek,
\omterm{def:g10:universal-gravitation:weight}{gewicht},
\omterm{def:g10:inertia:inventory}{wrijving} --- en klokt de
betalingsnelheid: \omterm{def:g11:work-of-force:power}{vermogen}.

\section{Arbeid van een constante kracht}

\begin{definition}[Arbeid van een constante kracht]\label{def:g11:work-of-force:work}
Een constante \omterm{def:g10:inertia:force}{kracht} $\vect F$ toegepast
op een voorwerp dat langs een recht segment van $A$ naar $B$ beweegt,
levert de \emph{arbeid}\index{arbeid (van een kracht)}
\[
W_{AB}(\vect F) = F \times AB \times \cos\theta ,
\]
waar $F$ de grootte van de \omterm{def:g10:inertia:force}{kracht} is (in
\unit{N}), $AB$ de lengte van de verplaatsing (in \unit{m}) en $\theta$
de hoek tussen $\vect F$ en $\vect{AB}$. Arbeid wordt gemeten in
\emph{joule}\index{joule} (\unit{J}): één joule is de arbeid van een
\omterm{def:g10:inertia:force}{één-newton}-\omterm{def:g10:inertia:force}{kracht}
langs één \omterm{def:g10:orders-of-magnitude:unit}{meter} van haar eigen
richting.
\end{definition}

\begin{remark}\label{rem:g11:work-of-force:scal}
$F \times AB \times \cos\theta$ is precies het inwendig product
$\vect F \cdot \vect{AB}$ van het wiskundeboek van dit jaar. Alleen de
component $F\cos\theta$ \emph{langs} de beweging levert iets op.
\end{remark}

\begin{example}[Koffer aan een riem]\label{ex:g11:work-of-force:suitcase}
Een reiziger sleept een koffer \qty{300}{m} door een terminal, de riem
getrokken met $F = \qty{40}{N}$ onder $\theta = \qty{50}{\degree}$ met
de vloer: $W = 40 \times 300 \times \cos\qty{50}{\degree} \approx
\qty{7.7}{kJ}$. Plat getrokken ($\theta = 0$) zou dezelfde
\omterm{def:g10:inertia:force}{kracht} \qty{12}{kJ} leveren: een derde van
de trek tilt niets.
\end{example}

\begin{definition}[Motorische, weerstands- en nularbeid]\label{def:g11:work-of-force:sign}
Het teken van $\cos\theta$ sorteert elke
\omterm{def:g10:inertia:force}{kracht} in drie regimes:
\begin{itemize}
  \item $\theta < \qty{90}{\degree}$: $W > 0$, de
        \omterm{def:g11:work-of-force:work}{arbeid} is
        \emph{motorisch}\index{motorische arbeid} --- de
        \omterm{def:g10:inertia:force}{kracht} voedt
        \omterm{def:g10:energy-conservation:energy}{energie} in de
        beweging;
  \item $\theta = \qty{90}{\degree}$: $W = 0$, de
        \omterm{def:g10:inertia:force}{kracht}
        \omterm{def:g11:work-of-force:work}{arbeidt} helemaal niet, hoe
        groot ook;
  \item $\theta > \qty{90}{\degree}$: $W < 0$, de
        \omterm{def:g11:work-of-force:work}{arbeid} is
        \emph{weerstand}\index{weerstandsarbeid} --- de
        \omterm{def:g10:inertia:force}{kracht} onttrekt
        \omterm{def:g10:energy-conservation:energy}{energie}. Het pure geval
        is een rem op $\qty{180}{\degree}$: \qty{800}{N} over \qty{45}{m}
        haalt $800 \times 45 \times (-1) = \qty{-36}{kJ}$ uit een auto,
        in hete schijven.
\end{itemize}
\end{definition}

\section{De arbeid van het gewicht}

\begin{proposition}[Arbeid van het gewicht]\label{prop:g11:work-of-force:weight}
Wanneer een voorwerp van massa $m$ van $A$ (hoogte $z_A$) naar $B$
(hoogte $z_B$) beweegt langs \emph{elk} pad gemaakt van rechte segmenten,
levert haar \omterm{def:g10:universal-gravitation:weight}{gewicht}
$\vect P$ de \omterm{def:g11:work-of-force:work}{arbeid}
\[
W_{AB}(\vect P) = mg\,(z_A - z_B) = \pm\, mgh ,
\]
waar $h = \abs{z_A - z_B}$ de hoogtedaling is: $+mgh$ naar beneden,
$-mgh$ naar boven --- alleen de daling telt, niet de route.
\end{proposition}

\begin{proof}
Op één recht segment $AB$ maakt het
\omterm{def:g10:universal-gravitation:weight}{gewicht} (grootte $mg$,
recht omlaag) een hoek $\theta$ met $\vect{AB}$, en $AB\cos\theta$ is
precies de projectie van de verplaatsing op de neerwaartse verticaal:
$z_A - z_B$. Dus $W = mg \times AB\cos\theta = mg\,(z_A - z_B)$.
Langs een keten van segmenten tellen de
\omterm{def:g11:work-of-force:work}{arbeiden} op en de dalingen
telescooppen: $mg(z_A - z_C) + mg(z_C - z_B) = mg(z_A - z_B)$.
\end{proof}

\begin{proposition}[Elk pad]\label{prop:g11:work-of-force:anypath}
Dezelfde formule geldt langs elk gekromd pad van $A$ naar $B$.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:work-of-force:anypath}
Aannemelijk --- een kromme is zo dicht als we willen bij een gebroken
lijn van vele korte segmenten --- maar ``zo dicht als we willen'' eerlijk
maken is calculus, gedaan in \cref{ch:g12:work-and-energy}.
\end{remark}

\begin{example}[Wandelaar]\label{ex:g11:work-of-force:hiker}
Een \qty{78}{kg}-wandelaar (rugzak inbegrepen) klimt \qty{450}{m}, via
welk pad ook: $W(\vect P) = -78 \times 9.81 \times 450 \approx

omwegen ook.
\end{example}

\section{De arbeid van de wrijving}

\begin{proposition}[Arbeid van glijdende wrijving]\label{prop:g11:work-of-force:friction}
Een \omterm{def:g10:inertia:inventory}{wrijving}-\omterm{def:g10:inertia:force}{kracht}
van constante grootte $f$, elk moment tegengesteld aan de beweging,
levert de \omterm{def:g11:work-of-force:work}{arbeid} $W = -fL$ langs een
pad van totale lengte $L$: altijd
\omterm{def:g11:work-of-force:sign}{weerstand}, en evenredig met de
\emph{lengte van het afgelegde pad} --- niet met de verplaatsing.
\end{proposition}

\begin{proof}
Elk kort recht stuk heeft de \omterm{def:g10:inertia:force}{kracht} op
$\qty{180}{\degree}$ met de beweging, bijdragend
$-f \times (\text{zijn lengte})$; de stukken tellen op tot $-fL$. Een
gekromd pad wordt eveneens in korte stukken gehakt, de limiet aangenomen
zoals voor het \omterm{def:g10:universal-gravitation:weight}{gewicht}
(\cref{rem:g11:work-of-force:anypath}).
\end{proof}

\begin{remark}[Wrijving tegen gewicht: de grote scheiding]\label{rem:g11:work-of-force:divide}
De \omterm{def:g11:work-of-force:work}{arbeid} van het
\omterm{def:g10:universal-gravitation:weight}{gewicht} vergeet het pad; die
van \omterm{def:g10:inertia:inventory}{wrijving} is een tol per
\omterm{def:g10:orders-of-magnitude:unit}{meter}. Op een heen-en-weer-reis
netto't het \omterm{def:g10:universal-gravitation:weight}{gewicht} nul ---
wat het bergop neemt, betaalt het bergaf terug --- terwijl
\omterm{def:g10:inertia:inventory}{wrijving} $-fL$ rekent op elk been en
niets terugbetaalt: de \omterm{def:g11:work-of-force:work}{arbeid} van het
\omterm{def:g10:universal-gravitation:weight}{gewicht} kan worden opgeslagen
en herwonnen (potentiële
\omterm{def:g10:energy-conservation:energy}{energie},
\cref{ch:g11:mechanical-energy}), die van
\omterm{def:g10:inertia:inventory}{wrijving} is verloren als warmte.
\end{remark}

\begin{example}[Heen en terug]\label{ex:g11:work-of-force:roundtrip}
Een kist glijdt \qty{5.0}{m} over een vloer en \qty{5.0}{m} terug,
tegen $f = \qty{60}{N}$: \omterm{def:g10:inertia:inventory}{wrijving} levert
$-60 \times 10 = \qty{-600}{J}$ hoewel de kist eindigt waar ze begon.
\omterm{def:g10:universal-gravitation:weight}{Gewicht} en normaalkracht,
loodrecht op de beweging: nul doorheen.
\end{example}

\begin{remark}[En een veer?]\label{rem:g11:work-of-force:spring}
Een veer trekt harder naarmate ze verder uitgerekt is: haar
\omterm{def:g10:inertia:force}{kracht} is niet constant en
\cref{def:g11:work-of-force:work} geldt niet. Haar
\omterm{def:g11:work-of-force:work}{arbeid} betekent de rek in
infinitesimale stappen snijden en sommeren --- calculus, geleverd in
\cref{ch:g12:work-and-energy}.
\end{remark}

\section{Vermogen}

\begin{definition}[Vermogen]\label{def:g11:work-of-force:power}
Het \emph{vermogen}\index{vermogen!mechanisch} van een
\omterm{def:g10:inertia:force}{kracht} is de snelheid waarmee ze
\omterm{def:g11:work-of-force:work}{arbeidt}, $P = W/\Delta t$, in
\emph{watt}\index{watt} ($\unit{W} = \unit{J/s}$) --- dezelfde watt als
in \cref{ch:g11:circuits-and-power}. Een pre-SI
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} overleeft op
motorbadges: de \emph{paardenkracht}\index{paardenkracht},
$\qty{1}{hp} = \qty{736}{W}$.
\end{definition}

\begin{proposition}[Vermogen bij constante snelheid]\label{prop:g11:work-of-force:pfv}
Als het voorwerp beweegt met constante snelheid $v$, levert een
constante \omterm{def:g10:inertia:force}{kracht} $\vect F$ onder hoek
$\theta$ met de beweging het \omterm{def:g11:work-of-force:power}{vermogen}
$P = F v \cos\theta$ --- in het bijzonder $P = Fv$ voor een
\omterm{def:g10:inertia:force}{kracht} langs de beweging.
\end{proposition}

\begin{proof}
In een tijd $\Delta t$ legt het voorwerp $d = v\,\Delta t$ af, dus
$W = F d \cos\theta = F v \cos\theta\,\Delta t$, en
$P = W/\Delta t$.
\end{proof}

\begin{example}[Een hijskraan]\label{ex:g11:work-of-force:crane}
Een kraan tilt \qty{800}{kg} met constante snelheid \qty{0.40}{m/s}:
$F = mg = \qty{7.85e3}{N}$, $P = Fv = 7850 \times 0.40 =
\qty{3.1}{kW} \approx \qty{4.3}{hp}$. In \qty{20}{s} is
$W = P\,\Delta t = \qty{63}{kJ}$ --- of $mgh$ met $h = 8.0$~m.
\end{example}

\begin{method}[De zigzag-rekening]\label{met:g11:work-of-force:zigzag}
Om te klimmen: dezelfde $mgh$ (de
\omterm{def:g11:work-of-force:work}{arbeid} van het
\omterm{def:g10:universal-gravitation:weight}{gewicht} is padonafhankelijk),
maar een langere zachte helling verkleint de vereiste
\omterm{def:g10:inertia:force}{kracht} en het
\omterm{def:g11:work-of-force:power}{vermogen}, ten koste van meer
\omterm{def:g10:inertia:inventory}{wrijving} over de langere route. Wegen
zigzaggen; trappen doen hetzelfde.
\end{method}

\section{Probleem: De dag van de verhuizer}

\begin{problem}[{Weekendprobleem --- de rekening van een verhuisdag: van
dozen over de vloer tot dozen de trap op, met de zigzag van de bergweg
ertussen, en de watt van een mens tegen die van een waterkoker}]
\label{pb:g11:work-of-force:1}
Een verhuisploeg besteedt de dag aan dozen schuiven, tillen en klimmen.
Elke handeling is een \omterm{def:g11:work-of-force:work}{arbeids}-rekening:
kracht maal pad, teken, padonafhankelijkheid of niet. Data waar nodig:
$g = \qty{9.81}{N/kg}$.

\medskip
\noindent\textbf{Deel I --- Schuiven.}
Een doos glijdt \qty{8.0}{m} over een vloer onder een horizontale duw
van \qty{110}{N}; de \omterm{def:g10:inertia:inventory}{wrijving} is
\qty{95}{N}.
\begin{enumerate}
  \item Bereken de \omterm{def:g11:work-of-force:work}{arbeid} van de duw.
  \item Bereken de \omterm{def:g11:work-of-force:work}{arbeid} van
        wrijving, gewicht en normaalkracht.
  \item Totale \omterm{def:g11:work-of-force:work}{arbeid}? Versnelt of
        vertraagt de doos? Welke duw houdt constante snelheid?
  \item De doos neemt de diagonaal van een \qty{6}{m}$\times$\qty{8}{m}
        kamer of de muren. Wrijvingsarbeid in elk geval?
  \item Heen en terug over \qty{10}{m}: wrijvings- en gewichtsarbeid?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Tillen en de helling.}
Een \qty{30}{kg}-doos moet \qty{9.0}{m} omhoog (verdieping).
\begin{enumerate}[resume]
  \item Arbeid van het gewicht; minimum arbeid van de ploeg?
  \item Met de verhuizer (\qty{70}{kg}): totale gewichtsarbeid; aandeel
        van de doos?
  \item Helling van lengte \qty{30}{m}, wrijving \qty{40}{N}: duwkracht
        en arbeid bij constante snelheid?
  \item Vergelijk met een verticale lier: kracht, arbeid. Wat koopt de
        helling?
  \item Lier tilt op \qty{0.50}{m/s}: tijd en vermogen?
  \item Lierendement $0.60$: elektrisch vermogen? Controleer $P = Fv$.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De bergweg.}
Een \qty{3500}{kg}-vrachtwagen klimt \qty{240}{m} over een steile baan
(helling $24\%$, lengte \qty{1000}{m}) of een weg (helling $3.0\%$,
lengte \qty{8000}{m}).
\begin{enumerate}[resume]
  \item Controleer dat beide op dezelfde top uitkomen.
  \item Vereiste kracht (geen wrijving) op elke route?
  \item Arbeid op elke route? Commentaar.
  \item Rolweerstand \qty{400}{N}: extra wrijvingsarbeid op elke route?
  \item Op \qty{36}{km/h}: vermogen op elke route, in kW en hp?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De dagrekening.}
\begin{enumerate}[resume]
  \item Zestig dozen zoals Deel~I (duw) plus Deel~II (lier zonder
        wrijving): totale arbeid?
  \item Twintig trappen van \qty{70}{kg} over \qty{9.0}{m}:
        gewichtsarbeid; dagtotaal?
  \item Over \qty{8.0}{h}: gemiddeld vermogen? Hoe lang zou een
        \qty{2.0}{kW}-waterkoker dezelfde energie nemen?
  \item Punchline: wat betaalt de joule, wat negeert ze, en waarom
        gebruiken verhuizers machines?
\end{enumerate}
\end{problem}
