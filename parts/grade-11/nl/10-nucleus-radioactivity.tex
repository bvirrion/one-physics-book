\chapter{De kern en de radioactiviteit}\label{ch:g11:nucleus-radioactivity}

Op dit moment exploderen zo'n achtduizend atoomkernen elke seconde in je
lichaam. Niets valt ze aan: bepaalde kernen worden onstabiel geboren, en
elke transformeert zich vroeger of later, en slingert een fragment uit
met een flinke fractie van de lichtsnelheid. Dit hoofdstuk keert terug
naar de $10^{-15}$-meter-verdieping van
\cref{ch:g11:fundamental-interactions} om te zien welke kernen fragiel
zijn, hoe ze breken, en hoe hun regelmatige ongeduld grotten dateert en
plafonds bewaakt.

\section{Kernen, nucleonen en isotopen}

\begin{definition}[Nucleaire notatie]\label{def:g11:nucleus-radioactivity:notation}
Een \omterm{def:g11:fundamental-interactions:ladder}{kern} bevat $Z$
protonen (lading $+e$ elk) en $N$ neutronen (geen lading),
$A = Z + N$ \omterm{def:g11:fundamental-interactions:ladder}{nucleonen} in
totaal; ze wordt geschreven ${}^{A}_{Z}\mathrm{X}$, waar $\mathrm X$ het
chemisch symbool is, $Z$ het \emph{atoomnummer}\index{atoomnummer} en $A$
het \emph{massagetal}\index{massagetal}. Een soort gespecificeerd door $A$
en $Z$ is een \emph{nuclide}\index{nuclide}; nucliden met dezelfde $Z$
maar verschillende $A$ zijn \emph{isotopen}\index{isotoop} van één
element: dezelfde chemie (de elektronenwolk ziet alleen $Ze$),
verschillende kernen en stabiliteit.
\end{definition}

\begin{example}[Waterstof en koolstof]\label{ex:g11:nucleus-radioactivity:hc}
Waterstof heeft drie
\omterm{def:g11:nucleus-radioactivity:notation}{isotopen}:
${}^{1}_{1}\mathrm{H}$ (een eenzame proton), ${}^{2}_{1}\mathrm{H}$
(deuterium: één proton, één neutron), onstabiel ${}^{3}_{1}\mathrm{H}$
(tritium: twee neutronen). Natuurlijk koolstof: vooral
${}^{12}_{6}\mathrm{C}$ ($6$ protonen, $6$ neutronen), $1.1\%$
${}^{13}_{6}\mathrm{C}$, en één
\omterm{def:g11:fundamental-interactions:ladder}{atoom} op $10^{12}$ van
onstabiel ${}^{14}_{6}\mathrm{C}$ ($8$ neutronen), held van
\cref{ex:g11:nucleus-radioactivity:dating}.
\end{example}

\section{Stabiele en onstabiele kernen}

\begin{proposition}[De stabiliteitsvallei]\label{prop:g11:nucleus-radioactivity:valley}
Stabiele \omterm{def:g11:nucleus-radioactivity:notation}{nucliden} bezetten
een smalle band in het $(Z, N)$-vlak: $N \approx Z$ voor lichte kernen,
daarna een groeiend neutronenoverschot, tot $N \approx 1.5\,Z$. Elke
\omterm{def:g11:nucleus-radioactivity:notation}{nuclide} buiten de band --- te
veel neutronen, te veel protonen, of te groot (voorbij lood, $Z = 82$, is
geen \omterm{def:g11:nucleus-radioactivity:notation}{nuclide} echt stabiel)
--- transformeert vroeger of later.
\end{proposition}
\admitted

\begin{remark}
De band precies lokaliseren is kwantummechanica --- de bachelorvolumes. Het
patroon is het grootboek van \cref{ch:g11:fundamental-interactions}: sterke
lijm bindt alleen buren, Coulomb-afstoting overspant de hele
\omterm{def:g11:fundamental-interactions:ladder}{kern}, dus zware kernen
hebben extra neutronenlijm nodig; toch is een overschot aan neutronen ook
onstabiel (de
\omterm{def:g11:fundamental-interactions:weak}{zwakke wisselwerking} zorgt
ervoor).
\end{remark}

\begin{definition}[Radioactiviteit]\label{def:g11:nucleus-radioactivity:radioactivity}
\emph{Radioactiviteit}\index{radioactiviteit} is de spontane transformatie
van een onstabiele
\omterm{def:g11:fundamental-interactions:ladder}{kern} in een andere
\omterm{def:g11:nucleus-radioactivity:notation}{nuclide}, met emissie van
straling. Ze is \emph{willekeurig}: niets kondigt aan welke
\omterm{def:g11:fundamental-interactions:ladder}{kern} als volgende vervalt,
en geen \omterm{def:g10:pressure:pressure}{druk}, temperatuur of chemie kan
haar versnellen of vertragen; alleen grote populaties zijn voorspelbaar
(\cref{def:g11:nucleus-radioactivity:halflife}).
\end{definition}

\begin{remark}[Becquerel en de Curies]\label{rem:g11:nucleus-radioactivity:curie}
In 1896 vond Henri Becquerel dat uraniumzouten een fotografische plaat
beslaan door zwart papier, in een gesloten lade, zonder toegevoerde
\omterm{def:g10:energy-conservation:energy}{energie}. Marie en Pierre Curie
toonden dat het effect atomair is, maten het, en isoleerden polonium en
radium, een miljoen keer actiever. Het woord
\emph{\omterm{def:g11:nucleus-radioactivity:radioactivity}{radioactiviteit}}
is van Marie Curie; evenals twee Nobelprijzen.
\end{remark}

\section{Alfa, bèta, gamma}

\begin{definition}[De drie historische stralingen]\label{def:g11:nucleus-radioactivity:abg}
Drie stralingen, genoemd voordat iemand wist wat ze waren:
\begin{itemize}
  \item \emph{alfa}\index{alfastraling} ($\alpha$): een helium-\omterm{def:g11:fundamental-interactions:ladder}{kern}
        ${}^{4}_{2}\mathrm{He}$ geheel uitgeworpen --- zwaar, dubbel
        geladen, gestopt door een vel papier of centimeters lucht;
  \item \emph{bèta}\index{bètastraling} ($\beta^-$): een elektron
        ${}^{\;0}_{-1}\mathrm{e}$ gecreëerd in de
        \omterm{def:g11:fundamental-interactions:ladder}{kern} en
        uitgeworpen --- licht, snel, gestopt door millimeters aluminium;
  \item \emph{gamma}\index{gammastraling} ($\gamma$): elektromagnetische
        straling, veel energieker dan licht --- nooit echt gestopt, alleen
        \emph{verzwakt}\index{verzwakking}: een centimeter lood absorbeert
        de helft.
\end{itemize}
\end{definition}

\begin{proposition}[Behoud in nucleaire vergelijkingen]\label{prop:g11:nucleus-radioactivity:soddy}
In elke nucleaire transformatie zijn het totale
\omterm{def:g11:nucleus-radioactivity:notation}{massagetal} $A$ en de totale
lading --- de som van de onderste indices, met $-1$ voor een uitgezonden
elektron --- hetzelfde voor en na.
\end{proposition}
\admitted

\begin{remark}
Ladingbehoud regeerde al \cref{ch:g11:fundamental-interactions}; behoud van
$A$ zegt dat \omterm{def:g11:fundamental-interactions:ladder}{nucleonen}
alleen worden omgezet, nooit gecreëerd of vernietigd --- eerlijke
boekhouding voor de bachelorvolumes.
\end{remark}

\begin{method}[Een vervalvergelijking balanceren]\label{met:g11:nucleus-radioactivity:balance}
\begin{enumerate}
  \item Schrijf elke actor met beide labels: $\alpha = {}^{4}_{2}\mathrm{He}$,
        $\beta^- = {}^{\;0}_{-1}\mathrm{e}$,
        $\beta^+ = {}^{0}_{+1}\mathrm{e}$; een $\gamma$ draagt $A = Z = 0$.
  \item Stel de sommen van de $A$'s gelijk, daarna van de $Z$'s
        (\cref{prop:g11:nucleus-radioactivity:soddy}); los op, en lees het
        dochterelement af van haar $Z$ in het
        \omterm{def:g10:signals-and-waves:period}{periodiek} systeem.
\end{enumerate}
\end{method}

\begin{example}[Een alfaverval]\label{ex:g11:nucleus-radioactivity:alpha}
Radium-226, het radium van de Curies, is een $\alpha$-emitter:
${}^{226}_{88}\mathrm{Ra} \to {}^{A}_{Z}\mathrm{Y} + {}^{4}_{2}\mathrm{He}$
\omterm{def:g10:inertia:force}{dwingt} $A = 226 - 4 = 222$ en
$Z = 88 - 2 = 86$; element $86$ is radon, dus
${}^{226}_{88}\mathrm{Ra} \to {}^{222}_{86}\mathrm{Rn} + {}^{4}_{2}\mathrm{He}$
--- het radioactieve gas van slecht geventileerde kelders.
\end{example}

\begin{example}[Een bètaminus-verval]\label{ex:g11:nucleus-radioactivity:beta}
Koolstof-14 zit boven de band ($8$ neutronen tegen $6$ protonen):
${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} + {}^{\;0}_{-1}\mathrm{e}$.
Controle: $14 = 14 + 0$, $6 = 7 - 1$; een neutron werd een proton plus
het uitgeworpen elektron --- de
\omterm{def:g11:fundamental-interactions:weak}{zwakke wisselwerking} aan het
\omterm{def:g11:work-of-force:work}{werk}.
\end{example}

\begin{remark}[Bètaplus en gamma]\label{rem:g11:nucleus-radioactivity:betaplus}
Onder de band keert het spiegelverval $\beta^+$ een proton om in een
neutron plus een \emph{positron}\index{positron} ${}^{0}_{+1}\mathrm{e}$,
het antideeltje van het elektron:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$,
werkpaard van PET-scanners. En na de meeste vervallen settelt de dochter,
schuddend geboren, door een $\gamma$ erbovenop uit te zenden.
\end{remark}

\section{Halfwaardetijd en activiteit}

\begin{definition}[Halfwaardetijd]\label{def:g11:nucleus-radioactivity:halflife}
De \emph{halfwaardetijd}\index{halfwaardetijd} $T_{1/2}$ van een
\omterm{def:g11:nucleus-radioactivity:notation}{nuclide} is de tijd waarna de
helft van elk groot monster is vervallen. Elke verdere halfwaardetijd
halveert wat overblijft: van $N_0$ kernen blijven $N_0/2$ over bij
$T_{1/2}$, $N_0/4$ bij $2\,T_{1/2}$, $N_0/8$ bij $3\,T_{1/2}$,
$N_0/2^{\,n}$ na $n$ halfwaardetijden.
\end{definition}

\begin{remark}
Willekeur voor één
\omterm{def:g11:fundamental-interactions:ladder}{kern}, klokwerk voor
$10^{23}$: een enorme menigte muntuimelaars, één voor één
onvoorspelbaar, half geëlimineerd elke ronde. Interpoleren
\emph{tussen} hele halfwaardetijden vraagt de exponentiële functie ---
volgend jaar, met het wiskundeboek.
\end{remark}

\begin{example}[Twintig machten van tien]\label{ex:g11:nucleus-radioactivity:range}
\begin{center}\small\begin{tabular}{llll}
polonium-214 & $\alpha$ & \qty{1.6e-4}{s} & radiumketenschakel \\
technetium-99m & $\gamma$ & $6.0$ uur & medische beeldvorming \\
americium-241 & $\alpha$ & $432$ jaar & rookmelders \\
koolstof-14 & $\beta^-$ & \num{5700} jaar & archeologie \\
kalium-40 & $\beta$ & \num{1.25e9} jaar & in elke banaan \\
uranium-238 & $\alpha$ & \num{4.5e9} jaar & binnenwarmte van de Aarde \\
\end{tabular}\end{center}
\end{example}

\begin{example}[Koolstof-14-datering]\label{ex:g11:nucleus-radioactivity:dating}
Levende materie wisselt koolstof met de atmosfeer, dus houdt ze de
atmosferische verhouding koolstof-14; bij de dood stopt de inname en
halveert de verhouding elke \num{5700} jaar. Houtskool uit een
beschilderde grot toont een kwart van de levende verhouding: een kwart
is half van een half, dus het vuur brandde twee halfwaardetijden,
$2 \times \num{5700} = \num{11400}$ jaar, geleden.
\end{example}

\begin{definition}[Activiteit]\label{def:g11:nucleus-radioactivity:activity}
De \emph{activiteit}\index{activiteit} $\mathcal A$ van een monster is
haar aantal vervallen per seconde, gemeten in
\emph{becquerel}\index{becquerel}: $\qty{1}{Bq} = 1$ verval per seconde.
Minder overlevende kernen betekent minder vervallen, dus de activiteit
halveert ook bij elke
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}.
\end{definition}

\begin{example}[Ordes van grootte]\label{ex:g11:nucleus-radioactivity:orders}
Alles is licht radioactief. Een banaan: ongeveer \qty{15}{Bq}
(kalium-40). Een menselijk lichaam: ongeveer \qty{8000}{Bq} --- de haak
van dit hoofdstuk, zo'n vijfhonderd bananen waard. Een kubieke
\omterm{def:g10:orders-of-magnitude:unit}{meter} graniet: rond
\qty{3e6}{Bq} (uranium en haar keten, vandaar kelder-radon). Eén
technetium-injectie: ongeveer \qty{5e8}{Bq}, binnen dagen weg.
\end{example}

\section{Dienaren en gevaren}

\begin{example}[Drie carrières van een onstabiele kern]\label{ex:g11:nucleus-radioactivity:uses}
\emph{Geneeskunde}: technetium-99m, een pure $\gamma$-emitter met
$T_{1/2} = 6$ uur, wordt vastgezet aan een
\omterm{def:g11:fundamental-interactions:ladder}{molecule} die het
doelorgaan absorbeert; een camera filmt de $\gamma$'s van buiten, en de
volgende dag is de tracer grotendeels verdwenen --- terwijl radiotherapie
de logica omkeert, bundels focust om een tumor te vernietigen.
\emph{Datering}: koolstof-14 voor hout, bot en stof tot zo'n
\num{50000} jaar; uranium-238 voor rotsen --- en de Aarde, op
\num{4.5e9} jaar. \emph{Rookmelders}: een stipje americium-241 ioniseert
de lucht van een kleine kamer, zodat een minuscule stroom vloeit; rook
smoort de stroom en het alarm gaat af --- de $\alpha$'s sterven in
centimeters lucht.
\end{example}

\begin{remark}[Dosis en de sievert]\label{rem:g11:nucleus-radioactivity:dose}
Straling rukt elektronen van
\omterm{def:g11:fundamental-interactions:ladder}{moleculen} en kan DNA
beschadigen. De biologische schade wordt gevolgd door de
\emph{dosis}\index{dosis}, in \emph{sievert}\index{sievert} (\unit{Sv}),
die de \omterm{def:g10:energy-conservation:energy}{energie} afgezet per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} weefsel weegt naar hoe
schadelijk elke straling is. Drie regels: afstand, afscherming ($\alpha$:
huid volstaat --- tenzij de emitter wordt \emph{ingeademd}, de misdaad van
radon; $\beta$: aluminium; $\gamma$: lood), en tijd. Eigenlijke
dosimetrie is bachelormateriaal.
\end{remark}

De uranium-238-keten eindigt, vele stappen later, bij stabiel
${}^{206}_{82}\mathrm{Pb}$; elke stap is een $\alpha$ of een $\beta^-$.
Gebruik alleen behoud van $A$ om het aantal $\alpha$-stappen te vinden;
daarna, met behoud van $Z$, het aantal $\beta^-$-stappen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:13}
De kobalt-60-bron van een ziekenhuis ($T_{1/2} = 5.3$ jaar) heeft
\omterm{def:g11:nucleus-radioactivity:activity}{activiteit} \qty{6.4}{TBq};
regels laten afvoer toe onder \qty{0.1}{TBq}. Na hoeveel jaar mag ze
worden afgevoerd?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:14}
De \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} van je lichaam
is ongeveer \qty{8000}{Bq}. Hoeveel van je kernen vervallen per dag? Over
een $80$-jarig leven? Een banaan voegt ongeveer \qty{15}{Bq} toe terwijl
je haar verteert: commentaar, in één zin, op koppen die elke
\omterm{def:g11:nucleus-radioactivity:activity}{becquerel} vrezen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:15}
Een rookmelder houdt \qty{0.2}{\micro\gram} americium-241
($T_{1/2} = 432$ jaar), van
\omterm{def:g11:nucleus-radioactivity:activity}{activiteit} ongeveer
\qty{25}{kBq}. Hoeveel vervallen is dat per dag? Na tien jaar dienst,
is het americium veel minder dan half, ongeveer half, of veel meer
vervallen --- en is bronuitputting waarom melders worden vervangen?
Ten slotte, waarom is deze $\alpha$-bron ongevaarlijk aan het plafond,
maar gevaarlijk als de capsule vermalen en het stof ingeademd wordt?
\end{exercise}

\section{Probleem: De mummie, de reactor en de rookmelder}

\begin{problem}[{Weekendprobleem --- drie carrières van een onstabiele
kern: een mummie gedateerd door halveringen, de stamboom van uranium
gebalanceerd, een rookmelder geaudit, en waarom de sterren sterfelijk
moesten zijn}]
\label{pb:g11:nucleus-radioactivity:1}
Drie onstabiele kernen, drie banen: koolstof-14 tijdstempelt elke dode
plant en elk dier, uranium-238 verwarmt een planeet van binnen,
americium-241 bewaakt plafonds. Gegevens: $T_{1/2} = \num{5700}$ jaar
(C-14), \num{4.5e9} jaar (U-238), $432$ jaar (Am-241); levend koolstof
toont $13.6$ vervallen per minuut per gram;
\omterm{def:g11:nucleus-radioactivity:notation}{atoomnummers}: N $7$, O $8$,
Pb $82$, Bi $83$, Po $84$, Rn $86$, Ra $88$, Th $90$, Pa $91$, U $92$,
Np $93$, Am $95$.

\medskip
\noindent\textbf{Deel I --- De mummie.}
\begin{enumerate}
  \item Geef de samenstelling (protonen, neutronen) van
        ${}^{12}_{6}\mathrm{C}$ en ${}^{14}_{6}\mathrm{C}$. Waarom
        identieke chemie in een lichaam?
  \item Waarom is de verhouding koolstof-14 constant in een levend
        lichaam, en waarom begint ze te dalen bij de dood?
  \item Schrijf de vervalvergelijking van koolstof-14. Zou haar
        straling de linnen windselen van de mummie ontsnappen?
  \item Een gram koolstof uit het linnen toont $6.8$ vervallen per
        minuut. Welke fractie van de levende snelheid is dat? Dateer
        de mummie.
  \item Een tweede ``mummie'', de prijs van een privécollectie, toont
        $13.2$ vervallen per minuut per gram. Oordeel?
  \item Schat de snelheid van een monster tien halfwaardetijden oud,
        en verklaar waarom koolstofdatering vervaagt voorbij grofweg
        \num{57000} jaar.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De reactor onder de weide.}
\begin{enumerate}[resume]
  \item Geef de samenstelling van ${}^{238}_{92}\mathrm{U}$. Waarom
        hebben zware kernen zo'n neutronenoverschot nodig?
  \item Schrijf haar $\alpha$-vervalvergelijking.
  \item De dochter vervalt dan $\beta^-$, en de kleindochter opnieuw
        $\beta^-$. Schrijf beide vergelijkingen. Welk element
        verschijnt opnieuw?
  \item De keten eindigt bij ${}^{206}_{82}\mathrm{Pb}$. Tel haar
        $\alpha$-stappen, daarna haar $\beta^-$-stappen.
  \item De Aarde ontstond \num{4.5e9} jaar geleden. Welke fractie van
        haar oeruranium-238 blijft vandaag over? En over $5$ miljard
        jaar meer?
  \item In één zin: wat drijft deze begraven vervalwarmte aan het
        oppervlak aan? (Denk vulkanen en drijvende continenten.)
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De rookmelder.}
\begin{enumerate}[resume]
  \item Schrijf de $\alpha$-vervalvergelijking van
        ${}^{241}_{95}\mathrm{Am}$.
  \item Haar \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} is
        \qty{33}{kBq}: hoeveel vervallen per seconde? Per dag?
  \item Verklaar de melder: wat doen de $\alpha$-deeltjes met de lucht
        van de kamer, en wat verandert rook?
  \item Waarom is een $\alpha$-emitter hier de juiste keuze --- en een
        $\gamma$-emitter van gelijke
        \omterm{def:g11:nucleus-radioactivity:activity}{activiteit} de
        slechtste?
  \item Na één
        \omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}
        ($432$ jaar!) zou de
        \omterm{def:g11:nucleus-radioactivity:activity}{activiteit}
        \qty{16.5}{kBq} zijn; de handleiding vraagt vervanging na tien
        jaar. Is de bron de zwakke schakel?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Waarom de sterren sterfelijk moesten zijn.}
\begin{enumerate}[resume]
  \item Stervende sterren smeedden en verspreidden elk element zwaarder
        dan helium. Welke fractie van uranium-238 gesmeed \num{9e9} jaar
        geleden overleeft vandaag? Waarom maakt de resterende
        overvloed van de Aarde de planeet veel jonger dan de oudste
        sterren?
  \item De radioactieve binnenwarmte van de Aarde drijft vulkanisme en
        platentektoniek, en houdt de kern kolkend die het magnetisch
        schild van \cref{ch:g11:magnetic-fields} uitstoot. Hoe zou een
        planeet van alleen stabiele
        \omterm{def:g11:fundamental-interactions:ladder}{atomen} eruitzien?
  \item Eén regel, als poëzie en als boekhouding: waar werden jouw
        \omterm{def:g11:fundamental-interactions:ladder}{atomen} gesmeed, en
        wat verwarmt nog de grond onder je voeten?
\end{enumerate}
\end{problem}
