\chapter{Magnetisme en magnetische velden}\label{ch:g11:magnetic-fields}

Zeelieden stuurden duizend jaar lang op gemagnetiseerde naalden voordat
iemand wist waarom ze naar het noorden wijzen. Het antwoord kwam in
1820, toen een stroom een kompas deed trillen: magnetisme wordt gemaakt
door bewegende ladingen. Dit hoofdstuk brengt het
\omterm{def:g11:magnetic-fields:bfield}{magnetisch veld} in kaart --- van
\omterm{def:g11:magnetic-fields:magnet}{magneten}, van de Aarde, van
stromen --- en sluit de lus: het
\omterm{def:g11:electric-gravitational-fields:field}{veld} duwt terug op
stromen, en zo draait elke elektromotor.

\section{Magneten en het magnetisch veld}

\begin{definition}[Magneet en polen]\label{def:g11:magnetic-fields:magnet}
Een \emph{magneet}\index{magneet} trekt ijzer aan en, aan een draad
opgehangen, draait om het noorden te zoeken. Haar werking concentreert
zich in twee \emph{magnetische polen}\index{magnetische pool},
\emph{noord} (het uiteinde dat geografisch noorden zoekt) en
\emph{zuid}: gelijke polen stoten af, ongelijke trekken aan. Een
magneet doormidden zagen levert twee complete magneten --- geen
experiment heeft ooit een pool geïsoleerd.
\end{definition}

\begin{definition}[Magnetisch veld]\label{def:g11:magnetic-fields:bfield}
Een \omterm{def:g11:magnetic-fields:magnet}{magneet} wijzigt de ruimte
eromheen (\cref{ch:g11:electric-gravitational-fields}): ze creëert overal
een \emph{magnetisch veld}\index{magnetisch veld} $\vect B$. Een kompas
leest haar richting (de naald richt zich langs $\vect B$, noordpunt
vooruit); een \emph{Hall-sonde}\index{Hall-sonde} meet haar grootte, in
\emph{tesla}\index{tesla} (\unit{T}).
\end{definition}

\begin{definition}[Magnetische veldlijnen]\label{def:g11:magnetic-fields:lines}
\emph{Magnetische veldlijnen}\index{magnetische veldlijn} zijn krommen
raak aan $\vect B$, die opeenpakken waar het
\omterm{def:g11:electric-gravitational-fields:field}{veld} sterk is. Buiten
een \omterm{def:g11:magnetic-fields:magnet}{magneet} lopen ze van noordpool
naar zuidpool; in tegenstelling tot
\omterm{def:g11:electric-gravitational-fields:efield}{elektrische
veldlijnen} hebben ze geen einden: elk sluit door het lichaam van de
\omterm{def:g11:magnetic-fields:magnet}{magneet}.
\end{definition}

\section{De Aarde is een magneet}

\begin{definition}[Het magnetisch veld van de Aarde]\label{def:g11:magnetic-fields:earth}
Stromen in de gesmolten ijzerkern van de Aarde maken de planeet tot een
reusachtige \omterm{def:g11:magnetic-fields:magnet}{magneet}: aan het
oppervlak $B \approx \qty{50}{\micro T}$, en een vrije naald richt zich
langs de horizontale component van het
\omterm{def:g11:electric-gravitational-fields:field}{veld} --- dit is het
kompas. De geomagnetische polen zijn niet de geografische; de hoek tussen
kompasnoorden en ware noorden is de
\emph{declinatie}\index{declinatie}, gecorrigeerd op elke peiling.
\end{definition}

\begin{example}[Een peiling corrigeren]\label{ex:g11:magnetic-fields:bearing}
Waar de \omterm{def:g11:magnetic-fields:earth}{declinatie} $\ang{7}$ west is,
drijft een schip dat ``kompasnoorden'' stevent over \qty{5.0}{km}
$5000 \times \sin\ang{7} \approx \qty{6.1e2}{m}$ west van zijn koers: om
ware noorden te varen, stev $\ang{7}$ \emph{oost} van de naald.
\end{example}

\section{Het magnetisch veld van een stroom}

\begin{remark}[Oersted, 1820]\label{rem:g11:magnetic-fields:oersted}
Tijdens een college in 1820 zag Hans Christian Oersted een
kompasnaald zwenken toen hij een circuit dichtbij sloot: elektriciteit
en magnetisme waren één vak --- \emph{stromen creëren
\omterm{def:g11:magnetic-fields:bfield}{magnetische velden}}.
\end{remark}

\begin{proposition}[Veld van een rechte draad]\label{prop:g11:magnetic-fields:wire}
Een lange rechte draad die een stroom $I$ voert, creëert een
\omterm{def:g11:electric-gravitational-fields:field}{veld} waarvan de lijnen
cirkels zijn gecentreerd op de draad, in vlakken loodrecht erop. Op
afstand $d$ van de draad,
\[
B = \frac{\mu_0\, I}{2\pi d},
\qquad \mu_0 = 4\pi\times 10^{-7}\ \unit{T.m/A} \approx \qty{1.26e-6}{T.m/A}.
\]
\end{proposition}
\admitted

\begin{example}[Een draad is een zwakke magneet]\label{ex:g11:magnetic-fields:wire}
Op $d = \qty{1.0}{cm}$ van een draad die $I = \qty{15}{A}$ voert:
$B = \num{2e-7} \times 15 / 0.010 = \qty{0.30}{mT}$ --- zes
Aarde-\omterm{def:g11:electric-gravitational-fields:field}{velden}, uit een
stroom die een huishoudelijke zekering zou doen doorslaan. Sterke
\omterm{def:g11:electric-gravitational-fields:field}{velden} vragen een betere
meetkunde.
\end{example}

\begin{definition}[Spoel en solenoïde]\label{def:g11:magnetic-fields:solenoid}
Het wikkelen van de draad concentreert haar
\omterm{def:g11:electric-gravitational-fields:field}{veld}. Een \emph{platte
spoel}\index{spoel} ($N$ windingen in een schijf) gedraagt zich als een
\omterm{def:g11:lenses-and-eye:lens}{dunne}
\omterm{def:g11:magnetic-fields:magnet}{magneet}, één vlak noord, het andere
zuid; een \emph{solenoïde}\index{solenoïde} --- een lange cilindrische
wikkeling, $n$ windingen per
\omterm{def:g10:orders-of-magnitude:unit}{meter} --- als een
staaf\omterm{def:g11:magnetic-fields:magnet}{magneet}.
\end{definition}

\begin{proposition}[Veld binnen een solenoïde]\label{prop:g11:magnetic-fields:solenoidfield}
Binnen een lange \omterm{def:g11:magnetic-fields:solenoid}{solenoïde}, weg van
de einden, is het
\omterm{def:g11:electric-gravitational-fields:field}{veld}
\emph{homogeen}: evenwijdig aan de as en hetzelfde in elk inwendig punt,
van grootte $B = \mu_0\, n\, I$, onafhankelijk van de boringbreedte.
Buiten is het \omterm{def:g11:electric-gravitational-fields:field}{veld} zwak.
\end{proposition}
\admitted

\begin{remark}
Beide formules worden eerlijk afgeleid in het volume van bachelorjaar~1,
uit de wet die de circulatie van een
\omterm{def:g11:electric-gravitational-fields:field}{veld} relateert aan de
stroom die ze omcirkelt. Let op wat telt: $n$, windingen \emph{per
\omterm{def:g10:orders-of-magnitude:unit}{meter}} --- wikkel dichter, niet
langer.
\end{remark}

\begin{method}[De rechterhand, tweemaal]\label{met:g11:magnetic-fields:righthand}
\begin{itemize}
  \item \emph{Draad}: duim $=$ stroom; gekrulde vingers $=$ draaiing van
        de lijnen.
  \item \emph{\omterm{def:g11:magnetic-fields:solenoid}{Spoel},
        \omterm{def:g11:magnetic-fields:solenoid}{solenoïde}}: gekrulde
        vingers $=$ stroom in de windingen; duim $=$ $\vect B$ binnen,
        wijzend naar het noordvlak.
\end{itemize}
\end{method}

\begin{example}[Solenoïde-getallen]\label{ex:g11:magnetic-fields:solenoid}
Bij $10$ windingen per centimeter ($n = \qty{1000}{m^{-1}}$) en
$I = \qty{5.0}{A}$: $B = \num{1.26e-6} \times 1000 \times 5.0 \approx
\qty{6.3}{mT}$ --- honderd
Aarde-\omterm{def:g11:electric-gravitational-fields:field}{velden},
regelbaar met een knop.
\end{example}

\section{Elektromagneten}

\begin{definition}[Elektromagneet]\label{def:g11:magnetic-fields:electromagnet}
Een \emph{elektromagneet}\index{elektromagneet} is een
\omterm{def:g11:magnetic-fields:solenoid}{solenoïde} gewikkeld op een zachte
ijzeren kern: het ijzer magnetiseert in het
\omterm{def:g11:electric-gravitational-fields:field}{veld} van de
\omterm{def:g11:magnetic-fields:solenoid}{spoel} en vermenigvuldigt het,
typisch met $100$ tot $1000$. In tegenstelling tot een permanente
\omterm{def:g11:magnetic-fields:magnet}{magneet} schakelt ze uit met de
stroom.
\end{definition}

\begin{remark}[Drie machines]\label{rem:g11:magnetic-fields:machines}
De hijskraan van de schroothoop tilt een auto met een
\omterm{def:g11:magnetic-fields:electromagnet}{elektromagneet} en --- het
hele punt --- \emph{laat} hem \emph{vallen} door een schakelaar te
openen. Het kleine
\omterm{def:g11:magnetic-fields:electromagnet}{elektromagneetje} van een
relais trekt een contact dicht: een zwakke stroom schakelt een sterke
(\cref{ch:g11:circuits-and-power}). Een MRI-scanner houdt een homogeen
\qty{1.5}{T} rond de patiënt met een supergeleidende
\omterm{def:g11:magnetic-fields:solenoid}{solenoïde}.
\end{remark}

Ordes van grootte, van het sterrenstelsel tot een dode ster:

een koelkast\omterm{def:g11:magnetic-fields:magnet}{magneet} & $\sim \qty{5e-3}{T}$ \\

\end{tabular}
\end{center}

\section{De Laplace-kracht}

\begin{proposition}[Laplace-kracht]\label{prop:g11:magnetic-fields:laplace}
Een rechte draad van lengte $L$, die een stroom $I$ voert in een
\omterm{def:g11:electric-gravitational-fields:capacitor}{homogeen veld}
$\vect B$ \emph{loodrecht} op de draad, voelt de
\emph{Laplace-kracht}\index{Laplace-kracht} van grootte $F = I\,L\,B$,
loodrecht op zowel draad als $\vect B$: platte rechterhand, duim langs
de stroom, gestrekte vingers langs $\vect B$ --- de palm duwt langs
$\vect F$. Een draad \emph{evenwijdig} aan het
\omterm{def:g11:electric-gravitational-fields:field}{veld} voelt geen
\omterm{def:g10:inertia:force}{kracht}.
\end{proposition}
\admitted

\begin{remark}[De springende rail]\label{rem:g11:magnetic-fields:rail}
Het volume van bachelorjaar~1 leidt dit af uit de magnetische
\omterm{def:g10:inertia:force}{kracht} op elke bewegende lading. In het
laboratorium: een koperen staaf rust over twee horizontale rails tussen
de polen van een \omterm{def:g11:magnetic-fields:magnet}{magneet}
(\omterm{def:g11:electric-gravitational-fields:field}{veld} de bladzijde in
in de figuur); sluit de schakelaar, en de staaf schiet langs de rails.
Keer de stroom om, of keer de
\omterm{def:g11:magnetic-fields:magnet}{magneet}: ze schiet de andere kant
op.
\end{remark}

\begin{example}[Laplace-getallen]\label{ex:g11:magnetic-fields:laplace}
Een staaf van lengte $L = \qty{5.0}{cm}$ die $I = \qty{8.0}{A}$ voert
over $B = \qty{0.50}{T}$: $F = 8.0 \times 0.050 \times 0.50 =
\qty{0.20}{N}$ --- het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van \qty{20}{g}, op
een draad van enkele grammen.
\end{example}

\begin{remark}[De DC-motor]\label{rem:g11:magnetic-fields:motor}
Buig de draad tot een lus in het
\omterm{def:g11:electric-gravitational-fields:field}{veld}: de twee zijden
loodrecht op $\vect B$ voeren tegengestelde stromen, dus hun
\omterm{prop:g11:magnetic-fields:laplace}{Laplace-krachten} zijn
tegengesteld --- een paar dat de lus doet draaien. Een halve slag later
zou het paar haar terugdraaien, dus een roterende schakelaar (de
\emph{collector}\index{collector}) keert de stroom elke halve slag om.
Elke ventilator en boormachine is deze lus, vermenigvuldigd.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:1}
Een staaf\omterm{def:g11:magnetic-fields:magnet}{magneet} wordt tussen haar
polen doormidden gezaagd. Welke polen draagt elk stuk? Weer
samengevoegd, trekken de vers gezaagde vlakken aan of stoten ze af? Kan
enige zaagschema de noordpool isoleren?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:2}
Waar of onwaar, met één reden elk: (a) twee
\omterm{def:g11:magnetic-fields:lines}{magnetische veldlijnen} kruisen waar
het \omterm{def:g11:electric-gravitational-fields:field}{veld} sterk is; (b)
buiten een \omterm{def:g11:magnetic-fields:magnet}{magneet} lopen
\omterm{def:g11:electric-gravitational-fields:lines}{veldlijnen} van noordpool
naar zuidpool; (c) elke
\omterm{def:g11:magnetic-fields:lines}{magnetische veldlijn} is een gesloten
lus; (d) een naald komt tot rust loodrecht op de lijn erdoor.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:3}
Bereken het \omterm{def:g11:electric-gravitational-fields:field}{veld}
\qty{2.0}{cm} van een rechte draad die \qty{10}{A} voert. Hoeveel keer
de \qty{50}{\micro T} van de Aarde is dat?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:4}
Een \omterm{def:g11:magnetic-fields:solenoid}{solenoïde} van $800$ windingen
over \qty{40}{cm} voert $I = \qty{1.5}{A}$. Bereken $n$, daarna $B$
binnen; wat wordt $B$ als de stroom verdubbelt?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:5}
Een draadsegment van lengte \qty{10}{cm} voert \qty{5.0}{A} loodrecht op
een \qty{0.20}{T}-\omterm{def:g11:electric-gravitational-fields:field}{veld}.
Bereken de \omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht}; het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van welke massa
evenaart haar?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:6}
Waar een wandelaar staat, is de
\omterm{def:g11:magnetic-fields:earth}{declinatie} $\ang{8}$ west. Ze loopt
\qty{2.0}{km} ``noord per kompas'': hoe ver, en naar welke kant van
ware noorden, drijft ze? Welke koers zou haar ware noorden hebben
gewandeld?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:7}
Op welke afstand van een rechte draad die \qty{20}{A} voert, evenaart
haar \omterm{def:g11:electric-gravitational-fields:field}{veld} de
\qty{50}{\micro T} van de Aarde? Wat zegt dit over het vertrouwen van
een kompas nabij onder spanning staande kabels?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:8}
Ontwerp een \omterm{def:g11:magnetic-fields:solenoid}{solenoïde} die
$B = \qty{10}{mT}$ produceert met $I = \qty{4.0}{A}$: bereken de
vereiste $n$, daarna het windingenaantal voor een \qty{25}{cm}-\omterm{def:g11:magnetic-fields:solenoid}{spoel}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:9}
In het railexperiment voert de staaf (massa \qty{20}{g}, lengte
$L = \qty{12}{cm}$ tussen de rails) \qty{10}{A} in een
\qty{0.50}{T} loodrecht
\omterm{def:g11:electric-gravitational-fields:field}{veld}. Bereken de
\omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht}, daarna de
beginversnelling van de staaf; vergelijk met $g$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:10}
Een horizontale draad van massa \qty{5.0}{g} en lengte \qty{20}{cm}
zit in een horizontaal
\omterm{def:g11:electric-gravitational-fields:field}{veld}
$B = \qty{0.10}{T}$ loodrecht erop. Welke stroom maakt de
\omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht} het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van de draad
balanceren, zodat ze zweeft? Welke kant moet de
\omterm{def:g10:inertia:force}{kracht} op wijzen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:11}
Geef de richting van de
\omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht} (of zeg dat ze
verdwijnt): (a) stroom naar rechts op de bladzijde, $\vect B$ de
bladzijde in; (b) stroom naar de top van de bladzijde, $\vect B$ de
bladzijde uit; (c) stroom evenwijdig aan $\vect B$. Gebruik de platte
rechterhand.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:12}
Een horizontale draad loopt geografisch noord--zuid, \qty{3.0}{cm}
boven een kompas; de horizontale component van de Aarde daar is
\qty{20}{\micro T}. De draad voert $I = \qty{5.0}{A}$: bereken haar
\omterm{def:g11:electric-gravitational-fields:field}{veld} bij het kompas,
geef haar richting, daarna de afwijking van de naald
(\omterm{def:g11:electric-gravitational-fields:field}{velden} tellen op als
vectoren).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:13}
Een schroot-\omterm{def:g11:magnetic-fields:electromagnet}{elektromagneet} is
een \omterm{def:g11:magnetic-fields:solenoid}{spoel} van $400$ windingen over
\qty{20}{cm}, \omterm{def:g11:circuits-and-power:resistance}{weerstand}
\qty{2.0}{\ohm}, op een \qty{12}{V}-voeding; haar ijzeren kern
vermenigvuldigt het kale
\omterm{def:g11:electric-gravitational-fields:field}{veld} met $100$. Bereken
de stroom, het
\omterm{def:g11:magnetic-fields:solenoid}{kale-spoel}-\omterm{def:g11:electric-gravitational-fields:field}{veld},
het \omterm{def:g11:electric-gravitational-fields:field}{veld} met de kern, en
het \omterm{def:g10:energy-conservation:power}{vermogen} dat de
\omterm{def:g11:magnetic-fields:solenoid}{spoel} dissipeert
(\cref{ch:g11:circuits-and-power}). Waarom valt het schroot op het
ogenblik dat de schakelaar opent?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:14}
De rechthoekige lus van een motor heeft $N = 50$ windingen die
$I = \qty{2.0}{A}$ voeren; haar twee zijden van lengte \qty{5.0}{cm}
staan loodrecht op een
\omterm{def:g11:electric-gravitational-fields:field}{veld}
$B = \qty{0.30}{T}$. Bereken de
\omterm{def:g10:inertia:force}{kracht} op elk van deze zijden; waarom
zijn de twee \omterm{def:g10:inertia:force}{krachten} tegengesteld, en wat
doen ze met de lus? Waarom voelen de andere twee zijden soms geen
\omterm{def:g10:inertia:force}{kracht}? Waarom moet de stroom elke halve
slag omgekeerd worden?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:15}
Hoe dicht bij een kabel die \qty{100}{A} voert zou je moeten zijn opdat

\omterm{def:g11:electric-gravitational-fields:field}{velden} uit
\omterm{def:g11:magnetic-fields:solenoid}{solenoïden} worden gebouwd, niet
uit enkele draden. Dan: het
oppervlakte-\omterm{def:g11:electric-gravitational-fields:field}{veld} van

\end{exercise}

\section{Probleem: Het kompas, de kraan en de MRI}

\begin{problem}[{Weekendprobleem --- drie magneten aan het werk: een
kompas dat een oceaan oversteekt, een elektromagneet die een auto tilt,
en een MRI-solenoïde --- één veld, spanning over tien machten van tien,
drie banen}]
\label{pb:g11:magnetic-fields:1}
Dezelfde vector $\vect B$ stuurt een zeilboot met \qty{50}{\micro T},
tilt schrootijzer met \qty{1}{T}, en beeldt een knie af met
\qty{1.5}{T} --- drie machines voor de drie formules van dit
hoofdstuk.

\medskip
\noindent\textbf{Deel I --- Het kompas.}
Op de positie van het schip heeft het
\omterm{def:g11:electric-gravitational-fields:field}{veld} van de Aarde
horizontale component \qty{20}{\micro T} en verticale component
\qty{44}{\micro T}; de
\omterm{def:g11:magnetic-fields:earth}{declinatie} is $\ang{6}$ west.
\begin{enumerate}
  \item Waarom wijst een kompasnaald überhaupt (magnetisch) noorden?
  \item Bereken de grootte van het totale
        \omterm{def:g11:electric-gravitational-fields:field}{veld} en haar
        hoek met de horizontaal.
  \item ``Kompasnoorden'' stevend over \qty{30}{km}, hoe ver west van
        de ware-noorden-koers eindigt het schip?
  \item Stalen lading buigt de naald nog $\ang{4}$ west af: dezelfde
        vraag.
  \item Nabij de geomagnetische pool is een kompas nutteloos: welke
        component is de schuldige, en waarom?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De kraan.}
De schroot-\omterm{def:g11:magnetic-fields:electromagnet}{elektromagneet} is
een \omterm{def:g11:magnetic-fields:solenoid}{solenoïde} van $600$ windingen
over \qty{30}{cm}, van
\omterm{def:g11:circuits-and-power:resistance}{weerstand} \qty{1.5}{\ohm},
gevoed door een \qty{12}{V}-voeding; haar ijzeren kern vermenigvuldigt
het kale \omterm{def:g11:electric-gravitational-fields:field}{veld} met $50$.
\begin{enumerate}[resume]
  \item Bereken $n$, de windingdichtheid.
  \item Bereken de stroom in de
        \omterm{def:g11:magnetic-fields:solenoid}{spoel}
        (\cref{ch:g11:circuits-and-power}).
  \item Bereken het
        \omterm{def:g11:magnetic-fields:solenoid}{kale-spoel}-\omterm{def:g11:electric-gravitational-fields:field}{veld}
        $\mu_0 n I$.
  \item Bereken het
        \omterm{def:g11:electric-gravitational-fields:field}{veld} met de
        kern; vergelijk met de \qty{0.5}{T} van neodymium.
  \item Bereken het
        \omterm{def:g10:energy-conservation:power}{vermogen} gedissipeerd
        in de \omterm{def:g11:magnetic-fields:solenoid}{spoel}, daarna de
        \omterm{def:g10:energy-conservation:energy}{energie} voor een
        \qty{10}{\minute}-hijsploeg.
  \item Een permanente
        \omterm{def:g11:magnetic-fields:magnet}{magneet} zo sterk bestaat.
        Waarom toch een
        \omterm{def:g11:magnetic-fields:electromagnet}{elektromagneet}?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De lier-motor.}
De lier van de kraan is een DC-motor: een rechthoekige lus van
$N = 100$ windingen, zijden van lengte \qty{4.0}{cm} loodrecht op
$B = \qty{0.25}{T}$, die $I = \qty{3.0}{A}$ voert.
\begin{enumerate}[resume]
  \item Bereken de
        \omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht} op elk
        zijdebundel loodrecht op $\vect B$.
  \item De twee \omterm{def:g10:inertia:force}{krachten} zijn gelijk en
        tegengesteld, toch heffen hun effecten niet op. Wat produceren
        ze, en waarom moeten ze \emph{tegengesteld} zijn?
  \item Welke \omterm{def:g10:inertia:force}{kracht} werkt op de andere twee
        zijden, wanneer evenwijdig aan $\vect B$?
  \item Na een halve slag zouden dezelfde
        \omterm{def:g10:inertia:force}{krachten} de rotatie ongedaan maken.
        Wat doet de \omterm{rem:g11:magnetic-fields:motor}{collector}, en
        wanneer?
  \item Geef drie aparte ontwerpwijzigingen die elk het
        \omterm{def:g10:inertia:force}{kracht}paar verdubbelen.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De MRI.}
De \omterm{def:g11:magnetic-fields:solenoid}{solenoïde} van de scanner houdt
een homogeen \qty{1.5}{T} met een stroom $I = \qty{500}{A}$.
\begin{enumerate}[resume]
  \item Hoeveel keer het
        \omterm{def:g11:electric-gravitational-fields:field}{veld} van de
        Aarde is \qty{1.5}{T}?
  \item Bereken de windingdichtheid $n$ nodig \emph{zonder} enige
        ijzeren kern.
  \item Als de wikkeling
        \omterm{def:g11:circuits-and-power:resistance}{weerstand}
        \qty{0.20}{\ohm} had, welk
        \omterm{def:g10:energy-conservation:power}{vermogen} zou ze
        dissiperen? Welke eigenschap van de werkelijke wikkeling
        vermijdt dit, ten koste van extreme koude?
  \item Punchline: plaats de
        \omterm{def:g11:electric-gravitational-fields:field}{velden} van het
        schip, de kraan en de MRI op de machten-van-tien-ladder van
        interstellaire ruimte (\qty{e-10}{T}) tot een neutronenster
        (\qty{e8}{T}); wat veranderde tussen de machines --- de fysica,
        of de \omterm{def:g11:circuits-and-power:current}{ampères}?
\end{enumerate}
\end{problem}
