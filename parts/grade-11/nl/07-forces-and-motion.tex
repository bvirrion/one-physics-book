\chapter{Krachten en beweging}\label{ch:g11:forces-and-motion}

Gooi een sleutel die draaiend door de kamer vliegt: de einden wiebelen
wild, toch tekent één punt erin een perfecte parabool.
\Cref{ch:g10:inertia} regelde wat er gebeurt wanneer
\omterm{def:g10:inertia:force}{krachten}
\omterm{def:g10:inertia:compensate}{compenseren} --- niets. Dit hoofdstuk
vraagt wat er gebeurt wanneer ze dat niet doen: de overblijvende
\omterm{def:g10:inertia:force}{kracht} \emph{verandert de snelheid} ---
lichamen versnellen, vertragen, keren --- en zegt met hoeveel, in
verhoudingen zo niet nog in formules.

\section{Het zwaartepunt}

\begin{definition}[Zwaartepunt]\label{def:g11:forces-and-motion:com}
Het \emph{zwaartepunt}\index{zwaartepunt} van een lichaam of stelsel is
zijn balanspunt --- de gemiddelde positie van zijn massa: het
geometrisch middelpunt van een homogeen symmetrisch lichaam; voor twee
puntmassa's, het punt van het verbindingssegment op afstanden in de
\emph{omgekeerde} verhouding van de massa's,
$\frac{d_1}{d_2} = \frac{m_2}{m_1}$ --- dichter bij de zwaardere.
\end{definition}

\begin{example}[Een scheve dumbbell]\label{ex:g11:forces-and-motion:dumbbell}
Ballen van \qty{2.0}{kg} en \qty{1.0}{kg} zitten aan de einden van een
lichte staaf \qty{0.90}{m} lang. Het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} deelt haar in de
omgekeerde verhouding $d_1/d_2 = 1.0/2.0$: $\qty{0.30}{m}$ van de zware
bal, $\qty{0.60}{m}$ van de lichte --- balanceer de staaf daar.
\end{example}

\begin{remark}[De geworpen sleutel]\label{rem:g11:forces-and-motion:wrench}
Film een tuimelende sleutel: elk punt volgt een ingewikkelde lusende
kromme --- behalve het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt}, dat de schone parabool
tekent die een geworpen bal zou. De beweging van het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} is eenvoudig; de rest is
spin \emph{om} het --- daarom mag de mechanica een auto, een planeet of
een gymnast behandelen als één punt.
\end{remark}

\section{Een resulterende kracht verandert de snelheid}

\begin{definition}[Resulterende kracht]\label{def:g11:forces-and-motion:netforce}
De \emph{resulterende kracht}\index{resulterende kracht} op een lichaam
is de vectorsom van alle \omterm{def:g10:inertia:force}{krachten} die erop
werken, $\vect F = \vect F_1 + \vect F_2 + \dots$ (punt aan punt, zoals
in het wiskundeboek); de \omterm{def:g10:inertia:force}{krachten}
\omterm{def:g10:inertia:compensate}{compenseren} precies wanneer
$\vect F = \vect 0$.
\end{definition}

\begin{theorem}[Effect van een resulterende kracht --- semi-kwantitatief]\label{thm:g11:forces-and-motion:secondlaw}
\index{tweede wet van de beweging}
Een \omterm{def:g11:forces-and-motion:netforce}{resulterende kracht}
$\vect F$ die gedurende een korte tijd $\Delta t$ op een lichaam van
massa $m$ werkt, verandert de snelheid van haar
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} met een vector
$\Delta \vect v$ die de \emph{richting} van $\vect F$ heeft, en waarvan
de grootte groeit evenredig met $F$ en met $\Delta t$ en krimpt
evenredig met de massa:
\[
\Delta v \propto \frac{F \, \Delta t}{m} .
\]
\end{theorem}
\admitted

\begin{remark}
Verdubbel de \omterm{def:g10:inertia:force}{kracht} of haar duur: tweemaal de
verandering; verdubbel de massa: de helft. Volgend jaar verscherpt dit
tot een exacte vergelijking, en het volume van bachelorjaar~1 vraagt in
welke stelsels ze geldt; dit jaar volstaan verhoudingen --- ze runnen
al airbags, remmen en
\omterm{def:g10:universal-gravitation:satellite}{banen}.
\end{remark}

\begin{proposition}[De drie canonieke gevallen]\label{prop:g11:forces-and-motion:cases}
Laat een lichaam bewegen met snelheid $\vect v$ onder een
\omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} $\vect F$.
\begin{enumerate}
  \item $\vect F$ langs $\vect v$: het \emph{versnelt} in een rechte
        lijn;
  \item $\vect F$ tegengesteld aan $\vect v$: het \emph{vertraagt} in
        een rechte lijn;
  \item $\vect F$ loodrecht op $\vect v$: het \emph{keert} zonder
        snelheid te veranderen.
\end{enumerate}
\end{proposition}

\begin{proof}
Voeg $\Delta\vect v$, gericht langs $\vect F$
(\cref{thm:g11:forces-and-motion:secondlaw}), punt aan punt toe op
$\vect v$: langs $\vect v$ stapelen de pijlen (langer); tegengesteld
heffen ze deels op (korter); loodrecht kantelt elke kleine
$\Delta \vect v$ de pijl zonder hem te verlengen --- alleen de richting
keert.
\end{proof}

\begin{example}[Een zijdelingse trap]\label{ex:g11:forces-and-motion:kick}
Een puck glijdt oostwaarts op \qty{6.0}{m/s}; een korte zijdelingse slag
voegt $\Delta\vect v = \qty{8.0}{m/s}$ noordwaarts toe. De nieuwe
snelheid is de vectorsom: grootte $\sqrt{6.0^2 + 8.0^2} =
\qty{10.0}{m/s}$, richting $\tan\theta = 8.0/6.0$, d.w.z.\
$\theta \approx 53^\circ$ noord van oost --- de oude snelheid wordt niet
gewist, de verandering wordt eraan toegevoegd.
\end{example}

\section{Veranderingen vergelijken: kracht, tijd, massa}

\begin{method}[Verhoudingsredenering]\label{met:g11:forces-and-motion:ratio}
Om twee situaties te vergelijken, los nooit een vergelijking op ---
vorm verhoudingen: de snelheidsveranderingsverhouding is de $F$-verhouding,
maal de $\Delta t$-verhouding, gedeeld door de $m$-verhouding. Lees
daarna af welke factor veranderde, met hoeveel.
\end{method}

\begin{example}[De beladen kar]\label{ex:g11:forces-and-motion:cart}
Dezelfde duw, gedurende dezelfde seconde toegepast, stuurt een lege
\qty{20}{kg}-winkelwagen weg op \qty{1.2}{m/s}. Geladen tot
\qty{60}{kg} --- driedubbele massa, zelfde $F\Delta t$ --- vertrekt ze
op een derde daarvan, \qty{0.40}{m/s}. Massa is de koppigheid van
materie.
\end{example}

\begin{remark}[Crashes, airbags, gebogen knieën]\label{rem:g11:forces-and-motion:airbag}
In een crash zijn $\Delta v$ en $m$ niet onderhandelbaar --- de
snelheid van de inzittende moet nul worden --- dus $F\,\Delta t$ is
vast. De enige vrijheid die overblijft is \emph{tijd}: strek $\Delta t$
en de \omterm{def:g10:inertia:force}{kracht} krimpt in dezelfde verhouding.
Een hoofd dat op een airbag stopt in \qty{0.08}{s} in plaats van op het
stuur in \qty{0.01}{s} voelt een gemiddelde
\omterm{def:g10:inertia:force}{kracht} acht keer kleiner; kreukelzones,
helmschuim en gebogen knieën spelen dezelfde truc.
\end{remark}

\section{Wrijving: de kracht die glijden vecht}

\begin{definition}[Statische en kinetische wrijving]\label{def:g11:forces-and-motion:friction}
\omterm{def:g10:inertia:inventory}{Wrijving} tussen rakende oppervlakken
werkt hun \emph{relatief glijden} tegen. \emph{Statische
wrijving}\index{wrijving!statisch} werkt terwijl ze niet glijden, en
past zich aan tot een maximum om op te heffen wat de glij probeert te
starten; \emph{kinetische wrijving}\index{wrijving!kinetisch}, eens
glijden begonnen is, is ongeveer constant, gericht tegen het glijden, en
meestal \emph{zwakker} dan het statische maximum.
\end{definition}

\begin{example}[De kast duwen]\label{ex:g11:forces-and-motion:wardrobe}
Duw een kast zachtjes: ze beweegt niet --- statische
\omterm{def:g10:inertia:inventory}{wrijving} heft je \qty{150}{N} op. Duw
harder: bij zo'n \qty{400}{N} schiet ze in beweging, en glijdt daarna
met minder inspanning --- de zwakkere
\omterm{def:g11:forces-and-motion:friction}{kinetische wrijving} heeft
overgenomen.
\end{example}

\begin{remark}[Waarom ABS-remmen werken]\label{rem:g11:forces-and-motion:abs}
Het contactpunt van een rollend wiel glijdt niet op de weg: de band
grijpt met \emph{statische}
\omterm{def:g10:inertia:inventory}{wrijving}. Een geblokkeerd wiel slipt en
krijgt alleen de zwakkere \emph{kinetische}
\omterm{def:g10:inertia:inventory}{wrijving}, die tegen de slip wijst
ongeacht wat de wielen doen --- sturen sterft. ABS laat de rem los op
het ogenblik dat een wiel blokkeert: meer grip, kortere stop, een
bestuurder die kan sturen.
\end{remark}

\section{Keren: circulaire beweging eist een inwaartse kracht}

\begin{definition}[Eenparig cirkelvormige beweging]\label{def:g11:forces-and-motion:ucm}
Een lichaam is in \emph{eenparig cirkelvormige
beweging}\index{eenparig cirkelvormige beweging} wanneer haar
\omterm{def:g10:relative-motion:trajectory}{baan} een cirkel is die met
constante snelheid wordt afgelegd. De snelheid, raak aan de cirkel,
verandert elk moment van richting --- dus is ze nooit constant.
\end{definition}

\begin{proposition}[De inwaartse kracht]\label{prop:g11:forces-and-motion:inward}
Een lichaam in
\omterm{def:g11:forces-and-motion:ucm}{eenparig cirkelvormige beweging}
ondergaat een \omterm{def:g11:forces-and-motion:netforce}{resulterende
kracht} die nooit nul is: elk moment wijst ze van het lichaam naar het
middelpunt van de cirkel.
\end{proposition}

\begin{proof}
De snelheid is constant, dus de
\omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} heeft geen
deel langs $\vect v$ (gevallen 1 en 2 van
\cref{prop:g11:forces-and-motion:cases}): ze is loodrecht op de raaklijn,
dus langs de straal --- en inwaarts, omdat elke $\Delta \vect v$ het pad
naar het middelpunt buigt.
\end{proof}

\begin{example}[Het touw, de bocht, de Maan]\label{ex:g11:forces-and-motion:moon}
Elke gestage bocht verbergt een inwaartse
\omterm{def:g10:inertia:force}{kracht}. De rondgezwaaide bal: de
\omterm{def:g10:inertia:inventory}{spanning} van het touw. De auto in een
bocht: de zijdelingse statische
\omterm{def:g10:inertia:inventory}{wrijving} van weg op banden --- weg op
ijs, en de auto gaat rechtdoor. De Maan: de gravitationele trek van de
Aarde (\cref{ch:g10:universal-gravitation}), die haar in \num{27.3} dagen
ronddraait --- niet \emph{omhoog} gehouden maar zijwaarts getrokken in
een eeuwige bocht, vallend om de Aarde; volgend jaar maakt dit de
mechanica van
\omterm{def:g10:universal-gravitation:satellite}{satellieten}.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:1}
Nul of niet-nul \omterm{def:g11:forces-and-motion:netforce}{resulterende
kracht}? Rechtvaardig uit de snelheid: (a) een auto die rechtuit
kruist op gestage \qty{90}{km/h}; (b) remmend in een rechte lijn;
(c) een bocht nemend op gestage \qty{50}{km/h}; (d) een skydiver op
eindsnelheid.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:2}
Ballen van \qty{3.0}{kg} en \qty{1.0}{kg} zitten aan de einden van een
lichte staaf \qty{0.80}{m} lang. Lokaliseer het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt}; waar bij gelijke
massa's?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:3}
Dezelfde korte duw wordt gegeven aan een lege \qty{15}{kg}-bagagetrolley
en aan dezelfde trolley geladen tot \qty{45}{kg}. Vergelijk de
snelheidsveranderingen; welke duw zou de beladen trolley de $\Delta v$
van de lege geven?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:4}
Een tram die op \qty{10}{m/s} beweegt remt; één seconde later beweegt ze
op \qty{8}{m/s}, zelfde richting. Geef $\Delta \vect v$ en de richting
van de \omterm{def:g11:forces-and-motion:netforce}{resulterende kracht}. Welk
geval van \cref{prop:g11:forces-and-motion:cases} is dit?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:5}
Statische of \omterm{def:g11:forces-and-motion:friction}{kinetische wrijving}
--- en welke kant op gericht? (a) Een kist die je (te zachte) duw
weerstaan; (b) dezelfde kist eens ze glijdt; (c) een boek op een licht
gekanteld dienblad, niet glijdend.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:6}
Een \omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} $F$
toegepast gedurende $\Delta t$ op een massa $m$ produceert
$\Delta v = \qty{4.0}{m/s}$. Voorspel $\Delta v$ voor: (a)
$2F, \Delta t, m$; (b) $F, \Delta t/2, m$; (c) $F, \Delta t, 2m$;
(d) $3F, 2\Delta t, 6m$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:7}
Een puck glijdt oostwaarts op \qty{8.0}{m/s}; een slag voegt
$\Delta \vect v = \qty{6.0}{m/s}$ noordwaarts toe. Bereken de nieuwe
snelheid en richting. Waarom is de snelheid niet
$8.0 + 6.0 = \qty{14}{m/s}$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:8}
Een keeper stopt hetzelfde schot op twee manieren: armen stijf (de bal
stopt in \qty{0.02}{s}) of ``meegevend'' met de bal (\qty{0.08}{s}).
Vergelijk de gemiddelde \omterm{def:g10:inertia:force}{krachten}, en zeg

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:9}
Welke uitwendige \omterm{def:g10:inertia:force}{kracht} duwt een sprinter
vooruit bij de start? Waarom helpen spikes --- en waarom kan niemand

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:10}
Bij een noodstop, waarom stopt een auto met ABS (wielen rollend

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:11}
Een kogelslingeraar zwaait de bal aan haar draad. Welke
\omterm{def:g10:inertia:force}{kracht} keert de bal? Schets haar pad na

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:12}
Een bezem is een \qty{1.0}{kg} homogene steel (eigen
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} \qty{0.60}{m} van de
top) met een \qty{2.0}{kg}-kop waarvan het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} \qty{1.20}{m} van de
top is. Behandel elk deel als een puntmassa, lokaliseer het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} van de bezem en
controleer de omgekeerde-verhoudingsregel van
\cref{def:g11:forces-and-motion:com}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:13}
Een auto van massa $m$ en een beladen bus van massa $2m$ remmen vanaf
dezelfde snelheid met dezelfde rem-\omterm{def:g10:inertia:force}{kracht}.
\begin{enumerate}
  \item Vergelijk de tijden nodig om te stoppen.
  \item Elk verliest gestadig snelheid, reist dus met dezelfde
        \emph{gemiddelde} snelheid: vergelijk de stopafstanden.
  \item De auto remt nu vanaf \emph{dubbele} snelheid, zelfde
        \omterm{def:g10:inertia:force}{kracht}: vergelijk tijd en afstand
        met haar eerste stop. Moraal voor snelheidslimieten?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:14}
De Maan \omterm{def:g10:universal-gravitation:satellite}{draait} op
$R \approx \qty{3.84e8}{m}$ in $T = \num{27.3}$ dagen.
\begin{enumerate}
  \item Bereken haar omloopsnelheid (omtrek over
        \omterm{def:g10:signals-and-waves:period}{periode}).
  \item Welke kant wijst haar snelheidsverandering, en welke
        \omterm{def:g10:inertia:force}{kracht} levert haar?
  \item Verbeter, met \cref{prop:g11:forces-and-motion:inward}: ``de Maan
        valt niet omdat ze snel genoeg beweegt''.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:15}
Een ei van massa \qty{50}{g} raakt de vloer op \qty{4.0}{m/s}: op tegel
stopt het in ongeveer \qty{1}{ms}, op dik schuim in ongeveer
\qty{50}{ms}. Vergelijk de gemiddelde
\omterm{def:g10:inertia:force}{krachten}. Verklaar, met dezelfde
redenering, waarom je je knieën buigt bij een spronglanding, en waaruit
de valmat van een gymnast bestaat.
\end{exercise}

\section{Probleem: De crashtest}

\begin{problem}[{Weekendprobleem --- de zachte botsing ontwerpen: omdat
een crash de snelheidsverandering fixeert, is elk veiligheidsapparaat
in een auto een machine om tijd te strekken}]
\label{pb:g11:forces-and-motion:1}
Een crashtestlaboratorium rijdt een auto op \qty{50}{km/h} tegen een
rigide muur, met een \qty{70}{kg}-volwassen dummy en een
\qty{20}{kg}-kinddummy aan boord; camera's timen elke stop. Jouw taak:
vind waar het geweld van een crash leeft, en ontwerp het weg.

\medskip
\noindent\textbf{Deel I --- De maat van een crash.}
\begin{enumerate}
  \item Zet \qty{50}{km/h} om in \unit{m/s}.
  \item Bij elke frontale stop heeft de snelheidsverandering van de
        inzittenden dezelfde grootte. Welke? Waarom kan geen toestel
        haar wijzigen?
  \item Citeer \cref{thm:g11:forces-and-motion:secondlaw}: met $\Delta v$
        en $m$ vast is $F\,\Delta t$ vast. De enige knop van de
        ontwerper?
  \item De muur stopt de auto in \qty{0.050}{s}; een kreukelzone
        strekt dit tot \qty{0.150}{s}. Vergelijk de gemiddelde
        \omterm{def:g10:inertia:force}{krachten}.
  \item In welke richting wijzen $\Delta \vect v$ en de
        \omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} op
        de dummy's tijdens de stop?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De kreukelzone.}
\begin{enumerate}[resume]
  \item Waarom, in één zin, ontwerpen ingenieurs de voorkant van de
        auto om vernietigd te worden?
  \item In een vintage rigide auto (stop in \qty{0.050}{s}) is de dummy
        rigide aangespt aan de stoel. Vergelijk haar
        \omterm{def:g10:inertia:force}{kracht} met de moderne
        \qty{0.150}{s}-stop.
  \item De snelheid van de stoppende auto daalt gestadig (gemiddelde:
        de helft van de beginwaarde). Hoe ver reist ze in
        \qty{0.150}{s}? Vergelijk met een echte kreukelzone.
  \item Waarom geen \qty{10}{m}-kreukelzone? En welk deel van de auto
        mag \emph{niet} kreukelen?
  \item Vat Deel~II samen als een slogan: ``je kunt $\Delta v$ niet
        kiezen, dus moet je \ldots kiezen''.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De gordel, de airbag en het kind.}
\begin{enumerate}[resume]
  \item Een onaangespte dummy houdt haar \qty{14}{m/s}
        (\cref{ch:g10:inertia}) tot de voorruit haar stopt in
        \qty{0.005}{s}; een aangespte stopt met de auto in
        \qty{0.150}{s}. Vergelijk de gemiddelde
        \omterm{def:g10:inertia:force}{krachten}.
  \item Waarom is een gordel die licht rekt beter dan een
        onrekbaar stalen harnas?
  \item Het hoofd zou op het stuur stoppen in \qty{0.010}{s}; de airbag
        strekt dit tot \qty{0.080}{s}. Vergelijk de
        \omterm{def:g10:inertia:force}{krachten}.
  \item Volwassene en kind ondergaan dezelfde $\Delta v$ in dezelfde
        $\Delta t$. Vergelijk de
        \omterm{def:g11:forces-and-motion:netforce}{resulterende krachten}
        die de twee gordels moeten leveren.
  \item Een kind op een schoot: de armen moeten de snelheid van een
        \qty{10}{kg}-baby met \qty{14}{m/s} veranderen in
        \qty{0.150}{s}, terwijl zwaartekracht een snelheid met
        \qty{9.8}{m/s} per seconde verandert. Hoeveel keer sneller
        moeten de armen handelen --- het vasthouden van welke massa
        zou zo zwaar voelen? Concludeer.
  \item Kinderen voelen kleinere
        \omterm{def:g10:inertia:force}{krachten} --- waarom dan
        kinderzitjes? Denk aan \emph{waar} de gordel haar
        \omterm{def:g10:inertia:force}{kracht} toepast, en aan $\Delta t$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De veiligheidskooi, en het oordeel.}
\begin{enumerate}[resume]
  \item De passagierscel is rigide terwijl beide einden kreukelen.
        Waarom mag de cel niet vervormen, hoewel rigiditeit gewelddadige
        stops betekent?
  \item Twee identieke auto's botsen frontaal, elk op \qty{50}{km/h}.
        Waar stopt elk? Vergelijk elke crash met de muurtest.
  \item Een kleine auto ontmoet een beladen vrachtwagen frontaal. De
        \omterm{def:g10:inertia:force}{krachten} die ze op elkaar uitoefenen
        zijn gelijk en tegengesteld, wederzijds zoals elke wisselwerking
        (\cref{ch:g11:fundamental-interactions}). Wiens snelheid
        verandert meer?
  \item Som de veiligheidsapparaten van dit probleem en de enige
        hoeveelheid die elk van hen strekt.
  \item Finale: formuleer de ontwerpwette van de zachte botsing in één
        zin met alleen \emph{massa}, \emph{snelheidsverandering},
        \emph{\omterm{def:g10:inertia:force}{kracht}} en \emph{tijd}.
\end{enumerate}
\end{problem}
