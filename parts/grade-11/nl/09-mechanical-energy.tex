\chapter{Mechanische energie en haar behoud}\label{ch:g11:mechanical-energy}

Een achtbaan heeft geen motor voorbij de eerste heuvel: elke lus, elke
snelheidsuitbarsting erna wordt besteed uit de rekening die op de weg
omhoog geopend werd. Dit hoofdstuk zet de boekhouding om in twee
getallen --- kinetische en potentiële
\omterm{def:g10:energy-conservation:energy}{energie} --- en één regel:
zolang alleen het \omterm{def:g10:universal-gravitation:weight}{gewicht}
\omterm{def:g11:work-of-force:work}{arbeidt}, verandert hun som niet.
\omterm{def:g10:inertia:inventory}{Wrijving} is de tol; we leren haar van
de boeken af te lezen.

\section{Kinetische energie}

\begin{definition}[Kinetische energie]\label{def:g11:mechanical-energy:kinetic}
Een lichaam van massa $m$ (\unit{kg}) dat beweegt met snelheid $v$
(\unit{m/s}) draagt de \emph{kinetische energie}\index{kinetische energie}
\[
E_k = \tfrac12\, m v^2 .
\]
\omterm{def:g10:orders-of-magnitude:unit}{Eenheden}-check:
$\unit{kg.m^2/s^2} = \unit{N.m} = \unit{J}$ ---
\omterm{def:g11:work-of-force:work}{joule}, zoals elke
\omterm{def:g10:energy-conservation:energy}{energie}
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Ordes van grootte]\label{ex:g11:mechanical-energy:orders}
\begin{itemize}
  \item een voetganger, \qty{70}{kg} op \qty{1.4}{m/s}:
        $E_k = \tfrac12 \times 70 \times 1.4^2 \approx \qty{69}{J}$;
  \item een auto, \qty{1300}{kg} op \qty{130}{km/h} $= \qty{36.1}{m/s}$:
        $E_k \approx \qty{8.5e5}{J}$;
  \item een hogesnelheidstrein, $m = \qty{3.85e5}{kg}$, op
        \qty{320}{km/h} $= \qty{88.9}{m/s}$: $E_k \approx \qty{1.5e9}{J}$;
  \item een geweer kogel, \qty{8.0}{g} op \qty{800}{m/s}:
        $E_k \approx \qty{2.6}{kJ}$.
\end{itemize}
Zeven ordes van grootte scheiden de wandeling van de trein. En $E_k$
groeit als het \emph{kwadraat} van de snelheid: op \qty{130}{km/h}
draagt de auto $4$ keer de
\omterm{def:g10:energy-conservation:energy}{energie} die ze op
\qty{65}{km/h} heeft --- vier keer wat haar remmen moeten verwijderen.
\end{example}

\section{De kinetische-energie-stelling}

\begin{theorem}[Kinetische-energie-stelling]\label{thm:g11:mechanical-energy:ketheorem}
\index{kinetische-energie-stelling}
Voor een lichaam van massa $m$ dat langs een rechte lijn van $A$ naar
$B$ beweegt, is de verandering van
\omterm{def:g11:mechanical-energy:kinetic}{kinetische energie} gelijk aan
de totale \omterm{def:g11:work-of-force:work}{arbeid}
(\cref{ch:g11:work-of-force}) ontvangen:
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
de som over alle \omterm{def:g10:inertia:force}{krachten} die op het lichaam
werken. Ze strekt zich uit tot beweging langs elk pad, zoals we hier
ook aannemen.
\end{theorem}
\admitted

\begin{example}[Vrije val, frame voor frame gecontroleerd]\label{ex:g11:mechanical-energy:freefall}
Een \qty{0.200}{kg}-bal wordt losgelaten en gefilmd; de frames geven:
\begin{center}\small
\begin{tabular}{lcccc}
$t$ (\unit{s}) & 0.10 & 0.20 & 0.30 & 0.40\\
gevallen hoogte $h$ (\unit{m}) & 0.049 & 0.196 & 0.441 & 0.785\\
snelheid $v$ (\unit{m/s}) & 0.98 & 1.96 & 2.94 & 3.92\\
$E_k = \tfrac12 mv^2$ (\unit{J}) & 0.096 & 0.384 & 0.864 & 1.54\\
$W = mgh$ (\unit{J}) & 0.096 & 0.385 & 0.865 & 1.54
\end{tabular}
\end{center}
Startend van $E_k = 0$, matcht $E_k$ bij elk frame de
\omterm{def:g11:work-of-force:work}{arbeid} $mgh$ van het
\omterm{def:g10:universal-gravitation:weight}{gewicht}:
$\Delta E_k = W(\vect P)$, d.w.z.\ $v^2 = 2gh$ --- de stelling,
geverifieerd tot de precisie van de aflezingen.
\end{example}

\begin{remark}[Waar het bewijs woont]\label{rem:g11:mechanical-energy:proof}
De algemene stelling --- geldig voor elke baan, met elk stelsel
\omterm{def:g10:inertia:force}{krachten} --- is een oefening in calculus
en vectoren, geleverd in \cref{ch:g12:work-and-energy}. Dit jaar
gebruiken we de stelling; we bewijzen haar later.
\end{remark}

\section{Gravitationele potentiële energie}

\begin{definition}[Gravitationele potentiële energie]\label{def:g11:mechanical-energy:potential}
Nabij de Aarde is de \emph{gravitationele potentiële
energie}\index{potentiële energie} van een massa $m$ op hoogte $z$
boven een gekozen referentielvlak
\[
E_p = m g z .
\]
\end{definition}

\begin{remark}[De referentie]\label{rem:g11:mechanical-energy:reference}
Alleen \emph{verschillen} van $E_p$ tellen: verschuif het
\omterm{def:g11:mechanical-energy:potential}{referentielvlak} met
$z_0$ en elke $E_p$ verschuift met $mgz_0$, maar $\Delta E_p$ is
onveranderd. Kies de vloer, de grond, de top van de glijbaan --- wat
ook.
\end{remark}

\begin{example}[Op de plank]\label{ex:g11:mechanical-energy:shelf}
Een \qty{1.5}{kg}-boek op een plank \qty{1.8}{m} boven de vloer:
$E_p = 1.5 \times 9.81 \times 1.8 \approx \qty{26}{J}$ ten opzichte
van de vloer, $0$ ten opzichte van de plank. Val: $\Delta E_p =
-\qty{26}{J}$, en $\Delta E_k = +\qty{26}{J}$ bij aankomst.
\end{example}

\section{Mechanische energie en haar behoud}

\begin{definition}[Mechanische energie]\label{def:g11:mechanical-energy:em}
De \emph{mechanische energie}\index{mechanische energie} van een
lichaam is
\[
E_m = E_k + E_p = \tfrac12 m v^2 + m g z .
\]
\end{definition}

\begin{theorem}[Behoud van mechanische energie]\label{thm:g11:mechanical-energy:conservation}
\index{behoud van mechanische energie}
Als de enige \omterm{def:g10:inertia:force}{kracht} die
\omterm{def:g11:work-of-force:work}{arbeidt} het
\omterm{def:g10:universal-gravitation:weight}{gewicht} is (geen
\omterm{def:g10:inertia:inventory}{wrijving}, geen sleep, geen motor),
dan is $E_m$ constant langs de baan:
\[
E_m(A) = E_m(B) \qquad\text{d.w.z.}\qquad
\tfrac12 m v_A^2 + m g z_A = \tfrac12 m v_B^2 + m g z_B .
\]
\end{theorem}

\begin{proof}
De kinetische-energie-stelling geeft $\Delta E_k = W(\vect P) =
mg(z_A - z_B) = -\Delta E_p$, dus $\Delta E_k + \Delta E_p = 0$.
\end{proof}

\begin{example}[De glijbaan]\label{ex:g11:mechanical-energy:slide}
Een kind glijdt van \qty{3.0}{m} omlaag (geen wrijving). Op de bodem
$E_p = 0$: $\tfrac12 v^2 = g \times 3.0$, dus
$v = \sqrt{2 \times 9.81 \times 3.0} \approx \qty{7.7}{m/s}$ ---
ongeacht de vorm van de glijbaan.
\end{example}

\begin{example}[De slinger]\label{ex:g11:mechanical-energy:pendulum}
Een bob opgetild tot hoogte $h$ boven haar laagste punt en losgelaten:
op de bodem $v = \sqrt{2gh}$; ze stijgt terug tot dezelfde $h$ aan de
andere kant. De draad \omterm{def:g11:work-of-force:work}{arbeidt} niet
(loodrecht op de beweging).
\end{example}

\begin{example}[De schansspringer]\label{ex:g11:mechanical-energy:skijump}
Lancering op \qty{25}{m/s} van \qty{40}{m} hoogte. Landingssnelheid:
$v = \sqrt{25^2 + 2 \times 9.81 \times 40} \approx \qty{40}{m/s}$.
\end{example}

\section{Dissipatie door wrijving}

\begin{definition}[Dissipatie]\label{def:g11:mechanical-energy:dissipation}
Wanneer \omterm{def:g10:inertia:inventory}{wrijving} of weerstand
\omterm{def:g11:work-of-force:work}{arbeidt}, daalt $E_m$: de
\omterm{def:g10:energy-conservation:energy}{energie} wordt
\emph{gedissipeerd}\index{dissipatie} als warmte.
\end{definition}

\begin{proposition}[Het tekort]\label{prop:g11:mechanical-energy:deficit}
Langs een pad van $A$ naar $B$,
\[
E_m(B) - E_m(A) = W_{\text{niet-conservatief}} = W(\vect f) + \cdots
\]
waar $W(\vect f) = -fL$ voor glijdende wrijving over lengte $L$.
\end{proposition}

\begin{proof}
De kinetische-energie-stelling: $\Delta E_k = W(\vect P) + W(\vect f)
+ \cdots = -\Delta E_p + W(\vect f) + \cdots$.
\end{proof}

\begin{example}[De tobogan]\label{ex:g11:mechanical-energy:toboggan}
Een \qty{40}{kg}-tobogan daalt \qty{5.0}{m} over \qty{20}{m} met
$f = \qty{50}{N}$. $\Delta E_m = W_f = -50 \times 20 = \qty{-1.0}{kJ}$;
$E_k(\text{bodem}) = mgh + W_f = 1962 - 1000 = \qty{962}{J}$, dus
$v \approx \qty{6.9}{m/s}$ (tegen \qty{9.9}{m/s} zonder wrijving).
\end{example}

\begin{method}[Energieboekhouding]\label{met:g11:mechanical-energy:bookkeeping}
\begin{enumerate}
  \item Kies een \omterm{def:g11:mechanical-energy:potential}{referentielvlak} en houd het vast.
  \item Schrijf $E_m$ bij begin en einde.
  \item Als alleen gewicht arbeidt: $E_m$ behouden; los $v$ of $z$ op.
  \item Met wrijving: $E_m(B) = E_m(A) + W_f$; lees de tol af.
\end{enumerate}
\end{method}

\section{Probleem: De lus ontwerpen}

\begin{problem}[{Weekendprobleem --- een achtbaanlus veilig maken: de
minimale lanceerhoogte, elke snelheid langs de baan, de wrijvingsrekening
en de rem, allemaal uit $E_m$}]
\label{pb:g11:mechanical-energy:1}
Een achtbaanlus van straal $R = \qty{8.0}{m}$ eist $v \geq \sqrt{gR}$
op de top. Massa van de trein $m = \qty{2000}{kg}$. Lanceerhoogte $H$
boven de bodem van de lus.

\medskip
\noindent\textbf{Deel I --- De veilige top.}
\begin{enumerate}
  \item Minimale $v_{\text{top}}$?
  \item $E_m$ op de top bij die snelheid (referentie: bodem)?
  \item Minimale $H$ zonder wrijving? Toon $H_{\min} = 5R/2$.
  \item Bij $H = \qty{28}{m}$: $v_{\text{top}}$ en verhouding tot minimum?
  \item Waarom hangt de veiligheid niet van $m$ af?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Snelheden langs de baan ($H = \qty{24}{m}$).}
\begin{enumerate}[resume]
  \item $v$ op de bodem?
  \item $v$ op de zijkant ($z = R$)?
  \item $(E_p, E_k)$ bij lancering, bodem, zijkant, top?
  \item $v$ op $z = \qty{12}{m}$?
  \item Waar is ze het snelst? Traagst in de lus?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Wrijving.}
Twee sensoren \qty{60}{m} uit elkaar op dezelfde hoogte lezen
\qty{20.4}{m/s} en \qty{19.2}{m/s}.
\begin{enumerate}[resume]
  \item $\Delta E_m$?
  \item $W(\vect f)$ en $f$?
  \item Waar gaat de energie heen?
  \item Fractie van $E_m$ verloren per \qty{60}{m}? Gevaar bij
        $H_{\min}$?
  \item Met dit verlies vanaf $H = \qty{24}{m}$: is de top nog veilig?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De rem.}
De trein komt op \qty{18.0}{m/s} aan en stopt over \qty{45}{m}.
\begin{enumerate}[resume]
  \item $E_k$ bij aankomst?
  \item Gemiddelde remkracht?
  \item Remvertraging in $g$'s?
  \item Wrijvingsarbeid op de baan plus remarbeid: audit tegen $mgH$?
  \item Punchline: wat fixeerde $H = \qty{24}{m}$, en wat is de prijs
        van wrijving negeren?
\end{enumerate}
\end{problem}
