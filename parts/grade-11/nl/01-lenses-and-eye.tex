\chapter{Lenzen, beelden en het oog}\label{ch:g11:lenses-and-eye}

Elk scherp beeld --- op de sensor van een telefoon, een bioscoopscherm,
of de achterkant van je eigen oog --- komt van dezelfde truc: een
gebogen transparant medium buigt stralen door
\omterm{def:g10:refraction:refraction}{breking} (\cref{ch:g10:refraction})
zodat alle stralen die van één punt vertrekken, opnieuw samenkomen in
een ander. Dit hoofdstuk temt de truc: één
\omterm{def:g11:lenses-and-eye:lens}{lens}, drie stralen, één formule ---
en richt die dan op camera's, brillen, en de levende
\omterm{def:g11:lenses-and-eye:lens}{lens} die deze zin leest.

\section{Convergerende en divergerende lenzen}

\begin{definition}[Dunne lens]\label{def:g11:lenses-and-eye:lens}
Een \emph{lens}\index{lens} is een transparant medium begrensd door
twee oppervlakken, waarvan minstens één gebogen is. Dikker in het
midden dan aan de rand, is ze \emph{convergerend}\index{convergerende
lens}: evenwijdige stralen worden naar elkaar toe gebogen; dikker aan de
rand, \emph{divergerend}\index{divergerende lens}: ze spreiden uiteen.
Een \emph{dunne}\index{dunne lens} lens heeft verwaarloosbare dikte: we
tekenen een segment met pijlpunten naar buiten (convergerend) of naar
binnen (divergerend). Haar middelpunt $O$ is het \emph{optisch
centrum}\index{optisch centrum}; de loodlijn door $O$, de
\emph{hoofdas}\index{hoofdas}.
\end{definition}

\begin{definition}[Brandpunten en brandpuntsafstand]\label{def:g11:lenses-and-eye:focal}
Stralen die evenwijdig aan de as van een
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} aankomen, snijden de
as allemaal in één punt erachter, het \emph{beeldbrandpunt}\index{brandpunt}
$F'$; symmetrisch verlaten stralen die uit het
\emph{voorwerpsbrandpunt} $F$ komen (vóór, met $OF = OF'$) de lens
evenwijdig aan de as. De \emph{brandpuntsafstand}\index{brandpuntsafstand}
$f' = \overline{OF'}$ is positief voor een
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens}; door een
\omterm{def:g11:lenses-and-eye:lens}{divergerende lens} verlaten evenwijdige
stralen alsof ze uit een punt $F'$ \emph{vóór} de
\omterm{def:g11:lenses-and-eye:lens}{lens} komen: $f' < 0$.
\end{definition}

\begin{definition}[Sterkte]\label{def:g11:lenses-and-eye:vergence}
De \emph{sterkte}\index{sterkte} van een
\omterm{def:g11:lenses-and-eye:lens}{lens} met
\omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand} $f'$ (in meters)
is $C = 1/f'$, gemeten in \emph{dioptrie}\index{dioptrie} (symbool D):
$1~\text{D} = \qty{1}{m^{-1}}$;
\omterm{def:g11:lenses-and-eye:lens}{divergerende} lenzen hebben $C < 0$.
Opticiens schrijven voor in dioptrie: de sterktes van
\omterm{def:g11:lenses-and-eye:lens}{dunne} lenzen in contact tellen gewoon
op (hier aangenomen; afgeleid in het volume van bachelorjaar~1).
\end{definition}

\begin{example}[Een voorschrift lezen]\label{ex:g11:lenses-and-eye:vergence}
Een \omterm{def:g11:lenses-and-eye:lens}{lens} met $f' = \qty{0.50}{m}$ heeft
$C = 1/0.50 = +2.0$~D: \omterm{def:g11:lenses-and-eye:lens}{convergerend}.
Een voorschrift van $-4.0$~D betekent $f' = \qty{-0.25}{m}$:
\omterm{def:g11:lenses-and-eye:lens}{divergerend},
\omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand} \qty{25}{cm}.
\end{example}

\section{Constructie van het beeld}

\begin{method}[De drie constructiestralen]\label{met:g11:lenses-and-eye:rays}
Laat $AB$ een voorwerp loodrecht op de as in $A$ zijn. Onder alle stralen
die van de top $B$ vertrekken, hebben er drie een bekend pad:
\begin{enumerate}
  \item de straal door het \omterm{def:g11:lenses-and-eye:lens}{optisch centrum}
        $O$ gaat rechtdoor;
  \item de straal evenwijdig aan de as verlaat de lens door $F'$;
  \item de straal door $F$ verlaat de lens evenwijdig aan de as.
\end{enumerate}
Twee volstaan: hun snijpunt is het beeldpunt $B'$, en het beeld $A'B'$
staat loodrecht op de as in $A'$. Als de uittredende stralen divergeren,
ontmoeten hun \emph{achterwaartse} verlengingen (gestippeld) elkaar in
een \omterm{def:g11:lenses-and-eye:images}{virtueel beeldpunt}.
\end{method}

\begin{definition}[Reële en virtuele beelden]\label{def:g11:lenses-and-eye:images}
Een beeld is \emph{reëel}\index{reëel beeld} wanneer de uittredende
stralen er werkelijk doorheen gaan: een scherm daar vangt het op;
\emph{virtueel}\index{virtueel beeld} wanneer alleen hun achterwaartse
verlengingen samenkomen: geen scherm vangt het, maar een oog dat in de
\omterm{def:g11:lenses-and-eye:lens}{lens} kijkt ziet het prima. Een
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} geeft een reëel,
omgekeerd beeld van elk voorwerp voorbij $F$; schuif het voorwerp binnen
de \omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand} en het beeld
wordt virtueel, rechtop, vergroot.
\end{definition}

\section{Eén formule: de geconjugeerde relatie}

Constructies tonen \emph{waar} het beeld staat; een formule berekent het.
Oriënteer de as langs het licht en gebruik \emph{algebraïsche maten}:
$\overline{OA}$ is de coördinaat van $A$ ten opzichte van $O$ (negatief
voor een \omterm{def:g11:lenses-and-eye:images}{reëel} voorwerp vóór); hoogtes
$\overline{AB}$ tellen positief omhoog.

\begin{theorem}[Geconjugeerde relatie van de dunne lens]\label{thm:g11:lenses-and-eye:conjugate}
Door een \omterm{def:g11:lenses-and-eye:lens}{dunne lens} met
\omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand} $f'$ wordt een
voorwerpspunt $A$ op de as afgebeeld in het punt $A'$ met
\[
\frac{1}{\overline{OA'}} - \frac{1}{\overline{OA}} = \frac{1}{f'} = C .
\]
\end{theorem}
\admitted

\begin{remark}
De relatie volgt uit \omterm{thm:g10:refraction:snell}{de wet van Snellius}
toegepast op stralen dicht bij de as; de afleiding --- en de grenzen van
het \omterm{def:g11:lenses-and-eye:lens}{dunne-lens}-model --- worden eerlijk
gemaakt in het volume van bachelorjaar~1.
\end{remark}

\begin{definition}[Vergroting]\label{def:g11:lenses-and-eye:magnification}
De \emph{vergroting}\index{vergroting} van het beeld is
\[
\gamma = \frac{\overline{A'B'}}{\overline{AB}}
       = \frac{\overline{OA'}}{\overline{OA}} .
\]
$\abs{\gamma}$ vergelijkt groottes, het teken oriëntaties ($\gamma < 0$:
omgekeerd); het teken van $\overline{OA'}$ leest de aard (positief:
\omterm{def:g11:lenses-and-eye:images}{reëel}, achter de
\omterm{def:g11:lenses-and-eye:lens}{lens}; negatief:
\omterm{def:g11:lenses-and-eye:images}{virtueel}, vóór).
\end{definition}

\begin{example}[De formule aan het werk]\label{ex:g11:lenses-and-eye:conjugate}
Een voorwerp staat \qty{30}{cm} vóór een
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} met
$f' = \qty{10}{cm}$: $\overline{OA} = \qty{-30}{cm}$, dus
\[
\frac{1}{\overline{OA'}} = \frac{1}{10} - \frac{1}{30} = \frac{1}{15},
\qquad \overline{OA'} = +\qty{15}{cm},
\qquad \gamma = \frac{15}{-30} = -0.50 :
\]
een \omterm{def:g11:lenses-and-eye:images}{reëel beeld} \qty{15}{cm} achter de
\omterm{def:g11:lenses-and-eye:lens}{lens}, omgekeerd, half zo groot --- precies
wat de drie stralen tekenen.
\end{example}

\section{Het oog}

\begin{definition}[Het oog als optisch stelsel]\label{def:g11:lenses-and-eye:eye}
Optisch is het oog een \omterm{def:g11:lenses-and-eye:lens}{convergerende lens}
tegenover een scherm: het \emph{hoornvlies}\index{hoornvlies} en de
\emph{ooglens}\index{ooglens} werken als één
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} met regelbare
\omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand}; het
\emph{netvlies}\index{netvlies}, ongeveer \qty{17}{mm} erachter, is het
vaste scherm. Scherpstellen gebeurt door de
\omterm{def:g11:lenses-and-eye:lens}{lens} samen te knijpen, niet door ze te
verplaatsen zoals in een camera: \emph{accommodatie}\index{accommodatie}
is de verandering van $f'$ door de ciliaire spieren. Het dichtstbijzijnde
punt dat scherp gezien wordt bij volle accommodatie is het
\emph{nabijheidspunt}\index{nabijheidspunt} --- ongeveer \qty{25}{cm}
voor het \emph{standaardoog}\index{standaardoog} --- het verst, in rust,
het \emph{verheidspunt}\index{verheidspunt} (oneindig voor het
standaardoog).
\end{definition}

\begin{example}[De sterkte van je oog]\label{ex:g11:lenses-and-eye:eyerange}
Het \omterm{def:g11:lenses-and-eye:eye}{netvlies} zit op
$\overline{OA'} = +\qty{0.017}{m}$. Ver voorwerp
($1/\overline{OA} \approx 0$): $C = 1/0.017 \approx 59$~D.
\omterm{def:g11:lenses-and-eye:eye}{Nabijheidspunt}
($\overline{OA} = \qty{-0.25}{m}$): $C = 1/0.017 + 1/0.25 \approx
63$~D. Lezen kost ongeveer $4$~D: het oog is een $59$~D
\omterm{def:g11:lenses-and-eye:lens}{lens} met een ingebouwde fijnregeling van
$+4$~D.
\end{example}

\section{Afwijkingen, correcties en de loep}

\begin{definition}[Bijziendheid en verziendheid]\label{def:g11:lenses-and-eye:defects}
Een \emph{bijziend}\index{bijziendheid} (myoop) oog is te
\omterm{def:g11:lenses-and-eye:lens}{convergerend} voor zijn lengte: evenwijdige
stralen focussen \emph{vóór} het \omterm{def:g11:lenses-and-eye:eye}{netvlies},
verre voorwerpen zijn wazig, het \omterm{def:g11:lenses-and-eye:eye}{verheidspunt}
ligt op eindige afstand; een
\omterm{def:g11:lenses-and-eye:lens}{divergerende} bril-\omterm{def:g11:lenses-and-eye:lens}{lens}
corrigeert het. Een \emph{verziend}\index{verziendheid} (hypermetroop)
oog is niet \omterm{def:g11:lenses-and-eye:lens}{convergerend} genoeg: nabije
voorwerpen zouden achter het \omterm{def:g11:lenses-and-eye:eye}{netvlies}
focussen, het \omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} wijkt terug;
een \omterm{def:g11:lenses-and-eye:lens}{convergerende lens} corrigeert het.
Veroudering verstijft de \omterm{def:g11:lenses-and-eye:eye}{ooglens}
(\emph{ouderdomsverziendheid}\index{ouderdomsverziendheid}), wat de
\omterm{def:g11:lenses-and-eye:eye}{accommodatie} verkleint --- zelfde remedie:
leesbril.
\end{definition}

\begin{example}[Voorschrijven voor een bijziende]\label{ex:g11:lenses-and-eye:correction}
Een \omterm{def:g11:lenses-and-eye:defects}{bijziend} oog heeft zijn
\omterm{def:g11:lenses-and-eye:eye}{verheidspunt} op \qty{50}{cm}: de
corrigerende \omterm{def:g11:lenses-and-eye:lens}{lens} moet verre voorwerpen
\emph{op het \omterm{def:g11:lenses-and-eye:eye}{verheidspunt}} tonen --- voorwerp
in oneindig, beeld op $\overline{OA'} = \qty{-0.50}{m}$
(\omterm{def:g11:lenses-and-eye:images}{virtueel}, vóór). Dan
$1/f' = 1/\overline{OA'}$: $f' = \qty{-0.50}{m}$, $C = -2.0$~D; het
ontspannen oog kijkt naar dat
\omterm{def:g11:lenses-and-eye:images}{virtueel beeld} en ziet het scherp.
\end{example}

\begin{definition}[De loep]\label{def:g11:lenses-and-eye:magnifier}
Een \emph{loep}\index{loep} is een
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} met korte
\omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand}, gebruikt met het
voorwerp \emph{binnen} de \omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand}:
het beeld is \omterm{def:g11:lenses-and-eye:images}{virtueel}, rechtop, vergroot
(\cref{def:g11:lenses-and-eye:images}), en het oog bekijkt het comfortabel
ver weg.
\end{definition}

\begin{example}[De juweliersloep]\label{ex:g11:lenses-and-eye:magnifier}
Loep met $f' = \qty{5.0}{cm}$, steen op \qty{4.0}{cm}:
$1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$, dus
$\overline{OA'} = \qty{-20}{cm}$ en $\gamma = (-20)/(-4.0) = +5.0$: een
\omterm{def:g11:lenses-and-eye:images}{virtueel}, rechtop beeld, vijf keer
groter, een comfortabele \qty{20}{cm} van de
\omterm{def:g11:lenses-and-eye:lens}{lens}.
\end{example}

\begin{exercise}[$\star\star$]\label{exo:g11:lenses-and-eye:9}

\begin{exercise}[$\star\star$]\label{exo:g11:lenses-and-eye:11}

\begin{exercise}[$\star\star\star$]\label{exo:g11:lenses-and-eye:12}

        scherpgesteld op oneindig, daarna op het standaard

        \omterm{def:g11:lenses-and-eye:eye}{accommodatie} van het

  \item Een ouder oog heeft het
        \omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} op \qty{1.0}{m}.
        Bereken zijn \omterm{def:g10:signals-and-waves:amplitude}{amplitude} en

\begin{exercise}[$\star\star\star$]\label{exo:g11:lenses-and-eye:14}

\begin{enumerate}
  \item Bereken $f'$.
  \item Waar moet een voorwerp staan opdat zijn
        \omterm{def:g11:lenses-and-eye:images}{virtueel beeld}
        \qty{25}{cm} vóór de \omterm{def:g11:lenses-and-eye:lens}{lens} ligt?
        Bereken $\gamma$; vergelijk met de geadverteerde $\times 3$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:lenses-and-eye:15}
Een \omterm{def:g11:lenses-and-eye:images}{reëel} voorwerp staat op afstand $d > 0$ vóór
een \omterm{def:g11:lenses-and-eye:lens}{convergerende lens}
($\overline{OA} = -d$, $f' > 0$). Toon dat
$\overline{OA'} = \frac{f'd}{d - f'}$ en
$\gamma = \frac{f'}{f' - d}$. Leid af: \omterm{def:g11:lenses-and-eye:images}{reëel}
en omgekeerd wanneer $d > f'$;
\omterm{def:g11:lenses-and-eye:images}{virtueel}, rechtop, \emph{vergroot}
($\gamma > 1$) wanneer $0 < d < f'$ --- elke
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} is een
\omterm{def:g11:lenses-and-eye:magnifier}{loep}, dicht genoeg gebruikt.
\end{exercise}

\section{Probleem: Eén formule, drie instrumenten}

\begin{problem}[{Weekendprobleem --- dezelfde geconjugeerde relatie
stelt een camera scherp, schrijft een leesbril voor, ontwerpt een loep,
en meet de zoom ingebouwd in je eigen oog}]
\label{pb:g11:lenses-and-eye:1}
Een fotograaf verschuift een \qty{50}{mm}
\omterm{def:g11:lenses-and-eye:lens}{lens}, een grootmoeder houdt de krant op
armlengte, een juwelier buigt over een diamant: één regel algebra,
$1/\overline{OA'} - 1/\overline{OA} = 1/f'$
(\cref{thm:g11:lenses-and-eye:conjugate}), bedient alle drie de ambachten
--- en daarna het oog zelf.

\medskip
\noindent\textbf{Deel I --- Eén formule.} Een
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens} heeft
$f' = \qty{20}{cm}$.
\begin{enumerate}
  \item Een voorwerp op \qty{60}{cm}: bereken $\overline{OA'}$ en
        $\gamma$; beschrijf het beeld.
  \item Verplaats het voorwerp naar \qty{30}{cm}; herbereken. Wat is
        omgewisseld met vraag~1? (Lichtpaden zijn omkeerbaar.)
  \item Verplaats het naar \qty{10}{cm}, binnen $f'$: herbereken en
        beschrijf.
  \item Samenvatting: terwijl het voorwerp van heel ver naar $F$
        schuift, en dan voorbij, hoe beweegt en verandert het beeld
        van aard?
  \item Toon dat een zeer ver voorwerp ($1/\overline{OA} \approx 0$)
        altijd in het brandvlak afbeeldt: $\overline{OA'} = f'$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De fotograaf.}
Een camera-\omterm{def:g11:lenses-and-eye:lens}{lens} heeft $f' = \qty{50}{mm}$; de
sensor, \qty{24}{mm} hoog, kan ten opzichte van de
\omterm{def:g11:lenses-and-eye:lens}{lens} worden verplaatst.
\begin{enumerate}[resume]
  \item Waar staat de sensor bij scherpstelling op een ver landschap?
  \item Nu een gezicht \qty{1.0}{m} ver: bereken de nieuwe afstand
        lens--sensor en de verplaatsing vanaf de landschapsstand.
  \item Het mechanisme laat hoogstens \qty{55}{mm} afstand
        lens--sensor toe. Bereken de minimale scherpstelafstand.
  \item Bij die dichtste scherpstelling, bereken $\gamma$. Past het
        beeld van een \qty{20}{cm} gezicht op de sensor?
  \item Ware-grootte-``macro'' betekent $\gamma = -1$: waar moeten
        voorwerp en sensor dan staan, en waarom eist dit een speciale
        \omterm{def:g11:lenses-and-eye:lens}{lens}?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De grootmoeder.}
Met de leeftijd wijkt het \omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt}
terug: dat van grootmoeder ligt op \qty{1.0}{m}.
\begin{enumerate}[resume]
  \item Verklaar de krant op armlengte.
  \item Om te lezen op \qty{25}{cm} moet haar bril een bladzijde op
        \qty{25}{cm} omzetten in een
        \omterm{def:g11:lenses-and-eye:images}{virtueel beeld} op \qty{1.0}{m}:
        bereken de vereiste \omterm{def:g11:lenses-and-eye:vergence}{sterkte}.
  \item Bereken $\gamma$. Het beeld is vier keer groter, maar lijkt
        niet groter: verklaar, door groottes \emph{en} afstanden te
        vergelijken.
  \item Toon dat drukwerk alleen scherp is tussen \qty{25}{cm} en
        $f' \approx \qty{33}{cm}$. Wat met een \emph{ver} voorwerp
        door de bril gezien?
  \item Leid in één zin af waarom bifocale lenzen bestaan.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De juwelier, en de zoom van het oog zelf.}
De loep van de juwelier heeft $f' = \qty{5.0}{cm}$; zijn
\omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} ligt op \qty{25}{cm}; zijn
\omterm{def:g11:lenses-and-eye:eye}{netvlies} is \qty{17}{mm} achter de
\omterm{def:g11:lenses-and-eye:lens}{lens} van zijn oog.
\begin{enumerate}[resume]
  \item Een steen \qty{4.0}{cm} onder de loep: positie, aard,
        \omterm{def:g11:lenses-and-eye:magnification}{vergroting}?
  \item Waar moet de steen staan opdat zijn beeld op zijn
        \omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} valt, en wat is
        $\gamma$ dan?
  \item Waarom plaatsen ervaren juweliers de steen precies op $F$
        (beeld in oneindig)?
  \item Nu het oog alleen: bereken zijn
        \omterm{def:g11:lenses-and-eye:vergence}{sterkte} bij het bekijken van
        een ver voorwerp, daarna één op \qty{25}{cm}: hoeveel
        \omterm{def:g11:lenses-and-eye:vergence}{dioptrie} aan
        \omterm{def:g11:lenses-and-eye:eye}{accommodatie}?
  \item Zet beide \omterm{def:g11:lenses-and-eye:vergence}{sterktes} om in
        \omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstanden}: met
        hoeveel, in millimeters en procent, verandert het oog zijn
        eigen \omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand} --- de
        zoom die de camera met verplaatsing deed en grootmoeder met
        $+3$~D?
\end{enumerate}
\end{problem}
