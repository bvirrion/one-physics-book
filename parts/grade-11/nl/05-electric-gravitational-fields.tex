\chapter{Elektrische en gravitatievelden}\label{ch:g11:electric-gravitational-fields}

Hoe trekt de Aarde een appel die ze nooit aanraakt? Newton zelf noemde
werking op afstand ``zo'n grote absurditeit'' dat hij weigerde ze te
verdedigen. Het moderne antwoord: een massa, of een lading, wijzigt de
ruimte eromheen --- ze creëert een
\emph{\omterm{def:g11:electric-gravitational-fields:field}{veld}} --- en wat
daar geplaatst wordt, reageert op het
\omterm{def:g11:electric-gravitational-fields:field}{veld} op zijn eigen
plaats. We bouwen de twee grote voorbeelden naast elkaar, tot de
analogie een identiteit van vorm wordt.

\section{Van werking op afstand naar het veld}

\begin{definition}[Veld]\label{def:g11:electric-gravitational-fields:field}
Een \emph{veld}\index{veld} wijst aan elk punt van de ruimte een vector
toe. Een bron (een lading, een massa) creëert haar veld overal eromheen,
of er iets is om het te voelen of niet; een testobject voelt dan een
\omterm{def:g10:inertia:force}{kracht} bepaald door het veld \emph{op zijn
eigen plaats} alleen. Dit tweestappenbeeld vervangt werking op afstand.
\end{definition}

\begin{remark}
Het \omterm{def:g11:electric-gravitational-fields:field}{veld} is geen
boekhoudtruc: het draagt \omterm{def:g10:energy-conservation:energy}{energie},
en veranderingen in de bron reizen erdoorheen met eindige snelheid ---
licht zelf is zo'n reizend
\omterm{def:g11:electric-gravitational-fields:field}{veld}
(\cref{ch:g12:light-as-wave}). Hier bestuderen we
\omterm{def:g11:electric-gravitational-fields:field}{velden} die niet in de
tijd veranderen.
\end{remark}

\section{Het elektrisch veld}

\begin{definition}[Elektrisch veld]\label{def:g11:electric-gravitational-fields:efield}
Als een kleine testlading $q$ geplaatst in een punt $M$ een elektrische
\omterm{def:g10:inertia:force}{kracht} $\vect F$ voelt, is het
\emph{elektrisch veld}\index{elektrisch veld} in $M$
\[
\vect E = \frac{\vect F}{q}
\qquad \text{(in \unit{N/C}, ook geschreven \unit{V/m})}.
\]
$\vect E$ hangt alleen af van de bronnen en van $M$, niet van de
testlading: $q$ verdubbelen verdubbelt $\vect F$, en laat het quotiënt
onveranderd.
\end{definition}

\begin{proposition}[Kracht op een lading]\label{prop:g11:electric-gravitational-fields:force}
Een lading $q$ geplaatst waar het
\omterm{def:g11:electric-gravitational-fields:field}{veld} $\vect E$ is,
voelt de \omterm{def:g10:inertia:force}{kracht} $\vect F = q \vect E$: langs
$\vect E$ als $q > 0$, tegengesteld aan $\vect E$ als $q < 0$, van
grootte $F = \abs{q}\,E$.
\end{proposition}

\begin{proof}
Herschik de definitie; de tekenregel is de algebra van een negatief
veelvoud.
\end{proof}

\begin{example}[Een veld lezen]\label{ex:g11:electric-gravitational-fields:reading}
Een testlading $q = \qty{+2.0e-8}{C}$ in $M$ voelt een
\omterm{def:g10:inertia:force}{kracht} van \qty{6.0e-4}{N} naar het oosten. Het
\omterm{def:g11:electric-gravitational-fields:field}{veld} in $M$ is
$E = \num{6.0e-4} / \num{2.0e-8} = \qty{3.0e4}{N/C}$, naar het oosten;
$q' = \qty{-1.0e-8}{C}$ geplaatst in hetzelfde punt voelt
$F = \num{1.0e-8} \times \num{3.0e4} = \qty{3.0e-4}{N}$ --- naar het
westen.
\end{example}

\begin{proposition}[Veld van een puntlading]\label{prop:g11:electric-gravitational-fields:point}
Een puntlading $Q$ creëert, op afstand $d$, een
\omterm{def:g11:electric-gravitational-fields:field}{veld} langs de lijn die
lading en punt verbindt --- weg van $Q$ als $Q > 0$, ernaartoe als
$Q < 0$ --- van grootte
\[
E = k\,\frac{\abs{Q}}{d^2},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{proposition}

\begin{proof}
De wet van Coulomb (\cref{ch:g11:fundamental-interactions}) geeft de
\omterm{def:g10:inertia:force}{kracht} op een testlading $q$ op afstand $d$:
grootte $k \abs{qQ} / d^2$, langs de verbindingslijn, afstotend bij
gelijke tekens. Deel door $q$ en pas
\cref{prop:g11:electric-gravitational-fields:force} toe.
\end{proof}

\begin{example}[Een geladen bol]\label{ex:g11:electric-gravitational-fields:sphere}
Een Van de Graaff-bol draagt $Q = \qty{+5.0e-7}{C}$ (een kleine geladen
bol werkt als een puntlading in haar middelpunt). Op $d = \qty{0.50}{m}$:
\[
E = \num{8.99e9} \times \num{5.0e-7} / 0.50^2 \approx \qty{1.8e4}{N/C},
\]
radiaal van de bol af. Op \qty{1.0}{m} is ze gedaald tot een kwart: de
$1/d^2$ van de \omterm{def:g10:inertia:force}{kracht} overleeft in het
\omterm{def:g11:electric-gravitational-fields:field}{veld}.
\end{example}

\begin{definition}[Veldlijnen]\label{def:g11:electric-gravitational-fields:lines}
Een \emph{veldlijn}\index{veldlijn} is een kromme overal raak aan het
\omterm{def:g11:electric-gravitational-fields:field}{veld}, met een pijl die
de richting van het
\omterm{def:g11:electric-gravitational-fields:field}{veld} geeft. Drie
regels:
\begin{itemize}
  \item ze vertrekken van positieve ladingen en komen aan bij negatieve;
  \item waar lijnen opeenpakken is het
        \omterm{def:g11:electric-gravitational-fields:field}{veld} sterk; waar
        ze spreiden, zwak;
  \item twee lijnen kruisen nooit: het
        \omterm{def:g11:electric-gravitational-fields:field}{veld} heeft één
        richting in elk punt.
\end{itemize}
\end{definition}

\section{Het homogeen veld van een condensator}

\begin{definition}[Parallelle-platencondensator]\label{def:g11:electric-gravitational-fields:capacitor}
Twee tegenoverstaande parallelle metalen platen, dicht bijeen en met
tegengestelde ladingen, vormen een \emph{condensator}\index{condensator}.
Tussen de platen is het
\omterm{def:g11:electric-gravitational-fields:field}{veld}
\emph{homogeen}\index{homogeen veld}: overal dezelfde grootte en
richting --- loodrecht op de platen, van de positieve plaat naar de
negatieve.
\end{definition}

\begin{proposition}[Veld tussen de platen]\label{prop:g11:electric-gravitational-fields:uniform}
Als de \omterm{def:g11:circuits-and-power:voltage}{spanning} tussen de platen
(\cref{ch:g11:circuits-and-power}) $U$ is en hun scheiding $d$, heeft het
\omterm{def:g11:electric-gravitational-fields:field}{veld} ertussen grootte
\[
E = \frac{U}{d} \qquad \text{(in \unit{V/m})}.
\]
\end{proposition}
\admitted

\begin{remark}
Homogeniteit en $E = U/d$ worden eerlijk afgeleid in het volume van
bachelorjaar~1; de formule verklaart de
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} \unit{V/m}.
\omterm{def:g11:lenses-and-eye:lens}{Dunne} spleten maken sterke
\omterm{def:g11:electric-gravitational-fields:field}{velden}: \qty{100}{V} over
een millimeter is al \qty{e5}{V/m}.
\end{remark}

\begin{example}[Een elektronenbundel sturen]\label{ex:g11:electric-gravitational-fields:beam}
In een \omterm{def:g10:signals-and-waves:oscilloscope}{oscilloscoop} gaat de
bundel tussen platen met $U = \qty{100}{V}$ en $d = \qty{5.0}{mm}$:
$E = 100 / \num{5.0e-3} = \qty{2.0e4}{V/m}$, en elk elektron voelt
$F = \num{1.60e-19} \times \num{2.0e4} = \qty{3.2e-15}{N}$ naar de
positieve plaat --- $\num{4e14}$ keer zijn
\omterm{def:g10:universal-gravitation:weight}{gewicht} van \qty{8.9e-30}{N},
dus de bundel buigt zichtbaar af (het gekromde pad is de zaak van
\cref{ch:g12:newtons-laws}).
\end{example}

\section{Het gravitatieveld}

\begin{definition}[Gravitatieveld]\label{def:g11:electric-gravitational-fields:gfield}
Als een testmassa $m$ in een punt een
\omterm{def:g10:universal-gravitation:force}{gravitationele kracht} (een
\omterm{def:g10:universal-gravitation:weight}{gewicht}) $\vect P$ voelt, is
het \emph{gravitatieveld}\index{gravitatieveld} daar
\[
\vect g = \frac{\vect P}{m} \qquad \text{(in \unit{N/kg})},
\]
en omgekeerd $\vect P = m \vect g$. Een eerder hoofdstuk introduceerde
het getal $g$; $\vect g$ voegt de richting toe waarin een losgelaten
steen begint te bewegen.
\end{definition}

\begin{proposition}[Veld van de Aarde]\label{prop:g11:electric-gravitational-fields:earth}
Op afstand $d$ van het middelpunt van de Aarde (massa $M$) wijst het
\omterm{def:g11:electric-gravitational-fields:gfield}{gravitatieveld} naar
het middelpunt en heeft grootte
\[
g = G\,\frac{M}{d^2},
\qquad G = \qty{6.67e-11}{N.m^2/kg^2}.
\]
\end{proposition}

\begin{proof}
Universele \omterm{def:g11:fundamental-interactions:four}{gravitatie}
(\cref{ch:g10:universal-gravitation}) geeft de
\omterm{def:g10:inertia:force}{kracht} op een testmassa $m$: grootte
$G M m / d^2$, naar het middelpunt; deel door $m$. Aan het oppervlak is
$d = R$ en dit is de $g = GM/R^2 \approx \qty{9.81}{N/kg}$ van het
eerdere hoofdstuk.
\end{proof}

\begin{example}[Het veld waar de Maan leeft]\label{ex:g11:electric-gravitational-fields:moon}
Op de afstand van de Maan $d = \qty{3.84e8}{m}$:
\[
g = \num{6.67e-11} \times \num{5.97e24} / (\num{3.84e8})^2
\approx \qty{2.7e-3}{N/kg}.
\]
Het \omterm{def:g11:electric-gravitational-fields:field}{veld} van de Aarde
stopt nooit --- het \omterm{def:g11:lenses-and-eye:lens}{dunner}t alleen uit als
$1/d^2$; dit is het
\omterm{def:g11:electric-gravitational-fields:field}{veld} dat de Maan op
haar \omterm{def:g10:universal-gravitation:satellite}{baan} houdt.
\end{example}

\begin{remark}[Homogeen nabij het oppervlak]\label{rem:g11:electric-gravitational-fields:uniformg}
Over een laboratorium of zelfs een berg verandert $d$ nauwelijks
vergeleken met $R = \qty{6.37e6}{m}$, dus $\vect g$ is
\omterm{def:g11:electric-gravitational-fields:capacitor}{homogeen} tot hoge
nauwkeurigheid: verticale lijnen, constant \qty{9.81}{N/kg} --- het
\omterm{def:g11:electric-gravitational-fields:field}{veld} van de Aarde lijkt
op dat van de
\omterm{def:g11:electric-gravitational-fields:capacitor}{condensator}, net
zoals de ronde Aarde plat lijkt vanuit een tuin.
\end{remark}

\section{Eén idee, twee krachten}

De twee \omterm{def:g11:electric-gravitational-fields:field}{velden} van dit
hoofdstuk zijn dezelfde wiskunde in twee kostuums:

\begin{center}
\begin{tabular}{lcc}
 & \textbf{elektrisch} & \textbf{gravitatie} \\
\hline
bron & lading $Q$ & massa $M$ \\
\omterm{def:g11:electric-gravitational-fields:field}{veld} & $\vect E = \vect F / q$ & $\vect g = \vect P / m$ \\
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} & \unit{N/C} = \unit{V/m} & \unit{N/kg} \\
puntbron & $E = k \abs{Q} / d^2$ & $g = G M / d^2$ \\
constante & $k = \qty{8.99e9}{N.m^2/C^2}$ & $G = \qty{6.67e-11}{N.m^2/kg^2}$ \\
\omterm{def:g10:inertia:force}{kracht} op een test & $\vect F = q \vect E$ & $\vect P = m \vect g$ \\
\omterm{def:g11:electric-gravitational-fields:lines}{veldlijnen} & uit $+$, in $-$ & altijd in de massa \\
\end{tabular}
\end{center}

Elke formule links wordt die rechts onder het woordenboek
$q \leftrightarrow m$, $\vect E \leftrightarrow \vect g$,
$k \leftrightarrow G$.

\begin{remark}[Waar de analogie breekt]\label{rem:g11:electric-gravitational-fields:noanalogue}
Lading komt in twee tekens; massa in één. Dus
\omterm{def:g11:electric-gravitational-fields:efield}{elektrische velden}
kunnen opgeheven worden: de herschikte ladingen van een metalen doos
wissen elk uitwendig
\omterm{def:g11:electric-gravitational-fields:field}{veld} erin uit (een
\emph{kooi van Faraday} --- waarom een lift je telefoon-\omterm{def:g10:signals-and-waves:signal}{signaal}
doodt), en bulk-materie, met gebalanceerde $+$ en $-$, is elektrisch
stil. Zwaartekracht kan noch afgeschermd noch geneutraliseerd worden:
elke \omterm{def:g10:orders-of-magnitude:unit}{kilogram} voegt haar trek toe
--- daarom bestuurt de \omterm{def:g10:inertia:force}{kracht} die $\num{e39}$
zwakker is (\cref{exo:g11:electric-gravitational-fields:10}) het heelal op
grote schalen.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:electric-gravitational-fields:1}
Een testlading $q = \qty{+2.0e-6}{C}$ in een punt $M$ voelt een
\omterm{def:g10:inertia:force}{kracht} van \qty{5.0e-3}{N} naar het noorden.
Geef het \omterm{def:g11:electric-gravitational-fields:field}{veld} in $M$
(grootte en richting), daarna de
\omterm{def:g10:inertia:force}{kracht} op $q' = \qty{-4.0e-6}{C}$ geplaatst
in $M$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:electric-gravitational-fields:2}
Een puntlading $Q = \qty{+5.0}{nC}$ zit in $O$.
\begin{enumerate}
  \item Bereken het \omterm{def:g11:electric-gravitational-fields:field}{veld}
        op $d = \qty{30}{cm}$: grootte, richting.
  \item Op welke afstand is het
        \omterm{def:g11:electric-gravitational-fields:field}{veld} half zo
        sterk?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:electric-gravitational-fields:3}
Een \omterm{def:g11:electric-gravitational-fields:capacitor}{condensator} heeft
$U = \qty{600}{V}$ over platen \qty{3.0}{mm} uit elkaar. Bereken het
\omterm{def:g11:electric-gravitational-fields:field}{veld}, daarna de
\omterm{def:g10:inertia:force}{kracht} op één elektron daar
($e = \qty{1.60e-19}{C}$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:electric-gravitational-fields:4}
Op Mars weegt een \qty{250}{kg}-sonde \qty{930}{N}. Bereken het Martiaanse
oppervlakte-\omterm{def:g11:electric-gravitational-fields:field}{veld}
$g_{\text{Mars}}$, daarna het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van een
\qty{80}{kg}-astronaut daar.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:electric-gravitational-fields:5}
Waar of onwaar, met één reden elk: (a) twee
\omterm{def:g11:electric-gravitational-fields:lines}{veldlijnen} mogen
kruisen waar het
\omterm{def:g11:electric-gravitational-fields:field}{veld} sterk is; (b)
\omterm{def:g11:electric-gravitational-fields:lines}{veldlijnen} vertrekken
van negatieve ladingen en komen aan bij positieve; (c) opeengepakte
\omterm{def:g11:electric-gravitational-fields:lines}{veldlijnen}
\omterm{def:g10:signals-and-waves:signal}{signaleren} een sterk
\omterm{def:g11:electric-gravitational-fields:field}{veld}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:electric-gravitational-fields:6}
Het International Space Station
\omterm{def:g10:universal-gravitation:satellite}{draait} op hoogte
\qty{420}{km} ($R = \qty{6.37e6}{m}$, $M = \qty{5.97e24}{kg}$).
\begin{enumerate}
  \item Bereken $g$ bij het station; welke fractie van de
        oppervlaktewaarde?
  \item Astronauten zweven. Verenig dit met je antwoord in één zin.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:electric-gravitational-fields:7}
Welke \omterm{def:g11:electric-gravitational-fields:field}{veldgrootte} zou
het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een elektron
balanceren ($m_e = \qty{9.11e-31}{kg}$)? Vergelijk met de
\qty{2.0e4}{V/m} van \cref{ex:g11:electric-gravitational-fields:beam};
concludeer over de plaats van zwaartekracht in elektronenfysica.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:electric-gravitational-fields:8}
Een stofkorrel van massa \qty{2.0e-6}{kg} zweeft in rust tussen
horizontale platen \qty{2.0}{cm} uit elkaar met $U = \qty{5.0}{kV}$
ertover.
\begin{enumerate}
  \item Bereken het
        \omterm{def:g11:electric-gravitational-fields:field}{veld}, daarna de
        grootte van de lading van de korrel.
  \item De lading van de korrel is positief: welke plaat is de
        positieve?
  \item Hoeveel
        \omterm{def:g11:fundamental-interactions:elementary}{elementaire
        ladingen} draagt de korrel?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:electric-gravitational-fields:9}
Ladingen $+Q$ en $+4Q$ zitten \qty{30}{cm} uit elkaar. Waar op het
segment dat hen verbindt is het totale
\omterm{def:g11:electric-gravitational-fields:field}{veld} nul? Verklaar
eerst waarom het punt tussen de ladingen moet liggen, dichter bij de
kleinere.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:electric-gravitational-fields:10}
In een waterstof-\omterm{def:g11:fundamental-interactions:ladder}{atoom} zitten
het proton ($m_p = \qty{1.67e-27}{kg}$, lading $+e$) en het elektron
($m_e = \qty{9.11e-31}{kg}$, lading $-e$) $d = \qty{5.3e-11}{m}$ uit
elkaar. Bereken de elektrische en de
\omterm{def:g10:universal-gravitation:force}{gravitationele krachten}
ertussen, en hun verhouding.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:electric-gravitational-fields:11}
In een punt $M$ zou één bron alleen een
\omterm{def:g11:electric-gravitational-fields:field}{veld} van \qty{300}{N/C}

\section{Probleem: De druppel van Millikan}

\begin{problem}[{Weekendprobleem --- de lading van het elektron wegen: een
oliedruppel geparkeerd in de lucht tussen twee platen onthult dat lading
in ondeelbare stappen van $\num{1.60e-19}$
\omterm{def:g11:fundamental-interactions:charge}{coulomb} komt}]
\label{pb:g11:electric-gravitational-fields:1}
In 1909 spoot Robert Millikan oliedruppels tussen de horizontale platen
van een \omterm{def:g11:electric-gravitational-fields:capacitor}{condensator}.
Door de \omterm{def:g11:circuits-and-power:voltage}{spanning} te stemmen tot
een gekozen druppel bewegingloos hing --- elektrische
\omterm{def:g10:inertia:force}{kracht} precies in balans met
\omterm{def:g10:universal-gravitation:weight}{gewicht} --- kon hij haar
lading wegen. Onze druppel heeft massa $m = \qty{4.90e-15}{kg}$
(gemeten uit haar trage val met het
\omterm{def:g11:electric-gravitational-fields:field}{veld} uit; we nemen ze
als gegeven); de platen staan $d = \qty{8.0}{mm}$ uit elkaar, de
\emph{bovenste} positief; de lading van de druppel is negatief.

\medskip
\noindent\textbf{Deel I --- Het podium.}
\begin{enumerate}
  \item Beschrijf het
        \omterm{def:g11:electric-gravitational-fields:field}{veld} tussen de
        platen; wat levert ``\omterm{def:g11:electric-gravitational-fields:capacitor}{homogeen}''?
  \item Voor $U = \qty{800}{V}$, bereken $E$.
  \item In welke richting is de elektrische
        \omterm{def:g10:inertia:force}{kracht} op de druppel? Waarom moest de
        \emph{bovenste} plaat de positieve zijn?
  \item Bereken het
        \omterm{def:g10:universal-gravitation:weight}{gewicht} van de
        druppel.
  \item Schrijf de rustvoorwaarde als een gelijkheid van twee
        \omterm{def:g10:inertia:force}{kracht}groottes.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De balans.}
\begin{enumerate}[resume]
  \item Leid eruit $\abs{q}$ af in termen van $m$, $g$, $d$ en $U$.
  \item De druppel hangt in rust bij $U = \qty{800}{V}$: bereken
        $\abs{q}$.
  \item Deel door $e = \qty{1.60e-19}{C}$. Hoeveel overtollige
        elektronen draagt de druppel?
  \item De \omterm{def:g11:circuits-and-power:voltage}{spanning} wordt
        opgeschoven naar \qty{900}{V}. In welke richting wijst de netto
        \omterm{def:g10:inertia:force}{kracht} nu?
  \item Toon dat de balancerende
        \omterm{def:g11:circuits-and-power:voltage}{spanning}
        $U = mgd/\abs{q}$ is: hoe groter de lading, hoe kleiner de
        \omterm{def:g11:circuits-and-power:voltage}{spanning}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De treden.}
Een stralingsuitbarsting kan elektronen op of van de druppel slaan; na
elke verandering stemt Millikan de
\omterm{def:g11:circuits-and-power:voltage}{spanning} opnieuw om te balanceren.
\begin{enumerate}[resume]
  \item Na één uitbarsting eist balans $U = \qty{1200}{V}$. Bereken de
        nieuwe $\abs{q}$. Heeft de druppel een elektron gewonnen of
        verloren?
  \item Opeenvolgende balansen worden gevonden bij $U = 1200$, $800$,
        $600$, $480$ en \qty{400}{V}. Bereken de vijf ladingen.
  \item Toon dat alle vijf gehele veelvouden van één hoeveelheid zijn;
        geef haar waarde.
  \item Tussen opeenvolgende balansen, met hoeveel verandert de lading?
        Interpreteer.
  \item Verklaar waarom deze druppel nooit in rust kan hangen bij
        $U = \qty{700}{V}$, hoe geduldig men ook stemt.
  \item Millikan herhaalde dit op honderden druppels: elke gemeten
        lading was een geheel veelvoud van dezelfde waarde. Formuleer
        de conclusie, en noem de gemeten hoeveelheid.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Waarom er geen gravitationele Millikan is.}
\begin{enumerate}[resume]
  \item In het woordenboek van dit hoofdstuk, wat speelt de rollen van
        $q$, $E$ en $qE$ in de balans van Deel~II?
  \item Zou een massa erboven de druppel omhoog kunnen houden? Bereken
        het \omterm{def:g11:electric-gravitational-fields:field}{veld} van een
        \qty{1000}{kg}-loodbol \qty{1.0}{m} boven de druppel, de
        \omterm{def:g10:inertia:force}{kracht} op de druppel, en vergelijk met
        haar \omterm{def:g10:universal-gravitation:weight}{gewicht}.
  \item De balans heeft een opwaartse trek nodig sterk genoeg om de
        hele Aarde te bevechten. Waarom kan een plaat lading dat leveren
        en geen opstelling van massa's?
  \item De \emph{massa} van de druppel verandert ook in stappen (één
        olie-\omterm{def:g11:fundamental-interactions:ladder}{molecule}:
        ongeveer \qty{5e-25}{kg}). Vergelijk de relatieve sprongen in $m$
        en in $\abs{q}$ wanneer één elektron landt, en concludeer:
        waarom ziet deze balans de
        \omterm{def:g11:fundamental-interactions:ladder}{atomen} van
        elektriciteit maar niet de
        \omterm{def:g11:fundamental-interactions:ladder}{moleculen} van olie?
        Punchline: lading is gekwantiseerd, en haar
        \omterm{def:g11:fundamental-interactions:ladder}{atoom} is
        $e = \qty{1.60e-19}{C}$.
\end{enumerate}
\end{problem}
