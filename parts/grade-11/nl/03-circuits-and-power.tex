\chapter{Elektrische circuits en vermogen}\label{ch:g11:circuits-and-power}

Draai een schakelaar: een lamp brandt, omdat een generator vele
kilometers verder lading door een lus van koper door je muur duwt.
Eerdere jaren brachten het circuit in kaart --- stroomsterkte,
\omterm{def:g11:circuits-and-power:voltage}{spanning},
\omterm{prop:g11:circuits-and-power:ohm}{wet van Ohm}. Dit hoofdstuk
opent het opnieuw met het rekenboek van \cref{ch:g10:energy-conservation}:
wat de meter rekent, waarom \omterm{def:g11:lenses-and-eye:lens}{dunne} draden
gloeien, waarom het net op \qty{400}{kV} draait.

\section{Lading in beweging}

\begin{definition}[Stroomsterkte en de coulomb]\label{def:g11:circuits-and-power:current}
Een \emph{elektrische stroom}\index{elektrische stroom} is een geordende
stroom van lading (elektronen, in een metaal). Haar stroomsterkte $I$,
in \emph{ampĂ¨re}\index{ampĂ¨re} (\unit{A}), is de lading die per seconde
door een dwarsdoorsnede van de draad gaat: een constante stroom $I$
gedurende een tijd $t$ transporteert de lading
\[ q = I\,t , \]
in \emph{coulomb}\index{coulomb} (\unit{C}), met $\qty{1}{C} =
\qty{1}{A.s}$. De elementaire lading is $e = \qty{1.6e-19}{C}$. Stroom
wordt gemeten met een ampĂ¨remeter, in serie ingevoegd.
\end{definition}

\begin{example}[Bliksem versus telefoon]\label{ex:g11:circuits-and-power:coulomb}
Een blikseminslag draagt zo'n $\qty{3.0e4}{A}$ gedurende ongeveer
$\qty{1.0e-4}{s}$: $q = 3.0\times10^{4} \times 10^{-4} = \qty{3.0}{C}$.
Je telefoon, op \qty{2.0}{A}, verplaatst dezelfde lading elke $1.5$
seconden.
\end{example}

\begin{definition}[Spanning en de volt]\label{def:g11:circuits-and-power:voltage}
De \emph{spanning}\index{spanning} $U$ tussen twee punten van een circuit,
in \emph{volt}\index{volt} (\unit{V}), is de
\omterm{def:g10:energy-conservation:energy}{energie} uitgewisseld per
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} lading die ertussen reist:
$\qty{1}{V} = \qty{1}{J/C}$, dus een lading $q$ die een toestel onder
spanning $U$ doorkruist, wisselt de
\omterm{def:g10:energy-conservation:energy}{energie} $E = qU$ ermee uit. Een
voltmeter, parallel aangesloten, meet ze.
\end{definition}

\begin{proposition}[Circuitwetten]\label{prop:g11:circuits-and-power:laws}
In een circuit in stationaire toestand:
\begin{enumerate}
  \item \emph{knooppuntenwet:} de stromen die in een knooppunt aankomen
        tellen op tot de stromen die eruit vertrekken;
  \item \emph{seriewet:} \omterm{def:g11:circuits-and-power:voltage}{spanningen}
        langs een pad tellen op: $U_{AC} = U_{AB} + U_{BC}$;
  \item \emph{parallelwet:} twee takken die dezelfde twee punten
        verbinden dragen dezelfde
        \omterm{def:g11:circuits-and-power:voltage}{spanning}.
\end{enumerate}
\end{proposition}

\begin{proof}
In stationaire toestand stapelt lading nergens op, dus wat in een
knooppunt instroomt, stroomt eruit.
\omterm{def:g10:energy-conservation:energy}{Energie} per
\omterm{def:g11:circuits-and-power:current}{coulomb} is additief langs een
pad en hangt alleen af van de eindpunten --- niet van de gekozen tak
ertussen.
\end{proof}

\section{Wet van Ohm en weerstandsnetwerken}

\begin{definition}[Weerstand]\label{def:g11:circuits-and-power:resistance}
De \emph{weerstand}\index{weerstand} van een geleider is de verhouding
$R = U/I$ van de \omterm{def:g11:circuits-and-power:voltage}{spanning} erover tot
de stroom erdoor, gemeten in \emph{ohm}\index{ohm} (\unit{\ohm}):
$\qty{1}{\ohm} = \qty{1}{V/A}$.
\end{definition}

\begin{proposition}[Wet van Ohm]\label{prop:g11:circuits-and-power:ohm}
\index{wet van Ohm}
Voor een metallische geleider op vaste temperatuur is $R$ constant:
\omterm{def:g11:circuits-and-power:voltage}{spanning} en stroom zijn evenredig,
\[ U = R\,I . \]
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:circuits-and-power:ohm-limits}
Een experimentele wet, geen universele: een diode, een gloeidraad van
een verwarmingslamp of je eigen huid gehoorzamen er niet aan. Waarom
metalen dat wel doen, wordt afgeleid in de bachelorvolumes.
\end{remark}

\begin{proposition}[Weerstanden in serie en parallel]\label{prop:g11:circuits-and-power:networks}
\index{equivalente weerstand}
Twee weerstanden in serie zijn equivalent met Ă©Ă©n weerstand
$R_{\text{s}} = R_1 + R_2$; twee parallel, met Ă©Ă©n weerstand
$R_{\text{p}}$ gegeven door
\[
\frac{1}{R_{\text{p}}} = \frac{1}{R_1} + \frac{1}{R_2} ,
\qquad\text{d.w.z.}\quad
R_{\text{p}} = \frac{R_1 R_2}{R_1 + R_2} < \min(R_1, R_2) .
\]
\end{proposition}

\begin{proof}
Serie: dezelfde $I$ doorkruist beide en
\omterm{def:g11:circuits-and-power:voltage}{spanningen} tellen op
(\cref{prop:g11:circuits-and-power:laws}), dus $U = (R_1 + R_2)\,I$.
Parallel: beide staan onder dezelfde $U$ en stromen tellen op, dus
$I = \frac{U}{R_1} + \frac{U}{R_2}$.
\end{proof}

\begin{example}[Equivalente weerstand]\label{ex:g11:circuits-and-power:networks}
$R_1 = \qty{100}{\ohm}$ en $R_2 = \qty{150}{\ohm}$: in serie,
\qty{250}{\ohm}; parallel, $\frac{100 \times 150}{250} =
\qty{60}{\ohm}$ --- minder dan elk van beide: de stroom krijgt een
tweede weg.
\end{example}

\section{Energie en vermogen in een circuit}

\begin{proposition}[Elektrisch vermogen]\label{prop:g11:circuits-and-power:power}
\index{elektrisch vermogen}
Een toestel onder \omterm{def:g11:circuits-and-power:voltage}{spanning} $U$
doorkruist door een stroom $I$ ontvangt
\omterm{def:g10:energy-conservation:energy}{energie} met de snelheid
\[ P = U\,I , \qquad\text{d.w.z.}\quad E = U\,I\,t \ \text{in een tijd } t . \]
\end{proposition}

\begin{proof}
In een tijd $t$ doorkruist de lading $q = It$ het toestel
(\cref{def:g11:circuits-and-power:current}) en elke
\omterm{def:g11:circuits-and-power:current}{coulomb} levert $U$
\omterm{def:g10:energy-conservation:energy}{joule} af
(\cref{def:g11:circuits-and-power:voltage}): $E = qU = UIt$.
\end{proof}

\begin{proposition}[Vermogen in een weerstand]\label{prop:g11:circuits-and-power:ri2}
Een weerstand $R$ die een stroom $I$ voert, dissipeert
\[ P = R\,I^2 = \frac{U^2}{R} . \]
\end{proposition}

\begin{proof}
Substitueer \omterm{prop:g11:circuits-and-power:ohm}{de wet van Ohm} in
$P = UI$: $P = (RI)I = RI^2 = U(U/R) = U^2/R$.
\end{proof}

\begin{example}[Een typeplaatje lezen]\label{ex:g11:circuits-and-power:nameplate}
Een waterkoker gemarkeerd ``\qty{230}{V} -- \qty{2200}{W}'' trekt
$I = P/U = 2200/230 = \qty{9.6}{A}$ en heeft
\omterm{def:g11:circuits-and-power:resistance}{weerstand}
$R = U^2/P = 230^2/2200 = \qty{24}{\ohm}$. Gedurende \qty{150}{s}
gebruikt ze $E = 2200 \times 150 = \qty{3.3e5}{J} \approx
\qty{0.09}{kW.h}$ --- ongeveer twee cent.
\end{example}

\begin{remark}[Joules en kilowatturen]\label{rem:g11:circuits-and-power:kwh}
De meter telt \omterm{def:g10:energy-conservation:kwh}{kilowatturen}
(\cref{def:g10:energy-conservation:kwh}): $\qty{1}{kW.h} =
\qty{3.6e6}{J}$. Joules zijn de
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} van de fysicus;
\omterm{def:g10:energy-conservation:kwh}{kilowatturen} prijzen de
rekeningen.
\end{remark}

\section{Het joule-effect: van toaster tot hoogspanningsmast}

\begin{definition}[Joule-effect]\label{def:g11:circuits-and-power:joule}
De dissipatie $P = RI^2$ die
\omterm{def:g10:energy-conservation:forms}{elektrische energie} in warmte
omzet in elke weerstand is het \emph{joule-effect}\index{joule-effect}.
Verwarmers, waterkokers, toasters en ouderwetse gloeilampen zijn
weerstanden by design; een \emph{zekering}\index{zekering} is een
\omterm{def:g11:lenses-and-eye:lens}{dunne} draad gekalibreerd om te smelten
--- het circuit te openen --- wanneer de stroom haar rating overschrijdt.
\end{definition}

\begin{example}[Waarom masten hoog lopen]\label{ex:g11:circuits-and-power:transmission}
Een dorp heeft $P = \qty{100}{kW}$ nodig door een lijn van
\omterm{def:g11:circuits-and-power:resistance}{weerstand}
$R_\ell = \qty{5.0}{\ohm}$. Geleverd bij $U = \qty{500}{V}$: $I = P/U =
\qty{200}{A}$ en de lijn verspilt $R_\ell I^2 = \qty{200}{kW}$ ---
tweemaal de levering: absurd. Bij $U = \qty{20}{kV}$: $I = \qty{5.0}{A}$
en $R_\ell I^2 = \qty{125}{W}$. Veertig keer minder stroom, $1600$ keer
minder verlies.
\end{example}

$U$ verhogen bij vaste $P = UI$ verkleint $I$, en het verlies daalt met

\end{omfigure}

\begin{remark}[De ijzeren regel van het net]\label{rem:g11:circuits-and-power:grid}
Bij vast geleverd \omterm{def:g10:energy-conservation:power}{vermogen}
$P = UI$ daalt het lijnverlies $R_\ell I^2 = R_\ell P^2/U^2$ met het
\emph{kwadraat} van de transport-\omterm{def:g11:circuits-and-power:voltage}{spanning}.
Vandaar: omhoog transformeren (\qty{400}{kV}) om te reizen, omlaag
(\qty{230}{V}) om mee te leven.
\end{remark}

\begin{remark}[Huishoudelijke veiligheid]\label{rem:g11:circuits-and-power:safety}
Een \qty{230}{V}-lijn gezekerd op \qty{16}{A} levert hoogstens
$230 \times 16 \approx \qty{3.7}{kW}$: twee waterkokers tegelijk smelten
de \omterm{def:g11:circuits-and-power:joule}{zekering} --- by design de
dunste, zwakste schakel, veel goedkoper dan een brand in de muur.
\end{remark}

\section{Echte batterijen: elektromotorische kracht}

\begin{definition}[EMK en inwendige weerstand]\label{def:g11:circuits-and-power:emf}
Een batterij zet chemische om in
\omterm{def:g10:energy-conservation:forms}{elektrische energie}. Haar
\emph{elektromotorische kracht}\index{elektromotorische kracht} (EMK)
$\mathcal{E}$, in \omterm{def:g11:circuits-and-power:voltage}{volt}, is de
\omterm{def:g10:energy-conservation:energy}{energie} die ze aan elke
\omterm{def:g11:circuits-and-power:current}{coulomb} overdraagt die haar
doorkruist; haar \emph{inwendige weerstand}\index{inwendige weerstand}
$r$ verantwoordt het \omterm{def:g11:circuits-and-power:joule}{joule-effect}
binnen haar eigen chemie. Model: een ideale bron $\mathcal{E}$ in serie
met $r$ (krullende $\mathcal{E}$: $E$ is
\omterm{def:g10:energy-conservation:energy}{energie}).
\end{definition}

\begin{proposition}[Klemspanning]\label{prop:g11:circuits-and-power:terminal}
Een batterij die een stroom $I$ levert, toont aan haar klemmen
\[ U = \mathcal{E} - r\,I . \]
\end{proposition}

\begin{proof}
De chemie levert $\mathcal{E}I$, de
\omterm{def:g11:circuits-and-power:emf}{inwendige weerstand} eet $rI^2$, het
circuit ontvangt $UI$:
\omterm{def:g10:energy-conservation:energy}{energiebehoud}
(\cref{ch:g10:energy-conservation}) geeft $UI = \mathcal{E}I - rI^2$.
\end{proof}

\begin{example}[Een batterij onder belasting]\label{ex:g11:circuits-and-power:discharge}
Een batterij ($\mathcal{E} = \qty{9.0}{V}$, $r = \qty{1.0}{\ohm}$) voedt
een lampje van \omterm{def:g11:circuits-and-power:resistance}{weerstand}
$R = \qty{8.0}{\ohm}$. De lus gehoorzaamt $\mathcal{E} = rI + RI$, dus
$I = \mathcal{E}/(R + r) = \qty{1.0}{A}$ en $U = 9.0 - 1.0 \times 1.0 =
\qty{8.0}{V}$: het lampje ontvangt \qty{8.0}{W} terwijl de batterij
\qty{1.0}{W} verspilt aan zichzelf te verwarmen.
\end{example}

\begin{remark}[Ontvangers]\label{rem:g11:circuits-and-power:receivers}
Een motor of een oplaadende batterij draait de omzetting om: ze
absorbeert onder $U = \mathcal{E}' + r'I$, zet $\mathcal{E}'I$ om in
beweging of chemie en verliest $r'I^2$. Bronnen duwen, ontvangers duwen
terug.
\end{remark}

\begin{example}[Rendement van een batterij]\label{ex:g11:circuits-and-power:efficiency}
Het \omterm{def:g10:energy-conservation:efficiency}{rendement}
(\cref{def:g10:energy-conservation:efficiency}) van de ontlading hierboven:
nuttig $UI = \qty{8.0}{W}$ uit $\mathcal{E}I = \qty{9.0}{W}$, dus
$\eta = U/\mathcal{E} = 8/9 \approx 0.89$ --- dalend als de stroom
stijgt: hardwerkende batterijen worden heet.
\end{example}

\section{Oefeningen}

\omterm{def:g11:circuits-and-power:joule}{zekering}? Iemand voegt een
\qty{1.5}{kW}-verwarming toe: toon dat ze doorslaat.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:circuits-and-power:11}

Bereken de stroom en het joule-verlies als het

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:circuits-and-power:12}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:circuits-and-power:13}
Een batterij ($\mathcal{E} = \qty{4.5}{V}$, $r = \qty{1.5}{\ohm}$) voedt
een weerstand $R = \qty{6.0}{\ohm}$.
\begin{enumerate}
  \item Bereken $I$, $U$, het nuttige
        \omterm{def:g10:energy-conservation:power}{vermogen} in $R$, het
        \omterm{def:g10:energy-conservation:power}{vermogen} verloren in $r$,
        en het \omterm{def:g10:energy-conservation:efficiency}{rendement}.
  \item Toon dat het \omterm{def:g10:energy-conservation:power}{vermogen}
        ontvangen door een variabele $R$,
        $P = \mathcal{E}^2 R/(R+r)^2$, maximaal is wanneer $R = r$
        (hint: $(R+r)^2/R = (R-r)^2/R + 4r$), en bereken dat maximum.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:circuits-and-power:14}
Een \qty{2.2}{kW}-waterkoker van
\omterm{def:g10:energy-conservation:efficiency}{rendement} $0.90$ brengt

verwarmen kost \qty{4180}{J} per
\omterm{def:g10:orders-of-magnitude:unit}{kilogram} per graad). Bereken de
vereiste warmte, de getrokken
\omterm{def:g10:energy-conservation:forms}{elektrische energie}, de
verwarmingstijd, en de kostprijs bij $0.25$ euro per \unit{kW.h}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:circuits-and-power:15}

\begin{enumerate}
  \item Zet de capaciteit om in
        \omterm{def:g11:circuits-and-power:current}{coulomb}, en schat de
        opgeslagen \omterm{def:g10:energy-conservation:energy}{energie}
        ($E \approx \mathcal{E} q$) in
        \omterm{def:g10:energy-conservation:energy}{joule} en
        \omterm{def:g10:energy-conservation:kwh}{kilowattuur}.
  \item De starter trekt \qty{150}{A}: bereken de klem-\omterm{def:g11:circuits-and-power:voltage}{spanning},
        het geleverde \omterm{def:g10:energy-conservation:power}{vermogen}, en
        het inwendig verloren \omterm{def:g10:energy-conservation:power}{vermogen}.
  \item Verklaar waarom de koplampen dimmen terwijl de motor start.
\end{enumerate}
\end{exercise}

\section{Probleem: Van de dam tot de toaster}

\begin{problem}[{Weekendprobleem --- de duizend-kilometerreis van een
kilowattuur, van vallend water tot bruinerend brood, met de volle
rekening van de trip}]
\label{pb:g11:circuits-and-power:1}
Een bergdam houdt water \qty{200}{m} boven haar turbines, die
\qty{50}{m^3} --- \qty{5.0e4}{kg} --- per seconde inslikken; turbine en
alternator zetten om met
\omterm{def:g10:energy-conservation:efficiency}{rendement} $0.90$. Door een
lijn van totale \omterm{def:g11:circuits-and-power:resistance}{weerstand}
$R_\ell = \qty{32}{\ohm}$ (\qty{1000}{km} heen en terug) voedt de
centrale een stad waar je \qty{900}{W}-toaster wacht op \qty{230}{V}. De
alternator levert \qty{20}{kV}; transformatoren
(\omterm{def:g10:energy-conservation:efficiency}{rendement} $0.99$ elk)
zetten \omterm{def:g11:circuits-and-power:voltage}{spanningen} om naar
behoefte. Elektriciteit kost $0.25$ euro per \unit{kW.h};
$g = \qty{9.81}{N/kg}$.

\medskip
\noindent\textbf{Deel I --- Bij de dam.}
\begin{enumerate}
  \item Hoeveel potentiĂ«le \omterm{def:g10:energy-conservation:energy}{energie}
        geeft het vallende water elke seconde vrij? Leid het
        \omterm{def:g10:energy-conservation:power}{vermogen} af dat voor de
        turbines beschikbaar is.
  \item Welk \omterm{prop:g11:circuits-and-power:power}{elektrisch vermogen}
        verlaat de alternator?
  \item Bij \qty{20}{kV} rechtstreeks van de alternator, welke stroom
        vloeit?
  \item Bereken het joule-verlies $R_\ell I^2$ bij die stroom en
        vergelijk met de output van de centrale. Concludeer.
  \item Het \omterm{def:g10:energy-conservation:power}{vermogen} $P = UI$ van
        de stad is vast door de vraag. Wat is de enige manier om $I$ te
        verkleinen, en welk toestel doet dat?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Op de lijn.}
\begin{enumerate}[resume]
  \item Een transformator, voor deze vraag als verliesvrij behandeld,
        transformeert de \qty{88}{MW} omhoog naar \qty{400}{kV}: welke
        stroom vloeit nu?
  \item Bereken het nieuwe lijnverlies, in megawatt.
  \item Welke fractie van de output van de centrale gaat verloren in de
        lijn?
  \item Toon dat de verloren fractie gelijk is aan $R_\ell P/U^2$, en
        controleer dat dit de factor tussen vragen~4 en~7 verklaart.
  \item Bereken de \omterm{def:g11:circuits-and-power:voltage}{spanningsval}
        $R_\ell I$ langs de lijn bij \qty{400}{kV}. Is ze consistent met
        vraag~8?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- In de keuken.}
\begin{enumerate}[resume]
  \item Zet \qty{1}{kW.h} om in
        \omterm{def:g10:energy-conservation:energy}{joule}; wat vermenigvuldigt
        de meter?
  \item Bereken de stroom en
        \omterm{def:g11:circuits-and-power:resistance}{weerstand} van de toaster
        bij \qty{230}{V}.
  \item Drie minuten toasten: de
        \omterm{def:g10:energy-conservation:energy}{energie} in
        \omterm{def:g10:energy-conservation:energy}{joule} en
        \omterm{def:g10:energy-conservation:kwh}{kilowattuur}, en haar prijs.
  \item De toaster dissipeert $RI^2$: verklaar in Ă©Ă©n zin waarom dezelfde
        formule het verlies op de lijn is, het product in de keuken.
  \item De toaster deelt een \qty{16}{A}
        \omterm{def:g11:circuits-and-power:joule}{zekering} met een
        \qty{2.2}{kW}-waterkoker: houdt de
        \omterm{def:g11:circuits-and-power:joule}{zekering}? Een
        \qty{2.0}{kW}-verwarming komt erbij: en nu?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De audit.}
\begin{enumerate}[resume]
  \item De eigen bedrading van het huis,
        \omterm{def:g11:circuits-and-power:resistance}{weerstand}
        \qty{0.05}{\ohm}, voert de \qty{13.5}{A} van vraag~15: bereken
        het verlies en zijn fractie van de ongeveer \qty{3.1}{kW}
        geleverd.
  \item Keten de rendementen (turbine--alternator, omhoogtransformeren,
        lijn, omlaagtransformeren, huisbedrading) tot Ă©Ă©n overall
        net-\omterm{def:g10:energy-conservation:efficiency}{rendement}.
  \item Hoeveel \omterm{def:g10:orders-of-magnitude:unit}{kilogram} (en
        kubieke \omterm{def:g10:orders-of-magnitude:unit}{meter}) damwater
        moet vallen voor \emph{Ă©Ă©n}
        \omterm{def:g10:energy-conservation:kwh}{kilowattuur} om je toaster
        te bereiken?
  \item Evalueer de verloren fractie $R_\ell P/U^2$ van vraag~9 bij
        $U = \qty{20}{kV}$. Wat betekent een waarde boven $1$?
  \item De punchline --- \emph{de prijs van afstand}: in twee zinnen,
        wat kost Ă©Ă©n getoast
        \omterm{def:g10:energy-conservation:kwh}{kilowattuur} de berg, welke
        fractie sterft onderweg, en welke enkele truc maakt het mogelijk?
\end{enumerate}
\end{problem}
