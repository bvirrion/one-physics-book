\chapter{De fundamentele wisselwerkingen}\label{ch:g11:fundamental-interactions}

Wrijf een ballon over je haar en hij raapt stukjes papier op --- en
verslaat, met een paar vierkante centimeters rubber, de gravitatietrek
van de hele planeet. Die makkelijke overwinning is waarom
\omterm{def:g11:fundamental-interactions:ladder}{atomen} bijeenhouden en
waarom zware kernen uiteindelijk splijten
(\cref{ch:g11:nucleus-radioactivity}). Het heelal draait op precies vier
wisselwerkingen; dit hoofdstuk ontmoet ze, weegt ze tegen elkaar af, en
wijst elke haar verdieping toe.

\section{Materie van quarks tot sterrenstelsels}

\begin{definition}[De structuurladder]\label{def:g11:fundamental-interactions:ladder}
Materie is gebouwd in verdiepingen, elk samengesteld uit die eronder:
drie \emph{quarks}\index{quark} (puntvormig, kleiner dan \qty{e-18}{m})
binden tot een \emph{nucleon}\index{nucleon} --- proton of neutron,
ongeveer \qty{e-15}{m} groot; nucleonen pakken tot een
\emph{kern}\index{kern} (\qty{e-15}{m} tot \qty{e-14}{m}); een kern plus
haar elektronenwolk is een \emph{atoom}\index{atoom} (\qty{e-10}{m});
atomen binden tot \emph{moleculen}\index{molecule} en kristallen
(\qty{e-9}{m} op), die samenkomen tot alledaagse voorwerpen
(\qty{1}{m}), planeten (\qty{e7}{m}), planetenstelsels (\qty{e11}{m}),
sterrenstelsels (\qty{e21}{m}) en het waarneembare heelal
(\qty{e26}{m}).
\end{definition}

\begin{definition}[De vier wisselwerkingen]\label{def:g11:fundamental-interactions:four}
Een \emph{fundamentele wisselwerking}\index{fundamentele wisselwerking}
is een \omterm{def:g10:inertia:force}{kracht} die niet reduceerbaar is tot
andere \omterm{def:g10:inertia:force}{krachten}; er zijn er precies vier:
\emph{gravitatie}\index{gravitatie}, de \emph{elektromagnetische
wisselwerking}\index{elektromagnetische wisselwerking}, de
\omterm{def:g11:fundamental-interactions:strong}{sterke wisselwerking} en de
\omterm{def:g11:fundamental-interactions:weak}{zwakke wisselwerking}.
\end{definition}

\begin{remark}\label{rem:g11:fundamental-interactions:empty}
De verdiepingen zijn gescheiden door leegtes: een
\omterm{def:g11:fundamental-interactions:ladder}{atoom} is $10^{5}$ keer
breder dan haar \omterm{def:g11:fundamental-interactions:ladder}{kern} ---
schaalt de \omterm{def:g11:fundamental-interactions:ladder}{kern} op tot een
\qty{1}{cm}-knikker en de elektronenwolk is een kilometer breed. Iets
moet deze schaarse verdiepingen bijeenhouden, en dat is bijna nooit
zwaartekracht.
\end{remark}

\section{Elektrische lading}

\begin{definition}[Elektrische lading]\label{def:g11:fundamental-interactions:charge}
\emph{Elektrische lading}\index{elektrische lading} is de eigenschap van
materie waarop de
\omterm{def:g11:fundamental-interactions:four}{elektromagnetische
wisselwerking} werkt, zoals massa de eigenschap is waarop
\omterm{def:g11:fundamental-interactions:four}{gravitatie} werkt. Ze wordt
gemeten in \emph{coulomb}\index{coulomb} (\unit{C}) en komt in
\emph{twee tekens}, positief en negatief, die bij optelling opheffen;
gelijke ladingen stoten af, ongelijke trekken aan.
\end{definition}

\begin{definition}[Elementaire lading]\label{def:g11:fundamental-interactions:elementary}
Lading is korrelig: elke waargenomen lading is een geheel veelvoud van
de \emph{elementaire lading}\index{elementaire lading}
$e = \qty{1.60e-19}{C}$. Het proton draagt $+e$, het elektron $-e$, het
neutron $0$; een lading $q$ bevat $N = \abs{q}/e$ elementaire ladingen.
\end{definition}

\begin{proposition}[Behoud van lading]\label{prop:g11:fundamental-interactions:conservation}
De totale lading van een geïsoleerd systeem verandert nooit: lading
wordt noch gecreëerd noch vernietigd, alleen overgedragen --- in de
praktijk door elektronen, de lichtste dragers.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:conservation}
Nooit is een schending waargenomen; de diepe reden, een symmetrie van
het elektromagnetisme, wordt gegeven in de bachelorvolumes.
\end{remark}

\begin{definition}[Opladen door wrijving en door contact]\label{def:g11:fundamental-interactions:charging}
Twee isolatoren wrijven trekt elektronen van de ene naar de andere,
met achterlating van tegengestelde ladingen van gelijke grootte
(\emph{opladen door wrijving}\index{opladen door wrijving}); een geladen
voorwerp een neutraal aanraken deelt de lading (\emph{opladen door
contact}\index{opladen door contact}). Alleen elektronen bewegen; de
totale lading blijft behouden.
\end{definition}

\begin{example}[Elektronen tellen]\label{ex:g11:fundamental-interactions:balloon}
Een ballon gewreven over haar krijgt $q = \qty{-1.6e-8}{C}$: hij heeft
\emph{gewonnen} $N = \num{1.6e-8} / \num{1.6e-19} = \num{1.0e11}$
elektronen, en het haar blijft achter met precies $+\qty{1.6e-8}{C}$.
\end{example}

\section{De wet van Coulomb}

\begin{theorem}[Wet van Coulomb]\label{thm:g11:fundamental-interactions:coulomb}
Twee puntladingen $q_1$ en $q_2$ op afstand $d$ oefenen op elkaar
\omterm{def:g10:inertia:force}{krachten} uit langs de verbindingslijn, van
gelijke groottes en tegengestelde richtingen --- afstotend bij gelijke
tekens, aantrekkend bij ongelijke --- met
\[
F = k\,\frac{\abs{q_1 q_2}}{d^{2}},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{theorem}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:torsion}
Coulomb stelde de wet vast in 1785 door de twist van een
\omterm{def:g11:lenses-and-eye:lens}{dunne} torsievezel te meten; waarom de
\omterm{def:g10:orders-of-magnitude:scinot}{exponent} precies $2$ is, wordt
beantwoord in de bachelorvolumes.
\end{remark}

\begin{remark}[Een formele tweeling van gravitatie]\label{rem:g11:fundamental-interactions:twin}
De wet van Coulomb is, symbool voor symbool, de wet van universele
\omterm{def:g11:fundamental-interactions:four}{gravitatie}
(\cref{ch:g10:universal-gravitation}) met ladingen in plaats van massa's:
\begin{center}
\begin{tabular}{lll}
bron & massa $m$ & lading $q$ \\
wet & $F = G\,\dfrac{m_1 m_2}{d^2}$ & $F = k\,\dfrac{\abs{q_1 q_2}}{d^2}$ \\[1.5ex]
constante & $G = \qty{6.67e-11}{N.m^2/kg^2}$ & $k = \qty{8.99e9}{N.m^2/C^2}$ \\
teken & altijd aantrekkend & aantrekkend of afstotend \\
\end{tabular}
\end{center}
De twee verschillen drijven alles hieronder: elektriciteit is
overweldigend sterker, en alleen zij kan zichzelf opheffen.
\end{remark}

\begin{example}[Twee protonen, beide krachten tegelijk]\label{ex:g11:fundamental-interactions:protons}
Twee protonen ($m_p = \qty{1.67e-27}{kg}$) zitten $d = \qty{1.0e-15}{m}$
uit elkaar --- nucleaire buren. Elektrische afstoting:
\[
F_E = \num{8.99e9} \times
\frac{(\num{1.60e-19})^{2}}{(\num{1.0e-15})^{2}}
\approx \qty{2.3e2}{N}
\]
--- het \omterm{def:g10:universal-gravitation:weight}{gewicht} van een
$23$-kg-koffer, gedragen door een deeltje van $10^{-27}$
\omterm{def:g10:orders-of-magnitude:unit}{kilogram}. Gravitationele
aantrekking: $F_G = \num{6.67e-11} \times (\num{1.67e-27})^{2} /
(\num{1.0e-15})^{2} \approx \qty{1.9e-34}{N}$. De verhouding
$F_E / F_G = k e^{2} / (G m_p^{2}) \approx \num{1.2e36}$ is
afstandsonafhankelijk ($d^{2}$ valt weg): tussen elementaire deeltjes
is zwaartekracht $10^{36}$ keer te zwak om te tellen --- op \emph{elke}
afstand.
\end{example}

\section{De sterke en zwakke wisselwerkingen}

Het voorbeeld laat een schandaal achter: een helium-\omterm{def:g11:fundamental-interactions:ladder}{kern}
houdt twee protonen die met tientallen
\omterm{def:g10:inertia:force}{newton} afstoten --- en toch bestaat
helium.

\begin{definition}[De sterke wisselwerking]\label{def:g11:fundamental-interactions:strong}
De \emph{sterke wisselwerking}\index{sterke wisselwerking} bindt
\omterm{def:g11:fundamental-interactions:ladder}{quarks} tot
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen} en
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen} tot kernen. Op
\qty{e-15}{m} is ze aantrekkend en ver sterker dan de elektrische
afstoting (duizenden \omterm{def:g10:inertia:force}{newton} tussen
\omterm{def:g11:fundamental-interactions:ladder}{nucleonen}); voorbij enkele
$10^{-15}$ \omterm{def:g10:orders-of-magnitude:unit}{meter} verdwijnt ze: haar
\emph{reikwijdte}\index{reikwijdte (van een wisselwerking)} is ongeveer
\qty{e-15}{m}. Ze grijpt protonen en neutronen gelijk en negeert lading.
\end{definition}

\begin{remark}[Waarom kernen bestaan --- en waarom zware bezwijken]\label{rem:g11:fundamental-interactions:nuclei}
De twee \omterm{def:g10:inertia:force}{krachten} in de
\omterm{def:g11:fundamental-interactions:ladder}{kern} schalen verschillend. De
sterke lijm is kortbereik: een
\omterm{def:g11:fundamental-interactions:ladder}{nucleon} bindt alleen haar
enkele buren binnen \qty{e-15}{m}. De elektrische afstoting is
langbereik: elk proton duwt op \emph{elk ander} proton. Naarmate kernen
groeien, stapelt afstoting sneller op dan lijm: zware kernen hebben
extra neutronen nodig (lijm zonder lading), en voorbij ongeveer $80$
protonen faalt zelfs dat --- de zwaarste kernen leven op geleende tijd
(\cref{ch:g11:nucleus-radioactivity}).
\end{remark}

\begin{definition}[De zwakke wisselwerking]\label{def:g11:fundamental-interactions:weak}
De \emph{zwakke wisselwerking}\index{zwakke wisselwerking}, van
\omterm{def:g11:fundamental-interactions:strong}{reikwijdte} nog korter, laat
een neutron in een proton veranderen (en terug): ze drijft de
$\beta$-radioactiviteit van \cref{ch:g11:nucleus-radioactivity}.
\end{definition}

\section{Wie regeert welke verdieping}

\begin{method}[Een schaal auditen]\label{met:g11:fundamental-interactions:audit}
Om te vinden welke wisselwerking een structuur van grootte $d$ regeert,
check \omterm{def:g11:fundamental-interactions:strong}{reikwijdte} en
opheffing:
\begin{enumerate}
  \item $d \leq \qty{e-14}{m}$: de
        \emph{\omterm{def:g11:fundamental-interactions:strong}{sterke
        wisselwerking}} regeert --- ze verslaat de elektrische afstoting,
        en zwaartekracht is $10^{36}$ te weinig.
  \item \omterm{def:g11:fundamental-interactions:ladder}{atomen} tot bergen
        ($\qty{e-10}{m}$ tot $\qty{e4}{m}$): de sterke
        \omterm{def:g10:inertia:force}{kracht} is buiten
        \omterm{def:g11:fundamental-interactions:strong}{reikwijdte}; de
        \emph{\omterm{def:g11:fundamental-interactions:four}{elektromagnetische
        wisselwerking}} regeert --- bindingen, cohesie,
        \omterm{def:g10:inertia:inventory}{wrijving}, contact.
  \item planeten en verder ($d \geq \qty{e6}{m}$): materie is neutraal
        tot fantastische precisie, dus elektriciteit heft zichzelf op;
        massa heeft één teken en telt altijd op ---
        \emph{\omterm{def:g11:fundamental-interactions:four}{gravitatie}}
        regeert de hemel.
\end{enumerate}
De \omterm{def:g11:fundamental-interactions:weak}{zwakke wisselwerking} houdt
geen verdieping: ze bindt niets, maar arbitreert transformaties
($\beta$-verval).
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g11:fundamental-interactions:1}
Geef de \omterm{def:g10:orders-of-magnitude:oom}{orde van grootte} van elke
grootte en haar verdieping van de ladder: proton \qty{1.7e-15}{m}; water-\omterm{def:g11:fundamental-interactions:ladder}{molecule}
\qty{2.8e-10}{m}; zandkorrel \qty{2e-4}{m}; Aarde \qty{1.3e7}{m}; Melkweg
\qty{9.5e20}{m}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:fundamental-interactions:2}
Een ballon gewreven over haar draagt $q = \qty{-4.8e-9}{C}$. Heeft hij
elektronen gewonnen of verloren, en hoeveel? Wat is de lading van het
haar erna, en welke wet zegt dat?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:fundamental-interactions:3}
Ladingen $q_1 = \qty{+2.0e-6}{C}$ en $q_2 = \qty{-3.0e-6}{C}$ zitten
\qty{30}{cm} uit elkaar. Bereken de
\omterm{def:g10:inertia:force}{kracht}; aantrekkend of afstotend? Vergelijk
de \omterm{def:g10:inertia:force}{krachten} op $q_1$ en op $q_2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:fundamental-interactions:4}
Twee ladingen trekken aan met een
\omterm{def:g10:inertia:force}{kracht} $F_0$. Wat wordt ze als: de afstand
verdubbelt; de afstand door $3$ wordt gedeeld; één lading verdubbelt;
beide ladingen \emph{en} de afstand verdubbelen?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:fundamental-interactions:5}
Noem de wisselwerking verantwoordelijk voor: (a) de Maan die om de Aarde
draait; (b) de cohesie van een zoutkristal; (c) de cohesie van een
helium-\omterm{def:g11:fundamental-interactions:ladder}{kern}; (d) het
$\beta$-verval van koolstof-14; (e) de ballon van
\cref{exo:g11:fundamental-interactions:2} die aan een muur plakt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:fundamental-interactions:6}
De twee protonen van een helium-\omterm{def:g11:fundamental-interactions:ladder}{kern}
zitten $d = \qty{2.0e-15}{m}$ uit elkaar ($m_p = \qty{1.67e-27}{kg}$).
Bereken hun elektrische afstoting, hun gravitationele aantrekking, en de
verhouding. Wat houdt de
\omterm{def:g11:fundamental-interactions:ladder}{kern} heel?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:fundamental-interactions:7}

het elektron ($m_e = \qty{9.11e-31}{kg}$) $d = \qty{5.3e-11}{m}$ van het

\omterm{def:g11:fundamental-interactions:ladder}{atomen} bijeen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:fundamental-interactions:8}
Bol A draagt $q_A = \qty{+8.0e-9}{C}$; een identieke bol B is neutraal.

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:fundamental-interactions:9}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:fundamental-interactions:10}
Twee protonen in naburige
\omterm{def:g11:fundamental-interactions:ladder}{moleculen} zitten

Wat draagt de \omterm{def:g11:fundamental-interactions:strong}{sterke

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:fundamental-interactions:11}
Elk van twee munten houdt ongeveer $\num{e23}$ elektronen. Verplaats één
elektron op een \emph{miljoen} ertussen en zet de munten \qty{1.0}{m} uit

\omterm{def:g10:inertia:force}{kracht}. Het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van welke massa
evenaart haar? Concludeer over de neutraliteit van alledaagse materie.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:fundamental-interactions:12}
Een uranium-\omterm{def:g11:fundamental-interactions:ladder}{kern} heeft diameter
\qty{1.4e-14}{m} en $92$ protonen. Bereken de afstoting tussen twee
protonen aan tegenoverliggende einden. Kan de
\omterm{def:g11:fundamental-interactions:strong}{sterke wisselwerking} dit paar
binden --- waarom niet? Hoeveel afstotende protonenparen zijn er?
Waarom hebben zware kernen extra neutronen nodig, en waarom kan voorbij
een zekere grootte zelfs neutronen hen niet redden?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:fundamental-interactions:13}
Twee balletjes van massa \qty{1.0}{g} hangen aan \qty{50}{cm}-draden
gebonden aan hetzelfde punt. Met dezelfde lading $q$ komen ze tot rust
\qty{6.0}{cm} uit elkaar. Elk balletje is in evenwicht onder zijn
\omterm{def:g10:universal-gravitation:weight}{gewicht}, de
\omterm{def:g10:inertia:inventory}{spanning} van de draad en de elektrische
\omterm{def:g10:inertia:force}{kracht}: toon dat $F = mg\tan\theta$
($\theta$: hoek van de draad met de verticaal) en dat hier
$\tan\theta \approx 0.060$. Bereken $F$, daarna $q$, daarna het aantal
\omterm{def:g11:fundamental-interactions:elementary}{elementaire ladingen}
verplaatst.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:fundamental-interactions:14}
Gegevens: $M_E = \qty{5.97e24}{kg}$, $M_M = \qty{7.35e22}{kg}$,
Aarde--Maan-afstand $d = \qty{3.84e8}{m}$. Bereken de
Aarde--Maan-\omterm{def:g10:universal-gravitation:force}{gravitationele
kracht}. Zwaartekracht uitgeschakeld: welke ladingen $+Q$ op de Aarde en
$-Q$ op de Maan houden dezelfde aantrekking? Hoeveel elektronen is $Q$,
en wat wegen ze ($m_e = \qty{9.11e-31}{kg}$)? Commentaar.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:fundamental-interactions:15}
In een zoutkristal zitten Na$^{+}$- en Cl$^{-}$-ionen (ladingen $\pm e$)
$d = \qty{2.8e-10}{m}$ uit elkaar; een Cl$^{-}$-ion heeft massa

naburige ionen en het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van een Cl$^{-}$-ion;
geef de verhouding. Verklaar daarna hoe zwaartekracht wint op grote

\end{exercise}

\section{Probleem: De audit van de vier krachten}

\begin{problem}[{Weekendprobleem --- waarom de wereld noch instort noch
uiteen vliegt: de vier wisselwerkingen geaudit verdieping per
verdieping, tot de neutraliteit van materie tot twee delen op $10^{18}$}]
\label{pb:g11:fundamental-interactions:1}
\omterm{def:g11:fundamental-interactions:ladder}{Atomen} implodeert niet,
kernen houden grotendeels, de Maan crasht noch ontsnapt; dit probleem
audit wie de eer verdient. Gegevens: $e = \qty{1.60e-19}{C}$,
$k = \qty{8.99e9}{N.m^2/C^2}$, $G = \qty{6.67e-11}{N.m^2/kg^2}$,
$m_p = \qty{1.67e-27}{kg}$, $m_e = \qty{9.11e-31}{kg}$,
$M_E = \qty{5.97e24}{kg}$, $M_M = \qty{7.35e22}{kg}$, Aarde--Maan-afstand
\qty{3.84e8}{m}.

\medskip
\noindent\textbf{Deel I --- De ladder.}
\begin{enumerate}
  \item Som de verdiepingen van de ladder van
        \omterm{def:g11:fundamental-interactions:ladder}{quark} tot
        sterrenstelsel, één
        \omterm{def:g10:orders-of-magnitude:oom}{orde van grootte} van
        grootte elk.
  \item Schaalmodel: de
        \omterm{def:g11:fundamental-interactions:ladder}{kern} wordt een
        \qty{1}{cm}-knikker; hoe breed is het
        \omterm{def:g11:fundamental-interactions:ladder}{atoom}?
  \item Welke fractie van het volume van een
        \omterm{def:g11:fundamental-interactions:ladder}{atoom} neemt haar
        \omterm{def:g11:fundamental-interactions:ladder}{kern} in?
  \item Een haar is \qty{70}{\micro m} dik. Hoeveel
        \omterm{def:g11:fundamental-interactions:ladder}{atomen} overspannen
        haar?
  \item Hoeveel \omterm{def:g10:energy-conservation:power}{machten} van tien
        van het \omterm{def:g11:fundamental-interactions:ladder}{quark}-plafond
        (\qty{e-18}{m}) tot een sterrenstelsel (\qty{e21}{m})?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De elektrische verdieping.}
\begin{enumerate}[resume]
  \item In een waterstof-\omterm{def:g11:fundamental-interactions:ladder}{atoom}
        zit het elektron $d = \qty{5.3e-11}{m}$ van het proton: bereken
        de elektrische \omterm{def:g10:inertia:force}{kracht} ertussen.
  \item Bereken de
        \omterm{def:g10:universal-gravitation:force}{gravitationele kracht}
        ertussen, en de verhouding van de twee. Welke houdt het
        \omterm{def:g11:fundamental-interactions:ladder}{atoom} bijeen?
  \item Wat verandert als zwaartekracht binnen
        \omterm{def:g11:fundamental-interactions:ladder}{atomen} wordt
        uitgeschakeld? En als elektriciteit?
  \item Waarom zijn de ``contact''-\omterm{def:g10:inertia:force}{krachten}
        van het alledaagse leven --- de vloer die op je voeten duwt,
        \omterm{def:g10:inertia:inventory}{wrijving} --- elektromagnetisch in
        vermomming?
  \item Elk van twee munten houdt ongeveer $\num{e23}$ elektronen.
        Transfereer één elektron op een \emph{miljard} ertussen, zet ze
        \qty{1.0}{m} uit elkaar: bereken de ladingen en de
        \omterm{def:g10:inertia:force}{kracht}. Conclusie?
  \item Welke wisselwerking regeert de verdiepingen van \qty{e-10}{m}
        tot \qty{e4}{m}, en waarom concurreert de
        \omterm{def:g11:fundamental-interactions:strong}{sterke
        wisselwerking} niet?
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De nucleaire verdieping.}
\begin{enumerate}[resume]
  \item Twee protonen zitten \qty{1.0e-15}{m} uit elkaar in een
        \omterm{def:g11:fundamental-interactions:ladder}{kern}: bereken hun
        elektrische afstoting.
  \item Ongebreideld, welke versnelling zou deze
        \omterm{def:g10:inertia:force}{kracht} een proton geven? Vergelijk
        met $g$.
  \item Som de drie eigenschappen die de
        \omterm{def:g11:fundamental-interactions:strong}{sterke
        wisselwerking} nodig heeft om te verklaren waarom kernen
        bestaan, waarom neutronen ook gegrepen worden, en waarom kernen
        minuscuul zijn.
  \item In een uranium-\omterm{def:g11:fundamental-interactions:ladder}{kern}
        (diameter \qty{1.4e-14}{m}, $92$ protonen), bereken de afstoting
        tussen twee protonen aan tegenoverliggende einden. Kan de
        \omterm{def:g11:fundamental-interactions:strong}{sterke
        wisselwerking} dat paar rechtstreeks binden? Verklaar waarom
        afstoting wint naarmate kernen groeien, en waar de neutronen
        voor dienen.
  \item In één zin, zoals beloofd: wat doet de
        \omterm{def:g11:fundamental-interactions:weak}{zwakke
        wisselwerking}?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De astronomische verdieping, en het oordeel.}
\begin{enumerate}[resume]
  \item Bereken de
        \omterm{def:g10:universal-gravitation:force}{gravitationele kracht}
        tussen de Aarde en de Maan.
  \item De Aarde bevat ongeveer $\num{1.8e51}$ protonen (en evenveel
        elektronen), de Maan ongeveer $\num{2.2e49}$. Welke fractie $f$
        van de protonlading van elk lichaam, onopgeheven gelaten, zou
        de elektrische \omterm{def:g10:inertia:force}{kracht} de
        gravitationele laten evenaren?
  \item Waarom regeert
        \omterm{def:g11:fundamental-interactions:four}{gravitatie}, de
        zwakkeling van vraag~7, de astronomie? Twee redenen.
  \item Oordeel --- één zin per verdieping
        (\omterm{def:g11:fundamental-interactions:ladder}{kern},
        \omterm{def:g11:fundamental-interactions:ladder}{atoom} tot berg,
        planeet en omhoog): wat houdt haar bijeen? Beantwoord daarna de
        titel: waarom stort de wereld noch in, noch vliegt ze uiteen?
\end{enumerate}
\end{problem}
