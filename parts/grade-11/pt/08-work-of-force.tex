\chapter{Trabalho de uma Força}\label{ch:g11:work-of-force}

Empurre um guarda-roupa pelo quarto e você terá trabalhado; empurrar contra um
parede a tarde toda e, independentemente do relatório dos seus músculos, a física conta
zero. \omterm{def:g11:work-of-force:work}{Trabalhar} é o que \omterm{def:g10:inertia:force}{vigor} entrega \emph{ao longo de um movimento} --- energia
em trânsito (\cref{ch:g10:energy-conservation}), pago em \omterm{def:g11:work-of-force:work}{joules}. Isto
preços do capítulo o forças de \cref{ch:g11:forces-and-motion} --- puxa
em um ângulo, \omterm{def:g10:universal-gravitation:weight}{peso}, \omterm{def:g10:inertia:inventory}{atrito} --- e cronometra a taxa de pagamento: \omterm{def:g11:work-of-force:power}{poder}.

\section{Trabalho de uma força constante}

\begin{definition}[Trabalho de uma força constante]\label{def:g11:work-of-force:work}
Uma constante \omterm{def:g10:inertia:force}{vigor} $\vect F $ aplicado a um objeto movendo-se ao longo de uma linha reta
segmento de $ A $ para $ B $ faz o \emph{trabalhar}\index{trabalhar}
\[
W_{AB}(\vect F) = F \times AB \times \cos\theta ,
\]
onde $ F $ é a magnitude do \omterm{def:g10:inertia:force}{vigor} (em \unit{N}), $ AB $ o comprimento
do deslocamento (em \unit{m}) e $\theta $ o ângulo entre
$\vect F $ e $\vect{AB}$. O trabalho é medido em
\emph{joules}\index{joules} (\unit{J}): um \omterm{def:g10:energy-conservation:energy}{joule} é o trabalho de um
um newton \omterm{def:g10:inertia:force}{vigor} ao longo de um \omterm{def:g10:orders-of-magnitude:unit}{medidor} de sua própria direção.
\end{definition}

\begin{remark}\label{rem:g11:work-of-force:scal}
$ F \times AB \times \cos\theta $ é exatamente o produto escalar
$\vect F \cdot \vect{AB}$ do volume de matemática deste ano. Apenas o
componente $ F\cos\theta $ \emph{junto} o movimento ganha qualquer coisa.
\end{remark}

\begin{example}[Mala na coleira]\label{ex:g11:work-of-force:suitcase}
Um viajante arrasta uma mala \qty{300}{m} através de um terminal, a cinta
puxado com $ F = \qty{40}{N}$ no $\theta = \qty{50}{\degree}$ para o
andar: $ W = 40 \times 300 \times \cos\qty{50}{\degree} \approx
\qty{7.7}{kJ}$. Puxado plano ($\theta = 0$) o mesmo \omterm{def:g10:inertia:force}{vigor} faria
entregar \qty{12}{kJ}: um terço da tração não levanta nada.
\end{example}

\begin{definition}[Motor, resistivo, trabalho zero]\label{def:g11:work-of-force:sign}
O \omterm{def:g10:signals-and-waves:signal}{sinal} de $\cos\theta $ classifica cada \omterm{def:g10:inertia:force}{vigor} em três regimes:
\begin{itemize}
  \item $\theta < \qty{90}{\degree}$: $ W > 0$, o \omterm{def:g11:work-of-force:work}{trabalhar} é
        \emph{motor}\index{motor} --- o \omterm{def:g10:inertia:force}{vigor} alimenta energia em
        o movimento;
  \item $\theta = \qty{90}{\degree}$: $ W = 0$, o \omterm{def:g10:inertia:force}{vigor} funciona não em
        todos, por maiores que sejam;
  \item $\theta > \qty{90}{\degree}$: $ W < 0$, o \omterm{def:g11:work-of-force:work}{trabalhar} é
        \emph{resistivo}\index{resistivo} --- o \omterm{def:g10:inertia:force}{vigor} drenos
        energia. O caso puro é um freio em $\qty{180}{\degree}$:
        \qty{800}{N} sobre \qty{45}{m} leva $800 \times 45 \times (-1)
        = \qty{-36}{kJ}$ fora de um carro, em discos quentes.
\end{itemize}
\end{definition}

\section{O trabalho do peso}

\begin{proposition}[Trabalho do peso]\label{prop:g11:work-of-force:weight}
Quando um objeto de massa $ m $ se move de $ A $ (altitude $ z_A $) para $ B $
(altitude $ z_B $) junto \emph{qualquer} caminho feito de segmentos retos, seu
\omterm{def:g10:universal-gravitation:weight}{peso} $\vect P $ faz o \omterm{def:g11:work-of-force:work}{trabalhar}
\[
W_{AB}(\vect P) = mg\,(z_A - z_B) = \pm\, mgh ,
\]
onde $ h = \abs{z_A - z_B}$ é a queda de altura: $+mgh $ descendo,
$-mgh $ subindo --- só a queda importa, não a rota.
\end{proposition}

\begin{proof}
Em um segmento reto $ AB $, o \omterm{def:g10:universal-gravitation:weight}{peso} (magnitude $ mg $, direto
para baixo) faz um ângulo $\theta $ com $\vect{AB}$, e $ AB\cos\theta $ é
precisamente a projeção do deslocamento na \omterm{def:g10:universal-gravitation:weight}{vertical} descendente:
$ z_A - z_B $. Por isso $ W = mg \times AB\cos\theta = mg\,(z_A - z_B)$.
Ao longo de uma cadeia de segmentos, o funciona adicione e o telescópio de gotas:
$ mg(z_A - z_C) + mg(z_C - z_B) = mg(z_A - z_B)$.
\end{proof}

\begin{proposition}[Qualquer caminho]\label{prop:g11:work-of-force:anypath}
A mesma fórmula é válida ao longo de qualquer caminho curvo de $ A $ para $ B $.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:work-of-force:anypath}
Plausível --- uma curva é o mais próximo que gostaríamos de uma linha quebrada de muitos
segmentos curtos --- mas tornar tão próximo quanto quisermos honesto é
cálculo, feito em \cref{ch:g12:work-and-energy}.
\end{remark}

\begin{example}[Caminhante]\label{ex:g11:work-of-force:hiker}
UM \qty{78}{kg} caminhante (pacote incluído) sobe \qty{450}{m}, por qualquer
trilha: $ W(\vect P) = -78 \times 9.81 \times 450 \approx \qty{-344}{kJ}$,
e exatamente $+\qty{344}{kJ}$ recuar, quaisquer que sejam os desvios.
\end{example}

\section{O trabalho de fricção}

\begin{proposition}[Trabalho de fricção deslizante]\label{prop:g11:work-of-force:friction}
UM \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g10:inertia:force}{vigor} de magnitude constante $ f $, em cada instante oposto
o movimento, o \omterm{def:g11:work-of-force:work}{trabalhar} $ W = -fL $ ao longo de um caminho de comprimento total $ L $:
sempre \omterm{def:g11:work-of-force:sign}{resistivo}, e proporcional ao \emph{comprimento do caminho
viajou} --- não ao deslocamento.
\end{proposition}

\begin{proof}
Cada peça curta e reta tem o \omterm{def:g10:inertia:force}{vigor} no $\qty{180}{\degree}$ para o
movimento, contribuindo $-f \times (\text{its length})$; as peças adicionam
até $-fL $. Um caminho curvo é cortado em pedaços curtos da mesma forma, o
limite admitido quanto ao \omterm{def:g10:universal-gravitation:weight}{peso} (\cref{rem:g11:work-of-force:anypath}).
\end{proof}

\begin{remark}[Atrito contra peso: a grande divisão]\label{rem:g11:work-of-force:divide}
O pesode \omterm{def:g11:work-of-force:work}{trabalhar} esquece o caminho; atritoé um pedágio por \omterm{def:g10:orders-of-magnitude:unit}{medidor}. Ligado
uma viagem de ida e volta \omterm{def:g10:universal-gravitation:weight}{peso} rede zero --- o que é preciso para subir, ele reembolsa
em declive --- enquanto \omterm{def:g10:inertia:inventory}{atrito} cobranças $-fL $ em todas as etapas e reembolsos
nada: o pesode \omterm{def:g11:work-of-force:work}{trabalhar} pode ser armazenado e recuperado (potencial
energia, \cref{ch:g11:mechanical-energy}), \omterm{def:g10:inertia:inventory}{atrito}'s é perdido como calor.
\end{remark}

\begin{example}[Lá e de volta]\label{ex:g11:work-of-force:roundtrip}
Uma caixa desliza \qty{5.0}{m} através de um chão e \qty{5.0}{m} de volta,
contra $ f = \qty{60}{N}$: \omterm{def:g10:inertia:inventory}{atrito} faz $-60 \times 10 =
\qty{-600}{J}$ mesmo que a caixa termine onde começou. \omterm{def:g10:universal-gravitation:weight}{Peso} e
normais \omterm{def:g10:inertia:force}{vigor}, perpendicular ao movimento: zero por toda parte.
\end{example}

\begin{remark}[E uma mola?]\label{rem:g11:work-of-force:spring}
Uma mola puxa com mais força quanto mais é esticada: sua \omterm{def:g10:inertia:force}{vigor} não é
constante e \cref{def:g11:work-of-force:work} não se aplica.
Isso é \omterm{def:g11:work-of-force:work}{trabalhar} significa cortar o alongamento em etapas infinitesimais e
somando --- cálculo, entregue em \cref{ch:g12:work-and-energy}.
\end{remark}

\section{Poder}

\begin{definition}[Poder]\label{def:g11:work-of-force:power}
O \emph{poder}\index{poder} de um \omterm{def:g10:inertia:force}{vigor} é a taxa em
qual é funciona, $ P = W/\Delta t $, em \emph{watts}\index{watts}
($\unit{W} = \unit{J/s}$) --- o mesmo watt que em
\cref{ch:g11:circuits-and-power}. Um pré-SI \omterm{def:g10:orders-of-magnitude:unit}{unidade} sobrevive no \omterm{def:g11:work-of-force:sign}{motor}
emblemas: o \emph{cavalos de potência}\index{cavalos de potência},
$\qty{1}{hp} = \qty{736}{W}$.
\end{definition}

\begin{proposition}[Potência em velocidade constante]\label{prop:g11:work-of-force:pfv}
Se o objeto se move com velocidade constante $ v $, uma constante \omterm{def:g10:inertia:force}{vigor} $\vect F $
em ângulo $\theta $ ao movimento entrega o \omterm{def:g11:work-of-force:power}{poder}
$ P = F v \cos\theta $ --- em particular $ P = Fv $ por um \omterm{def:g10:inertia:force}{vigor} ao longo do
movimento.
\end{proposition}

\begin{proof}
Em um tempo $\Delta t $ o objeto cobre $ d = v\,\Delta t $, então
$ W = F \times v\,\Delta t \times \cos\theta $; dividir por $\Delta t $.
\end{proof}

\begin{example}[Guindaste]\label{ex:g11:work-of-force:crane}
Um guindaste levanta um \qty{600}{kg} palete em uma posição constante \qty{0.90}{m/s}: o
cabo \omterm{def:g10:inertia:force}{vigor} é igual ao \omterm{def:g10:universal-gravitation:weight}{peso}, $ mg = \qty{5.89e3}{N}$, então
$ P = Fv = 5890 \times 0.90 \approx \qty{5.3}{kW} \approx \qty{7.2}{hp}$
--- uma parelha de sete cavalos em uma caixa de aço.
\end{example}

\begin{method}[Subindo com menos força: o ziguezague]\label{met:g11:work-of-force:zigzag}
Para elevar uma carga a uma altura $ h $ com um limite \omterm{def:g10:inertia:force}{vigor}:
\begin{enumerate}
  \item A conta é fixa: por \cref{prop:g11:work-of-force:weight},
        qualquer caminho para o topo custa o \omterm{def:g11:work-of-force:work}{trabalhar} $ mgh $.
  \item Esticar o caminho: em uma encosta de comprimento $ L $ o \omterm{def:g10:inertia:force}{vigor} necessário
        (velocidade constante, \omterm{def:g10:inertia:inventory}{atrito} de lado) é $ F = mgh/L $ --- o dobro
        comprimento, divida pela metade \omterm{def:g10:inertia:force}{vigor}.
  \item Pague a sobretaxa: \omterm{def:g10:inertia:inventory}{atrito} cobranças por parte \omterm{def:g10:orders-of-magnitude:unit}{medidor}
        (\cref{prop:g11:work-of-force:friction}), adicionando $ fL $.
\end{enumerate}
Estradas de montanha ziguezagueiam na etapa 2: mesma altura, menos \omterm{def:g10:inertia:force}{vigor}, mesmo \omterm{def:g11:work-of-force:work}{trabalhar}.
\end{method}

\section{Exercícios}

\section{Problema: o dia da mudança}

\begin{problem}[{Problema de fim de semana --- a física paga por joule: a
dia de deslizamentos, içamentos e curvas fechadas, terminando com a humilhação
descoberta do que uma chaleira pensa sobre o trabalho honesto}]
\label{pb:g11:work-of-force:1}
A equipe de uma transportadora esvazia um armazém e transporta a carga até o terceiro andar
plana e depois dirige a van até uma vila no topo de uma colina. Cada tarefa tem um preço
em \omterm{def:g11:work-of-force:work}{joules}; a chaleira da cozinha auditará o dia. Caixas: \qty{30}{kg}.

\medskip
\noindent\textbf{Part I --- Across the warehouse floor.}
Uma caixa é empurrada horizontalmente \omterm{def:g10:inertia:force}{vigor} $ F = \qty{110}{N}$ sobre
$ d = \qty{8.0}{m}$; deslizando \omterm{def:g10:inertia:inventory}{atrito} é $ f = \qty{95}{N}$.
\begin{enumerate}
  \item Calcule o \omterm{def:g11:work-of-force:work}{trabalhar} do empurrão.
  \item Calcule o \omterm{def:g11:work-of-force:work}{trabalhar} de \omterm{def:g10:inertia:inventory}{atrito}, do \omterm{def:g10:universal-gravitation:weight}{peso}, e do normal
        \omterm{def:g10:inertia:force}{vigor} (justifique os zeros).
  \item Total \omterm{def:g11:work-of-force:work}{trabalhar} na caixa e o significado do seu \omterm{def:g10:signals-and-waves:signal}{sinal} (a caixa começou
        \omterm{def:g10:relative-motion:frame}{em repouso})? Qual \omterm{def:g10:inertia:inventory}{impulso} mantém a velocidade constante
        (\cref{ch:g11:forces-and-motion})?
  \item O canto mais distante pode ser alcançado ao longo do \qty{10.0}{m} diagonal
        ou ao longo de duas paredes (\qty{6.0}{m} então \qty{8.0}{m}). \omterm{def:g10:inertia:inventory}{Atrito}
        \omterm{def:g11:work-of-force:work}{trabalhar} em cada rota?
  \item Em uma viagem diagonal de ida e volta (ida e volta), calcule o \omterm{def:g11:work-of-force:work}{trabalhar} de
        \omterm{def:g10:inertia:inventory}{atrito} e o \omterm{def:g11:work-of-force:work}{trabalhar} do \omterm{def:g10:universal-gravitation:weight}{peso}. Qual \omterm{def:g10:inertia:force}{vigor} reembolsos?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Up to the third floor ($ h = \qty{9.0}{m}$).}
\begin{enumerate}[resume]
  \item Calcule o \omterm{def:g11:work-of-force:work}{trabalhar} do \omterm{def:g10:universal-gravitation:weight}{peso} em uma caixa elevada ao apartamento.
        O que \omterm{def:g11:work-of-force:work}{trabalhar} a tripulação deve fornecer pelo menos?
  \item Levado escada acima por um \qty{70}{kg} \omterm{def:g11:work-of-force:sign}{motor}, cargas de gravidade
        para \omterm{def:g11:work-of-force:sign}{motor} \emph{e} caixa: total \omterm{def:g11:work-of-force:work}{trabalhar} contra a gravidade e o
        fração realmente gasta na caixa?
  \item Uma rampa de bonde é \qty{30}{m} anseio pelo mesmo \qty{9.0}{m}
        subir, com \qty{40}{N} de rolar \omterm{def:g10:inertia:inventory}{atrito}. Em velocidade constante,
        calcular o empurrão necessário ao longo da rampa e seu \omterm{def:g11:work-of-force:work}{trabalhar}.
  \item Compare a rampa com uma talha \omterm{def:g10:universal-gravitation:weight}{vertical} reta: \omterm{def:g10:inertia:force}{vigor} necessário,
        \omterm{def:g11:work-of-force:work}{trabalhar} feito. O que a rampa compra e a que preço?
  \item Um guincho de corda e polia eleva a caixa a \qty{0.50}{m/s}.
        Duração e mecânica \omterm{def:g11:work-of-force:power}{poder}?
  \item Recupere isso \omterm{def:g11:work-of-force:power}{poder} como $ P = Fv $. A talha \omterm{def:g11:work-of-force:sign}{motor} é $60\%$
        eficiente: \omterm{prop:g11:circuits-and-power:power}{energia elétrica} retirou?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The zigzag road.}
A van carregada (\qty{3500}{kg}) deve chegar a uma aldeia \qty{240}{m}
acima do vale: pista reta, \qty{1.0}{km} no $24\%$ grau, ou
estrada pavimentada, \qty{8.0}{km} no $3.0\%$ nota (nota $= \sin\alpha $).
\begin{enumerate}[resume]
  \item Verifique se ambas as rotas sobem da mesma forma \qty{240}{m}.
  \item Em velocidade constante e negligenciando \omterm{def:g10:inertia:inventory}{atrito}, calcule a direção
        \omterm{def:g10:inertia:force}{vigor} necessários ao longo de cada rota.
  \item \omterm{def:g11:work-of-force:work}{Trabalhar} da condução \omterm{def:g10:inertia:force}{vigor} em cada rota? Compare com $ mgh $.
  \item Rolando \omterm{def:g11:circuits-and-power:resistance}{resistência} é \qty{400}{N} em ambos: calcule o extra
        \omterm{def:g11:work-of-force:work}{trabalhar} em cada rota. Quanto custa o conforto do ziguezague?
  \item No \qty{36}{km/h}, calcule o \omterm{def:g11:work-of-force:sign}{motor} \omterm{def:g11:work-of-force:power}{poder} necessário em cada
        rota (em \unit{kW} e \unit{hp}). Por que as estradas ziguezagueiam?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The day's ledger.}
\begin{enumerate}[resume]
  \item A tripulação desliza $60$ caixas pelo chão (o empurrão de
        questão 1) e iça todos $60$ até o \qty{9.0}{m}
        (questão 6). Totalmente mecânico \omterm{def:g11:work-of-force:work}{trabalhar} entregue nas caixas?
  \item Adicione o próprio corpo do suporte da escada: $20$ viaja às \qty{70}{kg}
        (os reembolsos em declive vão para os joelhos, não para o livro-razão). Dia
        total?
  \item A tripulação trabalhou $8.0$ horas: mecânica média \omterm{def:g11:work-of-force:power}{poder}? Quanto tempo
        seria o \qty{2.0}{kW} chaleira leva para gastar o mesmo energia?
  \item Finale: em duas frases, diga qual é o \omterm{def:g10:energy-conservation:energy}{joule} paga por --- e
        não (a parede empurrou a tarde toda, a caixa segurou
        imóvel) --- e por que os transportadores possuem guinchos, rampas e zigue-zague
        estradas em vez de músculos maiores.
\end{enumerate}
\end{problem}
