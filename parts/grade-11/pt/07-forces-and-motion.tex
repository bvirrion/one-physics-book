\section{O centro de massa}

\begin{definition}[Centro de massa]\label{def:g11:forces-and-motion:com}
O \emph{centro de massa}\index{centro de massa} de um corpo ou sistema é
seu ponto de equilíbrio --- a posição média de sua massa: o geométrico
centro de um corpo homogêneo e simétrico; para duas massas pontuais, o ponto
do segmento de união em distâncias no \emph{inverso} proporção do
massas, $\frac{d_1}{d_2} = \frac{m_2}{m_1}$ --- mais perto do mais pesado.
\end{definition}

\begin{example}[Um haltere torto]\label{ex:g11:forces-and-motion:dumbbell}
Bolas de \qty{2.0}{kg} e \qty{1.0}{kg} sente-se nas extremidades de uma barra de luz
\qty{0.90}{m} longo. O \omterm{def:g11:forces-and-motion:com}{centro de massa} divide na razão inversa
$ d_1/d_2 = 1.0/2.0$: $\qty{0.30}{m}$ da bola pesada,
$\qty{0.60}{m}$ do leve --- equilibre a barra ali.
\end{example}

\begin{remark}[A chave lançada]\label{rem:g11:forces-and-motion:wrench}
Filme uma chave inglesa: cada ponto segue uma complicada curva em loop
--- exceto o \omterm{def:g11:forces-and-motion:com}{centro de massa}, que traça a parábola limpa a lançada
bola faria. O movimento do \omterm{def:g11:forces-and-motion:com}{centro de massa} é simples; o resto é giro
\emph{em volta} isso --- é por isso que os mecânicos podem tratar um carro, um planeta ou
uma ginasta como um único ponto.
\end{remark}

\section{Uma força resultante altera a velocidade}

\begin{definition}[Força líquida]\label{def:g11:forces-and-motion:netforce}
O \emph{força resultante}\index{força resultante} em um corpo é a soma vetorial de
todos os forças agindo sobre isso, $\vect F = \vect F_1 + \vect F_2 + \dots $
(ponta a cauda, como no volume de matemática); o forças \omterm{def:g10:inertia:compensate}{compensar}
exatamente quando $\vect F = \vect 0$.
\end{definition}

\begin{theorem}[Efeito de uma força resultante --- semiquantitativa]\label{thm:g11:forces-and-motion:secondlaw}
\index{segunda lei do movimento}
UM \omterm{def:g11:forces-and-motion:netforce}{força resultante} $\vect F $ agindo sobre um corpo de massa $ m $ por um curto \omterm{def:g10:signals-and-waves:period}{período}
$\Delta t $ altera a velocidade de seu \omterm{def:g11:forces-and-motion:com}{centro de massa} por um vetor
$\Delta \vect v $ isso tem o \emph{direção} de $\vect F $, e cujo
magnitude cresce proporcionalmente $ F $ e para $\Delta t $ e se encolhe
proporcional à massa:
\[
\Delta v \propto \frac{F \, \Delta t}{m} .
\]
\end{theorem}
\admitted

\begin{remark}
Dobre o \omterm{def:g10:inertia:force}{vigor} ou a sua duração: o dobro da mudança; dobrar a massa:
metade. O próximo ano transforma isso em uma equação exata, e o Ano 1
o volume pergunta em quais frames ele contém; este ano as proporções são suficientes ---
eles já usam airbags, freios e \omterm{def:g10:universal-gravitation:satellite}{órbitas}.
\end{remark}

\begin{proposition}[Os três casos canônicos]\label{prop:g11:forces-and-motion:cases}
Deixe um corpo se mover com velocidade $\vect v $ sob um \omterm{def:g11:forces-and-motion:netforce}{força resultante} $\vect F $.
\begin{enumerate}
  \item $\vect F $ junto $\vect v $: isto \emph{acelera} em linha reta;
  \item $\vect F $ oposto $\vect v $: isto \emph{desacelera} em linha reta;
  \item $\vect F $ perpendicular a $\vect v $: isto \emph{voltas} sem
        mudando a velocidade.
\end{enumerate}
\end{proposition}

\begin{proof}
Adicionar $\Delta\vect v $, dirigido junto $\vect F $
(\cref{thm:g11:forces-and-motion:secondlaw}), ponta a ponta
$\vect v $: junto $\vect v $ as flechas se acumulam (mais longas); oposto, eles
cancelar parcialmente (mais curto); perpendicular, cada pequeno $\Delta \vect v $
inclina a seta sem alongá-la --- apenas a direção gira.
\end{proof}

\begin{example}[Um chute lateral]\label{ex:g11:forces-and-motion:kick}
Um disco desliza para o leste em \qty{6.0}{m/s}; um breve golpe lateral acrescenta
$\Delta\vect v = \qty{8.0}{m/s}$ devido ao norte. A nova velocidade é a
soma vetorial: magnitude $\sqrt{6.0^2 + 8.0^2} = \qty{10.0}{m/s}$,
direção $\tan\theta = 8.0/6.0$, ou seja\ $\theta \approx 53^\circ $ norte
do leste --- a velocidade antiga não é apagada, a mudança é adicionada a ela.
\end{example}

\section{Comparando mudanças: força, tempo, massa}

\begin{method}[Raciocínio de razão]\label{met:g11:forces-and-motion:ratio}
Para comparar duas situações, nunca resolva uma equação --- forme proporções: o
razão de mudança de velocidade é a $ F $ proporção, vezes o $\Delta t $ proporção,
dividido pelo $ m $ razão. Em seguida, leia qual fator mudou e em quanto.
\end{method}

\begin{example}[O carrinho carregado]\label{ex:g11:forces-and-motion:cart}
O mesmo push, aplicado no mesmo \omterm{def:g10:orders-of-magnitude:unit}{segundo}, envia um vazio
\qty{20}{kg} carrinho de compras em \qty{1.2}{m/s}. Carregado para
\qty{60}{kg} --- massa tripla, igual $ F\Delta t $ --- sai em um
terço disso, \qty{0.40}{m/s}. A massa é a teimosia da matéria.
\end{example}

\begin{remark}[Acidentes, airbags, joelhos dobrados]\label{rem:g11:forces-and-motion:airbag}
Em um acidente $\Delta v $ e $ m $ não são negociáveis --- o ocupante
a velocidade deve chegar a zero --- então $ F\,\Delta t $ é fixo. A única liberdade
esquerda é \emph{tempo}: esticar $\Delta t $ e o \omterm{def:g10:inertia:force}{vigor} encolhe no
mesma proporção. Uma cabeça parada em um airbag em \qty{0.08}{s} em vez de ligado
a roda em \qty{0.01}{s} sente uma média \omterm{def:g10:inertia:force}{vigor} oito vezes menor;
zonas de deformação, espuma do capacete e joelhos dobrados fazem o mesmo truque.
\end{remark}

\section{Fricção: a força que combate o deslizamento}

\begin{definition}[Fricção estática e cinética]\label{def:g11:forces-and-motion:friction}
\omterm{def:g10:inertia:inventory}{Atrito} entre superfícies em contato se opõem à sua \emph{relativo
deslizando}. \emph{Fricção estática}\index{Fricção estática} age enquanto eles
não deslize, ajustando-se ao máximo para cancelar qualquer tentativa
para iniciar o slide; \emph{atrito cinético}\index{atrito cinético},
uma vez iniciado o deslizamento, é aproximadamente constante, direcionado contra o
deslizando e geralmente \emph{mais fraco} do que o máximo estático.
\end{definition}

\begin{example}[Empurrando o guarda-roupa]\label{ex:g11:forces-and-motion:wardrobe}
Empurre um guarda-roupa suavemente: ele não se move --- estático \omterm{def:g10:inertia:inventory}{atrito} cancela
seu \qty{150}{N}. Empurre com mais força: em alguns \qty{400}{N} ele entra em
movimento, então desliza com menos esforço --- o mais fraco \omterm{def:g11:forces-and-motion:friction}{atrito cinético}
assumiu.
\end{example}

\begin{remark}[Por que os freios ABS funcionam]\label{rem:g11:forces-and-motion:abs}
O ponto de contato de uma roda em movimento não desliza na estrada: o pneu
agarra com \emph{estático} \omterm{def:g10:inertia:inventory}{atrito}. Uma roda travada derrapa e atinge apenas
o mais fraco \emph{cinético} \omterm{def:g10:inertia:inventory}{atrito}, que aponta contra a derrapagem
o que quer que as rodas façam --- a direção morre. O ABS libera o freio
instantaneamente uma roda trava: mais aderência, parada mais curta, um motorista que consegue dirigir.
\end{remark}

\section{Torneamento: o movimento circular precisa de uma força interna}

\begin{definition}[Movimento circular uniforme]\label{def:g11:forces-and-motion:ucm}
Um corpo está em \emph{movimento circular uniforme}\index{movimento circular uniforme} quando é \omterm{def:g10:relative-motion:trajectory}{trajetória} é um círculo percorrido com velocidade constante. O
velocidade, tangente ao círculo, muda de direção a cada instante ---
então nunca é constante.
\end{definition}

\begin{proposition}[A força interior]\label{prop:g11:forces-and-motion:inward}
Um corpo em \omterm{def:g11:forces-and-motion:ucm}{movimento circular uniforme} está sujeito a um \omterm{def:g11:forces-and-motion:netforce}{força resultante} isso é
nunca zero: a cada instante aponta do corpo para o centro
do círculo.
\end{proposition}

\begin{proof}
A velocidade é constante, então o \omterm{def:g11:forces-and-motion:netforce}{força resultante} não tem nenhuma parte junto $\vect v $
(casos 1 e 2 de \cref{prop:g11:forces-and-motion:cases}): é
perpendicular à tangente, portanto ao longo do raio --- e para dentro,
já que cada $\Delta \vect v $ curva o caminho em direção ao centro.
\end{proof}

\begin{example}[A corda, a curva, a Lua]\label{ex:g11:forces-and-motion:moon}
Cada curva constante esconde um interior \omterm{def:g10:inertia:force}{vigor}. A bola girada: o barbante
\omterm{def:g11:circuits-and-power:voltage}{tensão}. O carro em uma curva: a estática lateral \omterm{def:g10:inertia:inventory}{atrito} da estrada em
pneus --- ficaram no gelo e o carro seguiu em frente. A Lua: a da Terra
atração gravitacional (\cref{ch:g10:universal-gravitation}), girando-o
por aí \num{27.3} dias --- não realizado \emph{acima} mas puxado para o lado em um
giro perpétuo, caindo ao redor da Terra; no próximo ano faz deste o
mecânica de \omterm{def:g10:universal-gravitation:satellite}{satélites}.
\end{example}

\section{Problema: o teste de colisão}

\begin{problem}[{Problema de fim de semana --- projetando a colisão suave:
já que um acidente corrige a mudança de velocidade, todos os dispositivos de segurança em um carro
é uma máquina para esticar o tempo}]
\label{pb:g11:forces-and-motion:1}
Um laboratório de testes de colisão dirige um carro em \qty{50}{km/h} em um rígido
parede, com um \qty{70}{kg} manequim adulto e um \qty{20}{kg} boneco de criança
a bordo; as câmeras cronometram cada parada. Seu trabalho: descobrir onde está a violência de um
colidir com vidas e projetá-las.

\medskip
\noindent\textbf{Part I --- The measure of a crash.}
\begin{enumerate}
  \item Converter \qty{50}{km/h} para \unit{m/s}.
  \item Em qualquer parada frontal a mudança de velocidade dos ocupantes tem a mesma
        magnitude. Qual deles? Por que nenhum dispositivo pode alterá-lo?
  \item Citar \cref{thm:g11:forces-and-motion:secondlaw}: com $\Delta v $
        e $ m $ fixo, $ F\,\Delta t $ é fixo. O único mostrador do designer?
  \item A parede para o carro \qty{0.050}{s}; uma zona de deformação
        estende isso para \qty{0.150}{s}. Compare a média forças.
  \item Em que direção fazer $\Delta \vect v $ e o \omterm{def:g11:forces-and-motion:netforce}{força resultante} ligado
        os manequins apontam durante a parada?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crumple zone.}
\begin{enumerate}[resume]
  \item Por que, em uma frase, os engenheiros projetam a frente do carro
        ser destruído?
  \item Em um carro rígido antigo (pare em \qty{0.050}{s}) o manequim está amarrado
        rigidamente ao assento. Compare seu \omterm{def:g10:inertia:force}{vigor} com o moderno \qty{0.150}{s} parar.
  \item A velocidade do carro que para, cai continuamente (média: metade da inicial).
        Quão longe ele viaja \qty{0.150}{s}? Compare com uma zona de deformação real.
  \item Por que não um \qty{10}{m} zona de deformação? E qual parte do carro
        deve \emph{não} amassar?
  \item Resuma a Parte II como um slogan: você não pode escolher $\Delta v $,
        então você deve escolher \dots''.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The belt, the airbag and the child.}
\begin{enumerate}[resume]
  \item Um manequim sem cinto mantém seu \qty{14}{m/s} (\cref{ch:g10:inertia})
        até que o para-brisa o pare \qty{0.005}{s}; um cinto
        para com o carro dentro \qty{0.150}{s}. Compare a média forças.
  \item Por que um cinto que estica um pouco melhor do que um
        arnês de aço inextensível?
  \item A cabeça parava no volante em \qty{0.010}{s}; o airbag
        estende isso para \qty{0.080}{s}. Compare o forças.
  \item Adulto e criança passam pelo mesmo $\Delta v $ no mesmo
        $\Delta t $. Compare o forças líquidas as duas correias devem fornecer.
  \item Uma criança no colo: os braços devem mudar \qty{10}{kg} bebê
        velocidade por \qty{14}{m/s} em \qty{0.150}{s}, enquanto a gravidade
        altera uma velocidade em \qty{9.8}{m/s} cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}. Quantas vezes
        mais rápido os braços devem agir --- segurando que massa sentiria isso
        pesado? Concluir.
  \item As crianças se sentem menores forças --- então por que cadeiras infantis? Pense em
        \emph{onde} o cinto aplica seu \omterm{def:g10:inertia:force}{vigor}, e de $\Delta t$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The safety cage, and the verdict.}
\begin{enumerate}[resume]
  \item A célula do passageiro fica rígida enquanto ambas as extremidades se dobram. Por que deve
        a célula não se deforma, embora a rigidez signifique paradas violentas?
  \item Dois carros idênticos colidem frontalmente, cada um a uma \qty{50}{km/h}.
        Onde cada um para? Compare cada colisão com o teste de parede.
  \item Um carro pequeno bate de frente com um caminhão carregado. O forças eles exercem
        uns sobre os outros são iguais e opostos, mútuos como toda interação
        é (\cref{ch:g11:fundamental-interactions}). Cuja velocidade
        muda mais?
  \item Liste os dispositivos de segurança deste problema e a quantidade única
        cada um deles se estica.
  \item Finale: declare a lei de projeto da colisão suave em um
        frase usando apenas \emph{massa}, \emph{mudança de velocidade},
        \emph{\omterm{def:g10:inertia:force}{vigor}} e \emph{tempo}.
\end{enumerate}
\end{problem}
