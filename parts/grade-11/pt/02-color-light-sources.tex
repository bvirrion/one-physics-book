\chapter{Fontes de cor e luz}\label{ch:g11:color-light-sources}

Pergunte a um físico qual a cor de um tomate e você receberá uma contra-pergunta:
sob que luz? O tomate só \omterm{def:g11:color-light-sources:objectcolor}{difunde} o que é dado, a lâmpada
só dá o que é \omterm{def:g10:light-spectra:white}{espectro} contém, e o olho comprime tudo o que
chega em três números. A cor é uma negociação entre fonte,
objeto e olho --- este capítulo interroga todas as três partes por vez.

\section{Vendo em três números}

\begin{definition}[Cones e visão tricromática]\label{def:g11:color-light-sources:cones}
O \omterm{def:g11:lenses-and-eye:eye}{retina} (\cref{ch:g11:lenses-and-eye}) carrega duas famílias de
células sensíveis à luz: as \emph{varas}\index{varas}, que respondem a
luz fraca, mas ignore \omterm{def:g10:light-spectra:wavelength}{comprimento de onda} (visão noturna, em cinza), e o
\emph{cones}\index{cones}, em três tipos mais sensíveis perto
$\qty{420}{nm}$ (azul), $\qty{530}{nm}$ (verde) e $\qty{560}{nm}$
(vermelho). A visão humana das cores é
\emph{tricromático}\index{tricromático}: seja qual for o \omterm{def:g10:light-spectra:white}{espectro},
o cérebro recebe apenas três respostas cônicas, e a cor é o que ele
faz desse triplo.
\end{definition}

\begin{definition}[Cor como percepção]\label{def:g11:color-light-sources:perception}
A cor é, portanto, um \emph{percepção}, não é uma propriedade apenas da luz:
um \omterm{def:g10:light-spectra:white}{espectro} especifica uma intensidade em cada \omterm{def:g10:light-spectra:wavelength}{comprimento de onda}, o olho mantém
três números. Luzes com \omterm{def:g10:light-spectra:white}{espectros} diferentes, mas os mesmos três cone
as respostas parecem estritamente as mesmas: são \emph{metâmeros}\index{metâmeros}.
\end{definition}

\begin{example}[Dois amarelos]\label{ex:g11:color-light-sources:yellows}
O $\qty{589}{nm}$ linha de uma lâmpada de sódio excita o vermelho e o verde
\omterm{def:g11:color-light-sources:cones}{cones} quase igualmente: o cérebro diz \emph{amarelo}. Uma iluminação de tela
é vermelho ($\qty{610}{nm}$) e verde ($\qty{540}{nm}$) subpixels dá
o mesmo amarelo sem luz próxima $\qty{589}{nm}$. Um prisma separa
os dois instantaneamente; o olho nunca pode.
\end{example}

\section{Adicionando luzes}

\begin{definition}[Síntese aditiva]\label{def:g11:color-light-sources:additive}
A superposição de feixes de luz em uma tela branca adiciona seus \omterm{def:g10:light-spectra:white}{espectros}: isto é
\emph{síntese aditiva}\index{síntese aditiva}. Três bem escolhidos
luzes, vermelho, verde e azul --- o \emph{aditivo
primárias}\index{aditivo
primárias} --- misture em quase todos os perceptíveis
cor: R $+$ G $=$ amarelo, R $+$ B $=$ magenta, G $+$ B $=$ ciano e
R $+$ G $+$ B $=$ branco.
\end{definition}

\begin{proposition}[Cores complementares]\label{prop:g11:color-light-sources:complementary}
Duas luzes cuja soma é percebida como branca são
\emph{complementar}\index{complementar}. Os três pares
amarelo/azul, ciano/vermelho e magenta/verde são \omterm{prop:g11:color-light-sources:complementary}{complementares}.
\end{proposition}

\begin{proof}
Amarelo é R $+$ G; adicionar azul completa o triplo do branco. O outro
dois pares são o mesmo argumento, um primário de cada vez.
\end{proof}

\begin{example}[Telas]\label{ex:g11:color-light-sources:screen}
Cada pixel da tela de um telefone tem três subpixels --- vermelho, verde, azul,
sob $\qty{0.1}{mm}$ largo --- muito perto para o olho separar, então
suas luzes adicionam \omterm{def:g11:lenses-and-eye:eye}{retina}: amarelo é R e G aceso, laranja é R em
cheio poder e G na metade, o branco são os três. Uma \omterm{def:g11:lenses-and-eye:magnifier}{lupa} em um
a área branca revela o truque: nenhum branco em lugar nenhum, apenas R, G, B.
\end{example}

\section{Subtraindo da luz branca}

\begin{definition}[Síntese subtrativa]\label{def:g11:color-light-sources:subtractive}
UM \emph{filtro}\index{filtro} (vidro colorido, um gel, uma camada de tinta ou
tinta) transmite parte do \omterm{def:g10:light-spectra:white}{espectro} ele recebe e absorve o resto.
Produzindo cores por \emph{removendo} bandas de \omterm{def:g10:light-spectra:white}{luz branca} é
\emph{síntese subtrativa}\index{síntese subtrativa}. É
primárias, complementos das aditivas, cada uma subtrai uma banda:
ciano $=$ branco $-$ R, magenta $=$ branco $-$ G, amarelo $=$ branco $-$ B.
Filtros empilhados subtraem conjuntamente: ciano $+$ folhas amarelas verdes, ciano
$+$ azul magenta, magenta $+$ amarelo vermelho, todos os três pretos. Impressão
e pintura assim, com tintas C, M, Y.
\end{definition}

\begin{method}[Escrituração de espectro]\label{met:g11:color-light-sources:bookkeeping}
Para prever uma cor através de qualquer cadeia de fontes, \omterm{def:g11:color-light-sources:subtractive}{filtros} e objetos,
rastreie as três bandas primárias: escreva a luz da fonte como seu conteúdo
em R, G, B; em cada \omterm{def:g11:color-light-sources:subtractive}{filtro} ou superfície, eliminar as bandas absorvidas; nome
a cor do que sobrevive (nada resta $=$ preto).
\end{method}

\begin{example}[Uma cadeia de filtros]\label{ex:g11:color-light-sources:chain}
\omterm{def:g10:light-spectra:white}{Luz branca} $(R, G, B)$ cruza um amarelo \omterm{def:g11:color-light-sources:subtractive}{filtro}: a faixa azul é
absorvido, deixando $(R, G)$ --- amarelo. Um ciano \omterm{def:g11:color-light-sources:subtractive}{filtro} então absorve o
faixa vermelha, saindo $(G)$: verde. Trocando o \omterm{def:g11:color-light-sources:subtractive}{filtros} não muda nada:
cada um absorve sua própria banda, independentemente da ordem.
\end{example}

\section{A cor de um objeto}

\begin{definition}[Difusão e cor do objeto]\label{def:g11:color-light-sources:objectcolor}
Um objeto opaco absorve parte da luz que recebe e
\emph{difunde}\index{difunde} o resto --- reemite em todos
direção, em direção aos nossos olhos, entre outros. O \emph{cor de um
objeto}\index{cor de um
objeto} é a cor da luz que ela difunde
\emph{sob a iluminação dada}: o objeto fornece sua absorção
hábitos, a fonte a matéria-prima. Troque a lâmpada, mude a cor.
\end{definition}

\begin{example}[A camisa vermelha]\label{ex:g11:color-light-sources:redshirt}
Uma camisa parece vermelha à luz do dia: ela absorve as faixas verdes e azuis e
\omterm{def:g11:color-light-sources:objectcolor}{difunde} o vermelho. Sob uma luz verde pura não há faixa vermelha para
difusa: a camisa absorve tudo e fica preta. Sob magenta
luz $(R, B)$ isto \omterm{def:g11:color-light-sources:objectcolor}{difunde} a faixa vermelha sozinha: ainda vermelha. Um branco
objeto \omterm{def:g11:color-light-sources:objectcolor}{difunde} cada banda que recebe (sob luz verde parece
verde); um objeto preto absorve todos eles sob cada luz --- o que é
também por que as roupas pretas esquentam ao sol.
\end{example}

\section{Fontes de luz e temperatura de cor}

\begin{definition}[Fontes incandescentes]\label{def:g11:color-light-sources:incandescent}
Uma fonte que brilha porque é quente --- chama, filamento, estrela --- brilha
por \emph{incandescência}\index{incandescência}. Isso é \omterm{def:g10:light-spectra:white}{espectro} é o
térmico contínuo \omterm{def:g10:light-spectra:white}{espectro} de \cref{ch:g10:light-spectra}, e seu pico
obedece à lei de Wien (\cref{prop:g10:light-spectra:wien}),
\[
\lambda_{\max} \times T = \qty{2.90e-3}{m.K},
\]
nossa ferramenta quantitativa a partir de agora: pico medido, temperatura conhecida.
\end{definition}

\begin{example}[Uma lâmpada muito ineficiente]\label{ex:g11:color-light-sources:bulb}
Um filamento de tungstênio em $ T = \qty{2700}{K}$ picos em
$\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{2700}{K} = \qty{1.07e-6}{m}
= \qty{1070}{nm}$, profundamente no \omterm{def:g10:light-spectra:wavelength}{infravermelho}: a maior parte da eletricidade poder
sai como radiação invisível, apenas cerca de $5\%$ como luz --- o
a lâmpada incandescente era principalmente um aquecedor.
\end{example}

\begin{definition}[Lâmpadas espectrais e lasers]\label{def:g11:color-light-sources:lamps}
UM \emph{lâmpada espectral}\index{lâmpada espectral} --- um baixa \omterm{def:g10:pressure:pressure}{pressão} gás
excitado por uma descarga: sódio, mercúrio, néon --- emite um \omterm{def:g10:light-spectra:emission}{espectro de linha}
(\cref{ch:g10:light-spectra}): alguns isolados comprimentos de onda, nada em
entre. UM \emph{laser}\index{laser} vai além: um único
\omterm{def:g10:light-spectra:wavelength}{comprimento de onda}, luz monocromática, em um feixe estreito.
\end{definition}

\begin{definition}[LEDs]\label{def:g11:color-light-sources:led}
UM \emph{diodo emissor de luz}\index{diodo emissor de luz} (LED) emite um
\emph{quase monocromático}\index{quase monocromático} banda com algumas dezenas de nanômetros de largura: estreita
o suficiente para parecer uma cor definida, muito mais ampla do que um \omterm{def:g11:color-light-sources:lamps}{laser} linha. Um branco
LED é um LED azul (pico próximo $\qty{450}{nm}$) revestido com um
\emph{fósforo}\index{fósforo}, um material que absorve parte do azul
e reemitindo-o como uma larga faixa amarela: azul e amarelo sendo
\omterm{prop:g11:color-light-sources:complementary}{complementar} (\cref{prop:g11:color-light-sources:complementary}), o
a mistura parece branca.
\end{definition}

\begin{definition}[Temperatura de cor]\label{def:g11:color-light-sources:colortemp}
O \emph{temperatura de cor}\index{temperatura de cor} de um branco
fonte é a temperatura do corpo incandescente cuja tonalidade sua luz
\omterm{def:g11:color-light-sources:led}{fósforos}: vela $\qty{1800}{K}$, lâmpada incandescente $\qty{2700}{K}$,
halogênio $\qty{3400}{K}$, sol do meio-dia $\qty{5500}{K}$, luz do dia nublado
$\qty{6500}{K}$. É uma etiqueta colorida, não uma leitura de termômetro --- um
$\qty{6500}{K}$ O LED funciona à temperatura ambiente --- e o vocabulário funciona
ao contrário: baixa temperatura de cor indica laranja quente, alta fria
azulado.
\end{definition}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:color-light-sources:1}
Dê a cor percebida quando uma tela branca recebe simultaneamente:
(a) luz vermelha e verde; (b) luz verde e azul; (c) todos os três
primárias. Qual luz é \omterm{prop:g11:color-light-sources:complementary}{complementar} ao azul?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:color-light-sources:2}
Quais subpixels (R, G, B) uma tela acende para exibir: amarelo,
magenta, branco, preto, laranja?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:color-light-sources:3}
\omterm{def:g10:light-spectra:white}{Luz branca} cruza um magenta \omterm{def:g11:color-light-sources:subtractive}{filtro}. Qual banda é absorvida e qual
surge a cor? Um ciano \omterm{def:g11:color-light-sources:subtractive}{filtro} é então adicionado atrás dele: o que surge agora?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:color-light-sources:4}
Sob \omterm{def:g10:light-spectra:white}{luz branca} uma bandeira mostra uma faixa amarela. Quais bandas fazem isso
difuso, o que ele absorve? Sua cor em (a) vermelho; (b) luz azul?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:color-light-sources:5}
Uma estrela \omterm{def:g10:light-spectra:continuous}{espectro contínuo} picos em $\lambda_{\max} = \qty{580}{nm}$.
Calcule sua temperatura superficial.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:color-light-sources:6}
A chama de uma vela fica a cerca de $\qty{1800}{K}$. (a) Calcular
$\lambda_{\max}$ e dê seu domínio. (b) A chama, no entanto, parece
laranja: explique.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:color-light-sources:7}
Um ciclista usa uma jaqueta vermelha e luvas ciano. Usando
\cref{met:g11:color-light-sources:bookkeeping}, dê a cor aparente
de cada item em (a) \omterm{def:g10:light-spectra:white}{luz branca}; (b) luz vermelha; (c) luz verde.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:color-light-sources:8}
Uma rua antiga é iluminada por baixa \omterm{def:g10:pressure:pressure}{pressão} lâmpadas de sódio (uma única linha em
$\qty{589}{nm}$). (a) Que tipo de \omterm{def:g10:light-spectra:white}{espectro} é isso e quais cores
pode \emph{qualquer} objeto tomar sob ele? (b) Um carro parece preto sob estes
lâmpadas e azul à luz do dia: explique.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:color-light-sources:9}
Um verde \omterm{def:g11:color-light-sources:lamps}{laser} ponteiro emite em $\qty{532}{nm}$. (a) Seu feixe atravessa uma
prisma: compare o resultado com o de um feixe branco e nomeie o
propriedade ilustrada. (b) Ele então encontra um vermelho \omterm{def:g11:color-light-sources:subtractive}{filtro}: o que sai?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:color-light-sources:10}
Descreva o \omterm{def:g10:light-spectra:white}{espectro} de um LED branco (diodo azul mais \omterm{def:g11:color-light-sources:led}{fósforo}) e
explique por que sua luz é percebida como branca. Por que um objeto vermelho escuro
parece mais opaco sob um LED branco barato do que à luz do dia?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:color-light-sources:11}
Calcular $\lambda_{\max}$ para um corpo incandescente em (a) uma vela
\omterm{def:g11:color-light-sources:colortemp}{temperatura de cor} ($\qty{1800}{K}$); (b) dia nublado
($\qty{6500}{K}$). Qual das duas luzes é chamada de quente? Comentário.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:color-light-sources:12}
Uma lâmpada de sódio ($\qty{589}{nm}$) e uma tela exibindo amarelo
(subpixels em $\qty{610}{nm}$ e $\qty{540}{nm}$) parecem exatamente iguais
cor. (a) Como são chamadas duas dessas luzes e qual é a sua função?
existência prova sobre cor? (b) Proponha dois experimentos diferentes que
diferencie-os e preveja o resultado de cada um.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:color-light-sources:13}
Um spot de halogênio de teatro funciona em $\qty{3200}{K}$. (a) Calcular
$\lambda_{\max}$: esta luz é mais quente ou mais fria que a luz do dia? (b) Para
imita a luz do dia, a mancha é coberta por um gel azul claro. O que faz
o gel é removido e por que o palco fica mais escuro? (c) Explique por que não
gel poderia em vez disso \emph{fortalecer} a extremidade azul do local \omterm{def:g10:light-spectra:white}{espectro}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:color-light-sources:14}
Um açougueiro ilumina o \omterm{def:g10:orders-of-magnitude:scinot}{expositor} de carnes com LEDs de \omterm{def:g11:color-light-sources:colortemp}{temperatura de cor}
$\qty{2700}{K}$; a peixaria do lado usa $\qty{6500}{K}$.
(a) Descreva a tonalidade de cada iluminação e seu efeito na carne vermelha e no
peixe branco. (b) Algum dos comerciantes trocou suas mercadorias? Diga exatamente
onde a mudança acontece (\cref{def:g11:color-light-sources:objectcolor}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:color-light-sources:15}
Uma cidade próxima a um observatório deve reacender suas ruas e hesita
entre baixa \omterm{def:g10:pressure:pressure}{pressão} lâmpadas de sódio e amplo \omterm{def:g10:light-spectra:white}{espectro} LEDs brancos.
(a) Que escolha preferem os astrónomos e porquê? (Pense no que
solteiro \omterm{def:g11:color-light-sources:subtractive}{filtro} podem remover de suas imagens.) (b) Qual escolha
motoristas e pedestres preferem? Justifique com as cores dos objetos.
(c) Propor um compromisso, formulado com \omterm{def:g11:color-light-sources:colortemp}{temperatura de cor}.
\end{exercise}

\section{Problema: cor de engenharia}

\begin{problem}[{Problema de fim de semana --- luzes do palco, ruas de sódio e
o LED branco: como a cor é projetada, a partir de três holofotes que podem
pinte qualquer matiz no chip que ilumina esta página}]
\label{pb:g11:color-light-sources:1}
Um designer de iluminação possui apenas holofotes vermelhos, verdes e azuis; uma rua
engenheiro possui uma lâmpada que emite um único \omterm{def:g10:light-spectra:wavelength}{comprimento de onda}; um fabricante de chips possui
um diodo que brilha apenas em azul. Todos os três vendem cores. Este problema é executado
seus negócios, por sua vez --- mistura aditiva no palco, economia monocromática em
a rua, \omterm{def:g11:color-light-sources:led}{fósforo} alquimia no chip --- e termina escolhendo o
luz para um quarto e para uma janela de joalheria.

\medskip
\noindent\textbf{Part I --- The stage: painting with three spotlights.}
\begin{enumerate}
  \item Os três pontos se sobrepõem em um fundo branco. Nomeie a cor de
        cada sobreposição de pares e da sobreposição tripla.
  \item Como o designer faz o laranja? E um rosa claro? (Descreva
        o local poderes qualitativamente.)
  \item Uma dançarina está diante do cenário, iluminada pelo vermelho e verde
        apenas manchas. Aparecem duas sombras coloridas: cor de cada uma? Explicar.
  \item O cenário é repintado de azul profundo. Como é
        sob a mancha vermelha sozinho? Sob todos os três?
  \item Por que três manchas podem reproduzir quase todas as cores? Qual
        pedaço de anatomia fixa o número três?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Costumes: subtraction at work.}
\begin{enumerate}[resume]
  \item Uma fantasia é amarela em \omterm{def:g10:light-spectra:white}{luz branca}. Qual banda faz sua tintura
        absorver?
  \item Preveja sua cor sob a mancha vermelha, a mancha verde, a azul
        local.
  \item O estilista quer que o traje mude de brilhante para preto em
        um toque de um botão. Para qual ponto o filme muda?
  \item Um gel magenta é deslizado na frente de um ponto branco iluminando o
        traje amarelo. Acompanhe as bandas
        (\cref{met:g11:color-light-sources:bookkeeping}): cor do traje?
  \item Por que os géis de palco vêm em ciano, magenta e amarelo em vez de
        vermelho, verde e azul? (O que um gel vermelho faria com uma mancha branca
        poder?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Sodium streets: the monochrome economy.}
\begin{enumerate}[resume]
  \item UM baixa \omterm{def:g10:pressure:pressure}{pressão} lâmpada de sódio emite essencialmente uma linha em
        $\qty{589}{nm}$. Que tipo de \omterm{def:g10:light-spectra:white}{espectro} é isso, e o que significa
        toda a rua parece embaixo dela?
  \item Sob esta lâmpada, dê a cor aparente de uma parede branca, um
        marcação rodoviária amarela, um carro vermelho, um carro azul.
  \item Por que os astrônomos adoraram essas lâmpadas? (O que alguém pode \omterm{def:g11:color-light-sources:subtractive}{filtro} fazer
        para o brilho de toda a cidade?)
  \item Uma lâmpada de rua de tungstênio em $\qty{2700}{K}$ é proposto em seu lugar.
        Calcule seu $\lambda_{\max}$: onde está a maior parte de sua poder ir?
  \item A lâmpada de sódio converte cerca de $30\%$ da sua elétrica poder
        em luz visível, a lâmpada de tungstênio cerca de $5\%$. Quantos
        \omterm{def:g10:energy-conservation:power}{watts} de luz cada um fornece por $\qty{100}{W}$ de poder?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The white LED and the choice of a light.}
\begin{enumerate}[resume]
  \item Um LED branco é um diodo azul ($\qty{450}{nm}$) sob um amarelo
        \omterm{def:g11:color-light-sources:led}{fósforo}. Explique, com cores \omterm{prop:g11:color-light-sources:complementary}{complementares}, por que sua saída
        lê como branco.
  \item Descreva seu \omterm{def:g10:light-spectra:white}{espectro} e compare-o com o contínuo
        \omterm{def:g10:light-spectra:white}{espectro} da luz do dia. Qual banda está sub-representada?
  \item Deduza quais objetos tal LED renderiza mal e conecte
        isso com o açougueiro de \cref{exo:g11:color-light-sources:14}.
  \item Uma lâmpada LED é vendida como ``$\qty{2700}{K}$''. Tem alguma coisa dentro
        isso em $\qty{2700}{K}$? O que o rótulo promete? Calcule o
        $\lambda_{\max}$ do corpo incandescente que ele imita.
  \item Finale --- escolha, com uma frase de justificativa cada, uma
        \omterm{def:g11:color-light-sources:colortemp}{temperatura de cor} para uma luminária de quarto e outra para uma joalheria
        janela; em seguida, declare a lei das cores do capítulo em uma linha.
\end{enumerate}
\end{problem}
