\chapter{Campos Elétricos e Gravitacionais}\label{ch:g11:electric-gravitational-fields}

Como a Terra puxa uma maçã que nunca toca? O próprio Newton ligou
ação à distância um absurdo tão grande que ele se recusou a defender
isso. A resposta moderna: uma massa, ou carga, modifica o espaço ao seu redor
--- cria um \emph{campo} --- e tudo o que for colocado lá responde
para o campo em seu próprio local. Construímos os dois grandes exemplos
lado a lado, até que a analogia se torne uma identidade de forma.

\section{Da ação à distância para o campo}

\begin{definition}[Campo]\label{def:g11:electric-gravitational-fields:field}
UM \emph{campo}\index{campo} atribui a cada ponto do espaço um vetor. Um
fonte (uma carga, uma massa) cria seu campo em todos os lugares ao seu redor,
se há ou não algo para senti-lo; um objeto de teste então sente uma
\omterm{def:g10:inertia:force}{vigor} determinado pelo campo \emph{em seu próprio local} sozinho. Isto
a imagem em duas etapas substitui a ação à distância.
\end{definition}

\begin{remark}
O campo não é um truque de contabilidade: carrega energia, e mudanças em
a fonte viaja para fora através dela a uma velocidade finita --- a própria luz é
uma viagem dessas campo (\cref{ch:g12:light-as-wave}). Aqui estudamos
campos que não mudam com o tempo.
\end{remark}

\section{O campo elétrico}

\begin{definition}[Campo elétrico]\label{def:g11:electric-gravitational-fields:efield}
Se uma pequena carga de teste $q $ colocado em um ponto $ M $ sente uma eletricidade \omterm{def:g10:inertia:force}{vigor}
$\vect F $, o \emph{campo elétrico}\index{campo elétrico} no $ M $ é
\[
\vect E = \frac{\vect F}{q}
\qquad \text{(in \unit{N/C}, also written \unit{V/m})}.
\]
$\vect E $ depende apenas das fontes e $ M $, não no teste
cobrança: duplicação $ q $ dobra $\vect F $, deixando o quociente inalterado.
\end{definition}

\begin{proposition}[Forçar uma carga]\label{prop:g11:electric-gravitational-fields:force}
Uma cobrança $ q $ colocado onde o campo é $\vect E $ sente o \omterm{def:g10:inertia:force}{vigor}
$\vect F = q \vect E $: junto $\vect E $ se $ q > 0$, oposto a $\vect E $
se $ q < 0$, de magnitude $ F = \abs{q}\,E $.
\end{proposition}

\begin{proof}
Reorganize a definição; a regra dos sinais é a álgebra de um múltiplo negativo.
\end{proof}

\begin{example}[Lendo um campo]\label{ex:g11:electric-gravitational-fields:reading}
Uma carga de teste $ q = \qty{+2.0e-8}{C}$ no $ M $ sente um \omterm{def:g10:inertia:force}{vigor} de
\qty{6.0e-4}{N} apontando para o leste. O campo no $ M $ é
$ E = \num{6.0e-4} / \num{2.0e-8} = \qty{3.0e4}{N/C}$, apontando para leste;
$ q' = \qty{-1.0e-8}{C}$ colocado no mesmo ponto parece
$ F = \num{1.0e-8} \times \num{3.0e4} = \qty{3.0e-4}{N}$ --- apontando para oeste.
\end{example}

\begin{proposition}[Campo de uma carga pontual]\label{prop:g11:electric-gravitational-fields:point}
Uma carga pontual $ Q $ cria, à distância $ d $, um campo ao longo da linha
juntando a carga ao ponto --- longe de $ Q $ se $ Q > 0$, em direção a ele se
$ Q < 0$ --- de magnitude
\[
E = k\,\frac{\abs{Q}}{d^2},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{proposition}

\begin{proof}
Lei de Coulomb (\cref{ch:g11:fundamental-interactions}) dá o \omterm{def:g10:inertia:force}{vigor}
em uma carga de teste $ q $ à distância $ d $: magnitude $ k \abs{qQ} / d^2$,
ao longo da linha de união, repulsivo para sinais semelhantes. Dividir por $ q $ e
aplicar \cref{prop:g11:electric-gravitational-fields:force}.
\end{proof}

\begin{example}[Uma esfera carregada]\label{ex:g11:electric-gravitational-fields:sphere}
Uma esfera de Van de Graaff carrega $ Q = \qty{+5.0e-7}{C}$ (uma pequena carga
esfera atua como uma carga pontual em seu centro). No $ d = \qty{0.50}{m}$:
\[
E = \num{8.99e9} \times \num{5.0e-7} / 0.50^2 \approx \qty{1.8e4}{N/C},
\]
apontando radialmente para longe da esfera. No \qty{1.0}{m} caiu
a um quarto disto: o $1/d^2$ do \omterm{def:g10:inertia:force}{vigor} sobrevive no campo.
\end{example}

\begin{definition}[Linhas de campo]\label{def:g11:electric-gravitational-fields:lines}
UM \emph{linha de campo}\index{linha de campo} é uma curva em todos os lugares tangente a
o campo, uma flecha dando o campodireção. Três regras:
\begin{itemize}
  \item saem de cargas positivas e chegam a cargas negativas;
  \item onde as linhas lotam o campo é \omterm{def:g11:fundamental-interactions:strong}{forte}; onde eles se espalham, fracos;
  \item duas linhas nunca se cruzam: a campo tem uma direção em cada ponto.
\end{itemize}
\end{definition}

\section{O campo uniforme de um capacitor}

\begin{definition}[Capacitor de placa paralela]\label{def:g11:electric-gravitational-fields:capacitor}
Duas placas metálicas paralelas voltadas uma para a outra, próximas uma da outra e transportando
encargos, formar um \emph{capacitor}\index{capacitor}. Entre as placas
o campo é \emph{uniforme}\index{uniforme}: mesma magnitude e
direção em todos os lugares --- perpendicular às placas, do positivo
placa para o negativo.
\end{definition}

\begin{proposition}[Campo entre as placas]\label{prop:g11:electric-gravitational-fields:uniform}
Se o \omterm{def:g11:circuits-and-power:voltage}{tensão} entre as placas (\cref{ch:g11:circuits-and-power}) é
$ U $ e sua separação é $ d $, o campo entre eles tem magnitude
\[
E = \frac{U}{d} \qquad \text{(in \unit{V/m})}.
\]
\end{proposition}
\admitted

\begin{remark}
Uniformidade e $ E = U/d $ são derivados honestamente no volume do Ano 1; o
fórmula explica o \omterm{def:g10:orders-of-magnitude:unit}{unidade} \unit{V/m}. \omterm{def:g11:lenses-and-eye:lens}{Afinar} lacunas fortalecem campos:
\qty{100}{V} através de um milímetro já é \qty{e5}{V/m}.
\end{remark}

\begin{example}[Direcionando um feixe de elétrons]\label{ex:g11:electric-gravitational-fields:beam}
Em um \omterm{def:g10:signals-and-waves:oscilloscope}{osciloscópio}, o feixe passa entre placas com
$ U = \qty{100}{V}$ e $ d = \qty{5.0}{mm}$:
$ E = 100 / \num{5.0e-3} = \qty{2.0e4}{V/m}$, e cada elétron sente
$ F = \num{1.60e-19} \times \num{2.0e4} = \qty{3.2e-15}{N}$ em direção ao
placa positiva --- $\num{4e14}$ vezes é \omterm{def:g10:universal-gravitation:weight}{peso} de \qty{8.9e-30}{N},
então o feixe desvia visivelmente (o caminho curvo é
\cref{ch:g12:newtons-laws}negócio).
\end{example}

\section{O campo gravitacional}

\begin{definition}[Campo gravitacional]\label{def:g11:electric-gravitational-fields:gfield}
Se uma massa de teste $ m $ colocado em um ponto parece um força gravitacional (um
\omterm{def:g10:universal-gravitation:weight}{peso}) $\vect P $, o \emph{campo gravitacional}\index{campo gravitacional} há
\[
\vect g = \frac{\vect P}{m} \qquad \text{(in \unit{N/kg})},
\]
e inversamente $\vect P = m \vect g $. Um capítulo anterior introduziu o
número $ g $; $\vect g $ adiciona a direção em que uma pedra caída começa a se mover.
\end{definition}

\begin{proposition}[Campo da Terra]\label{prop:g11:electric-gravitational-fields:earth}
À distância $ d $ do centro da Terra (massa $ M $), o gravitacional
campo aponta para o centro e tem magnitude
\[
g = G\,\frac{M}{d^2},
\qquad G = \qty{6.67e-11}{N.m^2/kg^2}.
\]
\end{proposition}

\begin{proof}
Universal \omterm{def:g11:fundamental-interactions:four}{gravitação} (\cref{ch:g10:universal-gravitation}) dá o
\omterm{def:g10:inertia:force}{vigor} em uma massa de teste $ m $: magnitude $ G M m / d^2$, em direção ao centro;
dividir por $ m $. Na superfície, $ d = R $ e este é o
$ g = GM/R^2 \approx \qty{9.81}{N/kg}$ do capítulo anterior.
\end{proof}

\begin{example}[O campo onde vive a Lua]\label{ex:g11:electric-gravitational-fields:moon}
À distância da Lua $ d = \qty{3.84e8}{m}$:
\[
g = \num{6.67e-11} \times \num{5.97e24} / (\num{3.84e8})^2
\approx \qty{2.7e-3}{N/kg}.
\]
da Terra campo nunca para --- apenas afina fora como $1/d^2$; este é o
campo que mantém a Lua em seu \omterm{def:g10:universal-gravitation:satellite}{órbita}.
\end{example}

\begin{remark}[Uniforme perto da superfície]\label{rem:g11:electric-gravitational-fields:uniformg}
Sobre um laboratório ou mesmo uma montanha, $ d $ quase não muda em comparação com
$ R = \qty{6.37e6}{m}$, então $\vect g $ é uniforme com alta precisão:
linhas verticais, constantes \qty{9.81}{N/kg} --- da Terra campo parece
como o \omterm{def:g11:electric-gravitational-fields:capacitor}{capacitor}'s, assim como a Terra redonda parece plana vista de um jardim.
\end{remark}

\section{Uma ideia, duas forças}

Os dois campos deste capítulo são a mesma matemática em dois trajes:

\begin{center}
\begin{tabular}{lcc}
 & \textbf{electric} & \textbf{gravitational} \\
\hline
source & charge $ Q $ & mass $ M $ \\
campo & $\vect E = \vect F / q $ & $\vect g = \vect P / m $ \\
\omterm{def:g10:orders-of-magnitude:unit}{unidade} & \unit{N/C} = \unit{V/m} & \unit{N/kg} \\
point source & $ E = k \abs{Q} / d^2$ & $ g = G M / d^2$ \\
constant & $ k = \qty{8.99e9}{N.m^2/C^2}$ & $ G = \qty{6.67e-11}{N.m^2/kg^2}$ \\
\omterm{def:g10:inertia:force}{vigor} on a test & $\vect F = q \vect E $ & $\vect P = m \vect g $ \\
linhas de campo & out of $+$, into $-$ & always into the mass \\
\end{tabular}
\end{center}

Cada fórmula à esquerda se torna a da direita sob o
dicionário $ q \leftrightarrow m $, $\vect E \leftrightarrow \vect g $, $ k \leftrightarrow G $.

\begin{remark}[Onde a analogia quebra]\label{rem:g11:electric-gravitational-fields:noanalogue}
A cobrança vem em dois sinais; massa em um. Então campos elétricos pode ser
cancelado: as cargas reorganizadas de uma caixa de metal eliminam qualquer
campo dentro dele (um \emph{Gaiola de Faraday} --- por que um elevador mata seu
telefone \omterm{def:g10:signals-and-waves:signal}{sinal}), e matéria a granel, com equilíbrio $+$ e $-$, é
eletricamente silencioso. A gravidade não pode ser protegida nem neutralizada:
cada \omterm{def:g10:orders-of-magnitude:unit}{quilograma} adiciona sua atração --- e é por isso que o \omterm{def:g10:inertia:force}{vigor} mais fraco por
$\num{e39}$ (\cref{exo:g11:electric-gravitational-fields:10}) executa o
universo em grandes escalas.
\end{remark}

\end{exercise}

\section{Problema: gota de Millikan}

\begin{problem}[{Problema de fim de semana --- pesando a carga do elétron: um
gota de óleo estacionada no ar entre duas placas revela essa carga
vem em etapas indivisíveis de $\num{1.60e-19}$ \omterm{def:g11:fundamental-interactions:charge}{Coulombs}}]
\label{pb:g11:electric-gravitational-fields:1}
Em 1909, Robert Millikan pulverizou gotas de óleo entre as camadas horizontais
placas de um \omterm{def:g11:electric-gravitational-fields:capacitor}{capacitor}. Ao sintonizar o \omterm{def:g11:circuits-and-power:voltage}{tensão} até que uma gota escolhida
pendurado imóvel --- elétrico \omterm{def:g10:inertia:force}{vigor} exatamente equilibrando \omterm{def:g10:universal-gravitation:weight}{peso} --- ele
poderia pesar sua carga. Nossa gota tem massa $ m = \qty{4.90e-15}{kg}$
(medido a partir de sua queda lenta com o campo desligado; nós tomamos isso como dado);
as placas são $ d = \qty{8.0}{mm}$ à parte, o \emph{superior} um
positivo; a carga da gota é negativa.

\medskip
\noindent\textbf{Part I --- The stage.}
\begin{enumerate}
  \item Descreva o campo entre as placas; o que significa ``uniforme'' comprar?
  \item Para $ U = \qty{800}{V}$, calcular $ E $.
  \item Em que direção está a eletricidade \omterm{def:g10:inertia:force}{vigor} na gota? Por que
        fez o \emph{superior} placa tem que ser positiva?
  \item Calcule a gota \omterm{def:g10:universal-gravitation:weight}{peso}.
  \item Escreva a condição de repouso como uma igualdade de dois \omterm{def:g10:inertia:force}{vigor}
        magnitudes.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The balance.}
\begin{enumerate}[resume]
  \item A partir dele, expresse $\abs{q}$ em termos de $ m $, $ g $, $ d $ e $ U $.
  \item A gota fica \omterm{def:g10:relative-motion:frame}{em repouso} por $ U = \qty{800}{V}$: calcular
        $\abs{q}$.
  \item Dividir por $ e = \qty{1.60e-19}{C}$. Quantos elétrons em excesso
        a gota carrega?
  \item O \omterm{def:g11:circuits-and-power:voltage}{tensão} é empurrado para cima \qty{900}{V}. Em que direção
        a rede \omterm{def:g10:inertia:force}{vigor} agora aponta?
  \item Mostre que o equilíbrio \omterm{def:g11:circuits-and-power:voltage}{tensão} é $ U = mgd/\abs{q}$: o
        quanto maior a carga, menor será \omterm{def:g11:circuits-and-power:voltage}{tensão}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The steps.}
Uma explosão de radiação pode lançar elétrons para dentro ou para fora da gota; depois
cada mudança, Millikan reajusta o \omterm{def:g11:circuits-and-power:voltage}{tensão} para equilibrar.
\begin{enumerate}[resume]
  \item Depois de uma explosão, o equilíbrio requer $ U = \qty{1200}{V}$. Calcular
        o novo $\abs{q}$. A gota ganhou ou perdeu um elétron?
  \item Os saldos sucessivos são encontrados em $ U = 1200$, $800$, $600$,
        $480$ e \qty{400}{V}. Calcule as cinco acusações.
  \item Mostre que todos os cinco são múltiplos inteiros de uma quantidade; dar
        seu valor.
  \item Entre saldos consecutivos, quanto custa a cobrança
        mudar? Interpretar.
  \item Explique por que essa gota nunca pode ficar parada \omterm{def:g10:relative-motion:frame}{em repouso}
        $ U = \qty{700}{V}$, por mais pacientemente que se sintonize.
  \item Millikan repetiu isso em centenas de gotículas: cada
        a carga era um múltiplo inteiro do mesmo valor. Indique o
        conclusão e nomeie a quantidade medida.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Why there is no gravitational Millikan.}
\begin{enumerate}[resume]
  \item No dicionário deste capítulo, o que desempenha o papel de $ q $, $ E $
        e $ qE $ no balanço da Parte II?
  \item Em vez disso, uma massa acima poderia segurar a gota? Calcule o
        campo de um \qty{1000}{kg} esfera de chumbo \qty{1.0}{m} acima do
        gota, o \omterm{def:g10:inertia:force}{vigor} na gota e compare com seu \omterm{def:g10:universal-gravitation:weight}{peso}.
  \item O equilíbrio precisa de um \omterm{def:g10:inertia:inventory}{impulso} ascendente \omterm{def:g11:fundamental-interactions:strong}{forte} o suficiente para combater o
        toda a Terra. Por que uma placa de cobrança pode fornecê-lo e não
        arranjo de massas pode?
  \item A gota \emph{massa} também muda em etapas (um óleo
        molécula: sobre \qty{5e-25}{kg}). Compare os saltos relativos em
        $ m $ e em $\abs{q}$ quando um elétron pousa, e concluo: por que
        esse equilíbrio vê o \omterm{def:g11:fundamental-interactions:ladder}{átomos} de eletricidade, mas não
        \omterm{def:g11:fundamental-interactions:ladder}{moléculas} de petróleo? Conclusão: a carga é quantizada e seu \omterm{def:g11:fundamental-interactions:ladder}{átomo}
        é $ e = \qty{1.60e-19}{C}$.
\end{enumerate}
\end{problem}
