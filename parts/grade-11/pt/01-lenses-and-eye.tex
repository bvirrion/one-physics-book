\chapter{Lentes, imagens e o olho}\label{ch:g11:lenses-and-eye}

Cada imagem nítida --- em um sensor de telefone, tela de cinema ou na parte traseira
do seu próprio olho --- vem do mesmo truque: uma curva transparente
meio curva os raios por refração (\cref{ch:g10:refraction}) para que todos
os raios que saem de um ponto se encontram novamente em outro. Este capítulo doma
o truque: um \omterm{def:g11:lenses-and-eye:lens}{lente}, três raios, uma fórmula --- então aponta para
câmeras, óculos e os vivos \omterm{def:g11:lenses-and-eye:lens}{lente} lendo esta frase.

\section{Lentes convergentes e divergentes}

\begin{definition}[Lente fina]\label{def:g11:lenses-and-eye:lens}
UM \emph{lente}\index{lente} é um meio transparente delimitado por dois
superfícies, pelo menos uma curva. Mais grosso no centro do que na borda,
é \emph{convergente}\index{convergente}: raios paralelos são dobrados
um em direção ao outro; mais grosso na borda, \emph{divergente}\index{divergente}: eles se espalharam. UM \emph{afinar}\index{afinar} lente tem
espessura desprezível: desenhamos um segmento com saída (convergente) ou
pontas de seta para dentro (divergentes). Seu centro $O $ é o \emph{óptico
centro}\index{óptico
centro}; a reta perpendicular que passa $ O $, o
\emph{eixo principal}\index{eixo principal}.
\end{definition}

\begin{definition}[Pontos focais e distância focal]\label{def:g11:lenses-and-eye:focal}
Os raios que chegam paralelos ao eixo de um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} todos atravessam o
eixo em um ponto atrás dele, o \emph{ponto focal da imagem}\index{ponto focal da imagem} $ F'$; simetricamente, os raios saindo do \emph{ponto focal do objeto}
$ F $ (na frente, com $ OF = OF'$) sai paralelamente ao eixo. O
\emph{distância focal}\index{distância focal} $ f' = \overline{OF'}$ é
positivo para um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente}; através de um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{divergente}, raios paralelos
sair como se fosse de um ponto $ F'$ \emph{na frente} do \omterm{def:g11:lenses-and-eye:lens}{lente}: $ f' < 0$.
\end{definition}

\begin{definition}[Vergência]\label{def:g11:lenses-and-eye:vergence}
O \emph{vergência}\index{vergência} de um \omterm{def:g11:lenses-and-eye:lens}{lente} de \omterm{def:g11:lenses-and-eye:focal}{distância focal} $ f'$
(em metros) é $ C = 1/f'$,
medido em \emph{dioptrias}\index{dioptrias} (símbolo D):
$1~\text{D} = \qty{1}{m^{-1}}$; \omterm{def:g11:lenses-and-eye:lens}{divergente} \omterm{def:g11:lenses-and-eye:lens}{lentes} têm $ C < 0$.
Os oftalmologistas prescrevem em dioptrias: vergências de \omterm{def:g11:lenses-and-eye:lens}{afinar} \omterm{def:g11:lenses-and-eye:lens}{lentes} de contato
simplesmente adicione (admitido aqui; derivado no volume do Ano 1).
\end{definition}

\begin{example}[Lendo uma receita]\label{ex:g11:lenses-and-eye:vergence}
UM \omterm{def:g11:lenses-and-eye:lens}{lente} com $ f' = \qty{0.50}{m}$ tem $ C = 1/0.50 = +2.0$~D:
\omterm{def:g11:lenses-and-eye:lens}{convergente}. UM $-4.0$ Prescrição ~D significa $ f' = \qty{-0.25}{m}$:
\omterm{def:g11:lenses-and-eye:lens}{divergente}, \omterm{def:g11:lenses-and-eye:focal}{distância focal} \qty{25}{cm}.
\end{example}

\section{Construindo a imagem}

\begin{method}[Os três raios de construção]\label{met:g11:lenses-and-eye:rays}
Deixar $ AB $ ser um objeto perpendicular ao eixo em $ A $. Entre todos os raios
deixando a ponta $ B $, três têm caminhos conhecidos:
\begin{enumerate}
  \item o raio através do centro óptico $ O $ segue em frente;
  \item o raio paralelo ao eixo sai por $ F'$;
  \item o raio através $ F $ sai paralelamente ao eixo.
\end{enumerate}
Quaisquer dois são suficientes: sua intersecção é o ponto da imagem $ B'$, e o
imagem $ A'B'$ fica perpendicular ao eixo em $ A'$. Se a saída
os raios divergem, seus \emph{para trás} extensões (tracejadas) se encontram em um
imagem \omterm{def:g11:lenses-and-eye:images}{virtual} apontar.
\end{method}

\begin{definition}[Imagens reais e virtuais]\label{def:g11:lenses-and-eye:images}
Uma imagem é \emph{real}\index{real} quando os raios que saem
realmente passa por ele: uma tela colocada ali o captura;
\emph{virtual}\index{virtual} quando apenas suas extensões para trás
encontro: nenhuma tela o captura, mas um olho olhando para o \omterm{def:g11:lenses-and-eye:lens}{lente} vê isso
perfeitamente bem. UM \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} fornece uma imagem real e invertida de qualquer
objeto além $ F $; deslize o objeto dentro do \omterm{def:g11:lenses-and-eye:focal}{distância focal} e o
a imagem fica virtual, \omterm{def:g10:universal-gravitation:weight}{vertical}, ampliada.
\end{definition}

\section{Uma fórmula: a relação conjugada}

Mostra de construções \emph{onde} a imagem é; uma fórmula calcula isso.
Oriente o eixo ao longo da luz e use \emph{medidas algébricas}:
$\overline{OA}$ é a coordenada de $ A $ em relação a $ O $ (negativo para
um real objeto na frente); alturas $\overline{AB}$ conte positivo para cima.

\begin{theorem}[Relação conjugada de lente fina]\label{thm:g11:lenses-and-eye:conjugate}
Através de um \omterm{def:g11:lenses-and-eye:lens}{lente} fina de \omterm{def:g11:lenses-and-eye:focal}{distância focal} $ f'$, um ponto de objeto $ A $ no
imagens do eixo no ponto $ A'$ com
\[
\frac{1}{\overline{OA'}} - \frac{1}{\overline{OA}} = \frac{1}{f'} = C .
\]
\end{theorem}
\admitted

\begin{remark}
A relação decorre de Lei de Snell aplicado a raios próximos ao
eixo; a derivação --- e os limites do \omterm{def:g11:lenses-and-eye:lens}{lente} fina modelo --- são
tornado honesto no volume do Ano 1.
\end{remark}

\begin{definition}[Ampliação]\label{def:g11:lenses-and-eye:magnification}
O \emph{ampliação}\index{ampliação} da imagem é
\[
\gamma = \frac{\overline{A'B'}}{\overline{AB}}
       = \frac{\overline{OA'}}{\overline{OA}} .
\]
$\abs{\gamma}$ compara tamanhos, suas orientações de \omterm{def:g10:signals-and-waves:signal}{sinal} ($\gamma < 0$:
invertido); o \omterm{def:g10:signals-and-waves:signal}{sinal} de $\overline{OA'}$ lê a natureza (positivo:
real, atrás do \omterm{def:g11:lenses-and-eye:lens}{lente}; negativo: \omterm{def:g11:lenses-and-eye:images}{virtual}, na frente).
\end{definition}

\begin{example}[A fórmula no trabalho]\label{ex:g11:lenses-and-eye:conjugate}
Um objeto fica \qty{30}{cm} na frente de um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} com
$ f' = \qty{10}{cm}$: $\overline{OA} = \qty{-30}{cm}$, então
\[
\frac{1}{\overline{OA'}} = \frac{1}{10} - \frac{1}{30} = \frac{1}{15},
\qquad \overline{OA'} = +\qty{15}{cm},
\qquad \gamma = \frac{15}{-30} = -0.50 :
\]
um imagem real \qty{15}{cm} atrás do \omterm{def:g11:lenses-and-eye:lens}{lente}, invertido, meio tamanho ---
exatamente o que os três raios desenham.
\end{example}

\section{O olho}

\begin{definition}[O olho como sistema óptico]\label{def:g11:lenses-and-eye:eye}
Opticamente, o olho é um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} de frente para uma tela: o
\emph{córnea}\index{córnea} e \emph{lente cristalina}\index{lente cristalina} agir como um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} de ajustável \omterm{def:g11:lenses-and-eye:focal}{distância focal}; o
\emph{retina}\index{retina}, sobre \qty{17}{mm} atrás, é o fixo
tela. A focagem é feita apertando o \omterm{def:g11:lenses-and-eye:lens}{lente}, não movendo-o como em
uma câmera: \emph{alojamento}\index{alojamento} é a mudança de
$ f'$ produzido pelos músculos ciliares. O ponto mais próximo visto nitidamente em plena
alojamento é o \emph{ponto próximo}\index{ponto próximo} --- sobre
\qty{25}{cm} para o \emph{olho padrão}\index{olho padrão} --- o
mais distante, \omterm{def:g10:relative-motion:frame}{em repouso}, o \emph{ponto distante}\index{ponto distante} (infinito para
o olho padrão).
\end{definition}

\begin{example}[A vergência do seu olho]\label{ex:g11:lenses-and-eye:eyerange}
O \omterm{def:g11:lenses-and-eye:eye}{retina} senta-se em $\overline{OA'} = +\qty{0.017}{m}$. Objeto distante
($1/\overline{OA} \approx 0$): $ C = 1/0.017 \approx 59$~D. \omterm{def:g11:lenses-and-eye:eye}{Ponto próximo}
($\overline{OA} = \qty{-0.25}{m}$): $ C = 1/0.017 + 1/0.25 \approx
63$~D. A leitura custa cerca de $4$~D: o olho é um $59$~D \omterm{def:g11:lenses-and-eye:lens}{lente} com um
embutido $+4$~D sintonia fina.
\end{example}

\section{Defeitos, correções e a lupa}

\begin{definition}[Miopia e hipermetropia]\label{def:g11:lenses-and-eye:defects}
UM \emph{míope}\index{míope} olho (míope) é muito \omterm{def:g11:lenses-and-eye:lens}{convergente} para
seu comprimento: foco de raios paralelos \emph{em frente de} o \omterm{def:g11:lenses-and-eye:eye}{retina}, distante
objetos ficam borrados, o \omterm{def:g11:lenses-and-eye:eye}{ponto distante} está a uma distância finita; um \omterm{def:g11:lenses-and-eye:lens}{divergente}
espetáculo \omterm{def:g11:lenses-and-eye:lens}{lente} corrige isso. UM \emph{hipermetrópico}\index{hipermetrópico}
olho (clarividente) não é \omterm{def:g11:lenses-and-eye:lens}{convergente} suficiente: objetos próximos focariam
atrás do \omterm{def:g11:lenses-and-eye:eye}{retina}, o \omterm{def:g11:lenses-and-eye:eye}{ponto próximo} retrocede; um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} corrige
isso. O envelhecimento endurece o \omterm{def:g11:lenses-and-eye:eye}{lente cristalina}
(\emph{presbiopia}\index{presbiopia}), encolhendo \omterm{def:g11:lenses-and-eye:eye}{alojamento} --- mesmo
remédio: óculos de leitura.
\end{definition}

\begin{example}[Prescrição para um míope]\label{ex:g11:lenses-and-eye:correction}
UM \omterm{def:g11:lenses-and-eye:defects}{míope} olho tem seu \omterm{def:g11:lenses-and-eye:eye}{ponto distante} no \qty{50}{cm}: a correção \omterm{def:g11:lenses-and-eye:lens}{lente}
deve mostrar objetos distantes \emph{no \omterm{def:g11:lenses-and-eye:eye}{ponto distante}} --- objeto em
infinito, imagem em $\overline{OA'} = \qty{-0.50}{m}$ (\omterm{def:g11:lenses-and-eye:images}{virtual}, em
frontal). Então $1/f' = 1/\overline{OA'}$: $ f' = \qty{-0.50}{m}$,
$ C = -2.0$~D; o olho relaxado olha para isso imagem \omterm{def:g11:lenses-and-eye:images}{virtual} e vê isso
bruscamente.
\end{example}

\begin{definition}[A lupa]\label{def:g11:lenses-and-eye:magnifier}
UM \emph{lupa}\index{lupa} é um \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} de curto focal
comprimento usado com o objeto \emph{dentro} o \omterm{def:g11:lenses-and-eye:focal}{distância focal}: a imagem
é \omterm{def:g11:lenses-and-eye:images}{virtual}, \omterm{def:g10:universal-gravitation:weight}{vertical}, ampliado (\cref{def:g11:lenses-and-eye:images}),
e o olho vê-o confortavelmente longe.
\end{definition}

\begin{example}[Lupa de joalheiro]\label{ex:g11:lenses-and-eye:magnifier}
\omterm{def:g11:lenses-and-eye:magnifier}{Lupa} com $ f' = \qty{5.0}{cm}$, pedra em \qty{4.0}{cm}:
$1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$, então
$\overline{OA'} = \qty{-20}{cm}$ e $\gamma = (-20)/(-4.0) = +5.0$: um
\omterm{def:g11:lenses-and-eye:images}{virtual}, imagem \omterm{def:g10:universal-gravitation:weight}{vertical}, cinco vezes maior, confortável \qty{20}{cm}
do \omterm{def:g11:lenses-and-eye:lens}{lente}.
\end{example}

\section{Problema: Uma fórmula, três instrumentos}

\begin{problem}[{Problema de fim de semana --- a mesma relação conjugada
foca uma câmera, prescreve óculos de leitura, projeta uma lupa e
mede o zoom embutido em seu próprio olho}]
\label{pb:g11:lenses-and-eye:1}
Um fotógrafo faz uma \qty{50}{mm} \omterm{def:g11:lenses-and-eye:lens}{lente}, uma avó segura o
jornal com o braço estendido, um joalheiro se inclina sobre um diamante: uma linha de
álgebra, $1/\overline{OA'} - 1/\overline{OA} = 1/f'$
(\cref{thm:g11:lenses-and-eye:conjugate}), executa todas as três negociações ---
e então o próprio olho.

\medskip
\noindent\textbf{Part I --- One formula.} UM \omterm{def:g11:lenses-and-eye:lens}{lente} \omterm{def:g11:lenses-and-eye:lens}{convergente} tem
$ f' = \qty{20}{cm}$.
\begin{enumerate}
  \item Um objeto em \qty{60}{cm}: calcular $\overline{OA'}$ e
        $\gamma $; descreva a imagem.
  \item Mova o objeto para \qty{30}{cm}; recalcular. O que foi trocado
        com a pergunta 1? (Os caminhos da luz são reversíveis.)
  \item Mova-o para \qty{10}{cm}, dentro $ f'$: recalcular e descrever.
  \item Resumindo: à medida que o objeto desliza de muito longe para $ F $,
        depois disso, como a imagem se move e muda a natureza?
  \item Mostre que um objeto muito distante ($1/\overline{OA} \approx 0$)
        sempre imagens no plano focal: $\overline{OA'} = f'$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The photographer.}
Uma câmera \omterm{def:g11:lenses-and-eye:lens}{lente} tem $ f' = \qty{50}{mm}$; o sensor, \qty{24}{mm} alto,
pode ser movido em relação ao \omterm{def:g11:lenses-and-eye:lens}{lente}.
\begin{enumerate}[resume]
  \item Onde está o sensor quando focado em uma paisagem distante?
  \item Agora um rosto \qty{1.0}{m} distância: calcule o novo sensor de \omterm{def:g11:lenses-and-eye:lens}{lente}
        distância e o deslocamento do cenário paisagístico.
  \item O mecanismo permite no máximo \qty{55}{mm} da lente-sensor
        distância. Calcule a distância mínima de foco.
  \item No foco mais próximo, calcule $\gamma $. Será que a imagem de um
        \qty{20}{cm} rosto caber no sensor?
  \item ``macro'' em tamanho real significa $\gamma = -1$: onde deve objetar e
        sensor então é, e por que isso exige um especial \omterm{def:g11:lenses-and-eye:lens}{lente}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The grandmother.}
Com a idade o \omterm{def:g11:lenses-and-eye:eye}{ponto próximo} recua: a avó está em \qty{1.0}{m}.
\begin{enumerate}[resume]
  \item Explique o jornal à distância.
  \item Para ler em \qty{25}{cm}, seus óculos devem virar uma página em
        \qty{25}{cm} em um imagem \omterm{def:g11:lenses-and-eye:images}{virtual} no \qty{1.0}{m}: calcular o
        obrigatório \omterm{def:g11:lenses-and-eye:vergence}{vergência}.
  \item Calcular $\gamma $. A imagem é quatro vezes maior, mas não parece
        maior: explicar, comparar tamanhos \emph{e} distâncias.
  \item Mostre que a impressão é nítida apenas entre \qty{25}{cm} e
        $ f' \approx \qty{33}{cm}$. Que tal um \emph{distante} objeto visto
        através dos óculos?
  \item Deduza em uma frase por que existem \omterm{def:g11:lenses-and-eye:lens}{lentes} bifocais.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The jeweler, and the eye's own zoom.}
A \omterm{def:g11:lenses-and-eye:magnifier}{lupa} do joalheiro tem $ f' = \qty{5.0}{cm}$; dele \omterm{def:g11:lenses-and-eye:eye}{ponto próximo} está em
\qty{25}{cm}; dele \omterm{def:g11:lenses-and-eye:eye}{retina} é \qty{17}{mm} atrás do olho dele \omterm{def:g11:lenses-and-eye:lens}{lente}.
\begin{enumerate}[resume]
  \item Uma pedra \qty{4.0}{cm} sob a \omterm{def:g11:lenses-and-eye:magnifier}{lupa}: posição, natureza,
        \omterm{def:g11:lenses-and-eye:magnification}{ampliação}?
  \item Onde deve ficar a pedra para que sua imagem caia em sua direção? perto
        ponto, e o que é $\gamma $ então?
  \item Por que joalheiros experientes colocam a pedra exatamente no $ F $
        em vez disso (imagem no infinito)?
  \item Agora só o olho: calcule seu \omterm{def:g11:lenses-and-eye:vergence}{vergência} vendo um distante
        objeto, então um em \qty{25}{cm}: quantos \omterm{def:g11:lenses-and-eye:vergence}{dioptrias} de
        \omterm{def:g11:lenses-and-eye:eye}{alojamento}?
  \item Converta ambos \omterm{def:g11:lenses-and-eye:vergence}{vergências} para distâncias focais: em quanto, em
        milímetros e por cento, o olho muda seu próprio focal
        comprimento --- o zoom que a câmera fez com a viagem e a avó
        com $+3$~D?
\end{enumerate}
\end{problem}
