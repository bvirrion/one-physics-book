\chapter{As interações fundamentais}\label{ch:g11:fundamental-interactions}

Esfregue um balão no seu cabelo e ele pega pedaços de papel --- batendo,
com alguns centímetros quadrados de borracha, a atração gravitacional do
planeta inteiro. Essa vitória fácil é o motivo \omterm{def:g11:fundamental-interactions:ladder}{átomos} manter juntos e por que
\omterm{def:g11:fundamental-interactions:ladder}{núcleos} pesados eventualmente se quebram (\cref{ch:g11:nucleus-radioactivity}).
O universo funciona exatamente em quatro interações; este capítulo atende
eles, pesa-os uns contra os outros e atribui a cada um o seu piso.

\section{Matéria dos quarks às galáxias}

\begin{definition}[A escada da estrutura]\label{def:g11:fundamental-interactions:ladder}
A matéria é construída em andares, cada um montado a partir do que está abaixo: três
\emph{quarks}\index{quarks} (pontual, menor que \qty{e-18}{m}) ligar
em um \emph{núcleo}\index{núcleo} --- próton ou nêutron, cerca de
\qty{e-15}{m} entre; núcleons se agrupam em um \emph{núcleo}\index{núcleo}
(\qty{e-15}{m} para \qty{e-14}{m}); um núcleo mais sua nuvem de elétrons é
um \emph{átomo}\index{átomo} (\qty{e-10}{m}); átomos se ligam
\emph{moléculas}\index{moléculas} e cristais (\qty{e-9}{m} para cima), que
montar em objetos do cotidiano (\qty{1}{m}), planetas (\qty{e7}{m}),
sistemas planetários (\qty{e11}{m}), galáxias (\qty{e21}{m}) e o
universo observável (\qty{e26}{m}).
\end{definition}

\begin{definition}[As quatro interações]\label{def:g11:fundamental-interactions:four}
UM \emph{interação fundamental}\index{interação fundamental} é um
\omterm{def:g10:inertia:force}{vigor} não redutível a outros forças; existem exatamente quatro:
\emph{gravitação}\index{gravitação}, o \emph{interação eletromagnética}\index{interação eletromagnética},
o interação \omterm{def:g11:fundamental-interactions:strong}{forte} e o interação fraca.
\end{definition}

\begin{remark}\label{rem:g11:fundamental-interactions:empty}
Os pisos são separados por vazios: um \omterm{def:g11:fundamental-interactions:ladder}{átomo} é $10^{5}$ vezes mais largo que
é \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- dimensionar o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} até um \qty{1}{cm} mármore e o
nuvem de elétrons tem um quilômetro de diâmetro. Algo deve conter esses esparsos
andares juntos, e a gravidade quase nunca é isso.
\end{remark}

\section{Carga elétrica}

\begin{definition}[Carga elétrica]\label{def:g11:fundamental-interactions:charge}
\emph{Carga elétrica}\index{Carga elétrica} é propriedade da matéria
em que o \omterm{def:g11:fundamental-interactions:four}{interação eletromagnética} atua, já que a massa é propriedade
\omterm{def:g11:fundamental-interactions:four}{gravitação} atua. É medido em \emph{Coulombs}\index{Coulombs}
(\unit{C}) e entra \emph{dois sinais}, positivo e negativo, que
cancelar quando adicionado; cargas semelhantes se repelem, cargas diferentes se atraem.
\end{definition}

\begin{definition}[Carga elementar]\label{def:g11:fundamental-interactions:elementary}
A carga é granulada: cada carga observada é um múltiplo inteiro da
\emph{carga elementar}\index{carga elementar}
$ e = \qty{1.60e-19}{C}$. O próton carrega $+e $, o elétron $-e $, o
nêutron $0$; uma cobrança $ q $ detém $ N = \abs{q}/e $ cobranças elementares.
\end{definition}

\begin{proposition}[Conservação de carga]\label{prop:g11:fundamental-interactions:conservation}
A carga total de um sistema isolado nunca muda: a carga não é nem
criado nem destruído, apenas transferido --- na prática por elétrons,
os transportadores mais leves.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:conservation}
Nenhuma violação jamais foi observada; a razão profunda, uma simetria de
eletromagnetismo, é dado nos volumes universitários.
\end{remark}

\begin{definition}[Carregamento por fricção e por contato]\label{def:g11:fundamental-interactions:charging}
Esfregar dois isolantes arranca elétrons de um e vai para o outro,
deixando cargas opostas de igual magnitude (\emph{cobrando por
fricção}\index{cobrando por
fricção}); tocar um objeto carregado em um
o neutro compartilha a carga (\emph{cobrando por
contato}\index{cobrando por
contato}). Apenas os elétrons se movem; o total
carga é conservada.
\end{definition}

\begin{example}[Contando elétrons]\label{ex:g11:fundamental-interactions:balloon}
Um balão esfregado no cabelo adquire $ q = \qty{-1.6e-8}{C}$: tem
\emph{ganhou} $ N = \num{1.6e-8} / \num{1.6e-19} = \num{1.0e11}$
elétrons, e o cabelo fica exatamente com $+\qty{1.6e-8}{C}$.
\end{example}

\section{Lei de Coulomb}

\begin{theorem}[Lei de Coulomb]\label{thm:g11:fundamental-interactions:coulomb}
Cargas de dois pontos $ q_1$ e $ q_2$ uma distância $ d $ separados exercem sobre cada
outro forças ao longo da linha que os une, de iguais magnitudes e
direções opostas --- repulsivo para sinais semelhantes, atraente para sinais diferentes
sinais --- com
\[
F = k\,\frac{\abs{q_1 q_2}}{d^{2}},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{theorem}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:torsion}
Coulomb estabeleceu a lei em 1785 medindo a torção de um \omterm{def:g11:lenses-and-eye:lens}{afinar}
fibra de torção; por que o expoente é exatamente $2$ é respondido nos volumes da universidade.
\end{remark}

\begin{remark}[Um gêmeo formal da gravitação]\label{rem:g11:fundamental-interactions:twin}
A lei de Coulomb é, símbolo por símbolo, a lei da universalidade. \omterm{def:g11:fundamental-interactions:four}{gravitação}
(\cref{ch:g10:universal-gravitation}) com cargas no lugar de massas:
\begin{center}
\begin{tabular}{lll}
source & mass $ m $ & charge $ q $ \\
law & $ F = G\,\dfrac{m_1 m_2}{d^2}$ & $ F = k\,\dfrac{\abs{q_1 q_2}}{d^2}$ \\[1.5ex]
constant & $ G = \qty{6.67e-11}{N.m^2/kg^2}$ & $ k = \qty{8.99e9}{N.m^2/C^2}$ \\
sign & always attractive & attractive or repulsive \\
\end{tabular}
\end{center}
As duas diferenças orientam tudo abaixo: a eletricidade é
esmagadoramente mais \omterm{def:g11:fundamental-interactions:strong}{forte}, e só ele pode anular-se.
\end{remark}

\begin{example}[Dois prótons, ambas as forças ao mesmo tempo]\label{ex:g11:fundamental-interactions:protons}
Dois prótons ($ m_p = \qty{1.67e-27}{kg}$) sentar $ d = \qty{1.0e-15}{m}$
separados --- vizinhos nucleares. Repulsão elétrica:
\[
F_E = \num{8.99e9} \times
\frac{(\num{1.60e-19})^{2}}{(\num{1.0e-15})^{2}}
\approx \qty{2.3e2}{N}
\]
--- o \omterm{def:g10:universal-gravitation:weight}{peso} de um $23$ mala de -kg, transportada por uma partícula de
$10^{-27}$ \omterm{def:g10:orders-of-magnitude:unit}{quilogramas}. Atração gravitacional:
$ F_G = \num{6.67e-11} \times (\num{1.67e-27})^{2} /
(\num{1.0e-15})^{2} \approx \qty{1.9e-34}{N}$. A proporção
$ F_E / F_G = k e^{2} / (G m_p^{2}) \approx \num{1.2e36}$ é
independente da distância ($ d^{2}$ cancela): entre partículas elementares,
a gravidade é $10^{36}$ vezes fraco demais para importar --- em \emph{qualquer} distância.
\end{example}

\section{As interações fortes e fracas}

O exemplo deixa um escândalo: um hélio \omterm{def:g11:fundamental-interactions:ladder}{núcleo} contém dois prótons que
repelir com dezenas de \omterm{def:g10:inertia:force}{newtons} --- e ainda assim o hélio existe.

\begin{definition}[A forte interação]\label{def:g11:fundamental-interactions:strong}
O \emph{forte}\index{forte} liga \omterm{def:g11:fundamental-interactions:ladder}{quarks}
em núcleons e núcleons em \omterm{def:g11:fundamental-interactions:ladder}{núcleos}. No \qty{e-15}{m} é
atraente e muito mais forte que a repulsão elétrica (milhares de
\omterm{def:g10:inertia:force}{newtons} entre núcleons); além de alguns $10^{-15}$ metros desaparece:
é \emph{faixa}\index{faixa} é sobre
\qty{e-15}{m}. Ele agarra prótons e nêutrons e ignora a carga.
\end{definition}

\begin{remark}[Por que existem núcleos --- e por que os pesados cederam]\label{rem:g11:fundamental-interactions:nuclei}
Os dois forças no \omterm{def:g11:fundamental-interactions:ladder}{núcleo} escalar de forma diferente. A cola \omterm{def:g11:fundamental-interactions:strong}{forte} é
curto alcance: um \omterm{def:g11:fundamental-interactions:ladder}{núcleo} vincula apenas seus poucos vizinhos dentro
\qty{e-15}{m}. A repulsão elétrica é de longo alcance: cada próton
empurra \emph{todos os outros} próton. À medida que os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} crescem, a repulsão
acumula mais rápido que a cola: \omterm{def:g11:fundamental-interactions:ladder}{núcleos} pesados precisam de nêutrons extras (cola
sem custo), e além de cerca de $80$ prótons, mesmo isso falha --- o
os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} mais pesados vivem em tempo emprestado (\cref{ch:g11:nucleus-radioactivity}).
\end{remark}

\begin{definition}[A interação fraca]\label{def:g11:fundamental-interactions:weak}
O \emph{briga}\index{briga}, de \omterm{def:g11:fundamental-interactions:strong}{faixa} mais curto
ainda assim, permite que um nêutron se transforme em um próton (e vice-versa): ele aciona o
$\beta $ radioatividade de \cref{ch:g11:nucleus-radioactivity}.
\end{definition}

\section{Quem governa qual andar}

\begin{method}[Auditando uma escala]\label{met:g11:fundamental-interactions:audit}
Para descobrir qual interação rege uma estrutura de tamanho $ d $, verificar \omterm{def:g11:fundamental-interactions:strong}{faixa}
e cancelamento:
\begin{enumerate}
  \item $ d \leq \qty{e-14}{m}$: o \emph{interação \omterm{def:g11:fundamental-interactions:strong}{forte}} regras ---
        ele vence a repulsão elétrica, e a gravidade é $10^{36}$ fora.
  \item \omterm{def:g11:fundamental-interactions:ladder}{átomos} para as montanhas ($\qty{e-10}{m}$ para $\qty{e4}{m}$): o
        \omterm{def:g11:fundamental-interactions:strong}{forte} \omterm{def:g10:inertia:force}{vigor} está fora de \omterm{def:g11:fundamental-interactions:strong}{faixa}; o \emph{eletromagnético
        interação} regras --- títulos, coesão, \omterm{def:g10:inertia:inventory}{atrito}, entre em contato.
  \item planetas e além ($ d \geq \qty{e6}{m}$): a matéria é neutra para
        precisão fantástica, então a eletricidade se cancela; massa tem
        um \omterm{def:g10:signals-and-waves:signal}{sinal} e sempre acrescenta --- \emph{\omterm{def:g11:fundamental-interactions:four}{gravitação}} governa o céu.
\end{enumerate}
O interação fraca não tem palavra: não vincula nada, mas arbitra
transformações ($\beta $ decadência).
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:fundamental-interactions:1}
Dê o \omterm{def:g10:orders-of-magnitude:oom}{ordem de grandeza} de cada tamanho e seu piso da escada:
próton \qty{1.7e-15}{m}; água molécula \qty{2.8e-10}{m}; grão de areia
\qty{2e-4}{m}; Terra \qty{1.3e7}{m}; Via Láctea \qty{9.5e20}{m}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:fundamental-interactions:2}
Um balão esfregado no cabelo carrega $ q = \qty{-4.8e-9}{C}$. Ganhou
ou elétrons perdidos, e quantos? Qual é a carga do cabelo depois,
e qual lei diz isso?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:fundamental-interactions:3}
Encargos $ q_1 = \qty{+2.0e-6}{C}$ e $ q_2 = \qty{-3.0e-6}{C}$ sentar
\qty{30}{cm} separado. Calcule o \omterm{def:g10:inertia:force}{vigor}; atraente ou repulsivo?
Compare o forças sobre $ q_1$ e assim por diante $ q_2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:fundamental-interactions:4}
Duas cargas se atraem com \omterm{def:g10:inertia:force}{vigor} $ F_0$. O que acontece se: o
a distância dobra; a distância é dividida por $3$; uma carga dobra;
ambas as acusações \emph{e} a distância é o dobro?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:fundamental-interactions:5}
Cite a interação responsável por: (a) a Lua orbitando a Terra;
(b) a coesão de um cristal de sal; (c) a coesão de um hélio
\omterm{def:g11:fundamental-interactions:ladder}{núcleo}; (d) o $\beta $ decadência do carbono-14; (e) o balão de
\cref{exo:g11:fundamental-interactions:2} grudado em uma parede.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:fundamental-interactions:6}
Os dois prótons de um hélio \omterm{def:g11:fundamental-interactions:ladder}{núcleo} sentar $ d = \qty{2.0e-15}{m}$ separado

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:fundamental-interactions:7}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:fundamental-interactions:8}

\begin{exercise}[$\star\star $]\label{exo:g11:fundamental-interactions:9}

\begin{exercise}[$\star\star $]\label{exo:g11:fundamental-interactions:10}

\begin{exercise}[$\star\star $]\label{exo:g11:fundamental-interactions:11}
Cada uma das duas moedas contém cerca de $\num{e23}$ elétrons. Mova um elétron

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:fundamental-interactions:12}

interação \omterm{def:g11:fundamental-interactions:strong}{forte} vincular este par --- por que não? Quantos repelindo
pares de prótons existem? Por que os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} pesados precisam de nêutrons extras, e
por que, além de algum tamanho, nem mesmo os nêutrons podem salvá-los?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:fundamental-interactions:13}
Duas bolas de massa \qty{1.0}{g} pendurar em \qty{50}{cm} fios amarrados
o mesmo ponto. Dada a mesma cobrança $ q $, eles se acomodam \qty{6.0}{cm}
separado. Cada bola está em equilíbrio sob seu \omterm{def:g10:universal-gravitation:weight}{peso}, o tópico
\omterm{def:g11:circuits-and-power:voltage}{tensão} e o elétrico \omterm{def:g10:inertia:force}{vigor}: mostre isso $ F = mg\tan\theta $ ($\theta $:
ângulo da linha com a \omterm{def:g10:universal-gravitation:weight}{vertical}) e isso aqui $\tan\theta \approx 0.060$.
Calcular $ F $, então $ q $, então o número de cargas elementares movido.
\end{exercise}

\section{Problema: A auditoria das quatro forças}

\begin{problem}[{Problema de fim de semana --- porque o mundo não entra em colapso
nem voa: as quatro interações auditadas andar por andar, até
neutralidade da matéria para duas partes em $10^{18}$}]
\label{pb:g11:fundamental-interactions:1}
\omterm{def:g11:fundamental-interactions:ladder}{Átomos} não implodem, a maioria dos \omterm{def:g11:fundamental-interactions:ladder}{núcleos} se sustentam, a Lua não bate nem
escapa; este problema audita quem merece o crédito. Dados:
$ e = \qty{1.60e-19}{C}$,
$ k = \qty{8.99e9}{N.m^2/C^2}$, $ G = \qty{6.67e-11}{N.m^2/kg^2}$,
$ m_p = \qty{1.67e-27}{kg}$, $ m_e = \qty{9.11e-31}{kg}$,
$ M_E = \qty{5.97e24}{kg}$, $ M_M = \qty{7.35e22}{kg}$, Terra-Lua
distância \qty{3.84e8}{m}.

\medskip
\noindent\textbf{Part I --- The ladder.}
\begin{enumerate}
  \item Liste os andares da escada de quark para a galáxia, um ordem
        de magnitude de tamanho cada.
  \item Modelo em escala: o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} torna-se um \qty{1}{cm} mármore; quão largo
        é o \omterm{def:g11:fundamental-interactions:ladder}{átomo}?
  \item Que fração de um átomoo volume faz o seu \omterm{def:g11:fundamental-interactions:ladder}{núcleo} ocupar?
  \item Um cabelo é \qty{70}{\micro m} espesso. Quantos \omterm{def:g11:fundamental-interactions:ladder}{átomos} abranger isso?
  \item Quantos poderes de dez do quark teto (\qty{e-18}{m})
        para uma galáxia (\qty{e21}{m})?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The electric floor.}
\begin{enumerate}[resume]
  \item Em um hidrogênio \omterm{def:g11:fundamental-interactions:ladder}{átomo} o elétron fica $ d = \qty{5.3e-11}{m}$
        do próton: calcule a \omterm{prop:g11:circuits-and-power:power}{energia elétrica} \omterm{def:g10:inertia:force}{vigor} entre eles.
  \item Calcule o força gravitacional entre eles e a proporção de
        os dois. Qual deles detém o \omterm{def:g11:fundamental-interactions:ladder}{átomo} junto?
  \item O que muda se a gravidade for desligada por dentro \omterm{def:g11:fundamental-interactions:ladder}{átomos}? E se
        eletricidade é?
  \item Por que o ``contato'' forças da vida cotidiana --- o chão
        empurrando seus pés, \omterm{def:g10:inertia:inventory}{atrito} --- eletromagnético disfarçado?
  \item Cada uma das duas moedas contém cerca de $\num{e23}$ elétrons. Transferência
        um elétron em um \emph{bilhão} entre eles, coloque-os
        \qty{1.0}{m} separados: calcule as cargas e o \omterm{def:g10:inertia:force}{vigor}. Conclusão?
  \item De qual interação rege os andares \qty{e-10}{m} para
        \qty{e4}{m}, e por que o interação \omterm{def:g11:fundamental-interactions:strong}{forte} não competir?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The nuclear floor.}
\begin{enumerate}[resume]
  \item Dois prótons estão sentados \qty{1.0e-15}{m} separados em um \omterm{def:g11:fundamental-interactions:ladder}{núcleo}: calcular
        sua repulsão elétrica.
  \item Sem restrições, que aceleração seria essa \omterm{def:g10:inertia:force}{vigor} dar um próton?
        Compare com $ g $.
  \item Liste as três propriedades que interação \omterm{def:g11:fundamental-interactions:strong}{forte} precisa
        explicar por que os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} existem, por que os nêutrons também são capturados e por que
        os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} são minúsculos.
  \item Em um urânio \omterm{def:g11:fundamental-interactions:ladder}{núcleo} (diâmetro \qty{1.4e-14}{m}, $92$ prótons),
        calcule a repulsão entre dois prótons em extremidades opostas. Pode
        o interação \omterm{def:g11:fundamental-interactions:strong}{forte} vincular esse par diretamente? Explique por que
        a repulsão vence à medida que os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} crescem e para que servem os nêutrons.
  \item Em uma frase, como prometido: o que o interação fraca fazer?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The astronomical floor, and the verdict.}
\begin{enumerate}[resume]
  \item Calcule o força gravitacional entre a Terra e a Lua.
  \item A Terra contém cerca de $\num{1.8e51}$ prótons (e tantos
        elétrons), a Lua sobre $\num{2.2e49}$. Que fração $ f$ de
        a carga de prótons de cada corpo, se não fosse cancelada, faria com que o
        elétrico \omterm{def:g10:inertia:force}{vigor} corresponde ao gravitacional?
  \item Por que \omterm{def:g11:fundamental-interactions:four}{gravitação}, o fraco da questão 7, regra
        astronomia? Duas razões.
  \item Veredicto --- uma frase por andar (\omterm{def:g11:fundamental-interactions:ladder}{núcleo}, \omterm{def:g11:fundamental-interactions:ladder}{átomo} para a montanha,
        planeta e acima): o que o mantém unido? Então responda o título:
        por que o mundo não entra em colapso nem se desintegra?
\end{enumerate}
\end{problem}
