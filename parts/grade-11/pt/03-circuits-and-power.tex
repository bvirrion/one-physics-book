\chapter{Circuitos Elétricos e Energia}\label{ch:g11:circuits-and-power}

Acione um interruptor: uma lâmpada acende, porque um gerador a muitos quilômetros de distância
empurra a carga em torno de um laço de cobre através de sua parede. Anos anteriores
mapeou o circuito --- corrente, \omterm{def:g11:circuits-and-power:voltage}{tensão}, Lei de Ohm. Este capítulo reabre
com o livro de contas de \cref{ch:g10:energy-conservation}: o que
cobranças do \omterm{def:g10:orders-of-magnitude:unit}{medidor}, por que \omterm{def:g11:lenses-and-eye:lens}{afinar} fios brilham, por que a rede funciona em \qty{400}{kV}.

\section{Carregue em movimento}

\begin{definition}[Atual e o Coulomb]\label{def:g11:circuits-and-power:current}
Um \emph{corrente elétrica}\index{corrente elétrica} é um fluxo ordenado de
carga (elétrons, em um metal). Sua intensidade $I $, em
\emph{amperes}\index{amperes} (\unit{A}), é a carga que passa por um
seção transversal do fio por \omterm{def:g10:orders-of-magnitude:unit}{segundo}: uma corrente constante $ I $ fluindo por um
hora $ t $ transporta a carga
\[ q = I\,t , \]
em \emph{Coulombs}\index{Coulombs} (\unit{C}), com $\qty{1}{C} =
\qty{1}{A.s}$. A carga elementar é $ e = \qty{1.6e-19}{C}$. Atual
é medido por um ammeter, inserido em série.
\end{definition}

\begin{example}[Relâmpago versus telefone]\label{ex:g11:circuits-and-power:coulomb}
Um raio carrega alguns $\qty{3.0e4}{A}$ por cerca de
$\qty{1.0e-4}{s}$: $ q = 3.0\times10^{4} \times 10^{-4} = \qty{3.0}{C}$.
Seu telefone, em \qty{2.0}{A}, move a mesma carga a cada $1.5$ \omterm{def:g10:orders-of-magnitude:unit}{segundos}.
\end{example}

\begin{definition}[Tensão e o volt]\label{def:g11:circuits-and-power:voltage}
O \emph{tensão}\index{tensão} $ U $ entre dois pontos de um circuito, em
\emph{volts}\index{volts} (\unit{V}), é o energia trocado por \omterm{def:g10:orders-of-magnitude:unit}{unidade} de
cobrar viajando entre eles: $\qty{1}{V} = \qty{1}{J/C}$, então uma cobrança
$ q $ atravessando um dispositivo sob tensão $ U $ troca o energia $ E = qU $
com isso. Um voltmeter, conectado em paralelo, mede isso.
\end{definition}

\begin{proposition}[Leis de circuito]\label{prop:g11:circuits-and-power:laws}
Em um circuito funcionando continuamente:
\begin{enumerate}
  \item \emph{lei de junção:} as correntes que chegam a uma junção se somam
        às correntes que o deixam;
  \item \emph{lei da série:} tensões ao longo de um caminho some:
        $ U_{AC} = U_{AB} + U_{BC}$;
  \item \emph{lei paralela:} dois ramos unindo os mesmos dois pontos
        carregar o mesmo \omterm{def:g11:circuits-and-power:voltage}{tensão}.
\end{enumerate}
\end{proposition}

\begin{proof}
Em um estado estacionário, a carga não se acumula em lugar nenhum, então o que flui para um
junção flui para fora. \omterm{def:g10:energy-conservation:energy}{Energia} por coulomb é aditivo ao longo de um caminho e
depende apenas de seus pontos finais --- não da ramificação entre eles.
\end{proof}

\section{Lei de Ohm e redes de Res}

\begin{definition}[Resistência]\label{def:g11:circuits-and-power:resistance}
O \emph{resistência}\index{resistência} de um condutor é a razão
$ R = U/I $ do \omterm{def:g11:circuits-and-power:voltage}{tensão} através dele até a corrente que passa por ele, medida em
\emph{ohms}\index{ohms} (\unit{\ohm}): $\qty{1}{\ohm} = \qty{1}{V/A}$.
\end{definition}

\begin{proposition}[Lei de Ohm]\label{prop:g11:circuits-and-power:ohm}
\index{Lei de Ohm}
Para um condutor metálico mantido a uma temperatura fixa, $ R $ é constante:
\omterm{def:g11:circuits-and-power:voltage}{tensão} e a corrente são proporcionais,
\[ U = R\,I . \]
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:circuits-and-power:ohm-limits}
Uma lei experimental, não universal: um diodo, uma lâmpada de aquecimento
filamento ou sua própria pele não obedecem. Por que os metais fazem isso é derivado no
volumes universitários.
\end{remark}

\begin{proposition}[Resistores em série e paralelo]\label{prop:g11:circuits-and-power:networks}
\index{resistência equivalente}
Dois Res em série equivalem a um R $ R_{\text{s}} =
R_1 + R_2$; dois em paralelo, para um R $ R_{\text{p}}$ dado por
\[
\frac{1}{R_{\text{p}}} = \frac{1}{R_1} + \frac{1}{R_2} ,
\qquad\text{i.e.}\quad
R_{\text{p}} = \frac{R_1 R_2}{R_1 + R_2} < \min(R_1, R_2) .
\]
\end{proposition}

\begin{proof}
Série: a mesma $ I $ atravessa ambos e tensões adicionar
(\cref{prop:g11:circuits-and-power:laws}), então $ U = (R_1 + R_2)\,I $.
Paralelo: ambos sentam-se sob o mesmo $ U $ e as correntes se somam, então
$ I = \frac{U}{R_1} + \frac{U}{R_2}$.
\end{proof}

\begin{example}[Resistência equivalente]\label{ex:g11:circuits-and-power:networks}
$ R_1 = \qty{100}{\ohm}$ e $ R_2 = \qty{150}{\ohm}$: em série,
\qty{250}{\ohm}; em paralelo, $\frac{100 \times 150}{250} =
\qty{60}{\ohm}$ --- menos que qualquer um: a corrente ganha uma segunda estrada.
\end{example}

\section{Energia e potência em um circuito}

\begin{proposition}[Energia elétrica]\label{prop:g11:circuits-and-power:power}
\index{energia elétrica}
Um dispositivo sob \omterm{def:g11:circuits-and-power:voltage}{tensão} $ U $ atravessado por uma corrente $ I $ recebe energia em
a taxa
\[ P = U\,I , \qquad\text{i.e.}\quad E = U\,I\,t \ \text{in a time } t . \]
\end{proposition}

\begin{proof}
Em um tempo $ t $ a cobrança $ q = It $ atravessa o dispositivo
(\cref{def:g11:circuits-and-power:current}) e cada coulomb entrega
$ U $ \omterm{def:g10:energy-conservation:energy}{joules} (\cref{def:g11:circuits-and-power:voltage}): $ E = qU = UIt $.
\end{proof}

\begin{proposition}[Potência em um R]\label{prop:g11:circuits-and-power:ri2}
Um R $ R $ transportando uma corrente $ I $ dissipa
\[ P = R\,I^2 = \frac{U^2}{R} . \]
\end{proposition}

\begin{proof}
Substituir Lei de Ohm em $ P = UI $: $ P = (RI)I = RI^2 = U(U/R) = U^2/R $.
\end{proof}

\begin{example}[Lendo uma placa de identificação]\label{ex:g11:circuits-and-power:nameplate}
Uma chaleira marcada ``\qty{230}{V} -- \qty{2200}{W}'' empates $ I = P/U =
2200/230 = \qty{9.6}{A}$ e tem \omterm{def:g11:circuits-and-power:resistance}{resistência} $ R = U^2/P = 230^2/2200 =
\qty{24}{\ohm}$. Correndo \qty{150}{s} ele usa $ E = 2200 \times 150 =
\qty{3.3e5}{J} \approx \qty{0.09}{kW.h}$ --- cerca de dois centavos.
\end{example}

\begin{remark}[Joules e quilowatts-hora]\label{rem:g11:circuits-and-power:kwh}
O \omterm{def:g10:orders-of-magnitude:unit}{medidor} conta \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} (\cref{def:g10:energy-conservation:kwh}):
$\qty{1}{kW.h} = \qty{3.6e6}{J}$. Joules são os do físico \omterm{def:g10:orders-of-magnitude:unit}{unidade};
\omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} precificar as contas.
\end{remark}

\section{O efeito Joule: da torradeira ao poste}

\begin{definition}[Efeito Joule]\label{def:g11:circuits-and-power:joule}
A dissipação $ P = RI^2$ girando \omterm{prop:g11:circuits-and-power:power}{energia elétrica} em calor em cada
R é o \emph{Efeito Joule}\index{Efeito Joule}. Aquecedores,
chaleiras, torradeiras e lâmpadas antigas são Res por design; um
\emph{destruição}\index{destruição} é um \omterm{def:g11:lenses-and-eye:lens}{afinar} fio calibrado para derreter --- abrindo o
circuito --- quando a corrente excede sua classificação.
\end{definition}

\begin{example}[Por que os postes são altos]\label{ex:g11:circuits-and-power:transmission}
Uma aldeia precisa $ P = \qty{100}{kW}$ através de uma linha de \omterm{def:g11:circuits-and-power:resistance}{resistência}
$ R_\ell = \qty{5.0}{\ohm}$. Entregue em $ U = \qty{500}{V}$: $ I = P/U =
\qty{200}{A}$ e a linha desperdiça $ R_\ell I^2 = \qty{200}{kW}$ --- duas vezes
a entrega: absurda. No $ U = \qty{20}{kV}$: $ I = \qty{5.0}{A}$ e
$ R_\ell I^2 = \qty{125}{W}$. Quarenta vezes menos atual, $1600$ vezes menos
perda.
\end{example}

\begin{remark}[A regra de ferro da rede]\label{rem:g11:circuits-and-power:grid}
Em entrega fixa poder $ P = UI $, a perda de linha $ R_\ell I^2 = R_\ell
P^2/U^2$ cai como o \emph{quadrado} do transporte \omterm{def:g11:circuits-and-power:voltage}{tensão}. Portanto:
transformar (\qty{400}{kV}) viajar, descer (\qty{230}{V}) para conviver.
\end{remark}

\begin{remark}[Segurança doméstica]\label{rem:g11:circuits-and-power:safety}
UM \qty{230}{V} linha fundida em \qty{16}{A} suprimentos no máximo $230 \times 16
\approx \qty{3.7}{kW}$: duas chaleiras ao mesmo tempo derretem o fusível --- por design
o elo mais fino e fraco, muito mais barato que um incêndio dentro da parede.
\end{remark}

\section{Baterias reais: força eletromotriz}

\begin{definition}[EMF e resistência interna]\label{def:g11:circuits-and-power:emf}
Uma battery converte produtos químicos em \omterm{prop:g11:circuits-and-power:power}{energia elétrica}. Isso é
\emph{força eletromotriz}\index{força eletromotriz} (EMF)
$\mathcal{E}$, em \omterm{def:g11:circuits-and-power:voltage}{volts}, é o energia entrega cada um coulomb cruzando-o;
isso é \emph{resistência interna}\index{resistência interna} $ r $ contas
para o \omterm{def:g11:circuits-and-power:joule}{Efeito Joule} dentro de sua própria química. Modelo: uma fonte ideal
$\mathcal{E}$ em série com $ r $ (encaracolado $\mathcal{E}$: $ E $ é energia).
\end{definition}

\begin{proposition}[Tensão terminal]\label{prop:g11:circuits-and-power:terminal}
Uma battery fornecendo uma corrente $ I $ mostra em seus terminais
\[ U = \mathcal{E} - r\,I . \]
\end{proposition}

\begin{proof}
Os suprimentos de química $\mathcal{E}I $, o \omterm{def:g11:circuits-and-power:emf}{resistência interna} come
$ rI^2$, o circuito recebe $ UI $: energia conservação
(\cref{ch:g10:energy-conservation}) dá $ UI = \mathcal{E}I - rI^2$.
\end{proof}

\begin{example}[Uma battery sob carga]\label{ex:g11:circuits-and-power:discharge}
Uma battery ($\mathcal{E} = \qty{9.0}{V}$, $ r = \qty{1.0}{\ohm}$) alimenta um
lâmpada de \omterm{def:g11:circuits-and-power:resistance}{resistência} $ R = \qty{8.0}{\ohm}$. O laço obedece $\mathcal{E} =
rI + RI $, então $ I = \mathcal{E}/(R + r) = \qty{1.0}{A}$ e $ U = 9.0 -
1.0 \times 1.0 = \qty{8.0}{V}$: a lâmpada recebe \qty{8.0}{W} enquanto o
resíduos de battery \qty{1.0}{W} aquecendo-se.
\end{example}

\begin{remark}[Receptores]\label{rem:g11:circuits-and-power:receivers}
Um motor ou uma battery em carga faz a conversão ao contrário: absorve
sob $ U = \mathcal{E}' + r'I $, convertendo $\mathcal{E}'I $ em movimento ou
química e perder $ r'I^2$. As fontes empurram, os receptores empurram de volta.
\end{remark}

\begin{example}[Eficiência de uma battery]\label{ex:g11:circuits-and-power:efficiency}
O \omterm{def:g10:energy-conservation:efficiency}{eficiência} (\cref{def:g10:energy-conservation:efficiency}) do
descarga acima: útil $ UI = \qty{8.0}{W}$ fora de $\mathcal{E}I =
\qty{9.0}{W}$, então $\eta = U/\mathcal{E} = 8/9 \approx 0.89$ --- caindo
à medida que a corrente aumenta: as batterys de trabalho pesado esquentam.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:circuits-and-power:1}
A battery de um telefone carrega a uma velocidade constante \qty{2.0}{A} para \qty{90}{min}. O que
carga passa pelo cabo, em \omterm{def:g11:circuits-and-power:current}{Coulombs}? Quantas cargas elementares
é isso ($ e = \qty{1.6e-19}{C}$)?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:circuits-and-power:2}
UM \qty{220}{\ohm} R através \qty{12}{V}: que corrente flui? Um
elemento de aquecimento atrai \qty{0.50}{A} no \qty{230}{V}: o que \omterm{def:g11:circuits-and-power:resistance}{resistência}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:circuits-and-power:3}
Calcule o \omterm{prop:g11:circuits-and-power:networks}{resistência equivalente} de $ R_1 = \qty{120}{\ohm}$ e $ R_2 =
\qty{60}{\ohm}$ em série e depois em paralelo. Que é menor que ambos,
e por quê?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:circuits-and-power:4}
Um ferro de viagem desenha \qty{8.7}{A} no \qty{230}{V}. Calcule seu poder, então
o energia ele usa em \qty{4.0}{min}, em \omterm{def:g10:energy-conservation:energy}{joules} e em \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:circuits-and-power:5}
Um carregador de laptop oferece \qty{60}{W} para \qty{5.0}{h} por dia: diariamente
energia em \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} e \omterm{def:g10:energy-conservation:energy}{joules}, e custo em $0.25$ euros por
\unit{kW.h}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:circuits-and-power:6}
Um aquecedor elétrico é classificado \qty{1200}{W} no \qty{230}{V}.
\begin{enumerate}
  \item Calcule seu \omterm{def:g11:circuits-and-power:resistance}{resistência} e a corrente que ele atrai.
  \item Conectado (mesmo \omterm{def:g11:circuits-and-power:resistance}{resistência}) em um \qty{115}{V} fornecimento, o que poder
        entrega? Por que não metade?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:circuits-and-power:7}
$ R_1 = \qty{40}{\ohm}$ e $ R_2 = \qty{20}{\ohm}$ em série ao longo
\qty{12}{V}. Calcule a corrente, a \omterm{def:g11:circuits-and-power:voltage}{tensão} em cada R e

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:circuits-and-power:8}
$ R_1 = \qty{60}{\ohm}$ e $ R_2 = \qty{30}{\ohm}$ em paralelo através

\omterm{prop:g11:circuits-and-power:networks}{resistência equivalente} duas maneiras.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:circuits-and-power:9}
Uma battery fornece $ U = \qty{5.5}{V}$ no $ I = \qty{0.50}{A}$, e

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:circuits-and-power:10}
UM \qty{230}{V} linha de cozinha é protegida por um \qty{16}{A} fusível. UM

\begin{exercise}[$\star\star $]\label{exo:g11:circuits-and-power:11}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:circuits-and-power:12}

\begin{exercise}[$\star\star\star $]\label{exo:g11:circuits-and-power:13}
Uma battery ($\mathcal{E} = \qty{4.5}{V}$, $ r = \qty{1.5}{\ohm}$) alimenta um
R $ R = \qty{6.0}{\ohm}$.
\begin{enumerate}
  \item Calcular $ I $, $ U $, o útil poder em $ R $, o poder perdido em $ r $,
        e o \omterm{def:g10:energy-conservation:efficiency}{eficiência}.
  \item Mostre que o poder recebido por uma variável $ R $,
        $ P = \mathcal{E}^2 R/(R+r)^2$, é maior quando $ R = r $ (dica:

\end{enumerate}
\end{exercise}

\section{Problema: Da barragem à torradeira}

\begin{problem}[{Problema de fim de semana --- a jornada de mil quilômetros de um
quilowatt-hora, desde a queda da água até o pão dourado, com a conta cheia
da viagem}]
\label{pb:g11:circuits-and-power:1}
Uma represa na montanha contém água \qty{200}{m} acima de suas turbinas, que engolem
\qty{50}{m^3} --- \qty{5.0e4}{kg} --- por \omterm{def:g10:orders-of-magnitude:unit}{segundo}; turbina e alternador
converter com \omterm{def:g10:energy-conservation:efficiency}{eficiência} $0.90$. Através de uma linha de total \omterm{def:g11:circuits-and-power:resistance}{resistência}
$ R_\ell = \qty{32}{\ohm}$ (\qty{1000}{km} ida e volta) a plant se alimenta
uma cidade onde seu \qty{900}{W} torradeira espera em \qty{230}{V}. O
saídas do alternador \qty{20}{kV}; transformadores (\omterm{def:g10:energy-conservation:efficiency}{eficiência} $0.99$ cada)
converter tensões à vontade. A electricidade é vendida a $0.25$ euros por
\unit{kW.h}; $ g = \qty{9.81}{N/kg}$.

\medskip
\noindent\textbf{Part I --- At the dam.}
\begin{enumerate}
  \item Quanto potencial energia a água que cai libera cada
        \omterm{def:g10:orders-of-magnitude:unit}{segundo}? Deduza o poder disponível para as turbinas.
  \item O que \omterm{prop:g11:circuits-and-power:power}{energia elétrica} sai do alternador?
  \item No \qty{20}{kV} direto do alternador, que corrente flui?
  \item Calcule a perda de Joule $ R_\ell I^2$ naquela corrente e compare
        com a produção da plant. Concluir.
  \item A cidade poder $ P = UI $ é fixado pela demanda. Qual é o único
        maneira de encolher $ I $, e qual dispositivo executa isso?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- On the line.}
\begin{enumerate}[resume]
  \item Um transformador, tratado como sem perdas para esta questão, percorre o
        \qty{88}{MW} até \qty{400}{kV}: que corrente flui agora?
  \item Calcule a perda da nova linha, em megawatts.
  \item Que fração da produção da plant é perdida na linha?
  \item Mostre que a fração perdida é igual $ R_\ell P/U^2$ e verifique isso
        explica o fator entre as questões 4 e 7.
  \item Calcule o \omterm{def:g11:circuits-and-power:voltage}{tensão} derrubar $ R_\ell I $ ao longo da linha em
        \qty{400}{kV}. É consistente com a questão 8?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- In the kitchen.}
\begin{enumerate}[resume]
  \item Converter \qty{1}{kW.h} para \omterm{def:g10:energy-conservation:energy}{joules}; o que o \omterm{def:g10:orders-of-magnitude:unit}{medidor} multiplica?
  \item Calcule a corrente e a \omterm{def:g11:circuits-and-power:resistance}{resistência} no \qty{230}{V}.
  \item Três minutos de torradas: o energia em \omterm{def:g10:energy-conservation:energy}{joules} e
        \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora}, e seu preço.
  \item A torradeira se dissipa $ RI^2$: explique em uma frase por que o
        mesma fórmula é a perda na linha, o produto na cozinha.
  \item A torradeira compartilha um \qty{16}{A} fusível com um \qty{2.2}{kW}
        chaleira: o fusível segurar? UM \qty{2.0}{kW} aquecedor junta: e agora?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The audit.}
\begin{enumerate}[resume]
  \item A fiação da própria casa, \omterm{def:g11:circuits-and-power:resistance}{resistência} \qty{0.05}{\ohm}, carrega o
        \qty{13.5}{A} da questão 15: calcule a perda e sua fração
        dos aproximadamente \qty{3.1}{kW} entregue.
  \item Encadeie as \omterm{def:g10:energy-conservation:efficiency}{eficiências} (turbina-alternador, elevador, linha,
        abaixador, fiação doméstica) em uma grade geral \omterm{def:g10:energy-conservation:efficiency}{eficiência}.
  \item Quantos \omterm{def:g10:orders-of-magnitude:unit}{quilogramas} (e cúbico metros) da água da barragem deve cair para
        \emph{um} \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} para alcançar sua torradeira?
  \item Avalie a fração perdida $ R_\ell P/U^2$ da questão 9 em
        $ U = \qty{20}{kV}$. O que significa um valor acima $1$ significar?
  \item A piada --- \emph{o preço da distância}: em duas frases,
        o que um brinde \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} custou a montanha, o que
        fração morre no caminho, e qual truque torna isso possível?
\end{enumerate}
\end{problem}
