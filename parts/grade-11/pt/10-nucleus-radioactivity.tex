\chapter{O Núcleo e a Radioatividade}\label{ch:g11:nucleus-radioactivity}

Neste momento, cerca de oito mil \omterm{def:g11:fundamental-interactions:ladder}{núcleos} atômicos explodem dentro do seu corpo a cada
\omterm{def:g10:orders-of-magnitude:unit}{segundo}. Nada os ataca: certos \omterm{def:g11:fundamental-interactions:ladder}{núcleos} nascem instáveis, e cada
mais cedo ou mais tarde se transforma, lançando um fragmento em uma boa fração
da velocidade da luz. Este capítulo retorna ao $10^{-15}$-metro de piso de
\cref{ch:g11:fundamental-interactions} para ver quais \omterm{def:g11:fundamental-interactions:ladder}{núcleos} são frágeis, como
eles quebram, e como sua impaciência regular data cavernas e guarda tetos.

\section{Núcleos, núcleons e isótopos}

\begin{definition}[Notação nuclear]\label{def:g11:nucleus-radioactivity:notation}
UM \omterm{def:g11:fundamental-interactions:ladder}{núcleo} contém $ Z $ prótons (carga $+e $ cada) e $ N $ nêutrons (não
cobrar), $ A = Z + N $ núcleons em tudo; está escrito ${}^{A}_{Z}\mathrm{X}$,
onde $\mathrm X $ é o símbolo químico, $ Z $ o \emph{atômico
número}\index{atômico
número} e $ A $ o \emph{número de massa}\index{número de massa}. Uma espécie especificada por $ A $ e $ Z $ é um \emph{nuclídeo}\index{nuclídeo};
nuclídeos com o mesmo $ Z $ mas diferente $ A $ são \emph{isótopos}\index{isótopos}
de um elemento: mesma química (a nuvem de elétrons vê apenas $ Ze $),
diferentes \omterm{def:g11:fundamental-interactions:ladder}{núcleos} e estabilidade.
\end{definition}

\begin{example}[Hidrogênio e carbono]\label{ex:g11:nucleus-radioactivity:hc}
O hidrogênio tem três \omterm{def:g11:nucleus-radioactivity:notation}{isótopos}: ${}^{1}_{1}\mathrm{H}$ (um próton solitário),
${}^{2}_{1}\mathrm{H}$ (deutério: um próton, um nêutron), instável
${}^{3}_{1}\mathrm{H}$ (trítio: dois nêutrons). Carbono natural: principalmente
${}^{12}_{6}\mathrm{C}$ ($6$ prótons, $6$ nêutrons), $1.1\%$ de
${}^{13}_{6}\mathrm{C}$, e um \omterm{def:g11:fundamental-interactions:ladder}{átomo} em $10^{12}$ de instável
${}^{14}_{6}\mathrm{C}$ ($8$ nêutrons), herói de \cref{ex:g11:nucleus-radioactivity:dating}.
\end{example}

\section{Núcleos estáveis e instáveis}

\begin{proposition}[O vale da estabilidade]\label{prop:g11:nucleus-radioactivity:valley}
Estável \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeos} ocupam uma \omterm{def:g11:fundamental-interactions:strong}{faixa} estreita no $(Z, N)$ avião: $ N \approx Z $
para \omterm{def:g11:fundamental-interactions:ladder}{núcleos} leves, então um excesso crescente de nêutrons, até $ N \approx 1.5\,Z $.
Todo \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo} fora da banda --- muitos nêutrons, muitos prótons ou muito
grande (além do chumbo, $ Z = 82$, não \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo} é verdadeiramente estável) --- se transforma mais cedo ou mais tarde.
\end{proposition}
\admitted

\begin{remark}
Localizar a banda com exatidão é a mecânica quântica  os volumes universitários.
O padrão é o livro-razão de \cref{ch:g11:fundamental-interactions}: \omterm{def:g11:fundamental-interactions:strong}{forte}
a cola une apenas os vizinhos, a repulsão de Coulomb abrange todo o \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, então
\omterm{def:g11:fundamental-interactions:ladder}{núcleos} pesados precisam de cola de nêutrons extra; ainda assim, um excesso de nêutrons é instável
também (o interação fraca cuida disso).
\end{remark}

\begin{definition}[Radioatividade]\label{def:g11:nucleus-radioactivity:radioactivity}
\emph{Radioatividade}\index{Radioatividade} é a transformação espontânea
de um instável \omterm{def:g11:fundamental-interactions:ladder}{núcleo} em outro \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo}, com emissão de radiação. Isso é \emph{aleatório}: nada anuncia qual \omterm{def:g11:fundamental-interactions:ladder}{núcleo} decai em seguida, e não
\omterm{def:g10:pressure:pressure}{pressão}, a temperatura ou a química podem acelerá-lo ou atrasá-lo; apenas grande
populações são previsíveis (\cref{def:g11:nucleus-radioactivity:halflife}).
\end{definition}

\begin{remark}[Becquerel e os Curie]\label{rem:g11:nucleus-radioactivity:curie}
Em 1896, Henri Becquerel descobriu que os sais de urânio embaçavam uma chapa fotográfica
em papel preto, em gaveta fechada, sem energia fornecido. Maria e
Pierre Curie mostrou que o efeito é atômico, mediu-o e isolou polônio
e o rádio, um milhão de vezes mais ativo. A palavra \emph{radioatividade} é
de Marie Curie; o mesmo acontece com dois prêmios Nobel.
\end{remark}

\section{Alfa, beta, gama}

\begin{definition}[As três radiações históricas]\label{def:g11:nucleus-radioactivity:abg}
Três radiações, nomeadas antes que alguém soubesse o que eram:
\begin{itemize}
  \item \emph{alfa}\index{alfa} ($\alpha $): um hélio \omterm{def:g11:fundamental-interactions:ladder}{núcleo}
        ${}^{4}_{2}\mathrm{He}$ ejetado inteiro --- pesado, duplamente carregado,
        parado por uma folha de papel ou centímetros de ar;
  \item \emph{beta}\index{beta} ($\beta^-$): um elétron
        ${}^{\;0}_{-1}\mathrm{e}$ criado no \omterm{def:g11:fundamental-interactions:ladder}{núcleo} e ejetado ---
        leve, rápido, parado por milímetros de alumínio;
  \item \emph{gama}\index{gama} ($\gamma $): eletromagnético
        radiação, muito mais energética que a luz --- nunca parou, apenas
        \emph{atenuado}\index{atenuado}: um centímetro de chumbo absorve metade.
\end{itemize}
\end{definition}

\begin{proposition}[Conservação em equações nucleares]\label{prop:g11:nucleus-radioactivity:soddy}
Em cada transformação nuclear, o total \omterm{def:g11:nucleus-radioactivity:notation}{número de massa} $ A $ e o total
cobrar --- a soma dos índices mais baixos, contando $-1$ para uma emissão
elétron --- são iguais antes e depois.
\end{proposition}
\admitted

\begin{remark}
Conservação de cobrança já regulamentada \cref{ch:g11:fundamental-interactions};
conservação de $ A $ diz núcleons são apenas convertidos, nunca criados ou
destruído --- contabilidade honesta para os volumes universitários.
\end{remark}

\begin{method}[Equilibrando uma equação de decaimento]\label{met:g11:nucleus-radioactivity:balance}
\begin{enumerate}
  \item Escreva cada ator com ambos os rótulos: $\alpha = {}^{4}_{2}\mathrm{He}$,
        $\beta^- = {}^{\;0}_{-1}\mathrm{e}$, $\beta^+ = {}^{0}_{+1}\mathrm{e}$; um $\gamma $ carrega $ A = Z = 0$.
  \item Iguale as somas dos $ A $'s, então do $ Z $ de
        (\cref{prop:g11:nucleus-radioactivity:soddy}); resolva e leia o
        elemento filha fora de seu $ Z $ no \omterm{def:g10:signals-and-waves:period}{periódico} mesa.
\end{enumerate}
\end{method}

\begin{example}[Uma decadência alfa]\label{ex:g11:nucleus-radioactivity:alpha}
O rádio-226, o rádio dos Curie, é um $\alpha $ emissor:
${}^{226}_{88}\mathrm{Ra} \to {}^{A}_{Z}\mathrm{Y} + {}^{4}_{2}\mathrm{He}$
forças $ A = 226 - 4 = 222$ e $ Z = 88 - 2 = 86$; elemento $86$ é radônio, então
${}^{226}_{88}\mathrm{Ra} \to {}^{222}_{86}\mathrm{Rn} + {}^{4}_{2}\mathrm{He}$
--- o gás radioativo de porões não ventilados.
\end{example}

\begin{example}[Uma decadência beta-menos]\label{ex:g11:nucleus-radioactivity:beta}
Carbono-14 fica acima da banda ($8$ nêutrons contra $6$ prótons):
${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} + {}^{\;0}_{-1}\mathrm{e}$.
Verificar: $14 = 14 + 0$, $6 = 7 - 1$; um nêutron se tornou um próton mais o
elétron ejetado --- o interação fraca no \omterm{def:g11:work-of-force:work}{trabalhar}.
\end{example}

\begin{remark}[Beta-plus e gama]\label{rem:g11:nucleus-radioactivity:betaplus}
Abaixo da banda, o espelho decai $\beta^+$ transforma um próton em um nêutron
mais um \emph{pósitron}\index{pósitron} ${}^{0}_{+1}\mathrm{e}$, a antipartícula do elétron:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$,
carro-chefe dos scanners PET. E depois da maior decadência a filha, nasce tremendo,
se instala emitindo um $\gamma $ no topo.
\end{remark}

\section{Meia-vida e atividade}

\begin{definition}[Meia-vida]\label{def:g11:nucleus-radioactivity:halflife}
O \emph{meia-vida}\index{meia-vida} $ T_{1/2}$ de um \omterm{def:g11:nucleus-radioactivity:notation}{nuclídeo} é a hora
após o que metade de qualquer amostra grande decaiu. Cada meia-vida adicional
metade do que resta: de $ N_0$ \omterm{def:g11:fundamental-interactions:ladder}{núcleos}, $ N_0/2$ são deixados em $ T_{1/2}$, $ N_0/4$
no $2\,T_{1/2}$, $ N_0/8$ no $3\,T_{1/2}$, $ N_0/2^{\,n}$ depois $ n $ meias-vidas.
\end{definition}

\begin{remark}
Aleatoriedade para um \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, relógio para $10^{23}$: uma vasta multidão de
lança-moedas, imprevisíveis um por um, eliminados pela metade a cada rodada.
Interpolando \emph{entre} meias-vidas inteiras tomam o exponencial
função --- no próximo ano, com o volume de matemática.
\end{remark}

\begin{example}[Vinte potências de dez]\label{ex:g11:nucleus-radioactivity:range}
\begin{center}\small\begin{tabular}{llll}
polonium-214 & $\alpha $ & \qty{1.6e-4}{s} & radium chain link \\
technetium-99m & $\gamma $ & $6.0$ hours & medical imaging \\
americium-241 & $\alpha $ & $432$ years & smoke detectors \\
carbon-14 & $\beta^-$ & \num{5700} years & archaeology \\
potassium-40 & $\beta $ & \num{1.25e9} years & in every banana \\
uranium-238 & $\alpha $ & \num{4.5e9} years & Earth's inner heat \\
\end{tabular}\end{center}
\end{example}

\begin{example}[Datação por carbono-14]\label{ex:g11:nucleus-radioactivity:dating}
A matéria viva troca carbono com a atmosfera, por isso mantém o
proporção atmosférica de carbono-14; na morte, a ingestão é interrompida e o
proporção divide pela metade cada \num{5700} anos. Carvão de uma caverna pintada mostra
um quarto da proporção vital: um quarto é metade da metade, então o
o fogo queimou duas meias-vidas, $2 \times \num{5700} = \num{11400}$ anos, atrás.
\end{example}

\begin{definition}[Atividade]\label{def:g11:nucleus-radioactivity:activity}
O \emph{atividade}\index{atividade} $\mathcal A $ de uma amostra é o seu número
de decaimentos por \omterm{def:g10:orders-of-magnitude:unit}{segundo}, medido em \emph{bequerels}\index{bequerels}:
$\qty{1}{Bq} = 1$ decadência por \omterm{def:g10:orders-of-magnitude:unit}{segundo}. Menos \omterm{def:g11:fundamental-interactions:ladder}{núcleos} sobreviventes significa menos
decai, então a atividade também cai pela metade a cada \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida}.
\end{definition}

\begin{example}[Ordens de magnitude]\label{ex:g11:nucleus-radioactivity:orders}
Tudo está ligeiramente radioativo. Uma banana: cerca de \qty{15}{Bq}
(potássio-40). Um corpo humano: cerca de \qty{8000}{Bq} --- o gancho deste capítulo,
cerca de quinhentas bananas. Um cúbico \omterm{def:g10:orders-of-magnitude:unit}{medidor} de granito: ao redor
\qty{3e6}{Bq} (urânio e sua cadeia, de onde vem o radônio da adega). Um tecnécio
injeção: cerca de \qty{5e8}{Bq}, desapareceu em poucos dias.
\end{example}

\section{Servos e perigos}

\begin{example}[Três carreiras de um núcleo instável]\label{ex:g11:nucleus-radioactivity:uses}
\emph{Medicamento}: tecnécio-99m, um puro $\gamma $ emissor com
$ T_{1/2} = 6$ horas, é fixado em um molécula o órgão alvo absorve; um
câmera filma o $\gamma $ s de fora, e no dia seguinte o rastreador
praticamente desapareceu --- enquanto a radioterapia inverte a lógica, focando os feixes para
destruir um tumor. \emph{Namorando}: carbono-14 para costas de madeira, osso e tecido
alguns \num{50000} anos; urânio-238 para rochas --- e a Terra, em
\num{4.5e9} anos. \emph{Detectores de fumaça}: uma partícula de amerício-241 ioniza
o ar de uma pequena câmara, deixando fluir uma pequena corrente; a fumaça sufoca o
atual e o alarme dispara --- o $\alpha $ morrem em centímetros de ar.
\end{example}

\begin{remark}[Dose e peneira]\label{rem:g11:nucleus-radioactivity:dose}
A radiação arranca elétrons \omterm{def:g11:fundamental-interactions:ladder}{moléculas} e pode danificar o DNA. O biológico
o dano é rastreado pelo \emph{dose}\index{dose}, em
\emph{sieverts}\index{sieverts} (\unit{Sv}), que
pesa o energia depositado por \omterm{def:g10:orders-of-magnitude:unit}{quilograma} de tecido pela forma como danifica cada
radiação é. Três regras: distância, blindagem ($\alpha $: pele é suficiente ---
a menos que o emissor seja \emph{inalado}, crime do radônio; $\beta $: alumínio;
$\gamma $: lead) e tempo. A dosimetria propriamente dita é material universitário.
\end{remark}

\section{Problema: A múmia, o reator e o detector de fumaça}

\begin{problem}[{Problema de fim de semana --- três carreiras de um núcleo instável:
uma múmia datada pela metade, árvore genealógica do urânio equilibrada, um detector de fumaça
auditado, e por que as estrelas tinham que ser mortais}]
\label{pb:g11:nucleus-radioactivity:1}
Três \omterm{def:g11:fundamental-interactions:ladder}{núcleos} instáveis, três empregos: o carbono-14 marca a data e hora de cada plant morta
e animal, o urânio-238 aquece um planeta por dentro, o amerício-241 observa
tetos. Dados: $ T_{1/2} = \num{5700}$ anos (C-14), \num{4.5e9} anos
(U-238), $432$ anos (Am-241); shows de carbono vivo $13.6$ decai por minuto
por grama; números atômicos:N $7$, Ó $8$, Pb $82$, Bi $83$, Pó $84$, Rn $86$,
Rá $88$, O $90$, pai $91$, você $92$, Np $93$, Sou $95$.

\medskip
\noindent\textbf{Part I --- The mummy.}
\begin{enumerate}
  \item Dê a composição (prótons, nêutrons) de ${}^{12}_{6}\mathrm{C}$
        e ${}^{14}_{6}\mathrm{C}$. Por que química idêntica em um corpo?
  \item Por que a proporção de carbono-14 é constante em um corpo vivo, e
        por que começa a cair com a morte?
  \item Escreva a equação de decaimento do carbono-14. Sua radiação escaparia
        através dos embrulhos de linho da múmia?
  \item Um grama de carbono do linho mostra $6.8$ decai por minuto.
        Que fração da taxa de vida é essa? Namore a múmia.
  \item Uma segunda múmia, prêmio de coleção particular, mostra
        $13.2$ decai por minuto por grama. Veredicto?
  \item Estime a taxa a partir de uma amostra de dez meias-vidas e explique
        por que a datação por carbono desaparece além de aproximadamente \num{57000} anos.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The reactor under the meadow.}
\begin{enumerate}[resume]
  \item Dê a composição de ${}^{238}_{92}\mathrm{U}$. Por que pesado
        os \omterm{def:g11:fundamental-interactions:ladder}{núcleos} precisam de tal excedente de nêutrons?
  \item Escreva seu $\alpha $ equação de decaimento.
  \item A filha então decai $\beta^-$ e a neta
        $\beta^-$ de novo. Escreva ambas as equações. Qual elemento reaparece?
  \item A cadeia termina em ${}^{206}_{82}\mathrm{Pb}$. Conte o seu $\alpha $
        passos, então é $\beta^-$ passos.
  \item A Terra se formou \num{4.5e9} anos atrás. Que fração de seu
        o urânio-238 primordial permanece até hoje? E em $5$ mais bilhões de anos?
  \item Em uma frase: o que esse calor de decomposição enterrado \omterm{def:g11:work-of-force:power}{poder} no
        superfície? (Pense em vulcões e continentes à deriva.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The smoke detector.}
\begin{enumerate}[resume]
  \item Escreva o $\alpha $ equação de decaimento de ${}^{241}_{95}\mathrm{Am}$.
  \item Isso é \omterm{def:g11:nucleus-radioactivity:activity}{atividade} é \qty{33}{kBq}: quantas decaições por \omterm{def:g10:orders-of-magnitude:unit}{segundo}? Por dia?
  \item Explique o detector: o que o $\alpha $ partículas fazem ao ar
        da câmara, e o que a fumaça muda?
  \item Por que é um $\alpha $ emissor a escolha certa aqui --- e um
        $\gamma $ emissor de igual \omterm{def:g11:nucleus-radioactivity:activity}{atividade} o pior?
  \item Depois de um \omterm{def:g11:nucleus-radioactivity:halflife}{meia-vida} ($432$ anos!) o \omterm{def:g11:nucleus-radioactivity:activity}{atividade} seria
        \qty{16.5}{kBq}; o manual pede substituição após dez anos.
        A fonte é o elo mais fraco?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Why the stars had to be mortal.}
\begin{enumerate}[resume]
  \item Estrelas moribundas forjaram e espalharam todos os elementos mais pesados que
        hélio. Que fração de urânio-238 foi forjada \num{9e9} anos atrás
        sobrevive hoje? Por que a abundância restante da Terra faz com que a
        planeta muito mais jovem que as estrelas mais antigas?
  \item O calor radioativo interno da Terra impulsiona o vulcanismo e as placas
        tectônica, e mantém o \omterm{def:g11:fundamental-interactions:ladder}{núcleo} produzindo o escudo magnético de
        \cref{ch:g11:magnetic-fields}. O que seria um planeta apenas estável
        \omterm{def:g11:fundamental-interactions:ladder}{átomos} ser como?
  \item Uma linha, como poesia e como escrituração contábil: onde estavam seus \omterm{def:g11:fundamental-interactions:ladder}{átomos}
        forjado, e o que ainda aquece o chão sob seus pés?
\end{enumerate}
\end{problem}
