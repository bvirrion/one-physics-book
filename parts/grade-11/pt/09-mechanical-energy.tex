\chapter{Energia Mecânica e sua Conservação}\label{ch:g11:mechanical-energy}

Uma montanha-russa não tem \omterm{def:g11:work-of-force:sign}{motor} depois da primeira colina: cada volta, cada
A explosão de velocidade depois é gasta na conta aberta na subida.
Este capítulo transforma a contabilidade em dois números --- cinético e
potencial energia --- e uma regra: enquanto apenas \omterm{def:g10:universal-gravitation:weight}{peso} funciona, sua soma
não muda. \omterm{def:g10:inertia:inventory}{Atrito} é a taxa; aprenderemos a lê-lo no
livros.

\section{Energia cinética}

\begin{definition}[Energia cinética]\label{def:g11:mechanical-energy:kinetic}
Um corpo de massa $m $ (\unit{kg}) movendo-se em velocidade $ v $ (\unit{m/s}) carrega
o \emph{energia cinética}\index{energia cinética}
\[
E_k = \tfrac12\, m v^2 .
\]
\omterm{def:g10:orders-of-magnitude:unit}{Unidades} verificar: $\unit{kg.m^2/s^2} = \unit{N.m} = \unit{J}$ --- \omterm{def:g11:work-of-force:work}{joules},
como todo energia (\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Ordens de magnitude]\label{ex:g11:mechanical-energy:orders}
\begin{itemize}
  \item um pedestre, \qty{70}{kg} no \qty{1.4}{m/s}:
        $ E_k = \tfrac12 \times 70 \times 1.4^2 \approx \qty{69}{J}$;
  \item um carro, \qty{1300}{kg} no \qty{130}{km/h} $= \qty{36.1}{m/s}$:
        $ E_k \approx \qty{8.5e5}{J}$;
  \item um trem de alta velocidade, $ m = \qty{3.85e5}{kg}$, no \qty{320}{km/h}
        $= \qty{88.9}{m/s}$: $ E_k \approx \qty{1.5e9}{J}$;
  \item uma bala de rifle, \qty{8.0}{g} no \qty{800}{m/s}: $ E_k \approx \qty{2.6}{kJ}$.
\end{itemize}
Sete ordens de grandeza separam o passeio do trem. E $ E_k $
cresce como o \emph{quadrado} da velocidade: em \qty{130}{km/h} o carro
carrega $4$ vezes o energia tem em \qty{65}{km/h} --- quatro vezes
o que seus freios devem remover.
\end{example}

\section{O teorema da energia cinética}

\begin{theorem}[Teorema da energia cinética]\label{thm:g11:mechanical-energy:ketheorem}
\index{teorema da energia cinética}
Para um corpo de massa $ m $ movendo-se ao longo de uma linha reta de $ A $ para $ B $,
a mudança de \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} é igual ao total \omterm{def:g11:work-of-force:work}{trabalhar}
(\cref{ch:g11:work-of-force}) recebido:
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
a soma que cobre tudo forças agindo sobre o corpo. Estende-se até
movimento ao longo de qualquer caminho, como também admitimos aqui.
\end{theorem}
\admitted

\begin{example}[Queda livre, verificada quadro a quadro]\label{ex:g11:mechanical-energy:freefall}
UM \qty{0.200}{kg} a bola é lançada e filmada; os quadros dão:
\begin{center}\small
\begin{tabular}{lcccc}
$ t $ (\unit{s}) & 0.10 & 0.20 & 0.30 & 0.40\\
fallen height $ h $ (\unit{m}) & 0.049 & 0.196 & 0.441 & 0.785\\
speed $ v $ (\unit{m/s}) & 0.98 & 1.96 & 2.94 & 3.92\\
$ E_k = \tfrac12 mv^2$ (\unit{J}) & 0.096 & 0.384 & 0.864 & 1.54\\
$ W = mgh $ (\unit{J}) & 0.096 & 0.385 & 0.865 & 1.54
\end{tabular}
\end{center}
A partir de $ E_k = 0$, em cada quadro $ E_k $ corresponde ao \omterm{def:g11:work-of-force:work}{trabalhar} $ mgh $ do \omterm{def:g10:universal-gravitation:weight}{peso}: $\Delta E_k = W(\vect P)$, ou seja\ $ v^2 = 2gh $ --- o teorema,
verificada a precisão das leituras.
\end{example}

\begin{remark}[Onde mora a prova]\label{rem:g11:mechanical-energy:proof}
A afirmação geral --- válida ao longo de qualquer caminho, não apenas em linha reta
linha --- segue a segunda lei de Newton e um pouco de cálculo, ambos
mais adiante neste livro (\cref{ch:g12:newtons-laws,%
ch:g12:work-and-energy}); até então, a verificação de queda livre é nossa
garantia, e admitimos a versão de caminho curvo e a usamos livremente.
\end{remark}

\section{Energia potencial gravitacional}

\begin{definition}[Energia potencial gravitacional]\label{def:g11:mechanical-energy:potential}
Um corpo de massa $ m $ em altitude $ z $, medido para cima a partir de um escolhido
\emph{nível de referência}\index{nível de referência} onde $ z = 0$, armazena o
\emph{energia potencial gravitacional}\index{energia potencial gravitacional}
\[
E_p = m g z, \qquad g = \qty{9.81}{N/kg}.
\]
\end{definition}

\begin{remark}[Só as diferenças importam]\label{rem:g11:mechanical-energy:reference}
Movendo-se de $ A $ para $ B $ mudanças $ E_p $ por $ mg(z_B - z_A) =
-W_{AB}(\vect P)$: o oposto do \omterm{def:g11:work-of-force:work}{trabalhar} do \omterm{def:g10:universal-gravitation:weight}{peso}
(\cref{ch:g11:work-of-force}). O \omterm{def:g11:mechanical-energy:potential}{nível de referência} é uma escolha livre ---
piso, nível do mar, tampo da mesa --- mudança $ E_p $ em todos os lugares por uma constante
isso desaparece de toda diferença. Coloque onde estão os números
mais simples e não o mova no meio do problema.
\end{remark}

\begin{example}[Um livro em uma estante]\label{ex:g11:mechanical-energy:shelf}
UM \qty{1.2}{kg} livro fica \qty{1.8}{m} acima do chão, ele próprio
\qty{9.0}{m} acima da rua. Referência no chão:
$ E_p = 1.2 \times 9.81 \times 1.8 \approx \qty{21}{J}$; na rua:
$ E_p \approx \qty{127}{J}$. Caindo no chão libera \qty{21}{J} em
ambas as contas.
\end{example}

\section{Energia mecânica e sua conservação}

\begin{definition}[Energia mecânica]\label{def:g11:mechanical-energy:em}
O \emph{energia mecânica}\index{energia mecânica} de um corpo é
\[
E_m = E_k + E_p = \tfrac12 m v^2 + m g z .
\]
\end{definition}

\begin{theorem}[Conservação de energia mecânica]\label{thm:g11:mechanical-energy:conservation}
\index{conservação de energia mecânica}
Se o único \omterm{def:g10:inertia:force}{vigor} \omterm{def:g11:work-of-force:work}{trabalhar} em um corpo é o seu \omterm{def:g10:universal-gravitation:weight}{peso} (não \omterm{def:g10:inertia:inventory}{atrito}; outro
forças, como a reação de uma pista sem \omterm{def:g10:inertia:inventory}{atrito}, perpendicular ao
movimento), $ E_m $ é conservado: para quaisquer dois pontos $ A $, $ B $ do movimento,
\[
\tfrac12 m v_A^2 + m g z_A = \tfrac12 m v_B^2 + m g z_B .
\]
\end{theorem}

\begin{proof}
Pelo \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} teorema, $ E_k(B) - E_k(A) = W_{AB}(\vect P) =
mg(z_A - z_B) = E_p(A) - E_p(B)$: o que $ E_k $ \omterm{def:g10:signals-and-waves:oscilloscope}{ganhos}, $ E_p $ perde.
\end{proof}

\begin{example}[Toboágua]\label{ex:g11:mechanical-energy:slide}
Uma criança começa \omterm{def:g10:relative-motion:frame}{em repouso} no topo de um escorregador de altura sem \omterm{def:g10:inertia:inventory}{atrito}
$ h = \qty{3.2}{m}$. Referência na parte inferior: $ mgh = \tfrac12 mv^2$, então
$ v = \sqrt{2gh} \approx \qty{7.9}{m/s}$ (\qty{29}{km/h}) --- tanto faz
a massa \emph{e qualquer que seja o formato do slide}: reto, curvo
ou espiral, apenas entra a queda de altura.
\end{example}

\begin{example}[Pêndulo]\label{ex:g11:mechanical-energy:pendulum}
Um pêndulo é puxado para o lado até subir $ h = \qty{12}{cm}$ acima
seu ponto mais baixo e então liberado. Nos extremos $ v = 0$; na parte inferior
o todo $ mgh $ é cinético, $ v = \sqrt{2gh} \approx \qty{1.5}{m/s}$;
do outro lado, o bob sobe exatamente \qty{12}{cm}.
\end{example}

\section{Dissipação por fricção}

\begin{definition}[Dissipação]\label{def:g11:mechanical-energy:dissipation}
\omterm{def:g10:energy-conservation:energy}{Energia} sair da conta mecânica é considerado
\emph{dissipado}\index{dissipado}: reaparece como energia térmica em
as superfícies de \omterm{def:g10:inertia:inventory}{atrito} e o ar. Total energia é conservado
(\cref{ch:g10:energy-conservation}); \omterm{def:g11:mechanical-energy:em}{energia mecânica} sozinho não é.
\end{definition}

\begin{proposition}[O déficit mede o atrito]\label{prop:g11:mechanical-energy:deficit}
Se um \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g10:inertia:force}{vigor} $\vect f $ também funciona ao longo do movimento,
\[
\Delta E_m = E_m(B) - E_m(A) = W_{AB}(\vect f) < 0:
\]
a perda de \omterm{def:g11:mechanical-energy:em}{energia mecânica} é igual ao \omterm{def:g11:work-of-force:work}{trabalhar} de \omterm{def:g10:inertia:inventory}{atrito}.
\end{proposition}

\begin{proof}
O \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} teorema agora é lido $\Delta E_k = W_{AB}(\vect P) +
W_{AB}(\vect f) = -\Delta E_p + W_{AB}(\vect f)$; mover $\Delta E_p $
entre. \omterm{def:g10:inertia:inventory}{Atrito} se opõe ao movimento, então é \omterm{def:g11:work-of-force:work}{trabalhar} é negativo.
\end{proof}

\begin{example}[Auditoria de tobogã]\label{ex:g11:mechanical-energy:toboggan}
Uma criança e um trenó ($ m = \qty{40}{kg}$) comece \omterm{def:g10:relative-motion:frame}{em repouso}, solte
$ h = \qty{5.0}{m}$ junto \qty{12}{m} de neve, e chegar a
\qty{7.0}{m/s} em vez do sem \omterm{def:g10:inertia:inventory}{atrito}
$\sqrt{2gh} \approx \qty{9.9}{m/s}$. Referência na parte inferior:
$\Delta E_m = \tfrac12 \times 40 \times 7.0^2 - 40 \times 9.81
\times 5.0 = 980 - 1962 = \qty{-982}{J}$. Então $ W(\vect f) =
\qty{-982}{J}$ sobre \qty{12}{m}: uma média \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g10:inertia:force}{vigor}
$ f = 982/12 \approx \qty{82}{N}$, e \qty{982}{J} de calor no
corredores e a neve.
\end{example}

\begin{method}[Escrituração de energia]\label{met:g11:mechanical-energy:bookkeeping}
\begin{enumerate}
  \item Escolha um \omterm{def:g11:mechanical-energy:potential}{nível de referência} e guarde-o; nome $ A $ (dados), $ B $
        (pergunta).
  \item Escrever $ E_m = \tfrac12 mv^2 + mgz $ no $ A $ e em $ B $.
  \item Não \omterm{def:g10:inertia:inventory}{atrito}: $ E_m(A) = E_m(B)$; resolver --- normalmente
        $ v_B = \sqrt{v_A^2 + 2g(z_A - z_B)}$.
  \item \omterm{def:g10:inertia:inventory}{Atrito}: $ E_m(B) - E_m(A) = -f\ell $ ($\ell $ = comprimento do caminho);
        resolver o desconhecido.
  \item Verifique: o $ E_k $/$ E_p $ pilha de barras para $ E_m $, nível, a menos que \omterm{def:g10:inertia:inventory}{atrito}.
\end{enumerate}
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:mechanical-energy:1}
Calcule o \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} de (a) um \qty{90}{kg} scooter e piloto em
\qty{25}{km/h}; (b) um \qty{1200}{kg} carro em \qty{130}{km/h}. (c) Por que
fator faz $ E_k $ mudar quando a velocidade dobrar? Triplos?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:mechanical-energy:2}
UM \qty{75}{kg} \omterm{def:g10:signals-and-waves:oscilloscope}{ganhos} de alpinista \qty{850}{m} de altitude. Calcule o \omterm{def:g10:signals-and-waves:oscilloscope}{ganho}
de potencial energia. Depende do \omterm{def:g11:mechanical-energy:potential}{nível de referência}? Na rota?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:mechanical-energy:3}
Um vaso de flores cai \qty{20}{m} de uma varanda. Desprezando a \omterm{def:g11:circuits-and-power:resistance}{resistência} do ar,
encontre sua velocidade no solo em \unit{m/s} e \unit{km/h}. Para onde foi a massa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:mechanical-energy:4}
UM \qty{1200}{kg} freios do carro de \qty{25}{m/s} descansar em \qty{50}{m}.
Use o \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} teorema para encontrar o \omterm{def:g11:work-of-force:work}{trabalhar} dos freios, então o
frenagem média \omterm{def:g10:inertia:force}{vigor}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:mechanical-energy:5}
Um pêndulo é liberado \qty{8.0}{cm} acima do seu ponto mais baixo. Encontrar
sua velocidade na parte inferior e a altura alcançada no outro lado.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:mechanical-energy:6}
Usando \cref{ex:g11:mechanical-energy:orders}: por qual fator o
trem \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} exceder a bala? Até que altura poderia o seu próprio
$ E_k $ levante o trem ($ h = v^2/2g $)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:mechanical-energy:7}
Dois escorregadores sem \omterm{def:g10:inertia:inventory}{atrito} caem do mesmo \qty{3.0}{m} plataforma, um
reto e íngreme, um longo e sinuoso. Compare as velocidades de chegada
(calculá-los) e os tempos de chegada (sem cálculo).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:mechanical-energy:8}
Uma bola é lançada diretamente para cima em \qty{12}{m/s}. Encontre a altura máxima
alcançado, então a velocidade em \qty{4.0}{m} --- subindo e subindo
para baixo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:mechanical-energy:9}
Um saltador de esqui sai da decolagem às \qty{23}{m/s} e terras \qty{35}{m}
mais baixo. Prever a velocidade de pouso sem \omterm{def:g11:circuits-and-power:resistance}{resistência} do ar; é o verdadeiro
valor maior ou menor, e por quê?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:mechanical-energy:10}
UM \qty{45}{kg} trenó começa \omterm{def:g10:relative-motion:frame}{em repouso}, cai \qty{6.0}{m} junto
\qty{15}{m} de declive, chegando a \qty{8.0}{m/s}. Encontre o mecânico
energia perdido, então a média \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g10:inertia:force}{vigor}. Onde foi que energia ir?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:mechanical-energy:11}
A estaca de Galileu: um pêndulo liberado \qty{15}{cm} acima do seu ponto mais baixo
encontra uma estaca bloqueando a metade superior da corda na \omterm{def:g10:universal-gravitation:weight}{vertical}. Encontrar
a velocidade na parte inferior e a subida além do pino; o que o pino
mudar, e o que não?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:mechanical-energy:12}
Um skatista cai em um half-pipe vindo de \qty{2.5}{m}; cada travessia
(lado a lado) se dissipa $10\%$ do \omterm{def:g11:mechanical-energy:em}{energia mecânica}. Encontre o
velocidade na parte inferior da primeira descida, então --- calculando a
alturas de subida sucessivas --- a primeira travessia terminando abaixo \qty{1.0}{m}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:mechanical-energy:13}
De um \qty{45}{m} no topo de um penhasco, uma pedra é atirada \qty{15}{m/s} ---
para cima, horizontalmente ou em qualquer ângulo. Mostre que a velocidade de pouso é a
mesmo em todos os casos e calcule-o. O que \emph{faz} depende do ângulo?

\section{Problema: Projetando o loop}

\begin{problem}[{Problema de fim de semana --- uma auditoria energética de um rolo
montanha-russa: a altura de lançamento que um loop determina, as velocidades que ele promete, o
atrito confessado por dois sensores, os freios que fecham a conta ---
todo o passeio pendurado em um número}]
\label{pb:g11:mechanical-energy:1}
Um trem de massa $ m = \qty{2.0e3}{kg}$ é liberado \omterm{def:g10:relative-motion:frame}{em repouso} de um lançamento
colina de altura $ H $ e costas, sem \omterm{def:g11:work-of-force:sign}{motor}, através de uma circular \omterm{def:g10:universal-gravitation:weight}{vertical}
loop de raio $ R = \qty{8.0}{m}$ cujo topo fica na altura $2R $.
A análise do movimento circular (ano que vem) fornece o único dado que tomamos emprestado: em
o topo do loop que o trem pressiona nos trilhos somente se
$ v_{\text{top}}^2 \geq gR $. Partes I e II ignoram \omterm{def:g10:inertia:inventory}{atrito}.

\medskip
\noindent\textbf{Part I --- The hill the loop dictates.}
\begin{enumerate}
  \item Calcule a velocidade mínima segura no topo do loop.
  \item Terreno como \omterm{def:g11:mechanical-energy:potential}{nível de referência}: calcule o trem mecânico
        energia no topo do loop nessa velocidade mínima.
  \item Deduzir $ H_{\min}$ e mostrar $ H_{\min} = 5R/2$ --- sem massa, não $ g $.
  \item A equipe leva $ H = \qty{24}{m}$: calcula o topo do loop real
        velocidade e proporção $ v_{\text{top}}^2/(gR)$.
  \item Onde foi $ m $ ir? Por que o design cabe cheio e vazio
        trens iguais?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- What the height promises} ($ H = \qty{24}{m}$).
\begin{enumerate}[resume]
  \item Velocidade na parte inferior do loop ($ z = 0$), em \unit{m/s} e
        \unit{km/h}?
  \item Velocidade na lateral do loop ($ z = R $)?
  \item Calcular $ E_p $ e $ E_k $ no lançamento, na parte inferior, lateral e superior do
        o laço; verifique se cada par soma $ mgH $.
  \item UM \qty{12}{m} A colisão nas costas do camelo segue o loop: velocidade no topo?
  \item Dê a fórmula geral para $ v(z)$; onde é o passeio
        mais rápido e mais lento?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g10:inertia:inventory}{atrito} audit.}
Dois sensores na mesma altura $ z = \qty{2.0}{m}$, \qty{60}{m} de
rastrear, ler $ v_1 = \qty{20.4}{m/s}$ então $ v_2 = \qty{19.2}{m/s}$.
\begin{enumerate}[resume]
  \item Calcule a mudança de \omterm{def:g11:mechanical-energy:em}{energia mecânica} entre os sensores.
        O que tornou inteligente o posicionamento na mesma altura?
  \item Deduza o \omterm{def:g11:work-of-force:work}{trabalhar} de \omterm{def:g10:inertia:inventory}{atrito} e a média \omterm{def:g10:inertia:inventory}{atrito} \omterm{def:g10:inertia:force}{vigor}.
  \item Onde está o desaparecido energia perdido? Nomeie o formulário e os locais.
  \item Que fração do trem $ E_m $ está perdido por \qty{60}{m}? Por que
        seria o nu $ H_{\min}$ ser inseguro no mundo real e difícil?
  \item Sobre \qty{90}{m} da trilha lançamento separado e topo do loop: estimativa
        $ E_m $ lá; a margem da pergunta 4 ainda salva o loop?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Closing the account.}
O último sensor, na entrada da frenagem horizontal reta
($ z = 0$), lê \qty{18.0}{m/s}.
\begin{enumerate}[resume]
  \item Calcule o trem \omterm{def:g11:mechanical-energy:kinetic}{energia cinética} lá.
  \item Os freios devem pará-lo $ d = \qty{45}{m}$: calcular o
        frenagem necessária (constante) \omterm{def:g10:inertia:force}{vigor}.
  \item Verificação de conforto: calcule a desaceleração $ a = F/m $; comparar com $ g $.
  \item Quanto energia rastreou \omterm{def:g10:inertia:inventory}{atrito} dissipar-se por todo
        passeio? Verifique isso \omterm{def:g10:inertia:inventory}{atrito} $+$ freios $= mgH$, para o \omterm{def:g10:energy-conservation:energy}{joule}.
  \item Punchline, uma frase: qual número fixou o loop
        segurança, cada velocidade, o \omterm{def:g10:inertia:inventory}{atrito} conta e o freio \omterm{def:g10:inertia:force}{vigor} ---
        e qual foi o preço de ignorar \omterm{def:g10:inertia:inventory}{atrito}?
\end{enumerate}
\end{problem}
