\chapter{Magnetismo e Campos Magnéticos}\label{ch:g11:magnetic-fields}

Marinheiros guiados por agulhas magnetizadas mil anos antes
alguém sabia por que eles apontam para o norte. A resposta veio em 1820, quando um
a corrente fez a bússola se contorcer: o magnetismo é produzido por cargas em movimento.
Este capítulo mapeia o \omterm{def:g11:magnetic-fields:bfield}{campo magnético} --- de ímãs, da Terra, de
correntes --- então fecha o loop: o campo empurra de volta as correntes,
é assim que todo motor elétrico gira.

\section{Ímãs e o campo magnético}

\begin{definition}[Ímã e pólos]\label{def:g11:magnetic-fields:magnet}
UM \emph{quero}\index{quero} atrai o ferro e, pendurado por um fio,
vira-se para o norte. Sua ação concentra-se em dois \emph{magnético
postagens}\index{magnético
postagens}, \emph{norte} (o fim que busca
norte geográfico) e \emph{sul}: pólos semelhantes se repelem, ao contrário de se atraem.
Serrar um ímã ao meio resulta em dois ímãs completos --- sem experimento
já isolou um poste.
\end{definition}

\begin{definition}[Campo magnético]\label{def:g11:magnetic-fields:bfield}
UM ímã modifica o espaço ao seu redor
(\cref{ch:g11:electric-gravitational-fields}): cria em todos os lugares um
\emph{campo magnético}\index{campo magnético} $\vect B $. Uma bússola lê
sua direção (a agulha se alinha com $\vect B $, extremidade norte para frente); um
\emph{Salão Sonda}\index{Salão Sonda} mede sua magnitude, em
\emph{teslas}\index{teslas} (\unit{T}).
\end{definition}

\begin{definition}[Linhas de campo magnético]\label{def:g11:magnetic-fields:lines}
\emph{Linhas de campo magnético}\index{Linhas de campo magnético} são curvas
tangente a $\vect B $, aglomeração onde campo é \omterm{def:g11:fundamental-interactions:strong}{forte}. Fora de um
ímã eles vão do pólo norte ao sul; diferente \omterm{def:g11:electric-gravitational-fields:efield}{campo elétrico} linhas
eles não têm fim: cada um se fecha através do ímãcorpo.
\end{definition}

\section{A Terra é um ímã}

\begin{definition}[O campo magnético da Terra]\label{def:g11:magnetic-fields:earth}
As correntes no \omterm{def:g11:fundamental-interactions:ladder}{núcleo} de ferro fundido da Terra tornam o planeta um gigante
ímã: na superfície $ B \approx \qty{50}{\micro T}$ e um grátis
agulha se alinha com o campocomponente horizontal de --- este é o
bússola. Os pólos geomagnéticos não são os geográficos; o ângulo
entre o norte da bússola e o norte verdadeiro é o
\emph{declinação}\index{declinação}, corrigido em cada rolamento.
\end{definition}

\begin{example}[Corrigindo um rolamento]\label{ex:g11:magnetic-fields:bearing}
Onde o \omterm{def:g11:magnetic-fields:earth}{declinação} é $\ang{7}$ oeste, um navio dirigindo a `` bússola
norte'' para \qty{5.0}{km} desvios $5000 \times \sin\ang{7} \approx
\qty{6.1e2}{m}$ oeste de sua rota: para navegar para o norte verdadeiro, vire $\ang{7}$
\emph{leste} da agulha.
\end{example}

\section{O campo magnético de uma corrente}

\begin{remark}[Oersted, 1820]\label{rem:g11:magnetic-fields:oersted}
Durante uma palestra em 1820, Hans Christian Oersted viu a agulha de uma bússola
balanço quando ele fechou um circuito próximo: eletricidade e magnetismo foram
um assunto --- \emph{correntes criam campos magnéticos}.
\end{remark}

\begin{proposition}[Campo de um fio reto]\label{prop:g11:magnetic-fields:wire}
Um fio longo e reto transportando uma corrente $ I $ cria um campo cujo
linhas são círculos centrados no fio, em planos perpendiculares a ele.
À distância $ d $ do fio,
\[
B = \frac{\mu_0\, I}{2\pi d},
\qquad \mu_0 = 4\pi\times 10^{-7}\ \unit{T.m/A} \approx \qty{1.26e-6}{T.m/A}.
\]
\end{proposition}
\admitted

\begin{example}[Um fio é um ímã fraco]\label{ex:g11:magnetic-fields:wire}
No $ d = \qty{1.0}{cm}$ de um fio que transporta $ I = \qty{15}{A}$:
$ B = \num{2e-7} \times 15 / 0.010 = \qty{0.30}{mT}$ --- seis Terra
campos, de uma corrente que desarmaria um disjuntor doméstico. \omterm{def:g11:fundamental-interactions:strong}{Forte}
campos precisa de uma geometria melhor.
\end{example}

\begin{definition}[Bobina e solenóide]\label{def:g11:magnetic-fields:solenoid}
Enrolar o fio concentra seu campo. UM \emph{plano
bobina}\index{plano
bobina} ($ N $ vira um disco) se comporta como um \omterm{def:g11:lenses-and-eye:lens}{afinar} ímã,
uma está voltada para o norte, a outra para o sul; um \emph{solenóide}\index{solenóide} ---
um longo enrolamento cilíndrico, $ n $ voltas por \omterm{def:g10:orders-of-magnitude:unit}{medidor} --- como um bar ímã.
\end{definition}

\begin{proposition}[Campo dentro de um solenóide]\label{prop:g11:magnetic-fields:solenoidfield}
Dentro de um longo \omterm{def:g11:magnetic-fields:solenoid}{solenóide}, longe das extremidades, o campo é
\emph{uniforme}: paralelo ao eixo e igual em todos os interiores
ponto, de magnitude $ B = \mu_0\, n\, I $, independente do furo
largura. Lá fora, o campo é fraco.
\end{proposition}
\admitted

\begin{remark}
Ambas as fórmulas são derivadas honestamente no volume do Ano 1, da lei
relacionando um campoda circulação para a corrente que ela circunda. Observe o que
assuntos: $ n $, vira \emph{por \omterm{def:g10:orders-of-magnitude:unit}{medidor}} --- vento mais \omterm{def:g11:fundamental-interactions:strong}{forte}, não mais.
\end{remark}

\begin{method}[A mão direita, duas vezes]\label{met:g11:magnetic-fields:righthand}
\begin{itemize}
  \item \emph{Arame}: polegar = atual; dedos enrolados = giro das linhas.
  \item \emph{Bobina, \omterm{def:g11:magnetic-fields:solenoid}{solenóide}}: dedos enrolados = corrente nas voltas;
        polegar = $\vect B $ dentro, apontando para a face norte.
\end{itemize}
\end{method}

\end{example}

\section{Eletroímãs}

\begin{definition}[Eletroímã]\label{def:g11:magnetic-fields:electromagnet}
Um \emph{eletroímã}\index{eletroímã} é um \omterm{def:g11:magnetic-fields:solenoid}{solenóide} ferida em um
\omterm{def:g11:fundamental-interactions:ladder}{núcleo} de ferro macio: o ferro magnetiza no bobinade campo e multiplica
isso, normalmente por $100$ para $1000$. Ao contrário de um permanente ímã, isso
desliga com a corrente.
\end{definition}

\begin{remark}[Três máquinas]\label{rem:g11:magnetic-fields:machines}
O guindaste de ferro-velho levanta um carro com um \omterm{def:g11:magnetic-fields:electromagnet}{eletroímã} e --- o
ponto inteiro --- \emph{gotas} abrindo um interruptor. Um relé é pequeno
\omterm{def:g11:magnetic-fields:electromagnet}{eletroímã} fecha um contato: uma corrente fraca comuta um
\omterm{def:g11:fundamental-interactions:strong}{forte} (\cref{ch:g11:circuits-and-power}). Um scanner de ressonância magnética segura um uniforme
\qty{1.5}{T} ao redor do paciente com um supercondutor \omterm{def:g11:magnetic-fields:solenoid}{solenóide}.
\end{remark}

Ordens de magnitude, da galáxia a uma estrela morta:
\begin{center}
\small
\begin{tabular}{lr}
interstellar space & $\sim \qty{e-10}{T}$ \\
the Earth's surface campo & \qty{5e-5}{T} \\
a fridge ímã & $\sim \qty{5e-3}{T}$ \\
a neodymium ímã, at its pole & $\sim \qty{0.5}{T}$ \\
an MRI \omterm{def:g11:magnetic-fields:solenoid}{solenóide} & \qty{1.5}{T} \\
surface of a neutron star & $\sim \qty{e8}{T}$ \\
\end{tabular}
\end{center}

\section{A força de Laplace}

\begin{proposition}[Força de Laplace]\label{prop:g11:magnetic-fields:laplace}
Um fio reto de comprimento $ L $, transportando uma corrente $ I $ em um uniforme
campo $\vect B $ \emph{perpendicular} ao fio, sente o
\emph{Força de Laplace}\index{Força de Laplace} de magnitude $ F = I\,L\,B $,
perpendicular ao fio e $\vect B $: mão direita espalmada, polegar junto
os dedos atuais e retos ao longo $\vect B $ --- a palma empurra junto
$\vect F $. Um fio \emph{paralelo} para o campo não sente \omterm{def:g10:inertia:force}{vigor}.
\end{proposition}
\admitted

\begin{remark}[O trilho de salto]\label{rem:g11:magnetic-fields:rail}
O volume do Ano 1 deriva isso do magnético \omterm{def:g10:inertia:force}{vigor} em cada movimento
cobrar. No laboratório: uma barra de cobre repousa sobre duas
trilhos entre os pólos de um ímã (campo na página do
figura); feche a chave e a haste dispara ao longo dos trilhos. Reverso
a corrente ou inverta o ímã: atira para o outro lado.
\end{remark}

\begin{example}[Números de Laplace]\label{ex:g11:magnetic-fields:laplace}
Uma haste de comprimento $ L = \qty{5.0}{cm}$ carregando $ I = \qty{8.0}{A}$ entre
$ B = \qty{0.50}{T}$: $ F = 8.0 \times 0.050 \times 0.50 = \qty{0.20}{N}$
--- o \omterm{def:g10:universal-gravitation:weight}{peso} de \qty{20}{g}, em um fio pesando alguns gramas.
\end{example}

\begin{remark}[O motor CC]\label{rem:g11:magnetic-fields:motor}
Dobre o fio em um laço no campo: os dois lados perpendiculares a
$\vect B $ carregam correntes opostas, então seus Forças de Laplace são opostos
--- um par que gira o loop. Meia volta depois o par giraria
de volta, então uma chave rotativa (o \emph{comutador}\index{comutador}) inverte o
corrente a cada meia volta. Cada leque e furadeira é esse loop, multiplicado.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g11:magnetic-fields:1}
Um bar ímã é serrado ao meio entre seus postes. Quais pólos cada
peça transportar? Reunidos novamente, os rostos recém-cortados atraem
ou repelir? Qualquer esquema de corte pode isolar o pólo norte?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:magnetic-fields:2}
Verdadeiro ou falso, com um motivo cada: (a) dois linhas de \omterm{def:g11:magnetic-fields:bfield}{campo magnético}
cruz onde o campo é \omterm{def:g11:fundamental-interactions:strong}{forte}; (b) fora de um ímã, linhas de campo correr
do pólo norte ao pólo sul; (c) cada linha de \omterm{def:g11:magnetic-fields:bfield}{campo magnético} é um
circuito fechado; (d) uma agulha fica perpendicular à linha que a atravessa.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:magnetic-fields:3}
Calcule o campo \qty{2.0}{cm} de um fio reto que transporta
\qty{10}{A}. Quantas vezes a Terra \qty{50}{\micro T} é isso?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:magnetic-fields:4}
UM \omterm{def:g11:magnetic-fields:solenoid}{solenóide} de $800$ vira a ferida \qty{40}{cm} carrega
$ I = \qty{1.5}{A}$. Calcular $ n $, então $ B $ dentro; o que faz $ B $ tornar-se
se a corrente dobrar?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g11:magnetic-fields:5}
Um segmento de fio de comprimento \qty{10}{cm} carrega \qty{5.0}{A}
perpendicular a um \qty{0.20}{T} campo. Calcule o \omterm{prop:g11:magnetic-fields:laplace}{Força de Laplace}; o

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:magnetic-fields:6}
Onde um caminhante está, o \omterm{def:g11:magnetic-fields:earth}{declinação} é $\ang{8}$ oeste. Ela anda
\qty{2.0}{km} ``norte pela bússola'': quão longe e para que lado da verdade

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:magnetic-fields:7}
A que distância de um fio reto que transporta \qty{20}{A} faz o seu

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:magnetic-fields:8}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:magnetic-fields:9}

\begin{exercise}[$\star\star $]\label{exo:g11:magnetic-fields:10}

\begin{exercise}[$\star\star\star $]\label{exo:g11:magnetic-fields:12}

\begin{exercise}[$\star\star\star $]\label{exo:g11:magnetic-fields:13}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:magnetic-fields:14}
A espira retangular de um motor tem $ N = 50$ vira carregando
$ I = \qty{2.0}{A}$; seus dois lados de comprimento \qty{5.0}{cm} são
perpendicular a um campo $ B = \qty{0.30}{T}$. Calcule o \omterm{def:g10:inertia:force}{vigor} em cada
desses lados; por que os dois estão forças oposto, e o que eles fazem para
o laço? Por que os outros dois lados às vezes não sentem \omterm{def:g10:inertia:force}{vigor}? Por que deve
a corrente seja invertida a cada meia volta?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:magnetic-fields:15}
Quão perto de um cabo que transporta \qty{100}{A} você teria que ser para
é campo alcançar \qty{1}{T}? Compare com o milímetro do cabo
raio e conclua por que a força de Tesla campos são construídos a partir
\omterm{def:g11:magnetic-fields:solenoid}{solenóides}, não fios únicos. Então: a superfície de uma estrela de nêutrons campo de
\qty{e8}{T} é quantos poderes de dez acima da Terra?
\end{exercise}

\section{Problema: A bússola, o guindaste e a ressonância magnética}

\begin{problem}[{Problema de fim de semana --- três ímãs em ação: uma bússola
cruzando um oceano, um eletroímã levantando um carro e um solenóide de ressonância magnética
--- um campo, abrangendo dez potências de dez, realizando três trabalhos}]
\label{pb:g11:magnetic-fields:1}
O mesmo vetor $\vect B $ dirige um veleiro com \qty{50}{\micro T},
levanta sucata com \qty{1}{T}, e imagens de um joelho com \qty{1.5}{T}
--- três máquinas para as três fórmulas deste capítulo.

\medskip
\noindent\textbf{Part I --- The compass.}
Na posição da nave, a Terra campo tem componente horizontal
\qty{20}{\micro T} e componente \omterm{def:g10:universal-gravitation:weight}{vertical} \qty{44}{\micro T};
o \omterm{def:g11:magnetic-fields:earth}{declinação} é $\ang{6}$ oeste.
\begin{enumerate}
  \item Por que a agulha de uma bússola aponta para o norte (magnético)?
  \item Calcule o total campoa magnitude de e seu ângulo com a horizontal.
  \item Direcionando a ``bússola norte'' para \qty{30}{km}, a que distância a oeste de
        na trilha norte verdadeiro o navio termina?
  \item Carga de aço desvia ainda mais a agulha $\ang{4}$ oeste: mesma pergunta.
  \item Perto do pólo geomagnético uma bússola é inútil: qual componente
        é o culpado, e por quê?
\end{enumerate}

\noindent\textbf{Part II --- The crane.}
O ferro-velho \omterm{def:g11:magnetic-fields:electromagnet}{eletroímã} é um \omterm{def:g11:magnetic-fields:solenoid}{solenóide} de $600$ vira a ferida
\qty{30}{cm}, de \omterm{def:g11:circuits-and-power:resistance}{resistência} \qty{1.5}{\ohm}, alimentado por um \qty{12}{V}
fornecer; seu \omterm{def:g11:fundamental-interactions:ladder}{núcleo} de ferro multiplica o nu campo por $50$.
\begin{enumerate}[resume]
  \item Calcular $ n $, a densidade de giro.
  \item Calcule a corrente no bobina (\cref{ch:g11:circuits-and-power}).
  \item Calcule o bobina nua campo $\mu_0 n I $.
  \item Calcule o campo com o \omterm{def:g11:fundamental-interactions:ladder}{núcleo}; compare com o neodímio \qty{0.5}{T}.
  \item Calcule o poder dissipado no bobina, então o energia para
        um \qty{10}{\minute} turno de levantamento.
  \item Um permanente ímã esse \omterm{def:g11:fundamental-interactions:strong}{forte} existe. Por que um \omterm{def:g11:magnetic-fields:electromagnet}{eletroímã} de qualquer forma?
\end{enumerate}

\noindent\textbf{Part III --- The winch motor.}
O guincho do guindaste é um motor DC: um laço retangular de $ N = 100$ voltas,
lados do comprimento \qty{4.0}{cm} perpendicular a $ B = \qty{0.25}{T}$,
carregando $ I = \qty{3.0}{A}$.
\begin{enumerate}[resume]
  \item Calcule o \omterm{prop:g11:magnetic-fields:laplace}{Força de Laplace} em cada feixe lateral perpendicular a $\vect B $.
  \item Os dois forças são iguais e opostos, mas seu efeito não
        cancelar. O que eles produzem e por que devem ser \emph{oposto}?
  \item O que \omterm{def:g10:inertia:force}{vigor} atua nos outros dois lados, quando paralelo a $\vect B $?
  \item Depois de meia volta, o mesmo forças desfaria a rotação.
        O que o \omterm{rem:g11:magnetic-fields:motor}{comutador} fazer e quando?
  \item Forneça três alterações de design separadas, cada uma com o dobro do \omterm{def:g10:inertia:force}{vigor} par.
\end{enumerate}

\noindent\textbf{Part IV --- The MRI.}
O scanner \omterm{def:g11:magnetic-fields:solenoid}{solenóide} segura um uniforme \qty{1.5}{T} com uma corrente
$ I = \qty{500}{A}$.
\begin{enumerate}[resume]
  \item Quantas vezes a Terra campo é \qty{1.5}{T}?
  \item Calcule a densidade de giro $ n$ necessário \emph{sem} qualquer \omterm{def:g11:fundamental-interactions:ladder}{núcleo} de ferro.
  \item Se o enrolamento tivesse \omterm{def:g11:circuits-and-power:resistance}{resistência} \qty{0.20}{\ohm}, o que poder
        isso se dissiparia? Que propriedade do enrolamento real evita
        isso, ao preço do frio extremo?
  \item Punchline: coloque o navio, o guindaste e a ressonância magnética campos
        na escada das potências de dez do espaço interestelar
        (\qty{e-10}{T}) para uma estrela de nêutrons (\qty{e8}{T}); o que mudou
        entre as máquinas --- a física, ou o \omterm{def:g11:circuits-and-power:current}{amperes}?
\end{enumerate}
\end{problem}
