\chapter{The Nucleus and Radioactivity}\label{ch:g11:nucleus-radioactivity}

Right now, some eight thousand atomic nuclei explode inside your body every
second. Nothing attacks them: certain nuclei are born unstable, and each
sooner or later transforms itself, hurling out a fragment at a good fraction
of the speed of light. This chapter returns to the $10^{-15}$-metre floor of
\cref{ch:g11:fundamental-interactions} to see which nuclei are fragile, how
they break, and how their regular impatience dates caves and guards ceilings.

\section{Nuclei, nucleons and isotopes}

\begin{definition}[Nuclear notation]\label{def:g11:nucleus-radioactivity:notation}
A nucleus contains $Z$ protons (charge $+e$ each) and $N$ neutrons (no
charge), $A = Z + N$ nucleons in all; it is written ${}^{A}_{Z}\mathrm{X}$,
where $\mathrm X$ is the chemical symbol, $Z$ the \emph{atomic
number}\index{atomic number} and $A$ the \emph{mass number}\index{mass
number}. A species specified by $A$ and $Z$ is a \emph{nuclide}\index{nuclide};
nuclides with the same $Z$ but different $A$ are \emph{isotopes}\index{isotope}
of one element: same chemistry (the electron cloud sees only $Ze$),
different nuclei and stability.
\end{definition}

\begin{example}[Hydrogen and carbon]\label{ex:g11:nucleus-radioactivity:hc}
Hydrogen has three isotopes: ${}^{1}_{1}\mathrm{H}$ (a lone proton),
${}^{2}_{1}\mathrm{H}$ (deuterium: one proton, one neutron), unstable
${}^{3}_{1}\mathrm{H}$ (tritium: two neutrons). Natural carbon: mostly
${}^{12}_{6}\mathrm{C}$ ($6$ protons, $6$ neutrons), $1.1\%$ of
${}^{13}_{6}\mathrm{C}$, and one atom in $10^{12}$ of unstable
${}^{14}_{6}\mathrm{C}$ ($8$ neutrons), hero of \cref{ex:g11:nucleus-radioactivity:dating}.
\end{example}

\section{Stable and unstable nuclei}

\begin{proposition}[The valley of stability]\label{prop:g11:nucleus-radioactivity:valley}
Stable nuclides occupy a narrow band in the $(Z, N)$ plane: $N \approx Z$
for light nuclei, then a growing neutron excess, up to $N \approx 1.5\,Z$.
Every nuclide off the band --- too many neutrons, too many protons, or too
big (beyond lead, $Z = 82$, no nuclide is truly stable) --- transforms sooner or later.
\end{proposition}
\admitted

\begin{remark}
Locating the band exactly is quantum mechanics --- the university volumes.
The pattern is the ledger of \cref{ch:g11:fundamental-interactions}: strong
glue binds only neighbours, Coulomb repulsion spans the whole nucleus, so
heavy nuclei need extra neutron glue; yet an excess of neutrons is unstable
too (the weak interaction sees to it).
\end{remark}

\begin{definition}[Radioactivity]\label{def:g11:nucleus-radioactivity:radioactivity}
\emph{Radioactivity}\index{radioactivity} is the spontaneous transformation
of an unstable nucleus into another nuclide, with emission of radiation. It
is \emph{random}: nothing announces which nucleus decays next, and no
pressure, temperature or chemistry can hasten or delay it; only large
populations are predictable (\cref{def:g11:nucleus-radioactivity:halflife}).
\end{definition}

\begin{remark}[Becquerel and the Curies]\label{rem:g11:nucleus-radioactivity:curie}
In 1896 Henri Becquerel found that uranium salts fog a photographic plate
through black paper, in a closed drawer, with no energy supplied. Marie and
Pierre Curie showed the effect is atomic, measured it, and isolated polonium
and radium, a million times more active. The word \emph{radioactivity} is
Marie Curie's; so are two Nobel prizes.
\end{remark}

\section{Alpha, beta, gamma}

\begin{definition}[The three historic radiations]\label{def:g11:nucleus-radioactivity:abg}
Three radiations, named before anyone knew what they were:
\begin{itemize}
  \item \emph{alpha}\index{alpha radiation} ($\alpha$): a helium nucleus
        ${}^{4}_{2}\mathrm{He}$ ejected whole --- heavy, doubly charged,
        stopped by a sheet of paper or centimetres of air;
  \item \emph{beta}\index{beta radiation} ($\beta^-$): an electron
        ${}^{\;0}_{-1}\mathrm{e}$ created in the nucleus and ejected ---
        light, fast, stopped by millimetres of aluminium;
  \item \emph{gamma}\index{gamma radiation} ($\gamma$): electromagnetic
        radiation, far more energetic than light --- never quite stopped, only
        \emph{attenuated}\index{attenuation}: a centimetre of lead absorbs half.
\end{itemize}
\end{definition}

\begin{proposition}[Conservation in nuclear equations]\label{prop:g11:nucleus-radioactivity:soddy}
In every nuclear transformation, the total mass number $A$ and the total
charge --- the sum of the lower indices, counting $-1$ for an emitted
electron --- are the same before and after.
\end{proposition}
\admitted

\begin{remark}
Charge conservation already ruled \cref{ch:g11:fundamental-interactions};
conservation of $A$ says nucleons are only converted, never created or
destroyed --- honest bookkeeping for the university volumes.
\end{remark}

\begin{method}[Balancing a decay equation]\label{met:g11:nucleus-radioactivity:balance}
\begin{enumerate}
  \item Write every actor with both labels: $\alpha = {}^{4}_{2}\mathrm{He}$,
        $\beta^- = {}^{\;0}_{-1}\mathrm{e}$, $\beta^+ = {}^{0}_{+1}\mathrm{e}$; a $\gamma$ carries $A = Z = 0$.
  \item Equate the sums of the $A$'s, then of the $Z$'s
        (\cref{prop:g11:nucleus-radioactivity:soddy}); solve, and read the
        daughter element off its $Z$ in the periodic table.
\end{enumerate}
\end{method}

\begin{example}[An alpha decay]\label{ex:g11:nucleus-radioactivity:alpha}
Radium-226, the Curies' radium, is an $\alpha$ emitter:
${}^{226}_{88}\mathrm{Ra} \to {}^{A}_{Z}\mathrm{Y} + {}^{4}_{2}\mathrm{He}$
forces $A = 226 - 4 = 222$ and $Z = 88 - 2 = 86$; element $86$ is radon, so
${}^{226}_{88}\mathrm{Ra} \to {}^{222}_{86}\mathrm{Rn} + {}^{4}_{2}\mathrm{He}$
--- the radioactive gas of unventilated cellars.
\end{example}

\begin{example}[A beta-minus decay]\label{ex:g11:nucleus-radioactivity:beta}
Carbon-14 sits above the band ($8$ neutrons against $6$ protons):
${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} + {}^{\;0}_{-1}\mathrm{e}$.
Check: $14 = 14 + 0$, $6 = 7 - 1$; a neutron became a proton plus the
ejected electron --- the weak interaction at work.
\end{example}

\begin{remark}[Beta-plus and gamma]\label{rem:g11:nucleus-radioactivity:betaplus}
Below the band, the mirror decay $\beta^+$ turns a proton into a neutron
plus a \emph{positron}\index{positron} ${}^{0}_{+1}\mathrm{e}$, the electron's antiparticle:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$,
workhorse of PET scanners. And after most decays the daughter, born shaking,
settles by emitting a $\gamma$ on top.
\end{remark}

\section{Half-life and activity}

\begin{definition}[Half-life]\label{def:g11:nucleus-radioactivity:halflife}
The \emph{half-life}\index{half-life} $T_{1/2}$ of a nuclide is the time
after which half of any large sample has decayed. Each further half-life
halves what remains: of $N_0$ nuclei, $N_0/2$ are left at $T_{1/2}$, $N_0/4$
at $2\,T_{1/2}$, $N_0/8$ at $3\,T_{1/2}$, $N_0/2^{\,n}$ after $n$ half-lives.
\end{definition}

\begin{remark}
Randomness for one nucleus, clockwork for $10^{23}$: a vast crowd of
coin-flippers, unpredictable one by one, half eliminated each round.
Interpolating \emph{between} whole half-lives takes the exponential
function --- next year, with the mathematics volume.
\end{remark}

\begin{example}[Twenty powers of ten]\label{ex:g11:nucleus-radioactivity:range}
\begin{center}\small\begin{tabular}{llll}
polonium-214 & $\alpha$ & \qty{1.6e-4}{s} & radium chain link \\
technetium-99m & $\gamma$ & $6.0$ hours & medical imaging \\
americium-241 & $\alpha$ & $432$ years & smoke detectors \\
carbon-14 & $\beta^-$ & \num{5700} years & archaeology \\
potassium-40 & $\beta$ & \num{1.25e9} years & in every banana \\
uranium-238 & $\alpha$ & \num{4.5e9} years & Earth's inner heat \\
\end{tabular}\end{center}
\end{example}

\begin{example}[Carbon-14 dating]\label{ex:g11:nucleus-radioactivity:dating}
Living matter exchanges carbon with the atmosphere, so it holds the
atmospheric proportion of carbon-14; at death the intake stops and the
proportion halves every \num{5700} years. Charcoal from a painted cave shows
one quarter of the living proportion: a quarter is half of a half, so the
fire burned two half-lives, $2 \times \num{5700} = \num{11400}$ years, ago.
\end{example}

\begin{definition}[Activity]\label{def:g11:nucleus-radioactivity:activity}
The \emph{activity}\index{activity} $\mathcal A$ of a sample is its number
of decays per second, measured in \emph{becquerels}\index{becquerel}:
$\qty{1}{Bq} = 1$ decay per second. Fewer surviving nuclei means fewer
decays, so activity also halves at each half-life.
\end{definition}

\begin{example}[Orders of magnitude]\label{ex:g11:nucleus-radioactivity:orders}
Everything is slightly radioactive. A banana: about \qty{15}{Bq}
(potassium-40). A human body: about \qty{8000}{Bq} --- this chapter's hook,
some five hundred bananas' worth. A cubic metre of granite: around
\qty{3e6}{Bq} (uranium and its chain, whence cellar radon). One technetium
injection: about \qty{5e8}{Bq}, gone within days.
\end{example}

\section{Servants and dangers}

\begin{example}[Three careers of an unstable nucleus]\label{ex:g11:nucleus-radioactivity:uses}
\emph{Medicine}: technetium-99m, a pure $\gamma$ emitter with
$T_{1/2} = 6$ hours, is fixed to a molecule the target organ absorbs; a
camera films the $\gamma$s from outside, and by the next day the tracer has
mostly vanished --- while radiotherapy reverses the logic, focusing beams to
destroy a tumour. \emph{Dating}: carbon-14 for wood, bone and cloth back
some \num{50000} years; uranium-238 for rocks --- and the Earth, at
\num{4.5e9} years. \emph{Smoke detectors}: a speck of americium-241 ionizes
the air of a small chamber, letting a tiny current flow; smoke chokes the
current and the alarm fires --- the $\alpha$s die in centimetres of air.
\end{example}

\begin{remark}[Dose and the sievert]\label{rem:g11:nucleus-radioactivity:dose}
Radiation tears electrons off molecules and can damage DNA. The biological
harm is tracked by the \emph{dose}\index{dose}, in
\emph{sieverts}\index{sievert} (\unit{Sv}), which
weighs the energy deposited per kilogram of tissue by how damaging each
radiation is. Three rules: distance, shielding ($\alpha$: skin suffices ---
unless the emitter is \emph{inhaled}, radon's crime; $\beta$: aluminium;
$\gamma$: lead), and time. Dosimetry proper is university material.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:1}
Give the number of protons and of neutrons in ${}^{16}_{8}\mathrm{O}$,
${}^{27}_{13}\mathrm{Al}$, ${}^{60}_{27}\mathrm{Co}$, ${}^{235}_{92}\mathrm{U}$
and ${}^{3}_{1}\mathrm{H}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:2}
Among ${}^{14}_{6}\mathrm{C}$, ${}^{14}_{7}\mathrm{N}$, ${}^{12}_{6}\mathrm{C}$,
${}^{13}_{6}\mathrm{C}$ and ${}^{16}_{8}\mathrm{O}$, which are isotopes of one
another? Why are ${}^{14}_{6}\mathrm{C}$ and ${}^{14}_{7}\mathrm{N}$ \emph{not}
isotopes, despite equal mass numbers?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:3}
Polonium-210, ${}^{210}_{84}\mathrm{Po}$, is an $\alpha$ emitter. Write the
decay equation and name the daughter ($Z = 81$: thallium, $82$: lead, $83$:
bismuth).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:4}
Write the $\beta^-$ decay equations of cobalt-60 (${}^{60}_{27}\mathrm{Co}$)
and of iodine-131 (${}^{131}_{53}\mathrm{I}$); daughters: $Z = 28$ nickel,
$Z = 54$ xenon.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:5}
An iodine-131 source ($T_{1/2} = 8.0$ days) has activity \qty{800}{MBq}
today. What is its activity after $16$ days? After $32$ days?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:6}
Four radiations cross a strong magnetic field, then screens. Identify each:
(a) stopped by paper, barely deflected by the field; (b) crosses paper,
stopped by \qty{4}{mm} of aluminium, strongly deflected; (c) crosses both,
only attenuated by lead, undeflected; (d) like (b), deflected the other way.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:7}
Radon-222 (${}^{222}_{86}\mathrm{Rn}$) opens a fast chain: an $\alpha$
decay, then another $\alpha$, then a $\beta^-$. Write the three equations
($Z = 82$: lead, $83$: bismuth, $84$: polonium).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:8}
Fluorine-18, ${}^{18}_{9}\mathrm{F}$, is the tracer of PET scanners; stable
fluorine is ${}^{19}_{9}\mathrm{F}$. Which side of the stability band is it
on? Predict its decay mode and write the equation ($Z = 8$: oxygen).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:9}
A patient receives \qty{500}{MBq} of technetium-99m ($T_{1/2} = 6.0$ hours)
at 08:00. What is the activity at 08:00 the next morning? After how many
hours does it first drop below \qty{1}{MBq}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:10}
A bone from a peat bog shows a carbon-14 proportion one eighth of a living
bone's ($T_{1/2} = \num{5700}$ years). How old is the bone? And why is
carbon-14 useless for dating dinosaur bones (age about $10^{8}$ years)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:11}
Using the valley of stability, predict the decay mode ($\alpha$, $\beta^-$ or
$\beta^+$) of ${}^{14}_{6}\mathrm{C}$, ${}^{11}_{6}\mathrm{C}$ and
${}^{238}_{92}\mathrm{U}$, justifying each in one line (stable carbon:
${}^{12}_{6}\mathrm{C}$, ${}^{13}_{6}\mathrm{C}$).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:12}
The uranium-238 chain ends, many steps later, at stable
${}^{206}_{82}\mathrm{Pb}$; every step is an $\alpha$ or a $\beta^-$. Using
conservation of $A$ alone, find the number of $\alpha$ steps; then, with
conservation of $Z$, the number of $\beta^-$ steps.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:13}
A hospital's cobalt-60 source ($T_{1/2} = 5.3$ years) has activity
\qty{6.4}{TBq}; regulations allow disposal below \qty{0.1}{TBq}. After how
many years may it be disposed of?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:14}
Your body's activity is about \qty{8000}{Bq}. How many of your nuclei decay
per day? Over an $80$-year life? A banana adds about \qty{15}{Bq} while you
digest it: comment, in a sentence, on headlines that fear every becquerel.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:15}
A smoke detector holds \qty{0.2}{\micro\gram} of americium-241
($T_{1/2} = 432$ years), of activity about \qty{25}{kBq}. How many decays is
that per day? After ten years of service, has the americium decayed by much
less than half, about half, or much more --- and is source exhaustion why
detectors are replaced? Finally, why is this $\alpha$ source harmless on the
ceiling, yet dangerous if the capsule is ground up and the dust inhaled?
\end{exercise}

\section{Problem: The mummy, the reactor and the smoke detector}

\begin{problem}[{Weekend problem --- three careers of an unstable nucleus:
a mummy dated by halvings, uranium's family tree balanced, a smoke detector
audited, and why the stars had to be mortal}]
\label{pb:g11:nucleus-radioactivity:1}
Three unstable nuclei, three jobs: carbon-14 timestamps every dead plant
and animal, uranium-238 heats a planet from inside, americium-241 watches
ceilings. Data: $T_{1/2} = \num{5700}$ years (C-14), \num{4.5e9} years
(U-238), $432$ years (Am-241); living carbon shows $13.6$ decays per minute
per gram; atomic numbers: N $7$, O $8$, Pb $82$, Bi $83$, Po $84$, Rn $86$,
Ra $88$, Th $90$, Pa $91$, U $92$, Np $93$, Am $95$.

\medskip
\noindent\textbf{Part I --- The mummy.}
\begin{enumerate}
  \item Give the composition (protons, neutrons) of ${}^{12}_{6}\mathrm{C}$
        and ${}^{14}_{6}\mathrm{C}$. Why identical chemistry in a body?
  \item Why is the proportion of carbon-14 constant in a living body, and
        why does it start dropping at death?
  \item Write the decay equation of carbon-14. Would its radiation escape
        through the mummy's linen wrappings?
  \item A gram of carbon from the linen shows $6.8$ decays per minute.
        What fraction of the living rate is that? Date the mummy.
  \item A second ``mummy'', the prize of a private collection, shows
        $13.2$ decays per minute per gram. Verdict?
  \item Estimate the rate from a sample ten half-lives old, and explain
        why carbon dating fades out beyond roughly \num{57000} years.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The reactor under the meadow.}
\begin{enumerate}[resume]
  \item Give the composition of ${}^{238}_{92}\mathrm{U}$. Why do heavy
        nuclei need such a neutron surplus?
  \item Write its $\alpha$ decay equation.
  \item The daughter then decays $\beta^-$, and the granddaughter
        $\beta^-$ again. Write both equations. Which element reappears?
  \item The chain ends at ${}^{206}_{82}\mathrm{Pb}$. Count its $\alpha$
        steps, then its $\beta^-$ steps.
  \item The Earth formed \num{4.5e9} years ago. What fraction of its
        primordial uranium-238 remains today? And in $5$ billion more years?
  \item In one sentence: what does this buried decay heat power at the
        surface? (Think volcanoes and drifting continents.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The smoke detector.}
\begin{enumerate}[resume]
  \item Write the $\alpha$ decay equation of ${}^{241}_{95}\mathrm{Am}$.
  \item Its activity is \qty{33}{kBq}: how many decays per second? Per day?
  \item Explain the detector: what do the $\alpha$ particles do to the air
        of the chamber, and what does smoke change?
  \item Why is an $\alpha$ emitter the right choice here --- and a
        $\gamma$ emitter of equal activity the worst one?
  \item After one half-life ($432$ years!) the activity would be
        \qty{16.5}{kBq}; the manual asks for replacement after ten years.
        Is the source the weak link?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Why the stars had to be mortal.}
\begin{enumerate}[resume]
  \item Dying stars forged and scattered every element heavier than
        helium. What fraction of uranium-238 forged \num{9e9} years ago
        survives today? Why does Earth's remaining abundance make the
        planet far younger than the oldest stars?
  \item Earth's radioactive inner heat drives volcanism and plate
        tectonics, and keeps the core churning out the magnetic shield of
        \cref{ch:g11:magnetic-fields}. What would a planet of only stable
        atoms be like?
  \item One line, as poetry and as bookkeeping: where were your atoms
        forged, and what still warms the ground under your feet?
\end{enumerate}
\end{problem}
