\chapter{Work of a Force}\label{ch:g11:work-of-force}

Push a wardrobe across the room and you have worked; push against a
wall all afternoon and, whatever your muscles report, physics counts
zero. \omterm{def:g11:work-of-force:work}{Work} is what a \omterm{def:g10:inertia:force}{force} delivers \emph{along a motion} --- \omterm{def:g10:energy-conservation:energy}{energy}
in transit (\cref{ch:g10:energy-conservation}), paid in \omterm{def:g11:work-of-force:work}{joules}. This
chapter prices the \omterm{def:g10:inertia:force}{forces} of \cref{ch:g11:forces-and-motion} --- pulls
at an angle, \omterm{def:g10:universal-gravitation:weight}{weight}, \omterm{def:g10:inertia:inventory}{friction} --- and clocks the payment rate: \omterm{def:g11:work-of-force:power}{power}.

\section{Work of a constant force}

\begin{definition}[Work of a constant force]\label{def:g11:work-of-force:work}
A constant \omterm{def:g10:inertia:force}{force} $\vect F$ applied to an object moving along a straight
segment from $A$ to $B$ does the \emph{work}\index{work (of a force)}
\[
W_{AB}(\vect F) = F \times AB \times \cos\theta ,
\]
where $F$ is the magnitude of the \omterm{def:g10:inertia:force}{force} (in \unit{N}), $AB$ the length
of the displacement (in \unit{m}) and $\theta$ the angle between
$\vect F$ and $\vect{AB}$. Work is measured in
\emph{joules}\index{joule} (\unit{J}): one joule is the work of a
\omterm{def:g10:inertia:force}{one-newton} \omterm{def:g10:inertia:force}{force} along one \omterm{def:g10:orders-of-magnitude:unit}{metre} of its own direction.
\end{definition}

\begin{remark}\label{rem:g11:work-of-force:scal}
$F \times AB \times \cos\theta$ is exactly the scalar product
$\vect F \cdot \vect{AB}$ of this year's mathematics volume. Only the
component $F\cos\theta$ \emph{along} the motion earns anything.
\end{remark}

\begin{example}[Suitcase on a leash]\label{ex:g11:work-of-force:suitcase}
A traveller drags a suitcase \qty{300}{m} across a terminal, the strap
pulled with $F = \qty{40}{N}$ at $\theta = \qty{50}{\degree}$ to the
floor: $W = 40 \times 300 \times \cos\qty{50}{\degree} \approx
\qty{7.7}{kJ}$. Pulled flat ($\theta = 0$) the same \omterm{def:g10:inertia:force}{force} would
deliver \qty{12}{kJ}: a third of the pull lifts nothing.
\end{example}

\begin{definition}[Motor, resistive, zero work]\label{def:g11:work-of-force:sign}
The sign of $\cos\theta$ sorts every \omterm{def:g10:inertia:force}{force} into three regimes:
\begin{itemize}
  \item $\theta < \qty{90}{\degree}$: $W > 0$, the \omterm{def:g11:work-of-force:work}{work} is
        \emph{motor}\index{motor work} --- the \omterm{def:g10:inertia:force}{force} feeds \omterm{def:g10:energy-conservation:energy}{energy} into
        the motion;
  \item $\theta = \qty{90}{\degree}$: $W = 0$, the \omterm{def:g10:inertia:force}{force} \omterm{def:g11:work-of-force:work}{works} not at
        all, however large;
  \item $\theta > \qty{90}{\degree}$: $W < 0$, the \omterm{def:g11:work-of-force:work}{work} is
        \emph{resistive}\index{resistive work} --- the \omterm{def:g10:inertia:force}{force} drains
        \omterm{def:g10:energy-conservation:energy}{energy}. The pure case is a brake at $\qty{180}{\degree}$:
        \qty{800}{N} over \qty{45}{m} takes $800 \times 45 \times (-1)
        = \qty{-36}{kJ}$ out of a car, into hot discs.
\end{itemize}
\end{definition}

\section{The work of the weight}

\begin{proposition}[Work of the weight]\label{prop:g11:work-of-force:weight}
When an object of mass $m$ moves from $A$ (altitude $z_A$) to $B$
(altitude $z_B$) along \emph{any} path made of straight segments, its
\omterm{def:g10:universal-gravitation:weight}{weight} $\vect P$ does the \omterm{def:g11:work-of-force:work}{work}
\[
W_{AB}(\vect P) = mg\,(z_A - z_B) = \pm\, mgh ,
\]
where $h = \abs{z_A - z_B}$ is the height drop: $+mgh$ going down,
$-mgh$ going up --- the drop alone matters, not the route.
\end{proposition}

\begin{proof}
On one straight segment $AB$, the \omterm{def:g10:universal-gravitation:weight}{weight} (magnitude $mg$, straight
down) makes an angle $\theta$ with $\vect{AB}$, and $AB\cos\theta$ is
precisely the projection of the displacement on the downward vertical:
$z_A - z_B$. Hence $W = mg \times AB\cos\theta = mg\,(z_A - z_B)$.
Along a chain of segments the \omterm{def:g11:work-of-force:work}{works} add and the drops telescope:
$mg(z_A - z_C) + mg(z_C - z_B) = mg(z_A - z_B)$.
\end{proof}

\begin{proposition}[Any path]\label{prop:g11:work-of-force:anypath}
The same formula holds along any curved path from $A$ to $B$.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:work-of-force:anypath}
Plausible --- a curve is as close as we like to a broken line of many
short segments --- but making ``as close as we like'' honest is
calculus, done in \cref{ch:g12:work-and-energy}.
\end{remark}

\begin{example}[Hiker]\label{ex:g11:work-of-force:hiker}
A \qty{78}{kg} hiker (pack included) climbs \qty{450}{m}, by whichever
trail: $W(\vect P) = -78 \times 9.81 \times 450 \approx \qty{-344}{kJ}$,
and exactly $+\qty{344}{kJ}$ back down, whatever the detours.
\end{example}

\section{The work of friction}

\begin{proposition}[Work of sliding friction]\label{prop:g11:work-of-force:friction}
A \omterm{def:g10:inertia:inventory}{friction} \omterm{def:g10:inertia:force}{force} of constant magnitude $f$, at every instant opposite
the motion, does the \omterm{def:g11:work-of-force:work}{work} $W = -fL$ along a path of total length $L$:
always \omterm{def:g11:work-of-force:sign}{resistive}, and proportional to the \emph{length of the path
travelled} --- not to the displacement.
\end{proposition}

\begin{proof}
Each short straight piece has the \omterm{def:g10:inertia:force}{force} at $\qty{180}{\degree}$ to the
motion, contributing $-f \times (\text{its length})$; the pieces add
up to $-fL$. A curved path is chopped into short pieces likewise, the
limit admitted as for the \omterm{def:g10:universal-gravitation:weight}{weight} (\cref{rem:g11:work-of-force:anypath}).
\end{proof}

\begin{remark}[Friction against weight: the great divide]\label{rem:g11:work-of-force:divide}
The \omterm{def:g10:universal-gravitation:weight}{weight}'s \omterm{def:g11:work-of-force:work}{work} forgets the path; \omterm{def:g10:inertia:inventory}{friction}'s is a toll per \omterm{def:g10:orders-of-magnitude:unit}{metre}. On
a round trip the \omterm{def:g10:universal-gravitation:weight}{weight} nets zero --- what it takes uphill it refunds
downhill --- while \omterm{def:g10:inertia:inventory}{friction} charges $-fL$ on every leg and refunds
nothing: the \omterm{def:g10:universal-gravitation:weight}{weight}'s \omterm{def:g11:work-of-force:work}{work} can be stored and recovered (potential
\omterm{def:g10:energy-conservation:energy}{energy}, \cref{ch:g11:mechanical-energy}), \omterm{def:g10:inertia:inventory}{friction}'s is lost as heat.
\end{remark}

\begin{example}[There and back]\label{ex:g11:work-of-force:roundtrip}
A crate slides \qty{5.0}{m} across a floor and \qty{5.0}{m} back,
against $f = \qty{60}{N}$: \omterm{def:g10:inertia:inventory}{friction} does $-60 \times 10 =
\qty{-600}{J}$ even though the crate ends where it started. \omterm{def:g10:universal-gravitation:weight}{Weight} and
normal \omterm{def:g10:inertia:force}{force}, perpendicular to the motion: zero throughout.
\end{example}

\begin{remark}[What about a spring?]\label{rem:g11:work-of-force:spring}
A spring pulls harder the further it is stretched: its \omterm{def:g10:inertia:force}{force} is not
constant and \cref{def:g11:work-of-force:work} does not apply.
Its \omterm{def:g11:work-of-force:work}{work} means cutting the stretch into infinitesimal steps and
summing --- calculus, delivered in \cref{ch:g12:work-and-energy}.
\end{remark}

\section{Power}

\begin{definition}[Power]\label{def:g11:work-of-force:power}
The \emph{power}\index{power!mechanical} of a \omterm{def:g10:inertia:force}{force} is the rate at
which it \omterm{def:g11:work-of-force:work}{works}, $P = W/\Delta t$, in \emph{watts}\index{watt}
($\unit{W} = \unit{J/s}$) --- the same watt as in
\cref{ch:g11:circuits-and-power}. A pre-SI \omterm{def:g10:orders-of-magnitude:unit}{unit} survives on engine
badges: the \emph{horsepower}\index{horsepower},
$\qty{1}{hp} = \qty{736}{W}$.
\end{definition}

\begin{proposition}[Power at constant velocity]\label{prop:g11:work-of-force:pfv}
If the object moves at constant speed $v$, a constant \omterm{def:g10:inertia:force}{force} $\vect F$
at angle $\theta$ to the motion delivers the \omterm{def:g11:work-of-force:power}{power}
$P = F v \cos\theta$ --- in particular $P = Fv$ for a \omterm{def:g10:inertia:force}{force} along the
motion.
\end{proposition}

\begin{proof}
In a time $\Delta t$ the object covers $d = v\,\Delta t$, so
$W = F \times v\,\Delta t \times \cos\theta$; divide by $\Delta t$.
\end{proof}

\begin{example}[Crane]\label{ex:g11:work-of-force:crane}
A crane hoists a \qty{600}{kg} pallet at a steady \qty{0.90}{m/s}: the
cable's \omterm{def:g10:inertia:force}{force} equals the \omterm{def:g10:universal-gravitation:weight}{weight}, $mg = \qty{5.89e3}{N}$, so
$P = Fv = 5890 \times 0.90 \approx \qty{5.3}{kW} \approx \qty{7.2}{hp}$
--- a seven-horse team in a steel box.
\end{example}

\begin{method}[Climbing with less force: the zigzag]\label{met:g11:work-of-force:zigzag}
To raise a load a height $h$ with a limited \omterm{def:g10:inertia:force}{force}:
\begin{enumerate}
  \item The bill is fixed: by \cref{prop:g11:work-of-force:weight},
        any route to the top costs the \omterm{def:g11:work-of-force:work}{work} $mgh$.
  \item Stretch the path: on a slope of length $L$ the \omterm{def:g10:inertia:force}{force} needed
        (constant speed, \omterm{def:g10:inertia:inventory}{friction} aside) is $F = mgh/L$ --- double the
        length, halve the \omterm{def:g10:inertia:force}{force}.
  \item Pay the surcharge: \omterm{def:g10:inertia:inventory}{friction} charges by the \omterm{def:g10:orders-of-magnitude:unit}{metre}
        (\cref{prop:g11:work-of-force:friction}), adding $fL$.
\end{enumerate}
Mountain roads zigzag for step 2: same height, less \omterm{def:g10:inertia:force}{force}, same \omterm{def:g11:work-of-force:work}{work}.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:1}
A constant \omterm{def:g10:inertia:force}{force} $F = \qty{40}{N}$ acts over a straight \qty{50}{m}
displacement: compute its \omterm{def:g11:work-of-force:work}{work} for $\theta = \qty{0}{\degree}$,
$\qty{60}{\degree}$, $\qty{90}{\degree}$ and $\qty{120}{\degree}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:2}
A \qty{12}{kg} toolbox is lowered \qty{2.5}{m} from a van, then later
raised back. \omterm{def:g11:work-of-force:work}{Work} of its \omterm{def:g10:universal-gravitation:weight}{weight} in each case?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:3}
\omterm{def:g11:work-of-force:sign}{Motor}, \omterm{def:g11:work-of-force:sign}{resistive} or zero? (a) the \omterm{def:g10:universal-gravitation:weight}{weight} of a falling apple; (b) the
\omterm{def:g10:universal-gravitation:weight}{weight} of a rising ball; (c) the normal \omterm{def:g10:inertia:force}{force} on a sliding box; (d)
\omterm{def:g10:inertia:inventory}{friction} on a braking bicycle; (e) the string's \omterm{def:g10:inertia:inventory}{tension} on a swinging
pendulum bob.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:4}
A crane lifts \qty{600}{kg} by \qty{25}{m} at constant speed. \omterm{def:g11:work-of-force:work}{Work} of
the lifting \omterm{def:g10:inertia:force}{force} and of the \omterm{def:g10:universal-gravitation:weight}{weight}? What is their sum, and why?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:5}
An escalator \omterm{def:g11:work-of-force:sign}{motor} does \qty{90}{kJ} of \omterm{def:g11:work-of-force:work}{work} every minute. Its \omterm{def:g11:work-of-force:power}{power}
in \omterm{def:g11:work-of-force:power}{watts}? In \omterm{def:g11:work-of-force:power}{horsepower}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:6}
A sled is towed \qty{200}{m} by a rope of \omterm{def:g10:inertia:inventory}{tension} \qty{120}{N} at
$\qty{30}{\degree}$ above the snow; \omterm{def:g10:inertia:inventory}{friction} is \qty{45}{N}. \omterm{def:g11:work-of-force:work}{Work} of

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:7}
A cyclist rides at a constant \qty{25}{km/h} against a total drag of
\qty{18}{N}. \omterm{def:g10:energy-conservation:power}{Power} delivered? \omterm{def:g11:work-of-force:work}{Work} done in one hour?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:8}
A \qty{65}{kg} skier descends \qty{120}{m} of height, by a \qty{800}{m}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:9}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:10}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:11}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:work-of-force:12}

track of $20\%$ grade or a zigzag road of $6\%$ grade (grade $=

Conclude in one sentence.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:work-of-force:13}
A winch raises \qty{250}{kg} of tiles \qty{15}{m} up scaffolding at a

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:work-of-force:14}
A wardrobe is slid against $f = \qty{140}{N}$ of \omterm{def:g10:inertia:inventory}{friction} from one
corner of a room to the opposite one: straight along the \qty{6.0}{m}
diagonal, or \qty{3.6}{m} then \qty{4.8}{m} along the walls. \omterm{def:g10:inertia:inventory}{Friction}
\omterm{def:g11:work-of-force:work}{work} on each route, then on a diagonal round trip? What does the
\omterm{def:g10:universal-gravitation:weight}{weight} do on that round trip, and what deep difference between the
two \omterm{def:g10:inertia:force}{forces} does this expose?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:work-of-force:15}
A cyclist (\qty{80}{kg} with bike) climbs a pass: \qty{600}{m} of
height over \qty{12}{km} of road, at a constant \qty{15}{km/h},
against \qty{14}{N} of drag and rolling \omterm{def:g11:circuits-and-power:resistance}{resistance}. Compute the \omterm{def:g11:work-of-force:work}{work}
supplied against gravity, against \omterm{def:g10:inertia:inventory}{friction}, and in total; the
duration of the climb; the average \omterm{def:g11:work-of-force:power}{power}, in \omterm{def:g11:work-of-force:power}{watts} and \omterm{def:g11:work-of-force:power}{horsepower}.
\end{exercise}

\section{Problem: The mover's day}

\begin{problem}[{Weekend problem --- physics pays by the joule: a
day of sliding, hoisting and hairpin bends, ending with the humbling
discovery of what a kettle thinks of honest labour}]
\label{pb:g11:work-of-force:1}
A mover's crew empties a warehouse, hauls the load up to a third-floor
flat, then drives the van to a hilltop village. Every task is priced
in \omterm{def:g11:work-of-force:work}{joules}; the kitchen kettle will audit the day. Boxes: \qty{30}{kg}.

\medskip
\noindent\textbf{Part I --- Across the warehouse floor.}
A box is pushed with a horizontal \omterm{def:g10:inertia:force}{force} $F = \qty{110}{N}$ over
$d = \qty{8.0}{m}$; sliding \omterm{def:g10:inertia:inventory}{friction} is $f = \qty{95}{N}$.
\begin{enumerate}
  \item Compute the \omterm{def:g11:work-of-force:work}{work} of the push.
  \item Compute the \omterm{def:g11:work-of-force:work}{work} of \omterm{def:g10:inertia:inventory}{friction}, of the \omterm{def:g10:universal-gravitation:weight}{weight}, and of the normal
        \omterm{def:g10:inertia:force}{force} (justify the zeros).
  \item Total \omterm{def:g11:work-of-force:work}{work} on the box, and its sign's meaning (the box started
        at rest)? What push keeps the speed constant
        (\cref{ch:g11:forces-and-motion})?
  \item The far corner can be reached along the \qty{10.0}{m} diagonal
        or along two walls (\qty{6.0}{m} then \qty{8.0}{m}). \omterm{def:g10:inertia:inventory}{Friction}
        \omterm{def:g11:work-of-force:work}{work} on each route?
  \item On a diagonal round trip (out and back), compute the \omterm{def:g11:work-of-force:work}{work} of
        \omterm{def:g10:inertia:inventory}{friction} and the \omterm{def:g11:work-of-force:work}{work} of the \omterm{def:g10:universal-gravitation:weight}{weight}. Which \omterm{def:g10:inertia:force}{force} refunds?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Up to the third floor ($h = \qty{9.0}{m}$).}
\begin{enumerate}[resume]
  \item Compute the \omterm{def:g11:work-of-force:work}{work} of the \omterm{def:g10:universal-gravitation:weight}{weight} on a box raised to the flat.
        What \omterm{def:g11:work-of-force:work}{work} must the crew supply at least?
  \item Carried up the stairs by a \qty{70}{kg} mover, gravity charges
        for mover \emph{and} box: total \omterm{def:g11:work-of-force:work}{work} against gravity, and the
        fraction actually spent on the box?
  \item A trolley ramp is \qty{30}{m} long for the same \qty{9.0}{m}
        rise, with \qty{40}{N} of rolling \omterm{def:g10:inertia:inventory}{friction}. At constant speed,
        compute the push needed along the ramp and its \omterm{def:g11:work-of-force:work}{work}.
  \item Compare the ramp with a straight vertical hoist: \omterm{def:g10:inertia:force}{force} needed,
        \omterm{def:g11:work-of-force:work}{work} done. What does the ramp buy, and at what price?
  \item A rope-and-pulley hoist raises the box at \qty{0.50}{m/s}.
        Duration and mechanical \omterm{def:g11:work-of-force:power}{power}?
  \item Recover that \omterm{def:g11:work-of-force:power}{power} as $P = Fv$. The hoist's \omterm{def:g11:work-of-force:sign}{motor} is $60\%$
        efficient: \omterm{prop:g11:circuits-and-power:power}{electrical power} drawn?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The zigzag road.}
The loaded van (\qty{3500}{kg}) must reach a village \qty{240}{m}
above the valley: straight lane, \qty{1.0}{km} at $24\%$ grade, or
paved road, \qty{8.0}{km} at $3.0\%$ grade (grade $= \sin\alpha$).
\begin{enumerate}[resume]
  \item Check that both routes climb the same \qty{240}{m}.
  \item At constant speed and neglecting \omterm{def:g10:inertia:inventory}{friction}, compute the driving
        \omterm{def:g10:inertia:force}{force} needed along each route.
  \item \omterm{def:g11:work-of-force:work}{Work} of the driving \omterm{def:g10:inertia:force}{force} on each route? Compare with $mgh$.
  \item Rolling \omterm{def:g11:circuits-and-power:resistance}{resistance} is \qty{400}{N} on both: compute the extra
        \omterm{def:g11:work-of-force:work}{work} on each route. What does the zigzag's comfort cost?
  \item At \qty{36}{km/h}, compute the engine \omterm{def:g11:work-of-force:power}{power} needed on each
        route (in \unit{kW} and \unit{hp}). Why do roads zigzag?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The day's ledger.}
\begin{enumerate}[resume]
  \item The crew slides $60$ boxes across the floor (the push of
        question 1) and hoists all $60$ up the \qty{9.0}{m}
        (question 6). Total mechanical \omterm{def:g11:work-of-force:work}{work} delivered to the boxes?
  \item Add the stair carrier's own body: $20$ trips up at \qty{70}{kg}
        (downhill refunds go to hot knees, not to the ledger). Day's
        total?
  \item The crew worked $8.0$ hours: average mechanical \omterm{def:g11:work-of-force:power}{power}? How long
        would the \qty{2.0}{kW} kettle take to expend the same \omterm{def:g10:energy-conservation:energy}{energy}?
  \item Finale: in two sentences, say what the \omterm{def:g11:work-of-force:work}{joule} pays for --- and
        does not (the wall pushed all afternoon, the box held
        motionless) --- and why movers own winches, ramps and zigzag
        roads rather than bigger muscles.
\end{enumerate}
\end{problem}
