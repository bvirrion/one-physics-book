\chapter{चुंबकत्व और चुंबकीय क्षेत्र}\label{ch:g11:magnetic-fields}

हज़ार साल पहले से नाविक चुम्बकित सुइयों से चलते थे
किसी को पता था कि वे उत्तर की ओर क्यों इशारा करते हैं। इसका उत्तर 1820 में आया, जब ए
करंट ने कंपास को हिलाया: चुंबकत्व गतिमान आवेशों से बनता है।
यह अध्याय मानचित्रित करता है \omterm{def:g11:magnetic-fields:bfield}{चुंबकीय क्षेत्र} --- का मैग्नेट, पृथ्वी का, का
धाराएँ --- फिर लूप बंद कर देता है: \omterm{def:g11:electric-gravitational-fields:field}{मैदान} धाराओं को पीछे धकेलता है,
प्रत्येक \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} मोटर इसी प्रकार घूमती है।

\section{चुम्बक और चुंबकीय क्षेत्र}

\begin{definition}[चुंबक और ध्रुव]\label{def:g11:magnetic-fields:magnet}
ए \emph{चुंबक}\index{चुंबक} लोहे को आकर्षित करता है और धागे से लटकाया जाता है,
उत्तर की ओर मुख कर लेता है. इसकी क्रिया दो पर केन्द्रित होती है \emph{चुंबकीय डंडे}\index{चुंबकीय डंडे}, \emph{उत्तर} (अंत वह चाहता है
भौगोलिक उत्तर) और \emph{दक्षिण}: जैसे ध्रुव प्रतिकर्षित करते हैं, वैसे ध्रुव आकर्षित करते हैं।
एक चुम्बक को आधे में काटने से दो पूर्ण चुम्बक प्राप्त होते हैं --- कोई प्रयोग नहीं
कभी किसी ध्रुव को अलग किया है.
\end{definition}

\begin{definition}[चुंबकीय क्षेत्र]\label{def:g11:magnetic-fields:bfield}
ए \omterm{def:g11:magnetic-fields:magnet}{चुंबक} इसके चारों ओर की जगह को संशोधित करता है
(\cref{ch:g11:electric-gravitational-fields}): यह हर जगह बनाता है a
\emph{चुंबकीय क्षेत्र}\index{चुंबकीय क्षेत्र} $\vect B $. एक कम्पास पढ़ता है
इसकी दिशा (सुई संरेखित होती है $\vect B $, उत्तरी छोर आगे); ए
\emph{हॉल जांच}\index{हॉल जांच} इसके परिमाण को मापता है, में
\emph{टेस्ला}\index{टेस्ला} (\unit{T}).
\end{definition}

\begin{definition}[चुंबकीय क्षेत्र रेखाएँ]\label{def:g11:magnetic-fields:lines}
\emph{चुंबकीय क्षेत्र रेखाएँ}\index{चुंबकीय क्षेत्र रेखाएँ} वक्र हैं
स्पर्शरेखा $\vect B $, जहाँ भीड़ है \omterm{def:g11:electric-gravitational-fields:field}{मैदान} मजबूत है. बाहर ए
\omterm{def:g11:magnetic-fields:magnet}{चुंबक} वे उत्तरी ध्रुव से दक्षिण की ओर दौड़ते हैं; भिन्न \omterm{def:g11:electric-gravitational-fields:efield}{विद्युत क्षेत्र} पंक्तियाँ
उनका कोई अंत नहीं है: प्रत्येक के माध्यम से बंद हो जाता है चुंबकका शरीर.
\end{definition}

\section{पृथ्वी एक चुंबक है}

\begin{definition}[पृथ्वी का चुंबकीय क्षेत्र]\label{def:g11:magnetic-fields:earth}
पृथ्वी के पिघले हुए लौह कोर में धाराएँ ग्रह को विशाल बनाती हैं
\omterm{def:g11:magnetic-fields:magnet}{चुंबक}: सतह पर $ B \approx \qty{50}{\micro T}$, और एक मुफ़्त
सुई के साथ संरेखित होती है मैदानका क्षैतिज घटक --- यह है
दिशा सूचक यंत्र। भू-चुंबकीय ध्रुव भौगोलिक नहीं हैं; कोण
कम्पास उत्तर और वास्तविक उत्तर के बीच है
\emph{झुकाव}\index{झुकाव}, हर बियरिंग पर सही किया गया।
\end{definition}

\begin{example}[एक बियरिंग को ठीक करना]\label{ex:g11:magnetic-fields:bearing}
जहां \omterm{def:g11:magnetic-fields:earth}{झुकाव} है $\ang{7}$ पश्चिम, एक जहाज संचालन ``कम्पास
उत्तर'' के लिए \qty{5.0}{km} बहाव $5000 \times \sin\ang{7} \approx
\qty{6.1e2}{m}$ इसके ट्रैक के पश्चिम में: सही उत्तर की ओर जाना, चलाना $\ang{7}$
\emph{पूर्व} सुई का.
\end{example}

\section{धारा का चुंबकीय क्षेत्र}

\begin{remark}[ओर्स्टेड, 1820]\label{rem:g11:magnetic-fields:oersted}
1820 में एक व्याख्यान के दौरान, हंस क्रिश्चियन ओर्स्टेड ने एक कम्पास सुई देखी
झूले जब उसने पास के एक सर्किट को बंद कर दिया: बिजली और चुंबकत्व थे
एक विषय --- \emph{धाराएँ निर्मित करती हैं \omterm{def:g11:magnetic-fields:bfield}{चुंबकीय क्षेत्र}}.
\end{remark}

\begin{proposition}[एक सीधे तार का क्षेत्र]\label{prop:g11:magnetic-fields:wire}
करंट प्रवाहित करने वाला एक लंबा सीधा तार $ I $ बनाता है \omterm{def:g11:electric-gravitational-fields:field}{मैदान} किसका
रेखाएँ तार पर केन्द्रित वृत्त हैं, जो उसके लंबवत तल में हैं।
दूरी पर $ d $ तार से,
\[
B = \frac{\mu_0\, I}{2\pi d},
\qquad \mu_0 = 4\pi\times 10^{-7}\ \unit{T.m/A} \approx \qty{1.26e-6}{T.m/A}.
\]
\end{proposition}
\admitted

($\odot $: वर्तमान पृष्ठ से बाहर; $\otimes $: यह में)। तार पकड़ो
दाहिने हाथ से, अंगूठा धारा की दिशा में: उंगलियाँ रास्ता मोड़ती हैं
$\vect B $ बदल जाता है.}
\end{omfigure}

\begin{example}[तार एक कमजोर चुम्बक है]\label{ex:g11:magnetic-fields:wire}
पर $ d = \qty{1.0}{cm}$ एक तार ले जाने से $ I = \qty{15}{A}$:
$ B = \num{2e-7} \times 15 / 0.010 = \qty{0.30}{mT}$ --- छह पृथ्वी
खेत, एक ऐसे करंट से जो घर को तोड़ देगा। मज़बूत
खेत एक बेहतर ज्यामिति की आवश्यकता है.
\end{example}

\begin{definition}[कुंडल और सोलनॉइड]\label{def:g11:magnetic-fields:solenoid}
तार को घुमाने से उसका संकेन्द्रण हो जाता है \omterm{def:g11:electric-gravitational-fields:field}{मैदान}. ए \emph{समतल कुंडल}\index{समतल कुंडल} ($ N $ एक डिस्क में बदल जाता है) की तरह व्यवहार करता है \omterm{def:g11:lenses-and-eye:lens}{पतला} \omterm{def:g11:magnetic-fields:magnet}{चुंबक},
एक का मुख उत्तर की ओर, दूसरे का दक्षिण की ओर; ए \emph{solenoid}\index{solenoid} ---
एक लंबी बेलनाकार घुमावदार, $ n $ प्रति बदल जाता है \omterm{def:g10:orders-of-magnitude:unit}{मीटर} --- एक बार की तरह \omterm{def:g11:magnetic-fields:magnet}{चुंबक}.
\end{definition}

\begin{proposition}[सोलनॉइड के अंदर फ़ील्ड]\label{prop:g11:magnetic-fields:solenoidfield}
एक लंबे अंदर \omterm{def:g11:magnetic-fields:solenoid}{solenoid}, छोर से दूर, \omterm{def:g11:electric-gravitational-fields:field}{मैदान} है
\emph{\omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी}}: अक्ष के समानांतर और प्रत्येक आंतरिक भाग पर समान
बिंदु, परिमाण का $ B = \mu_0\, n\, I $, बोर से स्वतंत्र
चौड़ाई. इसके बाहर \omterm{def:g11:electric-gravitational-fields:field}{मैदान} कमज़ोर है.
\end{proposition}
\admitted

\begin{remark}
दोनों सूत्र ईमानदारी से वर्ष 1 खंड में कानून से प्राप्त किए गए हैं
संबंधित ए मैदानइसका परिसंचरण उस धारा तक होता है जिसे यह घेरता है। ध्यान दें क्या
मायने रखता है: $ n $, बदल जाता है \emph{प्रति \omterm{def:g10:orders-of-magnitude:unit}{मीटर}} --- हवा तेज़, अधिक लंबी नहीं।
\end{remark}

\begin{method}[दाहिना हाथ, दो बार]\label{met:g11:magnetic-fields:righthand}
\begin{itemize}
  \item \emph{तार}: अंगूठा = करंट; उँगलियाँ टेढ़ी होना = रेखाओं का मुड़ना।
  \item \emph{कुंडल, \omterm{def:g11:magnetic-fields:solenoid}{solenoid}}: मुड़ी हुई उंगलियाँ = घुमावों में करंट;
        अंगूठा= $\vect B $ अंदर, उत्तर मुख की ओर इशारा करते हुए।
\end{itemize}
\end{method}

$ I = \qty{5.0}{A}$: $ B = \num{1.26e-6} \times 1000 \times 5.0 \approx
\qty{6.3}{mT}$ --- एक सौ पृथ्वी खेत, एक घुंडी द्वारा समायोज्य।
\end{example}

\section{विद्युत चुम्बकों}

\begin{definition}[विद्युत]\label{def:g11:magnetic-fields:electromagnet}
एक \emph{विद्युत}\index{विद्युत} एक है \omterm{def:g11:magnetic-fields:solenoid}{solenoid} एक पर घाव
नरम लौह कोर: लोहा चुम्बकित होता है कुंडल'एस \omterm{def:g11:electric-gravitational-fields:field}{मैदान} और गुणा करता है
यह, आम तौर पर द्वारा $100$ को $1000$. स्थायी के विपरीत \omterm{def:g11:magnetic-fields:magnet}{चुंबक}, यह
करंट के साथ स्विच ऑफ हो जाता है।
\end{definition}

\begin{remark}[तीन मशीनें]\label{rem:g11:magnetic-fields:machines}
स्क्रैपयार्ड क्रेन एक कार को उठाती है \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} और ---
संपूर्ण बिंदु --- \emph{चला जाता है} यह एक स्विच खोलकर. एक रिले छोटा है
\omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} किसी संपर्क को खींचकर बंद कर देता है: एक कमजोर धारा स्विच को बंद कर देती है
मजबूत एक (\cref{ch:g11:circuits-and-power}). एमआरआई स्कैनर एक \omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी} धारण करता है
\qty{1.5}{T} एक सुपरकंडक्टिंग के साथ रोगी के चारों ओर \omterm{def:g11:magnetic-fields:solenoid}{solenoid}.
\end{remark}

\section{लाप्लास बल}

\begin{proposition}[लाप्लास बल]\label{prop:g11:magnetic-fields:laplace}
लम्बाई का एक सीधा तार $ L $, करंट ले जाना $ I $ में एक \omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी}
फ़ील्ड $\vect B $ \emph{सीधा} तार को, लगता है
\emph{लाप्लास बल}\index{लाप्लास बल} परिमाण का $ F = I\,L\,B $,
तार और दोनों के लंबवत $\vect B $: दाहिना हाथ सपाट, अंगूठा साथ में
वर्तमान, सीधी उँगलियाँ साथ में $\vect B $ ---हथेली साथ में धकेलती है
$\vect F $. तार \emph{समानांतर} तक \omterm{def:g11:electric-gravitational-fields:field}{मैदान} नहीं लगता बल.
\end{proposition}
\admitted

\begin{remark}[जंपिंग रेल]\label{rem:g11:magnetic-fields:rail}
वर्ष 1 खंड इसे \omterm{def:g11:magnetic-fields:magnet}{चुंबक}ीय से प्राप्त करता है बल प्रत्येक गति पर
आरोप. प्रयोगशाला में: एक तांबे की \omterm{def:g11:color-light-sources:cones}{छड़} दो क्षैतिज रेखाओं पर टिकी होती है
ए के ध्रुवों के बीच रेल \omterm{def:g11:magnetic-fields:magnet}{चुंबक} (\omterm{def:g11:electric-gravitational-fields:field}{मैदान} पेज में
चित्रा); स्विच बंद करें, और रॉड रेल के साथ-साथ शूट हो जाएगी। उलटा
वर्तमान, या पलटें \omterm{def:g11:magnetic-fields:magnet}{चुंबक}: यह दूसरे तरीके से शूट करता है।
\end{remark}

\begin{example}[लाप्लास संख्याएँ]\label{ex:g11:magnetic-fields:laplace}
लम्बाई की एक \omterm{def:g11:color-light-sources:cones}{छड़} $ L = \qty{5.0}{cm}$ भार उठाते $ I = \qty{8.0}{A}$ आर-पार
$ B = \qty{0.50}{T}$: $ F = 8.0 \times 0.050 \times 0.50 = \qty{0.20}{N}$
--- द \omterm{def:g10:universal-gravitation:weight}{वज़न} का \qty{20}{g}, कुछ ग्राम वजन वाले तार पर।
\end{example}

\begin{remark}[डीसी मोटर]\label{rem:g11:magnetic-fields:motor}
तार को एक लूप में मोड़ें \omterm{def:g11:electric-gravitational-fields:field}{मैदान}: दोनों भुजाएँ लंबवत्
$\vect B $ विपरीत धाराएँ लेकर चलते हैं, इसलिए उनका \omterm{prop:g11:magnetic-fields:laplace}{लाप्लास बल} विपरीत हैं
--- एक जोड़ी जो लूप को घुमाती है। आधे चक्कर के बाद जोड़ा घूमेगा
यह वापस, तो एक घूर्णन स्विच (द \emph{कम्यूटेटर}\index{कम्यूटेटर}) को उलट देता है
हर आधे मोड़ पर करंट। हर पंखा और ड्रिल यह लूप है, गुणा।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g11:magnetic-fields:1}

टुकड़ा ले जाना? एक साथ वापस लाए गए, ताज़ा कटे हुए चेहरे आकर्षित करते हैं

उत्तरी ध्रुव से दक्षिणी ध्रुव तक; (सी) प्रत्येक \omterm{def:g11:magnetic-fields:bfield}{चुंबकीय क्षेत्र} रेखा एक है
बंद लूप; (डी) एक सुई इसके माध्यम से रेखा के लंबवत स्थित होती है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g11:magnetic-fields:9}

करंट बनाता है \omterm{prop:g11:magnetic-fields:laplace}{लाप्लास बल} तार को संतुलित करें \omterm{def:g10:universal-gravitation:weight}{वज़न}, ताकि यह

की दिशा बताएं \omterm{prop:g11:magnetic-fields:laplace}{लाप्लास बल} (या कहें कि यह गायब हो जाता है):

क्या हर आधे मोड़ पर धारा उलट दी जाएगी?

\section{समस्या: कंपास, क्रेन और एमआरआई}

\begin{problem}[{सप्ताहांत समस्या--- तीन चुम्बक काम कर रहे हैं: एक दिशा सूचक यंत्र
एक महासागर को पार करना, एक कार को उठाने वाला एक विद्युत चुंबक, और एक एमआरआई सोलनॉइड
--- एक क्षेत्र, दस की दस शक्तियों तक फैला हुआ, तीन कार्य कर रहा है}]
\label{pb:g11:magnetic-fields:1}
वही वेक्टर $\vect B $ एक नौकायन नाव चलाता है \qty{50}{\micro T},
लोहे के स्क्रैप को उठाता है \qty{1}{T}, और चित्र एक घुटने के साथ \qty{1.5}{T}
--- इस अध्याय के तीन सूत्रों के लिए तीन मशीनें।

\medskip
\noindent\textbf{Part I --- The compass.}
जहाज़ की स्थिति में, पृथ्वी की \omterm{def:g11:electric-gravitational-fields:field}{मैदान} क्षैतिज घटक है
\qty{20}{\micro T} और ऊर्ध्वाधर घटक \qty{44}{\micro T};
 \omterm{def:g11:magnetic-fields:earth}{झुकाव} है $\ang{6}$ पश्चिम.
\begin{enumerate}
  \item कम्पास की सुई आखिर उत्तर की ओर (चुम्बकीय) क्यों होती है?
  \item कुल की गणना करें मैदानका परिमाण और क्षैतिज के साथ उसका कोण।
  \item स्टीयरिंग ``कम्पास नॉर्थ'' के लिए \qty{30}{km}, पश्चिम से कितनी दूर
        क्या जहाज सच्चे-उत्तरी ट्रैक पर समाप्त होता है?
  \item स्टील कार्गो सुई को और अधिक विक्षेपित करता है $\ang{4}$ पश्चिम: वही प्रश्न.
  \item भू-चुम्बकीय ध्रुव के पास दिशा सूचक यंत्र बेकार है: कौन सा घटक
        दोष देना है, और क्यों?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crane.}
कबाड़खाना \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} एक है \omterm{def:g11:magnetic-fields:solenoid}{solenoid} का $600$ घाव को पलट देता है
\qty{30}{cm}, का \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} \qty{1.5}{\ohm}, एक द्वारा खिलाया गया \qty{12}{V}
आपूर्ति; इसका लौह कोर नंगे को कई गुना बढ़ा देता है \omterm{def:g11:electric-gravitational-fields:field}{मैदान} द्वारा $50$.
\begin{enumerate}[resume]
  \item गणना करना $ n $, बारी घनत्व।
  \item में धारा की गणना करें कुंडल (\cref{ch:g11:circuits-and-power}).
  \item की गणना करें नंगे कुंडल \omterm{def:g11:electric-gravitational-fields:field}{मैदान} $\mu_0 n I $.
  \item की गणना करें \omterm{def:g11:electric-gravitational-fields:field}{मैदान} कोर के साथ; नियोडिमियम से तुलना करें \qty{0.5}{T}.
  \item की गणना करें \omterm{def:g10:energy-conservation:power}{शक्ति} में नष्ट हो गया कुंडल, फिर ऊर्जा एक के लिए \qty{10}{\minute} उठाने की पारी.
  \item एक स्थायी \omterm{def:g11:magnetic-fields:magnet}{चुंबक} यह मजबूत मौजूद है. क्यों एक \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} फिर भी?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The winch motor.}
क्रेन की चरखी एक डीसी मोटर है: एक आयताकार लूप $ N = 100$ बदल जाता है,
लंबाई के किनारे \qty{4.0}{cm} के लिए सीधा $ B = \qty{0.25}{T}$,
ले जाना $ I = \qty{3.0}{A}$.
\begin{enumerate}[resume]
  \item की गणना करें \omterm{prop:g11:magnetic-fields:laplace}{लाप्लास बल} प्रत्येक तरफ लंबवत बंडल करें $\vect B $.
  \item दो बल समान और विपरीत हैं, फिर भी उनका प्रभाव नहीं पड़ता
        रद्द करें. वे क्या उत्पादन करते हैं, और उन्हें क्यों होना चाहिए \emph{विलोम}?
  \item क्या बल समानांतर होने पर, अन्य दो पक्षों पर कार्य करता है $\vect B $?
  \item आधे चक्कर के बाद वही स्थिति बल रोटेशन को पूर्ववत कर देगा.
        क्या करता है \omterm{rem:g11:magnetic-fields:motor}{कम्यूटेटर} करो, और कब?
  \item तीन अलग-अलग डिज़ाइन परिवर्तन दें जो प्रत्येक को दोगुना कर दें बल जोड़ा।
\end{enumerate}

\noindent\textbf{Part IV --- The MRI.}
स्कैनर का \omterm{def:g11:magnetic-fields:solenoid}{solenoid} एक \omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी} धारण करता है \qty{1.5}{T} एक करंट के साथ
$ I = \qty{500}{A}$.
\begin{enumerate}[resume]
  \item पृथ्वी की कितनी बार \omterm{def:g11:electric-gravitational-fields:field}{मैदान} है \qty{1.5}{T}?
  \item टर्न घनत्व की गणना करें $ n$ आवश्यकता है \emph{बिना} कोई भी लौह कोर.
  \item यदि वाइंडिंग होती \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} \qty{0.20}{\ohm}, क्या \omterm{def:g10:energy-conservation:power}{शक्ति}
        क्या यह नष्ट हो जाएगा? वास्तविक वाइंडिंग की कौन सी संपत्ति से बचा जाता है?
        यह, अत्यधिक ठंड की कीमत पर?
  \item पंचलाइन: जहाज, क्रेन और एमआरआई रखें खेत
        अंतरतारकीय अंतरिक्ष से दस शक्तियों की सीढ़ी पर
        (\qty{e-10}{T}) एक न्यूट्रॉन स्टार तक (\qty{e8}{T}); क्या बदला
        मशीनों के बीच --- भौतिकी, या \omterm{def:g11:circuits-and-power:current}{एम्पीयर}?
\end{enumerate}
\end{problem}
