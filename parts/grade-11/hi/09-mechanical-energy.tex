\chapter{यांत्रिक ऊर्जा और उसका संरक्षण}\label{ch:g11:mechanical-energy}

एक रोलर कोस्टर में पहली पहाड़ी से आगे कोई इंजन नहीं होता: हर लूप, हर
बाद में बर्स्ट ऑफ स्पीड ऊपर जाते समय खोले गए खाते से खर्च किया जाता है।
यह अध्याय बहीखाता को दो संख्याओं में बदल देता है --- गतिज और
संभावित ऊर्जा --- और एक नियम: केवल जबकि \omterm{def:g10:universal-gravitation:weight}{वज़न} \omterm{def:g11:work-of-force:work}{काम} करता है, उनका योग
नहीं बदलता. \omterm{def:g10:inertia:inventory}{टकराव} शुल्क है; हम इसे पढ़ना सीखेंगे
किताबें.

\section{गतिज ऊर्जा}

\begin{definition}[गतिज ऊर्जा]\label{def:g11:mechanical-energy:kinetic}
द्रव्यमान का एक पिंड $m $ (\unit{kg}) गति से घूमना $ v $ (\unit{m/s}) ले जाता है
द \emph{गतिज ऊर्जा}\index{गतिज ऊर्जा}
\[
E_k = \tfrac12\, m v^2 .
\]
इकाइयों जाँच करना: $\unit{kg.m^2/s^2} = \unit{N.m} = \unit{J}$ --- \omterm{def:g11:work-of-force:work}{जूल},
हर की तरह ऊर्जा (\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[परिमाण के आदेश]\label{ex:g11:mechanical-energy:orders}
\begin{itemize}
  \item एक पैदल यात्री, \qty{70}{kg} पर \qty{1.4}{m/s}:
        $ E_k = \tfrac12 \times 70 \times 1.4^2 \approx \qty{69}{J}$;
  \item एक कार, \qty{1300}{kg} पर \qty{130}{km/h} $= \qty{36.1}{m/s}$:
        $ E_k \approx \qty{8.5e5}{J}$;
  \item एक हाई-स्पीड ट्रेन, $ m = \qty{3.85e5}{kg}$, पर \qty{320}{km/h}
        $= \qty{88.9}{m/s}$: $ E_k \approx \qty{1.5e9}{J}$;
  \item एक राइफल की गोली, \qty{8.0}{g} पर \qty{800}{m/s}: $ E_k \approx \qty{2.6}{kJ}$.
\end{itemize}
परिमाण के सात क्रम टहलने को ट्रेन से अलग करते हैं। और $ E_k $
की तरह बढ़ता है \emph{वर्ग} गति का: पर \qty{130}{km/h} कार
ले जाता है $4$ कई बार ऊर्जा यह पर है \qty{65}{km/h} --- चार बार
इसके ब्रेक को क्या हटाना होगा।
\end{example}

\section{गतिज-ऊर्जा प्रमेय}

\begin{theorem}[गतिज-ऊर्जा प्रमेय]\label{thm:g11:mechanical-energy:ketheorem}
\index{गतिज-ऊर्जा प्रमेय}
द्रव्यमान के एक पिंड के लिए $ m $ से एक सीधी रेखा में आगे बढ़ना $ A $ को $ B $,
का परिवर्तन \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} कुल के बराबर है \omterm{def:g11:work-of-force:work}{काम}
(\cref{ch:g11:work-of-force}) प्राप्त हुआ:
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
कुल योग सब से ऊपर चल रहा है बल शरीर पर क्रिया करना. तक फैला हुआ है
किसी भी पथ पर गति, जैसा कि हम यहां भी स्वीकार करते हैं।
\end{theorem}
\admitted

\begin{example}[फ्री फॉल, फ्रेम दर फ्रेम चेक किया गया]\label{ex:g11:mechanical-energy:freefall}
ए \qty{0.200}{kg} गेंद को गिराया जाता है और फिल्माया जाता है; फ़्रेम देते हैं:
\begin{center}\small
\begin{tabular}{lcccc}
$ t $ (\unit{s}) & 0.10 & 0.20 & 0.30 & 0.40\\
fallen height $ h $ (\unit{m}) & 0.049 & 0.196 & 0.441 & 0.785\\
speed $ v $ (\unit{m/s}) & 0.98 & 1.96 & 2.94 & 3.92\\
$ E_k = \tfrac12 mv^2$ (\unit{J}) & 0.096 & 0.384 & 0.864 & 1.54\\
$ W = mgh $ (\unit{J}) & 0.096 & 0.385 & 0.865 & 1.54
\end{tabular}
\end{center}
से शुरू $ E_k = 0$, हर फ्रेम पर $ E_k $ से मेल खाता है \omterm{def:g11:work-of-force:work}{काम} $ mgh $ की \omterm{def:g10:universal-gravitation:weight}{वज़न}: $\Delta E_k = W(\vect P)$, यानी\ $ v^2 = 2gh $ --- प्रमेय,
रीडिंग की सटीकता से सत्यापित।
\end{example}

\begin{remark}[जहां सबूत रहता है]\label{rem:g11:mechanical-energy:proof}
सामान्य कथन---किसी भी रास्ते पर मान्य है, न कि केवल सीधे रास्ते पर
पंक्ति --- \omterm{def:g10:inertia:force}{न्यूटन} के दूसरे नियम और थोड़ा कैलकुलस, दोनों से अनुसरण करती है
बाद में इस पुस्तक में (\cref{ch:g12:newtons-laws,%
ch:g12:work-and-energy}); तब तक फ्री-फ़ॉल चेक हमारा है
वारंट, और हम घुमावदार-पथ संस्करण को स्वीकार करते हैं और इसका स्वतंत्र रूप से उपयोग करते हैं।
\end{remark}

\section{गुरुत्वाकर्षण स्थितिज ऊर्जा}

\begin{definition}[गुरुत्वाकर्षण स्थितिज ऊर्जा]\label{def:g11:mechanical-energy:potential}
द्रव्यमान का एक पिंड $ m $ ऊंचाई पर $ z $, चुने हुए से ऊपर की ओर मापा गया
\emph{संदर्भ स्तर}\index{संदर्भ स्तर} कहाँ $ z = 0$, भंडारित करता है
\emph{गुरुत्वाकर्षण स्थितिज ऊर्जा}\index{गुरुत्वाकर्षण स्थितिज ऊर्जा}
\[
E_p = m g z, \qquad g = \qty{9.81}{N/kg}.
\]
\end{definition}

\begin{remark}[केवल मतभेद मायने रखते हैं]\label{rem:g11:mechanical-energy:reference}
से चलती है $ A $ को $ B $ परिवर्तन $ E_p $ द्वारा $ mg(z_B - z_A) =
-W_{AB}(\vect P)$: के विपरीत \omterm{def:g11:work-of-force:work}{काम} की \omterm{def:g10:universal-gravitation:weight}{वज़न}
(\cref{ch:g11:work-of-force}). \omterm{def:g11:mechanical-energy:potential}{संदर्भ स्तर} एक स्वतंत्र विकल्प है ---
फर्श, समुद्र तल, टेबल टॉप --- स्थानांतरण $ E_p $ हर जगह एक स्थिरांक द्वारा
वह हर अंतर से बाहर हो जाता है। इसे वहां रखें जहां संख्याएं हैं
सबसे सरल, और इसे समस्या के बीच में न ले जाएँ।
\end{remark}

\begin{example}[शेल्फ पर एक किताब]\label{ex:g11:mechanical-energy:shelf}
ए \qty{1.2}{kg} किताब बैठती है \qty{1.8}{m} मंजिल के ऊपर, स्वयं
\qty{9.0}{m} सड़क के ऊपर. फर्श पर संदर्भ:
$ E_p = 1.2 \times 9.81 \times 1.8 \approx \qty{21}{J}$; गली में:
$ E_p \approx \qty{127}{J}$. फर्श पर गिरने से मुक्ति मिलती है \qty{21}{J} में
दोनों खाते.
\end{example}

\section{यांत्रिक ऊर्जा और उसका संरक्षण}

\begin{definition}[मेकेनिकल ऊर्जा]\label{def:g11:mechanical-energy:em}
 \emph{मेकेनिकल ऊर्जा}\index{मेकेनिकल ऊर्जा} एक शरीर का है
\[
E_m = E_k + E_p = \tfrac12 m v^2 + m g z .
\]
\end{definition}

\begin{theorem}[यांत्रिक ऊर्जा का संरक्षण]\label{thm:g11:mechanical-energy:conservation}
\index{यांत्रिक ऊर्जा का संरक्षण}
यदि एकमात्र बल एक शरीर पर \omterm{def:g11:work-of-force:work}{काम} करना उसका है \omterm{def:g10:universal-gravitation:weight}{वज़न} (नहीं \omterm{def:g10:inertia:inventory}{टकराव}; अन्य
बल, एक घर्षण रहित ट्रैक की प्रतिक्रिया की तरह, लंबवत
गति), $ E_m $ संरक्षित है: किन्हीं दो बिंदुओं के लिए $ A $, $ B $ गति का,
\[
\tfrac12 m v_A^2 + m g z_A = \tfrac12 m v_B^2 + m g z_B .
\]
\end{theorem}

\begin{proof}
से \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} प्रमेय, $ E_k(B) - E_k(A) = W_{AB}(\vect P) =
mg(z_A - z_B) = E_p(A) - E_p(B)$: क्या $ E_k $ लाभ, $ E_p $ हार जाता है.
\end{proof}

\begin{example}[पानी की स्लाइड]\label{ex:g11:mechanical-energy:slide}
एक बच्चा ऊंचाई की घर्षण रहित स्लाइड पर आराम करना शुरू करता है
$ h = \qty{3.2}{m}$. नीचे संदर्भ: $ mgh = \tfrac12 mv^2$, इसलिए
$ v = \sqrt{2gh} \approx \qty{7.9}{m/s}$ (\qty{29}{km/h}) --- जो भी हो
जन \emph{और स्लाइड का आकार कुछ भी हो}: सीधा, घुमावदार
या सर्पिल, केवल ऊंचाई की बूंद ही प्रवेश करती है।
\end{example}

\begin{example}[लंगर]\label{ex:g11:mechanical-energy:pendulum}
एक पेंडुलम बॉब को तब तक एक तरफ खींचा जाता है जब तक वह ऊपर न उठ जाए $ h = \qty{12}{cm}$ ऊपर
इसका निम्नतम बिंदु, फिर जारी किया गया। चरम सीमा पर $ v = 0$; सबसे नीचे
संपूर्ण $ mgh $ गतिज है, $ v = \sqrt{2gh} \approx \qty{1.5}{m/s}$;
दूर की तरफ बॉब बिल्कुल वापस चढ़ जाता है \qty{12}{cm}.
\end{example}

\section{घर्षण द्वारा अपव्यय}

\begin{definition}[अपव्यय]\label{def:g11:mechanical-energy:dissipation}
ऊर्जा यांत्रिक खाता छोड़ना कहा जाता है
\emph{व्यस्त}\index{व्यस्त}: यह फिर से प्रकट होता है \omterm{def:g10:energy-conservation:forms}{थर्मल ऊर्जा} में
रगड़ने वाली सतहें और हवा। कुल ऊर्जा संरक्षित है
(\cref{ch:g10:energy-conservation}); \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} अकेला नहीं है.
\end{definition}

\begin{proposition}[घाटा घर्षण को मापता है]\label{prop:g11:mechanical-energy:deficit}
यदि एक \omterm{def:g10:inertia:inventory}{टकराव} बल $\vect f $ भी \omterm{def:g11:work-of-force:work}{काम} करता है गति के साथ,
\[
\Delta E_m = E_m(B) - E_m(A) = W_{AB}(\vect f) < 0:
\]
का नुकसान \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} के बराबर है \omterm{def:g11:work-of-force:work}{काम} का \omterm{def:g10:inertia:inventory}{टकराव}.
\end{proposition}

\begin{proof}
 \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} प्रमेय अब पढ़ता है $\Delta E_k = W_{AB}(\vect P) +
W_{AB}(\vect f) = -\Delta E_p + W_{AB}(\vect f)$; कदम $\Delta E_p $
आर-पार। \omterm{def:g10:inertia:inventory}{टकराव} प्रस्ताव का विरोध करता है, इसलिए यह \omterm{def:g11:work-of-force:work}{काम} नकारात्मक है.
\end{proof}

\begin{example}[टोबोगन ऑडिट]\label{ex:g11:mechanical-energy:toboggan}
एक बच्चा और स्लेज ($ m = \qty{40}{kg}$) \omterm{def:g10:relative-motion:frame}{आराम से} शुरू करें, छोड़ें
$ h = \qty{5.0}{m}$ साथ में \qty{12}{m} बर्फ़ की, और पहुंचें
\qty{7.0}{m/s} घर्षण रहित के बजाय
$\sqrt{2gh} \approx \qty{9.9}{m/s}$. नीचे संदर्भ:
$\Delta E_m = \tfrac12 \times 40 \times 7.0^2 - 40 \times 9.81
\times 5.0 = 980 - 1962 = \qty{-982}{J}$. इसलिए $ W(\vect f) =
\qty{-982}{J}$ ऊपर \qty{12}{m}: एक औसत \omterm{def:g10:inertia:inventory}{टकराव} बल
$ f = 982/12 \approx \qty{82}{N}$, और \qty{982}{J} में गर्माहट का
धावक और बर्फ.
\end{example}

यह --- प्रत्येक क्षण में गिरावट अब तक उत्पन्न गर्मी है।}
\end{omfigure}

\begin{method}[ऊर्जा बहीखाता]\label{met:g11:mechanical-energy:bookkeeping}
\begin{enumerate}
  \item एक विकल्प चुनें \omterm{def:g11:mechanical-energy:potential}{संदर्भ स्तर} और इसे रखो; नाम $ A $ (डेटा), $ B $
        (सवाल)।
  \item लिखना $ E_m = \tfrac12 mv^2 + mgz $ पर $ A $ और कम से $ B $.
  \item नहीं \omterm{def:g10:inertia:inventory}{टकराव}: $ E_m(A) = E_m(B)$; हल करें --- सामान्यतः
        $ v_B = \sqrt{v_A^2 + 2g(z_A - z_B)}$.
  \item \omterm{def:g10:inertia:inventory}{टकराव}: $ E_m(B) - E_m(A) = -f\ell $ ($\ell $ = पथ की लंबाई);
        अज्ञात का समाधान करें.
  \item जाँचें $ E_k $/$ E_p $ सलाखें ढेर हो गईं $ E_m $, स्तर जब तक \omterm{def:g10:inertia:inventory}{टकराव}.
\end{enumerate}
\end{method}

\section{समस्या: लूप को डिज़ाइन करना}

\begin{problem}[{सप्ताहांत समस्या---एक रोलर का ऊर्जा ऑडिट
कोस्टर: लॉन्च ऊंचाई एक लूप तय करती है, जिस गति का यह वादा करती है, वह
घर्षण को दो सेंसरों द्वारा स्वीकार किया गया, ब्रेक जो खाता बंद करते हैं ---
पूरी सवारी एक नंबर से लटकी हुई है}]
\label{pb:g11:mechanical-energy:1}
जनसमूह की एक रेलगाड़ी $ m = \qty{2.0e3}{kg}$ एक प्रक्षेपण से आराम की स्थिति में जारी किया जाता है
ऊंचाई की पहाड़ी $ H $ और तट, इंजन रहित, एक ऊर्ध्वाधर गोलाकार के माध्यम से
त्रिज्या का पाश $ R = \qty{8.0}{m}$ जिसका शीर्ष ऊंचाई पर बैठता है $2R $.
सर्कुलर-मोशन विश्लेषण (अगले वर्ष) वह डेटा प्रदान करता है जो हम उधार लेते हैं: पर
ट्रेन के ऊपर का लूप तभी ट्रैक पर दबता है
$ v_{\text{top}}^2 \geq gR $. भाग I और II अनदेखा करें \omterm{def:g10:inertia:inventory}{टकराव}.

\medskip
\noindent\textbf{Part I --- The hill the loop dictates.}
\begin{enumerate}
  \item लूप के शीर्ष पर न्यूनतम सुरक्षित गति की गणना करें।
  \item \omterm{def:g11:electric-gravitational-fields:field}{मैदान} के रूप में \omterm{def:g11:mechanical-energy:potential}{संदर्भ स्तर}: ट्रेन की गणना करें \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} उस न्यूनतम गति पर लूप टॉप पर।
  \item परिणाम निकालना $ H_{\min}$ और दिखाओ $ H_{\min} = 5R/2$ --- कोई द्रव्यमान नहीं, नहीं $ g $.
  \item टीम लेती है $ H = \qty{24}{m}$: वास्तविक शीर्ष-लूप की गणना करें
        गति और अनुपात $ v_{\text{top}}^2/(gR)$.
  \item जहाँ किया $ m $ जाना? डिज़ाइन पूर्ण और खाली क्यों फिट होता है?
        ट्रेनें एक जैसी?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- What the height promises} ($ H = \qty{24}{m}$).
\begin{enumerate}[resume]
  \item लूप के नीचे गति ($ z = 0$), में \unit{m/s} और
        \unit{km/h}?
  \item लूप के किनारे पर गति ($ z = R $)?
  \item गणना करना $ E_p $ और $ E_k $ लॉन्च के समय, नीचे, किनारे और ऊपर
        पाश; जांचें कि प्रत्येक जोड़ी का योग कितना है $ mgH $.
  \item ए \qty{12}{m} ऊँट-पीठ की टक्कर लूप का अनुसरण करती है: गति अपने शिखर पर?
  \item का सामान्य सूत्र बताइये $ v(z)$; सवारी कहाँ है
        सबसे तेज़, और सबसे धीमा?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g10:inertia:inventory}{टकराव} audit.}
एक ही ऊंचाई पर दो सेंसर $ z = \qty{2.0}{m}$, \qty{60}{m} का
अलग ट्रैक करें, पढ़ें $ v_1 = \qty{20.4}{m/s}$ तब $ v_2 = \qty{19.2}{m/s}$.
\begin{enumerate}[resume]
  \item के परिवर्तन की गणना करें \omterm{def:g11:mechanical-energy:em}{मेकेनिकल ऊर्जा} सेंसर के बीच.
        समान-ऊंचाई वाले प्लेसमेंट को किस चीज़ ने चतुर बनाया?
  \item निष्कर्ष निकालें \omterm{def:g11:work-of-force:work}{काम} का \omterm{def:g10:inertia:inventory}{टकराव} और औसत \omterm{def:g10:inertia:inventory}{टकराव} बल.
  \item कहाँ गायब है ऊर्जा गया? स्वरूप एवं स्थानों के नाम बतायें।
  \item ट्रेन का कितना अंश $ E_m $ प्रति खो गया है \qty{60}{m}? क्यों
        नंगे होंगे $ H_{\min}$ वास्तविक, घिनौनी दुनिया में असुरक्षित रहें?
  \item के बारे में \qty{90}{m} ट्रैक का अलग लॉन्च और लूप टॉप: अनुमान
        $ E_m $ वहाँ; क्या प्रश्न 4 का मार्जिन अभी भी लूप को बचाता है?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Closing the account.}
अंतिम सेंसर, सीधे क्षैतिज ब्रेकिंग रन के प्रवेश पर
($ z = 0$), पढ़ता है \qty{18.0}{m/s}.
\begin{enumerate}[resume]
  \item ट्रेन की गणना करें \omterm{def:g11:mechanical-energy:kinetic}{गतिज ऊर्जा} वहाँ।
  \item ब्रेक को इसे अंदर रोकना होगा $ d = \qty{45}{m}$: गणना करें
        आवश्यक (निरंतर) ब्रेक लगाना बल.
  \item आराम की जांच: मंदी की गणना करें $ a = F/m $; से तुलना करें $ g $.
  \item कितना ऊर्जा ट्रैक किया \omterm{def:g10:inertia:inventory}{टकराव} पूरी तरह से नष्ट हो जाना
        सवारी? सत्यापित करें कि \omterm{def:g10:inertia:inventory}{टकराव} $+$ ब्रेक $= mgH$, तक \omterm{def:g10:energy-conservation:energy}{जौल}.
  \item पंचलाइन, एक वाक्य: किस एकल संख्या ने लूप को ठीक किया
        सुरक्षा, हर गति, \omterm{def:g10:inertia:inventory}{टकराव} बिल और ब्रेक बल ---
        और अनदेखी की कीमत क्या थी? \omterm{def:g10:inertia:inventory}{टकराव}?
\end{enumerate}
\end{problem}
