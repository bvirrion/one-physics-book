\chapter{बल और गति}\label{ch:g11:forces-and-motion}

पूरे कमरे में एक रिंच घुमाते हुए फेंकें: फिर भी सिरे बेतहाशा हिलते रहते हैं
इसके अंदर एक बिंदु एक पूर्ण परवलय का पता लगाता है। \Cref{ch:g10:inertia}
तय हो गया कि कब क्या होगा बल \omterm{def:g10:inertia:compensate}{मुआवजा} --- कुछ नहीं। यह अध्याय
पूछता है कि जब वे ऐसा नहीं करते तो क्या होता है: बचा हुआ बल \emph{परिवर्तन वेग} --- शरीरों को तेज़ करना, उन्हें धीमा करना, उन्हें मोड़ना
--- और कहता है कि कितना, अनुपात में यदि अभी तक सूत्रों में नहीं।

\section{जनसमूह का केंद्र}

\begin{definition}[सेंटर ऑफ मास]\label{def:g11:forces-and-motion:com}
 \emph{सेंटर ऑफ मास}\index{सेंटर ऑफ मास} एक शरीर या प्रणाली का है
इसका संतुलन बिंदु --- इसके द्रव्यमान की औसत स्थिति: ज्यामितीय
एक सजातीय सममित शरीर का केंद्र; दो बिंदु द्रव्यमानों के लिए, बिंदु
की दूरी पर जुड़ने वाले खंड की \emph{श्लोक में} का अनुपात
जनता, $\frac{d_1}{d_2} = \frac{m_2}{m_1}$ --- भारी के करीब.
\end{definition}

\begin{example}[एक असंतुलित डम्बल]\label{ex:g11:forces-and-motion:dumbbell}
की गेंदें \qty{2.0}{kg} और \qty{1.0}{kg} एक हल्की पट्टी के सिरों पर बैठें
\qty{0.90}{m} लंबा। \omterm{def:g11:forces-and-motion:com}{सेंटर ऑफ मास} इसे व्युत्क्रम अनुपात में विभाजित करता है
$ d_1/d_2 = 1.0/2.0$: $\qty{0.30}{m}$ भारी गेंद से,
$\qty{0.60}{m}$ प्रकाश वाले से --- वहां बार को संतुलित करें।
\end{example}

\begin{remark}[फेंका हुआ रिंच]\label{rem:g11:forces-and-motion:wrench}
एक टंबलिंग रिंच को फिल्माएं: प्रत्येक बिंदु एक जटिल लूपिंग वक्र का अनुसरण करता है
--- को छोड़कर \omterm{def:g11:forces-and-motion:com}{सेंटर ऑफ मास}, जो फेंके गए स्वच्छ परवलय का पता लगाता है
गेंद होगी. की गति \omterm{def:g11:forces-and-motion:com}{सेंटर ऑफ मास} साधारण है; बाकी स्पिन है
\emph{आस-पास} यह --- यही कारण है कि यांत्रिकी एक कार, एक ग्रह या का इलाज कर सकते हैं
एक एकल बिंदु के रूप में एक जिमनास्ट।
\end{remark}

\section{एक शुद्ध बल वेग को बदल देता है}

\begin{definition}[शुद्ध बल]\label{def:g11:forces-and-motion:netforce}
 \emph{शुद्ध बल}\index{शुद्ध बल} किसी पिंड पर सदिश योग होता है
सभी बल उस पर कार्य करना, $\vect F = \vect F_1 + \vect F_2 + \dots $
(टिप से टेल तक, जैसा कि गणित के खंड में है); बल \omterm{def:g10:inertia:compensate}{मुआवजा}
बिल्कुल कब $\vect F = \vect 0$.
\end{definition}

\begin{theorem}[शुद्ध बल का प्रभाव---अर्ध-मात्रात्मक]\label{thm:g11:forces-and-motion:secondlaw}
\index{गति का दूसरा नियम}
ए \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} $\vect F $ द्रव्यमान के पिंड पर कार्य करना $ m $ छोटी \omterm{def:g10:signals-and-waves:period}{अवधि} के लिए
$\Delta t $ इसके वेग को बदल देता है \omterm{def:g11:forces-and-motion:com}{सेंटर ऑफ मास} एक वेक्टर द्वारा
$\Delta \vect v $ उसके पास है \emph{दिशा} का $\vect F $, और किसका
परिमाण अनुपात में बढ़ता है $ F $ और को $\Delta t $ और सिकुड़ जाता है
द्रव्यमान का अनुपात:
\[
\Delta v \propto \frac{F \, \Delta t}{m} .
\]
\end{theorem}
\admitted

\begin{remark}
दोगुना करें बल या इसकी \omterm{def:g10:signals-and-waves:period}{अवधि}: दो बार परिवर्तन; द्रव्यमान दोगुना करें:
आधा. अगला वर्ष इसे एक सटीक समीकरण में बदल देता है, और वर्ष 1
वॉल्यूम पूछता है कि यह किस फ्रेम में है; इस वर्ष अनुपात पर्याप्त हैं ---
वे पहले से ही एयरबैग, ब्रेक आदि चलाते हैं कक्षाओं.
\end{remark}

\begin{proposition}[तीन विहित मामले]\label{prop:g11:forces-and-motion:cases}
किसी पिंड को वेग से चलने दें $\vect v $ एक के तहत \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} $\vect F $.
\begin{enumerate}
  \item $\vect F $ साथ में $\vect v $: यह \emph{रफ्तार बढ़ाओ} एक सीधी रेखा में;
  \item $\vect F $ विलोम $\vect v $: यह \emph{धीरे करता है} एक सीधी रेखा में;
  \item $\vect F $ के लिए सीधा $\vect v $: यह \emph{मोड़ों} बिना
        बदलती गति.
\end{enumerate}
\end{proposition}

\begin{proof}
जोड़ना $\Delta\vect v $, साथ निर्देशित $\vect F $
(\cref{thm:g11:forces-and-motion:secondlaw}), टिप टू टेल ऑन
$\vect v $: साथ में $\vect v $ तीर ढेर हो जाते हैं (लंबे समय तक); इसके विपरीत, वे
आंशिक रूप से रद्द (छोटा); लंबवत, प्रत्येक छोटा $\Delta \vect v $
तीर को लम्बा किये बिना झुका देता है--- केवल दिशा बदल जाती है।
\end{proof}

\begin{example}[एक बग़ल में लात]\label{ex:g11:forces-and-motion:kick}
एक पक पूर्व की ओर उड़ता है \qty{6.0}{m/s}; एक संक्षिप्त बग़ल में झटका जोड़ता है
$\Delta\vect v = \qty{8.0}{m/s}$ प्रधान दिक्सूचक नोक 0 या 360 डिग्री पर। नया वेग है
वेक्टर योग: परिमाण $\sqrt{6.0^2 + 8.0^2} = \qty{10.0}{m/s}$,
दिशा $\tan\theta = 8.0/6.0$, यानी\ $\theta \approx 53^\circ $ उत्तर
पूर्व का---पुराना वेग मिटता नहीं, उसमें परिवर्तन जुड़ जाता है।
\end{example}

\section{तुलनात्मक परिवर्तन: बल, समय, द्रव्यमान}

\begin{method}[अनुपात तर्क]\label{met:g11:forces-and-motion:ratio}
दो स्थितियों की तुलना करने के लिए, कभी भी एक समीकरण को हल न करें --- अनुपात बनाएं:
वेग-परिवर्तन अनुपात है $ F $ अनुपात, गुना $\Delta t $ अनुपात,
द्वारा विभाजित $ m $ अनुपात। फिर पढ़िए कि कौन सा कारक कितना बदला।
\end{method}

\begin{example}[भरी हुई गाड़ी]\label{ex:g11:forces-and-motion:cart}
एक ही धक्का, एक ही सेकंड के लिए लगाया गया, एक खाली भेजता है
\qty{20}{kg} शॉपिंग कार्ट बंद \qty{1.2}{m/s}. पर लोड किया गया
\qty{60}{kg} --- त्रिगुण द्रव्यमान, समान $ F\Delta t $ --- यह ए पर निकल जाता है
उसका तीसरा, \qty{0.40}{m/s}. द्रव्यमान पदार्थ का हठ है।
\end{example}

\begin{remark}[दुर्घटनाएं, एयरबैग, मुड़े हुए घुटने]\label{rem:g11:forces-and-motion:airbag}
एक दुर्घटना में $\Delta v $ और $ m $ परक्राम्य नहीं हैं --- कब्जेदार के
गति शून्य तक पहुंचनी चाहिए --- इसलिए $ F\,\Delta t $ तय है. एकमात्र स्वतंत्रता
बायां है \emph{समय}: खींचना $\Delta t $ और यह बल में सिकुड़ जाता है
समान अनुपात. एक सिर अंदर एयरबैग पर रुक रहा है \qty{0.08}{s} के बजाय पर
पहिया अंदर \qty{0.01}{s} औसत लगता है बल आठ गुना छोटा;
क्रम्पल जोन, हेलमेट फोम और मुड़े हुए घुटने एक ही चाल खेलते हैं।
\end{remark}

\section{घर्षण: वह बल जो फिसलने से लड़ता है}

\begin{definition}[स्थैतिक और गतिज घर्षण]\label{def:g11:forces-and-motion:friction}
\omterm{def:g10:inertia:inventory}{टकराव} स्पर्श करने वाली सतहों के बीच उनका विरोध होता है \emph{रिश्तेदार फिसलन}. \emph{स्थैतिक घर्षण}\index{स्थैतिक घर्षण} जब वे कार्य करते हैं
स्लाइड न करें, जो भी प्रयास करता है उसे रद्द करने के लिए खुद को अधिकतम तक समायोजित करें
स्लाइड शुरू करने के लिए; \emph{गतिज घर्षण}\index{गतिज घर्षण},
एक बार फिसलन शुरू हो जाने के बाद, लगभग स्थिर रहती है, इसके विपरीत निर्देशित होती है
फिसलन, और आमतौर पर \emph{कमज़ोर} स्थैतिक अधिकतम से.
\end{definition}

\begin{example}[अलमारी को धक्का देना]\label{ex:g11:forces-and-motion:wardrobe}
अलमारी को धीरे से धकेलें: यह हिलती-डुलती नहीं --- स्थिर \omterm{def:g10:inertia:inventory}{टकराव} रद्द करता है
आपका \qty{150}{N}. \omterm{def:g10:inertia:inventory}{जोर} से धक्का दें: कुछ पर \qty{400}{N} यह अंदर घुस जाता है
गति, फिर कम प्रयास से फिसलती है --- कमजोर \omterm{def:g11:forces-and-motion:friction}{गतिज घर्षण}
पर कब्ज़ा कर लिया है.
\end{example}

\begin{remark}[एबीएस ब्रेक क्यों काम करते हैं?]\label{rem:g11:forces-and-motion:abs}
घूमते हुए पहिये का संपर्क बिंदु सड़क पर नहीं फिसलता: टायर
के साथ पकड़ता है \emph{स्थिर} \omterm{def:g10:inertia:inventory}{टकराव}. एक बंद पहिया फिसलता है और केवल मिलता है
कमज़ोर \emph{गतिज} \omterm{def:g10:inertia:inventory}{टकराव}, जो स्किड के विरुद्ध इंगित करता है
पहिये कुछ भी करें --- स्टीयरिंग मर जाता है। एबीएस ब्रेक रिलीज करता है
तुरंत एक पहिया लॉक हो जाता है: अधिक पकड़, कम स्टॉप, एक ड्राइवर जो चला सकता है।
\end{remark}

\section{घूमना: वृत्ताकार गति के लिए आंतरिक बल की आवश्यकता होती है}

\begin{definition}[एकसमान वृत्तीय गति]\label{def:g11:forces-and-motion:ucm}
एक शरीर अंदर है \emph{एकसमान वृत्तीय गति}\index{एकसमान वृत्तीय गति} कब इसकी \omterm{def:g10:relative-motion:trajectory}{प्रक्षेपवक्र} स्थिर गति से चला गया एक वृत्त है। द
वेग, वृत्त की स्पर्श रेखा, हर पल दिशा बदलती है ---
इसलिए यह कभी स्थिर नहीं रहता.
\end{definition}

\begin{proposition}[भीतर की शक्ति]\label{prop:g11:forces-and-motion:inward}
एक शरीर में \omterm{def:g11:forces-and-motion:ucm}{एकसमान वृत्तीय गति} ए के अधीन है \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} वह है
कभी शून्य नहीं: हर क्षण यह शरीर से केंद्र की ओर इंगित करता है
वृत्त का.
\end{proposition}

\begin{proof}
गति स्थिर है, इसलिए \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} साथ में कोई भाग नहीं है $\vect v $
(केस 1 और 2 \cref{prop:g11:forces-and-motion:cases}): यह है
स्पर्शरेखा के लंबवत, इसलिए त्रिज्या के अनुदिश --- और अंदर की ओर,
प्रत्येक के बाद से $\Delta \vect v $ मार्ग को केंद्र की ओर मोड़ देता है।
\end{proof}

\section{समस्या: क्रैश टेस्ट}

\begin{problem}[{सप्ताहांत समस्या --- हल्की टक्कर को डिज़ाइन करना:
चूँकि एक दुर्घटना कार के वेग परिवर्तन को ठीक कर देती है, इसलिए कार का प्रत्येक सुरक्षा उपकरण
समय बढ़ाने की एक मशीन है}]
\label{pb:g11:forces-and-motion:1}
एक क्रैश-टेस्ट प्रयोगशाला एक कार चलाती है \qty{50}{km/h} एक कठोर में
दीवार, एक के साथ \qty{70}{kg} वयस्क डमी और ए \qty{20}{kg} बच्चा डमी
सवार; हर पड़ाव पर कैमरे का समय। आपका काम: पता लगाएं कि हिंसा कहां है
क्रैश जीवन, और इसे दूर डिज़ाइन करें।

\medskip
\noindent\textbf{Part I --- The measure of a crash.}
\begin{enumerate}
  \item बदलना \qty{50}{km/h} को \unit{m/s}.
  \item किसी भी फ्रंटल स्टॉप में रहने वालों का वेग परिवर्तन समान होता है
        परिमाण. कौन सा? कोई भी उपकरण इसे क्यों नहीं बदल सकता?
  \item उद्धरण \cref{thm:g11:forces-and-motion:secondlaw}: साथ $\Delta v $
        और $ m $ तय, $ F\,\Delta t $ तय है. डिज़ाइनर का एकमात्र डायल?
  \item दीवार कार को अंदर रोक देती है \qty{0.050}{s}; एक क्रम्पल ज़ोन
        इसे फैलाता है \qty{0.150}{s}. औसत की तुलना करें बल.
  \item किस दिशा में करें $\Delta \vect v $ और यह \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} पर
        रुकने के दौरान डमी इशारा करते हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crumple zone.}
\begin{enumerate}[resume]
  \item एक वाक्य में कहें तो इंजीनियर कार का अगला भाग क्यों डिज़ाइन करते हैं?
        नष्ट हो जाना है?
  \item एक पुरानी कठोर कार में (अंदर रुकें) \qty{0.050}{s}) डमी बंधी हुई है
        सीट पर सख्ती से। इसकी तुलना करें बल आधुनिक के साथ \qty{0.150}{s} रुकना।
  \item रुकने वाली कार की गति लगातार कम हो जाती है (औसत: प्रारंभिक की आधी)।
        यह कितनी दूर तक यात्रा करता है \qty{0.150}{s}? वास्तविक क्रम्पल जोन से तुलना करें।
  \item क्यों नहीं ए \qty{10}{m} सिकुड़ने की गुंजाइश का स्थान? और कार का कौन सा पार्ट
        अवश्य \emph{नहीं} सिकुड़ना?
  \item भाग II को एक नारे के रूप में सारांशित करें: ``आप चुन नहीं सकते $\Delta v $,
        इसलिए आपको चुनना होगा \dots''.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The belt, the airbag and the child.}
\begin{enumerate}[resume]
  \item एक बिना बेल्ट वाली डमी इसे रखती है \qty{14}{m/s} (\cref{ch:g10:inertia})
        जब तक कि विंडशील्ड उसे रोक न दे \qty{0.005}{s}; एक बेल्ट वाला
        कार के साथ रुकता है \qty{0.150}{s}. औसत की तुलना करें बल.
  \item एक बेल्ट जो थोड़ी सी खिंचती है वह बेल्ट से बेहतर क्यों होती है?
        अनस्ट्रेचेबल स्टील हार्नेस?
  \item सिर अंदर पहिये पर रुक जाएगा \qty{0.010}{s}; एयरबैग
        इसे फैलाता है \qty{0.080}{s}. तुलना करें बल.
  \item वयस्क और बच्चे समान रूप से गुजरते हैं $\Delta v $ ठीक उसी प्रकार
        $\Delta t $. तुलना करें \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल} दो बेल्ट की आपूर्ति होनी चाहिए।
  \item गोद में एक बच्चा: बाहें बदलनी होंगी \qty{10}{kg} बच्चे का
        द्वारा वेग \qty{14}{m/s} में \qty{0.150}{s}, जबकि गुरुत्वाकर्षण
        द्वारा एक वेग बदलता है \qty{9.8}{m/s} प्रत्येक सेकंड. कितनी बार
        हथियारों को तेजी से कार्य करना चाहिए --- जो द्रव्यमान धारण करेगा वह महसूस होगा
        भारी? निष्कर्ष.
  \item बच्चे छोटा महसूस करते हैं बल --- तो बच्चों की सीटें क्यों? के बारे में सोचें
        \emph{कहाँ} बेल्ट इसे लागू करता है बल, और का $\Delta t$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The safety cage, and the verdict.}
\begin{enumerate}[resume]
  \item पैसेंजर सेल कठोर है जबकि दोनों सिरे मुड़े हुए हैं। क्यों चाहिए?
        कोशिका विकृत नहीं होती, यद्यपि कठोरता का अर्थ हिंसक रूप से रुकना है?
  \item दो एक जैसी कारें आमने-सामने टकराती हैं \qty{50}{km/h}.
        प्रत्येक कहाँ रुकता है? प्रत्येक दुर्घटना की तुलना दीवार परीक्षण से करें।
  \item एक छोटी कार एक भरे हुए ट्रक से आमने-सामने मिलती है। बल वे प्रयास करते हैं
        एक-दूसरे पर बराबर और विपरीत, हर बातचीत की तरह परस्पर हैं
        है (\cref{ch:g11:fundamental-interactions}). जिसका वेग
        और अधिक परिवर्तन?
  \item इस समस्या के सुरक्षा उपकरणों और एकल मात्रा की सूची बनाएं
        उनमें से हर एक खिंचता है।
  \item समापन: हल्की टक्कर के डिज़ाइन नियम को एक में बताएं
        केवल वाक्य का प्रयोग \emph{द्रव्यमान}, \emph{वेग परिवर्तन},
        \emph{बल} और \emph{समय}.
