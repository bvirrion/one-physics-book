\chapter{विद्युत और गुरुत्वाकर्षण क्षेत्र}\label{ch:g11:electric-gravitational-fields}

पृथ्वी उस सेब को कैसे खींचती है जिसे वह कभी नहीं छूती? \omterm{def:g10:inertia:force}{न्यूटन} ने खुद बुलाया था
दूरी पर कार्रवाई ``इतनी बड़ी बेतुकी'' कि उन्होंने बचाव करने से इनकार कर दिया
यह. आधुनिक उत्तर: एक द्रव्यमान, या एक आवेश, अपने आस-पास के स्थान को संशोधित करता है
--- यह एक बनाता है \emph{\omterm{def:g11:electric-gravitational-fields:field}{मैदान}} --- और जो कुछ भी वहां रखा जाता है वह प्रतिक्रिया करता है
को \omterm{def:g11:electric-gravitational-fields:field}{मैदान} अपने ही स्थान पर. हम दो महान उदाहरण पक्ष बनाते हैं
साथ-साथ, जब तक सादृश्य रूप की पहचान न बन जाए।

\section{दूर से कार्रवाई से लेकर मैदान तक}

\begin{definition}[मैदान]\label{def:g11:electric-gravitational-fields:field}
ए \emph{मैदान}\index{मैदान} अंतरिक्ष के प्रत्येक बिंदु को एक वेक्टर निर्दिष्ट करता है। ए
स्रोत (एक आवेश, एक द्रव्यमान) अपने चारों ओर हर जगह अपना क्षेत्र बनाता है,
उसे महसूस करने लायक कुछ है या नहीं; एक परीक्षण वस्तु तब महसूस होती है
बल क्षेत्र द्वारा निर्धारित \emph{अपने ही स्थान पर} अकेला। यह
दो-चरणीय चित्र दूरी पर क्रिया को प्रतिस्थापित करता है।
\end{definition}

\begin{remark}
 \omterm{def:g11:electric-gravitational-fields:field}{मैदान} यह बहीखाता चाल नहीं है: यह चलता है ऊर्जा, और में परिवर्तन
स्रोत इसके माध्यम से सीमित गति से बाहर की ओर यात्रा करता है --- प्रकाश ही है
ऐसी यात्रा \omterm{def:g11:electric-gravitational-fields:field}{मैदान} (\cref{ch:g12:light-as-wave}). यहाँ हम पढ़ते हैं
खेत जो समय के साथ नहीं बदलता.
\end{remark}

\section{विद्युत क्षेत्र}

\begin{definition}[विद्युत क्षेत्र]\label{def:g11:electric-gravitational-fields:efield}
यदि एक छोटा सा परीक्षण शुल्क $q $ एक बिंदु पर रखा गया $ M $ बिजली सी महसूस होती है बल
$\vect F $, द \emph{विद्युत क्षेत्र}\index{विद्युत क्षेत्र} पर $ M $ है
\[
\vect E = \frac{\vect F}{q}
\qquad \text{(in \unit{N/C}, also written \unit{V/m})}.
\]
$\vect E $ केवल स्रोतों आदि पर निर्भर करता है $ M $, परीक्षण पर नहीं
चार्ज: दोगुना $ q $ दोगुना हो जाता है $\vect F $, भागफल को अपरिवर्तित छोड़कर।
\end{definition}

\begin{proposition}[आरोप पर बल लगाना]\label{prop:g11:electric-gravitational-fields:force}
एक आरोप $ q $ कहाँ रखा गया \omterm{def:g11:electric-gravitational-fields:field}{मैदान} है $\vect E $ महसूस करता है बल
$\vect F = q \vect E $: साथ में $\vect E $ अगर $ q > 0$, के विपरीत $\vect E $
अगर $ q < 0$, परिमाण का $ F = \abs{q}\,E $.
\end{proposition}

\begin{proof}
परिभाषा को पुनर्व्यवस्थित करें; चिह्न नियम ऋणात्मक गुणज का बीजगणित है।
\end{proof}

\begin{example}[एक फ़ील्ड पढ़ना]\label{ex:g11:electric-gravitational-fields:reading}
एक परीक्षण शुल्क $ q = \qty{+2.0e-8}{C}$ पर $ M $ एक महसूस होता है बल का
\qty{6.0e-4}{N} पूर्व की ओर इशारा करते हुए. \omterm{def:g11:electric-gravitational-fields:field}{मैदान} पर $ M $ है
$ E = \num{6.0e-4} / \num{2.0e-8} = \qty{3.0e4}{N/C}$, पूर्व की ओर इशारा करते हुए;
$ q' = \qty{-1.0e-8}{C}$ एक ही बिंदु पर रखा हुआ महसूस होता है
$ F = \num{1.0e-8} \times \num{3.0e4} = \qty{3.0e-4}{N}$ --- पश्चिम की ओर इशारा करते हुए.
\end{example}

\begin{proposition}[एक बिंदु आवेश का क्षेत्र]\label{prop:g11:electric-gravitational-fields:point}
एक बिंदु प्रभार $ Q $ दूरी पर बनाता है $ d $, ए \omterm{def:g11:electric-gravitational-fields:field}{मैदान} लाइन के साथ
चार्ज से बिंदु तक जुड़ना --- से दूर $ Q $ अगर $ Q > 0$, इसकी ओर यदि
$ Q < 0$ --- परिमाण का
\[
E = k\,\frac{\abs{Q}}{d^2},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{proposition}

\begin{proof}
कूलम्ब का नियम (\cref{ch:g11:fundamental-interactions}) देता है बल
एक परीक्षण शुल्क पर $ q $ दूरी पर $ d $: परिमाण $ k \abs{qQ} / d^2$,
जुड़ने वाली रेखा के साथ, समान \omterm{def:g10:signals-and-waves:signal}{संकेत}ों के लिए प्रतिकारक। द्वारा विभाजित करें $ q $ और
लागू करें \cref{prop:g11:electric-gravitational-fields:force}.
\end{proof}

\begin{example}[एक आवेशित गोला]\label{ex:g11:electric-gravitational-fields:sphere}
एक वैन डी ग्रेफ़ क्षेत्र ले जाता है $ Q = \qty{+5.0e-7}{C}$ (एक छोटा सा चार्ज
गोला अपने केंद्र पर एक बिंदु आवेश की तरह कार्य करता है)। पर $ d = \qty{0.50}{m}$:
\[
E = \num{8.99e9} \times \num{5.0e-7} / 0.50^2 \approx \qty{1.8e4}{N/C},
\]
गोले से रेडियल रूप से दूर की ओर इशारा करते हुए। पर \qty{1.0}{m} यह गिर गया है
इसके एक चौथाई तक: द $1/d^2$ की बल में जीवित रहता है \omterm{def:g11:electric-gravitational-fields:field}{मैदान}.
\end{example}

\begin{definition}[फ़ील्ड रेखाएँ]\label{def:g11:electric-gravitational-fields:lines}
ए \emph{फ़ील्ड लाइन}\index{फ़ील्ड लाइन} हर जगह एक वक्र स्पर्शरेखा है
द \omterm{def:g11:electric-gravitational-fields:field}{मैदान}, एक तीर दे रहा है मैदानकी दिशा. तीन नियम:
\begin{itemize}
  \item वे सकारात्मक चार्ज छोड़ते हैं और नकारात्मक चार्ज पर पहुंचते हैं;
  \item जहां लाइनें भीड़ लगाती हैं \omterm{def:g11:electric-gravitational-fields:field}{मैदान} मजबूत है; जहां वे फैलते हैं, कमजोर;
  \item दो रेखाएं कभी नहीं कटतीं: द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} प्रत्येक बिंदु पर एक दिशा होती है।
\end{itemize}
\end{definition}

\section{संधारित्र का एकसमान क्षेत्र}

\begin{definition}[समानांतर-प्लेट संधारित्र]\label{def:g11:electric-gravitational-fields:capacitor}
दो आमने-सामने की समानांतर धातु की प्लेटें, एक साथ बंद होती हैं और विपरीत दिशा में चलती हैं
आरोप, फॉर्म ए \emph{संधारित्र}\index{संधारित्र}. प्लेटों के बीच
द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} है \emph{वर्दी}\index{वर्दी}: same magnitude and
हर जगह दिशा --- प्लेटों के लंबवत, सकारात्मक से
plate to the negative.
\end{definition}

\begin{proposition}[प्लेटों के बीच का क्षेत्र]\label{prop:g11:electric-gravitational-fields:uniform}
यदि \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} प्लेटों के बीच (\cref{ch:g11:circuits-and-power}) है
$ U $ और उनका अलगाव है $ d $, द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} उनके बीच परिमाण है
\[
E = \frac{U}{d} \qquad \text{(in \unit{V/m})}.
\]
\end{proposition}
\admitted

\begin{remark}
एकरूपता और $ E = U/d $ वर्ष 1 खंड में ईमानदारी से प्राप्त किए गए हैं; द
सूत्र बताता है \omterm{def:g10:orders-of-magnitude:unit}{इकाई} \unit{V/m}. \omterm{def:g11:lenses-and-eye:lens}{पतला} अंतराल मजबूत बनाते हैं खेत:
\qty{100}{V} एक मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} के पार पहले से ही है \qty{e5}{V/m}.
\end{remark}

\begin{example}[एक इलेक्ट्रॉन किरण का संचालन]\label{ex:g11:electric-gravitational-fields:beam}
एक में \omterm{def:g10:signals-and-waves:oscilloscope}{आस्टसीलस्कप}, किरण प्लेटों के बीच से गुजरती है
$ U = \qty{100}{V}$ और $ d = \qty{5.0}{mm}$:
$ E = 100 / \num{5.0e-3} = \qty{2.0e4}{V/m}$, और प्रत्येक इलेक्ट्रॉन महसूस करता है
$ F = \num{1.60e-19} \times \num{2.0e4} = \qty{3.2e-15}{N}$ की ओर
सकारात्मक प्लेट --- $\num{4e14}$ कई बार यह \omterm{def:g10:universal-gravitation:weight}{वज़न} का \qty{8.9e-30}{N},
इसलिए किरण स्पष्ट रूप से विक्षेपित होती है (घुमावदार पथ है
\cref{ch:g12:newtons-laws}का व्यवसाय)।
\end{example}

\section{गुरुत्वाकर्षण क्षेत्र}

\begin{definition}[गुरुत्वाकर्षण क्षेत्र]\label{def:g11:electric-gravitational-fields:gfield}
यदि एक परीक्षण जन $ m $ एक बिंदु पर रखा हुआ महसूस होता है \omterm{def:g10:universal-gravitation:force}{गुरुत्वाकर्षण बल} (ए
\omterm{def:g10:universal-gravitation:weight}{वज़न}) $\vect P $, द \emph{गुरुत्वाकर्षण क्षेत्र}\index{गुरुत्वाकर्षण क्षेत्र} वहाँ है
\[
\vect g = \frac{\vect P}{m} \qquad \text{(in \unit{N/kg})},
\]
और इसके विपरीत $\vect P = m \vect g $. पहले के एक अध्याय में इसका परिचय दिया गया था
संख्या $ g $; $\vect g $ वह दिशा जोड़ता है जिस दिशा में गिरा हुआ पत्थर चलना शुरू करता है।
\end{definition}

\begin{proposition}[पृथ्वी का क्षेत्र]\label{prop:g11:electric-gravitational-fields:earth}
दूरी पर $ d $ पृथ्वी के केंद्र से (द्रव्यमान) $ M $), द गुरुत्वाकर्षण
फ़ील्ड केंद्र की ओर इंगित करता है और इसका परिमाण है
\[
g = G\,\frac{M}{d^2},
\qquad G = \qty{6.67e-11}{N.m^2/kg^2}.
\]
\end{proposition}

\begin{proof}
सार्वभौमिक \omterm{def:g11:fundamental-interactions:four}{आकर्षण-शक्ति} (\cref{ch:g10:universal-gravitation}) देता है
बल एक परीक्षण द्रव्यमान पर $ m $: परिमाण $ G M m / d^2$, केंद्र की ओर;
द्वारा विभाजित करें $ m $. सतह पर, $ d = R $ और यह है
$ g = GM/R^2 \approx \qty{9.81}{N/kg}$ पहले अध्याय का.
\end{proof}

\begin{example}[वह क्षेत्र जहाँ चंद्रमा रहता है]\label{ex:g11:electric-gravitational-fields:moon}
चंद्रमा की दूरी पर $ d = \qty{3.84e8}{m}$:
\[
g = \num{6.67e-11} \times \num{5.97e24} / (\num{3.84e8})^2
\approx \qty{2.7e-3}{N/kg}.
\]
पृथ्वी का \omterm{def:g11:electric-gravitational-fields:field}{मैदान} कभी नहीं रुकता --- केवल यही \omterm{def:g11:lenses-and-eye:lens}{पतला} के रूप में बाहर $1/d^2$; ये है
\omterm{def:g11:electric-gravitational-fields:field}{मैदान} जो चंद्रमा को अपने ऊपर धारण करता है \omterm{def:g10:universal-gravitation:satellite}{कक्षा}.
\end{example}

\begin{remark}[सतह के निकट वर्दी]\label{rem:g11:electric-gravitational-fields:uniformg}
किसी प्रयोगशाला या पहाड़ के ऊपर भी, $ d $ की तुलना में बमुश्किल बदलता है
$ R = \qty{6.37e6}{m}$, इसलिए $\vect g $ है \omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी} उच्च सटीकता के लिए:
ऊर्ध्वाधर रेखाएँ, स्थिर \qty{9.81}{N/kg} ---पृथ्वी का \omterm{def:g11:electric-gravitational-fields:field}{मैदान} दिखता है
की तरह संधारित्रठीक वैसे ही, जैसे बगीचे से देखने पर गोल पृथ्वी चपटी दिखती है।
\end{remark}

दो खेत इस अध्याय में दो वेशभूषाओं में एक ही गणित है:

\begin{tabular}{lcc}
 & \textbf{electric} & \textbf{gravitational} \\
\hline
source & charge $ Q $ & mass $ M $ \\

\omterm{def:g10:orders-of-magnitude:unit}{इकाई} & \unit{N/C} = \unit{V/m} & \unit{N/kg} \\

बायीं ओर का प्रत्येक सूत्र दायीं ओर के नीचे वाला बन जाता है
शब्दकोष $ q \leftrightarrow m $, $\vect E \leftrightarrow \vect g $, $ k \leftrightarrow G $.

\begin{remark}[जहां सादृश्य टूट जाता है]\label{rem:g11:electric-gravitational-fields:noanalogue}
चार्ज दो \omterm{def:g10:signals-and-waves:signal}{संकेत}ों में आता है; एक में द्रव्यमान. इसलिए \omterm{def:g11:electric-gravitational-fields:efield}{विद्युत क्षेत्र} हो सकता है
रद्द: धातु बॉक्स के पुनर्व्यवस्थित चार्ज किसी भी बाहरी को मिटा देते हैं
\omterm{def:g11:electric-gravitational-fields:field}{मैदान} इसके अंदर (ए \emph{फैराडे गुफ़ा} --- क्यों एक लिफ्ट आपकी जान ले लेती है
फ़ोन \omterm{def:g10:signals-and-waves:signal}{संकेत}), और थोक पदार्थ, संतुलित के साथ $+$ और $-$, है
विद्युतीय रूप से मौन. गुरुत्वाकर्षण को न तो जांचा जा सकता है और न ही निष्प्रभावी किया जा सकता है:
प्रत्येक \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} अपना खिंचाव जोड़ता है --- यही कारण है कि बल द्वारा कमजोर
$\num{e39}$ (\cref{exo:g11:electric-gravitational-fields:10}) चलाता है
बड़े पैमाने पर ब्रह्मांड.
\end{remark}

\section{समस्या: मिलिकन की बूंद}

\begin{problem}[{सप्ताहांत समस्या --- इलेक्ट्रॉन के चार्ज का वजन: एक
दो प्लेटों के बीच हवा में खड़ी तेल की बूंद उस आवेश को प्रकट करती है
के अविभाज्य चरणों में आता है $\num{1.60e-19}$ \omterm{def:g11:fundamental-interactions:charge}{कूलंब}}]
\label{pb:g11:electric-gravitational-fields:1}
1909 में रॉबर्ट मिलिकन ने क्षैतिज के बीच तेल की बूंदों का छिड़काव किया
ए की प्लेटें \omterm{def:g11:electric-gravitational-fields:capacitor}{संधारित्र}. ट्यूनिंग करके \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} एक चुनी हुई बूंद तक
गतिहीन लटका हुआ --- बिजली बल बिल्कुल संतुलन \omterm{def:g10:universal-gravitation:weight}{वज़न} --- वह
इसके चार्ज का वजन कर सकता है। हमारी बूंद में द्रव्यमान होता है $ m = \qty{4.90e-15}{kg}$
(इसकी धीमी गिरावट से मापा जाता है \omterm{def:g11:electric-gravitational-fields:field}{मैदान} बंद; हम इसे दिए गए अनुसार लेते हैं);
प्लेटें हैं $ d = \qty{8.0}{mm}$ अलग, \emph{अपर} एक
सकारात्मक; बूंद का आवेश ऋणात्मक है।

\medskip
\noindent\textbf{Part I --- The stage.}
\begin{enumerate}
  \item का वर्णन करें \omterm{def:g11:electric-gravitational-fields:field}{मैदान} प्लेटों के बीच; क्या करता है ``\omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी}'' खरीदना?
  \item के लिए $ U = \qty{800}{V}$, गणना करें $ E $.
  \item बिजली किस दिशा में है बल बूंद पर? क्यों
        किया \emph{अपर} प्लेट सकारात्मक होनी चाहिए?
  \item बूंदों की गणना करें \omterm{def:g10:universal-gravitation:weight}{वज़न}.
  \item आराम की स्थिति को दो की समानता के रूप में लिखें बल
        परिमाण.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The balance.}
\begin{enumerate}[resume]
  \item इससे, व्यक्त करें $\abs{q}$ के अनुसार $ m $, $ g $, $ d $ और $ U $.
  \item बूंद आराम की स्थिति में लटकी रहती है $ U = \qty{800}{V}$: गणना करें
        $\abs{q}$.
  \item द्वारा विभाजित करें $ e = \qty{1.60e-19}{C}$. कितने अतिरिक्त इलेक्ट्रॉन
        क्या बूंद ले जाती है?
  \item  \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} तक उकसाया जाता है \qty{900}{V}. किस दिशा में
        नेट करता है बल अब बिंदु?
  \item संतुलन दिखाओ \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} है $ U = mgd/\abs{q}$: द
        जितना बड़ा चार्ज, उतना ही छोटा \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The steps.}
विकिरण का एक विस्फोट इलेक्ट्रॉनों को बूंद पर या उससे बाहर गिरा सकता है; के बाद
प्रत्येक परिवर्तन, मिलिकन पुनः ट्यून करता है \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} बलेंस करने के लिए।
\begin{enumerate}[resume]
  \item एक विस्फोट के बाद संतुलन की आवश्यकता होती है $ U = \qty{1200}{V}$. गणना करें
        नया $\abs{q}$. क्या बूंद ने इलेक्ट्रॉन प्राप्त किया या खोया?
  \item उत्तरोत्तर शेष पाए जाते हैं $ U = 1200$, $800$, $600$,
        $480$ और \qty{400}{V}. पाँच आवेशों की गणना कीजिए।
  \item दिखाएँ कि सभी पाँच एक ही मात्रा के पूर्णांक गुणज हैं; देना
        इसका मूल्य.
  \item लगातार शेष के बीच, चार्ज कितना होता है
        परिवर्तन? व्याख्या करना।
  \item बताएं कि यह बूंद कभी आराम की स्थिति में क्यों नहीं लटक सकती
        $ U = \qty{700}{V}$, हालाँकि धैर्यपूर्वक कोई धुन बजाता है।
  \item मिलिकन ने इसे सैकड़ों बूंदों पर दोहराया: प्रत्येक को मापा गया
        आवेश समान मान का पूर्णांक गुणज था। बताएं
        निष्कर्ष, और मापी गई मात्रा का नाम बताएं।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Why there is no gravitational Millikan.}
\begin{enumerate}[resume]
  \item इस अध्याय के शब्दकोष में किसकी क्या भूमिका है $ q $, $ E $
        और $ qE $ भाग II के संतुलन में?
  \item क्या इसके स्थान पर कोई द्रव्यमान ओवरहेड बूंद को रोक सकता है? की गणना करें
        \omterm{def:g11:electric-gravitational-fields:field}{मैदान} एक का \qty{1000}{kg} सीसा क्षेत्र \qty{1.0}{m} के ऊपर
        छोटी बूंद, बल बूंद पर, और उससे तुलना करें \omterm{def:g10:universal-gravitation:weight}{वज़न}.
  \item संतुलन को लड़ने के लिए पर्याप्त रूप से ऊपर की ओर खींचने की आवश्यकता होती है
