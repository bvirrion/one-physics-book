\chapter{एक बल का कार्य}\label{ch:g11:work-of-force}

पूरे कमरे में एक अलमारी रख दें और आपका \omterm{def:g11:work-of-force:work}{काम} हो गया; ए के विरुद्ध धक्का
पूरी दोपहर दीवार पर रहें और, आपकी मांसपेशियां जो भी रिपोर्ट करती हैं, भौतिकी मायने रखती है
शून्य. \omterm{def:g11:work-of-force:work}{काम} क्या है? बल उद्धार \emph{एक गति के साथ} --- ऊर्जा
पारगमन में (\cref{ch:g10:energy-conservation}), में भुगतान किया गया \omterm{def:g11:work-of-force:work}{जूल}. यह
अध्याय की कीमतें बल का \cref{ch:g11:forces-and-motion} --- खींचता है
एक कोण पर, \omterm{def:g10:universal-gravitation:weight}{वज़न}, \omterm{def:g10:inertia:inventory}{टकराव} --- और भुगतान दर देखता है: \omterm{def:g11:work-of-force:power}{शक्ति}.

\section{निरंतर बल का कार्य}

\begin{definition}[निरंतर बल का कार्य]\label{def:g11:work-of-force:work}
एक स्थिरांक बल $\vect F $ सीधी दिशा में घूम रही किसी वस्तु पर लागू होता है
से खंड $ A $ को $ B $ करता है \emph{काम}\index{काम}
\[
W_{AB}(\vect F) = F \times AB \times \cos\theta ,
\]
कहाँ $ F $ का परिमाण है बल (में \unit{N}), $ AB $ लंबाई
विस्थापन का (में) \unit{m}) और $\theta $ के बीच का कोण
$\vect F $ और $\vect{AB}$. काम को मापा जाता है
\emph{जूल}\index{जूल} (\unit{J}): एक जूल एक का कार्य है
एक \omterm{def:g10:inertia:force}{न्यूटन} बल एक के साथ \omterm{def:g10:orders-of-magnitude:unit}{मीटर} अपनी ही दिशा का.
\end{definition}

\begin{remark}\label{rem:g11:work-of-force:scal}
$ F \times AB \times \cos\theta $ बिल्कुल अदिश गुणनफल है
$\vect F \cdot \vect{AB}$ इस वर्ष के गणित खंड का। केवल
घटक $ F\cos\theta $ \emph{साथ में} गति कुछ भी कमाती है.
\end{remark}

\begin{example}[पट्टे पर सूटकेस]\label{ex:g11:work-of-force:suitcase}
एक यात्री सूटकेस खींचता है \qty{300}{m} एक टर्मिनल के पार, पट्टा
साथ खींचा $ F = \qty{40}{N}$ पर $\theta = \qty{50}{\degree}$ को
मंजिल: $ W = 40 \times 300 \times \cos\qty{50}{\degree} \approx
\qty{7.7}{kJ}$. सपाट खींचा गया ($\theta = 0$) जो उसी बल होगा
पहुंचाना \qty{12}{kJ}: खिंचाव का एक तिहाई हिस्सा कुछ भी नहीं उठाता है।
\end{example}

\begin{definition}[मोटर, प्रतिरोधक, शून्य कार्य]\label{def:g11:work-of-force:sign}
का चिन्ह $\cos\theta $ प्रत्येक को क्रमबद्ध करता है बल तीन व्यवस्थाओं में:
\begin{itemize}
  \item $\theta < \qty{90}{\degree}$: $ W > 0$, द \omterm{def:g11:work-of-force:work}{काम} है
        \emph{मोटर}\index{मोटर} --- द बल फ़ीड ऊर्जा में
        गति;
  \item $\theta = \qty{90}{\degree}$: $ W = 0$, द बल \omterm{def:g11:work-of-force:work}{काम} करता है पर नहीं
        सभी, चाहे कितने भी बड़े हों;
  \item $\theta > \qty{90}{\degree}$: $ W < 0$, द \omterm{def:g11:work-of-force:work}{काम} है
        \emph{प्रतिरोधक}\index{प्रतिरोधक} --- द बल नालियों
        ऊर्जा. शुद्ध मामला एक ब्रेक है $\qty{180}{\degree}$:
        \qty{800}{N} ऊपर \qty{45}{m} लेता है $800 \times 45 \times (-1)
        = \qty{-36}{kJ}$ कार से बाहर, हॉट डिस्क में।
\end{itemize}
\end{definition}

\section{वज़न का काम}

\begin{proposition}[वज़न का काम]\label{prop:g11:work-of-force:weight}
जब कोई द्रव्यमान की वस्तु $ m $ से चलता है $ A $ (ऊंचाई $ z_A $) को $ B $
(ऊंचाई $ z_B $) साथ में \emph{कोई} सीधे खंडों से बना पथ, इसका
\omterm{def:g10:universal-gravitation:weight}{वज़न} $\vect P $ करता है \omterm{def:g11:work-of-force:work}{काम}
\[
W_{AB}(\vect P) = mg\,(z_A - z_B) = \pm\, mgh ,
\]
कहाँ $ h = \abs{z_A - z_B}$ ऊंचाई में गिरावट है: $+mgh $ नीचे जा रहा हूँ,
$-mgh $ ऊपर जाना --- केवल बूंद मायने रखती है, मार्ग नहीं।
\end{proposition}

\begin{proof}
एक सीधे खंड पर $ AB $, द \omterm{def:g10:universal-gravitation:weight}{वज़न} (परिमाण $ mg $, सीधा
नीचे) एक कोण बनाता है $\theta $ साथ $\vect{AB}$, और $ AB\cos\theta $ है
सटीक रूप से नीचे की ओर ऊर्ध्वाधर पर विस्थापन का प्रक्षेपण:
$ z_A - z_B $. इस तरह $ W = mg \times AB\cos\theta = mg\,(z_A - z_B)$.
खंडों की एक श्रृंखला के साथ \omterm{def:g11:work-of-force:work}{काम} करता है जोड़ें और बूँदें दूरबीन:
$ mg(z_A - z_C) + mg(z_C - z_B) = mg(z_A - z_B)$.
\end{proof}

\begin{proposition}[कोई भी रास्ता]\label{prop:g11:work-of-force:anypath}
यही सूत्र किसी भी घुमावदार पथ पर लागू होता है $ A $ को $ B $.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:work-of-force:anypath}
प्रशंसनीय --- एक वक्र उतना ही करीब होता है जितना हम कई की टूटी हुई रेखा के करीब होते हैं
छोटे खंड --- लेकिन ``हम जितना चाहें उतना करीब'' बनाना ईमानदार है
कैलकुलस, में किया गया \cref{ch:g12:work-and-energy}.
\end{remark}

\begin{example}[यात्री]\label{ex:g11:work-of-force:hiker}
ए \qty{78}{kg} यात्री (पैक शामिल) चढ़ता है \qty{450}{m}, जो भी हो
निशान: $ W(\vect P) = -78 \times 9.81 \times 450 \approx \qty{-344}{kJ}$,
और बिल्कुल $+\qty{344}{kJ}$ चाहे जो भी मोड़ हो, वापस आ जाओ।
\end{example}

\section{घर्षण का कार्य}

\begin{proposition}[फिसलने वाले घर्षण का कार्य]\label{prop:g11:work-of-force:friction}
ए \omterm{def:g10:inertia:inventory}{टकराव} बल स्थिर परिमाण का $ f $, हर क्षण विपरीत
गति, करता है \omterm{def:g11:work-of-force:work}{काम} $ W = -fL $ कुल लंबाई के पथ के साथ $ L $:
हमेशा \omterm{def:g11:work-of-force:sign}{प्रतिरोधक}, और के आनुपातिक \emph{पथ की लंबाई यात्रा की} ---विस्थापन को नहीं.
\end{proposition}

\begin{proof}
प्रत्येक छोटे सीधे टुकड़े में है बल पर $\qty{180}{\degree}$ को
गति, योगदान $-f \times (\text{its length})$; टुकड़े जोड़ते हैं
तक $-fL $. एक घुमावदार रास्ते को इसी तरह छोटे-छोटे टुकड़ों में काटा जाता है
सीमा के लिए के रूप में स्वीकार किया गया \omterm{def:g10:universal-gravitation:weight}{वज़न} (\cref{rem:g11:work-of-force:anypath}).
\end{proof}

\begin{remark}[वज़न के विरुद्ध घर्षण: महान विभाजन]\label{rem:g11:work-of-force:divide}
 \omterm{def:g10:universal-gravitation:weight}{वज़न}'एस \omterm{def:g11:work-of-force:work}{काम} रास्ता भूल जाता है; टकरावप्रति टोल है \omterm{def:g10:orders-of-magnitude:unit}{मीटर}. पर
एक दौर की यात्रा \omterm{def:g10:universal-gravitation:weight}{वज़न} नेट ज़ीरो --- यह जितना ऊपर चढ़ता है उतना वापस कर देता है
डाउनहिल --- जबकि \omterm{def:g10:inertia:inventory}{टकराव} प्रभार $-fL $ हर पैर पर और रिफंड
कुछ भी नहीं: \omterm{def:g10:universal-gravitation:weight}{वज़न}'एस \omterm{def:g11:work-of-force:work}{काम} संग्रहीत और पुनर्प्राप्त किया जा सकता है (संभावित)।
ऊर्जा, \cref{ch:g11:mechanical-energy}), टकरावकी गर्मी के रूप में खो जाता है.
\end{remark}

\begin{example}[वहाँ और वापस]\label{ex:g11:work-of-force:roundtrip}
एक टोकरा सरकता है \qty{5.0}{m} एक मंजिल के पार और \qty{5.0}{m} वापस,
विरुद्ध $ f = \qty{60}{N}$: \omterm{def:g10:inertia:inventory}{टकराव} करता है $-60 \times 10 =
\qty{-600}{J}$ भले ही टोकरा वहीं खत्म हो जाता है जहां से शुरू हुआ था। \omterm{def:g10:universal-gravitation:weight}{वज़न} और
सामान्य बल, गति के लंबवत: संपूर्ण शून्य।
\end{example}

\begin{remark}[वसंत के बारे में क्या?]\label{rem:g11:work-of-force:spring}
एक स्प्रिंग जितनी अधिक खींची जाती है उतनी ही अधिक खींचती है: इसकी बल नहीं है
स्थिर और \cref{def:g11:work-of-force:work} लागू नहीं होता.
इसका \omterm{def:g11:work-of-force:work}{काम} इसका अर्थ है खिंचाव को छोटे-छोटे चरणों में काटना और
सारांश --- कलन, दिया गया \cref{ch:g12:work-and-energy}.
\end{remark}

\section{शक्ति}

\begin{definition}[शक्ति]\label{def:g11:work-of-force:power}
 \emph{शक्ति}\index{शक्ति} एक का बल दर पर है
जो यह \omterm{def:g11:work-of-force:work}{काम} करता है, $ P = W/\Delta t $, में \emph{वाट}\index{वाट}
($\unit{W} = \unit{J/s}$) --- के समान ही वाट
\cref{ch:g11:circuits-and-power}. एक पूर्व एसआई \omterm{def:g10:orders-of-magnitude:unit}{इकाई} इंजन पर जीवित रहता है
बैज: \emph{घोड़े की शक्ति}\index{घोड़े की शक्ति},
$\qty{1}{hp} = \qty{736}{W}$.
\end{definition}

\begin{proposition}[निरंतर वेग पर शक्ति]\label{prop:g11:work-of-force:pfv}
यदि वस्तु स्थिर गति से चलती है $ v $, एक स्थिरांक बल $\vect F $
कोण पर $\theta $ गति को प्रदान करता है \omterm{def:g11:work-of-force:power}{शक्ति}
$ P = F v \cos\theta $ --- विशेष रूप से $ P = Fv $ एक के लिए बल साथ में
गति.
\end{proposition}

\begin{proof}
एक समय में $\Delta t $ वस्तु कवर करती है $ d = v\,\Delta t $, इसलिए
$ W = F \times v\,\Delta t \times \cos\theta $; द्वारा विभाजित करें $\Delta t $.
\end{proof}

\begin{example}[क्रेन]\label{ex:g11:work-of-force:crane}
एक क्रेन फहराती है \qty{600}{kg} एक स्थिर स्थिति में फूस \qty{0.90}{m/s}: द
केबल का बल के बराबर है \omterm{def:g10:universal-gravitation:weight}{वज़न}, $ mg = \qty{5.89e3}{N}$, इसलिए
$ P = Fv = 5890 \times 0.90 \approx \qty{5.3}{kW} \approx \qty{7.2}{hp}$
--- स्टील के डिब्बे में सात घोड़ों की टीम।
\end{example}

\begin{method}[कम बल के साथ चढ़ना: ज़िगज़ैग]\label{met:g11:work-of-force:zigzag}
किसी भार को ऊँचाई तक उठाना $ h $ एक सीमित के साथ बल:
\begin{enumerate}
  \item बिल तय है: द्वारा \cref{prop:g11:work-of-force:weight},
        शीर्ष तक किसी भी मार्ग की लागत होती है \omterm{def:g11:work-of-force:work}{काम} $ mgh $.
  \item पथ बढ़ाएँ: लंबाई की ढलान पर $ L $  बल जरुरत
        (निरंतर गति, \omterm{def:g10:inertia:inventory}{टकराव} एक तरफ) है $ F = mgh/L $ --- दोगुना करें
        लंबाई, आधी बल.
  \item अधिभार का भुगतान करें: \omterm{def:g10:inertia:inventory}{टकराव} द्वारा आरोप \omterm{def:g10:orders-of-magnitude:unit}{मीटर}
        (\cref{prop:g11:work-of-force:friction}), जोड़ना $ fL $.
\end{enumerate}
चरण 2 के लिए पर्वतीय सड़कें टेढ़ी-मेढ़ी: समान ऊँचाई, कम बल, वही \omterm{def:g11:work-of-force:work}{काम}.
\end{method}

\section{समस्या: प्रस्तावक का दिन}

\begin{problem}[{सप्ताहांत समस्या --- भौतिकी जूल द्वारा भुगतान करती है: ए
फिसलन, उत्थापन और हेयरपिन के झुकने का दिन, जो नम्रता के साथ समाप्त होता है
एक केतली ईमानदार श्रम के बारे में क्या सोचती है इसकी खोज}]
\label{pb:g11:work-of-force:1}
एक मूवर का दल एक गोदाम को खाली कर देता है, सामान को तीसरी मंजिल तक ले जाता है
समतल, फिर वैन को एक पहाड़ी की चोटी पर स्थित गाँव की ओर ले जाता है। हर कार्य की कीमत होती है
में \omterm{def:g11:work-of-force:work}{जूल}; रसोई की केतली दिन का ऑडिट करेगी। बक्से: \qty{30}{kg}.

\medskip
\noindent\textbf{Part I --- Across the warehouse floor.}
एक बॉक्स को क्षैतिज रूप से धकेला जाता है बल $ F = \qty{110}{N}$ ऊपर
$ d = \qty{8.0}{m}$; रपट \omterm{def:g10:inertia:inventory}{टकराव} है $ f = \qty{95}{N}$.
\begin{enumerate}
  \item की गणना करें \omterm{def:g11:work-of-force:work}{काम} धक्का का.
  \item की गणना करें \omterm{def:g11:work-of-force:work}{काम} का \omterm{def:g10:inertia:inventory}{टकराव}, की \omterm{def:g10:universal-gravitation:weight}{वज़न}, और सामान्य का
        बल (शून्य का औचित्य सिद्ध करें)।
  \item कुल \omterm{def:g11:work-of-force:work}{काम} बॉक्स पर, और उसके चिन्ह का अर्थ (बॉक्स शुरू हुआ)।
        आराम पर)? कौन सा धक्का गति को स्थिर रखता है
        (\cref{ch:g11:forces-and-motion})?
  \item दूर \omterm{def:g11:color-light-sources:cones}{कोन}े तक पहुंचा जा सकता है \qty{10.0}{m} विकर्ण
        या दो दीवारों के साथ (\qty{6.0}{m} तब \qty{8.0}{m}). \omterm{def:g10:inertia:inventory}{टकराव}
        \omterm{def:g11:work-of-force:work}{काम} प्रत्येक मार्ग पर?
  \item एक विकर्ण राउंड ट्रिप (बाहर और पीछे) पर, गणना करें \omterm{def:g11:work-of-force:work}{काम} का
        \omterm{def:g10:inertia:inventory}{टकराव} और यह \omterm{def:g11:work-of-force:work}{काम} की \omterm{def:g10:universal-gravitation:weight}{वज़न}. कौन बल रिफंड?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Up to the third floor ($ h = \qty{9.0}{m}$).}
\begin{enumerate}[resume]
  \item की गणना करें \omterm{def:g11:work-of-force:work}{काम} की \omterm{def:g10:universal-gravitation:weight}{वज़न} फ्लैट की ओर उठाए गए एक बक्से पर।
        क्या \omterm{def:g11:work-of-force:work}{काम} क्या क्रू को कम से कम आपूर्ति करनी चाहिए?
  \item ए द्वारा सीढ़ियों से ऊपर ले जाया गया \qty{70}{kg} मूवर, गुरुत्वाकर्षण शुल्क
        प्रस्तावक के लिए \emph{और} बॉक्स: कुल \omterm{def:g11:work-of-force:work}{काम} गुरुत्वाकर्षण के विरुद्ध, और
        अंश वास्तव में बॉक्स पर खर्च किया गया?
  \item एक ट्रॉली रैंप है \qty{30}{m} उसी के लिए लंबे समय तक \qty{9.0}{m}
        उठो, साथ \qty{40}{N} लुढ़कने का \omterm{def:g10:inertia:inventory}{टकराव}. निरंतर गति से,
        रैम्प और उसके साथ आवश्यक धक्का की गणना करें \omterm{def:g11:work-of-force:work}{काम}.
  \item रैंप की तुलना सीधे ऊर्ध्वाधर \omterm{def:g10:signals-and-waves:wave}{लहर}ा से करें: बल आवश्यकता है,
        \omterm{def:g11:work-of-force:work}{काम} हो गया। रैम्प क्या खरीदता है, और किस कीमत पर?
  \item एक रस्सी और चरखी \omterm{def:g10:signals-and-waves:wave}{लहर}ा बॉक्स को ऊपर उठाती है \qty{0.50}{m/s}.
        \omterm{def:g10:signals-and-waves:period}{अवधि} और यांत्रिक \omterm{def:g11:work-of-force:power}{शक्ति}?
  \item उसे पुनर्प्राप्त करें \omterm{def:g11:work-of-force:power}{शक्ति} जैसा $ P = Fv $. \omterm{def:g10:signals-and-waves:wave}{लहर}ा का \omterm{def:g11:work-of-force:sign}{मोटर} है $60\%$
        कुशल: \omterm{prop:g11:circuits-and-power:power}{विद्युत शक्ति} अनिर्णित?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The zigzag road.}
भरी हुई वैन (\qty{3500}{kg}) एक गाँव तक पहुँचना चाहिए \qty{240}{m}
घाटी के ऊपर: सीधी लेन, \qty{1.0}{km} पर $24\%$ ग्रेड, या
पक्की सड़क, \qty{8.0}{km} पर $3.0\%$ ग्रेड (ग्रेड $= \sin\alpha $).
\begin{enumerate}[resume]
  \item जांचें कि दोनों मार्ग एक ही चढ़ाई पर हैं \qty{240}{m}.
  \item निरंतर गति और उपेक्षा से \omterm{def:g10:inertia:inventory}{टकराव}, ड्राइविंग की गणना करें
        बल प्रत्येक मार्ग पर आवश्यक है।
  \item \omterm{def:g11:work-of-force:work}{काम} ड्राइविंग का बल प्रत्येक मार्ग पर? साथ तुलना करें $ mgh $.
  \item रोलिंग \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} है \qty{400}{N} दोनों पर: अतिरिक्त की गणना करें
        \omterm{def:g11:work-of-force:work}{काम} प्रत्येक मार्ग पर. ज़िगज़ैग के आराम की कीमत क्या है?
  \item पर \qty{36}{km/h}, इंजन की गणना करें \omterm{def:g11:work-of-force:power}{शक्ति} प्रत्येक पर जरूरत है
        मार्ग (में \unit{kW} और \unit{hp}). सड़कें टेढ़ी-मेढ़ी क्यों हो जाती हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The day's ledger.}
\begin{enumerate}[resume]
  \item दल स्लाइड करता है $60$ फर्श के पार बक्से (का धक्का)।
        प्रश्न 1) और सभी को फहराता है $60$ ऊपर \qty{9.0}{m}
        (प्रश्न 6). पूर्ण यांत्रिक \omterm{def:g11:work-of-force:work}{काम} बक्सों तक पहुँचाया गया?
  \item सीढ़ी वाहक का अपना शरीर जोड़ें: $20$ पर यात्राएँ \qty{70}{kg}
        (डाउनहिल रिफंड गर्म घुटनों तक जाता है, बहीखाता तक नहीं)। दिन का
        कुल?
  \item दल ने \omterm{def:g11:work-of-force:work}{काम} किया $8.0$ घंटे: औसत यांत्रिक \omterm{def:g11:work-of-force:power}{शक्ति}? कब तक
        होगा \qty{2.0}{kW} केतली वही खर्च करने के लिए ले लो ऊर्जा?
  \item समापन: दो वाक्यों में, क्या कहें? \omterm{def:g10:energy-conservation:energy}{जौल} --- और के लिए भुगतान करता है

        गतिहीन) --- और क्यों मूवर्स के पास चरखी, रैंप और ज़िगज़ैग होते हैं
        बड़ी मांसपेशियों के बजाय सड़कें।
