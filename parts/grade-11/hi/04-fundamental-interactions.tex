\chapter{मौलिक बातचीत}\label{ch:g11:fundamental-interactions}

अपने बालों पर एक गुब्बारा रगड़ें और यह कागज के टुकड़े उठाता है --- पिटाई,
रबर के कुछ वर्ग सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} के साथ, का गुरुत्वाकर्षण खिंचाव
संपूर्ण ग्रह. वह आसान जीत इसीलिए है परमाणुओं एक साथ रहो और क्यों
भारी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} अंततः टूट जाते हैं (\cref{ch:g11:nucleus-radioactivity}).
ब्रह्माण्ड ठीक चार अंतःक्रियाओं पर चलता है; यह अध्याय मिलता है
उन्हें, एक-दूसरे के विरुद्ध तौलता है, और प्रत्येक को उसकी मंजिल निर्धारित करता है।

\section{क्वार्क से लेकर आकाशगंगाओं तक का पदार्थ}

\begin{definition}[संरचना सीढ़ी]\label{def:g11:fundamental-interactions:ladder}
पदार्थ फर्शों में निर्मित होता है, प्रत्येक को नीचे दिए गए एक से इकट्ठा किया जाता है: तीन
\emph{क्वार्क}\index{क्वार्क} (बिंदु जैसा, उससे छोटा \qty{e-18}{m}) बाँधना
एक में \emph{न्यूक्लियॉन}\index{न्यूक्लियॉन} --- प्रोटॉन या न्यूट्रॉन, के बारे में
\qty{e-15}{m} आर-पार; न्यूक्लियंस एक में पैक होते हैं \emph{नाभिक}\index{नाभिक}
(\qty{e-15}{m} को \qty{e-14}{m}); एक नाभिक और उसका इलेक्ट्रॉन बादल है
एक \emph{एटम}\index{एटम} (\qty{e-10}{m}); परमाणु आपस में जुड़ते हैं
\emph{अणुओं}\index{अणुओं} और क्रिस्टल (\qty{e-9}{m} ऊपर), जो
रोजमर्रा की वस्तुओं में इकट्ठा करें (\qty{1}{m}), ग्रह (\qty{e7}{m}),
ग्रहीय प्रणालियाँ (\qty{e11}{m}), आकाशगंगाएँ (\qty{e21}{m}) और
देखने योग्य ब्रह्मांड (\qty{e26}{m}).
\end{definition}

\begin{definition}[चार इंटरैक्शन]\label{def:g11:fundamental-interactions:four}
ए \emph{मौलिक बातचीत}\index{मौलिक बातचीत} एक है
बल दूसरे के लिए कम करने योग्य नहीं बल; बिल्कुल चार हैं:
\emph{आकर्षण-शक्ति}\index{आकर्षण-शक्ति}, द \emph{विद्युत चुम्बकीय संपर्क}\index{विद्युत चुम्बकीय संपर्क},
द \omterm{def:g11:fundamental-interactions:strong}{मजबूत अंतःक्रिया} और यह \omterm{def:g11:fundamental-interactions:weak}{कमजोर अंतःक्रिया}.
\end{definition}

\begin{remark}\label{rem:g11:fundamental-interactions:empty}
फर्शों को रिक्तियों द्वारा अलग किया जाता है: a \omterm{def:g11:fundamental-interactions:ladder}{एटम} है $10^{5}$ से कई गुना अधिक व्यापक
इसका \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} --- स्केल करें \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} एक तक \qty{1}{cm} संगमरमर और
इलेक्ट्रॉन बादल एक किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} के दायरे में है। कुछ तो इन विरलों को बनाए रखना चाहिए
फर्श एक साथ हैं, और गुरुत्वाकर्षण लगभग कभी नहीं है।
\end{remark}

\section{बिजली का आवेश}

\begin{definition}[बिजली का आवेश]\label{def:g11:fundamental-interactions:charge}
\emph{बिजली का आवेश}\index{बिजली का आवेश} पदार्थ का गुण है
जिस पर \omterm{def:g11:fundamental-interactions:four}{विद्युत चुम्बकीय संपर्क} कार्य करता है, क्योंकि द्रव्यमान गुण है
\omterm{def:g11:fundamental-interactions:four}{आकर्षण-शक्ति} पर कार्य करता है. इसमें मापा जाता है \emph{कूलंब}\index{कूलंब}
(\unit{C}) और अंदर आता है \emph{दो लक्षण}, सकारात्मक और नकारात्मक, जो
जोड़े जाने पर रद्द करें; जैसे आवेश प्रतिकर्षित करते हैं, वैसे आवेश आकर्षित करते हैं।
\end{definition}

\begin{definition}[प्राथमिक प्रभार]\label{def:g11:fundamental-interactions:elementary}
आवेश कणित होता है: प्रत्येक अवलोकित आवेश इसका पूर्ण गुणज होता है
\emph{प्राथमिक प्रभार}\index{प्राथमिक प्रभार}
$ e = \qty{1.60e-19}{C}$. प्रोटॉन वहन करता है $+e $, इलेक्ट्रॉन $-e $, द
न्यूट्रॉन $0$; एक आरोप $ q $ रखती है $ N = \abs{q}/e $ प्राथमिक आरोप.
\end{definition}

\begin{proposition}[आवेश का संरक्षण]\label{prop:g11:fundamental-interactions:conservation}
किसी पृथक प्रणाली का कुल आवेश कभी नहीं बदलता: आवेश न तो होता है
न तो बनाया गया और न ही नष्ट किया गया, केवल स्थानांतरित किया गया --- व्यवहार में इलेक्ट्रॉनों द्वारा,
सबसे हल्के वाहक.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:conservation}
कभी कोई उल्लंघन नहीं देखा गया; गहरा कारण, की एक समरूपता
विद्युत चुम्बकत्व, विश्वविद्यालय के संस्करणों में दिया गया है।
\end{remark}

\begin{definition}[घर्षण और संपर्क द्वारा चार्ज करना]\label{def:g11:fundamental-interactions:charging}
दो इंसुलेटर को रगड़ने से एक से इलेक्ट्रॉन टूटकर दूसरे से अलग हो जाते हैं,
समान परिमाण के विपरीत आवेशों को छोड़ना (\emph{द्वारा चार्ज करना घर्षण}\index{द्वारा चार्ज करना घर्षण}); किसी आवेशित वस्तु को स्पर्श करना
तटस्थ व्यक्ति चार्ज साझा करता है (\emph{द्वारा चार्ज करना संपर्क करें}\index{द्वारा चार्ज करना संपर्क करें}). केवल इलेक्ट्रॉन चलते हैं; कुल
चार्ज संरक्षित है.
\end{definition}

\begin{example}[इलेक्ट्रॉनों की गिनती]\label{ex:g11:fundamental-interactions:balloon}
बालों पर गुब्बारा रगड़ने से लाभ होता है $ q = \qty{-1.6e-8}{C}$: यह है
\emph{प्राप्त की} $ N = \num{1.6e-8} / \num{1.6e-19} = \num{1.0e11}$
इलेक्ट्रॉन, और बाल बिल्कुल बचे रह जाते हैं $+\qty{1.6e-8}{C}$.
\end{example}

\section{कूलम्ब का नियम}

\begin{theorem}[कूलम्ब का नियम]\label{thm:g11:fundamental-interactions:coulomb}
दो बिंदु शुल्क $ q_1$ और $ q_2$ एक दूरी $ d $ प्रत्येक पर अलग से प्रयास करें
अन्य बल उन्हें जोड़ने वाली रेखा के अनुदिश, समान परिमाण का और
विपरीत दिशाएँ --- समान चिह्नों के लिए प्रतिकारक, विपरीत चिह्नों के लिए आकर्षक
\omterm{def:g10:signals-and-waves:signal}{संकेत} --- के साथ
\[
F = k\,\frac{\abs{q_1 q_2}}{d^{2}},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{theorem}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:torsion}
कूलम्ब ने 1785 में एक के मोड़ को मापकर कानून की स्थापना की \omterm{def:g11:lenses-and-eye:lens}{पतला}
मरोड़ फाइबर; क्यों \omterm{def:g10:orders-of-magnitude:scinot}{प्रतिपादक} बिलकुल है $2$ इसका उत्तर विश्वविद्यालय के संस्करणों में दिया गया है।
\end{remark}

\begin{remark}[गुरुत्वाकर्षण का एक औपचारिक जुड़वां]\label{rem:g11:fundamental-interactions:twin}
कूलम्ब का नियम, प्रतीक के लिए प्रतीक, सार्वभौमिक का नियम है \omterm{def:g11:fundamental-interactions:four}{आकर्षण-शक्ति}
(\cref{ch:g10:universal-gravitation}) द्रव्यमान के स्थान पर आवेशों के साथ:
\begin{center}
\begin{tabular}{lll}
source & mass $ m $ & charge $ q $ \\
law & $ F = G\,\dfrac{m_1 m_2}{d^2}$ & $ F = k\,\dfrac{\abs{q_1 q_2}}{d^2}$ \\[1.5ex]
constant & $ G = \qty{6.67e-11}{N.m^2/kg^2}$ & $ k = \qty{8.99e9}{N.m^2/C^2}$ \\
sign & always attractive & attractive or repulsive \\
\end{tabular}
\end{center}
नीचे दी गई हर चीज़ को दो अंतर संचालित करते हैं: बिजली है
अत्यधिक मजबूत, और यह अकेले ही स्वयं को रद्द कर सकता है।
\end{remark}

\begin{example}[दो प्रोटॉन, दोनों बल एक साथ]\label{ex:g11:fundamental-interactions:protons}
दो प्रोटोन ($ m_p = \qty{1.67e-27}{kg}$) बैठना $ d = \qty{1.0e-15}{m}$
अलग --- परमाणु पड़ोसी। विद्युत प्रतिकर्षण:
\[
F_E = \num{8.99e9} \times
\frac{(\num{1.60e-19})^{2}}{(\num{1.0e-15})^{2}}
\approx \qty{2.3e2}{N}
\]
--- द \omterm{def:g10:universal-gravitation:weight}{वज़न} एक का $23$-किग्रा सूटकेस, का एक कण द्वारा ले जाया गया
$10^{-27}$ \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम}. गुरुत्वाकर्षण आकर्षण:
$ F_G = \num{6.67e-11} \times (\num{1.67e-27})^{2} /
(\num{1.0e-15})^{2} \approx \qty{1.9e-34}{N}$. अनुपात
$ F_E / F_G = k e^{2} / (G m_p^{2}) \approx \num{1.2e36}$ है
दूरी-स्वतंत्र ($ d^{2}$ रद्द): प्राथमिक कणों के बीच,
गुरुत्वाकर्षण है $10^{36}$ कई बार बात करने के लिए बहुत कमजोर --- पर \emph{कोई} दूरी।
\end{example}

\section{मजबूत और कमजोर इंटरैक्शन}

उदाहरण एक घोटाला छोड़ता है: एक हीलियम \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} इसमें दो प्रोटॉन होते हैं
दसियों से प्रतिकर्षित करें \omterm{def:g10:inertia:force}{न्यूटन} --- और फिर भी हीलियम मौजूद है।

\begin{definition}[मजबूत अंतःक्रिया]\label{def:g11:fundamental-interactions:strong}
 \emph{मजबूत अंतःक्रिया}\index{मजबूत अंतःक्रिया} बांध \omterm{def:g11:fundamental-interactions:ladder}{क्वार्क}
में न्युक्लियोन और न्युक्लियोन \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} में. पर \qty{e-15}{m} यह है
विद्युत प्रतिकर्षण (हजारों) की तुलना में आकर्षक और कहीं अधिक मजबूत
\omterm{def:g10:inertia:force}{न्यूटन} बीच में न्युक्लियोन); कुछ से परे $10^{-15}$ एम-eters यह गायब हो जाता है:
इसका \emph{श्रेणी}\index{श्रेणी} के बारे में है
\qty{e-15}{m}. यह प्रोटॉन और न्यूट्रॉन को समान रूप से पकड़ता है और चार्ज को अनदेखा करता है।
\end{definition}

\begin{remark}[नाभिक क्यों मौजूद होते हैं --- और भारी नाभिक रास्ता क्यों छोड़ देते हैं]\label{rem:g11:fundamental-interactions:nuclei}

लघु-श्रेणी: ए \omterm{def:g11:fundamental-interactions:ladder}{न्यूक्लियॉन} केवल अपने कुछ पड़ोसियों को ही भीतर बांधता है
\qty{e-15}{m}. विद्युत प्रतिकर्षण लंबी दूरी का होता है: प्रत्येक प्रोटॉन
आगे बढ़ाता है \emph{हर दूसरे} प्रोटोन. जैसे-जैसे \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} बढ़ता है, प्रतिकर्षण
गोंद की तुलना में तेजी से जमा होता है: भारी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} को अतिरिक्त न्यूट्रॉन (गोंद) की आवश्यकता होती है
बिना किसी शुल्क के), और उससे भी आगे $80$ प्रोटॉन यहां तक कि विफल रहता है ---
सबसे भारी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} उधार के समय पर जीवित रहते हैं (\cref{ch:g11:nucleus-radioactivity}).
\end{remark}

\begin{definition}[कमज़ोर अंतःक्रिया]\label{def:g11:fundamental-interactions:weak}
 \emph{कमजोर अंतःक्रिया}\index{कमजोर अंतःक्रिया}, का \omterm{def:g11:fundamental-interactions:strong}{श्रेणी} छोटा
फिर भी, एक न्यूट्रॉन को एक प्रोटॉन (और वापस) में बदल देता है: यह ड्राइव करता है
$\beta $ की रेडियोधर्मिता \cref{ch:g11:nucleus-radioactivity}.
\end{definition}

\section{किस मंजिल पर कौन शासन करता है}

\begin{method}[एक पैमाने का ऑडिट करना]\label{met:g11:fundamental-interactions:audit}
यह पता लगाने के लिए कि कौन सा इंटरैक्शन आकार की संरचना को नियंत्रित करता है $ d $, जाँच करना \omterm{def:g11:fundamental-interactions:strong}{श्रेणी}
और रद्दीकरण:
\begin{enumerate}
  \item $ d \leq \qty{e-14}{m}$: द \emph{\omterm{def:g11:fundamental-interactions:strong}{मजबूत अंतःक्रिया}} नियम ---
        यह विद्युत प्रतिकर्षण और गुरुत्वाकर्षण को मात देता है $10^{36}$ बाहर।
  \item परमाणुओं पहाड़ों तक ($\qty{e-10}{m}$ को $\qty{e4}{m}$): मजबूत बल से बाहर है \omterm{def:g11:fundamental-interactions:strong}{श्रेणी}; \emph{विद्युत चुम्बकीय बातचीत} नियम --- बंधन, सामंजस्य, \omterm{def:g10:inertia:inventory}{टकराव}, संपर्क करना।
  \item ग्रह और उससे आगे ($ d \geq \qty{e6}{m}$:) पदार्थ तटस्थ है
        शानदार परिशुद्धता, इसलिए बिजली स्वयं को रद्द कर देती है; द्रव्यमान है
        एक चिह्न और हमेशा जोड़ता है --- \emph{\omterm{def:g11:fundamental-interactions:four}{आकर्षण-शक्ति}} आकाश पर राज करता है.
\end{enumerate}
 \omterm{def:g11:fundamental-interactions:weak}{कमजोर अंतःक्रिया} कोई आधार नहीं रखता: यह किसी को बांधता नहीं, बल्कि मध्यस्थता करता है
परिवर्तन ($\beta $ क्षय)।
\end{method}

\begin{exercise}[$\star\star\star $]\label{exo:g11:fundamental-interactions:12}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g11:fundamental-interactions:13}
द्रव्यमान की दो गेंदें \qty{1.0}{g} से लटका \qty{50}{cm} धागे बंधे हुए हैं

अलग। प्रत्येक गेंद अपने नीचे संतुलन में है \omterm{def:g10:universal-gravitation:weight}{वज़न}, धागे का
\omterm{def:g10:inertia:inventory}{तनाव} और बिजली बल: बताते हैं कि $ F = mg\tan\theta $ ($\theta $:
धागे का ऊर्ध्वाधर से कोण) और वह यहां $\tan\theta \approx 0.060$.
गणना करना $ F $, तब $ q $, फिर की संख्या प्राथमिक आरोप ले जाया गया.
\end{exercise}

\section{समस्या: चारों सेनाओं का ऑडिट}

\begin{problem}[{सप्ताहांत समस्या---दुनिया का पतन क्यों नहीं होता
न ही अलग-अलग उड़ता है: चार अंतःक्रियाओं का फर्श दर फर्श, नीचे तक ऑडिट किया जाता है
पदार्थ की तटस्थता दो भागों में $10^{18}$}]
\label{pb:g11:fundamental-interactions:1}
परमाणुओं विस्फोट न करें, \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} अधिकांशतः स्थिर रहें, चंद्रमा न तो दुर्घटनाग्रस्त होता है और न ही
भाग जाता है; यह समस्या ऑडिट करती है कि श्रेय का हकदार कौन है। डेटा:
$ e = \qty{1.60e-19}{C}$,
$ k = \qty{8.99e9}{N.m^2/C^2}$, $ G = \qty{6.67e-11}{N.m^2/kg^2}$,
$ m_p = \qty{1.67e-27}{kg}$, $ m_e = \qty{9.11e-31}{kg}$,
$ M_E = \qty{5.97e24}{kg}$, $ M_M = \qty{7.35e22}{kg}$, पृथ्वी - चंद्रमा
दूरी \qty{3.84e8}{m}.

\medskip
\noindent\textbf{Part I --- The ladder.}
\begin{enumerate}
  \item सीढ़ी के फर्शों की सूची बनाएं \omterm{def:g11:fundamental-interactions:ladder}{क्वार्क} आकाशगंगा के लिए, एक \omterm{def:g10:orders-of-magnitude:oom}{आदेश का आकार} प्रत्येक आकार का.
  \item स्केल मॉडल: \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} एक बन जाता है \qty{1}{cm} संगमरमर; कितना चौड़ा
        है \omterm{def:g11:fundamental-interactions:ladder}{एटम}?
  \item एक का कौन सा अंश एटमका वॉल्यूम अपना काम करता है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} कब्ज़ा?
  \item एक बाल है \qty{70}{\micro m} मोटा। कितने परमाणुओं इसे फैलाओ?
  \item कितने पॉवर्स दस में से \omterm{def:g11:fundamental-interactions:ladder}{क्वार्क} छत (\qty{e-18}{m})
        एक आकाशगंगा के लिए (\qty{e21}{m})?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The electric floor.}
\begin{enumerate}[resume]
  \item एक हाइड्रोजन में \omterm{def:g11:fundamental-interactions:ladder}{एटम} इलेक्ट्रॉन बैठता है $ d = \qty{5.3e-11}{m}$
        प्रोटॉन से: विद्युत की गणना करें बल उन दोनों के बीच।
  \item की गणना करें \omterm{def:g10:universal-gravitation:force}{गुरुत्वाकर्षण बल} उनके बीच, और का अनुपात
        दो. कौन सा धारण करता है \omterm{def:g11:fundamental-interactions:ladder}{एटम} एक साथ?
  \item अगर अंदर गुरुत्वाकर्षण बंद कर दिया जाए तो क्या बदलाव आएगा? परमाणुओं? और यदि
        बिजली है?
  \item ``संपर्क'' क्यों हैं बल रोजमर्रा की जिंदगी की --- मंजिल
        अपने पैरों पर \omterm{def:g10:inertia:inventory}{जोर} देकर, \omterm{def:g10:inertia:inventory}{टकराव} --- भेष में विद्युत चुम्बकीय?
  \item प्रत्येक दो सिक्के लगभग धारण करते हैं $\num{e23}$ इलेक्ट्रॉन. स्थानांतरण
        एक में एक इलेक्ट्रॉन \emph{अरब} उनके बीच, उन्हें सेट करें
        \qty{1.0}{m} अलग: आरोपों की गणना करें और बल. निष्कर्ष?
  \item कौन सी अंतःक्रिया से मंजिलों पर शासन होता है \qty{e-10}{m} को
        \qty{e4}{m}, और क्यों करता है \omterm{def:g11:fundamental-interactions:strong}{मजबूत अंतःक्रिया} प्रतिस्पर्धा नहीं?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The nuclear floor.}
\begin{enumerate}[resume]
  \item दो प्रोटोन बैठते हैं \qty{1.0e-15}{m} एक में अलग \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}: गणना करें
        उनका विद्युत प्रतिकर्षण.
  \item बेलगाम, ये कैसी तेजी होगी बल एक प्रोटॉन दें?
        साथ तुलना करें $ g $.
  \item तीन संपत्तियों की सूची बनाएं \omterm{def:g11:fundamental-interactions:strong}{मजबूत अंतःक्रिया} की जरूरत है
        बताएं कि \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} क्यों मौजूद हैं, न्यूट्रॉन भी क्यों जकड़े हुए हैं, और क्यों
        \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} छोटे होते हैं।
  \item यूरेनियम में \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} (व्यास \qty{1.4e-14}{m}, $92$ प्रोटोन),
        विपरीत छोर पर स्थित दो प्रोटॉनों के बीच प्रतिकर्षण की गणना करें। हो सकता है \omterm{def:g11:fundamental-interactions:strong}{मजबूत अंतःक्रिया} उस जोड़े को सीधे बांधें? स्पष्ट करें क्यों
        जैसे-जैसे \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} बढ़ता है, प्रतिकर्षण जीतता है, और न्यूट्रॉन किस लिए होते हैं।
  \item एक वाक्य में, जैसा कि वादा किया गया था: क्या करता है \omterm{def:g11:fundamental-interactions:weak}{कमजोर अंतःक्रिया} करना?
\end{enumerate}

\noindent\textbf{Part IV --- The astronomical floor, and the verdict.}
\begin{enumerate}[resume]
  \item की गणना करें \omterm{def:g10:universal-gravitation:force}{गुरुत्वाकर्षण बल} पृथ्वी और चंद्रमा के बीच.
  \item पृथ्वी में लगभग शामिल है $\num{1.8e51}$ प्रोटॉन (और उतने ही)।
        इलेक्ट्रॉन), चंद्रमा के बारे में $\num{2.2e49}$. कौन सा अंश $ f$ का
        प्रत्येक पिंड का प्रोटॉन चार्ज, बिना रद्द किए छोड़ दिया जाएगा
        बिजली बल गुरुत्वाकर्षण से मेल खायें?
  \item क्यों करता है \omterm{def:g11:fundamental-interactions:four}{आकर्षण-शक्ति}, प्रश्न 7 का कमज़ोर नियम
        खगोल विज्ञान? दो कारण।
  \item फैसला --- प्रति मंजिल एक वाक्य (\omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, \omterm{def:g11:fundamental-interactions:ladder}{एटम} पहाड़ की ओर,

        दुनिया न तो ढहती है और न ही बिखरती है?
\end{enumerate}
