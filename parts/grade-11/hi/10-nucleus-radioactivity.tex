\chapter{नाभिक और रेडियोधर्मिता}\label{ch:g11:nucleus-radioactivity}

अभी, आपके शरीर के अंदर लगभग आठ हजार परमाणु \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} फटते हैं
\omterm{def:g10:orders-of-magnitude:unit}{दूसरा}. कुछ भी उन पर हमला नहीं करता: कुछ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} अस्थिर पैदा होते हैं, और प्रत्येक
देर-सबेर अपने आप को रूपांतरित कर लेता है, एक अच्छे अंश पर एक टुकड़ा फेंक देता है
प्रकाश की गति का. यह अध्याय वापस आता है $10^{-15}$-मीटर का फर्श
\cref{ch:g11:fundamental-interactions} यह देखने के लिए कि कौन से \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} नाजुक हैं, कैसे
वे टूट जाते हैं, और कैसे उनकी नियमित अधीरता गुफाओं और छतों की रखवाली करती है।

\section{नाभिक, न्यूक्लियॉन और आइसोटोप}

\begin{definition}[परमाणु संकेतन]\label{def:g11:nucleus-radioactivity:notation}
ए \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} रोकना $ Z $ प्रोटॉन (आवेश $+e $ प्रत्येक) और $ N $ न्यूट्रॉन (न.)
चार्ज), $ A = Z + N $ न्युक्लियोन सभी में; यह लिखा है ${}^{A}_{Z}\mathrm{X}$,
कहाँ $\mathrm X $ रासायनिक प्रतीक है, $ Z $  \emph{परमाणु संख्या}\index{परमाणु संख्या} और $ A $  \emph{द्रव्यमान संख्या}\index{द्रव्यमान संख्या}. द्वारा निर्दिष्ट एक प्रजाति $ A $ और $ Z $ एक है \emph{न्यूक्लाइड}\index{न्यूक्लाइड};
उसी के साथ न्यूक्लाइड होता है $ Z $ लेकिन अलग $ A $ हैं \emph{आइसोटोप}\index{आइसोटोप}
एक तत्व का: समान रसायन शास्त्र (इलेक्ट्रॉन बादल केवल देखता है $ Ze $),
विभिन्न \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} और स्थिरता।
\end{definition}

\begin{example}[हाइड्रोजन और कार्बन]\label{ex:g11:nucleus-radioactivity:hc}
हाइड्रोजन में तीन हैं \omterm{def:g11:nucleus-radioactivity:notation}{आइसोटोप}: ${}^{1}_{1}\mathrm{H}$ (एक अकेला प्रोटॉन),
${}^{2}_{1}\mathrm{H}$ (ड्यूटेरियम: एक प्रोटॉन, एक न्यूट्रॉन), अस्थिर
${}^{3}_{1}\mathrm{H}$ (ट्रिटियम: दो न्यूट्रॉन)। प्राकृतिक कार्बन: अधिकतर
${}^{12}_{6}\mathrm{C}$ ($6$ प्रोटान, $6$ न्यूट्रॉन), $1.1\%$ का
${}^{13}_{6}\mathrm{C}$, और एक \omterm{def:g11:fundamental-interactions:ladder}{एटम} में $10^{12}$ अस्थिर का
${}^{14}_{6}\mathrm{C}$ ($8$ न्यूट्रॉन), का नायक \cref{ex:g11:nucleus-radioactivity:dating}.
\end{example}

\section{स्थिर और अस्थिर नाभिक}

\begin{proposition}[स्थिरता की घाटी]\label{prop:g11:nucleus-radioactivity:valley}
स्थिर \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड} में एक संकीर्ण बैंड पर कब्जा करें $(Z, N)$ विमान: $ N \approx Z $
हल्के \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} के लिए, फिर न्यूट्रॉन की बढ़ती हुई अधिकता, तक $ N \approx 1.5\,Z $.
प्रत्येक \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड} बैंड से बाहर --- बहुत अधिक न्यूट्रॉन, बहुत अधिक प्रोटॉन, या बहुत अधिक
बड़ा (लीड से परे, $ Z = 82$, नहीं \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड} वास्तव में स्थिर है) --- देर-सबेर रूपांतरित हो जाता है।
\end{proposition}
\admitted

\begin{remark}
बैंड का सटीक रूप से पता लगाना क्वांटम यांत्रिकी है --- विश्वविद्यालय वॉल्यूम।
पैटर्न का बही है \cref{ch:g11:fundamental-interactions}: मजबूत
गोंद केवल पड़ोसियों को बांधता है, कूलम्ब का प्रतिकर्षण संपूर्ण को फैलाता है \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, तो
भारी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}ों को अतिरिक्त न्यूट्रॉन गोंद की आवश्यकता होती है; फिर भी न्यूट्रॉन की अधिकता अस्थिर होती है
भी (द \omterm{def:g11:fundamental-interactions:weak}{कमजोर अंतःक्रिया} इसे देखता है)।
\end{remark}

\begin{definition}[रेडियोधर्मिता]\label{def:g11:nucleus-radioactivity:radioactivity}
\emph{रेडियोधर्मिता}\index{रेडियोधर्मिता} सहज परिवर्तन है
एक अस्थिर का \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} दूसरे में \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड}, विकिरण के उत्सर्जन के साथ। यह है \emph{यादृच्छिक}: कुछ भी घोषणा नहीं करता कि कौन सी \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} आगे क्षय होता है, और नहीं
\omterm{def:g10:pressure:pressure}{दबाव}, तापमान या रसायन विज्ञान इसे तेज़ या विलंबित कर सकता है; केवल बड़ा
जनसंख्या पूर्वानुमानित है (\cref{def:g11:nucleus-radioactivity:halflife}).
\end{definition}

\begin{remark}[बेकरेल और क्यूरीज़]\label{rem:g11:nucleus-radioactivity:curie}
1896 में हेनरी बेकरेल ने पाया कि यूरेनियम लवण एक फोटोग्राफिक प्लेट को धुंधला कर देता है
काले कागज के माध्यम से, एक बंद दराज में, नं. के साथ ऊर्जा आपूर्ति की गई. मैरी और
पियरे क्यूरी ने दिखाया कि प्रभाव परमाणु है, इसे मापा और पोलोनियम को अलग कर दिया
और रेडियम, एक लाख गुना अधिक सक्रिय। शब्द \emph{\omterm{def:g11:nucleus-radioactivity:radioactivity}{रेडियोधर्मिता}} है
मैरी क्यूरी की; तो दो नोबेल पुरस्कार हैं।
\end{remark}

\section{अल्फा, बीटा, गामा}

\begin{definition}[तीन ऐतिहासिक विकिरण]\label{def:g11:nucleus-radioactivity:abg}
तीन विकिरण, जिनका नाम किसी के जानने से पहले ही रख लिया गया कि वे क्या हैं:
\begin{itemize}
  \item \emph{अल्फा}\index{अल्फा} ($\alpha $): एक हीलियम \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}
        ${}^{4}_{2}\mathrm{He}$ पूरा बाहर निकाला गया --- भारी, दोगुना चार्ज किया हुआ,
        कागज की एक शीट या हवा के सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} द्वारा रोका गया;
  \item \emph{बीटा}\index{बीटा} ($\beta^-$): एक इलेक्ट्रॉन
        ${}^{\;0}_{-1}\mathrm{e}$ में बनाया गया \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} और बाहर निकाल दिया गया ---
        हल्का, तेज़, एल्यूमीनियम के मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} द्वारा रोका गया;
  \item \emph{गामा}\index{गामा} ($\gamma $): \omterm{def:g11:magnetic-fields:electromagnet}{विद्युत} चुम्बकीय
        विकिरण, प्रकाश से कहीं अधिक ऊर्जावान --- कभी पूरी तरह से नहीं रुका, केवल
        \emph{तनु}\index{तनु}: एक सेंटी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} सीसा आधा सोख लेता है।
\end{itemize}
\end{definition}

\begin{proposition}[परमाणु समीकरणों में संरक्षण]\label{prop:g11:nucleus-radioactivity:soddy}
प्रत्येक परमाणु परिवर्तन में, कुल \omterm{def:g11:nucleus-radioactivity:notation}{द्रव्यमान संख्या} $ A $ और कुल
प्रभार---निचले सूचकांकों का योग, गिनती $-1$ एक उत्सर्जित के लिए
इलेक्ट्रॉन --- पहले और बाद में समान हैं।
\end{proposition}
\admitted

\begin{remark}
चार्ज संरक्षण पर पहले ही फैसला सुनाया जा चुका है \cref{ch:g11:fundamental-interactions};
का संरक्षण $ A $ कहते हैं न्युक्लियोन केवल रूपांतरित होते हैं, कभी निर्मित नहीं होते या
नष्ट कर दिया गया --- विश्वविद्यालय के संस्करणों के लिए ईमानदार बहीखाता पद्धति।
\end{remark}

\begin{method}[क्षय समीकरण को संतुलित करना]\label{met:g11:nucleus-radioactivity:balance}
\begin{enumerate}
  \item प्रत्येक अभिनेता को दोनों लेबलों के साथ लिखें: $\alpha = {}^{4}_{2}\mathrm{He}$,
        $\beta^- = {}^{\;0}_{-1}\mathrm{e}$, $\beta^+ = {}^{0}_{+1}\mathrm{e}$; ए $\gamma $ किया जाता है $ A = Z = 0$.
  \item का योग बराबर करें $ A $का, फिर का $ Z $'एस
        (\cref{prop:g11:nucleus-radioactivity:soddy}); हल करें, और पढ़ें
        बेटी का तत्व इससे हट गया $ Z $ में \omterm{def:g10:signals-and-waves:period}{आवधिक} मेज़।
\end{enumerate}
\end{method}

\begin{example}[एक अल्फ़ा क्षय]\label{ex:g11:nucleus-radioactivity:alpha}
रेडियम-226, क्यूरीज़ का रेडियम, एक है $\alpha $ उत्सर्जक:
${}^{226}_{88}\mathrm{Ra} \to {}^{A}_{Z}\mathrm{Y} + {}^{4}_{2}\mathrm{He}$
बल $ A = 226 - 4 = 222$ और $ Z = 88 - 2 = 86$; तत्व $86$ रेडॉन है, इसलिए
${}^{226}_{88}\mathrm{Ra} \to {}^{222}_{86}\mathrm{Rn} + {}^{4}_{2}\mathrm{He}$
--- बिना हवादार तहखानों की रेडियोधर्मी गैस।
\end{example}

\begin{example}[एक बीटा-माइनस क्षय]\label{ex:g11:nucleus-radioactivity:beta}
कार्बन-14 बैंड के ऊपर बैठता है ($8$ न्यूट्रॉन के विरुद्ध $6$ प्रोटॉन):
${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} + {}^{\;0}_{-1}\mathrm{e}$.
जाँच करना: $14 = 14 + 0$, $6 = 7 - 1$; एक न्यूट्रॉन प्लस एक प्रोटॉन बन गया
उत्सर्जित इलेक्ट्रॉन --- \omterm{def:g11:fundamental-interactions:weak}{कमजोर अंतःक्रिया} पर \omterm{def:g11:work-of-force:work}{काम}.
\end{example}

\begin{remark}[बीटा-प्लस और गामा]\label{rem:g11:nucleus-radioactivity:betaplus}
बैंड के नीचे, दर्पण सड़ जाता है $\beta^+$ प्रोटॉन को न्यूट्रॉन में बदल देता है
प्लस ए \emph{पोजीट्रान}\index{पोजीट्रान} ${}^{0}_{+1}\mathrm{e}$, इलेक्ट्रॉन का प्रतिकण:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$,
पीईटी स्कैनर का वर्कहॉर्स। और सबसे सड़ने के बाद बेटी पैदा हुई, हिलती हुई,
का उत्सर्जन करके स्थिर हो जाता है $\gamma $ शीर्ष पर।
\end{remark}

\section{आधा जीवन और गतिविधि}

\begin{definition}[हाफ लाइफ]\label{def:g11:nucleus-radioactivity:halflife}
 \emph{हाफ लाइफ}\index{हाफ लाइफ} $ T_{1/2}$ एक का \omterm{def:g11:nucleus-radioactivity:notation}{न्यूक्लाइड} समय है
जिसके बाद किसी भी बड़े नमूने का आधा हिस्सा सड़ गया है। प्रत्येक आगे आधा जीवन
जो बचता है उसे आधा कर देता है: का $ N_0$ \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, $ N_0/2$ पर छोड़े गए हैं $ T_{1/2}$, $ N_0/4$
पर $2\,T_{1/2}$, $ N_0/8$ पर $3\,T_{1/2}$, $ N_0/2^{\,n}$ बाद $ n $ आधा जीवन.
\end{definition}

\begin{remark}
एक के लिए यादृच्छिकता \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, के लिए घड़ी की कल $10^{23}$: की एक विशाल भीड़
सिक्का उछालने वाले, एक-एक करके अप्रत्याशित, प्रत्येक दौर में आधे-आधे बाहर हो गए।
प्रक्षेप करना \emph{बीच में} संपूर्ण अर्ध-आयु घातांकीय होती है
समारोह --- अगले वर्ष, गणित की मात्रा के साथ।
\end{remark}

\begin{example}[दस की बीस शक्तियाँ]\label{ex:g11:nucleus-radioactivity:range}
\begin{center}\small\begin{tabular}{llll}
polonium-214 & $\alpha $ & \qty{1.6e-4}{s} & radium chain link \\
technetium-99m & $\gamma $ & $6.0$ hours & medical imaging \\
americium-241 & $\alpha $ & $432$ years & smoke detectors \\
carbon-14 & $\beta^-$ & \num{5700} years & archaeology \\
potassium-40 & $\beta $ & \num{1.25e9} years & in every banana \\

\begin{remark}[खुराक और छलनी]\label{rem:g11:nucleus-radioactivity:dose}
विकिरण इलेक्ट्रॉनों को तोड़ देता है \omterm{def:g11:fundamental-interactions:ladder}{अणुओं} और डीएनए को नुकसान पहुंचा सकता है। जैविक

\emph{सिवर्ट्स}\index{सिवर्ट्स} (\unit{Sv}), जो
का वजन करता है ऊर्जा प्रति जमा किया गया \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} प्रत्येक ऊतक कितना हानिकारक है
विकिरण है. तीन नियम: दूरी, परिरक्षण ($\alpha $: त्वचा पर्याप्त है ---
जब तक कि उत्सर्जक न हो \emph{साँस}, रेडॉन का अपराध; $\beta $: एल्यूमीनियम;
$\gamma $: लीड), और समय। डोसिमेट्री उचित विश्वविद्यालय सामग्री है।
\end{remark}

\section{समस्या: ममी, रिएक्टर और स्मोक डिटेक्टर}

\begin{problem}[{सप्ताहांत समस्या --- एक अस्थिर केंद्रक के तीन करियर:
आधा-अधूरा करके दिनांकित एक ममी, यूरेनियम का वंश वृक्ष संतुलित, एक धूम्रपान डिटेक्टर

\label{pb:g11:nucleus-radioactivity:1}
तीन अस्थिर \omterm{def:g11:fundamental-interactions:ladder}{नाभिक}, तीन कार्य: प्रत्येक मृत पौधे पर कार्बन-14 टाइमस्टैम्प
और जानवर, यूरेनियम-238 ग्रह को अंदर से गर्म करता है, अमेरिकियम-241 देखता है
छत. डेटा: $ T_{1/2} = \num{5700}$ वर्ष (सी-14), \num{4.5e9} साल
(यू-238), $432$ वर्ष (एएम-241); जीवित कार्बन शो $13.6$ प्रति मिनट क्षय होता है
प्रति ग्राम; परमाणु क्रमांक: एन $7$, ओ $8$, पी.बी $82$, द्वि $83$, पो $84$, आरएन $86$,
रा $88$, वां $90$, पा $91$, यू $92$, एन.पी $93$, पूर्वाह्न $95$.

\noindent\textbf{Part I --- The mummy.}
\begin{enumerate}
  \item (प्रोटॉन, न्यूट्रॉन) का संघटन दीजिए ${}^{12}_{6}\mathrm{C}$
        और ${}^{14}_{6}\mathrm{C}$. एक शरीर में समान रसायन क्यों?
  \item जीवित शरीर में कार्बन-14 का अनुपात स्थिर क्यों होता है, और

  \item कार्बन-14 का क्षय समीकरण लिखिए। क्या इसका विकिरण बच जायेगा

  \item लिनेन से एक ग्राम कार्बन निकलता है $6.8$ प्रति मिनट क्षय होता है।
        वह जीवन दर का कौन सा अंश है? मम्मी को डेट करो.
  \item एक \omterm{def:g10:orders-of-magnitude:unit}{दूसरा} ``मम्मी'', एक निजी संग्रह का पुरस्कार, दिखाता है
        $13.2$ प्रति ग्राम प्रति मिनट क्षय होता है। निर्णय?
  \item दस आधे जीवन पुराने नमूने से दर का अनुमान लगाएं और समझाएं
        कार्बन डेटिंग मोटे तौर पर क्यों लुप्त होती जा रही है? \num{57000} साल।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The reactor under the meadow.}
\begin{enumerate}[resume]
  \item की रचना बताइये ${}^{238}_{92}\mathrm{U}$. भारी क्यों करते हो
        क्या \omterm{def:g11:fundamental-interactions:ladder}{नाभिक} को ऐसे न्यूट्रॉन अधिशेष की आवश्यकता है?
  \item इसे लिखें $\alpha $ क्षय समीकरण.
  \item फिर बेटी सड़ जाती है $\beta^-$, और पोती
        $\beta^-$ दोबारा। दोनों समीकरण लिखिए. कौन सा तत्व पुनः प्रकट होता है?
  \item शृंखला यहीं समाप्त होती है ${}^{206}_{82}\mathrm{Pb}$. इसकी गणना करें $\alpha $
        चरण, फिर यह $\beta^-$ कदम.
  \item पृथ्वी का निर्माण हुआ \num{4.5e9} साल पहले। इसका कितना अंश
        प्राइमर्डियल यूरेनियम-238 आज भी बचा हुआ है? और में $5$ अरब वर्ष और?
  \item एक वाक्य में: यह दबी हुई क्षय ऊष्मा क्या करती है? \omterm{def:g11:work-of-force:power}{शक्ति} पर
        सतह? (ज्वालामुखियों और बहते महाद्वीपों के बारे में सोचें।)
\end{enumerate}

  \item लिखना $\alpha $ का क्षय समीकरण ${}^{241}_{95}\mathrm{Am}$.
  \item इसका \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} है \qty{33}{kBq}: प्रति सेकंड कितने क्षय? प्रति दिन?
  \item डिटेक्टर को समझाएं: क्या करते हैं $\alpha $ कण हवा से करते हैं
        चैम्बर का, और धुआं क्या बदलता है?
  \item एक क्यों है? $\alpha $ यहाँ सही विकल्प उत्सर्जित करें --- और ए
        $\gamma $ बराबर का उत्सर्जक \omterm{def:g11:nucleus-radioactivity:activity}{गतिविधि} सबसे ख़राब?

\noindent\textbf{Part IV --- Why the stars had to be mortal.}
\begin{enumerate}[resume]
  \item मरते हुए तारों ने प्रत्येक भारी तत्व को गढ़ा और बिखेर दिया
        हीलियम. यूरेनियम-238 का कौन सा अंश जाली है? \num{9e9} वर्षों पहले
        आज जीवित है? पृथ्वी की शेष प्रचुरता क्यों बनती है?
        सबसे पुराने तारों से भी छोटा ग्रह?
  \item पृथ्वी की रेडियोधर्मी आंतरिक ऊष्मा ज्वालामुखी और प्लेट को संचालित करती है
        टेक्टोनिक्स, और कोर को \omterm{def:g11:magnetic-fields:magnet}{चुंबक}ीय ढाल से बाहर निकालता रहता है
        \cref{ch:g11:magnetic-fields}. क्या कोई ग्रह केवल स्थिर होगा
        परमाणुओं उस जैसे रहो?
  \item एक पंक्ति, कविता के रूप में और बही-खाते के रूप में: कहाँ थे तुम्हारे परमाणुओं
        जाली, और क्या अभी भी आपके पैरों के नीचे की ज़मीन को गर्म करता है?
\end{enumerate}
\end{problem}
