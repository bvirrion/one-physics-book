\chapter{लेंस, छवियाँ, और आँख}\label{ch:g11:lenses-and-eye}

प्रत्येक तीक्ष्ण चित्र --- फ़ोन सेंसर, सिनेमा स्क्रीन, या पीछे
आपकी अपनी आँख --- उसी चाल से आती है: एक घुमावदार पारदर्शी
माध्यम किरणों को मोड़ता है \omterm{def:g10:refraction:refraction}{अपवर्तन} (\cref{ch:g10:refraction}) तो वह सब
किरणें एक बिंदु से निकलकर दूसरे बिंदु पर पुनः मिलती हैं। यह अध्याय वश में करता है
युक्ति: एक \omterm{def:g11:lenses-and-eye:lens}{लेंस}, तीन किरणें, एक सूत्र --- फिर इसे इंगित करता है
कैमरे, चश्मा, और रहन-सहन \omterm{def:g11:lenses-and-eye:lens}{लेंस} इस वाक्य को पढ़ रहे हैं.

\section{अभिसरण और अपसारी लेंस}

\begin{definition}[पतला लेंस]\label{def:g11:lenses-and-eye:lens}
ए \emph{लेंस}\index{लेंस} दो से घिरा एक पारदर्शी माध्यम है
सतहें, कम से कम एक घुमावदार। किनारे की तुलना में केंद्र में अधिक मोटा,
यह है \emph{अभिसारी}\index{अभिसारी}: समानान्तर किरणें मुड़ी हुई होती हैं
एक दूसरे की ओर; किनारे पर मोटा, \emph{अपसारी}\index{अपसारी}: वे अलग-अलग फैल गए। ए \emph{पतला}\index{पतला} लेंस है
नगण्य मोटाई: हम बाहर की ओर (अभिसरण) या के साथ एक खंड खींचते हैं
अंदर की ओर (अपसारी) तीर के सिरे। इसका केंद्र $O $ है \emph{ऑप्टिकल केंद्र}\index{ऑप्टिकल केंद्र}; के माध्यम से लंबवत रेखा $ O $, द
\emph{मुख्य धुरी}\index{मुख्य धुरी}.
\end{definition}

\begin{definition}[फोकल बिंदु और फोकल लंबाई]\label{def:g11:lenses-and-eye:focal}
किरणें a की धुरी के समानांतर आती हैं \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} सब पार करो
इसके पीछे एक बिंदु पर अक्ष, \emph{छवि केंद्र बिंदु}\index{छवि केंद्र बिंदु} $ F'$; सममित रूप से, किरणें निकल रही हैं \emph{वस्तु का केंद्र बिंदु}
$ F $ (सामने, साथ $ OF = OF'$) अक्ष के समानांतर बाहर निकलें।
\emph{फोकल लम्बाई}\index{फोकल लम्बाई} $ f' = \overline{OF'}$ है
ए के लिए सकारात्मक \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस}; किसी के जरिए \omterm{def:g11:lenses-and-eye:lens}{अपसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस}, समानांतर किरणें
बाहर निकलें जैसे कि एक बिंदु से $ F'$ \emph{सामने} की \omterm{def:g11:lenses-and-eye:lens}{लेंस}: $ f' < 0$.
\end{definition}

\begin{definition}[vergence]\label{def:g11:lenses-and-eye:vergence}
 \emph{vergence}\index{vergence} एक का \omterm{def:g11:lenses-and-eye:lens}{लेंस} का \omterm{def:g11:lenses-and-eye:focal}{फोकल लम्बाई} $ f'$
(\omterm{def:g10:orders-of-magnitude:unit}{मीटर} में) है $ C = 1/f'$,
में मापा गया \emph{diopters}\index{diopters} (प्रतीक डी):
$1~\text{D} = \qty{1}{m^{-1}}$; \omterm{def:g11:lenses-and-eye:lens}{अपसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} है $ C < 0$.
ऑप्टिशियंस डायोप्टर में लिखते हैं: के सत्यापन \omterm{def:g11:lenses-and-eye:lens}{पतला} संपर्क में \omterm{def:g11:lenses-and-eye:lens}{लेंस}
बस जोड़ें (यहां स्वीकृत; वर्ष 1 खंड में लिया गया)।
\end{definition}

\begin{example}[एक नुस्खा पढ़ना]\label{ex:g11:lenses-and-eye:vergence}
ए \omterm{def:g11:lenses-and-eye:lens}{लेंस} साथ $ f' = \qty{0.50}{m}$ है $ C = 1/0.50 = +2.0$~डी:
\omterm{def:g11:lenses-and-eye:lens}{अभिसारी}. ए $-4.0$~D नुस्खे का मतलब है $ f' = \qty{-0.25}{m}$:
\omterm{def:g11:lenses-and-eye:lens}{अपसारी}, \omterm{def:g11:lenses-and-eye:focal}{फोकल लम्बाई} \qty{25}{cm}.
\end{example}

\section{छवि का निर्माण}

\begin{method}[तीन निर्माण किरणें]\label{met:g11:lenses-and-eye:rays}
होने देना $ AB $ अक्ष पर लंबवत कोई वस्तु हो $ A $. सभी किरणों के बीच
टिप छोड़ रहा हूँ $ B $, तीन के ज्ञात पथ हैं:
\begin{enumerate}
  \item के माध्यम से किरण \omterm{def:g11:lenses-and-eye:lens}{ऑप्टिकल केंद्र} $ O $ सीधे चला जाता है;
  \item अक्ष के समानांतर किरण बाहर निकलती है $ F'$;
  \item के माध्यम से किरण $ F $ अक्ष के समानांतर बाहर निकलता है।
\end{enumerate}
कोई भी दो पर्याप्त हैं: उनका प्रतिच्छेदन छवि बिंदु है $ B'$, और
छवि $ A'B'$ पर अक्ष पर लंबवत \omterm{def:g10:universal-gravitation:weight}{खड़ा} है $ A'$. यदि निवर्तमान
किरणें अलग हो जाती हैं, उनकी \emph{पिछड़ा} एक्सटेंशन (धराशायी) एक पर मिलते हैं
\omterm{def:g11:lenses-and-eye:images}{आभासी} छवि बिंदु।
\end{method}

\begin{definition}[वास्तविक और आभासी छवियाँ]\label{def:g11:lenses-and-eye:images}
एक छवि है \emph{असली}\index{असली} जब बाहर जाने वाली किरणें
वास्तव में इसके माध्यम से गुजरें: वहां रखी एक स्क्रीन इसे पकड़ लेती है;
\emph{आभासी}\index{आभासी} जब केवल उनका पिछड़ा विस्तार होता है
मिलें: कोई स्क्रीन इसे नहीं पकड़ती, बल्कि एक नज़र इसे पकड़ती है \omterm{def:g11:lenses-and-eye:lens}{लेंस} इसे देखता है
बिल्कुल ठीक. ए \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} किसी की वास्तविक, उलटी छवि देता है
परे वस्तु $ F $; ऑब्जेक्ट को अंदर स्लाइड करें \omterm{def:g11:lenses-and-eye:focal}{फोकल लम्बाई} और
छवि आभासी, सीधी, बड़ी हो जाती है।
\end{definition}

\section{एक सूत्र: संयुग्मी संबंध}

निर्माण दिखाते हैं \emph{कहाँ} छवि है; एक सूत्र इसकी गणना करता है।
अक्ष को प्रकाश के अनुदिश उन्मुख करें और उपयोग करें \emph{बीजगणितीय उपाय}:
$\overline{OA}$ का समन्वय है $ A $ के सापेक्ष $ O $ (के लिए नकारात्मक
ए \omterm{def:g11:lenses-and-eye:images}{असली} सामने वस्तु); ऊंचाइयों $\overline{AB}$ सकारात्मक को ऊपर की ओर गिनें।

\begin{theorem}[पतले-लेंस संयुग्म संबंध]\label{thm:g11:lenses-and-eye:conjugate}
किसी के जरिए \omterm{def:g11:lenses-and-eye:lens}{पतला} \omterm{def:g11:lenses-and-eye:lens}{लेंस} का \omterm{def:g11:lenses-and-eye:focal}{फोकल लम्बाई} $ f'$, एक वस्तु बिंदु $ A $ पर
बिंदु पर अक्ष छवियाँ $ A'$ साथ
\[
\frac{1}{\overline{OA'}} - \frac{1}{\overline{OA}} = \frac{1}{f'} = C .
\]
\end{theorem}
\admitted

\begin{remark}
संबंध इस प्रकार है \omterm{thm:g10:refraction:snell}{स्नेल का नियम} के निकट किरणों पर लगाया जाता है
अक्ष; व्युत्पत्ति --- और की सीमाएँ पतली \omterm{def:g11:lenses-and-eye:lens}{लेंस} मॉडल --- हैं
वर्ष 1 खंड में ईमानदार बनाया गया।
\end{remark}

\begin{definition}[बढ़ाई]\label{def:g11:lenses-and-eye:magnification}
 \emph{बढ़ाई}\index{बढ़ाई} छवि का है
\[
\gamma = \frac{\overline{A'B'}}{\overline{AB}}
       = \frac{\overline{OA'}}{\overline{OA}} .
\]
$\abs{\gamma}$ आकार, उसके चिह्न अभिविन्यास की तुलना करता है ($\gamma < 0$:
उलटा); का \omterm{def:g10:signals-and-waves:signal}{संकेत} $\overline{OA'}$ प्रकृति को पढ़ता है (सकारात्मक:
\omterm{def:g11:lenses-and-eye:images}{असली}, पीछे \omterm{def:g11:lenses-and-eye:lens}{लेंस}; नकारात्मक: \omterm{def:g11:lenses-and-eye:images}{आभासी}, सामने)।
\end{definition}

\begin{example}[काम पर सूत्र]\label{ex:g11:lenses-and-eye:conjugate}
एक वस्तु खड़ी है \qty{30}{cm} ए के सामने \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} साथ
$ f' = \qty{10}{cm}$: $\overline{OA} = \qty{-30}{cm}$, इसलिए
\[
\frac{1}{\overline{OA'}} = \frac{1}{10} - \frac{1}{30} = \frac{1}{15},
\qquad \overline{OA'} = +\qty{15}{cm},
\qquad \gamma = \frac{15}{-30} = -0.50 :
\]
ए वास्तविक छवि \qty{15}{cm} पीछे \omterm{def:g11:lenses-and-eye:lens}{लेंस}, उलटा, आधे आकार का ---
बिल्कुल वही जो तीन किरणें खींचती हैं।
\end{example}

\section{आंख}

\begin{definition}[एक ऑप्टिकल प्रणाली के रूप में आँख]\label{def:g11:lenses-and-eye:eye}
प्रकाशिक रूप से, आँख एक है \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} एक स्क्रीन का सामना करना:
\emph{कॉर्निया}\index{कॉर्निया} और \emph{क्रिस्टलीय लेंस}\index{क्रिस्टलीय लेंस} एक के रूप में कार्य करें \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} समायोज्य का \omterm{def:g11:lenses-and-eye:focal}{फोकल लम्बाई};
\emph{रेटिना}\index{रेटिना}, के बारे में \qty{17}{mm} पीछे, निश्चित है
स्क्रीन. को निचोड़कर फोकस किया जाता है \omterm{def:g11:lenses-and-eye:lens}{लेंस}, इसे अंदर की तरह ले जाकर नहीं
एक कैमरा: \emph{आवास}\index{आवास} का परिवर्तन है
$ f'$ सिलिअरी मांसपेशियों द्वारा निर्मित। निकटतम बिंदु पूरी तरह से तेजी से देखा गया
आवास है \emph{निकट बिंदु}\index{निकट बिंदु} --- के बारे में
\qty{25}{cm} के लिए \emph{मानक आँख}\index{मानक आँख} --- द
सबसे दूर, आराम पर, \emph{सुदूर बिंदु}\index{सुदूर बिंदु} (अनंत के लिए
मानक आँख)।
\end{definition}

\section{दोष, सुधार और आवर्धक}

\begin{definition}[मायोपिया और हाइपरोपिया]\label{def:g11:lenses-and-eye:defects}
ए \emph{कमबीन}\index{कमबीन} (अदूरदर्शी) आँख भी है \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} के लिए
इसकी लंबाई: समानांतर किरणें फोकस करती हैं \emph{के सामने}  \omterm{def:g11:lenses-and-eye:eye}{रेटिना}, दूर
वस्तुएँ धुंधली हो जाती हैं, \omterm{def:g11:lenses-and-eye:eye}{सुदूर बिंदु} परिमित दूरी पर है; ए \omterm{def:g11:lenses-and-eye:lens}{अपसारी}
तमाशा \omterm{def:g11:lenses-and-eye:lens}{लेंस} इसे ठीक करता है. ए \emph{हाइपरोपिक}\index{हाइपरोपिक}
(दूरदर्शी) आँख नहीं है \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} पर्याप्त: निकट की वस्तुएं फोकस करेंगी
के पीछे \omterm{def:g11:lenses-and-eye:eye}{रेटिना}, द \omterm{def:g11:lenses-and-eye:eye}{निकट बिंदु} पीछे हटना; ए \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} सही करता है
यह. उम्र बढ़ने से अकड़न हो जाती है \omterm{def:g11:lenses-and-eye:eye}{क्रिस्टलीय लेंस}
(\emph{जरादूरदृष्टि}\index{जरादूरदृष्टि}), सिकुड़ रहा है \omterm{def:g11:lenses-and-eye:eye}{आवास} --- वही
उपाय: पढ़ने का चश्मा.
\end{definition}

\end{example}

\section{समस्या: एक सूत्र, तीन उपकरण}

\begin{problem}[{सप्ताहांत समस्या---वही संयुग्मी संबंध
एक कैमरा फोकस करता है, पढ़ने के लिए चश्मा निर्धारित करता है, एक लूप डिजाइन करता है, और
आपकी अपनी आंख में निर्मित ज़ूम को मापता है}]
\label{pb:g11:lenses-and-eye:1}
एक फोटोग्राफर रैक करता है \qty{50}{mm} \omterm{def:g11:lenses-and-eye:lens}{लेंस}, एक दादी रखती है
हाथ की दूरी पर अखबार, एक जौहरी हीरे पर झुक रहा है: की एक पंक्ति
बीजगणित, $1/\overline{OA'} - 1/\overline{OA} = 1/f'$
(\cref{thm:g11:lenses-and-eye:conjugate}), तीनों ट्रेड चलाता है ---
और फिर आंख ही.

\medskip
\noindent\textbf{Part I --- One formula.} ए \omterm{def:g11:lenses-and-eye:lens}{अभिसारी} \omterm{def:g11:lenses-and-eye:lens}{लेंस} है
$ f' = \qty{20}{cm}$.
\begin{enumerate}
  \item पर एक वस्तु \qty{60}{cm}: गणना करें $\overline{OA'}$ और
        $\gamma $; छवि का वर्णन करें.
  \item ऑब्जेक्ट को यहां ले जाएं \qty{30}{cm}; पुनर्गणना क्या आदान-प्रदान हुआ
        प्रश्न 1 के साथ? (प्रकाश पथ प्रतिवर्ती हैं।)
  \item इसे यहां ले जाएं \qty{10}{cm}, अंदर $ f'$: पुनर्गणना करें और वर्णन करें।
  \item संक्षेप में बताएं: जैसे ही वस्तु बहुत दूर से अंदर की ओर सरकती है $ F $,
        फिर इसके पीछे, छवि कैसे चलती है और प्रकृति को कैसे बदलती है?
  \item दिखाएँ कि एक बहुत दूर की वस्तु ($1/\overline{OA} \approx 0$)
        हमेशा फोकल प्लेन में छवियाँ: $\overline{OA'} = f'$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The photographer.}
एक कैमरा \omterm{def:g11:lenses-and-eye:lens}{लेंस} है $ f' = \qty{50}{mm}$; सेंसर, \qty{24}{mm} लंबा,
के सापेक्ष स्थानांतरित किया जा सकता है \omterm{def:g11:lenses-and-eye:lens}{लेंस}.
\begin{enumerate}[resume]
  \item दूर के परिदृश्य पर ध्यान केंद्रित करने पर सेंसर कहाँ होता है?
  \item अब एक चेहरा \qty{1.0}{m} दूर: नए लेंस--सेंसर की गणना करें
        परिदृश्य सेटिंग से दूरी और यात्रा।
  \item तंत्र अधिकतम अनुमति देता है \qty{55}{mm} \omterm{def:g11:lenses-and-eye:lens}{लेंस} का--सेंसर
        दूरी. न्यूनतम फोकसिंग दूरी की गणना करें।
  \item उस निकटतम फोकस पर, गणना करें $\gamma $. क्या ए की छवि
        \qty{20}{cm} सेंसर पर चेहरा फिट?
  \item आदमकद ``मैक्रो'' का अर्थ है $\gamma = -1$: कहां आपत्ति होनी चाहिए और
        सेंसर तो हो, और यह एक विशेष की मांग क्यों करता है \omterm{def:g11:lenses-and-eye:lens}{लेंस}?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The grandmother.}
उम्र के साथ \omterm{def:g11:lenses-and-eye:eye}{निकट बिंदु} पीछे हटना: दादी का घर है \qty{1.0}{m}.
\begin{enumerate}[resume]
  \item बांह की लंबाई वाले समाचार पत्र की व्याख्या करें।
  \item पर पढ़ने के लिए \qty{25}{cm}, उसके चश्मे को एक पृष्ठ अवश्य पलटना चाहिए
        \qty{25}{cm} एक में \omterm{def:g11:lenses-and-eye:images}{आभासी} छवि पर \qty{1.0}{m}: गणना करें
        आवश्यक \omterm{def:g11:lenses-and-eye:vergence}{vergence}.
  \item गणना करना $\gamma $. छवि चार गुना बड़ी है, फिर भी वैसी नहीं दिखती
        बड़ा: आकारों की तुलना करके समझाएं \emph{और} दूरियाँ.
  \item दिखाएँ कि प्रिंट केवल बीच में ही तेज़ है \qty{25}{cm} और
        $ f' \approx \qty{33}{cm}$. का क्या? \emph{दूरस्थ} वस्तु देखी
        चश्मे के माध्यम से?
  \item एक वाक्य में निष्कर्ष निकालें कि बाइफोकल \omterm{def:g11:lenses-and-eye:lens}{लेंस} क्यों मौजूद हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The jeweler, and the eye's own zoom.}
जौहरी का लोपा है $ f' = \qty{5.0}{cm}$; उसका \omterm{def:g11:lenses-and-eye:eye}{निकट बिंदु} पर है
\qty{25}{cm}; उसका \omterm{def:g11:lenses-and-eye:eye}{रेटिना} है \qty{17}{mm} उसकी आँख के पीछे \omterm{def:g11:lenses-and-eye:lens}{लेंस}.
\begin{enumerate}[resume]
  \item एक पत्थर \qty{4.0}{cm} लूप के अंतर्गत: स्थिति, प्रकृति,
        \omterm{def:g11:lenses-and-eye:magnification}{बढ़ाई}?
  \item पत्थर को कहाँ बैठना चाहिए ताकि उसकी छवि उस पर गिरे \omterm{def:g11:lenses-and-eye:eye}{निकट
        बिंदु}, और क्या है $\gamma $ तब?
  \item अनुभवी जौहरी पत्थर को बिल्कुल ठीक स्थान पर क्यों रखते हैं? $ F $
        इसके बजाय (अनंत पर छवि)?
  \item अब केवल आँख: इसकी गणना करें \omterm{def:g11:lenses-and-eye:vergence}{vergence} दूर से देखना
        ऑब्जेक्ट, फिर एक पर \qty{25}{cm}: कितने \omterm{def:g11:lenses-and-eye:vergence}{diopters} का
        \omterm{def:g11:lenses-and-eye:eye}{आवास}?
  \item दोनों को रूपांतरित करें सत्यापन को फोकल लंबाई: कितने से, में
        मिली\omterm{def:g10:orders-of-magnitude:unit}{मीटर} और प्रतिशत, क्या आँख अपने आप बदल जाती है \omterm{def:g11:lenses-and-eye:focal}{फोकल लम्बाई} --- यात्रा और दादी के साथ कैमरे ने जो ज़ूम किया
        के साथ $+3$~डी?
\end{enumerate}
\end{problem}
