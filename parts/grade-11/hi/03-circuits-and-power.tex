\chapter{इलेक्ट्रिक सर्किट और पावर}\label{ch:g11:circuits-and-power}

एक स्विच पलटें: एक लैंप जलता है, क्योंकि एक जनरेटर कई किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} दूर है
आपकी दीवार के माध्यम से तांबे के एक लूप के चारों ओर चार्ज को धकेलता है। पहले के वर्ष
सर्किट को मैप किया --- करंट, \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}, \omterm{prop:g11:circuits-and-power:ohm}{ओम कानून}. यह अध्याय पुनः खुलता है
इसका हिसाब किताब के साथ \cref{ch:g10:energy-conservation}: क्या
\omterm{def:g10:orders-of-magnitude:unit}{मीटर} चार्ज, क्यों \omterm{def:g11:lenses-and-eye:lens}{पतला} तार चमकते हैं, ग्रिड क्यों चलता है \qty{400}{kV}.

\section{चलते-फिरते चार्ज करें}

\begin{definition}[धारा और कूलम्ब]\label{def:g11:circuits-and-power:current}
एक \emph{विद्युत धारा}\index{विद्युत धारा} का एक क्रमबद्ध प्रवाह है
चार्ज (इलेक्ट्रॉन, एक धातु में)। इसकी तीव्रता $I $, में
\emph{एम्पीयर}\index{एम्पीयर} (\unit{A}), क्या आवेश a से होकर गुजर रहा है
प्रति सेकंड तार का क्रॉस-सेक्शन: एक स्थिर धारा $ I $ एक के लिए बह रहा है
समय $ t $ चार्ज का परिवहन करता है
\[ q = I\,t , \]
में \emph{कूलंब}\index{कूलंब} (\unit{C}), साथ $\qty{1}{C} =
\qty{1}{A.s}$. प्राथमिक आरोप है $ e = \qty{1.6e-19}{C}$. वर्तमान
श्रृंखला में डाले गए एमीटर द्वारा मापा जाता है।
\end{definition}

\begin{example}[बिजली बनाम फोन]\label{ex:g11:circuits-and-power:coulomb}
एक बिजली का झटका कुछ ले जाता है $\qty{3.0e4}{A}$ के बारे में
$\qty{1.0e-4}{s}$: $ q = 3.0\times10^{4} \times 10^{-4} = \qty{3.0}{C}$.
आपका फ़ोन, पर \qty{2.0}{A}, हर एक समान चार्ज को स्थानांतरित करता है $1.5$ सेकंड.
\end{example}

\begin{definition}[वोल्टेज और वोल्ट]\label{def:g11:circuits-and-power:voltage}
 \emph{वोल्टेज}\index{वोल्टेज} $ U $ एक सर्किट के दो बिंदुओं के बीच, में
\emph{वोल्ट}\index{वोल्ट} (\unit{V}), है ऊर्जा प्रति का आदान-प्रदान किया गया \omterm{def:g10:orders-of-magnitude:unit}{इकाई} का
उनके बीच यात्रा शुल्क: $\qty{1}{V} = \qty{1}{J/C}$, तो एक शुल्क
$ q $ वोल्टेज के तहत एक उपकरण को पार करना $ U $ का आदान-प्रदान करता है ऊर्जा $ E = qU $
इसके साथ. समानांतर में जुड़ा एक वोल्टमीटर इसे मापता है।
\end{definition}

\begin{proposition}[सर्किट कानून]\label{prop:g11:circuits-and-power:laws}
स्थिर रूप से चलने वाले सर्किट में:
\begin{enumerate}
  \item \emph{जंक्शन कानून:} जंक्शन पर पहुंचने वाली धाराएं जुड़ती हैं
        उसे छोड़ने वाली धाराओं को;
  \item \emph{श्रृंखला कानून:} \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} एक पथ के साथ जोड़ें:
        $ U_{AC} = U_{AB} + U_{BC}$;
  \item \emph{समानांतर कानून:} दो शाखाएं समान दो बिंदुओं को जोड़ती हैं
        वही ले जाओ \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}.
\end{enumerate}
\end{proposition}

\begin{proof}
स्थिर अवस्था में आवेश कहीं भी एकत्रित नहीं होता है, इसलिए क्या प्रवाहित होता है?
जंक्शन बहता है। ऊर्जा प्रति कूलम्ब एक पथ के साथ योगात्मक है और
केवल इसके अंतिम बिंदुओं पर निर्भर करता है --- उनके बीच ली गई शाखा पर नहीं।
\end{proof}

\section{ओम का नियम और अवरोधक नेटवर्क}

\begin{definition}[प्रतिरोध]\label{def:g11:circuits-and-power:resistance}
 \emph{प्रतिरोध}\index{प्रतिरोध} एक कंडक्टर का अनुपात है
$ R = U/I $ की \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} इसके पार इसके माध्यम से प्रवाहित होने वाली धारा को मापा जाता है
\emph{ओम}\index{ओम} (\unit{\ohm}): $\qty{1}{\ohm} = \qty{1}{V/A}$.
\end{definition}

\begin{proposition}[ओम कानून]\label{prop:g11:circuits-and-power:ohm}
\index{ओम कानून}
निश्चित तापमान पर रखे गए धात्विक कंडक्टर के लिए, $ R $ स्थिर है:
\omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} और धारा आनुपातिक हैं,
\[ U = R\,I . \]
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:circuits-and-power:ohm-limits}
एक प्रायोगिक नियम, सार्वभौमिक नहीं: एक डायोड, एक हीटिंग लैंप
फिलामेंट या आपकी अपनी त्वचा इसका पालन नहीं करती है। धातुएँ क्यों उत्पन्न होती हैं?
विश्वविद्यालय की मात्रा.
\end{remark}

\begin{proposition}[श्रृंखला और समानांतर प्रतिरोधक]\label{prop:g11:circuits-and-power:networks}
\index{समतुल्य प्रतिरोध}
श्रृंखला में दो प्रतिरोधक एक प्रतिरोधक के बराबर होते हैं $ R_{\text{s}} =
R_1 + R_2$; एक प्रतिरोधक के समानांतर दो $ R_{\text{p}}$ द्वारा दिए गए
\[
\frac{1}{R_{\text{p}}} = \frac{1}{R_1} + \frac{1}{R_2} ,
\qquad\text{i.e.}\quad
R_{\text{p}} = \frac{R_1 R_2}{R_1 + R_2} < \min(R_1, R_2) .
\]
\end{proposition}

\begin{proof}
शृंखला: वही $ I $ और दोनों को पार करता है \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} जोड़ना
(\cref{prop:g11:circuits-and-power:laws}), इसलिए $ U = (R_1 + R_2)\,I $.
समानांतर: दोनों एक ही के नीचे बैठते हैं $ U $ और धाराएँ जुड़ती हैं, इसलिए
$ I = \frac{U}{R_1} + \frac{U}{R_2}$.
\end{proof}

\begin{example}[समतुल्य प्रतिरोध]\label{ex:g11:circuits-and-power:networks}
$ R_1 = \qty{100}{\ohm}$ और $ R_2 = \qty{150}{\ohm}$: शृंखला में,
\qty{250}{\ohm}; समानांतर में, $\frac{100 \times 150}{250} =
\qty{60}{\ohm}$ --- दोनों में से कम: धारा को दूसरी सड़क मिलती है।
\end{example}

\section{एक सर्किट में ऊर्जा और शक्ति}

\begin{proposition}[विद्युत शक्ति]\label{prop:g11:circuits-and-power:power}
\index{विद्युत शक्ति}
के अंतर्गत एक उपकरण \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} $ U $ एक धारा द्वारा पार किया गया $ I $ प्राप्त करता है ऊर्जा पर
दर
\[ P = U\,I , \qquad\text{i.e.}\quad E = U\,I\,t \ \text{in a time } t . \]
\end{proposition}

\begin{proof}
एक समय में $ t $ आरोप $ q = It $ डिवाइस को पार करता है
(\cref{def:g11:circuits-and-power:current}) और प्रत्येक कूलम्ब सौंपना
$ U $ जूल (\cref{def:g11:circuits-and-power:voltage}): $ E = qU = UIt $.
\end{proof}

\begin{proposition}[एक अवरोधक में शक्ति]\label{prop:g11:circuits-and-power:ri2}
एक अवरोधक $ R $ करंट ले जाना $ I $ dissipates
\[ P = R\,I^2 = \frac{U^2}{R} . \]
\end{proposition}

\begin{proof}
विकल्प \omterm{prop:g11:circuits-and-power:ohm}{ओम कानून} में $ P = UI $: $ P = (RI)I = RI^2 = U(U/R) = U^2/R $.
\end{proof}

\begin{example}[नेमप्लेट पढ़ना]\label{ex:g11:circuits-and-power:nameplate}
एक केतली पर `` अंकित है\qty{230}{V} -- \qty{2200}{W}'' खींचता है $ I = P/U =
2200/230 = \qty{9.6}{A}$ और हैं \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} $ R = U^2/P = 230^2/2200 =
\qty{24}{\ohm}$. दौड़ना \qty{150}{s} यह उपयोगकर्ता है $ E = 2200 \times 150 =
\qty{3.3e5}{J} \approx \qty{0.09}{kW.h}$ --- लगभग दो सेंट.
\end{example}

\begin{remark}[जूल और किलोवाट-घंटे]\label{rem:g11:circuits-and-power:kwh}
\omterm{def:g10:orders-of-magnitude:unit}{मीटर} मायने रखता है किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे (\cref{def:g10:energy-conservation:kwh}):
$\qty{1}{kW.h} = \qty{3.6e6}{J}$. जूल भौतिक विज्ञानी हैं \omterm{def:g10:orders-of-magnitude:unit}{इकाई};
किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे बिलों का मूल्य तय करें.
\end{remark}

\section{जूल प्रभाव: टोस्टर से तोरण तक}

\begin{definition}[जूल प्रभाव]\label{def:g11:circuits-and-power:joule}
अपव्यय $ P = RI^2$ मोड़ \omterm{def:g10:energy-conservation:forms}{विद्युतीय ऊर्जा} हर में गर्मी में
अवरोधक है \emph{जूल प्रभाव}\index{जूल प्रभाव}. हीटर,
केतली, टोस्टर और पुरानी शैली के बल्ब डिज़ाइन द्वारा \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}ी हैं; ए
\emph{फ्यूज}\index{फ्यूज} एक है \omterm{def:g11:lenses-and-eye:lens}{पतला} तार को पिघलाने के लिए कैलिब्रेट किया गया --- खोलना
सर्किट --- जब करंट अपनी रेटिंग से अधिक हो जाता है।
\end{definition}

\begin{example}[तोरण ऊँचे क्यों चलते हैं?]\label{ex:g11:circuits-and-power:transmission}
एक गांव की जरूरत है $ P = \qty{100}{kW}$ की एक पंक्ति के माध्यम से \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}
$ R_\ell = \qty{5.0}{\ohm}$. पर सुपुर्दगी $ U = \qty{500}{V}$: $ I = P/U =
\qty{200}{A}$ और लाइन बर्बाद हो जाती है $ R_\ell I^2 = \qty{200}{kW}$ --- दो बार
वितरण: बेतुका. पर $ U = \qty{20}{kV}$: $ I = \qty{5.0}{A}$ और
$ R_\ell I^2 = \qty{125}{W}$. चालीस गुना कम वर्तमान, $1600$ कई गुना कम
हानि.
\end{example}

\begin{remark}[ग्रिड का लौह नियम]\label{rem:g11:circuits-and-power:grid}
निश्चित समय पर वितरित किया गया \omterm{def:g10:energy-conservation:power}{शक्ति} $ P = UI $, लाइन लॉस $ R_\ell I^2 = R_\ell
P^2/U^2$ के रूप में गिरता है \emph{वर्ग} परिवहन का \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}. इसलिए:
बदलना (\qty{400}{kV}) यात्रा करने के लिए, नीचे (\qty{230}{V}) साथ रहना।
\end{remark}

\begin{remark}[घरेलू सुरक्षा]\label{rem:g11:circuits-and-power:safety}
ए \qty{230}{V} लाइन आपस में जुड़ी हुई है \qty{16}{A} अधिकतम आपूर्ति $230 \times 16
\approx \qty{3.7}{kW}$: दो केतली एक साथ पिघलती हैं \omterm{def:g11:circuits-and-power:joule}{फ्यूज} ---डिज़ाइन द्वारा
सबसे पतली, सबसे कमज़ोर कड़ी, दीवार के अंदर लगी आग से भी कहीं सस्ती।
\end{remark}

\section{असली बैटरियां: इलेक्ट्रोमोटिव बल}

\begin{definition}[ईएमएफ और आंतरिक प्रतिरोध]\label{def:g11:circuits-and-power:emf}
एक बैटरी रसायन को परिवर्तित करती है \omterm{def:g10:energy-conservation:forms}{विद्युतीय ऊर्जा}. इसका
\emph{वैद्युतवाहक बल}\index{वैद्युतवाहक बल} (ईएमएफ)
$\mathcal{E}$, में \omterm{def:g11:circuits-and-power:voltage}{वोल्ट}, है ऊर्जा यह प्रत्येक को हाथ लगाता है कूलम्ब इसे पार करना;
इसका \emph{आंतरिक प्रतिरोध}\index{आंतरिक प्रतिरोध} $ r $ खाते
के लिए \omterm{def:g11:circuits-and-power:joule}{जूल प्रभाव} अपनी ही रसायन शास्त्र के अंदर. मॉडल: एक आदर्श स्रोत
$\mathcal{E}$ के साथ श्रृंखला में $ r $ (घुँघराले $\mathcal{E}$: $ E $ है ऊर्जा).
\end{definition}

\begin{proposition}[टर्मिनल का वोल्टेज]\label{prop:g11:circuits-and-power:terminal}
करंट देने वाली बैटरी $ I $ अपने टर्मिनलों पर दिखाता है
\[ U = \mathcal{E} - r\,I . \]
\end{proposition}

\begin{proof}
रसायन विज्ञान आपूर्ति करता है $\mathcal{E}I $, द \omterm{def:g11:circuits-and-power:emf}{आंतरिक प्रतिरोध} खाता है
$ rI^2$, सर्किट प्राप्त करता है $ UI $: \omterm{thm:g10:energy-conservation:conservation}{ऊर्जा संरक्षण}
(\cref{ch:g10:energy-conservation}) देता है $ UI = \mathcal{E}I - rI^2$.
\end{proof}

\begin{example}[लोड के तहत एक बैटरी]\label{ex:g11:circuits-and-power:discharge}
एक बैटरी ($\mathcal{E} = \qty{9.0}{V}$, $ r = \qty{1.0}{\ohm}$) फ़ीड ए
का बल्ब \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} $ R = \qty{8.0}{\ohm}$. पाश पालन करता है $\mathcal{E} =
rI + RI $, इसलिए $ I = \mathcal{E}/(R + r) = \qty{1.0}{A}$ और $ U = 9.0 -
1.0 \times 1.0 = \qty{8.0}{V}$: बल्ब प्राप्त करता है \qty{8.0}{W} जबकि
बैटरी की बर्बादी \qty{1.0}{W} खुद को गर्म करना.
\end{example}

\begin{remark}[रिसीवर]\label{rem:g11:circuits-and-power:receivers}
एक मोटर या चार्जिंग बैटरी रूपांतरण को पीछे की ओर चलाती है: यह अवशोषित करती है
के अंतर्गत $ U = \mathcal{E}' + r'I $, परिवर्तित करना $\mathcal{E}'I $ गति में या
रसायन शास्त्र और हार $ r'I^2$. स्रोत धक्का देते हैं, रिसीवर पीछे धकेलते हैं।
\end{remark}

\begin{example}[बैटरी की दक्षता]\label{ex:g11:circuits-and-power:efficiency}
 \omterm{def:g10:energy-conservation:efficiency}{क्षमता} (\cref{def:g10:energy-conservation:efficiency}) का
ऊपर निर्वहन: उपयोगी $ UI = \qty{8.0}{W}$ से बाहर $\mathcal{E}I =
\qty{9.0}{W}$, इसलिए $\eta = U/\mathcal{E} = 8/9 \approx 0.89$ --- गिराना
जैसे ही करंट चढ़ता है: कड़ी मेहनत करने वाली बैटरियां गर्म हो जाती हैं।
\end{example}

\begin{exercise}[$\star\star\star $]\label{exo:g11:circuits-and-power:13}

अवरोधक $ R = \qty{6.0}{\ohm}$.
\begin{enumerate}
  \item गणना करना $ I $, $ U $, उपयोगी \omterm{def:g10:energy-conservation:power}{शक्ति} में $ R $, द \omterm{def:g10:energy-conservation:power}{शक्ति} में खो गया $ r $,

  \item दिखाओ कि \omterm{def:g10:energy-conservation:power}{शक्ति} एक चर द्वारा प्राप्त किया गया $ R $,

        $(R+r)^2/R = (R-r)^2/R + 4r $), और उस अधिकतम की गणना करें।
\end{enumerate}
\end{exercise}

\section{समस्या: बांध से टोस्टर तक}

\begin{problem}[{वीकेंड की समस्या---हजारों किलोमीटर का सफर
किलोवाट-घंटा, पानी गिरने से लेकर ब्राउनिंग ब्रेड तक, पूरे बिल के साथ
यात्रा का}]
\label{pb:g11:circuits-and-power:1}
एक पहाड़ी बांध में पानी होता है \qty{200}{m} इसके टर्बाइनों के ऊपर, जो निगल जाते हैं
\qty{50}{m^3} --- \qty{5.0e4}{kg} ---प्रति सेकंड; टरबाइन और अल्टरनेटर
के साथ कनवर्ट करें \omterm{def:g10:energy-conservation:efficiency}{क्षमता} $0.90$. कुल की एक पंक्ति के माध्यम से \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}
$ R_\ell = \qty{32}{\ohm}$ (\qty{1000}{km} वहाँ और पीछे) पौधा खाता है
एक शहर जहां आपका \qty{900}{W} टोस्टर इंतज़ार कर रहा है \qty{230}{V}. द
अल्टरनेटर आउटपुट \qty{20}{kV}; ट्रांसफार्मर (\omterm{def:g10:energy-conservation:efficiency}{क्षमता} $0.99$ प्रत्येक)
परिवर्तित करें \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} इच्छानुसार। पर बिजली बिकती है $0.25$ यूरो प्रति
\unit{kW.h}; $ g = \qty{9.81}{N/kg}$.

\medskip
\noindent\textbf{Part I --- At the dam.}
\begin{enumerate}
  \item कितनी सम्भावना है ऊर्जा क्या गिरता हुआ पानी प्रत्येक को छोड़ता है?
        \omterm{def:g10:orders-of-magnitude:unit}{दूसरा}? निष्कर्ष निकालें \omterm{def:g10:energy-conservation:power}{शक्ति} टर्बाइनों के लिए उपलब्ध है।
  \item क्या \omterm{prop:g11:circuits-and-power:power}{विद्युत शक्ति} अल्टरनेटर छोड़ देता है?
  \item पर \qty{20}{kV} अल्टरनेटर से सीधे कौन सी धारा प्रवाहित होती है?
  \item जूल हानि की गणना करें $ R_\ell I^2$ उस वर्तमान पर और तुलना करें
        यह पौधे के आउटपुट के साथ है। निष्कर्ष.
  \item शहर का \omterm{def:g10:energy-conservation:power}{शक्ति} $ P = UI $ मांग द्वारा तय किया जाता है। एकमात्र क्या है
        सिकुड़ने का तरीका $ I $, और कौन सा उपकरण इसे निष्पादित करता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- On the line.}
\begin{enumerate}[resume]
  \item एक ट्रांसफार्मर, जिसे इस प्रश्न के लिए दोषरहित माना जाता है, कदम उठाता है
        \qty{88}{MW} तक \qty{400}{kV}: अब कौन सी धारा प्रवाहित होती है?
  \item नई लाइन लॉस की गणना मेगा\omterm{def:g10:energy-conservation:power}{वाट} में करें।
  \item संयंत्र के उत्पादन का कितना भाग लाइन में नष्ट हो जाता है?
  \item दिखाएँ कि खोया हुआ अंश बराबर है $ R_\ell P/U^2$, और उसकी जांच करें
        यह प्रश्न 4 और 7 के बीच के कारक की व्याख्या करता है।
  \item की गणना करें \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} बूँद $ R_\ell I $ पर लाइन के साथ
        \qty{400}{kV}. क्या यह प्रश्न 8 के अनुरूप है?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- In the kitchen.}
\begin{enumerate}[resume]
  \item बदलना \qty{1}{kW.h} को जूल; \omterm{def:g10:orders-of-magnitude:unit}{मीटर} क्या गुणा करता है?
  \item टोस्टर की धारा की गणना करें और \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} पर \qty{230}{V}.
  \item टोस्टिंग के तीन मिनट: ऊर्जा में जूल और
        किलो\omterm{def:g10:energy-conservation:power}{वाट} घंटे, और इसकी कीमत।
  \item टोस्टर नष्ट हो जाता है $ RI^2$: एक वाक्य में स्पष्ट करें कि क्यों
        एक ही फार्मूला है लाइन पर नुकसान, रसोई में उत्पाद।
  \item टोस्टर एक साझा करता है \qty{16}{A} \omterm{def:g11:circuits-and-power:joule}{फ्यूज} के साथ \qty{2.2}{kW}
        केतली: करता है \omterm{def:g11:circuits-and-power:joule}{फ्यूज} पकड़ना? ए \qty{2.0}{kW} हीटर जुड़ता है: और अब?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The audit.}
\begin{enumerate}[resume]
  \item घर की अपनी वायरिंग, \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध} \qty{0.05}{\ohm}, ले जाता है
        \qty{13.5}{A} प्रश्न 15 का: हानि और उसके अंश की गणना करें
        मोटे तौर पर \qty{3.1}{kW} पहुंचा दिया।
  \item दक्षताओं को श्रृंखलाबद्ध करें (टरबाइन--अल्टरनेटर, स्टेप-अप, लाइन,
        स्टेप-डाउन, हाउस वायरिंग) को एक समग्र ग्रिड में \omterm{def:g10:energy-conservation:efficiency}{क्षमता}.
  \item कितने \omterm{def:g10:orders-of-magnitude:unit}{किलोग्राम} (और घन एम-eters) बांध का पानी गिरना चाहिए
        \emph{एक} \omterm{def:g10:energy-conservation:kwh}{किलोवाट्ट घन्ता} अपने टोस्टर तक पहुँचने के लिए?
  \item खोए हुए अंश का मूल्यांकन करें $ R_\ell P/U^2$ प्रश्न 9 पर
        $ U = \qty{20}{kV}$. उपरोक्त मान का क्या मतलब है $1$ अर्थ?
  \item पंचलाइन --- \emph{दूरी की कीमत}: दो वाक्यों में,
        कोई क्या टोस्ट करता है \omterm{def:g10:energy-conservation:kwh}{किलोवाट्ट घन्ता} पहाड़ की कीमत, क्या
        अंश रास्ते में ही मर जाता है, और कौन सी एक युक्ति इसे संभव बनाती है?
\end{enumerate}
\end{problem}
