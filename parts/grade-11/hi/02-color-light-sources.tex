\chapter{रंग और प्रकाश स्रोत}\label{ch:g11:color-light-sources}

किसी भौतिक विज्ञानी से टमाटर का रंग पूछें और आपको एक प्रति-प्रश्न मिलेगा:
किस रोशनी में? केवल टमाटर \omterm{def:g11:color-light-sources:objectcolor}{फैलता है} यह क्या दिया जाता है, दीपक
केवल वही देता है जो उसका है \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} शामिल है, और आंख जो कुछ भी संपीड़ित करती है
तीन संख्या में आता है. रंग स्रोत के बीच एक समझौता है,
वस्तु और आँख --- यह अध्याय तीनों पक्षों से बारी-बारी से पूछताछ करता है।

\section{तीन संख्या में देख रहे हैं}

\begin{definition}[शंकु और त्रिवर्णीय दृष्टि]\label{def:g11:color-light-sources:cones}
 \omterm{def:g11:lenses-and-eye:eye}{रेटिना} (\cref{ch:g11:lenses-and-eye}) के दो परिवारों को ले जाता है
प्रकाश-संवेदनशील कोशिकाएँ: \emph{छड़}\index{छड़}, जो प्रतिक्रिया देता है
फीकी रोशनी लेकिन नज़रअंदाज़ करें \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य} (रात्रि दृष्टि, भूरे रंग में), और
\emph{कोन}\index{कोन}, तीन प्रकार में सबसे संवेदनशील पास
$\qty{420}{nm}$ (नीला), $\qty{530}{nm}$ (हरा) और $\qty{560}{nm}$
(लाल)। मानव रंग दृष्टि है
\emph{तीन रंगो}\index{तीन रंगो}: जो भी \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम},
मस्तिष्क को केवल तीन शंकु प्रतिक्रियाएँ प्राप्त होती हैं, और रंग ही वह है
उस त्रिगुण से बनता है।
\end{definition}

\begin{definition}[एक धारणा के रूप में रंग]\label{def:g11:color-light-sources:perception}
रंग इसलिए एक है \emph{धारणा}, अकेले प्रकाश की संपत्ति नहीं:
ए \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} प्रत्येक पर एक तीव्रता निर्दिष्ट करता है \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य}, नजर रखती है
तीन नंबर. अलग-अलग स्पेक्ट्रा वाली रोशनी लेकिन तीन समान \omterm{def:g11:color-light-sources:cones}{कोन}
प्रतिक्रियाएँ बिल्कुल एक जैसी दिखती हैं: वे हैं \emph{मेटामर्स}\index{मेटामर्स}.
\end{definition}

\begin{example}[दो पीले]\label{ex:g11:color-light-sources:yellows}
 $\qty{589}{nm}$ सोडियम लैंप की रेखा लाल और हरे रंग को उत्तेजित करती है
\omterm{def:g11:color-light-sources:cones}{कोन} लगभग समान रूप से: मस्तिष्क कहता है \emph{पीला}. एक स्क्रीन लाइटिंग
यह लाल है ($\qty{610}{nm}$) और हरा ($\qty{540}{nm}$) उपपिक्सेल देता है
वही पीला, पास में कोई रोशनी नहीं $\qty{589}{nm}$. एक प्रिज्म अलग हो जाता है
दो तुरन्त; आँख कभी नहीं कर सकती.
\end{example}

\section{रोशनी जोड़ना}

\begin{definition}[योगात्मक संश्लेषण]\label{def:g11:color-light-sources:additive}
सफ़ेद स्क्रीन पर प्रकाश किरणों को सुपरपोज़ करने से उनका स्पेक्ट्रा जुड़ जाता है: यह है
\emph{योगात्मक संश्लेषण}\index{योगात्मक संश्लेषण}. तीन अच्छी तरह से चुने गए
रोशनी, लाल, हरा और नीला --- \emph{additive प्राथमिक}\index{additive प्राथमिक} --- लगभग हर बोधगम्य में मिश्रण करें
रंग: आर $+$ जी $=$ पीला, आर $+$ बी $=$ मजेंटा, जी $+$ बी $=$ सियान, और
आर $+$ जी $+$ बी $=$ सफ़ेद।
\end{definition}

\begin{proposition}[सहायक रंग]\label{prop:g11:color-light-sources:complementary}
दो बत्तियाँ जिनका योग सफेद माना जाता है
\emph{पूरक}\index{पूरक}. तीन जोड़े
पीला/नीला, सियान/लाल और मैजेंटा/हरा \omterm{prop:g11:color-light-sources:complementary}{पूरक} हैं।
\end{proposition}

\begin{proof}
पीला R है $+$ जी; नीला जोड़ने से सफ़ेद का तिगुना पूरा हो जाता है। \omterm{def:g10:orders-of-magnitude:unit}{दूसरा}
दो जोड़े एक ही तर्क हैं, एक समय में एक प्राथमिक।
\end{proof}

\begin{example}[स्क्रीन]\label{ex:g11:color-light-sources:screen}
फ़ोन स्क्रीन का प्रत्येक पिक्सेल तीन उपपिक्सेल होता है --- लाल, हरा, नीला,
के अंतर्गत $\qty{0.1}{mm}$ चौड़ा --- आंख के अलग होने के लिए बहुत करीब, इसलिए
उनकी लाइटें चार चांद लगा देती हैं \omterm{def:g11:lenses-and-eye:eye}{रेटिना}: पीला R और G है, नारंगी R है
पूर्ण \omterm{def:g10:energy-conservation:power}{शक्ति} और जी आधे पर, सफेद तीनों हैं। एक पर एक आवर्धक कांच
सफेद क्षेत्र से चाल का पता चलता है: कहीं भी सफेद नहीं, केवल आर, जी, बी।
\end{example}

\section{श्वेत प्रकाश से घटाना}

\begin{definition}[घटाव संश्लेषण]\label{def:g11:color-light-sources:subtractive}
ए \emph{फ़िल्टर}\index{फ़िल्टर} (रंगीन कांच, जेल, स्याही की परत या)
पेंट) का हिस्सा संचारित करता है \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} यह शेष को प्राप्त और अवशोषित करता है।
द्वारा रंगों का उत्पादन \emph{को हटाने} बैंड से \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} है
\emph{घटाव संश्लेषण}\index{घटाव संश्लेषण}. यह
प्राथमिक, योगात्मक लोगों के \omterm{prop:g11:color-light-sources:complementary}{पूरक}, प्रत्येक एक बैंड घटाते हैं:
सियान $=$ सफ़ेद $-$ आर, मैजेंटा $=$ सफ़ेद $-$ जी, पीला $=$ सफ़ेद $-$ बी.
स्टैक्ड फिल्टर संयुक्त रूप से घटाते हैं: सियान $+$ पीले पत्ते हरे, सियान
$+$ मैजेंटा नीला, मैजेंटा $+$ पीला लाल, तीनों काले। मुद्रण
और पेंटिंग इस तरह से काम करती है, सी, एम, वाई स्याही के साथ।
\end{definition}

\begin{method}[स्पेक्ट्रम बहीखाता पद्धति]\label{met:g11:color-light-sources:bookkeeping}
स्रोतों की किसी भी श्रृंखला के माध्यम से रंग की भविष्यवाणी करने के लिए, फिल्टर और वस्तुएं,
तीन प्राथमिक बैंडों को ट्रैक करें: स्रोत के प्रकाश को उसकी सामग्री के रूप में लिखें
आर, जी, बी में; हरेक पर \omterm{def:g11:color-light-sources:subtractive}{फ़िल्टर} या सतह, अवशोषित बैंड को हटा दें; नाम
जो बचता है उसका रंग (कुछ भी नहीं बचा)। $=$ काला)।
\end{method}

\begin{example}[एक फिल्टर श्रृंखला]\label{ex:g11:color-light-sources:chain}
\omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी} $(R, G, B)$ एक पीले रंग को पार करता है \omterm{def:g11:color-light-sources:subtractive}{फ़िल्टर}: नीला बैंड है
अवशोषित, छोड़ना $(R, G)$ --- पीला। एक सियान \omterm{def:g11:color-light-sources:subtractive}{फ़िल्टर} फिर अवशोषित कर लेता है
लाल बैंड, जा रहा है $(G)$: हरा। स्वैपिंग फिल्टर कुछ भी नहीं बदलता:
आदेश की परवाह किए बिना प्रत्येक अपने स्वयं के बैंड को अवशोषित करता है।
\end{example}

\section{किसी वस्तु का रंग}

\begin{definition}[प्रसार और वस्तु का रंग]\label{def:g11:color-light-sources:objectcolor}
एक अपारदर्शी वस्तु प्राप्त प्रकाश का कुछ भाग अवशोषित कर लेती है
\emph{फैलता है}\index{फैलता है} बाकी --- इसे प्रत्येक में पुनः उत्सर्जित करता है
दिशा, दूसरों के बीच हमारी आँखों की ओर। \emph{एक का रंग वस्तु}\index{एक का रंग वस्तु} यह फैलने वाले प्रकाश का रंग है
\emph{दी गई रोशनी के तहत}: वस्तु अपना अवशोषण प्रदान करती है
आदतें, स्रोत कच्चा माल। दीपक बदलो, रंग बदलो.
\end{definition}

\begin{example}[लाल शर्ट]\label{ex:g11:color-light-sources:redshirt}
एक शर्ट दिन के उजाले में लाल दिखती है: यह हरे और नीले बैंड को अवशोषित कर लेती है
\omterm{def:g11:color-light-sources:objectcolor}{फैलता है} एक लाल वाला। शुद्ध हरे प्रकाश के नीचे कोई लाल पट्टी नहीं होती
फैलाना: शर्ट सब कुछ सोख लेती है और काली दिखाई देती है। मजेंटा के तहत
रोशनी $(R, B)$ यह \omterm{def:g11:color-light-sources:objectcolor}{फैलता है} अकेले लाल बैंड: अभी भी लाल. एक सफ़ेद
वस्तु \omterm{def:g11:color-light-sources:objectcolor}{फैलता है} प्रत्येक बैंड इसे प्राप्त होता है (हरी रोशनी के तहत यह दिखता है
हरा); एक काली वस्तु प्रत्येक प्रकाश के अंतर्गत उन सभी को अवशोषित कर लेती है --- जो कि है
काले कपड़े धूप में क्यों गर्म होते हैं?
\end{example}

\section{प्रकाश स्रोत और रंग तापमान}

\begin{definition}[गरमागरम स्रोत]\label{def:g11:color-light-sources:incandescent}
एक स्रोत चमक रहा है क्योंकि यह गर्म है --- ज्वाला, रेशा, तारा --- चमकता है
द्वारा \emph{जोश}\index{जोश}. इसका \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} है
निरंतर थर्मल \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} का \cref{ch:g10:light-spectra}, और इसकी चरम सीमा
वीन के नियम का पालन करता है (\cref{prop:g10:light-spectra:wien}),
\[
\lambda_{\max} \times T = \qty{2.90e-3}{m.K},
\]
अब से हमारा मात्रात्मक उपकरण: अधिकतम माप, तापमान ज्ञात।
\end{definition}

\begin{example}[एक बहुत ही अकुशल दीपक]\label{ex:g11:color-light-sources:bulb}
एक टंगस्टन फिलामेंट $ T = \qty{2700}{K}$ शिखर पर
$\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{2700}{K} = \qty{1.07e-6}{m}
= \qty{1070}{nm}$, गहराई में \omterm{def:g10:light-spectra:wavelength}{अवरक्त}: अधिकांश विद्युत \omterm{def:g10:energy-conservation:power}{शक्ति}
अदृश्य विकिरण के रूप में निकलता है, केवल के बारे में $5\%$ प्रकाश के रूप में ---
गरमागरम बल्ब \omterm{def:g10:refraction:fiber}{मुख्य} रूप से एक हीटर था।
\end{example}

\begin{definition}[स्पेक्ट्रल लैंप और लेजर]\label{def:g11:color-light-sources:lamps}
ए \emph{वर्णक्रमीय दीपक}\index{वर्णक्रमीय दीपक} --- ए कम \omterm{def:g10:pressure:pressure}{दबाव} गैस
एक निर्वहन से उत्साहित: सोडियम, पारा, नियॉन --- उत्सर्जित करता है \omterm{def:g10:light-spectra:emission}{लाइन स्पेक्ट्रम}
(\cref{ch:g10:light-spectra}): कुछ पृथक \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य}, अंदर कुछ भी नहीं
के बीच. ए \emph{लेज़र}\index{लेज़र} आगे बढ़ता है: एक एकल
\omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य}, एकवर्णी प्रकाश, एक संकीर्ण किरण में।
\end{definition}

\begin{definition}[एल ई डी]\label{def:g11:color-light-sources:led}
ए \emph{प्रकाश उत्सर्जक डायोड}\index{प्रकाश उत्सर्जक डायोड} (एलईडी) उत्सर्जित करता है
\emph{अर्ध एकरंगा}\index{अर्ध एकरंगा} कुछ दसियों नैनो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} चौड़ा बैंड: संकीर्ण
एक निश्चित रंग देखने के लिए पर्याप्त, एक से कहीं अधिक व्यापक \omterm{def:g11:color-light-sources:lamps}{लेज़र} रेखा। एक सफ़ेद
एलईडी एक नीली एलईडी है (शीर्ष के पास)। $\qty{450}{nm}$) के साथ लेपित
\emph{भास्वर}\index{भास्वर}, नीले रंग के भाग को अवशोषित करने वाला पदार्थ
और इसे एक चौड़ी पीली पट्टी के रूप में पुनः उत्सर्जित कर रहा है: नीला और पीला होना
\omterm{prop:g11:color-light-sources:complementary}{पूरक} (\cref{prop:g11:color-light-sources:complementary}), द
मिश्रण सफेद के रूप में पढ़ता है।
\end{definition}

\begin{definition}[रंग तापमान]\label{def:g11:color-light-sources:colortemp}
 \emph{रंग तापमान}\index{रंग तापमान} एक सफ़ेद रंग का
स्रोत गरमागरम वस्तु का तापमान है जिसका रंग उसकी रोशनी है
माचिस: मोमबत्ती $\qty{1800}{K}$, गरमागरम बल्ब $\qty{2700}{K}$,
हलोजन $\qty{3400}{K}$, दोपहर का सूरज $\qty{5500}{K}$, घटाटोप दिन का उजाला
$\qty{6500}{K}$. यह एक रंग लेबल है, थर्मा\omterm{def:g10:orders-of-magnitude:unit}{मीटर} रीडिंग नहीं --- ए
$\qty{6500}{K}$ एलईडी कमरे के तापमान पर चलती है --- और शब्दावली चलती है
पीछे की ओर: कम रंग का तापमान ``गर्म'' नारंगी, उच्च ``ठंडा'' पढ़ता है
नीला।
\end{definition}

\section{समस्या: इंजीनियरिंग रंग}

\begin{problem}[{सप्ताहांत समस्या--- स्टेज लाइट, सोडियम सड़कें और
सफेद एलईडी: तीन स्पॉटलाइट से रंग कैसे इंजीनियर किया जा सकता है
इस पृष्ठ को रोशन करने वाली चिप पर कोई भी रंग पेंट करें}]
\label{pb:g11:color-light-sources:1}
एक प्रकाश डिजाइनर के पास केवल लाल, हरे और नीले स्पॉटलाइट होते हैं; एक सड़क
इंजीनियर के पास एक लैंप है जो एक सिंगल उत्सर्जित करता है \omterm{def:g10:light-spectra:wavelength}{तरंग दैर्ध्य}; एक चिप निर्माता का मालिक है
एक डायोड जो केवल नीला चमकता है। तीनों रंग बेचते हैं। यह समस्या चलती है
बदले में उनके व्यापार --- मंच पर योगात्मक मिश्रण, मोनोक्रोम अर्थव्यवस्था में
सड़क, \omterm{def:g11:color-light-sources:led}{भास्वर} चिप में कीमिया --- और चुनने से समाप्त होता है
शयनकक्ष के लिए और जौहरी की खिड़की के लिए प्रकाश।

\medskip
\noindent\textbf{Part I --- The stage: painting with three spotlights.}
\begin{enumerate}
  \item तीन धब्बे एक सफेद पृष्ठभूमि पर ओवरलैप होते हैं। के रंग का नाम बताएं
        प्रत्येक जोड़ीवार ओवरलैप और ट्रिपल ओवरलैप।
  \item डिज़ाइनर नारंगी कैसे बनाता है? और हल्का गुलाबी? (वर्णन करें
        स्थान पॉवर्स गुणात्मक रूप से।)
  \item एक नर्तकी पृष्ठभूमि के सामने खड़ी है, जो लाल और हरे रंग से जगमगा रही है
        केवल धब्बे. दो रंगीन छायाएँ दिखाई देती हैं: प्रत्येक का रंग? व्याख्या करना।
  \item पृष्ठभूमि गहरे नीले रंग में रंगी गई है। यह कैसा दिखता है
        अकेले लाल धब्बे के नीचे? तीनों के अंतर्गत?
  \item तीन धब्बे ``लगभग हर'' रंग को पुन: उत्पन्न क्यों कर सकते हैं? जो
        शरीर रचना विज्ञान का टुकड़ा संख्या तीन तय करता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Costumes: subtraction at work.}
\begin{enumerate}[resume]
  \item एक पोशाक पीले रंग की है \omterm{def:g10:light-spectra:white}{सफ़ेद रोशनी}. कौन सा बैंड इसकी रंगाई करता है
        अवशोषित?
  \item लाल धब्बे, हरे धब्बे, नीले के नीचे इसके रंग का अनुमान लगाएं
        स्थान.
  \item डिज़ाइनर चाहता है कि पोशाक चमकीली से काली हो जाए
        स्विच का एक झटका. फ़्लिक किस स्थान पर स्विच होता है?
  \item एक मैजेंटा जेल को प्रकाश देने वाले सफेद स्थान के सामने सरकाया जाता है
        पीली पोशाक. बैंड को ट्रैक करें
        (\cref{met:g11:color-light-sources:bookkeeping}): पोशाक का रंग?
  \item स्टेज जैल सियान, मैजेंटा और पीले रंग में क्यों आते हैं?
        लाल, हरा और नीला? (लाल जेल सफेद दाग पर क्या करेगा
        \omterm{def:g10:energy-conservation:power}{शक्ति}?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Sodium streets: the monochrome economy.}
\begin{enumerate}[resume]
  \item ए कम \omterm{def:g10:pressure:pressure}{दबाव} सोडियम लैंप मूलतः एक लाइन उत्सर्जित करता है
        $\qty{589}{nm}$. किस प्रकार का \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} यह क्या है, और क्या करता है
        पूरी सड़क इसके नीचे दिखती है?
  \item इस दीपक के नीचे, एक सफेद दीवार का स्पष्ट रंग दें, ए
        पीली सड़क चिह्न, एक लाल कार, एक नीली कार।
  \item खगोलविदों को ये लैंप क्यों पसंद आए? (कोई क्या कर सकता है \omterm{def:g11:color-light-sources:subtractive}{फ़िल्टर} करो
        पूरे शहर की चमक?)
  \item एक टंगस्टन स्ट्रीट बल्ब $\qty{2700}{K}$ इसके स्थान पर प्रस्तावित है।
        इसकी गणना करें $\lambda_{\max}$: इसका अधिकांश हिस्सा कहां है \omterm{def:g10:energy-conservation:power}{शक्ति} जाना?
  \item सोडियम लैंप लगभग परिवर्तित हो जाता है $30\%$ इसकी बिजली का \omterm{def:g10:energy-conservation:power}{शक्ति}
        दृश्य प्रकाश में, टंगस्टन बल्ब के बारे में $5\%$. कितने
        \omterm{def:g10:energy-conservation:power}{वाट} प्रत्येक प्रकाश का प्रति प्रदान करता है $\qty{100}{W}$ का \omterm{def:g10:energy-conservation:power}{शक्ति}?
\end{enumerate}

        \omterm{def:g11:color-light-sources:led}{भास्वर}. समझाओ, साथ में सहायक रंग, इसका आउटपुट क्यों

        \omterm{def:g10:light-spectra:white}{स्पेक्ट्रम} दिन के उजाले का. किस बैंड का प्रतिनिधित्व कम है?
