\begin{solution}{exo:g11:lenses-and-eye:2}
Diverging ($f' < 0$). No: for any object distance $d$,
$\gamma = f'/(f' - d)$ (shown in \cref{exo:g11:lenses-and-eye:15})
lies between $0$ and $1$ --- the image is
always virtual, upright and \emph{reduced}. The print appears upright
and shrunk, never magnified.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:3}
The object sits at $2f'$; the central and parallel rays cross
\qty{20}{cm} behind the lens. Check:
$1/\overline{OA'} = 1/10 - 1/20 = 1/20$, so
$\overline{OA'} = +\qty{20}{cm}$ and $\gamma = 20/(-20) = -1$: real,
inverted, life-size --- as drawn.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:4}
$1/\overline{OA'} = 1/8.0 - 1/24 = 1/12$:
$\overline{OA'} = +\qty{12}{cm}$, $\gamma = 12/(-24) = -0.50$. Real
image \qty{12}{cm} behind the lens, inverted, half-size.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:5}
Cornea and crystalline lens form the converging lens; the retina is
the screen. Accommodation changes the focal length (vergence) of the
crystalline lens; the lens--retina distance cannot change. Near
point: closest point seen sharply at full accommodation ---
\qty{25}{cm} for the standard eye.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:6}
Landscape: sensor in the focal plane, \qty{50}{mm} from the lens. At
\qty{3.0}{m}: $1/\overline{OA'} = 1/50 - 1/3000 = 59/3000$, so
$\overline{OA'} \approx \qty{50.8}{mm}$. Travel: about \qty{0.8}{mm}.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:7}
$1/\overline{OA'} = 1/6.0 - 1/5.0 = -1/30$:
$\overline{OA'} = \qty{-30}{cm}$ --- virtual, upright,
$\gamma = (-30)/(-5.0) = +6.0$: six times the stamp, \qty{30}{cm}

\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:8}
$1/f' = 1/40 + 1/40 = 1/20$: $f' = \qty{20}{cm}$;
$\gamma = 40/(-40) = -1$. Symmetry forces
$\abs{\overline{OA'}} = \abs{\overline{OA}}$, hence
$\abs{\gamma} = 1$; the image is real and inverted, so $\gamma = -1$:
exactly life-size.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:9}
$1/\overline{OA'} = -1/25 - 1/50 = -3/50$:
$\overline{OA'} \approx \qty{-16.7}{cm}$,
$\gamma = (-16.7)/(-50) = +0.33$: virtual, upright, one third of the
size. A myope's glasses are diverging: everything seen through them
--- including, from outside, the wearer's eyes --- is reduced.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:10}
Image of infinity at the far point: $f' = \qty{-0.40}{m}$, so
$C = -2.5$~D. The mountain's image sits \qty{40}{cm} in front of the
lens --- exactly the far point, which the relaxed eye sees sharply
with zero accommodation.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:12}
\emph{1.} $\gamma = \overline{OA'}/\overline{OA} = -50$ gives
$\overline{OA'} = -50\,\overline{OA}$; with
$\overline{OA'} - \overline{OA} = \qty{5.1}{m}$,
$-51\,\overline{OA} = \qty{5.1}{m}$: $\overline{OA} = \qty{-0.10}{m}$,
$\overline{OA'} = +\qty{5.0}{m}$.

\emph{2.} $1/f' = 1/5.0 + 1/0.10 = 10.2$: $f' \approx \qty{9.8}{cm}$.

\emph{3.} $50 \times \qty{24}{mm} = \qty{1.2}{m}$, inverted --- which
is why slides are loaded upside down.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:13}
\emph{1.} $C_\infty = 1/0.017 \approx 59$~D;
$C_{25} = 1/0.017 + 1/0.25 \approx 63$~D. Amplitude:
$1/0.25 = 4.0$~D.

\emph{2.} Amplitude $= 1/1.0 = 1.0$~D: one quarter ($25\%$) of the
standard eye's $4.0$~D remains.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:14}
\emph{1.} $f' = 0.25/3 \approx \qty{0.083}{m} = \qty{8.3}{cm}$.

\emph{2.} $\overline{OA'} = \qty{-25}{cm}$:
$1/\overline{OA} = -1/25 - 3/25 = -4/25$, so
$\overline{OA} = \qty{-6.25}{cm}$ and
$\gamma = (-25)/(-6.25) = +4.0$ --- better than the advertised
$\times 3$, which assumes the image at infinity; pulling it to the
near point gains one unit.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:15}
$1/\overline{OA'} = 1/f' - 1/d = (d - f')/(f'd)$, so
$\overline{OA'} = f'd/(d - f')$ and
$\gamma = \overline{OA'}/(-d) = f'/(f' - d)$. If $d > f'$:
$\overline{OA'} > 0$ (real) and $\gamma < 0$ (inverted). If
$0 < d < f'$: $\overline{OA'} < 0$ (virtual) and
$\gamma = f'/(f' - d) > 1$: upright and enlarged --- the magnifier
setting.
\end{solution}

\begin{solution}{pb:g11:lenses-and-eye:1}
\textbf{1.} $1/\overline{OA'} = 1/20 - 1/60 = 1/30$:
$\overline{OA'} = +\qty{30}{cm}$, $\gamma = -0.50$ --- real, inverted,
half-size.

\textbf{2.} $\overline{OA'} = +\qty{60}{cm}$, $\gamma = -2$: the
positions \qty{30}{cm} and \qty{60}{cm} exchanged roles, the
magnifications are reciprocal --- reversed light follows the same
path.

\textbf{3.} $1/\overline{OA'} = 1/20 - 1/10 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, $\gamma = +2$ --- virtual, upright,
doubled.

\textbf{4.} From infinity the image starts in the focal plane; as the
object approaches $F$ the real, inverted image recedes to infinity and
grows; inside $f'$ it turns virtual, upright, enlarged, in front of
the lens.

\textbf{5.} $1/\overline{OA} \approx 0$ leaves
$1/\overline{OA'} = 1/f'$: $\overline{OA'} = f'$, the focal plane.

\textbf{6.} In the focal plane, \qty{50}{mm} behind the lens.

\textbf{7.} $1/\overline{OA'} = 1/50 - 1/1000 = 19/1000$:
$\overline{OA'} \approx \qty{52.6}{mm}$; travel $\approx \qty{2.6}{mm}$.

\textbf{8.} $1/\overline{OA} = 1/55 - 1/50 = -1/550$: minimum focusing
distance \qty{550}{mm} = \qty{55}{cm}.

\textbf{9.} $\gamma = 55/(-550) = -0.10$: the face images at
\qty{20}{mm}, just inside the \qty{24}{mm} sensor.

\textbf{10.} $\gamma = -1$ forces $\overline{OA'} = -\overline{OA}$,
so $2/\overline{OA'} = 1/f'$: object \qty{100}{mm} ($= 2f'$) in front,
sensor \qty{100}{mm} behind --- double the landscape draw, far beyond
the \qty{55}{mm} mechanism; macro lenses provide the extra travel.

\textbf{11.} Anything nearer than her near point (\qty{1.0}{m}) blurs:
stretched arms bring the page as close to \qty{1.0}{m} as possible ---
sharp, if small.

\textbf{12.} $C = 1/(-1.0) - 1/(-0.25) = -1.0 + 4.0 = +3.0$~D.

\textbf{13.} $\gamma = (-1.0)/(-0.25) = +4.0$: four times larger but
four times farther --- the same angular size; the gain is not size but
\emph{sharpness}, since the image now lies at her near point.

\textbf{14.} Sharp needs the image beyond \qty{1.0}{m}: from the page
at \qty{25}{cm} (image at \qty{1.0}{m}) to the page at
$f' = 1/3.0 \approx \qty{0.33}{m}$ (image at infinity). A distant
object images \qty{33}{cm} \emph{behind} the glasses --- a real image
the eye cannot use: the distant world blurs.

\textbf{15.} Hence bifocals: reading power in the lower half of the
lens, none (or the distance correction) in the upper half.

\textbf{16.} $1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, virtual, upright, $\gamma = +5.0$.

\textbf{17.} $1/\overline{OA} = -1/25 - 1/5.0 = -6/25$:
$\overline{OA} \approx \qty{-4.2}{cm}$; $\gamma = (-25)/(-25/6) = +6.0$.

\textbf{18.} Stone at $F$: image at infinity, viewed with zero
accommodation --- hours of inspection without eye strain.

\textbf{19.} $C_\infty = 1/0.017 \approx 59$~D;
$C_{25} = 1/0.017 + 1/0.25 \approx 63$~D: about $4$~D of
accommodation.

\textbf{20.} $f' = 1/58.8 \approx \qty{17.0}{mm}$ against
$1/62.8 \approx \qty{15.9}{mm}$: about \qty{1.1}{mm}, some $6\%$ ---
the built-in zoom the camera imitated with \qty{2.6}{mm} of travel and
grandmother patched with $+3$~D.
\end{solution}
