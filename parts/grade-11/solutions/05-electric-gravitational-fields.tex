\begin{solution}{exo:g11:electric-gravitational-fields:8}
\emph{1.} $E = \num{5.0e3} / \num{2.0e-2} = \qty{2.5e5}{V/m}$; at rest

$q = \num{2.0e-6} \times 9.81 / \num{2.5e5} \approx \qty{7.8e-11}{C}$.

\emph{2.} The \omterm{def:g10:inertia:force}{force} on the positive grain must point up, so $\vect E$

\emph{3.} $n = \num{7.85e-11} / \num{1.60e-19} \approx \num{4.9e8}$

\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:9}
Outside the segment both \omterm{def:g11:electric-gravitational-fields:field}{fields} point the same way; between, they
oppose --- and balance needs the point nearer the weaker source $+Q$.
With $x$ the distance to $+Q$: $kQ/x^2 = 4kQ/(0.30 - x)^2$, so
$0.30 - x = 2x$ and $x = \qty{0.10}{m}$ from $+Q$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:10}

\begin{solution}{exo:g11:electric-gravitational-fields:12}
$GM/(R+h)^2 = \tfrac12\,GM/R^2$ gives $R + h = R\sqrt{2}$, so
$h = (\sqrt{2} - 1)R \approx \qty{2.6e6}{m}$ (about \qty{2600}{km}). At
$h = R$, $d = 2R$: one quarter of the surface value.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:13}
\emph{1.} $AM = \sqrt{3.0^2 + 4.0^2} = \qty{5.0}{cm}$; each charge
creates $E_1 = \num{8.99e9} \times \num{2.0e-9} / 0.050^2
\approx \qty{7.2e3}{N/C}$.

\emph{2.} The components along $AB$ cancel by symmetry; each \omterm{def:g11:electric-gravitational-fields:field}{field}
contributes $E_1 \times 4/5$ along the bisector, away from the segment:
$E = 2 \times \num{7.19e3} \times 0.80 \approx \qty{1.2e4}{N/C}$,
directed along the bisector, away from $[AB]$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:14}
The box's free charges move under the external \omterm{def:g11:electric-gravitational-fields:field}{field}: $+$ piles up on
one face, $-$ on the other, and these displaced charges create an
internal \omterm{def:g11:electric-gravitational-fields:field}{field} opposing the external one. They keep moving until the
total \omterm{def:g11:electric-gravitational-fields:field}{field} inside is exactly zero --- then equilibrium. Gravity offers
only one sign of source: every mass \emph{adds} an attracting \omterm{def:g11:electric-gravitational-fields:field}{field},
none can oppose it. A gravity shield would need negative mass, which
does not exist.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{Weight} $m\vect g$ (down), \omterm{def:g10:inertia:inventory}{tension} $\vect T$ (along the
thread), electric \omterm{def:g10:inertia:force}{force} $q\vect E$ (horizontal). At rest,
$T\sin\theta = qE$ and $T\cos\theta = mg$; divide:
$\tan\theta = qE/(mg)$.

\emph{2.} $\tan\theta = \num{2.0e-7} \times \num{1.0e4} /
(\num{5.0e-4} \times 9.81) \approx 0.41$, so $\theta \approx \ang{22}$;
$T = mg/\cos\theta \approx \qty{5.3e-3}{N}$.
\end{solution}

\begin{solution}{pb:g11:electric-gravitational-fields:1}
\textbf{1.} \omterm{def:g11:electric-gravitational-fields:capacitor}{Uniform}, perpendicular to the plates, pointing down (from
the $+$ upper plate to the $-$ lower one); uniformity means the \omterm{def:g10:inertia:force}{force}
on the droplet is the same wherever it drifts.
\textbf{2.} $E = U/d = 800 / \num{8.0e-3} = \qty{1.0e5}{V/m}$.
\textbf{3.} The charge is negative, so the \omterm{def:g10:inertia:force}{force} is \emph{opposite}
$\vect E$: upward. Only with the $+$ plate on top does $\vect E$ point
down and the electric \omterm{def:g10:inertia:force}{force} fight the \omterm{def:g10:universal-gravitation:weight}{weight}.
\textbf{4.} $P = mg = \num{4.90e-15} \times 9.81 \approx
\qty{4.8e-14}{N}$.
\textbf{5.} $\abs{q}E = mg$.
\textbf{6.} $\abs{q} = mg/E = mgd/U$.
\textbf{7.} $\abs{q} = \num{4.81e-14} \times \num{8.0e-3} / 800
\approx \qty{4.8e-19}{C}$.
\textbf{8.} $n = \num{4.8e-19} / \num{1.60e-19} = 3$ excess electrons.
\textbf{9.} $E$ grows, so $\abs{q}E > mg$: the net \omterm{def:g10:inertia:force}{force} points up.
\textbf{10.} From $\abs{q}\,U/d = mg$, $U = mgd/\abs{q}$ --- inversely
proportional to the charge.
\textbf{11.} $\abs{q} = mgd/U = \num{3.85e-16}/1200 \approx
\qty{3.2e-19}{C} = 2e$: down from $3e$, the droplet \emph{lost} one
electron.
\textbf{12.} $\abs{q} = \num{3.85e-16}/U$: \qty{3.2e-19}{C},
\qty{4.8e-19}{C}, \qty{6.4e-19}{C}, \qty{8.0e-19}{C},
\qty{9.6e-19}{C}.
\textbf{13.} They are $2e, 3e, 4e, 5e, 6e$ with
$e = \qty{1.60e-19}{C}$.
\textbf{14.} Each step changes $\abs{q}$ by exactly
\qty{1.6e-19}{C}: one electron lands at a time.
\textbf{15.} It would need $\abs{q} = \num{3.85e-16}/700 \approx
\qty{5.5e-19}{C} = 3.4\,e$ --- not a whole number of electrons, so no
burst of radiation can ever produce it.
\textbf{16.} Charge is \emph{quantized}: it comes in integer multiples
of the \omterm{def:g11:fundamental-interactions:elementary}{elementary charge} $e = \qty{1.60e-19}{C}$ --- the quantity
Millikan measured.
\textbf{17.} $q \leftrightarrow m$, $E \leftrightarrow g$, and
$\abs{q}E \leftrightarrow mg$: the \omterm{def:g10:universal-gravitation:weight}{weight} itself.
\textbf{18.} $g_{\text{sphere}} = GM/d^2 = \num{6.67e-11} \times 1000 /
1.0^2 = \qty{6.7e-8}{N/kg}$; \omterm{def:g10:inertia:force}{force} on the droplet
$\num{4.90e-15} \times \num{6.7e-8} \approx \qty{3.3e-22}{N}$ --- about
$\num{1.5e8}$ times smaller than its \omterm{def:g10:universal-gravitation:weight}{weight}. Hopeless.
\textbf{19.} Because $G$ is so small, only a planet-sized mass overhead
could rival the Earth's pull --- and no mass repels. The enormous $k$
and the two signs of charge let a bench-top plate out-pull the planet
in either direction.
\textbf{20.} One electron: $\Delta\abs{q}/\abs{q} = e/3e \approx 33\%$;
one oil \omterm{def:g11:fundamental-interactions:ladder}{molecule}: $\Delta m/m = \num{5e-25} / \num{4.90e-15}
\approx \num{e-10}$. The charge staircase moves the balancing \omterm{def:g11:circuits-and-power:voltage}{voltage}
by hundreds of \omterm{def:g11:circuits-and-power:voltage}{volts}; the mass staircase is invisible at any \omterm{def:g11:circuits-and-power:voltage}{voltage}.
The balance therefore resolves the \omterm{def:g11:fundamental-interactions:ladder}{atom} of electricity ---
$e = \qty{1.60e-19}{C}$ --- while the \omterm{def:g11:fundamental-interactions:ladder}{molecules} of oil blur into a
continuum.
\end{solution}
