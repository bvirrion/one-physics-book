\section*{Hoofdstuk \ref{ch:g11:forces-and-motion} --- Krachten en beweging}

\begin{solution}{exo:g11:forces-and-motion:1}
(a) Constante snelheid: \omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} nul. (b) Snelheid
verandert: niet-nul, tegengesteld aan de beweging. (c) Richting
verandert (snelheid is een vector): niet-nul, naar de binnenkant van
de bocht. (d) Constante snelheid: nul ---
\omterm{def:g10:universal-gravitation:weight}{gewicht} en weerstand
\omterm{def:g10:inertia:compensate}{compenseren}.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:2}
$d_1/d_2 = 1.0/3.0$ met $d_1 + d_2 = \qty{0.80}{m}$:
$d_1 = \qty{0.20}{m}$ van de \qty{3.0}{kg}-bal. Gelijke massa's: het
midden, \qty{0.40}{m} van elk.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:3}
Zelfde $F\Delta t$, driedubbele massa: $\Delta v$ driemaal kleiner
geladen. Om de $\Delta v$ van de lege trolley te evenaren, duw driemaal
harder (of driemaal langer).
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:4}
$\Delta \vect v$: \qty{2}{m/s}, \emph{achterwaarts} gericht (van
\num{10} naar \qty{8}{m/s}). De
\omterm{def:g11:forces-and-motion:netforce}{resulterende kracht} wijst dezelfde
kant op: tegengesteld aan $\vect v$ --- geval 2, vertragen.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:5}
(a) Statisch, tegengesteld aan je duw (nog geen glijden). (b) Kinetisch,
tegengesteld aan het glijden. (c) Statisch, \emph{omhoog} de helling ---
tegen de glij die zwaartekracht probeert te starten.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:6}
$\Delta v \propto F\Delta t/m$: (a) \qty{8.0}{m/s}; (b) \qty{2.0}{m/s};
(c) \qty{2.0}{m/s}; (d) $4.0 \times \frac{3 \times 2}{6} =
\qty{4.0}{m/s}$ --- de drie veranderingen heffen op.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:7}
$v' = \sqrt{8.0^2 + 6.0^2} = \qty{10.0}{m/s}$, bij $\tan\theta =
6.0/8.0$, d.w.z.\ $\theta \approx 37^\circ$ noord van oost. Snelheden
tellen op als \emph{vectoren}: loodrechte bijdragen combineren via
Pythagoras, niet door optelling van groottes.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:8}
Vooraf vast: de massa van de bal en haar $\Delta v$ (inkomende snelheid
tot nul), vandaar het product $F\Delta t$. $\Delta t$ strekken van
\qty{0.02}{s} tot \qty{0.08}{s} deelt de gemiddelde
\omterm{def:g10:inertia:force}{kracht} door $4$.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:9}
De statische \omterm{def:g10:inertia:inventory}{wrijving} van de baan op de
schoen: de voet duwt achterwaarts op de grond, en de wederzijdse
\omterm{def:g10:inertia:contact}{contactkracht} duwt de sprinter vooruit.
Spikes verhogen de maximale statische
\omterm{def:g10:inertia:inventory}{wrijving} (geen slip bij volle inspanning).
Op wrijvingsloos ijs bestaat geen uitwendige horizontale
\omterm{def:g10:inertia:force}{kracht}, dus de snelheid van het
\omterm{def:g11:forces-and-motion:com}{zwaartepunt} kan niet veranderen ---
geen start, geen wandelen.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:10}
Het contactpunt van een rollend wiel glijdt niet: statische
\omterm{def:g10:inertia:inventory}{wrijving}, sterker dan de
\omterm{def:g11:forces-and-motion:friction}{kinetische wrijving} van een slip
(\cref{def:g11:forces-and-motion:friction}) --- kortere stop. De
\omterm{def:g11:forces-and-motion:friction}{kinetische wrijving} van een
slippende band wijst tegengesteld aan het glijden, ongeacht de hoek van
de wielen: sturen verandert niets, sturen is verloren.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:11}
De \omterm{def:g10:inertia:inventory}{spanning} van de draad, inwaarts naar de
atleet. Na loslating vliegt de bal af in een rechte lijn langs de
raaklijn, met de snelheid die ze had (traagheid). De ``draad'' van de
Maan is de gravitationele trek van de Aarde; schakel zwaartekracht uit en
ze vertrekt langs haar raaklijn, recht en uniform, voor altijd.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:12}
Van de top: $x = \dfrac{1.0 \times 0.60 + 2.0 \times 1.20}{3.0}
= \qty{1.00}{m}$. Afstanden tot de twee deelmidden: \qty{0.40}{m}
(steel) en \qty{0.20}{m} (kop), verhouding $2 : 1$ --- het omgekeerde
van de massaverhouding $1.0 : 2.0$, zoals
\cref{def:g11:forces-and-motion:com} eist.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:13}
\emph{1.} Zelfde $F$ en $\Delta v$, dubbele $m$: de bus heeft tweemaal
de tijd nodig. \emph{2.} Zelfde
\omterm{def:g10:relative-motion:speed}{gemiddelde snelheid} voor tweemaal de
tijd: tweemaal de afstand. \emph{3.} Dubbele $\Delta v$ bij dezelfde $F$
en $m$: tweemaal de tijd; maar de
\omterm{def:g10:relative-motion:speed}{gemiddelde snelheid} verdubbelt ook,
dus de afstand wordt met $4$ vermenigvuldigd. Je snelheid verdubbelen
verviervoudigt je remweg.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:14}
\emph{1.} $T = 27.3 \times \qty{86400}{s} \approx \qty{2.36e6}{s}$, dus
$v = \dfrac{2\pi \times \num{3.84e8}}{\num{2.36e6}} \approx
\qty{1.0e3}{m/s} \approx \qty{1}{km/s}$. \emph{2.} Naar de Aarde toe
(inwaarts); de gravitationele trek van de Aarde. \emph{3.} De Maan
\emph{valt}: haar snelheidsverandering wijst elk moment aardwaarts. Haar
tangentiële snelheid zorgt er alleen voor dat ze blijft missen --- ze
valt \emph{om} de Aarde heen in plaats van erin.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:15}
Zelfde $m$ en $\Delta v$, dus $F\Delta t$ is vast: $\Delta t$ vijftig
keer langer op schuim betekent een gemiddelde
\omterm{def:g10:inertia:force}{kracht} vijftig keer kleiner. De knieën buigen
strekt de $\Delta t$ van de landing van een schok tot een lange flex;
een valmat is, mechanisch gesproken, gemaakt van extra milliseconden.
\end{solution}

\begin{solution}{pb:g11:forces-and-motion:1}
\textbf{1.} $50/3.6 \approx \qty{13.9}{m/s}$ (noem het \qty{14}{m/s}).
\textbf{2.} $\Delta v \approx \qty{14}{m/s}$: de inzittende beweegt op
\qty{14}{m/s} ervoor en op nul erna --- beide einden zijn vast door de
crash, dus geen toestel kan $\Delta v$ raken.
\textbf{3.} $\Delta v \propto F\Delta t/m$
(\cref{thm:g11:forces-and-motion:secondlaw}) met $\Delta v$ en $m$ vast
dwingt $F\Delta t$ vast te zijn. De enige knop van de ontwerper is
$\Delta t$.
\textbf{4.} $0.150/0.050 = 3$: de kreukelzone deelt de gemiddelde
\omterm{def:g10:inertia:force}{kracht} door $3$.
\textbf{5.} Achterwaarts, tegengesteld aan de beweging --- de snelheid
krimpt van \qty{14}{m/s} tot nul.
\textbf{6.} De plooiende voorkant is een machine om de stop langer te
laten duren: metaal dat kreukelt koopt milliseconden.
\textbf{7.} Zelfde $\Delta v$ en $m$, $\Delta t$ driemaal korter: de
vintage dummy neemt driemaal de
\omterm{def:g10:inertia:force}{kracht}.
\textbf{8.} \omterm{def:g10:relative-motion:speed}{Gemiddelde snelheid}
$\approx \qty{6.9}{m/s}$; afstand $\approx 6.9 \times 0.150 \approx
\qty{1.0}{m}$ --- ongeveer de
\omterm{def:g10:orders-of-magnitude:unit}{meter} vervormbare auto die een
echte kreukelzone levert.
\textbf{9.} Een \qty{10}{m}-neus is onbestuurbaar en onparkeerbaar, en
haar eigen massa zou elke crash verergeren. De passagierscel mag niet
kreukelen: ze bewaart de overlevingsruimte.
\textbf{10.} ``\ldots dus moet je $\Delta t$ kiezen.''
\textbf{11.} $0.150/0.005 = 30$: de voorruitstop is dertig keer
gewelddadiger dan de aangespte.
\textbf{12.} Een licht rekkende gordel verlengt de eigen $\Delta t$ van
de inzittende iets meer (en spreidt de
\omterm{def:g10:inertia:force}{kracht}); een stalen harnas zou de kortste
stop van de carrosserie opleggen aan de zachtste passagier.
\textbf{13.} $0.080/0.010 = 8$: de airbag deelt de
\omterm{def:g10:inertia:force}{kracht} op het hoofd door $8$.
\textbf{14.} Zelfde $\Delta v$ en $\Delta t$: de
\omterm{def:g10:inertia:force}{kracht} schaalt met de massa, $70/20 = 3.5$
keer groter voor de volwassene.
\textbf{15.} Vereiste verandering: $14/0.150 \approx \qty{93}{m/s}$
per seconde --- ongeveer $9.5$ keer wat zwaartekracht produceert
(\qty{9.8}{m/s} per seconde). De armen moeten ongeveer $9.5$ keer het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van de baby trekken:
alsof ze een \qty{95}{kg}-last vasthouden. Geen armen kunnen dat; de
baby vliegt vooruit.
\textbf{16.} De gordel van een volwassene leidt
\omterm{def:g10:inertia:force}{kracht} naar heupen en borst; op een kind
zou ze buik en nek belasten. De kinderzitje past de (kleinere)
\omterm{def:g10:inertia:force}{kracht} toe op sterke lichaamsdelen en haar
padding strekt $\Delta t$ verder.
\textbf{17.} Als de cel vervormt, verliezen de inzittenden hun
overlevingsruimte en worden geraakt door de structuur zelf. Niets mag
de rigide cel rechtstreeks raken: de stops van de inzittenden worden
afgehandeld door gordel en airbag (lange $\Delta t$), terwijl de cel
de kreukelende einden verankert.
\textbf{18.} Door symmetrie stoppen beide auto's op het contactvlak:
elke bestuurder ondergaat $\Delta v \approx \qty{14}{m/s}$ over een
kreukelzone-stop --- in wezen de muurtest, niet ``dubbele crash''.
\textbf{19.} De \omterm{def:g10:inertia:force}{krachten} zijn gelijk en
tegengesteld, maar $\Delta v \propto F\Delta t/m$: de kleine auto, met
de kleine $m$, lijdt de grote snelheidsverandering.
\textbf{20.} Kreukelzone, rekkende veiligheidsgordel, airbag,
kinderzitjepadding --- elk van hen strekt dezelfde hoeveelheid:
$\Delta t$.
\textbf{21.} De crash fixeert de massa en de snelheidsverandering,
vandaar het product \omterm{def:g10:inertia:force}{kracht} $\times$ tijd: een
zachte botsing is een lange --- strek de tijd, en de
\omterm{def:g10:inertia:force}{kracht} daalt in dezelfde verhouding.
\end{solution}
