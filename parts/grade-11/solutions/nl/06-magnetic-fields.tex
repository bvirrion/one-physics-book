\section*{Hoofdstuk \ref{ch:g11:magnetic-fields} --- Magnetisme en magnetische velden}

\begin{solution}{exo:g11:magnetic-fields:1}
Elk stuk is een complete \omterm{def:g11:magnetic-fields:magnet}{magneet} met
een noord- en een zuidpool: op elk zaagvlak verschijnt een nieuwe pool.
De verse vlakken zijn een N en een Z, dus ze trekken aan --- de stukken
proberen weer samen te komen. Geen schema isoleert een pool: elke snede
creëert een paar.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:2}
(a) Onwaar: het \omterm{def:g11:electric-gravitational-fields:field}{veld} heeft
één richting in elk punt. (b) Waar. (c) Waar: elke lijn sluit door het
lichaam van de \omterm{def:g11:magnetic-fields:magnet}{magneet}. (d) Onwaar:
de naald komt tot rust \emph{langs} de lijn, raak eraan.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:3}
$B = \num{2e-7} \times 10 / 0.020 = \qty{1.0e-4}{T}$ --- tweemaal het
\omterm{def:g11:electric-gravitational-fields:field}{veld} van de Aarde.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:4}
$n = 800/0.40 = \qty{2000}{m^{-1}}$;
$B = \num{1.26e-6} \times 2000 \times 1.5 \approx \qty{3.8}{mT}$.
$B \propto I$: verdubbelen van de stroom geeft \qty{7.5}{mT}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:5}
$F = ILB = 5.0 \times 0.10 \times 0.20 = \qty{0.10}{N}$ --- het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van
$m = F/g = 0.10/9.81 \approx \qty{10}{g}$.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:6}
\emph{1.} Haar spoor wijst $\ang{8}$ west van ware noorden:
$2000 \times \sin\ang{8} \approx \qty{2.8e2}{m}$ naar het westen.

\emph{2.} Stev $\ang{8}$ \emph{oost} van het noorden van de naald.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:7}
$d = \mu_0 I / (2\pi B) = \num{2e-7} \times 20 / \num{5.0e-5}
= \qty{8.0}{cm}$. Binnen een decimeter van zo'n kabel leest het kompas
de draad, niet de planeet --- houd hem weg van onder spanning staande
geleiders.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:8}
$n = B/(\mu_0 I) = 0.010 / (\num{1.26e-6} \times 4.0)
\approx \qty{2.0e3}{m^{-1}}$; over \qty{25}{cm},
$N = 1989 \times 0.25 \approx 500$ windingen.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:9}
$F = 10 \times 0.12 \times 0.50 = \qty{0.60}{N}$;
$a = F/m = 0.60/0.020 = \qty{30}{m/s^2}$ --- ongeveer $3g$: de staaf
springt echt.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:10}
Balans: $ILB = mg$, dus
$I = \num{5.0e-3} \times 9.81 / (0.20 \times 0.10) \approx
\qty{2.5}{A}$. De \omterm{def:g10:inertia:force}{kracht} moet omhoog wijzen;
met $\vect B$ horizontaal en loodrecht op de draad fixeert de platte
rechterhand (palm omhoog) de vereiste stroomrichting.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:11}
Platte rechterhand: (a) naar de top van de bladzijde; (b) naar rechts;
(c) geen \omterm{def:g10:inertia:force}{kracht} --- de draad is evenwijdig
aan het \omterm{def:g11:electric-gravitational-fields:field}{veld}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:12}
\emph{1.} $B = \num{2e-7} \times 5.0 / 0.030 \approx
\qty{33}{\micro T}$; de lijnen cirkelen om de draad, dus er recht
onder is het \omterm{def:g11:electric-gravitational-fields:field}{veld}
horizontaal, loodrecht op de draad: oost--west.

\emph{2.} De naald volgt de vectorsom: $\tan\theta = 33/20 = 1.67$,
dus $\theta \approx \ang{59}$ van noorden af.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:13}
\emph{1.} $I = U/R = 12/2.0 = \qty{6.0}{A}$;
$n = 400/0.20 = \qty{2000}{m^{-1}}$;
$B_0 = \num{1.26e-6} \times 2000 \times 6.0 \approx \qty{15}{mT}$;
met de kern, $100 B_0 \approx \qty{1.5}{T}$.

\emph{2.} $P = UI = 12 \times 6.0 = \qty{72}{W}$.

\emph{3.} Geen stroom, geen
\omterm{def:g11:electric-gravitational-fields:field}{veld} --- en zacht
ijzer blijft niet gemagnetiseerd: de last laat meteen los, by design.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:14}
\emph{1.} $F = NILB = 50 \times 2.0 \times 0.050 \times 0.30 =
\qty{1.5}{N}$ op elke zijde. De stroom loopt tegengestelde kanten op in
de twee zijden, dus de
\omterm{def:g10:inertia:force}{krachten} zijn tegengesteld: een koppel dat
de lus draait.

\emph{2.} Wanneer evenwijdig aan $\vect B$, voelt een draad geen
\omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht}.

\emph{3.} Na een halve slag zou het koppel omkeren en de rotatie
ongedaan maken; de
\omterm{rem:g11:magnetic-fields:motor}{collector} keert de stroom elke
halve slag om zodat de lus één kant op blijft draaien.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:15}
\emph{1.} $d = \num{2e-7} \times 100 / 1 = \qty{2.0e-5}{m} =
\qty{20}{\micro m}$ --- diep \emph{binnen} het millimeter-dikke koper.
Geen toegankelijk punt nabij een enkele draad bereikt \qty{1}{T}:
sterke \omterm{def:g11:electric-gravitational-fields:field}{velden} stapelen
duizenden windingen (een
\omterm{def:g11:magnetic-fields:solenoid}{solenoïde}), plus ijzer of
supergeleiders.

\emph{2.} $\num{e8} / \num{5e-5} = \num{2e12}$: ongeveer twaalf
\omterm{def:g10:energy-conservation:power}{machten} van tien.
\end{solution}

\begin{solution}{pb:g11:magnetic-fields:1}
\textbf{1.} De naald is zelf een kleine
\omterm{def:g11:magnetic-fields:magnet}{magneet}; het
\omterm{def:g11:electric-gravitational-fields:field}{veld} van de Aarde
richt haar, noordpunt langs de horizontale component van $\vect B$,
d.w.z.\ naar magnetisch noorden.

\textbf{2.} $B = \sqrt{20^2 + 44^2} = \sqrt{2336} \approx
\qty{48}{\micro T}$, gekanteld $\tan^{-1}(44/20) \approx \ang{66}$
onder de horizontaal.

\textbf{3.} Het spoor loopt $\ang{6}$ west van ware noorden:
$30 \times \sin\ang{6} \approx \qty{3.1}{km}$ west.

\textbf{4.} Fouten tellen op: $\ang{10}$, dus $30 \times \sin\ang{10}
\approx \qty{5.2}{km}$ --- vijf kilometer voor tien graden.

\textbf{5.} De horizontale component: nabij de pool is het
\omterm{def:g11:electric-gravitational-fields:field}{veld} bijna verticaal,
het horizontale deel krimpt naar nul, en de naald heeft niets meer om
zich op te richten.

\textbf{6.} $n = 600/0.30 = \qty{2000}{m^{-1}}$.

\textbf{7.} $I = U/R = 12/1.5 = \qty{8.0}{A}$.

\textbf{8.} $B_0 = \mu_0 n I = \num{1.26e-6} \times 2000 \times 8.0
\approx \qty{20}{mT}$.

\textbf{9.} $50 \times \qty{20}{mT} = \qty{1.0}{T}$ --- tweemaal het
\omterm{def:g11:electric-gravitational-fields:field}{veld} aan een
neodymiumpool, over een veel groter vlak.

\textbf{10.} $P = UI = 12 \times 8.0 = \qty{96}{W}$; in
\qty{10}{\minute}: $E = 96 \times 600 \approx \qty{58}{kJ}$.

\textbf{11.} Omdat ze loslaat: open de schakelaar, het
\omterm{def:g11:electric-gravitational-fields:field}{veld} sterft, de auto
valt precies waar gewenst. Een permanente
\omterm{def:g11:magnetic-fields:magnet}{magneet} grijpt voor altijd.

\textbf{12.} $F = NILB = 100 \times 3.0 \times 0.040 \times 0.25 =
\qty{3.0}{N}$.

\textbf{13.} Een koppel: toegepast op tegenoverliggende zijden van de
as, draaien de twee tegengestelde
\omterm{def:g10:inertia:force}{krachten} de lus dezelfde kant op. Gelijke
\emph{evenwijdige} \omterm{def:g10:inertia:force}{krachten} zouden de lus
alleen opzij duwen, niet laten draaien.

\textbf{14.} Geen: een draad evenwijdig aan het
\omterm{def:g11:electric-gravitational-fields:field}{veld} voelt geen
\omterm{prop:g11:magnetic-fields:laplace}{Laplace-kracht}.

\textbf{15.} Ze keert de stroom in de lus om bij elke halve slag, precies
wanneer het koppel van teken zou veranderen --- zodat het koppel altijd
dezelfde rotatie drijft.

\textbf{16.} Verdubbel $N$, of verdubbel $I$, of verdubbel $B$: het
\omterm{def:g10:inertia:force}{kracht}paar $F = NILB$ is evenredig met elk.

\textbf{17.} $1.5 / \num{5.0e-5} = \num{3.0e4}$: dertigduizend Aarden.

\textbf{18.} $n = B/(\mu_0 I) = 1.5 / (\num{1.26e-6} \times 500)
\approx \qty{2.4e3}{m^{-1}}$ --- en elk van die windingen voert
\qty{500}{A}.

\textbf{19.} $P = RI^2 = 0.20 \times 500^2 = \qty{50}{kW}$ --- de
verwarming van een buurt. De werkelijke wikkeling is supergeleidend:
nul \omterm{def:g11:circuits-and-power:resistance}{weerstand}, nul
dissipatie, mits ze enkele graden boven het absolute nulpunt gehouden
wordt.

\textbf{20.} Kompas $\num{5e-5}$, kraan $1$, MRI $1.5$: het schip zit
vijf sporten onder de twee machines, die een decade delen --- tussen
interstellair $\num{e-10}$ en neutronenster $\num{e8}$. Niets
veranderde behalve de
\omterm{def:g11:circuits-and-power:current}{ampères} (en de windingen, ijzer
en supergeleider die ze vermenigvuldigen): één
\omterm{def:g11:electric-gravitational-fields:field}{veld}, drie banen.
\end{solution}
