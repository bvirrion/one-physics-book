\section*{Hoofdstuk \ref{ch:g11:electric-gravitational-fields} --- Elektrische en gravitatievelden}

\begin{solution}{exo:g11:electric-gravitational-fields:1}
$E = \num{5.0e-3} / \num{2.0e-6} = \qty{2.5e3}{N/C}$, naar het noorden.
Op $q'$: $F = \num{4.0e-6} \times \num{2.5e3} = \qty{1.0e-2}{N}$, naar
het \emph{zuiden} ($q' < 0$).
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:2}
\emph{1.} $E = \num{8.99e9} \times \num{5.0e-9} / 0.30^2
\approx \qty{5.0e2}{N/C}$, van $O$ af wijzend ($Q > 0$).

\emph{2.} $k\abs{Q}/d^2$ halveren eist $d^2$ verdubbeld:
$d' = 0.30\sqrt{2} \approx \qty{0.42}{m}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:3}
$E = 600 / \num{3.0e-3} = \qty{2.0e5}{V/m}$;
$F = eE = \num{1.60e-19} \times \num{2.0e5} = \qty{3.2e-14}{N}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:4}
$g_{\text{Mars}} = P/m = 930/250 = \qty{3.7}{N/kg}$. Astronaut:
$P = 80 \times 3.72 \approx \qty{3.0e2}{N}$ (ongeveer \qty{780}{N} op
Aarde).
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:5}
(a) Onwaar: het \omterm{def:g11:electric-gravitational-fields:field}{veld} heeft
één richting in elk punt, dus lijnen kruisen nooit. (b) Onwaar: ze
vertrekken van $+$ en komen aan bij $-$. (c) Waar: opeenpakking codeert
sterkte.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:6}
\emph{1.} $d = \num{6.37e6} + \num{4.2e5} = \qty{6.79e6}{m}$, dus
$g = \num{6.67e-11} \times \num{5.97e24} / (\num{6.79e6})^2
\approx \qty{8.6}{N/kg}$ --- $88\%$ van de oppervlaktewaarde.

\emph{2.} Ze zweven omdat station en astronaut samen vallen (vrije
val), niet omdat zwaartekracht afwezig is.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:7}
$E = m_e g / e = \num{9.11e-31} \times 9.81 / \num{1.60e-19}
\approx \qty{5.6e-11}{V/m}$: zo'n $\num{4e14}$ keer zwakker dan het
\omterm{def:g11:electric-gravitational-fields:field}{veld} van de
\omterm{def:g10:signals-and-waves:oscilloscope}{oscilloscoop}. Zwaartekracht
is volstrekt verwaarloosbaar in elektronenfysica.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:8}
\emph{1.} $E = \num{5.0e3} / \num{2.0e-2} = \qty{2.5e5}{V/m}$; in rust
$qE = mg$, dus
$q = \num{2.0e-6} \times 9.81 / \num{2.5e5} \approx \qty{7.8e-11}{C}$.

\emph{2.} De \omterm{def:g10:inertia:force}{kracht} op de positieve korrel
moet omhoog wijzen, dus $\vect E$ wijst omhoog: de \emph{onderste} plaat
is positief.

\emph{3.} $n = \num{7.85e-11} / \num{1.60e-19} \approx \num{4.9e8}$
\omterm{def:g11:fundamental-interactions:elementary}{elementaire ladingen}.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:9}
Buiten het segment wijzen beide
\omterm{def:g11:electric-gravitational-fields:field}{velden} dezelfde kant
op; ertussen staan ze tegenover elkaar --- en balans eist het punt
dichter bij de zwakkere bron $+Q$. Met $x$ de afstand tot $+Q$:
$kQ/x^2 = 4kQ/(0.30 - x)^2$, dus $0.30 - x = 2x$ en
$x = \qty{0.10}{m}$ van $+Q$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:10}
$F_e = ke^2/d^2 = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{5.3e-11})^2 \approx \qty{8.2e-8}{N}$;
$F_g = G m_p m_e / d^2 = \num{6.67e-11} \times \num{1.67e-27} \times
\num{9.11e-31} / (\num{5.3e-11})^2 \approx \qty{3.6e-47}{N}$. Verhouding
$F_e / F_g \approx \num{2.3e39}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:11}
$E = \sqrt{300^2 + 400^2} = \qty{500}{N/C}$, op
$\tan^{-1}(400/300) \approx \ang{53}$ noord van oost. Op $q < 0$:
$F = \num{2.0e-6} \times 500 = \qty{1.0e-3}{N}$, tegengesteld aan het
\omterm{def:g11:electric-gravitational-fields:field}{veld} --- $\ang{53}$
zuid van west.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:12}
$GM/(R+h)^2 = \tfrac12\,GM/R^2$ geeft $R + h = R\sqrt{2}$, dus
$h = (\sqrt{2} - 1)R \approx \qty{2.6e6}{m}$ (ongeveer \qty{2600}{km}).
Op $h = R$, $d = 2R$: een kwart van de oppervlaktewaarde.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:13}
\emph{1.} $AM = \sqrt{3.0^2 + 4.0^2} = \qty{5.0}{cm}$; elke lading
creëert $E_1 = \num{8.99e9} \times \num{2.0e-9} / 0.050^2
\approx \qty{7.2e3}{N/C}$.

\emph{2.} De componenten langs $AB$ heffen door symmetrie op; elk
\omterm{def:g11:electric-gravitational-fields:field}{veld} draagt
$E_1 \times 4/5$ bij langs de middelloodlijn, van het segment af:
$E = 2 \times \num{7.19e3} \times 0.80 \approx \qty{1.2e4}{N/C}$,
gericht langs de middelloodlijn, van $[AB]$ af.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:14}
De vrije ladingen van de doos bewegen onder het uitwendige
\omterm{def:g11:electric-gravitational-fields:field}{veld}: $+$ stapelt op
op één vlak, $-$ op het andere, en deze verplaatste ladingen creëren een
inwendig \omterm{def:g11:electric-gravitational-fields:field}{veld} dat het
uitwendige tegengaat. Ze blijven bewegen tot het totale
\omterm{def:g11:electric-gravitational-fields:field}{veld} binnen precies nul
is --- dan evenwicht. Zwaartekracht biedt slechts één teken bron: elke
massa \emph{voegt} een aantrekkend
\omterm{def:g11:electric-gravitational-fields:field}{veld} toe, geen kan het
tegenwerken. Een zwaartekrachtschild zou negatieve massa nodig hebben,
die niet bestaat.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{Gewicht} $m\vect g$
(omlaag), \omterm{def:g10:inertia:inventory}{spanning} $\vect T$ (langs de
draad), elektrische \omterm{def:g10:inertia:force}{kracht} $q\vect E$
(horizontaal). In rust $T\sin\theta = qE$ en $T\cos\theta = mg$; deel:
$\tan\theta = qE/(mg)$.

\emph{2.} $\tan\theta = \num{2.0e-7} \times \num{1.0e4} /
(\num{5.0e-4} \times 9.81) \approx 0.41$, dus $\theta \approx \ang{22}$;
$T = mg/\cos\theta \approx \qty{5.3e-3}{N}$.
\end{solution}

\begin{solution}{pb:g11:electric-gravitational-fields:1}
\textbf{1.} \omterm{def:g11:electric-gravitational-fields:capacitor}{Homogeen},
loodrecht op de platen, omlaag wijzend (van de $+$-bovenplaat naar de
$-$-onderplaat); homogeniteit betekent dat de
\omterm{def:g10:inertia:force}{kracht} op de druppel hetzelfde is waar ze ook
drijft.
\textbf{2.} $E = U/d = 800 / \num{8.0e-3} = \qty{1.0e5}{V/m}$.
\textbf{3.} De lading is negatief, dus de
\omterm{def:g10:inertia:force}{kracht} is \emph{tegengesteld} aan $\vect E$:
omhoog. Alleen met de $+$-plaat boven wijst $\vect E$ omlaag en vecht de
elektrische \omterm{def:g10:inertia:force}{kracht} het
\omterm{def:g10:universal-gravitation:weight}{gewicht} af.
\textbf{4.} $P = mg = \num{4.90e-15} \times 9.81 \approx
\qty{4.8e-14}{N}$.
\textbf{5.} $\abs{q}E = mg$.
\textbf{6.} $\abs{q} = mg/E = mgd/U$.
\textbf{7.} $\abs{q} = \num{4.81e-14} \times \num{8.0e-3} / 800
\approx \qty{4.8e-19}{C}$.
\textbf{8.} $n = \num{4.8e-19} / \num{1.60e-19} = 3$ overtollige
elektronen.
\textbf{9.} $E$ groeit, dus $\abs{q}E > mg$: de netto
\omterm{def:g10:inertia:force}{kracht} wijst omhoog.
\textbf{10.} Uit $\abs{q}\,U/d = mg$, $U = mgd/\abs{q}$ --- omgekeerd
evenredig met de lading.
\textbf{11.} $\abs{q} = mgd/U = \num{3.85e-16}/1200 \approx
\qty{3.2e-19}{C} = 2e$: van $3e$ omlaag, de druppel heeft \emph{één
elektron verloren}.
\textbf{12.} $\abs{q} = \num{3.85e-16}/U$: \qty{3.2e-19}{C},
\qty{4.8e-19}{C}, \qty{6.4e-19}{C}, \qty{8.0e-19}{C},
\qty{9.6e-19}{C}.
\textbf{13.} Ze zijn $2e, 3e, 4e, 5e, 6e$ met
$e = \qty{1.60e-19}{C}$.
\textbf{14.} Elke trede verandert $\abs{q}$ met precies
\qty{1.6e-19}{C}: één elektron landt per keer.
\textbf{15.} Ze zou $\abs{q} = \num{3.85e-16}/700 \approx
\qty{5.5e-19}{C} = 3.4\,e$ nodig hebben --- geen geheel aantal
elektronen, dus geen stralingsuitbarsting kan het ooit produceren.
\textbf{16.} Lading is \emph{gekwantiseerd}: ze komt in gehele
veelvouden van de
\omterm{def:g11:fundamental-interactions:elementary}{elementaire lading}
$e = \qty{1.60e-19}{C}$ --- de hoeveelheid die Millikan mat.
\textbf{17.} $q \leftrightarrow m$, $E \leftrightarrow g$, en
$\abs{q}E \leftrightarrow mg$: het
\omterm{def:g10:universal-gravitation:weight}{gewicht} zelf.
\textbf{18.} $g_{\text{bol}} = GM/d^2 = \num{6.67e-11} \times 1000 /
1.0^2 = \qty{6.7e-8}{N/kg}$; \omterm{def:g10:inertia:force}{kracht} op de
druppel $\num{4.90e-15} \times \num{6.7e-8} \approx \qty{3.3e-22}{N}$
--- ongeveer $\num{1.5e8}$ keer kleiner dan haar
\omterm{def:g10:universal-gravitation:weight}{gewicht}. Hopeloos.
\textbf{19.} Omdat $G$ zo klein is, zou alleen een planeet-grote massa
erboven de trek van de Aarde kunnen evenaren --- en geen massa stoot af.
De enorme $k$ en de twee tekens van lading laten een tafelbladplaat de
planeet in beide richtingen overtreffen.
\textbf{20.} Eén elektron: $\Delta\abs{q}/\abs{q} = e/3e \approx 33\%$;
één olie-\omterm{def:g11:fundamental-interactions:ladder}{molecule}:
$\Delta m/m = \num{5e-25} / \num{4.90e-15} \approx \num{e-10}$. De
ladingstrap verplaatst de balancerende
\omterm{def:g11:circuits-and-power:voltage}{spanning} met honderden
\omterm{def:g11:circuits-and-power:voltage}{volt}; de massatrap is onzichtbaar
bij elke \omterm{def:g11:circuits-and-power:voltage}{spanning}. De balans
lost dus het \omterm{def:g11:fundamental-interactions:ladder}{atoom} van
elektriciteit op --- $e = \qty{1.60e-19}{C}$ --- terwijl de
\omterm{def:g11:fundamental-interactions:ladder}{moleculen} van olie tot een
continuum vervagen.
\end{solution}
