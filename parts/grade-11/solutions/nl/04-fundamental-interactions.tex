\section*{Hoofdstuk \ref{ch:g11:fundamental-interactions} --- De fundamentele wisselwerkingen}

\begin{solution}{exo:g11:fundamental-interactions:1}
Proton: $10^{-15}$ (\omterm{def:g11:fundamental-interactions:ladder}{nucleon}-verdieping);
water-\omterm{def:g11:fundamental-interactions:ladder}{molecule}: $10^{-10}$
(\omterm{def:g11:fundamental-interactions:ladder}{molecule}-verdieping ---
enkele \omterm{def:g11:fundamental-interactions:ladder}{moleculen} zitten op
atomaire groottes); zand: $10^{-4}$ (alledaagse materie); Aarde: $10^{7}$
(planeet); Melkweg: $10^{21}$ (sterrenstelsel). Alles in
\omterm{def:g10:orders-of-magnitude:unit}{meter}.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:2}
Negatieve lading: hij heeft \emph{gewonnen}
$N = \num{4.8e-9}/\num{1.6e-19} = \num{3.0e10}$ elektronen. Het haar
draagt $+\qty{4.8e-9}{C}$, door ladingbehoud
(\cref{prop:g11:fundamental-interactions:conservation}).
\end{solution}

\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:6}
$F_E = \num{8.99e9} \times (\num{1.60e-19})^2/(\num{2.0e-15})^2
= \num{2.30e-28}/\num{4.0e-30} \approx \qty{58}{N}$;
$F_G = \num{6.67e-11} \times (\num{1.67e-27})^2/\num{4.0e-30}
\approx \qty{4.7e-35}{N}$. Verhouding $F_E/F_G \approx \num{1.2e36}$. De
\omterm{def:g11:fundamental-interactions:strong}{sterke wisselwerking} houdt
de \omterm{def:g11:fundamental-interactions:ladder}{kern}, zwaartekracht is
irrelevant.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:7}
$F_E = \num{2.30e-28}/(\num{5.3e-11})^2 = \num{2.30e-28}/\num{2.81e-21}
\approx \qty{8.2e-8}{N}$;
$F_G = \num{6.67e-11} \times \num{1.67e-27} \times \num{9.11e-31} /
\num{2.81e-21} \approx \qty{3.6e-47}{N}$. Verhouding $\approx
\num{2.3e39}$: de
\omterm{def:g11:fundamental-interactions:four}{elektromagnetische
wisselwerking} houdt \omterm{def:g11:fundamental-interactions:ladder}{atomen}
bijeen.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:8}
Door contact wordt de lading gedeeld: $q_A = q_B = \qty{+4.0e-9}{C}$
(totaal nog $\qty{8.0e-9}{C}$). Dan
$F = \num{8.99e9} \times (\num{4.0e-9})^2/0.10^2 =
\qty{1.4e-5}{N}$, afstotend (gelijke tekens).
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:9}
$d = \sqrt{k q^2/F} = \sqrt{\num{8.99e9} \times \num{1.0e-12}/0.90}
\approx \qty{0.10}{m}$. Een \omterm{def:g10:inertia:force}{kracht} $9$ keer
kleiner eist een afstand $3$ keer groter: $d = \qty{0.30}{m}$.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:10}
$F = \num{2.30e-28}/(\num{1.0e-10})^2 = \qty{2.3e-8}{N}$. De
\omterm{def:g11:fundamental-interactions:strong}{sterke wisselwerking}
draagt niets bij: $10^{-10}$~m is $10^{5}$ keer haar
\omterm{def:g11:fundamental-interactions:strong}{reikwijdte}.
\omterm{def:g11:fundamental-interactions:ladder}{Moleculen} worden gebonden
door de
\omterm{def:g11:fundamental-interactions:four}{elektromagnetische
wisselwerking}.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:11}
$q = \num{e23}/\num{e6} \times \num{1.6e-19} = \qty{1.6e-2}{C}$ op elke
munt. $F = \num{8.99e9} \times (\num{1.6e-2})^2/1.0^2 \approx
\qty{2.3e6}{N}$ --- het
\omterm{def:g10:universal-gravitation:weight}{gewicht} van
$\num{2.3e6}/9.81 \approx \qty{2.3e5}{kg}$, een beladen goederentrein,
uit een miljoenste van de elektronen. Alledaagse materie moet (en is)
neutraal tot veel beter dan één deel op een miljoen.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:12}
$F = \num{2.30e-28}/(\num{1.4e-14})^2 = \num{2.30e-28}/\num{1.96e-28}
\approx \qty{1.2}{N}$ --- nog enorm op deze schaal. Nee: ze staan $14$
keer de sterke \omterm{def:g11:fundamental-interactions:strong}{reikwijdte}
uit elkaar; elk is alleen aan zijn naaste buren gelijmd. Paren:
$\frac{92 \times 91}{2} = 4186$, \emph{alle} afstotend op elke
afstand, terwijl de lijm niet meegroeit met de grootte. Neutronen voegen
aantrekking toe zonder afstoting en verdunnen de protonen; maar
afstoting groeit met het kwadraat van het protonenaantal en lijm alleen
met het aantal buren, dus voorbij ongeveer $80$ protonen balanceert geen
neutronenbudget de boeken --- zulke kernen vervallen
(\cref{ch:g11:nucleus-radioactivity}).
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:13}
Evenwicht: \omterm{def:g10:universal-gravitation:weight}{verticaal}
$T\cos\theta = mg$, horizontaal $T\sin\theta = F$, dus
$F = mg\tan\theta$. Hier $\sin\theta = 3.0/50 = 0.060$, en voor kleine
hoeken $\tan\theta \approx \sin\theta = 0.060$. Dan
$F = \num{1.0e-3} \times 9.81 \times 0.060 \approx \qty{5.9e-4}{N}$;
$q = d\sqrt{F/k} = 0.060 \times \sqrt{\num{5.9e-4}/\num{8.99e9}}
\approx \qty{1.54e-8}{C}$; $N = \num{1.54e-8}/\num{1.6e-19} \approx
\num{9.6e10}$
\omterm{def:g11:fundamental-interactions:elementary}{elementaire ladingen}.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:14}
$F = \num{6.67e-11} \times \num{5.97e24} \times \num{7.35e22} /
(\num{3.84e8})^2 = \num{2.93e37}/\num{1.47e17} \approx \qty{2.0e20}{N}$.
Stellen $kQ^2/d^2 = F$:
$Q = d\sqrt{F/k} = \num{3.84e8} \times \sqrt{\num{2.0e20}/\num{8.99e9}}
\approx \qty{5.7e13}{C}$. Dat is
$N = \num{5.7e13}/\num{1.6e-19} \approx \num{3.6e32}$ elektronen, van
massa $\num{3.6e32} \times \num{9.11e-31} \approx \qty{3.2e2}{kg}$: een
paar honderd \omterm{def:g10:orders-of-magnitude:unit}{kilogram} elektronen
zou het werk kunnen doen van $\qty{7.35e22}{kg}$ Maan --- de sterkte van
elektriciteit en de zwakte van zwaartekracht in één getal.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:15}
$F = \num{2.30e-28}/(\num{2.8e-10})^2 = \num{2.30e-28}/\num{7.84e-20}
\approx \qty{2.9e-9}{N}$;
\omterm{def:g10:universal-gravitation:weight}{gewicht}
$P = \num{5.9e-26} \times 9.81 \approx \qty{5.8e-25}{N}$; verhouding
$F/P \approx \num{5e15}$. Elke binding heeft een \emph{vaste} sterkte,
maar \omterm{def:g10:universal-gravitation:weight}{gewicht} groeit met de
hele massa erboven: stapel rots en de last op de onderste laag groeit
zonder limiet terwijl haar bindingen dat niet doen. Rond \qty{e4}{m}
rots verplettert de last de elektrische bindingen --- zwaartekracht
wint niet door sterkte per deeltje maar door \emph{optellen} wanneer
lading opheft.
\end{solution}

\begin{solution}{pb:g11:fundamental-interactions:1}
\textbf{1.} \omterm{def:g11:fundamental-interactions:ladder}{Quark}
$< \qty{e-18}{m}$;
\omterm{def:g11:fundamental-interactions:ladder}{nucleon} $\qty{e-15}{m}$;
\omterm{def:g11:fundamental-interactions:ladder}{kern}
$10^{-15}$--$\qty{e-14}{m}$;
\omterm{def:g11:fundamental-interactions:ladder}{atoom} $\qty{e-10}{m}$;
\omterm{def:g11:fundamental-interactions:ladder}{molecule} $\qty{e-9}{m}$;
alledaags voorwerp $\qty{1}{m}$; planeet $\qty{e7}{m}$; planetenstelsel
$\qty{e11}{m}$; sterrenstelsel $\qty{e21}{m}$.

\textbf{2.} Schaalfactor $10^{5}$: het
\omterm{def:g11:fundamental-interactions:ladder}{atoom} is
$\qty{1}{cm} \times 10^{5} = \qty{1}{km}$ breed.

\textbf{3.} $(10^{-15}/10^{-10})^3 = 10^{-15}$: materie is
$\num{99.9999999999999}\,\%$ leeg.

\textbf{4.} $\num{7e-5}/\num{e-10} = \num{7e5}$
\omterm{def:g11:fundamental-interactions:ladder}{atomen} --- ongeveer een
miljoen over één haar.

\textbf{5.} Van $10^{-18}$ tot $10^{21}$: $39$
\omterm{def:g10:energy-conservation:power}{machten} van tien.

\textbf{6.} $F_E = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{5.3e-11})^2 \approx \qty{8.2e-8}{N}$.

\textbf{7.} $F_G = \num{6.67e-11} \times \num{1.67e-27} \times
\num{9.11e-31}/(\num{5.3e-11})^2 \approx \qty{3.6e-47}{N}$; verhouding
$F_E/F_G \approx \num{2.3e39}$. Elektriciteit houdt het
\omterm{def:g11:fundamental-interactions:ladder}{atoom}; zwaartekracht is
toeschouwer.

\textbf{8.} Zonder zwaartekracht: niets meetbaars verandert (ze is
$10^{39}$ te weinig). Zonder elektriciteit: geen
\omterm{def:g11:fundamental-interactions:ladder}{atomen}, geen
\omterm{def:g11:fundamental-interactions:ladder}{moleculen}, geen materie
helemaal.

\textbf{9.} ``Contact'' is de elektrische afstoting tussen de
elektronenwolken van
\omterm{def:g11:fundamental-interactions:ladder}{atomen} gebracht tot
ongeveer $\qty{e-10}{m}$: niets raakt ooit. Vloerreactie en
\omterm{def:g10:inertia:inventory}{wrijving} zijn de wet van Coulomb in bulk.

\textbf{10.} $q = \num{e23} \times \num{e-9} \times \num{1.6e-19} =
\qty{1.6e-5}{C}$ elk;
$F = \num{8.99e9} \times (\num{1.6e-5})^2 \approx \qty{2.3}{N}$ --- een
duidelijk zichtbare \omterm{def:g10:inertia:force}{kracht} uit één elektron
op een miljard. Alledaagse materie is neutraal tot veel beter dan
$10^{-9}$.

\textbf{11.} De
\omterm{def:g11:fundamental-interactions:four}{elektromagnetische
wisselwerking}. De sterke \omterm{def:g10:inertia:force}{kracht} is buiten
\omterm{def:g11:fundamental-interactions:strong}{reikwijdte} voorbij ongeveer
$\qty{e-14}{m}$ --- $10^{4}$ keer kleiner dan één
\omterm{def:g11:fundamental-interactions:ladder}{atoom}.

\textbf{12.} $F = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{1.0e-15})^2 \approx \qty{2.3e2}{N}$.

\textbf{13.} $a = F/m_p = 230/\num{1.67e-27} \approx
\qty{1.4e29}{m/s^2}$, ongeveer $\num{1.4e28}$ keer $g$: geen alledaagse
\omterm{def:g10:inertia:force}{kracht} komt in de buurt.

\textbf{14.} Sterker dan \qty{230}{N} op $\qty{e-15}{m}$ (kernen
bestaan); werkend op protonen en neutronen gelijk, blind voor lading
(neutronen worden gegrepen);
\omterm{def:g11:fundamental-interactions:strong}{reikwijdte} ongeveer
$\qty{e-15}{m}$ (kernen blijven minuscuul --- de lijm reikt niet verder).

\textbf{15.} $F = \num{2.30e-28}/(\num{1.4e-14})^2 \approx
\qty{1.2}{N}$. Nee: $14$ keer de
\omterm{def:g11:fundamental-interactions:strong}{reikwijdte}. Afstoting
werkt over \emph{alle} $\frac{92\times91}{2} = 4186$ protonenparen; de
lijm bindt alleen naaste buren, dus groei begunstigt afstoting.
Neutronen voegen lijm en afstand toe zonder afstoting toe te voegen ---
tot, voorbij ongeveer $80$ protonen, geen mengsel balanceert en de
\omterm{def:g11:fundamental-interactions:ladder}{kern} vervalt.

\textbf{16.} De \omterm{def:g11:fundamental-interactions:weak}{zwakke
wisselwerking} verandert een neutron in een proton (en terug),
veroorzaakt $\beta$-verval --- ze transformeert; ze bindt nooit.

\textbf{17.} $F = \num{6.67e-11} \times \num{5.97e24} \times
\num{7.35e22}/(\num{3.84e8})^2 \approx \qty{2.0e20}{N}$.

\textbf{18.} $Q_E = \num{1.8e51} \times \num{1.6e-19} =
\qty{2.9e32}{C}$, $Q_M = \qty{3.5e30}{C}$. Gelijke
\omterm{def:g10:inertia:force}{krachten} betekent
$f^2\,k\,Q_E Q_M = G M_E M_M$:
\[
f = \sqrt{\frac{\num{2.93e37}}{\num{8.99e9} \times \num{2.9e32}
\times \num{3.5e30}}} = \sqrt{\num{3.2e-36}} \approx \num{1.8e-18} :
\]
een onevenwicht van \emph{twee delen op $10^{18}$} zou zwaartekracht
opheffen.

\textbf{19.} Lading komt in twee tekens en heft op (vraag~18 toont hoe
perfect), dus de sterkere \omterm{def:g10:inertia:force}{kracht} schakelt
zichzelf uit op grote schalen; massa heeft één teken, telt eeuwig op, en
niets schermt haar af.

\textbf{20.} \omterm{def:g11:fundamental-interactions:ladder}{Kern}: de
\omterm{def:g11:fundamental-interactions:strong}{sterke wisselwerking}
overtreft de afstoting van de protonen op $\qty{e-15}{m}$.
\omterm{def:g11:fundamental-interactions:ladder}{Atoom} tot berg: de
\omterm{def:g11:fundamental-interactions:four}{elektromagnetische
wisselwerking} bindt elektronen aan kernen en
\omterm{def:g11:fundamental-interactions:ladder}{atomen} aan elkaar. Planeet
en omhoog: \omterm{def:g11:fundamental-interactions:four}{gravitatie},
onbestreden eens lading opheft, houdt planeten,
\omterm{def:g10:universal-gravitation:satellite}{banen} en sterrenstelsels.
De wereld stort niet in omdat elke verdieping is geschraagd door een
binder stijver dan de last, en vliegt niet uiteen omdat materie neutraal
is tot ongeveer $10^{-18}$ --- dus elke schaal heeft precies één
\omterm{def:g10:inertia:force}{kracht} aan het roer, en ze is aantrekkend
waar ze dat moet zijn.
\end{solution}
