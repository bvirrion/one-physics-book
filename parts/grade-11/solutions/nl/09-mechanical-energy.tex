\section*{Hoofdstuk \ref{ch:g11:mechanical-energy} --- Mechanische energie en haar behoud}

\begin{solution}{exo:g11:mechanical-energy:1}
(a) $v = \qty{6.94}{m/s}$:
$E_k = \tfrac12 \times 90 \times 6.94^2 \approx \qty{2.2e3}{J}$.
(b) $v = \qty{36.1}{m/s}$:
$E_k = \tfrac12 \times 1200 \times 36.1^2 \approx \qty{7.8e5}{J}$.
(c) $\times 4$; $\times 9$ --- het kwadraat.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:2}
$\Delta E_p = 75 \times 9.81 \times 850 \approx \qty{6.3e5}{J}$. Nee:
de referentie verschuift beide waarden evenveel. Nee: alleen het
hoogteverschil telt.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:3}
$v = \sqrt{2 \times 9.81 \times 20} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$. $mgh = \tfrac12 mv^2$: de massa valt weg.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:4}
$W = \Delta E_k = 0 - \tfrac12 \times 1200 \times 25^2 =
\qty{-3.75e5}{J}$; $F = \num{3.75e5}/50 = \qty{7.5e3}{N}$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:5}
$v = \sqrt{2 \times 9.81 \times 0.080} \approx \qty{1.3}{m/s}$; ze
stijgt terug tot \qty{8.0}{cm} (behoud).
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:6}
$\num{1.5e9}/\num{2.6e3} \approx \num{5.9e5}$: ongeveer zeshonderdduizend
kogels. $h = 88.9^2/(2 \times 9.81) \approx \qty{4.0e2}{m}$ --- haar
eigen snelheid zou de trein $400$ meter kunnen tillen.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:7}
Zelfde snelheid $v = \sqrt{2 \times 9.81 \times 3.0} \approx
\qty{7.7}{m/s}$ (alleen de daling telt). De steile glijbaan wint op
tijd: ze bereikt hoge snelheid sneller en haar pad is korter.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:8}
$h = v^2/2g = 144/19.62 \approx \qty{7.3}{m}$. Op \qty{4.0}{m}:
$v = \sqrt{144 - 2 \times 9.81 \times 4.0} \approx \qty{8.1}{m/s}$ ---
dezelfde beide kanten op: $E_m$ hangt af van hoogte, niet van
richting van reizen.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:9}
$v = \sqrt{23^2 + 2 \times 9.81 \times 35} = \sqrt{1216} \approx
\qty{35}{m/s}$. Kleiner in werkelijkheid: luchtweerstand dissipeert een
deel van $E_m$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:10}
$\Delta E_m = \tfrac12 \times 45 \times 8.0^2 - 45 \times 9.81 \times
6.0 = 1440 - 2649 \approx \qty{-1.2e3}{J}$;
$f = 1209/15 \approx \qty{81}{N}$. Naar warmte, in de glijders en de
sneeuw.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:11}
$v = \sqrt{2 \times 9.81 \times 0.15} \approx \qty{1.7}{m/s}$; ze
stijgt weer tot \qty{15}{cm}. De pen verandert het pad (een strakkere
cirkel), niet de \omterm{def:g10:energy-conservation:energy}{energie}:
hoogtes en snelheden blijven onaangeroerd.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:12}
$v = \sqrt{2 \times 9.81 \times 2.5} = \qty{7.0}{m/s}$. Stijghoogtes
schalen met $E_m$, dus $\times 0.9$ per doortocht: $2.25$, $2.03$,
$1.82$, $1.64$, $1.48$, $1.33$, $1.20$, $1.08$, $0.97$ --- voor het
eerst onder \qty{1.0}{m} bij de $9$e doortocht.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:13}
Zelfde begin-$E_m = \tfrac12 m \times 15^2 + m \times 9.81 \times 45$
ongeacht de hoek, dus de grondsnelheid is
$v = \sqrt{225 + 2 \times 9.81 \times 45} = \sqrt{1108} \approx
\qty{33}{m/s}$. De hoek fixeert de \emph{richting} van de
landingssnelheid, de vliegtijd en de
\omterm{def:g11:fundamental-interactions:strong}{reikwijdte} --- niet de
landingssnelheid.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:14}
$\tfrac12 mv^2 = mgh - f\ell = 80 \times 9.81 \times 120 - 65 \times
800 = \num{94176} - \num{52000} = \qty{4.22e4}{J}$, dus
$v = \sqrt{2 \times \num{42176}/80} \approx \qty{32}{m/s}$
(\qty{117}{km/h}). Fractie
\omterm{def:g11:mechanical-energy:dissipation}{gedissipeerd}:
$52000/94176 \approx 55\%$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:15}
$m = \qty{2.0e9}{kg}$:
$E_p = \num{2.0e9} \times 9.81 \times 300 \approx \qty{5.9e12}{J}
= \num{5.9e12}/\num{3.6e6} \approx \qty{1.6e6}{kW.h}$. Herwonnen:
$0.85 \times \num{1.64e6} \approx \qty{1.4e6}{kW.h}$ --- ongeveer
$\num{1.4e5}$ huizen voor een dag.
\end{solution}

\begin{solution}{pb:g11:mechanical-energy:1}
\textbf{1.} $v_{\text{top}} = \sqrt{gR} = \sqrt{9.81 \times 8.0}
\approx \qty{8.9}{m/s}$.

\textbf{2.} $E_m = mg\,(2R) + \tfrac12 m gR = \num{3.14e5} +
\num{7.8e4} \approx \qty{3.92e5}{J}$.

\textbf{3.} $mgH_{\min} = E_m$ geeft $H_{\min} = \qty{20}{m}$;
algebraïsch $H_{\min} = 2R + \frac{gR}{2g} = \frac{5R}{2}$.

\textbf{4.} $v_{\text{top}} = \sqrt{2g(H - 2R)} =
\sqrt{2 \times 9.81 \times 8.0} \approx \qty{12.5}{m/s}$;
$v_{\text{top}}^2/(gR) = 2.0$ --- tweemaal het minimum.

\textbf{5.} Elke \omterm{def:g10:energy-conservation:energy}{energie} is
evenredig met $m$, dus $m$ valt weg: dezelfde snelheden en dezelfde
veiligheid, vol of leeg.

\textbf{6.} $v = \sqrt{2 \times 9.81 \times 24} \approx \qty{21.7}{m/s}
\approx \qty{78}{km/h}$.

\textbf{7.} $v = \sqrt{2 \times 9.81 \times (24 - 8.0)} \approx
\qty{17.7}{m/s}$.

\textbf{8.} $mgH = \qty{4.71e5}{J}$. $(E_p, E_k)$: lancering
$(\num{4.71e5}, 0)$; bodem $(0, \num{4.71e5})$; zijkant
$(\num{1.57e5}, \num{3.14e5})$; top $(\num{3.14e5}, \num{1.57e5})$ ---
elk paar sommeert tot \qty{4.71e5}{J}.

\textbf{9.} $v = \sqrt{2 \times 9.81 \times (24 - 12)} \approx
\qty{15.3}{m/s}$.

\textbf{10.} $v(z) = \sqrt{2g(H - z)}$: snelst op het laagste punt van
de baan, traagst op de top van de lus.

\textbf{11.} Zelfde hoogte, dus $\Delta E_p = 0$ en
$\Delta E_m = \Delta E_k = \tfrac12 \times 2000 \times (19.2^2 -
20.4^2) \approx \qty{-4.75e4}{J}$: de plaatsing maakt het tekort puur
kinetisch, afleesbaar van twee snelheidsmeters.

\textbf{12.} $W(\vect f) = \qty{-4.75e4}{J}$;
$f = \num{4.75e4}/60 \approx \qty{7.9e2}{N}$.

\textbf{13.} \omterm{def:g10:energy-conservation:forms}{Thermische energie},
in de wielen, de rails en de lucht.

\textbf{14.} $E_m$ bij de eerste sensor: $\num{4.16e5} + \num{3.9e4} =
\qty{4.55e5}{J}$; verlies $\num{4.75e4}/\num{4.55e5} \approx 10\%$ per
\qty{60}{m}. Bij $H_{\min}$ wordt de top van de lus \emph{precies}
bereikt op $\sqrt{gR}$; elk verlies brengt haar onder de
veiligheidssnelheid.

\textbf{15.} Met $\approx 10\%$ verlies per \qty{60}{m} is ongeveer
$15\%$ van $mgH$ ($\approx \qty{7.1e4}{J}$) weg: $E_m \approx
\qty{4.00e5}{J}$, nog boven de vereiste \qty{3.92e5}{J} --- veilig, met
slechts $2\%$ speling. De marge was geen luxe.

\textbf{16.} $E_k = \tfrac12 \times 2000 \times 18.0^2 =
\qty{3.24e5}{J}$.

\textbf{17.} $F = \num{3.24e5}/45 = \qty{7.2e3}{N}$.

\textbf{18.} $a = 7200/2000 = \qty{3.6}{m/s^2} \approx 0.37\,g$:
stevig maar comfortabel.

\textbf{19.} Baan\omterm{def:g10:inertia:inventory}{wrijving}:
$mgH - \num{3.24e5} = \num{470880} - \num{324000} = \qty{1.47e5}{J}$.
Audit: $\num{1.47e5} + \num{3.24e5} = \num{4.71e5} = mgH$ --- elke
\omterm{def:g11:work-of-force:work}{joule} verantwoord.

\textbf{20.} De lanceerhoogte $H = \qty{24}{m}$ fixeerde de veiligheid
van de lus, elke snelheid $\sqrt{2g(H - z)}$, de
\omterm{def:g10:inertia:inventory}{wrijving}srekening en de rem-\omterm{def:g10:inertia:force}{kracht};
de prijs van \omterm{def:g10:inertia:inventory}{wrijving} negeren is de
marge --- de $\qty{4}{m}$ heuvel die de wrijvende wereld stilzwijgend
opslokt.
\end{solution}
