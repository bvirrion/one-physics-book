\section*{Hoofdstuk \ref{ch:g11:color-light-sources} --- Kleur en lichtbronnen}

\begin{solution}{exo:g11:color-light-sources:1}
(a) geel; (b) cyaan; (c) wit. Complementair aan blauw: geel
(blauw $+$ geel $=$ blauw $+$ rood $+$ groen $=$ wit).
\end{solution}

\begin{solution}{exo:g11:color-light-sources:2}
Geel: R, G. Magenta: R, B. Wit: R, G, B. Zwart: geen. Oranje: R op
vol \omterm{def:g10:energy-conservation:power}{vermogen}, G op verminderd
\omterm{def:g10:energy-conservation:power}{vermogen}.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:3}
Magenta absorbeert de groene band: $(R, B)$ treedt uit, magenta licht.
Het cyaan \omterm{def:g11:color-light-sources:subtractive}{filter} absorbeert
daarna de rode band: alleen $(B)$ overleeft --- blauw.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:4}
Gele streep: \omterm{def:g11:color-light-sources:objectcolor}{diffundeert} R en
G, absorbeert B. (a) Onder rood licht
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} ze R: rood.
(b) Onder blauw licht absorbeert ze alles: zwart.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:5}
$T = \dfrac{\qty{2.90e-3}{m.K}}{\qty{5.80e-7}{m}} = \qty{5000}{K}$.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:6}
\emph{(a)} $\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{1800}{K}
= \qty{1.61e-6}{m} = \qty{1610}{nm}$:
\omterm{def:g10:light-spectra:wavelength}{infrarood}.

\emph{(b)} Het \omterm{def:g10:light-spectra:white}{spectrum} is continu: de
zichtbare staart is klein maar niet nul, en het blauwe uiteinde vervaagt
eerst --- wat zichtbaar blijft is vooral rood-oranje, vandaar de kleur
van de vlam.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:7}
Jack (\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} R): (a) rood;
(b) rood; (c) zwart. Handschoenen (diffunderen G en B, absorberen R):
(a) cyaan; (b) zwart; (c) groen.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:8}
\emph{(a)} Een \omterm{def:g10:light-spectra:emission}{emissielijnenspectrum}.
Elk voorwerp kan alleen $\qty{589}{nm}$ diffunderen of niets: de straat
is tinten geel-oranje en zwart.

\emph{(b)} De blauwe auto
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} alleen de blauwe
band. Daglicht bevat die: blauwe auto. De natriumlijn draagt geen blauw:
alles wordt geabsorbeerd, de auto lijkt zwart.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:9}
\emph{(a)} De witte bundel waaiert uit tot een regenboog; de
\omterm{def:g11:color-light-sources:lamps}{laser}bundel wordt afgebogen maar
blijft één bundel van hetzelfde groen: ze is
\omterm{def:g10:light-spectra:white}{monochromatisch} --- één
\omterm{def:g10:light-spectra:wavelength}{golflengte}, niets voor het
prisma om te sorteren.

\emph{(b)} Het rode \omterm{def:g11:color-light-sources:subtractive}{filter}
laat alleen de rode band door en absorbeert $\qty{532}{nm}$: bijna niets
komt eruit; de vlek verdwijnt.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:10}
\omterm{def:g10:light-spectra:white}{Spectrum}: een smalle blauwe piek bij
$\qty{450}{nm}$ plus de brede gele band van de
\omterm{def:g11:color-light-sources:led}{fosfor} (ongeveer
$500$--$\qty{700}{nm}$). Blauw en geel zijn complementair, dus de
\omterm{def:g11:color-light-sources:cones}{kegeltjes}-triple leest wit. De band
vervaagt naar diep rood, dus een dieprood voorwerp ontvangt weinig dat
het kan diffunderen: het lijkt doffer dan bij daglicht, waarvan het
\omterm{def:g10:light-spectra:continuous}{continue spectrum} het volledig
voedt.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:11}
(a) $\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{1800}{K}
= \qty{1610}{nm}$, \omterm{def:g10:light-spectra:wavelength}{infrarood};
(b) $\qty{2.90e-3}{m.K}/\qty{6500}{K} = \qty{446}{nm}$, blauw-violet.
Het kaarslicht heet ``warm'' --- de bron met de \emph{lagere}
temperatuur: de woordenschat loopt omgekeerd aan de
\omterm{def:g10:light-spectra:kelvin}{kelvin}-schaal.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:12}
\emph{(a)} \omterm{def:g11:color-light-sources:perception}{Metameren}. Hun
bestaan bewijst dat kleur een
\omterm{def:g11:color-light-sources:perception}{waarneming} is --- drie
\omterm{def:g11:color-light-sources:cones}{kegeltjes}-responsen --- en niet
het \omterm{def:g10:light-spectra:white}{spectrum} zelf: oneindig veel
spectra mappen op dezelfde triple.

\emph{(b)} Prisma of \omterm{def:g10:light-spectra:white}{spectroscoop}: één
lijn op $\qty{589}{nm}$ tegenover twee banden op $540$ en
$\qty{610}{nm}$. Rood \omterm{def:g11:color-light-sources:subtractive}{filter}:
het absorbeert $\qty{589}{nm}$, de natriumvlek wordt donker, terwijl de
$\qty{610}{nm}$-subpixel van het scherm overleeft --- de schermvlek
wordt rood.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:13}
\emph{(a)} $\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{3200}{K} =
\qty{906}{nm}$. \omterm{def:g11:color-light-sources:colortemp}{Kleurtemperatuur}
$\qty{3200}{K} < \qty{6500}{K}$: warmer (meer oranje) dan daglicht.

\emph{(b)} De gel absorbeert een deel van het rood-oranje overschot om
het \omterm{def:g10:light-spectra:white}{spectrum} naar blauw te
herbalanceren; omdat een
\omterm{def:g11:color-light-sources:subtractive}{filter} alleen licht
verwijdert, daalt het totale
\omterm{def:g10:energy-conservation:power}{vermogen} op het podium.

\emph{(c)} Een \omterm{def:g11:color-light-sources:subtractive}{filter} is
passief: het kan bij elke
\omterm{def:g10:light-spectra:wavelength}{golflengte} absorberen maar nooit
uitzenden, dus geen gel kan de blauwe intensiteit verhogen boven wat de
spot al produceert.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:14}
\emph{(a)} $\qty{2700}{K}$: warm, roodrijk licht --- de rode band van
het vlees wordt goed gevoed, het vlees lijkt levendig fris.
$\qty{6500}{K}$: koel blauwwit --- de vis
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} de volle triple
en lijkt helder wit, vers op ijs.

\emph{(b)} Geen van beiden heeft zijn waren veranderd: de
absorptiegewoonten van vlees en vis zijn onaangeroerd. Wat veranderde is
de belichting, dus het gediffundeerde licht --- de
\omterm{def:g11:color-light-sources:objectcolor}{kleur van een voorwerp} is
alleen gedefinieerd onder een gegeven belichting.
\end{solution}

\begin{solution}{exo:g11:color-light-sources:15}
\emph{(a)} Natrium: de hele stadsgloei zit in één lijn op
$\qty{589}{nm}$, die één smal
\omterm{def:g11:color-light-sources:subtractive}{filter} uit
telescoopbeelden verwijdert. Een breed LED-\omterm{def:g10:light-spectra:white}{spectrum}
kan niet weggefilterd worden zonder ook het sterlicht te verwerpen.

\emph{(b)} Bestuurders en voetgangers verkiezen de LED: onder haar volle
\omterm{def:g10:light-spectra:white}{spectrum} houden voorwerpen
onderscheidbare kleuren (een rode jas blijft rood), terwijl natriumlicht
alles geel-oranje of zwart maakt.

\emph{(c)} Warmwitte LED's, rond $\qty{2700}{K}$: aanvaardbare
kleurweergave voor de straat, en een
\omterm{def:g10:light-spectra:white}{spectrum} arm aan het blauwe uiteinde,
het deel dat de nachthemel het meest vervuilt.
\end{solution}

\begin{solution}{pb:g11:color-light-sources:1}
\textbf{1.} R $+$ G $=$ geel, R $+$ B $=$ magenta, G $+$ B $=$ cyaan;
driedubbele overlap wit.

\textbf{2.} Oranje: rood op vol \omterm{def:g10:energy-conservation:power}{vermogen},
groen op deels \omterm{def:g10:energy-conservation:power}{vermogen}, blauw uit.
Bleek roze: alle drie aan (witte basis) met rood iets dominant.

\textbf{3.} De schaduw geworpen vanuit de rode spot ontvangt alleen de
groene spot: groene schaduw; symmetrisch is de andere schaduw rood.

\textbf{4.} Diepblauwe verf
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} alleen B. Onder
alleen de rode spot: zwart. Onder alle drie:
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} ze de blauwe
band --- blauw.

\textbf{5.} Het oog reduceert elk
\omterm{def:g10:light-spectra:white}{spectrum} tot drie
\omterm{def:g11:color-light-sources:cones}{kegeltjes}-responsen; de triple
evenaren volstaat om de kleur te evenaren. Drie
\omterm{def:g11:color-light-sources:cones}{kegeltjes}-types fixeren het
aantal spots.

\textbf{6.} Geel $=$ wit $-$ blauw: de kleurstof absorbeert de blauwe
band.

\textbf{7.} Rode spot: \omterm{def:g11:color-light-sources:objectcolor}{diffundeert}
R --- rood. Groene spot:
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} G --- groen.
Blauwe spot: absorbeert alles --- zwart.

\textbf{8.} Belicht de scène met alleen de blauwe spot; de schakelaar is
wit $\to$ blauw. Het gele kostuum
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} R en G maar
absorbeert B: helder onder wit, zwart onder blauw.

\textbf{9.} Magenta gel: $(R, G, B) \to (R, B)$. Het kostuum
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} R en G van wat
het ontvangt: het \omterm{def:g11:color-light-sources:objectcolor}{diffundeert}
R, absorbeert B --- rood kostuum.

\textbf{10.} Een rode gel houdt één band van drie en dumpt twee derden
van het \omterm{def:g10:energy-conservation:power}{vermogen} van de witte
spot; elke C-, M-, Y-gel verwijdert één band, houdt twee derden, en
twee van hen stapelen levert nog elke primair.

\textbf{11.} Een \omterm{def:g10:light-spectra:emission}{emissielijnenspectrum}.
De straat wordt een monochrome wereld: elk oppervlak is geel-oranje,
dof of helder, of zwart.

\textbf{12.} Witte muur: geel-oranje
(\omterm{def:g11:color-light-sources:objectcolor}{diffundeert} de lijn). Gele
markering: helder geel-oranje. Rode auto: donker --- de
$\qty{589}{nm}$-lijn ligt buiten de band die ze
\omterm{def:g11:color-light-sources:objectcolor}{diffundeert}. Blauwe auto:
zwart.

\textbf{13.} Eén smal \omterm{def:g11:color-light-sources:subtractive}{filter}
gecentreerd op $\qty{589}{nm}$ verwijdert de hele stadsgloei uit een
beeld terwijl het bijna geen sterlicht van de brede spectra offert.

\textbf{14.} $\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{2700}{K} =
\qty{1.07e-6}{m} = \qty{1070}{nm}$: de piek, en het grootste deel van
het \omterm{def:g10:energy-conservation:power}{vermogen}, ligt in het
\omterm{def:g10:light-spectra:wavelength}{infrarood} --- warmte, niet licht.

\textbf{15.} Natrium: $0.30 \times \qty{100}{W} = \qty{30}{W}$
zichtbaar licht; wolfraam: $0.05 \times \qty{100}{W} = \qty{5}{W}$.
Zes keer meer licht voor dezelfde rekening.

\textbf{16.} De \omterm{def:g11:color-light-sources:led}{fosfor} zet een deel
van het blauw om in een brede gele band; blauw en geel zijn
complementair, dus blauw $+$ geel voltooit de triple: wit.

\textbf{17.} Een smalle blauwe piek plus een brede gele bult --- tegenover
het egale continuum van daglicht. Het dieprode uiteinde (en een dip
tussen blauw en geel) is ondervertegenwoordigd.

\textbf{18.} Dieprode voorwerpen worden dof en bruinachtig weergegeven.
Vandaar de $\qty{2700}{K}$ warme LED's van de slager, waarvan de
versterkte rode inhoud het vlees levendig houdt.

\textbf{19.} Niets binnenin is op $\qty{2700}{K}$: de chip loopt bij
kamertemperatuur. Het label belooft alleen de \emph{tint} van een
$\qty{2700}{K}$-gloeilichaam, waarvan de piek zou zijn
$\qty{2.90e-3}{m.K}/\qty{2700}{K} = \qty{1.07e-6}{m} = \qty{1070}{nm}$.

\textbf{20.} Slaapkamer: ongeveer $\qty{2700}{K}$ --- warm, blauwarm
licht om te rusten. Juwelier: $5500$--$\qty{6500}{K}$ --- daglichtachtig
wit, zodat witten wit lezen en stenen elke band teruggeven. De wet:
waargenomen kleur $=$ bron-\omterm{def:g10:light-spectra:white}{spectrum}
$\times$ voorwerpsabsorptie, gelezen door drie
\omterm{def:g11:color-light-sources:cones}{kegeltjes}.
\end{solution}
