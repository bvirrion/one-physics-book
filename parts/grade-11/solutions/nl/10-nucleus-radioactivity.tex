\section*{Hoofdstuk \ref{ch:g11:nucleus-radioactivity} --- De kern en de radioactiviteit}

\begin{solution}{exo:g11:nucleus-radioactivity:1}
${}^{16}_{8}\mathrm{O}$: $8$ p, $8$ n; ${}^{27}_{13}\mathrm{Al}$: $13$ p,
$14$ n; ${}^{60}_{27}\mathrm{Co}$: $27$ p, $33$ n;
${}^{235}_{92}\mathrm{U}$: $92$ p, $143$ n; ${}^{3}_{1}\mathrm{H}$: $1$ p,
$2$ n. ($N = A - Z$ elke keer.)
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:2}
De drie koolstoffen ($Z = 6$) zijn
\omterm{def:g11:nucleus-radioactivity:notation}{isotopen} van elkaar.
${}^{14}_{6}\mathrm{C}$ en ${}^{14}_{7}\mathrm{N}$ delen $A = 14$ maar
niet $Z$: verschillende elementen (verschillende elektronenwolken), dus
geen \omterm{def:g11:nucleus-radioactivity:notation}{isotopen}.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:3}
$A = 210 - 4 = 206$, $Z = 84 - 2 = 82$:
${}^{210}_{84}\mathrm{Po} \to {}^{206}_{82}\mathrm{Pb} +
{}^{4}_{2}\mathrm{He}$ --- de dochter is lood-206.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:8}
Fluor-18 heeft $9$ neutronen waar stabiel fluor-19 er $10$ heeft:
neutronarm, onder de band, dus $\beta^+$:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:9}
$24$ uur $= 4\,T_{1/2}$: $500/16 \approx \qty{31}{MBq}$. Omdat
$2^9 = 512$, brengen negen halfwaardetijden $500/512 < \qty{1}{MBq}$:
na $9 \times 6 = \qty{54}{h}$.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:10}
$\frac18 = \frac1{2^3}$: drie halfwaardetijden, $3 \times \num{5700} =
\num{17100}$ jaar. Een dinosaurusbot is ongeveer $\num{17500}$
halfwaardetijden oud: een fractie $1/2^{17500}$ blijft --- geen enkel
\omterm{def:g11:fundamental-interactions:ladder}{atoom} koolstof-14 (een gram
koolstof houdt slechts ongeveer $\num{5e22}$
\omterm{def:g11:fundamental-interactions:ladder}{atomen}).
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:11}
${}^{14}_{6}\mathrm{C}$: $8$ neutronen tegen $6$--$7$ in stabiel koolstof,
boven de band, $\beta^-$. ${}^{11}_{6}\mathrm{C}$: $5$ neutronen,
neutronarm, $\beta^+$. ${}^{238}_{92}\mathrm{U}$: $Z = 92 > 83$, te

\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:12}
Alleen $\alpha$ verandert $A$: $238 - 206 = 32 = 4x$, dus $x = 8$
\omterm{def:g11:nucleus-radioactivity:abg}{alfa's}. Ze verwijderen $16$ van
$Z$: $92 - 16 + y = 82$ geeft $y = 6$ bètaminus-vervallen.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:13}
$6.4/0.1 = 64 = 2^{6}$: zes halfwaardetijden, $6 \times 5.3 = 31.8
\approx 32$ jaar in afgeschermde opslag.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:14}
Per dag: $8000 \times 86400 \approx \num{6.9e8}$ vervallen. Over $80$
jaar: $\num{6.9e8} \times 365 \times 80 \approx \num{2e13}$. Het leven
heeft altijd op deze achtergrond gedraaid --- één
\omterm{def:g11:nucleus-radioactivity:activity}{becquerel} is één
\omterm{def:g11:fundamental-interactions:ladder}{atoom} per seconde uit de
$\sim 10^{27}$ van het lichaam; de
\omterm{def:g10:orders-of-magnitude:unit}{eenheid} is minuscuul, en een
angstaanjagend klinkend aantal
\omterm{def:g11:nucleus-radioactivity:activity}{becquerels} kan de waarde
van een paar bananen zijn.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:15}
$\num{25000} \times 86400 \approx \num{2.2e9}$ vervallen per dag. Tien
jaar is $10/432 \approx 2\%$ van een
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}: veel minder
dan de helft is vervallen, dus de elektronica en stof, niet de bron,
\omterm{def:g10:inertia:force}{dwingen} vervanging. Buiten sterven de
$\alpha$'s in de lucht en de behuizing van de kamer; ingeademd stof
parkeert de emitter in de longen, waar elke $\alpha$ haar
\omterm{def:g10:energy-conservation:energy}{energie} in levend weefsel
dumpt --- een grote \omterm{rem:g11:nucleus-radioactivity:dose}{dosis}
(\omterm{rem:g11:nucleus-radioactivity:dose}{sievert}) uit een bescheiden
\omterm{def:g11:nucleus-radioactivity:activity}{activiteit}.
\end{solution}

\begin{solution}{pb:g11:nucleus-radioactivity:1}
\textbf{1.} ${}^{12}_{6}\mathrm{C}$: $6$ p, $6$ n; ${}^{14}_{6}\mathrm{C}$:
$6$ p, $8$ n. Chemie ziet alleen de elektronenwolk, vastgelegd door
$Z = 6$: identiek gedrag.

\textbf{2.} Eten en ademen vernieuwen constant de koolstof van een
levend lichaam op de atmosferische verhouding; bij de dood stopt de
inname en loopt verval onbestreden.

\textbf{3.} ${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} +
{}^{\;0}_{-1}\mathrm{e}$ ($\beta^-$). Deze $\beta$'s zijn zacht:
\omterm{def:g10:orders-of-magnitude:unit}{meters} gewikkeld linnen stoppen
ze, dus weinig ontsnapt de mummie --- daarom meet men een klein
koolstofmonster uit het object zelf.

\textbf{4.} $6.8/13.6 = \frac12$: één
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}. De mummie is
ongeveer \num{5700} jaar oud.

\textbf{5.} $13.2/13.6 \approx 0.97$: het linnen is in wezen modern ---
hoogstens enkele eeuwen oud. Een vervalsing.

\textbf{6.} $13.6/2^{10} = 13.6/1024 \approx 0.013$ vervallen per minuut
--- één telling elke $75$ minuten per gram, verdronken in natuurlijke
achtergrond. Tien halfwaardetijden, ongeveer \num{57000} jaar, is de
praktische horizon.

\textbf{7.} $92$ protonen, $146$ neutronen. Coulomb-afstoting werkt
tussen alle protonenparen over de
\omterm{def:g11:fundamental-interactions:ladder}{kern} terwijl sterke lijm
alleen buren bindt: zware kernen overleven alleen met surplusneutronen,
lijm zonder afstoting.

\textbf{8.} ${}^{238}_{92}\mathrm{U} \to {}^{234}_{90}\mathrm{Th} +
{}^{4}_{2}\mathrm{He}$.

\textbf{9.} ${}^{234}_{90}\mathrm{Th} \to {}^{234}_{91}\mathrm{Pa} +
{}^{\;0}_{-1}\mathrm{e}$, daarna ${}^{234}_{91}\mathrm{Pa} \to
{}^{234}_{92}\mathrm{U} + {}^{\;0}_{-1}\mathrm{e}$: uranium verschijnt
opnieuw, als uranium-234.

\textbf{10.} $A$: $238 - 206 = 32$, dus $8$ $\alpha$-stappen; $Z$:
$92 - 16 + y = 82$, dus $6$ $\beta^-$-stappen.

\textbf{11.} $\num{4.5e9}$ jaar is één
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}: $\frac12$
blijft. Vijf miljard jaar meer $\approx$ één verdere
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}: ongeveer
$\frac14$.

\textbf{12.} Ze drijft vulkanen, platentektoniek (drijvende continenten,
aardbevingen) en geothermische warmte: de oppervlaktegeologie van een
planeet van binnen verwarmd.

\textbf{13.} ${}^{241}_{95}\mathrm{Am} \to {}^{237}_{93}\mathrm{Np} +
{}^{4}_{2}\mathrm{He}$.

\textbf{14.} $\num{3.3e4}$ vervallen per seconde;
$\num{33000} \times 86400 \approx \num{2.9e9}$ per dag.

\textbf{15.} Elke $\alpha$ ioniseert de lucht die ze doorkruist; de
ionen voeren een minuscule stroom tussen de elektroden van de kamer.
Rookdeeltjes vangen de ionen, de stroom daalt, het alarm gaat af.

\textbf{16.} De $\alpha$'s besteden al hun
\omterm{def:g10:energy-conservation:energy}{energie} binnen de kamer en
geen ontsnapt de behuizing: maximale ionisatie, nul lekkage. Een
$\gamma$-bron van gelijke
\omterm{def:g11:nucleus-radioactivity:activity}{activiteit} zou de lucht van
de kamer nauwelijks ioniseren en in plaats daarvan de hele kamer
bestralen.

\textbf{17.} Tien jaar is $10/432 \approx 2\%$ van een
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}: de
\omterm{def:g11:nucleus-radioactivity:activity}{activiteit} is in wezen
onveranderd. De zwakke schakels zijn stof, insecten en elektronica ---
de \omterm{def:g11:fundamental-interactions:ladder}{kern} overleeft het
apparaat.

\textbf{18.} $\num{9e9}$ jaar $= 2$ halfwaardetijden: $\frac14$ overleeft.
Het uranium van de Aarde is nog overvloedig, dus het is hoogstens enkele
halfwaardetijden geleden gesmeed --- de planeet condenseerde uit verse
sterrenas, miljarden jaren na de eerste sterren.

\textbf{19.} Geologisch dood: een koude binnenkant, geen vulkanisme of
platentektoniek om lucht en rots te recyclen, en geen kolkende kern ---
vandaar geen magnetisch schild tegen de zonnewind.

\textbf{20.} Elk \omterm{def:g11:fundamental-interactions:ladder}{atoom} van
jou voorbij helium werd gesmeed in een stervende ster, en de onstabiele
resten van die smederij verwarmen nog de grond waarop je staat:
sterrenstof, in leven gehouden door haar eigen
\omterm{def:g11:nucleus-radioactivity:halflife}{halfwaardetijd}.
\end{solution}
