\section*{Hoofdstuk \ref{ch:g11:work-of-force} --- Arbeid van een kracht}

\begin{solution}{exo:g11:work-of-force:1}
$W = 40 \times 50 \times \cos\theta$: $+\qty{2000}{J}$;
$+\qty{1000}{J}$; $0$; $-\qty{1000}{J}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:2}
$mgh = 12 \times 9.81 \times 2.5 = \qty{294}{J}$: neergelaten,
$+\qty{294}{J}$ (\omterm{def:g10:universal-gravitation:weight}{gewicht} langs
de val); opgetild, $-\qty{294}{J}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:3}
(a) \omterm{def:g11:work-of-force:sign}{motorisch}; (b)
\omterm{def:g11:work-of-force:sign}{weerstand}; (c) nul (loodrecht); (d)
\omterm{def:g11:work-of-force:sign}{weerstand}; (e) nul --- de
\omterm{def:g10:inertia:inventory}{spanning} van het touw blijft loodrecht op
de beweging van de bob.
\end{solution}

\begin{solution}{exo:g11:work-of-force:4}
Constante snelheid: hijskracht $= mg = \qty{5.89e3}{N}$, dus
$W_{\text{hijs}} = 5886 \times 25 \approx +\qty{147}{kJ}$ en
$W(\vect P) = -\qty{147}{kJ}$. Som nul: bij constante snelheid balanceren
de \omterm{def:g10:inertia:force}{krachten}
(\cref{ch:g11:forces-and-motion}), dus hun
\omterm{def:g11:work-of-force:work}{arbeiden} heffen op.
\end{solution}

\begin{solution}{exo:g11:work-of-force:5}
$P = \num{9.0e4}/60 = \qty{1.5}{kW}$; $1500/736 \approx \qty{2.0}{hp}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:6}
Touw: $120 \times 200 \times \cos\qty{30}{\degree} \approx
+\qty{20.8}{kJ}$; \omterm{def:g10:inertia:inventory}{wrijving}:
$-45 \times 200 = -\qty{9.0}{kJ}$;
\omterm{def:g10:universal-gravitation:weight}{gewicht} en normaalkracht,
loodrecht: $0$. Totaal $+\qty{11.8}{kJ}$: de slee wint snelheid.
\end{solution}

\begin{solution}{exo:g11:work-of-force:7}
$v = 25/3.6 = \qty{6.94}{m/s}$; $P = Fv = 18 \times 6.94 \approx
\qty{125}{W}$. In één uur, $W = P\,\Delta t = 125 \times 3600 =
\qty{450}{kJ}$ (equivalent $18 \times \qty{25000}{m}$).
\end{solution}

\begin{solution}{exo:g11:work-of-force:8}
\omterm{def:g10:universal-gravitation:weight}{Gewicht}, beide routes:
$+65 \times 9.81 \times 120 \approx +\qty{76.5}{kJ}$ (alleen daling).
\omterm{def:g10:inertia:inventory}{Wrijving}: $-30 \times 800 =
-\qty{24}{kJ}$ op de helling, $-30 \times 1500 = -\qty{45}{kJ}$ op de
baan. Alleen \omterm{def:g10:inertia:inventory}{wrijving} geeft om de route.
\end{solution}

\begin{solution}{exo:g11:work-of-force:9}
Stijging $h = 4.0\sin\qty{25}{\degree} = \qty{1.69}{m}$:
$W(\vect P) = -20 \times 9.81 \times 1.69 \approx -\qty{332}{J}$;
$W_f = -30 \times 4.0 = -\qty{120}{J}$. Constante snelheid: totale
\omterm{def:g11:work-of-force:work}{arbeid} nul, dus
$W_{\text{duw}} = +\qty{452}{J}$ en $F = 452/4.0 \approx \qty{113}{N}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:10}
$v = 130/3.6 = \qty{36.1}{m/s}$; constante snelheid, dus
$F = P/v = \num{5.0e4}/36.1 \approx \qty{1.4}{kN}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:11}
$P = Fv\cos\theta = 800 \times 1.0 \times \cos\qty{20}{\degree}
\approx \qty{752}{W}$; in
\omterm{def:g11:work-of-force:power}{paardenkracht}, $752/736 \approx
\num{1.02}$ --- precies wat een sterk paard aanhoudt, de vergelijking
waarop Watt zijn \omterm{def:g10:orders-of-magnitude:unit}{eenheid} bouwde.
\end{solution}

\begin{solution}{exo:g11:work-of-force:12}
Lengtes $L = h/\text{helling}$: \qty{1500}{m} en \qty{5000}{m}.
\omterm{def:g10:inertia:force}{Krachten} $F = mg \times \text{helling}$:
$1200 \times 9.81 \times 0.20 \approx \qty{2.35}{kN}$ en
$\qty{0.71}{kN}$. \omterm{def:g11:work-of-force:work}{Arbeiden} $FL$:
$2354 \times 1500 \approx \qty{3.53}{MJ}$ en
$706 \times 5000 \approx \qty{3.53}{MJ}$ --- beide gelijk $mgh$: de weg
ruilt \omterm{def:g10:inertia:force}{kracht} voor lengte, nooit
\omterm{def:g11:work-of-force:work}{arbeid}.
\end{solution}

\begin{solution}{exo:g11:work-of-force:13}
$F = mg = \qty{2.45e3}{N}$; $W = 2453 \times 15 \approx \qty{36.8}{kJ}$;
$\Delta t = 15/0.40 = \qty{37.5}{s}$;
$P = 36800/37.5 \approx \qty{981}{W} = 2453 \times 0.40$ ($Fv$
klopt); elektrisch: $981/0.70 \approx \qty{1.4}{kW}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:14}
Diagonaal: $-140 \times 6.0 = -\qty{840}{J}$; muren:
$-140 \times 8.4 \approx -\qty{1.18}{kJ}$; heen en terug:
$-140 \times 12 = -\qty{1.68}{kJ}$. Het
\omterm{def:g10:universal-gravitation:weight}{gewicht} levert nul op de
rondreis (geen netto daling). De
\omterm{def:g11:work-of-force:work}{arbeid} van
\omterm{def:g10:inertia:inventory}{wrijving} hangt af van het pad en
betaalt nooit terug; die van het
\omterm{def:g10:universal-gravitation:weight}{gewicht} hangt alleen af van
de eindpunten (\cref{rem:g11:work-of-force:divide}).
\end{solution}

\begin{solution}{exo:g11:work-of-force:15}
Zwaartekracht: $80 \times 9.81 \times 600 \approx \qty{471}{kJ}$;
\omterm{def:g10:inertia:inventory}{wrijving}: $14 \times 12000 =
\qty{168}{kJ}$; totaal $\qty{639}{kJ}$. $v = \qty{4.17}{m/s}$, dus
$\Delta t = 12000/4.17 = \qty{2880}{s} = \qty{48}{min}$ en
$P = \num{6.39e5}/2880 \approx \qty{222}{W} \approx \qty{0.30}{hp}$ ---
een sterke amateur, een derde van een paard.
\end{solution}

\begin{solution}{pb:g11:work-of-force:1}
\textbf{1.} $W = 110 \times 8.0 = +\qty{880}{J}$.

\textbf{2.} \omterm{def:g10:inertia:inventory}{Wrijving}:
$-95 \times 8.0 = -\qty{760}{J}$.
\omterm{def:g10:universal-gravitation:weight}{Gewicht} en normaalkracht zijn
loodrecht op de beweging: elk nul.

\textbf{3.} Totaal $+\qty{120}{J}$: netto
\omterm{def:g11:work-of-force:sign}{motorische arbeid}, de doos versnelt.
Een \qty{95}{N}-duw balanceert
\omterm{def:g10:inertia:inventory}{wrijving} --- nul
\omterm{def:g11:forces-and-motion:netforce}{resulterende kracht}, constante
snelheid, nul totale \omterm{def:g11:work-of-force:work}{arbeid}.

\textbf{4.} Diagonaal: $-95 \times 10.0 = -\qty{950}{J}$; muren:
$-95 \times 14.0 = -\qty{1.33}{kJ}$:
\omterm{def:g10:inertia:inventory}{wrijving} rekent per
\omterm{def:g10:orders-of-magnitude:unit}{meter}.

\textbf{5.} \omterm{def:g10:inertia:inventory}{Wrijving}:
$-95 \times 20.0 = -\qty{1.9}{kJ}$;
\omterm{def:g10:universal-gravitation:weight}{gewicht}: $0$ (geen netto
daling). Alleen het
\omterm{def:g10:universal-gravitation:weight}{gewicht} betaalt terug.

\textbf{6.} $W(\vect P) = -30 \times 9.81 \times 9.0 =
-\qty{2.65}{kJ}$; de ploeg moet minstens $+\qty{2.65}{kJ}$ leveren.

\textbf{7.} $(70 + 30) \times 9.81 \times 9.0 = \qty{8.83}{kJ}$; het
aandeel van de doos is $2.65/8.83 \approx 30\%$ --- dragen is vooral
zelftransport.

\textbf{8.} $\sin\alpha = 9.0/30 = 0.30$:
$F = mg\sin\alpha + f = 88.3 + 40 = \qty{128}{N}$;
$W = 128 \times 30 = \qty{3.85}{kJ}$ ($= 2.65 + 1.2$).

\textbf{9.} Verticale hijs: $F = mg = \qty{294}{N}$,
$W = \qty{2.65}{kJ}$. De helling koopt een kleinere
\omterm{def:g10:inertia:force}{kracht} ($128$ tegen \qty{294}{N}) ten koste
van meer \omterm{def:g11:work-of-force:work}{arbeid}: de
\qty{1.2}{kJ}-tol van \omterm{def:g10:inertia:inventory}{wrijving}.

\textbf{10.} $\Delta t = 9.0/0.50 = \qty{18}{s}$;
$P = 2649/18 \approx \qty{147}{W}$.

\textbf{11.} $P = Fv = 294 \times 0.50 = \qty{147}{W}$ --- hetzelfde.
Elektrisch: $147/0.60 \approx \qty{245}{W}$.

\textbf{12.} $1000 \times 0.24 = \qty{240}{m}$ en
$8000 \times 0.030 = \qty{240}{m}$: dezelfde top.

\textbf{13.} $F = mg \times \text{helling}$: steile baan
$3500 \times 9.81 \times 0.24 \approx \qty{8.2}{kN}$; weg
$\approx \qty{1.0}{kN}$.

\textbf{14.} $8240 \times 1000 \approx \qty{8.24}{MJ}$ en
$1030 \times 8000 \approx \qty{8.24}{MJ}$ --- beide $= mgh = 3500
\times 9.81 \times 240$.

\textbf{15.} Rolweerstand: $400 \times 1000 = \qty{0.40}{MJ}$ tegen
$400 \times 8000 = \qty{3.2}{MJ}$: het comfort van de zachte weg kost
\qty{2.8}{MJ} extra \omterm{def:g10:inertia:inventory}{wrijving}.

\textbf{16.} $v = \qty{10}{m/s}$. Weg:
$(1030 + 400) \times 10 = \qty{14.3}{kW} \approx \qty{19}{hp}$; baan:
$(8240 + 400) \times 10 = \qty{86.4}{kW} \approx \qty{117}{hp}$ ---
boven een beladen bestelwagen. Wegen zigzagggen zodat bescheiden motoren
(en banden) kunnen klimmen: zelfde
\omterm{def:g11:work-of-force:work}{arbeid}, \omterm{def:g11:lenses-and-eye:lens}{dun}
uitgespreid.

\textbf{17.} Per doos: $880 + 2649 = \qty{3.53}{kJ}$; voor $60$ dozen,
$60 \times 3529 \approx \qty{212}{kJ}$.

\textbf{18.} Trappen: $20 \times 70 \times 9.81 \times 9.0 \approx
\qty{124}{kJ}$. Dagtotaal $\approx \qty{336}{kJ} = \qty{0.34}{MJ}$.

\textbf{19.} $P = \num{3.36e5}/28800 \approx \qty{12}{W}$. De waterkoker:
$\num{3.36e5}/2000 = \qty{168}{s}$ --- onder drie minuten voor de hele
dag arbeid.

\textbf{20.} De \omterm{def:g11:work-of-force:work}{joule} betaalt voor
\omterm{def:g10:inertia:force}{kracht} geleverd \emph{langs} een beweging ---
niets voor de geduwde muur of de vastgehouden doos, hoe uitputtend ook
(spieren verbranden \omterm{def:g10:energy-conservation:forms}{chemische
energie} zelfs bij nul mechanische
\omterm{def:g11:work-of-force:work}{arbeid}). Omdat menselijk
\omterm{def:g11:work-of-force:power}{vermogen} slechts enkele tientallen
\omterm{def:g11:work-of-force:power}{watt} is, laten verhuizers machines de
\omterm{def:g10:inertia:force}{kracht} hervormen --- lier, helling, zigzag ---
terwijl de rekening $mgh$ nooit verandert.
\end{solution}
