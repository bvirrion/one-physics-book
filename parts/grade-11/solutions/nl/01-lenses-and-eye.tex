\section*{Hoofdstuk \ref{ch:g11:lenses-and-eye} --- Lenzen, beelden en het oog}

\begin{solution}{exo:g11:lenses-and-eye:1}
$C = 1/0.50 = +2.0$~D; $1/(-0.20) = -5.0$~D; $1/0.0040 = +250$~D.
Een $+8.0$~D \omterm{def:g11:lenses-and-eye:lens}{lens}:
$f' = 1/8.0 = \qty{0.125}{m} = \qty{12.5}{cm}$.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:2}
\omterm{def:g11:lenses-and-eye:lens}{Divergerend} ($f' < 0$). Nee: voor elke
voorwerpsafstand $d$ ligt $\gamma = f'/(f' - d)$ (getoond in
\cref{exo:g11:lenses-and-eye:15}) tussen $0$ en $1$ --- het beeld is
altijd \omterm{def:g11:lenses-and-eye:images}{virtueel}, rechtop en
\emph{verkleind}. De druk verschijnt rechtop en verkleind, nooit
vergroot.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:3}
Het voorwerp staat op $2f'$; de centrale en evenwijdige stralen snijden
elkaar \qty{20}{cm} achter de \omterm{def:g11:lenses-and-eye:lens}{lens}. Controle:
$1/\overline{OA'} = 1/10 - 1/20 = 1/20$, dus
$\overline{OA'} = +\qty{20}{cm}$ en $\gamma = 20/(-20) = -1$:
\omterm{def:g11:lenses-and-eye:images}{reëel}, omgekeerd, ware grootte --- zoals
getekend.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:4}
$1/\overline{OA'} = 1/8.0 - 1/24 = 1/12$:
$\overline{OA'} = +\qty{12}{cm}$, $\gamma = 12/(-24) = -0.50$.
\omterm{def:g11:lenses-and-eye:images}{Reëel beeld} \qty{12}{cm} achter de
\omterm{def:g11:lenses-and-eye:lens}{lens}, omgekeerd, half zo groot.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:5}
\omterm{def:g11:lenses-and-eye:eye}{Hoornvlies} en
\omterm{def:g11:lenses-and-eye:eye}{ooglens} vormen de
\omterm{def:g11:lenses-and-eye:lens}{convergerende lens}; het
\omterm{def:g11:lenses-and-eye:eye}{netvlies} is het scherm.
\omterm{def:g11:lenses-and-eye:eye}{Accommodatie} verandert de
\omterm{def:g11:lenses-and-eye:focal}{brandpuntsafstand}
(\omterm{def:g11:lenses-and-eye:vergence}{sterkte}) van de
\omterm{def:g11:lenses-and-eye:eye}{ooglens}; de afstand lens--netvlies kan
niet veranderen. \omterm{def:g11:lenses-and-eye:eye}{Nabijheidspunt}: dichtstbijzijnde
punt scherp gezien bij volle \omterm{def:g11:lenses-and-eye:eye}{accommodatie} ---
\qty{25}{cm} voor het \omterm{def:g11:lenses-and-eye:eye}{standaardoog}.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:6}
Landschap: sensor in het brandvlak, \qty{50}{mm} van de
\omterm{def:g11:lenses-and-eye:lens}{lens}. Op \qty{3.0}{m}:
$1/\overline{OA'} = 1/50 - 1/3000 = 59/3000$, dus
$\overline{OA'} \approx \qty{50.8}{mm}$. Verplaatsing: ongeveer
\qty{0.8}{mm}.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:7}
$1/\overline{OA'} = 1/6.0 - 1/5.0 = -1/30$:
$\overline{OA'} = \qty{-30}{cm}$ --- \omterm{def:g11:lenses-and-eye:images}{virtueel},
rechtop, $\gamma = (-30)/(-5.0) = +6.0$: zes keer de postzegel,
\qty{30}{cm} boven de \omterm{def:g11:lenses-and-eye:lens}{lens}.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:8}
$1/f' = 1/40 + 1/40 = 1/20$: $f' = \qty{20}{cm}$;
$\gamma = 40/(-40) = -1$. Symmetrie
\omterm{def:g10:inertia:force}{dwingt}
$\abs{\overline{OA'}} = \abs{\overline{OA}}$, vandaar
$\abs{\gamma} = 1$; het beeld is
\omterm{def:g11:lenses-and-eye:images}{reëel} en omgekeerd, dus $\gamma = -1$:
precies ware grootte.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:9}
$1/\overline{OA'} = -1/25 - 1/50 = -3/50$:
$\overline{OA'} \approx \qty{-16.7}{cm}$,
$\gamma = (-16.7)/(-50) = +0.33$: \omterm{def:g11:lenses-and-eye:images}{virtueel},
rechtop, een derde van de grootte. De bril van een bijziende is
\omterm{def:g11:lenses-and-eye:lens}{divergerend}: alles wat erdoor gezien wordt
--- inclusief, van buiten, de ogen van de drager --- is verkleind.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:10}
Beeld van oneindig op het \omterm{def:g11:lenses-and-eye:eye}{verheidspunt}:
$f' = \qty{-0.40}{m}$, dus $C = -2.5$~D. Het beeld van de berg zit
\qty{40}{cm} vóór de \omterm{def:g11:lenses-and-eye:lens}{lens} --- precies het
\omterm{def:g11:lenses-and-eye:eye}{verheidspunt}, dat het ontspannen oog scherp
ziet met nul \omterm{def:g11:lenses-and-eye:eye}{accommodatie}.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:11}
$C = 1/\overline{OA'} - 1/\overline{OA} = 1/(-0.80) - 1/(-0.25)
= -1.25 + 4.00 = +2.75$~D.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:12}
\emph{1.} $\gamma = \overline{OA'}/\overline{OA} = -50$ geeft
$\overline{OA'} = -50\,\overline{OA}$; met
$\overline{OA'} - \overline{OA} = \qty{5.1}{m}$,
$-51\,\overline{OA} = \qty{5.1}{m}$: $\overline{OA} = \qty{-0.10}{m}$,
$\overline{OA'} = +\qty{5.0}{m}$.

\emph{2.} $1/f' = 1/5.0 + 1/0.10 = 10.2$: $f' \approx \qty{9.8}{cm}$.

\emph{3.} $50 \times \qty{24}{mm} = \qty{1.2}{m}$, omgekeerd --- daarom
worden dia's ondersteboven geladen.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:13}
\emph{1.} $C_\infty = 1/0.017 \approx 59$~D;
$C_{25} = 1/0.017 + 1/0.25 \approx 63$~D.
\omterm{def:g10:signals-and-waves:amplitude}{Amplitude}: $1/0.25 = 4.0$~D.

\emph{2.} \omterm{def:g10:signals-and-waves:amplitude}{Amplitude} $= 1/1.0 = 1.0$~D:
een kwart ($25\%$) van de $4.0$~D van het
\omterm{def:g11:lenses-and-eye:eye}{standaardoog} blijft over.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:14}
\emph{1.} $f' = 0.25/3 \approx \qty{0.083}{m} = \qty{8.3}{cm}$.

\emph{2.} $\overline{OA'} = \qty{-25}{cm}$:
$1/\overline{OA} = -1/25 - 3/25 = -4/25$, dus
$\overline{OA} = \qty{-6.25}{cm}$ en
$\gamma = (-25)/(-6.25) = +4.0$ --- beter dan de geadverteerde
$\times 3$, die het beeld in oneindig veronderstelt; het naar het
\omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} trekken wint één
\omterm{def:g10:orders-of-magnitude:unit}{eenheid}.
\end{solution}

\begin{solution}{exo:g11:lenses-and-eye:15}
$1/\overline{OA'} = 1/f' - 1/d = (d - f')/(f'd)$, dus
$\overline{OA'} = f'd/(d - f')$ en
$\gamma = \overline{OA'}/(-d) = f'/(f' - d)$. Als $d > f'$:
$\overline{OA'} > 0$ (\omterm{def:g11:lenses-and-eye:images}{reëel}) en $\gamma < 0$
(omgekeerd). Als $0 < d < f'$: $\overline{OA'} < 0$
(\omterm{def:g11:lenses-and-eye:images}{virtueel}) en
$\gamma = f'/(f' - d) > 1$: rechtop en vergroot --- de
\omterm{def:g11:lenses-and-eye:magnifier}{loep}-stand.
\end{solution}

\begin{solution}{pb:g11:lenses-and-eye:1}
\textbf{1.} $1/\overline{OA'} = 1/20 - 1/60 = 1/30$:
$\overline{OA'} = +\qty{30}{cm}$, $\gamma = -0.50$ ---
\omterm{def:g11:lenses-and-eye:images}{reëel}, omgekeerd, half zo groot.

\textbf{2.} $\overline{OA'} = +\qty{60}{cm}$, $\gamma = -2$: de
posities \qty{30}{cm} en \qty{60}{cm} hebben rollen geruild, de
\omterm{def:g11:lenses-and-eye:magnification}{vergrotingen} zijn omgekeerd ---
omgekeerd licht volgt hetzelfde pad.

\textbf{3.} $1/\overline{OA'} = 1/20 - 1/10 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, $\gamma = +2$ ---
\omterm{def:g11:lenses-and-eye:images}{virtueel}, rechtop, verdubbeld.

\textbf{4.} Vanuit oneindig begint het beeld in het brandvlak; terwijl
het voorwerp $F$ nadert, wijkt het \omterm{def:g11:lenses-and-eye:images}{reële},
omgekeerde beeld terug naar oneindig en groeit; binnen $f'$ wordt het
\omterm{def:g11:lenses-and-eye:images}{virtueel}, rechtop, vergroot, vóór de
\omterm{def:g11:lenses-and-eye:lens}{lens}.

\textbf{5.} $1/\overline{OA} \approx 0$ laat
$1/\overline{OA'} = 1/f'$ over: $\overline{OA'} = f'$, het brandvlak.

\textbf{6.} In het brandvlak, \qty{50}{mm} achter de
\omterm{def:g11:lenses-and-eye:lens}{lens}.

\textbf{7.} $1/\overline{OA'} = 1/50 - 1/1000 = 19/1000$:
$\overline{OA'} \approx \qty{52.6}{mm}$; verplaatsing
$\approx \qty{2.6}{mm}$.

\textbf{8.} $1/\overline{OA} = 1/55 - 1/50 = -1/550$: minimale
scherpstelafstand \qty{550}{mm} = \qty{55}{cm}.

\textbf{9.} $\gamma = 55/(-550) = -0.10$: het gezicht beeldt af op
\qty{20}{mm}, net binnen de \qty{24}{mm}-sensor.

\textbf{10.} $\gamma = -1$ \omterm{def:g10:inertia:force}{dwingt}
$\overline{OA'} = -\overline{OA}$, dus $2/\overline{OA'} = 1/f'$:
voorwerp \qty{100}{mm} ($= 2f'$) vóór, sensor \qty{100}{mm} achter ---
dubbel de landschapstrek, ver voorbij het \qty{55}{mm}-mechanisme;
macrolenzen leveren de extra verplaatsing.

\textbf{11.} Alles dichter dan haar
\omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} (\qty{1.0}{m}) is wazig:
gestrekte armen brengen de bladzijde zo dicht mogelijk bij \qty{1.0}{m}
--- scherp, zij het klein.

\textbf{12.} $C = 1/(-1.0) - 1/(-0.25) = -1.0 + 4.0 = +3.0$~D.

\textbf{13.} $\gamma = (-1.0)/(-0.25) = +4.0$: vier keer groter maar
vier keer verder --- dezelfde hoekgrootte; de winst is niet de grootte
maar de \emph{scherpte}, omdat het beeld nu op haar
\omterm{def:g11:lenses-and-eye:eye}{nabijheidspunt} ligt.

\textbf{14.} Scherp vereist het beeld voorbij \qty{1.0}{m}: van de
bladzijde op \qty{25}{cm} (beeld op \qty{1.0}{m}) tot de bladzijde op
$f' = 1/3.0 \approx \qty{0.33}{m}$ (beeld in oneindig). Een ver
voorwerp beeldt \qty{33}{cm} \emph{achter} de bril af --- een
\omterm{def:g11:lenses-and-eye:images}{reëel beeld} dat het oog niet kan
gebruiken: de verre wereld is wazig.

\textbf{15.} Vandaar bifocals: lees-\omterm{def:g10:energy-conservation:power}{vermogen}
in de onderste helft van de \omterm{def:g11:lenses-and-eye:lens}{lens}, geen
(of de afstands-correctie) in de bovenste helft.

\textbf{16.} $1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, \omterm{def:g11:lenses-and-eye:images}{virtueel},
rechtop, $\gamma = +5.0$.

\textbf{17.} $1/\overline{OA} = -1/25 - 1/5.0 = -6/25$:
$\overline{OA} \approx \qty{-4.2}{cm}$; $\gamma = (-25)/(-25/6) = +6.0$.

\textbf{18.} Steen op $F$: beeld in oneindig, bekeken met nul
\omterm{def:g11:lenses-and-eye:eye}{accommodatie} --- uren inspectie zonder
oogvermoeidheid.

\textbf{19.} $C_\infty = 1/0.017 \approx 59$~D;
$C_{25} = 1/0.017 + 1/0.25 \approx 63$~D: ongeveer $4$~D
\omterm{def:g11:lenses-and-eye:eye}{accommodatie}.

\textbf{20.} $f' = 1/58.8 \approx \qty{17.0}{mm}$ tegen
$1/62.8 \approx \qty{15.9}{mm}$: ongeveer \qty{1.1}{mm}, zo'n $6\%$ ---
de ingebouwde zoom die de camera imiteerde met \qty{2.6}{mm}
verplaatsing en grootmoeder met $+3$~D compenseerde.
\end{solution}
