\section*{अध्याय \ref{ch:g11:magnetic-fields} --- चुंबकत्व और चुंबकीय क्षेत्र}

\begin{solution}{exo:g11:magnetic-fields:1}
प्रत्येक टुकड़ा पूर्ण है \omterm{def:g11:magnetic-fields:magnet}{चुंबक} एक उत्तरी और एक दक्षिणी ध्रुव के साथ: एक नया
प्रत्येक कटे चेहरे पर ध्रुव दिखाई देता है। ताज़ा चेहरे N और S हैं, इसलिए
वे आकर्षित करते हैं --- टुकड़े पुनः एकत्रित होने का प्रयास करते हैं। कोई भी योजना अलग नहीं होती
ध्रुव: प्रत्येक कट से एक जोड़ी बनती है।
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:2}
(ए) मिथ्या: द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} प्रत्येक बिंदु पर एक दिशा होती है। (बी) सच है.
(सी) सत्य: प्रत्येक पंक्ति इसके माध्यम से बंद होती है चुंबकका शरीर. (घ) मिथ्या: द
सुई बैठ जाती है \emph{साथ में} रेखा, इसके स्पर्शरेखा.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:3}
$B = \num{2e-7} \times 10 / 0.020 = \qty{1.0e-4}{T}$ --- दो बार
पृथ्वी का \omterm{def:g11:electric-gravitational-fields:field}{मैदान}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:4}
$ n = 800/0.40 = \qty{2000}{m^{-1}}$;
$ B = \num{1.26e-6} \times 2000 \times 1.5 \approx \qty{3.8}{mT}$.
$ B \propto I $: धारा को दोगुना करने से लाभ मिलता है \qty{7.5}{mT}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:5}
$ F = ILB = 5.0 \times 0.10 \times 0.20 = \qty{0.10}{N}$ --- द \omterm{def:g10:universal-gravitation:weight}{वज़न}
का $ m = F/g = 0.10/9.81 \approx \qty{10}{g}$.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:6}
\emph{1.} उसके ट्रैक पॉइंट $\ang{8}$ सच्चे उत्तर के पश्चिम:
$2000 \times \sin\ang{8} \approx \qty{2.8e2}{m}$ पश्चिम की ओर.

\emph{2.} रास्ते पर लाना $\ang{8}$ \emph{पूर्व} सुई के उत्तर की ओर.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:7}
$ d = \mu_0 I / (2\pi B) = \num{2e-7} \times 20 / \num{5.0e-5}
= \qty{8.0}{cm}$. ऐसी केबल के एक डेसी\omterm{def:g10:orders-of-magnitude:unit}{मीटर} के भीतर कम्पास पढ़ता है
तार, ग्रह नहीं --- इसे जीवित कंडक्टरों से दूर रखें।
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:8}
$ n = B/(\mu_0 I) = 0.010 / (\num{1.26e-6} \times 4.0)
\approx \qty{2.0e3}{m^{-1}}$; ऊपर \qty{25}{cm},
$ N = 1989 \times 0.25 \approx 500$ बदल जाता है.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:9}
$ F = 10 \times 0.12 \times 0.50 = \qty{0.60}{N}$;
$ a = F/m = 0.60/0.020 = \qty{30}{m/s^2}$ --- के बारे में $3g $: \omterm{def:g11:color-light-sources:cones}{छड़}ी
वास्तव में कूदता है.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:10}
संतुलन: $ ILB = mg $, इसलिए
$ I = \num{5.0e-3} \times 9.81 / (0.20 \times 0.10) \approx
\qty{2.5}{A}$. बल ऊपर इंगित करना चाहिए; साथ $\vect B $ क्षैतिज और
तार के लंबवत, सपाट दाहिना हाथ (हथेली ऊपर) ठीक करता है
आवश्यक वर्तमान दिशा.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:11}
सपाट दाहिना हाथ: (ए) पृष्ठ के शीर्ष की ओर; (बी) दाईं ओर;
(सी) नहीं बल --- तार के समानांतर है \omterm{def:g11:electric-gravitational-fields:field}{मैदान}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:12}
\emph{1.} $ B = \num{2e-7} \times 5.0 / 0.030 \approx
\qty{33}{\micro T}$; रेखाएं तार को घेरती हैं, इसलिए सीधे इसके नीचे
द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} क्षैतिज है, तार के लंबवत है: पूर्व--पश्चिम।

\emph{2.} सुई वेक्टर योग का अनुसरण करती है: $\tan\theta = 33/20 =
1.67$, इसलिए $\theta \approx \ang{59}$ उत्तर से दूर.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:13}
\emph{1.} $ I = U/R = 12/2.0 = \qty{6.0}{A}$;
$ n = 400/0.20 = \qty{2000}{m^{-1}}$;
$ B_0 = \num{1.26e-6} \times 2000 \times 6.0 \approx \qty{15}{mT}$;
कोर के साथ, $100 B_0 \approx \qty{1.5}{T}$.

\emph{2.} $ P = UI = 12 \times 6.0 = \qty{72}{W}$.

\emph{3.} कोई करंट नहीं, नहीं \omterm{def:g11:electric-gravitational-fields:field}{मैदान} ---और नरम लोहा टिकता नहीं
चुम्बकित: डिजाइन के अनुसार लोड तुरंत रिलीज होता है।
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:14}
\emph{1.} $ F = NILB = 50 \times 2.0 \times 0.050 \times 0.30 =
\qty{1.5}{N}$ हर तरफ. दोनों में धारा विपरीत दिशा में चलती है
पक्ष, तो बल विपरीत हैं: एक जोड़ा जो लूप को घुमाता है।

\emph{2.} जब के समानांतर $\vect B $, एक तार नहीं लगता है \omterm{prop:g11:magnetic-fields:laplace}{लाप्लास बल}.

\emph{3.} आधे मोड़ के बाद जोड़ा उलट जाएगा और पूर्ववत कर देगा
घूर्णन; \omterm{rem:g11:magnetic-fields:motor}{कम्यूटेटर} प्रत्येक आधे मोड़ पर लूप में धारा प्रवाहित होती है
एक तरफ घूमता रहता है.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:15}
\emph{1.} $ d = \num{2e-7} \times 100 / 1 = \qty{2.0e-5}{m} =
\qty{20}{\micro m}$ --- गहरा \emph{अंदर} मिलीमीटर-मोटा तांबा।
एक भी तार के पास कोई सुलभ बिंदु नहीं पहुंचता है \qty{1}{T}: मज़बूत
खेत हजारों घुमावों को ढेर करें (ए \omterm{def:g11:magnetic-fields:solenoid}{solenoid}), प्लस आयरन या
अतिचालक.

\emph{2.} $\num{e8} / \num{5e-5} = \num{2e12}$: लगभग बारह पॉवर्स अक्सर।
\end{solution}

\begin{solution}{pb:g11:magnetic-fields:1}
\textbf{1.} सुई स्वयं छोटी है \omterm{def:g11:magnetic-fields:magnet}{चुंबक}; पृथ्वी का \omterm{def:g11:electric-gravitational-fields:field}{मैदान}
इसे क्षैतिज घटक के साथ उत्तरी छोर पर संरेखित करता है $\vect B $,
यानी\ \omterm{def:g11:magnetic-fields:magnet}{चुंबक}ीय उत्तर की ओर.

\textbf{2.} $ B = \sqrt{20^2 + 44^2} = \sqrt{2336} \approx
\qty{48}{\micro T}$, झुका हुआ $\tan^{-1}(44/20) \approx \ang{66}$ नीचे
क्षैतिज.

\textbf{3.} ट्रैक चलता है $\ang{6}$ सच्चे उत्तर के पश्चिम:
$30 \times \sin\ang{6} \approx \qty{3.1}{km}$ पश्चिम.

\textbf{4.} त्रुटियाँ जोड़ें: $\ang{10}$, इसलिए $30 \times \sin\ang{10} \approx
\qty{5.2}{km}$ --- दस डिग्री के लिए पांच किलो\omterm{def:g10:orders-of-magnitude:unit}{मीटर}।

\textbf{5.} क्षैतिज घटक: ध्रुव के पास \omterm{def:g11:electric-gravitational-fields:field}{मैदान} है
लगभग ऊर्ध्वाधर, क्षैतिज भाग शून्य की ओर सिकुड़ जाता है, और
सुई के पास संरेखित करने के लिए कुछ भी नहीं बचा है।

\textbf{6.} $ n = 600/0.30 = \qty{2000}{m^{-1}}$.

\textbf{7.} $ I = U/R = 12/1.5 = \qty{8.0}{A}$.

\textbf{8.} $ B_0 = \mu_0 n I = \num{1.26e-6} \times 2000 \times 8.0
\approx \qty{20}{mT}$.

\textbf{9.} $50 \times \qty{20}{mT} = \qty{1.0}{T}$ --- दो बार
\omterm{def:g11:electric-gravitational-fields:field}{मैदान} एक नियोडिमियम ध्रुव पर, कहीं अधिक बड़े चेहरे पर।

\textbf{10.} $ P = UI = 12 \times 8.0 = \qty{96}{W}$; में
\qty{10}{\minute}: $ E = 96 \times 600 \approx \qty{58}{kJ}$.

\textbf{11.} क्योंकि यह जाने देता है: स्विच खोलें, \omterm{def:g11:electric-gravitational-fields:field}{मैदान} मर जाता है,
कार वहीं गिरती है जहां चाहिए। एक स्थायी \omterm{def:g11:magnetic-fields:magnet}{चुंबक} हमेशा के लिए पकड़ लेता है.

\textbf{12.} $ F = NILB = 100 \times 3.0 \times 0.040 \times 0.25 =
\qty{3.0}{N}$.

\textbf{13.} एक युगल: अक्ष के विपरीत पक्षों पर लागू, दो
विपरीत बल दोनों लूप को एक ही तरह से घुमाते हैं। बराबर \emph{समानांतर}
बल केवल लूप को किनारे की ओर धकेलेगा, घुमाएगा नहीं।

\textbf{14.} कोई नहीं: के समानांतर एक तार \omterm{def:g11:electric-gravitational-fields:field}{मैदान} नहीं लगता \omterm{prop:g11:magnetic-fields:laplace}{लाप्लास
बल}.

\textbf{15.} यह प्रत्येक आधे-मोड़ पर लूप में धारा को उलट देता है,
जिस तरह जोड़ा \omterm{def:g10:signals-and-waves:signal}{संकेत} बदलता है --- उसी तरह टॉर्क हमेशा चलता रहता है
वही घुमाव.

\textbf{16.} दोहरा $ N $, या दोगुना $ I $, या दोगुना $ B $: द बल जोड़ा
$ F = NILB $ प्रत्येक के लिए आनुपातिक है.

\textbf{17.} $1.5 / \num{5.0e-5} = \num{3.0e4}$: तीस हजार
पृथ्वी.

\textbf{18.} $ n = B/(\mu_0 I) = 1.5 / (\num{1.26e-6} \times 500)
\approx \qty{2.4e3}{m^{-1}}$ --- और उनमें से प्रत्येक मोड़ वहन करता है
\qty{500}{A}.

\textbf{19.} $ P = RI^2 = 0.20 \times 500^2 = \qty{50}{kW}$ --- ए
पड़ोस को गर्म करने लायक। \omterm{def:g11:lenses-and-eye:images}{असली} वाइंडिंग है
अतिचालक: शून्य \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}, शून्य अपव्यय, बशर्ते यह हो
पूर्ण शून्य से कुछ डिग्री ऊपर रखा गया।

\textbf{20.} दिशा सूचक यंत्र $\num{5e-5}$, क्रेन $1$, एमआरआई $1.5$: जहाज
दो मशीनों से पाँच पायदान नीचे बैठता है, जो एक दशक साझा करते हैं ---
अंतरतारकीय के बीच $\num{e-10}$ और न्यूट्रॉन-स्टार $\num{e8}$. कुछ नहीं
बदल गया लेकिन \omterm{def:g11:circuits-and-power:current}{एम्पीयर} (और मोड़, लोहा और सुपरकंडक्टर
उन्हें गुणा करें): एक \omterm{def:g11:electric-gravitational-fields:field}{मैदान}, तीन नौकरियाँ।
\end{solution}
