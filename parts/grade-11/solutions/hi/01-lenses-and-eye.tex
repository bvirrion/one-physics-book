\begin{solution}{exo:g11:lenses-and-eye:15}
$1/\overline{OA'} = 1/f' - 1/d = (d - f')/(f'd)$, इसलिए
$\overline{OA'} = f'd/(d - f')$ और
$\gamma = \overline{OA'}/(-d) = f'/(f' - d)$. अगर $ d > f'$:
$\overline{OA'} > 0$ (\omterm{def:g11:lenses-and-eye:images}{असली}) और $\gamma < 0$ (उल्टा)। अगर
$0 < d < f'$: $\overline{OA'} < 0$ (\omterm{def:g11:lenses-and-eye:images}{आभासी}) और
$\gamma = f'/(f' - d) > 1$: सीधा और बड़ा --- \omterm{def:g11:lenses-and-eye:magnifier}{ताल}
सेटिंग।
\end{solution}

\begin{solution}{pb:g11:lenses-and-eye:1}
\textbf{1.} $1/\overline{OA'} = 1/20 - 1/60 = 1/30$:
$\overline{OA'} = +\qty{30}{cm}$, $\gamma = -0.50$ --- \omterm{def:g11:lenses-and-eye:images}{असली}, उलटा,
आधे आकार का.

\textbf{2.} $\overline{OA'} = +\qty{60}{cm}$, $\gamma = -2$: द
पद \qty{30}{cm} और \qty{60}{cm} भूमिकाओं का आदान-प्रदान किया गया
आवर्धन पारस्परिक हैं --- उलटा प्रकाश उसी का अनुसरण करता है
पथ.

\textbf{3.} $1/\overline{OA'} = 1/20 - 1/10 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, $\gamma = +2$ --- \omterm{def:g11:lenses-and-eye:images}{आभासी}, सीधा,
दोगुना.

\textbf{4.} अनंत से छवि फोकल तल में शुरू होती है; के रूप में
वस्तु दृष्टिकोण $ F $  \omterm{def:g11:lenses-and-eye:images}{असली}, उलटी छवि अनंत तक पीछे हट जाती है और
बढ़ता है; अंदर $ f'$ यह बदलता है \omterm{def:g11:lenses-and-eye:images}{आभासी}, सीधा, बढ़ा हुआ, सामने
द \omterm{def:g11:lenses-and-eye:lens}{लेंस}.

\textbf{5.} $1/\overline{OA} \approx 0$ पत्तियों
$1/\overline{OA'} = 1/f'$: $\overline{OA'} = f'$, फोकल विमान।

\textbf{6.} फोकल प्लेन में, \qty{50}{mm} पीछे \omterm{def:g11:lenses-and-eye:lens}{लेंस}.

\textbf{7.} $1/\overline{OA'} = 1/50 - 1/1000 = 19/1000$:
$\overline{OA'} \approx \qty{52.6}{mm}$; यात्रा $\approx \qty{2.6}{mm}$.

\textbf{8.} $1/\overline{OA} = 1/55 - 1/50 = -1/550$: न्यूनतम फोकसिंग
दूरी \qty{550}{mm} = \qty{55}{cm}.

\textbf{9.} $\gamma = 55/(-550) = -0.10$: चेहरे की छवियां
\qty{20}{mm}, बस के अंदर \qty{24}{mm} सेंसर.

\textbf{10.} $\gamma = -1$ बल $\overline{OA'} = -\overline{OA}$,
इसलिए $2/\overline{OA'} = 1/f'$: वस्तु \qty{100}{mm} ($= 2f'$) सामने,
सेंसर \qty{100}{mm} पीछे --- लैंडस्केप ड्रा को दोगुना करें, बहुत आगे
द \qty{55}{mm} तंत्र; मैक्रो \omterm{def:g11:lenses-and-eye:lens}{लेंस} अतिरिक्त यात्रा प्रदान करते हैं।

\textbf{11.} उससे भी करीब कुछ भी \omterm{def:g11:lenses-and-eye:eye}{निकट बिंदु} (\qty{1.0}{m}) धुंधला:
फैली हुई भुजाएँ पृष्ठ को जितना करीब लाती हैं \qty{1.0}{m} यथासंभव ---
तीक्ष्ण, यदि छोटा हो।

\textbf{12.} $ C = 1/(-1.0) - 1/(-0.25) = -1.0 + 4.0 = +3.0$~डी.

\textbf{13.} $\gamma = (-1.0)/(-0.25) = +4.0$: चार गुना बड़ा लेकिन
चार गुना दूर --- समान कोणीय आकार; लाभ आकार नहीं है लेकिन
\emph{कुशाग्रता}, चूँकि छवि अब उसके पास है \omterm{def:g11:lenses-and-eye:eye}{निकट बिंदु}.

\textbf{14.} शार्प को परे की छवि की आवश्यकता है \qty{1.0}{m}: पेज से
पर \qty{25}{cm} (छवि पर \qty{1.0}{m}) पृष्ठ पर
$ f' = 1/3.0 \approx \qty{0.33}{m}$ (अनंत पर छवि). एक दूर
वस्तु छवियाँ \qty{33}{cm} \emph{पीछे} चश्मा --- ए वास्तविक छवि
आँख उपयोग नहीं कर सकती: दूर की दुनिया धुंधली हो जाती है।

\textbf{15.} इसलिए बाइफोकल्स: पढ़ना \omterm{def:g10:energy-conservation:power}{शक्ति} के निचले आधे भाग में
\omterm{def:g11:lenses-and-eye:lens}{लेंस}, ऊपरी आधे भाग में कोई नहीं (या दूरी सुधार)।

\textbf{16.} $1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, \omterm{def:g11:lenses-and-eye:images}{आभासी}, सीधा, $\gamma = +5.0$.

\textbf{17.} $1/\overline{OA} = -1/25 - 1/5.0 = -6/25$:

\omterm{def:g11:lenses-and-eye:eye}{आवास} --- आंखों पर \omterm{def:g10:pressure:pressure}{दबाव} डाले बिना घंटों निरीक्षण।

\textbf{19.} $ C_\infty = 1/0.017 \approx 59$~डी;
$ C_{25} = 1/0.017 + 1/0.25 \approx 63$~डी: के बारे में $4$~डी का
