\begin{solution}{exo:g11:work-of-force:4}
लगातार गति: उठाना बल $= mg = \qty{5.89e3}{N}$, इसलिए
$ W_{\text{lift}} = 5886 \times 25 \approx +\qty{147}{kJ}$ और
$ W(\vect P) = -\qty{147}{kJ}$. योग शून्य: स्थिर वेग पर
बल संतुलन (\cref{ch:g11:forces-and-motion}), तो उनका \omterm{def:g11:work-of-force:work}{काम} करता है
रद्द करना।
\end{solution}

\end{solution}

\end{solution}

\begin{solution}{exo:g11:work-of-force:7}
$ v = 25/3.6 = \qty{6.94}{m/s}$; $ P = Fv = 18 \times 6.94 \approx
\qty{125}{W}$. एक घंटे में, $ W = P\,\Delta t = 125 \times 3600 =
\qty{450}{kJ}$ (समान रूप से $18 \times \qty{25000}{m}$).
\end{solution}

\begin{solution}{exo:g11:work-of-force:8}
\omterm{def:g10:universal-gravitation:weight}{वज़न}, दोनों मार्ग: $+65 \times 9.81 \times 120 \approx
+\qty{76.5}{kJ}$ (केवल ड्रॉप). \omterm{def:g10:inertia:inventory}{टकराव}: $-30 \times 800 =
-\qty{24}{kJ}$ ढलान पर, $-30 \times 1500 = -\qty{45}{kJ}$ पटरी पर। केवल \omterm{def:g10:inertia:inventory}{टकराव} मार्ग की परवाह करता है.
\end{solution}

\begin{solution}{exo:g11:work-of-force:9}
उठना $ h = 4.0\sin\qty{25}{\degree} = \qty{1.69}{m}$:
$ W(\vect P) = -20 \times 9.81 \times 1.69 \approx -\qty{332}{J}$;
$ W_f = -30 \times 4.0 = -\qty{120}{J}$. लगातार गति: कुल \omterm{def:g11:work-of-force:work}{काम}
शून्य, तो $ W_{\text{push}} = +\qty{452}{J}$ और
$ F = 452/4.0 \approx \qty{113}{N}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:10}
$ v = 130/3.6 = \qty{36.1}{m/s}$; स्थिर गति, इसलिए
$ F = P/v = \num{5.0e4}/36.1 \approx \qty{1.4}{kN}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:11}
$ P = Fv\cos\theta = 800 \times 1.0 \times \cos\qty{20}{\degree}
\approx \qty{752}{W}$; में \omterm{def:g11:work-of-force:power}{घोड़े की शक्ति}, $752/736 \approx \num{1.02}$ ---
लगभग वैसा ही जैसा एक मजबूत घोड़ा कायम रखता है, तुलना वॉट
उसका निर्माण किया \omterm{def:g10:orders-of-magnitude:unit}{इकाई} पर।
\end{solution}

\begin{solution}{exo:g11:work-of-force:12}
लंबाई $ L = h/\text{grade}$: \qty{1500}{m} और \qty{5000}{m}. बल
$ F = mg \times \text{grade}$: $1200 \times 9.81 \times 0.20 \approx
\qty{2.35}{kN}$ और $\qty{0.71}{kN}$. \omterm{def:g11:work-of-force:work}{काम} करता है $ FL $: $2354 \times 1500
\approx \qty{3.53}{MJ}$ और $706 \times 5000 \approx \qty{3.53}{MJ}$
--- दोनों बराबर $ mgh $: सड़क व्यापार बल लंबाई के लिए, कभी नहीं \omterm{def:g11:work-of-force:work}{काम}.
\end{solution}

\begin{solution}{exo:g11:work-of-force:13}
$ F = mg = \qty{2.45e3}{N}$; $ W = 2453 \times 15 \approx \qty{36.8}{kJ}$;
$\Delta t = 15/0.40 = \qty{37.5}{s}$;
$ P = 36800/37.5 \approx \qty{981}{W} = 2453 \times 0.40$ ($ Fv $ सहमत);
\omterm{def:g11:magnetic-fields:electromagnet}{विद्युत}: $981/0.70 \approx \qty{1.4}{kW}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:14}
विकर्ण: $-140 \times 6.0 = -\qty{840}{J}$; दीवारें:
$-140 \times 8.4 \approx -\qty{1.18}{kJ}$; राउंड ट्रिप:
$-140 \times 12 = -\qty{1.68}{kJ}$. \omterm{def:g10:universal-gravitation:weight}{वज़न} राउंड पर शून्य करता है
यात्रा (कोई नेट ड्रॉप नहीं)। \omterm{def:g10:inertia:inventory}{टकराव}'एस \omterm{def:g11:work-of-force:work}{काम} पथ पर निर्भर करता है और कभी नहीं
रिफंड; वज़नयह केवल अंतिम बिंदुओं पर निर्भर करता है
(\cref{rem:g11:work-of-force:divide}).
\end{solution}

\begin{solution}{exo:g11:work-of-force:15}
गुरुत्वाकर्षण: $80 \times 9.81 \times 600 \approx \qty{471}{kJ}$; \omterm{def:g10:inertia:inventory}{टकराव}:
$14 \times 12000 = \qty{168}{kJ}$; कुल $\qty{639}{kJ}$.
$ v = \qty{4.17}{m/s}$, इसलिए $\Delta t = 12000/4.17 = \qty{2880}{s} =
\qty{48}{min}$ और $ P = \num{6.39e5}/2880 \approx \qty{222}{W} \approx
\qty{0.30}{hp}$ --- एक मजबूत शौकिया, घोड़े का एक तिहाई।
\end{solution}

\begin{solution}{pb:g11:work-of-force:1}
\textbf{1.} $ W = 110 \times 8.0 = +\qty{880}{J}$.

\textbf{2.} \omterm{def:g10:inertia:inventory}{टकराव}: $-95 \times 8.0 = -\qty{760}{J}$. \omterm{def:g10:universal-gravitation:weight}{वज़न} और
सामान्य बल गति के लंबवत हैं: प्रत्येक शून्य।

\textbf{3.} कुल $+\qty{120}{J}$: जाल \omterm{def:g11:work-of-force:sign}{मोटर} कार्य, बॉक्स की गति तेज हो जाती है।
ए \qty{95}{N} पुश संतुलन \omterm{def:g10:inertia:inventory}{टकराव} --- शून्य \omterm{def:g11:forces-and-motion:netforce}{शुद्ध बल}, स्थिर
गति, शून्य कुल \omterm{def:g11:work-of-force:work}{काम}.

\textbf{4.} विकर्ण: $-95 \times 10.0 = -\qty{950}{J}$; दीवारें:
$-95 \times 14.0 = -\qty{1.33}{kJ}$: \omterm{def:g10:inertia:inventory}{टकराव} द्वारा आरोप \omterm{def:g10:orders-of-magnitude:unit}{मीटर}.

\textbf{5.} \omterm{def:g10:inertia:inventory}{टकराव}: $-95 \times 20.0 = -\qty{1.9}{kJ}$; \omterm{def:g10:universal-gravitation:weight}{वज़न}:
$0$ (कोई नेट ड्रॉप नहीं)। केवल \omterm{def:g10:universal-gravitation:weight}{वज़न} रिफंड.

\textbf{6.} $ W(\vect P) = -30 \times 9.81 \times 9.0 =
-\qty{2.65}{kJ}$; चालक दल को कम से कम आपूर्ति करनी चाहिए $+\qty{2.65}{kJ}$.

\textbf{7.} $(70 + 30) \times 9.81 \times 9.0 = \qty{8.83}{kJ}$; द
बॉक्स का हिस्सा है $2.65/8.83 \approx 30\%$ --- अधिकतर ले जाना है
स्व-परिवहन।

\textbf{8.} $\sin\alpha = 9.0/30 = 0.30$:
$ F = mg\sin\alpha + f = 88.3 + 40 = \qty{128}{N}$;
$ W = 128 \times 30 = \qty{3.85}{kJ}$ ($= 2.65 + 1.2$).

\textbf{9.} लंबवत \omterm{def:g10:signals-and-waves:wave}{लहर}ा: $ F = mg = \qty{294}{N}$,
$ W = \qty{2.65}{kJ}$. रैंप छोटा खरीदता है बल ($128$ ख़िलाफ़
\qty{294}{N}) अधिक कीमत पर \omterm{def:g11:work-of-force:work}{काम}: \omterm{def:g10:inertia:inventory}{टकराव}'एस \qty{1.2}{kJ}
टोल।

\textbf{10.} $\Delta t = 9.0/0.50 = \qty{18}{s}$;
$ P = 2649/18 \approx \qty{147}{W}$.

\textbf{11.} $ P = Fv = 294 \times 0.50 = \qty{147}{W}$ --- वही।
\omterm{def:g11:magnetic-fields:electromagnet}{विद्युत}: $147/0.60 \approx \qty{245}{W}$.

\textbf{12.} $1000 \times 0.24 = \qty{240}{m}$ और
$8000 \times 0.030 = \qty{240}{m}$: वही शिखर.

\textbf{13.} $ F = mg \times \text{grade}$: लेन
$3500 \times 9.81 \times 0.24 \approx \qty{8.2}{kN}$; सड़क
$\approx \qty{1.0}{kN}$.

\textbf{14.} $8240 \times 1000 \approx \qty{8.24}{MJ}$ और
$1030 \times 8000 \approx \qty{8.24}{MJ}$ --- दोनों $= mgh = 3500
\times 9.81 \times 240$.

\textbf{15.} रोलिंग \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}: $400 \times 1000 = \qty{0.40}{MJ}$
ख़िलाफ़ $400 \times 8000 = \qty{3.2}{MJ}$: सौम्य सड़क का आराम
लागत \qty{2.8}{MJ} अतिरिक्त का \omterm{def:g10:inertia:inventory}{टकराव}.

\textbf{16.} $ v = \qty{10}{m/s}$. सड़क:
$(1030 + 400) \times 10 = \qty{14.3}{kW} \approx \qty{19}{hp}$; लेन:
$(8240 + 400) \times 10 = \qty{86.4}{kW} \approx \qty{117}{hp}$ ---
एक भरी हुई वैन से परे. सड़कें टेढ़ी-मेढ़ी हैं ताकि मामूली इंजन (और टायर)
चढ़ सकते हैं: वही \omterm{def:g11:work-of-force:work}{काम}, फैलाना \omterm{def:g11:lenses-and-eye:lens}{पतला}.

\textbf{17.} प्रति बॉक्स: $880 + 2649 = \qty{3.53}{kJ}$; के लिए $60$ बक्से,
$60 \times 3529 \approx \qty{212}{kJ}$.

\textbf{18.} सीढ़ियाँ: $20 \times 70 \times 9.81 \times 9.0 \approx
\qty{124}{kJ}$. दिन का कुल योग $\approx \qty{336}{kJ} = \qty{0.34}{MJ}$.

\textbf{19.} $ P = \num{3.36e5}/28800 \approx \qty{12}{W}$. केतली:
$\num{3.36e5}/2000 = \qty{168}{s}$ --- तीन मिनट से कम के लिए
पूरे दिन का श्रम.

\textbf{20.}  \omterm{def:g10:energy-conservation:energy}{जौल} के लिए भुगतान करता है बल पहुंचा दिया \emph{साथ में} एक प्रस्ताव
--- चाहे कितनी भी थकावट क्यों न हो, दीवार को धक्का देने या बक्से को पकड़ने के लिए कुछ भी नहीं
(मांसपेशियां जलती हैं \omterm{def:g10:energy-conservation:forms}{रसायन ऊर्जा} यहाँ तक कि शून्य यांत्रिकी भी कर रहा हूँ \omterm{def:g11:work-of-force:work}{काम}). तब से
मानव \omterm{def:g11:work-of-force:power}{शक्ति} आउटपुट कुछ दर्जन है \omterm{def:g11:work-of-force:power}{वाट}, मूवर्स मशीनों को नया आकार देते हैं
द बल --- चरखी, रैंप, ज़िगज़ैग --- जबकि बिल $ mgh$ कभी नहीं बदलता।
\end{solution}
