\section*{अध्याय \ref{ch:g11:electric-gravitational-fields} --- विद्युत और गुरुत्वाकर्षण क्षेत्र}

\begin{solution}{exo:g11:electric-gravitational-fields:1}
$E = \num{5.0e-3} / \num{2.0e-6} = \qty{2.5e3}{N/C}$, उत्तर की ओर इशारा करते हुए।
पर $ q'$: $ F = \num{4.0e-6} \times \num{2.5e3} = \qty{1.0e-2}{N}$,
इशारा करते हुए \emph{दक्षिण} ($ q' < 0$).
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:2}
\emph{1.} $ E = \num{8.99e9} \times \num{5.0e-9} / 0.30^2
\approx \qty{5.0e2}{N/C}$, की ओर इशारा करते हुए $ O $ ($ Q > 0$).

\emph{2.} संयोग $ k\abs{Q}/d^2$ आवश्यकताओं $ d^2$ दोगुना:
$ d' = 0.30\sqrt{2} \approx \qty{0.42}{m}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:3}
$ E = 600 / \num{3.0e-3} = \qty{2.0e5}{V/m}$;
$ F = eE = \num{1.60e-19} \times \num{2.0e5} = \qty{3.2e-14}{N}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:4}
$ g_{\text{Mars}} = P/m = 930/250 = \qty{3.7}{N/kg}$. अंतरिक्ष यात्री:
$ P = 80 \times 3.72 \approx \qty{3.0e2}{N}$ (के बारे में $\qty{780}{N}$ पर
पृथ्वी).
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:5}
(ए) मिथ्या: द \omterm{def:g11:electric-gravitational-fields:field}{मैदान} प्रत्येक बिंदु पर एक दिशा होती है, इसलिए रेखाएँ कभी नहीं होतीं
पार करना. (बी) गलत: वे चले जाते हैं $+$ और पहुंचें $-$. (सी) सच: भीड़
कोड ताकत.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:6}
\emph{1.} $ d = \num{6.37e6} + \num{4.2e5} = \qty{6.79e6}{m}$, इसलिए
$ g = \num{6.67e-11} \times \num{5.97e24} / (\num{6.79e6})^2
\approx \qty{8.6}{N/kg}$ --- $88\%$ सतही मूल्य का.

\emph{2.} वे तैरते हैं क्योंकि स्टेशन और अंतरिक्ष यात्री एक साथ गिरते हैं (मुक्त)।
गिरना), इसलिए नहीं कि गुरुत्वाकर्षण अनुपस्थित है।
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:7}
$ E = m_e g / e = \num{9.11e-31} \times 9.81 / \num{1.60e-19}
\approx \qty{5.6e-11}{V/m}$: कुछ $\num{4e14}$ से कई गुना कमजोर
\omterm{def:g10:signals-and-waves:oscilloscope}{आस्टसीलस्कप}'एस \omterm{def:g11:electric-gravitational-fields:field}{मैदान}. इलेक्ट्रॉन भौतिकी में गुरुत्वाकर्षण बिल्कुल नगण्य है।
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:8}
\emph{1.} $ E = \num{5.0e3} / \num{2.0e-2} = \qty{2.5e5}{V/m}$; \omterm{def:g10:relative-motion:frame}{आराम से}
$ qE = mg $, इसलिए
$ q = \num{2.0e-6} \times 9.81 / \num{2.5e5} \approx \qty{7.8e-11}{C}$.

\emph{2.}  बल सकारात्मक अनाज पर इंगित करना चाहिए, इसलिए $\vect E $
ऊपर की ओर इंगित करता है: \emph{निचला} प्लेट सकारात्मक है.

\emph{3.} $ n = \num{7.85e-11} / \num{1.60e-19} \approx \num{4.9e8}$
प्राथमिक आरोप.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:9}
दोनों खंड के बाहर खेत उसी तरह इंगित करें; बीच में, वे
विरोध करें --- और संतुलन के लिए कमजोर स्रोत के \omterm{def:g11:lenses-and-eye:eye}{निकट बिंदु} की आवश्यकता होती है $+Q $.
साथ $ x $ की दूरी $+Q $: $ kQ/x^2 = 4kQ/(0.30 - x)^2$, इसलिए
$0.30 - x = 2x $ और $ x = \qty{0.10}{m}$ से $+Q $.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:10}
$ F_e = ke^2/d^2 = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{5.3e-11})^2 \approx \qty{8.2e-8}{N}$;
$ F_g = G m_p m_e / d^2 = \num{6.67e-11} \times \num{1.67e-27} \times
\num{9.11e-31} / (\num{5.3e-11})^2 \approx \qty{3.6e-47}{N}$. अनुपात
$ F_e / F_g \approx \num{2.3e39}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:11}
$ E = \sqrt{300^2 + 400^2} = \qty{500}{N/C}$, पर
$\tan^{-1}(400/300) \approx \ang{53}$ पूर्व के उत्तर. पर $ q < 0$:
$ F = \num{2.0e-6} \times 500 = \qty{1.0e-3}{N}$, के विपरीत \omterm{def:g11:electric-gravitational-fields:field}{मैदान} ---
$\ang{53}$ पश्चिम के दक्षिण.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:12}
$ GM/(R+h)^2 = \tfrac12\,GM/R^2$ देता है $ R + h = R\sqrt{2}$, इसलिए
$ h = (\sqrt{2} - 1)R \approx \qty{2.6e6}{m}$ (के बारे में \qty{2600}{km}). पर
$ h = R $, $ d = 2R $: सतही मूल्य का एक चौथाई.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:13}
\emph{1.} $ AM = \sqrt{3.0^2 + 4.0^2} = \qty{5.0}{cm}$; प्रत्येक आरोप
बनाता है $ E_1 = \num{8.99e9} \times \num{2.0e-9} / 0.050^2
\approx \qty{7.2e3}{N/C}$.

\emph{2.} साथ में घटक $ AB $ समरूपता द्वारा रद्द करें; प्रत्येक \omterm{def:g11:electric-gravitational-fields:field}{मैदान}
योगदान $ E_1 \times 4/5$ समद्विभाजक के अनुदिश, खंड से दूर:
$ E = 2 \times \num{7.19e3} \times 0.80 \approx \qty{1.2e4}{N/C}$,
से दूर, द्विभाजक के अनुदिश निर्देशित $[AB]$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:14}
बॉक्स के मुक्त चार्ज बाहरी के नीचे चलते हैं \omterm{def:g11:electric-gravitational-fields:field}{मैदान}: $+$ ढेर लग जाता है
एक चेहरा, $-$ दूसरी ओर, और ये विस्थापित आरोप एक बनाते हैं
आंतरिक \omterm{def:g11:electric-gravitational-fields:field}{मैदान} बाहरी का विरोध. वे तब तक चलते रहते हैं जब तक
कुल \omterm{def:g11:electric-gravitational-fields:field}{मैदान} अंदर बिल्कुल शून्य है --- फिर संतुलन। गुरुत्वाकर्षण प्रदान करता है
स्रोत का केवल एक ही \omterm{def:g10:signals-and-waves:signal}{संकेत}: प्रत्येक द्रव्यमान \emph{कहते हैं} एक आकर्षक \omterm{def:g11:electric-gravitational-fields:field}{मैदान},
कोई भी इसका विरोध नहीं कर सकता. एक गुरुत्वाकर्षण ढाल के लिए नकारात्मक द्रव्यमान की आवश्यकता होगी, जो
अस्तित्व में नहीं है.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{वज़न} $ m\vect g $ (नीचे), \omterm{def:g10:inertia:inventory}{तनाव} $\vect T $ (साथ में
धागा), बिजली बल $ q\vect E $ (क्षैतिज)। \omterm{def:g10:relative-motion:frame}{आराम से},
$ T\sin\theta = qE $ और $ T\cos\theta = mg $; विभाजित करना:
$\tan\theta = qE/(mg)$.

\emph{2.} $\tan\theta = \num{2.0e-7} \times \num{1.0e4} /
(\num{5.0e-4} \times 9.81) \approx 0.41$, इसलिए $\theta \approx \ang{22}$;
$ T = mg/\cos\theta \approx \qty{5.3e-3}{N}$.
\end{solution}

\begin{solution}{pb:g11:electric-gravitational-fields:1}
\textbf{1.} \omterm{def:g11:electric-gravitational-fields:capacitor}{वर्दी}, प्लेटों के लंबवत, नीचे की ओर इशारा करते हुए (से
द $+$ ऊपरी प्लेट को $-$ नीचे की); एकरूपता का मतलब है बल
बूंद जहां भी बहती है, वहां एक समान होती है।
\textbf{2.} $ E = U/d = 800 / \num{8.0e-3} = \qty{1.0e5}{V/m}$.
\textbf{3.} चार्ज नकारात्मक है, इसलिए बल है \emph{विलोम}
$\vect E $: ऊपर की ओर. केवल के साथ $+$ शीर्ष पर प्लेट करता है $\vect E $ बिंदु
नीचे और बिजली बल लड़ो \omterm{def:g10:universal-gravitation:weight}{वज़न}.
\textbf{4.} $ P = mg = \num{4.90e-15} \times 9.81 \approx
\qty{4.8e-14}{N}$.
\textbf{5.} $\abs{q}E = mg $.
\textbf{6.} $\abs{q} = mg/E = mgd/U $.
\textbf{7.} $\abs{q} = \num{4.81e-14} \times \num{8.0e-3} / 800
\approx \qty{4.8e-19}{C}$.
\textbf{8.} $ n = \num{4.8e-19} / \num{1.60e-19} = 3$ अतिरिक्त इलेक्ट्रॉन.
\textbf{9.} $ E $ बढ़ता है, इसलिए $\abs{q}E > mg $: नेट बल ऊपर की ओर इशारा करता है।
\textbf{10.} से $\abs{q}\,U/d = mg $, $ U = mgd/\abs{q}$ ---विपरीत
चार्ज के आनुपातिक.
\textbf{11.} $\abs{q} = mgd/U = \num{3.85e-16}/1200 \approx
\qty{3.2e-19}{C} = 2e $: से नीचे $3e $, बूंद \emph{खो गया} एक
इलेक्ट्रॉन.
\textbf{12.} $\abs{q} = \num{3.85e-16}/U $: \qty{3.2e-19}{C},
\qty{4.8e-19}{C}, \qty{6.4e-19}{C}, \qty{8.0e-19}{C},
\qty{9.6e-19}{C}.
\textbf{13.} वे हैं $2e, 3e, 4e, 5e, 6e $ साथ
$ e = \qty{1.60e-19}{C}$.
\textbf{14.} हर कदम बदलता है $\abs{q}$ बिलकुल सही से
\qty{1.6e-19}{C}: एक समय में एक इलेक्ट्रॉन उतरता है।
\textbf{15.} इसकी आवश्यकता होगी $\abs{q} = \num{3.85e-16}/700 \approx
\qty{5.5e-19}{C} = 3.4\,e $ --- इलेक्ट्रॉनों की पूर्ण संख्या नहीं, इसलिए नहीं
विकिरण का विस्फोट कभी भी इसे उत्पन्न कर सकता है।
\textbf{16.} आरोप है \emph{मात्रा निर्धारित}: यह पूर्णांक गुणज में आता है
का \omterm{def:g11:fundamental-interactions:elementary}{प्राथमिक प्रभार} $ e = \qty{1.60e-19}{C}$ ---मात्रा
मिलिकन ने नापा।
\textbf{17.} $ q \leftrightarrow m $, $ E \leftrightarrow g $, और
$\abs{q}E \leftrightarrow mg $: द \omterm{def:g10:universal-gravitation:weight}{वज़न} स्वयं.
\textbf{18.} $ g_{\text{sphere}} = GM/d^2 = \num{6.67e-11} \times 1000 /
1.0^2 = \qty{6.7e-8}{N/kg}$; बल बूंद पर
$\num{4.90e-15} \times \num{6.7e-8} \approx \qty{3.3e-22}{N}$ --- के बारे में
$\num{1.5e8}$ उससे कई गुना छोटा \omterm{def:g10:universal-gravitation:weight}{वज़न}. निराशाजनक।
\textbf{19.} क्योंकि $ G $ यह इतना छोटा है कि ऊपर केवल एक ग्रह के आकार का द्रव्यमान है
पृथ्वी के खिंचाव का मुकाबला कर सकता है --- और कोई द्रव्यमान प्रतिकर्षित नहीं करता। विशाल $ k $
और आवेश के दो लक्षण एक बेंच-टॉप प्लेट को ग्रह से बाहर खींचने देते हैं
किसी भी दिशा में.
\textbf{20.} एक इलेक्ट्रॉन: $\Delta\abs{q}/\abs{q} = e/3e \approx 33\%$;
एक तेल अणु: $\Delta m/m = \num{5e-25} / \num{4.90e-15}
\approx \num{e-10}$. चार्ज सीढ़ी संतुलन को आगे बढ़ाती है \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}
सैकड़ों द्वारा \omterm{def:g11:circuits-and-power:voltage}{वोल्ट}; सामूहिक सीढ़ियाँ बिल्कुल भी अदृश्य हैं \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}.
इसलिए संतुलन का समाधान होता है \omterm{def:g11:fundamental-interactions:ladder}{एटम} बिजली का ---
$ e = \qty{1.60e-19}{C}$ --- जब \omterm{def:g11:fundamental-interactions:ladder}{अणुओं} तेल का एक में धुंधला हो जाना
सातत्य.
\end{solution}
