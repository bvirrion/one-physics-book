\section*{अध्याय \ref{ch:g11:circuits-and-power} --- इलेक्ट्रिक सर्किट और पावर}

\begin{solution}{exo:g11:circuits-and-power:1}
$q = It = 2.0 \times 90 \times 60 = \qty{1.1e4}{C}$. प्राथमिक की संख्या
शुल्क: $ q/e = 1.08\times10^{4}/1.6\times10^{-19} \approx \num{6.8e22}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:2}
$ I = U/R = 12/220 = \qty{0.055}{A} = \qty{55}{mA}$;
$ R = U/I = 230/0.50 = \qty{460}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:3}
शृंखला: $120 + 60 = \qty{180}{\ohm}$. समानांतर: $\frac{120 \times 60}{180}
= \qty{40}{\ohm}$ --- दोनों से छोटा : समान \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज}, लेकिन वर्तमान
\omterm{def:g10:orders-of-magnitude:unit}{दूसरा} पथ मिलता है, इसलिए कुल धारा अधिक और कम होती है \omterm{def:g11:circuits-and-power:resistance}{प्रतिरोध}.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:4}
$ P = UI = 230 \times 8.7 = \qty{2.0}{kW}$. में $ t = \qty{240}{s}$:
$ E = Pt = 2000 \times 240 = \qty{4.8e5}{J} = 4.8\times10^{5}/3.6
\times10^{6} \approx \qty{0.13}{kW.h}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:5}
$ E = 60 \times 5.0 = \qty{300}{W.h} = \qty{0.30}{kW.h} = 0.30 \times
3.6\times10^{6} = \qty{1.1e6}{J}$. लागत: $0.30 \times 0.25 = 0.075$
एक दिन में यूरो.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:6}
\emph{1.} $ R = U^2/P = 230^2/1200 = \qty{44}{\ohm}$;
$ I = P/U = 1200/230 = \qty{5.2}{A}$.

\emph{2.} $ P' = U'^2/R = 115^2/44.1 = \qty{300}{W}$: एक चौथाई, नहीं
आधा, क्योंकि आधा करना $ U $ आधा भी $ I $, और $ P = UI \propto U^2$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:7}
$ R_{\text{s}} = \qty{60}{\ohm}$, इसलिए $ I = 12/60 = \qty{0.20}{A}$.
$ U_1 = R_1 I = \qty{8.0}{V}$, $ U_2 = \qty{4.0}{V}$ (जोड़ \qty{12}{V});
$ P_1 = U_1 I = \qty{1.6}{W}$, $ P_2 = \qty{0.8}{W}$: कुल \qty{2.4}{W}
$= UI = 12 \times 0.20$, जैसा कि होना चाहिए।
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:8}
$ I_1 = 12/60 = \qty{0.20}{A}$, $ I_2 = 12/30 = \qty{0.40}{A}$, कुल
$ I = \qty{0.60}{A}$. समकक्ष: $ R = U/I = 12/0.60 = \qty{20}{\ohm}$,
सचमुच $\frac{60 \times 30}{60 + 30} = \qty{20}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:9}
$\mathcal{E} - 0.50\,r = 5.5$ और $\mathcal{E} - 2.0\,r = 4.0$;
घटाना, $1.5\,r = 1.5$, इसलिए $ r = \qty{1.0}{\ohm}$ और $\mathcal{E} =
\qty{6.0}{V}$. शार्ट सर्किट: $\mathcal{E}/r = \qty{6.0}{A}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:10}
केतली $2200/230 = \qty{9.6}{A}$, टोस्टर $1000/230 = \qty{4.3}{A}$:
कुल \qty{13.9}{A} $< \qty{16}{A}$, द \omterm{def:g11:circuits-and-power:joule}{फ्यूज} धारण करता है. हीटर $1500/230 =
\qty{6.5}{A}$: कुल \qty{20.4}{A} $> \qty{16}{A}$, यह उड़ता है
(समान रूप से, $\qty{4.7}{kW} > 230 \times 16 \approx \qty{3.7}{kW}$).
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:11}
पर \qty{200}{V}: $ I = 10^4/200 = \qty{50}{A}$, नुकसान $ RI^2 = 2.0 \times
50^2 = \qty{5.0}{kW}$ --- आधी डिलीवरी. पर \qty{4.0}{kV}: $ I =
\qty{2.5}{A}$, नुकसान $2.0 \times 2.5^2 = \qty{12.5}{W}$. अनुपात $400 =
20^2$: बीस गुना \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} मतलब बीस गुना कम करंट, और
हानि वैसे ही होती है $ I^2$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:12}
चार लेआउट: सभी श्रृंखला में, $3 \times 60 = \qty{180}{\ohm}$; सभी में
समानांतर, $60/3 = \qty{20}{\ohm}$; एक श्रृंखला में और दो समानांतर में,
$60 + 30 = \qty{90}{\ohm}$; एक समानांतर में और दो श्रृंखला में,
$\frac{60 \times 120}{180} = \qty{40}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:13}
\emph{1.} $ I = \mathcal{E}/(R + r) = 4.5/7.5 = \qty{0.60}{A}$;
$ U = RI = \qty{3.6}{V}$; उपयोगी $ P = UI = \qty{2.2}{W}$; खो गया $ rI^2 =
1.5 \times 0.60^2 = \qty{0.54}{W}$; $\eta = U/\mathcal{E} = 3.6/4.5 =
0.80$.

\emph{2.} \omterm{def:g10:signals-and-waves:signal}{संकेत} से, $ P = \mathcal{E}^2 \big/ \big[(R - r)^2/R +
4r\big]$; ब्रैकेट सबसे कम है ($= 4r $) बिल्कुल कब $ R = r $, इसलिए
$ P_{\max} = \mathcal{E}^2/(4r) = 4.5^2/6.0 = \qty{3.4}{W}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:14}
गर्मी: $ Q = 1.0 \times 4180 \times (100 - 20) = \qty{3.3e5}{J}$.
विद्युत: $ E = Q/0.90 = \qty{3.7e5}{J}$. समय: $ t = E/P =
3.7\times10^{5}/2200 \approx \qty{170}{s}$, तीन मिनट से कम। लागत:
$3.7\times10^{5}/3.6\times10^{6} \approx \qty{0.10}{kW.h}$, के बारे में $2.6$
सेंट.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:15}
\emph{1.} $ q = 60 \times 3600 = \qty{2.2e5}{C}$; $ E \approx \mathcal{E}q
= 12.7 \times 2.16\times10^{5} = \qty{2.7e6}{J} \approx \qty{0.76}{kW.h}$.

\emph{2.} $ U = 12.7 - 0.020 \times 150 = \qty{9.7}{V}$; पहुंचा दिया
$ UI = \qty{1.5}{kW}$; खो गया $ rI^2 = 0.020 \times 150^2 = \qty{450}{W}$.

\emph{3.} हेडलाइट्स टर्मिनलों के पार बैठती हैं: क्रैंक करते समय,
टर्मिनल \omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} से गिरता है $12.7$ को $\qty{9.7}{V}$, तो वे प्राप्त करते हैं
कम \omterm{def:g10:energy-conservation:power}{शक्ति} और मंद.
\end{solution}

\begin{solution}{pb:g11:circuits-and-power:1}
\textbf{1.} प्रत्येक सेकंड, $ E_p = mgh = 5.0\times10^{4} \times 9.81 \times
200 = \qty{9.8e7}{J}$: उपलब्ध \omterm{def:g10:energy-conservation:power}{शक्ति} \qty{98}{MW}.

\textbf{2.} $0.90 \times 98 = \qty{88}{MW}$.

\textbf{3.} $ I = P/U = 8.8\times10^{7}/2.0\times10^{4} = \qty{4.4e3}{A}$.

\textbf{4.} $ R_\ell I^2 = 32 \times (4.4\times10^{3})^2 \approx
\qty{6.2e8}{W} = \qty{620}{MW}$ --- संयंत्र का सात गुना उत्पादन:
पर संचारित करना \qty{20}{kV} असंभव है.

\textbf{5.} साथ $ P = UI $ निश्चित, केवल उठाना $ U $ सिकुड़ता $ I $: ए
ट्रांसफार्मर.

\textbf{6.} $ I = 8.8\times10^{7}/4.0\times10^{5} = \qty{220}{A}$.

\textbf{7.} $ R_\ell I^2 = 32 \times 220^2 \approx \qty{1.6}{MW}$.

\textbf{8.} $1.6/88 \approx 0.018$: के बारे में $1.8\%$.

\textbf{9.} अंश $= R_\ell I^2/P = R_\ell (P/U)^2/P = R_\ell P/U^2
\propto 1/U^2$: से \qty{20}{kV} को \qty{400}{kV} यह गिर जाता है
$20^2 = 400$, सचमुच $\qty{620}{MW}/400 \approx \qty{1.6}{MW}$.

\textbf{10.} $ R_\ell I = 32 \times 220 \approx \qty{7.1}{kV}$: $1.8\%$
का \qty{400}{kV}, चूँकि प्रश्न 8 के समान भिन्न है
$ R_\ell I/U = R_\ell I^2/UI $.

\textbf{11.} $\qty{1}{kW.h} = \qty{3.6e6}{J}$: \omterm{def:g10:orders-of-magnitude:unit}{मीटर} गुणा करता है
\omterm{def:g10:energy-conservation:power}{शक्ति} समय के अनुसार इसे खींचा जाता है, और जोड़ा जाता है।

\textbf{12.} $ I = 900/230 = \qty{3.9}{A}$;
$ R = U^2/P = 230^2/900 \approx \qty{59}{\ohm}$.

\textbf{13.} $ E = 900 \times 180 = \qty{1.6e5}{J} = \qty{0.045}{kW.h}$:
के बारे में $1.1$ सेंट.

\textbf{14.} का भौतिकी $ RI^2$ समान है; केवल उद्देश्य
अंतर --- लाइन पर गर्मी अवांछित है, टोस्टर में गर्मी
\emph{है} उत्पाद.

\textbf{15.} $3.9 + 9.6 = \qty{13.5}{A} < \qty{16}{A}$: रखता है. हीटर
$2000/230 = \qty{8.7}{A}$: कुल $\qty{22.2}{A} > \qty{16}{A}$, द \omterm{def:g11:circuits-and-power:joule}{फ्यूज}
मारता है.

\textbf{16.} $0.05 \times 13.5^2 = \qty{9.1}{W}$: $9.1/3100 \approx
0.3\%$ वितरित का \omterm{def:g10:energy-conservation:power}{शक्ति} --- छोटे तारों पर नगण्य।

\textbf{17.} $\eta = 0.90 \times 0.99 \times 0.982 \times 0.99 \times
0.997 \approx 0.86$.

\textbf{18.} एक टोस्ट किया हुआ \omterm{def:g10:energy-conservation:kwh}{किलोवाट्ट घन्ता} आवश्यकताओं $3.6\times10^{6}/0.86
\approx \qty{4.2e6}{J}$ अपस्ट्रीम: $ m = E/(gh) = 4.2\times10^{6}/(9.81
\times 200) \approx \qty{2.1e3}{kg}$, के बारे में \qty{2.1}{m^3} पानी डा।

\textbf{19.} $ R_\ell P/U^2 = 32 \times 8.8\times10^{7}/(2.0\times
10^{4})^2 \approx 7$. ऊपर एक ``अंश'' $1$ मतलब लाइन होगी
जितना वह वहन करता है उससे अधिक नष्ट कर देता है: उस समय वितरण नहीं हो सकता
\omterm{def:g11:circuits-and-power:voltage}{वोल्टेज} बिल्कुल भी।

\textbf{20.} एक टोस्ट किया हुआ \omterm{def:g10:energy-conservation:kwh}{किलोवाट्ट घन्ता} पहाड़ की कीमत लगभग दो है
टन --- \qty{2.1}{m^3} --- पानी गिरने का \qty{200}{m}, और कुछ
$14\%$ की ऊर्जा रास्ते में ही मर जाता है, ज्यादातर टर्बाइनों और ट्रांसफार्मरों में
(अंडर $2\%$ में \qty{1000}{km} रेखा की)। एकल युक्ति जो बनाती है
यात्रा संभव है ट्रांसफार्मर: की ओर कदम बढ़ाना \qty{400}{kV}
द्वारा धारा को विभाजित करता है $20$ और जूल हानि $400$.
\end{solution}
