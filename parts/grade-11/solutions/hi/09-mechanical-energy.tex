\begin{solution}{exo:g11:mechanical-energy:1}
(ए) $v = \qty{6.94}{m/s}$:
$ E_k = \tfrac12 \times 90 \times 6.94^2 \approx \qty{2.2e3}{J}$.
(बी) $ v = \qty{36.1}{m/s}$:

\end{solution}

\begin{solution}{exo:g11:mechanical-energy:8}

\begin{solution}{exo:g11:mechanical-energy:13}
वही प्रारंभिक $E_m = \tfrac12 m \times 15^2 + m \times 9.81 \times 45$

\begin{solution}{exo:g11:mechanical-energy:14}
$\tfrac12 mv^2 = mgh - f\ell = 80 \times 9.81 \times 120 - 65 \times

$ v = \sqrt{2 \times \num{42176}/80} \approx \qty{32}{m/s}$
(\qty{117}{km/h}). अंश \omterm{def:g11:mechanical-energy:dissipation}{व्यस्त}: $52000/94176 \approx 55\%$.
\end{solution}

\begin{solution}{pb:g11:mechanical-energy:1}
\textbf{1.} $ v_{\text{top}} = \sqrt{gR} = \sqrt{9.81 \times 8.0}
\approx \qty{8.9}{m/s}$.

\textbf{2.} $ E_m = mg\,(2R) + \tfrac12 m gR = \num{3.14e5} +
\num{7.8e4} \approx \qty{3.92e5}{J}$.

\textbf{3.} $ mgH_{\min} = E_m $ देता है $ H_{\min} = \qty{20}{m}$;
बीजगणित $ H_{\min} = 2R + \frac{gR}{2g} = \frac{5R}{2}$.

\textbf{4.} $ v_{\text{top}} = \sqrt{2g(H - 2R)} =
\sqrt{2 \times 9.81 \times 8.0} \approx \qty{12.5}{m/s}$;
$ v_{\text{top}}^2/(gR) = 2.0$ --- न्यूनतम दो बार.

\textbf{5.} प्रत्येक ऊर्जा के लिए आनुपातिक है $ m $, इसलिए $ m $ रद्द: वही
गति और समान सुरक्षा, पूर्ण या खाली।

\textbf{6.} $ v = \sqrt{2 \times 9.81 \times 24} \approx \qty{21.7}{m/s}
\approx \qty{78}{km/h}$.

\textbf{7.} $ v = \sqrt{2 \times 9.81 \times (24 - 8.0)} \approx
\qty{17.7}{m/s}$.

\textbf{8.} $ mgH = \qty{4.71e5}{J}$. $(E_p, E_k)$: शुरू करना
$(\num{4.71e5}, 0)$; तल $(0, \num{4.71e5})$; ओर
$(\num{1.57e5}, \num{3.14e5})$; शीर्ष $(\num{3.14e5}, \num{1.57e5})$ ---
प्रत्येक जोड़ी का योग है \qty{4.71e5}{J}.

\textbf{9.} $ v = \sqrt{2 \times 9.81 \times (24 - 12)} \approx
\qty{15.3}{m/s}$.

\textbf{10.} $ v(z) = \sqrt{2g(H - z)}$: निम्नतम बिंदु पर सबसे तेज़
ट्रैक, लूप के शीर्ष पर सबसे धीमा।

\textbf{11.} समान ऊंचाई, इसलिए $\Delta E_p = 0$ और
$\Delta E_m = \Delta E_k = \tfrac12 \times 2000 \times (19.2^2 -
20.4^2) \approx \qty{-4.75e4}{J}$: प्लेसमेंट घाटे को शुद्ध बनाता है
गतिज, दो स्पीडो\omterm{def:g10:orders-of-magnitude:unit}{मीटर} पढ़ें।

\textbf{12.} $ W(\vect f) = \qty{-4.75e4}{J}$;
$ f = \num{4.75e4}/60 \approx \qty{7.9e2}{N}$.

\textbf{13.} \omterm{def:g10:energy-conservation:forms}{थर्मल ऊर्जा}, पहियों, रेलों और हवा में।

\textbf{14.} $ E_m $ पहले सेंसर पर: $\num{4.16e5} + \num{3.9e4} =
\qty{4.55e5}{J}$; नुकसान $\num{4.75e4}/\num{4.55e5} \approx 10\%$ प्रति
\qty{60}{m}. पर $ H_{\min}$ लूप के शीर्ष पर पहुंच गया है \emph{बिल्कुल}
पर $\sqrt{gR}$; कोई भी हानि इसे सुरक्षा गति से नीचे गिरा देती है।

\textbf{15.} हार $\approx 10\%$ प्रति \qty{60}{m}, के बारे में $15\%$ का
$ mgH $ ($\approx \qty{7.1e4}{J}$) चला गया है: $ E_m \approx \qty{4.00e5}{J}$,
अभी भी आवश्यक से ऊपर है \qty{3.92e5}{J} --- सुरक्षित, केवल के साथ $2\%$ बचा। मार्जिन कोई विलासिता नहीं थी.

\textbf{16.} $ E_k = \tfrac12 \times 2000 \times 18.0^2 =
\qty{3.24e5}{J}$.

\textbf{17.} $ F = \num{3.24e5}/45 = \qty{7.2e3}{N}$.

\textbf{18.} $ a = 7200/2000 = \qty{3.6}{m/s^2} \approx 0.37\,g $:
दृढ़ लेकिन आरामदायक.

\textbf{19.} रास्ता \omterm{def:g10:inertia:inventory}{टकराव}: $ mgH - \num{3.24e5} = \num{470880} -
\num{324000} = \qty{1.47e5}{J}$. अंकेक्षण: $\num{1.47e5} + \num{3.24e5} =
\num{4.71e5} = mgH $ --- प्रत्येक \omterm{def:g10:energy-conservation:energy}{जौल} के लिए हिसाब।

\textbf{20.} लॉन्च ऊंचाई $ H = \qty{24}{m}$ लूप को ठीक किया
सुरक्षा, हर गति $\sqrt{2g(H - z)}$, द \omterm{def:g10:inertia:inventory}{टकराव} बिल और ब्रेक
बल; अनदेखी की कीमत \omterm{def:g10:inertia:inventory}{टकराव} मार्जिन है --- $\qty{4}{m}$
पहाड़ी की जिसे रगड़ने वाली दुनिया चुपचाप खा जाती है।
\end{solution}
