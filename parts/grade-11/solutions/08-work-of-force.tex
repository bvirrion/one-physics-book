\section*{Chapter \ref{ch:g11:work-of-force} --- Work of a Force}

\begin{solution}{exo:g11:work-of-force:1}
$W = 40 \times 50 \times \cos\theta$: $+\qty{2000}{J}$;
$+\qty{1000}{J}$; $0$; $-\qty{1000}{J}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:2}
$mgh = 12 \times 9.81 \times 2.5 = \qty{294}{J}$: lowered,
$+\qty{294}{J}$ (weight along the drop); raised, $-\qty{294}{J}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:3}
(a) motor; (b) resistive; (c) zero (perpendicular); (d) resistive;
(e) zero --- the string's tension stays perpendicular to the bob's
motion.
\end{solution}

\begin{solution}{exo:g11:work-of-force:4}
Constant speed: lifting force $= mg = \qty{5.89e3}{N}$, so
$W_{\text{lift}} = 5886 \times 25 \approx +\qty{147}{kJ}$ and
$W(\vect P) = -\qty{147}{kJ}$. Sum zero: at constant velocity the
forces balance (\cref{ch:g11:forces-and-motion}), so their works
cancel.
\end{solution}

\begin{solution}{exo:g11:work-of-force:5}
$P = \num{9.0e4}/60 = \qty{1.5}{kW}$; $1500/736 \approx \qty{2.0}{hp}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:6}
Rope: $120 \times 200 \times \cos\qty{30}{\degree} \approx
+\qty{20.8}{kJ}$; friction: $-45 \times 200 = -\qty{9.0}{kJ}$; weight
and normal force, perpendicular: $0$. Total $+\qty{11.8}{kJ}$: the
sled gains speed.
\end{solution}

\begin{solution}{exo:g11:work-of-force:7}
$v = 25/3.6 = \qty{6.94}{m/s}$; $P = Fv = 18 \times 6.94 \approx
\qty{125}{W}$. In one hour, $W = P\,\Delta t = 125 \times 3600 =
\qty{450}{kJ}$ (equivalently $18 \times \qty{25000}{m}$).
\end{solution}

\begin{solution}{exo:g11:work-of-force:8}
Weight, both routes: $+65 \times 9.81 \times 120 \approx
+\qty{76.5}{kJ}$ (drop only). Friction: $-30 \times 800 =
-\qty{24}{kJ}$ on the slope, $-30 \times 1500 = -\qty{45}{kJ}$ on the
track. Only friction cares about the route.
\end{solution}

\begin{solution}{exo:g11:work-of-force:9}
Rise $h = 4.0\sin\qty{25}{\degree} = \qty{1.69}{m}$:
$W(\vect P) = -20 \times 9.81 \times 1.69 \approx -\qty{332}{J}$;
$W_f = -30 \times 4.0 = -\qty{120}{J}$. Constant speed: total work
zero, so $W_{\text{push}} = +\qty{452}{J}$ and
$F = 452/4.0 \approx \qty{113}{N}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:10}
$v = 130/3.6 = \qty{36.1}{m/s}$; constant speed, so
$F = P/v = \num{5.0e4}/36.1 \approx \qty{1.4}{kN}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:11}
$P = Fv\cos\theta = 800 \times 1.0 \times \cos\qty{20}{\degree}
\approx \qty{752}{W}$; in horsepower, $752/736 \approx \num{1.02}$ ---
almost exactly what a strong horse sustains, the comparison Watt
built his unit on.
\end{solution}

\begin{solution}{exo:g11:work-of-force:12}
Lengths $L = h/\text{grade}$: \qty{1500}{m} and \qty{5000}{m}. Forces
$F = mg \times \text{grade}$: $1200 \times 9.81 \times 0.20 \approx
\qty{2.35}{kN}$ and $\qty{0.71}{kN}$. Works $FL$: $2354 \times 1500
\approx \qty{3.53}{MJ}$ and $706 \times 5000 \approx \qty{3.53}{MJ}$
--- both equal $mgh$: the road trades force for length, never work.
\end{solution}

\begin{solution}{exo:g11:work-of-force:13}
$F = mg = \qty{2.45e3}{N}$; $W = 2453 \times 15 \approx \qty{36.8}{kJ}$;
$\Delta t = 15/0.40 = \qty{37.5}{s}$;
$P = 36800/37.5 \approx \qty{981}{W} = 2453 \times 0.40$ ($Fv$ agrees);
electrical: $981/0.70 \approx \qty{1.4}{kW}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:14}
Diagonal: $-140 \times 6.0 = -\qty{840}{J}$; walls:
$-140 \times 8.4 \approx -\qty{1.18}{kJ}$; round trip:
$-140 \times 12 = -\qty{1.68}{kJ}$. The weight does zero on the round
trip (no net drop). Friction's work depends on the path and never
refunds; the weight's depends only on the endpoints
(\cref{rem:g11:work-of-force:divide}).
\end{solution}

\begin{solution}{exo:g11:work-of-force:15}
Gravity: $80 \times 9.81 \times 600 \approx \qty{471}{kJ}$; friction:
$14 \times 12000 = \qty{168}{kJ}$; total $\qty{639}{kJ}$.
$v = \qty{4.17}{m/s}$, so $\Delta t = 12000/4.17 = \qty{2880}{s} =
\qty{48}{min}$ and $P = \num{6.39e5}/2880 \approx \qty{222}{W} \approx
\qty{0.30}{hp}$ --- a strong amateur, a third of a horse.
\end{solution}

\begin{solution}{pb:g11:work-of-force:1}
\textbf{1.} $W = 110 \times 8.0 = +\qty{880}{J}$.

\textbf{2.} Friction: $-95 \times 8.0 = -\qty{760}{J}$. Weight and
normal force are perpendicular to the motion: zero each.

\textbf{3.} Total $+\qty{120}{J}$: net motor work, the box speeds up.
A \qty{95}{N} push balances friction --- zero net force, constant
speed, zero total work.

\textbf{4.} Diagonal: $-95 \times 10.0 = -\qty{950}{J}$; walls:
$-95 \times 14.0 = -\qty{1.33}{kJ}$: friction charges by the metre.

\textbf{5.} Friction: $-95 \times 20.0 = -\qty{1.9}{kJ}$; weight:
$0$ (no net drop). Only the weight refunds.

\textbf{6.} $W(\vect P) = -30 \times 9.81 \times 9.0 =
-\qty{2.65}{kJ}$; the crew must supply at least $+\qty{2.65}{kJ}$.

\textbf{7.} $(70 + 30) \times 9.81 \times 9.0 = \qty{8.83}{kJ}$; the
box's share is $2.65/8.83 \approx 30\%$ --- carrying is mostly
self-transport.

\textbf{8.} $\sin\alpha = 9.0/30 = 0.30$:
$F = mg\sin\alpha + f = 88.3 + 40 = \qty{128}{N}$;
$W = 128 \times 30 = \qty{3.85}{kJ}$ ($= 2.65 + 1.2$).

\textbf{9.} Vertical hoist: $F = mg = \qty{294}{N}$,
$W = \qty{2.65}{kJ}$. The ramp buys a smaller force ($128$ against
\qty{294}{N}) at the price of more work: friction's \qty{1.2}{kJ}
toll.

\textbf{10.} $\Delta t = 9.0/0.50 = \qty{18}{s}$;
$P = 2649/18 \approx \qty{147}{W}$.

\textbf{11.} $P = Fv = 294 \times 0.50 = \qty{147}{W}$ --- same.
Electrical: $147/0.60 \approx \qty{245}{W}$.

\textbf{12.} $1000 \times 0.24 = \qty{240}{m}$ and
$8000 \times 0.030 = \qty{240}{m}$: same summit.

\textbf{13.} $F = mg \times \text{grade}$: lane
$3500 \times 9.81 \times 0.24 \approx \qty{8.2}{kN}$; road
$\approx \qty{1.0}{kN}$.

\textbf{14.} $8240 \times 1000 \approx \qty{8.24}{MJ}$ and
$1030 \times 8000 \approx \qty{8.24}{MJ}$ --- both $= mgh = 3500
\times 9.81 \times 240$.

\textbf{15.} Rolling resistance: $400 \times 1000 = \qty{0.40}{MJ}$
against $400 \times 8000 = \qty{3.2}{MJ}$: the gentle road's comfort
costs \qty{2.8}{MJ} of extra friction.

\textbf{16.} $v = \qty{10}{m/s}$. Road:
$(1030 + 400) \times 10 = \qty{14.3}{kW} \approx \qty{19}{hp}$; lane:
$(8240 + 400) \times 10 = \qty{86.4}{kW} \approx \qty{117}{hp}$ ---
beyond a loaded van. Roads zigzag so that modest engines (and tyres)
can climb: same work, spread thin.

\textbf{17.} Per box: $880 + 2649 = \qty{3.53}{kJ}$; for $60$ boxes,
$60 \times 3529 \approx \qty{212}{kJ}$.

\textbf{18.} Stairs: $20 \times 70 \times 9.81 \times 9.0 \approx
\qty{124}{kJ}$. Day's total $\approx \qty{336}{kJ} = \qty{0.34}{MJ}$.

\textbf{19.} $P = \num{3.36e5}/28800 \approx \qty{12}{W}$. The kettle:
$\num{3.36e5}/2000 = \qty{168}{s}$ --- under three minutes for the
whole day's labour.

\textbf{20.} The joule pays for force delivered \emph{along} a motion
--- nothing for the wall pushed or the box held, however exhausting
(muscles burn chemical energy even doing zero mechanical work). Since
human power output is a few dozen watts, movers let machines reshape
the force --- winch, ramp, zigzag --- while the bill $mgh$ never
changes.
\end{solution}
