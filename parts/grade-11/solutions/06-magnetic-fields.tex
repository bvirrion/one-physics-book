\section*{Chapter \ref{ch:g11:magnetic-fields} --- Magnetism and Magnetic Fields}

\begin{solution}{exo:g11:magnetic-fields:1}
Each piece is a complete \omterm{def:g11:magnetic-fields:magnet}{magnet} with a north and a south pole: a new
pole appears on each cut face. The fresh faces are an N and an S, so
they attract --- the pieces try to reassemble. No scheme isolates a
pole: every cut creates a pair.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:2}
(a) False: the \omterm{def:g11:electric-gravitational-fields:field}{field} has one direction at each point. (b) True.
(c) True: each line closes through the \omterm{def:g11:magnetic-fields:magnet}{magnet}'s body. (d) False: the
needle settles \emph{along} the line, tangent to it.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:4}

\begin{solution}{exo:g11:magnetic-fields:5}

\end{solution}

\end{solution}

\begin{solution}{exo:g11:magnetic-fields:7}
$d = \mu_0 I / (2\pi B) = \num{2e-7} \times 20 / \num{5.0e-5}
= \qty{8.0}{cm}$. Within a decimetre of such a cable the compass reads

\end{solution}

\begin{solution}{exo:g11:magnetic-fields:8}

\approx \qty{2.0e3}{m^{-1}}$; over \qty{25}{cm},
$N = 1989 \times 0.25 \approx 500$ turns.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:9}
$F = 10 \times 0.12 \times 0.50 = \qty{0.60}{N}$;

\end{solution}

\begin{solution}{exo:g11:magnetic-fields:10}
Balance: $ILB = mg$, so
$I = \num{5.0e-3} \times 9.81 / (0.20 \times 0.10) \approx
\qty{2.5}{A}$. The \omterm{def:g10:inertia:force}{force} must point up; with $\vect B$ horizontal and
perpendicular to the wire, the flat right hand (palm up) fixes the
required current direction.
\end{solution}

(c) no \omterm{def:g10:inertia:force}{force} --- the wire is parallel to the \omterm{def:g11:electric-gravitational-fields:field}{field}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:12}

\end{solution}

\begin{solution}{exo:g11:magnetic-fields:13}
\emph{1.} $I = U/R = 12/2.0 = \qty{6.0}{A}$;
$n = 400/0.20 = \qty{2000}{m^{-1}}$;
$B_0 = \num{1.26e-6} \times 2000 \times 6.0 \approx \qty{15}{mT}$;
with the core, $100 B_0 \approx \qty{1.5}{T}$.

\emph{2.} $P = UI = 12 \times 6.0 = \qty{72}{W}$.

\emph{3.} No current, no \omterm{def:g11:electric-gravitational-fields:field}{field} --- and soft iron does not stay
magnetized: the load releases instantly, by design.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:14}
\emph{1.} $F = NILB = 50 \times 2.0 \times 0.050 \times 0.30 =
\qty{1.5}{N}$ on each side. The current runs opposite ways in the two
sides, so the \omterm{def:g10:inertia:force}{forces} are opposite: a couple that rotates the loop.

\emph{2.} When parallel to $\vect B$, a wire feels no \omterm{prop:g11:magnetic-fields:laplace}{Laplace force}.

\emph{3.} After half a turn the couple would reverse and undo the
rotation; the \omterm{rem:g11:magnetic-fields:motor}{commutator} flips the current each half-turn so the loop
keeps spinning one way.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:15}
\emph{1.} $d = \num{2e-7} \times 100 / 1 = \qty{2.0e-5}{m} =
\qty{20}{\micro m}$ --- deep \emph{inside} the millimetre-thick copper.
No accessible point near a single wire reaches \qty{1}{T}: strong
\omterm{def:g11:electric-gravitational-fields:field}{fields} stack thousands of turns (a \omterm{def:g11:magnetic-fields:solenoid}{solenoid}), plus iron or
superconductors.

\emph{2.} $\num{e8} / \num{5e-5} = \num{2e12}$: about twelve \omterm{def:g10:energy-conservation:power}{powers} of
ten.
\end{solution}

\begin{solution}{pb:g11:magnetic-fields:1}
\textbf{1.} The needle is itself a small \omterm{def:g11:magnetic-fields:magnet}{magnet}; the Earth's \omterm{def:g11:electric-gravitational-fields:field}{field}
aligns it, north end along the horizontal component of $\vect B$,
i.e.\ toward magnetic north.

\textbf{2.} $B = \sqrt{20^2 + 44^2} = \sqrt{2336} \approx
\qty{48}{\micro T}$, tilted $\tan^{-1}(44/20) \approx \ang{66}$ below
the horizontal.

\textbf{3.} The track runs $\ang{6}$ west of true north:
$30 \times \sin\ang{6} \approx \qty{3.1}{km}$ west.

\textbf{4.} Errors add: $\ang{10}$, so $30 \times \sin\ang{10} \approx
\qty{5.2}{km}$ --- five kilometres for ten degrees.

\textbf{5.} The horizontal component: near the pole the \omterm{def:g11:electric-gravitational-fields:field}{field} is
almost vertical, the horizontal part shrinks toward zero, and the
needle has nothing left to align with.

\textbf{6.} $n = 600/0.30 = \qty{2000}{m^{-1}}$.

\textbf{7.} $I = U/R = 12/1.5 = \qty{8.0}{A}$.

\textbf{8.} $B_0 = \mu_0 n I = \num{1.26e-6} \times 2000 \times 8.0
\approx \qty{20}{mT}$.

\textbf{9.} $50 \times \qty{20}{mT} = \qty{1.0}{T}$ --- twice the
\omterm{def:g11:electric-gravitational-fields:field}{field} at a neodymium pole, over a far larger face.

\textbf{10.} $P = UI = 12 \times 8.0 = \qty{96}{W}$; in
\qty{10}{\minute}: $E = 96 \times 600 \approx \qty{58}{kJ}$.

\textbf{11.} Because it lets go: open the switch, the \omterm{def:g11:electric-gravitational-fields:field}{field} dies, the
car drops exactly where wanted. A permanent \omterm{def:g11:magnetic-fields:magnet}{magnet} grips forever.

\textbf{12.} $F = NILB = 100 \times 3.0 \times 0.040 \times 0.25 =
\qty{3.0}{N}$.

\textbf{13.} A couple: applied on opposite sides of the axis, the two
opposite \omterm{def:g10:inertia:force}{forces} both turn the loop the same way. Equal \emph{parallel}
\omterm{def:g10:inertia:force}{forces} would only push the loop sideways, not spin it.

\textbf{14.} None: a wire parallel to the \omterm{def:g11:electric-gravitational-fields:field}{field} feels no \omterm{prop:g11:magnetic-fields:laplace}{Laplace
force}.

\textbf{15.} It reverses the current in the loop at each half-turn,
just as the couple would change sign --- so the torque always drives
the same rotation.

\textbf{16.} Double $N$, or double $I$, or double $B$: the \omterm{def:g10:inertia:force}{force} pair
$F = NILB$ is proportional to each.

\textbf{17.} $1.5 / \num{5.0e-5} = \num{3.0e4}$: thirty thousand
Earths.

\textbf{18.} $n = B/(\mu_0 I) = 1.5 / (\num{1.26e-6} \times 500)
\approx \qty{2.4e3}{m^{-1}}$ --- and every one of those turns carries
\qty{500}{A}.

\textbf{19.} $P = RI^2 = 0.20 \times 500^2 = \qty{50}{kW}$ --- a
neighbourhood's worth of heating. The real winding is
superconducting: zero \omterm{def:g11:circuits-and-power:resistance}{resistance}, zero dissipation, provided it is
kept a few degrees above absolute zero.

\textbf{20.} Compass $\num{5e-5}$, crane $1$, MRI $1.5$: the ship
sits five rungs below the two machines, which share a decade ---
between interstellar $\num{e-10}$ and neutron-star $\num{e8}$. Nothing
changed but the \omterm{def:g11:circuits-and-power:current}{amperes} (and the turns, iron and superconductor that
multiply them): one \omterm{def:g11:electric-gravitational-fields:field}{field}, three jobs.
\end{solution}
