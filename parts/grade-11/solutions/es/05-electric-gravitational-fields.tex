\begin{solution}{exo:g11:electric-gravitational-fields:1}

\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:2}

\begin{solution}{exo:g11:electric-gravitational-fields:6}

\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:7}
$E = m_e g / e = \num{9.11e-31} \times 9.81 / \num{1.60e-19}

\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:8}
\emph{1.}$E = \num{5.0e3} / \num{2.0e-2} = \qty{2.5e5}{V/m}$; en reposo
$qE = mg$, entonces
$q = \num{2.0e-6} \times 9.81 / \num{2.5e5} \approx \qty{7.8e-11}{C}$.

\emph{2.} El \omterm{def:g10:inertia:force}{fuerza} en la fibra positiva debe apuntar hacia arriba, por lo que $\vect E$
apunta hacia arriba: la placa \emph{más bajo} es positiva.

\emph{3.} $n = \num{7.85e-11} / \num{1.60e-19} \approx \num{4.9e8}$
\omterm{def:g11:fundamental-interactions:elementary}{cargas elementales}.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:9}
Fuera del segmento ambos \omterm{def:g11:electric-gravitational-fields:field}{campos} apuntan en la misma dirección; entre, ellos
oponerse --- y el equilibrio necesita el punto más cercano a la fuente más débil $+Q$.
Con $x$ la distancia a $+Q$: $kQ/x^2 = 4kQ/(0.30 - x)^2$, entonces
$0.30 - x = 2x$ y $x = \qty{0.10}{m}$ de $+Q$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:10}
$F_e = ke^2/d^2 = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{5.3e-11})^2 \approx \qty{8.2e-8}{N}$;
$F_g = G m_p m_e / d^2 = \num{6.67e-11} \times \num{1.67e-27} \times
\num{9.11e-31} / (\num{5.3e-11})^2 \approx \qty{3.6e-47}{N}$. proporción
$F_e / F_g \approx \num{2.3e39}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:11}
$E = \sqrt{300^2 + 400^2} = \qty{500}{N/C}$, en
$\tan^{-1}(400/300) \approx \ang{53}$ al norte del este. En $q < 0$:
$F = \num{2.0e-6} \times 500 = \qty{1.0e-3}{N}$, frente al \omterm{def:g11:electric-gravitational-fields:field}{campo} ---
$\ang{53}$ al sur del oeste.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:12}
$GM/(R+h)^2 = \tfrac12\,GM/R^2$ da $R + h = R\sqrt{2}$, entonces
$h = (\sqrt{2} - 1)R \approx \qty{2.6e6}{m}$ (sobre \qty{2600}{km}). en
$h = R$, $d = 2R$: una cuarta parte del valor de la superficie.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:13}
\emph{1.}$AM = \sqrt{3.0^2 + 4.0^2} = \qty{5.0}{cm}$; cada carga
crea $E_1 = \num{8.99e9} \times \num{2.0e-9} / 0.050^2
\approx \qty{7.2e3}{N/C}$.

\emph{2.} Los componentes a lo largo de $AB$ se cancelan por simetría; cada \omterm{def:g11:electric-gravitational-fields:field}{campo}
contribuye $E_1 \times 4/5$ a lo largo de la bisectriz, lejos del segmento:
$E = 2 \times \num{7.19e3} \times 0.80 \approx \qty{1.2e4}{N/C}$,
dirigido a lo largo de la bisectriz, lejos de $[AB]$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:14}
Los cargos gratuitos de la caja se mueven bajo el \omterm{def:g11:electric-gravitational-fields:field}{campo} externo: $+$ se acumula en
una cara, $-$ en la otra, y estas cargas desplazadas crean una
\omterm{def:g11:electric-gravitational-fields:field}{campo} interno opuesto al externo. Siguen moviéndose hasta que
El total de \omterm{def:g11:electric-gravitational-fields:field}{campo} en el interior es exactamente cero --- entonces el equilibrio. Ofertas de gravedad
sólo un signo de fuente: cada masa \emph{agrega} y una atracción \omterm{def:g11:electric-gravitational-fields:field}{campo},
nadie puede oponerse a ello. Un escudo de gravedad necesitaría masa negativa, lo que
no existe.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{Peso} $m\vect g$ (abajo), \omterm{def:g10:inertia:inventory}{tensión} $\vect T$ (a lo largo del
rosca), eléctrico \omterm{def:g10:inertia:force}{fuerza} $q\vect E$ (horizontal). En reposo,
$T\sin\theta = qE$ y $T\cos\theta = mg$; dividir:
$\tan\theta = qE/(mg)$.

\emph{2.} $\tan\theta = \num{2.0e-7} \times \num{1.0e4} /
(\num{5.0e-4} \times 9.81) \approx 0.41$, so$\theta \approx \ang{22}$;
$T = mg/\cos\theta \approx \qty{5.3e-3}{N}$.
\end{solution}

\begin{solution}{pb:g11:electric-gravitational-fields:1}
\textbf{1.} \omterm{def:g11:electric-gravitational-fields:capacitor}{Uniforme}, perpendicular a las placas, apuntando hacia abajo (desde
la placa superior $+$ a la inferior $-$); uniformidad significa el \omterm{def:g10:inertia:force}{fuerza}
en la gota es el mismo dondequiera que se desplace.
\textbf{2.}$E = U/d = 800 / \num{8.0e-3} = \qty{1.0e5}{V/m}$.
\textbf{3.} La carga es negativa, por lo que \omterm{def:g10:inertia:force}{fuerza} es \emph{opuesto}
$\vect E$: hacia arriba. Sólo con la placa $+$ arriba apunta $\vect E$
abajo y el \omterm{def:g10:inertia:force}{fuerza} eléctrico lucha contra el \omterm{def:g10:universal-gravitation:weight}{peso}.
\textbf{4.} $P = mg = \num{4.90e-15} \times 9.81 \approx
\qty{4.8e-14}{N}$.
\textbf{5.} $\abs{q}E = mg$.
\textbf{6.} $\abs{q} = mg/E = mgd/U$.
\textbf{7.} $\abs{q} = \num{4.81e-14} \times \num{8.0e-3} / 800
\approx \qty{4.8e-19}{C}$.
\textbf{8.} $n = \num{4.8e-19} / \num{1.60e-19} = 3$ exceso de electrones.
\textbf{9.} $E$ crece, por lo que $\abs{q}E > mg$: el \omterm{def:g10:inertia:force}{fuerza} neto apunta hacia arriba.
\textbf{10.} De $\abs{q}\,U/d = mg$, $U = mgd/\abs{q}$ --- inversamente
proporcional a la carga.
\textbf{11.} $\abs{q} = mgd/U = \num{3.85e-16}/1200 \approx
\qty{3.2e-19}{C} = 2e$: desde$3e$, la gota \emph{perdido} una
electrón.
\textbf{12.} $\abs{q} = \num{3.85e-16}/U$: \qty{3.2e-19}{C},
\qty{4.8e-19}{C}, \qty{6.4e-19}{C}, \qty{8.0e-19}{C},
\qty{9.6e-19}{C}.
\textbf{13.} Son $2e, 3e, 4e, 5e, 6e$ con
$e = \qty{1.60e-19}{C}$.
\textbf{14.} Cada paso cambia $\abs{q}$ exactamente
\qty{1.6e-19}{C}: aterriza un electrón a la vez.
\textbf{15.} Necesitaría $\abs{q} = \num{3.85e-16}/700 \approx
\qty{5.5e-19}{C} = 3.4\,e$ --- no un número entero de electrones, por lo que no
Una ráfaga de radiación alguna vez puede producirlo.
\textbf{16.} El cargo es \emph{cuantificado}: viene en múltiplos enteros
del \omterm{def:g11:fundamental-interactions:elementary}{carga elemental} $e = \qty{1.60e-19}{C}$ --- la cantidad
Millikan midió.
\textbf{17.} $q \leftrightarrow m$, $E \leftrightarrow g$ y
$\abs{q}E \leftrightarrow mg$: el propio \omterm{def:g10:universal-gravitation:weight}{peso}.
\textbf{18.}$g_{\text{sphere}} = GM/d^2 = \num{6.67e-11} \times 1000 /
1.0^2 = \qty{6.7e-8}{N/kg}$; \omterm{def:g10:inertia:force}{fuerza} en la gota
$\num{4.90e-15} \times \num{6.7e-8} \approx \qty{3.3e-22}{N}$ --- acerca de
$\num{1.5e8}$ veces más pequeño que su \omterm{def:g10:universal-gravitation:weight}{peso}. Desesperanzado.
\textbf{19.} Debido a que $G$ es tan pequeño, solo hay una masa del tamaño de un planeta sobre su cabeza.
podría rivalizar con la atracción de la Tierra, y ninguna masa la repele. El enorme $k$
y los dos signos de carga permiten que una placa de mesa extraiga el planeta
en cualquier dirección.
\textbf{20.} Un electrón: $\Delta\abs{q}/\abs{q} = e/3e \approx 33\%$;
un aceite \omterm{def:g11:fundamental-interactions:ladder}{molécula}: $\Delta m/m = \num{5e-25} / \num{4.90e-15}
\approx \num{e-10}$. La escalera de carga mueve el equilibrio \omterm{def:g11:circuits-and-power:voltage}{Voltaje}
por cientos de \omterm{def:g11:circuits-and-power:voltage}{voltios}; la escalera masiva es invisible en cualquier \omterm{def:g11:circuits-and-power:voltage}{Voltaje}.
El saldo por tanto resuelve el \omterm{def:g11:fundamental-interactions:ladder}{átomo} de electricidad ---
$e = \qty{1.60e-19}{C}$ --- mientras que el \omterm{def:g11:fundamental-interactions:ladder}{moléculas} de aceite se difumina en un
continuo.
\end{solution}
