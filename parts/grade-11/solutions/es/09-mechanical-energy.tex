\begin{solution}{exo:g11:mechanical-energy:8}

\end{solution}

\begin{solution}{exo:g11:mechanical-energy:13}
Misma inicial $E_m = \tfrac12 m \times 15^2 + m \times 9.81 \times 45$
cualquiera que sea el ángulo, por lo que la velocidad sobre el terreno es
$v = \sqrt{225 + 2 \times 9.81 \times 45} = \sqrt{1108} \approx
\qty{33}{m/s}$. El ángulo fija el \emph{dirección} del rellano.
velocidad, el tiempo de vuelo y el \omterm{def:g11:fundamental-interactions:strong}{rango} --- no la velocidad de aterrizaje.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:14}
$\tfrac12 mv^2 = mgh - f\ell = 80 \times 9.81 \times 120 - 65 \times
800 = \num{94176} - \num{52000} = \qty{4.22e4}{J}$, entonces
$v = \sqrt{2 \times \num{42176}/80} \approx \qty{32}{m/s}$
(\qty{117}{km/h}). Fracción \omterm{def:g11:mechanical-energy:dissipation}{disipado}: $52000/94176 \approx 55\%$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:15}
$m = \qty{2.0e9}{kg}$:
$E_p = \num{2.0e9} \times 9.81 \times 300 \approx \qty{5.9e12}{J}
= \num{5.9e12}/\num{3.6e6} \approx \qty{1.6e6}{kW.h}$. Recuperado:
$0.85 \times \num{1.64e6} \approx \qty{1.4e6}{kW.h}$ --- acerca de
$\num{1.4e5}$ viviendas por un día.
\end{solution}

\begin{solution}{pb:g11:mechanical-energy:1}
\textbf{1.} $v_{\text{top}} = \sqrt{gR} = \sqrt{9.81 \times 8.0}
\approx \qty{8.9}{m/s}$.

\textbf{2.} $E_m = mg\,(2R) + \tfrac12 m gR = \num{3.14e5} +
\num{7.8e4} \approx \qty{3.92e5}{J}$.

\textbf{3.} $mgH_{\min} = E_m$ da $H_{\min} = \qty{20}{m}$;
algebraicamente $H_{\min} = 2R + \frac{gR}{2g} = \frac{5R}{2}$.

\textbf{4.}$v_{\text{top}} = \sqrt{2g(H - 2R)} =
\sqrt{2 \times 9.81 \times 8.0} \approx \qty{12.5}{m/s}$;
$v_{\text{top}}^2/(gR) = 2.0$ --- el doble del mínimo.

\textbf{5.} Cada \omterm{def:g10:energy-conservation:energy}{energía} es proporcional a $m$, por lo que $m$ cancela: igual
velocidades y la misma seguridad, lleno o vacío.

\textbf{6.} $v = \sqrt{2 \times 9.81 \times 24} \approx \qty{21.7}{m/s}
\approx \qty{78}{km/h}$.

\textbf{7.} $v = \sqrt{2 \times 9.81 \times (24 - 8.0)} \approx
\qty{17.7}{m/s}$.

\textbf{8.} $mgH = \qty{4.71e5}{J}$. $(E_p, E_k)$: lanzamiento
$(\num{4.71e5}, 0)$; abajo $(0, \num{4.71e5})$; lado
$(\num{1.57e5}, \num{3.14e5})$; arriba $(\num{3.14e5}, \num{1.57e5})$ ---
cada par suma \qty{4.71e5}{J}.

\textbf{9.} $v = \sqrt{2 \times 9.81 \times (24 - 12)} \approx
\qty{15.3}{m/s}$.

\textbf{10.} $v(z) = \sqrt{2g(H - z)}$: más rápido en el punto más bajo de
la pista, más lento en la parte superior del circuito.

\textbf{11.} Misma altura, por lo que $\Delta E_p = 0$ y
$\Delta E_m = \Delta E_k = \tfrac12 \times 2000 \times (19.2^2 -
20.4^2) \approx \qty{-4.75e4}{J}$: la colocación purifica el déficit
cinético, lea dos velocímetros.

\textbf{12.} $W(\vect f) = \qty{-4.75e4}{J}$;
$f = \num{4.75e4}/60 \approx \qty{7.9e2}{N}$.

\textbf{13.} \omterm{def:g10:energy-conservation:forms}{Energía térmica}, en las ruedas, los rieles y el aire.

\textbf{14.} $E_m$ en el primer sensor: $\num{4.16e5} + \num{3.9e4} =
\qty{4.55e5}{J}$; pérdida$\num{4.75e4}/\num{4.55e5} \approx 10\%$ por
\qty{60}{m}. En $H_{\min}$ se alcanza la parte superior del bucle \emph{exactamente}
en $\sqrt{gR}$; cualquier pérdida lo sitúa por debajo de la velocidad de seguridad.

\textbf{15.} Perdiendo $\approx 10\%$ por \qty{60}{m}, aproximadamente $15\%$ de
$mgH$ ($\approx \qty{7.1e4}{J}$) ya no está: $E_m \approx \qty{4.00e5}{J}$,
todavía por encima del \qty{3.92e5}{J} requerido --- seguro, con solo $2\%$ para
repuesto. El margen no fue un lujo.

\textbf{16.} $E_k = \tfrac12 \times 2000 \times 18.0^2 =
\qty{3.24e5}{J}$.

\textbf{17.} $F = \num{3.24e5}/45 = \qty{7.2e3}{N}$.

\textbf{18.}$a = 7200/2000 = \qty{3.6}{m/s^2} \approx 0.37\,g$:
firme pero cómodo.

\textbf{19.} Pista \omterm{def:g10:inertia:inventory}{fricción}: $mgH - \num{3.24e5} = \num{470880} -
\num{324000} = \qty{1.47e5}{J}$. Auditoría:$\num{1.47e5} + \num{3.24e5} =
\num{4.71e5} = mgH$ --- cada \omterm{def:g11:work-of-force:work}{joule} contabilizado.

\textbf{20.} La altura de lanzamiento $H = \qty{24}{m}$ arregló el bucle.
seguridad, cada velocidad $\sqrt{2g(H - z)}$, la factura \omterm{def:g10:inertia:inventory}{fricción} y el freno
\omterm{def:g10:inertia:force}{fuerza}; el precio de ignorar \omterm{def:g10:inertia:inventory}{fricción} es el margen --- el $\qty{4}{m}$
de cerro que el mundo del frotamiento consume silenciosamente.
\end{solution}
