\begin{solution}{exo:g11:circuits-and-power:3}
Serie: $120 + 60 = \qty{180}{\ohm}$. Paralelo: $\frac{120 \times 60}{180}
= \qty{40}{\ohm}$ --- más pequeño que ambos: mismo \omterm{def:g11:circuits-and-power:voltage}{Voltaje}, pero el actual
obtiene una segunda ruta, por lo que hay más corriente total y menos \omterm{def:g11:circuits-and-power:resistance}{resistencia}.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:4}
$P = UI = 230 \times 8.7 = \qty{2.0}{kW}$. In $t = \qty{240}{s}$:
$E = Pt = 2000 \times 240 = \qty{4.8e5}{J} = 4.8\times10^{5}/3.6
\times10^{6} \approx \qty{0.13}{kW.h}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:5}
$E = 60 \times 5.0 = \qty{300}{W.h} = \qty{0.30}{kW.h} = 0.30 \times
3.6\times10^{6} = \qty{1.1e6}{J}$. Costo:$0.30 \times 0.25 = 0.075$

\end{solution}

\begin{solution}{exo:g11:circuits-and-power:6}
\emph{1.} $R = U^2/P = 230^2/1200 = \qty{44}{\ohm}$;
$I = P/U = 1200/230 = \qty{5.2}{A}$.

\emph{2.} $P' = U'^2/R = 115^2/44.1 = \qty{300}{W}$: un cuarto, no
la mitad, porque reducir a la mitad $U$ también reduce a la mitad $I$ y $P = UI \propto U^2$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:7}
$R_{\text{s}} = \qty{60}{\ohm}$, entonces $I = 12/60 = \qty{0.20}{A}$.
$U_1 = R_1 I = \qty{8.0}{V}$, $U_2 = \qty{4.0}{V}$ (suma \qty{12}{V});
$P_1 = U_1 I = \qty{1.6}{W}$, $P_2 = \qty{0.8}{W}$: total \qty{2.4}{W}
$= UI = 12 \times 0.20$, como debe ser.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:8}
$I_1 = 12/60 = \qty{0.20}{A}$, $I_2 = 12/30 = \qty{0.40}{A}$, totales
$I = \qty{0.60}{A}$. Equivalente: $R = U/I = 12/0.60 = \qty{20}{\ohm}$,
y de hecho $\frac{60 \times 30}{60 + 30} = \qty{20}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:9}
$\mathcal{E} - 0.50\,r = 5.5$ y $\mathcal{E} - 2.0\,r = 4.0$;
restando, $1.5\,r = 1.5$, entonces $r = \qty{1.0}{\ohm}$ y $\mathcal{E} =
\qty{6.0}{V}$. Cortocircuito:$\mathcal{E}/r = \qty{6.0}{A}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:10}
Hervidor $2200/230 = \qty{9.6}{A}$, tostadora $1000/230 = \qty{4.3}{A}$:
total \qty{13.9}{A} $< \qty{16}{A}$, el \omterm{def:g11:circuits-and-power:joule}{fusible} se mantiene. Calentador $1500/230 =
\qty{6.5}{A}$: total \qty{20.4}{A}$> \qty{16}{A}$, explota
(equivalentemente, $\qty{4.7}{kW} > 230 \times 16 \approx \qty{3.7}{kW}$).
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:11}
En \qty{200}{V}: $I = 10^4/200 = \qty{50}{A}$, pérdida $RI^2 = 2.0 \times
50^2 = \qty{5.0}{kW}$--- la mitad de la entrega. En \qty{4.0}{kV}:$I =
\qty{2.5}{A}$, pérdida$2.0 \times 2.5^2 = \qty{12.5}{W}$. Relación$400 =
20^2$: veinte veces la \omterm{def:g11:circuits-and-power:voltage}{Voltaje} significa veinte veces menos corriente, y
la pérdida es $I^2$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:12}
Cuatro diseños: todos en serie, $3 \times 60 = \qty{180}{\ohm}$; todo dentro
paralelo, $60/3 = \qty{20}{\ohm}$; uno en serie con dos en paralelo,
$60 + 30 = \qty{90}{\ohm}$; uno en paralelo con dos en serie,
$\frac{60 \times 120}{180} = \qty{40}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:13}
\emph{1.} $I = \mathcal{E}/(R + r) = 4.5/7.5 = \qty{0.60}{A}$;
$U = RI = \qty{3.6}{V}$; útil $P = UI = \qty{2.2}{W}$; perdido $rI^2 =
1.5 \times 0.60^2 = \qty{0.54}{W}$;$\eta = U/\mathcal{E} = 3.6/4.5 =
0.80$.

\emph{2.} Por la pista, $P = \mathcal{E}^2 \big/ \big[(R - r)^2/R +
4r\big]$; el soporte es menor ($= 4r$) exactamente cuando$R = r$, por lo que
$P_{\max} = \mathcal{E}^2/(4r) = 4.5^2/6.0 = \qty{3.4}{W}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:14}
Calor: $Q = 1.0 \times 4180 \times (100 - 20) = \qty{3.3e5}{J}$.
Eléctrico: $E = Q/0.90 = \qty{3.7e5}{J}$. Hora: $t = E/P =
3.7\times10^{5}/2200 \approx \qty{170}{s}$, menos de tres minutos. Costo:
$3.7\times10^{5}/3.6\times10^{6} \approx \qty{0.10}{kW.h}$, sobre $2.6$
centavos.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:15}
\emph{1.} $q = 60 \times 3600 = \qty{2.2e5}{C}$; $E \approx \mathcal{E}q
= 12.7 \times 2.16\times10^{5} = \qty{2.7e6}{J} \approx \qty{0.76}{kW.h}$.

\emph{2.}$U = 12.7 - 0.020 \times 150 = \qty{9.7}{V}$; entregado
$UI = \qty{1.5}{kW}$; perdido $rI^2 = 0.020 \times 150^2 = \qty{450}{W}$.

\emph{3.} Los faros se encuentran a través de los terminales: mientras arranca, el
terminal \omterm{def:g11:circuits-and-power:voltage}{Voltaje} cae de $12.7$ a $\qty{9.7}{V}$, por lo que reciben
menos \omterm{def:g10:energy-conservation:power}{fuerza} y tenue.
\end{solution}

\begin{solution}{pb:g11:circuits-and-power:1}
\textbf{1.} Cada segundo, $E_p = mgh = 5.0\times10^{4} \times 9.81 \times
200 = \qty{9.8e7}{J}$: disponible \omterm{def:g10:energy-conservation:power}{fuerza} \qty{98}{MW}.

\textbf{2.} $0.90 \times 98 = \qty{88}{MW}$.

\textbf{3.} $I = P/U = 8.8\times10^{7}/2.0\times10^{4} = \qty{4.4e3}{A}$.

\textbf{4.} $R_\ell I^2 = 32 \times (4.4\times10^{3})^2 \approx
\qty{6.2e8}{W} = \qty{620}{MW}$ --- siete veces la producción de la planta:
transmitir en \qty{20}{kV} es imposible.

\textbf{5.} Con $P = UI$ fijo, sólo al elevar $U$ se encoge $I$: a
transformador.

\textbf{6.} $I = 8.8\times10^{7}/4.0\times10^{5} = \qty{220}{A}$.

\textbf{7.} $R_\ell I^2 = 32 \times 220^2 \approx \qty{1.6}{MW}$.

\textbf{8.} $1.6/88 \approx 0.018$: sobre $1.8\%$.

\textbf{9.} Fracción $= R_\ell I^2/P = R_\ell (P/U)^2/P = R_\ell P/U^2
\propto 1/U^2$: de \qty{20}{kV} a \qty{400}{kV} cae por
$20^2 = 400$ y, de hecho, $\qty{620}{MW}/400 \approx \qty{1.6}{MW}$.

\textbf{10.} $R_\ell I = 32 \times 220 \approx \qty{7.1}{kV}$: $1.8\%$
de \qty{400}{kV}, la misma fracción que la pregunta 8, ya que
$R_\ell I/U = R_\ell I^2/UI$.

\textbf{11.} $\qty{1}{kW.h} = \qty{3.6e6}{J}$: el medidor multiplica el
\omterm{def:g10:energy-conservation:power}{fuerza} extraído en el momento en que se extrae, y suma.

\textbf{12.} $I = 900/230 = \qty{3.9}{A}$;
$R = U^2/P = 230^2/900 \approx \qty{59}{\ohm}$.

\textbf{13.}$E = 900 \times 180 = \qty{1.6e5}{J} = \qty{0.045}{kW.h}$:
alrededor de $1.1$ centavos.

\textbf{14.} La física de $RI^2$ es idéntica; solo el propósito
difiere --- en la línea el calor no es deseado, en la tostadora el calor
\emph{is} el producto.

\textbf{15.} $3.9 + 9.6 = \qty{13.5}{A} < \qty{16}{A}$: retiene. Calentador
$2000/230 = \qty{8.7}{A}$: total $\qty{22.2}{A} > \qty{16}{A}$, el \omterm{def:g11:circuits-and-power:joule}{fusible}
golpes.

\textbf{16.} $0.05 \times 13.5^2 = \qty{9.1}{W}$: $9.1/3100 \approx
0.3\%$ del \omterm{def:g10:energy-conservation:power}{fuerza} entregado --- insignificante en cables cortos.

\textbf{17.} $\eta = 0.90 \times 0.99 \times 0.982 \times 0.99 \times
0.997 \approx 0.86$.

\textbf{18.} Un \omterm{def:g10:energy-conservation:kwh}{kilovatio-hora} tostado necesita $3.6\times10^{6}/0.86
\approx \qty{4.2e6}{J}$aguas arriba:$m = E/(gh) = 4.2\times10^{6}/(9.81
\times 200) \approx \qty{2.1e3}{kg}$, aproximadamente \qty{2.1}{m^3} de agua.

\textbf{19.} $R_\ell P/U^2 = 32 \times 8.8\times10^{7}/(2.0\times
10^{4})^2 \approx 7$. Una ``fracción'' arriba de$1$ significa que la línea
disipar más de lo que lleva: la entrega no puede ocurrir en ese
\omterm{def:g11:circuits-and-power:voltage}{Voltaje} en absoluto.

\textbf{20.} Un tostado \omterm{def:g10:energy-conservation:kwh}{kilovatio-hora} le cuesta a la montaña unos dos
toneladas --- \qty{2.1}{m^3} --- de agua que cae \qty{200}{m}, y algunas
$14\%$ del \omterm{def:g10:energy-conservation:energy}{energía} muere en ruta, principalmente en turbinas y transformadores.
(bajo $2\%$ en \qty{1000}{km} de línea). El único truco que hace que
El viaje posible es el transformador: avanzando a \qty{400}{kV}.
divide la corriente por $20$ y la pérdida en julios por $400$.
\end{solution}
