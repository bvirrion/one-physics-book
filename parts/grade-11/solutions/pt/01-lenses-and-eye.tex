\begin{solution}{exo:g11:lenses-and-eye:2}
\omterm{def:g11:lenses-and-eye:lens}{Divergente} ($ f' < 0$). Não: para qualquer distância do objeto $ d $,

\begin{solution}{exo:g11:lenses-and-eye:15}
$1/\overline{OA'} = 1/f' - 1/d = (d - f')/(f'd)$, então
$\overline{OA'} = f'd/(d - f')$ e
$\gamma = \overline{OA'}/(-d) = f'/(f' - d)$. Se $ d > f'$:
$\overline{OA'} > 0$ (real) e $\gamma < 0$ (invertido). Se
$0 < d < f'$: $\overline{OA'} < 0$ (\omterm{def:g11:lenses-and-eye:images}{virtual}) e
$\gamma = f'/(f' - d) > 1$: \omterm{def:g10:universal-gravitation:weight}{vertical} e ampliado --- o \omterm{def:g11:lenses-and-eye:magnifier}{lupa}
contexto.
\end{solution}

\begin{solution}{pb:g11:lenses-and-eye:1}
\textbf{1.} $1/\overline{OA'} = 1/20 - 1/60 = 1/30$:
$\overline{OA'} = +\qty{30}{cm}$, $\gamma = -0.50$ --- real, invertido,
metade do tamanho.

\textbf{2.} $\overline{OA'} = +\qty{60}{cm}$, $\gamma = -2$: o
posições \qty{30}{cm} e \qty{60}{cm} trocaram de papéis, o
ampliações são recíprocos --- a luz invertida segue o mesmo
caminho.

\textbf{3.} $1/\overline{OA'} = 1/20 - 1/10 = -1/20$:
$\overline{OA'} = \qty{-20}{cm}$, $\gamma = +2$ --- \omterm{def:g11:lenses-and-eye:images}{virtual}, em pé,
dobrou.

\textbf{4.} Do infinito a imagem começa no plano focal; como o
abordagens de objetos $ F $ o real, a imagem invertida retrocede ao infinito e
cresce; dentro $ f'$ acontece \omterm{def:g11:lenses-and-eye:images}{virtual}, \omterm{def:g10:universal-gravitation:weight}{vertical}, ampliado, na frente de
o \omterm{def:g11:lenses-and-eye:lens}{lente}.

\textbf{5.} $1/\overline{OA} \approx 0$ folhas
$1/\overline{OA'} = 1/f'$: $\overline{OA'} = f'$, o plano focal.

\textbf{6.} No plano focal, \qty{50}{mm} atrás do \omterm{def:g11:lenses-and-eye:lens}{lente}.

\textbf{7.} $1/\overline{OA'} = 1/50 - 1/1000 = 19/1000$:
$\overline{OA'} \approx \qty{52.6}{mm}$; viagem $\approx \qty{2.6}{mm}$.

\textbf{8.} $1/\overline{OA} = 1/55 - 1/50 = -1/550$: focagem mínima
distância \qty{550}{mm} = \qty{55}{cm}.

\textbf{9.} $\gamma = 55/(-550) = -0.10$: as imagens do rosto em
\qty{20}{mm}, logo dentro do \qty{24}{mm} sensor.

\textbf{10.} $\gamma = -1$ forças $\overline{OA'} = -\overline{OA}$,
então $2/\overline{OA'} = 1/f'$: objeto \qty{100}{mm} ($= 2f'$) na frente,
sensor \qty{100}{mm} atrás --- o dobro da paisagem, muito além
o \qty{55}{mm} mecanismo; \omterm{def:g11:lenses-and-eye:lens}{lentes} macro fornecem deslocamento extra.

\textbf{11.} Qualquer coisa mais perto do que ela \omterm{def:g11:lenses-and-eye:eye}{ponto próximo} (\qty{1.0}{m}) desfoca:
braços esticados aproximam a página o mais possível \qty{1.0}{m} possível ---
afiado, se pequeno.

\textbf{12.} $ C = 1/(-1.0) - 1/(-0.25) = -1.0 + 4.0 = +3.0$~D.

\textbf{13.} $\gamma = (-1.0)/(-0.25) = +4.0$: quatro vezes maior, mas
quatro vezes mais longe --- o mesmo tamanho angular; o \omterm{def:g10:signals-and-waves:oscilloscope}{ganho} não é tamanho, mas
\emph{nitidez}, já que a imagem agora está com ela \omterm{def:g11:lenses-and-eye:eye}{ponto próximo}.

\textbf{14.} A Sharp precisa da imagem além \qty{1.0}{m}: da página
em \qty{25}{cm} (imagem em \qty{1.0}{m}) para a página em

\textbf{15.} Daí os bifocais: leitura poder na metade inferior do

\textbf{18.} Pedra em $ F $: imagem no infinito, vista com zero

\textbf{19.} $ C_\infty = 1/0.017 \approx 59$~D;
$ C_{25} = 1/0.017 + 1/0.25 \approx 63$~D: sobre $4$~D de

\textbf{20.} $ f' = 1/58.8 \approx \qty{17.0}{mm}$ contra

avó remendada com $+3$~D.
\end{solution}
