\begin{solution}{exo:g11:work-of-force:4}
Velocidade constante: levantamento \omterm{def:g10:inertia:force}{vigor} $= mg = \qty{5.89e3}{N}$, então
$ W_{\text{lift}} = 5886 \times 25 \approx +\qty{147}{kJ}$ e
$ W(\vect P) = -\qty{147}{kJ}$. Soma zero: a velocidade constante o
forças equilíbrio (\cref{ch:g11:forces-and-motion}), então seu funciona
cancelar.
\end{solution}

\begin{solution}{exo:g11:work-of-force:6}
Corda: $120 \times 200 \times \cos\qty{30}{\degree} \approx
+\qty{20.8}{kJ}$; \omterm{def:g10:inertia:inventory}{atrito}: $-45 \times 200 = -\qty{9.0}{kJ}$; \omterm{def:g10:universal-gravitation:weight}{peso}
e normais \omterm{def:g10:inertia:force}{vigor}, perpendicular: $0$. Total $+\qty{11.8}{kJ}$: o

\end{solution}

\begin{solution}{exo:g11:work-of-force:7}
$ v = 25/3.6 = \qty{6.94}{m/s}$; $ P = Fv = 18 \times 6.94 \approx
\qty{125}{W}$. Em uma hora, $ W = P\,\Delta t = 125 \times 3600 =
\qty{450}{kJ}$ (equivalentemente $18 \times \qty{25000}{m}$).
\end{solution}

\begin{solution}{exo:g11:work-of-force:8}
\omterm{def:g10:universal-gravitation:weight}{Peso}, ambas as rotas: $+65 \times 9.81 \times 120 \approx
+\qty{76.5}{kJ}$ (somente queda). \omterm{def:g10:inertia:inventory}{Atrito}: $-30 \times 800 =
-\qty{24}{kJ}$ na encosta, $-30 \times 1500 = -\qty{45}{kJ}$ no
pista. Apenas \omterm{def:g10:inertia:inventory}{atrito} se preocupa com a rota.
\end{solution}

\begin{solution}{exo:g11:work-of-force:9}
Ascender $ h = 4.0\sin\qty{25}{\degree} = \qty{1.69}{m}$:
$ W(\vect P) = -20 \times 9.81 \times 1.69 \approx -\qty{332}{J}$;
$ W_f = -30 \times 4.0 = -\qty{120}{J}$. Velocidade constante: total \omterm{def:g11:work-of-force:work}{trabalhar}
zero, então $ W_{\text{push}} = +\qty{452}{J}$ e
$ F = 452/4.0 \approx \qty{113}{N}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:10}
$ v = 130/3.6 = \qty{36.1}{m/s}$; velocidade constante, então
$ F = P/v = \num{5.0e4}/36.1 \approx \qty{1.4}{kN}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:11}
$ P = Fv\cos\theta = 800 \times 1.0 \times \cos\qty{20}{\degree}
\approx \qty{752}{W}$; em \omterm{def:g11:work-of-force:power}{cavalos de potência}, $752/736 \approx \num{1.02}$ ---
quase exatamente o que um cavalo \omterm{def:g11:fundamental-interactions:strong}{forte} sustenta, a comparação Watt
construiu o seu \omterm{def:g10:orders-of-magnitude:unit}{unidade} sobre.
\end{solution}

\begin{solution}{exo:g11:work-of-force:12}
Comprimentos $ L = h/\text{grade}$: \qty{1500}{m} e \qty{5000}{m}. Forças
$ F = mg \times \text{grade}$: $1200 \times 9.81 \times 0.20 \approx
\qty{2.35}{kN}$ e $\qty{0.71}{kN}$. Funciona $ FL $: $2354 \times 1500
\approx \qty{3.53}{MJ}$ e $706 \times 5000 \approx \qty{3.53}{MJ}$
--- ambos iguais $ mgh $: o comércio rodoviário \omterm{def:g10:inertia:force}{vigor} por comprimento, nunca \omterm{def:g11:work-of-force:work}{trabalhar}.
\end{solution}

\begin{solution}{exo:g11:work-of-force:13}
$ F = mg = \qty{2.45e3}{N}$; $ W = 2453 \times 15 \approx \qty{36.8}{kJ}$;
$\Delta t = 15/0.40 = \qty{37.5}{s}$;
$ P = 36800/37.5 \approx \qty{981}{W} = 2453 \times 0.40$ ($ Fv $ concorda);
elétrica: $981/0.70 \approx \qty{1.4}{kW}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:14}
Diagonal: $-140 \times 6.0 = -\qty{840}{J}$; paredes:
$-140 \times 8.4 \approx -\qty{1.18}{kJ}$; ida e volta:
$-140 \times 12 = -\qty{1.68}{kJ}$. O \omterm{def:g10:universal-gravitation:weight}{peso} faz zero na rodada
viagem (sem queda líquida). Atritode \omterm{def:g11:work-of-force:work}{trabalhar} depende do caminho e nunca
reembolsos; o pesodepende apenas dos endpoints
(\cref{rem:g11:work-of-force:divide}).
\end{solution}

\begin{solution}{exo:g11:work-of-force:15}
Gravidade: $80 \times 9.81 \times 600 \approx \qty{471}{kJ}$; \omterm{def:g10:inertia:inventory}{atrito}:
$14 \times 12000 = \qty{168}{kJ}$; total $\qty{639}{kJ}$.
$ v = \qty{4.17}{m/s}$, então $\Delta t = 12000/4.17 = \qty{2880}{s} =
\qty{48}{min}$ e $ P = \num{6.39e5}/2880 \approx \qty{222}{W} \approx
\qty{0.30}{hp}$ --- um amador \omterm{def:g11:fundamental-interactions:strong}{forte}, um terço de um cavalo.
\end{solution}

\begin{solution}{pb:g11:work-of-force:1}
\textbf{1.} $ W = 110 \times 8.0 = +\qty{880}{J}$.

\textbf{2.} \omterm{def:g10:inertia:inventory}{Atrito}: $-95 \times 8.0 = -\qty{760}{J}$. \omterm{def:g10:universal-gravitation:weight}{Peso} e
normais \omterm{def:g10:inertia:force}{vigor} são perpendiculares ao movimento: zero cada.

\textbf{3.} Total $+\qty{120}{J}$: líquido trabalho \omterm{def:g11:work-of-force:sign}{motor}, a caixa acelera.
UM \qty{95}{N} empurrar saldos \omterm{def:g10:inertia:inventory}{atrito} --- zero \omterm{def:g11:forces-and-motion:netforce}{força resultante}, constante
velocidade, total zero \omterm{def:g11:work-of-force:work}{trabalhar}.

\textbf{4.} Diagonal: $-95 \times 10.0 = -\qty{950}{J}$; paredes:
$-95 \times 14.0 = -\qty{1.33}{kJ}$: \omterm{def:g10:inertia:inventory}{atrito} cobranças por parte \omterm{def:g10:orders-of-magnitude:unit}{medidor}.

\textbf{5.} \omterm{def:g10:inertia:inventory}{Atrito}: $-95 \times 20.0 = -\qty{1.9}{kJ}$; \omterm{def:g10:universal-gravitation:weight}{peso}:
$0$ (sem queda líquida). Apenas o \omterm{def:g10:universal-gravitation:weight}{peso} reembolsos.

\textbf{6.} $ W(\vect P) = -30 \times 9.81 \times 9.0 =
-\qty{2.65}{kJ}$; a tripulação deve fornecer pelo menos $+\qty{2.65}{kJ}$.

\textbf{7.} $(70 + 30) \times 9.81 \times 9.0 = \qty{8.83}{kJ}$; o
a participação da caixa é $2.65/8.83 \approx 30\%$ --- carregar é principalmente
auto-transporte.

\textbf{8.} $\sin\alpha = 9.0/30 = 0.30$:
$ F = mg\sin\alpha + f = 88.3 + 40 = \qty{128}{N}$;
$ W = 128 \times 30 = \qty{3.85}{kJ}$ ($= 2.65 + 1.2$).

\textbf{9.} Elevador \omterm{def:g10:universal-gravitation:weight}{vertical}: $ F = mg = \qty{294}{N}$,
$ W = \qty{2.65}{kJ}$. A rampa compra um menor \omterm{def:g10:inertia:force}{vigor} ($128$ contra
\qty{294}{N}) ao preço de mais \omterm{def:g11:work-of-force:work}{trabalhar}: atritode \qty{1.2}{kJ}
pedágio.

\textbf{10.} $\Delta t = 9.0/0.50 = \qty{18}{s}$;
$ P = 2649/18 \approx \qty{147}{W}$.

\textbf{11.} $ P = Fv = 294 \times 0.50 = \qty{147}{W}$ --- mesmo.
Elétrica: $147/0.60 \approx \qty{245}{W}$.

\textbf{12.} $1000 \times 0.24 = \qty{240}{m}$ e
$8000 \times 0.030 = \qty{240}{m}$: mesmo cume.

\textbf{13.} $ F = mg \times \text{grade}$: \omterm{def:g11:fundamental-interactions:strong}{faixa}
$3500 \times 9.81 \times 0.24 \approx \qty{8.2}{kN}$; estrada
$\approx \qty{1.0}{kN}$.

\textbf{14.} $8240 \times 1000 \approx \qty{8.24}{MJ}$ e
$1030 \times 8000 \approx \qty{8.24}{MJ}$ --- ambos $= mgh = 3500
\times 9.81 \times 240$.

\textbf{15.} Rolando \omterm{def:g11:circuits-and-power:resistance}{resistência}: $400 \times 1000 = \qty{0.40}{MJ}$
contra $400 \times 8000 = \qty{3.2}{MJ}$: o conforto da estrada suave
custos \qty{2.8}{MJ} de extras \omterm{def:g10:inertia:inventory}{atrito}.

\textbf{16.} $ v = \qty{10}{m/s}$. Estrada:
$(1030 + 400) \times 10 = \qty{14.3}{kW} \approx \qty{19}{hp}$; \omterm{def:g11:fundamental-interactions:strong}{faixa}:
$(8240 + 400) \times 10 = \qty{86.4}{kW} \approx \qty{117}{hp}$ ---
além de uma van carregada. As estradas ziguezagueiam para que \omterm{def:g11:work-of-force:sign}{motores} (e pneus) modestos
pode escalar: o mesmo \omterm{def:g11:work-of-force:work}{trabalhar}, espalhar \omterm{def:g11:lenses-and-eye:lens}{afinar}.

\textbf{17.} Por caixa: $880 + 2649 = \qty{3.53}{kJ}$; para $60$ caixas,
$60 \times 3529 \approx \qty{212}{kJ}$.

\textbf{18.} Escadaria: $20 \times 70 \times 9.81 \times 9.0 \approx
\qty{124}{kJ}$. Total do dia $\approx \qty{336}{kJ} = \qty{0.34}{MJ}$.

\textbf{19.} $ P = \num{3.36e5}/28800 \approx \qty{12}{W}$. A chaleira:
$\num{3.36e5}/2000 = \qty{168}{s}$ --- menos de três minutos para o
um dia inteiro de trabalho.

\textbf{20.} O \omterm{def:g10:energy-conservation:energy}{joule} paga por \omterm{def:g10:inertia:force}{vigor} entregue \emph{junto} um movimento
--- nada para a parede empurrada ou a caixa segurada, por mais exaustiva que seja
(músculos queimam energia química mesmo fazendo zero mecânica \omterm{def:g11:work-of-force:work}{trabalhar}). Desde
humano \omterm{def:g11:work-of-force:power}{poder} a produção é de algumas dezenas \omterm{def:g11:work-of-force:power}{watts}, os \omterm{def:g11:work-of-force:sign}{motores} permitem que as máquinas se remodelem
o \omterm{def:g10:inertia:force}{vigor} --- guincho, rampa, ziguezague --- enquanto a conta $ mgh$ nunca
mudanças.
\end{solution}
