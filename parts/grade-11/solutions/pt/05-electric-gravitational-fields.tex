\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:6}

\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:7}

\begin{solution}{exo:g11:electric-gravitational-fields:8}
\emph{1.} $ E = \num{5.0e3} / \num{2.0e-2} = \qty{2.5e5}{V/m}$; \omterm{def:g10:relative-motion:frame}{em repouso}
$ qE = mg $, então
$ q = \num{2.0e-6} \times 9.81 / \num{2.5e5} \approx \qty{7.8e-11}{C}$.

\emph{2.} O \omterm{def:g10:inertia:force}{vigor} no grão positivo deve apontar para cima, então $\vect E $
aponta para cima: o \emph{mais baixo} placa é positiva.

\emph{3.} $ n = \num{7.85e-11} / \num{1.60e-19} \approx \num{4.9e8}$
cargas elementares.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:9}
Fora do segmento ambos campos aponte da mesma maneira; entre, eles
opor-se --- e o equilíbrio precisa do ponto mais próximo da fonte mais fraca $+Q $.
Com $ x $ a distância para $+Q $: $ kQ/x^2 = 4kQ/(0.30 - x)^2$, então
$0.30 - x = 2x $ e $ x = \qty{0.10}{m}$ de $+Q $.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:10}
$ F_e = ke^2/d^2 = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{5.3e-11})^2 \approx \qty{8.2e-8}{N}$;
$ F_g = G m_p m_e / d^2 = \num{6.67e-11} \times \num{1.67e-27} \times
\num{9.11e-31} / (\num{5.3e-11})^2 \approx \qty{3.6e-47}{N}$. Razão
$ F_e / F_g \approx \num{2.3e39}$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:11}
$ E = \sqrt{300^2 + 400^2} = \qty{500}{N/C}$, no
$\tan^{-1}(400/300) \approx \ang{53}$ norte do leste. Sobre $ q < 0$:
$ F = \num{2.0e-6} \times 500 = \qty{1.0e-3}{N}$, oposto ao campo ---

\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:12}
$ GM/(R+h)^2 = \tfrac12\,GM/R^2$ dá $ R + h = R\sqrt{2}$, então
$ h = (\sqrt{2} - 1)R \approx \qty{2.6e6}{m}$ (sobre \qty{2600}{km}). No
$ h = R $, $ d = 2R $: um quarto do valor da superfície.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:13}
\emph{1.} $ AM = \sqrt{3.0^2 + 4.0^2} = \qty{5.0}{cm}$; cada cobrança
cria $ E_1 = \num{8.99e9} \times \num{2.0e-9} / 0.050^2
\approx \qty{7.2e3}{N/C}$.

\emph{2.} Os componentes junto $ AB $ cancelar por simetria; cada campo
contribui $ E_1 \times 4/5$ ao longo da bissetriz, longe do segmento:
$ E = 2 \times \num{7.19e3} \times 0.80 \approx \qty{1.2e4}{N/C}$,
direcionado ao longo da bissetriz, longe de $[AB]$.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:14}
As cargas gratuitas da caixa movem-se sob o exterior campo: $+$ se acumula
um rosto, $-$ por outro lado, e essas cargas deslocadas criam uma
interno campo opondo-se ao externo. Eles continuam se movendo até o
total campo dentro é exatamente zero --- então equilíbrio. Ofertas de gravidade
apenas um \omterm{def:g10:signals-and-waves:signal}{sinal} de origem: cada missa \emph{adiciona} uma atração campo,
ninguém pode se opor a isso. Um escudo gravitacional precisaria de massa negativa, o que
não existe.
\end{solution}

\begin{solution}{exo:g11:electric-gravitational-fields:15}
\emph{1.} \omterm{def:g10:universal-gravitation:weight}{Peso} $ m\vect g $ (abaixo), \omterm{def:g11:circuits-and-power:voltage}{tensão} $\vect T $ (ao longo do
fio), elétrico \omterm{def:g10:inertia:force}{vigor} $ q\vect E $ (horizontal). \omterm{def:g10:relative-motion:frame}{Em repouso},
$ T\sin\theta = qE $ e $ T\cos\theta = mg $; dividir:
$\tan\theta = qE/(mg)$.

\emph{2.} $\tan\theta = \num{2.0e-7} \times \num{1.0e4} /
(\num{5.0e-4} \times 9.81) \approx 0.41$, então $\theta \approx \ang{22}$;
$ T = mg/\cos\theta \approx \qty{5.3e-3}{N}$.
\end{solution}

\begin{solution}{pb:g11:electric-gravitational-fields:1}
\textbf{1.} Uniforme, perpendicular às placas, apontando para baixo (de
o $+$ placa superior para o $-$ inferior); uniformidade significa \omterm{def:g10:inertia:force}{vigor}
na gota é o mesmo onde quer que ela flutue.
\textbf{2.} $ E = U/d = 800 / \num{8.0e-3} = \qty{1.0e5}{V/m}$.
\textbf{3.} A carga é negativa, então o \omterm{def:g10:inertia:force}{vigor} é \emph{oposto}
$\vect E $: para cima. Somente com o $+$ placa em cima faz $\vect E $ ponto
para baixo e o elétrico \omterm{def:g10:inertia:force}{vigor} lutar contra o \omterm{def:g10:universal-gravitation:weight}{peso}.
\textbf{4.} $ P = mg = \num{4.90e-15} \times 9.81 \approx
\qty{4.8e-14}{N}$.
\textbf{5.} $\abs{q}E = mg $.
\textbf{6.} $\abs{q} = mg/E = mgd/U $.
\textbf{7.} $\abs{q} = \num{4.81e-14} \times \num{8.0e-3} / 800
\approx \qty{4.8e-19}{C}$.
\textbf{8.} $ n = \num{4.8e-19} / \num{1.60e-19} = 3$ excesso de elétrons.
\textbf{9.} $ E $ cresce, então $\abs{q}E > mg $: a rede \omterm{def:g10:inertia:force}{vigor} aponta para cima.
\textbf{10.} De $\abs{q}\,U/d = mg $, $ U = mgd/\abs{q}$ --- inversamente
proporcional à carga.
\textbf{11.} $\abs{q} = mgd/U = \num{3.85e-16}/1200 \approx
\qty{3.2e-19}{C} = 2e $: abaixo de $3e $, a gota \emph{perdido} um
elétron.
\textbf{12.} $\abs{q} = \num{3.85e-16}/U $: \qty{3.2e-19}{C},
\qty{4.8e-19}{C}, \qty{6.4e-19}{C}, \qty{8.0e-19}{C},
\qty{9.6e-19}{C}.
\textbf{13.} Eles são $2e, 3e, 4e, 5e, 6e $ com
$ e = \qty{1.60e-19}{C}$.
\textbf{14.} Cada etapa muda $\abs{q}$ por exatamente
\qty{1.6e-19}{C}: um elétron pousa de cada vez.
\textbf{15.} Seria necessário $\abs{q} = \num{3.85e-16}/700 \approx
\qty{5.5e-19}{C} = 3.4\,e $ --- não um número inteiro de elétrons, então não
explosão de radiação pode produzi-lo.
\textbf{16.} A cobrança é \emph{quantizado}: vem em múltiplos inteiros
do \omterm{def:g11:fundamental-interactions:elementary}{carga elementar} $ e = \qty{1.60e-19}{C}$ --- a quantidade
Millikan mediu.
\textbf{17.} $ q \leftrightarrow m $, $ E \leftrightarrow g $, e
$\abs{q}E \leftrightarrow mg $: o \omterm{def:g10:universal-gravitation:weight}{peso} em si.
\textbf{18.} $ g_{\text{sphere}} = GM/d^2 = \num{6.67e-11} \times 1000 /
1.0^2 = \qty{6.7e-8}{N/kg}$; \omterm{def:g10:inertia:force}{vigor} na gota
$\num{4.90e-15} \times \num{6.7e-8} \approx \qty{3.3e-22}{N}$ --- sobre
$\num{1.5e8}$ vezes menor que seu \omterm{def:g10:universal-gravitation:weight}{peso}. Desesperado.
\textbf{19.} Porque $ G $ é tão pequeno, apenas uma massa do tamanho de um planeta acima
poderia rivalizar com a atração da Terra --- e nenhuma repulsão em massa. O enorme $ k $
e os dois sinais de carga permitem que uma placa de bancada puxe o planeta
em qualquer direção.
\textbf{20.} Um elétron: $\Delta\abs{q}/\abs{q} = e/3e \approx 33\%$;
um óleo molécula: $\Delta m/m = \num{5e-25} / \num{4.90e-15}
\approx \num{e-10}$. A escada de carga move o equilíbrio \omterm{def:g11:circuits-and-power:voltage}{tensão}
por centenas de \omterm{def:g11:circuits-and-power:voltage}{volts}; a escadaria de massa é invisível em qualquer \omterm{def:g11:circuits-and-power:voltage}{tensão}.
O equilíbrio, portanto, resolve o \omterm{def:g11:fundamental-interactions:ladder}{átomo} de eletricidade ---
$ e = \qty{1.60e-19}{C}$ --- enquanto o \omterm{def:g11:fundamental-interactions:ladder}{moléculas} de óleo borrão em um
contínuo.
\end{solution}
