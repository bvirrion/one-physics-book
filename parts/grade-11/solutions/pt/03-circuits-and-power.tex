\begin{solution}{exo:g11:circuits-and-power:1}

encargos: $ q/e = 1.08\times10^{4}/1.6\times10^{-19} \approx \num{6.8e22}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:2}
$ I = U/R = 12/220 = \qty{0.055}{A} = \qty{55}{mA}$;
$ R = U/I = 230/0.50 = \qty{460}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:3}
Série: $120 + 60 = \qty{180}{\ohm}$. Paralelo: $\frac{120 \times 60}{180}
= \qty{40}{\ohm}$ --- menor que ambos: o mesmo \omterm{def:g11:circuits-and-power:voltage}{tensão}, mas a atual
obtém um \omterm{def:g10:orders-of-magnitude:unit}{segundo} caminho, então mais corrente total e menos \omterm{def:g11:circuits-and-power:resistance}{resistência}.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:4}
$ P = UI = 230 \times 8.7 = \qty{2.0}{kW}$. Em $ t = \qty{240}{s}$:
$ E = Pt = 2000 \times 240 = \qty{4.8e5}{J} = 4.8\times10^{5}/3.6
\times10^{6} \approx \qty{0.13}{kW.h}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:5}
$ E = 60 \times 5.0 = \qty{300}{W.h} = \qty{0.30}{kW.h} = 0.30 \times
3.6\times10^{6} = \qty{1.1e6}{J}$. Custo: $0.30 \times 0.25 = 0.075$
euros por dia.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:6}
\emph{1.} $ R = U^2/P = 230^2/1200 = \qty{44}{\ohm}$;
$ I = P/U = 1200/230 = \qty{5.2}{A}$.

\emph{2.} $ P' = U'^2/R = 115^2/44.1 = \qty{300}{W}$: um quarto, não
metade, porque reduzir pela metade $ U $ também metades $ I $, e $ P = UI \propto U^2$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:7}
$ R_{\text{s}} = \qty{60}{\ohm}$, então $ I = 12/60 = \qty{0.20}{A}$.
$ U_1 = R_1 I = \qty{8.0}{V}$, $ U_2 = \qty{4.0}{V}$ (soma \qty{12}{V});
$ P_1 = U_1 I = \qty{1.6}{W}$, $ P_2 = \qty{0.8}{W}$: total \qty{2.4}{W}
$= UI = 12 \times 0.20$, como deveria.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:8}
$ I_1 = 12/60 = \qty{0.20}{A}$, $ I_2 = 12/30 = \qty{0.40}{A}$, total
$ I = \qty{0.60}{A}$. Equivalente: $ R = U/I = 12/0.60 = \qty{20}{\ohm}$,
e de fato $\frac{60 \times 30}{60 + 30} = \qty{20}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:9}
$\mathcal{E} - 0.50\,r = 5.5$ e $\mathcal{E} - 2.0\,r = 4.0$;
subtraindo, $1.5\,r = 1.5$, então $ r = \qty{1.0}{\ohm}$ e $\mathcal{E} =
\qty{6.0}{V}$. Curto-circuito: $\mathcal{E}/r = \qty{6.0}{A}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:10}
Chaleira $2200/230 = \qty{9.6}{A}$, torradeira $1000/230 = \qty{4.3}{A}$:
total \qty{13.9}{A} $< \qty{16}{A}$, o fusível segura. Aquecedor $1500/230 =
\qty{6.5}{A}$: total \qty{20.4}{A} $> \qty{16}{A}$, isso sopra
(equivalentemente, $\qty{4.7}{kW} > 230 \times 16 \approx \qty{3.7}{kW}$).
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:11}
No \qty{200}{V}: $ I = 10^4/200 = \qty{50}{A}$, perda $ RI^2 = 2.0 \times
50^2 = \qty{5.0}{kW}$ --- metade da entrega. No \qty{4.0}{kV}: $ I =
\qty{2.5}{A}$, perda $2.0 \times 2.5^2 = \qty{12.5}{W}$. Razão $400 =
20^2$: vinte vezes o \omterm{def:g11:circuits-and-power:voltage}{tensão} significa vinte vezes menos corrente, e
a perda vai como $ I^2$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:12}
Quatro layouts: todos em série, $3 \times 60 = \qty{180}{\ohm}$; tudo dentro
paralelo, $60/3 = \qty{20}{\ohm}$; um em série com dois em paralelo,
$60 + 30 = \qty{90}{\ohm}$; um em paralelo com dois em série,
$\frac{60 \times 120}{180} = \qty{40}{\ohm}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:13}
\emph{1.} $ I = \mathcal{E}/(R + r) = 4.5/7.5 = \qty{0.60}{A}$;
$ U = RI = \qty{3.6}{V}$; útil $ P = UI = \qty{2.2}{W}$; perdido $ rI^2 =
1.5 \times 0.60^2 = \qty{0.54}{W}$; $\eta = U/\mathcal{E} = 3.6/4.5 =
0.80$.

\emph{2.} Pela dica, $ P = \mathcal{E}^2 \big/ \big[(R - r)^2/R +
4r\big]$; o colchete é o menor ($= 4r $) exatamente quando $ R = r $, então
$ P_{\max} = \mathcal{E}^2/(4r) = 4.5^2/6.0 = \qty{3.4}{W}$.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:14}
Aquecer: $ Q = 1.0 \times 4180 \times (100 - 20) = \qty{3.3e5}{J}$.
Elétrica: $ E = Q/0.90 = \qty{3.7e5}{J}$. Tempo: $ t = E/P =
3.7\times10^{5}/2200 \approx \qty{170}{s}$, menos de três minutos. Custo:
$3.7\times10^{5}/3.6\times10^{6} \approx \qty{0.10}{kW.h}$, sobre $2.6$
centavos.
\end{solution}

\begin{solution}{exo:g11:circuits-and-power:15}
\emph{1.} $ q = 60 \times 3600 = \qty{2.2e5}{C}$; $ E \approx \mathcal{E}q
= 12.7 \times 2.16\times10^{5} = \qty{2.7e6}{J} \approx \qty{0.76}{kW.h}$.

\emph{2.} $ U = 12.7 - 0.020 \times 150 = \qty{9.7}{V}$; entregue
$ UI = \qty{1.5}{kW}$; perdido $ rI^2 = 0.020 \times 150^2 = \qty{450}{W}$.

\emph{3.} Os faróis ficam sobre os terminais: durante a partida, o
terminal \omterm{def:g11:circuits-and-power:voltage}{tensão} cai de $12.7$ para $\qty{9.7}{V}$, então eles recebem
menos poder e escuro.
\end{solution}

\begin{solution}{pb:g11:circuits-and-power:1}
\textbf{1.} Cada \omterm{def:g10:orders-of-magnitude:unit}{segundo}, $ E_p = mgh = 5.0\times10^{4} \times 9.81 \times
200 = \qty{9.8e7}{J}$: disponível poder \qty{98}{MW}.

\textbf{2.} $0.90 \times 98 = \qty{88}{MW}$.

\textbf{3.} $ I = P/U = 8.8\times10^{7}/2.0\times10^{4} = \qty{4.4e3}{A}$.

\textbf{4.} $ R_\ell I^2 = 32 \times (4.4\times10^{3})^2 \approx
\qty{6.2e8}{W} = \qty{620}{MW}$ --- sete vezes a produção da plant:
transmitindo em \qty{20}{kV} é impossível.

\textbf{5.} Com $ P = UI $ fixo, apenas aumentando $ U $ encolhe $ I $: um
transformador.

\textbf{6.} $ I = 8.8\times10^{7}/4.0\times10^{5} = \qty{220}{A}$.

\textbf{7.} $ R_\ell I^2 = 32 \times 220^2 \approx \qty{1.6}{MW}$.

\textbf{8.} $1.6/88 \approx 0.018$: sobre $1.8\%$.

\textbf{9.} Fração $= R_\ell I^2/P = R_\ell (P/U)^2/P = R_\ell P/U^2
\propto 1/U^2$: de \qty{20}{kV} para \qty{400}{kV} cai
$20^2 = 400$, e de fato $\qty{620}{MW}/400 \approx \qty{1.6}{MW}$.

\textbf{10.} $ R_\ell I = 32 \times 220 \approx \qty{7.1}{kV}$: $1.8\%$
de \qty{400}{kV}, a mesma fração da questão 8, já que
$ R_\ell I/U = R_\ell I^2/UI $.

\textbf{11.} $\qty{1}{kW.h} = \qty{3.6e6}{J}$: o \omterm{def:g10:orders-of-magnitude:unit}{medidor} multiplica o
poder desenhado no momento em que é desenhado e soma.

\textbf{12.} $ I = 900/230 = \qty{3.9}{A}$;
$ R = U^2/P = 230^2/900 \approx \qty{59}{\ohm}$.

\textbf{13.} $ E = 900 \times 180 = \qty{1.6e5}{J} = \qty{0.045}{kW.h}$:
sobre $1.1$ centavos.

\textbf{14.} A física de $ RI^2$ é idêntico; apenas o propósito
difere --- na linha o calor é indesejado, na torradeira o calor
\emph{é} o produto.

\textbf{15.} $3.9 + 9.6 = \qty{13.5}{A} < \qty{16}{A}$: segura. Aquecedor
$2000/230 = \qty{8.7}{A}$: total $\qty{22.2}{A} > \qty{16}{A}$, o fusível
golpes.

\textbf{16.} $0.05 \times 13.5^2 = \qty{9.1}{W}$: $9.1/3100 \approx
0.3\%$ do entregue poder --- insignificante em fios curtos.

\textbf{17.} $\eta = 0.90 \times 0.99 \times 0.982 \times 0.99 \times
0.997 \approx 0.86$.

\textbf{18.} Um torrado \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} precisa $3.6\times10^{6}/0.86
\approx \qty{4.2e6}{J}$ a montante: $ m = E/(gh) = 4.2\times10^{6}/(9.81
\times 200) \approx \qty{2.1e3}{kg}$, sobre \qty{2.1}{m^3} de água.

\textbf{19.} $ R_\ell P/U^2 = 32 \times 8.8\times10^{7}/(2.0\times
10^{4})^2 \approx 7$. Uma ``fração'' acima $1$ significa que a linha seria
dissipa mais do que carrega: a entrega não pode acontecer naquele
\omterm{def:g11:circuits-and-power:voltage}{tensão} de forma alguma.

\textbf{20.} Um torrado \omterm{def:g10:energy-conservation:kwh}{quilowatt-hora} custa à montanha cerca de dois
toneladas --- \qty{2.1}{m^3} --- de água caindo \qty{200}{m}, e alguns
$14\%$ do energia morre no caminho, principalmente em turbinas e transformadores
(sob $2\%$ em \qty{1000}{km} de linha). O único truque que faz o
jornada possível é o transformador: avançar para \qty{400}{kV}
divide a corrente por $20$ e a perda de Joule por $400$.
\end{solution}
