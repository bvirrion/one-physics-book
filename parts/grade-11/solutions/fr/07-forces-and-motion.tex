\section*{Chapitre \ref{ch:g11:forces-and-motion} --- Forces et mouvement}

\begin{solution}{exo:g11:forces-and-motion:1}
(a)~Vitesse constante~: \omterm{def:g11:forces-and-motion:netforce}{force résultante}
nulle. (b)~La vitesse change~: non nulle, opposée au mouvement.
(c)~La direction change (la vitesse est un vecteur)~: non nulle, vers
l'intérieur du virage. (d)~Vitesse constante~: nulle ---
\omterm{def:g10:universal-gravitation:weight}{poids} et traînée
\omterm{def:g10:inertia:compensate}{se compensent}.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:2}
$d_1/d_2 = 1.0/3.0$ avec $d_1 + d_2 = \qty{0.80}{m}$~:
$d_1 = \qty{0.20}{m}$ depuis la balle de \qty{3.0}{kg}. Masses égales~:
le milieu, \qty{0.40}{m} de chacune.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:3}
Même $F\Delta t$, triple masse~: $\Delta v$ trois fois plus petit
chargé. Pour égaler le $\Delta v$ du chariot vide, pousser trois fois
plus fort (ou trois fois plus longtemps).
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:4}
$\Delta \vect v$~: \qty{2}{m/s}, dirigé \emph{en arrière} (de
\num{10} à \qty{8}{m/s}). La
\omterm{def:g11:forces-and-motion:netforce}{force résultante} pointe le même sens~:
opposé à $\vect v$ --- cas~2, ralentissement.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:5}
(a)~Statique, opposé à votre poussée (pas encore de glissement).
(b)~Cinétique, opposé au glissement. (c)~Statique, pointant
\emph{vers le haut} de la pente --- contre le glissement que la
gravité tente de démarrer.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:6}
$\Delta v \propto F\Delta t/m$~: (a)~\qty{8.0}{m/s}~; (b)~\qty{2.0}{m/s}~;
(c)~\qty{2.0}{m/s}~; (d)~$4.0 \times \frac{3 \times 2}{6} =
\qty{4.0}{m/s}$ --- les trois changements s'annulent.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:7}
$v' = \sqrt{8.0^2 + 6.0^2} = \qty{10.0}{m/s}$, à $\tan\theta =
6.0/8.0$, c.-à-d. $\theta \approx 37^\circ$ nord de l'est. Les
vitesses s'ajoutent comme des \emph{vecteurs}~: des contributions
perpendiculaires se combinent par Pythagore, non par addition des
grandeurs.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:8}
Fixé d'avance~: la masse de la balle et son $\Delta v$ (vitesse
d'arrivée à zéro), d'où le produit $F\Delta t$. Étendre $\Delta t$ de
\qty{0.02}{s} à \qty{0.08}{s} divise la
\omterm{def:g10:inertia:force}{force} moyenne par $4$.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:9}
Le \omterm{def:g10:inertia:inventory}{frottement} statique de la piste sur la
chaussure~: le pied pousse vers l'arrière sur le sol, et la
\omterm{def:g10:inertia:contact}{force de contact} mutuelle pousse le
sprinter en avant. Les pointes élèvent le maximum de
\omterm{def:g10:inertia:inventory}{frottement} statique (pas de glissement à
plein effort). Sur glace sans frottement aucune
\omterm{def:g10:inertia:force}{force} horizontale externe n'existe, donc
la vitesse du \omterm{def:g11:forces-and-motion:com}{centre de masse} ne peut
changer --- pas de départ, pas de marche.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:10}
Le point de contact d'une roue qui roule ne glisse pas~:
\omterm{def:g10:inertia:inventory}{frottement} statique, plus fort que le
\omterm{def:g11:forces-and-motion:friction}{frottement cinétique} d'un dérapage
(\cref{def:g11:forces-and-motion:friction}) --- arrêt plus court. Le
\omterm{def:g11:forces-and-motion:friction}{frottement cinétique} d'un pneu qui
dérape pointe contre le glissement, quel que soit l'angle des roues~:
tourner le volant ne change rien, la direction est perdue.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:11}
La \omterm{def:g10:inertia:inventory}{tension} du fil, pointant vers
l'intérieur vers l'athlète. Après le lâcher la balle part en ligne
droite le long de la tangente, à la vitesse qu'elle avait (inertie).
Le «~fil~» de la Lune est l'attraction gravitationnelle de la Terre~;
coupez la gravité et elle part le long de sa tangente, droite et
uniforme, pour toujours.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:12}
Depuis le sommet~: $x = \dfrac{1.0 \times 0.60 + 2.0 \times 1.20}{3.0}
= \qty{1.00}{m}$. Distances aux deux centres de parties~: \qty{0.40}{m}
(bâton) et \qty{0.20}{m} (tête), rapport $2 : 1$ --- l'inverse du
rapport des masses $1.0 : 2.0$, comme l'exige la
\cref{def:g11:forces-and-motion:com}.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:13}
\emph{1.} Même $F$ et $\Delta v$, double $m$~: la camionnette a besoin
du double de temps. \emph{2.} Même
\omterm{def:g10:relative-motion:speed}{vitesse moyenne} pendant deux fois le
temps~: deux fois la distance. \emph{3.} Double $\Delta v$ au même $F$
et $m$~: deux fois le temps~; mais la
\omterm{def:g10:relative-motion:speed}{vitesse moyenne} double aussi, donc la
distance est multipliée par $4$. Doubler votre vitesse quadruple votre
distance de freinage.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:14}
\emph{1.} $T = 27.3 \times \qty{86400}{s} \approx \qty{2.36e6}{s}$, donc
$v = \dfrac{2\pi \times \num{3.84e8}}{\num{2.36e6}} \approx
\qty{1.0e3}{m/s} \approx \qty{1}{km/s}$. \emph{2.} Vers la Terre
(vers l'intérieur)~; l'attraction gravitationnelle de la Terre.
\emph{3.} La Lune \emph{tombe}~: son changement de vitesse pointe vers
la Terre à chaque instant. Sa vitesse tangentielle assure seulement
qu'elle continue de rater --- elle tombe \emph{autour} de la Terre
plutôt que dedans.
\end{solution}

\begin{solution}{exo:g11:forces-and-motion:15}
Même $m$ et $\Delta v$, donc $F\Delta t$ est fixé~: $\Delta t$
cinquante fois plus long sur mousse signifie une
\omterm{def:g10:inertia:force}{force} moyenne cinquante fois plus petite.
Plier les genoux étire le $\Delta t$ de l'atterrissage d'un à-coup à
une longue flexion~; un tapis de réception est, mécaniquement, fait de
millisecondes supplémentaires.
\end{solution}

\begin{solution}{pb:g11:forces-and-motion:1}
\textbf{1.} $50/3.6 \approx \qty{13.9}{m/s}$ (appelons-le \qty{14}{m/s}).
\textbf{2.} $\Delta v \approx \qty{14}{m/s}$~: l'occupant se déplace à
\qty{14}{m/s} avant et à zéro après --- les deux extrémités sont
fixées par l'accident, donc aucun dispositif ne peut toucher
$\Delta v$.
\textbf{3.} $\Delta v \propto F\Delta t/m$
(\cref{thm:g11:forces-and-motion:secondlaw}) avec $\Delta v$ et $m$
fixés force $F\Delta t$ à être fixé. Le seul cadran du concepteur est
$\Delta t$.
\textbf{4.} $0.150/0.050 = 3$~: la zone de déformation divise la
\omterm{def:g10:inertia:force}{force} moyenne par $3$.
\textbf{5.} Vers l'arrière, opposé au mouvement --- la vitesse se
contracte de \qty{14}{m/s} à zéro.
\textbf{6.} L'avant qui se plie est une machine pour faire durer
l'arrêt plus longtemps~: le métal qui se déforme achète des
millisecondes.
\textbf{7.} Même $\Delta v$ et $m$, $\Delta t$ trois fois plus court~:
le mannequin d'époque prend trois fois la
\omterm{def:g10:inertia:force}{force}.
\textbf{8.} \omterm{def:g10:relative-motion:speed}{Vitesse moyenne}
$\approx \qty{6.9}{m/s}$~; distance $\approx 6.9 \times 0.150 \approx
\qty{1.0}{m}$ --- juste le \omterm{def:g10:orders-of-magnitude:unit}{mètre} de
voiture déformable qu'une zone de déformation réelle fournit.
\textbf{9.} Un nez de \qty{10}{m} est impraticable et
imparkable, et sa propre masse aggraverait chaque accident. La cellule
passagers ne doit pas se déformer~: elle préserve l'espace de survie.
\textbf{10.} «~\dots donc vous devez choisir $\Delta t$.~»
\textbf{11.} $0.150/0.005 = 30$~: l'arrêt sur le pare-brise est trente
fois plus violent que celui ceinturé.
\textbf{12.} Une ceinture qui s'étire un peu allonge un peu plus le
propre $\Delta t$ de l'occupant (et répartit la
\omterm{def:g10:inertia:force}{force})~; un harnais d'acier imposerait
l'arrêt le plus court de la carrosserie au passager le plus souple.
\textbf{13.} $0.080/0.010 = 8$~: l'airbag divise la
\omterm{def:g10:inertia:force}{force} sur la tête par $8$.
\textbf{14.} Même $\Delta v$ et $\Delta t$~: la
\omterm{def:g10:inertia:force}{force} scale avec la masse,
$70/20 = 3.5$ fois plus grande pour l'adulte.
\textbf{15.} Changement requis~: $14/0.150 \approx \qty{93}{m/s}$
chaque seconde --- environ $9.5$ fois ce que produit la gravité
(\qty{9.8}{m/s} par seconde). Les bras doivent tirer environ $9.5$
fois le \omterm{def:g10:universal-gravitation:weight}{poids} du bébé~: comme tenir
une charge de \qty{95}{kg}. Aucun bras ne le peut~; le bébé vole en
avant.
\textbf{16.} La ceinture d'un adulte route la
\omterm{def:g10:inertia:force}{force} vers hanches et poitrine~; sur un
enfant elle chargerait ventre et cou. Le siège enfant applique la
(plus petite) \omterm{def:g10:inertia:force}{force} sur des parties
robustes du corps et son rembourrage étire encore $\Delta t$.
\textbf{17.} Si la cellule se déforme, les occupants perdent leur
espace de survie et sont frappés par la structure elle-même. Rien ne
doit frapper la cellule rigide directement~: les arrêts des occupants
sont gérés par ceinture et airbag (long $\Delta t$), tandis que la
cellule ancre les extrémités qui se déforment.
\textbf{18.} Par symétrie les deux voitures s'arrêtent au plan de
contact~: chaque conducteur subit $\Delta v \approx \qty{14}{m/s}$ sur
un arrêt de zone de déformation --- essentiellement l'essai contre le
mur, non «~le double de l'accident~».
\textbf{19.} Les \omterm{def:g10:inertia:force}{forces} sont égales et
opposées, mais $\Delta v \propto F\Delta t/m$~: la petite voiture, avec
la petite $m$, subit le grand changement de vitesse.
\textbf{20.} Zone de déformation, ceinture qui s'étire, airbag,
rembourrage de siège enfant --- chacun d'eux étire la même quantité~:
$\Delta t$.
\textbf{21.} L'accident fixe la masse et le changement de vitesse,
d'où le produit \omterm{def:g10:inertia:force}{force} $\times$ temps~: une
collision douce est une collision longue --- étirez le temps, et la
\omterm{def:g10:inertia:force}{force} tombe dans le même rapport.
\end{solution}
