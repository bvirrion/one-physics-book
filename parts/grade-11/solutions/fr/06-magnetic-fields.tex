\section*{Chapitre \ref{ch:g11:magnetic-fields} --- Magnétisme et champs magnétiques}

\begin{solution}{exo:g11:magnetic-fields:1}
Chaque morceau est un \omterm{def:g11:magnetic-fields:magnet}{aimant} complet avec un
pôle nord et un pôle sud~: un nouveau pôle apparaît sur chaque face de
coupe. Les faces fraîches sont un N et un S, donc elles s'attirent ---
les morceaux tentent de se réassembler. Aucun schéma n'isole un pôle~:
chaque coupe crée une paire.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:2}
(a)~Faux~: le \omterm{def:g11:electric-gravitational-fields:field}{champ} a une seule
direction en chaque point. (b)~Vrai. (c)~Vrai~: chaque ligne se ferme à
travers le corps de l'\omterm{def:g11:magnetic-fields:magnet}{aimant}. (d)~Faux~:
l'aiguille se place \emph{le long} de la ligne, tangente à elle.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:3}
$B = \num{2e-7} \times 10 / 0.020 = \qty{1.0e-4}{T}$ --- deux fois le
\omterm{def:g11:electric-gravitational-fields:field}{champ} terrestre.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:4}
$n = 800/0.40 = \qty{2000}{m^{-1}}$~;
$B = \num{1.26e-6} \times 2000 \times 1.5 \approx \qty{3.8}{mT}$.
$B \propto I$~: doubler le courant donne \qty{7.5}{mT}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:5}
$F = ILB = 5.0 \times 0.10 \times 0.20 = \qty{0.10}{N}$ --- le
\omterm{def:g10:universal-gravitation:weight}{poids} de $m = F/g = 0.10/9.81 \approx
\qty{10}{g}$.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:6}
\emph{1.} Sa piste pointe $\ang{8}$ ouest du nord vrai~:
$2000 \times \sin\ang{8} \approx \qty{2.8e2}{m}$ vers l'ouest.

\emph{2.} Gouverner $\ang{8}$ \emph{est} du nord de l'aiguille.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:7}
$d = \mu_0 I / (2\pi B) = \num{2e-7} \times 20 / \num{5.0e-5}
= \qty{8.0}{cm}$. Dans le décimètre d'un tel câble la boussole lit le
fil, non la planète --- la tenir loin des conducteurs sous tension.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:8}
$n = B/(\mu_0 I) = 0.010 / (\num{1.26e-6} \times 4.0)
\approx \qty{2.0e3}{m^{-1}}$~; sur \qty{25}{cm},
$N = 1989 \times 0.25 \approx 500$ spires.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:9}
$F = 10 \times 0.12 \times 0.50 = \qty{0.60}{N}$~;
$a = F/m = 0.60/0.020 = \qty{30}{m/s^2}$ --- environ $3g$~: la tige
saute vraiment.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:10}
Équilibre~: $ILB = mg$, donc
$I = \num{5.0e-3} \times 9.81 / (0.20 \times 0.10) \approx
\qty{2.5}{A}$. La \omterm{def:g10:inertia:force}{force} doit pointer vers le
haut~; avec $\vect B$ horizontal et perpendiculaire au fil, la main
droite à plat (paume vers le haut) fixe le sens de courant requis.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:11}
Main droite à plat~: (a)~vers le haut de la page~; (b)~vers la
droite~; (c)~aucune \omterm{def:g10:inertia:force}{force} --- le fil est
parallèle au \omterm{def:g11:electric-gravitational-fields:field}{champ}.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:12}
\emph{1.} $B = \num{2e-7} \times 5.0 / 0.030 \approx
\qty{33}{\micro T}$~; les lignes encerclent le fil, donc directement
sous lui le \omterm{def:g11:electric-gravitational-fields:field}{champ} est horizontal,
perpendiculaire au fil~: est--ouest.

\emph{2.} L'aiguille suit la somme vectorielle~: $\tan\theta = 33/20 =
1.67$, donc $\theta \approx \ang{59}$ hors du nord.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:13}
\emph{1.} $I = U/R = 12/2.0 = \qty{6.0}{A}$~;
$n = 400/0.20 = \qty{2000}{m^{-1}}$~;
$B_0 = \num{1.26e-6} \times 2000 \times 6.0 \approx \qty{15}{mT}$~;
avec le noyau, $100 B_0 \approx \qty{1.5}{T}$.

\emph{2.} $P = UI = 12 \times 6.0 = \qty{72}{W}$.

\emph{3.} Pas de courant, pas de
\omterm{def:g11:electric-gravitational-fields:field}{champ} --- et le fer doux ne reste
pas aimanté~: la charge se libère instantanément, par conception.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:14}
\emph{1.} $F = NILB = 50 \times 2.0 \times 0.050 \times 0.30 =
\qty{1.5}{N}$ de chaque côté. Le courant court en sens opposés dans les
deux côtés, donc les \omterm{def:g10:inertia:force}{forces} sont opposées~:
un couple qui fait tourner la boucle.

\emph{2.} Lorsqu'ils sont parallèles à $\vect B$, un fil ne ressent
aucune \omterm{prop:g11:magnetic-fields:laplace}{force de Laplace}.

\emph{3.} Après un demi-tour le couple s'inverserait et déferait la
rotation~; le \omterm{rem:g11:magnetic-fields:motor}{collecteur} inverse le courant à
chaque demi-tour pour que la boucle continue de tourner dans le même
sens.
\end{solution}

\begin{solution}{exo:g11:magnetic-fields:15}
\emph{1.} $d = \num{2e-7} \times 100 / 1 = \qty{2.0e-5}{m} =
\qty{20}{\micro m}$ --- profondément \emph{à l'intérieur} du cuivre
millimétrique. Aucun point accessible près d'un fil seul n'atteint
\qty{1}{T}~: les forts
\omterm{def:g11:electric-gravitational-fields:field}{champs} empilent des milliers de
spires (un \omterm{def:g11:magnetic-fields:solenoid}{solénoïde}), plus du fer ou des
supraconducteurs.

\emph{2.} $\num{e8} / \num{5e-5} = \num{2e12}$~: environ douze
\omterm{def:g10:energy-conservation:power}{puissances} de dix.
\end{solution}

\begin{solution}{pb:g11:magnetic-fields:1}
\textbf{1.} L'aiguille est elle-même un petit
\omterm{def:g11:magnetic-fields:magnet}{aimant}~; le
\omterm{def:g11:electric-gravitational-fields:field}{champ} terrestre l'aligne, extrémité
nord le long de la composante horizontale de $\vect B$, c.-à-d. vers le
nord magnétique.

\textbf{2.} $B = \sqrt{20^2 + 44^2} = \sqrt{2336} \approx
\qty{48}{\micro T}$, incliné de $\tan^{-1}(44/20) \approx \ang{66}$ sous
l'horizontale.

\textbf{3.} La piste court $\ang{6}$ ouest du nord vrai~:
$30 \times \sin\ang{6} \approx \qty{3.1}{km}$ à l'ouest.

\textbf{4.} Les erreurs s'ajoutent~: $\ang{10}$, donc
$30 \times \sin\ang{10} \approx \qty{5.2}{km}$ --- cinq kilomètres pour
dix degrés.

\textbf{5.} La composante horizontale~: près du pôle le
\omterm{def:g11:electric-gravitational-fields:field}{champ} est presque vertical, la
partie horizontale se contracte vers zéro, et l'aiguille n'a plus rien
sur quoi s'aligner.

\textbf{6.} $n = 600/0.30 = \qty{2000}{m^{-1}}$.

\textbf{7.} $I = U/R = 12/1.5 = \qty{8.0}{A}$.

\textbf{8.} $B_0 = \mu_0 n I = \num{1.26e-6} \times 2000 \times 8.0
\approx \qty{20}{mT}$.

\textbf{9.} $50 \times \qty{20}{mT} = \qty{1.0}{T}$ --- deux fois le
\omterm{def:g11:electric-gravitational-fields:field}{champ} à un pôle de néodyme, sur une
face bien plus grande.

\textbf{10.} $P = UI = 12 \times 8.0 = \qty{96}{W}$~; en
\qty{10}{\minute}~: $E = 96 \times 600 \approx \qty{58}{kJ}$.

\textbf{11.} Parce qu'il lâche~: ouvrir l'interrupteur, le
\omterm{def:g11:electric-gravitational-fields:field}{champ} meurt, la voiture tombe
exactement où l'on veut. Un
\omterm{def:g11:magnetic-fields:magnet}{aimant} permanent saisit pour toujours.

\textbf{12.} $F = NILB = 100 \times 3.0 \times 0.040 \times 0.25 =
\qty{3.0}{N}$.

\textbf{13.} Un couple~: appliquées de part et d'autre de l'axe, les
deux \omterm{def:g10:inertia:force}{forces} opposées font toutes deux
tourner la boucle dans le même sens. Des
\omterm{def:g10:inertia:force}{forces} égales \emph{parallèles} ne feraient
que pousser la boucle de côté, non la faire tourner.

\textbf{14.} Aucune~: un fil parallèle au
\omterm{def:g11:electric-gravitational-fields:field}{champ} ne ressent aucune
\omterm{prop:g11:magnetic-fields:laplace}{force de Laplace}.

\textbf{15.} Il inverse le courant dans la boucle à chaque demi-tour,
juste quand le couple changerait de signe --- donc le couple entraîne
toujours la même rotation.

\textbf{16.} Doubler $N$, ou doubler $I$, ou doubler $B$~: la paire de
\omterm{def:g10:inertia:force}{forces} $F = NILB$ est proportionnelle à
chacun.

\textbf{17.} $1.5 / \num{5.0e-5} = \num{3.0e4}$~: trente mille Terres.

\textbf{18.} $n = B/(\mu_0 I) = 1.5 / (\num{1.26e-6} \times 500)
\approx \qty{2.4e3}{m^{-1}}$ --- et chacune de ces spires porte
\qty{500}{A}.

\textbf{19.} $P = RI^2 = 0.20 \times 500^2 = \qty{50}{kW}$ --- la
chauffe d'un quartier. L'enroulement réel est supraconducteur~: zéro
\omterm{def:g11:circuits-and-power:resistance}{résistance}, zéro dissipation, pourvu qu'il
soit maintenu à quelques degrés au-dessus du zéro absolu.

\textbf{20.} Boussole $\num{5e-5}$, grue $1$, IRM $1.5$~: le navire
siège cinq échelons sous les deux machines, qui partagent une
décade --- entre l'interstellaire $\num{e-10}$ et l'étoile à neutrons
$\num{e8}$. Rien n'a changé que les
\omterm{def:g11:circuits-and-power:current}{ampères} (et les spires, le fer et le
supraconducteur qui les multiplient)~: un seul
\omterm{def:g11:electric-gravitational-fields:field}{champ}, trois métiers.
\end{solution}
