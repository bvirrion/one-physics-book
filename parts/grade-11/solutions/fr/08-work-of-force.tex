\section*{Chapitre \ref{ch:g11:work-of-force} --- Travail d'une force}

\begin{solution}{exo:g11:work-of-force:1}
$W = 40 \times 50 \times \cos\theta$~: $+\qty{2000}{J}$~;
$+\qty{1000}{J}$~; $0$~; $-\qty{1000}{J}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:2}
$mgh = 12 \times 9.81 \times 2.5 = \qty{294}{J}$~: abaissée,
$+\qty{294}{J}$ (\omterm{def:g10:universal-gravitation:weight}{poids} le long de la
descente)~; relevée, $-\qty{294}{J}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:3}
(a)~\omterm{def:g11:work-of-force:sign}{moteur}~; (b)~\omterm{def:g11:work-of-force:sign}{résistant}~;
(c)~nul (perpendiculaire)~; (d)~\omterm{def:g11:work-of-force:sign}{résistant}~;
(e)~nul --- la \omterm{def:g10:inertia:inventory}{tension} du fil reste
perpendiculaire au mouvement du bob.
\end{solution}

\begin{solution}{exo:g11:work-of-force:4}
Vitesse constante~: \omterm{def:g10:inertia:force}{force} de levage
$= mg = \qty{5.89e3}{N}$, donc
$W_{\text{levage}} = 5886 \times 25 \approx +\qty{147}{kJ}$ et
$W(\vect P) = -\qty{147}{kJ}$. Somme nulle~: à vitesse constante les
\omterm{def:g10:inertia:force}{forces} s'équilibrent
(\cref{ch:g11:forces-and-motion}), donc leurs
\omterm{def:g11:work-of-force:work}{travaux} s'annulent.
\end{solution}

\begin{solution}{exo:g11:work-of-force:5}
$P = \num{9.0e4}/60 = \qty{1.5}{kW}$~; $1500/736 \approx \qty{2.0}{hp}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:6}
Corde~: $120 \times 200 \times \cos\qty{30}{\degree} \approx
+\qty{20.8}{kJ}$~; \omterm{def:g10:inertia:inventory}{frottement}~:
$-45 \times 200 = -\qty{9.0}{kJ}$~;
\omterm{def:g10:universal-gravitation:weight}{poids} et
\omterm{def:g10:inertia:force}{force} normale, perpendiculaires~: $0$.
Total $+\qty{11.8}{kJ}$~: le traîneau gagne de la vitesse.
\end{solution}

\begin{solution}{exo:g11:work-of-force:7}
$v = 25/3.6 = \qty{6.94}{m/s}$~; $P = Fv = 18 \times 6.94 \approx
\qty{125}{W}$. En une heure, $W = P\,\Delta t = 125 \times 3600 =
\qty{450}{kJ}$ (équivalemment $18 \times \qty{25000}{m}$).
\end{solution}

\begin{solution}{exo:g11:work-of-force:8}
\omterm{def:g10:universal-gravitation:weight}{Poids}, les deux routes~:
$+65 \times 9.81 \times 120 \approx +\qty{76.5}{kJ}$ (descente seule).
\omterm{def:g10:inertia:inventory}{Frottement}~: $-30 \times 800 =
-\qty{24}{kJ}$ sur la pente, $-30 \times 1500 = -\qty{45}{kJ}$ sur la
piste. Seul le \omterm{def:g10:inertia:inventory}{frottement} se soucie de la
route.
\end{solution}

\begin{solution}{exo:g11:work-of-force:9}
Montée $h = 4.0\sin\qty{25}{\degree} = \qty{1.69}{m}$~:
$W(\vect P) = -20 \times 9.81 \times 1.69 \approx -\qty{332}{J}$~;
$W_f = -30 \times 4.0 = -\qty{120}{J}$. Vitesse constante~: travail
total nul, donc $W_{\text{poussée}} = +\qty{452}{J}$ et
$F = 452/4.0 \approx \qty{113}{N}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:10}
$v = 130/3.6 = \qty{36.1}{m/s}$~; vitesse constante, donc
$F = P/v = \num{5.0e4}/36.1 \approx \qty{1.4}{kN}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:11}
$P = Fv\cos\theta = 800 \times 1.0 \times \cos\qty{20}{\degree}
\approx \qty{752}{W}$~; en
\omterm{def:g11:work-of-force:power}{chevaux-vapeur}, $752/736 \approx \num{1.02}$ ---
presque exactement ce qu'un cheval fort soutient, la comparaison sur
laquelle Watt a bâti son \omterm{def:g10:orders-of-magnitude:unit}{unité}.
\end{solution}

\begin{solution}{exo:g11:work-of-force:12}
Longueurs $L = h/\text{pente}$~: \qty{1500}{m} et \qty{5000}{m}.
\omterm{def:g10:inertia:force}{Forces} $F = mg \times \text{pente}$~:
$1200 \times 9.81 \times 0.20 \approx \qty{2.35}{kN}$ et
$\qty{0.71}{kN}$. \omterm{def:g11:work-of-force:work}{Travaux} $FL$~:
$2354 \times 1500 \approx \qty{3.53}{MJ}$ et
$706 \times 5000 \approx \qty{3.53}{MJ}$ --- tous deux égaux à $mgh$~:
la route échange \omterm{def:g10:inertia:force}{force} contre longueur,
jamais \omterm{def:g11:work-of-force:work}{travail}.
\end{solution}

\begin{solution}{exo:g11:work-of-force:13}
$F = mg = \qty{2.45e3}{N}$~; $W = 2453 \times 15 \approx \qty{36.8}{kJ}$~;
$\Delta t = 15/0.40 = \qty{37.5}{s}$~;
$P = 36800/37.5 \approx \qty{981}{W} = 2453 \times 0.40$ ($Fv$
d'accord)~; électrique~: $981/0.70 \approx \qty{1.4}{kW}$.
\end{solution}

\begin{solution}{exo:g11:work-of-force:14}
Diagonale~: $-140 \times 6.0 = -\qty{840}{J}$~; murs~:
$-140 \times 8.4 \approx -\qty{1.18}{kJ}$~; aller-retour~:
$-140 \times 12 = -\qty{1.68}{kJ}$. Le
\omterm{def:g10:universal-gravitation:weight}{poids} fait zéro sur l'aller-retour
(pas de descente nette). Le \omterm{def:g11:work-of-force:work}{travail} du
\omterm{def:g10:inertia:inventory}{frottement} dépend du chemin et ne rembourse
jamais~; celui du \omterm{def:g10:universal-gravitation:weight}{poids} ne dépend
que des extrémités (\cref{rem:g11:work-of-force:divide}).
\end{solution}

\begin{solution}{exo:g11:work-of-force:15}
Gravité~: $80 \times 9.81 \times 600 \approx \qty{471}{kJ}$~;
\omterm{def:g10:inertia:inventory}{frottement}~: $14 \times 12000 = \qty{168}{kJ}$~;
total $\qty{639}{kJ}$. $v = \qty{4.17}{m/s}$, donc
$\Delta t = 12000/4.17 = \qty{2880}{s} = \qty{48}{min}$ et
$P = \num{6.39e5}/2880 \approx \qty{222}{W} \approx \qty{0.30}{hp}$ ---
un amateur fort, un tiers de cheval.
\end{solution}

\begin{solution}{pb:g11:work-of-force:1}
\textbf{1.} $W = 110 \times 8.0 = +\qty{880}{J}$.

\textbf{2.} \omterm{def:g10:inertia:inventory}{Frottement}~:
$-95 \times 8.0 = -\qty{760}{J}$. \omterm{def:g10:universal-gravitation:weight}{Poids}
et \omterm{def:g10:inertia:force}{force} normale sont perpendiculaires au
mouvement~: zéro chacun.

\textbf{3.} Total $+\qty{120}{J}$~: \omterm{def:g11:work-of-force:sign}{travail
moteur} net, la caisse accélère. Une poussée de \qty{95}{N} équilibre
le \omterm{def:g10:inertia:inventory}{frottement} ---
\omterm{def:g11:forces-and-motion:netforce}{force résultante} nulle, vitesse
constante, \omterm{def:g11:work-of-force:work}{travail} total nul.

\textbf{4.} Diagonale~: $-95 \times 10.0 = -\qty{950}{J}$~; murs~:
$-95 \times 14.0 = -\qty{1.33}{kJ}$~: le
\omterm{def:g10:inertia:inventory}{frottement} facture au
\omterm{def:g10:orders-of-magnitude:unit}{mètre}.

\textbf{5.} \omterm{def:g10:inertia:inventory}{Frottement}~:
$-95 \times 20.0 = -\qty{1.9}{kJ}$~;
\omterm{def:g10:universal-gravitation:weight}{poids}~: $0$ (pas de descente
nette). Seul le \omterm{def:g10:universal-gravitation:weight}{poids} rembourse.

\textbf{6.} $W(\vect P) = -30 \times 9.81 \times 9.0 =
-\qty{2.65}{kJ}$~; l'équipe doit fournir au moins $+\qty{2.65}{kJ}$.

\textbf{7.} $(70 + 30) \times 9.81 \times 9.0 = \qty{8.83}{kJ}$~; la
part de la caisse est $2.65/8.83 \approx 30\,\%$ --- porter, c'est
surtout se transporter soi-même.

\textbf{8.} $\sin\alpha = 9.0/30 = 0.30$~:
$F = mg\sin\alpha + f = 88.3 + 40 = \qty{128}{N}$~;
$W = 128 \times 30 = \qty{3.85}{kJ}$ ($= 2.65 + 1.2$).

\textbf{9.} Treuil vertical~: $F = mg = \qty{294}{N}$,
$W = \qty{2.65}{kJ}$. La rampe achète une
\omterm{def:g10:inertia:force}{force} plus petite ($128$ contre
\qty{294}{N}) au prix de plus de
\omterm{def:g11:work-of-force:work}{travail}~: le péage de
\qty{1.2}{kJ} du \omterm{def:g10:inertia:inventory}{frottement}.

\textbf{10.} $\Delta t = 9.0/0.50 = \qty{18}{s}$~;
$P = 2649/18 \approx \qty{147}{W}$.

\textbf{11.} $P = Fv = 294 \times 0.50 = \qty{147}{W}$ --- même.
Électrique~: $147/0.60 \approx \qty{245}{W}$.

\textbf{12.} $1000 \times 0.24 = \qty{240}{m}$ et
$8000 \times 0.030 = \qty{240}{m}$~: même sommet.

\textbf{13.} $F = mg \times \text{pente}$~: voie
$3500 \times 9.81 \times 0.24 \approx \qty{8.2}{kN}$~; route
$\approx \qty{1.0}{kN}$.

\textbf{14.} $8240 \times 1000 \approx \qty{8.24}{MJ}$ et
$1030 \times 8000 \approx \qty{8.24}{MJ}$ --- tous deux $= mgh = 3500
\times 9.81 \times 240$.

\textbf{15.} \omterm{def:g11:circuits-and-power:resistance}{Résistance} au
roulement~: $400 \times 1000 = \qty{0.40}{MJ}$ contre
$400 \times 8000 = \qty{3.2}{MJ}$~: le confort de la route douce coûte
\qty{2.8}{MJ} de \omterm{def:g10:inertia:inventory}{frottement} supplémentaire.

\textbf{16.} $v = \qty{10}{m/s}$. Route~:
$(1030 + 400) \times 10 = \qty{14.3}{kW} \approx \qty{19}{hp}$~; voie~:
$(8240 + 400) \times 10 = \qty{86.4}{kW} \approx \qty{117}{hp}$ ---
au-delà d'une camionnette chargée. Les routes serpentent pour que des
moteurs modestes (et des pneus) puissent monter~: même
\omterm{def:g11:work-of-force:work}{travail}, étalé
\omterm{def:g11:lenses-and-eye:lens}{mince}.

\textbf{17.} Par caisse~: $880 + 2649 = \qty{3.53}{kJ}$~; pour $60$
caisses, $60 \times 3529 \approx \qty{212}{kJ}$.

\textbf{18.} Escaliers~: $20 \times 70 \times 9.81 \times 9.0 \approx
\qty{124}{kJ}$. Total de la journée $\approx \qty{336}{kJ} =
\qty{0.34}{MJ}$.

\textbf{19.} $P = \num{3.36e5}/28800 \approx \qty{12}{W}$. La
bouilloire~: $\num{3.36e5}/2000 = \qty{168}{s}$ --- moins de trois
minutes pour tout le labeur de la journée.

\textbf{20.} Le \omterm{def:g11:work-of-force:work}{joule} paie la
\omterm{def:g10:inertia:force}{force} livrée \emph{le long} d'un
mouvement --- rien pour le mur poussé ou la caisse tenue, si épuisant
soit-ce (les muscles brûlent de l'
\omterm{def:g10:energy-conservation:forms}{énergie chimique} même en faisant zéro
\omterm{def:g11:work-of-force:work}{travail} mécanique). Puisque la
\omterm{def:g11:work-of-force:power}{puissance} humaine est de quelques dizaines
de \omterm{def:g11:work-of-force:power}{watts}, les déménageurs laissent les
machines remodeler la \omterm{def:g10:inertia:force}{force} --- treuil, rampe,
serpentines --- tandis que la facture $mgh$ ne change jamais.
\end{solution}
