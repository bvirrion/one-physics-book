\section*{Chapitre \ref{ch:g11:mechanical-energy} --- Énergie mécanique et sa conservation}

\begin{solution}{exo:g11:mechanical-energy:1}
(a)~$v = \qty{6.94}{m/s}$~:
$E_k = \tfrac12 \times 90 \times 6.94^2 \approx \qty{2.2e3}{J}$.
(b)~$v = \qty{36.1}{m/s}$~:
$E_k = \tfrac12 \times 1200 \times 36.1^2 \approx \qty{7.8e5}{J}$.
(c)~$\times 4$~; $\times 9$ --- le carré.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:2}

\end{solution}

\begin{solution}{exo:g11:mechanical-energy:3}
$v = \sqrt{2 \times 9.81 \times 20} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$. $mgh = \tfrac12 mv^2$~: la masse s'annule.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:4}
$W = \Delta E_k = 0 - \tfrac12 \times 1200 \times 25^2 =
\qty{-3.75e5}{J}$~; $F = \num{3.75e5}/50 = \qty{7.5e3}{N}$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:5}
$v = \sqrt{2 \times 9.81 \times 0.080} \approx \qty{1.3}{m/s}$~; elle
remonte à \qty{8.0}{cm} (conservation).
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:6}
$\num{1.5e9}/\num{2.6e3} \approx \num{5.9e5}$~: environ six cent mille
balles. $h = 88.9^2/(2 \times 9.81) \approx \qty{4.0e2}{m}$ --- sa
propre vitesse pourrait soulever le train de $400$~mètres.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:7}
Même vitesse $v = \sqrt{2 \times 9.81 \times 3.0} \approx \qty{7.7}{m/s}$
(seule la descente entre en jeu). Le toboggan raide gagne sur le temps~:
il atteint vite une grande vitesse et son chemin est plus court.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:8}
$h = v^2/2g = 144/19.62 \approx \qty{7.3}{m}$. À \qty{4.0}{m}~:
$v = \sqrt{144 - 2 \times 9.81 \times 4.0} \approx \qty{8.1}{m/s}$ ---
la même dans les deux sens~: $E_m$ dépend de la hauteur, non du sens
de parcours.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:9}
$v = \sqrt{23^2 + 2 \times 9.81 \times 35} = \sqrt{1216} \approx
\qty{35}{m/s}$. Plus petite en réalité~: la résistance de l'air dissipe
une partie de $E_m$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:10}
$\Delta E_m = \tfrac12 \times 45 \times 8.0^2 - 45 \times 9.81 \times
6.0 = 1440 - 2649 \approx \qty{-1.2e3}{J}$~;
$f = 1209/15 \approx \qty{81}{N}$. En chaleur, dans les skis et la
neige.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:11}
$v = \sqrt{2 \times 9.81 \times 0.15} \approx \qty{1.7}{m/s}$~; elle
remonte à \qty{15}{cm}. Le crochet change le chemin (un cercle plus
serré), non l'\omterm{def:g10:energy-conservation:energy}{énergie}~: hauteurs et
vitesses sont intactes.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:12}
$v = \sqrt{2 \times 9.81 \times 2.5} = \qty{7.0}{m/s}$. Les hauteurs de
remontée scalent avec $E_m$, donc $\times 0.9$ à chaque passage~:
$2.25$, $2.03$, $1.82$, $1.64$, $1.48$, $1.33$, $1.20$, $1.08$,
$0.97$ --- première fois sous \qty{1.0}{m} au $9$e passage.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:13}
Même $E_m$ initiale $= \tfrac12 m \times 15^2 + m \times 9.81 \times 45$
quel que soit l'angle, donc la vitesse au sol est
$v = \sqrt{225 + 2 \times 9.81 \times 45} = \sqrt{1108} \approx
\qty{33}{m/s}$. L'angle fixe la \emph{direction} de la vitesse
d'atterrissage, le temps de vol et la
\omterm{def:g11:fundamental-interactions:strong}{portée} --- non la vitesse
d'atterrissage.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:14}
$\tfrac12 mv^2 = mgh - f\ell = 80 \times 9.81 \times 120 - 65 \times
800 = \num{94176} - \num{52000} = \qty{4.22e4}{J}$, donc
$v = \sqrt{2 \times \num{42176}/80} \approx \qty{32}{m/s}$
(\qty{117}{km/h}). Fraction
\omterm{def:g11:mechanical-energy:dissipation}{dissipée}~: $52000/94176 \approx 55\,\%$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:15}
$m = \qty{2.0e9}{kg}$~:
$E_p = \num{2.0e9} \times 9.81 \times 300 \approx \qty{5.9e12}{J}
= \num{5.9e12}/\num{3.6e6} \approx \qty{1.6e6}{kW.h}$. Récupérée~:
$0.85 \times \num{1.64e6} \approx \qty{1.4e6}{kW.h}$ --- environ
$\num{1.4e5}$ foyers pour une journée.
\end{solution}

\begin{solution}{pb:g11:mechanical-energy:1}
\textbf{1.} $v_{\text{sommet}} = \sqrt{gR} = \sqrt{9.81 \times 8.0}
\approx \qty{8.9}{m/s}$.

\textbf{2.} $E_m = mg\,(2R) + \tfrac12 m gR = \num{3.14e5} +
\num{7.8e4} \approx \qty{3.92e5}{J}$.

\textbf{3.} $mgH_{\min} = E_m$ donne $H_{\min} = \qty{20}{m}$~;
algébriquement $H_{\min} = 2R + \frac{gR}{2g} = \frac{5R}{2}$.

\textbf{4.} $v_{\text{sommet}} = \sqrt{2g(H - 2R)} =
\sqrt{2 \times 9.81 \times 8.0} \approx \qty{12.5}{m/s}$~;
$v_{\text{sommet}}^2/(gR) = 2.0$ --- deux fois le minimum.

\textbf{5.} Chaque \omterm{def:g10:energy-conservation:energy}{énergie} est
proportionnelle à $m$, donc $m$ s'annule~: mêmes vitesses et même
sécurité, plein ou vide.

\textbf{6.} $v = \sqrt{2 \times 9.81 \times 24} \approx \qty{21.7}{m/s}
\approx \qty{78}{km/h}$.

\textbf{7.} $v = \sqrt{2 \times 9.81 \times (24 - 8.0)} \approx
\qty{17.7}{m/s}$.

\textbf{8.} $mgH = \qty{4.71e5}{J}$. $(E_p, E_k)$~: lancement
$(\num{4.71e5}, 0)$~; bas $(0, \num{4.71e5})$~; côté
$(\num{1.57e5}, \num{3.14e5})$~; sommet $(\num{3.14e5}, \num{1.57e5})$ ---
chaque paire somme à \qty{4.71e5}{J}.

\textbf{9.} $v = \sqrt{2 \times 9.81 \times (24 - 12)} \approx
\qty{15.3}{m/s}$.

\textbf{10.} $v(z) = \sqrt{2g(H - z)}$~: le plus rapide au point le plus
bas de la piste, le plus lent au sommet de la boucle.

\textbf{11.} Même hauteur, donc $\Delta E_p = 0$ et
$\Delta E_m = \Delta E_k = \tfrac12 \times 2000 \times (19.2^2 -
20.4^2) \approx \qty{-4.75e4}{J}$~: le placement rend le déficit purement
cinétique, lu sur deux compteurs de vitesse.

\textbf{12.} $W(\vect f) = \qty{-4.75e4}{J}$~;
$f = \num{4.75e4}/60 \approx \qty{7.9e2}{N}$.

\textbf{13.} \omterm{def:g10:energy-conservation:forms}{Énergie thermique}, dans les
roues, les rails et l'air.

\textbf{14.} $E_m$ au premier capteur~: $\num{4.16e5} + \num{3.9e4} =
\qty{4.55e5}{J}$~; perte $\num{4.75e4}/\num{4.55e5} \approx 10\,\%$ par
\qty{60}{m}. À $H_{\min}$ le sommet de la boucle est atteint
\emph{exactement} à $\sqrt{gR}$~; toute perte le fait descendre sous la
vitesse de sécurité.

\textbf{15.} En perdant $\approx 10\,\%$ par \qty{60}{m}, environ
$15\,\%$ de $mgH$ ($\approx \qty{7.1e4}{J}$) a disparu~:
$E_m \approx \qty{4.00e5}{J}$, encore au-dessus des \qty{3.92e5}{J}
requis --- sûr, avec seulement $2\,\%$ de marge. La marge n'était pas un
luxe.

\textbf{16.} $E_k = \tfrac12 \times 2000 \times 18.0^2 =
\qty{3.24e5}{J}$.

\textbf{17.} $F = \num{3.24e5}/45 = \qty{7.2e3}{N}$.

\textbf{18.} $a = 7200/2000 = \qty{3.6}{m/s^2} \approx 0.37\,g$~: ferme
mais confortable.

\textbf{19.} \omterm{def:g10:inertia:inventory}{Frottement} de piste~:
$mgH - \num{3.24e5} = \num{470880} - \num{324000} = \qty{1.47e5}{J}$.
Audit~: $\num{1.47e5} + \num{3.24e5} = \num{4.71e5} = mgH$ --- chaque
\omterm{def:g11:work-of-force:work}{joule} est comptabilisé.

\textbf{20.} La hauteur de lancement $H = \qty{24}{m}$ a fixé la
sécurité de la boucle, chaque vitesse $\sqrt{2g(H - z)}$, la facture de
\omterm{def:g10:inertia:inventory}{frottement} et la
\omterm{def:g10:inertia:force}{force} de freinage~; le prix d'ignorer le
\omterm{def:g10:inertia:inventory}{frottement} est la marge --- les
$\qty{4}{m}$ de colline que le monde frottant consomme silencieusement.
\end{solution}
