\begin{solution}{exo:g11:fundamental-interactions:1}
Proton~: $10^{-15}$ (étage \omterm{def:g11:fundamental-interactions:ladder}{nucléon})~;
\omterm{def:g11:fundamental-interactions:ladder}{molécule} d'eau~: $10^{-10}$ (étage
\omterm{def:g11:fundamental-interactions:ladder}{molécule} --- les
\omterm{def:g11:fundamental-interactions:ladder}{molécules} isolées siègent aux tailles
atomiques)~; sable~: $10^{-4}$ (matière du quotidien)~; Terre~: $10^{7}$
(planète)~; Voie lactée~: $10^{21}$ (galaxie). Tout en
\omterm{def:g10:orders-of-magnitude:unit}{mètres}.
\end{solution}

portent $+\qty{4.8e-9}{C}$, par conservation de la charge

\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:6}
$F_E = \num{8.99e9} \times (\num{1.60e-19})^2/(\num{2.0e-15})^2
= \num{2.30e-28}/\num{4.0e-30} \approx \qty{58}{N}$~;
$F_G = \num{6.67e-11} \times (\num{1.67e-27})^2/\num{4.0e-30}
\approx \qty{4.7e-35}{N}$. Rapport $F_E/F_G \approx \num{1.2e36}$.
L'\omterm{def:g11:fundamental-interactions:strong}{interaction forte} tient le
\omterm{def:g11:fundamental-interactions:ladder}{noyau}, la gravité est hors sujet.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:7}

\approx \qty{8.2e-8}{N}$~;
$F_G = \num{6.67e-11} \times \num{1.67e-27} \times \num{9.11e-31} /

\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:8}
Par contact la charge est partagée~: $q_A = q_B = \qty{+4.0e-9}{C}$
(total toujours $\qty{8.0e-9}{C}$). Alors
$F = \num{8.99e9} \times (\num{4.0e-9})^2/0.10^2 =
\qty{1.4e-5}{N}$, répulsive (mêmes signes).
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:9}
$d = \sqrt{k q^2/F} = \sqrt{\num{8.99e9} \times \num{1.0e-12}/0.90}
\approx \qty{0.10}{m}$. Une \omterm{def:g10:inertia:force}{force} $9$ fois
plus petite exige une distance $3$ fois plus grande~: $d = \qty{0.30}{m}$.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:10}
$F = \num{2.30e-28}/(\num{1.0e-10})^2 = \qty{2.3e-8}{N}$.
L'\omterm{def:g11:fundamental-interactions:strong}{interaction forte} ne contribue rien~:
$10^{-10}$~m est $10^{5}$ fois sa
\omterm{def:g11:fundamental-interactions:strong}{portée}. Les
\omterm{def:g11:fundamental-interactions:ladder}{molécules} sont liées par l'
\omterm{def:g11:fundamental-interactions:four}{interaction électromagnétique}.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:11}
$q = \num{e23}/\num{e6} \times \num{1.6e-19} = \qty{1.6e-2}{C}$ sur
chaque pièce. $F = \num{8.99e9} \times (\num{1.6e-2})^2/1.0^2 \approx
\qty{2.3e6}{N}$ --- le \omterm{def:g10:universal-gravitation:weight}{poids} de
$\num{2.3e6}/9.81 \approx \qty{2.3e5}{kg}$, un train de marchandises
chargé, pour un millionième des électrons. La matière ordinaire doit
être (et est) neutre bien mieux qu'à une partie par million.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:12}
$F = \num{2.30e-28}/(\num{1.4e-14})^2 = \num{2.30e-28}/\num{1.96e-28}
\approx \qty{1.2}{N}$ --- encore énorme à cette échelle. Non~: ils sont
$14$ fois la \omterm{def:g11:fundamental-interactions:strong}{portée} forte~; chacun n'est
collé qu'à ses plus proches voisins. Paires~:
$\frac{92 \times 91}{2} = 4186$, \emph{toutes} se repoussant à toute
distance, tandis que la colle ne croît pas avec la taille. Les neutrons
ajoutent de l'attraction sans répulsion et diluent les protons~; mais
la répulsion croît avec le carré du nombre de protons et la colle
seulement avec le nombre de voisins, donc au-delà d'environ $80$ protons
aucun budget de neutrons n'équilibre les comptes --- de tels noyaux se
désintègrent (\cref{ch:g11:nucleus-radioactivity}).
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:13}
Équilibre~: \omterm{def:g10:universal-gravitation:weight}{verticalement}
$T\cos\theta = mg$, horizontalement $T\sin\theta = F$, donc
$F = mg\tan\theta$. Ici $\sin\theta = 3.0/50 = 0.060$, et pour les
petits angles $\tan\theta \approx \sin\theta = 0.060$. Alors
$F = \num{1.0e-3} \times 9.81 \times 0.060 \approx \qty{5.9e-4}{N}$~;
$q = d\sqrt{F/k} = 0.060 \times \sqrt{\num{5.9e-4}/\num{8.99e9}}
\approx \qty{1.54e-8}{C}$~; $N = \num{1.54e-8}/\num{1.6e-19} \approx
\num{9.6e10}$ \omterm{def:g11:fundamental-interactions:elementary}{charges élémentaires}.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:14}
$F = \num{6.67e-11} \times \num{5.97e24} \times \num{7.35e22} /
(\num{3.84e8})^2 = \num{2.93e37}/\num{1.47e17} \approx \qty{2.0e20}{N}$.
En posant $kQ^2/d^2 = F$~:
$Q = d\sqrt{F/k} = \num{3.84e8} \times \sqrt{\num{2.0e20}/\num{8.99e9}}
\approx \qty{5.7e13}{C}$. C'est
$N = \num{5.7e13}/\num{1.6e-19} \approx \num{3.6e32}$ électrons, de
masse $\num{3.6e32} \times \num{9.11e-31} \approx \qty{3.2e2}{kg}$~:
quelques centaines de \omterm{def:g10:orders-of-magnitude:unit}{kilogrammes}
d'électrons pourraient faire le travail de $\qty{7.35e22}{kg}$ de Lune ---
la force de l'électricité et la faiblesse de la gravité en un seul
nombre.
\end{solution}

\begin{solution}{exo:g11:fundamental-interactions:15}
$F = \num{2.30e-28}/(\num{2.8e-10})^2 = \num{2.30e-28}/\num{7.84e-20}
\approx \qty{2.9e-9}{N}$~; \omterm{def:g10:universal-gravitation:weight}{poids}
$P = \num{5.9e-26} \times 9.81 \approx \qty{5.8e-25}{N}$~; rapport
$F/P \approx \num{5e15}$. Chaque liaison a une force \emph{fixe}, mais
le \omterm{def:g10:universal-gravitation:weight}{poids} croît avec toute la masse
au-dessus~: empilez de la roche et la charge sur la couche du bas croît
sans limite tandis que ses liaisons non. Vers \qty{e4}{m} de roche la
charge écrase les liaisons électriques --- la gravité gagne non par
force par particule mais en \emph{s'additionnant} lorsque la charge
s'annule.
\end{solution}

\begin{solution}{pb:g11:fundamental-interactions:1}
\textbf{1.} \omterm{def:g11:fundamental-interactions:ladder}{Quark} $< \qty{e-18}{m}$~;
\omterm{def:g11:fundamental-interactions:ladder}{nucléon} $\qty{e-15}{m}$~;
\omterm{def:g11:fundamental-interactions:ladder}{noyau} $10^{-15}$--$\qty{e-14}{m}$~;
\omterm{def:g11:fundamental-interactions:ladder}{atome} $\qty{e-10}{m}$~;
\omterm{def:g11:fundamental-interactions:ladder}{molécule} $\qty{e-9}{m}$~; objet du
quotidien $\qty{1}{m}$~; planète $\qty{e7}{m}$~; système planétaire
$\qty{e11}{m}$~; galaxie $\qty{e21}{m}$.

\textbf{2.} Facteur d'échelle $10^{5}$~: l'
\omterm{def:g11:fundamental-interactions:ladder}{atome} fait
$\qty{1}{cm} \times 10^{5} = \qty{1}{km}$ de large.

\textbf{3.} $(10^{-15}/10^{-10})^3 = 10^{-15}$~: la matière est
$\num{99.9999999999999}\,\%$ vide.

\textbf{4.} $\num{7e-5}/\num{e-10} = \num{7e5}$
\omterm{def:g11:fundamental-interactions:ladder}{atomes} --- environ un million sur un
cheveu.

\textbf{5.} De $10^{-18}$ à $10^{21}$~: $39$
\omterm{def:g10:energy-conservation:power}{puissances} de dix.

\textbf{6.} $F_E = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{5.3e-11})^2 \approx \qty{8.2e-8}{N}$.

\textbf{7.} $F_G = \num{6.67e-11} \times \num{1.67e-27} \times
\num{9.11e-31}/(\num{5.3e-11})^2 \approx \qty{3.6e-47}{N}$~; rapport
$F_E/F_G \approx \num{2.3e39}$. L'électricité tient l'
\omterm{def:g11:fundamental-interactions:ladder}{atome}~; la gravité est spectatrice.

\textbf{8.} Sans gravité~: rien de mesurable ne change (elle est
$10^{39}$ en dessous). Sans électricité~: pas d'
\omterm{def:g11:fundamental-interactions:ladder}{atomes}, pas de
\omterm{def:g11:fundamental-interactions:ladder}{molécules}, pas de matière du tout.

\textbf{9.} Le «~contact~» est la répulsion électrique entre les nuages
électroniques d'\omterm{def:g11:fundamental-interactions:ladder}{atomes} amenés à environ
$\qty{e-10}{m}$~: rien ne se touche jamais. Réaction du sol et
\omterm{def:g10:inertia:inventory}{frottement} sont la loi de Coulomb en
volume.

\textbf{10.} $q = \num{e23} \times \num{e-9} \times \num{1.6e-19} =
\qty{1.6e-5}{C}$ chacune~;
$F = \num{8.99e9} \times (\num{1.6e-5})^2 \approx \qty{2.3}{N}$ --- une
\omterm{def:g10:inertia:force}{force} bien visible pour un électron sur un
milliard. La matière ordinaire est neutre bien mieux que $10^{-9}$.

\textbf{11.} L'\omterm{def:g11:fundamental-interactions:four}{interaction
électromagnétique}. La \omterm{def:g10:inertia:force}{force} forte est hors de
\omterm{def:g11:fundamental-interactions:strong}{portée} au-delà d'environ
$\qty{e-14}{m}$ --- $10^{4}$ fois plus petit qu'un seul
\omterm{def:g11:fundamental-interactions:ladder}{atome}.

\textbf{12.} $F = \num{8.99e9} \times (\num{1.60e-19})^2 /
(\num{1.0e-15})^2 \approx \qty{2.3e2}{N}$.

\textbf{13.} $a = F/m_p = 230/\num{1.67e-27} \approx
\qty{1.4e29}{m/s^2}$, environ $\num{1.4e28}$ fois $g$~: aucune
\omterm{def:g10:inertia:force}{force} du quotidien n'approche.

\textbf{14.} Plus forte que \qty{230}{N} à $\qty{e-15}{m}$ (les noyaux
existent)~; agissant sur protons et neutrons indifféremment, aveugle à
la charge (les neutrons sont saisis)~;
\omterm{def:g11:fundamental-interactions:strong}{portée} d'environ $\qty{e-15}{m}$ (les
noyaux restent minuscules --- la colle ne peut aller plus loin).

\textbf{15.} $F = \num{2.30e-28}/(\num{1.4e-14})^2 \approx
\qty{1.2}{N}$. Non~: $14$ fois la
\omterm{def:g11:fundamental-interactions:strong}{portée}. La répulsion agit sur
\emph{toutes} les $\frac{92\times91}{2} = 4186$ paires de protons~; la
colle ne lie que les plus proches voisins, donc la croissance favorise
la répulsion. Les neutrons ajoutent de la colle et de l'espacement sans
ajouter de répulsion --- jusqu'à ce qu'au-delà d'environ $80$ protons
aucun mélange n'équilibre et le
\omterm{def:g11:fundamental-interactions:ladder}{noyau} se désintègre.

\textbf{16.} L'\omterm{def:g11:fundamental-interactions:weak}{interaction faible} transforme
un neutron en proton (et inversement), provoquant la désintégration
$\beta$ --- elle transforme~; elle ne lie jamais.

\textbf{17.} $F = \num{6.67e-11} \times \num{5.97e24} \times
\num{7.35e22}/(\num{3.84e8})^2 \approx \qty{2.0e20}{N}$.

\textbf{18.} $Q_E = \num{1.8e51} \times \num{1.6e-19} =
\qty{2.9e32}{C}$, $Q_M = \qty{3.5e30}{C}$. Égaler les
\omterm{def:g10:inertia:force}{forces} signifie
$f^2\,k\,Q_E Q_M = G M_E M_M$~:
\[
f = \sqrt{\frac{\num{2.93e37}}{\num{8.99e9} \times \num{2.9e32}
\times \num{3.5e30}}} = \sqrt{\num{3.2e-36}} \approx \num{1.8e-18} :
\]
un déséquilibre de \emph{deux parties sur $10^{18}$} annulerait la
gravité.

\textbf{19.} La charge a deux signes et s'annule (la question~18 montre
à quel point parfaitement), donc la \omterm{def:g10:inertia:force}{force}
plus forte s'éteint d'elle-même aux grandes échelles~; la masse n'a
qu'un signe, s'ajoute pour toujours, et rien ne l'écranne.

\textbf{20.} \omterm{def:g11:fundamental-interactions:ladder}{Noyau}~: l'
\omterm{def:g11:fundamental-interactions:strong}{interaction forte} domine la répulsion des
protons à $\qty{e-15}{m}$.
\omterm{def:g11:fundamental-interactions:ladder}{Atome} à montagne~: l'
\omterm{def:g11:fundamental-interactions:four}{interaction électromagnétique} lie les
électrons aux noyaux et les
\omterm{def:g11:fundamental-interactions:ladder}{atomes} entre eux. Planète et au-delà~:
la \omterm{def:g11:fundamental-interactions:four}{gravitation}, sans opposition une fois
la charge annulée, tient planètes,
\omterm{def:g10:universal-gravitation:satellite}{orbites} et galaxies. Le monde ne
s'effondre pas parce que chaque étage est soutenu par un liant plus
raide que la charge, et ne s'envole pas parce que la matière est neutre
à environ $10^{-18}$ --- donc chaque échelle a exactement une
\omterm{def:g10:inertia:force}{force} en charge, et elle est attractive là
où elle doit l'être.
\end{solution}
