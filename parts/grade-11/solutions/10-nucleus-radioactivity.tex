\section*{Chapter \ref{ch:g11:nucleus-radioactivity} --- The Nucleus and Radioactivity}

\begin{solution}{exo:g11:nucleus-radioactivity:1}
${}^{16}_{8}\mathrm{O}$: $8$ p, $8$ n; ${}^{27}_{13}\mathrm{Al}$: $13$ p,
$14$ n; ${}^{60}_{27}\mathrm{Co}$: $27$ p, $33$ n;
${}^{235}_{92}\mathrm{U}$: $92$ p, $143$ n; ${}^{3}_{1}\mathrm{H}$: $1$ p,
$2$ n. ($N = A - Z$ each time.)
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:2}
The three carbons ($Z = 6$) are isotopes of one another.
${}^{14}_{6}\mathrm{C}$ and ${}^{14}_{7}\mathrm{N}$ share $A = 14$ but not
$Z$: different elements (different electron clouds), so not isotopes.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:3}
$A = 210 - 4 = 206$, $Z = 84 - 2 = 82$:
${}^{210}_{84}\mathrm{Po} \to {}^{206}_{82}\mathrm{Pb} +
{}^{4}_{2}\mathrm{He}$ --- the daughter is lead-206.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:8}
Fluorine-18 has $9$ neutrons where stable fluorine-19 has $10$:
neutron-poor, below the band, so $\beta^+$:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:9}
$24$ hours $= 4\,T_{1/2}$: $500/16 \approx \qty{31}{MBq}$. Since
$2^9 = 512$, nine half-lives bring $500/512 < \qty{1}{MBq}$: after
$9 \times 6 = \qty{54}{h}$.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:10}
$\frac18 = \frac1{2^3}$: three half-lives, $3 \times \num{5700} =
\num{17100}$ years. A dinosaur bone is about $\num{17500}$ half-lives old:
a fraction $1/2^{17500}$ remains --- not one atom of carbon-14 (a gram of
carbon holds only about $\num{5e22}$ atoms).
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:11}
${}^{14}_{6}\mathrm{C}$: $8$ neutrons against $6$--$7$ in stable carbon,
above the band, $\beta^-$. ${}^{11}_{6}\mathrm{C}$: $5$ neutrons,
neutron-poor, $\beta^+$. ${}^{238}_{92}\mathrm{U}$: $Z = 92 > 83$, too
heavy, $\alpha$.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:12}
Only $\alpha$ changes $A$: $238 - 206 = 32 = 4x$, so $x = 8$ alphas. They
remove $16$ from $Z$: $92 - 16 + y = 82$ gives $y = 6$ beta-minus decays.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:13}
$6.4/0.1 = 64 = 2^{6}$: six half-lives, $6 \times 5.3 = 31.8 \approx 32$
years in shielded storage.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:14}
Per day: $8000 \times 86400 \approx \num{6.9e8}$ decays. Over $80$ years:
$\num{6.9e8} \times 365 \times 80 \approx \num{2e13}$. Life has always run
on this background --- one becquerel is one atom per second out of the
body's $\sim 10^{27}$; the unit is tiny, and a scary-sounding count of
becquerels may be a few bananas' worth.
\end{solution}

\begin{solution}{exo:g11:nucleus-radioactivity:15}
$\num{25000} \times 86400 \approx \num{2.2e9}$ decays per day. Ten years is
$10/432 \approx 2\%$ of a half-life: far less than half has decayed, so the
electronics and dust, not the source, force replacement. Outside, the
$\alpha$s die in the chamber's air and casing; inhaled dust parks the
emitter in the lungs, where every $\alpha$ dumps its energy into living
tissue --- a large dose (sieverts) from a modest activity.
\end{solution}

\begin{solution}{pb:g11:nucleus-radioactivity:1}
\textbf{1.} ${}^{12}_{6}\mathrm{C}$: $6$ p, $6$ n; ${}^{14}_{6}\mathrm{C}$:
$6$ p, $8$ n. Chemistry sees only the electron cloud, fixed by $Z = 6$:
identical behaviour.

\textbf{2.} Eating and breathing constantly renew a living body's carbon
at the atmospheric proportion; at death the intake stops and decay runs
unopposed.

\textbf{3.} ${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} +
{}^{\;0}_{-1}\mathrm{e}$ ($\beta^-$). These $\beta$s are soft: metres of
wrapped linen stop them, so little escapes the mummy --- which is why one
measures a small carbon sample taken from the object itself.

\textbf{4.} $6.8/13.6 = \frac12$: one half-life. The mummy is about
\num{5700} years old.

\textbf{5.} $13.2/13.6 \approx 0.97$: the linen is essentially modern ---
at most a few centuries old. A forgery.

\textbf{6.} $13.6/2^{10} = 13.6/1024 \approx 0.013$ decays per minute ---
one count every $75$ minutes per gram, drowned in natural background. Ten
half-lives, about \num{57000} years, is the practical horizon.

\textbf{7.} $92$ protons, $146$ neutrons. Coulomb repulsion acts between
all proton pairs across the nucleus while strong glue binds only
neighbours: heavy nuclei survive only with surplus neutrons, glue without
repulsion.

\textbf{8.} ${}^{238}_{92}\mathrm{U} \to {}^{234}_{90}\mathrm{Th} +
{}^{4}_{2}\mathrm{He}$.

\textbf{9.} ${}^{234}_{90}\mathrm{Th} \to {}^{234}_{91}\mathrm{Pa} +
{}^{\;0}_{-1}\mathrm{e}$, then ${}^{234}_{91}\mathrm{Pa} \to
{}^{234}_{92}\mathrm{U} + {}^{\;0}_{-1}\mathrm{e}$: uranium reappears, as
uranium-234.

\textbf{10.} $A$: $238 - 206 = 32$, so $8$ $\alpha$ steps; $Z$:
$92 - 16 + y = 82$, so $6$ $\beta^-$ steps.

\textbf{11.} $\num{4.5e9}$ years is one half-life: $\frac12$ remains. Five
billion more years $\approx$ one further half-life: about $\frac14$.

\textbf{12.} It drives volcanoes, plate tectonics (drifting continents,
earthquakes) and geothermal heat: the surface geology of a planet warmed
from within.

\textbf{13.} ${}^{241}_{95}\mathrm{Am} \to {}^{237}_{93}\mathrm{Np} +
{}^{4}_{2}\mathrm{He}$.

\textbf{14.} $\num{3.3e4}$ decays per second;
$\num{33000} \times 86400 \approx \num{2.9e9}$ per day.

\textbf{15.} Each $\alpha$ ionizes the air it crosses; the ions carry a
tiny current between the chamber's electrodes. Smoke particles capture the
ions, the current drops, the alarm fires.

\textbf{16.} The $\alpha$s spend all their energy inside the chamber and
none escapes the casing: maximal ionization, zero leakage. A $\gamma$
source of equal activity would ionize the chamber's air barely at all and
irradiate the whole room instead.

\textbf{17.} Ten years is $10/432 \approx 2\%$ of a half-life: the
activity is essentially unchanged. The weak links are dust, insects and
electronics --- the nucleus outlives the gadget.

\textbf{18.} $\num{9e9}$ years $= 2$ half-lives: $\frac14$ survives.
Earth's uranium is still abundant, so it was forged at most a few
half-lives ago --- the planet condensed from fresh star ash, billions of
years after the first stars.

\textbf{19.} Geologically dead: a cold interior, no volcanism or plate
tectonics to recycle air and rock, and no churning core --- hence no
magnetic shield against the solar wind.

\textbf{20.} Every atom of you beyond helium was forged in a dying star,
and the unstable leftovers of that forge still warm the ground you stand
on: stardust, kept alive by its own half-life.
\end{solution}
