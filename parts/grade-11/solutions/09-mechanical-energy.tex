\section*{Chapter \ref{ch:g11:mechanical-energy} --- Mechanical Energy and Its Conservation}

\begin{solution}{exo:g11:mechanical-energy:1}
(a) $v = \qty{6.94}{m/s}$:
$E_k = \tfrac12 \times 90 \times 6.94^2 \approx \qty{2.2e3}{J}$.
(b) $v = \qty{36.1}{m/s}$:
$E_k = \tfrac12 \times 1200 \times 36.1^2 \approx \qty{7.8e5}{J}$.
(c) $\times 4$; $\times 9$ --- the square.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:2}

the reference shifts both values equally. No: only the height
difference enters.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:3}
$v = \sqrt{2 \times 9.81 \times 20} = \qty{19.8}{m/s} \approx
\qty{71}{km/h}$. $mgh = \tfrac12 mv^2$: the mass cancels.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:4}
$W = \Delta E_k = 0 - \tfrac12 \times 1200 \times 25^2 =
\qty{-3.75e5}{J}$; $F = \num{3.75e5}/50 = \qty{7.5e3}{N}$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:5}
$v = \sqrt{2 \times 9.81 \times 0.080} \approx \qty{1.3}{m/s}$; it
rises back to \qty{8.0}{cm} (conservation).
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:6}
$\num{1.5e9}/\num{2.6e3} \approx \num{5.9e5}$: about six hundred
thousand bullets. $h = 88.9^2/(2 \times 9.81) \approx \qty{4.0e2}{m}$
--- its own speed could lift the train $400$ meters.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:7}
Same speed $v = \sqrt{2 \times 9.81 \times 3.0} \approx \qty{7.7}{m/s}$
(only the drop enters). The steep slide wins on time: it reaches high
speed sooner and its path is shorter.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:8}
$h = v^2/2g = 144/19.62 \approx \qty{7.3}{m}$. At \qty{4.0}{m}:
$v = \sqrt{144 - 2 \times 9.81 \times 4.0} \approx \qty{8.1}{m/s}$ ---
the same both ways: $E_m$ depends on height, not on direction of travel.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:9}
$v = \sqrt{23^2 + 2 \times 9.81 \times 35} = \sqrt{1216} \approx
\qty{35}{m/s}$. Smaller in reality: air resistance dissipates part of
$E_m$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:10}
$\Delta E_m = \tfrac12 \times 45 \times 8.0^2 - 45 \times 9.81 \times
6.0 = 1440 - 2649 \approx \qty{-1.2e3}{J}$;
$f = 1209/15 \approx \qty{81}{N}$. Into heat, in the runners and the
snow.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:11}
$v = \sqrt{2 \times 9.81 \times 0.15} \approx \qty{1.7}{m/s}$; it rises
to \qty{15}{cm} again. The peg changes the path (a tighter circle), not
the \omterm{def:g10:energy-conservation:energy}{energy}: heights and speeds are untouched.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:12}
$v = \sqrt{2 \times 9.81 \times 2.5} = \qty{7.0}{m/s}$. Rise heights
scale with $E_m$, so $\times 0.9$ each traverse: $2.25$, $2.03$, $1.82$,
$1.64$, $1.48$, $1.33$, $1.20$, $1.08$, $0.97$ --- first below
\qty{1.0}{m} on the $9$th traverse.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:13}
Same initial $E_m = \tfrac12 m \times 15^2 + m \times 9.81 \times 45$
whatever the angle, so the ground speed is
$v = \sqrt{225 + 2 \times 9.81 \times 45} = \sqrt{1108} \approx
\qty{33}{m/s}$. The angle fixes the \emph{direction} of the landing
velocity, the flight time and the \omterm{def:g11:fundamental-interactions:strong}{range} --- not the landing speed.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:14}
$\tfrac12 mv^2 = mgh - f\ell = 80 \times 9.81 \times 120 - 65 \times
800 = \num{94176} - \num{52000} = \qty{4.22e4}{J}$, so
$v = \sqrt{2 \times \num{42176}/80} \approx \qty{32}{m/s}$
(\qty{117}{km/h}). Fraction \omterm{def:g11:mechanical-energy:dissipation}{dissipated}: $52000/94176 \approx 55\%$.
\end{solution}

\begin{solution}{exo:g11:mechanical-energy:15}
$m = \qty{2.0e9}{kg}$:
$E_p = \num{2.0e9} \times 9.81 \times 300 \approx \qty{5.9e12}{J}
= \num{5.9e12}/\num{3.6e6} \approx \qty{1.6e6}{kW.h}$. Recovered:
$0.85 \times \num{1.64e6} \approx \qty{1.4e6}{kW.h}$ --- about
$\num{1.4e5}$ homes for a day.
\end{solution}

\begin{solution}{pb:g11:mechanical-energy:1}
\textbf{1.} $v_{\text{top}} = \sqrt{gR} = \sqrt{9.81 \times 8.0}
\approx \qty{8.9}{m/s}$.

\textbf{2.} $E_m = mg\,(2R) + \tfrac12 m gR = \num{3.14e5} +
\num{7.8e4} \approx \qty{3.92e5}{J}$.

\textbf{3.} $mgH_{\min} = E_m$ gives $H_{\min} = \qty{20}{m}$;
algebraically $H_{\min} = 2R + \frac{gR}{2g} = \frac{5R}{2}$.

\textbf{4.} $v_{\text{top}} = \sqrt{2g(H - 2R)} =
\sqrt{2 \times 9.81 \times 8.0} \approx \qty{12.5}{m/s}$;
$v_{\text{top}}^2/(gR) = 2.0$ --- twice the minimum.

\textbf{5.} Every \omterm{def:g10:energy-conservation:energy}{energy} is proportional to $m$, so $m$ cancels: same
speeds and same safety, full or empty.

\textbf{6.} $v = \sqrt{2 \times 9.81 \times 24} \approx \qty{21.7}{m/s}
\approx \qty{78}{km/h}$.

\textbf{7.} $v = \sqrt{2 \times 9.81 \times (24 - 8.0)} \approx
\qty{17.7}{m/s}$.

\textbf{8.} $mgH = \qty{4.71e5}{J}$. $(E_p, E_k)$: launch
$(\num{4.71e5}, 0)$; bottom $(0, \num{4.71e5})$; side
$(\num{1.57e5}, \num{3.14e5})$; top $(\num{3.14e5}, \num{1.57e5})$ ---
each pair sums to \qty{4.71e5}{J}.

\textbf{9.} $v = \sqrt{2 \times 9.81 \times (24 - 12)} \approx
\qty{15.3}{m/s}$.

\textbf{10.} $v(z) = \sqrt{2g(H - z)}$: fastest at the lowest point of
the track, slowest at the top of the loop.

\textbf{11.} Same height, so $\Delta E_p = 0$ and
$\Delta E_m = \Delta E_k = \tfrac12 \times 2000 \times (19.2^2 -
20.4^2) \approx \qty{-4.75e4}{J}$: the placement makes the deficit pure
kinetic, read off two speedometers.

\textbf{12.} $W(\vect f) = \qty{-4.75e4}{J}$;
$f = \num{4.75e4}/60 \approx \qty{7.9e2}{N}$.

\textbf{13.} \omterm{def:g10:energy-conservation:forms}{Thermal energy}, in the wheels, the rails and the air.

\textbf{14.} $E_m$ at the first sensor: $\num{4.16e5} + \num{3.9e4} =
\qty{4.55e5}{J}$; loss $\num{4.75e4}/\num{4.55e5} \approx 10\%$ per
\qty{60}{m}. At $H_{\min}$ the top of the loop is reached \emph{exactly}
at $\sqrt{gR}$; any loss drops it below the safety speed.

\textbf{15.} Losing $\approx 10\%$ per \qty{60}{m}, about $15\%$ of
$mgH$ ($\approx \qty{7.1e4}{J}$) is gone: $E_m \approx \qty{4.00e5}{J}$,
still above the required \qty{3.92e5}{J} --- safe, with only $2\%$ to
spare. The margin was no luxury.

\textbf{16.} $E_k = \tfrac12 \times 2000 \times 18.0^2 =
\qty{3.24e5}{J}$.

\textbf{17.} $F = \num{3.24e5}/45 = \qty{7.2e3}{N}$.

\textbf{18.} $a = 7200/2000 = \qty{3.6}{m/s^2} \approx 0.37\,g$:
firm but comfortable.

\textbf{19.} Track \omterm{def:g10:inertia:inventory}{friction}: $mgH - \num{3.24e5} = \num{470880} -
\num{324000} = \qty{1.47e5}{J}$. Audit: $\num{1.47e5} + \num{3.24e5} =
\num{4.71e5} = mgH$ --- every \omterm{def:g11:work-of-force:work}{joule} accounted for.

\textbf{20.} The launch height $H = \qty{24}{m}$ fixed the loop's
safety, every speed $\sqrt{2g(H - z)}$, the \omterm{def:g10:inertia:inventory}{friction} bill and the brake
\omterm{def:g10:inertia:force}{force}; the price of ignoring \omterm{def:g10:inertia:inventory}{friction} is the margin --- the $\qty{4}{m}$
of hill that the rubbing world quietly consumes.
\end{solution}
