\chapter{Color y fuentes de luz}\label{ch:g11:color-light-sources}

Pregúntele a un físico por el color de un tomate y obtendrá una contrapregunta:
¿Bajo qué luz? El tomate solo \omterm{def:g11:color-light-sources:objectcolor}{difunde} lo que se le da, la lamp.
sólo da lo que contiene \omterm{def:g10:light-spectra:white}{espectro}, y el ojo comprime lo que sea
llega a tres números. El color es una negociación entre fuente,
Objeto y ojo: este capítulo interroga a las tres partes por turno.

\section{Ver en tres números}

\begin{definition}[Conos y visión tricromática]\label{def:g11:color-light-sources:cones}
El \omterm{def:g11:lenses-and-eye:eye}{retina} (\cref{ch:g11:lenses-and-eye}) lleva dos familias de
células sensibles a la luz: las \emph{varillas}\index{vara}, que responden a
luz tenue pero ignora \omterm{def:g10:light-spectra:wavelength}{longitud de onda} (visión nocturna, en gris), y el
\emph{conos}\index{cono}, en tres tipos más sensibles cerca
$\qty{420}{nm}$ (azul), $\qty{530}{nm}$ (verde) y $\qty{560}{nm}$
(rojo). La visión humana del color es
\emph{tricromático}\index{visión tricromática}: cualquiera que sea el \omterm{def:g10:light-spectra:white}{espectro},
el cerebro recibe sólo tres respuestas de conos, y el color es lo que
hace de ese triple.
\end{definition}

\begin{definition}[El color como percepción.]\label{def:g11:color-light-sources:perception}
Por lo tanto, el color es un \emph{percepción}, no una propiedad únicamente de la luz:
un \omterm{def:g10:light-spectra:white}{espectro} especifica una intensidad en cada \omterm{def:g10:light-spectra:wavelength}{longitud de onda}, el ojo sigue
tres números. Luces con diferentes espectros pero los mismos tres \omterm{def:g11:color-light-sources:cones}{cono}
las respuestas parecen estrictamente iguales: son \emph{metámeros}\index{metámero}.
\end{definition}

\begin{example}[dos amarillos]\label{ex:g11:color-light-sources:yellows}
La línea $\qty{589}{nm}$ de lamps de sodio excita el rojo y el verde
\omterm{def:g11:color-light-sources:cones}{conos} casi por igual: el cerebro dice \emph{amarillo}. Una iluminación de pantalla
sus subpíxeles rojo ($\qty{610}{nm}$) y verde ($\qty{540}{nm}$) dan
el mismo amarillo sin luz cerca de $\qty{589}{nm}$. Un prisma separa
los dos al instante; el ojo nunca puede hacerlo.
\end{example}

\section{Agregar luces}

\begin{definition}[Síntesis aditiva]\label{def:g11:color-light-sources:additive}
La superposición de haces de luz sobre una pantalla blanca añade sus espectros: esto es
\emph{síntesis aditiva}\index{síntesis aditiva}. Tres bien elegidos
luces, rojo, verde y azul --- el \emph{aditivo
primarias}\index{colores primarios} --- se mezclan en casi todos los perceptibles
color: R $+$ G $=$ amarillo, R $+$ B $=$ magenta, G $+$ B $=$ cian y
R $+$ G $+$ B $=$ blanco.
\end{definition}

\begin{proposition}[Colores complementarios]\label{prop:g11:color-light-sources:complementary}
Dos luces cuya suma se percibe como blanca son
\emph{complementario}\index{colores complementarios}. los tres pares
amarillo/azul, cian/rojo y magenta/verde son complementarios.
\end{proposition}

\begin{proof}
El amarillo es R $+$ G; agregando azul se completa el triple al blanco. el otro
dos pares son el mismo argumento, uno primario a la vez.
\end{proof}

\begin{example}[Pantallas]\label{ex:g11:color-light-sources:screen}
Cada píxel de la pantalla de un teléfono consta de tres subpíxeles: rojo, verde, azul,
bajo $\qty{0.1}{mm}$ de ancho --- demasiado cerca para que el ojo se separe, por lo que
sus luces se agregan al \omterm{def:g11:lenses-and-eye:eye}{retina}: el amarillo es R y G encendido, el naranja es R en
\omterm{def:g10:energy-conservation:power}{fuerza} completo y G a la mitad, el blanco son los tres. Una lupa sobre un
El área blanca revela el truco: no hay blanco en ninguna parte, solo R, G, B.
\end{example}

\section{Restando de la luz blanca}

\begin{definition}[Síntesis sustractiva]\label{def:g11:color-light-sources:subtractive}
A \emph{filtrar}\index{filtrar} (vidrio coloreado, un gel, una capa de tinta o
pintura) transmite parte del \omterm{def:g10:light-spectra:white}{espectro} que recibe y absorbe el resto.
Producir colores mediante bandas \emph{eliminando} de \omterm{def:g10:light-spectra:white}{luz blanca} es
\emph{síntesis sustractiva}\index{síntesis sustractiva}. su
primarias, complementarias de las aditivas, cada una resta una banda:
cian $=$ blanco $-$ R, magenta $=$ blanco $-$ G, amarillo $=$ blanco $-$ B.
Los filtros apilados restan conjuntamente: cian $+$ hojas amarillas verdes, cian
$+$ azul magenta, magenta $+$ amarillo rojo, los tres negros. Impresión
y pintar de esta manera, con tintas C, M, Y.
\end{definition}

\begin{method}[Contabilidad del espectro]\label{met:g11:color-light-sources:bookkeeping}
Para predecir un color a través de cualquier cadena de fuentes, \omterm{def:g11:color-light-sources:subtractive}{filtros} y objetos,
rastrear las tres bandas principales: escribir la luz de la fuente como su contenido
en R, G, B; en cada \omterm{def:g11:color-light-sources:subtractive}{filtrar} o superficie, eliminar las bandas absorbidas; nombre
el color de lo que sobrevive (no queda nada $=$ negro).
\end{method}

\begin{example}[Una cadena de filtro]\label{ex:g11:color-light-sources:chain}
\omterm{def:g10:light-spectra:white}{luz blanca} $(R, G, B)$ cruza una franja amarilla \omterm{def:g11:color-light-sources:subtractive}{filtrar}: la franja azul es
absorbido, dejando $(R, G)$ --- amarillo. Luego, un cian \omterm{def:g11:color-light-sources:subtractive}{filtrar} absorbe el
banda roja, quedando $(G)$: verde. Cambiar el \omterm{def:g11:color-light-sources:subtractive}{filtros} no cambia nada:
cada uno absorbe su propia banda sin importar el orden.
\end{example}

\section{El color de un objeto}

\begin{definition}[Difusión y color de objeto.]\label{def:g11:color-light-sources:objectcolor}
Un objeto opaco absorbe parte de la luz que recibe y
\emph{difunde}\index{difusión} el resto --- lo reemite en cada
dirección, hacia nuestros ojos entre otros. El \emph{color de un
objeto}\index{color de un objeto} es el color de la luz que difunde
\emph{bajo la iluminación dada}: el objeto suministra su absorción
hábitos, la fuente la materia prima. Cambia la lamp, cambia el color.
\end{definition}

\begin{example}[La camisa roja]\label{ex:g11:color-light-sources:redshirt}
Una camisa parece roja a la luz del día: absorbe las bandas verdes y azules y
\omterm{def:g11:color-light-sources:objectcolor}{difunde} el rojo. Bajo una luz verde pura no hay una banda roja para
difusa: la camiseta lo absorbe todo y aparece negra. bajo magenta
enciende $(R, B)$ it \omterm{def:g11:color-light-sources:objectcolor}{difunde} la banda roja sola: todavía roja. un blanco
objeto \omterm{def:g11:color-light-sources:objectcolor}{difunde} cada banda que recibe (bajo luz verde parece
verde); un objeto negro los absorbe a todos bajo cada luz --- lo cual es
También es por eso que la ropa negra se calienta con el sol.
\end{example}

\section{Fuentes de luz y temperatura de color.}

\begin{definition}[fuentes incandescentes]\label{def:g11:color-light-sources:incandescent}
Una fuente que brilla porque está caliente --- llama, filamento, estrella --- brilla
por \emph{incandescencia}\index{incandescencia}. Su \omterm{def:g10:light-spectra:white}{espectro} es el
térmica continua \omterm{def:g10:light-spectra:white}{espectro} de \cref{ch:g10:light-spectra}, y su pico
obedece la ley de Wien (\cref{prop:g10:light-spectra:wien}),
\[
\lambda_{\max} \times T = \qty{2.90e-3}{m.K},
\]
nuestra herramienta cuantitativa a partir de ahora: pico medido, temperatura conocida.
\end{definition}

\begin{example}[Una lamp muy ineficiente]\label{ex:g11:color-light-sources:bulb}
Un filamento de tungsteno en $T = \qty{2700}{K}$ alcanza su punto máximo en
$\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{2700}{K} = \qty{1.07e-6}{m}
= \qty{1070}{nm}$, en lo profundo del \omterm{def:g10:light-spectra:wavelength}{infrarrojo}: la mayor parte del \omterm{def:g10:energy-conservation:power}{fuerza} eléctrico
sale como radiación invisible, sólo $5\%$ como luz --- el
La bombilla incandescente era principalmente un calentador.
\end{example}

\begin{definition}[Lámparas espectrales y láseres]\label{def:g11:color-light-sources:lamps}
Un gas \emph{lamp espectral}\index{lamp espectral} --- un gas \omterm{def:g10:pressure:pressure}{baja presión}
excitado por una descarga: sodio, mercurio, neón --- emite un \omterm{def:g10:light-spectra:emission}{espectro de líneas}
(\cref{ch:g10:light-spectra}): algunos aislados \omterm{def:g10:light-spectra:wavelength}{longitudes de onda}, nada en
entre. Un \emph{láser}\index{láser} va más allá: un solo
\omterm{def:g10:light-spectra:wavelength}{longitud de onda}, \omterm{def:g10:light-spectra:white}{luz monocromática}, en un haz estrecho.
\end{definition}

\begin{definition}[CONDUJO]\label{def:g11:color-light-sources:led}
Un \emph{diodo emisor de luz}\index{diodo emisor de luz} (LED) emite un
\emph{casi monocromático}\index{casi monocromático} banda de unas pocas decenas de nanómetros de ancho: estrecha
suficiente para lucir un color definido, mucho más amplio que una línea \omterm{def:g11:color-light-sources:lamps}{láser}. un blanco
El LED es un LED azul (pico cerca de $\qty{450}{nm}$) recubierto con una
\emph{fósforo}\index{fósforo}, un material que absorbe parte del azul
y reemitiéndolo como una amplia banda amarilla: azul y amarillo siendo
complementario (\cref{prop:g11:color-light-sources:complementary}), el
la mezcla se lee como blanca.
\end{definition}

\begin{definition}[temperatura de color]\label{def:g11:color-light-sources:colortemp}
El \emph{temperatura de color}\index{temperatura de color} de un blanquecino
La fuente es la temperatura del cuerpo incandescente cuyo tono emite su luz.
coincidencias: vela $\qty{1800}{K}$, bombilla incandescente $\qty{2700}{K}$,
halógeno $\qty{3400}{K}$, sol de mediodía $\qty{5500}{K}$, luz diurna cubierta
$\qty{6500}{K}$. Es una etiqueta de color, no la lectura de un termómetro.
$\qty{6500}{K}$ El LED funciona a temperatura ambiente --- y el vocabulario se ejecuta
al revés: la temperatura de color baja indica naranja "cálido", la temperatura alta "fría"
azulado.
\end{definition}

\section{Problema: color de ingeniería}

\begin{problem}[{Problema de fin de semana --- luces del escenario, calles de sodio y
el LED blanco: cómo se diseña el color, a partir de tres focos que pueden
pinta cualquier tono al chip que ilumina esta página}]
\label{pb:g11:color-light-sources:1}
Un diseñador de iluminación sólo posee focos rojos, verdes y azules; una calle
el ingeniero posee una lamp que emite un solo \omterm{def:g10:light-spectra:wavelength}{longitud de onda}; un fabricante de chips posee
un diodo que solo brilla en azul. Los tres venden color. Este problema se ejecuta
sus oficios a su vez: mezcla aditiva en el escenario, economía monocromática en
la calle, \omterm{def:g11:color-light-sources:led}{fósforo} alquimia en el chip --- y termina eligiendo el
Luz para un dormitorio y para un escaparate de joyería.

\medskip
\noindent\textbf{Parte I --- The stage: painting with three spotlights.}
\begin{enumerate}
  \item Los tres puntos se superponen sobre un fondo blanco. Nombra el color de
        cada superposición por pares y de la superposición triple.
  \item ¿Cómo hace el diseñador el naranja? ¿Y un rosa pálido? (Describir
        el lugar \omterm{def:g10:energy-conservation:power}{potestades} cualitativamente.)
  \item Una bailarina se encuentra ante el telón de fondo, iluminado por el rojo y el verde.
        solo manchas. Aparecen dos sombras de colores: ¿color de cada una? Explicar.
  \item El fondo está repintado de azul intenso. ¿Cómo se ve?
        ¿Solo bajo la mancha roja? ¿Bajo los tres?
  \item ¿Por qué tres manchas pueden reproducir "casi todos" los colores? cual
        ¿Un trozo de anatomía arregla el número tres?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Costumes: subtraction at work.}
\begin{enumerate}[resume]
  \item Un disfraz es amarillo en \omterm{def:g10:light-spectra:white}{luz blanca}. ¿Qué banda se tiñe?
        absorber?
  \item Predice su color debajo de la mancha roja, la mancha verde, el azul
        lugar.
  \item El diseñador quiere que el disfraz cambie de brillante a negro en
        un solo movimiento de un interruptor. ¿A qué lugar cambia la película?
  \item Se desliza un gel magenta delante de un punto blanco que ilumina la
        traje amarillo. Seguimiento de las bandas
        (\cref{met:g11:color-light-sources:bookkeeping}): ¿color del traje?
  \item ¿Por qué los geles de escenario vienen en cian, magenta y amarillo en lugar de
        rojo, verde y azul? (¿Qué le haría un gel rojo a una mancha blanca?
        \omterm{def:g10:energy-conservation:power}{fuerza}?)
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Sodium streets: the monochrome economy.}
\begin{enumerate}[resume]
  \item Una lamp de sodio \omterm{def:g10:pressure:pressure}{baja presión} emite esencialmente una línea a
        $\qty{589}{nm}$. ¿Qué tipo de \omterm{def:g10:light-spectra:white}{espectro} es este y qué hace?
        ¿Se ve toda la calle debajo?
  \item Debajo de esta lamp, da el color aparente de una pared blanca, una
        Señalización vial amarilla, un auto rojo, un auto azul.
  \item ¿Por qué a los astrónomos les encantaban estas lamps? (¿Qué puede hacer uno \omterm{def:g11:color-light-sources:subtractive}{filtrar}?
        al resplandor de toda la ciudad?)
  \item Se propone en su lugar una bombilla de tungsteno en $\qty{2700}{K}$.
        Calcule su $\lambda_{\max}$: ¿adónde va la mayor parte de su \omterm{def:g10:energy-conservation:power}{fuerza}?
  \item La lamp de sodio convierte aproximadamente $30\%$ de su \omterm{def:g10:energy-conservation:power}{fuerza} eléctrica
        en luz visible, la bombilla de tungsteno aproximadamente $5\%$. cuantos
        ¿\omterm{def:g10:energy-conservation:power}{vatios} de luz entrega cada uno por $\qty{100}{W}$ de \omterm{def:g10:energy-conservation:power}{fuerza}?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The white LED and the choice of a light.}
\begin{enumerate}[resume]
  \item Un LED blanco es un diodo azul ($\qty{450}{nm}$) debajo de un amarillo
        \omterm{def:g11:color-light-sources:led}{fósforo}. Explique, con \omterm{prop:g11:color-light-sources:complementary}{colores complementarios}, por qué su salida.
        se lee como blanco.
  \item Describe su \omterm{def:g10:light-spectra:white}{espectro} y compáralo con el \omterm{def:g10:light-spectra:continuous}{continuo
        espectro} de luz diurna. ¿Qué banda está subrepresentada?
  \item Deduzca qué objetos un LED de este tipo se reproduce mal y conéctelos
        esto con el carnicero de \cref{exo:g11:color-light-sources:14}.
  \item Una bombilla LED se vende como ``$\qty{2700}{K}$''. ¿Hay algo dentro?
        ¿Está en $\qty{2700}{K}$? ¿Qué promete la etiqueta? Calcular el
        $\lambda_{\max}$ del cuerpo incandescente que imita.
  \item Finale --- elige, con una frase de justificación cada uno, una
        \omterm{def:g11:color-light-sources:colortemp}{temperatura de color} para lamp de dormitorio y otra para joyería
        ventana; luego, indique la ley del color del capítulo en una línea.
\end{enumerate}
\end{problem}
