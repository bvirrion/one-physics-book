\chapter{Fuerzas y movimiento}\label{ch:g11:forces-and-motion}

Lanza una llave inglesa que gira por la habitación: los extremos se tambalean violentamente, pero
un punto en su interior traza una parábola perfecta. \Cref{ch:g10:inertia}
resolvió qué sucede cuando \omterm{def:g10:inertia:force}{efectivo} \omterm{def:g10:inertia:compensate}{compensar} --- nada. este capitulo
pregunta qué sucede cuando no lo hacen: los restos \omterm{def:g10:inertia:force}{fuerza} \emph{cambios

--- y dice cuánto, en proporciones si aún no en fórmulas.

\section{El centro de masa}

\begin{definition}[centro de masas]\label{def:g11:forces-and-motion:com}
El \emph{centro de masa}\index{centro de masa} de un cuerpo o sistema es
su punto de equilibrio --- la posición promedio de su masa: la geometría
centro de un cuerpo simétrico homogéneo; para dos masas puntuales, el punto
del segmento de unión a distancias en la relación \emph{inverso} del
masas, $\frac{d_1}{d_2} = \frac{m_2}{m_1}$ --- más cerca de las más pesadas.
\end{definition}

\begin{example}[Una mancuerna torcida]\label{ex:g11:forces-and-motion:dumbbell}
Las bolas de \qty{2.0}{kg} y \qty{1.0}{kg} se encuentran en los extremos de una barra de luz.
\qty{0.90}{m} de largo. El \omterm{def:g11:forces-and-motion:com}{centro de masa} lo divide en razón inversa.
$d_1/d_2 = 1.0/2.0$: $\qty{0.30}{m}$ de la bola pesada,
$\qty{0.60}{m}$ desde el ligero --- equilibra la barra allí.
\end{example}

\begin{remark}[La llave arrojada]\label{rem:g11:forces-and-motion:wrench}
Filme una llave inglesa: cada punto sigue una curva complicada
--- excepto el \omterm{def:g11:forces-and-motion:com}{centro de masa}, que traza la parábola limpia un lanzado
la pelota lo haría. El movimiento del \omterm{def:g11:forces-and-motion:com}{centro de masa} es simple; el resto es giro
\emph{alrededor} eso --- por eso los mecánicos pueden tratar un automóvil, un planeta o
una gimnasta como un solo punto.
\end{remark}

\section{Una fuerza neta cambia la velocidad.}

\begin{definition}[fuerza neta]\label{def:g11:forces-and-motion:netforce}
El \emph{fuerza neta}\index{fuerza neta} en un cuerpo es la suma vectorial de
todo el \omterm{def:g10:inertia:force}{efectivo} actuando sobre él, $\vect F = \vect F_1 + \vect F_2 + \dots$
(de punta a cola, como en el volumen de matemáticas); el \omterm{def:g10:inertia:force}{efectivo} \omterm{def:g10:inertia:compensate}{compensar}
exactamente cuando $\vect F = \vect 0$.
\end{definition}

\begin{theorem}[Efecto de una fuerza neta --- semicuantitativa]\label{thm:g11:forces-and-motion:secondlaw}
\index{segunda ley del movimiento}
Un \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} $\vect F$ que actúa sobre un cuerpo de masa $m$ por un corto período de tiempo.
$\Delta t$ cambia la velocidad de su \omterm{def:g11:forces-and-motion:com}{centro de masa} por un vector
$\Delta \vect v$ que tiene el \emph{dirección} de $\vect F$, y cuyo
magnitud crece en proporción a $F$ y a $\Delta t$ y se contrae en
proporción a la masa:
\[
\Delta v \propto \frac{F \, \Delta t}{m} .
\]
\end{theorem}
\admitted

\begin{remark}
Duplicar el \omterm{def:g10:inertia:force}{fuerza} o su duración: el doble del cambio; duplicar la masa:
la mitad. El año que viene afina esto en una ecuación exacta, y el Año 1
el volumen pregunta en qué cuadros se sostiene; este año los ratios son suficientes ---
Ya funcionan airbags, frenos y \omterm{def:g10:universal-gravitation:satellite}{órbitas}.
\end{remark}

\begin{proposition}[Los tres casos canónicos]\label{prop:g11:forces-and-motion:cases}
Dejemos que un cuerpo se mueva con velocidad $\vect v$ bajo un \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} $\vect F$.
\begin{enumerate}
  \item $\vect F$ a lo largo de $\vect v$: es \emph{acelera} en línea recta;
  \item $\vect F$ frente a $\vect v$: es \emph{se ralentiza} en línea recta;
  \item $\vect F$ perpendicular a $\vect v$: \emph{vueltas} sin
        cambiando de velocidad.
\end{enumerate}
\end{proposition}

\begin{proof}
Agregue $\Delta\vect v$, dirigido a lo largo de $\vect F$
(\cref{thm:g11:forces-and-motion:secondlaw}), punta a cola sobre
$\vect v$: a lo largo de $\vect v$ las flechas se acumulan (más largas); opuesto, ellos
cancelar parcialmente (más corto); perpendicular, cada uno pequeño $\Delta \vect v$
inclina la flecha sin alargarla --- solo gira la dirección.
\end{proof}

\begin{example}[Una patada lateral]\label{ex:g11:forces-and-motion:kick}
Un disco se desliza hacia el este en \qty{6.0}{m/s}; un breve golpe lateral añade
$\Delta\vect v = \qty{8.0}{m/s}$ hacia el norte. La nueva velocidad es la
suma vectorial: magnitud $\sqrt{6.0^2 + 8.0^2} = \qty{10.0}{m/s}$,
dirección $\tan\theta = 8.0/6.0$, es decir\ $\theta \approx 53^\circ$ norte
de este --- la antigua velocidad no se borra, se le añade el cambio.
\end{example}

\section{Comparar cambios: fuerza, tiempo, masa.}

\begin{method}[Razonamiento de razón]\label{met:g11:forces-and-motion:ratio}
Para comparar dos situaciones, nunca resuelvas una ecuación --- forma razones: la
La relación de cambio de velocidad es la relación $F$, multiplicada por la relación $\Delta t$,
dividido por la relación $m$. Luego lea qué factor cambió y en qué medida.
\end{method}

\begin{example}[El carro cargado]\label{ex:g11:forces-and-motion:cart}
El mismo empujón, aplicado durante el mismo segundo, envía un mensaje vacío.
\qty{20}{kg} carrito de compras apagado en \qty{1.2}{m/s}. Cargado a
\qty{60}{kg} --- triple masa, igual $F\Delta t$ --- sale a una
tercio de eso, \qty{0.40}{m/s}. La masa es la terquedad de la materia.
\end{example}

\begin{remark}[Choques, airbags, rodillas dobladas]\label{rem:g11:forces-and-motion:airbag}
En un accidente $\Delta v$ y $m$ no son negociables --- el valor del ocupante
la velocidad debe llegar a cero --- por lo que $F\,\Delta t$ es fijo. la unica libertad
a la izquierda está \emph{tiempo}: se estira $\Delta t$ y el \omterm{def:g10:inertia:force}{fuerza} se encoge en el
misma proporción. Una cabeza que se detiene en un airbag en \qty{0.08}{s} en lugar de en
la rueda en \qty{0.01}{s} se siente en promedio \omterm{def:g10:inertia:force}{fuerza} ocho veces más pequeña;
Las zonas de deformación, la espuma del casco y las rodillas dobladas juegan el mismo truco.
\end{remark}

\section{Fricción: la fuerza que combate el deslizamiento}

\begin{definition}[Fricción estática y cinética]\label{def:g11:forces-and-motion:friction}
\omterm{def:g10:inertia:inventory}{Fricción} entre superficies en contacto se opone a su \emph{relativo
corredera}. \emph{Fricción estática}\index{fricción!estático} actúa mientras
no se desliza, ajustándose al máximo para anular cualquier intento
para iniciar la diapositiva; \emph{fricción cinética}\index{fricción!cinético},
una vez que ha comenzado el deslizamiento, es aproximadamente constante, dirigido contra la
deslizante, y generalmente \emph{más débil} que el máximo estático.
\end{definition}

\begin{example}[Empujando el armario]\label{ex:g11:forces-and-motion:wardrobe}
Empujar un armario suavemente: no se mueve --- estática \omterm{def:g10:inertia:inventory}{fricción} anula
su \qty{150}{N}. Empuje más fuerte: en algún momento \qty{400}{N} se tambalea
movimiento, luego se desliza con menos esfuerzo --- el más débil \omterm{def:g11:forces-and-motion:friction}{fricción cinética}
ha tomado el relevo.
\end{example}

\begin{remark}[¿Por qué funcionan los frenos ABS?]\label{rem:g11:forces-and-motion:abs}
El punto de contacto de una rueda en movimiento no se desliza sobre la carretera: el neumático
empuñaduras con \emph{estático} \omterm{def:g10:inertia:inventory}{fricción}. Una rueda bloqueada patina y sólo se pone
el \emph{cinético} \omterm{def:g10:inertia:inventory}{fricción} más débil, que apunta en contra del patinaje
Hagan lo que hagan las ruedas, la dirección muere. El ABS libera el freno
Una rueda se bloquea al instante: más agarre, parada más corta, un conductor que puede conducir.
\end{remark}

\section{Girar: el movimiento circular necesita una fuerza hacia adentro}

\begin{definition}[movimiento circular uniforme]\label{def:g11:forces-and-motion:ucm}
Un cuerpo está en \emph{movimiento circular uniforme}\index{circular uniforme
movimiento} cuando su \omterm{def:g10:relative-motion:trajectory}{trayectoria} es un círculo recorrido con rapidez constante. el
La velocidad, tangente al círculo, cambia de dirección en cada instante.
por lo que nunca es constante.
\end{definition}

\begin{proposition}[La fuerza hacia adentro]\label{prop:g11:forces-and-motion:inward}
Un cuerpo en \omterm{def:g11:forces-and-motion:ucm}{movimiento circular uniforme} está sujeto a un \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} que es
nunca cero: en cada instante apunta desde el cuerpo hacia el centro
del círculo.
\end{proposition}

\begin{proof}
La velocidad es constante, por lo que el \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} no tiene pieza a lo largo del $\vect v$.
(casos 1 y 2 de \cref{prop:g11:forces-and-motion:cases}): es
perpendicular a la tangente, por lo tanto a lo largo del radio --- y hacia adentro,
ya que cada $\Delta \vect v$ dobla el camino hacia el centro.
\end{proof}

\begin{example}[La cuerda, la curva, la Luna.]\label{ex:g11:forces-and-motion:moon}
Cada giro constante esconde un \omterm{def:g10:inertia:force}{fuerza} hacia adentro. La bola girada: la cuerda
\omterm{def:g10:inertia:inventory}{tensión}. El coche en una curva: la estática lateral \omterm{def:g10:inertia:inventory}{fricción} de la carretera en
neumáticos --- se pusieron hielo y el auto va derecho. La Luna: la Tierra.
atracción gravitacional (\cref{ch:g10:universal-gravitation}), girándolo
alrededor en \num{27.3} días --- no sostenido \emph{up} pero tirado hacia un lado en un
giro perpetuo, cayendo alrededor de la Tierra; el año que viene hace que este sea el
mecánica de \omterm{def:g10:universal-gravitation:satellite}{satélites}.
\end{example}

el promedio \omterm{def:g10:inertia:force}{efectivo}. Explica, con el mismo razonamiento, por qué doblas tu
rodillas al realizar un salto y de qué está hecha la colchoneta de un gimnasta.
\end{exercise}

\section{Problema: la prueba de choque}

\begin{problem}[{Problema del fin de semana --- diseñando la colisión suave:
Dado que un choque soluciona el cambio de velocidad, todos los dispositivos de seguridad en un automóvil
es una máquina para estirar el tiempo}]
\label{pb:g11:forces-and-motion:1}
Un laboratorio de pruebas de choque conduce un automóvil en \qty{50}{km/h} contra una vía rígida.
pared, con un chupete de adulto \qty{70}{kg} y un chupete de niño \qty{20}{kg}
a bordo; Las cámaras miden cada parada. Tu trabajo: encontrar dónde se encuentra la violencia de un
estrellar vidas y diseñarlas para eliminarlas.

\medskip
\noindent\textbf{Parte I --- The measure of a crash.}
\begin{enumerate}
  \item Convierte \qty{50}{km/h} en \unit{m/s}.
  \item En cualquier parada frontal el cambio de velocidad de los ocupantes tiene el mismo
        magnitud. ¿Cuál? ¿Por qué ningún dispositivo puede alterarlo?
  \item Cita \cref{thm:g11:forces-and-motion:secondlaw}: con $\Delta v$
        y $m$ fijo, $F\,\Delta t$ es fijo. ¿La única esfera del diseñador?
  \item El muro detiene el auto en \qty{0.050}{s}; una zona de deformación
        extiende esto a \qty{0.150}{s}. Compare el promedio \omterm{def:g10:inertia:force}{efectivo}.
  \item ¿En qué dirección están $\Delta \vect v$ y \omterm{def:g11:forces-and-motion:netforce}{fuerza neta} en
        ¿El punto de los muñecos durante la parada?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The crumple zone.}
\begin{enumerate}[resume]
  \item ¿Por qué, en una frase, los ingenieros diseñan la parte delantera del coche?
        ser destruido?
  \item En un coche rígido antiguo (parada en \qty{0.050}{s}) el chupete está atado
        rígidamente al asiento. Compare su \omterm{def:g10:inertia:force}{fuerza} con el moderno tope \qty{0.150}{s}.
  \item La velocidad del coche que se detiene disminuye constantemente (promedio: la mitad de la inicial).
        ¿Qué distancia recorre en \qty{0.150}{s}? Compárese con una zona de deformación real.
  \item ¿Por qué no una zona de deformación \qty{10}{m}? ¿Y qué parte del auto?
        ¿\emph{no} debe arrugarse?
  \item Resuma la Parte II como un lema: ``no puedes elegir $\Delta v$,
        entonces debes elegir \dots''.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The belt, the airbag and the child.}
\begin{enumerate}[resume]
  \item Un chupete sin cinturón mantiene su \qty{14}{m/s} (\cref{ch:g10:inertia})
        hasta que el parabrisas lo detenga en \qty{0.005}{s}; uno con cinturón
        se detiene con el coche en \qty{0.150}{s}. Compare el promedio \omterm{def:g10:inertia:force}{efectivo}.
  \item ¿Por qué un cinturón que se estira ligeramente mejor que uno?
        ¿Arnés de acero inestirable?
  \item El cabezal se pararía sobre la rueda en \qty{0.010}{s}; el airbag
        extiende esto a \qty{0.080}{s}. Compare el \omterm{def:g10:inertia:force}{efectivo}.
  \item Adulto y niño pasan por el mismo $\Delta v$ en el mismo
        $\Delta t$. Compare el \omterm{def:g11:forces-and-motion:netforce}{fuerzas netas} que deben suministrar las dos correas.
  \item Un niño en el regazo: los brazos deben cambiar los de un bebé \qty{10}{kg}
        velocidad por \qty{14}{m/s} en \qty{0.150}{s}, mientras que la gravedad
        cambia una velocidad en \qty{9.8}{m/s} cada segundo. cuantas veces
        más rápido deben actuar los brazos --- sosteniendo qué masa sentiría que
        pesado? Concluir.
  \item Los niños se sienten más pequeños \omterm{def:g10:inertia:force}{efectivo} --- Entonces, ¿por qué sillas para niños? pensar en
        \emph{dónde} el cinturón aplica su \omterm{def:g10:inertia:force}{fuerza}, y de $\Delta t$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The safety cage, and the verdict.}
\begin{enumerate}[resume]
  \item El habitáculo está rígido y ambos extremos se arrugan. ¿Por qué debe
        ¿La celda no se deforma, aunque la rigidez significa paradas violentas?
  \item Dos coches idénticos chocan frontalmente, cada uno en \qty{50}{km/h}.
        ¿Dónde se detiene cada uno? Compara cada choque con la prueba de la pared.
  \item Un coche pequeño se encuentra de frente con un camión cargado. El \omterm{def:g10:inertia:force}{efectivo} que ejercen
        unos sobre otros son iguales y opuestos, mutuos como toda interacción
        es (\cref{ch:g11:fundamental-interactions}). Cuya velocidad
        ¿cambia más?
  \item Listar los dispositivos de seguridad de este problema y la cantidad única
        cada uno de ellos se estira.
  \item Final: establece la ley de diseño de la colisión suave en uno
        oración usando solo \emph{masa}, \emph{cambio de velocidad},
        \emph{\omterm{def:g10:inertia:force}{fuerza}} y \emph{tiempo}.
\end{enumerate}
\end{problem}
