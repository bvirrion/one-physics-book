\section{Carga en movimiento}

\begin{definition}[Corriente y culombio]\label{def:g11:circuits-and-power:current}
Un \emph{corriente eléctrica}\index{corriente eléctrica} es un flujo ordenado de
carga (electrones, en un metal). Su intensidad $I$, en
\emph{amperios}\index{amperio} (\unit{A}), es la carga que pasa por un
sección transversal del cable por segundo: una corriente constante $I$ que fluye durante un
momento $t$ transporta la carga
\[ q = I\,t , \]
en \emph{culombios}\index{culombio} (\unit{C}), con $\qty{1}{C} =
\qty{1}{A.s}$. La carga elemental es$e = \qty{1.6e-19}{C}$. Actual
se mide con un ammeter, insertado en serie.
\end{definition}

\begin{example}[Rayo versus teléfono]\label{ex:g11:circuits-and-power:coulomb}
Un rayo transporta algo de $\qty{3.0e4}{A}$ durante aproximadamente
$\qty{1.0e-4}{s}$: $q = 3.0\times10^{4} \times 10^{-4} = \qty{3.0}{C}$.
Tu teléfono, en \qty{2.0}{A}, mueve la misma carga cada $1.5$ segundos.
\end{example}

\begin{definition}[Voltaje y voltios]\label{def:g11:circuits-and-power:voltage}
El \emph{Voltaje}\index{Voltaje} $U$ entre dos puntos de un circuito, en
\emph{voltios}\index{voltio} (\unit{V}), ¿se intercambia el \omterm{def:g10:energy-conservation:energy}{energía} por \omterm{def:g10:orders-of-magnitude:unit}{unidad} de
cargo que viaja entre ellos: $\qty{1}{V} = \qty{1}{J/C}$, por lo que un cargo
$q$ cruzando un dispositivo bajo tensión $U$ intercambia el \omterm{def:g10:energy-conservation:energy}{energía} $E = qU$
con eso. Un voltmeter, conectado en paralelo, lo mide.
\end{definition}

\begin{proposition}[Leyes del circuito]\label{prop:g11:circuits-and-power:laws}
En un circuito que funciona de manera constante:
\begin{enumerate}
  \item \emph{ley de unión:} las corrientes que llegan a un cruce se suman
        a las corrientes que lo abandonan;
  \item \emph{ley de la serie:} \omterm{def:g11:circuits-and-power:voltage}{voltajes} a lo largo de una ruta se suman:
        $U_{AC} = U_{AB} + U_{BC}$;
  \item \emph{ley paralela:} dos ramas que unen los mismos dos puntos
        llevar el mismo \omterm{def:g11:circuits-and-power:voltage}{Voltaje}.
\end{enumerate}
\end{proposition}

\begin{proof}
En un estado estacionario la carga no se acumula en ninguna parte, por lo que lo que fluye hacia un
la unión fluye. \omterm{def:g10:energy-conservation:energy}{Energía} por \omterm{def:g11:circuits-and-power:current}{culombio} es aditivo a lo largo de una ruta y
depende sólo de sus puntos finales, no de la rama tomada entre ellos.
\end{proof}

\section{Ley de Ohm y redes de resistencias.}

\begin{definition}[Resistencia]\label{def:g11:circuits-and-power:resistance}
El \emph{resistencia}\index{resistencia} de un conductor es la relación
$R = U/I$ del \omterm{def:g11:circuits-and-power:voltage}{Voltaje} a través de él a la corriente que lo atraviesa, medida en
\emph{ohmios}\index{ohm} (\unit{\ohm}): $\qty{1}{\ohm} = \qty{1}{V/A}$.
\end{definition}

\begin{proposition}[Ley de Ohm]\label{prop:g11:circuits-and-power:ohm}
\index{ley de ohm}
Para un conductor metálico mantenido a temperatura fija, $R$ es constante:
\omterm{def:g11:circuits-and-power:voltage}{Voltaje} y la corriente son proporcionales,
\[ U = R\,I . \]
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:circuits-and-power:ohm-limits}
Una ley experimental, no universal: un diodo, una lamp calefactora
filamento o su propia piel no lo obedecen. ¿Por qué lo hacen los metales? Se deriva del
volúmenes universitarios.
\end{remark}

\begin{proposition}[Resistencias en serie y paralelo]\label{prop:g11:circuits-and-power:networks}
\index{resistencia equivalente}
Dos resistencias en serie equivalen a una resistencia $R_{\text{s}} =
R_1 + R_2$; dos en paralelo, a una resistencia$R_{\text{p}}$ dada por
\[
\frac{1}{R_{\text{p}}} = \frac{1}{R_1} + \frac{1}{R_2} ,
\qquad\text{i.e.}\quad
R_{\text{p}} = \frac{R_1 R_2}{R_1 + R_2} < \min(R_1, R_2) .
\]
\end{proposition}

\begin{proof}
Serie: la misma $I$ cruza ambas y \omterm{def:g11:circuits-and-power:voltage}{voltajes} suma
(\cref{prop:g11:circuits-and-power:laws}), entonces $U = (R_1 + R_2)\,I$.
Paralelo: ambos se encuentran bajo el mismo $U$ y las corrientes se suman, por lo que
$I = \frac{U}{R_1} + \frac{U}{R_2}$.
\end{proof}

\begin{example}[Resistencia equivalente]\label{ex:g11:circuits-and-power:networks}
$R_1 = \qty{100}{\ohm}$ y $R_2 = \qty{150}{\ohm}$: en serie,
\qty{250}{\ohm}; en paralelo, $\frac{100 \times 150}{250} =
\qty{60}{\ohm}$ --- menos que cualquiera de los dos: la corriente toma un segundo camino.
\end{example}

\section{Energía y potencia en un circuito.}

\begin{proposition}[Energía electrica]\label{prop:g11:circuits-and-power:power}
\index{energía eléctrica}
Un dispositivo bajo \omterm{def:g11:circuits-and-power:voltage}{Voltaje} $U$ atravesado por un $I$ actual recibe \omterm{def:g10:energy-conservation:energy}{energía} en
la tasa
\[ P = U\,I , \qquad\text{i.e.}\quad E = U\,I\,t \ \text{in a time } t . \]
\end{proposition}

\begin{proof}
En un tiempo $t$ la carga $q = It$ atraviesa el dispositivo
(\cref{def:g11:circuits-and-power:current}) y cada \omterm{def:g11:circuits-and-power:current}{culombio} entrega
$U$ \omterm{def:g10:energy-conservation:energy}{julios} (\cref{def:g11:circuits-and-power:voltage}): $E = qU = UIt$.
\end{proof}

\begin{proposition}[Potencia en una resistencia]\label{prop:g11:circuits-and-power:ri2}
Una resistencia $R$ que transporta una corriente $I$ se disipa
\[ P = R\,I^2 = \frac{U^2}{R} . \]
\end{proposition}

\begin{proof}
Sustituya \omterm{prop:g11:circuits-and-power:ohm}{ley de ohm} en $P = UI$: $P = (RI)I = RI^2 = U(U/R) = U^2/R$.
\end{proof}

\begin{example}[Leyendo una placa de identificación]\label{ex:g11:circuits-and-power:nameplate}
Una tetera marcada ``\qty{230}{V} -- \qty{2200}{W}'' extrae $I = P/U =
2200/230 = \qty{9.6}{A}$y tiene \omterm{def:g11:circuits-and-power:resistance}{resistencia}$R = U^2/P = 230^2/2200 =
\qty{24}{\ohm}$. Al ejecutar \qty{150}{s}, usa$E = 2200 \times 150 =
\qty{3.3e5}{J} \approx \qty{0.09}{kW.h}$ --- alrededor de dos centavos.
\end{example}

\begin{remark}[Julios y kilovatios-hora]\label{rem:g11:circuits-and-power:kwh}
El medidor cuenta \omterm{def:g10:energy-conservation:kwh}{kilovatios-hora} (\cref{def:g10:energy-conservation:kwh}):
$\qty{1}{kW.h} = \qty{3.6e6}{J}$. Los julios son el \omterm{def:g10:orders-of-magnitude:unit}{unidad} del físico;
\omterm{def:g10:energy-conservation:kwh}{kilovatios-hora} pone precio a las facturas.
\end{remark}

\section{El efecto Joule: de la tostadora al pilón}

\begin{definition}[Efecto Joule]\label{def:g11:circuits-and-power:joule}
La disipación $P = RI^2$ convierte \omterm{def:g10:energy-conservation:forms}{energía eléctrica} en calor en cada
La resistencia es la \emph{efecto julio}\index{efecto julio}. calentadores,
las teteras, tostadoras y bombillas antiguas son resistencias por diseño; un
\emph{fusible}\index{fusible} es un cable \omterm{def:g11:lenses-and-eye:lens}{delgado} calibrado para fundirse --- abriendo el
circuito --- cuando la corriente excede su clasificación.
\end{definition}

\begin{example}[¿Por qué las torres de alta tensión son altas?]\label{ex:g11:circuits-and-power:transmission}
Un pueblo necesita $P = \qty{100}{kW}$ a través de una línea de \omterm{def:g11:circuits-and-power:resistance}{resistencia}
$R_\ell = \qty{5.0}{\ohm}$. Entregado en $U = \qty{500}{V}$: $I = P/U =
\qty{200}{A}$y la línea desperdicia$R_\ell I^2 = \qty{200}{kW}$ --- dos veces
la entrega: absurda. En $U = \qty{20}{kV}$: $I = \qty{5.0}{A}$ y
$R_\ell I^2 = \qty{125}{W}$. Cuarenta veces menos actual, $1600$ veces menos
pérdida.
\end{example}

\begin{remark}[La regla de hierro de la reja.]\label{rem:g11:circuits-and-power:grid}
En \omterm{def:g10:energy-conservation:power}{fuerza} $P = UI$ de entrega fija, la pérdida de línea $R_\ell I^2 = R_\ell
P^2/U^2$ cae como \emph{cuadrado} del transporte \omterm{def:g11:circuits-and-power:voltage}{Voltaje}. Por lo tanto:
transfórmate arriba (\qty{400}{kV}) para viajar, abajo (\qty{230}{V}) para vivir.
\end{remark}

\begin{remark}[seguridad del hogar]\label{rem:g11:circuits-and-power:safety}
Una línea \qty{230}{V} fusionada en \qty{16}{A} suministra como máximo $230 \times 16
\approx \qty{3.7}{kW}$: dos calderas a la vez derriten el \omterm{def:g11:circuits-and-power:joule}{fusible} --- por diseño
el eslabón más delgado y débil, mucho más barato que un incendio dentro del muro.
\end{remark}

\section{Baterías reales: fuerza electromotriz}

\begin{definition}[CEM y resistencia interna]\label{def:g11:circuits-and-power:emf}
Una battery convierte el producto químico en \omterm{def:g10:energy-conservation:forms}{energía eléctrica}. su
\emph{fuerza electromotriz}\index{fuerza electromotriz} (CEM)
$\mathcal{E}$, en \omterm{def:g11:circuits-and-power:voltage}{voltios}, es el \omterm{def:g10:energy-conservation:energy}{energía} que entrega cada \omterm{def:g11:circuits-and-power:current}{culombio} que lo cruza;
sus cuentas \emph{resistencia interna}\index{resistencia interna} $r$
para el \omterm{def:g11:circuits-and-power:joule}{efecto julio} dentro de su propia química. Modelo: una fuente ideal
$\mathcal{E}$ en serie con $r$ (Rizado $\mathcal{E}$: $E$ es \omterm{def:g10:energy-conservation:energy}{energía}).
\end{definition}

\begin{proposition}[Terminal de tensión]\label{prop:g11:circuits-and-power:terminal}
Se muestra una battery que entrega una corriente $I$ en sus terminales
\[ U = \mathcal{E} - r\,I . \]
\end{proposition}

\begin{proof}
La química suministra $\mathcal{E}I$, el \omterm{def:g11:circuits-and-power:emf}{resistencia interna} come
$rI^2$, el circuito recibe $UI$: conservación \omterm{def:g10:energy-conservation:energy}{energía}
(\cref{ch:g10:energy-conservation}) da $UI = \mathcal{E}I - rI^2$.
\end{proof}

\begin{example}[Una battery baja cargada]\label{ex:g11:circuits-and-power:discharge}
Una battery ($\mathcal{E} = \qty{9.0}{V}$, $r = \qty{1.0}{\ohm}$) alimenta un
bombilla de \omterm{def:g11:circuits-and-power:resistance}{resistencia} $R = \qty{8.0}{\ohm}$. El bucle obedece a $\mathcal{E} =
rI + RI$, por lo que$I = \mathcal{E}/(R + r) = \qty{1.0}{A}$y$U = 9.0 -
1.0 \times 1.0 = \qty{8.0}{V}$: la bombilla recibe \qty{8.0}{W} mientras que el
La battery desperdicia \qty{1.0}{W} calentándose.
\end{example}

\begin{remark}[Receptores]\label{rem:g11:circuits-and-power:receivers}
Un motor o una battery en carga realiza la conversión al revés: absorbe
bajo $U = \mathcal{E}' + r'I$, convirtiendo $\mathcal{E}'I$ en movimiento o
química y perdiendo $r'I^2$. Las fuentes presionan, los receptores rechazan.
\end{remark}

\begin{example}[Eficiencia de una battery]\label{ex:g11:circuits-and-power:efficiency}
El \omterm{def:g10:energy-conservation:efficiency}{eficiencia} (\cref{def:g10:energy-conservation:efficiency}) del
descarga arriba: útil $UI = \qty{8.0}{W}$ de $\mathcal{E}I =
\qty{9.0}{W}$, por lo que$\eta = U/\mathcal{E} = 8/9 \approx 0.89$ --- eliminando
a medida que la corriente aumenta: las batterys que trabajan duro se calientan.
\end{example}

\section{Problema: De la presa a la tostadora}

\begin{problem}[{Problema del fin de semana --- el viaje de mil kilómetros de un
Kilovatio-hora, desde la caída del agua hasta el pan dorado, con la factura completa
del viaje}]
\label{pb:g11:circuits-and-power:1}
Una presa de montaña retiene agua \qty{200}{m} por encima de sus turbinas, que tragan
\qty{50}{m^3} --- \qty{5.0e4}{kg} --- por segundo; turbina y alternador
convertir con \omterm{def:g10:energy-conservation:efficiency}{eficiencia} $0.90$. A través de una línea del total \omterm{def:g11:circuits-and-power:resistance}{resistencia}
$R_\ell = \qty{32}{\ohm}$ (\qty{1000}{km} ida y vuelta) la planta se alimenta
una ciudad donde tu tostadora \qty{900}{W} te espera en \qty{230}{V}. el
salidas del alternador \qty{20}{kV}; transformadores (\omterm{def:g10:energy-conservation:efficiency}{eficiencia} $0.99$ cada uno)
convierta \omterm{def:g11:circuits-and-power:voltage}{voltajes} a voluntad. La electricidad se vende a $0.25$ euros por
\unit{kW.h}; $g = \qty{9.81}{N/kg}$.

\medskip
\noindent\textbf{Parte I --- At the dam.}
\begin{enumerate}
  \item ¿Cuánto potencial \omterm{def:g10:energy-conservation:energy}{energía} libera el agua que cae cada vez?
        segundo? Deducir el \omterm{def:g10:energy-conservation:power}{fuerza} del que disponen las turbinas.
  \item ¿Qué \omterm{prop:g11:circuits-and-power:power}{energía eléctrica} sale del alternador?
  \item En \qty{20}{kV} directamente desde el alternador, ¿qué corriente fluye?
  \item Calcule la pérdida en julios $R_\ell I^2$ en esa corriente y compare
        con la producción de la planta. Concluir.
  \item El \omterm{def:g10:energy-conservation:power}{fuerza} $P = UI$ de la ciudad se fija por demanda. ¿Cuál es el único
        ¿Cómo reducir $I$ y qué dispositivo lo realiza?
\end{enumerate}

\noindent\textbf{Parte II --- On the line.}

  \item Calcule la pérdida de la nueva línea, en megavatios.
  \item ¿Qué fracción de la producción de la planta se pierde en la línea?
