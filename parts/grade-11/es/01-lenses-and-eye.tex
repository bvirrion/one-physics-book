\chapter{Lentes, imágenes y el ojo}\label{ch:g11:lenses-and-eye}

Cada imagen nítida, en el sensor de un teléfono, en una pantalla de cine o en la parte trasera
de tu propio ojo --- proviene del mismo truco: una curva transparente
El medio dobla los rayos en \omterm{def:g10:refraction:refraction}{refracción} (\cref{ch:g10:refraction}) para que todos
los rayos que salen de un punto se vuelven a encontrar en otro. Este capítulo domestica
el truco: un \omterm{def:g11:lenses-and-eye:lens}{lente}, tres rayos, una fórmula --- luego apunta a
cámaras, gafas y los vivos \omterm{def:g11:lenses-and-eye:lens}{lente} leyendo esta frase.

\section{Lentes convergentes y divergentes}

\begin{definition}[Lente delgada]\label{def:g11:lenses-and-eye:lens}
Un \emph{lente}\index{lente} es un medio transparente delimitado por dos
superficies, al menos una curva. Más grueso en el centro que en el borde,
es \emph{convergente}\index{lente convergente}: los rayos paralelos están doblados
el uno hacia el otro; más gruesos en el borde, \emph{divergente}\index{divergente
lente}: se separan. Una lente \emph{delgado}\index{lente delgada} tiene
espesor insignificante: dibujamos un segmento con hacia afuera (convergente) o
puntas de flecha hacia adentro (divergentes). Su centro $O$ es el \emph{óptico
centro}\index{centro óptico}; la línea perpendicular que pasa por $O$, la
\emph{eje principal}\index{eje principal}.
\end{definition}

\begin{definition}[Puntos focales y distancia focal]\label{def:g11:lenses-and-eye:focal}
Los rayos que llegan paralelos al eje de un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} cruzan el
eje en un punto detrás de él, el \emph{punto focal de la imagen}\index{focales
punto} $F'$; simétricamente, los rayos salen del \emph{punto focal del objeto}
$F$ (delante, con $OF = OF'$) sale paralela al eje. el
\emph{longitud focal}\index{longitud focal} $f' = \overline{OF'}$ es
positivo para \omterm{def:g11:lenses-and-eye:lens}{lente convergente}; a través de un \omterm{def:g11:lenses-and-eye:lens}{lente divergente}, rayos paralelos
salir como desde un punto $F'$ \emph{Al frente} del \omterm{def:g11:lenses-and-eye:lens}{lente}: $f' < 0$.
\end{definition}

\begin{definition}[Vergencia]\label{def:g11:lenses-and-eye:vergence}
El \emph{vergencia}\index{vergencia} de un \omterm{def:g11:lenses-and-eye:lens}{lente} de \omterm{def:g11:lenses-and-eye:focal}{longitud focal} $f'$
(en metros) es $C = 1/f'$,
medido en \emph{dioptrías}\index{dioptría} (símbolo D):
$1~\text{D} = \qty{1}{m^{-1}}$; Las lentes \omterm{def:g11:lenses-and-eye:lens}{divergente} tienen $C < 0$.
Los ópticos prescriben en dioptrías: vergencias de lentes \omterm{def:g11:lenses-and-eye:lens}{delgado} en contacto
simplemente agregue (admitido aquí; derivado en el volumen del Año 1).
\end{definition}

\begin{example}[Lectura de una receta]\label{ex:g11:lenses-and-eye:vergence}
Un \omterm{def:g11:lenses-and-eye:lens}{lente} con $f' = \qty{0.50}{m}$ tiene $C = 1/0.50 = +2.0$~D:
\omterm{def:g11:lenses-and-eye:lens}{convergente}. Una prescripción $-4.0$~D significa $f' = \qty{-0.25}{m}$:
\omterm{def:g11:lenses-and-eye:lens}{divergente}, \omterm{def:g11:lenses-and-eye:focal}{longitud focal} \qty{25}{cm}.
\end{example}

\section{Construyendo la imagen}

\begin{method}[Los tres rayos de construccion]\label{met:g11:lenses-and-eye:rays}
Sea $AB$ un objeto perpendicular al eje en $A$. Entre todos los rayos
dejando la punta $B$, tres tienen caminos conocidos:
\begin{enumerate}
  \item el rayo que pasa por el \omterm{def:g11:lenses-and-eye:lens}{centro óptico} $O$ sigue recto;
  \item el rayo paralelo al eje sale por $F'$;
  \item el rayo que pasa por $F$ sale paralelo al eje.
\end{enumerate}
Dos cualesquiera son suficientes: su intersección es el punto de la imagen $B'$, y el
La imagen $A'B'$ es perpendicular al eje en $A'$. si el saliente
los rayos divergen, sus extensiones \emph{hacia atrás} (discontinuas) se encuentran en un
\omterm{def:g11:lenses-and-eye:images}{imagen virtual} punto.
\end{method}

\begin{definition}[Imágenes reales y virtuales]\label{def:g11:lenses-and-eye:images}
Una imagen es \emph{real}\index{imagen real} cuando los rayos salientes
pasar a través de él: una pantalla colocada allí lo atrapa;
\emph{virtual}\index{imagen virtual} cuando solo sus extensiones hacia atrás
Meet: ninguna pantalla lo capta, pero un ojo que mira hacia el \omterm{def:g11:lenses-and-eye:lens}{lente} lo ve.
perfectamente bien. Un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} proporciona una imagen real invertida de cualquier
objeto más allá de $F$; deslice el objeto dentro del \omterm{def:g11:lenses-and-eye:focal}{longitud focal} y el
La imagen se vuelve virtual, vertical y ampliada.
\end{definition}

\section{Una fórmula: la relación conjugada}

Las construcciones muestran \emph{dónde} la imagen es; una fórmula lo calcula.
Oriente el eje a lo largo de la luz y use \emph{medidas algebraicas}:
$\overline{OA}$ es la coordenada de $A$ relativa a $O$ (negativa para
un objeto \omterm{def:g11:lenses-and-eye:images}{real} al frente); las alturas $\overline{AB}$ cuentan positivas hacia arriba.

\begin{theorem}[Relación conjugada de lente delgada]\label{thm:g11:lenses-and-eye:conjugate}
A través de un \omterm{def:g11:lenses-and-eye:lens}{lente delgada} de \omterm{def:g11:lenses-and-eye:focal}{longitud focal} $f'$, un punto objeto $A$ en el
imágenes del eje en el punto $A'$ con
\[
\frac{1}{\overline{OA'}} - \frac{1}{\overline{OA}} = \frac{1}{f'} = C .
\]
\end{theorem}
\admitted

\begin{remark}
La relación se sigue de \omterm{thm:g10:refraction:snell}{ley de snell} aplicada a rayos cercanos al
eje; la derivación --- y los límites del modelo \omterm{def:g11:lenses-and-eye:lens}{lente delgada} --- son
hecho honesto en el volumen del Año 1.
\end{remark}

\begin{definition}[Ampliación]\label{def:g11:lenses-and-eye:magnification}
El \emph{aumento}\index{aumento} de la imagen es
\[
\gamma = \frac{\overline{A'B'}}{\overline{AB}}
       = \frac{\overline{OA'}}{\overline{OA}} .
\]
$\abs{\gamma}$ compara tamaños, sus orientaciones de señales ($\gamma < 0$:
invertido); el signo de $\overline{OA'}$ lee la naturaleza (positivo:
\omterm{def:g11:lenses-and-eye:images}{real}, detrás del \omterm{def:g11:lenses-and-eye:lens}{lente}; negativo: \omterm{def:g11:lenses-and-eye:images}{virtual}, al frente).
\end{definition}

\begin{example}[La fórmula en acción]\label{ex:g11:lenses-and-eye:conjugate}
Un objeto se encuentra \qty{30}{cm} frente a un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} con
$f' = \qty{10}{cm}$: $\overline{OA} = \qty{-30}{cm}$, entonces
\[
\frac{1}{\overline{OA'}} = \frac{1}{10} - \frac{1}{30} = \frac{1}{15},
\qquad \overline{OA'} = +\qty{15}{cm},
\qquad \gamma = \frac{15}{-30} = -0.50 :
\]
un \omterm{def:g11:lenses-and-eye:images}{imagen real} \qty{15}{cm} detrás del \omterm{def:g11:lenses-and-eye:lens}{lente}, invertido, de tamaño medio ---
exactamente lo que dibujan los tres rayos.
\end{example}

\section{el ojo}

\begin{definition}[El ojo como sistema óptico]\label{def:g11:lenses-and-eye:eye}
Ópticamente, el ojo es un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} frente a una pantalla: el
\emph{córnea}\index{córnea} y \emph{lente cristalina}\index{cristalino
lente} actúan como un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} de \omterm{def:g11:lenses-and-eye:focal}{longitud focal} ajustable; el
\emph{retina}\index{retina}, aproximadamente \qty{17}{mm} detrás, es el fijo
pantalla. El enfoque se realiza apretando el \omterm{def:g11:lenses-and-eye:lens}{lente}, no moviéndolo como en
una cámara: \emph{alojamiento}\index{alojamiento} es el cambio de
$f'$ producido por los músculos ciliares. El punto más cercano visto claramente en plena
el alojamiento es el \emph{punto cercano}\index{punto cercano} --- sobre
\qty{25}{cm} para \emph{ojo estándar}\index{ojo estándar} --- el
más lejano, en reposo, el \emph{punto lejano}\index{punto lejano} (infinito para
el ojo estándar).
\end{definition}

\end{example}

\section{Defectos, correcciones y la lupa.}

\begin{definition}[Miopía e hipermetropía]\label{def:g11:lenses-and-eye:defects}
Un ojo \emph{miope}\index{miopía} (miope) es demasiado \omterm{def:g11:lenses-and-eye:lens}{convergente} para
su longitud: rayos paralelos enfocan \emph{delante de} el \omterm{def:g11:lenses-and-eye:eye}{retina}, distante
los objetos se vuelven borrosos, el \omterm{def:g11:lenses-and-eye:eye}{punto lejano} está a una distancia finita; a \omterm{def:g11:lenses-and-eye:lens}{divergente}
El espectáculo \omterm{def:g11:lenses-and-eye:lens}{lente} lo corrige. A \emph{hipermétrope}\index{hipermetropía}
El ojo (hipermetropía) no es suficiente \omterm{def:g11:lenses-and-eye:lens}{convergente}: los objetos cercanos se enfocarían
detrás del \omterm{def:g11:lenses-and-eye:eye}{retina}, el \omterm{def:g11:lenses-and-eye:eye}{punto cercano} retrocede; a \omterm{def:g11:lenses-and-eye:lens}{lente convergente} corrige
eso. El envejecimiento endurece el \omterm{def:g11:lenses-and-eye:eye}{lente cristalina}
(\emph{presbicia}\index{presbicia}), encogiendo \omterm{def:g11:lenses-and-eye:eye}{alojamiento} --- igual
Remedio: gafas para leer.
\end{definition}

\begin{example}[Recetando para un miope]\label{ex:g11:lenses-and-eye:correction}
Un ojo \omterm{def:g11:lenses-and-eye:defects}{miope} tiene su \omterm{def:g11:lenses-and-eye:eye}{punto lejano} en \qty{50}{cm}: el \omterm{def:g11:lenses-and-eye:lens}{lente} corrector.
debe mostrar objetos distantes \emph{at the \omterm{def:g11:lenses-and-eye:eye}{punto lejano}} --- objeto en
infinito, imagen en $\overline{OA'} = \qty{-0.50}{m}$ (\omterm{def:g11:lenses-and-eye:images}{virtual}, en
frente). Entonces $1/f' = 1/\overline{OA'}$: $f' = \qty{-0.50}{m}$,
$C = -2.0$~D; el ojo relajado mira ese \omterm{def:g11:lenses-and-eye:images}{imagen virtual} y lo ve
bruscamente.
\end{example}

\begin{definition}[la lupa]\label{def:g11:lenses-and-eye:magnifier}
Un \emph{lupa}\index{lupa} es un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} de \omterm{def:g11:lenses-and-eye:focal}{focales
longitud} corto usado con el objeto \emph{adentro} o \omterm{def:g11:lenses-and-eye:focal}{longitud focal}: la imagen
es \omterm{def:g11:lenses-and-eye:images}{virtual}, vertical, ampliada (\cref{def:g11:lenses-and-eye:images}),
y el ojo lo ve cómodamente desde lejos.
\end{definition}

\begin{example}[lupa de joyero]\label{ex:g11:lenses-and-eye:magnifier}
Lupa con $f' = \qty{5.0}{cm}$, piedra en \qty{4.0}{cm}:
$1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$, entonces
$\overline{OA'} = \qty{-20}{cm}$ y $\gamma = (-20)/(-4.0) = +5.0$: un
\omterm{def:g11:lenses-and-eye:images}{virtual}, imagen vertical, cinco veces más grande, un cómodo \qty{20}{cm}
del \omterm{def:g11:lenses-and-eye:lens}{lente}.
\end{example}

\section{Problema: una fórmula, tres instrumentos}

\begin{problem}[{Problema de fin de semana --- la misma relación conjugada
Enfoca una cámara, prescribe gafas para leer, diseña una lupa y
mide el zoom integrado en tu propio ojo}]
\label{pb:g11:lenses-and-eye:1}
Un fotógrafo coloca un \qty{50}{mm} \omterm{def:g11:lenses-and-eye:lens}{lente}, una abuela sostiene el
periódico con el brazo extendido, un joyero se inclina sobre un diamante: una línea de
álgebra, $1/\overline{OA'} - 1/\overline{OA} = 1/f'$
(\cref{thm:g11:lenses-and-eye:conjugate}), ejecuta las tres operaciones ---
y luego el ojo mismo.

\medskip
\noindent\textbf{Parte I --- One formula.} Un \omterm{def:g11:lenses-and-eye:lens}{lente convergente} tiene
$f' = \qty{20}{cm}$.
\begin{enumerate}
  \item Un objeto en \qty{60}{cm}: calcular $\overline{OA'}$ y
        $\gamma$; describir la imagen.
  \item Mueva el objeto a \qty{30}{cm}; recalcular. lo que se intercambió
        con la pregunta 1? (Los caminos de luz son reversibles).
  \item Muévelo a \qty{10}{cm}, dentro de $f'$: recalcula y describe.
  \item Resumen: a medida que el objeto se desliza desde muy lejos hacia $F$,
        luego, más allá, ¿cómo se mueve la imagen y cambia la naturaleza?
  \item Muestra que un objeto muy distante ($1/\overline{OA} \approx 0$)
        siempre imágenes en el plano focal: $\overline{OA'} = f'$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The photographer.}
Una cámara \omterm{def:g11:lenses-and-eye:lens}{lente} tiene $f' = \qty{50}{mm}$; el sensor, \qty{24}{mm} alto,
se puede mover en relación con el \omterm{def:g11:lenses-and-eye:lens}{lente}.
\begin{enumerate}[resume]
  \item ¿Dónde está el sensor cuando se enfoca en un paisaje lejano?
  \item Ahora a una cara de distancia \qty{1.0}{m}: calcula la nueva lente-sensor
        distancia y el recorrido desde el entorno paisajístico.
  \item El mecanismo permite como máximo \qty{55}{mm} de lente--sensor
        distancia. Calcule la distancia mínima de enfoque.
  \item En el foco más cercano, calcule $\gamma$. ¿La imagen de un
        ¿\qty{20}{cm} encaja en la cara del sensor?
  \item ``macro'' de tamaño natural significa $\gamma = -1$: donde debe objetar y
        Entonces será el sensor, y ¿por qué esto exige un \omterm{def:g11:lenses-and-eye:lens}{lente} especial?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The grandmother.}
Con la edad el \omterm{def:g11:lenses-and-eye:eye}{punto cercano} retrocede: el de la abuela está en \qty{1.0}{m}.
\begin{enumerate}[resume]
  \item Explica el periódico en condiciones de plena competencia.
  \item Para leer en \qty{25}{cm}, sus gafas deben pasar página en
        \qty{25}{cm} en \omterm{def:g11:lenses-and-eye:images}{imagen virtual} en \qty{1.0}{m}: calcular el
        requerido \omterm{def:g11:lenses-and-eye:vergence}{vergencia}.
  \item Calcular $\gamma$. La imagen es cuatro veces más grande, pero no parece
        más grande: explique, comparando tamaños \emph{y} distancias.
  \item Muestra que la impresión es nítida sólo entre \qty{25}{cm} y
        $f' \approx \qty{33}{cm}$. ¿Qué pasa con un objeto \emph{distante} visto?
        a través de las gafas?
  \item Deduce en una frase por qué existen las lentes bifocales.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The jeweler, and the eye's own zoom.}
La lupa de joyero tiene $f' = \qty{5.0}{cm}$; su \omterm{def:g11:lenses-and-eye:eye}{punto cercano} está en
\qty{25}{cm}; su \omterm{def:g11:lenses-and-eye:eye}{retina} es \qty{17}{mm} detrás del \omterm{def:g11:lenses-and-eye:lens}{lente} de su ojo.
\begin{enumerate}[resume]
  \item Una piedra \qty{4.0}{cm} bajo la lupa: posición, naturaleza,
        \omterm{def:g11:lenses-and-eye:magnification}{aumento}?
  \item ¿Dónde debe asentarse la piedra para que su imagen caiga en su \omterm{def:g11:lenses-and-eye:eye}{cerca
        punto}, y qué es entonces $\gamma$?
  \item ¿Por qué los joyeros experimentados colocan la piedra exactamente en $F$?
        en cambio (imagen en el infinito)?
  \item Ahora solo el ojo: calcula su \omterm{def:g11:lenses-and-eye:vergence}{vergencia} viendo una distancia
        objeto, luego uno en \qty{25}{cm}: ¿cuántos \omterm{def:g11:lenses-and-eye:vergence}{dioptrías} de
        \omterm{def:g11:lenses-and-eye:eye}{alojamiento}?
  \item Convierta ambos \omterm{def:g11:lenses-and-eye:vergence}{vergencias} a \omterm{def:g11:lenses-and-eye:focal}{distancias focales}: por cuánto, en
        milímetros y porcentaje, ¿el ojo cambia su propio \omterm{def:g11:lenses-and-eye:focal}{focales
        longitud} --- el zoom que hizo la cámara con los viajes y la abuela?
        con $+3$~D?
\end{enumerate}
\end{problem}
