\section{Energía cinética}

\begin{definition}[Energía cinetica]\label{def:g11:mechanical-energy:kinetic}
Un cuerpo de masa $m$ (\unit{kg}) que se mueve con velocidad $v$ (\unit{m/s}) lleva
el \emph{energía cinética}\index{energía cinética}
\[
E_k = \tfrac12\, m v^2 .
\]
\omterm{def:g10:orders-of-magnitude:unit}{Unidades} comprobar: $\unit{kg.m^2/s^2} = \unit{N.m} = \unit{J}$ --- \omterm{def:g11:work-of-force:work}{julios},
como cada \omterm{def:g10:energy-conservation:energy}{energía} (\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Órdenes de magnitud]\label{ex:g11:mechanical-energy:orders}
\begin{itemize}
  \item un peatón, \qty{70}{kg} en \qty{1.4}{m/s}:
        $E_k = \tfrac12 \times 70 \times 1.4^2 \approx \qty{69}{J}$;
  \item un coche, \qty{1300}{kg} en \qty{130}{km/h} $= \qty{36.1}{m/s}$:
        $E_k \approx \qty{8.5e5}{J}$;
  \item un tren de alta velocidad, $m = \qty{3.85e5}{kg}$, en \qty{320}{km/h}
        $= \qty{88.9}{m/s}$: $E_k \approx \qty{1.5e9}{J}$;
  \item una bala de rifle, \qty{8.0}{g} en \qty{800}{m/s}: $E_k \approx \qty{2.6}{kJ}$.
\end{itemize}
Siete órdenes de magnitud separan el paseo del tren. Y $E_k$
crece como el \emph{cuadrado} de la velocidad: en \qty{130}{km/h} el coche
lleva $4$ veces el \omterm{def:g10:energy-conservation:energy}{energía} que tiene en \qty{65}{km/h} --- cuatro veces
lo que sus frenos deben quitar.
\end{example}

\section{El teorema de la energía cinética}

\begin{theorem}[Teorema de la energía cinética]\label{thm:g11:mechanical-energy:ketheorem}
\index{teorema de la energía cinética}
Para un cuerpo de masa $m$ que se mueve en línea recta desde $A$ hasta $B$,
el cambio de \omterm{def:g11:mechanical-energy:kinetic}{energía cinética} es igual al total \omterm{def:g11:work-of-force:work}{trabajar}
(\cref{ch:g11:work-of-force}) recibido:
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
la suma que recorre todos los \omterm{def:g10:inertia:force}{efectivo} que actúan sobre el cuerpo. Se extiende a
movimiento a lo largo de cualquier camino, como también admitimos aquí.
\end{theorem}
\admitted

\begin{example}[Caída libre, comprobada cuadro a cuadro]\label{ex:g11:mechanical-energy:freefall}
Se deja caer una pelota \qty{0.200}{kg} y se filma; los marcos dan:
\begin{center}\small
\begin{tabular}{lcccc}
$t$ (\unit{s}) & 0.10 & 0.20 & 0.30 & 0.40\\
fallen height $h$ (\unit{m}) & 0.049 & 0.196 & 0.441 & 0.785\\
speed $v$ (\unit{m/s}) & 0.98 & 1.96 & 2.94 & 3.92\\
$E_k = \tfrac12 mv^2$ (\unit{J}) & 0.096 & 0.384 & 0.864 & 1.54\\
$W = mgh$ (\unit{J}) & 0.096 & 0.385 & 0.865 & 1.54
\end{tabular}
\end{center}
A partir de $E_k = 0$, en cada cuadro $E_k$ coincide con \omterm{def:g11:work-of-force:work}{trabajar} $mgh$ de
el \omterm{def:g10:universal-gravitation:weight}{peso}: $\Delta E_k = W(\vect P)$, es decir\ $v^2 = 2gh$ --- el teorema,
verificado con la precisión de las lecturas.
\end{example}

\begin{remark}[Donde vive la prueba]\label{rem:g11:mechanical-energy:proof}
La afirmación general --- válida en cualquier camino, no sólo en línea recta
línea --- se deriva de la segunda ley de Newton y un poco de cálculo, ambos
más adelante en este libro (\cref{ch:g12:newtons-laws,%
ch:g12:work-and-energy}); Hasta entonces, el control de caída libre es nuestro.
garantía, y admitimos la versión de trayectoria curva y la usamos libremente.
\end{remark}

\section{Energía potencial gravitacional}

\begin{definition}[Energía potencial gravitacional]\label{def:g11:mechanical-energy:potential}
Un cuerpo de masa $m$ a la altitud $z$, medido hacia arriba desde un punto elegido
\emph{nivel de referencia}\index{nivel de referencia} donde $z = 0$, almacena el
\emph{energía potencial gravitacional}\index{energía potencial!gravitacional}
\[
E_p = m g z, \qquad g = \qty{9.81}{N/kg}.
\]
\end{definition}

\begin{remark}[Solo importan las diferencias]\label{rem:g11:mechanical-energy:reference}
Pasar de $A$ a $B$ cambia $E_p$ por $mg(z_B - z_A) =
-W_{AB}(\vect P)$: lo contrario del \omterm{def:g11:work-of-force:work}{trabajar} del \omterm{def:g10:universal-gravitation:weight}{peso}
(\cref{ch:g11:work-of-force}). El \omterm{def:g11:mechanical-energy:potential}{nivel de referencia} es una elección libre ---
piso, nivel del mar, mesa --- desplazando $E_p$ en todas partes mediante una constante
que desaparece de toda diferencia. Ponlo donde están los números.
lo más simple y no lo mueva a mitad del problema.
\end{remark}

\begin{example}[Un libro en una estantería]\label{ex:g11:mechanical-energy:shelf}
Un libro \qty{1.2}{kg} se encuentra \qty{1.8}{m} encima del suelo.
\qty{9.0}{m} encima de la calle. Referencia en el piso:
$E_p = 1.2 \times 9.81 \times 1.8 \approx \qty{21}{J}$; en la calle:
$E_p \approx \qty{127}{J}$. Caer al suelo libera \qty{21}{J} en
ambas cuentas.
\end{example}

\section{La energía mecánica y su conservación.}

\begin{definition}[Energia mecanica]\label{def:g11:mechanical-energy:em}
El \emph{energía mecánica}\index{energía mecánica} de un cuerpo es
\[
E_m = E_k + E_p = \tfrac12 m v^2 + m g z .
\]
\end{definition}

\begin{theorem}[Conservación de energía mecánica]\label{thm:g11:mechanical-energy:conservation}
\index{conservación de la energía mecánica}
Si el único \omterm{def:g10:inertia:force}{fuerza} que trabaja en un cuerpo es su \omterm{def:g10:universal-gravitation:weight}{peso} (no \omterm{def:g10:inertia:inventory}{fricción}; otros
\omterm{def:g10:inertia:force}{efectivo}, como la reacción de una pista sin fricción, perpendicular a la
movimiento), $E_m$ se conserva: para dos puntos cualesquiera $A$, $B$ del movimiento,
\[
\tfrac12 m v_A^2 + m g z_A = \tfrac12 m v_B^2 + m g z_B .
\]
\end{theorem}

\begin{proof}
Según el teorema \omterm{def:g10:energy-conservation:energy}{energía-cinética}, $E_k(B) - E_k(A) = W_{AB}(\vect P) =
mg(z_A - z_B) = E_p(A) - E_p(B)$: lo que gana$E_k$, lo pierde$E_p$.
\end{proof}

\begin{example}[tobogán de agua]\label{ex:g11:mechanical-energy:slide}
Un niño comienza en reposo sobre un tobogán de altura sin fricción.
$h = \qty{3.2}{m}$. Referencia en la parte inferior: $mgh = \tfrac12 mv^2$, por lo que
$v = \sqrt{2gh} \approx \qty{7.9}{m/s}$ (\qty{29}{km/h}) --- lo que sea
la masa \emph{y cualquiera que sea la forma del tobogán}: recta, curva
o espiral, sólo entra la caída de altura.
\end{example}

\begin{example}[Péndulo]\label{ex:g11:mechanical-energy:pendulum}
Se aparta una masa de péndulo hasta que se eleva $h = \qty{12}{cm}$ por encima
su punto más bajo y luego se suelta. En los extremos $v = 0$; en la parte inferior
todo el $mgh$ es cinético, $v = \sqrt{2gh} \approx \qty{1.5}{m/s}$;
en el otro lado, la masa vuelve a subir exactamente a \qty{12}{cm}.
\end{example}

\section{Disipación por fricción}

\begin{definition}[Disipación]\label{def:g11:mechanical-energy:dissipation}
\omterm{def:g10:energy-conservation:energy}{Energía} se dice que salir de la cuenta mecánica es
\emph{disipado}\index{disipación}: reaparece como \omterm{def:g10:energy-conservation:forms}{energía termal} en
las superficies de fricción y el aire. Se conserva el total \omterm{def:g10:energy-conservation:energy}{energía}
(\cref{ch:g10:energy-conservation}); \omterm{def:g11:mechanical-energy:em}{energía mecánica} por sí solo no lo es.
\end{definition}

\begin{proposition}[El déficit mide la fricción]\label{prop:g11:mechanical-energy:deficit}
Si un \omterm{def:g10:inertia:inventory}{fricción} \omterm{def:g10:inertia:force}{fuerza} $\vect f$ también \omterm{def:g11:work-of-force:work}{obras} a lo largo del movimiento,
\[
\Delta E_m = E_m(B) - E_m(A) = W_{AB}(\vect f) < 0:
\]
la pérdida de \omterm{def:g11:mechanical-energy:em}{energía mecánica} equivale a la \omterm{def:g11:work-of-force:work}{trabajar} de \omterm{def:g10:inertia:inventory}{fricción}.
\end{proposition}

\begin{proof}
El teorema \omterm{def:g10:energy-conservation:energy}{energía-cinética} ahora dice $\Delta E_k = W_{AB}(\vect P) +
W_{AB}(\vect f) = -\Delta E_p + W_{AB}(\vect f)$; mover$\Delta E_p$
a través. \omterm{def:g10:inertia:inventory}{Fricción} se opone a la moción, por lo que su \omterm{def:g11:work-of-force:work}{trabajar} es negativa.
\end{proof}

\begin{example}[Auditorio de Trineo]\label{ex:g11:mechanical-energy:toboggan}
Un niño y un trineo ($m = \qty{40}{kg}$) comienzan en reposo, bajan
$h = \qty{5.0}{m}$ por \qty{12}{m} de nieve, y llegar a
\qty{7.0}{m/s} en lugar del sin fricción
$\sqrt{2gh} \approx \qty{9.9}{m/s}$. Referencia en la parte inferior:
$\Delta E_m = \tfrac12 \times 40 \times 7.0^2 - 40 \times 9.81
\times 5.0 = 980 - 1962 = \qty{-982}{J}$. Entonces$W(\vect f) =
\qty{-982}{J}$ sobre \qty{12}{m}: un promedio \omterm{def:g10:inertia:inventory}{fricción} \omterm{def:g10:inertia:force}{fuerza}
$f = 982/12 \approx \qty{82}{N}$ y \qty{982}{J} de calidez en el
corredores y la nieve.
\end{example}

\begin{method}[Contabilidad de energía]\label{met:g11:mechanical-energy:bookkeeping}
\begin{enumerate}
  \item Elija un \omterm{def:g11:mechanical-energy:potential}{nivel de referencia} y consérvelo; nombre $A$ (datos), $B$
        (pregunta).
  \item Escriba $E_m = \tfrac12 mv^2 + mgz$ en $A$ y en $B$.
  \item No \omterm{def:g10:inertia:inventory}{fricción}: $E_m(A) = E_m(B)$; resolver --- típicamente
        $v_B = \sqrt{v_A^2 + 2g(z_A - z_B)}$.
  \item \omterm{def:g10:inertia:inventory}{Fricción}: $E_m(B) - E_m(A) = -f\ell$ ($\ell$ = longitud de la ruta);
        resolver lo desconocido.
  \item Verifique: las barras $E_k$/$E_p$ se apilan hasta $E_m$, niveladas a menos que \omterm{def:g10:inertia:inventory}{fricción}.
\end{enumerate}
\end{method}

\section{Problema: diseñar el bucle}

\begin{problem}[{Problema de fin de semana --- auditoría energética de un rodillo
montaña rusa: la altura de lanzamiento que dicta un bucle, las velocidades que promete, la
friccion confesada por dos sensores, los frenos que cierran la cuenta ---
todo el viaje colgado de un número}]
\label{pb:g11:mechanical-energy:1}
Un tren de masa $m = \qty{2.0e3}{kg}$ es liberado en reposo desde una lancha.
cerro de altura $H$ y costea, sin motor, a través de una circular vertical
bucle de radio $R = \qty{8.0}{m}$ cuya parte superior se encuentra a la altura $2R$.
El análisis del movimiento circular (el próximo año) proporciona el único dato que tomamos prestado: en
la parte superior del bucle el tren presiona sobre la vía sólo si
$v_{\text{top}}^2 \geq gR$. Las partes I y II ignoran \omterm{def:g10:inertia:inventory}{fricción}.

\medskip
\noindent\textbf{Parte I --- The hill the loop dictates.}
\begin{enumerate}
  \item Calcule la velocidad mínima segura en la parte superior del bucle.
  \item Tierra como \omterm{def:g11:mechanical-energy:potential}{nivel de referencia}: calcula el \omterm{def:g11:mechanical-energy:em}{mecanico
        energía} del tren en la parte superior del circuito a esa velocidad mínima.
  \item Deduzca $H_{\min}$ y muestre $H_{\min} = 5R/2$ --- sin masa, sin $g$.
  \item El equipo toma $H = \qty{24}{m}$: calcula la parte superior del bucle real
        velocidad y relación $v_{\text{top}}^2/(gR)$.
  \item ¿A dónde fue $m$? ¿Por qué el diseño encaja lleno y vacío?
        trenes iguales?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- What the height promises} ($H = \qty{24}{m}$).
\begin{enumerate}[resume]
  \item Velocidad en la parte inferior del bucle ($z = 0$), en \unit{m/s} y
        \unit{km/h}?
  \item ¿Velocidad al lado del bucle ($z = R$)?
  \item Calcule $E_p$ y $E_k$ en el inicio, en la parte inferior, lateral y superior de
        el bucle; compruebe que cada par sume $mgH$.
  \item Un bache \qty{12}{m} sigue el circuito: ¿velocidad en su punto máximo?
  \item Dé la fórmula general para $v(z)$; donde esta el paseo
        ¿El más rápido y el más lento?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The \omterm{def:g10:inertia:inventory}{fricción} audit.}
Dos sensores a la misma altura $z = \qty{2.0}{m}$, \qty{60}{m} de
Separe las pistas, lea $v_1 = \qty{20.4}{m/s}$ y luego $v_2 = \qty{19.2}{m/s}$.
\begin{enumerate}[resume]
  \item Calcular el cambio de \omterm{def:g11:mechanical-energy:em}{energía mecánica} entre los sensores.
        ¿Qué hizo que la colocación a la misma altura fuera inteligente?
  \item Deduce el \omterm{def:g11:work-of-force:work}{trabajar} de \omterm{def:g10:inertia:inventory}{fricción} y el promedio \omterm{def:g10:inertia:inventory}{fricción} \omterm{def:g10:inertia:force}{fuerza}.
  \item ¿Dónde se ha ido el \omterm{def:g10:energy-conservation:energy}{energía} desaparecido? Nombra la forma y los lugares.
  \item ¿Qué fracción del $E_m$ del tren se pierde por \qty{60}{m}? ¿Por qué?
        ¿El $H_{\min}$ desnudo sería inseguro en el mundo real?
  \item Acerca de \qty{90}{m} del inicio separado de la pista y la parte superior del bucle: estimación
        $E_m$ allí; ¿El margen de la pregunta 4 todavía salva el bucle?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Closing the account.}
El último sensor, a la entrada de la recta de frenado horizontal.
($z = 0$), se lee \qty{18.0}{m/s}.
\begin{enumerate}[resume]
  \item Calcula el \omterm{def:g11:mechanical-energy:kinetic}{energía cinética} del tren allí.
  \item Los frenos deben detenerlo en $d = \qty{45}{m}$: calcular el
        frenado requerido (constante) \omterm{def:g10:inertia:force}{fuerza}.
  \item Control de confort: calcula la desaceleración $a = F/m$; comparar con $g$.
  \item ¿Cuánto disipó \omterm{def:g10:energy-conservation:energy}{energía} la pista \omterm{def:g10:inertia:inventory}{fricción} en todo el
        montar? Verificar que \omterm{def:g10:inertia:inventory}{fricción} $+$ frena $= mgH$, al \omterm{def:g11:work-of-force:work}{joule}.
  \item Remate, una frase: ¿qué número arregló el bucle?
        seguridad, cada velocidad, la factura \omterm{def:g10:inertia:inventory}{fricción} y el freno \omterm{def:g10:inertia:force}{fuerza} ---
        ¿Y cuál fue el precio de ignorar \omterm{def:g10:inertia:inventory}{fricción}?
\end{enumerate}
\end{problem}
