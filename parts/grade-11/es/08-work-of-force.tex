\chapter{Trabajo de una fuerza}\label{ch:g11:work-of-force}

Empuja un armario al otro lado de la habitación y habrás trabajado; empujar contra un
pared toda la tarde y, sea cual sea lo que informen tus músculos, la física cuenta
cero. \omterm{def:g11:work-of-force:work}{Trabajar} es lo que entrega un \omterm{def:g10:inertia:force}{fuerza} \emph{a lo largo de un movimiento} --- \omterm{def:g10:energy-conservation:energy}{energía}
en tránsito (\cref{ch:g10:energy-conservation}), pagado en \omterm{def:g11:work-of-force:work}{julios}. esto
capítulo precios el \omterm{def:g10:inertia:force}{efectivo} de \cref{ch:g11:forces-and-motion} --- tira
en ángulo, \omterm{def:g10:universal-gravitation:weight}{peso}, \omterm{def:g10:inertia:inventory}{fricción} --- y registra la tasa de pago: \omterm{def:g11:work-of-force:power}{fuerza}.

\section{Trabajo de una fuerza constante.}

\begin{definition}[Trabajo de una fuerza constante]\label{def:g11:work-of-force:work}
Una constante \omterm{def:g10:inertia:force}{fuerza} $\vect F$ aplicada a un objeto que se mueve en línea recta.
segmento de $A$ a $B$ hace el \emph{trabajar}\index{trabajo (de una fuerza)}
\[
W_{AB}(\vect F) = F \times AB \times \cos\theta ,
\]
donde $F$ es la magnitud de \omterm{def:g10:inertia:force}{fuerza} (en \unit{N}), $AB$ la longitud
del desplazamiento (en \unit{m}) y $\theta$ el ángulo entre
$\vect F$ y $\vect{AB}$. El trabajo se mide en
\emph{julios}\index{joule} (\unit{J}): un julio es el trabajo de un
\omterm{def:g10:inertia:force}{un newton} \omterm{def:g10:inertia:force}{fuerza} a lo largo de un \omterm{def:g10:orders-of-magnitude:unit}{metro} de su propia dirección.
\end{definition}

\begin{remark}\label{rem:g11:work-of-force:scal}
$F \times AB \times \cos\theta$ es exactamente el producto escalar
$\vect F \cdot \vect{AB}$ del volumen de matemáticas de este año. Sólo el
componente $F\cos\theta$ \emph{a lo largo de} la moción gana algo.
\end{remark}

\begin{example}[Maleta con correa]\label{ex:g11:work-of-force:suitcase}
Un viajero arrastra una maleta \qty{300}{m} por una terminal, la correa
tirado con $F = \qty{40}{N}$ en $\theta = \qty{50}{\degree}$ al
piso: $W = 40 \times 300 \times \cos\qty{50}{\degree} \approx
\qty{7.7}{kJ}$. Tirado plano ($\theta = 0$) lo mismo \omterm{def:g10:inertia:force}{fuerza}
entregar \qty{12}{kJ}: un tercio del tirón no levanta nada.
\end{example}

\begin{definition}[Motor resistivo trabajo cero]\label{def:g11:work-of-force:sign}
El signo de $\cos\theta$ clasifica cada \omterm{def:g10:inertia:force}{fuerza} en tres regímenes:
\begin{itemize}
  \item $\theta < \qty{90}{\degree}$: $W > 0$, el \omterm{def:g11:work-of-force:work}{trabajar} es
        \emph{motor}\index{trabajo motor} --- el \omterm{def:g10:inertia:force}{fuerza} alimenta \omterm{def:g10:energy-conservation:energy}{energía} en
        la moción;
  \item $\theta = \qty{90}{\degree}$: $W = 0$, el \omterm{def:g10:inertia:force}{fuerza} \omterm{def:g11:work-of-force:work}{obras} no en
        todos, por grandes que sean;
  \item $\theta > \qty{90}{\degree}$: $W < 0$, el \omterm{def:g11:work-of-force:work}{trabajar} es
        \emph{resistador}\index{trabajo resistivo} --- el \omterm{def:g10:inertia:force}{fuerza} drena
        \omterm{def:g10:energy-conservation:energy}{energía}. El caso puro es un freno en $\qty{180}{\degree}$:
        \qty{800}{N} sobre \qty{45}{m} saca a $800 \times 45 \times (-1)
        = \qty{-36}{kJ}$ de un automóvil y lo coloca en discos calientes.
\end{itemize}
\end{definition}

\section{El trabajo del peso.}

\begin{proposition}[trabajo del peso]\label{prop:g11:work-of-force:weight}
Cuando un objeto de masa $m$ se mueve desde $A$ (altitud $z_A$) a $B$
(altitud $z_B$) por un camino \emph{cualquier} formado por tramos rectos, su
\omterm{def:g10:universal-gravitation:weight}{peso} $\vect P$ hace el \omterm{def:g11:work-of-force:work}{trabajar}
\[
W_{AB}(\vect P) = mg\,(z_A - z_B) = \pm\, mgh ,
\]
donde $h = \abs{z_A - z_B}$ es la caída de altura: $+mgh$ bajando,
$-mgh$ subiendo --- sólo importa el descenso, no la ruta.
\end{proposition}

\begin{proof}
En un segmento recto $AB$, el \omterm{def:g10:universal-gravitation:weight}{peso} (magnitud $mg$, recto
hacia abajo) forma un ángulo $\theta$ con $\vect{AB}$, y $AB\cos\theta$ es
Precisamente la proyección del desplazamiento sobre la vertical descendente:
$z_A - z_B$. Por lo tanto $W = mg \times AB\cos\theta = mg\,(z_A - z_B)$.
A lo largo de una cadena de segmentos, el \omterm{def:g11:work-of-force:work}{obras} suma y el telescopio cae:
$mg(z_A - z_C) + mg(z_C - z_B) = mg(z_A - z_B)$.
\end{proof}

\begin{proposition}[cualquier ruta]\label{prop:g11:work-of-force:anypath}
La misma fórmula se aplica a lo largo de cualquier trayectoria curva desde $A$ hasta $B$.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:work-of-force:anypath}
Plausible: una curva es lo más parecido que nos gusta a una línea discontinua de muchos
segmentos cortos --- pero hacer que "lo más cerca que queramos" sea honesto es
cálculo, realizado en \cref{ch:g12:work-and-energy}.
\end{remark}

\begin{example}[excursionista]\label{ex:g11:work-of-force:hiker}
Un excursionista \qty{78}{kg} (pack incluido) sube al \qty{450}{m}, por el que sea
sendero: $W(\vect P) = -78 \times 9.81 \times 450 \approx \qty{-344}{kJ}$,
y exactamente $+\qty{344}{kJ}$ retroceder, cualesquiera que sean los desvíos.
\end{example}

\section{El trabajo de fricción}

\begin{proposition}[Trabajo de fricción por deslizamiento]\label{prop:g11:work-of-force:friction}
Un \omterm{def:g10:inertia:inventory}{fricción} \omterm{def:g10:inertia:force}{fuerza} de magnitud constante $f$, en cada instante opuesto
el movimiento, hace el \omterm{def:g11:work-of-force:work}{trabajar} $W = -fL$ a lo largo de un camino de longitud total $L$:
siempre \omterm{def:g11:work-of-force:sign}{resistador}, y proporcional al \emph{longitud del camino
viajó} --- no al desplazamiento.
\end{proposition}

\begin{proof}
Cada pieza recta corta tiene el \omterm{def:g10:inertia:force}{fuerza} en $\qty{180}{\degree}$ al
moción, contribuyendo $-f \times (\text{its length})$; las piezas suman
hasta $-fL$. Un camino curvo se corta en pedazos cortos del mismo modo, el
límite admitido como para el \omterm{def:g10:universal-gravitation:weight}{peso} (\cref{rem:g11:work-of-force:anypath}).
\end{proof}

\begin{remark}[Fricción contra peso: la gran división]\label{rem:g11:work-of-force:divide}
El \omterm{def:g11:work-of-force:work}{trabajar} del \omterm{def:g10:universal-gravitation:weight}{peso} olvida el camino; \omterm{def:g10:inertia:inventory}{fricción} es un peaje según \omterm{def:g10:orders-of-magnitude:unit}{metro}. encendido
un viaje de ida y vuelta el \omterm{def:g10:universal-gravitation:weight}{peso} genera cero --- lo que se necesita cuesta arriba se reembolsa
cuesta abajo --- mientras que \omterm{def:g10:inertia:inventory}{fricción} cobra $-fL$ en cada tramo y reembolsa
nada: el \omterm{def:g11:work-of-force:work}{trabajar} del \omterm{def:g10:universal-gravitation:weight}{peso} se puede almacenar y recuperar (posible
\omterm{def:g10:energy-conservation:energy}{energía}, \cref{ch:g11:mechanical-energy}), \omterm{def:g10:inertia:inventory}{fricción} se pierde en forma de calor.
\end{remark}

\begin{example}[Ida y vuelta]\label{ex:g11:work-of-force:roundtrip}
Una caja desliza \qty{5.0}{m} por el suelo y \qty{5.0}{m} hacia atrás,
contra $f = \qty{60}{N}$: \omterm{def:g10:inertia:inventory}{fricción} hace $-60 \times 10 =
\qty{-600}{J}$ a pesar de que la caja termina donde comenzó. \omterm{def:g10:universal-gravitation:weight}{Peso} y
normal \omterm{def:g10:inertia:force}{fuerza}, perpendicular al movimiento: cero en todas partes.
\end{example}

\begin{remark}[¿Qué tal un recurso?]\label{rem:g11:work-of-force:spring}
Un resorte tira más fuerte cuanto más se estira: su \omterm{def:g10:inertia:force}{fuerza} no es
constante y \cref{def:g11:work-of-force:work} no se aplica.
Su \omterm{def:g11:work-of-force:work}{trabajar} significa cortar el tramo en pasos infinitesimales y
sumatoria --- cálculo, entregado en \cref{ch:g12:work-and-energy}.
\end{remark}

\section{Fuerza}

\begin{definition}[poder]\label{def:g11:work-of-force:power}
El \emph{fuerza}\index{fuerza!mecánico} de un \omterm{def:g10:inertia:force}{fuerza} es la tasa en
cual es \omterm{def:g11:work-of-force:work}{obras}, $P = W/\Delta t$, en \emph{vatios}\index{vatio}
($\unit{W} = \unit{J/s}$) --- el mismo vatio que en
\cref{ch:g11:circuits-and-power}. Un modelo anterior al SI \omterm{def:g10:orders-of-magnitude:unit}{unidad} sobrevive en el motor
insignias: el \emph{caballo de fuerza}\index{caballo de fuerza},
$\qty{1}{hp} = \qty{736}{W}$.
\end{definition}

\begin{proposition}[Potencia a velocidad constante]\label{prop:g11:work-of-force:pfv}
Si el objeto se mueve a velocidad constante $v$, una constante \omterm{def:g10:inertia:force}{fuerza} $\vect F$
en el ángulo $\theta$ con respecto al movimiento entrega el \omterm{def:g11:work-of-force:power}{fuerza}
$P = F v \cos\theta$ --- en particular $P = Fv$ para un \omterm{def:g10:inertia:force}{fuerza} a lo largo del
movimiento.
\end{proposition}

\begin{proof}
En un tiempo $\Delta t$ el objeto cubre $d = v\,\Delta t$, por lo que
$W = F \times v\,\Delta t \times \cos\theta$; dividir por $\Delta t$.
\end{proof}

\begin{example}[Grúa]\label{ex:g11:work-of-force:crane}
Una grúa iza un palet \qty{600}{kg} a una velocidad constante \qty{0.90}{m/s}: el
\omterm{def:g10:inertia:force}{fuerza} del cable es igual a \omterm{def:g10:universal-gravitation:weight}{peso}, $mg = \qty{5.89e3}{N}$, por lo que
$P = Fv = 5890 \times 0.90 \approx \qty{5.3}{kW} \approx \qty{7.2}{hp}$
--- un equipo de siete caballos en una caja de acero.
\end{example}

\begin{method}[Subir con menos fuerza: el zigzag]\label{met:g11:work-of-force:zigzag}
Para elevar una carga una altura $h$ con un \omterm{def:g10:inertia:force}{fuerza} limitado:
\begin{enumerate}
  \item La factura queda fijada: por \cref{prop:g11:work-of-force:weight},
        cualquier ruta hasta la cima cuesta el \omterm{def:g11:work-of-force:work}{trabajar} $mgh$.
  \item Estirar el camino: en una pendiente de longitud $L$ se necesita el \omterm{def:g10:inertia:force}{fuerza}
        (velocidad constante, aparte de \omterm{def:g10:inertia:inventory}{fricción}) es $F = mgh/L$ --- el doble de
        longitud, reduzca a la mitad el \omterm{def:g10:inertia:force}{fuerza}.
  \item Paga el recargo: \omterm{def:g10:inertia:inventory}{fricción} cobra por el \omterm{def:g10:orders-of-magnitude:unit}{metro}
        (\cref{prop:g11:work-of-force:friction}), agregando $fL$.
\end{enumerate}
Los caminos de montaña zigzaguean para el paso 2: misma altura, menos \omterm{def:g10:inertia:force}{fuerza}, mismo \omterm{def:g11:work-of-force:work}{trabajar}.
\end{method}

\section{Problema: el día de la mudanza}

\begin{problem}[{Problema del fin de semana --- la física se paga por julios: a
jornada de graduales, elevaciones y curvas cerradas, que finaliza con la humillación
descubrimiento de lo que piensa una tetera del trabajo honesto}]
\label{pb:g11:work-of-force:1}
El equipo de una empresa de mudanzas vacía un almacén y transporta la carga hasta el tercer piso.
Luego conduce la camioneta hasta un pueblo en la cima de una colina. Cada tarea tiene un precio.
en \omterm{def:g11:work-of-force:work}{julios}; la tetera de la cocina auditará el día. Cajas: \qty{30}{kg}.

\medskip
\noindent\textbf{Parte I --- Across the warehouse floor.}
Se empuja una caja con un \omterm{def:g10:inertia:force}{fuerza} $F = \qty{110}{N}$ horizontal sobre
$d = \qty{8.0}{m}$; deslizando \omterm{def:g10:inertia:inventory}{fricción} es $f = \qty{95}{N}$.
\begin{enumerate}
  \item Calcula el \omterm{def:g11:work-of-force:work}{trabajar} del push.
  \item Calcular el \omterm{def:g11:work-of-force:work}{trabajar} del \omterm{def:g10:inertia:inventory}{fricción}, del \omterm{def:g10:universal-gravitation:weight}{peso} y del normal
        \omterm{def:g10:inertia:force}{fuerza} (justificar los ceros).
  \item Total \omterm{def:g11:work-of-force:work}{trabajar} en el cuadro y el significado de su signo (el cuadro comenzó
        en reposo)? ¿Qué empuje mantiene constante la velocidad?
        (\cref{ch:g11:forces-and-motion})?
  \item Se puede llegar a la esquina más alejada siguiendo la diagonal \qty{10.0}{m}.
        o a lo largo de dos paredes (\qty{6.0}{m} y luego \qty{8.0}{m}). \omterm{def:g10:inertia:inventory}{Fricción}
        \omterm{def:g11:work-of-force:work}{trabajar} en cada ruta?
  \item En un viaje de ida y vuelta en diagonal (ida y vuelta), calcule el \omterm{def:g11:work-of-force:work}{trabajar} de
        \omterm{def:g10:inertia:inventory}{fricción} y el \omterm{def:g11:work-of-force:work}{trabajar} del \omterm{def:g10:universal-gravitation:weight}{peso}. ¿Qué reembolsos \omterm{def:g10:inertia:force}{fuerza}?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Up to the third floor ($h = \qty{9.0}{m}$).}
\begin{enumerate}[resume]
  \item Calcule el \omterm{def:g11:work-of-force:work}{trabajar} del \omterm{def:g10:universal-gravitation:weight}{peso} en una caja elevada al plano.
        ¿Qué \omterm{def:g11:work-of-force:work}{trabajar} debe suministrar al menos la tripulación?
  \item Transportado escaleras arriba por un motor \qty{70}{kg}, cargas de gravedad
        para la caja de mudanzas \emph{y}: total \omterm{def:g11:work-of-force:work}{trabajar} contra la gravedad, y el
        ¿fracción realmente gastada en la caja?
  \item Una rampa para carro tiene \qty{30}{m} de largo para el mismo \qty{9.0}{m}
        subida, con \qty{40}{N} de \omterm{def:g10:inertia:inventory}{fricción} rodante. A velocidad constante,
        Calcule el empuje necesario a lo largo de la rampa y su \omterm{def:g11:work-of-force:work}{trabajar}.
  \item Compare la rampa con un polipasto vertical recto: se necesita \omterm{def:g10:inertia:force}{fuerza},
        \omterm{def:g11:work-of-force:work}{trabajar} hecho. ¿Qué compra la rampa y a qué precio?
  \item Un polipasto de cuerda y polea eleva la caja en \qty{0.50}{m/s}.
        ¿Duración y mecánica \omterm{def:g11:work-of-force:power}{fuerza}?
  \item Recupera ese \omterm{def:g11:work-of-force:power}{fuerza} como $P = Fv$. El \omterm{def:g11:work-of-force:sign}{motor} del polipasto es $60\%$.
        eficiente: \omterm{prop:g11:circuits-and-power:power}{energía eléctrica} dibujado?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The zigzag road.}
La furgoneta cargada (\qty{3500}{kg}) debe llegar a un pueblo \qty{240}{m}.
sobre el valle: carril recto, \qty{1.0}{km} en grado $24\%$, o
camino asfaltado, \qty{8.0}{km} en grado $3.0\%$ (grado $= \sin\alpha$).
\begin{enumerate}[resume]
  \item Comprobar que ambas vías suben al mismo \qty{240}{m}.
  \item A velocidad constante y despreciando \omterm{def:g10:inertia:inventory}{fricción}, calcule la conducción
        \omterm{def:g10:inertia:force}{fuerza} necesario a lo largo de cada ruta.
  \item \omterm{def:g11:work-of-force:work}{Trabajar} del \omterm{def:g10:inertia:force}{fuerza} de conducción en cada ruta? Comparar con $mgh$.
  \item Rolling \omterm{def:g11:circuits-and-power:resistance}{resistencia} es \qty{400}{N} en ambos: calcule el extra
        \omterm{def:g11:work-of-force:work}{trabajar} en cada ruta. ¿Cuánto cuesta la comodidad del zigzag?
  \item En \qty{36}{km/h}, calcule el motor \omterm{def:g11:work-of-force:power}{fuerza} necesario en cada
        ruta (en \unit{kW} y \unit{hp}). ¿Por qué las carreteras zigzaguean?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The day's ledger.}
\begin{enumerate}[resume]
  \item La tripulación desliza las cajas $60$ por el suelo (el empujón de
        pregunta 1) y levanta todos los $60$ hasta el \qty{9.0}{m}
        (pregunta 6). ¿Mecánica Total \omterm{def:g11:work-of-force:work}{trabajar} entregada en las cajas?
  \item Añadir el propio cuerpo del portaescaleras: $20$ sube en \qty{70}{kg}
        (Los reembolsos cuesta abajo van a parar a las rodillas calientes, no al libro mayor). dia
        totales?
  \item La tripulación trabajó $8.0$ horas: ¿promedio mecánico \omterm{def:g11:work-of-force:power}{fuerza}? cuanto tiempo
        ¿La tetera \qty{2.0}{kW} tardaría en gastar lo mismo \omterm{def:g10:energy-conservation:energy}{energía}?
  \item Final: en dos oraciones, diga por qué paga el \omterm{def:g11:work-of-force:work}{joule} --- y
        no (la pared empujó toda la tarde, la caja sostenida
        inmóvil) --- y por qué los encargados de la mudanza poseen cabrestantes, rampas y zigzag
        caminos en lugar de músculos más grandes.
\end{enumerate}
\end{problem}
