\chapter{Magnetismo y campos magnéticos}\label{ch:g11:magnetic-fields}

Marineros guiados por agujas magnetizadas desde hace mil años
Alguien sabía por qué apuntan al norte. La respuesta llegó en 1820, cuando un
La corriente hizo que una brújula se moviera: el magnetismo se produce mediante cargas en movimiento.
Este capítulo mapea el \omterm{def:g11:magnetic-fields:bfield}{campo magnético} --- de \omterm{def:g11:magnetic-fields:magnet}{imanes}, de la Tierra, de
corrientes --- luego cierra el circuito: el \omterm{def:g11:electric-gravitational-fields:field}{campo} empuja hacia atrás las corrientes,
Así es como gira todo motor eléctrico.

\section{Imanes y el campo magnético.}

\begin{definition}[Imán y polos]\label{def:g11:magnetic-fields:magnet}
A \emph{imán}\index{imán} atrae el hierro y, colgado de un hilo,
se vuelve hacia el norte. Su acción se concentra en dos \emph{magnético
polos}\index{polo magnético}, \emph{norte} (el fin que busca
norte geográfico) y \emph{sur}: los polos iguales se repelen, los diferentes se atraen.
Cortar un imán por la mitad produce dos imanes completos --- sin experimento
alguna vez ha aislado un polo.
\end{definition}

\begin{definition}[campo magnético]\label{def:g11:magnetic-fields:bfield}
Un \omterm{def:g11:magnetic-fields:magnet}{imán} modifica el espacio a su alrededor.
(\cref{ch:g11:electric-gravitational-fields}): crea en todas partes un
\emph{campo magnético}\index{campo magnético} $\vect B$. Una brújula lee
su dirección (la aguja se alinea con $\vect B$, el extremo norte hacia adelante); un
\emph{Sonda Hall}\index{Sonda Hall} mide su magnitud, en
\emph{Tesla}\index{Tesla} (\unit{T}).
\end{definition}

\begin{definition}[Líneas de campo magnético]\label{def:g11:magnetic-fields:lines}
\emph{Líneas de campo magnético}\index{línea de campo magnético} son curvas
tangente a $\vect B$, apiñándose donde el \omterm{def:g11:electric-gravitational-fields:field}{campo} es fuerte. fuera de un
\omterm{def:g11:magnetic-fields:magnet}{imán} corren del polo norte al sur; a diferencia de las líneas \omterm{def:g11:electric-gravitational-fields:efield}{campo eléctrico}
no tienen extremos: cada uno se cierra a través del cuerpo de \omterm{def:g11:magnetic-fields:magnet}{imán}.
\end{definition}

\section{La Tierra es un imán}

\begin{definition}[El campo magnético de la Tierra]\label{def:g11:magnetic-fields:earth}
Las corrientes en el núcleo de hierro fundido de la Tierra hacen del planeta un gigante
\omterm{def:g11:magnetic-fields:magnet}{imán}: en superficie $B \approx \qty{50}{\micro T}$, y libre
La aguja se alinea con el componente horizontal del \omterm{def:g11:electric-gravitational-fields:field}{campo}. Este es el
brújula. Los polos geomagnéticos no son los geográficos; el ángulo
entre el norte cardinal y el norte verdadero es el
\emph{declinación}\index{declinación}, corregido en cada rodamiento.
\end{definition}

\begin{example}[Corrección de un rumbo]\label{ex:g11:magnetic-fields:bearing}
Donde el \omterm{def:g11:magnetic-fields:earth}{declinación} es $\ang{7}$ oeste, un barco que dirige la "brújula"
norte'' para \qty{5.0}{km} desvía $5000 \times \sin\ang{7} \approx
\qty{6.1e2}{m}$al oeste de su rumbo: para navegar hacia el norte verdadero, gire$\ang{7}$
\emph{este} de la aguja.
\end{example}

\section{El campo magnético de una corriente.}

\begin{remark}[Oersted, 1820]\label{rem:g11:magnetic-fields:oersted}
Durante una conferencia en 1820, Hans Christian Oersted vio la aguja de una brújula.
oscilar cuando cerró un circuito cercano: la electricidad y el magnetismo eran
un sujeto --- \emph{currents create \omterm{def:g11:magnetic-fields:bfield}{campos magnéticos}}.
\end{remark}

\begin{proposition}[Campo de un hilo recto]\label{prop:g11:magnetic-fields:wire}
Un cable largo y recto que transporta una corriente $I$ crea un \omterm{def:g11:electric-gravitational-fields:field}{campo} cuyo
Las líneas son círculos centrados en el alambre, en planos perpendiculares a él.
A la distancia $d$ del cable,
\[
B = \frac{\mu_0\, I}{2\pi d},
\qquad \mu_0 = 4\pi\times 10^{-7}\ \unit{T.m/A} \approx \qty{1.26e-6}{T.m/A}.
\]
\end{proposition}
\admitted

\begin{example}[Un cable es un imán débil]\label{ex:g11:magnetic-fields:wire}
En $d = \qty{1.0}{cm}$ de un cable que lleva $I = \qty{15}{A}$:
$B = \num{2e-7} \times 15 / 0.010 = \qty{0.30}{mT}$ --- seis Tierra
\omterm{def:g11:electric-gravitational-fields:field}{campos}, de una corriente que dispararía un disyuntor doméstico. fuerte
\omterm{def:g11:electric-gravitational-fields:field}{campos} necesita una mejor geometría.
\end{example}

\begin{definition}[Bobina y solenoide]\label{def:g11:magnetic-fields:solenoid}
Al enrollar el cable se concentra su \omterm{def:g11:electric-gravitational-fields:field}{campo}. Un \emph{plano
bobina}\index{bobina} ($N$ gira en un disco) se comporta como un \omterm{def:g11:lenses-and-eye:lens}{delgado} \omterm{def:g11:magnetic-fields:magnet}{imán},
uno mira al norte y el otro al sur; a \emph{solenoide}\index{solenoide} ---
un devanado cilíndrico largo, $n$ vueltas por \omterm{def:g10:orders-of-magnitude:unit}{metro} --- como una barra \omterm{def:g11:magnetic-fields:magnet}{imán}.
\end{definition}

\begin{proposition}[Campo dentro de un solenoide]\label{prop:g11:magnetic-fields:solenoidfield}
Dentro de un \omterm{def:g11:magnetic-fields:solenoid}{solenoide} largo, alejado de los extremos, se encuentra el \omterm{def:g11:electric-gravitational-fields:field}{campo}.
\emph{uniforme}: paralelo al eje e igual en todos los interiores
punto, de magnitud $B = \mu_0\, n\, I$, independiente de la
ancho. En el exterior, el \omterm{def:g11:electric-gravitational-fields:field}{campo} es débil.
\end{proposition}
\admitted

\begin{remark}
Ambas fórmulas se derivan honestamente en el volumen del Año 1, de la ley
relacionando la circulación de un \omterm{def:g11:electric-gravitational-fields:field}{campo} con la corriente que rodea. Tenga en cuenta lo que
importa: $n$, gira \emph{per \omterm{def:g10:orders-of-magnitude:unit}{metro}} --- viento más fuerte, no más.
\end{remark}

\begin{method}[La mano derecha, dos veces]\label{met:g11:magnetic-fields:righthand}
\begin{itemize}
  \item \emph{Cable}: pulgar = actual; dedos curvados = giro de las líneas.
  \item \emph{\omterm{def:g11:magnetic-fields:solenoid}{Bobina}, \omterm{def:g11:magnetic-fields:solenoid}{solenoide}}: dedos curvados = corriente en las vueltas;
        pulgar = $\vect B$ adentro, apuntando a la cara norte.
\end{itemize}
\end{method}

En $10$ vueltas por centímetro ($n = \qty{1000}{m^{-1}}$) y

\section{Electroimanes}

\begin{definition}[Electroimán]\label{def:g11:magnetic-fields:electromagnet}
Un \emph{electroimán}\index{electroimán} es una herida \omterm{def:g11:magnetic-fields:solenoid}{solenoide} en un
Núcleo de hierro dulce: el hierro se magnetiza en el \omterm{def:g11:electric-gravitational-fields:field}{campo} del \omterm{def:g11:magnetic-fields:solenoid}{bobina} y se multiplica
normalmente entre $100$ y $1000$. A diferencia de un \omterm{def:g11:magnetic-fields:magnet}{imán} permanente,
se apaga con la corriente.
\end{definition}

\begin{remark}[tres maquinas]\label{rem:g11:magnetic-fields:machines}
La grúa del desguace levanta un coche con un \omterm{def:g11:magnetic-fields:electromagnet}{electroimán} y --- el
punto completo --- \emph{gotas} abriendo un interruptor. Un relevo es pequeño.
\omterm{def:g11:magnetic-fields:electromagnet}{electroimán} cierra un contacto: una corriente débil conmuta un
uno fuerte (\cref{ch:g11:circuits-and-power}). Un escáner de resonancia magnética sostiene un uniforme
\qty{1.5}{T} alrededor del paciente con un \omterm{def:g11:magnetic-fields:solenoid}{solenoide} superconductor.
\end{remark}

\end{center}

\section{La fuerza de Laplace}

\begin{proposition}[Fuerza de Laplace]\label{prop:g11:magnetic-fields:laplace}
Un cable recto de longitud $L$, que transporta una corriente $I$ en un \omterm{def:g11:electric-gravitational-fields:capacitor}{uniforme
campo} $\vect B$ \emph{perpendicular} al cable, siente la
\emph{fuerza de Laplace}\index{fuerza de Laplace} de magnitud $F = I\,L\,B$,
perpendicular a ambos cables y $\vect B$: mano derecha plana, pulgar a lo largo
los dedos rectos actuales a lo largo de $\vect B$ --- la palma empuja
$\vect F$. Un cable \emph{paralelo} al \omterm{def:g11:electric-gravitational-fields:field}{campo} no siente \omterm{def:g10:inertia:force}{fuerza}.
\end{proposition}
\admitted

\begin{remark}[La barandilla de salto]\label{rem:g11:magnetic-fields:rail}
El volumen del Año 1 deriva esto del \omterm{def:g10:inertia:force}{fuerza} magnético en cada movimiento
cargo. En el laboratorio: una varilla de cobre descansa sobre dos
rieles entre los polos de un \omterm{def:g11:magnetic-fields:magnet}{imán} (\omterm{def:g11:electric-gravitational-fields:field}{campo} en la página en el
figura); Cierra el interruptor y la varilla se dispara a lo largo de los rieles. revertir
la corriente, o voltea el \omterm{def:g11:magnetic-fields:magnet}{imán}: dispara hacia el otro lado.
\end{remark}

\begin{example}[Números de Laplace]\label{ex:g11:magnetic-fields:laplace}
Una varilla de longitud $L = \qty{5.0}{cm}$ que lleva a $I = \qty{8.0}{A}$ a través
$B = \qty{0.50}{T}$: $F = 8.0 \times 0.050 \times 0.50 = \qty{0.20}{N}$
--- el \omterm{def:g10:universal-gravitation:weight}{peso} de \qty{20}{g}, sobre un cable de unos pocos gramos.
\end{example}

\begin{remark}[El motor CC]\label{rem:g11:magnetic-fields:motor}
Doble el cable formando un bucle en el \omterm{def:g11:electric-gravitational-fields:field}{campo}: los dos lados perpendiculares a
$\vect B$ transportan corrientes opuestas, por lo que sus \omterm{prop:g11:magnetic-fields:laplace}{Fuerzas de Laplace} son opuestas.
--- un par que hace girar el bucle. Media vuelta después la pareja giraría
hacia atrás, por lo que un interruptor giratorio (el \emph{conmutador}\index{conmutador}) invierte el
corriente cada media vuelta. Cada abanico y taladro es este bucle multiplicado.
\end{remark}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:1}
Una barra \omterm{def:g11:magnetic-fields:magnet}{imán} se corta por la mitad entre sus polos. ¿Qué polos tiene cada uno?
llevar pieza? Reunidos de nuevo, ¿las caras recién cortadas se atraen?
o repeler? ¿Puede algún plan de corte aislar el polo norte?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:2}
Verdadero o falso, con una razón cada uno: (a) dos \omterm{def:g11:magnetic-fields:lines}{líneas de campo magnético}
cruce donde el \omterm{def:g11:electric-gravitational-fields:field}{campo} es fuerte; (b) fuera de una ejecución \omterm{def:g11:magnetic-fields:magnet}{imán}, \omterm{def:g11:electric-gravitational-fields:lines}{líneas de campo}
del polo norte al polo sur; (c) cada \omterm{def:g11:magnetic-fields:lines}{línea de campo magnético} es un
circuito cerrado; (d) una aguja se sitúa perpendicular a la línea que la atraviesa.
\end{exercise}

\section{Problema: La brújula, la grúa y la resonancia magnética}

\begin{problem}[{Problema de fin de semana --- tres imanes en acción: una brújula
cruzar un océano, un electroimán que levanta un automóvil y un solenoide de resonancia magnética
--- un campo, que abarca diez potencias de diez, haciendo tres trabajos}]
\label{pb:g11:magnetic-fields:1}
El mismo vector $\vect B$ gobierna un velero con \qty{50}{\micro T},
levanta chatarra con \qty{1}{T} e imagina una rodilla con \qty{1.5}{T}
--- tres máquinas para las tres fórmulas de este capítulo.

\medskip
\noindent\textbf{Parte I --- The compass.}
En la posición de la nave, el \omterm{def:g11:electric-gravitational-fields:field}{campo} de la Tierra tiene componente horizontal.
\qty{20}{\micro T} y componente vertical \qty{44}{\micro T};
el \omterm{def:g11:magnetic-fields:earth}{declinación} es $\ang{6}$ al oeste.
\begin{enumerate}
  \item ¿Por qué la aguja de una brújula apunta (magnéticamente) al norte?
  \item Calcula la magnitud total de \omterm{def:g11:electric-gravitational-fields:field}{campo} y su ángulo con la horizontal.
  \item Dirigiendo la ``brújula norte'' para \qty{30}{km}, ¿a qué distancia al oeste de
        ¿En qué rumbo norte verdadero termina el barco?
  \item La carga de acero desvía la aguja $\ang{4}$ más hacia el oeste: la misma pregunta.
  \item Cerca del polo geomagnético una brújula es inútil: ¿qué componente?
        tiene la culpa y por qué?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The crane.}
El desguace \omterm{def:g11:magnetic-fields:electromagnet}{electroimán} es un \omterm{def:g11:magnetic-fields:solenoid}{solenoide} de vueltas $600$ enrolladas
\qty{30}{cm}, de \omterm{def:g11:circuits-and-power:resistance}{resistencia} \qty{1.5}{\ohm}, alimentado por un \qty{12}{V}
suministro; su núcleo de hierro multiplica el \omterm{def:g11:electric-gravitational-fields:field}{campo} desnudo por $50$.
\begin{enumerate}[resume]
  \item Calcula $n$, la densidad de giro.
  \item Calcule la corriente en el \omterm{def:g11:magnetic-fields:solenoid}{bobina} (\cref{ch:g11:circuits-and-power}).
  \item Calcular el \omterm{def:g11:magnetic-fields:solenoid}{bobina desnuda} \omterm{def:g11:electric-gravitational-fields:field}{campo} $\mu_0 n I$.
  \item Calcular el \omterm{def:g11:electric-gravitational-fields:field}{campo} con el núcleo; compárelo con el \qty{0.5}{T} de neodimio.
  \item Calcular el \omterm{def:g10:energy-conservation:power}{fuerza} disipado en el \omterm{def:g11:magnetic-fields:solenoid}{bobina}, luego el \omterm{def:g10:energy-conservation:energy}{energía} para
        un turno de elevación \qty{10}{\minute}.
  \item Existe un \omterm{def:g11:magnetic-fields:magnet}{imán} permanente así de fuerte. ¿Por qué un \omterm{def:g11:magnetic-fields:electromagnet}{electroimán} de todos modos?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The winch motor.}
El cabrestante de la grúa es un motor CC: un bucle rectangular de $N = 100$ vueltas,
lados de longitud \qty{4.0}{cm} perpendiculares a $B = \qty{0.25}{T}$,
llevando $I = \qty{3.0}{A}$.
\begin{enumerate}[resume]
  \item Calcule \omterm{prop:g11:magnetic-fields:laplace}{fuerza de Laplace} en cada paquete lateral perpendicular a $\vect B$.
  \item Los dos \omterm{def:g10:inertia:force}{efectivo} son iguales y opuestos, pero su efecto no
        cancelar. ¿Qué producen y por qué deben ser \emph{opuesto}?
  \item ¿Qué actúa \omterm{def:g10:inertia:force}{fuerza} en los otros dos lados, cuando es paralelo a $\vect B$?
  \item Después de media vuelta, el mismo \omterm{def:g10:inertia:force}{efectivo} desharía la rotación.
        ¿Qué hace el \omterm{rem:g11:magnetic-fields:motor}{conmutador} y cuándo?
  \item Proporcione tres cambios de diseño separados, cada uno de los cuales duplica el par \omterm{def:g10:inertia:force}{fuerza}.
\end{enumerate}

\end{problem}
