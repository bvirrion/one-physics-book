\chapter{El núcleo y la radiactividad}\label{ch:g11:nucleus-radioactivity}

En este momento, unos ocho mil núcleos atómicos explotan dentro de tu cuerpo cada
segundo. Nada los ataca: ciertos núcleos nacen inestables y cada uno
tarde o temprano se transforma, arrojando un fragmento en una buena fracción
de la velocidad de la luz. Este capítulo regresa al piso $10^{-15}$ metros de
\cref{ch:g11:fundamental-interactions} para ver qué núcleos son frágiles, cómo
se rompen, y cómo su habitual impaciencia data cuevas y protege techos.

\section{Núcleos, nucleones e isótopos.}

\begin{definition}[Notación nuclear]\label{def:g11:nucleus-radioactivity:notation}
Un \omterm{def:g11:fundamental-interactions:ladder}{núcleo} contiene protones $Z$ (carga $+e$ cada uno) y neutrones $N$ (sin
cargo), $A = Z + N$ \omterm{def:g11:fundamental-interactions:ladder}{nucleones} en total; está escrito ${}^{A}_{Z}\mathrm{X}$,
donde $\mathrm X$ es el símbolo químico, $Z$ el \emph{atómico
numero}\index{número atómico} y $A$ el \emph{número de masa}\index{masa
numero}. Una especie especificada por $A$ y $Z$ es \emph{nucleido}\index{nucleido};
nucleidos con el mismo $Z$ pero diferente $A$ son \emph{isótopos}\index{isótopo}
de un elemento: misma química (la nube de electrones solo ve $Ze$),
diferentes núcleos y estabilidad.
\end{definition}

\begin{example}[Hidrógeno y carbono]\label{ex:g11:nucleus-radioactivity:hc}
El hidrógeno tiene tres \omterm{def:g11:nucleus-radioactivity:notation}{isótopos}: ${}^{1}_{1}\mathrm{H}$ (un protón solitario),
${}^{2}_{1}\mathrm{H}$ (deuterio: un protón, un neutrón), inestable
${}^{3}_{1}\mathrm{H}$ (tritio: dos neutrones). Carbono natural: principalmente
${}^{12}_{6}\mathrm{C}$ ($6$ protones, $6$ neutrones), $1.1\%$ de
${}^{13}_{6}\mathrm{C}$ y un \omterm{def:g11:fundamental-interactions:ladder}{átomo} en $10^{12}$ de inestable
${}^{14}_{6}\mathrm{C}$ ($8$ neutrones), héroe de \cref{ex:g11:nucleus-radioactivity:dating}.
\end{example}

\section{Núcleos estables e inestables.}

\begin{proposition}[El valle de la estabilidad]\label{prop:g11:nucleus-radioactivity:valley}
Estable \omterm{def:g11:nucleus-radioactivity:notation}{nucleidos} ocupa una banda estrecha en el plano $(Z, N)$: $N \approx Z$
para núcleos ligeros, luego un exceso de neutrones creciente, hasta $N \approx 1.5\,Z$.
Cada \omterm{def:g11:nucleus-radioactivity:notation}{nucleido} fuera de la banda: demasiados neutrones, demasiados protones o demasiado
grande (más allá del plomo, $Z = 82$, ningún \omterm{def:g11:nucleus-radioactivity:notation}{nucleido} es realmente estable) --- se transforma tarde o temprano.
\end{proposition}
\admitted

\begin{remark}
Localizar la banda exactamente es mecánica cuántica: los volúmenes universitarios.
El patrón es el libro mayor de \cref{ch:g11:fundamental-interactions}: fuerte
el pegamento une sólo a los vecinos, la repulsión de Coulomb se extiende por todo el \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, por lo que
los núcleos pesados necesitan pegamento de neutrones adicional; sin embargo, un exceso de neutrones es inestable
también (el \omterm{def:g11:fundamental-interactions:weak}{interacción débil} se encarga de ello).
\end{remark}

\begin{definition}[Radiactividad]\label{def:g11:nucleus-radioactivity:radioactivity}
\emph{Radioactividad}\index{radioactividad} es la transformación espontánea
de un \omterm{def:g11:fundamental-interactions:ladder}{núcleo} inestable en otro \omterm{def:g11:nucleus-radioactivity:notation}{nucleido}, con emisión de radiación. eso
es \emph{aleatorio}: nada anuncia qué \omterm{def:g11:fundamental-interactions:ladder}{núcleo} decaerá a continuación, y no
\omterm{def:g10:pressure:pressure}{presión}, la temperatura o la química pueden acelerarlo o retrasarlo; solo grande
las poblaciones son predecibles (\cref{def:g11:nucleus-radioactivity:halflife}).
\end{definition}

\begin{remark}[Becquerel y los Curie]\label{rem:g11:nucleus-radioactivity:curie}
En 1896, Henri Becquerel descubrió que las sales de uranio empañaban una placa fotográfica.
a través de papel negro, en un cajón cerrado, sin \omterm{def:g10:energy-conservation:energy}{energía} suministrado. María y
Pierre Curie demostró que el efecto es atómico, lo midió y aisló el polonio.
y el radio, un millón de veces más activo. La palabra \emph{\omterm{def:g11:nucleus-radioactivity:radioactivity}{radioactividad}} es
el de Marie Curie; también lo son dos premios Nobel.
\end{remark}

\section{Alfa, beta, gamma}

\begin{definition}[Las tres radiaciones historicas]\label{def:g11:nucleus-radioactivity:abg}
Tres radiaciones, nombradas antes de que nadie supiera cuáles eran:
\begin{itemize}
  \item \emph{alfa}\index{radiación alfa} ($\alpha$): un helio \omterm{def:g11:fundamental-interactions:ladder}{núcleo}
        ${}^{4}_{2}\mathrm{He}$ expulsado entero --- pesado, doblemente cargado,
        detenido por una hoja de papel o centímetros de aire;
  \item \emph{beta}\index{radiación beta} ($\beta^-$): un electrón
        ${}^{\;0}_{-1}\mathrm{e}$ creado en el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} y expulsado ---
        ligero, rápido, detenido por milímetros de aluminio;
  \item \emph{gama}\index{radiación gamma} ($\gamma$): electromagnético
        radiación, mucho más energética que la luz, nunca se detiene del todo, sólo
        \emph{atenuado}\index{atenuación}: un centímetro de plomo absorbe la mitad.
\end{itemize}
\end{definition}

\begin{proposition}[Conservación en ecuaciones nucleares]\label{prop:g11:nucleus-radioactivity:soddy}
En cada transformación nuclear, el total \omterm{def:g11:nucleus-radioactivity:notation}{número de masa} $A$ y el total
cargo --- la suma de los índices inferiores, contando $-1$ para un emitido
electrón --- son los mismos antes y después.
\end{proposition}
\admitted

\begin{remark}
Conservación de carga ya dictaminada \cref{ch:g11:fundamental-interactions};
La conservación de $A$ dice que \omterm{def:g11:fundamental-interactions:ladder}{nucleones} solo se convierten, nunca se crean o
destruido --- contabilidad honesta de los volúmenes de la universidad.
\end{remark}

\begin{method}[Equilibrio de una ecuación de decaimiento]\label{met:g11:nucleus-radioactivity:balance}
\begin{enumerate}
  \item Escriba cada actor con ambas etiquetas: $\alpha = {}^{4}_{2}\mathrm{He}$,
        $\beta^- = {}^{\;0}_{-1}\mathrm{e}$, $\beta^+ = {}^{0}_{+1}\mathrm{e}$; un $\gamma$ lleva $A = Z = 0$.
  \item Igualar las sumas de los $A$, luego de los $Z$
        (\cref{prop:g11:nucleus-radioactivity:soddy}); resolver y leer el
        elemento hijo de su $Z$ en la tabla \omterm{def:g10:signals-and-waves:period}{periódico}.
\end{enumerate}
\end{method}

\begin{example}[Una desintegración alfa]\label{ex:g11:nucleus-radioactivity:alpha}
El radio-226, el radio de Curie, es un emisor $\alpha$:
${}^{226}_{88}\mathrm{Ra} \to {}^{A}_{Z}\mathrm{Y} + {}^{4}_{2}\mathrm{He}$
\omterm{def:g10:inertia:force}{efectivo} $A = 226 - 4 = 222$ y $Z = 88 - 2 = 86$; El elemento $86$ es radón, por lo que
${}^{226}_{88}\mathrm{Ra} \to {}^{222}_{86}\mathrm{Rn} + {}^{4}_{2}\mathrm{He}$
--- el gas radiactivo de los sótanos sin ventilación.
\end{example}

\begin{example}[Una desintegración beta-menos]\label{ex:g11:nucleus-radioactivity:beta}
El carbono-14 se encuentra por encima de la banda (neutrones $8$ contra protones $6$):
${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} + {}^{\;0}_{-1}\mathrm{e}$.
Verificar: $14 = 14 + 0$, $6 = 7 - 1$; un neutrón se convirtió en un protón más el
electrón expulsado --- el \omterm{def:g11:fundamental-interactions:weak}{interacción débil} en \omterm{def:g11:work-of-force:work}{trabajar}.
\end{example}

\begin{remark}[Beta-plus y gamma]\label{rem:g11:nucleus-radioactivity:betaplus}
Debajo de la banda, la desintegración del espejo $\beta^+$ convierte un protón en un neutrón
más una \emph{positrón}\index{positrón} ${}^{0}_{+1}\mathrm{e}$, la antipartícula del electrón:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$,
caballo de batalla de los escáneres PET. Y después de la mayor decadencia la hija, nacida temblando,
se asienta emitiendo un $\gamma$ en la parte superior.
\end{remark}

\section{Vida media y actividad.}

\begin{definition}[vida media]\label{def:g11:nucleus-radioactivity:halflife}
El \emph{vida media}\index{vida media} $T_{1/2}$ de un \omterm{def:g11:nucleus-radioactivity:notation}{nucleido} es el tiempo
después de lo cual la mitad de cualquier muestra grande se ha descompuesto. Cada vida media adicional
reduce a la mitad lo que queda: de los núcleos $N_0$, $N_0/2$ quedan en $T_{1/2}$, $N_0/4$
en $2\,T_{1/2}$, $N_0/8$ en $3\,T_{1/2}$, $N_0/2^{\,n}$ después de las vidas medias $n$.
\end{definition}

\begin{remark}
Aleatoriedad para un \omterm{def:g11:fundamental-interactions:ladder}{núcleo}, mecanismo de relojería para $10^{23}$: una gran multitud de
Los lanzadores de monedas, impredecibles uno por uno, eran eliminados a medias en cada ronda.
Al interpolar las vidas medias enteras de \emph{entre} se toma la exponencial
función --- el año que viene, con el volumen de matemáticas.
\end{remark}

\begin{example}[Veinte potencias de diez]\label{ex:g11:nucleus-radioactivity:range}
\begin{center}\small\begin{tabular}{llll}
polonium-214 & $\alpha$ & \qty{1.6e-4}{s} & radium chain link \\
technetium-99m & $\gamma$ & $6.0$ hours & medical imaging \\
americium-241 & $\alpha$ & $432$ years & smoke detectors \\
carbon-14 & $\beta^-$ & \num{5700} years & archaeology \\
potassium-40 & $\beta$ & \num{1.25e9} years & in every banana \\
uranium-238 & $\alpha$ & \num{4.5e9} years & Earth's inner heat \\
\end{tabular}\end{center}
\end{example}

\begin{example}[Cita con carbono-14]\label{ex:g11:nucleus-radioactivity:dating}
La materia viva intercambia carbono con la atmósfera, por lo que retiene el
proporción atmosférica de carbono-14; al morir, la ingesta se detiene y el
la proporción se reduce a la mitad cada \num{5700} años. El carbón de una cueva pintada muestra
un cuarto de la proporción viva: un cuarto es la mitad de la mitad, por lo que el
El fuego quemó dos vidas medias, hace $2 \times \num{5700} = \num{11400}$ años.
\end{example}

\begin{definition}[actividad]\label{def:g11:nucleus-radioactivity:activity}
El \emph{actividad}\index{actividad} $\mathcal A$ de una muestra es su número
de desintegraciones por segundo, medido en \emph{becquereles}\index{becquerel}:
$\qty{1}{Bq} = 1$ decaimiento por segundo. Menos núcleos supervivientes significa menos
decae, por lo que la actividad también se reduce a la mitad en cada \omterm{def:g11:nucleus-radioactivity:halflife}{vida media}.
\end{definition}

\begin{example}[Órdenes de magnitud]\label{ex:g11:nucleus-radioactivity:orders}
Todo es ligeramente radiactivo. Un plátano: sobre \qty{15}{Bq}
(potasio-40). Un cuerpo humano: sobre \qty{8000}{Bq} --- el gancho de este capítulo,

inyección: aproximadamente \qty{5e8}{Bq}, desapareció en unos días.
\end{example}

\begin{example}[Tres carreras de un núcleo inestable]\label{ex:g11:nucleus-radioactivity:uses}
\emph{Medicamento}: tecnecio-99m, un emisor puro $\gamma$ con
$T_{1/2} = 6$ horas, se fija a un \omterm{def:g11:fundamental-interactions:ladder}{molécula} que el órgano diana absorbe; un
La cámara filma los $\gamma$ desde afuera, y al día siguiente el rastreador ha
en su mayoría desapareció, mientras que la radioterapia invierte la lógica, enfocando los rayos para
destruir un tumor. \emph{Tener una cita}: carbono-14 para respaldo de madera, hueso y tela

\end{example}

\begin{remark}[Dosis y el sievert]\label{rem:g11:nucleus-radioactivity:dose}
La radiación arranca electrones de \omterm{def:g11:fundamental-interactions:ladder}{moléculas} y puede dañar el ADN. lo biológico
el daño es rastreado por el \emph{dosis}\index{dosis}, en
\emph{sieverts}\index{sievert} (\unit{Sv}), que
pesa el \omterm{def:g10:energy-conservation:energy}{energía} depositado por \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} de tejido por lo dañino que es cada uno
la radiación es. Tres reglas: distancia, blindaje ($\alpha$: la piel es suficiente ---
a menos que el emisor sea \emph{inhalado}, delito de radón; $\beta$: aluminio;
$\gamma$: plomo) y tiempo. La dosimetría propiamente dicha es material universitario.
\end{remark}
