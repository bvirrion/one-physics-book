\chapter{Campos eléctricos y gravitacionales}\label{ch:g11:electric-gravitational-fields}

¿Cómo arranca la Tierra una manzana que nunca toca? El propio Newton llamó
acción a distancia "un absurdo tan grande" que se negó a defender
eso. La respuesta moderna: una masa o carga modifica el espacio que la rodea
--- crea un \emph{\omterm{def:g11:electric-gravitational-fields:field}{campo}} --- y lo que se coloque allí responde
al \omterm{def:g11:electric-gravitational-fields:field}{campo} en su propia ubicación. Construimos el lado de dos grandes ejemplos.
uno al lado del otro, hasta que la analogía se convierte en identidad de forma.

\section{De la acción a distancia al campo}

\begin{definition}[Campo]\label{def:g11:electric-gravitational-fields:field}
Un \emph{campo}\index{campo} asigna a cada punto del espacio un vector. un
fuente (una carga, una masa) crea su campo en todas partes a su alrededor,
whether or not anything is there to feel it; a test object then feels a
\omterm{def:g10:inertia:force}{fuerza} determinado únicamente por el campo \emph{en su propia ubicación}. esto
two-step picture replaces action at a distance.
\end{definition}

\begin{remark}
El \omterm{def:g11:electric-gravitational-fields:field}{campo} no es un truco contable: lleva \omterm{def:g10:energy-conservation:energy}{energía} y los cambios en
la fuente viaja hacia afuera a través de ella a una velocidad finita --- la luz misma es
un viajero \omterm{def:g11:electric-gravitational-fields:field}{campo} (\cref{ch:g12:light-as-wave}). aqui estudiamos
\omterm{def:g11:electric-gravitational-fields:field}{campos} que no cambian en el tiempo.
\end{remark}

\section{El campo electrico}

\begin{definition}[campo electrico]\label{def:g11:electric-gravitational-fields:efield}
Si una pequeña carga de prueba $q$ colocada en un punto $M$ siente una descarga eléctrica \omterm{def:g10:inertia:force}{fuerza}
$\vect F$, el \emph{campo eléctrico}\index{campo eléctrico} en $M$ es
\[
\vect E = \frac{\vect F}{q}
\qquad \text{(in \unit{N/C}, also written \unit{V/m})}.
\]
$\vect E$ depende sólo de las fuentes y de $M$, no de la prueba
cargo: duplicar $q$ duplica $\vect F$, dejando el cociente sin cambios.
\end{definition}

\begin{proposition}[Fuerza sobre una carga]\label{prop:g11:electric-gravitational-fields:force}
Una carga $q$ colocada donde está \omterm{def:g11:electric-gravitational-fields:field}{campo} $\vect E$ siente la \omterm{def:g10:inertia:force}{fuerza}
$\vect F = q \vect E$: junto a $\vect E$ si $q > 0$, opuesto a $\vect E$
si $q < 0$, de magnitud $F = \abs{q}\,E$.
\end{proposition}

\begin{proof}
Reorganizar la definición; la regla de los signos es el álgebra de un múltiplo negativo.
\end{proof}

\begin{example}[Leyendo un campo]\label{ex:g11:electric-gravitational-fields:reading}
Una carga de prueba $q = \qty{+2.0e-8}{C}$ en $M$ siente una \omterm{def:g10:inertia:force}{fuerza} de
\qty{6.0e-4}{N} apuntando al este. El \omterm{def:g11:electric-gravitational-fields:field}{campo} en $M$ es
$E = \num{6.0e-4} / \num{2.0e-8} = \qty{3.0e4}{N/C}$, apuntando al este;
$q' = \qty{-1.0e-8}{C}$ colocado en el mismo punto se siente
$F = \num{1.0e-8} \times \num{3.0e4} = \qty{3.0e-4}{N}$ --- apuntando al oeste.
\end{example}

\begin{proposition}[Campo de una carga puntual]\label{prop:g11:electric-gravitational-fields:point}
Una carga puntual $Q$ crea, a la distancia $d$, una \omterm{def:g11:electric-gravitational-fields:field}{campo} a lo largo de la línea
uniendo la carga al punto --- lejos de $Q$ si $Q > 0$, hacia él si
$Q < 0$ --- de magnitud
\[
E = k\,\frac{\abs{Q}}{d^2},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{proposition}

\begin{proof}
La ley de Coulomb (\cref{ch:g11:fundamental-interactions}) da el \omterm{def:g10:inertia:force}{fuerza}
en una carga de prueba $q$ a la distancia $d$: magnitud $k \abs{qQ} / d^2$,
a lo largo de la línea de unión, repulsivo para signos similares. Dividir por $q$ y
aplicar \cref{prop:g11:electric-gravitational-fields:force}.
\end{proof}

\begin{example}[Una esfera cargada]\label{ex:g11:electric-gravitational-fields:sphere}
Una esfera de Van de Graaff lleva $Q = \qty{+5.0e-7}{C}$ (una pequeña esfera cargada
esfera actúa como una carga puntual en su centro). En $d = \qty{0.50}{m}$:
\[
E = \num{8.99e9} \times \num{5.0e-7} / 0.50^2 \approx \qty{1.8e4}{N/C},
\]
apuntando radialmente lejos de la esfera. En \qty{1.0}{m} ha bajado
a una cuarta parte de esto: el $1/d^2$ del \omterm{def:g10:inertia:force}{fuerza} sobrevive en el \omterm{def:g11:electric-gravitational-fields:field}{campo}.
\end{example}

\begin{definition}[Líneas de campo]\label{def:g11:electric-gravitational-fields:lines}
A \emph{linea de campo}\index{linea de campo} es una curva tangente en todas partes a
el \omterm{def:g11:electric-gravitational-fields:field}{campo}, una flecha que indica la dirección del \omterm{def:g11:electric-gravitational-fields:field}{campo}. Tres reglas:
\begin{itemize}
  \item salen cargas positivas y llegan a cargas negativas;
  \item donde las líneas se agolpan, el \omterm{def:g11:electric-gravitational-fields:field}{campo} es fuerte; donde se extienden, débiles;
  \item dos líneas nunca se cruzan: la \omterm{def:g11:electric-gravitational-fields:field}{campo} tiene una dirección en cada punto.
\end{itemize}
\end{definition}

\section{El campo uniforme de un condensador.}

\begin{definition}[Condensador de placas paralelas]\label{def:g11:electric-gravitational-fields:capacitor}
Dos placas metálicas paralelas enfrentadas, muy juntas y llevando
cargos, forma un \emph{condensador}\index{condensador}. entre los platos
el \omterm{def:g11:electric-gravitational-fields:field}{campo} es \emph{uniforme}\index{campo uniforme}: misma magnitud y
dirección en todas partes --- perpendicular a las placas, desde el positivo
placa al negativo.
\end{definition}

\begin{proposition}[Campo entre las placas]\label{prop:g11:electric-gravitational-fields:uniform}
Si el \omterm{def:g11:circuits-and-power:voltage}{Voltaje} entre las placas (\cref{ch:g11:circuits-and-power}) está
$U$ y su separación es $d$, el \omterm{def:g11:electric-gravitational-fields:field}{campo} entre ellos tiene magnitud
\[
E = \frac{U}{d} \qquad \text{(in \unit{V/m})}.
\]
\end{proposition}
\admitted

\begin{remark}
Uniformidad y $E = U/d$ se derivan honestamente en el volumen del Año 1; el
La fórmula explica el \omterm{def:g10:orders-of-magnitude:unit}{unidad} \unit{V/m}. \omterm{def:g11:lenses-and-eye:lens}{Delgado} los huecos se hacen fuertes \omterm{def:g11:electric-gravitational-fields:field}{campos}:
\qty{100}{V} en un milímetro ya es \qty{e5}{V/m}.
\end{remark}

\begin{example}[Dirección de un haz de electrones]\label{ex:g11:electric-gravitational-fields:beam}
En un \omterm{def:g10:signals-and-waves:oscilloscope}{osciloscopio}, el haz pasa entre placas con
$U = \qty{100}{V}$ y $d = \qty{5.0}{mm}$:
$E = 100 / \num{5.0e-3} = \qty{2.0e4}{V/m}$, y cada electrón se siente
$F = \num{1.60e-19} \times \num{2.0e4} = \qty{3.2e-15}{N}$ hacia el
placa positiva --- $\num{4e14}$ veces su \omterm{def:g10:universal-gravitation:weight}{peso} de \qty{8.9e-30}{N},
por lo que el haz se desvía visiblemente (la trayectoria curva es
\cref{ch:g12:newtons-laws}).
\end{example}

\section{El campo gravitacional}

\begin{definition}[campo gravitacional]\label{def:g11:electric-gravitational-fields:gfield}
Si una masa de prueba $m$ colocada en un punto siente un \omterm{def:g10:universal-gravitation:force}{fuerza gravitacional} (un
\omterm{def:g10:universal-gravitation:weight}{peso}) $\vect P$, el \emph{campo gravitacional}\index{gravitacional
campo} que hay
\[
\vect g = \frac{\vect P}{m} \qquad \text{(in \unit{N/kg})},
\]
y a la inversa $\vect P = m \vect g$. Un capítulo anterior presentó la
número $g$; $\vect g$ agrega la dirección en la que comienza a moverse una piedra caída.
\end{definition}

\begin{proposition}[campo de la tierra]\label{prop:g11:electric-gravitational-fields:earth}
A la distancia $d$ del centro de la Tierra (masa $M$), el \omterm{def:g11:electric-gravitational-fields:gfield}{gravitacional
campo} apunta hacia el centro y tiene magnitud
\[
g = G\,\frac{M}{d^2},
\qquad G = \qty{6.67e-11}{N.m^2/kg^2}.
\]
\end{proposition}

\begin{proof}
Universal \omterm{def:g11:fundamental-interactions:four}{gravitación} (\cref{ch:g10:universal-gravitation}) da la
\omterm{def:g10:inertia:force}{fuerza} sobre una masa de prueba $m$: magnitud $G M m / d^2$, hacia el centro;
dividir por $m$. En la superficie, $d = R$ y este es el
$g = GM/R^2 \approx \qty{9.81}{N/kg}$ del capítulo anterior.
\end{proof}

\begin{example}[El campo donde vive la Luna]\label{ex:g11:electric-gravitational-fields:moon}
A la distancia de la Luna $d = \qty{3.84e8}{m}$:
\[
g = \num{6.67e-11} \times \num{5.97e24} / (\num{3.84e8})^2
\approx \qty{2.7e-3}{N/kg}.
\]
El \omterm{def:g11:electric-gravitational-fields:field}{campo} de la Tierra nunca se detiene; solo aparece \omterm{def:g11:lenses-and-eye:lens}{adelgaza} como $1/d^2$; este es el
\omterm{def:g11:electric-gravitational-fields:field}{campo} que sostiene a la Luna en su \omterm{def:g10:universal-gravitation:satellite}{órbita}.
\end{example}

\begin{remark}[Uniforme cerca de la superficie]\label{rem:g11:electric-gravitational-fields:uniformg}
Sobre un laboratorio o incluso una montaña, $d$ apenas cambia en comparación con
$R = \qty{6.37e6}{m}$, por lo que $\vect g$ es \omterm{def:g11:electric-gravitational-fields:capacitor}{uniforme} con alta precisión:
líneas verticales, constante \qty{9.81}{N/kg} --- el aspecto \omterm{def:g11:electric-gravitational-fields:field}{campo} de la Tierra
como el \omterm{def:g11:electric-gravitational-fields:capacitor}{condensador}, así como la Tierra redonda parece plana desde un jardín.
\end{remark}

\section{Una idea, dos fuerzas}

Los dos \omterm{def:g11:electric-gravitational-fields:field}{campos} de este capítulo son las mismas matemáticas con dos disfraces:

\begin{center}
\begin{tabular}{lcc}
 & \textbf{electric} & \textbf{gravitational} \\
\hline
source & charge $Q$ & mass $M$ \\
\omterm{def:g11:electric-gravitational-fields:field}{campo} & $\vect E = \vect F / q$ & $\vect g = \vect P / m$ \\
\omterm{def:g10:orders-of-magnitude:unit}{unidad} & \unit{N/C} = \unit{V/m} & \unit{N/kg} \\
point source & $E = k \abs{Q} / d^2$ & $g = G M / d^2$ \\
constant & $k = \qty{8.99e9}{N.m^2/C^2}$ & $G = \qty{6.67e-11}{N.m^2/kg^2}$ \\
\omterm{def:g10:inertia:force}{fuerza} on a test & $\vect F = q \vect E$ & $\vect P = m \vect g$ \\
\omterm{def:g11:electric-gravitational-fields:lines}{líneas de campo} & out of $+$, into $-$ & always into the mass \\
\end{tabular}
\end{center}

Cada fórmula de la izquierda se convierte en la de la derecha debajo del
diccionario $q \leftrightarrow m$, $\vect E \leftrightarrow \vect g$, $k \leftrightarrow G$.

\begin{remark}[Donde se rompe la analogía]\label{rem:g11:electric-gravitational-fields:noanalogue}
La carga se presenta en dos signos; masa en uno. Entonces \omterm{def:g11:electric-gravitational-fields:efield}{campos electricos} puede ser
cancelado: las cargas reordenadas de una caja metálica borran cualquier exterior
\omterm{def:g11:electric-gravitational-fields:field}{campo} dentro de él (un \emph{jaula de faraday} --- por qué un ascensor mata tu
teléfono \omterm{def:g10:signals-and-waves:signal}{señal}), y la materia a granel, con $+$ y $-$ balanceados, es
eléctricamente silencioso. La gravedad no se puede proteger ni neutralizar:
cada \omterm{def:g10:orders-of-magnitude:unit}{kilogramo} agrega su atractivo --- razón por la cual el \omterm{def:g10:inertia:force}{fuerza} es más débil en
$\num{e39}$ (\cref{exo:g11:electric-gravitational-fields:10}) ejecuta el
universo a gran escala.
\end{remark}
