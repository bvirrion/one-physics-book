\chapter{Las interacciones fundamentales}\label{ch:g11:fundamental-interactions}

Frota un globo en tu cabello y este recoge trozos de papel --- golpeando,
Con unos pocos centímetros cuadrados de goma, la atracción gravitacional del
planeta entero. Esa victoria fácil es la razón por la que \omterm{def:g11:fundamental-interactions:ladder}{átomos} se mantiene unido y por qué
los núcleos pesados eventualmente se rompen (\cref{ch:g11:nucleus-radioactivity}).
El universo se basa en exactamente cuatro interacciones; este capitulo cumple
los pesa, los compara y asigna a cada uno su piso.

\section{Materia de quarks a galaxias}

\begin{definition}[La escalera de estructura]\label{def:g11:fundamental-interactions:ladder}
La materia se construye en pisos, cada uno ensamblado a partir del siguiente: tres
Enlace \emph{quarks}\index{cuarc} (puntiagudo, más pequeño que \qty{e-18}{m})
en un \emph{nucleón}\index{nucleón} --- protón o neutrón, aproximadamente
\qty{e-15}{m} a través; Los nucleones se empaquetan en un \emph{núcleo}\index{núcleo}.
(\qty{e-15}{m} a \qty{e-14}{m}); un núcleo más su nube de electrones es
un \emph{átomo}\index{átomo} (\qty{e-10}{m}); los átomos se unen en
\emph{moléculas}\index{molécula} y cristales (\qty{e-9}{m} arriba), que
ensamblar en objetos cotidianos (\qty{1}{m}), planetas (\qty{e7}{m}),
sistemas planetarios (\qty{e11}{m}), galaxias (\qty{e21}{m}) y el
universo observable (\qty{e26}{m}).
\end{definition}

\begin{definition}[Las cuatro interacciones]\label{def:g11:fundamental-interactions:four}
Un \emph{interacción fundamental}\index{interacción fundamental} es un
\omterm{def:g10:inertia:force}{fuerza} no reducible a otro \omterm{def:g10:inertia:force}{efectivo}; hay exactamente cuatro:
\emph{gravitación}\index{gravitación}, el \emph{interacción electromagnética}\index{interacción electromagnética},
el \omterm{def:g11:fundamental-interactions:strong}{fuerte interacción} y el \omterm{def:g11:fundamental-interactions:weak}{interacción débil}.
\end{definition}

\begin{remark}\label{rem:g11:fundamental-interactions:empty}
Los pisos están separados por huecos: un \omterm{def:g11:fundamental-interactions:ladder}{átomo} es $10^{5}$ veces más ancho que
es \omterm{def:g11:fundamental-interactions:ladder}{núcleo} --- escale el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} hasta una canica \qty{1}{cm} y el
La nube de electrones tiene un kilómetro de ancho. Algo debe contener estos escasos
pisos juntos, y la gravedad casi nunca lo es.
\end{remark}

\section{carga electrica}

\begin{definition}[carga electrica]\label{def:g11:fundamental-interactions:charge}
\emph{carga electrica}\index{carga electrica} es propiedad de la materia
sobre el cual actúa el \omterm{def:g11:fundamental-interactions:four}{interacción electromagnética}, ya que la masa es la propiedad
\omterm{def:g11:fundamental-interactions:four}{gravitación} actúa sobre. Se mide en \emph{culombios}\index{culombio}
(\unit{C}) y viene en \emph{dos signos}, positivo y negativo, lo cual
cancelar cuando se agregue; Las cargas similares se repelen, las cargas diferentes se atraen.
\end{definition}

\begin{definition}[Elemental de carga]\label{def:g11:fundamental-interactions:elementary}
La carga es granulada: cada carga observada es un múltiplo entero de la
\emph{carga elemental}\index{carga elemental}
$e = \qty{1.60e-19}{C}$. El protón transporta $+e$, el electrón $-e$, el
neutrón $0$; una carga $q$ contiene cargas elementales $N = \abs{q}/e$.
\end{definition}

\begin{proposition}[Conservación de la carga]\label{prop:g11:fundamental-interactions:conservation}
La carga total de un sistema aislado nunca cambia: la carga no es ni
creado ni destruido, sólo transferido --- en la práctica por electrones,
los transportistas más ligeros.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:conservation}
Nunca se ha observado ninguna violación; la razón profunda, una simetría de
electromagnetismo, se da en los volúmenes universitarios.
\end{remark}

\begin{definition}[Carga por fricción y por contacto]\label{def:g11:fundamental-interactions:charging}
Al frotar dos aisladores, los electrones se desprenden de uno y pasan al otro,
dejando cargas opuestas de igual magnitud (\emph{cargando por
fricción}\index{carga por fricción}); tocar un objeto cargado con un
el neutro comparte la carga (\emph{cargando por
contacto}\index{carga por contacto}). Sólo los electrones se mueven; el total
la carga se conserva.
\end{definition}

\begin{example}[Contando electrones]\label{ex:g11:fundamental-interactions:balloon}
Un globo frotado sobre el cabello adquiere $q = \qty{-1.6e-8}{C}$: tiene
\emph{ganado}$N = \num{1.6e-8} / \num{1.6e-19} = \num{1.0e11}$
electrones, y el cabello queda exactamente con $+\qty{1.6e-8}{C}$.
\end{example}

\section{ley de coulomb}

\begin{theorem}[ley de coulomb]\label{thm:g11:fundamental-interactions:coulomb}
Dos cargas puntuales $q_1$ y $q_2$ separadas por una distancia $d$ ejercen sobre cada una
otros \omterm{def:g10:inertia:force}{efectivo} a lo largo de la línea que los une, de iguales magnitudes y
direcciones opuestas --- repulsivo para signos similares, atractivo para signos diferentes
signos --- con
\[
F = k\,\frac{\abs{q_1 q_2}}{d^{2}},
\qquad k = \qty{8.99e9}{N.m^2/C^2}.
\]
\end{theorem}
\admitted

\begin{remark}\label{rem:g11:fundamental-interactions:torsion}
Coulomb estableció la ley en 1785 midiendo la torsión de un \omterm{def:g11:lenses-and-eye:lens}{delgado}
fibra de torsión; Por qué \omterm{def:g10:orders-of-magnitude:scinot}{exponente} es exactamente $2$ se responde en los volúmenes de la universidad.
\end{remark}

\begin{remark}[Un gemelo formal de la gravitación]\label{rem:g11:fundamental-interactions:twin}
La ley de Coulomb es, símbolo por símbolo, la ley del universal \omterm{def:g11:fundamental-interactions:four}{gravitación}
(\cref{ch:g10:universal-gravitation}) con cargas en lugar de masas:
\begin{center}
\begin{tabular}{lll}
source & mass $m$ & charge $q$ \\
law & $F = G\,\dfrac{m_1 m_2}{d^2}$ & $F = k\,\dfrac{\abs{q_1 q_2}}{d^2}$ \\[1.5ex]
constant & $G = \qty{6.67e-11}{N.m^2/kg^2}$ & $k = \qty{8.99e9}{N.m^2/C^2}$ \\
sign & always attractive & attractive or repulsive \\
\end{tabular}
\end{center}
Las dos diferencias impulsan todo lo siguiente: la electricidad es
abrumadoramente más fuerte, y sólo él puede cancelarse a sí mismo.
\end{remark}

\begin{example}[Dos protones, ambas fuerzas a la vez]\label{ex:g11:fundamental-interactions:protons}
Dos protones ($m_p = \qty{1.67e-27}{kg}$) se sientan $d = \qty{1.0e-15}{m}$
aparte --- vecinos nucleares. Repulsión eléctrica:
\[
F_E = \num{8.99e9} \times
\frac{(\num{1.60e-19})^{2}}{(\num{1.0e-15})^{2}}
\approx \qty{2.3e2}{N}
\]
--- el \omterm{def:g10:universal-gravitation:weight}{peso} de una maleta de $23$ kg, transportado por una partícula de
$10^{-27}$\omterm{def:g10:orders-of-magnitude:unit}{kilogramos}. Atracción gravitacional:
$F_G = \num{6.67e-11} \times (\num{1.67e-27})^{2} /
(\num{1.0e-15})^{2} \approx \qty{1.9e-34}{N}$. la proporción
$F_E / F_G = k e^{2} / (G m_p^{2}) \approx \num{1.2e36}$ es
independiente de la distancia ($d^{2}$ cancela): entre partículas elementales,
la gravedad es $10^{36}$ veces demasiado débil para importar --- a una distancia \emph{cualquier}.
\end{example}

\section{Las interacciones fuertes y débiles.}

El ejemplo deja un escándalo: un helio \omterm{def:g11:fundamental-interactions:ladder}{núcleo} contiene dos protones que
repeler con decenas de \omterm{def:g10:inertia:force}{newton} --- y aún así existe el helio.

\begin{definition}[La interacción fuerte]\label{def:g11:fundamental-interactions:strong}
El \emph{fuerte interacción}\index{fuerte interacción} se une a \omterm{def:g11:fundamental-interactions:ladder}{quarks}
en \omterm{def:g11:fundamental-interactions:ladder}{nucleones} y \omterm{def:g11:fundamental-interactions:ladder}{nucleones} en núcleos. En \qty{e-15}{m} es
atractivo y mucho más fuerte que la repulsión eléctrica (miles de
\omterm{def:g10:inertia:force}{newton} entre \omterm{def:g11:fundamental-interactions:ladder}{nucleones}); más allá de unos pocos $10^{-15}$ \omterm{def:g10:orders-of-magnitude:unit}{metros} desaparece:
su \emph{rango}\index{rango (de una interacción)} se trata de
\qty{e-15}{m}. Agarra protones y neutrones por igual e ignora la carga.
\end{definition}

\begin{remark}[Por qué existen los núcleos --- y por qué los pesados ceden]\label{rem:g11:fundamental-interactions:nuclei}
Los dos \omterm{def:g10:inertia:force}{efectivo} en el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} escalan de manera diferente. El pegamento fuerte es
de corto alcance: un \omterm{def:g11:fundamental-interactions:ladder}{nucleón} une sólo a sus pocos vecinos dentro
\qty{e-15}{m}. La repulsión eléctrica es de largo alcance: cada protón
empuja el protón \emph{uno sí y otro no}. A medida que los núcleos crecen, se produce repulsión.
se acumula más rápido que el pegamento: los núcleos pesados necesitan neutrones adicionales (pegamento
sin carga), y más allá de aproximadamente $80$ protones, incluso eso falla --- el
los núcleos más pesados viven de tiempo prestado (\cref{ch:g11:nucleus-radioactivity}).
\end{remark}

\begin{definition}[La interacción débil]\label{def:g11:fundamental-interactions:weak}
El \emph{interacción débil}\index{interacción débil}, de \omterm{def:g11:fundamental-interactions:strong}{rango} más corto
Aun así, dejamos que un neutrón se convierta en un protón (y viceversa): impulsa el
$\beta$ radiactividad de \cref{ch:g11:nucleus-radioactivity}.
\end{definition}

\section{¿Quién gobierna qué piso?}

\begin{method}[Auditando una báscula]\label{met:g11:fundamental-interactions:audit}
Para encontrar qué interacción rige una estructura de tamaño $d$, consulte \omterm{def:g11:fundamental-interactions:strong}{rango}
y cancelación:
\begin{enumerate}
  \item $d \leq \qty{e-14}{m}$: las reglas \emph{\omterm{def:g11:fundamental-interactions:strong}{fuerte interacción}} ---
        vence a la repulsión eléctrica y la gravedad está fuera $10^{36}$.
  \item \omterm{def:g11:fundamental-interactions:ladder}{átomos} a montañas ($\qty{e-10}{m}$ a $\qty{e4}{m}$): el
        fuerte \omterm{def:g10:inertia:force}{fuerza} está fuera de \omterm{def:g11:fundamental-interactions:strong}{rango}; las reglas \emph{\omterm{def:g11:fundamental-interactions:four}{electromagnético
        interacción}} --- vínculos, cohesión, \omterm{def:g10:inertia:inventory}{fricción}, contacto.
  \item planetas y más allá ($d \geq \qty{e6}{m}$): la materia es neutral para
        precisión fantástica, por lo que la electricidad se anula; masa tiene
        un signo y siempre agrega --- \emph{\omterm{def:g11:fundamental-interactions:four}{gravitación}} gobierna el cielo.
\end{enumerate}
El \omterm{def:g11:fundamental-interactions:weak}{interacción débil} no tiene derecho a palabra: no vincula nada, pero arbitra
transformaciones (desintegración $\beta$).
\end{method}

\section{Problema: La auditoría de las cuatro fuerzas}

\begin{problem}[{Problema del fin de semana --- por qué el mundo tampoco colapsa
ni se separa: las cuatro interacciones auditadas piso por piso, hasta
neutralidad de la materia a dos partes en $10^{18}$}]
\label{pb:g11:fundamental-interactions:1}
\omterm{def:g11:fundamental-interactions:ladder}{átomos} no implosionan, los núcleos en su mayoría se mantienen, la Luna no choca ni
escapa; Este problema audita quién merece el crédito. Datos:
$e = \qty{1.60e-19}{C}$,
$k = \qty{8.99e9}{N.m^2/C^2}$, $G = \qty{6.67e-11}{N.m^2/kg^2}$,
$m_p = \qty{1.67e-27}{kg}$, $m_e = \qty{9.11e-31}{kg}$,
$M_E = \qty{5.97e24}{kg}$, $M_M = \qty{7.35e22}{kg}$, Tierra-Luna
distancia \qty{3.84e8}{m}.

\medskip
\noindent\textbf{Parte I --- The ladder.}
\begin{enumerate}
  \item Enumere los pisos de la escalera desde \omterm{def:g11:fundamental-interactions:ladder}{cuarc} hasta la galaxia, un \omterm{def:g10:orders-of-magnitude:oom}{orden
        de magnitud} de tamaño cada uno.
  Modelo a escala \item: el \omterm{def:g11:fundamental-interactions:ladder}{núcleo} se convierte en una canica \qty{1}{cm}; que ancho
        ¿Cuál es el \omterm{def:g11:fundamental-interactions:ladder}{átomo}?
  \item ¿Qué fracción del volumen de un \omterm{def:g11:fundamental-interactions:ladder}{átomo} ocupa su \omterm{def:g11:fundamental-interactions:ladder}{núcleo}?
  \item Un cabello tiene un grosor \qty{70}{\micro m}. ¿Cuántos \omterm{def:g11:fundamental-interactions:ladder}{átomos} lo abarcan?
  \item ¿Cuántos \omterm{def:g10:energy-conservation:power}{potestades} de diez del techo \omterm{def:g11:fundamental-interactions:ladder}{cuarc} (\qty{e-18}{m})
        a una galaxia (\qty{e21}{m})?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The electric floor.}
\begin{enumerate}[resume]
  \item En un hidrógeno \omterm{def:g11:fundamental-interactions:ladder}{átomo} el electrón se encuentra $d = \qty{5.3e-11}{m}$
        del protón: calcular el \omterm{def:g10:inertia:force}{fuerza} eléctrico entre ellos.
  \item Calcule el \omterm{def:g10:universal-gravitation:force}{fuerza gravitacional} entre ellos y la relación de
        los dos. ¿Cuál mantiene unido el \omterm{def:g11:fundamental-interactions:ladder}{átomo}?
  \item ¿Qué cambia si la gravedad se apaga dentro de \omterm{def:g11:fundamental-interactions:ladder}{átomos}? y si
        la electricidad es?
  \item ¿Por qué son los ``contactos'' \omterm{def:g10:inertia:force}{efectivo} de la vida cotidiana --- el piso?
        empujando con los pies, \omterm{def:g10:inertia:inventory}{fricción} --- ¿magnetismo disfrazado?
  \item Cada una de las dos monedas contiene aproximadamente $\num{e23}$ electrones. Transferir
        un electrón en un \emph{mil millones} entre ellos, configúrelos
        \qty{1.0}{m} aparte: calcular los cargos y el \omterm{def:g10:inertia:force}{fuerza}. ¿Conclusión?
  \item ¿Qué interacción rige los pisos desde \qty{e-10}{m} hasta
        \qty{e4}{m}, ¿y por qué el \omterm{def:g11:fundamental-interactions:strong}{fuerte interacción} no compite?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The nuclear floor.}
\begin{enumerate}[resume]
  \item Dos protones están separados \qty{1.0e-15}{m} en un \omterm{def:g11:fundamental-interactions:ladder}{núcleo}: calcular
        su repulsión eléctrica.
  \item Desenfrenado, ¿qué aceleración le daría este \omterm{def:g10:inertia:force}{fuerza} a un protón?
        Comparar con $g$.
  \item Enumere las tres propiedades que el \omterm{def:g11:fundamental-interactions:strong}{fuerte interacción} necesita
        explicar por qué existen los núcleos, por qué los neutrones también están atrapados y por qué
        los núcleos son diminutos.
  \item En un uranio \omterm{def:g11:fundamental-interactions:ladder}{núcleo} (diámetro \qty{1.4e-14}{m}, protones $92$),
        Calcule la repulsión entre dos protones en extremos opuestos. puede
        ¿El \omterm{def:g11:fundamental-interactions:strong}{fuerte interacción} une ese par directamente? Explica por qué
        La repulsión gana a medida que crecen los núcleos y para qué sirven los neutrones.
  \item En una frase, como prometí: ¿qué hace el \omterm{def:g11:fundamental-interactions:weak}{interacción débil}?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The astronomical floor, and the verdict.}
\begin{enumerate}[resume]
  \item Calcula el \omterm{def:g10:universal-gravitation:force}{fuerza gravitacional} entre la Tierra y la Luna.
  \item La Tierra contiene alrededor de $\num{1.8e51}$ protones (y tantos
        electrones), la Luna alrededor de $\num{2.2e49}$. ¿Qué fracción $f$ de
        La carga de protones de cada cuerpo, si no se cancela, haría que el
        ¿El \omterm{def:g10:inertia:force}{fuerza} eléctrico coincide con el gravitacional?
  \item ¿Por qué manda \omterm{def:g11:fundamental-interactions:four}{gravitación}, el debilucho de la pregunta 7?
        ¿astronomía? Dos razones.
  Veredicto \item --- una oración por piso (\omterm{def:g11:fundamental-interactions:ladder}{núcleo}, \omterm{def:g11:fundamental-interactions:ladder}{átomo} a montaña,
        planeta y arriba): ¿qué lo mantiene unido? Luego responde el título:
        ¿Por qué el mundo no colapsa ni se desintegra?
\end{enumerate}
\end{problem}
