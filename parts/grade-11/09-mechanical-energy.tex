\chapter{Mechanical Energy and Its Conservation}\label{ch:g11:mechanical-energy}

A roller coaster has no engine past the first hill: every loop, every
burst of speed afterwards is spent from the account opened on the way up.
This chapter turns the bookkeeping into two numbers --- kinetic and
potential energy --- and one rule: while only weight works, their sum
does not change. Friction is the fee; we will learn to read it off the
books.

\section{Kinetic energy}

\begin{definition}[Kinetic energy]\label{def:g11:mechanical-energy:kinetic}
A body of mass $m$ (\unit{kg}) moving at speed $v$ (\unit{m/s}) carries
the \emph{kinetic energy}\index{kinetic energy}
\[
E_k = \tfrac12\, m v^2 .
\]
Units check: $\unit{kg.m^2/s^2} = \unit{N.m} = \unit{J}$ --- joules,
like every energy (\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Orders of magnitude]\label{ex:g11:mechanical-energy:orders}
\begin{itemize}
  \item a pedestrian, \qty{70}{kg} at \qty{1.4}{m/s}:
        $E_k = \tfrac12 \times 70 \times 1.4^2 \approx \qty{69}{J}$;
  \item a car, \qty{1300}{kg} at \qty{130}{km/h} $= \qty{36.1}{m/s}$:
        $E_k \approx \qty{8.5e5}{J}$;
  \item a high-speed train, $m = \qty{3.85e5}{kg}$, at \qty{320}{km/h}
        $= \qty{88.9}{m/s}$: $E_k \approx \qty{1.5e9}{J}$;
  \item a rifle bullet, \qty{8.0}{g} at \qty{800}{m/s}: $E_k \approx \qty{2.6}{kJ}$.
\end{itemize}
Seven orders of magnitude separate the stroll from the train. And $E_k$
grows like the \emph{square} of the speed: at \qty{130}{km/h} the car
carries $4$ times the energy it has at \qty{65}{km/h} --- four times
what its brakes must remove.
\end{example}

\section{The kinetic-energy theorem}

\begin{theorem}[Kinetic-energy theorem]\label{thm:g11:mechanical-energy:ketheorem}
\index{kinetic-energy theorem}
For a body of mass $m$ moving along a straight line from $A$ to $B$,
the change of kinetic energy equals the total work
(\cref{ch:g11:work-of-force}) received:
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
the sum running over all forces acting on the body. It extends to
motion along any path, as we also admit here.
\end{theorem}
\admitted

\begin{example}[Free fall, checked frame by frame]\label{ex:g11:mechanical-energy:freefall}
A \qty{0.200}{kg} ball is dropped and filmed; the frames give:
\begin{center}\small
\begin{tabular}{lcccc}
$t$ (\unit{s}) & 0.10 & 0.20 & 0.30 & 0.40\\
fallen height $h$ (\unit{m}) & 0.049 & 0.196 & 0.441 & 0.785\\
speed $v$ (\unit{m/s}) & 0.98 & 1.96 & 2.94 & 3.92\\
$E_k = \tfrac12 mv^2$ (\unit{J}) & 0.096 & 0.384 & 0.864 & 1.54\\
$W = mgh$ (\unit{J}) & 0.096 & 0.385 & 0.865 & 1.54
\end{tabular}
\end{center}
Starting from $E_k = 0$, at every frame $E_k$ matches the work $mgh$ of
the weight: $\Delta E_k = W(\vect P)$, i.e.\ $v^2 = 2gh$ --- the theorem,
verified to the precision of the readings.
\end{example}

\begin{remark}[Where the proof lives]\label{rem:g11:mechanical-energy:proof}
The general statement --- valid along any path, not only a straight
line --- follows from Newton's second law and a little calculus, both
later in this book (\cref{ch:g12:newtons-laws,%
ch:g12:work-and-energy}); until then the free-fall check is our
warrant, and we admit the curved-path version and use it freely.
\end{remark}

\section{Gravitational potential energy}

\begin{definition}[Gravitational potential energy]\label{def:g11:mechanical-energy:potential}
A body of mass $m$ at altitude $z$, measured upward from a chosen
\emph{reference level}\index{reference level} where $z = 0$, stores the
\emph{gravitational potential energy}\index{potential energy!gravitational}
\[
E_p = m g z, \qquad g = \qty{9.81}{N/kg}.
\]
\end{definition}

\begin{remark}[Only differences matter]\label{rem:g11:mechanical-energy:reference}
Moving from $A$ to $B$ changes $E_p$ by $mg(z_B - z_A) =
-W_{AB}(\vect P)$: the opposite of the work of the weight
(\cref{ch:g11:work-of-force}). The reference level is a free choice ---
floor, sea level, table top --- shifting $E_p$ everywhere by a constant
that drops out of every difference. Put it where the numbers are
simplest, and do not move it mid-problem.
\end{remark}

\begin{example}[A book on a shelf]\label{ex:g11:mechanical-energy:shelf}
A \qty{1.2}{kg} book sits \qty{1.8}{m} above the floor, itself
\qty{9.0}{m} above the street. Reference at the floor:
$E_p = 1.2 \times 9.81 \times 1.8 \approx \qty{21}{J}$; in the street:
$E_p \approx \qty{127}{J}$. Falling to the floor releases \qty{21}{J} in
both accounts.
\end{example}

\section{Mechanical energy and its conservation}

\begin{definition}[Mechanical energy]\label{def:g11:mechanical-energy:em}
The \emph{mechanical energy}\index{mechanical energy} of a body is
\[
E_m = E_k + E_p = \tfrac12 m v^2 + m g z .
\]
\end{definition}

\begin{theorem}[Conservation of mechanical energy]\label{thm:g11:mechanical-energy:conservation}
\index{conservation of mechanical energy}
If the only force working on a body is its weight (no friction; other
forces, like the reaction of a frictionless track, perpendicular to the
motion), $E_m$ is conserved: for any two points $A$, $B$ of the motion,
\[
\tfrac12 m v_A^2 + m g z_A = \tfrac12 m v_B^2 + m g z_B .
\]
\end{theorem}

\begin{proof}
By the kinetic-energy theorem, $E_k(B) - E_k(A) = W_{AB}(\vect P) =
mg(z_A - z_B) = E_p(A) - E_p(B)$: what $E_k$ gains, $E_p$ loses.
\end{proof}

\begin{example}[Water slide]\label{ex:g11:mechanical-energy:slide}
A child starts at rest atop a frictionless slide of height
$h = \qty{3.2}{m}$. Reference at the bottom: $mgh = \tfrac12 mv^2$, so
$v = \sqrt{2gh} \approx \qty{7.9}{m/s}$ (\qty{29}{km/h}) --- whatever
the mass \emph{and whatever the shape of the slide}: straight, curved
or spiral, only the height drop enters.
\end{example}

\begin{example}[Pendulum]\label{ex:g11:mechanical-energy:pendulum}
A pendulum bob is pulled aside until it rises $h = \qty{12}{cm}$ above
its lowest point, then released. At the extremes $v = 0$; at the bottom
the whole $mgh$ is kinetic, $v = \sqrt{2gh} \approx \qty{1.5}{m/s}$;
on the far side the bob climbs back to exactly \qty{12}{cm}.
\end{example}

\begin{example}[Ski jump landing speed]\label{ex:g11:mechanical-energy:skijump}
A ski jumper leaves the takeoff at $v_A = \qty{25}{m/s}$ and lands
$\Delta z = \qty{40}{m}$ lower. Neglecting air resistance,
$v_B = \sqrt{v_A^2 + 2g\,\Delta z} = \sqrt{625 + 785} \approx
\qty{38}{m/s}$ --- no knowledge of the flight path needed.
\end{example}

\section{Dissipation by friction}

\begin{definition}[Dissipation]\label{def:g11:mechanical-energy:dissipation}
Energy leaving the mechanical account is said to be
\emph{dissipated}\index{dissipation}: it reappears as thermal energy in
the rubbing surfaces and the air. Total energy is conserved
(\cref{ch:g10:energy-conservation}); mechanical energy alone is not.
\end{definition}

\begin{proposition}[The deficit measures friction]\label{prop:g11:mechanical-energy:deficit}
If a friction force $\vect f$ also works along the motion,
\[
\Delta E_m = E_m(B) - E_m(A) = W_{AB}(\vect f) < 0:
\]
the loss of mechanical energy equals the work of friction.
\end{proposition}

\begin{proof}
The kinetic-energy theorem now reads $\Delta E_k = W_{AB}(\vect P) +
W_{AB}(\vect f) = -\Delta E_p + W_{AB}(\vect f)$; move $\Delta E_p$
across. Friction opposes the motion, so its work is negative.
\end{proof}

\begin{example}[Toboggan audit]\label{ex:g11:mechanical-energy:toboggan}
A child and sled ($m = \qty{40}{kg}$) start at rest, drop
$h = \qty{5.0}{m}$ along \qty{12}{m} of snow, and arrive at
\qty{7.0}{m/s} instead of the frictionless
$\sqrt{2gh} \approx \qty{9.9}{m/s}$. Reference at the bottom:
$\Delta E_m = \tfrac12 \times 40 \times 7.0^2 - 40 \times 9.81
\times 5.0 = 980 - 1962 = \qty{-982}{J}$. So $W(\vect f) =
\qty{-982}{J}$ over \qty{12}{m}: an average friction force
$f = 982/12 \approx \qty{82}{N}$, and \qty{982}{J} of warmth in the
runners and the snow.
\end{example}

\begin{method}[Energy bookkeeping]\label{met:g11:mechanical-energy:bookkeeping}
\begin{enumerate}
  \item Choose a reference level and keep it; name $A$ (data), $B$
        (question).
  \item Write $E_m = \tfrac12 mv^2 + mgz$ at $A$ and at $B$.
  \item No friction: $E_m(A) = E_m(B)$; solve --- typically
        $v_B = \sqrt{v_A^2 + 2g(z_A - z_B)}$.
  \item Friction: $E_m(B) - E_m(A) = -f\ell$ ($\ell$ = path length);
        solve for the unknown.
  \item Check: the $E_k$/$E_p$ bars stack to $E_m$, level unless friction.
\end{enumerate}
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:1}

\end{exercise}

A \qty{1200}{kg} car brakes from \qty{25}{m/s} to rest in \qty{50}{m}.

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:5}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:7}
Two frictionless slides drop from the same \qty{3.0}{m} platform, one

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:8}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:9}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:10}
A \qty{45}{kg} sled starts at rest, drops \qty{6.0}{m} along

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:11}
Galileo's peg: a pendulum released \qty{15}{cm} above its lowest point

\begin{exercise}[$\star\star\star$]\label{exo:g11:mechanical-energy:13}

\begin{exercise}[$\star\star\star$]\label{exo:g11:mechanical-energy:14}

\begin{exercise}[$\star\star\star$]\label{exo:g11:mechanical-energy:15}

\end{exercise}

\section{Problem: Designing the loop}

\begin{problem}[{Weekend problem --- an energy audit of a roller
coaster: the launch height a loop dictates, the speeds it promises, the
friction confessed by two sensors, the brakes that close the account ---
the whole ride hanging from one number}]
\label{pb:g11:mechanical-energy:1}
A train of mass $m = \qty{2.0e3}{kg}$ is released at rest from a launch
hill of height $H$ and coasts, engineless, through a vertical circular
loop of radius $R = \qty{8.0}{m}$ whose top sits at height $2R$.
Circular-motion analysis (next year) supplies the one datum we borrow: at
the loop top the train presses on the track only if
$v_{\text{top}}^2 \geq gR$. Parts I and II ignore friction.

\medskip
\noindent\textbf{Part I --- The hill the loop dictates.}
\begin{enumerate}
  \item Compute the minimum safe speed at the top of the loop.
  \item Ground as reference level: compute the train's mechanical
        energy at the loop top at that minimum speed.
  \item Deduce $H_{\min}$ and show $H_{\min} = 5R/2$ --- no mass, no $g$.
  \item The team takes $H = \qty{24}{m}$: compute the actual top-of-loop
        speed and the ratio $v_{\text{top}}^2/(gR)$.
  \item Where did $m$ go? Why does the design fit full and empty
        trains alike?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- What the height promises} ($H = \qty{24}{m}$).
\begin{enumerate}[resume]
  \item Speed at the bottom of the loop ($z = 0$), in \unit{m/s} and
        \unit{km/h}?
  \item Speed at the side of the loop ($z = R$)?
  \item Compute $E_p$ and $E_k$ at launch, bottom, side and top of
        the loop; check that each pair sums to $mgH$.
  \item A \qty{12}{m} camel-back bump follows the loop: speed at its crest?
  \item Give the general formula for $v(z)$; where is the ride
        fastest, and slowest?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The friction audit.}
Two sensors at the same height $z = \qty{2.0}{m}$, \qty{60}{m} of
track apart, read $v_1 = \qty{20.4}{m/s}$ then $v_2 = \qty{19.2}{m/s}$.
\begin{enumerate}[resume]
  \item Compute the change of mechanical energy between the sensors.
        What made the same-height placement clever?
  \item Deduce the work of friction and the average friction force.
  \item Where has the missing energy gone? Name the form and the places.
  \item What fraction of the train's $E_m$ is lost per \qty{60}{m}? Why
        would the bare $H_{\min}$ be unsafe in the real, rubbing world?
  \item About \qty{90}{m} of track separate launch and loop top: estimate
        $E_m$ there; does the margin of question 4 still save the loop?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Closing the account.}
The last sensor, at the entry of the straight horizontal braking run
($z = 0$), reads \qty{18.0}{m/s}.
\begin{enumerate}[resume]
  \item Compute the train's kinetic energy there.
  \item The brakes must stop it in $d = \qty{45}{m}$: compute the
        required (constant) braking force.
  \item Comfort check: compute the deceleration $a = F/m$; compare to $g$.
  \item How much energy did track friction dissipate over the whole
        ride? Verify that friction $+$ brakes $= mgH$, to the joule.
  \item Punchline, one sentence: which single number fixed the loop's
        safety, every speed, the friction bill and the brake force ---
        and what was the price of ignoring friction?
\end{enumerate}
\end{problem}
