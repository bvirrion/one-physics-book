\chapter{Travail d'une force}\label{ch:g11:work-of-force}

Poussez une armoire à travers la pièce et vous avez travaillé~; poussez
contre un mur tout l'après-midi et, quoi qu'en disent vos muscles, la
physique compte zéro. Le \omterm{def:g11:work-of-force:work}{travail} est ce
qu'une \omterm{def:g10:inertia:force}{force} livre \emph{le long d'un
mouvement} --- de l'\omterm{def:g10:energy-conservation:energy}{énergie} en
transit (\cref{ch:g10:energy-conservation}), payée en
\omterm{def:g11:work-of-force:work}{joules}. Ce chapitre tarifie les
\omterm{def:g10:inertia:force}{forces} du
\cref{ch:g11:forces-and-motion} --- tractions en biais,
\omterm{def:g10:universal-gravitation:weight}{poids},
\omterm{def:g10:inertia:inventory}{frottement} --- et chronomètre le taux de
paiement~: la \omterm{def:g11:work-of-force:power}{puissance}.

\section{Travail d'une force constante}

\begin{definition}[Travail d'une force constante]\label{def:g11:work-of-force:work}
Une \omterm{def:g10:inertia:force}{force} constante $\vect F$ appliquée à un
objet se déplaçant le long d'un segment rectiligne de $A$ à $B$ fait le
\emph{travail}\index{travail (d'une force)}
\[
W_{AB}(\vect F) = F \times AB \times \cos\theta ,
\]
où $F$ est la grandeur de la \omterm{def:g10:inertia:force}{force} (en
\unit{N}), $AB$ la longueur du déplacement (en \unit{m}) et $\theta$
l'angle entre $\vect F$ et $\vect{AB}$. Le travail se mesure en
\emph{joules}\index{joule} (\unit{J})~: un joule est le travail d'une
\omterm{def:g10:inertia:force}{force} d'un newton le long d'un
\omterm{def:g10:orders-of-magnitude:unit}{mètre} de sa propre direction.
\end{definition}

\begin{remark}\label{rem:g11:work-of-force:scal}
$F \times AB \times \cos\theta$ est exactement le produit scalaire
$\vect F \cdot \vect{AB}$ du volume de mathématiques de cette année.
Seule la composante $F\cos\theta$ \emph{le long} du mouvement rapporte
quoi que ce soit.
\end{remark}

\begin{example}[Valise en laisse]\label{ex:g11:work-of-force:suitcase}
Un voyageur traîne une valise sur \qty{300}{m} à travers un terminal, la
sangle tirée avec $F = \qty{40}{N}$ à $\theta = \qty{50}{\degree}$ du
sol~: $W = 40 \times 300 \times \cos\qty{50}{\degree} \approx
\qty{7.7}{kJ}$. Tirée à plat ($\theta = 0$) la même
\omterm{def:g10:inertia:force}{force} livrerait \qty{12}{kJ}~: un tiers de
la traction ne soulève rien.
\end{example}

\begin{definition}[Travail moteur, résistant, nul]\label{def:g11:work-of-force:sign}
Le signe de $\cos\theta$ range chaque
\omterm{def:g10:inertia:force}{force} en trois régimes~:
\begin{itemize}
  \item $\theta < \qty{90}{\degree}$~: $W > 0$, le
        \omterm{def:g11:work-of-force:work}{travail} est
        \emph{moteur}\index{travail moteur} --- la
        \omterm{def:g10:inertia:force}{force} alimente de l'
        \omterm{def:g10:energy-conservation:energy}{énergie} dans le mouvement~;
  \item $\theta = \qty{90}{\degree}$~: $W = 0$, la
        \omterm{def:g10:inertia:force}{force} ne
        \omterm{def:g11:work-of-force:work}{travaille} pas du tout, si
        grande soit-elle~;
  \item $\theta > \qty{90}{\degree}$~: $W < 0$, le
        \omterm{def:g11:work-of-force:work}{travail} est
        \emph{résistant}\index{travail résistant} --- la
        \omterm{def:g10:inertia:force}{force} draine de l'
        \omterm{def:g10:energy-conservation:energy}{énergie}. Le cas pur est un frein
        à $\qty{180}{\degree}$~: \qty{800}{N} sur \qty{45}{m} retire
        $800 \times 45 \times (-1) = \qty{-36}{kJ}$ d'une voiture, dans
        des disques chauds.
\end{itemize}
\end{definition}

\section{Le travail du poids}

\begin{proposition}[Travail du poids]\label{prop:g11:work-of-force:weight}
Lorsqu'un objet de masse $m$ se déplace de $A$ (altitude $z_A$) à $B$
(altitude $z_B$) le long de \emph{n'importe quel} chemin fait de
segments rectilignes, son
\omterm{def:g10:universal-gravitation:weight}{poids} $\vect P$ fait le
\omterm{def:g11:work-of-force:work}{travail}
\[
W_{AB}(\vect P) = mg\,(z_A - z_B) = \pm\, mgh ,
\]
où $h = \abs{z_A - z_B}$ est la descente de hauteur~: $+mgh$ en
descendant, $-mgh$ en montant --- seule la descente compte, non la
route.
\end{proposition}

\begin{proof}
Sur un segment rectiligne $AB$, le
\omterm{def:g10:universal-gravitation:weight}{poids} (grandeur $mg$, droit vers
le bas) fait un angle $\theta$ avec $\vect{AB}$, et $AB\cos\theta$ est
précisément la projection du déplacement sur la verticale
descendante~: $z_A - z_B$. D'où
$W = mg \times AB\cos\theta = mg\,(z_A - z_B)$. Le long d'une chaîne de
segments les \omterm{def:g11:work-of-force:work}{travaux} s'ajoutent et les
descentes se télescopent~:
$mg(z_A - z_C) + mg(z_C - z_B) = mg(z_A - z_B)$.
\end{proof}

\begin{proposition}[Tout chemin]\label{prop:g11:work-of-force:anypath}
La même formule tient le long de tout chemin courbe de $A$ à $B$.
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:work-of-force:anypath}
Plausible --- une courbe est aussi proche qu'on veut d'une ligne brisée
de nombreux petits segments --- mais rendre «~aussi proche qu'on
veut~» honnête est du calcul, fait dans le
\cref{ch:g12:work-and-energy}.
\end{remark}

\begin{example}[Randonneur]\label{ex:g11:work-of-force:hiker}
Un randonneur de \qty{78}{kg} (sac compris) grimpe \qty{450}{m}, par
quelque sentier que ce soit~: $W(\vect P) = -78 \times 9.81 \times 450
\approx \qty{-344}{kJ}$, et exactement $+\qty{344}{kJ}$ en
redescendant, quels que soient les détours.
\end{example}

\section{Le travail du frottement}

\begin{proposition}[Travail du frottement de glissement]\label{prop:g11:work-of-force:friction}
Une \omterm{def:g10:inertia:force}{force} de
\omterm{def:g10:inertia:inventory}{frottement} de grandeur constante $f$, à
chaque instant opposée au mouvement, fait le
\omterm{def:g11:work-of-force:work}{travail} $W = -fL$ le long d'un chemin de
longueur totale $L$~: toujours
\omterm{def:g11:work-of-force:sign}{résistant}, et proportionnel à la
\emph{longueur du chemin parcouru} --- non au déplacement.
\end{proposition}

\begin{proof}
Chaque petit morceau rectiligne a la
\omterm{def:g10:inertia:force}{force} à $\qty{180}{\degree}$ du mouvement,
contribuant $-f \times (\text{sa longueur})$~; les morceaux s'ajoutent
en $-fL$. Un chemin courbe se découpe en petits morceaux de même, la
limite étant admise comme pour le
\omterm{def:g10:universal-gravitation:weight}{poids}
(\cref{rem:g11:work-of-force:anypath}).
\end{proof}

\begin{remark}[Frottement contre poids~: le grand partage]\label{rem:g11:work-of-force:divide}
Le \omterm{def:g11:work-of-force:work}{travail} du
\omterm{def:g10:universal-gravitation:weight}{poids} oublie le chemin~; celui du
\omterm{def:g10:inertia:inventory}{frottement} est un péage par
\omterm{def:g10:orders-of-magnitude:unit}{mètre}. Sur un aller-retour le
\omterm{def:g10:universal-gravitation:weight}{poids} fait un bilan nul --- ce qu'il
prend en montant il le rembourse en descendant --- tandis que le
\omterm{def:g10:inertia:inventory}{frottement} facture $-fL$ à chaque étape et
ne rembourse rien~: le \omterm{def:g11:work-of-force:work}{travail} du
\omterm{def:g10:universal-gravitation:weight}{poids} peut être stocké et récupéré
(énergie potentielle, \cref{ch:g11:mechanical-energy}), celui du
\omterm{def:g10:inertia:inventory}{frottement} est perdu en chaleur.
\end{remark}

\begin{example}[Aller et retour]\label{ex:g11:work-of-force:roundtrip}
Une caisse glisse \qty{5.0}{m} sur un plancher et \qty{5.0}{m} en
retour, contre $f = \qty{60}{N}$~: le
\omterm{def:g10:inertia:inventory}{frottement} fait $-60 \times 10 =
\qty{-600}{J}$ même si la caisse finit où elle a commencé.
\omterm{def:g10:universal-gravitation:weight}{Poids} et
\omterm{def:g10:inertia:force}{force} normale, perpendiculaires au
mouvement~: zéro tout du long.
\end{example}

\begin{remark}[Et un ressort~?]\label{rem:g11:work-of-force:spring}
Un ressort tire plus fort plus il est étiré~: sa
\omterm{def:g10:inertia:force}{force} n'est pas constante et la
\cref{def:g11:work-of-force:work} ne s'applique pas. Son
\omterm{def:g11:work-of-force:work}{travail} signifie découper l'étirement en
pas infinitésimaux et sommer --- calcul, livré dans le
\cref{ch:g12:work-and-energy}.
\end{remark}

\section{Puissance}

\begin{definition}[Puissance]\label{def:g11:work-of-force:power}
La \emph{puissance}\index{puissance!mécanique} d'une
\omterm{def:g10:inertia:force}{force} est le taux auquel elle
\omterm{def:g11:work-of-force:work}{travaille}, $P = W/\Delta t$, en
\emph{watts}\index{watt} ($\unit{W} = \unit{J/s}$) --- le même watt que
dans le \cref{ch:g11:circuits-and-power}. Une
\omterm{def:g10:orders-of-magnitude:unit}{unité} pré-SI survit sur les plaques
moteur~: le \emph{cheval-vapeur}\index{cheval-vapeur},
$\qty{1}{hp} = \qty{736}{W}$.
\end{definition}

\begin{proposition}[Puissance à vitesse constante]\label{prop:g11:work-of-force:pfv}
Si l'objet se déplace à vitesse constante $v$, une
\omterm{def:g10:inertia:force}{force} constante $\vect F$ à l'angle $\theta$
du mouvement livre la \omterm{def:g11:work-of-force:power}{puissance}
$P = F v \cos\theta$ --- en particulier $P = Fv$ pour une
\omterm{def:g10:inertia:force}{force} le long du mouvement.
\end{proposition}

\begin{proof}
En un temps $\Delta t$ l'objet parcourt $d = v\,\Delta t$, donc
$W = F \times v\,\Delta t \times \cos\theta$~; diviser par $\Delta t$.
\end{proof}

\begin{example}[Grue]\label{ex:g11:work-of-force:crane}
Une grue hisse une palette de \qty{600}{kg} à un régime permanent de
\qty{0.90}{m/s}~: la \omterm{def:g10:inertia:force}{force} du câble égale le
\omterm{def:g10:universal-gravitation:weight}{poids}, $mg = \qty{5.89e3}{N}$, donc
$P = Fv = 5890 \times 0.90 \approx \qty{5.3}{kW} \approx \qty{7.2}{hp}$
--- une équipe de sept chevaux dans une boîte d'acier.
\end{example}

\begin{method}[Monter avec moins de force~: le zigzag]\label{met:g11:work-of-force:zigzag}
Pour élever une charge d'une hauteur $h$ avec une
\omterm{def:g10:inertia:force}{force} limitée~:
\begin{enumerate}
  \item La facture est fixe~: par la
        \cref{prop:g11:work-of-force:weight}, toute route jusqu'au sommet
        coûte le \omterm{def:g11:work-of-force:work}{travail} $mgh$.
  \item Étirez le chemin~: sur une pente de longueur $L$ la
        \omterm{def:g10:inertia:force}{force} nécessaire (vitesse constante,
        \omterm{def:g10:inertia:inventory}{frottement} mis à part) est
        $F = mgh/L$ --- doublez la longueur, divisez la
        \omterm{def:g10:inertia:force}{force} par deux.
  \item Payez le supplément~: le
        \omterm{def:g10:inertia:inventory}{frottement} facture au
        \omterm{def:g10:orders-of-magnitude:unit}{mètre}
        (\cref{prop:g11:work-of-force:friction}), ajoutant $fL$.
\end{enumerate}
Les routes de montagne serpentent pour l'étape~2~: même hauteur, moins
de \omterm{def:g10:inertia:force}{force}, même
\omterm{def:g11:work-of-force:work}{travail}.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:1}
Une \omterm{def:g10:inertia:force}{force} constante $F = \qty{40}{N}$ agit
sur un déplacement rectiligne de \qty{50}{m}~: calculer son
\omterm{def:g11:work-of-force:work}{travail} pour $\theta = \qty{0}{\degree}$,
$\qty{60}{\degree}$, $\qty{90}{\degree}$ et $\qty{120}{\degree}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:2}
Une boîte à outils de \qty{12}{kg} est abaissée de \qty{2.5}{m} d'un
fourgon, puis relevée plus tard. \omterm{def:g11:work-of-force:work}{Travail} de son
\omterm{def:g10:universal-gravitation:weight}{poids} dans chaque cas~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:3}
\omterm{def:g11:work-of-force:sign}{Moteur},
\omterm{def:g11:work-of-force:sign}{résistant} ou nul~? (a)~le
\omterm{def:g10:universal-gravitation:weight}{poids} d'une pomme qui tombe~;
(b)~le \omterm{def:g10:universal-gravitation:weight}{poids} d'une balle qui
monte~; (c)~la \omterm{def:g10:inertia:force}{force} normale sur une caisse
qui glisse~; (d)~le \omterm{def:g10:inertia:inventory}{frottement} sur un
vélo qui freine~; (e)~la \omterm{def:g10:inertia:inventory}{tension} du fil
sur le bob d'un pendule qui oscille.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:4}

\omterm{def:g11:work-of-force:work}{Travail} de la
\omterm{def:g10:inertia:force}{force} de levage et du
\omterm{def:g10:universal-gravitation:weight}{poids}~? Quelle est leur somme, et

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:work-of-force:5}
Un \omterm{def:g11:work-of-force:sign}{moteur} d'escalier mécanique fait
\qty{90}{kJ} de \omterm{def:g11:work-of-force:work}{travail} chaque minute. Sa
\omterm{def:g11:work-of-force:power}{puissance} en
\omterm{def:g11:work-of-force:power}{watts}~? En
\omterm{def:g11:work-of-force:power}{chevaux-vapeur}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:6}
Un traîneau est tracté sur \qty{200}{m} par une corde de
\omterm{def:g10:inertia:inventory}{tension} \qty{120}{N} à
$\qty{30}{\degree}$ au-dessus de la neige~; le
\omterm{def:g10:inertia:inventory}{frottement} est \qty{45}{N}.
\omterm{def:g11:work-of-force:work}{Travail} de chacune des quatre
\omterm{def:g10:inertia:force}{forces} (corde,
\omterm{def:g10:inertia:inventory}{frottement},
\omterm{def:g10:universal-gravitation:weight}{poids}, normale) et total~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:7}

\omterm{def:g10:energy-conservation:power}{Puissance} livrée~?
\omterm{def:g11:work-of-force:work}{Travail} fait en une heure~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:8}
Un skieur de \qty{65}{kg} descend \qty{120}{m} de hauteur, par une pente

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:9}
Une caisse de \qty{20}{kg} est poussée de \qty{4.0}{m} sur une rampe à

\omterm{def:g10:universal-gravitation:weight}{poids} et du
\omterm{def:g10:inertia:inventory}{frottement}, puis (d'après la vitesse
constante) le \omterm{def:g11:work-of-force:work}{travail} et la grandeur de la
poussée, parallèle à la rampe.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:work-of-force:10}

\omterm{def:g11:work-of-force:sign}{résistante} totale combat-elle~?

\begin{exercise}[$\star\star\star$]\label{exo:g11:work-of-force:12}

\section{Problème~: La journée du déménageur}

\begin{problem}[{Problème du week-end --- la physique paie au joule~: une
journée de glissements, de levages et d'épingles à cheveux, s'achevant
par l'humiliante découverte de ce qu'une bouilloire pense d'un labeur
honnête}]
\label{pb:g11:work-of-force:1}
L'équipe d'un déménageur vide un entrepôt, hisse la charge jusqu'à un
appartement au troisième étage, puis conduit le fourgon jusqu'à un
village de colline. Chaque tâche est tarifée en
\omterm{def:g11:work-of-force:work}{joules}~; la bouilloire de la cuisine
auditera la journée. Caisses~: \qty{30}{kg}.

\medskip
\noindent\textbf{Partie I --- À travers le plancher de l'entrepôt.}
Une caisse est poussée avec une
\omterm{def:g10:inertia:force}{force} horizontale $F = \qty{110}{N}$ sur
$d = \qty{8.0}{m}$~; le \omterm{def:g10:inertia:inventory}{frottement} de
glissement est $f = \qty{95}{N}$.
\begin{enumerate}
  \item Calculer le \omterm{def:g11:work-of-force:work}{travail} de la poussée.
  \item Calculer le \omterm{def:g11:work-of-force:work}{travail} du
        \omterm{def:g10:inertia:inventory}{frottement}, du
        \omterm{def:g10:universal-gravitation:weight}{poids}, et de la
        \omterm{def:g10:inertia:force}{force} normale (justifier les
        zéros).
  \item \omterm{def:g11:work-of-force:work}{Travail} total sur la caisse, et
        le sens de son signe (la caisse est partie du repos)~? Quelle
        poussée maintient la vitesse constante
        (\cref{ch:g11:forces-and-motion})~?
  \item Le coin opposé peut être atteint le long de la diagonale de
        \qty{10.0}{m} ou le long de deux murs (\qty{6.0}{m} puis
        \qty{8.0}{m}). \omterm{def:g11:work-of-force:work}{Travail} du
        \omterm{def:g10:inertia:inventory}{frottement} sur chaque route~?
  \item Sur un aller-retour en diagonale (aller et retour), calculer le
        \omterm{def:g11:work-of-force:work}{travail} du
        \omterm{def:g10:inertia:inventory}{frottement} et le
        \omterm{def:g11:work-of-force:work}{travail} du
        \omterm{def:g10:universal-gravitation:weight}{poids}. Quelle
        \omterm{def:g10:inertia:force}{force} rembourse~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Jusqu'au troisième étage ($h = \qty{9.0}{m}$).}
\begin{enumerate}[resume]
  \item Calculer le \omterm{def:g11:work-of-force:work}{travail} du
        \omterm{def:g10:universal-gravitation:weight}{poids} sur une caisse
        élevée jusqu'à l'appartement. Quel
        \omterm{def:g11:work-of-force:work}{travail} l'équipe doit-elle au moins
        fournir~?
  \item Portée par les escaliers par un déménageur de \qty{70}{kg}, la
        gravité facture pour le déménageur \emph{et} la caisse~:
        \omterm{def:g11:work-of-force:work}{travail} total contre la gravité, et
        la fraction réellement dépensée sur la caisse~?
  \item Une rampe de chariot fait \qty{30}{m} de long pour la même
        montée de \qty{9.0}{m}, avec \qty{40}{N} de
        \omterm{def:g10:inertia:inventory}{frottement} de roulement. À vitesse
        constante, calculer la poussée nécessaire le long de la rampe et
        son \omterm{def:g11:work-of-force:work}{travail}.
  \item Comparer la rampe avec un treuil vertical droit~:
        \omterm{def:g10:inertia:force}{force} nécessaire,
        \omterm{def:g11:work-of-force:work}{travail} fait. Qu'achète la
        rampe, et à quel prix~?
  \item Un treuil à corde et poulie élève la caisse à \qty{0.50}{m/s}.
        Durée et \omterm{def:g11:work-of-force:power}{puissance} mécanique~?
  \item Retrouver cette \omterm{def:g11:work-of-force:power}{puissance} comme
        $P = Fv$. Le \omterm{def:g11:work-of-force:sign}{moteur} du treuil a un
        \omterm{def:g10:energy-conservation:efficiency}{rendement} de $60\,\%$~:
        \omterm{prop:g11:circuits-and-power:power}{puissance électrique} tirée~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La route en zigzag.}
Le fourgon chargé (\qty{3500}{kg}) doit atteindre un village à
\qty{240}{m} au-dessus de la vallée~: voie droite, \qty{1.0}{km} à
pente $24\,\%$, ou route pavée, \qty{8.0}{km} à pente $3{,}0\,\%$
(pente $= \sin\alpha$).
\begin{enumerate}[resume]
  \item Vérifier que les deux routes grimpent les mêmes \qty{240}{m}.
  \item À vitesse constante et en négligeant le
        \omterm{def:g10:inertia:inventory}{frottement}, calculer la
        \omterm{def:g10:inertia:force}{force} motrice nécessaire le long de
        chaque route.
  \item \omterm{def:g11:work-of-force:work}{Travail} de la
        \omterm{def:g10:inertia:force}{force} motrice sur chaque route~?
        Comparer avec $mgh$.
  \item La \omterm{def:g11:circuits-and-power:resistance}{résistance} au roulement est
        \qty{400}{N} sur les deux~: calculer le
        \omterm{def:g11:work-of-force:work}{travail} supplémentaire sur chaque
        route. Que coûte le confort du zigzag~?
  \item À \qty{36}{km/h}, calculer la
        \omterm{def:g11:work-of-force:power}{puissance} moteur nécessaire sur
        chaque route (en \unit{kW} et \unit{hp}). Pourquoi les routes
        serpentent-elles~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le grand livre de la journée.}
\begin{enumerate}[resume]
  \item L'équipe glisse $60$ caisses sur le plancher (la poussée de la
        question~1) et hisse les $60$ des \qty{9.0}{m} (question~6).
        \omterm{def:g11:work-of-force:work}{Travail} mécanique total livré aux
        caisses~?
  \item Ajouter le corps propre du porteur d'escalier~: $20$ allers en
        haut à \qty{70}{kg} (les remboursements en descente vont aux
        genoux chauds, non au grand livre). Total de la journée~?
  \item L'équipe a travaillé $8{,}0$~heures~:
        \omterm{def:g11:work-of-force:power}{puissance} mécanique moyenne~?
        Combien de temps la bouilloire de \qty{2.0}{kW} mettrait-elle à
        dépenser la même
        \omterm{def:g10:energy-conservation:energy}{énergie}~?
  \item Final~: en deux phrases, dire ce que le
        \omterm{def:g11:work-of-force:work}{joule} paie --- et ne paie pas (le
        mur poussé tout l'après-midi, la caisse tenue immobile) --- et
        pourquoi les déménageurs possèdent des treuils, des rampes et
        des routes en zigzag plutôt que de plus gros muscles.
\end{enumerate}
\end{problem}
