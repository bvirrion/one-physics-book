\chapter{Couleur et sources de lumière}\label{ch:g11:color-light-sources}

Demandez à un physicien la couleur d'une tomate et vous obtiendrez une
contre-question~: sous quelle lumière~? La tomate ne
\omterm{def:g11:color-light-sources:objectcolor}{diffuse} que ce qu'on lui donne, la lampe ne donne que ce
que son \omterm{def:g10:light-spectra:white}{spectre} contient, et l'œil compresse ce qui
arrive en trois nombres. La couleur est une négociation entre source,
objet et œil --- ce chapitre interroge les trois parties tour à tour.

\section{Voir en trois nombres}

\begin{definition}[Cônes et vision trichromatique]\label{def:g11:color-light-sources:cones}
La \omterm{def:g11:lenses-and-eye:eye}{rétine} (\cref{ch:g11:lenses-and-eye}) porte deux familles
de cellules photosensibles~: les \emph{bâtonnets}\index{bâtonnet}, qui
répondent à la faible lumière mais ignorent la
\omterm{def:g10:light-spectra:wavelength}{longueur d'onde} (vision nocturne, en gris), et
les \emph{cônes}\index{cône}, de trois types, les plus sensibles près de
$\qty{420}{nm}$ (bleu), $\qty{530}{nm}$ (vert) et $\qty{560}{nm}$
(rouge). La vision des couleurs humaine est
\emph{trichromatique}\index{vision trichromatique}~: quel que soit le
\omterm{def:g10:light-spectra:white}{spectre}, le cerveau ne reçoit que trois réponses de
cônes, et la couleur est ce qu'il en fait.
\end{definition}

\begin{definition}[La couleur comme perception]\label{def:g11:color-light-sources:perception}
La couleur est donc une \emph{perception}, non une propriété de la
lumière seule~: un \omterm{def:g10:light-spectra:white}{spectre} spécifie une intensité à
chaque \omterm{def:g10:light-spectra:wavelength}{longueur d'onde}, l'œil ne garde que trois
nombres. Des lumières de spectres différents mais de mêmes trois
réponses des \omterm{def:g11:color-light-sources:cones}{cônes} paraissent strictement
identiques~: ce sont des \emph{métamères}\index{métamère}.
\end{definition}

\begin{example}[Deux jaunes]\label{ex:g11:color-light-sources:yellows}
La raie à $\qty{589}{nm}$ d'une lampe à sodium excite les
\omterm{def:g11:color-light-sources:cones}{cônes} rouge et vert à peu près également~: le
cerveau dit \emph{jaune}. Un écran allumant ses sous-pixels rouge
($\qty{610}{nm}$) et vert ($\qty{540}{nm}$) donne le même jaune sans
aucune lumière près de $\qty{589}{nm}$. Un prisme sépare les deux
instantanément~; l'œil ne le peut jamais.
\end{example}

\section{Additionner des lumières}

\begin{definition}[Synthèse additive]\label{def:g11:color-light-sources:additive}
Superposer des faisceaux lumineux sur un écran blanc additionne leurs
spectres~: c'est la \emph{synthèse additive}\index{synthèse additive}.
Trois lumières bien choisies, rouge, verte et bleue --- les
\emph{primaires additives}\index{couleurs primaires} --- se mélangent
en presque toutes les couleurs perceptibles~: R $+$ V $=$ jaune,
R $+$ B $=$ magenta, V $+$ B $=$ cyan, et R $+$ V $+$ B $=$ blanc.
\end{definition}

\begin{proposition}[Couleurs complémentaires]\label{prop:g11:color-light-sources:complementary}
Deux lumières dont la somme est perçue comme blanche sont
\emph{complémentaires}\index{couleurs complémentaires}. Les trois
paires jaune/bleu, cyan/rouge et magenta/vert sont complémentaires.
\end{proposition}

\begin{proof}
Le jaune est R $+$ V~; y ajouter le bleu complète le triplet jusqu'au
blanc. Les deux autres paires relèvent du même argument, une primaire
à la fois.
\end{proof}

\begin{example}[Écrans]\label{ex:g11:color-light-sources:screen}
Chaque pixel d'un écran de téléphone est trois sous-pixels --- rouge,
vert, bleu, de largeur inférieure à $\qty{0.1}{mm}$ --- trop proches
pour que l'œil les sépare, donc leurs lumières s'ajoutent sur la
\omterm{def:g11:lenses-and-eye:eye}{rétine}~: le jaune est R et V allumés, l'orange est R à
pleine \omterm{def:g10:energy-conservation:power}{puissance} et V à moitié, le blanc est les
trois. Une loupe sur une zone blanche révèle le tour~: nulle part de
blanc, seulement R, V, B.
\end{example}

\section{Soustraire à la lumière blanche}

\begin{definition}[Synthèse soustractive]\label{def:g11:color-light-sources:subtractive}
Un \emph{filtre}\index{filtre} (verre coloré, gélatine, couche d'encre
ou de peinture) transmet une partie du
\omterm{def:g10:light-spectra:white}{spectre} qu'il reçoit et absorbe le reste. Produire
des couleurs en \emph{retirant} des bandes à la
\omterm{def:g10:light-spectra:white}{lumière blanche} est la \emph{synthèse
soustractive}\index{synthèse soustractive}. Ses primaires, complémentaires
des additives, soustraient chacune une bande~: cyan $=$ blanc $-$ R,
magenta $=$ blanc $-$ V, jaune $=$ blanc $-$ B. Des filtres empilés
soustraient conjointement~: cyan $+$ jaune laisse le vert, cyan $+$
magenta le bleu, magenta $+$ jaune le rouge, les trois le noir.
L'impression et la peinture fonctionnent ainsi, avec les encres C, M, J.
\end{definition}

\begin{method}[Comptabilité spectrale]\label{met:g11:color-light-sources:bookkeeping}
Pour prédire une couleur à travers toute chaîne de sources,
\omterm{def:g11:color-light-sources:subtractive}{filtres} et objets, suivre les trois bandes
primaires~: écrire la lumière de la source comme son contenu en R, V,
B~; à chaque \omterm{def:g11:color-light-sources:subtractive}{filtre} ou surface, supprimer
les bandes absorbées~; nommer la couleur de ce qui survit (rien de
restant $=$ noir).
\end{method}

\begin{example}[Une chaîne de filtres]\label{ex:g11:color-light-sources:chain}
La \omterm{def:g10:light-spectra:white}{lumière blanche} $(R, V, B)$ traverse un
\omterm{def:g11:color-light-sources:subtractive}{filtre} jaune~: la bande bleue est absorbée,
il reste $(R, V)$ --- jaune. Un \omterm{def:g11:color-light-sources:subtractive}{filtre}
cyan absorbe ensuite la bande rouge, il reste $(V)$~: vert. Intervertir
les \omterm{def:g11:color-light-sources:subtractive}{filtres} ne change rien~: chacun absorbe
sa propre bande indépendamment de l'ordre.
\end{example}

\section{La couleur d'un objet}

\begin{definition}[Diffusion et couleur d'un objet]\label{def:g11:color-light-sources:objectcolor}
Un objet opaque absorbe une partie de la lumière qu'il reçoit et
\emph{diffuse}\index{diffusion} le reste --- la réémet dans toutes les
directions, vers nos yeux entre autres. La \emph{couleur d'un
objet}\index{couleur d'un objet} est la couleur de la lumière qu'il
diffuse \emph{sous l'éclairage donné}~: l'objet fournit ses habitudes
d'absorption, la source la matière première. Changez la lampe, changez
la couleur.
\end{definition}

\begin{example}[La chemise rouge]\label{ex:g11:color-light-sources:redshirt}
Une chemise paraît rouge en lumière du jour~: elle absorbe les bandes
verte et bleue et \omterm{def:g11:color-light-sources:objectcolor}{diffuse} la bande rouge. Sous
une lumière purement verte, il n'y a pas de bande rouge à diffuser~:
la chemise absorbe tout et paraît noire. Sous une lumière magenta
$(R, B)$ elle \omterm{def:g11:color-light-sources:objectcolor}{diffuse} la bande rouge seule~:
toujours rouge. Un objet blanc \omterm{def:g11:color-light-sources:objectcolor}{diffuse}
chaque bande qu'il reçoit (sous lumière verte il paraît vert)~; un
objet noir les absorbe toutes sous toute lumière --- ce qui explique
aussi pourquoi les vêtements noirs chauffent au soleil.
\end{example}

\section{Sources de lumière et température de couleur}

\begin{definition}[Sources incandescentes]\label{def:g11:color-light-sources:incandescent}
Une source qui brille parce qu'elle est chaude --- flamme, filament,
étoile --- rayonne par \emph{incandescence}\index{incandescence}. Son
\omterm{def:g10:light-spectra:white}{spectre} est le \omterm{def:g10:light-spectra:white}{spectre}
thermique continu du \cref{ch:g10:light-spectra}, et son pic obéit à la
loi de Wien (\cref{prop:g10:light-spectra:wien}),
\[
\lambda_{\max} \times T = \qty{2.90e-3}{m.K},
\]
notre outil quantitatif désormais~: pic mesuré, température connue.
\end{definition}

\begin{example}[Une lampe très inefficace]\label{ex:g11:color-light-sources:bulb}
Un filament de tungstène à $T = \qty{2700}{K}$ a son pic en
$\lambda_{\max} = \qty{2.90e-3}{m.K}/\qty{2700}{K} = \qty{1.07e-6}{m}
= \qty{1070}{nm}$, profondément dans
l'\omterm{def:g10:light-spectra:wavelength}{infrarouge}~: la plus grande partie de la
\omterm{def:g10:energy-conservation:power}{puissance} électrique part en rayonnement
invisible, seulement environ $5\,\%$ en lumière --- l'ampoule à
incandescence était surtout un radiateur.
\end{example}

\begin{definition}[Lampes spectrales et lasers]\label{def:g11:color-light-sources:lamps}
Une \emph{lampe spectrale}\index{lampe spectrale} --- un gaz à
\omterm{def:g10:pressure:pressure}{basse pression} excité par une décharge~: sodium,
mercure, néon --- émet un \omterm{def:g10:light-spectra:emission}{spectre de raies}
(\cref{ch:g10:light-spectra})~: quelques
\omterm{def:g10:light-spectra:wavelength}{longueurs d'onde} isolées, rien entre elles. Un
\emph{laser}\index{laser} va plus loin~: une seule
\omterm{def:g10:light-spectra:wavelength}{longueur d'onde},
\omterm{def:g10:light-spectra:white}{lumière monochromatique}, en un faisceau étroit.
\end{definition}

\begin{definition}[LED]\label{def:g11:color-light-sources:led}
Une \emph{diode électroluminescente}\index{diode
électroluminescente} (DEL ou LED) émet une bande
\emph{quasi monochromatique}\index{quasi monochromatique} large de
quelques dizaines de nanomètres~: assez étroite pour paraître d'une
couleur définie, bien plus large qu'une raie de
\omterm{def:g11:color-light-sources:lamps}{laser}. Une LED blanche est une LED bleue (pic
près de $\qty{450}{nm}$) revêtue d'un
\emph{luminophore}\index{luminophore}, matériau qui absorbe une partie
du bleu et le réémet en une large bande jaune~: le bleu et le jaune
étant complémentaires
(\cref{prop:g11:color-light-sources:complementary}), le mélange se lit
blanc.
\end{definition}

\begin{definition}[Température de couleur]\label{def:g11:color-light-sources:colortemp}
La \emph{température de couleur}\index{température de couleur} d'une
source blanchâtre est la température du corps incandescent dont la
teinte de lumière correspond~: bougie $\qty{1800}{K}$, ampoule à
incandescence $\qty{2700}{K}$, halogène $\qty{3400}{K}$, soleil de midi
$\qty{5500}{K}$, lumière du jour voilée $\qty{6500}{K}$. C'est une
étiquette de couleur, non une lecture de thermomètre --- une LED à
$\qty{6500}{K}$ fonctionne à température ambiante --- et le vocabulaire
va à rebours~: une basse température de couleur se lit «~chaude~»
orangée, une haute «~froide~» bleuâtre.
\end{definition}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:color-light-sources:1}
Donner la couleur perçue lorsqu'un écran blanc reçoit simultanément~:
(a) de la lumière rouge et verte~; (b) de la lumière verte et bleue~;
(c) les trois primaires. Quelle lumière est complémentaire du bleu~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:color-light-sources:2}
Quels sous-pixels (R, V, B) un écran allume-t-il pour afficher~: le
jaune, le magenta, le blanc, le noir, l'orange~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:color-light-sources:3}
La \omterm{def:g10:light-spectra:white}{lumière blanche} traverse un
\omterm{def:g11:color-light-sources:subtractive}{filtre} magenta. Quelle bande est absorbée,
et quelle couleur émerge~? Un \omterm{def:g11:color-light-sources:subtractive}{filtre}
cyan est ensuite ajouté derrière~: qu'est-ce qui émerge maintenant~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:color-light-sources:4}
Sous \omterm{def:g10:light-spectra:white}{lumière blanche} un drapeau montre une bande
jaune. Quelles bandes diffuse-t-il, lesquelles absorbe-t-il~? Sa
couleur sous (a) lumière rouge~; (b) lumière bleue~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:color-light-sources:5}
Le \omterm{def:g10:light-spectra:continuous}{spectre continu} d'une étoile a son pic en
$\lambda_{\max} = \qty{580}{nm}$. Calculer sa température de surface.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:color-light-sources:6}
La flamme d'une bougie est à environ $\qty{1800}{K}$. (a)~Calculer
$\lambda_{\max}$ et indiquer son domaine. (b)~La flamme paraît
néanmoins orange~: expliquer.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:color-light-sources:7}
Un cycliste porte une veste rouge et des gants cyan. En utilisant le
\cref{met:g11:color-light-sources:bookkeeping}, donner la couleur
apparente de chaque pièce sous (a)~
\omterm{def:g10:light-spectra:white}{lumière blanche}~; (b)~lumière rouge~;
(c)~lumière verte.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:color-light-sources:8}
Une vieille rue est éclairée par des lampes à sodium à
\omterm{def:g10:pressure:pressure}{basse pression} (une seule raie à
$\qty{589}{nm}$). (a)~Quel type de \omterm{def:g10:light-spectra:white}{spectre}
est-ce, et quelles couleurs peut prendre \emph{n'importe quel} objet
sous cette lumière~? (b)~Une voiture paraît noire sous ces lampes et
bleue en lumière du jour~: expliquer.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:color-light-sources:9}
Un pointeur \omterm{def:g11:color-light-sources:lamps}{laser} vert émet à $\qty{532}{nm}$.
(a)~Son faisceau traverse un prisme~: comparer le résultat avec celui
d'un faisceau blanc, et nommer la propriété illustrée. (b)~Il rencontre
ensuite un \omterm{def:g11:color-light-sources:subtractive}{filtre} rouge~: qu'en sort-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:color-light-sources:10}
Décrire le \omterm{def:g10:light-spectra:white}{spectre} d'une LED blanche (diode bleue plus
\omterm{def:g11:color-light-sources:led}{luminophore}), et expliquer pourquoi sa lumière est
perçue comme blanche. Pourquoi un objet rouge profond paraît-il plus
terne sous une LED blanche bon marché qu'en lumière du jour~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:color-light-sources:11}
Calculer $\lambda_{\max}$ pour un corps incandescent à (a)~la
\omterm{def:g11:color-light-sources:colortemp}{température de couleur} d'une bougie
($\qty{1800}{K}$)~; (b)~celle de la lumière du jour voilée
($\qty{6500}{K}$). Laquelle des deux lumières est dite «~chaude~»~?
Commenter.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:color-light-sources:12}
Une lampe à sodium ($\qty{589}{nm}$) et un écran affichant du jaune
(sous-pixels à $\qty{610}{nm}$ et $\qty{540}{nm}$) ont exactement la
même couleur. (a)~Comment appelle-t-on deux telles lumières, et que
prouve leur existence sur la couleur~? (b)~Proposer deux expériences
différentes qui les distinguent, et prédire le résultat de chacune.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:color-light-sources:13}
Un projecteur halogène de théâtre fonctionne à $\qty{3200}{K}$.
(a)~Calculer $\lambda_{\max}$~: cette lumière est-elle plus chaude ou
plus froide que la lumière du jour~? (b)~Pour imiter la lumière du
jour, le projecteur est recouvert d'une gélatine bleu pâle. Que retire
la gélatine, et pourquoi la scène s'assombrit-elle~? (c)~Expliquer
pourquoi aucune gélatine ne pourrait au contraire
\emph{renforcer} l'extrémité bleue du \omterm{def:g10:light-spectra:white}{spectre} du
projecteur.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:color-light-sources:14}
Un boucher éclaire l'étal de viande avec des LED de

\begin{exercise}[$\star\star\star$]\label{exo:g11:color-light-sources:15}

\section{Problème~: Ingénierie de la couleur}

\begin{problem}[{Problème du week-end --- projecteurs de scène, rues au
sodium et LED blanche~: comment on fabrique la couleur, de trois
projecteurs qui peignent n'importe quelle teinte jusqu'à la puce qui
éclaire cette page}]
\label{pb:g11:color-light-sources:1}
Un éclairagiste ne possède que des projecteurs rouge, vert et bleu~; un
ingénieur de voirie possède une lampe qui n'émet qu'une seule
\omterm{def:g10:light-spectra:wavelength}{longueur d'onde}~; un fabricant de puces possède
une diode qui ne brille qu'en bleu. Tous trois vendent de la couleur.
Ce problème parcourt leurs métiers tour à tour --- mélange additif sur
scène, économie monochrome dans la rue, alchimie du
\omterm{def:g11:color-light-sources:led}{luminophore} dans la puce --- et s'achève en
choisissant la lumière d'une chambre et d'une vitrine de bijoutier.

\medskip
\noindent\textbf{Partie I --- La scène~: peindre avec trois projecteurs.}
\begin{enumerate}
  \item Les trois projecteurs se superposent sur un fond blanc. Nommer
        la couleur de chaque recouvrement deux à deux et du
        recouvrement triple.
  \item Comment l'éclairagiste fait-il de l'orange~? Et un rose pâle~?
        (Décrire qualitativement les
        \omterm{def:g10:energy-conservation:power}{puissances} des projecteurs.)
  \item Un danseur se tient devant le fond, éclairé par les projecteurs
        rouge et vert seulement. Deux ombres colorées apparaissent~:
        couleur de chacune~? Expliquer.
  \item Le fond est repeint en bleu profond. Que devient-il sous le
        projecteur rouge seul~? Sous les trois~?
  \item Pourquoi trois projecteurs peuvent-ils reproduire «~presque
        toutes~» les couleurs~? Quelle pièce d'anatomie fixe le nombre
        trois~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Costumes~: la soustraction au travail.}
\begin{enumerate}[resume]
  \item Un costume est jaune en \omterm{def:g10:light-spectra:white}{lumière blanche}.
        Quelle bande son teint absorbe-t-il~?
  \item Prédire sa couleur sous le projecteur rouge, le projecteur
        vert, le projecteur bleu.
  \item L'éclairagiste veut que le costume bascule de vif à noir d'un
        coup d'interrupteur. Vers quel projecteur bascule-t-on~?
  \item Une gélatine magenta est glissée devant un projecteur blanc
        éclairant le costume jaune. Suivre les bandes
        (\cref{met:g11:color-light-sources:bookkeeping})~: couleur du
        costume~?
  \item Pourquoi les gélatines de scène se présentent-elles en cyan,
        magenta et jaune plutôt qu'en rouge, vert et bleu~? (Que
        ferait une gélatine rouge à la
        \omterm{def:g10:energy-conservation:power}{puissance} d'un projecteur blanc~?)
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Rues au sodium~: l'économie monochrome.}
\begin{enumerate}[resume]
  \item Une lampe à sodium à \omterm{def:g10:pressure:pressure}{basse pression} émet
        essentiellement une raie à $\qty{589}{nm}$. Quel type de
        \omterm{def:g10:light-spectra:white}{spectre} est-ce, et à quoi ressemble toute la
        rue sous cette lumière~?
  \item Sous cette lampe, donner la couleur apparente d'un mur blanc,
        d'un marquage routier jaune, d'une voiture rouge, d'une voiture
        bleue.
  \item Pourquoi les astronomes aimaient-ils ces lampes~? (Que peut
        faire un seul \omterm{def:g11:color-light-sources:subtractive}{filtre} à toute la
        lueur de la ville~?)
  \item Une ampoule de rue au tungstène à $\qty{2700}{K}$ est proposée
        à la place. Calculer son $\lambda_{\max}$~: où va la plus grande
        partie de sa \omterm{def:g10:energy-conservation:power}{puissance}~?
  \item La lampe à sodium convertit environ $30\,\%$ de sa
        \omterm{def:g10:energy-conservation:power}{puissance} électrique en lumière visible,
        l'ampoule au tungstène environ $5\,\%$. Combien de
        \omterm{def:g10:energy-conservation:power}{watts} de lumière chacune délivre-t-elle
        par $\qty{100}{W}$ de \omterm{def:g10:energy-conservation:power}{puissance}~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La LED blanche et le choix d'une lumière.}
\begin{enumerate}[resume]
  \item Une LED blanche est une diode bleue ($\qty{450}{nm}$) sous un
        \omterm{def:g11:color-light-sources:led}{luminophore} jaune. Expliquer, avec les
        \omterm{prop:g11:color-light-sources:complementary}{couleurs complémentaires},
        pourquoi sa sortie se lit blanche.
  \item Décrire son \omterm{def:g10:light-spectra:white}{spectre} et le comparer au
        \omterm{def:g10:light-spectra:continuous}{spectre continu} de la lumière du jour.
        Quelle bande est sous-représentée~?
  \item En déduire quels objets une telle LED rend mal, et relier cela
        au boucher du \cref{exo:g11:color-light-sources:14}.
  \item Une ampoule LED est vendue «~$\qty{2700}{K}$~». Y a-t-il quelque
        chose dedans à $\qty{2700}{K}$~? Que promet l'étiquette~?
        Calculer le $\lambda_{\max}$ du corps incandescent qu'elle
        imite.
  \item Final --- choisir, en une phrase de justification chacune, une
        \omterm{def:g11:color-light-sources:colortemp}{température de couleur} pour une
        lampe de chambre et une pour une vitrine de bijoutier~; puis
        énoncer en une ligne la loi de la couleur du chapitre.
\end{enumerate}
\end{problem}
