\chapter{Forces et mouvement}\label{ch:g11:forces-and-motion}

Lancez une clé à molette qui tourne dans la pièce~: les extrémités
vacillent sauvagement, et pourtant un point à l'intérieur trace une
parabole parfaite. Le~\cref{ch:g10:inertia} a réglé ce qui se passe
lorsque les \omterm{def:g10:inertia:force}{forces}
\omterm{def:g10:inertia:compensate}{se compensent} --- rien. Ce chapitre
demande ce qui se passe lorsqu'elles ne le font pas~: la
\omterm{def:g10:inertia:force}{force} restante \emph{change la vitesse} ---
accélérant les corps, les ralentissant, les faisant tourner --- et dit
de combien, en rapports sinon encore en formules.

\section{Le centre de masse}

\begin{definition}[Centre de masse]\label{def:g11:forces-and-motion:com}
Le \emph{centre de masse}\index{centre de masse} d'un corps ou d'un
système est son point d'équilibre --- la position moyenne de sa masse~:
le centre géométrique d'un corps homogène et symétrique~; pour deux
masses ponctuelles, le point du segment qui les joint à des distances
dans le rapport \emph{inverse} des masses, $\frac{d_1}{d_2} =
\frac{m_2}{m_1}$ --- plus près de la plus lourde.
\end{definition}

\begin{example}[Une haltère déséquilibrée]\label{ex:g11:forces-and-motion:dumbbell}
Des balles de \qty{2.0}{kg} et \qty{1.0}{kg} sont aux extrémités d'une
barre légère de \qty{0.90}{m}. Le
\omterm{def:g11:forces-and-motion:com}{centre de masse} la divise dans le rapport
inverse $d_1/d_2 = 1.0/2.0$~: $\qty{0.30}{m}$ de la balle lourde,
$\qty{0.60}{m}$ de la légère --- équilibrer la barre là.
\end{example}

\begin{remark}[La clé lancée]\label{rem:g11:forces-and-motion:wrench}
Filmez une clé qui tumble~: chaque point suit une courbe en boucles
compliquées --- sauf le \omterm{def:g11:forces-and-motion:com}{centre de masse}, qui
trace la parabole nette qu'aurait une balle lancée. Le mouvement du
\omterm{def:g11:forces-and-motion:com}{centre de masse} est simple~; le reste est une
rotation \emph{autour} de lui --- d'où le fait que la mécanique peut
traiter une voiture, une planète ou une gymnaste comme un seul point.
\end{remark}

\section{Une force résultante change la vitesse}

\begin{definition}[Force résultante]\label{def:g11:forces-and-motion:netforce}
La \emph{force résultante}\index{force résultante} sur un corps est la
somme vectorielle de toutes les \omterm{def:g10:inertia:force}{forces}
qui agissent sur lui, $\vect F = \vect F_1 + \vect F_2 + \dots$
(pointe à queue, comme dans le volume de mathématiques)~; les
\omterm{def:g10:inertia:force}{forces}
\omterm{def:g10:inertia:compensate}{se compensent} exactement lorsque
$\vect F = \vect 0$.
\end{definition}

\begin{theorem}[Effet d'une force résultante --- semi-quantitatif]\label{thm:g11:forces-and-motion:secondlaw}
\index{deuxième loi du mouvement}
Une \omterm{def:g11:forces-and-motion:netforce}{force résultante} $\vect F$ agissant
sur un corps de masse $m$ pendant une courte durée $\Delta t$ change
la vitesse de son \omterm{def:g11:forces-and-motion:com}{centre de masse} d'un vecteur
$\Delta \vect v$ qui a la \emph{direction} de $\vect F$, et dont la
grandeur croît proportionnellement à $F$ et à $\Delta t$ et décroît
proportionnellement à la masse~:
\[
\Delta v \propto \frac{F \, \Delta t}{m} .
\]
\end{theorem}
\admitted

\begin{remark}
Doubler la \omterm{def:g10:inertia:force}{force} ou sa durée~: deux fois le
changement~; doubler la masse~: la moitié. L'année suivante affine cela
en une équation exacte, et le volume de l'année~1 demande dans quels
référentiels elle tient~; cette année les rapports suffisent --- ils
font déjà tourner airbags, freins et
\omterm{def:g10:universal-gravitation:satellite}{orbites}.
\end{remark}

\begin{proposition}[Les trois cas canoniques]\label{prop:g11:forces-and-motion:cases}
Soit un corps se déplaçant avec la vitesse $\vect v$ sous une
\omterm{def:g11:forces-and-motion:netforce}{force résultante} $\vect F$.
\begin{enumerate}
  \item $\vect F$ le long de $\vect v$~: il \emph{accélère} en ligne
        droite~;
  \item $\vect F$ opposée à $\vect v$~: il \emph{ralentit} en ligne
        droite~;
  \item $\vect F$ perpendiculaire à $\vect v$~: il \emph{tourne} sans
        changer de vitesse.
\end{enumerate}
\end{proposition}

\begin{proof}
Ajouter $\Delta\vect v$, dirigé le long de $\vect F$
(\cref{thm:g11:forces-and-motion:secondlaw}), pointe à queue sur
$\vect v$~: le long de $\vect v$ les flèches s'empilent (plus
longue)~; opposées, elles s'annulent partiellement (plus courte)~;
perpendiculaires, chaque petit $\Delta \vect v$ incline la flèche sans
l'allonger --- seule la direction tourne.
\end{proof}

\begin{example}[Un coup de côté]\label{ex:g11:forces-and-motion:kick}
Un palet glisse vers l'est à \qty{6.0}{m/s}~; un coup latéral bref
ajoute $\Delta\vect v = \qty{8.0}{m/s}$ plein nord. La nouvelle vitesse
est la somme vectorielle~: grandeur $\sqrt{6.0^2 + 8.0^2} =
\qty{10.0}{m/s}$, direction $\tan\theta = 8.0/6.0$, c.-à-d.
$\theta \approx 53^\circ$ nord de l'est --- l'ancienne vitesse n'est
pas effacée, le changement s'y ajoute.
\end{example}

\section{Comparer les changements~: force, temps, masse}

\begin{method}[Raisonnement par rapports]\label{met:g11:forces-and-motion:ratio}
Pour comparer deux situations, ne jamais résoudre une équation ---
former des rapports~: le rapport de changement de vitesse est le
rapport de $F$, multiplié par le rapport de $\Delta t$, divisé par le
rapport de $m$. Puis lire quel facteur a changé, de combien.
\end{method}

\begin{example}[Le chariot chargé]\label{ex:g11:forces-and-motion:cart}
La même poussée, appliquée pendant la même seconde, envoie un chariot
de courses vide de \qty{20}{kg} à \qty{1.2}{m/s}. Chargé à
\qty{60}{kg} --- triple masse, même $F\Delta t$ --- il part au tiers,
\qty{0.40}{m/s}. La masse est l'entêtement de la matière.
\end{example}

\begin{remark}[Accidents, airbags, genoux pliés]\label{rem:g11:forces-and-motion:airbag}
Dans un accident $\Delta v$ et $m$ ne se négocient pas --- la vitesse
de l'occupant doit atteindre zéro --- donc $F\,\Delta t$ est fixé. La
seule liberté restante est le \emph{temps}~: étirez $\Delta t$ et la
\omterm{def:g10:inertia:force}{force} se contracte dans le même rapport. Une
tête s'arrêtant sur un airbag en \qty{0.08}{s} au lieu du volant en
\qty{0.01}{s} ressent une \omterm{def:g10:inertia:force}{force} moyenne huit
fois plus petite~; zones de déformation, mousse de casque et genoux
pliés jouent le même tour.
\end{remark}

\section{Le frottement~: la force qui combat le glissement}

\begin{definition}[Frottements statique et cinétique]\label{def:g11:forces-and-motion:friction}
Le \omterm{def:g10:inertia:inventory}{frottement} entre surfaces en contact
s'oppose à leur \emph{glissement relatif}. Le \emph{frottement
statique}\index{frottement!statique} agit tant qu'elles ne glissent pas,
s'ajustant jusqu'à un maximum pour annuler ce qui tente de démarrer le
glissement~; le \emph{frottement cinétique}\index{frottement!cinétique},
une fois le glissement commencé, est à peu près constant, dirigé contre
le glissement, et en général \emph{plus faible} que le maximum
statique.
\end{definition}

\begin{example}[Pousser l'armoire]\label{ex:g11:forces-and-motion:wardrobe}
Poussez une armoire doucement~: elle ne bouge pas --- le frottement
statique annule vos \qty{150}{N}. Poussez plus fort~: vers \qty{400}{N}
elle bascule en mouvement, puis glisse avec moins d'effort --- le
\omterm{def:g11:forces-and-motion:friction}{frottement cinétique} plus faible a pris le
relais.
\end{example}

\begin{remark}[Pourquoi l'ABS fonctionne]\label{rem:g11:forces-and-motion:abs}
Le point de contact d'une roue qui roule ne glisse pas sur la route~:
le pneu adhère par \omterm{def:g10:inertia:inventory}{frottement}
\emph{statique}. Une roue bloquée dérape et n'a que le
\omterm{def:g10:inertia:inventory}{frottement} \emph{cinétique} plus faible, qui
pointe contre le dérapage quoi que fassent les roues --- la direction
meurt. L'ABS relâche le frein dès qu'une roue se bloque~: plus
d'adhérence, arrêt plus court, un conducteur qui peut diriger.
\end{remark}

\section{Tourner~: le mouvement circulaire et la force vers l'intérieur}

\begin{definition}[Mouvement circulaire uniforme]\label{def:g11:forces-and-motion:ucm}
Un corps est en \emph{mouvement circulaire uniforme}\index{mouvement
circulaire uniforme} lorsque sa
\omterm{def:g10:relative-motion:trajectory}{trajectoire} est un cercle parcouru
à vitesse constante. La vitesse, tangente au cercle, change de direction
à chaque instant --- elle n'est donc jamais constante.
\end{definition}

\begin{proposition}[La force vers l'intérieur]\label{prop:g11:forces-and-motion:inward}
Un corps en \omterm{def:g11:forces-and-motion:ucm}{mouvement circulaire uniforme}
est soumis à une \omterm{def:g11:forces-and-motion:netforce}{force résultante} qui
n'est jamais nulle~: à chaque instant elle pointe du corps vers le
centre du cercle.
\end{proposition}

\begin{proof}
La vitesse est constante, donc la
\omterm{def:g11:forces-and-motion:netforce}{force résultante} n'a pas de part le long de
$\vect v$ (cas 1 et 2 de la
\cref{prop:g11:forces-and-motion:cases})~: elle est perpendiculaire à la
tangente, donc le long du rayon --- et vers l'intérieur, puisque chaque
$\Delta \vect v$ courbe le chemin vers le centre.
\end{proof}

\begin{example}[Le fil, le virage, la Lune]\label{ex:g11:forces-and-motion:moon}
Chaque virage régulier cache une
\omterm{def:g10:inertia:force}{force} vers l'intérieur. La balle tournoyée~:
la \omterm{def:g10:inertia:inventory}{tension} du fil. La voiture en virage~:
le \omterm{def:g10:inertia:inventory}{frottement} statique latéral de la route
sur les pneus --- disparu sur la glace, et la voiture va tout droit. La
Lune~: l'attraction gravitationnelle de la Terre
(\cref{ch:g10:universal-gravitation}), qui la fait tourner en
\num{27.3}~jours --- non tenue \emph{en l'air} mais tirée de côté dans
un virage perpétuel, tombant autour de la Terre~; l'année suivante en
fait la mécanique des
\omterm{def:g10:universal-gravitation:satellite}{satellites}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:1}
\omterm{def:g11:forces-and-motion:netforce}{Force résultante} nulle ou non nulle~?
Justifier d'après la vitesse~: (a)~une voiture croisière en ligne
droite à \qty{90}{km/h} constants~; (b)~freinage en ligne droite~;
(c)~virage à \qty{50}{km/h} constants~; (d)~un parachutiste à vitesse
limite.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:2}
Des balles de \qty{3.0}{kg} et \qty{1.0}{kg} sont aux extrémités d'une
barre légère de \qty{0.80}{m}. Localiser le
\omterm{def:g11:forces-and-motion:com}{centre de masse}~; où pour des masses
égales~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:3}
La même poussée brève est donnée à un chariot à bagages vide de
\qty{15}{kg} et au même chariot chargé à \qty{45}{kg}. Comparer les
changements de vitesse~; quelle poussée donnerait au chariot chargé le
$\Delta v$ du vide~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:4}
Un tramway se déplaçant à \qty{10}{m/s} freine~; une seconde plus tard
il se déplace à \qty{8}{m/s}, même direction. Donner $\Delta \vect v$
et la direction de la
\omterm{def:g11:forces-and-motion:netforce}{force résultante}. Quel cas de la
\cref{prop:g11:forces-and-motion:cases} est-ce~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:forces-and-motion:5}
Frottement statique ou
\omterm{def:g11:forces-and-motion:friction}{cinétique} --- et dirigé dans quel
sens~? (a)~Une caisse résistant à votre poussée (trop douce)~; (b)~la
même caisse une fois qu'elle glisse~; (c)~un livre sur un plateau
légèrement incliné, qui ne glisse pas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:6}
Une \omterm{def:g11:forces-and-motion:netforce}{force résultante} $F$ appliquée
pendant $\Delta t$ à une masse $m$ produit $\Delta v = \qty{4.0}{m/s}$.
Prédire $\Delta v$ pour~: (a)~$2F, \Delta t, m$~; (b)~$F, \Delta t/2,
m$~; (c)~$F, \Delta t, 2m$~; (d)~$3F, 2\Delta t, 6m$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:7}
Un palet glisse vers l'est à \qty{8.0}{m/s}~; un coup ajoute
$\Delta \vect v = \qty{6.0}{m/s}$ plein nord. Calculer la nouvelle
vitesse et la direction. Pourquoi la vitesse n'est-elle pas
$8.0 + 6.0 = \qty{14}{m/s}$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:8}
Un gardien de but arrête le même tir de deux manières~: bras rigides
(la balle s'arrête en \qty{0.02}{s}) ou en «~accompagnant~» la balle
(\qty{0.08}{s}). Comparer les \omterm{def:g10:inertia:force}{forces}
moyennes, et indiquer quelles quantités étaient fixées avant son
choix.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:9}
Quelle \omterm{def:g10:inertia:force}{force} externe pousse un sprinter en
avant au départ~? Pourquoi les chaussures à pointes aident-elles --- et
pourquoi personne ne peut accélérer, ni même marcher, sur une glace
parfaitement sans frottement~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:10}
En arrêt d'urgence, pourquoi une voiture avec ABS (roues maintenues en
roulement) s'arrête-t-elle en général plus court qu'une avec roues
bloquées en dérapage --- et pourquoi la voiture qui dérape ne peut-elle
plus être dirigée~? Citer le
\cref{def:g11:forces-and-motion:friction}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:forces-and-motion:11}
Un lanceur de marteau fait tourner la balle sur son fil. Quelle
\omterm{def:g10:inertia:force}{force} fait tourner la balle~? Esquisser sa
trajectoire après le lâcher. Pourquoi la Lune n'a-t-elle pas besoin de
fil, et que deviendrait sa trajectoire sans gravité~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:12}
Un balai est un bâton uniforme de \qty{1.0}{kg} (propre
\omterm{def:g11:forces-and-motion:com}{centre de masse} à \qty{0.60}{m} du sommet)
avec une tête de \qty{2.0}{kg} dont le
\omterm{def:g11:forces-and-motion:com}{centre de masse} est à \qty{1.20}{m} du
sommet. En traitant chaque partie comme une masse ponctuelle, localiser
le \omterm{def:g11:forces-and-motion:com}{centre de masse} du balai et vérifier la
règle du rapport inverse de la
\cref{def:g11:forces-and-motion:com}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:13}
Une voiture de masse $m$ et une camionnette chargée de masse $2m$
freinent depuis la même vitesse avec la même
\omterm{def:g10:inertia:force}{force} de freinage.
\begin{enumerate}
  \item Comparer les temps nécessaires pour s'arrêter.
  \item Chacune perd de la vitesse régulièrement, donc parcourt à la
        même vitesse \emph{moyenne}~: comparer les distances
        d'arrêt.
  \item La voiture freine maintenant depuis le \emph{double} de la
        vitesse, même \omterm{def:g10:inertia:force}{force}~: comparer temps et
        distance avec son premier arrêt. Morale pour les limitations de
        vitesse~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:forces-and-motion:14}
La Lune \omterm{def:g10:universal-gravitation:satellite}{orbite} à
$R \approx \qty{3.84e8}{m}$ en $T = \num{27.3}$~jours.
\begin{enumerate}
  \item Calculer sa vitesse orbitale (circonférence sur
        \omterm{def:g10:signals-and-waves:period}{période}).
  \item Dans quel sens pointe son changement de vitesse, et quelle
        \omterm{def:g10:inertia:force}{force} le fournit~?
  \item Améliorer, en utilisant la
        \cref{prop:g11:forces-and-motion:inward}~: «~la Lune ne tombe pas
        parce qu'elle va assez vite~».
\end{enumerate}
\end{exercise}

Un œuf de masse \qty{50}{g} frappe le sol à \qty{4.0}{m/s}~: sur carreau
il s'arrête en environ \qty{1}{ms}, sur mousse épaisse en environ
\qty{50}{ms}. Comparer les \omterm{def:g10:inertia:force}{forces} moyennes.

\section{Problème~: L'essai de choc}

\begin{problem}[{Problème du week-end --- concevoir la collision douce~:
puisqu'un accident fixe le changement de vitesse, chaque dispositif de
sécurité d'une voiture est une machine pour étirer le temps}]
\label{pb:g11:forces-and-motion:1}
Un laboratoire d'essais de choc conduit une voiture à \qty{50}{km/h}
contre un mur rigide, avec un mannequin adulte de \qty{70}{kg} et un
mannequin enfant de \qty{20}{kg} à bord~; des caméras chronomètrent
chaque arrêt. Votre travail~: trouver où vit la violence d'un accident,
et la concevoir pour l'éloigner.

\medskip
\noindent\textbf{Partie I --- La mesure d'un accident.}
\begin{enumerate}
  \item Convertir \qty{50}{km/h} en \unit{m/s}.
  \item Dans tout arrêt frontal le changement de vitesse des occupants
        a la même grandeur. Laquelle~? Pourquoi aucun dispositif ne
        peut-il l'altérer~?
  \item Citer le \cref{thm:g11:forces-and-motion:secondlaw}~: avec
        $\Delta v$ et $m$ fixés, $F\,\Delta t$ est fixé. Le seul cadran
        du concepteur~?
  \item Le mur arrête la voiture en \qty{0.050}{s}~; une zone de
        déformation étire cela à \qty{0.150}{s}. Comparer les
        \omterm{def:g10:inertia:force}{forces} moyennes.
  \item Dans quelle direction $\Delta \vect v$ et la
        \omterm{def:g11:forces-and-motion:netforce}{force résultante} sur les
        mannequins pointent-elles pendant l'arrêt~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- La zone de déformation.}
\begin{enumerate}[resume]
  \item Pourquoi, en une phrase, les ingénieurs conçoivent-ils l'avant
        de la voiture pour qu'il soit détruit~?
  \item Dans une voiture rigide d'époque (arrêt en \qty{0.050}{s}) le
        mannequin est sanglé rigidement au siège. Comparer sa
        \omterm{def:g10:inertia:force}{force} avec l'arrêt moderne de
        \qty{0.150}{s}.
  \item La vitesse de la voiture qui s'arrête chute régulièrement
        (moyenne~: la moitié de l'initiale). Quelle distance parcourt-elle
        en \qty{0.150}{s}~? Comparer avec une zone de déformation réelle.
  \item Pourquoi pas une zone de déformation de \qty{10}{m}~? Et quelle
        partie de la voiture ne doit \emph{pas} se déformer~?
  \item Résumer la partie~II en slogan~: «~vous ne pouvez pas choisir
        $\Delta v$, donc vous devez choisir~\dots~».
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La ceinture, l'airbag et l'enfant.}
\begin{enumerate}[resume]
  \item Un mannequin non ceinturé garde ses \qty{14}{m/s}
        (\cref{ch:g10:inertia}) jusqu'à ce que le pare-brise l'arrête
        en \qty{0.005}{s}~; un ceinturé s'arrête avec la voiture en
        \qty{0.150}{s}. Comparer les
        \omterm{def:g10:inertia:force}{forces} moyennes.
  \item Pourquoi une ceinture qui s'étire un peu est-elle meilleure
        qu'un harnais d'acier inextensible~?
  \item La tête s'arrêterait sur le volant en \qty{0.010}{s}~; l'airbag
        étire cela à \qty{0.080}{s}. Comparer les
        \omterm{def:g10:inertia:force}{forces}.
  \item Adulte et enfant subissent le même $\Delta v$ dans le même
        $\Delta t$. Comparer les
        \omterm{def:g11:forces-and-motion:netforce}{forces résultantes} que les deux
        ceintures doivent fournir.
  \item Un enfant sur les genoux~: les bras doivent changer la vitesse
        d'un bébé de \qty{10}{kg} de \qty{14}{m/s} en \qty{0.150}{s},
        tandis que la gravité change une vitesse de \qty{9.8}{m/s}
        chaque seconde. Combien de fois plus vite les bras doivent-ils
        agir --- tenir quelle masse se sentirait aussi lourd~?
        Conclure.
  \item Les enfants ressentent de plus petites
        \omterm{def:g10:inertia:force}{forces} --- alors pourquoi des sièges
        enfants~? Penser à \emph{où} la ceinture applique sa
        \omterm{def:g10:inertia:force}{force}, et à $\Delta t$.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La cage de sécurité, et le verdict.}
\begin{enumerate}[resume]
  \item La cellule passagers est rigide tandis que les deux extrémités
        se déforment. Pourquoi la cellule ne doit-elle pas se déformer,
        bien que la rigidité signifie des arrêts violents~?
  \item Deux voitures identiques se heurtent de face, chacune à
        \qty{50}{km/h}. Où chacune s'arrête-t-elle~? Comparer chaque
        accident avec l'essai contre le mur.
  \item Une petite voiture rencontre un camion chargé de face. Les
        \omterm{def:g10:inertia:force}{forces} qu'elles exercent l'une sur
        l'autre sont égales et opposées, mutuelles comme toute
        interaction (\cref{ch:g11:fundamental-interactions}). La vitesse
        de qui change le plus~?
  \item Lister les dispositifs de sécurité de ce problème et la seule
        quantité que chacun d'eux étire.
  \item Final~: énoncer la loi de conception de la collision douce en
        une phrase en n'utilisant que \emph{masse}, \emph{changement de
        vitesse}, \emph{\omterm{def:g10:inertia:force}{force}} et
        \emph{temps}.
\end{enumerate}
\end{problem}
