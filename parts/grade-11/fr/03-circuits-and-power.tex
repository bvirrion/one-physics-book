\chapter{Circuits électriques et puissance}\label{ch:g11:circuits-and-power}

Actionnez un interrupteur~: une lampe s'allume, parce qu'un générateur
à des kilomètres pousse de la charge dans une boucle de cuivre à travers
votre mur. Les années précédentes ont cartographié le circuit ---
intensité, \omterm{def:g11:circuits-and-power:voltage}{tension},
\omterm{prop:g11:circuits-and-power:ohm}{loi d'Ohm}. Ce chapitre le rouvre
avec le livre de comptes du \cref{ch:g10:energy-conservation}~: ce que
compte le compteur, pourquoi les fils
\omterm{def:g11:lenses-and-eye:lens}{minces} chauffent, pourquoi le réseau
tourne à \qty{400}{kV}.

\section{La charge en mouvement}

\begin{definition}[Intensité et coulomb]\label{def:g11:circuits-and-power:current}
Un \emph{courant électrique}\index{courant électrique} est un
écoulement ordonné de charge (des électrons, dans un métal). Son
intensité $I$, en \emph{ampères}\index{ampère} (\unit{A}), est la charge
qui traverse une section du fil par seconde~: un courant permanent $I$
pendant un temps $t$ transporte la charge
\[ q = I\,t , \]
en \emph{coulombs}\index{coulomb} (\unit{C}), avec $\qty{1}{C} =
\qty{1}{A.s}$. La charge élémentaire est $e = \qty{1.6e-19}{C}$.
L'intensité se mesure avec un ampèremètre, inséré en série.
\end{definition}

\begin{example}[Foudre contre téléphone]\label{ex:g11:circuits-and-power:coulomb}
Un coup de foudre transporte quelque $\qty{3.0e4}{A}$ pendant environ
$\qty{1.0e-4}{s}$~: $q = 3.0\times10^{4} \times 10^{-4} = \qty{3.0}{C}$.
Votre téléphone, à \qty{2.0}{A}, déplace la même charge toutes les
$1.5$~secondes.
\end{example}

\begin{definition}[Tension et volt]\label{def:g11:circuits-and-power:voltage}
La \emph{tension}\index{tension} $U$ entre deux points d'un circuit, en
\emph{volts}\index{volt} (\unit{V}), est l'\omterm{def:g10:energy-conservation:energy}{énergie}
échangée par \omterm{def:g10:orders-of-magnitude:unit}{unité} de charge qui voyage
entre eux~: $\qty{1}{V} = \qty{1}{J/C}$, donc une charge $q$ traversant
un dipôle sous tension $U$ échange avec lui l'\omterm{def:g10:energy-conservation:energy}{énergie}
$E = qU$. Un voltmètre, branché en dérivation, la mesure.
\end{definition}

\begin{proposition}[Lois des circuits]\label{prop:g11:circuits-and-power:laws}
Dans un circuit en régime permanent~:
\begin{enumerate}
  \item \emph{loi des nœuds~:} les intensités qui arrivent à un nœud
        s'ajoutent aux intensités qui en partent~;
  \item \emph{loi des mailles (série)~:} les
        \omterm{def:g11:circuits-and-power:voltage}{tensions} le long d'un chemin
        s'ajoutent~: $U_{AC} = U_{AB} + U_{BC}$~;
  \item \emph{loi des branches parallèles~:} deux branches joignant les
        mêmes deux points portent la même
        \omterm{def:g11:circuits-and-power:voltage}{tension}.
\end{enumerate}
\end{proposition}

\begin{proof}
En régime permanent la charge ne s'accumule nulle part, donc ce qui
entre dans un nœud en sort.
L'\omterm{def:g10:energy-conservation:energy}{énergie} par
\omterm{def:g11:circuits-and-power:current}{coulomb} est additive le long d'un chemin et
ne dépend que de ses extrémités --- pas de la branche prise entre
elles.
\end{proof}

\section{Loi d'Ohm et réseaux de résistances}

\begin{definition}[Résistance]\label{def:g11:circuits-and-power:resistance}
La \emph{résistance}\index{résistance} d'un conducteur est le rapport
$R = U/I$ de la \omterm{def:g11:circuits-and-power:voltage}{tension} à ses bornes à
l'intensité qui le traverse, mesurée en \emph{ohms}\index{ohm}
(\unit{\ohm})~: $\qty{1}{\ohm} = \qty{1}{V/A}$.
\end{definition}

\begin{proposition}[Loi d'Ohm]\label{prop:g11:circuits-and-power:ohm}
\index{loi d'Ohm}
Pour un conducteur métallique maintenu à température fixe, $R$ est
constante~: \omterm{def:g11:circuits-and-power:voltage}{tension} et intensité sont
proportionnelles,
\[ U = R\,I . \]
\end{proposition}
\admitted

\begin{remark}\label{rem:g11:circuits-and-power:ohm-limits}
Une loi expérimentale, non universelle~: une diode, un filament de
lampe chauffante ou votre propre peau ne l'obéissent pas. Pourquoi les
métaux l'obéissent est dérivé dans les volumes universitaires.
\end{remark}

\begin{proposition}[Résistances en série et en parallèle]\label{prop:g11:circuits-and-power:networks}
\index{résistance équivalente}
Deux résistances en série sont équivalentes à une résistance
$R_{\text{s}} = R_1 + R_2$~; deux en parallèle, à une résistance
$R_{\text{p}}$ donnée par
\[
\frac{1}{R_{\text{p}}} = \frac{1}{R_1} + \frac{1}{R_2} ,
\qquad\text{c.-à-d.}\quad
R_{\text{p}} = \frac{R_1 R_2}{R_1 + R_2} < \min(R_1, R_2) .
\]
\end{proposition}

\begin{proof}
Série~: le même $I$ traverse les deux et les
\omterm{def:g11:circuits-and-power:voltage}{tensions} s'ajoutent
(\cref{prop:g11:circuits-and-power:laws}), donc $U = (R_1 + R_2)\,I$.
Parallèle~: les deux sont sous la même $U$ et les intensités
s'ajoutent, donc $I = \frac{U}{R_1} + \frac{U}{R_2}$.
\end{proof}

\begin{example}[Résistance équivalente]\label{ex:g11:circuits-and-power:networks}
$R_1 = \qty{100}{\ohm}$ et $R_2 = \qty{150}{\ohm}$~: en série,
\qty{250}{\ohm}~; en parallèle, $\frac{100 \times 150}{250} =
\qty{60}{\ohm}$ --- moins que chacune~: le courant a une seconde
route.
\end{example}

\section{Énergie et puissance dans un circuit}

\begin{proposition}[Puissance électrique]\label{prop:g11:circuits-and-power:power}
\index{puissance électrique}
Un dipôle sous \omterm{def:g11:circuits-and-power:voltage}{tension} $U$ traversé par une
intensité $I$ reçoit de l'\omterm{def:g10:energy-conservation:energy}{énergie} au taux
\[ P = U\,I , \qquad\text{c.-à-d.}\quad E = U\,I\,t \ \text{pendant un temps } t . \]
\end{proposition}

\begin{proof}
Pendant un temps $t$ la charge $q = It$ traverse le dipôle
(\cref{def:g11:circuits-and-power:current}) et chaque
\omterm{def:g11:circuits-and-power:current}{coulomb} cède $U$
\omterm{def:g10:energy-conservation:energy}{joules}
(\cref{def:g11:circuits-and-power:voltage})~: $E = qU = UIt$.
\end{proof}

\begin{proposition}[Puissance dans une résistance]\label{prop:g11:circuits-and-power:ri2}
Une résistance $R$ parcourue par une intensité $I$ dissipe
\[ P = R\,I^2 = \frac{U^2}{R} . \]
\end{proposition}

\begin{proof}
Substituer la \omterm{prop:g11:circuits-and-power:ohm}{loi d'Ohm} dans $P = UI$~:
$P = (RI)I = RI^2 = U(U/R) = U^2/R$.
\end{proof}

\begin{example}[Lire une plaque signalétique]\label{ex:g11:circuits-and-power:nameplate}
Une bouilloire marquée «~\qty{230}{V} -- \qty{2200}{W}~» tire
$I = P/U = 2200/230 = \qty{9.6}{A}$ et a une
\omterm{def:g11:circuits-and-power:resistance}{résistance} $R = U^2/P = 230^2/2200 =
\qty{24}{\ohm}$. Pendant \qty{150}{s} elle consomme $E = 2200 \times 150
= \qty{3.3e5}{J} \approx \qty{0.09}{kW.h}$ --- environ deux centimes.
\end{example}

\begin{remark}[Joules et kilowatt-heures]\label{rem:g11:circuits-and-power:kwh}
Le compteur compte des \omterm{def:g10:energy-conservation:kwh}{kilowatt-heures}
(\cref{def:g10:energy-conservation:kwh})~: $\qty{1}{kW.h} =
\qty{3.6e6}{J}$. Les joules sont l'\omterm{def:g10:orders-of-magnitude:unit}{unité} du
physicien~; les \omterm{def:g10:energy-conservation:kwh}{kilowatt-heures} tarifent les
factures.
\end{remark}

\section{L'effet Joule~: du grille-pain au pylône}

\begin{definition}[Effet Joule]\label{def:g11:circuits-and-power:joule}
La dissipation $P = RI^2$ qui transforme
l'\omterm{def:g10:energy-conservation:forms}{énergie électrique} en chaleur dans chaque
résistance est l'\emph{effet Joule}\index{effet Joule}. Radiateurs,
bouilloires, grille-pain et ampoules à filament sont des résistances
par conception~; un \emph{fusible}\index{fusible} est un fil
\omterm{def:g11:lenses-and-eye:lens}{mince} calibré pour fondre --- ouvrant le circuit ---
lorsque l'intensité dépasse son calibre.
\end{definition}

\begin{example}[Pourquoi les pylônes tournent haut]\label{ex:g11:circuits-and-power:transmission}
Un village a besoin de $P = \qty{100}{kW}$ à travers une ligne de
\omterm{def:g11:circuits-and-power:resistance}{résistance} $R_\ell = \qty{5.0}{\ohm}$. Livré à
$U = \qty{500}{V}$~: $I = P/U = \qty{200}{A}$ et la ligne gaspille
$R_\ell I^2 = \qty{200}{kW}$ --- le double de la livraison~: absurde. À
$U = \qty{20}{kV}$~: $I = \qty{5.0}{A}$ et $R_\ell I^2 = \qty{125}{W}$.
Quarante fois moins d'intensité, $1600$ fois moins de pertes.
\end{example}

\end{omfigure}

\begin{remark}[La règle de fer du réseau]\label{rem:g11:circuits-and-power:grid}
À \omterm{def:g10:energy-conservation:power}{puissance} livrée fixe $P = UI$, la perte en
ligne $R_\ell I^2 = R_\ell P^2/U^2$ chute comme le \emph{carré} de la
\omterm{def:g11:circuits-and-power:voltage}{tension} de transport. D'où~: élever
(\qty{400}{kV}) pour voyager, abaisser (\qty{230}{V}) pour vivre.
\end{remark}

\begin{remark}[Sécurité domestique]\label{rem:g11:circuits-and-power:safety}
Une ligne à \qty{230}{V} fusible à \qty{16}{A} fournit au plus
$230 \times 16 \approx \qty{3.7}{kW}$~: deux bouilloires à la fois
fondent le \omterm{def:g11:circuits-and-power:joule}{fusible} --- par conception le maillon
le plus mince, le plus faible, bien moins cher qu'un incendie dans le
mur.
\end{remark}

\section{Piles réelles~: force électromotrice}

\begin{definition}[f.é.m. et résistance interne]\label{def:g11:circuits-and-power:emf}
Une pile convertit l'énergie chimique en
\omterm{def:g10:energy-conservation:forms}{énergie électrique}. Sa \emph{force
électromotrice}\index{force électromotrice} (f.é.m.)
$\mathcal{E}$, en \omterm{def:g11:circuits-and-power:voltage}{volts}, est
l'\omterm{def:g10:energy-conservation:energy}{énergie} qu'elle cède à chaque
\omterm{def:g11:circuits-and-power:current}{coulomb} qui la traverse~; sa
\emph{résistance interne}\index{résistance interne} $r$ rend compte de
l'\omterm{def:g11:circuits-and-power:joule}{effet Joule} dans sa propre chimie. Modèle~: une
source idéale $\mathcal{E}$ en série avec $r$ (calligraphique
$\mathcal{E}$~: $E$ est l'\omterm{def:g10:energy-conservation:energy}{énergie}).
\end{definition}

\begin{proposition}[Tension aux bornes]\label{prop:g11:circuits-and-power:terminal}
Une pile délivrant une intensité $I$ présente à ses bornes
\[ U = \mathcal{E} - r\,I . \]
\end{proposition}

\begin{proof}
La chimie fournit $\mathcal{E}I$, la
\omterm{def:g11:circuits-and-power:emf}{résistance interne} mange $rI^2$, le circuit reçoit
$UI$~: la conservation de l'\omterm{def:g10:energy-conservation:energy}{énergie}
(\cref{ch:g10:energy-conservation}) donne $UI = \mathcal{E}I - rI^2$.
\end{proof}

\begin{example}[Une pile en charge]\label{ex:g11:circuits-and-power:discharge}
Une pile ($\mathcal{E} = \qty{9.0}{V}$, $r = \qty{1.0}{\ohm}$) alimente
une ampoule de \omterm{def:g11:circuits-and-power:resistance}{résistance} $R = \qty{8.0}{\ohm}$.
La boucle obéit à $\mathcal{E} = rI + RI$, donc
$I = \mathcal{E}/(R + r) = \qty{1.0}{A}$ et $U = 9.0 - 1.0 \times 1.0 =
\qty{8.0}{V}$~: l'ampoule reçoit \qty{8.0}{W} tandis que la pile gaspille
\qty{1.0}{W} en se chauffant.
\end{example}

\begin{remark}[Récepteurs]\label{rem:g11:circuits-and-power:receivers}
Un moteur ou une batterie en charge fait tourner la conversion à
l'envers~: il absorbe sous $U = \mathcal{E}' + r'I$, convertissant
$\mathcal{E}'I$ en mouvement ou en chimie et perdant $r'I^2$. Les
sources poussent, les récepteurs résistent.
\end{remark}

\begin{example}[Rendement d'une pile]\label{ex:g11:circuits-and-power:efficiency}
Le \omterm{def:g10:energy-conservation:efficiency}{rendement}
(\cref{def:g10:energy-conservation:efficiency}) de la décharge
ci-dessus~: utile $UI = \qty{8.0}{W}$ sur $\mathcal{E}I =
\qty{9.0}{W}$, donc $\eta = U/\mathcal{E} = 8/9 \approx 0.89$ ---
qui chute quand l'intensité monte~: les piles qui travaillent fort
chauffent.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:circuits-and-power:1}
Une batterie de téléphone se charge à un régime permanent de
\qty{2.0}{A} pendant \qty{90}{min}. Quelle charge traverse le câble, en
\omterm{def:g11:circuits-and-power:current}{coulombs}~? Combien de charges élémentaires
est-ce ($e = \qty{1.6e-19}{C}$)~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:circuits-and-power:2}
Une résistance de \qty{220}{\ohm} sous \qty{12}{V}~: quelle intensité
circule~? Un élément chauffant tire \qty{0.50}{A} sous \qty{230}{V}~:
quelle \omterm{def:g11:circuits-and-power:resistance}{résistance}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:circuits-and-power:3}
Calculer la \omterm{prop:g11:circuits-and-power:networks}{résistance équivalente} de
$R_1 = \qty{120}{\ohm}$ et $R_2 = \qty{60}{\ohm}$ en série, puis en
parallèle. Laquelle est plus petite que les deux, et pourquoi~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:circuits-and-power:4}
Un fer de voyage tire \qty{8.7}{A} sous \qty{230}{V}. Calculer sa
\omterm{def:g10:energy-conservation:power}{puissance}, puis
l'\omterm{def:g10:energy-conservation:energy}{énergie} qu'il consomme en \qty{4.0}{min},
en \omterm{def:g10:energy-conservation:energy}{joules} et en
\omterm{def:g10:energy-conservation:kwh}{kilowatt-heures}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:circuits-and-power:5}
Un chargeur d'ordinateur portable délivre \qty{60}{W} pendant
\qty{5.0}{h} par jour~: \omterm{def:g10:energy-conservation:energy}{énergie} quotidienne en
\omterm{def:g10:energy-conservation:kwh}{kilowatt-heures} et
\omterm{def:g10:energy-conservation:energy}{joules}, et coût à $0{,}25$~euro par
\unit{kW.h}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:circuits-and-power:6}
Un radiateur électrique est calibré à \qty{1200}{W} sous \qty{230}{V}.
\begin{enumerate}
  \item Calculer sa \omterm{def:g11:circuits-and-power:resistance}{résistance} et l'intensité
        qu'il tire.
  \item Branché (même \omterm{def:g11:circuits-and-power:resistance}{résistance}) sur une
        alimentation à \qty{115}{V}, quelle
        \omterm{def:g10:energy-conservation:power}{puissance} délivre-t-il~? Pourquoi pas
        la moitié~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:circuits-and-power:7}
$R_1 = \qty{40}{\ohm}$ et $R_2 = \qty{20}{\ohm}$ en série sous
\qty{12}{V}. Calculer l'intensité, la
\omterm{def:g11:circuits-and-power:voltage}{tension} aux bornes de chaque résistance, et
la \omterm{def:g10:energy-conservation:power}{puissance} dissipée dans chacune~; vérifier
que les \omterm{def:g10:energy-conservation:power}{puissances} s'ajoutent en $UI$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:circuits-and-power:8}
$R_1 = \qty{60}{\ohm}$ et $R_2 = \qty{30}{\ohm}$ en parallèle sous
\qty{12}{V}. Calculer l'intensité de chaque branche, l'intensité
totale, et la \omterm{prop:g11:circuits-and-power:networks}{résistance équivalente} de deux
manières.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:circuits-and-power:9}
Une pile délivre $U = \qty{5.5}{V}$ à $I = \qty{0.50}{A}$, et
$U = \qty{4.0}{V}$ à $I = \qty{2.0}{A}$. Trouver sa f.é.m.
$\mathcal{E}$ et sa \omterm{def:g11:circuits-and-power:emf}{résistance interne} $r$, puis
l'intensité de court-circuit $\mathcal{E}/r$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:circuits-and-power:10}
Une ligne de cuisine à \qty{230}{V} est protégée par un
\omterm{def:g11:circuits-and-power:joule}{fusible} de \qty{16}{A}. Une bouilloire de
\qty{2.2}{kW} et un grille-pain de \qty{1.0}{kW} tournent ensemble~: le

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:circuits-and-power:13}

\section{Problème~: Du barrage au grille-pain}

\begin{problem}[{Problème du week-end --- le voyage de mille kilomètres
d'un kilowatt-heure, de l'eau qui tombe jusqu'au pain qui dore, avec la
facture complète du trajet}]
\label{pb:g11:circuits-and-power:1}
Un barrage de montagne retient de l'eau à \qty{200}{m} au-dessus de ses
turbines, qui avalent \qty{50}{m^3} --- \qty{5.0e4}{kg} --- par seconde~;
turbine et alternateur convertissent avec un
\omterm{def:g10:energy-conservation:efficiency}{rendement} $0{,}90$. À travers une ligne de
\omterm{def:g11:circuits-and-power:resistance}{résistance} totale $R_\ell = \qty{32}{\ohm}$
(\qty{1000}{km} aller-retour) la centrale alimente une ville où votre
grille-pain de \qty{900}{W} attend sous \qty{230}{V}. L'alternateur
sort \qty{20}{kV}~; des transformateurs
(\omterm{def:g10:energy-conservation:efficiency}{rendement} $0{,}99$ chacun) convertissent
les \omterm{def:g11:circuits-and-power:voltage}{tensions} à volonté. L'électricité se vend
à $0{,}25$~euro par \unit{kW.h}~; $g = \qty{9.81}{N/kg}$.

\medskip
\noindent\textbf{Partie I --- Au barrage.}
\begin{enumerate}
  \item Quelle \omterm{def:g10:energy-conservation:energy}{énergie} potentielle l'eau qui
        tombe libère-t-elle chaque seconde~? En déduire la
        \omterm{def:g10:energy-conservation:power}{puissance} disponible pour les turbines.
  \item Quelle \omterm{prop:g11:circuits-and-power:power}{puissance électrique} quitte
        l'alternateur~?
  \item À \qty{20}{kV} directement depuis l'alternateur, quelle intensité
        circule~?
  \item Calculer la perte Joule $R_\ell I^2$ à cette intensité et la
        comparer à la sortie de la centrale. Conclure.
  \item La \omterm{def:g10:energy-conservation:power}{puissance} $P = UI$ de la ville est
        fixée par la demande. Quelle est la seule façon de diminuer $I$,
        et quel appareil la réalise~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Sur la ligne.}
\begin{enumerate}[resume]
  \item Un transformateur, traité comme sans pertes pour cette question,
        élève les \qty{88}{MW} à \qty{400}{kV}~: quelle intensité
        circule maintenant~?
  \item Calculer la nouvelle perte en ligne, en mégawatts.
  \item Quelle fraction de la sortie de la centrale est perdue en ligne~?
  \item Montrer que la fraction perdue vaut $R_\ell P/U^2$, et vérifier
        que cela explique le facteur entre les questions~4 et~7.
  \item Calculer la chute de \omterm{def:g11:circuits-and-power:voltage}{tension}
        $R_\ell I$ le long de la ligne à \qty{400}{kV}. Est-elle
        cohérente avec la question~8~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Dans la cuisine.}
\begin{enumerate}[resume]
  \item Convertir \qty{1}{kW.h} en
        \omterm{def:g10:energy-conservation:energy}{joules}~; que multiplie le compteur~?
  \item Calculer l'intensité et la
        \omterm{def:g11:circuits-and-power:resistance}{résistance} du grille-pain sous
        \qty{230}{V}.
  \item Trois minutes de toast~:
        l'\omterm{def:g10:energy-conservation:energy}{énergie} en
        \omterm{def:g10:energy-conservation:energy}{joules} et en
        \omterm{def:g10:energy-conservation:kwh}{kilowatt-heures}, et son prix.
  \item Le grille-pain dissipe $RI^2$~: expliquer en une phrase pourquoi
        la même formule est la perte sur la ligne, le produit dans la
        cuisine.
  \item Le grille-pain partage un
        \omterm{def:g11:circuits-and-power:joule}{fusible} de \qty{16}{A} avec une
        bouilloire de \qty{2.2}{kW}~: le
        \omterm{def:g11:circuits-and-power:joule}{fusible} tient-il~? Un radiateur de
        \qty{2.0}{kW} s'ajoute~: et maintenant~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- L'audit.}
\begin{enumerate}[resume]
  \item Le câblage propre de la maison,
        \omterm{def:g11:circuits-and-power:resistance}{résistance} \qty{0.05}{\ohm}, porte
        les \qty{13.5}{A} de la question~15~: calculer la perte et sa
        fraction des environ \qty{3.1}{kW} livrés.
  \item Enchaîner les rendements (turbine--alternateur, élévation, ligne,
        abaissement, câblage de maison) en un
        \omterm{def:g10:energy-conservation:efficiency}{rendement} global du réseau.
  \item Combien de \omterm{def:g10:orders-of-magnitude:unit}{kilogrammes} (et de mètres
        cubes) d'eau du barrage doivent tomber pour qu'\emph{un}
        \omterm{def:g10:energy-conservation:kwh}{kilowatt-heure} atteigne votre
        grille-pain~?
  \item Évaluer la fraction perdue $R_\ell P/U^2$ de la question~9 à
        $U = \qty{20}{kV}$. Que signifie une valeur supérieure à~$1$~?
  \item La chute --- \emph{le prix de la distance}~: en deux phrases,
        que coûte un \omterm{def:g10:energy-conservation:kwh}{kilowatt-heure} toasté à la
        montagne, quelle fraction meurt en route, et quel unique tour
        le rend possible~?
\end{enumerate}
\end{problem}
