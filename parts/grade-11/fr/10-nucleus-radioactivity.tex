\chapter{Le noyau et la radioactivité}\label{ch:g11:nucleus-radioactivity}

En ce moment même, quelque huit mille noyaux atomiques explosent dans
votre corps chaque seconde. Rien ne les attaque~: certains noyaux
naissent instables, et chacun tôt ou tard se transforme lui-même, en
projetant un fragment à une bonne fraction de la vitesse de la lumière.
Ce chapitre revient à l'étage des $10^{-15}$~mètres du
\cref{ch:g11:fundamental-interactions} pour voir quels noyaux sont
fragiles, comment ils se brisent, et comment leur impatience régulière
date les grottes et surveille les plafonds.

\section{Noyaux, nucléons et isotopes}

\begin{definition}[Notation nucléaire]\label{def:g11:nucleus-radioactivity:notation}
Un \omterm{def:g11:fundamental-interactions:ladder}{noyau} contient $Z$ protons
(charge $+e$ chacun) et $N$ neutrons (sans charge), $A = Z + N$
\omterm{def:g11:fundamental-interactions:ladder}{nucléons} en tout~; on l'écrit
${}^{A}_{Z}\mathrm{X}$, où $\mathrm X$ est le symbole chimique, $Z$ le
\emph{numéro atomique}\index{numéro atomique} et $A$ le \emph{nombre de
masse}\index{nombre de masse}. Une espèce spécifiée par $A$ et $Z$ est
un \emph{nucléide}\index{nucléide}~; des nucléides de même $Z$ mais de
$A$ différents sont des \emph{isotopes}\index{isotope} d'un même
élément~: même chimie (le nuage électronique ne voit que $Ze$), noyaux
et stabilité différents.
\end{definition}

\begin{example}[Hydrogène et carbone]\label{ex:g11:nucleus-radioactivity:hc}
L'hydrogène a trois \omterm{def:g11:nucleus-radioactivity:notation}{isotopes}~:
${}^{1}_{1}\mathrm{H}$ (un proton seul), ${}^{2}_{1}\mathrm{H}$
(deutérium~: un proton, un neutron), ${}^{3}_{1}\mathrm{H}$ instable
(tritium~: deux neutrons). Carbone naturel~: surtout
${}^{12}_{6}\mathrm{C}$ ($6$ protons, $6$ neutrons), $1{,}1\,\%$ de
${}^{13}_{6}\mathrm{C}$, et un \omterm{def:g11:fundamental-interactions:ladder}{atome}
sur $10^{12}$ de ${}^{14}_{6}\mathrm{C}$ instable ($8$ neutrons), héros
de l'\cref{ex:g11:nucleus-radioactivity:dating}.
\end{example}

\section{Noyaux stables et instables}

\begin{proposition}[La vallée de stabilité]\label{prop:g11:nucleus-radioactivity:valley}
Les \omterm{def:g11:nucleus-radioactivity:notation}{nucléides} stables occupent une
bande étroite dans le plan $(Z, N)$~: $N \approx Z$ pour les noyaux
légers, puis un excès de neutrons croissant, jusqu'à
$N \approx 1.5\,Z$. Tout \omterm{def:g11:nucleus-radioactivity:notation}{nucléide}
hors de la bande --- trop de neutrons, trop de protons, ou trop gros
(au-delà du plomb, $Z = 82$, aucun
\omterm{def:g11:nucleus-radioactivity:notation}{nucléide} n'est vraiment stable) --- se
transforme tôt ou tard.
\end{proposition}
\admitted

\begin{remark}
Localiser la bande exactement est de la mécanique quantique --- les
volumes universitaires. Le motif est le grand livre du
\cref{ch:g11:fundamental-interactions}~: la colle forte ne lie que les
voisins, la répulsion de Coulomb traverse tout le
\omterm{def:g11:fundamental-interactions:ladder}{noyau}, donc les noyaux lourds
ont besoin de colle neutronique supplémentaire~; et pourtant un excès de
neutrons est instable aussi (l'
\omterm{def:g11:fundamental-interactions:weak}{interaction faible} s'en charge).
\end{remark}

\begin{definition}[Radioactivité]\label{def:g11:nucleus-radioactivity:radioactivity}
La \emph{radioactivité}\index{radioactivité} est la transformation
spontanée d'un \omterm{def:g11:fundamental-interactions:ladder}{noyau} instable en un
autre \omterm{def:g11:nucleus-radioactivity:notation}{nucléide}, avec émission de
rayonnement. Elle est \emph{aléatoire}~: rien n'annonce quel
\omterm{def:g11:fundamental-interactions:ladder}{noyau} se désintègre ensuite, et aucune
\omterm{def:g10:pressure:pressure}{pression}, température ou chimie ne peut
la hâter ou la retarder~; seules les grandes populations sont
prévisibles (\cref{def:g11:nucleus-radioactivity:halflife}).
\end{definition}

\begin{remark}[Becquerel et les Curie]\label{rem:g11:nucleus-radioactivity:curie}
En 1896 Henri Becquerel trouva que des sels d'uranium voilent une plaque
photographique à travers du papier noir, dans un tiroir fermé, sans
\omterm{def:g10:energy-conservation:energy}{énergie} fournie. Marie et Pierre
Curie montrèrent que l'effet est atomique, le mesurèrent, et isolèrent
le polonium et le radium, un million de fois plus actifs. Le mot
\emph{\omterm{def:g11:nucleus-radioactivity:radioactivity}{radioactivité}} est de Marie
Curie~; deux prix Nobel aussi.
\end{remark}

\section{Alpha, bêta, gamma}

\begin{definition}[Les trois rayonnements historiques]\label{def:g11:nucleus-radioactivity:abg}
Trois rayonnements, nommés avant que quiconque sache ce qu'ils étaient~:
\begin{itemize}
  \item \emph{alpha}\index{rayonnement alpha} ($\alpha$)~: un
        \omterm{def:g11:fundamental-interactions:ladder}{noyau} d'hélium
        ${}^{4}_{2}\mathrm{He}$ éjecté entier --- lourd, doublement
        chargé, arrêté par une feuille de papier ou des centimètres
        d'air~;
  \item \emph{bêta}\index{rayonnement bêta} ($\beta^-$)~: un électron
        ${}^{\;0}_{-1}\mathrm{e}$ créé dans le
        \omterm{def:g11:fundamental-interactions:ladder}{noyau} et éjecté --- léger,
        rapide, arrêté par des millimètres d'aluminium~;
  \item \emph{gamma}\index{rayonnement gamma} ($\gamma$)~: rayonnement
        électromagnétique, bien plus énergétique que la lumière ---
        jamais tout à fait arrêté, seulement
        \emph{atténué}\index{atténuation}~: un centimètre de plomb
        absorbe la moitié.
\end{itemize}
\end{definition}

\begin{proposition}[Conservation dans les équations nucléaires]\label{prop:g11:nucleus-radioactivity:soddy}
Dans toute transformation nucléaire, le
\omterm{def:g11:nucleus-radioactivity:notation}{nombre de masse} total $A$ et la
charge totale --- la somme des indices inférieurs, en comptant $-1$ pour
un électron émis --- sont les mêmes avant et après.
\end{proposition}
\admitted

\begin{remark}
La conservation de la charge régnait déjà dans le
\cref{ch:g11:fundamental-interactions}~; la conservation de $A$ dit que
les \omterm{def:g11:fundamental-interactions:ladder}{nucléons} ne sont que
convertis, jamais créés ni détruits --- comptabilité honnête pour les
volumes universitaires.
\end{remark}

\begin{method}[Équilibrer une équation de désintégration]\label{met:g11:nucleus-radioactivity:balance}
\begin{enumerate}
  \item Écrire chaque acteur avec les deux étiquettes~:
        $\alpha = {}^{4}_{2}\mathrm{He}$,
        $\beta^- = {}^{\;0}_{-1}\mathrm{e}$,
        $\beta^+ = {}^{0}_{+1}\mathrm{e}$~; un $\gamma$ porte
        $A = Z = 0$.
  \item Égaler les sommes des $A$, puis des $Z$
        (\cref{prop:g11:nucleus-radioactivity:soddy})~; résoudre, et lire
        l'élément fils sur son $Z$ dans le
        \omterm{def:g10:signals-and-waves:period}{tableau périodique}.
\end{enumerate}
\end{method}

\begin{example}[Une désintégration alpha]\label{ex:g11:nucleus-radioactivity:alpha}
Le radium-226, le radium des Curie, est un émetteur $\alpha$~:
${}^{226}_{88}\mathrm{Ra} \to {}^{A}_{Z}\mathrm{Y} + {}^{4}_{2}\mathrm{He}$
\omterm{def:g10:inertia:force}{impose} $A = 226 - 4 = 222$ et
$Z = 88 - 2 = 86$~; l'élément $86$ est le radon, donc
${}^{226}_{88}\mathrm{Ra} \to {}^{222}_{86}\mathrm{Rn} + {}^{4}_{2}\mathrm{He}$
--- le gaz radioactif des caves mal ventilées.
\end{example}

\begin{example}[Une désintégration bêta-moins]\label{ex:g11:nucleus-radioactivity:beta}
Le carbone-14 siège au-dessus de la bande ($8$ neutrons contre $6$
protons)~:
${}^{14}_{6}\mathrm{C} \to {}^{14}_{7}\mathrm{N} + {}^{\;0}_{-1}\mathrm{e}$.
Vérification~: $14 = 14 + 0$, $6 = 7 - 1$~; un neutron est devenu un
proton plus l'électron éjecté --- l'
\omterm{def:g11:fundamental-interactions:weak}{interaction faible} au
\omterm{def:g11:work-of-force:work}{travail}.
\end{example}

\begin{remark}[Bêta-plus et gamma]\label{rem:g11:nucleus-radioactivity:betaplus}
Sous la bande, la désintégration miroir $\beta^+$ transforme un proton
en neutron plus un \emph{positon}\index{positon}
${}^{0}_{+1}\mathrm{e}$, l'antiparticule de l'électron~:
${}^{18}_{9}\mathrm{F} \to {}^{18}_{8}\mathrm{O} + {}^{0}_{+1}\mathrm{e}$,
cheval de trait des scanners TEP. Et après la plupart des
désintégrations le noyau fils, né en vibration, se calme en émettant un
$\gamma$ par-dessus.
\end{remark}

\section{Demi-vie et activité}

\begin{definition}[Demi-vie]\label{def:g11:nucleus-radioactivity:halflife}
La \emph{demi-vie}\index{demi-vie} $T_{1/2}$ d'un
\omterm{def:g11:nucleus-radioactivity:notation}{nucléide} est le temps au bout
duquel la moitié de tout grand échantillon s'est désintégrée. Chaque
demi-vie supplémentaire divise par deux ce qui reste~: sur $N_0$
noyaux, $N_0/2$ restent à $T_{1/2}$, $N_0/4$ à $2\,T_{1/2}$, $N_0/8$
à $3\,T_{1/2}$, $N_0/2^{\,n}$ après $n$ demi-vies.
\end{definition}

\begin{remark}
Aléatoire pour un \omterm{def:g11:fundamental-interactions:ladder}{noyau},
mécanique pour $10^{23}$~: une foule immense de joueurs de pile ou
face, imprévisibles un par un, la moitié éliminée à chaque tour.
Interpoler \emph{entre} des demi-vies entières prend la fonction
exponentielle --- l'année suivante, avec le volume de mathématiques.
\end{remark}

\begin{example}[Vingt puissances de dix]\label{ex:g11:nucleus-radioactivity:range}
\begin{center}\small\begin{tabular}{llll}
polonium-214 & $\alpha$ & \qty{1.6e-4}{s} & maillon de la chaîne du radium \\
technétium-99m & $\gamma$ & $6{,}0$ heures & imagerie médicale \\
américium-241 & $\alpha$ & $432$ ans & détecteurs de fumée \\
carbone-14 & $\beta^-$ & \num{5700} ans & archéologie \\
potassium-40 & $\beta$ & \num{1.25e9} ans & dans chaque banane \\
uranium-238 & $\alpha$ & \num{4.5e9} ans & chaleur intérieure de la Terre \\
\end{tabular}\end{center}
\end{example}

\begin{example}[Datation au carbone-14]\label{ex:g11:nucleus-radioactivity:dating}
La matière vivante échange du carbone avec l'atmosphère, donc elle
détient la proportion atmosphérique de carbone-14~; à la mort l'apport
s'arrête et la proportion se divise par deux tous les \num{5700}~ans.
Du charbon d'une grotte peinte montre un quart de la proportion
vivante~: un quart est la moitié d'une moitié, donc le feu a brûlé
deux demi-vies, $2 \times \num{5700} = \num{11400}$~ans, auparavant.
\end{example}

\begin{definition}[Activité]\label{def:g11:nucleus-radioactivity:activity}
L'\emph{activité}\index{activité} $\mathcal A$ d'un échantillon est
son nombre de désintégrations par seconde, mesurée en
\emph{becquerels}\index{becquerel}~: $\qty{1}{Bq} = 1$
désintégration par seconde. Moins de noyaux survivants signifie moins
de désintégrations, donc l'activité se divise aussi par deux à chaque
\omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie}.
\end{definition}

\begin{example}[Ordres de grandeur]\label{ex:g11:nucleus-radioactivity:orders}
Tout est légèrement radioactif. Une banane~: environ \qty{15}{Bq}
(potassium-40). Un corps humain~: environ \qty{8000}{Bq} --- le crochet
de ce chapitre, quelque cinq cents bananes. Un mètre cube de granite~:
autour de \qty{3e6}{Bq} (uranium et sa chaîne, d'où le radon des
caves). Une injection de technétium~: environ \qty{5e8}{Bq}, disparue
en quelques jours.
\end{example}

\section{Serviteurs et dangers}

\begin{example}[Trois carrières d'un noyau instable]\label{ex:g11:nucleus-radioactivity:uses}
\emph{Médecine}~: le technétium-99m, émetteur $\gamma$ pur de
$T_{1/2} = 6$~heures, est fixé à une
\omterm{def:g11:fundamental-interactions:ladder}{molécule} que l'organe cible
absorbe~; une caméra filme les $\gamma$ de l'extérieur, et dès le
lendemain le traceur a pour l'essentiel disparu --- tandis que la
radiothérapie inverse la logique, en focalisant des faisceaux pour
détruire une tumeur. \emph{Datation}~: carbone-14 pour le bois, l'os
et le tissu jusqu'à quelque \num{50000}~ans~; uranium-238 pour les
roches --- et la Terre, à \num{4.5e9}~ans. \emph{Détecteurs de
fumée}~: une pincée d'américium-241 ionise l'air d'une petite chambre,
laissant passer un petit courant~; la fumée étouffe le courant et
l'alarme sonne --- les $\alpha$ meurent en centimètres d'air.
\end{example}

\begin{remark}[Dose et sievert]\label{rem:g11:nucleus-radioactivity:dose}
Le rayonnement arrache des électrons aux
\omterm{def:g11:fundamental-interactions:ladder}{molécules} et peut endommager
l'ADN. Le dommage biologique est suivi par la
\emph{dose}\index{dose}, en \emph{sieverts}\index{sievert} (\unit{Sv}),
qui pèse l'\omterm{def:g10:energy-conservation:energy}{énergie} déposée par
\omterm{def:g10:orders-of-magnitude:unit}{kilogramme} de tissu selon la nocivité
de chaque rayonnement. Trois règles~: distance, blindage ($\alpha$~: la
peau suffit --- sauf si l'émetteur est \emph{inhalé}, le crime du
radon~; $\beta$~: aluminium~; $\gamma$~: plomb), et temps. La dosimétrie
proprement dite est matière universitaire.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:1}
Donner le nombre de protons et de neutrons dans
${}^{16}_{8}\mathrm{O}$, ${}^{27}_{13}\mathrm{Al}$,
${}^{60}_{27}\mathrm{Co}$, ${}^{235}_{92}\mathrm{U}$ et
${}^{3}_{1}\mathrm{H}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:2}
Parmi ${}^{14}_{6}\mathrm{C}$, ${}^{14}_{7}\mathrm{N}$,
${}^{12}_{6}\mathrm{C}$, ${}^{13}_{6}\mathrm{C}$ et
${}^{16}_{8}\mathrm{O}$, lesquels sont des
\omterm{def:g11:nucleus-radioactivity:notation}{isotopes} les uns des autres~?
Pourquoi ${}^{14}_{6}\mathrm{C}$ et ${}^{14}_{7}\mathrm{N}$ ne sont-ils
\emph{pas} des \omterm{def:g11:nucleus-radioactivity:notation}{isotopes}, malgré des
\omterm{def:g11:nucleus-radioactivity:notation}{nombres de masse} égaux~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:3}
Le polonium-210, ${}^{210}_{84}\mathrm{Po}$, est un émetteur $\alpha$.
Écrire l'équation de désintégration et nommer le noyau fils ($Z = 81$~:
thallium, $82$~: plomb, $83$~: bismuth).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:4}
Écrire les équations de désintégration $\beta^-$ du cobalt-60
(${}^{60}_{27}\mathrm{Co}$) et de l'iode-131
(${}^{131}_{53}\mathrm{I}$)~; noyaux fils~: $Z = 28$ nickel, $Z = 54$
xénon.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:nucleus-radioactivity:5}
Une source d'iode-131 ($T_{1/2} = 8{,}0$~jours) a une
\omterm{def:g11:nucleus-radioactivity:activity}{activité} de \qty{800}{MBq}
aujourd'hui. Quelle est son
\omterm{def:g11:nucleus-radioactivity:activity}{activité} après $16$~jours~? Après
$32$~jours~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:6}
Quatre rayonnements traversent un fort
\omterm{def:g11:magnetic-fields:bfield}{champ magnétique}, puis des écrans.
Identifier chacun~: (a)~arrêté par le papier, à peine dévié par le
\omterm{def:g11:electric-gravitational-fields:field}{champ}~; (b)~traverse le papier,
arrêté par \qty{4}{mm} d'aluminium, fortement dévié~; (c)~traverse les
deux, seulement \omterm{def:g11:nucleus-radioactivity:abg}{atténué} par le plomb,
non dévié~; (d)~comme (b), dévié de l'autre côté.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:7}
Le radon-222 (${}^{222}_{86}\mathrm{Rn}$) ouvre une chaîne rapide~: une
désintégration $\alpha$, puis une autre $\alpha$, puis une $\beta^-$.
Écrire les trois équations ($Z = 82$~: plomb, $83$~: bismuth, $84$~:
polonium).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:8}
Le fluor-18, ${}^{18}_{9}\mathrm{F}$, est le traceur des scanners TEP~;
le fluor stable est ${}^{19}_{9}\mathrm{F}$. De quel côté de la bande
de stabilité est-il~? Prédire son mode de désintégration et écrire
l'équation ($Z = 8$~: oxygène).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:9}
Un patient reçoit \qty{500}{MBq} de technétium-99m
($T_{1/2} = 6{,}0$~heures) à 08~h. Quelle est l'
\omterm{def:g11:nucleus-radioactivity:activity}{activité} à 08~h le lendemain matin~?
Après combien d'heures descend-elle pour la première fois sous
\qty{1}{MBq}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:10}
Un os d'une tourbière montre une proportion de carbone-14 d'un huitième
de celle d'un os vivant ($T_{1/2} = \num{5700}$~ans). Quel âge a l'os~?
Et pourquoi le carbone-14 est-il inutile pour dater des os de
dinosaures (âge d'environ $10^{8}$~ans)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:nucleus-radioactivity:11}
En utilisant la vallée de stabilité, prédire le mode de désintégration
($\alpha$, $\beta^-$ ou $\beta^+$) de ${}^{14}_{6}\mathrm{C}$,
${}^{11}_{6}\mathrm{C}$ et ${}^{238}_{92}\mathrm{U}$, en justifiant
chacun en une ligne (carbone stable~: ${}^{12}_{6}\mathrm{C}$,
${}^{13}_{6}\mathrm{C}$).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:12}
La chaîne de l'uranium-238 se termine, bien des étapes plus tard, au
${}^{206}_{82}\mathrm{Pb}$ stable~; chaque étape est une $\alpha$ ou
une $\beta^-$. En utilisant la conservation de $A$ seule, trouver le
nombre d'étapes $\alpha$~; puis, avec la conservation de $Z$, le nombre
d'étapes $\beta^-$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:13}
La source de cobalt-60 d'un hôpital ($T_{1/2} = 5{,}3$~ans) a une
\omterm{def:g11:nucleus-radioactivity:activity}{activité} de \qty{6.4}{TBq}~; la
réglementation autorise l'élimination sous \qty{0.1}{TBq}. Après combien
d'années peut-elle être éliminée~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:14}
L'\omterm{def:g11:nucleus-radioactivity:activity}{activité} de votre corps est
d'environ \qty{8000}{Bq}. Combien de vos noyaux se désintègrent par
jour~? Sur une vie de $80$~ans~? Une banane ajoute environ \qty{15}{Bq}
pendant que vous la digérez~: commenter, en une phrase, les titres qui
craignent chaque
\omterm{def:g11:nucleus-radioactivity:activity}{becquerel}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:nucleus-radioactivity:15}
Un détecteur de fumée détient \qty{0.2}{\micro\gram} d'américium-241
($T_{1/2} = 432$~ans), d'
\omterm{def:g11:nucleus-radioactivity:activity}{activité} d'environ \qty{25}{kBq}.
Combien de désintégrations est-ce par jour~? Après dix ans de service,
l'américium s'est-il désintégré de beaucoup moins de la moitié,
d'environ la moitié, ou de beaucoup plus --- et l'épuisement de la
source est-il la raison pour laquelle on remplace les détecteurs~?
Enfin, pourquoi cette source $\alpha$ est-elle inoffensive au plafond,
mais dangereuse si la capsule est broyée et la poussière inhalée~?
\end{exercise}

\section{Problème~: La momie, le réacteur et le détecteur de fumée}

\begin{problem}[{Problème du week-end --- trois carrières d'un noyau
instable~: une momie datée par divisions par deux, l'arbre généalogique
de l'uranium équilibré, un détecteur de fumée audité, et pourquoi les
étoiles ont dû être mortelles}]
\label{pb:g11:nucleus-radioactivity:1}
Trois noyaux instables, trois métiers~: le carbone-14 horodate chaque
plante et animal morts, l'uranium-238 chauffe une planète de l'intérieur,
l'américium-241 surveille les plafonds. Données~:
$T_{1/2} = \num{5700}$~ans (C-14), \num{4.5e9}~ans (U-238), $432$~ans
(Am-241)~; le carbone vivant montre $13{,}6$ désintégrations par minute
par gramme~; \omterm{def:g11:nucleus-radioactivity:notation}{numéros atomiques}~:
N $7$, O $8$, Pb $82$, Bi $83$, Po $84$, Rn $86$, Ra $88$, Th $90$, Pa
$91$, U $92$, Np $93$, Am $95$.

\medskip
\noindent\textbf{Partie I --- La momie.}
\begin{enumerate}
  \item Donner la composition (protons, neutrons) de
        ${}^{12}_{6}\mathrm{C}$ et ${}^{14}_{6}\mathrm{C}$. Pourquoi une
        chimie identique dans un corps~?
  \item Pourquoi la proportion de carbone-14 est-elle constante dans un
        corps vivant, et pourquoi commence-t-elle à chuter à la mort~?
  \item Écrire l'équation de désintégration du carbone-14. Son
        rayonnement s'échapperait-il à travers les bandelettes de lin de
        la momie~?
  \item Un gramme de carbone du lin montre $6{,}8$ désintégrations par
        minute. Quelle fraction du taux vivant est-ce~? Dater la momie.
  \item Une seconde «~momie~», prix d'une collection privée, montre
        $13{,}2$ désintégrations par minute par gramme. Verdict~?
  \item Estimer le taux d'un échantillon de dix demi-vies, et expliquer
        pourquoi la datation au carbone s'estompe au-delà d'environ
        \num{57000}~ans.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le réacteur sous le pré.}
\begin{enumerate}[resume]
  \item Donner la composition de ${}^{238}_{92}\mathrm{U}$. Pourquoi les
        noyaux lourds ont-ils besoin d'un tel surplus de neutrons~?
  \item Écrire son équation de désintégration $\alpha$.
  \item Le noyau fils se désintègre ensuite $\beta^-$, et la
        petite-fille $\beta^-$ encore. Écrire les deux équations. Quel
        élément réapparaît~?
  \item La chaîne se termine à ${}^{206}_{82}\mathrm{Pb}$. Compter ses
        étapes $\alpha$, puis ses étapes $\beta^-$.
  \item La Terre s'est formée il y a \num{4.5e9}~ans. Quelle fraction de
        son uranium-238 primordial reste aujourd'hui~? Et dans
        $5$~milliards d'années de plus~?
  \item En une phrase~: que \omterm{def:g11:work-of-force:power}{alimente} cette
        chaleur de désintégration enfouie à la surface~? (Penser
        volcans et continents dérivants.)
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le détecteur de fumée.}
\begin{enumerate}[resume]
  \item Écrire l'équation de désintégration $\alpha$ de
        ${}^{241}_{95}\mathrm{Am}$.
  \item Son \omterm{def:g11:nucleus-radioactivity:activity}{activité} est
        \qty{33}{kBq}~: combien de désintégrations par seconde~? Par
        jour~?
  \item Expliquer le détecteur~: que font les particules $\alpha$ à
        l'air de la chambre, et que change la fumée~?
  \item Pourquoi un émetteur $\alpha$ est-il le bon choix ici --- et un
        émetteur $\gamma$ d'
        \omterm{def:g11:nucleus-radioactivity:activity}{activité} égale le pire~?
  \item Après une \omterm{def:g11:nucleus-radioactivity:halflife}{demi-vie}
        ($432$~ans~!) l'\omterm{def:g11:nucleus-radioactivity:activity}{activité}
        serait \qty{16.5}{kBq}~; le manuel demande un remplacement après
        dix ans. La source est-elle le maillon faible~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Pourquoi les étoiles ont dû être mortelles.}
\begin{enumerate}[resume]
  \item Les étoiles mourantes ont forgé et dispersé chaque élément plus
        lourd que l'hélium. Quelle fraction de l'uranium-238 forgé il y
        a \num{9e9}~ans survit aujourd'hui~? Pourquoi l'abondance
        restante de la Terre rend-elle la planète bien plus jeune que
        les plus vieilles étoiles~?
  \item La chaleur radioactive intérieure de la Terre pilote le
        volcanisme et la tectonique des plaques, et maintient le noyau
        en brassage produisant le bouclier magnétique du
        \cref{ch:g11:magnetic-fields}. À quoi ressemblerait une planète
        de seuls \omterm{def:g11:fundamental-interactions:ladder}{atomes}
        stables~?
  \item Une ligne, comme poésie et comme comptabilité~: où vos
        \omterm{def:g11:fundamental-interactions:ladder}{atomes} ont-ils été
        forgés, et qu'est-ce qui chauffe encore le sol sous vos pieds~?
\end{enumerate}
\end{problem}
