\chapter{Magnétisme et champs magnétiques}\label{ch:g11:magnetic-fields}

Les marins se guidaient à des aiguilles aimantées pendant mille ans
avant que quiconque sache pourquoi elles pointent le nord. La réponse
est venue en 1820, lorsqu'un courant fit tiquer une boussole~: le
magnétisme est fait de charges en mouvement. Ce chapitre cartographie
le \omterm{def:g11:magnetic-fields:bfield}{champ magnétique} --- des
\omterm{def:g11:magnetic-fields:magnet}{aimants}, de la Terre, des courants --- puis
ferme la boucle~: le \omterm{def:g11:electric-gravitational-fields:field}{champ}
repousse les courants, ce qui fait tourner tout moteur électrique.

\section{Aimants et champ magnétique}

\begin{definition}[Aimant et pôles]\label{def:g11:magnetic-fields:magnet}
Un \emph{aimant}\index{aimant} attire le fer et, suspendu par un fil,
s'oriente vers le nord. Son action se concentre en deux \emph{pôles
magnétiques}\index{pôle magnétique}, \emph{nord} (l'extrémité qui
cherche le nord géographique) et \emph{sud}~: les pôles de même nom se
repoussent, de noms opposés s'attirent. Scier un aimant en deux donne
deux aimants complets --- aucune expérience n'a jamais isolé un pôle.
\end{definition}

\begin{definition}[Champ magnétique]\label{def:g11:magnetic-fields:bfield}
Un \omterm{def:g11:magnetic-fields:magnet}{aimant} modifie l'espace autour de
lui (\cref{ch:g11:electric-gravitational-fields})~: il crée partout un
\emph{champ magnétique}\index{champ magnétique} $\vect B$. Une boussole
lit sa direction (l'aiguille s'aligne sur $\vect B$, extrémité nord en
avant)~; une \emph{sonde de Hall}\index{sonde de Hall} mesure sa
grandeur, en \emph{teslas}\index{tesla} (\unit{T}).
\end{definition}

\begin{definition}[Lignes de champ magnétique]\label{def:g11:magnetic-fields:lines}
Les \emph{lignes de champ magnétique}\index{ligne de champ magnétique}
sont des courbes tangentes à $\vect B$, se resserrant là où le
\omterm{def:g11:electric-gravitational-fields:field}{champ} est fort. Hors d'un
\omterm{def:g11:magnetic-fields:magnet}{aimant} elles vont du pôle nord au pôle
sud~; contrairement aux lignes de
\omterm{def:g11:electric-gravitational-fields:efield}{champ électrique} elles n'ont
pas d'extrémités~: chacune se ferme à travers le corps de l'
\omterm{def:g11:magnetic-fields:magnet}{aimant}.
\end{definition}

\section{La Terre est un aimant}

\begin{definition}[Le champ magnétique terrestre]\label{def:g11:magnetic-fields:earth}
Des courants dans le noyau de fer fondu de la Terre font de la planète
un \omterm{def:g11:magnetic-fields:magnet}{aimant} géant~: en surface
$B \approx \qty{50}{\micro T}$, et une aiguille libre s'aligne sur la
composante horizontale du \omterm{def:g11:electric-gravitational-fields:field}{champ} ---
c'est la boussole. Les pôles géomagnétiques ne sont pas les
géographiques~; l'angle entre le nord de la boussole et le nord vrai
est la \emph{déclinaison}\index{déclinaison}, corrigée sur chaque
relèvement.
\end{definition}

\begin{example}[Corriger un relèvement]\label{ex:g11:magnetic-fields:bearing}
Là où la \omterm{def:g11:magnetic-fields:earth}{déclinaison} est de $\ang{7}$ ouest,
un navire gouvernant «~nord boussole~» sur \qty{5.0}{km} dérive de
$5000 \times \sin\ang{7} \approx \qty{6.1e2}{m}$ à l'ouest de sa
route~: pour naviguer plein nord vrai, gouverner $\ang{7}$ \emph{est}
de l'aiguille.
\end{example}

\section{Le champ magnétique d'un courant}

\begin{remark}[Ørsted, 1820]\label{rem:g11:magnetic-fields:oersted}
Pendant un cours en 1820, Hans Christian Ørsted vit une aiguille de
boussole basculer lorsqu'il ferma un circuit à proximité~: électricité
et magnétisme n'étaient qu'un seul sujet --- \emph{les courants créent
des \omterm{def:g11:magnetic-fields:bfield}{champs magnétiques}}.
\end{remark}

\begin{proposition}[Champ d'un fil rectiligne]\label{prop:g11:magnetic-fields:wire}
Un long fil rectiligne parcouru par une intensité $I$ crée un
\omterm{def:g11:electric-gravitational-fields:field}{champ} dont les lignes sont des
cercles centrés sur le fil, dans des plans perpendiculaires à lui. À la
distance $d$ du fil,
\[
B = \frac{\mu_0\, I}{2\pi d},
\qquad \mu_0 = 4\pi\times 10^{-7}\ \unit{T.m/A} \approx \qty{1.26e-6}{T.m/A}.
\]
\end{proposition}
\admitted

\begin{example}[Un fil est un aimant faible]\label{ex:g11:magnetic-fields:wire}
À $d = \qty{1.0}{cm}$ d'un fil portant $I = \qty{15}{A}$~:
$B = \num{2e-7} \times 15 / 0.010 = \qty{0.30}{mT}$ --- six
\omterm{def:g11:electric-gravitational-fields:field}{champs} terrestres, pour un courant
qui ferait sauter un disjoncteur domestique. De forts
\omterm{def:g11:electric-gravitational-fields:field}{champs} exigent une meilleure
géométrie.
\end{example}

\begin{definition}[Bobine et solénoïde]\label{def:g11:magnetic-fields:solenoid}
Enrouler le fil concentre son
\omterm{def:g11:electric-gravitational-fields:field}{champ}. Une \emph{bobine
plate}\index{bobine} ($N$ spires en disque) se comporte comme un
\omterm{def:g11:magnetic-fields:magnet}{aimant} \omterm{def:g11:lenses-and-eye:lens}{mince}, une face
nord, l'autre sud~; un \emph{solénoïde}\index{solénoïde} --- un long
enroulement cylindrique, $n$ spires par
\omterm{def:g10:orders-of-magnitude:unit}{mètre} --- comme un
\omterm{def:g11:magnetic-fields:magnet}{aimant} droit.
\end{definition}

\begin{proposition}[Champ à l'intérieur d'un solénoïde]\label{prop:g11:magnetic-fields:solenoidfield}
À l'intérieur d'un long \omterm{def:g11:magnetic-fields:solenoid}{solénoïde}, loin des
extrémités, le \omterm{def:g11:electric-gravitational-fields:field}{champ} est
\emph{uniforme}~: parallèle à l'axe et le même en tout point
intérieur, de grandeur $B = \mu_0\, n\, I$, indépendante de la largeur
du tube. À l'extérieur, le
\omterm{def:g11:electric-gravitational-fields:field}{champ} est faible.
\end{proposition}
\admitted

\begin{remark}
Les deux formules sont dérivées honnêtement dans le volume de
l'année~1, à partir de la loi reliant la circulation d'un
\omterm{def:g11:electric-gravitational-fields:field}{champ} au courant qu'elle enserre.
Noter ce qui compte~: $n$, spires \emph{par
\omterm{def:g10:orders-of-magnitude:unit}{mètre}} --- enrouler plus serré, non plus
long.
\end{remark}

\begin{method}[La main droite, deux fois]\label{met:g11:magnetic-fields:righthand}
\begin{itemize}
  \item \emph{Fil}~: pouce $=$ courant~; doigts enroulés $=$ sens de
        rotation des lignes.
  \item \emph{\omterm{def:g11:magnetic-fields:solenoid}{Bobine},
        \omterm{def:g11:magnetic-fields:solenoid}{solénoïde}}~: doigts enroulés $=$
        courant dans les spires~; pouce $=$ $\vect B$ à l'intérieur,
        pointant vers la face nord.
\end{itemize}
\end{method}

\begin{example}[Nombres de solénoïde]\label{ex:g11:magnetic-fields:solenoid}
À $10$ spires par centimètre ($n = \qty{1000}{m^{-1}}$) et

\qty{6.3}{mT}$ --- cent \omterm{def:g11:electric-gravitational-fields:field}{champs}
terrestres, réglables d'un bouton.
\end{example}

\section{Électroaimants}

\begin{definition}[Électroaimant]\label{def:g11:magnetic-fields:electromagnet}
Un \emph{électroaimant}\index{électroaimant} est un
\omterm{def:g11:magnetic-fields:solenoid}{solénoïde} enroulé sur un noyau de fer doux~:
le fer s'aimante dans le \omterm{def:g11:electric-gravitational-fields:field}{champ} de la
\omterm{def:g11:magnetic-fields:solenoid}{bobine} et le multiplie, typiquement par $100$ à
$1000$. Contrairement à un
\omterm{def:g11:magnetic-fields:magnet}{aimant} permanent, il s'éteint avec le courant.
\end{definition}

\begin{remark}[Trois machines]\label{rem:g11:magnetic-fields:machines}
La grue de casse lève une voiture avec un
\omterm{def:g11:magnetic-fields:electromagnet}{électroaimant} et --- tout l'intérêt --- la
\emph{lâche} en ouvrant un interrupteur. Le petit
\omterm{def:g11:magnetic-fields:electromagnet}{électroaimant} d'un relais tire un contact
fermé~: un faible courant commute un fort
(\cref{ch:g11:circuits-and-power}). Un scanner IRM maintient un
\qty{1.5}{T} uniforme autour du patient avec un
\omterm{def:g11:magnetic-fields:solenoid}{solénoïde} supraconducteur.
\end{remark}

\section{La force de Laplace}

\begin{proposition}[Force de Laplace]\label{prop:g11:magnetic-fields:laplace}
Un fil rectiligne de longueur $L$, parcouru par une intensité $I$ dans
un \omterm{def:g11:electric-gravitational-fields:capacitor}{champ uniforme} $\vect B$
\emph{perpendiculaire} au fil, ressent la \emph{force de
Laplace}\index{force de Laplace} de grandeur $F = I\,L\,B$,
perpendiculaire au fil et à $\vect B$~: main droite à plat, pouce dans
le sens du courant, doigts tendus le long de $\vect B$ --- la paume
pousse dans le sens de $\vect F$. Un fil \emph{parallèle} au
\omterm{def:g11:electric-gravitational-fields:field}{champ} ne ressent aucune
\omterm{def:g10:inertia:force}{force}.
\end{proposition}
\admitted

\begin{remark}[Le rail sauteur]\label{rem:g11:magnetic-fields:rail}
Le volume de l'année~1 dérive cela de la
\omterm{def:g10:inertia:force}{force} magnétique sur chaque charge en
mouvement. Au laboratoire~: une tige de cuivre repose sur deux rails
horizontaux entre les pôles d'un
\omterm{def:g11:magnetic-fields:magnet}{aimant}
(\omterm{def:g11:electric-gravitational-fields:field}{champ} entrant dans la page sur la
figure)~; fermez l'interrupteur, et la tige file le long des rails.
Inversez le courant, ou retournez l'
\omterm{def:g11:magnetic-fields:magnet}{aimant}~: elle file de l'autre côté.
\end{remark}

\begin{example}[Nombres de Laplace]\label{ex:g11:magnetic-fields:laplace}
Une tige de longueur $L = \qty{5.0}{cm}$ portant $I = \qty{8.0}{A}$
dans $B = \qty{0.50}{T}$~: $F = 8.0 \times 0.050 \times 0.50 =
\qty{0.20}{N}$ --- le \omterm{def:g10:universal-gravitation:weight}{poids} de
\qty{20}{g}, sur un fil de quelques grammes.
\end{example}

\begin{remark}[Le moteur à courant continu]\label{rem:g11:magnetic-fields:motor}
Pliez le fil en une boucle dans le
\omterm{def:g11:electric-gravitational-fields:field}{champ}~: les deux côtés
perpendiculaires à $\vect B$ portent des courants opposés, donc leurs
\omterm{prop:g11:magnetic-fields:laplace}{forces de Laplace} sont opposées --- un couple
qui fait tourner la boucle. Un demi-tour plus tard le couple la
ferait revenir, donc un commutateur tournant (le
\emph{collecteur}\index{collecteur}) inverse le courant à chaque
demi-tour. Chaque ventilateur et chaque perceuse est cette boucle,
multipliée.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:1}
Un \omterm{def:g11:magnetic-fields:magnet}{aimant} droit est scié en deux entre ses
pôles. Quels pôles porte chaque morceau~? Remis ensemble, les faces
fraîchement coupées s'attirent-elles ou se repoussent-elles~? Un
schéma de découpe peut-il isoler le pôle nord~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:2}
Vrai ou faux, avec une raison chacune~: (a)~deux
\omterm{def:g11:magnetic-fields:lines}{lignes de champ magnétique} se croisent là où le
\omterm{def:g11:electric-gravitational-fields:field}{champ} est fort~; (b)~hors d'un
\omterm{def:g11:magnetic-fields:magnet}{aimant}, les
\omterm{def:g11:electric-gravitational-fields:lines}{lignes de champ} vont du pôle nord au
pôle sud~; (c)~chaque
\omterm{def:g11:magnetic-fields:lines}{ligne de champ magnétique} est une boucle fermée~;
(d)~une aiguille se place perpendiculairement à la ligne qui la
traverse.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:3}
Calculer le \omterm{def:g11:electric-gravitational-fields:field}{champ} à \qty{2.0}{cm} d'un
fil rectiligne portant \qty{10}{A}. Combien de fois le
\qty{50}{\micro T} terrestre est-ce~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:4}
Un \omterm{def:g11:magnetic-fields:solenoid}{solénoïde} de $800$ spires enroulées sur
\qty{40}{cm} porte $I = \qty{1.5}{A}$. Calculer $n$, puis $B$ à
l'intérieur~; que devient $B$ si le courant double~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:magnetic-fields:5}
Un segment de fil de longueur \qty{10}{cm} porte \qty{5.0}{A}
perpendiculairement à un \omterm{def:g11:electric-gravitational-fields:field}{champ} de
\qty{0.20}{T}. Calculer la \omterm{prop:g11:magnetic-fields:laplace}{force de Laplace}~;
le \omterm{def:g10:universal-gravitation:weight}{poids} de quelle masse l'égale-t-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:6}
Là où une randonneuse se trouve, la
\omterm{def:g11:magnetic-fields:earth}{déclinaison} est de $\ang{8}$ ouest. Elle
marche \qty{2.0}{km} «~nord à la boussole~»~: de combien, et de quel
côté du nord vrai, dérive-t-elle~? Quel cap l'aurait menée plein nord
vrai~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:7}
À quelle distance d'un fil rectiligne portant \qty{20}{A} son
\omterm{def:g11:electric-gravitational-fields:field}{champ} égale-t-il le
\qty{50}{\micro T} terrestre~? Que dit cela de la confiance en une
boussole près de câbles sous tension~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:8}
Concevoir un \omterm{def:g11:magnetic-fields:solenoid}{solénoïde} produisant
$B = \qty{10}{mT}$ avec $I = \qty{4.0}{A}$~: calculer le $n$ requis,
puis le nombre de spires pour une
\omterm{def:g11:magnetic-fields:solenoid}{bobine} de \qty{25}{cm}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:9}
Dans l'expérience des rails, la tige (masse \qty{20}{g}, longueur
$L = \qty{12}{cm}$ entre les rails) porte \qty{10}{A} dans un
\omterm{def:g11:electric-gravitational-fields:field}{champ} perpendiculaire de
\qty{0.50}{T}. Calculer la
\omterm{prop:g11:magnetic-fields:laplace}{force de Laplace}, puis l'accélération
initiale de la tige~; comparer avec $g$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:10}
Un fil horizontal de masse \qty{5.0}{g} et de longueur \qty{20}{cm}
repose dans un \omterm{def:g11:electric-gravitational-fields:field}{champ}
horizontal $B = \qty{0.10}{T}$ perpendiculaire à lui. Quelle intensité
fait équilibrer la \omterm{prop:g11:magnetic-fields:laplace}{force de Laplace} le
\omterm{def:g10:universal-gravitation:weight}{poids} du fil, de sorte qu'il lévite~?
Dans quel sens la \omterm{def:g10:inertia:force}{force} doit-elle pointer~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:magnetic-fields:11}
Donner la direction de la
\omterm{prop:g11:magnetic-fields:laplace}{force de Laplace} (ou dire qu'elle s'annule)~:
(a)~courant vers la droite de la page, $\vect B$ entrant dans la
page~; (b)~courant vers le haut de la page, $\vect B$ sortant de la
page~; (c)~courant parallèle à $\vect B$. Utiliser la main droite à
plat.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:12}
Un fil horizontal court nord--sud géographique, à \qty{3.0}{cm}
au-dessus d'une boussole~; la composante horizontale terrestre y est
\qty{20}{\micro T}. Le fil porte $I = \qty{5.0}{A}$~: calculer son

\omterm{def:g11:electric-gravitational-fields:field}{champs} s'ajoutent comme des

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:13}

alimentation \qty{12}{V}~; son noyau de fer multiplie le
\omterm{def:g11:electric-gravitational-fields:field}{champ} nu par $100$. Calculer

Pourquoi la ferraille tombe-t-elle à l'instant où l'interrupteur
s'ouvre~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:magnetic-fields:14}
La boucle rectangulaire d'un moteur a $N = 50$ spires portant

\section{Problème~: La boussole, la grue et l'IRM}

\begin{problem}[{Problème du week-end --- trois aimants au travail~: une
boussole traversant un océan, un électroaimant levant une voiture, et
un solénoïde d'IRM --- un seul champ, s'étalant sur dix puissances de
dix, faisant trois métiers}]
\label{pb:g11:magnetic-fields:1}
Le même vecteur $\vect B$ gouverne un voilier avec \qty{50}{\micro T},
lève de la ferraille avec \qty{1}{T}, et image un genou avec
\qty{1.5}{T} --- trois machines pour les trois formules de ce
chapitre.

\medskip
\noindent\textbf{Partie I --- La boussole.}
À la position du navire, le
\omterm{def:g11:electric-gravitational-fields:field}{champ} terrestre a une composante
horizontale \qty{20}{\micro T} et une composante verticale
\qty{44}{\micro T}~; la \omterm{def:g11:magnetic-fields:earth}{déclinaison} est de
$\ang{6}$ ouest.
\begin{enumerate}
  \item Pourquoi une aiguille de boussole pointe-t-elle du tout le nord
        (magnétique)~?
  \item Calculer la grandeur du
        \omterm{def:g11:electric-gravitational-fields:field}{champ} total et son angle avec
        l'horizontale.
  \item En gouvernant «~nord boussole~» sur \qty{30}{km}, de combien à
        l'ouest de la route plein nord vrai le navire finit-il~?
  \item Une cargaison d'acier dévie encore l'aiguille de $\ang{4}$
        ouest~: même question.
  \item Près du pôle géomagnétique une boussole est inutile~: quelle
        composante est en cause, et pourquoi~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- La grue.}
L'\omterm{def:g11:magnetic-fields:electromagnet}{électroaimant} de casse est un
\omterm{def:g11:magnetic-fields:solenoid}{solénoïde} de $600$ spires enroulées sur
\qty{30}{cm}, de \omterm{def:g11:circuits-and-power:resistance}{résistance}
\qty{1.5}{\ohm}, alimenté sous \qty{12}{V}~; son noyau de fer multiplie
le \omterm{def:g11:electric-gravitational-fields:field}{champ} nu par $50$.
\begin{enumerate}[resume]
  \item Calculer $n$, la densité de spires.
  \item Calculer l'intensité dans la
        \omterm{def:g11:magnetic-fields:solenoid}{bobine}
        (\cref{ch:g11:circuits-and-power}).
  \item Calculer le \omterm{def:g11:electric-gravitational-fields:field}{champ} de la
        \omterm{def:g11:magnetic-fields:solenoid}{bobine} nue $\mu_0 n I$.
  \item Calculer le \omterm{def:g11:electric-gravitational-fields:field}{champ} avec le
        noyau~; comparer avec le \qty{0.5}{T} du néodyme.
  \item Calculer la \omterm{def:g10:energy-conservation:power}{puissance} dissipée dans
        la \omterm{def:g11:magnetic-fields:solenoid}{bobine}, puis
        l'\omterm{def:g10:energy-conservation:energy}{énergie} pour un poste de levage de
        \qty{10}{\minute}.
  \item Un \omterm{def:g11:magnetic-fields:magnet}{aimant} permanent aussi fort existe.
        Pourquoi un \omterm{def:g11:magnetic-fields:electromagnet}{électroaimant}
        quand même~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le moteur du treuil.}
Le treuil de la grue est un moteur à courant continu~: une boucle
rectangulaire de $N = 100$ spires, côtés de longueur \qty{4.0}{cm}
perpendiculaires à $B = \qty{0.25}{T}$, portant $I = \qty{3.0}{A}$.
\begin{enumerate}[resume]
  \item Calculer la \omterm{prop:g11:magnetic-fields:laplace}{force de Laplace} sur
        chaque faisceau de côtés perpendiculaires à $\vect B$.
  \item Les deux \omterm{def:g10:inertia:force}{forces} sont égales et opposées,
        et pourtant leur effet ne s'annule pas. Que produisent-elles, et
        pourquoi doivent-elles être \emph{opposées}~?
  \item Quelle \omterm{def:g10:inertia:force}{force} agit sur les deux autres
        côtés, lorsqu'ils sont parallèles à $\vect B$~?
  \item Après un demi-tour, les mêmes
        \omterm{def:g10:inertia:force}{forces} déferaient la rotation. Que
        fait le \omterm{rem:g11:magnetic-fields:motor}{collecteur}, et quand~?
  \item Donner trois modifications de conception distinctes qui
        doublent chacune la paire de
        \omterm{def:g10:inertia:force}{forces}.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- L'IRM.}
Le \omterm{def:g11:magnetic-fields:solenoid}{solénoïde} du scanner maintient un
\qty{1.5}{T} uniforme avec une intensité $I = \qty{500}{A}$.
\begin{enumerate}[resume]
  \item Combien de fois le
        \omterm{def:g11:electric-gravitational-fields:field}{champ} terrestre vaut
        \qty{1.5}{T}~?
  \item Calculer la densité de spires $n$ requise \emph{sans} aucun
        noyau de fer.
  \item Si l'enroulement avait une
        \omterm{def:g11:circuits-and-power:resistance}{résistance} \qty{0.20}{\ohm},
        quelle \omterm{def:g10:energy-conservation:power}{puissance} dissiperait-il~?
        Quelle propriété de l'enroulement réel évite cela, au prix d'un
        froid extrême~?
  \item Chute~: placer les
        \omterm{def:g11:electric-gravitational-fields:field}{champs} du navire, de la grue
        et de l'IRM sur l'échelle des puissances de dix de l'espace
        interstellaire (\qty{e-10}{T}) à une étoile à neutrons
        (\qty{e8}{T})~; qu'est-ce qui a changé entre les machines --- la
        physique, ou les
        \omterm{def:g11:circuits-and-power:current}{ampères}~?
\end{enumerate}
\end{problem}
