\chapter{Lentilles, images et l'œil}\label{ch:g11:lenses-and-eye}

Toute image nette --- sur le capteur d'un téléphone, un écran de cinéma,
ou le fond de votre propre œil --- repose sur le même tour~: un milieu
transparent courbe dévie les rayons par \omterm{def:g10:refraction:refraction}{réfraction}
(\cref{ch:g10:refraction}) de sorte que tous les rayons issus d'un point
se retrouvent en un autre. Ce chapitre dompte le tour~: une
\omterm{def:g11:lenses-and-eye:lens}{lentille}, trois rayons, une formule --- puis le pointe
vers les appareils photo, les lunettes, et la \omterm{def:g11:lenses-and-eye:lens}{lentille}
vivante qui lit cette phrase.

\section{Lentilles convergentes et divergentes}

\begin{definition}[Lentille mince]\label{def:g11:lenses-and-eye:lens}
Une \emph{lentille}\index{lentille} est un milieu transparent délimité
par deux surfaces, dont l'une au moins est courbe. Plus épaisse au
centre qu'au bord, elle est \emph{convergente}\index{lentille
convergente}~: des rayons parallèles sont déviés l'un vers l'autre~;
plus épaisse au bord, \emph{divergente}\index{lentille divergente}~: ils
s'écartent. Une lentille \emph{mince}\index{lentille mince} a une
épaisseur négligeable~: on la dessine comme un segment munis de
flèches vers l'extérieur (convergente) ou vers l'intérieur
(divergente). Son centre $O$ est le \emph{centre
optique}\index{centre optique}~; la droite perpendiculaire passant par
$O$, l'\emph{axe principal}\index{axe principal} (ou axe optique).
\end{definition}

\begin{definition}[Foyers et distance focale]\label{def:g11:lenses-and-eye:focal}
Les rayons arrivant parallèles à l'axe d'une
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} se croisent tous sur l'axe en un
point derrière elle, le \emph{foyer image}\index{foyer} $F'$~;
symétriquement, les rayons issus du \emph{foyer objet} $F$ (devant, avec
$OF = OF'$) sortent parallèles à l'axe. La \emph{distance
focale}\index{distance focale} $f' = \overline{OF'}$ est positive pour
une \omterm{def:g11:lenses-and-eye:lens}{lentille convergente}~; à travers une
\omterm{def:g11:lenses-and-eye:lens}{lentille divergente}, des rayons parallèles sortent comme
s'ils venaient d'un point $F'$ \emph{devant} la
\omterm{def:g11:lenses-and-eye:lens}{lentille}~: $f' < 0$.
\end{definition}

\begin{definition}[Vergence]\label{def:g11:lenses-and-eye:vergence}
La \emph{vergence}\index{vergence} d'une \omterm{def:g11:lenses-and-eye:lens}{lentille} de
\omterm{def:g11:lenses-and-eye:focal}{distance focale} $f'$ (en mètres) est $C = 1/f'$,
mesurée en \emph{dioptries}\index{dioptrie} (symbole $\delta$ ou D)~:
$1~\text{D} = \qty{1}{m^{-1}}$~; les lentilles
\omterm{def:g11:lenses-and-eye:lens}{divergentes} ont $C < 0$. Les opticiens prescrivent en
dioptries~: les vergences de lentilles \omterm{def:g11:lenses-and-eye:lens}{minces} en contact
s'ajoutent simplement (admis ici~; dérivé dans le volume de l'année~1).
\end{definition}

\begin{example}[Lire une ordonnance]\label{ex:g11:lenses-and-eye:vergence}
Une \omterm{def:g11:lenses-and-eye:lens}{lentille} avec $f' = \qty{0.50}{m}$ a
$C = 1/0.50 = +2.0$~D~: \omterm{def:g11:lenses-and-eye:lens}{convergente}. Une prescription
de $-4.0$~D signifie $f' = \qty{-0.25}{m}$~:
\omterm{def:g11:lenses-and-eye:lens}{divergente}, \omterm{def:g11:lenses-and-eye:focal}{distance focale}
\qty{25}{cm}.
\end{example}

\section{Construction de l'image}

\begin{method}[Les trois rayons de construction]\label{met:g11:lenses-and-eye:rays}
Soit $AB$ un objet perpendiculaire à l'axe en $A$. Parmi tous les
rayons issus de l'extrémité $B$, trois ont des trajets connus~:
\begin{enumerate}
  \item le rayon passant par le \omterm{def:g11:lenses-and-eye:lens}{centre optique} $O$ va tout
        droit~;
  \item le rayon parallèle à l'axe sort en passant par $F'$~;
  \item le rayon passant par $F$ sort parallèle à l'axe.
\end{enumerate}
Deux suffisent~: leur intersection est le point image $B'$, et l'image
$A'B'$ est perpendiculaire à l'axe en $A'$. Si les rayons émergents
divergent, leurs prolongements \emph{rétrogrades} (en pointillés) se
rencontrent en un point d'\omterm{def:g11:lenses-and-eye:images}{image virtuelle}.
\end{method}

\begin{definition}[Images réelles et virtuelles]\label{def:g11:lenses-and-eye:images}
Une image est \emph{réelle}\index{image réelle} lorsque les rayons
émergents y passent effectivement~: un écran placé là la recueille~;
\emph{virtuelle}\index{image virtuelle} lorsque seuls leurs
prolongements rétrogrades se rencontrent~: aucun écran ne la recueille,
mais un œil qui regarde dans la \omterm{def:g11:lenses-and-eye:lens}{lentille} la voit
parfaitement. Une \omterm{def:g11:lenses-and-eye:lens}{lentille convergente} donne une image
réelle, renversée, de tout objet au-delà de $F$~; glisser l'objet à
l'intérieur de la \omterm{def:g11:lenses-and-eye:focal}{distance focale} et l'image devient
virtuelle, droite, agrandie.
\end{definition}

\section{Une formule~: la relation de conjugaison}

Les constructions montrent \emph{où} est l'image~; une formule la
calcule. On oriente l'axe dans le sens de la lumière et on utilise des
\emph{mesures algébriques}~: $\overline{OA}$ est la coordonnée de $A$
par rapport à $O$ (négative pour un objet
\omterm{def:g11:lenses-and-eye:images}{réel} devant)~; les hauteurs $\overline{AB}$ comptent
positives vers le haut.

\begin{theorem}[Relation de conjugaison des lentilles minces]\label{thm:g11:lenses-and-eye:conjugate}
À travers une \omterm{def:g11:lenses-and-eye:lens}{lentille mince} de
\omterm{def:g11:lenses-and-eye:focal}{distance focale} $f'$, un point objet $A$ sur l'axe a
pour image le point $A'$ tel que
\[
\frac{1}{\overline{OA'}} - \frac{1}{\overline{OA}} = \frac{1}{f'} = C .
\]
\end{theorem}
\admitted

\begin{remark}
La relation découle de la \omterm{thm:g10:refraction:snell}{loi de Snell} appliquée
aux rayons voisins de l'axe~; la dérivation --- et les limites du modèle
de la \omterm{def:g11:lenses-and-eye:lens}{lentille mince} --- sont traitées honnêtement dans
le volume de l'année~1.
\end{remark}

\begin{definition}[Grandissement]\label{def:g11:lenses-and-eye:magnification}
Le \emph{grandissement}\index{grandissement} de l'image est
\[
\gamma = \frac{\overline{A'B'}}{\overline{AB}}
       = \frac{\overline{OA'}}{\overline{OA}} .
\]
$\abs{\gamma}$ compare les tailles, son signe les orientations
($\gamma < 0$~: renversée)~; le signe de $\overline{OA'}$ lit la nature
(positif~: \omterm{def:g11:lenses-and-eye:images}{réelle}, derrière la
\omterm{def:g11:lenses-and-eye:lens}{lentille}~; négatif~: \omterm{def:g11:lenses-and-eye:images}{virtuelle},
devant).
\end{definition}

\begin{example}[La formule au travail]\label{ex:g11:lenses-and-eye:conjugate}
Un objet se trouve à \qty{30}{cm} devant une
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} avec $f' = \qty{10}{cm}$~:
$\overline{OA} = \qty{-30}{cm}$, donc
\[
\frac{1}{\overline{OA'}} = \frac{1}{10} - \frac{1}{30} = \frac{1}{15},
\qquad \overline{OA'} = +\qty{15}{cm},
\qquad \gamma = \frac{15}{-30} = -0.50 :
\]
une \omterm{def:g11:lenses-and-eye:images}{image réelle} à \qty{15}{cm} derrière la
\omterm{def:g11:lenses-and-eye:lens}{lentille}, renversée, moitié de taille --- exactement ce
que dessinent les trois rayons.
\end{example}

\section{L'œil}

\begin{definition}[L'œil comme système optique]\label{def:g11:lenses-and-eye:eye}
Optiquement, l'œil est une \omterm{def:g11:lenses-and-eye:lens}{lentille convergente} face à un
écran~: la \emph{cornée}\index{cornée} et le
\emph{cristallin}\index{cristallin} agissent comme une seule
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} de
\omterm{def:g11:lenses-and-eye:focal}{distance focale} réglable~; la
\emph{rétine}\index{rétine}, à environ \qty{17}{mm} derrière, est
l'écran fixe. La mise au point se fait en déformant le
\omterm{def:g11:lenses-and-eye:lens}{cristallin}, non en le déplaçant comme dans un appareil
photo~: l'\emph{accommodation}\index{accommodation} est le changement
de $f'$ produit par les muscles ciliaires. Le point le plus proche vu
nettement à pleine accommodation est le \emph{punctum
proximum}\index{punctum proximum} --- environ \qty{25}{cm} pour
l'\emph{œil normal}\index{œil normal} --- le plus lointain, au repos,
le \emph{punctum remotum}\index{punctum remotum} (l'infini pour l'œil
normal).
\end{definition}

\begin{example}[La vergence de votre œil]\label{ex:g11:lenses-and-eye:eyerange}
La \omterm{def:g11:lenses-and-eye:eye}{rétine} est à $\overline{OA'} = +\qty{0.017}{m}$. Objet
lointain ($1/\overline{OA} \approx 0$)~: $C = 1/0.017 \approx 59$~D.
\omterm{def:g11:lenses-and-eye:eye}{Punctum proximum} ($\overline{OA} = \qty{-0.25}{m}$)~:
$C = 1/0.017 + 1/0.25 \approx 63$~D. Lire coûte environ $4$~D~: l'œil
est une \omterm{def:g11:lenses-and-eye:lens}{lentille} de $59$~D avec un réglage fin intégré
de $+4$~D.
\end{example}

\section{Défauts, corrections et la loupe}

\begin{definition}[Myopie et hypermétropie]\label{def:g11:lenses-and-eye:defects}
Un œil \emph{myope}\index{myopie} (vue de près) est trop
\omterm{def:g11:lenses-and-eye:lens}{convergent} pour sa longueur~: les rayons parallèles se
focalisent \emph{devant} la \omterm{def:g11:lenses-and-eye:eye}{rétine}, les objets lointains
sont flous, le \omterm{def:g11:lenses-and-eye:eye}{punctum remotum} est à distance finie~; une
\omterm{def:g11:lenses-and-eye:lens}{lentille divergente} de lunette le corrige. Un œil
\emph{hypermétrope}\index{hypermétropie} (vue de loin) n'est pas assez
\omterm{def:g11:lenses-and-eye:lens}{convergent}~: les objets proches se focaliseraient derrière
la \omterm{def:g11:lenses-and-eye:eye}{rétine}, le \omterm{def:g11:lenses-and-eye:eye}{punctum proximum}
s'éloigne~; une \omterm{def:g11:lenses-and-eye:lens}{lentille convergente} le corrige. Le
vieillissement rigidifie le \omterm{def:g11:lenses-and-eye:eye}{cristallin}
(\emph{presbytie}\index{presbytie}), réduisant
l'\omterm{def:g11:lenses-and-eye:eye}{accommodation} --- même remède~: des lunettes de
lecture.
\end{definition}

\begin{example}[Prescrire pour un myope]\label{ex:g11:lenses-and-eye:correction}
Un œil \omterm{def:g11:lenses-and-eye:defects}{myope} a son \omterm{def:g11:lenses-and-eye:eye}{punctum remotum}
à \qty{50}{cm}~: la \omterm{def:g11:lenses-and-eye:lens}{lentille} correctrice doit montrer les
objets lointains \emph{au \omterm{def:g11:lenses-and-eye:eye}{punctum remotum}} --- objet à
l'infini, image en $\overline{OA'} = \qty{-0.50}{m}$
(\omterm{def:g11:lenses-and-eye:images}{virtuelle}, devant). Alors $1/f' = 1/\overline{OA'}$~:
$f' = \qty{-0.50}{m}$, $C = -2.0$~D~; l'œil détendu regarde cette
\omterm{def:g11:lenses-and-eye:images}{image virtuelle} et la voit nette.
\end{example}

\begin{definition}[La loupe]\label{def:g11:lenses-and-eye:magnifier}
Une \emph{loupe}\index{loupe} est une
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} de courte
\omterm{def:g11:lenses-and-eye:focal}{distance focale} utilisée avec l'objet \emph{à
l'intérieur} de la \omterm{def:g11:lenses-and-eye:focal}{distance focale}~: l'image est
\omterm{def:g11:lenses-and-eye:images}{virtuelle}, droite, agrandie
(\cref{def:g11:lenses-and-eye:images}), et l'œil la regarde confortablement de
loin.
\end{definition}

\begin{example}[La loupe d'un bijoutier]\label{ex:g11:lenses-and-eye:magnifier}
Loupe avec $f' = \qty{5.0}{cm}$, pierre à \qty{4.0}{cm}~:
$1/\overline{OA'} = 1/5.0 - 1/4.0 = -1/20$, donc
$\overline{OA'} = \qty{-20}{cm}$ et $\gamma = (-20)/(-4.0) = +5.0$~: une
image \omterm{def:g11:lenses-and-eye:images}{virtuelle}, droite, cinq fois plus grande, à un
confortable \qty{20}{cm} de la \omterm{def:g11:lenses-and-eye:lens}{lentille}.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:lenses-and-eye:1}
Calculer les \omterm{def:g11:lenses-and-eye:vergence}{vergences} pour $f' = \qty{50}{cm}$,
$\qty{-20}{cm}$ et $\qty{4.0}{mm}$, puis la
\omterm{def:g11:lenses-and-eye:focal}{distance focale} d'une \omterm{def:g11:lenses-and-eye:lens}{lentille}
de $+8.0$~D.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:lenses-and-eye:2}
Une \omterm{def:g11:lenses-and-eye:lens}{lentille} est plus épaisse aux bords qu'au centre.
\omterm{def:g11:lenses-and-eye:lens}{Convergente} ou \omterm{def:g11:lenses-and-eye:lens}{divergente}~?
Peut-elle servir de \omterm{def:g11:lenses-and-eye:magnifier}{loupe}~? Que voit-on en la tenant
au-dessus d'une page imprimée~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:lenses-and-eye:3}
Une \omterm{def:g11:lenses-and-eye:lens}{lentille convergente} a $f' = \qty{10}{cm}$~; un objet
est à \qty{20}{cm} devant elle. Construire l'image avec deux des trois
rayons, puis vérifier sa position et sa taille avec la relation de
conjugaison.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:lenses-and-eye:4}
Un objet est à \qty{24}{cm} devant une
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} avec $f' = \qty{8.0}{cm}$. Calculer
$\overline{OA'}$ et $\gamma$, et décrire l'image (nature, orientation,
taille).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:lenses-and-eye:5}
Dans l'œil~: qu'est-ce qui forme la
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente}~? Qu'est-ce qui joue le rôle
d'écran~? Qu'est-ce qui change pendant
l'\omterm{def:g11:lenses-and-eye:eye}{accommodation} --- et qu'est-ce qui ne peut pas~?
Définir le \omterm{def:g11:lenses-and-eye:eye}{punctum proximum} de
l'\omterm{def:g11:lenses-and-eye:eye}{œil normal}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:lenses-and-eye:6}
Un objectif d'appareil photo a $f' = \qty{50}{mm}$. Où est le capteur
lorsqu'on met au point un paysage lointain~? Calculer la distance
lentille--capteur pour un sujet à \qty{3.0}{m}, et le débattement entre
les deux réglages.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:lenses-and-eye:7}
Une \omterm{def:g11:lenses-and-eye:magnifier}{loupe} a $f' = \qty{6.0}{cm}$~; un timbre est à
\qty{5.0}{cm} dessous. Position, nature, orientation et
\omterm{def:g11:lenses-and-eye:magnification}{grandissement} de l'image~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:lenses-and-eye:15}
Un objet \omterm{def:g11:lenses-and-eye:images}{réel} est à la distance $d > 0$ devant une
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} ($\overline{OA} = -d$, $f' > 0$).

$\gamma = \frac{f'}{f' - d}$. En déduire~: image
\omterm{def:g11:lenses-and-eye:images}{réelle} et renversée lorsque $d > f'$~;
\omterm{def:g11:lenses-and-eye:images}{virtuelle}, droite, \emph{agrandie} ($\gamma > 1$)

une \omterm{def:g11:lenses-and-eye:magnifier}{loupe}, utilisée assez près.
\end{exercise}

\section{Problème~: Une formule, trois instruments}

\begin{problem}[{Problème du week-end --- la même relation de conjugaison
met au point un appareil photo, prescrit des lunettes de lecture,
conçoit une loupe, et mesure le zoom intégré à votre propre œil}]
\label{pb:g11:lenses-and-eye:1}
Un photographe manœuvre un objectif de \qty{50}{mm}, une grand-mère
tient le journal à bout de bras, un bijoutier se penche sur un diamant~:
une ligne d'algèbre, $1/\overline{OA'} - 1/\overline{OA} = 1/f'$
(\cref{thm:g11:lenses-and-eye:conjugate}), traverse les trois métiers ---
puis l'œil lui-même.

\medskip
\noindent\textbf{Partie I --- Une formule.} Une
\omterm{def:g11:lenses-and-eye:lens}{lentille convergente} a $f' = \qty{20}{cm}$.
\begin{enumerate}
  \item Un objet à \qty{60}{cm}~: calculer $\overline{OA'}$ et
        $\gamma$~; décrire l'image.
  \item Déplacer l'objet à \qty{30}{cm}~; recalculer. Qu'est-ce qui
        s'est échangé avec la question~1~? (Les trajets lumineux sont
        réversibles.)
  \item Le placer à \qty{10}{cm}, à l'intérieur de $f'$~: recalculer et
        décrire.
  \item Résumer~: lorsque l'objet glisse de très loin vers $F$, puis
        le dépasse, comment l'image se déplace-t-elle et change-t-elle
        de nature~?
  \item Montrer qu'un objet très lointain ($1/\overline{OA} \approx 0$)
        se forme toujours dans le plan focal~: $\overline{OA'} = f'$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le photographe.}
Un objectif d'appareil photo a $f' = \qty{50}{mm}$~; le capteur, haut
de \qty{24}{mm}, peut se déplacer par rapport à la
\omterm{def:g11:lenses-and-eye:lens}{lentille}.
\begin{enumerate}[resume]
  \item Où est le capteur lorsqu'on met au point un paysage lointain~?
  \item Maintenant un visage à \qty{1.0}{m}~: calculer la nouvelle
        distance lentille--capteur et le débattement depuis le réglage
        paysage.
  \item Le mécanisme autorise au plus \qty{55}{mm} de distance
        lentille--capteur. Calculer la distance minimale de mise au
        point.
  \item À cette mise au point la plus proche, calculer $\gamma$.
        L'image d'un visage de \qty{20}{cm} tient-elle sur le capteur~?
  \item «~Macro~» grandeur nature signifie $\gamma = -1$~: où doivent
        alors se trouver l'objet et le capteur, et pourquoi cela
        exige-t-il un objectif spécial~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La grand-mère.}
Avec l'âge, le \omterm{def:g11:lenses-and-eye:eye}{punctum proximum} s'éloigne~: celui de
grand-mère est à \qty{1.0}{m}.
\begin{enumerate}[resume]
  \item Expliquer le journal à bout de bras.
  \item Pour lire à \qty{25}{cm}, ses lunettes doivent transformer une
        page à \qty{25}{cm} en une \omterm{def:g11:lenses-and-eye:images}{image virtuelle}
        à \qty{1.0}{m}~: calculer la
        \omterm{def:g11:lenses-and-eye:vergence}{vergence} requise.
  \item Calculer $\gamma$. L'image est quatre fois plus grande, et
        pourtant ne paraît pas plus grande~: expliquer, en comparant
        tailles \emph{et} distances.
  \item Montrer que l'imprimé n'est net qu'entre \qty{25}{cm} et
        $f' \approx \qty{33}{cm}$. Qu'en est-il d'un objet
        \emph{lointain} vu à travers les lunettes~?
  \item En déduire en une phrase pourquoi les verres bifocaux existent.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le bijoutier, et le zoom propre de l'œil.}
La loupe du bijoutier a $f' = \qty{5.0}{cm}$~; son
\omterm{def:g11:lenses-and-eye:eye}{punctum proximum} est à \qty{25}{cm}~; sa
\omterm{def:g11:lenses-and-eye:eye}{rétine} est à \qty{17}{mm} derrière la
\omterm{def:g11:lenses-and-eye:lens}{lentille} de son œil.
\begin{enumerate}[resume]
  \item Une pierre à \qty{4.0}{cm} sous la loupe~: position, nature,
        \omterm{def:g11:lenses-and-eye:magnification}{grandissement}~?
  \item Où doit se trouver la pierre pour que son image tombe à son
        \omterm{def:g11:lenses-and-eye:eye}{punctum proximum}, et que vaut alors $\gamma$~?
  \item Pourquoi les bijoutiers expérimentés placent-ils la pierre
        exactement en $F$ (image à l'infini)~?
  \item Maintenant l'œil seul~: calculer sa
        \omterm{def:g11:lenses-and-eye:vergence}{vergence} en regardant un objet lointain,
        puis un à \qty{25}{cm}~: combien de
        \omterm{def:g11:lenses-and-eye:vergence}{dioptries}
        d'\omterm{def:g11:lenses-and-eye:eye}{accommodation}~?
  \item Convertir les deux \omterm{def:g11:lenses-and-eye:vergence}{vergences} en
        \omterm{def:g11:lenses-and-eye:focal}{distances focales}~: de combien, en
        millimètres et en pour cent, l'œil change-t-il sa propre
        \omterm{def:g11:lenses-and-eye:focal}{distance focale} --- le zoom que l'appareil
        photo faisait par débattement et grand-mère par $+3$~D~?
\end{enumerate}
\end{problem}
