\chapter{Énergie mécanique et sa conservation}\label{ch:g11:mechanical-energy}

Un grand huit n'a plus de moteur passé la première colline~: chaque
boucle, chaque élan de vitesse ensuite se dépense depuis le compte
ouvert à la montée. Ce chapitre transforme la comptabilité en deux
nombres --- \omterm{def:g10:energy-conservation:energy}{énergie} cinétique et
potentielle --- et une règle~: tant que seul le
\omterm{def:g10:universal-gravitation:weight}{poids}
\omterm{def:g11:work-of-force:work}{travaille}, leur somme ne change pas. Le
\omterm{def:g10:inertia:inventory}{frottement} est le péage~; nous apprendrons
à le lire dans les livres.

\section{Énergie cinétique}

\begin{definition}[Énergie cinétique]\label{def:g11:mechanical-energy:kinetic}
Un corps de masse $m$ (\unit{kg}) se déplaçant à la vitesse $v$
(\unit{m/s}) porte l'\emph{énergie cinétique}\index{énergie cinétique}
\[
E_k = \tfrac12\, m v^2 .
\]
Vérification des \omterm{def:g10:orders-of-magnitude:unit}{unités}~:
$\unit{kg.m^2/s^2} = \unit{N.m} = \unit{J}$ --- des
\omterm{def:g11:work-of-force:work}{joules}, comme toute
\omterm{def:g10:energy-conservation:energy}{énergie}
(\cref{ch:g10:energy-conservation}).
\end{definition}

\begin{example}[Ordres de grandeur]\label{ex:g11:mechanical-energy:orders}
\begin{itemize}
  \item un piéton, \qty{70}{kg} à \qty{1.4}{m/s}~:
        $E_k = \tfrac12 \times 70 \times 1.4^2 \approx \qty{69}{J}$~;
  \item une voiture, \qty{1300}{kg} à \qty{130}{km/h} $= \qty{36.1}{m/s}$~:
        $E_k \approx \qty{8.5e5}{J}$~;
  \item un train à grande vitesse, $m = \qty{3.85e5}{kg}$, à
        \qty{320}{km/h} $= \qty{88.9}{m/s}$~: $E_k \approx \qty{1.5e9}{J}$~;
  \item une balle de fusil, \qty{8.0}{g} à \qty{800}{m/s}~:
        $E_k \approx \qty{2.6}{kJ}$.
\end{itemize}
Sept ordres de grandeur séparent la promenade du train. Et $E_k$ croît
comme le \emph{carré} de la vitesse~: à \qty{130}{km/h} la voiture porte
$4$ fois l'\omterm{def:g10:energy-conservation:energy}{énergie} qu'elle a à
\qty{65}{km/h} --- quatre fois ce que ses freins doivent retirer.
\end{example}

\section{Le théorème de l'énergie cinétique}

\begin{theorem}[Théorème de l'énergie cinétique]\label{thm:g11:mechanical-energy:ketheorem}
\index{théorème de l'énergie cinétique}
Pour un corps de masse $m$ se déplaçant en ligne droite de $A$ à $B$,
la variation d'\omterm{def:g11:mechanical-energy:kinetic}{énergie cinétique} égale le
\omterm{def:g11:work-of-force:work}{travail} total (\cref{ch:g11:work-of-force})
reçu~:
\[
\Delta E_k = E_k(B) - E_k(A) = \sum W_{AB}(\vect F),
\]
la somme portant sur toutes les \omterm{def:g10:inertia:force}{forces} agissant
sur le corps. Il s'étend au mouvement le long de tout chemin, comme nous
l'admettons aussi ici.
\end{theorem}
\admitted

\begin{example}[Chute libre, vérifiée image par image]\label{ex:g11:mechanical-energy:freefall}
Une balle de \qty{0.200}{kg} est lâchée et filmée~; les images donnent~:
\begin{center}\small
\begin{tabular}{lcccc}
$t$ (\unit{s}) & 0.10 & 0.20 & 0.30 & 0.40\\
hauteur tombée $h$ (\unit{m}) & 0.049 & 0.196 & 0.441 & 0.785\\
vitesse $v$ (\unit{m/s}) & 0.98 & 1.96 & 2.94 & 3.92\\
$E_k = \tfrac12 mv^2$ (\unit{J}) & 0.096 & 0.384 & 0.864 & 1.54\\
$W = mgh$ (\unit{J}) & 0.096 & 0.385 & 0.865 & 1.54
\end{tabular}
\end{center}
Partant de $E_k = 0$, à chaque image $E_k$ coïncide avec le
\omterm{def:g11:work-of-force:work}{travail} $mgh$ du
\omterm{def:g10:universal-gravitation:weight}{poids}~: $\Delta E_k = W(\vect P)$,
c.-à-d. $v^2 = 2gh$ --- le théorème, vérifié à la précision des lectures.
\end{example}

\begin{remark}[Où vit la preuve]\label{rem:g11:mechanical-energy:proof}
L'énoncé général --- valide le long de tout chemin, non seulement une
ligne droite --- découle de la deuxième loi de Newton et d'un peu de
calcul, tous deux plus tard dans ce livre
(\cref{ch:g12:newtons-laws,ch:g12:work-and-energy})~; d'ici là la
vérification en chute libre est notre caution, et nous admettons la
version en chemin courbe et l'utilisons librement.
\end{remark}

\section{Énergie potentielle de gravitation}

\begin{definition}[Énergie potentielle de gravitation]\label{def:g11:mechanical-energy:potential}
Un corps de masse $m$ à l'altitude $z$, mesurée vers le haut depuis un
\emph{niveau de référence}\index{niveau de référence} choisi où
$z = 0$, stocke l'\emph{énergie potentielle de
gravitation}\index{énergie potentielle!gravitationnelle}
\[
E_p = m g z, \qquad g = \qty{9.81}{N/kg}.
\]
\end{definition}

\begin{remark}[Seules les différences comptent]\label{rem:g11:mechanical-energy:reference}
Passer de $A$ à $B$ change $E_p$ de $mg(z_B - z_A) =
-W_{AB}(\vect P)$~: l'opposé du \omterm{def:g11:work-of-force:work}{travail} du
\omterm{def:g10:universal-gravitation:weight}{poids}
(\cref{ch:g11:work-of-force}). Le
\omterm{def:g11:mechanical-energy:potential}{niveau de référence} est un choix
libre --- sol, niveau de la mer, dessus de table --- décalant $E_p$
partout d'une constante qui s'élimine de toute différence. Placez-le là
où les nombres sont les plus simples, et ne le déplacez pas en cours de
problème.
\end{remark}

\begin{example}[Un livre sur une étagère]\label{ex:g11:mechanical-energy:shelf}
Un livre de \qty{1.2}{kg} est à \qty{1.8}{m} au-dessus du sol, lui-même
à \qty{9.0}{m} au-dessus de la rue. Référence au sol~:
$E_p = 1.2 \times 9.81 \times 1.8 \approx \qty{21}{J}$~; dans la
rue~: $E_p \approx \qty{127}{J}$. Tomber au sol libère \qty{21}{J}
dans les deux comptes.
\end{example}

\section{Énergie mécanique et sa conservation}

\begin{definition}[Énergie mécanique]\label{def:g11:mechanical-energy:em}
L'\emph{énergie mécanique}\index{énergie mécanique} d'un corps est
\[
E_m = E_k + E_p = \tfrac12 m v^2 + m g z .
\]
\end{definition}

\begin{theorem}[Conservation de l'énergie mécanique]\label{thm:g11:mechanical-energy:conservation}
\index{conservation de l'énergie mécanique}
Si la seule \omterm{def:g10:inertia:force}{force} qui travaille sur un corps
est son \omterm{def:g10:universal-gravitation:weight}{poids} (pas de
\omterm{def:g10:inertia:inventory}{frottement}~; d'autres
\omterm{def:g10:inertia:force}{forces}, comme la réaction d'une piste
sans frottement, perpendiculaires au mouvement), $E_m$ est
conservée~: pour deux points quelconques $A$, $B$ du mouvement,
\[
\tfrac12 m v_A^2 + m g z_A = \tfrac12 m v_B^2 + m g z_B .
\]
\end{theorem}

\begin{proof}
Par le théorème de l'\omterm{def:g10:energy-conservation:energy}{énergie
cinétique}, $E_k(B) - E_k(A) = W_{AB}(\vect P) = mg(z_A - z_B) =
E_p(A) - E_p(B)$~: ce que $E_k$ gagne, $E_p$ le perd.
\end{proof}

\begin{example}[Toboggan aquatique]\label{ex:g11:mechanical-energy:slide}
Un enfant part du repos au sommet d'un toboggan sans frottement de
hauteur $h = \qty{3.2}{m}$. Référence en bas~: $mgh = \tfrac12 mv^2$,
donc $v = \sqrt{2gh} \approx \qty{7.9}{m/s}$ (\qty{29}{km/h}) ---
quelle que soit la masse \emph{et quelle que soit la forme du
toboggan}~: droit, courbe ou spiralé, seule la descente de hauteur
entre en jeu.
\end{example}

\begin{example}[Pendule]\label{ex:g11:mechanical-energy:pendulum}
Un bob de pendule est écarté jusqu'à s'élever de $h = \qty{12}{cm}$
au-dessus de son point le plus bas, puis lâché. Aux extrêmes $v = 0$~;
en bas tout le $mgh$ est cinétique, $v = \sqrt{2gh} \approx
\qty{1.5}{m/s}$~; de l'autre côté le bob remonte exactement à
\qty{12}{cm}.
\end{example}

\begin{example}[Vitesse d'atterrissage au saut à ski]\label{ex:g11:mechanical-energy:skijump}
Un sauteur à ski quitte le tremplin à $v_A = \qty{25}{m/s}$ et
atterrit $\Delta z = \qty{40}{m}$ plus bas. En négligeant la résistance
de l'air, $v_B = \sqrt{v_A^2 + 2g\,\Delta z} = \sqrt{625 + 785} \approx
\qty{38}{m/s}$ --- aucune connaissance de la trajectoire de vol n'est
nécessaire.
\end{example}

\section{Dissipation par frottement}

\begin{definition}[Dissipation]\label{def:g11:mechanical-energy:dissipation}
L'\omterm{def:g10:energy-conservation:energy}{énergie} quittant le compte
mécanique est dite \emph{dissipée}\index{dissipation}~: elle réapparaît
comme \omterm{def:g10:energy-conservation:forms}{énergie thermique} dans les
surfaces qui frottent et l'air. L'
\omterm{def:g10:energy-conservation:energy}{énergie} totale est conservée
(\cref{ch:g10:energy-conservation})~; l'
\omterm{def:g11:mechanical-energy:em}{énergie mécanique} seule ne l'est pas.
\end{definition}

\begin{proposition}[Le déficit mesure le frottement]\label{prop:g11:mechanical-energy:deficit}
Si une \omterm{def:g10:inertia:force}{force} de
\omterm{def:g10:inertia:inventory}{frottement} $\vect f$
\omterm{def:g11:work-of-force:work}{travaille} aussi le long du mouvement,
\[
\Delta E_m = E_m(B) - E_m(A) = W_{AB}(\vect f) < 0:
\]
la perte d'\omterm{def:g11:mechanical-energy:em}{énergie mécanique} égale le
\omterm{def:g11:work-of-force:work}{travail} du
\omterm{def:g10:inertia:inventory}{frottement}.
\end{proposition}

\begin{proof}
Le théorème de l'\omterm{def:g10:energy-conservation:energy}{énergie
cinétique} se lit maintenant $\Delta E_k = W_{AB}(\vect P) +
W_{AB}(\vect f) = -\Delta E_p + W_{AB}(\vect f)$~; faire passer
$\Delta E_p$. Le \omterm{def:g10:inertia:inventory}{frottement} s'oppose au
mouvement, donc son \omterm{def:g11:work-of-force:work}{travail} est négatif.
\end{proof}

\begin{example}[Audit de luge]\label{ex:g11:mechanical-energy:toboggan}
Un enfant et sa luge ($m = \qty{40}{kg}$) partent du repos, descendent
$h = \qty{5.0}{m}$ le long de \qty{12}{m} de neige, et arrivent à
\qty{7.0}{m/s} au lieu du $\sqrt{2gh} \approx \qty{9.9}{m/s}$ sans
frottement. Référence en bas~: $\Delta E_m = \tfrac12 \times 40 \times
7.0^2 - 40 \times 9.81 \times 5.0 = 980 - 1962 = \qty{-982}{J}$. Donc
$W(\vect f) = \qty{-982}{J}$ sur \qty{12}{m}~: une
\omterm{def:g10:inertia:force}{force} de
\omterm{def:g10:inertia:inventory}{frottement} moyenne
$f = 982/12 \approx \qty{82}{N}$, et \qty{982}{J} de chaleur dans les
patins et la neige.
\end{example}

\begin{method}[Comptabilité de l'énergie]\label{met:g11:mechanical-energy:bookkeeping}
\begin{enumerate}
  \item Choisir un \omterm{def:g11:mechanical-energy:potential}{niveau de
        référence} et le garder~; nommer $A$ (données), $B$
        (question).
  \item Écrire $E_m = \tfrac12 mv^2 + mgz$ en $A$ et en $B$.
  \item Pas de \omterm{def:g10:inertia:inventory}{frottement}~:
        $E_m(A) = E_m(B)$~; résoudre --- typiquement
        $v_B = \sqrt{v_A^2 + 2g(z_A - z_B)}$.
  \item \omterm{def:g10:inertia:inventory}{Frottement}~:
        $E_m(B) - E_m(A) = -f\ell$ ($\ell$ = longueur du chemin)~;
        résoudre pour l'inconnue.
  \item Vérifier~: les barres $E_k$/$E_p$ s'empilent jusqu'à $E_m$,
        plates sauf \omterm{def:g10:inertia:inventory}{frottement}.
\end{enumerate}
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:1}
Calculer l'\omterm{def:g11:mechanical-energy:kinetic}{énergie cinétique} de
(a)~un scooter et son conducteur de \qty{90}{kg} à \qty{25}{km/h}~;
(b)~une voiture de \qty{1200}{kg} à \qty{130}{km/h}. (c)~Par quel
facteur $E_k$ change-t-elle lorsque la vitesse double~? Triple~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:2}
Un grimpeur de \qty{75}{kg} gagne \qty{850}{m} d'altitude. Calculer le
gain d'\omterm{def:g10:energy-conservation:energy}{énergie} potentielle. Dépend-il
du \omterm{def:g11:mechanical-energy:potential}{niveau de référence}~? De la
route~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:3}
Un pot de fleurs tombe de \qty{20}{m} d'un balcon. En négligeant la
résistance de l'air, trouver sa vitesse au sol en \unit{m/s} et
\unit{km/h}. Où est passée la masse~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:4}
Une voiture de \qty{1200}{kg} freine de \qty{25}{m/s} à l'arrêt sur
\qty{50}{m}. Utiliser le théorème de l'
\omterm{def:g10:energy-conservation:energy}{énergie cinétique} pour trouver le
\omterm{def:g11:work-of-force:work}{travail} des freins, puis la
\omterm{def:g10:inertia:force}{force} de freinage moyenne.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g11:mechanical-energy:5}
Un bob de pendule est lâché \qty{8.0}{cm} au-dessus de son point le
plus bas. Trouver sa vitesse en bas et la hauteur atteinte de l'autre
côté.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:6}
En utilisant l'\cref{ex:g11:mechanical-energy:orders}~: par quel
facteur l'\omterm{def:g11:mechanical-energy:kinetic}{énergie cinétique} du train
dépasse-t-elle celle de la balle~? À quelle hauteur sa propre $E_k$
pourrait-elle soulever le train ($h = v^2/2g$)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:7}
Deux toboggans sans frottement descendent de la même plate-forme de
\qty{3.0}{m}, l'un droit et raide, l'autre long et sinueux. Comparer
les vitesses d'arrivée (les calculer) et les temps d'arrivée (sans
calcul).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:8}
Une balle est lancée verticalement vers le haut à \qty{12}{m/s}.
Trouver la hauteur maximale atteinte, puis la vitesse à \qty{4.0}{m}
--- à la montée et à la descente.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:9}
Un sauteur à ski quitte le tremplin à \qty{23}{m/s} et atterrit
\qty{35}{m} plus bas. Prédire la vitesse d'atterrissage sans
résistance de l'air~; la valeur réelle est-elle plus grande ou plus
petite, et pourquoi~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:10}
Une luge de \qty{45}{kg} part du repos, descend \qty{6.0}{m} le long
de \qty{15}{m} de pente, arrivant à \qty{8.0}{m/s}. Trouver l'
\omterm{def:g11:mechanical-energy:em}{énergie mécanique} perdue, puis la
\omterm{def:g10:inertia:force}{force} de
\omterm{def:g10:inertia:inventory}{frottement} moyenne. Où est passée l'
\omterm{def:g10:energy-conservation:energy}{énergie}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g11:mechanical-energy:11}
Le crochet de Galilée~: un pendule lâché \qty{15}{cm} au-dessus de
son point le plus bas rencontre un crochet bloquant la moitié
supérieure du fil à la verticale. Trouver la vitesse en bas et la
remontée au-delà du crochet~; que change le crochet, et que ne
change-t-il pas~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g11:mechanical-energy:12}
Un skateur plonge dans un half-pipe de \qty{2.5}{m}~; chaque passage
(d'un côté à l'autre) dissipe $10\,\%$ de l'
\omterm{def:g11:mechanical-energy:em}{énergie mécanique}. Trouver la vitesse
en bas de la première descente, puis --- en calculant les hauteurs de
remontée successives --- le premier passage se terminant sous
\qty{1.0}{m}.
\end{exercise}

\section{Problème~: Concevoir la boucle}

\begin{problem}[{Problème du week-end --- un audit d'énergie d'un grand
huit~: la hauteur de lancement qu'une boucle dicte, les vitesses
qu'elle promet, le frottement avoué par deux capteurs, les freins qui
ferment le compte --- tout le parcours suspendu à un seul nombre}]
\label{pb:g11:mechanical-energy:1}
Un train de masse $m = \qty{2.0e3}{kg}$ est lâché du repos depuis une
colline de lancement de hauteur $H$ et côtoie, sans moteur, une boucle
circulaire verticale de rayon $R = \qty{8.0}{m}$ dont le sommet est à
la hauteur $2R$. L'analyse du mouvement circulaire (l'année suivante)
fournit la seule donnée que nous empruntons~: au sommet de la boucle
le train n'appuie sur la piste que si
$v_{\text{sommet}}^2 \geq gR$. Les parties~I et~II ignorent le
\omterm{def:g10:inertia:inventory}{frottement}.

\medskip
\noindent\textbf{Partie I --- La colline que dicte la boucle.}
\begin{enumerate}
  \item Calculer la vitesse minimale de sécurité au sommet de la
        boucle.
  \item Sol comme \omterm{def:g11:mechanical-energy:potential}{niveau de
        référence}~: calculer l'
        \omterm{def:g11:mechanical-energy:em}{énergie mécanique} du train au
        sommet de la boucle à cette vitesse minimale.
  \item En déduire $H_{\min}$ et montrer $H_{\min} = 5R/2$ --- ni
        masse, ni $g$.
  \item L'équipe prend $H = \qty{24}{m}$~: calculer la vitesse réelle
        au sommet de la boucle et le rapport
        $v_{\text{sommet}}^2/(gR)$.
  \item Où est passée $m$~? Pourquoi la conception convient-elle aux
        trains pleins et vides~?
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Ce que la hauteur promet} ($H = \qty{24}{m}$).
\begin{enumerate}[resume]
  \item Vitesse en bas de la boucle ($z = 0$), en \unit{m/s} et
        \unit{km/h}~?
  \item Vitesse sur le côté de la boucle ($z = R$)~?
  \item Calculer $E_p$ et $E_k$ au lancement, en bas, sur le côté et
        au sommet de la boucle~; vérifier que chaque paire somme à
        $mgH$.
  \item Une bosse de chameau de \qty{12}{m} suit la boucle~: vitesse à
        sa crête~?
  \item Donner la formule générale de $v(z)$~; où le parcours est-il
        le plus rapide, et le plus lent~?
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- L'audit du
\omterm{def:g10:inertia:inventory}{frottement}.}
Deux capteurs à la même hauteur $z = \qty{2.0}{m}$, distants de
\qty{60}{m} de piste, lisent $v_1 = \qty{20.4}{m/s}$ puis
$v_2 = \qty{19.2}{m/s}$.
\begin{enumerate}[resume]
  \item Calculer la variation d'
        \omterm{def:g11:mechanical-energy:em}{énergie mécanique} entre les
        capteurs. Qu'est-ce qui a rendu le placement à même hauteur
        astucieux~?
  \item En déduire le \omterm{def:g11:work-of-force:work}{travail} du
        \omterm{def:g10:inertia:inventory}{frottement} et la
        \omterm{def:g10:inertia:force}{force} de
        \omterm{def:g10:inertia:inventory}{frottement} moyenne.
  \item Où est passée l'\omterm{def:g10:energy-conservation:energy}{énergie}
        manquante~? Nommer la forme et les lieux.
  \item Quelle fraction de l'$E_m$ du train est perdue par
        \qty{60}{m}~? Pourquoi le simple $H_{\min}$ serait-il dangereux
        dans le monde réel qui frotte~?
  \item Environ \qty{90}{m} de piste séparent le lancement et le sommet
        de la boucle~: estimer $E_m$ là~; la marge de la question~4
        sauve-t-elle encore la boucle~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Fermer le compte.}
Le dernier capteur, à l'entrée de la ligne de freinage horizontale
droite ($z = 0$), lit \qty{18.0}{m/s}.
\begin{enumerate}[resume]
  \item Calculer l'\omterm{def:g11:mechanical-energy:kinetic}{énergie
        cinétique} du train là.
  \item Les freins doivent l'arrêter en $d = \qty{45}{m}$~: calculer
        la \omterm{def:g10:inertia:force}{force} de freinage (constante)
        requise.
  \item Contrôle de confort~: calculer la décélération $a = F/m$~;
        comparer à $g$.
  \item Combien d'\omterm{def:g10:energy-conservation:energy}{énergie} le
        \omterm{def:g10:inertia:inventory}{frottement} de piste a-t-il
        dissipée sur tout le parcours~? Vérifier que
        \omterm{def:g10:inertia:inventory}{frottement} $+$ freins $= mgH$,
        au \omterm{def:g11:work-of-force:work}{joule} près.
  \item Chute, une phrase~: quel seul nombre a fixé la sécurité de la
        boucle, chaque vitesse, la facture de
        \omterm{def:g10:inertia:inventory}{frottement} et la
        \omterm{def:g10:inertia:force}{force} de freinage --- et quel
        était le prix d'ignorer le
        \omterm{def:g10:inertia:inventory}{frottement}~?
\end{enumerate}
\end{problem}
